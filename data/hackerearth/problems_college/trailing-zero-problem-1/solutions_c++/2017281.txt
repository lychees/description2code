#include<iostream>
#include<bits/stdc++.h>
#include<cmath>
using namespace std;   
int main()
{
int t;
cin>>t;
while(t--)
{
	double m=0;
	int n,i,j=0;
	cin>>n;
	for(i=2;i<=n;i++)
	m=m+log10(i);
//	cout<<m;
//	m=(int)m+1;
	for(i=5;n/i>=1;i*=5)
	j=j+n/i;
	cout<<(int)ceil(m)-j<<endl;
}
       
       
}
