#include <algorithm>
#include <fstream>
#include <iostream>
#include <math.h>
#include <queue>
#include <stack>
#include <stdlib.h>
#include <string>
#include <vector>
const int N = 2005;
int n, d[N][N];

using namespace std;

int main() {
  cin >> n;
  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= n; j++)
      scanf("%d", &d[i][j]);
  bool ans = 1;
  for (int i = 1; i <= n; i++)
    for (int j = 1; j <= n; j++) {
      if (d[i][j] != d[j][i])
        ans = 0;
      if (i == j && d[i][j] != 0)
        ans = 0;
      if (i != j && d[i][j] == 0)
        ans = 0;
    }
  for (int i = 1; i <= n && ans; i++) {
    int child, Min = 1000 * 1000 * 1000 + 8;
    for (int j = 1; j <= n; j++) {
      if (i != j && d[i][j] < Min) {
        Min = d[i][j];
        child = j;
      }
    }
    for (int j = 1; j <= n && ans; j++)
      if (i != j && j != child && abs(d[i][j] - d[child][j]) != d[i][child]) {
        ans = 0;
        break;
      }
  }
  cout << (ans ? "YES" : "NO");

  return 0;
}
