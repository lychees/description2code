#include<bits/stdc++.h>
using namespace std;
int arr[1000001];
int main()
{

	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		memset(arr,0,sizeof(arr));
		for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			arr[x]++;
		}
		int max1=INT_MIN;
		for(int i=1;i<=1000000;i++)
		max1=max(max1,arr[i]);
		for(int i=1000000;i>=1;i--)
		{
			if(arr[i] == max1)
			cout<<i<<" ";
		}
		cout<<endl;
	}
	return 0;
}
