#include <iostream>
#include <algorithm>
#define IN(n) int n; cin>>n;
using namespace std;

int main()
{
    IN(n)
    IN(k)
    int arr[n];
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr,arr+n);
    if(arr[0]>k){
    	cout<<"0"<<endl;
    	return 0;
    }
    for(int i=1;i<n;i++){
    	if(arr[i]-arr[i-1]>k){
    		cout<<arr[i-1]<<endl;
    		return 0;
    	}
    }
    cout<<arr[n-1]<<endl;
    return 0;
}
