#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
	int t;
	long long n, a[1000001],i,j,c;
	cin>>t;
	while(t--)
	{
		c=0;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(int i=0;i<n;i++)
		{
			for(j=i;j<n;j++)
			{
				if(a[j]==a[i])
				c++;
			}
		}
		cout<<c<<"\n";
		
	}
	return 0;
}
