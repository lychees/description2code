#include <bits/stdc++.h>
using namespace std;

long long dp[200][11234];
long long a[11234];

long long f(long long x,long long y,long long n,long long p){
    if (y == p) return 0;
    if (x == n) return INT_MAX;

    long long &ans = dp[x][y];
    if (ans != -1) return ans;

    ans = INT_MAX;
    long long sum = 0;
    ans = min(ans,f(x+1,y,n,p));
    for(long long i = y; i < p; ++i){
        sum += a[i];
        ans = min(ans,max(sum,f(x+1,i+1,n,p)));
    }
    return ans;
}

void solve(){
    memset(dp,-1,sizeof(dp));
    long long n,p;
    cin >> n >> p;
    for(long long i = 0; i < p; ++i)
        cin >> a[i];
    long long ans = f(0,0,n,p);
    cout << ans << endl;
}

int main()
{
    solve();
    return 0;
}
