#include<bits/stdc++.h>
using namespace std;

bool solve(vector<int> &vec,int mid,int k)
{
	int prev=vec[0]+mid;
	k--;
	for(int i=1;i<vec.size();i++)
	{
		if(prev+mid>=vec[i])
			continue;
		if(k==0)
			return false;
		prev=vec[i]+mid;
		k--;
	}
	return true;
}
int main()
{
	int n,k;
	cin >> n >> k;
	vector<int> vec(n);
	for(int i=0;i<n;i++)
		cin >> vec[i];
	sort(vec.begin(),vec.end());
	int low=0,high=10000000,mid;
	while(low<=high)
	{
		mid=(low+(high-low)/2);
		bool flag=solve(vec,mid,k);
		if(flag==true&&solve(vec,mid-1,k)==false)
			break;
		if(flag==false)
			low=mid+1;
		else
			high=mid-1;
	}
	cout << mid << endl;
}
