#include<bits/stdc++.h>
using namespace std ;

vector<int> arr ;
int  maximum = INT_MAX/100 , val , ans = 0 ;

void fun(int i , int n , int a , int b , int c ) {

    if( i == n ) {

        val = max(a,max(b,c)) - min(a,min(b,c)) ;
        if( maximum > val ) {
            maximum = val ;
            ans = max(a,max(b,c)) ;
        }
        //cout << a << " " << b << " " <<c << endl  ;
        return ;
    }
    fun(i+1,n,a+arr[i],b,c) ;
    fun(i+1,n,a,b+arr[i],c) ;
    fun(i+1,n,a,b,c+arr[i]) ;
}

int main() {

    int n ;
    cin >> n ;
    arr.resize(n) ;
    for( int i = 0 ; i < n ; i++ ) {
        cin >> arr[i] ;
    }

        fun(0,n,0,0,0) ;
    cout << ans << endl ;

    return 0 ;

}
