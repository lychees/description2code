#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include <cstdio>

using namespace std;


int main(){
	int N;
	cin >> N;
	vector<long long> v(N);
	for(int i=0; i<N; i++) scanf("%lld", &v[i]);
	
	vector<long long> sum(N+1, 0);
	for(int i=1; i<sum.size(); i++){
		sum[i] = sum[i-1] + v[i-1];
	}
	vector< pair<long long, int> > p(N+1);
	for(int i=0; i<N+1; i++){
		p[i] = make_pair( sum[i], i );
	}

	sort(p.begin(), p.end());
	
	int min_pos = p[0].second;
	int max_len = 0;
	for(int i=0; i<=N; i++){
		max_len = max( max_len, p[i].second - min_pos );
		min_pos = min( min_pos, p[i].second );
	}

	if(max_len == 0){
		printf("%d\n", -1);
		return 0;
	}

	int count = 0;
	for(int i=0; i+max_len<=N; i++){
		if( sum[ i+max_len ] - sum[i] >= 0 ) count++;
	}

	printf("%d %d\n", max_len, count);
	
	return 0;
}
