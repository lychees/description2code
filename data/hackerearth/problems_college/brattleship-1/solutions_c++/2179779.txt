#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
    	int r,c,w;
    	cin>>r>>c>>w;
    	int sum=w-1;
    	if(c==w&&r==1) sum++;
    	else if(c==w&&r!=1) sum+=r;
    	else if(w==1) sum+=r*c;
    	else sum+=(r*c)-w;
    	cout<<"Case #"<<i<<": "<<sum<<endl;
    }
    return 0;
}
