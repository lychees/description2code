T=int(raw_input())

for _ in xrange(0,T):
    N=int(raw_input())
    maxCount=0
    countDict={}
    numList=[int(x) for x in raw_input().split()][:N]
    for num in numList:
        countDict[num]=countDict.get(num,0)+1
        if(maxCount<countDict[num]):
            maxCount=countDict[num]
    
    resultList=[]
    
    for k,v in countDict.iteritems():
        if(v==maxCount):
            resultList.append(k)

    print ' '.join(str(x) for x in sorted(resultList,reverse=True))
