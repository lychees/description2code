#include <iostream>
using namespace std;
     
int sum[105][105];  
int main() {
	int t;
    int coin[4] = {0,1,3,5};
    cin >> t;
    while(t--){
    	int num ;
    	cin >> num;
    	int i,j;
    	for(i=0;i<=3;i++){
    		sum[0][i] = 1;
    	}
    	for(i=1;i<=num;i++){
    		for(j=1;j<=3;j++){
    			int x,y;
    			x = (i-coin[j] >= 0)? sum[i - coin[j]][j]: 0;
                y = (j >= 2)? sum[i][j-1]: 0;
                sum[i][j] = x+y;
    		}
    	}
    	cout << sum[num][3] << endl;
    }
    return 0;
}
