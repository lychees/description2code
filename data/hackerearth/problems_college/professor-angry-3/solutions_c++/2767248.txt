#include <iostream>
using namespace std;

//int a[1005];
int main()
{
	int t;cin>>t;
	while(t--){
		int n,k,a,cnt=0; cin>>n>>k; 
		for(int i=0;i<n;i++){
			cin>>a;
			if(a<=0) cnt++;
			else continue;
		}
		if(cnt>=k) cout<<"NO\n";
		else  cout<<"YES\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
