#include <bits/stdc++.h>
using namespace std;

int main()
{
	int test;
	cin>>test;
	for(int j = 1;j<=test;j++){
		int n;
		cin>>n;
		long long int arr[n],dp[n];
		long long int maxdp[n];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		if(n==1){
			cout<<"Case "<<j<<": "<<arr[0]<<endl;
			continue;
		}	
		long long int amax;
		dp[0] = arr[0];
		dp[1] = arr[1];
		maxdp[0] = arr[0];
		maxdp[1] = max(arr[0],arr[1]);
		//amax = max(arr[0],arr[1]);
		for(int i=2;i<n;i++){
			dp[i] = maxdp[i-2] + arr[i];
			maxdp[i] = max(maxdp[i-1],dp[i]);
		//	amax = dp[i]>amax?dp[i]:amax;
		}
		cout<<"Case "<<j<<": "<<maxdp[n-1]<<endl;
	}
    return 0;
}
