int dp[12][3][3], x[12];
int len ;
int rec(int pos, int is3, int flag)
{
	if (pos == len)
	{
		if (is3 == 1)
			return 0;
		else
			return 1;
	}
	if (dp[pos][is3][flag] != -1)
		return dp[pos][is3][flag];
	int ans = 0;
	if (!flag)
	{
		for (int i = 0; i < x[pos]; i++)
		{
			ans += rec(pos + 1, is3 | (i == 3), 1);
		}
		ans += rec(pos + 1, is3 | (x[pos] == 3), 0);
	}
	else
	{
		for (int i = 0; i <= 9; i++)
		{
			ans += rec(pos + 1, is3 | (i == 3), 1);
		}
		dp[pos][is3][flag ] = ans;
	}
	return ans;
}
