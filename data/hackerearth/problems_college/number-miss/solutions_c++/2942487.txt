#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    cin>>n;
    int map1[10005]={0};
    for(int i=0;i<n;i++)
        {
        int x;
        cin>>x;
        map1[x]++;
    }
    cin>>m;
    int map2[10005]={0};
    for(int i=0;i<m;i++)
        {
        int x;
        cin>>x;
        map2[x]++;
    }
    for(int i=1;i<=10000;i++)
        {
        int diff=abs(map1[i]-map2[i]);
        if(diff)
            cout<<i<<" ";
    }
    
}
