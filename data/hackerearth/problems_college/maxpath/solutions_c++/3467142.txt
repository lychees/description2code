//This is a program to compute maximum gold path
#include <iostream>
#include <stdlib.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n][n];
    for (int i=0; i<n; ++i)
        for(int j=0; j<n; ++j) cin>>a[i][j];
    int fc[n][n];
    for(int i=0; i<n; ++i)
        for(int j=0; j<n; ++j) fc[i][j]=-999999;
    fc[0][0]=a[0][0];
    //cerr<<"\nThe 0th row and 0th column is "<<fc[0][0];
    for(int j=1; j<n; ++j) {fc[0][j]=fc[0][j-1]+a[0][j]; //cerr<<"\nThe 0th row and "<<j<<" th column is "<<fc[0][j];}
    }
    for(int i=1; i<n-1; ++i)
    {
        for(int j=i+1; j<n; ++j)
        {
            fc[i][j]=max(fc[i-1][j],fc[i][j-1])+a[i][j];
            //cerr<<"\nThe "<<i<<" th and "<<j<<" th element is "<<fc[i][j];
        }
    }
    fc[n-1][n-1]=max(fc[n-2][n-1],fc[n-1][n-2])+a[n-1][n-1];
    int out1=fc[n-1][n-1];
    //cout<<"\n"<<out1;
    for(int i=0; i<n; ++i)
    {
        for(int j=i; j<n; ++j)
        {
            int t=a[i][j];
            a[i][j]=a[j][i];
            a[j][i]=t;
        }
    }
    for(int i=0; i<n; ++i)
        for(int j=0; j<n; ++j) fc[i][j]=-999999;
    fc[0][0]=a[0][0];
    //cerr<<"\nThe 0th row and 0th column is "<<fc[0][0];
    for(int j=1; j<n; ++j) {fc[0][j]=fc[0][j-1]+a[0][j];// cerr<<"\nThe 0th row and "<<j<<" th column is "<<fc[0][j];
    }
    for(int i=1; i<n-1; ++i)
    {
        for(int j=i+1; j<n; ++j)
        {
            fc[i][j]=max(fc[i-1][j],fc[i][j-1])+a[i][j];
            //cerr<<"\nThe "<<i<<" th and "<<j<<" th element is "<<fc[i][j];
        }
    }
    fc[n-1][n-1]=max(fc[n-2][n-1],fc[n-1][n-2]);
    int out2=fc[n-1][n-1];
    //cout<<"\n"<<out2;
    cout<<out1+out2;
    return 0;
}
