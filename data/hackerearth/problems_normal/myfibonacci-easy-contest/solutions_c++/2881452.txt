#include <iostream>
#include <vector>

using namespace std;





int main()
{

    int a , b , n ;
    
    scanf("%d%d%d",&a,&b,&n) ;
    
    vector<int> val (n,0);
    
    val[0] = a , val[1] = b ;
    
    for(int i = 3 ; i<= n ; i++)
    {
        val[i-1] = val[i-2] + val[i-3] ; 
    }
    
    
    cout << val[n-1] ;

    return 0;
}
