#define MOD 1000000007

long long cal(long long x,long long n)
{
	long long t=x;
	x=1;
	while(n)
	{
		if(n%2==1)
			x=x*t%MOD;
		t=t*t%MOD;
		n/=2;
	}
	return x;
}
