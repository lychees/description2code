#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	char a[10],b[10];
   	int i,j,cnt=0;
   	
   	cin>>a;
   	cin>>b;
   	int m=strlen(a);
   	int n=strlen(b);
   	
   	int l;
   	if(m>=n)
   	{cnt=m-n;
   	l=n;
   	}
   	else
   	{
   	cnt=n-m;
   	l=m;
   	}
   	
   	for(i=0;i<l;i++)
   	{
   		if(a[i]==b[i])
   	     {}
   	     else
   	     cnt++;
   }
   if(cnt==0)
   cout<<"Exact"<<"\n";
   else
   cout<<"Non Exact "<<cnt<<"\n";
}
}
