n = input()

flag = 0
for i in range(0,n):
    pat = ""

    if flag == 0 or flag == n-1:
        for j in range(0,n):
            pat = pat + "*"
    else:
        for j in range(0,n):
            if j == (n-1)-flag:
                pat = pat + "*"
                break
            else:
                pat = pat + " "
    print pat
    flag = flag+1
