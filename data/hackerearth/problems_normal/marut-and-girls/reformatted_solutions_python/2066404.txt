required_no_of_qualities = int(raw_input())
required_qualities = raw_input().split(' ')
no_of_g = int(raw_input())
no_of_cond = 0
for x in range(0, no_of_g):
    x = raw_input().split(' ')
    if len(x) >= required_no_of_qualities:
        if len(set(x).intersection(required_qualities)
               ) == required_no_of_qualities:
            no_of_cond = no_of_cond + 1
print no_of_cond
