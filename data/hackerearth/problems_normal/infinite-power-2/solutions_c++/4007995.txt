#include <bits/stdc++.h>
using namespace std;
int digits1[19],digits2[19];
int countDigits(long long int n)
{
    int count=0;
    while(n>0)
    {  
	count++;
	n=n/10; }
    return count;
}
long long findJustLargerOrEqual(int pos, int digits){
	if(pos==0){
            if(digits2[pos]==9)digits2[pos]=20;
            else digits2[pos]=digits2[pos]+1;
            for(int i=pos+1;i<digits;i++)
                digits2[i]=0;
               }
        else if(pos>0){
        	if(digits2[pos]<9){
        		digits2[pos]=digits2[pos]+1;
			}
        	else if(digits2[pos]==9){
        		if(digits2[pos-1]<8){
        			digits2[pos]=0;
                    digits2[pos-1]+=2;
				}
				else{
					int temp=pos;
					while(digits2[temp-1]==8&&temp-1>=0){
                        digits2[temp]=0;
                        temp--;
                        }
                    if(temp==0)digits2[temp]=20;
                    else{
                    	digits2[temp]=0;
                         digits2[temp-1]+=2;
					}
				}
			}
			for(int i=pos+1;i<digits;i++)
                digits2[i]=0;
		}
		long long great=0;
		for(int i=0;i<digits;i++)
            great=great*10+digits2[i];
        return great;
}
long long findJustSmallerOrEqual(int pos, int digits){
        if(digits1[pos]==1)  digits1[pos]=0;
    	else digits1[pos]=digits1[pos]-1;
        for( int i=pos+1;i<digits;i++)
            digits1[i]=8;
        long long small=0;
    	for(int i=0;i<digits;i++)
            small=small*10+digits1[i];
        return small;
}
int main()
{
   long long int num;
   int t;
   
    scanf("%d",&t);
   while(t-->0){
    	int pos=-1;
		scanf("%lld", &num);
		int  d=countDigits(num);
		int dig=d;
		long long temp=num;
		while(temp>0){
			digits1[d-1]=temp%10;
            digits2[d-1]=temp%10;
            temp=temp/10;
            d--;
		}
        for(int i=0;i<dig;i++){
            if(digits1[i]%2){
                        pos=i;
                        break;
						}
                }
        if(pos==-1)
               printf("Unlimited Power\n");
        else{
        	long long justLarger=findJustLargerOrEqual(pos, dig);
        	long long justSmaller=findJustSmallerOrEqual(pos, dig);
        	long long difference1=justLarger-num;
        	long long difference2=num-justSmaller;
			if(difference1%difference2==0){
				printf("%lld/1\n",difference1/difference2);
                }
            else{
                long int gcdValue=__gcd(difference1,difference2);
//                long a=diff2/gdf;
                printf("%lld/",difference1/gcdValue);
                printf("%lld\n",difference2/gcdValue);
                }
		}
	}
	return 0;
}
