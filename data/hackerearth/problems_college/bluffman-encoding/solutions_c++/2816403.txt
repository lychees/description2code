#include<bits/stdc++.h>
using namespace std;
string str[27],inp;
pair <int,pair<int,char> > pp[27];
int main()
{
    cin.sync_with_stdio(false);
    int t,i,n,tmp;
    char ch;
    cin>>t;
    while(t--)
    {
        map <string,char> mp;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>ch>>tmp;
            pp[i]=make_pair(tmp,make_pair(i,ch));
        }
        sort(pp,pp+n);
        str[0]="0";
        for(i=1;i<n-1;i++)
            str[i]="1"+str[i-1];
        str[n-1]="";
        for(i=1;i<n;i++)
            str[n-1]+="1";
        for(i=0;i<n;i++)
            mp[str[i]]=pp[i].second.second;
        cin>>inp;
        tmp=n-1;
        int s=0,ct=0;
        for(i=0;i<inp.size();i++)
        {
            if(inp[i]=='0')
            {
                cout<<mp[inp.substr(s,i-s+1)];
                s=i+1;
            }
            if(i-s+1==tmp)
            {
                cout<<mp[inp.substr(s,i-s+1)];
                s=i+1;
            }
        }
        cout<<'\n';
    }
}
