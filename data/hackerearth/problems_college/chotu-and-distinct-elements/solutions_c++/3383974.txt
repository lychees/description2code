#include<bits/stdc++.h>
using namespace std;
long long int a[100005];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		set<long long int>set1;
		long long int n;
		cin>>n;
		int c=0;
		for(int i=0;i<n;i++)
		{
			long long int x;
			cin>>x;
			set1.insert(x);
		}
	c = set1.size();
	int i=0;
	set<long long int>::iterator it = set1.begin();
	while(it!=set1.end() && i<c)
	{
		a[i] = *it;
		i++;
		it++;
	}
	sort(a,a+c);
	i=0;
	int j=c-1;
	if(c%2==0)
	{
		while(i<j)
		{
			cout<<a[j]<<" "<<a[i]<<" ";
			i++;
			j--;
		}
	}
	else
	{
		while(i<j)
		{
			cout<<a[j]<<" "<<a[i]<<" ";
			i++;
			j--;
		}
		if(i==j)
		{
			cout<<a[j]<<" ";
		}
	}
	cout<<"\n";
	}
}
