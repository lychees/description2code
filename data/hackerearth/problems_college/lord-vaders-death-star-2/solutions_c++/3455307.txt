#include<bits/stdc++.h>

#define MAX 1000005
#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

ll a[10000000];

int main(){

	ll n,i,num=-1;

	scanf("%lld",&n);

	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);

	sort(a,a+n);
    
    if(a[n-1]!=a[n-2])
    	num=a[n-1];
    else{

	for(i=n-2;i>=1;i--){
        
        if(a[i]!=a[i-1] && a[i]!=a[i+1]){
        	num=a[i];
        	break;
        }
	}

	if(num==-1)
		if(a[0]!=a[1])
			num=a[0];
}

printf("%lld\n",num);
	return 0;
}
