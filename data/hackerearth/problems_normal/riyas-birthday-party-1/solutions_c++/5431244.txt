#include <iostream>
#include <iomanip>
#include <limits>
#include <cstdio>
#include <cstdlib>
#include <vector>

using namespace std;

typedef unsigned long long ll;

ll M=1000000007;

int main() {
	ll noOfInputs,number;
	
	cin>>noOfInputs;
	while(noOfInputs != 0 ){
		cin>>number;
		
		cout<<( ( (number%M)*( (2*number)%M - 1 ) )%M )<<endl; 

		--noOfInputs;
	}

	getchar();
	getchar();
	return 0;
}
