#include <bits/stdc++.h>
using namespace std;

vector<bool> vis(1000023,false);
vector<long long> sum(1000023);

bool isprime(long long n){
	for(long long i=2;i*i<=n;i++)
		if(n%i==0)
			return false;
	return true;
}
int main()
{
    //cout << "Hello World!" << endl;
    //memset(vis,false,sizeof(vis));
    ios::sync_with_stdio(false);
    cin.tie(0);
    for(int i=2;i<=1000000;i++){
    	sum[i]=sum[i-1];
    	if(!vis[i]){
    		
    		sum[i]+=i;
    		for(int j=i*2;j<=1000000;j+=i)
    			vis[j]=true;
    	}
    }
    vis[0]=true;
    vis[1]=true;
    
    long long t,a,b;
    cin>>t;
   	while(t--){
   		cin>>a>>b;
   		
   		//cout<<sum[b]<<" "<<sum[a-1]<<endl;
   		a = sum[b]-sum[a-1];
   		if(a!=0&&a!=1&&isprime(a))cout<<"YES\n";
   		else cout<<"NO\n";
   	}
    return 0;
}
