#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int main(){
	long n,k;
	cin >> n;
	cin >> k;
	long long *arr;
	long long *anotherarray;
	anotherarray = new long long[n];
	arr = new long long[n];
	for(long long i=0;i<n;i++){
		cin >> arr[i];
	}
	sort(arr,arr+n);
	long long g=0;
	g+=k-1;
	long long l=0;
	for(long long i=0;g+i<n;i++){
		anotherarray[l]=(arr[g+i]-arr[i]);

		l++;
		
	}
	sort(anotherarray,anotherarray+l);
	cout << anotherarray[0]<<endl;
}
