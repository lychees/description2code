#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
double dp[51][100001];
int main()
{
    int t,m,n,k;
    cin>>t;
    while(t--){
    	cin>>m>>n>>k;
    	memset(dp,0,sizeof(dp));
    	for(int i=1;i<=n;i++) dp[1][i] = 1.0/n;
    	for(int i=2;i<=m;i++){
    		for(int k=1;k<=(i-1)*n;k++){
    			for(int j=1;j<=n;j++){
    				dp[i][k+j] += 1.0*dp[i-1][k]*dp[1][j];
    			}
    		}
    	}
    	double f = dp[m][k];
    	//cout << f << "\n";
    	if (f==0.0){
    		cout << "0.000 0\n";
    		continue;
    	}
    	int y = 0;
    	while (f<1){
    		f *= 10.0;
    		y++;
    	}
    	printf("%.3f %d\n",f,y);
    }
    return 0;
}
