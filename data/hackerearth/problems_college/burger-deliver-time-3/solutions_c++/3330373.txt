#include <bits/stdc++.h>
using namespace std;
vector<pair<int,int> >vec;
int main()
{ pair<int,int>r;
    int n,q,w;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cin>>q>>w;
     r=make_pair(q+w,i);
     vec.push_back(r);
    	
    	
    }
    sort(vec.begin(),vec.end());
    for(int i=0;i<n;i++)
    cout<<vec[i].second<<" ";
}
