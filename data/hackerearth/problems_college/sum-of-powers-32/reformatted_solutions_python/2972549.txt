P = [0] * 1000001
P[1] = 1
for i in xrange(2, 1001):
    j = 2
    while int(pow(i, j)) <= 1000000:
        pos = int(pow(i, j))
        P[pos] = 1
        j += 1

for _ in range(int(raw_input())):
    x = int(raw_input())
    found = False
    for i in xrange(1, x+1):
        if P[i] and P[x-i]:
            found = True
            break
    print "Yes" if found else "No"
