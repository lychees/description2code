#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)cin>>a[i];
    	int right[n],left[n];
    	left[0]=a[0];
    	for(int i=1;i<n;i++)left[i]=left[i-1]+a[i];
    	right[n-1]=a[n-1];
    	for(int i=n-2;i>=0;i--)right[i]=right[i+1]+a[i];
    	int flag=0;
    	for(int i=0;i<n;i++){
    		if(left[i]-a[i]==right[i]-a[i]){flag=1;break;}
    	}
    	if(flag)cout<<"YES\n";
    	else cout<<"NO\n";
    }
    return 0;
}
