#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>
using namespace std;
const long m = 161700;
int gcd(int n, int m){
	return m == 0 ? n : gcd(m, n % m);
}
int main(int argc, char const *argv[])
{
	int t, A[105];
	long long x, z, d;
	memset(A, 0, sizeof(A));
	
	for (int i = 1; i <= 100; i++) {
		for (int j = i + 1; j <= 100; j++) {
			for (int k = j + 1; k <= 100; k++) {
				A[gcd(i, gcd(j, k))]++;
			}
		}
	}
	long p;
	double ans;
	cin >> t;
	while (t != 0) {
		cin >> x >> z;
		p = 0;
		d = (x - z < 0) ? z - x : x - z;
		for (int i = 1; i <= 33; i++)
			if (d % i == 0)
				p += A[i];

		ans = (double)p / (double)m;
		printf("%.6f\n", ans);
		t--;
	}
	return 0;
}
