#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int i,n,x,y,eff=0;
        scanf("%lld%lld%lld",&n,&x,&y);
        long long int fin[n],star[n];
        for(i=0;i<n;i++)
        {
            scanf("%lld%lld",&star[i],&fin[i]);
        }
        sort(star,star+n);
        sort(fin,fin+n);
        for(i=0;i<n;i++)
        {
            if(star[i]>fin[i])
            {
                eff+=(star[i]-fin[i])*y;
            }
            else if(star[i]<fin[i])
            {
                eff+=(fin[i]-star[i])*x;
            }
        }
        printf("%lld\n",eff);
        eff=0;
    }
}
