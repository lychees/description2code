#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
       int cases, n;
       int girls[10000], boys[10000];
       cin>>cases;
       for(int i=0; i<cases; i++){
       	
       	
               int ans = 0;
               cin>>n;
               for(int j=0; j<n; j++){
                       cin>>girls[j];
               }
               for (int j=0; j<n; j++){
                       cin>>boys[j];
               }
               sort(girls,girls+n);
               sort(boys,boys+n);
               for (int k=0; k<n; k++){
                       int boy = boys[n-k-1];
                       int girl = girls[k];
                       if ((boy%girl == 0) || (girl%boy == 0)){
                               ans+=1;
                       }
                       
                       
               }
               cout<<ans<<"\n";
               
       }
}
