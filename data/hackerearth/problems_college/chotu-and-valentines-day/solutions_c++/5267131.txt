#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int n,q;
	cin>>n>>q;
	
	pair<long long,long long> p[n+1];
	p[0].first=p[0].second=0;
	
	for(long long i=1;i<=n;i++)
	{
		cin>>p[i].first;
		p[i].second = p[i].first + p[i-1].second;
	}
	
	while(q--)
	{
		long long l,r;
		cin>>l>>r;
		cout<<p[r].second-p[l-1].second<<endl;
	}
	return 0;
}
