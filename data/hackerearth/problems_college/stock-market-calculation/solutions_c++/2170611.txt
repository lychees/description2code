#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
#define MAX 50000
void maxpr(int arr[],int n){
    int i,m_sofar,arr_b[MAX];
    long long int buy=0,sell=0,profit=0,net=0,count=0;
    
    m_sofar=arr[n-1];
    arr_b[n-1]=0;
    for(i=n-2;i>=0;i--){
        if(arr[i]>m_sofar){
               arr_b[i]=0;
            m_sofar=arr[i];
        }else
            {
            arr_b[i]=1;
       } 
    }
    for(i=0;i<n;i++){
        if(arr_b[i]==1){
            buy+=arr[i];
            count++;
        }else{
            sell=arr[i]*count;
            profit=sell-buy;
            net+=profit;
            count=0;
            buy=0;
            
        }
    }
    cout<<net<<endl;
    
}
int main(){

	int a[MAX];
	int i,t,j,n;
	cin>>t;
	for(i=0;i<t;i++){
		cin>>n;
		for(j=0;j<n;j++){
			cin>>a[j];
		}
	maxpr(a,n);
	}
	return 0;
}
