#define MODV 252

typedef long long ll;
typedef double dbl;
typedef vector<ll> vi;
int getint(){int a;scanf("%d",&a);return a;}
ll getll(){ll a;scanf(LL,&a);return a;}

class mat{
  typedef unsigned T;
  typedef vector<vector<T>> M;
  typedef vector<T> V;
  public:
  int r, c;
  M m, t;
  mat(const int &r, const int &c):r(r),c(c),m(r,V(c,0)),t(r,V(c,0)){}
  void mul(const mat &other){
    for(int i=0;i<r;i++)for(int j=0;j<c;j++){
      t[i][j]=0;
      for(int k=0;k<c;k++)t[i][j]+=m[i][k]*other.m[k][j];
    }
    swap(m,t);
  }
  void pow(ll b){
    assert(r==c);
    mat res(r,c);
    for(int i=0;i<r;i++)res.m[i][i]=1;
    while(b){
      if(b&1)res.mul(*this);
      mul(*this);
      b/=2;
    }
    *this=res;
  }
};
void doit(){
  int t, tt;
  ll n;
  unsigned res=0;
  mat v(MODV, MODV);
  for(int i=0;i<MODV;i++){
    t=i*10;
    for(int j=1;j<10;j++)
      if((t+j)%j==0){
        tt=(t+j)%MODV;
        v.m[i][tt]++;
      }
  }
  n=getll();
  v.pow(n);
  for(int i=0;i<MODV;i++){
    res += v.m[0][i];
  }
  cout<<res<<endl;
}

int main() {
  doit();
  return 0;
}
