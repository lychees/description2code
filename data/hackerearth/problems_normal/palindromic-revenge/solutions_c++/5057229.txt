 #include<bits/stdc++.h>
using namespace std;
#define ll long long
#define li long int 

int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		ll i=0;
		ll j=s.length()-1;
		ll count=0;
		while(i<j)
		{
			if(s[i]!=s[j])
			{
				if(s[i]>s[j])
				count=count+s[i]-s[j];
				else
				count=count+s[j]-s[i];
			}
			i++;
			j--;
		}
		cout<<count<<endl;
		
	}
	return 0;
}
