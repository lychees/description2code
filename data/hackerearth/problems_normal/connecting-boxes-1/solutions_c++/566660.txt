#include <iostream>
using namespace std;

int main()
{
    int n,q;
    cin>>n;
    cin>>q;
    int a[n];
    for(int i=0;i<n;i++)
      a[i]=0;
    int x,y,r,s,t,v,flag;
    for(int i=0;i<q;i++)
    {
    	cin>>s;
    	cin>>r;
    	if(a[s-1]!=0 || a[r-1]!=0 || s==r)
    	  cout<<"NO";
    	else
    	{
    		x=(r<s)?r:s;
    		y=(r>s)?r:s;
    		x--;
    		y--;
    		t=x+1;
    		flag=1;
    		while(t<y)
    		{
    			if(a[t]!=0)
    			{
    				if(a[t]<y && a[t]>x);
    				else
    				{
    					flag=0;
    					break;
    				}
    			}
    			t++;
    		}
    		if(flag==0)
    		  cout<<"NO";
    		else
    		{
    			t=(y+1)%n;
    			while(t!=x)
    			{
    				if(a[t]!=0)
    				{
    					if((a[t]>y && a[t]<n) || (a[t]<x && a[t]>=0));
    					else
    					{
    						flag=0;
    						break;
    					}
    				}
    				t=(t+1)%n;
    			}
    			if(flag==0)
    			   cout<<"NO";
    			else
    			 {
    			 	cout<<"YES";
    			 	a[x]=y;
    			 	a[y]=x;
    			 }
    		}
    	
    		
    	}
    	cout<<"\n";
    }
    return 0;
}
