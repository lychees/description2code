#include <iostream>
using namespace std;

int main()
{
    int L,N;
    cin>>L;
    cin>>N;
    for(int i=0;i<N;i++)
    {
     int W,H;
     cin >> W >> H;
     if(W<L || H<L)
     	cout<<"UPLOAD ANOTHER\n";
     else 
     {
     if(W==H)
     	cout<<"ACCEPTED\n";
     else if(W > L || H > L)
     	cout<<"CROP IT\n";
     }
    }
    return 0;
}
