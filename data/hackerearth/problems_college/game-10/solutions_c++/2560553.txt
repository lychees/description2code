#include <iostream>
using namespace std;

int main()
{
    int t,n,m;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	int x,y,temp=0,i;
    	long long sum=0;
    	if(n%2==0){
    		y=n-2;
    		x=(n/2)-1;
    	}
    	else{
    		y=n-1;
    		x=(n/2)-1;
    	}
    		if(m<=x){
    			sum=1;
    			for(i=0;i<m;i++){
    				sum+=2;
    			}
    			cout<<sum<<endl;
    			temp=1;
    		}
    	if(temp==0){
    		for(i=x+1;i<n;i++){
    			if(i==m){
    			cout<<y<<endl;
    			break;
    			}
    			else
    			y=y-2;
    		}
    	}
    }
    return 0;
}
