times = int(input())
d = []
for x in range(times):
    d.append([raw_input(), raw_input()])

for a in d:
    news = ""
    for p in a[0]:
        for q in range(a[1].count(p)):
            if a[1].find(p) > -1:
                news = news + a[1][a[1].find(p)]
                a[1].replace(p, "", 1)
    print(news)
