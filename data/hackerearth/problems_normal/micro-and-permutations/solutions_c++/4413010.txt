#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,x,y,cnt=0,f;
    cin>>n>>m;
    int adj[n+1][n+1];
    int v[n+1];
    for(int i=0;i<n;i++)
    	v[i] = i+1;
    memset(adj,0,sizeof adj);
    for(int i=1;i<=m;i++)
    {
    	cin>>x>>y;
    	adj[x][y] = 1;
    	adj[y][x] = 1;
    }
    
    do
    {
    	f=1;
    	for(int i=1;i<n;i++)
    	{
    		if(!adj[v[i-1]][v[i]])
    		{
    			f = 0;
    			break;
    		}
    	}
    	if(f)
    		cnt++;
    }while(next_permutation(v,v+n));
    cout<<cnt;
}
