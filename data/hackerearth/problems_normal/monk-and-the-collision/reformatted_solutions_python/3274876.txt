T = int(raw_input())
for _ in xrange(0, T):
    N = int(raw_input())
    numbers = [int(x) for x in raw_input().split()][:N]
    hashedArray = [0]*10
    collision = 0
    for number in numbers:
        if hashedArray[number % 10]:
            collision = collision+1
        else:
            hashedArray[number % 10] = 1
    print collision
