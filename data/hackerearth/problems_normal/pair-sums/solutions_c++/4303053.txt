int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int n,s,arr[1000005];
    cin>>n>>s;
    
    int x;
    
    rep(x,1,n,1)
    cin>>arr[x-1];
    
    sort(arr,arr+n);
    
    int l=0;
    int r=n-1;
    
    bool ans=0;
    
    while(l !=r )
    {
    	if(arr[l]+arr[r] == s) {ans=1;break;}
    	else if(arr[l]+arr[r]>s) r--;
    	else l++;
    	
    }
    
    if(ans) cout<<"YES\n";
    else cout<<"NO\n";
    
    return 0;
}
