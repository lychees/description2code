#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)

using namespace std;

int main()
{
	lli t,cx,cy,r,px,py,i,j,k,z,y;
	
	s(t);
	
	while (t--) {
		s(r);
		s(cx);
		s(cy);
		
		z = 0;
		
		for (i = 1; i <= 4; ++i) {
			s(px);
			s(py);
			
			j = (cx-px)*(cx-px) + (cy-py)*(cy-py);
			
			if (j > r*r)
				z = 1;
		}
		
		if (z == 0)
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
    return 0;
}
