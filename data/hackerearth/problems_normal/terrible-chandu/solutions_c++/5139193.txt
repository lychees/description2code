#include <iostream>
#include <vector>
#include <string>
using namespace std;

string Reverse(string apex)
{
	string alabaster(apex);
	int L = apex.length();
	for(int i = 0; i < L; i++) { alabaster[i] = apex[(L - 1) - i]; }
	return alabaster;
}

int main()
{
	int T;
	cin >> T;
	vector<string> relay;
	string temp;
	for(int i = 0; i < T; i++) { cin >> temp; relay.push_back(temp);}
	vector<string> blade;
	vector<string>::const_iterator cit;
	for(cit = relay.begin(); cit != relay.end(); cit++)
	{
		blade.push_back(Reverse(*cit));
	}
	for(cit = blade.begin(); cit != blade.end(); cit++)
	{
		cout << *cit << endl;
	}
	return 0;
}
