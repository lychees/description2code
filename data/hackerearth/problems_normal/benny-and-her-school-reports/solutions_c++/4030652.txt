#include <iostream>
using namespace std;
typedef long long ll;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,m,x;
    	cin>>n>>m>>x;
    	ll sum=0;
    	ll a[n+1];
    	
    	for(int i=0; i<n; i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	ll ans;
    	ans= x*(n+1)-sum;
    	if(ans<=0)
    	{
    		ans=1;
    		cout<<ans<<endl;
    	}	
    	else if(ans>m)
    		cout<<"Impossible"<<endl;
    	
    	else 
    		cout<<ans<<endl;
    }
}
