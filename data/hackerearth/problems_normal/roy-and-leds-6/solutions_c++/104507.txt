#include <iostream>
using namespace std;

int main()
{
    int  T, R, G, B;
    int r = 0, g = 0, b = 0, i;
    int rc = 0, gc = 0, bc = 0, cc = 0, blc = 0, yc = 0, wc = 0, mc = 0;
    for(i = 0; i <= 3; i++) {
    	switch(i) {
    		case 0:
    			cin >> T;
    			break;
    		case 1:
    			cin >> R;
    			break;
    		case 2:
    			cin >> G;
    			break;
    		case 3:
    			cin >> B;
    			break;
    	}
    }
    //cout << T << R << G << B;
    int l = R;
    if(l > G)
    	l = G;
    if(l > B)
    	l = B;
    for(i = 0; i < T; i++) {
    	if(i < l) {
    		r = 0;
    		b = 0;
    		g = 0;
    		blc++;
    	}
    	else {
	    	if(i%R == 0) {
	    		if(r == 0)
	    			r = 1;
	    		else
	    			r = 0;    			
	    	}
	    	if(i%G == 0) {
	    		if(g == 0)
	    			g = 1;
	    		else
	    			g = 0;    			
	    	}
	    	if(i%B == 0) {
	    		if(b == 0)
	    			b = 1;
	    		else
	    			b = 0;    			
	    	}
	    	if(r == 1 && b == 1 && g == 1)
	    		wc++;
	    	else if(r == 1 && b == 1)
	    		mc++;
	    	else if(r == 1 && g == 1)
	    		yc++;
	    	else if(g == 1 && b == 1)
	    		cc++;
	    	else if(r == 0 && g == 0 && b == 0)
	    		blc++;
	    	else if(r == 1)
	    		rc++;
	    	else if(g == 1)
	    		gc++;
	    	else if(b == 1)
	    		bc++;		
    	}
    }
    cout << rc << " " << gc << " " << bc << " " << yc << " " << cc << " " << mc << " " << wc << " " << blc;
    return 0;
}
