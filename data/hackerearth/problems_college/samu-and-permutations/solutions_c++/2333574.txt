LL func(LL ind, LL prev, LL mask) {
           if(ind == n)
           return 0;
           
           if(dp[prev][mask] != -1)
           return dp[prev][mask];
           
           LL l , i;
           
           l = 0;
           
           for(i=0;i<n;i++) {
                            if((mask&(1<<i)) == 0) {
                                             l = max(l,(ind+1)*abs(ar[prev]-ar[i]) + func(ind+1,i,(mask|(1<<i))));
                            }
           }
           
           
           return dp[prev][mask] = l;
}
