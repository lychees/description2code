#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstdlib>
using namespace std;

const int N = 100100;
typedef long long ll;
ll a[N];
ll sum;
int n, q;

int main()
{
    scanf("%d%d", &n, &q);
    for (int i = 0; i < n; i++)
    {
    	ll x, y;
    	scanf("%lld%lld", &x, &y);
    	sum += min(x, y);
    	a[i] = abs(x - y);
    }
    sort(a, a + n);
    for (int i = 0; i < n; i += 2)
    {
    	if (a[i] != a[i + 1])
    	{
    		sum = -1;
    		break;
    	}
    	sum += a[i];
    }
    while(q--)
    {
    	ll x;
    	scanf("%lld", &x);
    	if (x == sum)
    		printf("YES\n");
    	else
    		printf("NO\n");
    }
    return 0;
}
