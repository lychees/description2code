#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a[50][50];
    	int i,j,n,m,count = 0;
    	char x;
    	cin>>n>>m;
    	for(i = 0 ; i < n ;i++)
    	for(j = 0 ; j < m ; j++){
    	cin>>x;
    	if(x == '0')
    	a[i][j] = 0;
    	else a[i][j] = 1;
    	}
    
    	
    	for(i = n-1 ; i >= 0 ; i--)
    	{
    	for(j = m-1 ; j >= 0 ; j--)
    	{
    		if(a[i][j] == 0)
    		{
    			    count++;
    				for(int k = 0 ; k <= i; k++ )
    				{
	                for(int h = 0 ; h <= j ; h++)
	                  {
		                 if(a[k][h] == 0)
		                 a[k][h] = 1;
		                 else
		                 a[k][h] = 0;
	                  }
    				}  
    		}
    	}
    	}
    	cout<<count<<endl;
    	for(i = 0 ; i < n; i++)
    	for(j = 0 ; j < m ; j++)
    	a[i][j] = 0;
    }
    return 0;
}
