#include <bits/stdc++.h>
using namespace std;

int main() {
	map<string, int > m;
	int t;
	cin >> t;
	
	for(int i = 0; i < t; ++i) {
		string s;
		cin >> s;
		m[s]++;
	}
	
	vector<string> v;
	for(map<string, int>::iterator it = m.begin(); it != m.end(); it++) {
		v.push_back(it->first);
	}
	
	sort(v.begin(), v.end());
	
	t = v.size();
	
	for(int i = 0; i < t; ++i) cout << v[i] << " " << m[v[i]] << endl;
	return 0;
}
