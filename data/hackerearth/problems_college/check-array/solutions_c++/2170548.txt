#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
    	int i=0;
    	long int n,flag=1;
    	cin>>n;
    	long int sum=0,x;
    	long int *a=new long int[n];
    	cin>>x;
    	a[0]=x;
    	sum=x;
    	for(i=1;i<n;i++)
    	{ 		
    		cin>>x;
    		sum+=x;
    		a[i]=sum;
    	}
    	for(i=1;i<n-1;i++)
    	{
    		if(a[n-1]-a[i]==a[i-1])
    		{
    			cout<<"YES"<<endl;
    		    flag=0;
    		    break;
    		}
    		
    	}
    	if(flag)cout<<"NO"<<endl;
    }
    return 0;
}
