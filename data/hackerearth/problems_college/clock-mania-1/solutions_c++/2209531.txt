#include<iostream>
#include<math.h>
using namespace std;

int main()
{
	
int t;
cin>>t;
while(t--)
{
	float h,m;
	cin>>h;
	cin>>m;
	if(h==12)
	h=0;
	h=h*60+m;
	int hdeg=h*0.5;
	int mdeg=6*m;
	
	int ans=abs(hdeg-mdeg);
	if(ans<(360-ans))
	cout<<ans<<endl;
	else
	cout<<360-ans<<endl;
}
		
	
		
	
	return 0;
	
}
