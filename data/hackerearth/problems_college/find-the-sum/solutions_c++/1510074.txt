vector<string> split(const string &s, char c) {
	vector<string> v;
	stringstream ss(s);
	string x;
	while (getline(ss, x, c)) v.push_back(x);
	return v;
}

	cerr << it -> substr((*it)[0] == ' ', it -> length()) << " = " << a << '\n';
	err(++it, args...);
}

ll find(ll x) {
	ll lb = 1, rb = (ll) 1E9;
	x = 8 * x + 1;
	ll y = (ll) sqrt(x);
	if (y & 1) {
		if (y * y <= x) return (y >> 1);
	}
	return (y >> 1) - 1;
}

ll rec(ll x) {
	if (x == 0) return 0;
	ll ans = 0;
	ll y = find(x);
	ll t = (y + 1) >> 1;
	ans = t * t;
	x -= (y * (y + 1)) / 2;
	if (x & 1) ans += y + 1;
	return ans;
}
