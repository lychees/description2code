#include <cstdio>

using namespace std;

int solve(int n){
	int ans = 0;
	for(int i = 1; i < n; i++){
		for(int j = i + 1; j <= n; j++){
			if((i ^ j) <= n) ans++;
		}
	}
	return ans;
}

int main()
{
    int T;
    scanf("%d", &T);
    int n;
    while(T--){
    	scanf("%d", &n);
    	printf("%d\n", solve(n));
    }
    return 0;
}
