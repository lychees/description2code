#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    long long t,n,m,a[3000],i,sum=0,c=0;
    cin >> t;
    while(t--)
    {
    	cin >> n >> m;
    	for(i=0;i<n;i++)
    	{
    		cin >> a[i];
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{
    		sum=sum+a[i];
    		c++;
    		if(sum>m)
    		{
    			break;
    		}
    	}
    	cout << c-1 <<endl;
    	sum=0;
    	c=0;
    }
    return 0;
}
