'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
st = raw_input()
n = input()
for i in range(n):
    x = raw_input().split()
    a = x[0]
    b = x[1]
    a = int(a)-1
    b = int(b)-1
    l = len(st)
    if (a == b):
        print "Yes"
    a = a % l
    b = b % l
    if (st[a] == st[b]):
        print "Yes"
    else:
        print "No"
