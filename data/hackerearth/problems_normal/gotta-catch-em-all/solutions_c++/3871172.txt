#include <bits/stdc++.h>
using namespace std;

int main()
{
    int k;
    cin>>k;
    vector<int> v(k,0);
    for(int i = 0;i<k;i++){
    	cin>>v[i];
    }
    
    sort(v.begin(), v.end());
    reverse(v.begin(), v.end());
    int res = 0;
    for(int i = 0;i<k;i++){
    	res = max(res, i+1+v[i]);
    }
    cout<<res+1;
    return 0;
}
