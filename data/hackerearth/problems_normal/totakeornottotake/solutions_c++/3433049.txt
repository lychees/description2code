#include <bits/stdc++.h>
using namespace std;
int main()
{
   long long int t,i,j,ni=-1,po=1,n,data,ower=1;
    char ch;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	ni=1;po=1;
    	for(i=0;i<n;i++)
    	{
    		cin>>ch;
    		if(ch=='N')
    		{
    		  	po=abs(ni)>abs(po) ? abs(ni):abs(po);
    			ni=(-1)*po;
    		}
    		else
    		{  
    			cin>>data;
    			if(ch=='*')
    			{
    				po*=data;
    				ni*=data;
    			}
    			else if(ch=='-')
    			{
    				ni-=data;
    			}
    			else if(ch=='+')
    			{
    				po+=data;
    			}
    			else
    			{
    				if(ni==1)
  		            ni/=data;
    			}
    		}
    	}
    	cout<<po;
    	cout<<"\n";
    }
    return 0;
}
