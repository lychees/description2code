/* Strong form of pigeon hole priciple
If n+1 objects are places into n boxes,then atleast one box contains one or more objects
*/
#include<iostream>
using namespace std;
int main()
{
	int tc=0;
	cin>>tc;
	while(tc-->0)
	{
		int n=0;
		cin>>n;
		int arr[n+1];
		long long int sum=0;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			sum=sum+arr[i];
		}
		cout<<sum-n+1<<endl;

	}
	return 0;
}
