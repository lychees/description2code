#include <iostream>
using namespace std;

int main()
{
    int testcase;
    cin>>testcase;
    int query,i,j,k=0;
    int result[testcase];
    result[testcase]={0};
    while(testcase)
    {
        int count=0;
    	cin>>query;
    	for(i=1;i<query;i++)
    	{
    		for(j=i+1;j<=query;j++)
    		{
    			if((i^j)<=query)
    			   count++;
    		}
    	}
    	result[k++]=count;
    	testcase--;
    }
    k--;
    i=0;
    while(k>=0){
    	cout<<result[i]<<endl;
    	k--;
    	i++;
    }
    //system("pause");
    return 0;
}
