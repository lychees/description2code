#include <iostream>
using namespace std;


int main()
{
   char string1[50000];
   int n,l,r,i;
   long int k;
   int q;
   cin>>n>>q;
   cin>>string1;
  long int string2[26][50000]={0};
   long int j=0;
    for(i=0;i<n;i++)
    {
            string2[(int)string1[i]-97][i]++;
            if(i>0)
            for(int q=0;q<26;q++)
            {
                string2[q][i]=string2[q][i]+string2[q][i-1];
            }
    }

   for(int m=0;m<q;m++)
   {
       cin>>l>>r>>k;

    j=0;
    for(i=0;i<26;i++)
    {
    	if(l>1)
        j=j+(string2[i][r-1]-string2[i][l-2]);
        else if(l==1)
        j=j+(string2[i][r-1]);
        
        
        if(j>=k)
        break;
    }
    if(j>=k)
        cout<<((char)(i+97))<<endl;
    else
    cout<<"Out of range"<<endl;
   }
   return 0;
}
