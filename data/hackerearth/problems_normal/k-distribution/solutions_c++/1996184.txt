#include<iostream>
using namespace std;
#define mod 1000000007

long long int dp[102][100005];
long long int func(int *a,int n,int k,long long int sum)
{
if(sum<2*k)
return 0;
for(int i=0;i<=n;i++)
for(long long int j=0;j<k;j++)
{
if(i==0&&j==0)
dp[i][j]=1;
else if(i==0)
dp[i][j]=0;
else if(j==0)
dp[i][j]=dp[i-1][j];
else if(sum-j>=k)
{
if(a[i-1]<=j)
dp[i][j]=(dp[i-1][j-a[i-1]]+dp[i-1][j])%mod;
else
dp[i][j]=dp[i-1][j];
}
}

long long int ans=0,temp=1;
for(int i=0;i<k;i++)
ans=(ans+dp[n][i])%mod;
for(int i=0;i<n;i++)
temp=(temp*2)%mod;
ans=temp-2*ans;
if(ans<0)
ans=ans+mod;
else
ans=ans%mod;
return ans;
}

int main()
{
ios_base::sync_with_stdio(false);
int n,k;
cin>>n>>k;
int a[n];
long long int sum=0;
for(int i=0;i<n;i++)
{
cin>>a[i];
sum=sum+a[i];
}
cout<<func(a,n,k,sum)<<endl;
return 0;
}
