#include <iostream>
#include <unordered_map>
#include <vector>
#include <algorithm>

using namespace std;


int main()
{
    int test ;
    scanf("%d",&test) ;
    
    while(test --)
    {
    	int n ;
    	scanf("%d",&n) ; 
    	
    	unordered_map<int,int> mp ; 
    	int m = 0 ; 
    	for(int i = 0 ; i < n ; i++)
    	{
    		int temp ;
    		scanf("%d",&temp) ; 
    		
    		mp[temp]++ ;
    		m = max(m,mp[temp]) ;
    		
    	}
    	vector<int> v ;
    	for(unordered_map<int ,int> :: iterator it = mp.begin() ; it != mp.end() ; it++)
    	{
    		if(it->second==m)
    			v.push_back(it->first);
    	}
    	sort(v.begin(),v.end()) ;
    	int size = v.size() ;
    	for(int i = size-1 ; i >= 0 ; i--)
    	{
    		cout << v[i] << " " ;
    	}
    	
    	cout << endl ;
    }
    
    return 0;
}
