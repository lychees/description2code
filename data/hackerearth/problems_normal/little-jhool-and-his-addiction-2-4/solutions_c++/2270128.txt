#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
#define pb push_back




int main()
{
    ll t,n,i,mi,ma,k,ans;
    cin>>t;

    while(t--)
    {
        cin>>n>>k;
        ll a[n];
        for(i=0;i<n;i++)cin>>a[i];
        sort(a,a+n);
        ma = -1;
        mi = 1000000000;

        for(i=0;i<n/2;i++)
        {
            ma = max(ma,a[i]+a[n-i-1]);
            mi = min(mi,a[i]+a[n-i-1]);
        }

        ans = ma-mi;
        cout<<ans<<"\n";

        if(ans>k)
        {
            cout<<"No more girlfriends!\n";
        }
        else if(ans==k)
        {
            cout<<"Lucky chap!\n";
        }
        else
        {
            cout<<"Chick magnet Jhool!\n";
        }

    }

    return 0;
}
