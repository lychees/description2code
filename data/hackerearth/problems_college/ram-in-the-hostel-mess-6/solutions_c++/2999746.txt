#include <bits/stdc++.h>
using namespace std;
int fib[1000003];

void precompute()
{
	int i;
	fib[1]=1;
	fib[2]=1;
	for(i=3; i<=100003; i++)
		fib[i]=fib[i-1]+fib[i-2];
}


int main()
{
	int t, n;
	long long int temp;
	cin>>t;
	precompute();
	while(t--)
	{
		cin>>n;
		temp = fib[n+2];
		temp = temp*temp;
		cout<<temp<<endl;
	}

	return 0;
}
