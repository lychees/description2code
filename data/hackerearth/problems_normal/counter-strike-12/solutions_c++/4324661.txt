#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    int t,n,m,d;ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>d;
        int posx[n];
        int posy[n];
        int tarx[m];
        int tary[m];
        int shot[m];
        memset(shot,0,sizeof(shot));
        for(int i = 0;i<n;i++)
        {
            cin>>posx[i]>>posy[i];
        }
        for(int j=0;j<m;j++)
        {
            cin>>tarx[j]>>tary[j];
        }
        int count=0;int flag = 0;
        for(int p = 0;p<n;p++)
        {
            if(count>m/2)
            {
                flag = 1;
                break;
            }
            for(int q = 0;q<m;q++)
            {
                if(shot[q]==0&&(abs(posx[p]-tarx[q])+abs(posy[p]-tary[q]))<=d)
                {
                    shot[q]=1;
                    count++;
                }
                if(count>m/2)
                {
                    flag = 1;
                    break;
                }
            }
        }
        if(flag == 0)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
}
