#include<bits/stdc++.h>
using namespace std;
string str,tmp,temp;
int lps()
{
   int n = str.size();
   int i, j, cl;
   int L[n][n];
   for (i = 0; i < n; i++)
      L[i][i] = 1;
    for (cl=2; cl<=n; cl++)
    {
        for (i=0; i<n-cl+1; i++)
        {
            j = i+cl-1;
            if (str[i] == str[j] && cl == 2)
               L[i][j] = 2;
            else if (str[i] == str[j])
               L[i][j] = L[i+1][j-1] + 2;
            else
               L[i][j] = max(L[i][j-1], L[i+1][j]);
        }
    }

    return L[0][n-1];
}
int main()
{
    cin.sync_with_stdio(false);
    cin.tie(0);
    int t,n,i;
    cin>>t;
    while(t--)
    {
        int ans=0;
        cin>>tmp;
        for(i=1;i<=tmp.size();i++)
        {
            temp=tmp.substr(tmp.size()-i,i);
            reverse(temp.begin(),temp.end());
            str=tmp.substr(0,tmp.size()-i)+temp;
            ans=max(ans,lps());
        }
        cout<<ans<<'\n';
    }
    return 0;
}
