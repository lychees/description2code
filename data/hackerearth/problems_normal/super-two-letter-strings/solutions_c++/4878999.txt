#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define MOD 1000000007
#define MAX 100000
ll dp[MAX];
int main()
{
  int test;cin>>test;int t;
  int n,p;int i,j;
  for(t=0;t<test;t++)
  {
    memset(dp,0,sizeof(dp));
    cin>>n>>p;
    dp[0]=1;
    for(i=1;i<=n;i++)
    {
      dp[i]=0;
      for(j=0;j<p;j++)
      {
        if (i-j>=0)
        {
          dp[i]=(dp[i]+dp[i-j-1])%MOD;
        }
      }

    }
    cout<<dp[n]%MOD<<"\n";
  }//test

return 0;
}
