#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<vector>
#include<map>
#include<set>

using namespace std;

main()
{
	int n,ct=0;
	cin>>n;
	
	for(int i=4;i<=n;i++)
	{
		if(i%2==0)
		cout<<i<<endl;
		else
		{
		for(int j=1;j<=i;j++)
		{
			if(i%j==0)
			ct++;
		}
		if(ct!=2)
		cout<<i<<endl;
		}
		ct=0;
	}
}
