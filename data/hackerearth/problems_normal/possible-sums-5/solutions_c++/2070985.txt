#include <bits/stdc++.h>
using namespace std;
int a[100];
bool arr[101][10001];
int main(){
	int t,n,i,j,sum;
	scanf("%d",&t);
	while(t--){
		sum=0;
		scanf("%d",&n);
		a[0]=0;
		for(i=1;i<=n;i++){
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		sort(a,a+n);
//		bool arr[n+1][sum+1];
		for(i=0;i<=n;i++)	arr[i][0]=true;
		for(i=1;i<=sum;i++)	arr[0][i]=false;
		for(i=1;i<=n;i++){
			for(j=1;j<a[i];j++){
				arr[i][j]=arr[i-1][j];
			}
			if(j==a[i])	arr[i][j]=true;
			for(j=a[i]+1;j<=sum;j++){
				arr[i][j]=(arr[i-1][j] || arr[i-1][j-a[i]]);
				//if(j%a[i]==0)	arr[i][j]=arr[i-1][j];
				//else	arr[i][j]=arr[i][j-a[i]];
			}
		}
		int count=0;
		for(i=0;i<=sum;i++){
			if(arr[n][i])	count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
