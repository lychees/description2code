#include <iostream>
#include<string>
using namespace std;

int main()
{
   long long int test,i,j,k,t,l,z;
    cin>>test;
    string str;
    while(test--)
    {
    	cin>>str;
    	j=0;
    	char ans[1000000];
    	for(i=0;i<str.length();i++)
    	{
    		if(str[i]<'0'||str[i]>'9')
    		{
    			ans[j]=str[i];
    			j++;
    		}
    		else
    		{
    			t=j;
    			k=str[i]-'0';
    			for(l=0;l<k-1;l++)
    			{
    				for(z=0;z<t;z++)
    				{
    					ans[j]=ans[z];
    					j++;
    				}
    			}
    		}
    	}
    	for(i=0;i<j;i++)
    	cout<<ans[i];
    	cout<<endl;
    }
    return 0;
}
