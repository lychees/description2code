#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main()
{

    char str[10000];
    int i,j,cnt=0,tot=0,t;
    scanf("%d",&t);
    while(t--)
        {
            scanf("%s",str);
            i=0;
            j=strlen(str);
            j--;
            cnt=tot=0;
            while(i<j)
                {
                       cnt=(labs((*(str+i))-(*(str+j))));
                        tot+=cnt;
                        i++;
                        j--;
                }
        printf("%d\n",tot);
    }
    return 0;
}
