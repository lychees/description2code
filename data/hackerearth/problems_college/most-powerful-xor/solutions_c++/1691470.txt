struct node {
	ll x;
	ll y;
};
ll gcd(ll a, ll b)
{
	return b == 0 ? a : a > b ? gcd(b, a%b) : gcd(a, b%a);
}
bool myfunc(struct node a,struct node b)
{
	if(a.x == b.x) {
		return a.y < b.y;
	} else {
		return a.x < b.x;
	}
}
int main()
{
	int t;
	S(t);
	while(t--) {
		int l, r;
		S(l);
		S(r);
		int i;
		int j;
		int ans = 0;
		for(i = l; i <= r; i++) {
			for(j = i; j <= r; j++) {
				ans = max(ans, i^j);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
