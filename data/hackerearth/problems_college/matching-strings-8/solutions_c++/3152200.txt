#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	string a,b;
   	cin>>a;
   	cin>>b;
   	int i=0,cnt=0;
   	while(a[i]!='\0'&&b[i]!='\0')
   	{
   		if(a[i]==b[i])
   		cnt++;
   		i++;
   	}
   	cout<<cnt<<endl;
   }
    return 0;
}
