#include <iostream>
using namespace std;
long long mod = 1000000007;
long long table[510][510];

long long solution(int N, int H)
{    
    for(int i=0; i<=H; i++)
        table[0][i] = 1;
    for(int i=1; i<=N; i++)
        for(int j=1; j<=H; j++)
            for(int k=1; k<=i; k++)
                table[i][j] = (table[i][j] + table[k-1][j-1]*table[i-k][j-1])%mod;

    return table[N-1][H-1];
}
int main()
{
    int N,H;
    cin>>N>>H;
    long long ans = 0;
    solution(N,H);
    for(int k=1;k<=N;k++)
        ans = (ans + ((k*table[k-1][H-1])%mod*table[N-k][H-1])%mod)%mod;
    cout<<ans<<endl;
    return 0;
}
