int INP() {
	int n,flag;
	char c;
	n = 0;
	flag = 1;
	c = getchar();
	while (c < 33) {
		c = getchar();
	}
	if (c == '-') {
		flag = -1;
		c = getchar();
	}
	while (c > 32) {
		n = n * 10 + c - '0';
		c = getchar();
	}
	n = n * flag;
	return n;
}

int GCD (int a,int b) {
	return(a % b == 0 ? b : GCD(b,a % b));
}
