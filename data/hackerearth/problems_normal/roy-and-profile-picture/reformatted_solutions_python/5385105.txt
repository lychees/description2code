l = int(raw_input())
n = int(raw_input())
for i in range(0, n):
    x = raw_input().split()
    if int(x[0]) >= l and int(x[1]) >= l:
        if int(x[0]) == int(x[1]):
            print "ACCEPTED"
        else:
            print "CROP IT"
    else:
        print "UPLOAD ANOTHER"
