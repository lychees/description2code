#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int st[n];
        for(int i=0;i<=n;i++)
            st[i]=0;
        int top=0;
        st[0]=-1;
        for(int i=0;i<n;i++)
        {
            int temp;
            cin>>temp;
            if(temp>st[top])
                st[++top]=temp;
            else
            {
                int low=1;
                int high=top;
                while(low<=high)
                {
                    int mid=(low+high)/2;
                    if(temp>st[mid])
                        low=mid+1;
                    else
                        high=mid-1;
                }
                st[low]=temp;
            }
        }
        int l;
        cin>>l;
        if(l>top)
            cout<<"-1\n";
        else
            cout<<st[l]<<"\n";
    }
}
