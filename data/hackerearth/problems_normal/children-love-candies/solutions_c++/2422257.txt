#include <bits/stdc++.h>
using namespace std;
typedef long long int i64;
int main()
{
    i64 Q_cases,Initial_num,Days_check,X;
    scanf("%lld",&Q_cases);
    while(Q_cases--)
    {
    	scanf("%lld %lld",&Initial_num,&Days_check);
    	X=Initial_num;    	
    	while(Days_check--)
    	{
    		//for 9pm child
    		if(X%2==0)
    		X-=(X/2);
    		else
    		X-=((X+1)/2);
    		//for 10 pm child
    		X-=X/4;
    	}
    	printf("%lld\n",X);
    }
    return 0;
}
