#include<iostream>
using namespace std;
int main()
{
	float a,b,c,d;
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>c>>d;
		a=(c+d)/2;
		b=(c-d)/2;
		if(((int)a==a)&&((int)b==b))
		{	
		if(a>0&&b>0)
		{
			if(a>b)
				cout<<a<<" "<<b<<endl;
				else
				cout<<b<<" "<<a<<endl;
		}
		else
			cout<<"impossible"<<endl;
		}
		else
		cout<<"impossible"<<endl;
	}
return 0;
}
