#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while (t--) {
    	int n,p;
    	scanf("%d%d",&n,&p);
    	int a[n];
    	int ctr = 0,f = 0;
    	for (int i = 0;i < n; i++) {
    		scanf("%d",&a[i]);
    		if (a[i] == p) {
    			f = 1;
    		}
    		if (a[i] > p)
    			ctr++;
    	}
    	if (f)
    		printf("YES %d\n",ctr);
    	else
    		printf("NO %d\n",0);
    }
    return 0;
}
