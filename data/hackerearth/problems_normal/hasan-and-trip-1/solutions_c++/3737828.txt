#include <bits/stdc++.h>
using namespace std;
#define num long long int

num X[3010],Y[3010],h[3010],vis[3010];
double ans[3010],d[3010];

long double dist(num x,num y)
{
	return sqrt(pow(X[x]-X[y],2)+pow(Y[x]-Y[y],2));
}

double solve(num x)
{
	if(vis[x]==1)
		return ans[x];
	else
	{
		
		for(num i=1;i<x;i++)
		{
			num qty=solve(i)-dist(x,i);
			if(ans[x]<qty)
			{
				ans[x]=qty;
				d[x]=d[i]+dist(x,i);
			}
		}
		vis[x]=1;
		return ans[x]+h[x];
	}
		
}


int main() {
	// your code goes here
	num n;
	cin>>n;
	cout<<fixed;
	cout<<setprecision(6);
	for(num i=1;i<=n;i++)
	{
		cin>>X[i]>>Y[i]>>h[i];
	}
	ans[1]=h[1];
	d[1]=0;
	for(num i=2;i<=n;i++)
	{
		ans[i]=-10000000000;
		for(num j=1;j<=i;j++)
		{
			ans[i]=max((long double)ans[i],ans[j]-dist(i,j));
		}
		ans[i]+=h[i];
	}
	cout<<ans[n]<<endl;
	return 0;
}
