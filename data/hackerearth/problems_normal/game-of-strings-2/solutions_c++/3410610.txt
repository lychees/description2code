#define N 1000000
int pf[N];
void pre(string s) {
    int n = (int) s.length();
    reset(pf,0);
    int  j = 0;
    for (int i=1; i<n; ++i) {
//        int j = pf[i-1];
        while (j > 0 && s[i] != s[j])
            j = pf[j-1];
        if (s[i] == s[j])  ++j;
        pf[i] = j;
    }
//    printA(pf,0,s.size());
}

        string s;int k,maxi,ans=0;

        pre(s);
        set<int> alps;
        int j = pf[s.size()-1];
        while(j){alps.insert(j);j = pf[j-1];}
        for (int i = 1;i < k; i++) {
            int cur = ((alps.find(pf[i])!=alps.end())?pf[i]:0);
            if (ans < cur ) {
                maxi = i;
                ans = cur;
