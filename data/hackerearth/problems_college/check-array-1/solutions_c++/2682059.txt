#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n],b[n],c[n];
		/*for(int i=0;i<n;i++)
		b[i]=0;*/
		for(int i=0;i<n;i++)
		cin>>a[i];
		int s=0,g=0;
		for(int i=0;i<n;i++)
		{	s+=a[i];
			b[i]=s;
		}
		for(int i=n-1;i>=0;i--)
		{
			g+=a[i];
			c[i]=g;
		}
		bool f=true;
		for(int i=0;i<n;i++)
		{
			if((b[i]-a[i])==(c[i]-a[i]))
			f=false;
		}
		if(f)
		cout<<"NO\n";
		else
		cout<<"YES\n";
		
	}
	return 0;
}
