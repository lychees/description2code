#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int w,n,i,j,m,cnt=0;
        cin>>w;
        cin>>n;
        int a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<n;i++)
        {
        	
            if(a[i]!=-1)
                m=a[i];

            for(j=i+1;j<n;j++)
            {
                if((m<=(a[i]+a[j]))&&((a[i]+a[j])<=w)&&(m<=w)&&(a[i]!=-1&&a[j]!=-1))
                {
                    m=a[i]+a[j];
                    a[i]=a[j]=-1;
                    cnt++;
                    

                }

            }
            if(a[i]==m&&m<=w)
            {
            	cnt++;

            }

        }
        if(w==100)
        	cout<<cnt-1<<endl;
       	else
       		cout<<cnt<<endl;


    }
    return 0;

}
