#include <bits/stdc++.h>
using namespace std;
vector<int> v;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		v.clear();
		int n;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			int x;
			scanf("%d",&x);
			v.push_back(x);
		}
		sort(v.begin(),v.end());
		int maxi = 0;
		int ans[3];
		vector<int>:: iterator it;
		for(int i=n-1;i>1;i--)
		{
			if(v[i-2]+v[i-1]>v[i])
			{
				maxi = 1;
				ans[0] = v[i-2];
				ans[1] = v[i-1];
				ans[2] = v[i];
				i = 0;
			}
		}
		if(maxi==0)
			printf("-1\n");
		else printf("%d %d %d\n",ans[0],ans[1],ans[2]);
	}
}
