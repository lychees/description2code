#include <cstdio>
#include <algorithm>
using namespace std;

int go(int n, int pos, int skip) {
    if (!skip) {
        if (pos <= (n+1)/2)
            return 2*pos - 1;
        return 2*go(n/2, pos - (n+1)/2, n & 1);
    }
    if (pos <= n/2)
        return 2*pos;
    return 2*go((n+1)/2, pos - n/2, 1 - (n & 1)) - 1;
}

int main() {
    int T, n, pos;
    scanf("%d", &T);
    while (T--) {
        scanf("%d %d", &n, &pos);
        printf("%d\n", go(n, pos, 0));
    }
}
