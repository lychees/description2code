#include <iostream>
using namespace std;

int main()
{
    int n,m,a[100000],b[100000];
    long long int sum = 0,p=0;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    	cin>>a[i];
    for(int j=0;j<m;j++)
    	cin>>b[j];
    for(int j=0;j<m;j++)
    	sum+=b[j];
    for(int i=0;i<n;i++)
    	p+=a[i]*sum;
    cout<<p;
    return 0;
}
