

#include<bits/stdc++.h>
using namespace std;

int n,m,k,x;

int main()
{
    int t=0;
    cin>>t;
    while(t){
	int i=0,tmp=0;
	n=0,m=0,k=0;
	scanf("%d%d%d",&n,&m,&k);
	
		if(k==1&&m!=k) x=1;
		else x=0;
		for(i=2;i*i<=m;i++)
		{
			if(m%i==0&&(m/i>=k||i>=k)) x=1;
		}
		if(x&&n%2==1) printf("Bernard\n");
		else printf("Havanese\n");
		t--;
	
	}
	return 0;
}
