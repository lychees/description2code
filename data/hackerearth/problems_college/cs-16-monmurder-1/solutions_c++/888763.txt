const long MOD = 1000000007;
const int MAX = 1000005;
bool a[MAX];
int p[MAX], A[MAX], k = 0;

void s(){
    ll i, j;
    for(i = 2;i < MAX;i++)
    {
        if(a[i] == false)
        {
            p[k++] = i;
            for(j = i*i;j < MAX;j += i)
            {
                a[j] = true;
            }
        }
    }
}

int main()
{
    int n, ans, t;
    //freopen("input.txt", "r", stdin);
    s();
    read(t);
    for(;t--;)
    {
        read(n);
        for(int i = 0;i < k;i++)
        {
            A[i] = 0;
            while(n % p[i] == 0)
            {
                A[i]++;
                n /= p[i];
            }
        }
        ans = A[0];
        for(int i = 1;i < k;i++)
        {
            ans ^= A[i];
        }
        if(ans == 0)
        {
            cout<<"Pappu Bhujia"<<endl;
        }
        else
            cout<<"Printf{}"<<endl;
    }
    return 0;
}
