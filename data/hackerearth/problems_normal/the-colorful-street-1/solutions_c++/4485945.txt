#include<iostream>
using namespace std;
int n;
int **cost;
int dp[21][3];
int mincost(int house,int prev)
{
    if(house==n)
        return 0;
    if( prev!=0 && dp[house][prev-1] != -1)
    {
        return dp[house][prev - 1];
    }
    int ans=10000000;
    for(int colour=1;colour<=3;colour++)
    {
        if((colour)!=prev)
        {
            ans=min(ans,mincost(house+1,colour)+cost[house][colour -1]);
        }
    }
    return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cost=new int*[n];
        int i,j;
        for(i=0;i<n;i++)
        {
           cost[i] = new int[3];
           cin>>cost[i][0]>>cost[i][1]>>cost[i][2];
        }
        for(i=0;i<n;i++)
        {
            for(j=0;j<3;j++)
                dp[i][j]=-1;
        }
        cout<<mincost(0,0)<<endl;
    }
}
