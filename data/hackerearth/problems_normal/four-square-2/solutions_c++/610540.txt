ll divisors(ll n)
{
    ll i=1;
    while(i*i < n)
    {
        if(n%i==0)
            {
                if(i%2==0) e=e+i;
                else o=o+i;
                if((n/i)%2==0) e=e+n/i;
                else o=o+n/i;
            }
        i++;
    }
