#include <iostream>
using namespace std;

int main()
{
    int q;
    cin>>q;
    while(q--)
    {
    	long long int n, t, i;
    	cin>>n>>t;
    	for(i=0; i<t; ++i)
    	{
    		if(n%2==0)
    		n = n - (n/2);
    		else
    		n = n - ((n+1)/2);
    		
    		n = n - (n/4);
    	}
    	cout<<n<<endl;
    }
    return 0;
}
