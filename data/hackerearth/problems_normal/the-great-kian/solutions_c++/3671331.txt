#include <iostream>
using namespace std;
typedef long long int lld;
int main()
{
    lld t,sum1=0,sum2=0,sum3=0;
    scanf("%lld",&t);
    lld ar[t];
   	for(int i=0;i<t;i++)
   	{
   		scanf("%lld",&ar[i]);
   		if(i%3==1)
   		{
   			sum1+=ar[i];
   		}
   		else if(i%3==2)
   		{
   			sum2+=ar[i];
   		}
   		else
   		{
   			sum3+=ar[i];
   		}
   	}
   	printf("%lld %lld %lld",sum3,sum1,sum2);
   	
    return 0;
}
