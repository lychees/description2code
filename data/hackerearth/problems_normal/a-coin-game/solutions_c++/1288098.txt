#include <bits/stdc++.h>
using namespace std;

int done[1000006], lose[1000006][2];

int find(int N1, int N2, int low, int high)
{
	if(low<=high)
	{
		int mid = (low+high)/2;
		
		if(lose[mid][0]>N1)
		{
			if(find(N1, N2, low, mid-1))
				return 1;
		}
		
		else if(lose[mid][0]<N1)
		{
			if(find(N1, N2, mid+1, high))
				return 1;
		}
		
		else
		{
			if(lose[mid][1]==N2)
				return 1;
		}
	}
	
	return 0;
}

int main()
{
	int j = 0;
	
    for(int i=0; i<=1000000; i++)
    {
    	if(done[i]==0)
    	{
    		done[i] = 1;
    		
    		lose[j][1] = i;
    		lose[j][0] = i+j;
    		
    		if(i+j<=1000000)
	    		done[i+j] = 1;
    		
    		j++;
    	}
    }
    
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int N1, N2;
    	scanf("%d %d", &N1, &N2);
    	
    	if(N1<N2)
    		swap(N1, N2);
    	
    	if(find(N1, N2, 0, j-1))
    		printf("Don't Play\n");
    	else
    		printf("Play\n");
    }
    
    return 0;
}
