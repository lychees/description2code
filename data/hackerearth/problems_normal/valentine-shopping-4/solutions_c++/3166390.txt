#include <iostream>
#include <climits>
using namespace std;

int main()
{
    int t,N,M;
    double a,b,c;
    double disc,minval;
    int shop;
    cin >> t;
    for(int i =0; i< t; i++){
    	cin >> N >> M;
    	for(int j = 0; j < N; j++){
    		minval = INT_MAX;
    		for(int k = 1; k<= M;k++){
    			cin >> a >> b >> c;
    			//cout << a << ' ' << b << ' ' << c <<endl;
    			disc = (1-(a/100))*(1-(b/100))*(1-(c/100));
    			//cout << disc <<endl;
    			if(disc < minval){
    				minval = disc;
    				shop = k;
    			}
    		}
    		cout << shop << " " ;
    	}
    	cout << endl;
    }
    return 0;
}
