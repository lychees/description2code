int main()
{

    #endif
	int t,n,i,ans1,ans2;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		scanf("%s",a);
		//case 1 is that start with boy
		ans1=0;
		for(i=0;i<2*n;i++)
		{
			if(i%2==0)
			{
				//here should be boy
				if(a[i]=='B')
					continue;
				ans1++;
			}
			else
			{
				//here should be girl
				if(a[i]=='G')
					continue;
				ans1++;
			}
		}
		//case 2 is that start with girl
		ans2=0;
		for(i=0;i<2*n;i++)
		{
			if(i%2==1)
			{
				//here should be boy
				if(a[i]=='B')
					continue;
				ans2++;
			}
			else
			{
				//here should be girl
				if(a[i]=='G')
					continue;
				ans2++;
			}
		}
		ans1/=2;
		ans2/=2;
		printf("%d\n",min(ans1,ans2));
	}
    return 0;
}
