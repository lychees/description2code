

for i in range(input()):
    n = input()
    total = n*12
    freq = total/26
    rem = total % 26
    frequency = []
    value = []
    value = map(int, raw_input().strip().split())
    for i in xrange(1, 27, 1):
        frequency.append(freq)
    for i in range(rem):
        frequency[i] += 1
    answer = 0
    for i in range(26):
        answer += frequency[i]*value[i]
        answer += (frequency[i]*(frequency[i]-1))/2
    print answer
