#include <iostream>
#include <cstring>
using namespace std;

bool prime[1000001];

void sieve(){
	
	memset(prime,true,sizeof(prime));
	int p;
	for(p=2;p*p<=1000000;p++)
	{
		if(prime[p])
		{
			int i;
			for(i=2*p;i<=1000000;i+=p)
			prime[i]=false;
		}
	}

}

int main()
{
    sieve();
    
    int t;
    cin>>t;
    while(t--){
    	
    int n;
    cin>>n;
    int k=0;
    int j=n/2;
    int ans;
    while(k==0&&j>=2)
    {
    	if(prime[j]&&prime[n-j]&&j!=n-j)
    	{
    		ans=j;
    		k++;
    	
    	}
    	j--;
    }
    if(k==0)
    cout<<"-1 -1\n";
    else cout<<ans<<" "<<n-ans<<endl;}
    return 0;
}
