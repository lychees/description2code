#include <bits/stdc++.h>
using namespace std;

int c[(int)1e5 + 6];

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);
  
  int n; cin >> n;
  for (int i = 0; i < n; ++i) {
    cin >> c[i];
    if (i) c[i] += c[i - 1];
  }

  int q; cin >> q;
  while (q--) {
    int x; cin >> x;
    int l = 0, r = n - 1;
    while (l < r) {
      int m = (l + r) >> 1;
      if (c[m] >= x) r = m;
      else l = m + 1;
    }
    cout << ++l << endl;
  }


  return 0;
}
