#include<bits/stdc++.h>
using namespace std;
bool fun(const pair<int,string> &a, 
               const pair<int,string> &b)
{	//	if(a.first > b.first)
	if(a.first==b.first)
	return (a.second < b.second)
	;
	else
       return (a.first > b.first);
}
int main()
{
	int i,n;
	long int m;
	string str;
	 cin>>m;
	vector< pair <int,string> > v;
	for (int i=0; i<m; i++)
	{	
		cin>>str>>n;
        v.push_back( make_pair(n,str));
	}
		sort(v.begin(),v.end(),fun);
		
		for (int i=0; i<m; i++)
    {
        cout << v[i].second << " "
             << v[i].first << endl;
    }
 
    return 0;
}
