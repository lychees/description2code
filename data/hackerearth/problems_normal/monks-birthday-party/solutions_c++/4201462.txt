#include <iostream>
#include <set>
#include <string>
using namespace std;

int main()
{
	int tc;
	cin>>tc;
	while(tc--){
    int n;
    cin>>n;
    set <string> s;
    set <string>::iterator it;
    string guest;
    while(n--)
    {
        cin>>guest;
        it = s.find(guest);
        if(it==s.end())
            s.insert(guest);
    }
    for(it=s.begin();it!=s.end();it++)
            cout<<*it<<endl;
	}
}
