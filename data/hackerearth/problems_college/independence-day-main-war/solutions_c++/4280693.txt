#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n,k;
    	cin >> n;
    	k = n;
    	double X = 0,Y = 0;
    	while(k--)
    	{
    		double x,y;
    		cin >> x >> y;
    		X += x;
    		Y += y;
    	}
    	printf("(%.2f , %.2f)\n",X/n,Y/n);
    }
    return 0;
}
