#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	long long n,m;
    	cin>>n>>m;
    	string output = "";
    	int counter = 0;
    	while(m)
    	{
    		if(m&0x1)
    		{
    			if(output!="")
    				output ="("+to_string(n)+ "<<"+to_string(counter)+") + " + output;
    			else
    				output ="("+to_string(n)+ "<<"+to_string(counter)+")";
    		}
    		m = m>>1;
    		counter++;
    	}
    	cout<<output<<endl;
    }
    return 0;
}
