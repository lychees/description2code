#include<iostream>
#define ll long long
ll count(ll c)
{
	return (c*(c+1))/2;
}
using namespace std;
int main()
{
	int t;
	cin>>t;
	string s;
	for(int i=0;i<t;i++)
	{
		cin>>s;
		bool visited[256]={0};
		ll j=0;
		ll c=0;
		while(s[j])
		{
		  if(visited[s[j]]==0)
		  {
		    c++;
		    visited[s[j]]=1;
		  } 
		  
		  j++;	
		}
		cout<<count(c)<<"\n";
	}
	system("pause");
}
