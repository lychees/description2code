#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	long long int n,m,i,j,k,flag=0;
	cin>>n>>m;
	long long int a[n];
	for(i=0;i<n;i++)
		cin>>a[i];
	sort(a, a+n);
	for(i=0;a[i]<=m;i++){
		for(j=i+1;a[j]<=m-a[i];j++){
		for(k=j+1;a[k]<=m-a[i]-a[j];k++){
			if(a[i]+a[j]+a[k]==m){
				flag=1;
				break;
			}
		}
		if(flag==1)
			break;
		}
		if(flag==1)
			break;
	}
	if(flag==1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
	return 0;
}
