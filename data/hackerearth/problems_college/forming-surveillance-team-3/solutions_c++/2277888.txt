#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstdlib>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		int i;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int c=0;
		sort(a,a+n);
		for(i=0;i<n;i++)
		{
			if(a[i]!=a[i+1])
			{
				c++;
			}
		}
		cout<<c<<"\n";
	}

}
