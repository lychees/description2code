#include <bits/stdc++.h>

using namespace std;

int arr[240000];

int gcd(int a, int b) {
	if (b == 0) return a;
	return gcd(b, a%b);
}

int main()
{
	for (int h = 0; h < 24; h++) {
		for (int m = 0; m < 60; m++) {
			for (int s = 0; s < 60; s++) {
				int g = gcd(h, m);
				g = gcd(g, s);
				if (g == 1) {
					arr[h*10000 + m*100 + s] = 1;
				}
			}
		}
	}

	for (int i = 235959; i >= 0; i--) {
		arr[i] += arr[i+1];
	}

	int t, x, y;
	int h, m, s;
	// 23 46 22 = 85582
	cin >> t;
	while (t--) {
		cin >> h >> m >> s;
		x = arr[h*10000 + m*100 + s];
		y = 86400 - (h*3600 + m*60 + s) - x;
		if (x != 0 && y != 0) {
			h = gcd(x, y);
			x = x / h;
			y = y / h;
		}
		printf("%d:%d\n", y, x);
	}
}
