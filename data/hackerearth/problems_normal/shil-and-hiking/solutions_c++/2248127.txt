#include<bits/stdc++.h>
using namespace std;
int main(){
long long n;
cin>>n;
double x[n],y[n];
double m,t;
long long ans=0;
for(int i=0;i<n;i++) cin>>x[i]>>y[i];
for(int i=1;i<n;i++){
    m=INFINITY;
    for(int j=i-1;j>=0;j--){
        t=(y[j]-y[i])/(x[j]-x[i]);
        if(t<=m){
            m=t;
            ans++;
        }
    }
}
cout<<ans;
}
