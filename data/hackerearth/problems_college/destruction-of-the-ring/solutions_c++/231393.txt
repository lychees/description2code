#include<stdio.h>
#define MOD 1000000007
#define LL long long
LL factorial[2000005];
/*
Tutorial TOPCODER
http://community.topcoder.com/tc?module=Static&d1=tutorials&d2=primeNumbers
a^?(n)=1(mod n).
a^(?(n)-1)=a^(-1) mod n
//Invers modulo
*/
LL pow(LL a, LL b, LL mod) {
    LL x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
LL modInverse(LL a, LL m){
    return pow(a,m-2,m);
}
int main()
{
        int t,a,b,n,i,m;
        factorial[1]=1,factorial[0]=1;
        for (i=2;i<2000005;i++)
                factorial[i]=(factorial[i-1]*i)%(MOD);
        scanf("%d",&t);
        while(t--)
        {
                scanf("%d%d",&n,&m);
                /*
                Ans=(n+m-2)!/(n-1)!
                */

                LL nCr=((factorial[n+m-2])*(modInverse(factorial[n-1],(MOD))))%(MOD);
                nCr=((nCr)*(modInverse(factorial[m-1],(MOD))))%(MOD);
                printf("%lld\n",nCr);
        }
        return 0;
}
