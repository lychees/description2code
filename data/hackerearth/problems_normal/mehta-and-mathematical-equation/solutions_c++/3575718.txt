#include<bits/stdc++.h>
using  namespace std;
typedef long long int lli;
lli dp[70][3][3];
lli p;
int num[70];
int indexx=0;
#define mod 1000000007
int conv(lli val)
 {
 	
 	while(val!=0)
 	{
 		  num[indexx]=val%2;
 		  val/=2;
 		  indexx++;
	 }
 	return 0;
 }
 

lli solve(int idx,int cary,int flag)
{
	// cout<<idx<<" "<<cary<<" "<<flag<<endl;
	  if(idx>=indexx)
	  {
	  	 if(cary==0 && flag==1)
	  	    {
	  	 	 return 1;
		    }
		   else
		   {
		   	return 0;
		   }
	  }
	  
	  else if(dp[idx][cary][flag]!=-1)
	   {
	   	//cout<<" call comp "<<dp[idx][cary][flag]<<endl;
	   	 return dp[idx][cary][flag];
	   }
	   else
	   {
	   	  lli ret=0;
	   	  
	   	  for(int i=0;i<=1;i++)
	   	   {
	   	   	for(int j=0;j<=1;j++)
	   	   	{
	   	   	 for(int k=0;k<=1;k++)
	   	   	  {
	   	   	 	if((cary+i+j+k)%2==num[idx])
	   	   	 	 {
	   	   	 	 	
	   	   	 	  	ret=(ret+solve(idx+1,(i+j+k+cary)/2, (flag |(i & j & k))))%mod;
				 }
			  }
			}
		   }
		    
			
			dp[idx][cary][flag]=ret;
			return ret;
	   }
}
int main()
{
	
	int t;
	  cin>>t;
	  while(t--)
	  {
	  	 indexx=0;
	  	 memset(dp,-1,sizeof dp);
	  	 cin>>p;
	  	 conv(p);
	     lli ans=solve(0,0,0);
	     cout<<ans<<endl;
	  }
}
