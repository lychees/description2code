#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,j,i,sum=0;
    cin>>n;
    vector<int>v,w;
    for(i=0;i<n;i++)
    {
    	cin>>j;
    	v.push_back(j);
    }
    while(n!=2)
    {
    for(i=1;i<n-1;i++)
    {
    	j=v[i-1]*v[i+1];
    	w.push_back(j);
    }
    int k,u,max=0;
    for(i=0;i<w.size();i++)
    {
    	if(max<w[i])
    	{
    		max=w[i];
    		k=i;
    	}
    }
    cout<<endl;
    u=k+1;
    sum+=max;
    v.erase(v.begin()+(u));
    n=v.size();
    w.clear();
    }
    cout<<sum<<endl;
    return 0;
}
