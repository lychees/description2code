#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        long long x, y;
        cin>>x>>y;
        cin>>n;
        long a[n];
        for(int i=0; i<n; i++)
            cin>>a[i];
        sort(a, a+n);
        long long s=0, cx=0, cy=0;
        int i=0;
        while(s<=x)
        {
            s+=a[i];
            i++;
            cx++;
        }
        s=0;
        i=0;
        while(s<=y)
        {
            s+=a[i];
            i++;
            cy++;
        }
        if(cx>cy)
            cout<<"Chef Won"<<endl;
        if(cx==cy)
            cout<<"Tie"<<endl;
        if(cx<cy)
            cout<<"Jhool Won"<<endl;
    }
    return 0;
}
