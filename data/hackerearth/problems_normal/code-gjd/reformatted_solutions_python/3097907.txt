for _ in range(input()):
    noe = input()
    elem = map(int, raw_input().split())
    elem.sort(reverse=True)
    if noe == 1:
        print "0"
    else:
        sums = elem[0] + elem[1]
        last = sums
        for i in range(2, noe):
            last += elem[i]
            sums += last
        print sums
