#include <iostream>
using namespace std;
void swap(int *a,int *b){
	int temp = *a;
	*a = *b;
	*b = temp;
}
int main()
{
	int n;
    int arr[100000][2];
    cin>>n;
    int i=0;
    while(i<n){
    	arr[i][0]=i+1;
    	int a,b;
    	cin>>a>>b;
    	arr[i][1] = a+b;
    	i++;
    }
 /*   for(int m=0;m<n;m++){
    	cout<<arr[m][1]<<" ";
    	cout<<arr[m][0]<<" "<<endl;
    }*/
    for(int j=0;j<n;j++){
    	for(int k=0;k<n-j-1;k++){
    		if(arr[k][1]>arr[k+1][1]){
    			swap(&arr[k][1],&arr[k+1][1]);
    			swap(&arr[k][0],&arr[k+1][0]);
    		}
    	}
    }
    for(int m=0;m<n;m++){
    	cout<<arr[m][0]<<" ";
    }
    return 0;
}
