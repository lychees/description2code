#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
     ll n,i,a,b,j;
     double m,s;
	 cin>>n;
	 vector< pair < ll,ll > > v;
	 for(i=0;i<n;i++)
	 {
	     cin>>a>>b;
		 v.push_back(pair<ll, ll> (a,b));
	 }
	 ll cnt=0,prev;
	 for(i=1;i<n;i++)
	 {
	     m=INT_MAX;
	     for(j=i-1;j>=0;j--)
		 {
		     
			     s=(double)(v[j].second-v[i].second)/(v[j].first-v[i].first);
			     if(s<=m)
				 {
				     cnt++;
					 m=s;
					 //cout<<i<<" "<<j<<endl;
				 }
		 }
		 
	 }
	 cout<<cnt;
	 return 0;
}
