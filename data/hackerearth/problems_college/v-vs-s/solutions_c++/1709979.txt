inline void Solve(){
    string s;
    cin>>s;
    reverse(s.begin(), s.end());
    bool flag = 0;
    for(int i = 0; i < s.length(); i++){
    	if(s[i] != '0'){
    		flag = 1;
    	}
    }
    if(flag == 0){
    	printf("0\n");	
    	return;
    }
    int rem = 0, t = 1;
    for(int i = 0; i < min((int)(s.length()), 2); i++){
        rem += (t * (s[i] - '0'));
        t *= 3;
    }
    rem %= 9;
    if(rem == 0) rem = 9;
    printf("%d\n", rem);
}

int main(){

    sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
