tc = int(raw_input())
while(tc>0):
    n = int(raw_input())
    num = []
    xx = raw_input().split(' ')
    sum = 0
    for i in range(0,n):
        num.append(int(xx[i]))
        sum = sum+num[i]
    m = []
    for i in range(0,sum+1):
        m.append(0)
    m[0] = 1
    for i in range(0,n):
        for j in range(sum,num[i]-1,-1):
            m[j] = m[j]|m[j-num[i]]
    ans = 0
    for i in range(0,sum+1):
        if(m[i]==1):
            ans = ans+1
    print ans
    tc = tc-1
