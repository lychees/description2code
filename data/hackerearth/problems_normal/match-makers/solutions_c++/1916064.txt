#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int g[n],b[n];
    	for(int i=0;i<n;i++){
    		cin>>g[i];
    	}
    	for(int i=0;i<n;i++){
    		cin>>b[i];
    	}
    	sort(g,g+n);
    	sort(b,b+n,std::greater<int>());
    int count=0;
    for(int i=0;i<n;i++){
    		if((b[i]%g[i]==0) ||(g[i]%b[i]==0))
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
