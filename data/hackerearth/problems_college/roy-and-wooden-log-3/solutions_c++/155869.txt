// Hackerearth - Roy and Wooden Log

#include <stdio.h>
#include <iostream>

using namespace std;
typedef long long LL;

LL length, ways, minCuts;

void solve()
{
  ways = length / 2 + 1;

  if(length % 2 == 0) minCuts = length / 2 - 1;
  else minCuts = length / 2;
}

int main()
{
  int t;

  cin >> t;
  while(t--)
  {
    cin >> length;

    solve();
    printf("%lld %lld\n", ways, minCuts);
  }

  return 0;
}
