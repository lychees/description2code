#include <iostream>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,m=0;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){cin>>a[i];}
		
		for(int i=0;i<n;i++){
			for(int j=0;j<n;j++){
				if(j>=i && a[i]==a[j]){m++;}
			}
		}  cout << m << endl;
	}
  
    return 0;
}
