#include <iostream>

#define MOD (1000000007LL)

using namespace std;

typedef long long int ll;

ll pow(ll base, int expo) {
  ll res = 1LL;
  ll k = base;
  while (expo) {
    if (expo % 2) {
      res *= k;
      res %= MOD;
    }
    k *= k;
    k %= MOD;
    expo /= 2;
  }
  return res;
}

int n;

int main() {
  int t;
  scanf("%d", &t);
  for (int i = 1; i <= t; i++) {
    scanf("%d", &n);
    ll s = 0LL;
    for (int j = 1; j <= n; j++) {
      int temp;
      scanf("%d", &temp);
      s += (ll)temp;
    }
    printf("%d\n", (pow(2LL, n - 1) * s) % MOD);
  }
  return 0;
}
