#include <iostream>
#include <string>
using namespace std;

int main ()
{
	string str, split[10000];
	int i, j = 0, l = 0, n;
	
	getline (cin, str);
	cin >> n;
	
	for (i=0; i<((str.length()/n)+1); i++)
	{
		split[j] = str.substr (l, n);
		l = l + n;
		j++;
		// cout << endl << l << endl;
	}
	
	for (i=0; i<j; i++)
		cout << split[i] << endl;
		
	return 0;
}
