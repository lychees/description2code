#include <bits/stdc++.h>
 
using namespace std;
 
const int N = 2002;
 
int n;

long long ans;
 
bitset < N > a[N];
 
int main()
{
  scanf("%d", &n);
  for (int i = 1;i <= n;i++)	
  	for (int j = 1;j <= n;j++)
  	{
  		int x; cin >> x; a[i][j] = x;
  	}
  for (int i = 1;i <= n;i++)
  	for (int j = i + 1;j <= n;j++)
  	{
  		int cnt = (a[i] & a[j]).count();
  		ans += (1ll * cnt * (cnt - 1)) / 2;
  	}
  cout << ans / 2;
	return 0;
}
