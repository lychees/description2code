#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
    int t;
    string s;
    long long l,i;
    cin>>t;
    while(t--){
    	cin>>s;int q=1;
    	i=0;l=0;
    	while(i<(s.length()-1)){
    		if(s[i]=='2'&&s[i+1]=='1'){
    			cout<<"The streak is broken!\n";q=0;break;
    		}
    		l=l*10+(s[i]-'0');i++;
    	}
    	if(q){
    		l=l*10+(s[i]-'0');
    	//	cout<<l<<endl;
          if(!(l%21))cout<<"The streak is broken!\n";
          else cout<<"The streak lives still in our heart!\n";
    	}
    }
    
    return 0;
}
