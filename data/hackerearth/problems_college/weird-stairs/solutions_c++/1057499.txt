#include <iostream>
using namespace std;


int main()
{
	ios_base::sync_with_stdio(false);
	int t,n,last,current,i,f;
	long long w,sum;
	cin>>t;
	while(t--) {
		cin>>w>>n;
		sum = 0;
		last = 0;
		for(i=1;i<=n;i++) {
			cin>>f;
			if(f == 1) {
				sum = sum + i*2 - last;
				last = i*2;
			} else {
				sum = sum + 1;
				last = sum;
			}
		}
		sum = sum + w * n;
		cout<<sum<<endl;
	}
    return 0;
}
