#include <bits/stdc++.h>

using namespace std;

#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	register int n,i,j,ans,k;
	//scanf("%d",&n);
	scanint(n);
	int a[n];
	for (i = 0; i < n; i++) {
		//scanf("%d",&a[i]);
		scanint(a[i]);
	}
	ans = 0, k = 1;
	for (i = 1; i < n; i++) {
		if (abs(a[i] - a[i-1]) == 1) {
			k++;
			if (k > ans) {
				ans = k;
			}
		} else {
			k = 1;
		}
	}
	if (k > ans) {
		ans = k;
	}
	printf("%d\n",ans);
	
	return 0;
}
