#include<bits/stdc++.h>
using namespace std;
int main(){ 
 int i,j,k,N[1000][1000],n,t,V[1000],W[1000],flag,w;
  char C;
    cin>>t;
    while(t--){
     cin>>n>>w;
        for(i=1;i<=n;i++){
            cin>>W[i];
        }
         for(i=1;i<=n;i++){
            cin>>V[i];
        }
        for(i=0;i<=n;i++){
            for(j=0;j<=w;j++){
                
                 if(j==0 ||i==0){
                     N[i][j]=0;
                     continue;
                 }
                 if(j<W[i]){
                        N[i][j]=N[i-1][j];
                     k=N[i][j];
                    }
                if(j>=W[i]){
                      N[i][j]=max((V[i]+N[i-1][j-W[i]]),N[i-1][j]);
                      k=N[i][j];
                                                                 
                      }
                                  }
                                  }
            cout<<k<<"\n";
                                  }
                                  
       
    return 0;
               }
