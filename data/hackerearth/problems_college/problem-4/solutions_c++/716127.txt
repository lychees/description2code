#include <iostream>
#include<cstdio>
using namespace std;

int main()
{

    char a[100];
   gets(a);
  char t;
  
   for(int i = 0; a[i+1]!='\0';i++)
   {
     for(int j = i+1; a[j]!='\0';j++)
      if(a[j] < a[i])
       {
	t = a[i];
	a[i] = a[j];
	a[j] = t;
       }
   }
   cout << a;
    return 0;
}
