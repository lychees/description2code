#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;
vector<char > ch;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
    	ch.clear();
        char str[101];
        gets(str);
        int l=strlen(str);
        for(int i=0;i<l;i++)
        {
        	if(str[i]!='.'&&str[i]!=','&&str[i]!='?'&&str[i]!='!'&&str[i]!=' ')
        	ch.push_back(str[i]);
        }
        for(int i=0;i<ch.size();i++)
        if(ch[i]=='*')
        {
        	cout<<ch[ch.size()-i-1]<<endl;
        	break;
        }
        // your code here
        
        test_cases--;
    }
    return 0;
}
