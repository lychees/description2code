#include<cstdio>
#include<cmath>
#define mod 1000000007
#define ll long long 
using namespace std;
ll catalan[10005];
ll catalanDP(long long  int n)
{
    catalan[0] = catalan[1] = 1;
    for (int i=2; i<=n; i++)
    {
        catalan[i] = 0;
        for (int j=0; j<i; j++)
            catalan[i]+=(catalan[j]%mod)*(catalan[i-j-1]%mod)%mod;
            catalan[i]%=mod;
    }
    return catalan[n];
}
int main()
{
    ll t,n;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	printf("%lld\n",catalanDP(n)%mod);
    }
    return 0;
}
