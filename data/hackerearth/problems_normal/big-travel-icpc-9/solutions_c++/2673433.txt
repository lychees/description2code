#include <bits/stdc++.h>
using namespace std;

int i,n,t;
const long MOD=1000000007;


int main()
{	ios_base::sync_with_stdio(false);
	cin.tie(false);
	int a,b;
	vector< pair<long,long> > XY,YX; 
    for(cin>>n,i=0;i<n;i++)
    {	cin>>a>>b;
    	XY.push_back(make_pair(a,b));
    	YX.push_back(make_pair(b,a));
    }
    sort(XY.begin(),XY.end());
    sort(YX.begin(),YX.end());
    long ans=0;
    long XarrSum=0,YarrSum=0;
    for(i=0;i<n;i++)
    {
    	//for(t=i+1;t<n;t++)
    	 
    		ans+= (XY[i].first*i -XarrSum)+(YX[i].first*i -YarrSum);   //(abs(Pt[i].first-Pt[t].first) + abs(Pt[i].second-Pt[t].second));
    		if(ans>=MOD) ans%=MOD;
    		XarrSum+=XY[i].first;
    		YarrSum+=YX[i].first;
    }
    cout<<ans<<endl;
    
    return 0;
}
