#include <iostream>
#include<string>
using namespace std;
bool f(string str)
{ 	int a[26]={0};
 for(int i=0;i<str.length();i++)
   	a[str[i]-97]++;
   	int len=a[0],k=0;
   	for(int i=0;i<26;i++)
   	if(len==a[i])
     k++;
     if(k==26)
     return true;
     return false;
 }
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
  
   	string str;
   	cin>>str;
    if(f(str))
    cout<<"Yes"<<endl;
    else
    cout<<"No"<<endl;
    }
    return 0;
}
