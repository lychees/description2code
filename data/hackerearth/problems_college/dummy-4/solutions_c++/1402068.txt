#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,i,j,ans,p;
	char s[10000];
	ans = 0;
	scanf("%d",&t);
	while (t --) {
		scanf("%s",s);
		
		for (i = 0; i < strlen(s); i++) {
			ans += s[i]-48;
		}
		
	}
	int sum = 0;
		while (ans/10 != 0) {
			p = ans;
			sum = 0;
			while (p != 0) {
				sum += p % 10;
				p = p / 10;
			}
			ans = sum;
			if (ans / 10 == 0) {
				break;
			}
		}
		printf("%d\n",ans);
	
	return 0;
}
