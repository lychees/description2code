#include <bits/stdc++.h>
using namespace std;
vector<int>adj[1010];
bool v[1010]={0};
int l[1010]={-1};
void bfs(int s){
	queue<int>q;
	q.push(s);
	v[s]=1;
	l[s]=0;
	while(!q.empty()){
		int p=q.front();
		q.pop();
		for(int i=0;i<adj[p].size();i++){
			if(!v[adj[p][i]]){
				l[adj[p][i]]=l[p]+1;
				q.push(adj[p][i]);
				v[adj[p][i]]=1;
			}
		}
	}
}
int main()
{
    int n,m; cin>>n>>m;
    while(m--){
    	int a,b; cin>>a>>b;
    	adj[a].push_back(b);
    	adj[b].push_back(a);
    }
    bfs(0);
    for(int i=1;i<n;i++){
    	cout<<l[i]<<endl;
    }
    return 0;
}
