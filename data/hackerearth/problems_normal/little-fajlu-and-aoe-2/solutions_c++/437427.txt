int solve(){
    int f, g, s, p, c, istr;
    int gainstr = 0;
    cin>>f>>g>>s>>p>>c>>istr;
    for(int i = 0; i <= f / 100; i++){
        for(int j = 0; j <= (f - i*100)/125 && j <= g/50; j++){
            int x = (f-(i*100) - (j*125))/50;
            x = min(x, (g - (j*50))/100 );
            gainstr = max(gainstr, i*s + j*p + x*c);

        }
    }
    if(istr >= gainstr)cout<<-1<<endl;
    else cout<<gainstr-istr<<endl;
    return 0;
}
