#include<bits/stdc++.h>
using namespace std;

char s[10000],r[100000];
int main() {
   int i,t,k;
   
   cin >> t;
   while(t--) {
     cin >> s;
     k=strlen(s);
     for(i=0;i<=k/2;i++) {
     	r[i]=s[k-i-1];
     	r[k-i-1]=s[i];
     }
     r[k]='\0';

     for(i=1;i<k;i++) {
     	if(abs(s[i]-s[i-1])!=abs(r[i]-r[i-1])) {
     		cout << "Not Funny" << endl;
     		break;
     	}
     } 

     if(i==k)
     	cout << "Funny" << endl;
    
     
   }
   return 0;

}
