#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	
	int t;
	cin>>t;
	while(t-->=0)
	{
	
	char a[101];
	int s;
	gets(a);
	s=strlen(a);
	char rev[s];
	int i,j,l,last_l;
	j=0;
	for(i=s-1;i>=0;i--,j++)
	{
		rev[j]=a[i];
	}

	rev[j++]='\0';

	int words=0;
    char str[s];
 for (int l =0, last_l = 0;; l++) {
        if (rev[l] == ' ' || rev[l] == '\0')
        {
            for (int m = l - 1; m >= last_l; m--) { 
                str[words] = rev[m];            
                words++;                            
            }
            last_l = l + 1;                         
                                                    
            str[words] = rev[l];                
            words++;
        }
        if (rev[l] == '\0')                     
            break;
    }

    cout << str << endl;
}

    return 0;
}
