#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		long int a[n];
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		long int sum=0;
		for(int i=n-1;i>=n-k && i>=0;i--)
		{
			sum+=a[i];
		}
		cout<<sum<<endl;
	}
}
