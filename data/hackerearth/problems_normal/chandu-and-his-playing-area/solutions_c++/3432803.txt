#include <iostream>
#include <math.h>

using namespace std;


int main() {
        int T;
        cin>>T;
        while(T--) {
            long long int ans, temp, mid, rest;
            cin >> ans;
            mid = ans/2;
            if(mid%2!=0) {
                mid = mid+1;
                rest = ans-mid;
            } else {
                rest = mid;
            }
            cout << (mid/2)*(rest/2) << endl;
        }
        return 0;
}
