#include <iostream>
using namespace std;

int nth_digit(int a,int b,int n){
    while((n--)>1){
        a*=10;
        a%=b;
    }
    return (a*10)/b;
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    
    int t,a,b,n;
    cin>>t;
    while(t--){
        cin>>a>>b>>n;
        a%=b;
        cout<<nth_digit(a,b,n)<<'\n';
    }
    return 0;
}
