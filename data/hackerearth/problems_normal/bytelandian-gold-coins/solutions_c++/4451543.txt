#include <iostream>
#include <map>
using namespace std;
map<long long int , long long int>mp;

long long int max(long long int i,long long int j)
{
    return (i<j)?j:i;
}
long long int coin (long long int n)
{
    if(mp.find(n) == mp.end())
    {
        mp[n] = max(n,coin(n/2)+coin(n/3)+coin(n/4));
    }
    return mp[n];
}
int main()
{
    long long int  t;
//  cin>>t;
//  while(t--)
    map<long long int , long long int>mp;
    {
        mp[0] = 0;
        mp[1] = 1;
        mp[2] = 2;
        cin>>t;
        cout<<coin(t);
        cout<<endl;
    }
    
    return 0;
}
