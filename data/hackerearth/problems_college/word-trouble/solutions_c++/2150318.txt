#include<bits/stdc++.h>
using namespace std;

#define MAX 100005
#define MOD 1000000007
typedef pair<int,int>pp;
typedef long long ll;
int N, M;

int main()
{
	string s1,s2;
	int test;
	cin>>test;
	while(test--)
	{
		cin>>s1>>s2;
		int cnt =0;
		int len = min(s1.length(),s2.length());
		int ans = max(s1.length(),s2.length());

		for(int i = 0 ;i < len ; i++)
		{
			if(s1[i] == s2[i])
            {
                cnt++;
            }
		}
		ans  = ans-cnt;

		if(ans == 0)
			cout<<"Exact"<<endl;
		else
        {
		    cout<<"Non Exact "<<ans<<endl;

		}
	}
	return 0;
}
