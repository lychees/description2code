#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    string s;
    int count=0;
   	while(t--){
    	stack<char> v;
    	cin>>s;
    	v.push(s[0]);
    	for(int i=1;i<s.length();i++){
    		if(v.empty()!=true)
    		{
    			if(s[i]==v.top())
    				v.pop();
    			else 
    				v.push(s[i]);
    		}
    		else 
    			v.push(s[i]);
    	}
    	if(v.empty()==true)
    	count++;
    	
    }
    cout<<count;
}
