#include <iostream>
using namespace std;
#include<cmath>

int main()
{int t;
cin>>t;
long n;
long p;
while(t--)
{
	cin>>n;
	p=(long)sqrt(n);
	if(n==p*p)
	cout<<"YES\n";
	else
	{
		if(n-p*p<(p+1)*(p+1)-n)
		{
			cout<<"-"<<n-p*p<<"\n";
		}
		else
			cout<<"+"<<(p+1)*(p+1)-n<<"\n";
	}
	
	
}

    //out << "Hello World!" << endl;
    return 0;
}
