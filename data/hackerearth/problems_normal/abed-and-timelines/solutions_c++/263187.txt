#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int main(){
	int t;cin>>t;
	for(int i=0; i<t; i++){
		int rows, cols;
		int x1, y1, x2, y2, p;
		cin>>rows>>cols;
		cin>>x1>>y1>>x2>>y2>>p;
		int val = abs(x1-x2);
		int val1 = rows-val;
		int val2 = val<val1?val:val1;
		val = abs(y1-y2);
		val1 = cols-val;
		int sum = val<val1?val:val1;
		sum = (val2 + sum)*p;
		if(i!=0)
			cout<<endl;
		if(sum > 1000){
			cout<<sum<<endl;
			cout<<"Let go of the group.";
		}else{
			cout<<sum<<endl;
			cout<<"You saved the group.";
		}
	}	
	return 0;
}
