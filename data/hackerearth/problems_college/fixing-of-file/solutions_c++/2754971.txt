#include <cstdio>
#include <map>
#include <iostream>
#include <string>

using namespace std;

char input[110101];

void Process(int num)
{
	int n,m,ans=0;
	char buf[110101];
	int i_buf = 0;
	cin >> n >> m;
	
	map< string, int> hash;
	
	for(int i = 0; i < n; ++i)
	{
		i_buf = 0;
		cin >> input;
		for(int j = 1; ; ++j)
		{
			if(input[j] == '/' || input[j] == '\0')
			{
				buf[i_buf] = '\0';
				string tmp(buf);
			//	cout << buf << endl;
				buf[i_buf++] = '/';
				hash[tmp] = 1;
			}
			else
			{
				buf[i_buf++] = input[j];
			}
			
			if(input[j] == '\0')
				break;
		}
	}
	
	for(int i = 0; i < m; ++i)
	{
		i_buf = 0;
		cin >> input;
		for(int j = 1; ; ++j)
		{
			if(input[j] == '/' || input[j] == '\0')
			{
				buf[i_buf] = '\0';
				string tmp(buf);
			//	cout << buf << endl;
				buf[i_buf++] = '/';
				if(hash[tmp] == 0)
				{
					++ans;
					hash[tmp] = 1;
				}
			}
			else
			{
				buf[i_buf++] = input[j];
			}
			
			if(input[j] == '\0')
				break;
		}
	}
	
	printf("Case #%d: %d\n",num,ans);
}

int main()
{
	int T,num=1;
	cin >> T;
	while(T--)
	{
		Process(num++);
	}
	return 0;
}
