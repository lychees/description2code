#!/usr/bin/env python
# -*- coding: utf-8 -*-
import sys
from itertools import combinations


def main():
    testcases = int(sys.stdin.next().rstrip())
    for testcase_num in range(testcases):
        sys.stdin.next()
        nums = list([int(x) for x in sys.stdin.next().rstrip().split()])
        target_sum = int(sys.stdin.next().rstrip())
        found = False
        for i in range(1, len(nums) + 1):
            if found:
                break
            for combination in combinations(nums, i):
                if (sum(combination)) == target_sum:
                    print('YES')
                    found = True
                    break
        if not found:
            print('NO')


if __name__ == '__main__':
    main()
