m=input()
p=input()
w=map(int,raw_input().split())
def f(a,w,p):
    i=0
    p1=0
    while(i<len(w) and p1<p):
        s=0
        while(i<len(w) and s+w[i]<=a):
            s+=w[i]
            i+=1
        p1+=1
    if i==len(w): return True
    return False
a=-1
b=sum(w)
while(a<b-1):
    c=(a+b)/2
    r=f(c,w,m)
    if r:
        b=c
    else:
        a=c
