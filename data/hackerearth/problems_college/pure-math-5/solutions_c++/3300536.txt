    #include<bits/stdc++.h>
    #define ll long long 
    int main()
    {
    ll int n,m,k,cnt,i,j;
    ll int size;
    int t;
    scanf("%d",&t);
   
    while(t--)
    {
        cnt=0;
        scanf("%lld",&size);
     //	cin >> size;
        for(j=2;j<=size;j++)
        {
            i=1;
            if(((j*j)-1) % i == 0 && ((i*i)-1) % j==0)
            {
                cnt++;
                n=j;
                m=((j*j) -1)/i;
                while(m<=size)
                {
                    cnt++;
                    k=m;
                    m=((k*k)-1)/n;
                    n=k;
                }
            }
        }
    printf("%lld\n",cnt);
    //cout << cnt << endl;
    }
    return 0;
    }
