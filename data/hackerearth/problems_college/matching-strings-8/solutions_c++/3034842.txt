#include <iostream>
using namespace std;
 
int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	string s1, s2;
    	cin>>s1>>s2;
    	int i, count=0, l1, l2;
    	l1=s1.length();
    	l2=s2.length();
    	if(l1<=l2)
    	{
    		for(i=0; i<s1.length(); i++)
    		if(s1[i]==s2[i])
    		count++;
    	}
    	else
    	{
    		for(i=0; i<s2.length(); i++)
    		if(s1[i]==s2[i])
    		count++;
    	}
    	
    	cout<<count<<endl;
    }
return 0;
}
