#include<stdio.h>
#include<iostream>
#include<string.h>
#include<algorithm>

using namespace std;

int abs(int a)
{
    return (a<0)?(a*(-1)):a;
}
int main()
{
    int n;
    scanf("%d",&n);
    int ar[n];
    int i=0;
    while(i<n)
        scanf("%d",&ar[i++]);
    sort(ar,ar+n);
    int ans=0,j;
    i=1;
    ans=1;
    j=ar[0];
    while(i<n)
    {
       if(ar[i]>j+4)
            ans++,j=ar[i];
       i++;
    }
    printf("%d\n",ans);
    return 0;
}
