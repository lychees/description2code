#include <iostream>
using namespace std;

int main()
{
   int n=0;
   cin>>n;
   int a[n];
   int sum=0;
   for(int i=0; i<n;i++)
   {
   	 cin>>a[i];
   	 sum+=a[i];
   	 for(int j=0; j<i; j++)
   	 {
   	 	if(a[j]>a[i])
   	 	{
   	 		int temp=a[i];
   	 		a[i]=a[j];
   	 		a[j]=temp;
   	 	}
   	 }
   }
   
   int k=0;
   	if(sum%2 != 0)
   	{
   		cout<<sum<<endl;
   	}
   	else
   	{
   		while(k<n)
   		{
   			if(a[k]%2 == 1)
   			{
   				sum=sum-a[k];
   				cout<<sum<<endl;
   				break;
   			}
   			k++;
   		}
   		if(sum%2 == 0)
   		{
   			cout<<":("<<endl;
   		}
   	}
   	 
   
   return 0;
}
