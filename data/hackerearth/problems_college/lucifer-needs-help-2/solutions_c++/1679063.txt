#include <stdio.h>

long long int a[1000000]={0},b[1000000]={0};
int main()
{
  //  printf("Hello World!\n");
  long long int n,m,i,j,k,l,r;
  scanf("%lld", &n);
  for (int i = 0; i < n; ++i)
  	scanf("%lld", &a[i]);
  	scanf("%lld", &m);
  for (i = 0;i < m; ++i) {
  	scanf("%lld %lld %lld", &l, &r, &k);
  	b[l-1] += k;
  	if(r+1 <= n)
  	b[r] += -k;
  }
  long long int tmp = 0;
  for (i = 0; i < n; ++i) {
  	tmp +=b[i];
  	printf("%lld\n", tmp+a[i]);
  }
    return 0;
}
