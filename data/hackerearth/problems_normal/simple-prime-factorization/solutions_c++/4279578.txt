#include<bits/stdc++.h>
using namespace std;
#define MAX 1000006
int lp[MAX];
int primes[MAX];
int pr;
void siever(void)
{
	int i,j;
	pr= 0;
	for(i=2;i<MAX;++i)
	{
		if(lp[i]==0)
		{
			primes[pr++]=i;
			for(j=i;j<MAX;j+=i)
				lp[j]=i;
		}
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    siever();
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	int p;
    	cin>>n;
    	p=n;
    	map<int,int> mp;
    	mp[2]=0;
    	while(n!=1)
    	{
    		int d = lp[n];
    		while(n%d == 0)
    		{
    			n/=d;
    			mp[d]++;
    		}
    	}
    	if(lp[p] == p)
    	{
    		cout<<"2^0\n";
    		continue;
    	}
    	map<int,int>::iterator it,ic;
    	for(it=mp.begin();it!=mp.end();++it)
    	{
    		ic = it;
    		ic++;
    		if(ic != mp.end()) cout<<it->first<<"^"<<it->second<<"*";
    		else cout<<it->first<<"^"<<it->second<<"\n";
    		
    	}
    	
    }
    
    return 0;
}
