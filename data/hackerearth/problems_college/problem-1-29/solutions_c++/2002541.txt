bool isok(int x,int y){
	return (x>=0 && x<n && y>=0 && y<n && arr[x][y]==1);
}
int solve(int x,int y){
	if(x==n-1 && y==n-1)return 1;
	if(isok(x,y)){
		if(solve(x+1,y))return 1;
		if(solve(x,y+1))return 1;
		return 0;
	}
	return 0;
}
int main()
{
    int t;

    while(t--){
    	cin>>n;
    	memset(arr,0,sizeof(arr));
    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++){
    			cin>>arr[i][j];

    	if(solve(0,0))cout<<"POSSIBLE\n";

    	for(int i=0;i<n;i++){
    		for(int j=0;j<n;j++){
    			arr[i][j]=0;
    		}
    	}
    }
    return 0;
}
