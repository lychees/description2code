#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,n;
    scanf("%d",&n);
    vector<int> v(n+1);
    v[1]=2;
    for(i=2;i<=n;i++)
    {
    	if(v[i]==0)
    	{
    		for(j=2*i;j<=n;j=j+i)
    		{
    			v[j]=1;
    		}
    	}
    }
    for(i=1;i<=n;i++)
    {
    	if(v[i]==1)
    		printf("%d\n",i);
    }
    return 0;
}
