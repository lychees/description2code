#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	long long int n,m,k;
	while(t--)
	{
		scanf("%lld %lld %lld",&n,&m,&k);
		long long int a,b,jump=0,x=1,y=1;
		for(long long int i=0;i<k;i++)
		{
			scanf("%lld %lld",&a,&b);
			long long int tempx,tempy;
			while(1)
			{
				tempx = x+a;
				tempy = y+b;
				if(tempx>n || tempy>m || tempx<1 || tempy<1)
				{
					break;
				}
				x = tempx;
				y =tempy;
				jump++;
			}
		}
		printf("%lld\n",jump);
	}

    return 0;
}
