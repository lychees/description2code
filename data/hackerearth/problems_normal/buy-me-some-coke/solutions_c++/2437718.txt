#include <iostream>
using namespace std;

int st[200][150][100];
bool marked[200][150][100] = {false};

int min(int a, int b) {
	if (a < b) return a;
	return b;
}

int steps(int cokes, int n5, int n10) {
	if (cokes == 0) return 0;
	if (marked[cokes][n5][n10]) return st[cokes][n5][n10];
	
	int ret = 1e9;
	
	if (n10 > 0) ret = min(ret, 1+steps(cokes-1, n5, n10-1));
	if (n10 > 0) ret = min(ret, 4+steps(cokes-1, n5+1, n10-1));
	if (n5 > 1)  ret = min(ret, 2+steps(cokes-1, n5-2, n10));
	if (n5 > 0)  ret = min(ret, 4+steps(cokes-1, n5-1, n10));
	
	if (n5 == 0 && n10 == 0) ret = min(ret, 8*cokes+steps(0, n5, n10));
	
	st[cokes][n5][n10] = ret;
	marked[cokes][n5][n10] = true;
	
	return ret;
}

int main()
{
    int tests;
    cin >> tests;
    
    
    
    while (tests-- > 0) {
    	int c, n1, n5, n10;
    	cin >> c >> n1 >> n5 >> n10;
    	
    	if (n10*10 + n5*5 + n1 < 8*c) {
    		cout << -1 << endl;
    		continue;
    	}
    	
    	cout << steps(c, n5, n10) << endl;
    }
    
    return 0;
}
