#! /usr/bin/env python

try:
  sequence_length = int(raw_input())
  
  sequence = raw_input()
  
  no_of_queries = int(raw_input())
  
  MAP = {}
  
  for element in sequence:
    element_count = MAP.get(element, 0)
    MAP[element] = element_count + 1
  
  
  for _ in xrange(no_of_queries):
    start, end = raw_input().split()
  
    if start != end:
      print MAP.get(start, 0) * MAP.get(end, 0)
    else:
      print MAP.get(start, 0) * (MAP.get(end, 0) - 1) / 2
except:
  pass
