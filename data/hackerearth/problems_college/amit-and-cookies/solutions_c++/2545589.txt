#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k,c=0,i;
    cin>>n>>k;
    int *a = (int *)malloc(sizeof(int)*n);
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    sort(a,a+n);
    for(i=0;i<n;i++){
    	if(k>=a[i]){
    		k-=a[i];
    		c++;
    	}else{
    		break;
    	}
    }
    cout<<c<<endl;
    return 0;
}
