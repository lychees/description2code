int a[MAX];
int main()
{
	int n, left, right,t;
	si(t);
	while(t--){(cin >> n);
	
		forall(i, 0, n)
			si(a[i]);
		left = 0, right = 0;
		forall(i,0,n)
		{
			if (a[i] == -1)
				left++;
			else
				break;
		}
		forrev(i,n-1,0)
		{
			if (a[i] == 1)
				right++;
			else
				break;
		}

		int leftgoingants = 0;
		forall(i,left,n-right)
			if (a[i] == -1)
				leftgoingants++;
		
		if (left >= right)
			pi(left + leftgoingants);
		else
			pi(left + leftgoingants + 1);
	}
	return 0;
}
