#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

#define MAXN 1010
int DP[MAXN][MAXN],n,m;

int solve(int i,int j)
{
	if(DP[i][j]!=-1)
		return DP[i][j];
	if(i+j==n)
	{
		if(i>j)
			return (DP[i][j]=1);
		else
			return (DP[i][j]=0);
	}	
	DP[i][j] = 0;
	if(i>=m*(j+1))
		DP[i][j] += solve(i,j+1);
	DP[i][j] += solve(i+1,j);
	DP[i][j] %= MOD;
	return DP[i][j];
}

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--)
	{
		cin >> n >> m;
		memset(DP,-1,sizeof DP);
		cout << solve(0,0) << endl;
	}
	return 0;
}
