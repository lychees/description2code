#include <iostream>
using namespace std;

int main()
{
    int test_cases=0;
    cin>>test_cases;
    while(test_cases--){
    	int index,n;
    	cin>>index>>n;
    	if(n==0){
    		cout<<index<<endl;
    		continue;
    	}
    	switch(index){
    		case 0:
    			if(n%2==0){
    				cout<<index<<endl;
    			}
    			else{
    				cout<<index+1<<endl;
    			}
    			break;
    		case 1:
    			if(n%2==0){
    				cout<<index<<endl;
    			}
    			else{
    				cout<<index-1<<endl;
    			}
    			break;
    		case 2:
    			if(n%2==0){
    				cout<<index-2<<endl;
    			}
    			else{
    				cout<<index-1<<endl;
    			}
    			break;
    			
    	}
    }
    return 0;
}
