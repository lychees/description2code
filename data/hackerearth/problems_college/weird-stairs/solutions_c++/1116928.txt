#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
int height[100010];
int a[100010];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        memset(height,0,sizeof(height));
        int n,width;
        scanf("%d%d",&width,&n);
        for(int i=1;i<=n;i++) scanf("%d",&a[i]);
        for(int i=1;i<=n;i++) {
            if(a[i]==1) height[i]=2*i;
            else height[i]=height[i-1]+1;
        }
        for(int i=n;i>=1;i--) height[i]-=height[i-1];
        ll ans=0;
        for(int i=1;i<=n;i++) {
            ans+=(width+height[i]);
        }
        printf("%lld\012",ans);
    }
    return 0;
}
