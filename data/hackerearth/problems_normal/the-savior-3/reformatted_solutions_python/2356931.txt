

tc = int(raw_input())

    n = int(raw_input())

    ans = 0
    for i in xrange(n):
        for j in xrange(n):
            if l[i] == l[j]:

            else:
                if (l[i]+l[j]) % 2 == 0:
                    ans = ans + 1
