#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    
    string s1,s2,s3;
    getline(cin,s3);
    while(t--)
    {
    	int a[26]={0},b[26]={0};
    	int temp;
    	getline(cin,s1);
    	getline(cin,s2);
    	
    	int len1=s1.length();
    	
    	int len2=s2.length();
    	for(int i=0;i<len1;i++)
    	{
    		a[s1[i]-'a']++;
    	}
    	for(int i=0;i<len2;i++)
    	{
    		b[s2[i]-'a']++;
    	}
    	for(int i=0;i<26;i++)
    	{
    		temp=a[i];
    		a[i]=a[i]-b[i];
    		b[i]=b[i]-temp;
    	}
    	int flag1=0;
    	int flag2=0;
    	for(int i=0;i<26;i++)
    	{
    		if(a[i]>0)
    		{
    			flag1=1;
    		}
    	}
    	for(int i=0;i<26;i++)
    	{
    		if(b[i]>0)
    		{
    			flag2=1;
    		}
    	}
    	if(flag1==1 && flag2==0)
    	{
    		cout<<"You win some."<<endl;
    	}
    	else if(flag1==0 && flag2==1)
    	{
    		cout<<"You lose some."<<endl;
    	}
    	else
    	{
    		cout<<"You draw some."<<endl;
    	}
    }
}
