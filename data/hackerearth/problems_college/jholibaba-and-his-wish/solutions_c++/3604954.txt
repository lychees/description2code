#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[1000000];
int main()
{
	ll n,i;
	cin>>n;
	ll sum=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum+=a[i];
	}
	ll ans=0;
	for(i=0;i<n;i++)
	{
		ans+=(i+1)*a[i];
	}
	ll vivek=ans;
	ll val,po=0;
	while(true)
	{
		val=vivek-sum;
		val+=n*a[po];
		vivek=val;
		ans=max(val,ans);
		po++;
		if(po>=n)
			break;
	}
	cout<<ans;
}
