for _ in range(input()):
    a, b = map(int, raw_input().split())
    l = []
    for i in range(a):
        s = list(raw_input())
        l.append(s)
    equality = False
    for k in range(len(l)-1):
        for r in range(k+1, len(l)):
            if l[k] == l[r]:
                equality = True
                break
    if equality:
        print "No"
    else:
        print "Yes"
