t = int(raw_input())
for i in range(t):
	nm = map(int, raw_input().split())
	output = []
	for j in range(nm[0]):
		min_price = 1000000000
		result = 1
		for k in range(nm[1]):
			discounts = map(int, raw_input().split())
			total_price = ((100 - discounts[0]) * (100 - discounts[1]) * (100 - discounts[2]))
			if total_price < min_price:
				result = k + 1
				min_price = total_price
		output.append(result)
	print ' '.join(map(str, output))
