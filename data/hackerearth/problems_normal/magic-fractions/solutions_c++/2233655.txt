#include<bits/stdc++.h>
using namespace std;
long long int arr[502] , pr[502];
void f( void )
{
  for( int i = 2 ; i < 501 ; ++i )
  {
    if( !arr[i] )
    {
      for( int j = 2*i ; j < 501 ; j += i )
      {
        arr[j]++;
      }
    }
  }
  pr[0]  = 0;
  pr[1] = 0;
  pr[2] = 1;
  for( int i = 3 ; i < 501 ; ++i)
  {
    pr[i] = pr[i-1];
    if( arr[i] == 0 )
      pr[i]++;
 //   cout << pr[i] << " ";
  }
}
int main()
{
  f();
  long long int n;
  scanf("%lld" , &n);
  long long int mag[n+2];
  mag[0] = mag[1] = 0;
  mag[2] = 1;
  mag[3] = 3;
  for( int i = 4 ; i <= n ; ++i )
  {
 //   cout << pr[i] << " " ;
    mag[i] = mag[i-1];
    mag[i] += pow( 2 , pr[i] - 1 );
  }
  cout << mag[n];
  return 0;
}
