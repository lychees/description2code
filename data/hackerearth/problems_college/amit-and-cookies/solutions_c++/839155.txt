#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k;
    cin >> n >> k;
    int arr[100002];
    for(int i=0;i<n;i++){
    	cin >> arr[i];
    	
    }
    sort(arr,arr+n);
    int i=0;
    while(i<n && k > 0){
    	if(arr[i] <= k){
    		k -= arr[i];
    		i++;
    	}
    	else k = 0;	
    }
    cout << i << endl;
    return 0;
}
