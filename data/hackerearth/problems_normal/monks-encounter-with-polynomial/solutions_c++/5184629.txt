#include <bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll a,b,c,k;
		cin>>a>>b>>c>>k;
		ll lb=0,ub=100000;
		ll ans=-1;
		while(lb<=ub)
		{
			ll mid=(lb+ub)/2;
			ll val = a*(mid*mid) + b*mid + c;
			if(val >= k){
				ans = mid;
				ub = mid-1;
			}
			else
				lb = mid+1;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
