#include <stdio.h>
#include <iostream>
#include <string>
#include <string.h>
#include <cstdlib>
using namespace std;


int **matrix;

int main()
{
	int N , Q;
	cin>>N>>Q;
	matrix = new int*[N];
	for(int i = 0 ; i < N ; i++)
	{
		matrix[i] = new int[26];
		memset(matrix[i],0,sizeof(int)*26);
	}
	string s;
	cin>>s;
	for(int i = s.size()-1 ; i >= 0  ; i--)
	{
		int val = (int)s[i] - (int)'a';
		matrix[i][val] = 1;
		if(i==s.size()-1)
			continue;
		for(int j = 0 ; j < 26 ; j++)
		{
			matrix[i][j] += matrix[i+1][j];
		}
	}

	// for(int i = s.size()-1 ; i >= 0  ; i--)
	// {
	// 	for(int j = 0 ; j < 26 ; j++)
	// 	{
	// 		cout<<matrix[i][j]<<" ";
	// 	}
	// 	cout<<endl;
	// }


	while(Q--)
	{
		int l,r,k;
		cin>>l>>r>>k;
		int start = 0;
		int curr = 0;
		while(start<k && curr<26)
		{
			int change = 0;
			if(r!=s.size())
			 	change = matrix[l-1][curr] - matrix[r][curr];
			else
			 	change = matrix[l-1][curr];
			if(change>0)
				start+=change;
			curr++;
		}
		if(start<k)
			cout<<"Out of range"<<endl;
		else
		{
			int val = curr+(int)'a'-1;
			cout<<(char)(val)<<endl;
		}
	}
}
