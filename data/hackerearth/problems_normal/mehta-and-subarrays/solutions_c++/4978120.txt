#include <bits/stdc++.h>
 
using namespace std;
 
const int N = 1e6 + 10;

pair <long long, int> a[N];
 
int main() {
	int n;
	long long x;
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) {
		scanf("%lld", &x);
		a[i].first = x + a[i - 1].first;
		a[i].second = i;
	}
	sort(a, a + n + 1);
	int l = a[0].second;
	int max_len = 0, count = 0;
	for (int i = 0; i <= n; i++) {
		int len = a[i].second - l;
		if (len > max_len) {
			max_len = len;
			count = 1;
		} else if (len == max_len) {
			count++;
		}
		l = min(l, a[i].second);
	}
	if (max_len == 0) {
		printf("-1\n");
	} else {
		printf("%d %d\n", max_len, count);
	}
	return 0;
}
