#include <bits/stdc++.h>

using namespace std;

map<string, int> gold, silver, bron;

struct arr {
	int g;
	int s;
	int b;
	string name;
}out[1000];

bool comp(struct arr a, struct arr bb)
{
	if(a.g > bb.g) {
		return true;
	}
	else if((a.g == bb.g) && (a.s > bb.s)) {
		return true;
	}
	else if((a.g == bb.g) && (a.s == bb.s) && (a.b > bb.b)) {
		return true;
	}
	else if((a.g == bb.g) && (a.s == bb.s) && (a.b == bb.b) && (a.name < bb.name)) {
		return true;
	}
	return false;
}

int main()
{
	int n;
	scanf("%d",&n);
	set<string> as;
	string x;
	for(int i = 0; i < 3*n; i++) {
		cin >> x;
		as.insert(x);
		if(i % 3 == 0) {
			gold[x]++;
		}
		else if(i % 3 == 1) {
			silver[x]++;
		}
		else {
			bron[x]++;
		}
	}
	int i = 0;
	for(set<string>::iterator it = as.begin(); it != as.end(); it++) {
		out[i].b = bron[*it];
		out[i].g = gold[*it];
		out[i].name = (*it);
		out[i].s = silver[*it];
		i++;
	}
	n = i;
	sort(out, out+n, comp);
	for(int i = 0; i < n; i++) {
		cout << out[i].name << " " << out[i].g << " " << out[i].s << " "<< out[i].b << endl;
	}
	return 0;
}
