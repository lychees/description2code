#include<bits/stdc++.h>
using namespace std;
int main(){
	long long t;
	scanf("%lld",&t);
	while(t--){
		string str;
		cin>>str;
		int a,b;
		scanf("%d %d",&a,&b);
		int min=a<b?a:b;
		int j=str.length()-1;
		int i=0;
		long long sum=0;
		while(i<j){
			if(str[i]=='/'&&str[j]=='/')
				sum+=2*min;
			else if(str[i]=='/'||str[j]=='/'){
				if(str[i]=='/'){
					if(str[j]=='a')
						sum+=a;
					else if(str[j]=='b')
						sum+=b;
				}
				else if(str[j]=='/'){
						if(str[i]=='a')
						sum+=a;
					else if(str[i]=='b')
						sum+=b;
				}
			}
			else if(str[i]!=str[j]){
				printf("-1\n");
				break;
			}
			i++;j--;
		}
		if(i>=j)
		printf("%lld\n",sum);
	}
}
