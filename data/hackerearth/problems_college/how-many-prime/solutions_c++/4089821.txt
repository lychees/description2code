#include <bits/stdc++.h>
#define LL long long unsigned int
#define pb push_back
#define Max 1000050
using namespace std;
typedef vector <int> V;
typedef pair <int,int> P;
int A[Max];	V v;
void sieve()
{
	for(int i=3;i<=Max;i+=2)
	{
		A[i]=1;
	}
	A[2]=1;	int s=sqrt(Max);
	for(int i=3;i<=s;i+=2)
	{
		if(A[i])
		{
			for(int j=i*i;j<=Max;j+=2*i)
				A[j]=0;
		}
	}
	v.pb(2);
	for(int i=3;i<=Max;i+=2)
	{
		if(A[i])
			v.pb(i);
	}
}

int main()
{
	sieve();	int t,ct,a,b;
	ios::sync_with_stdio(false);	int n;
	cin>>t;
	while(t--)
	{
		cin>>n;	int i=0;	int flag=0;
		while(v[i]<=n/2)
		{
			if(A[n-v[i]] && (n-v[i]!=v[i]))
			{
				a=v[i],b=n-v[i];
				flag=1;
			}
			i+=1;
		}
		if(!flag)
			cout<<"-1"<<" "<<"-1"<<"\n";
		else
			cout<<min(a,b)<<" "<<max(a,b)<<"\n";
		
	}
	return 0;
}
