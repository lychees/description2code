def same(a, b, cc):

    aa = (abs(a[1]-b[1])) % 2
    bb = (abs(a[0]-b[0])) % 2
    if aa == bb and cc == 1:
        return True
    elif aa != bb and cc == 0:
        return True
    else:
        return False


t = input()
for x in range(t):
    n = input()
    s = []
    for y in range(n):
        s.append(map(int, raw_input().split()))
    flag = True
    for i in xrange(1):
        if not flag:
            break
        for j in xrange(i+1, n):
            if (s[i][2] % 2) == (s[j][2] % 2):
                flag = same(s[i], s[j], 1)
                if not flag:
                    break
            else:
                flag = same(s[i], s[j], 0)
                if not flag:
                    break
    if flag:
        print 'Yes'
    else:
        print 'No'
