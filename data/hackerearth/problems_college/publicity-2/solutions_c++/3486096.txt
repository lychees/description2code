#define ll long long int

#define mem(a, v) memset(a, v, sizeof(a))

using namespace std;

vector<int> G[100005];
bool mark[100005];

void dfs(int st) {
    int sz = G[st].size();
    mark[st] = true;
    for (int i = 0; i < sz; i++) {
        if (!mark[G[st][i]]) {
            dfs(G[st][i]);
        }
    }
}

int main()
{
    int t, n, e, a, b;
    cin >> t;

    while (t--) {
        cin >> n >> e;
        for (int i = 0; i <= 100000; i++) {
            G[i].clear();
            mark[i] = false;
        }
        for (int i = 0; i < e; i++) {
            cin >> a >> b;
            
            G[a].push_back(b);
            G[b].push_back(a);
        }
        int cnt = 0;
        for (int i = 0; i < n; i++) {
            if (!mark[i]) {
                cnt++;
                dfs(i);
            }
        }
        cout << cnt << endl;
    }
    return 0;
}
