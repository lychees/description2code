#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int n,i,s;
    scanf("%ld",&n);

    long int a[n];

    for(i = 0;i < n;i++)
        scanf("%ld",&a[i]);

    scanf("%ld",&s);

        sort(a,a+n);

        long int sum = 0,p = 0,m;

        m = n;

        while(m--)
        {
           sum = sum + a[m];
           p++;

           if(sum >= s)
           break;

        }

        printf("%ld\n",p);


    return 0;
}
