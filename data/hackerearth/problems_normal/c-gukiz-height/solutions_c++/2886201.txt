#include <bits/stdc++.h>
using namespace std;

int n;

int m;

long long k;
long long ak;

long long tab[5000007];
long long sum[5000007];
long long s;

long long bsa, bsb, bss;

long long aktu;

int czy;

int main()
{
    scanf("%d%lld", &n, &k);
    for (int i=1; i<=n; i++)
    {
        scanf("%lld", &tab[i]);
    }
    m=n;
    while(m+n<=1000000)
    {
        for (int i=1; i<=n; i++)
        {
            tab[m+i]=tab[i];
        }
        m+=n;
    }
    for (int i=1; i<=m; i++)
    {
        sum[i]=sum[i-1]+tab[i];
        s+=tab[i];
    }
    for (int i=m+1; i<=5*m; i++)
    {
        tab[i]=tab[i-m];
        sum[i]=sum[i-1]+tab[i];
    }
    for (long long i=1; i<=5*m; i++)
    {
        if (sum[i]>=k-((i*(i+1))/2))
        {
            printf("%lld\n", i);
            return 0;
        }
    }
    bsa=0;
    bsb=2000000000/m+2;
    while(bsa<bsb)
    {
        czy=0;
        bss=(bsa+bsb)/2;
        for (long long i=bss*m+1; i<=bss*m+m; i++)
        {
            if (bss*s+sum[(i-1)%m+1]>=k-i*(i+1)/2)
            {
                czy=1;
            }
        }
        if (czy)
        {
            bsb=bss;
        }
        else
        {
            bsa=bss+1;
        }
    }
    bsa=max(bsa-2, (long long) 0);
    for (long long i=bsa*m+1; 1; i++)
    {
        if (bsa*s+sum[i-bsa*m]>=k-((i*(i+1))/2))
        {
            printf("%lld\n", i);
            return 0;
        }
    }
    return 0;
}
