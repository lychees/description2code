#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int n,i,len,sum,x,y;
    char arr[100001];
    cin>>n;
    while(n--)
    {
    	cin>>arr;
    	len=strlen(arr);
    	sum=0;
    	i=0;
    	while(i<len)
    	{
    		if(arr[i]=='X')
    		{
    			x=0;
    			while(arr[i]=='X')
    			{
    				x++;
    				i++;
    			}
    			if(x>1)
    			sum=sum+x-1;
    		}
    		else
    		{
    			y=0;
    			while(arr[i]=='Y')
    			{
    				y++;
    				i++;
    			}
    			if(y>1)
    			sum=sum+y-1;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
