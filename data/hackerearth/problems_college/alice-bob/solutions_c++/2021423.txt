#include <cstdlib>
#include <iostream>
#include <set>

using namespace std;
set<int> A;
set<int>::iterator it1;
set<int>::iterator it2;

int play()
{
     for(it1 = A.begin();it1!=A.end();++it1){
        // printf("Entered1\n");    
         it2 = it1;
         advance(it2,1);    
         for(;it2!=A.end();++it2){
             int value = *it2-*it1;
             set<int>::iterator it3 = A.find(value);
            // printf("Entered2   inserting %d\n", value); 
            // cout << *it3 << ' ' << *A.end() << '\n';  
             if(it3 == A.end()){
                A.insert(value);
                //printf("Entered 3\n");
                return 1;
             }
         }
     }
     return 0;                                   
 }

int main()
{
    int n,temp,move = 1;
    int turn = true;
    scanf("%d",&n);

    for(int i = 0;i<n;++i){
        scanf("%d",&temp);
        A.insert(temp);
    }

       move = (move+1)%2;

}
