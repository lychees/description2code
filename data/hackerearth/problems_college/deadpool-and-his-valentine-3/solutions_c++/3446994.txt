#include <iostream>
using namespace std;

int main()
{
    int k,m,x,y,i,a[101]={0};
    cin>>k>>m;
    while(k){
    	cin>>x;
    	for(i=0;i<x;i++){cin>>y;a[y]++;}
    	k--;
    }
    for(i=1;i<=m;i++)if(a[i]==0)break;
    if(i==m+1)cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    return 0;
}
