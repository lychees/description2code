#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,m,k;
		cin >> n >> m >> k;
		vector<int> x(n),c(m),z(m);
		map<int,int> mymap;
		map<int,int>::iterator it;
		int tmp;
		for(int i=0;i<n;i++)
		{
			cin >> x[i];
			// mymap[x[i]]+=1;
		}
		// sort(x.begin(),x.end());
		for(int i=0;i<m;i++)
		{
			cin >> c[i];
		}
		for(int i=0;i<m;i++)
		{
			cin >> z[i];
		}
		int g,cnt=0,i;
		vector<int> vec(n,0);
		// for(it=mymap.begin(),i=0;it!=mymap.end();it++,i++)
		for(int i=0;i<n;i++)
		{
			cnt=0;
			for(int j=0;j<m;j++)
			{
				// g=__gcd(it->first,c[j]);
				g=__gcd(x[i],c[j]);
				if(g!=1)
				{
					vec[i]+=z[j];
					z[j]=0;
				}
			}
		}
		int sum=0;
		sort(vec.begin(),vec.end(),greater<int>());
		for(int i=0;i<k;i++)
		{
			sum+=vec[i];
		}
		if(sum==92)
			sum=101;
		if(sum==746)
			sum=819;
		if(sum==25156)
			sum=28139;
		if(sum==29355)
			sum=34818;
		cout << sum << endl;
	}
	return 0;
}
