t = int(input())


def dgt(n):
    sum = 0
    while n > 0:
        sum = sum+n % 10
        n = int(n/10)
    return sum


while t > 0:
    n = int(input())
    sum = 0
    while n > 0:
        r = raw_input().split()
        arr = [int(i) for i in r]
        m = arr[0]
        k = arr[1]
        sum += m*k
        n -= 1
    while sum > 9:
        sum = dgt(sum)
    print(sum)

    t -= 1
