#include <bits/stdc++.h>
using namespace std;

int A[10000007];

int main()
{
    for(int i=1; i<=10000000; i++)
    	A[i] = i;
    
    for(int i=2; i<=10000000; i++)
    {
    	if(A[i]==i)
    	{
    		for(int j=i; j<=10000000; j+= i)
    			A[j] = A[j]/i*(i-1);
    	}
    }
    
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int N;
    	scanf("%d", &N);
    	
    	long long ans = 0;
    	
    	for(int i=1; i<=N; i++)
    		ans+= A[i];
    	
    	printf("%lld\n", ans);
    }
    
    return 0;
}
