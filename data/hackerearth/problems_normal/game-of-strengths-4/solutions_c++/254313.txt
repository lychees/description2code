#include <cstdio>
#include <algorithm>

#define MOD 1000000007
using namespace std;


int main()
{
	int notc,n;
	scanf("%d", &notc);
	while(notc--)
	{
		scanf("%d", &n);
		long long int a[n];
		for(int i=0; i<n ;++i)
			scanf("%lld", &a[i]);
		sort(a, a+n);
		long long int sum=0, sum2=0;
		long long int maxa=-1;
		
		for(long long int i=n-1;i>=0;--i)
		{
			sum= (sum + (a[i]*i)%MOD)%MOD;
			if (a[i]>maxa)	maxa=a[i];
		}
		
		for(long long int i=n-1;i>=0;--i)
			sum2= (sum2 + (a[i]*(n-i-1))%MOD)%MOD;
		
		if (sum > sum2)
			sum = (sum-sum2)%MOD;
		else
			sum = (MOD+(sum-sum2))%MOD;
		printf("%lld\n", (sum*maxa)%MOD); 
	}
	return 0;
}
