#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	string a;
    	long int i,l,c=1;
    	cin>>a;
    	
    	l=a.size();
    	
    	for(i=0;i<l-1;i++){
    		if(a[i]=='X' && a[i+1]=='Y')c++;
    		else if(a[i]=='Y' && a[i+1]=='X')c++;
    	}
    	
    	cout<<l-c<<endl;
    }
    
    return 0;
}
