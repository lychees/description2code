#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int i,m,n,j,x,y,k,f,u;
    	cin>>n>>m;
    	string s;
    	int a[n][m],b[n][m];
    	for(i=0;i<n;i++){
    		cin>>s;
    		for(j=0;j<m;j++){
    			a[i][j]=int(s[j])-48;
    			b[i][j]=0;
    		}
    	}
    	k=0;
    	for(i=n-1;i>=0;i--){
    		for(j=m-1;j>=0;j--){
    			u = 0;
    			for(x=i;x<n;x++){
    				for(y=j;y<m;y++){
    						u += b[x][y];
    					}
    				}
    				if(a[i][j]==0 && u%2==0){
    					k++;
    					b[i][j]=1;
    				}
    				else if(a[i][j]==1 && u%2==1){
    					k++;
    					b[i][j]=1;
    				}
    			}
    		}
    	cout<<k<<endl;
    }
    return 0;
}
