import math as mt
times = input()
for _ in range(times):
    inp = raw_input()
    lst = inp.split()
    start = int(lst[0])

    end = int(lst[1])

    first = mt.sqrt(start)

    last = int(mt.sqrt(end))

    new = int(first)

    if (new*new == start):
        first = new
    else:
        first = new+1
    if last*last > end:
        last -= 1
    number = last+1-first
    print number
