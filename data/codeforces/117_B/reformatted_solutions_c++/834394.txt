int a, b, mod, m;
int main() {
  scanf("%d%d%d", &a, &b, &mod);
  m = min(a, mod);
  for (int i = 0; i <= m; i++)
    if ((mod - (i * 1000000000LL) % mod) % mod > b) {
      printf("1 %09d", i);
      return 0;
    }
  printf("2");
}
