#include <iostream>
using namespace std;
int gcd(int x,int y){
    if(x==0){
        return y;
    }
    if(y==0){
        return x;
    }
    int a;
    if(x>y){
        a=x;
        x=y;
        y=a;
    }
    return gcd(x,y%x);

}

int main()
{
    int n;
    cin >>n;
    int arr[100000];
    int i;
    for(i=0;i<n;i++){
      cin >>arr[i];
    }
    int ans =0;
    for(int i=0;i<n;i++){
      for(int j=i+1;(j<n );j++){
         if((arr[i]>arr[j]) &&(gcd(arr[i],arr[j])==1)){
            ans++ ;
         }
      }
    }
    cout <<ans;
    return 0;
    }
