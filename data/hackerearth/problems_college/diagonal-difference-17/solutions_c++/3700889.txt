#include <iostream>
using namespace std;

int main()
{
    int a[100][100],n,s=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		if(i==j)
    			s=s+a[i][j];
    	}
    	
    }
     int r=0;
    int c=n-1;
    int p=0;
    while(c>=0)
   {
     p=p+a[r][c];
    r++;
    c--;
	}
	int q;
	if(p>s)
		q=p-s;
	else
		q=s-p;
		
   cout<<q; 	
    
    return 0;
}
