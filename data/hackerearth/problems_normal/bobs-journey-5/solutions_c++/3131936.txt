#include<bits/stdc++.h>
using namespace std;
#define pdn(n) printf("%d\n",n)
#define sl(n) scanf("%lld",&n)
#define sd(n) scanf("%d",&n)
int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
	int n,i,flag=0,arr[26]={};
	string s;
	cin >> n;

	for(i=0; i<n; i++)
	{
	    cin >> s;
	    
	    arr[s[0]-'a']++;
	}
	for(i=0; i<26; i++)
	    if(arr[i]>1)flag=1;
	if(flag==0)
	    printf("YES\n");
	else
	    printf("NO\n");
    }
    return 0;
}
