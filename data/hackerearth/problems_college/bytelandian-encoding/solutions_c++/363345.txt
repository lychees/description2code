#include <iostream>
#include<string.h>
using namespace std;


int main()
{
    int t;
    cin>>t;
    
    for(int i=1;i<=t;i++)
    	{
    		string s,k;
    		cin>>s;
    		
    		int length=0;
    		length=s.length();
    	//	cout<<"length="<<length<<endl;
    		int length1=length;
    		int position=0,position1=0;;
    	    
    	    while(!s.empty())
    	    	{
    	    		if(length1%2!=0)
    	    			{
    	    				position=(length1+1)/2;
    	    				k.push_back(s[position-1]);
    	    				//cout<<"position "<<position<<endl;
    	    				s.erase(position-1,1);
    	    				//cout<<"k="<<k<<" "<<"s="<<s<<endl;
    	    				length1=s.length();
    	    			}
    	    		else if (length1%2==0)
    	    			{
    	    				position=length1/2;
    	    				//cout<<"position"<<position<<endl;
    	    				position1=position+1;
    	    				//cout<<"position1 "<<position1<<endl;
    	    				if((int)s[position-1]<=(int)s[position1-1])
    	    				 	{
    	    				 		k.append(1u,s[position-1]);
    	    						s.erase(position-1,1);
    	    				 	}
    	    				 else if((int)s[position1-1]<(int)s[position-1])
    	    				 {
    	    				 		k.append(1u,s[position1-1]);
    	    						s.erase(position1-1,1);
    	    				 }	
    	    				//cout<<"k="<<k<<" "<<"s="<<s<<endl; 	
    	    				length1=s.length();
    	    			}
    	    	}
    	    	cout<<k<<endl;;
    	}
    return 0;
}
