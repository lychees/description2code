#include <iostream>
#include <stdlib.h>
#include <string.h>

using namespace std;

void CheckStatus(char input[], int size)
{
 int arr[26];
 memset(arr, 0, sizeof(int)*26);
 for(int i = 0; i <size; i++ )
 {
 	arr[input[i]-'a'] = 1;
 }
 
 for(int i =0; i < 26; i++)
 {
 	if(arr[i] == 0)
 	{
 		cout<<"NO"<<endl;
 		return;
 	}
 	
 }
 cout<<"YES"<<endl;
}

int main()
{
    int n;
    cin>>n;
    char input[101];
    for(int i  = 0 ; i < n; i++)
    {
    	cin>> input;
    	CheckStatus(input, strlen(input));
    }
    return 0;
}
