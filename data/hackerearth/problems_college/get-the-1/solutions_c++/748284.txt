#include <iostream>
using namespace std;

bool Got1(int);

int main()
{
	int t = 0;
	int i;

	cin>>t;
	if(t>0 && t<=1000)
	{
		int *pN = new int[t];
		for(i=0; i<t; i++)
			cin>>pN[i];
		for(i=0; i<t; i++)
			cout<<"Case #"<<i+1<<":"<<(Got1(pN[i])?"YES":"NO")<<endl;
	}
	return 0;
}

bool Got1(int k)
{
	int s=0, r=0;

	while(true)
	{
		while(k!=0)
		{
			r = k%10;
			s += (r*r);
			k /= 10;
		}
		if((s/10) == 0) break;
		k = s;
		s = 0;
	}
	return (s==1?true:false);
}
