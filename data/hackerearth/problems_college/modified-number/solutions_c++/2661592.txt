#include <cmath>
#include <iostream>
using namespace std;
int A[500];
int main() {
	const int Base = 10;
	const int N = 7;
	int Paddy_cnt = 0;
	for (int nz=1; nz<=N; nz++)
		for (unsigned long long int k=pow((double)Base,nz-1); k<pow((double)Base,nz); k++)
			if ((k*(k-1))%(Base-1) == 0)
				for (int n=nz; n<nz*2; n++){
					const unsigned long long int B = pow((double)Base,n);
					const double nr = k*(B-k)/(B-1);
					const int q = k-nr;
					if ((k*k==q*B+nr && 0<nr)){
						A[Paddy_cnt++]=k;
						break;
				}}
	int a,b;
	cin>>a>>b;
	bool x=0;
	int i=0;
	while(A[i]<a)i++;
	while(A[i]<=b){
		x=1;
		cout<<A[i]<<" ";
		i++;
	}
	if(x==1)cout<<endl;
	else cout<<"INVALID RANGE\n";
	return 0;
}
