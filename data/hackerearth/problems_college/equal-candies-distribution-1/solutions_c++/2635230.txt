#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long n;
    	scanf("%ld",&n);
    	long arr[n];
    	long i,j,k;
    	for(i=0;i<n;i++)
    	  scanf("%ld",&arr[i]);
        sort(arr,arr+n);
        long diff[10005];
        k=0;

        for(i=1;i<n;i++)
        {
            diff[k++]=arr[i]-arr[i-1];
        }
        long long min_op=0;
        for(i=0;i<k;i++)
        {
        	long long tmp,x;
        	x=diff[i];
        	tmp=(x/5)+((x%5)/2)+(((x%5)%2)/1);
        	min_op+=tmp;
        	if(i<k-1)
        	 diff[i+1]+=diff[i];
        }
        cout<<min_op<<endl;
    }
    return 0;
}
