#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    for(int i=0;i<n;i++)
    	cin>>arr[i];
    vector<int> v1,v2;
    int sum1 = 0,sum2 = 0;
	sort(arr,arr+n);
	for(int i=0;i<n;i++){
		if(arr[i]&1){
			sum2 += arr[i];
			v2.push_back(arr[i]);
		}
		else{
			sum1 += arr[i];
			v1.push_back(arr[i]);
		}
	}
	v2.push_back(sum2);
	v1.push_back(sum1);
	for(int i=0;i<v1.size();i++){
		cout<<v1[i]<<" ";
	}
	for(int i=0;i<v2.size();i++){
		cout<<v2[i]<<" ";
	}
    return 0;
}
