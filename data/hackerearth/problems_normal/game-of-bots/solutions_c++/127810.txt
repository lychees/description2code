# include<cstdio>
# include<cstring>
# include<algorithm>
#include <iostream>
using namespace std;
int main ()
{
		int t=0;
		scanf("%d",&t);
		
		while(t--)
		{
				int a1=0,a2=0,b1=0,b2=0,c1=0,c2=0;
				char a[1000005];
				char b[1000005];
				scanf("%s %s",a,b);
				int posa=0,posb=0;
				int len =strlen(a);
				for(int i=0;i<len;i++)
				{
					if(a[i] == 'A') a1++;
					if(b[i] == 'A')	a2++;
					if(a[i] == 'B') b1++;
					if(b[i] == 'B')	b2++;
					if(a[i] == '#') c1++;
					if(b[i] == '#') c2++;
					
				}
				for(int i=0;i<len;i++)
				{
					if(a[i]=='A') {posa=i;}
				}
				for(int i=0;i<len;i++)
				{
					if(a[i]=='B') {posb=i;break;}
				}
				
				
				
			if(a1 != a2 && b1 != b2 && c1 != c2){ printf("No\n");continue;}
			int flag=0;
			for(int i=0;i<len;i++){
			
		//	printf("--> a %d b %d \n ",posa,posb);
			if((a[i] == 'A' && b[i] == 'B' ) || (a[i] == 'B' && b[i] == 'A'))
				{flag=1;break;}
				
			if( (a[i] == '#' && b[i] == 'A' && posa < i) || (a[i] == '#' && b[i] == 'B' && posb > i))
				{flag =1; break;}
			
			}
					
				
			if(flag ==1 ) printf("No\n");
			else printf("Yes\n");
				
		}
	return 0;
}
