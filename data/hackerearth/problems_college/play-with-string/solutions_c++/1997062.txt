#include <bits/stdc++.h>
using namespace std;
int main() {
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    string s,s1;
        cin>>s;
        s1=s;
        int f=0,i=0;
        reverse(s1.begin(),s1.end());
        for(i=1;i<s.size();i++)
            if(abs(s[i]-s[i-1])!=abs(s1[i]-s1[i-1]))f=1;
               if(f)
               cout<<"Not Funny"<<endl;
               else
               cout<<"Funny"<<endl;
               
    }
    return 0;
}
