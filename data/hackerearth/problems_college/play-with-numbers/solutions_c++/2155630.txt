#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    long int i,j,x,y,z,sqt,c;
    cin>>T;
    while(T--)
    {
        cin>>z>>x>>y;
        sqt=sqrt(z);
        c=0;
        for(i=1;i<=sqt;i++)
        {
            if(z%i==0)
            {
                j=z/i;
                if(i!=j)
                {
                    if(i<=x && j<=y)
                        c++;
                    if(i<=y && j<=x)
                        c++;
                }
                else
                {
                    if(i<=x && j<=y)
                        c++;
                }
            }
        }
        cout<<c<<"\n";
    }
}
