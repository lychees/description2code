def main():
    n = int(raw_input())
    lst = map(long, raw_input().split())
    summ = 0
    i = 0
    while i < n:
        j = i
        value = 0
        while j < n:
            value = value+lst[j]
            if value > summ:
                summ = value
            j = j+1
        i = i+1
    print summ


main()
