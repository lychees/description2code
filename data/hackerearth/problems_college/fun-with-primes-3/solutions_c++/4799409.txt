#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i;
    cin>>t;
    for(;t--;)
    {
    	long int n,sum=0;
    	cin>>n;
    	while(n>0)
    	{
    		sum+=(n%10);
    		n/=10;
    	}
    	for(i=2;i<=sum;i++)
    	{
    		if(sum%i==0)
    		break;
    	}
    	if(sum==i)
    	cout<<("YES\n");
    	else 
    	cout<<("NO\n");
    }
    return 0;
}
