#include <bits/stdc++.h>
using namespace std;
pair<int,int>arr[1001];
pair<int,int>brr[1001];
bool func(int n,int m,int d)
{
	int cnt=0,x,y,x1,y1;
	for(int i=0;i<m;i++)
	{
	    x=brr[i].first;
	    y=brr[i].second;
		for(int j=0;j<n;j++)
		{
		    x1=arr[j].first;
		    y1=arr[j].second;
		    if((abs(x-x1)+abs(y-y1))<=d){cnt++;break;}
		}
	}
	if(cnt>=n/2)return true;
	else return false;
}
int main()
{
    int t,x,y;
    cin >> t;
    while(t--)
    {
    	int n,m,d;
    	cin >> n >> m >> d;
    	for(int i=0;i<n;i++)
    	{
    		cin >> x >> y ;
    		arr[i]=make_pair(x,y);
    	}
    	for(int i=0;i<m;i++)
    	{
    		cin >> x >> y;
    		brr[i]=make_pair(x,y);
    	}
    	if(func(n,m,d))cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
