#include<bits/stdc++.h>
using namespace std;
#define f(i,a,b) for(int i=(int)(a);i<(int)(b);i++)
#define rf(i,a,b) for(int i=(int)(a);i>=(int)(b);i--)
#define s(n)                        scanf("%d",&n)
#define p(n)			    printf("%d\n",n)
#define sl(n)                       scanf("%lld",&n)
#define pl(n)			    printf("%lld\n",n)
#define ll long long int
int main()
{
    int t;
    s(t);
    while(t--)
    {
         char s[100001];
         int count=0;
         cin>>s;
         int n=strlen(s);
         for(int i=0;i<n;i++)
         {
             if(s[i]==s[i+1])
                count++;
         }
         p(count);
    }
    return 0;
}
