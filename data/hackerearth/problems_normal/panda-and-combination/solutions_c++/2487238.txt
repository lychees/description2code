#include<iostream>
#include<algorithm>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long n,k;
        long long c=0;
       scanf("%lld%lld",&n,&k);
        while(n>0)
        {
            c=c+n%k;
            n=n/k;
        }
        printf("%lld\n",c);
    }
    return 0;
}
