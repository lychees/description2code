#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char s[100];
    	cin>>s;
    	int l=strlen(s);
    	if(l%2==0)
    	{
    		int a[126]={0};
    		for(int i=0;i<l;i++)
    		{
    			a[s[i]]++;
    		}
    		int i;
    		for(i=0;i<126;i++)
    		{
    			if(a[i]%2!=0)
    			break;
    		}
    		if(i!=126)
    		cout<<"No"<<endl;
    		else 
    		cout<<"Yes"<<endl;
    		
    	}
    	else if(l%2==1)
    	{
    	int a[126]={0};
    		for(int i=0;i<l;i++)
    		{
    			a[s[i]]=a[s[i]]+1;
    		}	
    		int count=0;
    		for(int i=0;i<126;i++)
    		{
    			if(a[i]%2!=0)
    		count++;
    		}
    		if(count!=1)
    		cout<<"No"<<endl;
    		else 
    		cout<<"Yes"<<endl;
    	}
    }
}
