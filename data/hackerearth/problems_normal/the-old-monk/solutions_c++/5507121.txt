#include <cstdio>
#define ll long long int
#define l long int
#define MAX 100005
using namespace std;

ll a1[MAX],a2[MAX];

int compare (const void * a, const void * b)
{
  return ( *(int *)b - *(int *)a );
}

l binary_search(l start,l end,l key)
{
	l l_grt=0;
	while(start<=end)
	{
		l mid = (start+end)/2;

		if(a2[mid] >= key)
		{
			l_grt = mid;
			start = mid+1;
		}
		else
			end = mid-1;
	}
	return l_grt;
}

int main()
{
	int t;
	scanf("%d",&t);

	while(t--)
	{
		l n;
		scanf("%ld",&n);

		for(l i=0;i<n;i++)
			scanf("%lld",&a1[i]);

		for(l i=0;i<n;i++)
			scanf("%lld",&a2[i]);

		l max=0;
		for(l i=0;i<n;i++)
		{
			if (a1[i]<=a2[i])
			{
				l monkiness = binary_search(i+1,n-1,a1[i]);
				monkiness = monkiness - i;
				if(monkiness>max)
					max = monkiness;
			}	
		}
		printf("%ld\n",max);	
	}
	return 0;
}
