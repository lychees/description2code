#include <iostream>
#include<climits>
using namespace std;

int main()
{
    int t,s=0,max=INT_MIN;
    cin>>t;
    while(t--)
    {
    	int a;
    	cin>>a;
    	if((s+a)>0)
    	{
    		s=s+a;
    	}
    	else{
    		s=0;
    	}
    	if(max<s)
    	{
    		max=s;
    	}
    }
    cout<<max<<endl;
    return 0;
}
