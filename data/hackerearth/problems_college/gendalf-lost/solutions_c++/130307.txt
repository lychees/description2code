#include<iostream>
#include<stdlib.h>
#include<string.h>

using namespace std;

char arr[1000006][11];
int hashing[1000006];
struct node
{
	char str[11];
	int index;
};

int cmp(const void*x, const void* y)
{
	return strcmp((char*)x,(char*)y);
}

int cmpt(const void*x, const void* y)
{
	return strcmp(((struct node*)x)->str,((struct node*)y)->str);
}

int main()
{
	int n,a,b,i,q,j,mid,x;
	char str[11];

	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}

	qsort(arr,n,sizeof(char[11]),cmp);
	
	cin>>q;

	struct node *temp=new struct node[1000006];

	for(i=0;i<q;i++)
	{
		cin>>temp[i].str;
		temp[i].index=i;
	}
	qsort(temp,q,sizeof(struct node),cmpt);
	
	i=j=0;
	
	while((i<n)&&(j<q))
	{
		x=strcmp(arr[i],temp[j].str);

		if(x>0)
		{
			hashing[temp[j].index]=-2;
			j++;
		}
		else if(x<0)
			i++;
		else
		{
			hashing[temp[j].index]=i;
			j++;
		}
	}

	while(j<q)
	{
		hashing[temp[j].index]=-1;
		j++;
	}


	for(i=0;i<q;i++)
		cout<<hashing[i]+1<<endl;

	return 0;

}
