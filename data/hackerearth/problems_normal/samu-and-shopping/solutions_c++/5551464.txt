#include <bits/stdc++.h>
using namespace std;
long long int arr[100010][3],dp[100010][3];
int main()
{
    int t; cin>>t;
    while(t--){
    	int n; cin>>n;
    	memset(arr,0,sizeof(arr));
    	memset(dp,0,sizeof(dp));
    	for(int i=0;i<n;i++){
    		for(int j=0;j<3;j++){
    			cin>>arr[i][j];
    		}
    	}
    	for(int i=n-1;i>=0;i--){
    		if(i==n-1){
    			for(int j=0;j<3;j++) dp[i][j]=arr[i][j];
    		}
    		else{
    			dp[i][0]=min(dp[i+1][1],dp[i+1][2])+arr[i][0];
    			dp[i][1]=min(dp[i+1][0],dp[i+1][2])+arr[i][1];
    			dp[i][2]=min(dp[i+1][0],dp[i+1][1])+arr[i][2];
    			//cout<<i<<" "<<dp[i][0]<<dp[i][1]<<dp[i][2]<<endl;
    		}
    	}
    	cout<<min(dp[0][0],min(dp[0][1],dp[0][2]))<<endl;
    }
    
    return 0;
}
