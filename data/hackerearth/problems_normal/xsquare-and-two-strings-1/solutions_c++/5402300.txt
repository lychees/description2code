#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--){
    	string s1,s2;
    	cin>>s1>>s2;
    	
    	int a[30],b[30];
    	for(int i=0;i<30;i++){
    		a[i]=0;b[i]=0;}
    	
    	int n1=s1.length();
    	int n2=s2.length();
    	for(int i=0;i<n1;i++){
    		a[s1[i]-'a']++;
    	}
    	
    	for(int i=0;i<n2;i++){
    		b[s2[i]-'a']++;
    	}
    	
    	bool temp=false;
    	for(int i=0;i<26;i++){
    		if(a[i]>0 and b[i]>0)
    			temp=true;
    	}
    	
    	if(temp)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
}
