#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
    cin>>s;
    int arr[256];
    memset(arr,0,sizeof(arr));
    int c=0;
    for(int i=0;i<s.length();i++){
    	if(arr[s[i]]==0){
    		c++;
    		arr[s[i]]=1;
    	}else continue;
    }
    if(c%2==0){
    	cout<<"CHAT WITH HER!";
    }else{
    	cout<<"IGNORE HIM!";
    }
    return 0;
}
