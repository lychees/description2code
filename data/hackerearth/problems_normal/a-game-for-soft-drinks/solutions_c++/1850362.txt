#include<bits/stdc++.h>
#define max 101
using namespace std;
int dp[max][max][max];
int main()
{
	int i,j,k;
	dp[0][0][0]=-1;
int flag=0;
	for(i=0;i<max;i++)
	{
	  for(j=0;j<max;j++)
	  {
	     for(k=0;k<max;k++)
	     {
	        flag=0;
	           if(i>0 && dp[i-1][j][k]==-1)
	          flag=1;
	           if(j>0 && dp[i][j-1][k]==-1)
	          flag=1;
	           if(k>0 && dp[i][j][k-1]==-1)
	          flag=1;
	           if(i>0 && j>0 && dp[i-1][j-1][k]==-1)
	          flag=1;
	           if(i>0 && k>0 && dp[i-1][j][k-1]==-1)
	          flag=1;
	           if(j>0 && k>0 && dp[i][j-1][k-1]==-1)
	          flag=1;
	           if(i>0 && j>0 && k>0 && dp[i-1][j-1][k-1]==-1)
	          flag=1;
	           
	           if(flag==1)
	           dp[i][j][k]=1;
	           else
	           dp[i][j][k]=-1;
		 }
	  }
	}
	int t;
	cin>>t;
	while(t--)
	{
	   int n1,n2,n3;
	   cin>>n1>>n2>>n3;
	   
	   if(dp[n1][n2][n3]==-1)
	     cout<<"Aishwarya"<<endl;
	     else
	     cout<<"Ashima"<<endl;
	}
return 0;
}
