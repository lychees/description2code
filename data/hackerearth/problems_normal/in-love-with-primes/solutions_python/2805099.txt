sieve = [False] * 100000
for i in range(2, 100000):
	if not sieve[i]:
		for j in range(i + i, 100000, i):
			sieve[j] = True
def tc(n):
	for i in range(2, n // 2 + 1):
		if not sieve[i] and not sieve[n - i]:
			return True
	return False
for t in range(int(raw_input())):
	print 'Deepa' if tc(int(raw_input())) else 'Arjit'
