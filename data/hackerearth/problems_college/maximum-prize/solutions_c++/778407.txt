#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int main(){
    int t,i,n,k;
    unsigned long long sum;
    cin>>t;
    while(t--){
        sum =0;
        scanf("%d%d",&n,&k);
        int x[n];
        for(i=0;i<n;i++){
            cin>>x[i];
        }
        sort(x,x+n,greater<int>());
        for(i=0;i<k&&i<n;i++){
            sum +=x[i];
        }
        cout<<sum<<endl;
    }     
    return 0;
}
