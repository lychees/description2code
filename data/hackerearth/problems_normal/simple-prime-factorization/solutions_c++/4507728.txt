#include<bits/stdc++.h>
using namespace std;
bool prime[1000006];
int arr[1000006];
int p=0;
void seive()
{
	p=0;
	int i=0;
	prime[0]=prime[1]=1;
	for(i=2;i*i<=1000000;i++)
	{
		if(!prime[i])
		{
			int j=2;
			while(j*i<=1000000)
			{
				prime[j*i]=1;
				j++;
			}
		}
	}
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	int t,n,i,j;
	seive();
	scanf("%d",&t);
	while(t--)
	{
		int cnt=0,flag=0;
		scanf("%d",&n);
		if(!prime[n])
		{
			printf("2^0\n");
			continue;
		}
		for(i=2;i*i<=n;i++)
		{
			if(n%i==0)
			{
				cnt=0;
				while(n%i==0)
				{
					cnt++;
					n/=i;
				}
				flag=1;
				if(n!=1)
					printf("%d^%d*",i,cnt);
				else
					printf("%d^%d",i,cnt);
			}
			if(flag==0 && i==2)
			{
				printf("2^0*");
				flag=1;
			}
		}
		if(n>1)
		{
			printf("%d^1",n);
		}
		printf("\n");
	}
	//fclose(stdout);
	return 0;
}
