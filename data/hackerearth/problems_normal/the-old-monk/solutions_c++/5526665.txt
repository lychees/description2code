#include <bits/stdc++.h>
using namespace std;

int main(){
    int t;
    scanf("%d", &t);

    while(t--){
        int n, max = 0, flag = 0;
        cin >> n;
        long long int a[n], b[n];

        for(int i = 0; i < n; i++){
            //cin >> a[i];
            scanf("%lld", &a[i]);
        }

        for(int i = 0; i < n; i++){
            //cin >> b[i];
            scanf("%lld", &b[i]);
        }

        int p1 = 0, p2;

        while(p1 < n){

            p2 = p1;

            while(p2 < n){

                if((p2 >= p1) && (b[p2] >= a[p1])){
                    int monkiness = p2 - p1;
                    if(monkiness > max)
                        max = monkiness;
                    p2++;
                }
                else if(b[p2] < a[p1]){
                    break;
                }
            }

            p1++;

        }

        //cout << max << endl;
        printf("%d\n", max);

    }


    return 0;
}
