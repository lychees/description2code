def main():
	tc=int(raw_input())
	while(tc>0):
		m=int(raw_input())
		bin=raw_input()
		cnt1=0
		cnt2=0
		cnt3=0
		cnt4=0
		for i in range(0,2*m,1):
			if(i%2==0):
				if(bin[i]!=bin[0]):
					cnt1+=1
				if(bin[i]!=bin[1]):
					cnt3+=1
			else:
				if(bin[i]!=bin[1]):
					cnt2+=1
				if(bin[i]!=bin[0]):
					cnt4+=1
		ans=min(min(cnt1,cnt2),min(cnt3,cnt4))
		print ans
		tc-=1
main()
