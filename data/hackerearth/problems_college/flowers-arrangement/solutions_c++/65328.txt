#include<cstdio>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	int t,i,j,k;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		cin>>s;
		int flag=0;
		for(i=0;i<s.length();i++)
		{
			if(s[0]!='R')
			{
				flag=1;
				break;
			}
			if(s[i]=='Y')
			{
				if(s[i-1]=='R'||(s[i-1]=='Y'&&s[i-2]=='R'))
	            continue;
	            else
	            {
				flag=1;
				break;
	            }
			}
			else if(s[i]!='R'&&s[i]!='Y')
			{
				flag=1;
				break;
			}
			else 
			continue;
		}
		if(flag==0)
		printf("YES\n");
		else
		printf("NO\n");
	}
return 0;	
}
