template <class T>
inline void IN(T&p)
{
    register T n=0,sign=1,a=GC();
    while((a<'0')or(a>'9'))
    {
        if(a=='-')
        {
            sign=-1;
            a=GC();
            break;
        }
        a=GC();
    }
    while((a>='0')and(a<='9'))
    {
        n=(n<<1)+(n<<3)+a-'0';
        a=GC();
    }
    p=n*sign;
}

template <class T>
inline void OUT(T n)
{
    register char d[20];
    register int8_t i=0;
    if(n<0)
    {
        PC('-');
        n=-n;
    }
    do
    {
        d[i++]=n%10+'0';
        n=n/10;
    }while(n);
    while(i)
        PC(d[--i]);
    PC(nl);
}
