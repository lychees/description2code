#include <bits/stdc++.h>
using namespace std;

bool isprime(int n){
	for(int i=2;i*i<=n;i++){
		
		if(n%i==0)
			return false;
	}	
	return true;
}

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	if(n==0 || n==1 || n==2)
    		cout<<"4\n";
    	else if(isprime(n+1))
    		cout<<n+2<<endl;
    	else
    		cout<<n+1<<endl;
    }
    return 0;
}
