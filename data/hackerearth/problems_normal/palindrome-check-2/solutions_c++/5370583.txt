#include <iostream>
#include <string>
using namespace std;

int main()
{
    char s[100];
    cin>>s;
    int n=0;
    int count=0;
    for(int j=0;j<100;j++)
    {
    	if(s[j]!='\0')
    	{
    		n++;
    	}
    	else
    	{
    		break;
    	}
    }
    for(int i=0;i<(n/2);i++)
    {
    	if(s[i]==s[n-1-i])
    	{
    		count=0;	
    	}
    	else
    	{
    		count++;
    	}
    }
    if(count==0)
    {
    	cout<<"YES";
    }
    else
    {
    	cout<<"NO";
    }
    return 0;
}
