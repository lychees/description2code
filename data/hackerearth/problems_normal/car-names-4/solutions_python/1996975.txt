from collections import Counter
for _ in xrange(input()):
	s=list(raw_input())
	c=Counter(s)
	
	if s==list('carcar'):
		print "Not OK"
		continue
	
	if len(c)!=3:
		print "Not OK"
	else:
		temp=0
		flag=0
		for k,v in c.most_common(10000):
			if temp==0:
				temp=v
			elif temp!=v:
				flag=1
				break
		if flag:
			print "Not OK"
		else:
			print "OK"
