#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007

/*
F: 0 0 0 0
G: 0 1 d d 
H: 0 f f*d f*d f*d ....
*/

int main()
{
	ll a, b, c, d, e, f;
	cin >> a >> b >> c >> d >> e >> f;
	int q;
	cin >> q;
	while(q--){
		ll n;
		cin >> n;
		if(n == 0){
			cout << 0 << endl;
		} else if(n == 1){
			cout << f << endl;
		} else {
			cout << ((((n-1)*f)%MOD*d)%MOD+f)%MOD << endl;
		}
	}
    return 0;
}
