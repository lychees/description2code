#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		string s,a;
		cin >> s >> a;
		
		long int len = s.length(), lena = a.length();
		vector<long int> word(26,0);
		for(long int i=0;i<lena;i++)
			word[a[i]-97]++;
		
		for(long int i=0;i<len;i++)
		{
			long int x=s[i]-97;
			for(long int j=0;j<x;j++)
			{
				if(word[j] > 0)
				{
					s[i] = char(j+97);
					word[j]--;
					break;
				}
			}
		}
		cout << s << endl;
	}
	return 0;
}
