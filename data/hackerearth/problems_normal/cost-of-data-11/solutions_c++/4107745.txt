#include<iostream>
using namespace std;
typedef long long ll;
struct tree
{
    tree* next[26];
};
void tree(tree *l)
{
    
    for(int j=0;j<26;j++)
    {
        l->next[j]=NULL;
    }
    
}
ll c=1;
void insert(struct tree *root,string s)
{
    struct tree *p=new struct tree();
    p=root;
    for(int i=0;i<s.length();i++)
    {
        if(p->next[s[i]-'a']!=NULL)
            
        {
        
            p=p->next[s[i]-'a'];
        }
        else
        {
            p->next[s[i]-'a']=new struct tree();
            c++;
            p= p->next[s[i]-'a'];
        }
        
    }
}
int main()
{
	cin.sync_with_stdio(0);cin.tie(0);
    ll n;
    cin>>n;
    struct tree*k=new struct tree();
    while(n--)
    {
        string s;
        cin>>s;
        insert(k,s);
    }
    cout<<c;
}
