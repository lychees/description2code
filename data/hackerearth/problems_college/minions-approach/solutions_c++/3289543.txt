#include <iostream>
#include <cmath>

using namespace std;

int gcd(int x,int y){
	if(x==y)return x;
	int a=max(x,y)-min(x,y);
	int b=min(x,y);
	return gcd(a,b);
}

bool prime(int n){
	if(n==1)return false;
	if(n==2)return true;
	if(n%2==0)return false;
	int check=0;
	for(int i=3;i<=sqrt(n);i++){
		if(n%i==0){
			check++;
			break;
		}
	}
	if(check==0)return true;
	return false;
}

int main(){

	int T,N;
	cin >> T;
	int i,j;
	int result;

	while(T--){
		cin >> N;
		result=0;
		for(i=1;i<N;i++){
			if(gcd(i,N)==1)
				result++;
		}
		if(prime(result))
			cout<<"Correct"<<endl;
		else
			cout<<"Incorrect"<<endl;
	}

	return 0;
}
