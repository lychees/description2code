#include <algorithm>
#include <map>
#include <math.h>
#include <queue>
#include <set>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <vector>
using namespace std;
typedef long long ll;

const int inf = 1 << 29;

int a[2010];
int main() {
  int n;
  scanf("%d", &n);

  for (int i = 0; i < n; i++) {
    scanf("%d", &a[i]);
    a[i] = abs(a[i]);
  }
  int ans = 0;
  for (int i = 0; i < n; i++) {
    int x = 0, y = 0;
    for (int j = 0; j < i; j++)
      if (a[j] < a[i])
        x++;
    for (int j = i + 1; j < n; j++)
      if (a[j] < a[i])
        y++;
    ans += min(x, y);
  }
  printf("%d\n", ans);
}
