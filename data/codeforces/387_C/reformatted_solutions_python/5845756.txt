def lcmp(a, b):
    return a < b if len(a) == len(b) else len(a) < len(b)


line = readLine().strip()
ll = len(line)

cPos = 0
nPos = 0
prev = ''
Ans = 0

    nPos = 1
    while cPos + nPos < ll and line[cPos + nPos] == '0':
        nPos += 1
    cur = line[cPos: cPos + nPos]
    if lcmp(prev, cur):
        Ans = 1
    else:
        Ans += 1
    prev += cur
    cPos = cPos + nPos
