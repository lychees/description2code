unsigned long long int power(unsigned long long int x, unsigned long long int y, unsigned long long int BIGNUM)
{
	int retVal;
	long long pbytwo;
	if (x == 0) {
		retVal = 0;
		return retVal;
	}
	if (y == 0) {
		retVal = 1;
		return retVal;
	}
	if (y == 1) {
		retVal = x % BIGNUM;
		return retVal;
	}
	if (y % 2 == 0) {
		pbytwo = power(x, y/2, BIGNUM);	
		pbytwo =  (pbytwo * pbytwo);
		retVal = pbytwo % BIGNUM;
		return retVal;
	}
	else {
		pbytwo = power(x, (y-1)/2, BIGNUM);
		pbytwo =  (pbytwo * pbytwo);
		pbytwo = pbytwo % BIGNUM;
		pbytwo = pbytwo * x;
		retVal = pbytwo % BIGNUM;
		return retVal;
	}
}
