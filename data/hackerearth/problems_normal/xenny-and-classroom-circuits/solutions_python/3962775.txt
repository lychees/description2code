def process(A,n):
	pairs = 0
	bg_reversed = 0
	gb_reversed = 0
	for e in A:
		if e == ('B','G'):
			bg_reversed += 1
		elif e == ('G','B'):
			gb_reversed += 1
		else:
			pairs += 1
	return (pairs/2) + min(bg_reversed, gb_reversed)

for _ in xrange(input()):
	n = input()
	s = raw_input()
	A = zip(s[::2], s[1::2])
	print process(A, n)
