#include "bits/stdc++.h"
using namespace std;
const int N = 1e5 + 5;
int n;
long long d;
vector < pair < int , int > > v[N];
int t , x , y;
long long ans = 0;
int cur = 0;
long long done = 0;
int main(){
    scanf("%d %lld" , &n , &d);
    for(int i = 1 ; i <= n ; ++i){
        scanf("%d %d %d" , &t , &x , &y);
        v[t].emplace_back(make_pair(y , -x));
    }
    for(int i = 1 ; i < N ; ++i){
        done += cur;
        if(done >= d){
            break;
        }
        if(!v[i].empty()){
            auto it = *max_element(v[i].begin() , v[i].end());
            if(it.first > cur){
                cur = it.first;
                ans -= it.second;
            }
        }
    }
    printf("%lld\n" , ans);
}
