#include <iostream>
#include <unordered_map>
using namespace std;
#define boost std::ios::sync_with_stdio(false);

int main()
{   boost;
    char s[10002],m;
    int i;
    unordered_map<char,int> ctr;
    cin.getline(s,1002);
    for(i=0;s[i]!='\0';i++){
    	ctr[s[i]]++;
    }
    int max=ctr[s[0]];
    m=s[0];
    for(i=1;s[i]!='\0';i++){
    	if(ctr[s[i]]>max){
    		m=s[i];
    		max=ctr[s[i]];
    	}
    	else if(ctr[s[i]]==max){
    		if(s[i]<m){
    		m=s[i];
    		max=ctr[s[i]];
    		}
    	}
    }
    cout<<m<<" "<<max;
    return 0;
}
