#include <bits/stdc++.h>
using namespace std;

long long sol(vector<string> brd,int n,int m)
{
	long long ans=0;
	vector<vector<long long> > DP(n,vector<long long>(m,-1));
	for(int i=0;i<m;i++)
		DP[0][i]=1;
	for(int i=0;i<n;i++)
		DP[i][0]=1;
	for(int i=1;i<n;i++)
	{
		for(int j=1;j<m;j++)
		{
			if(brd[i][j]=='#')
			{
				if(brd[i-1][j-1]=='#' && brd[i-1][j]=='_' && brd[i][j-1]=='_')
				{
					DP[i][j]=1+min(DP[i-1][j-1],min(DP[i-1][j],DP[i][j-1]));
				}
				else
					DP[i][j]=1;
			}
			else
			{
				if(brd[i-1][j-1]=='_' && brd[i-1][j]=='#' && brd[i][j-1]=='#')
				{
					DP[i][j]=1+min(DP[i-1][j-1],min(DP[i-1][j],DP[i][j-1]));
				}
				else
					DP[i][j]=1;
			}
			if(brd[i][j]=='_' && DP[i][j]>=32)
				ans++;
		}
	}
	return ans;
}

int main()
{
    int m,n;
 	cin>>n>>m;
 	vector<string> brd;
 	string x;
 	for(int i=0;i<n;i++){
 		cin>>x;
 		brd.push_back(x);
 	}
 	
 	cout<<sol(brd,n,m);
    return 0;
}
