#include <iostream>
#include <vector>
#include <cstdio>
#include <cstring>
using namespace std;

typedef vector<int> vi;

bool isValid(vi &gate,vi &person){
	for(int i=0;i<26; i++)
		if(person[i]>gate[i])
			return false;
	return true;
}

int main()
{
    vi gate(26,0);
    vi person(26,0);
    char seq1[100001];
    char seq[100001];
    int t,length,length1;
    cin>>t;
    
    while(t--){
		for(int i=0;i<26;i++)
			gate[i]=0,person[i]=0;
		
		scanf("%s",seq1);
		length1 = strlen(seq1);
		scanf("%s",seq);
		length = strlen(seq);
		if(length > length1){
			printf("NO\n");
			continue;
		}
		
		for(int i=0;i<length;i++){
			person[seq[i]-'a']++;
			gate[seq1[i]-'a']++;
		}
		
		if(isValid(gate,person)){
			printf("YES\n");
			continue;
		}
		
		
		bool result = false;
		
		for(int i=1;i<=length1-length;i++){
			gate[seq1[i+length-1]-'a']++;
			gate[seq1[i-1]-'a']--;
			if(isValid(gate,person)){
				result = true;
				break;
			}
		}

		if(result)
			printf("YES\n");
		else
			printf("NO\n");
	}
    return 0;
}
