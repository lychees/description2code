#include<iostream>

using namespace std;
#define MOD 1000000007

int main()
{
	int t, n;

	long long int tribonacci[100010];
	tribonacci[0] = tribonacci[1] = 0;
	tribonacci[2] = 1;
	for(int i=3; i<100010; i++)
		tribonacci[i] = (tribonacci[i-1]+tribonacci[i-2]+tribonacci[i-3])%MOD;

	cin >> t;
	while(t--)
	{
		cin >> n;
		cout << tribonacci[n+2] << endl;
	}
}
