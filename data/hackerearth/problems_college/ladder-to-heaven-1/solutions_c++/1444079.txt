#include <iostream>
using namespace std;

int main()
{
    unsigned long long int t,n,l,l1,sum,k;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	l=0;
    	l1=1;
    	k=l1+l;
    	l1=k;
    	l=l1;
    	sum=0;
    	while(k<n)
    	{
    		k=l+l1;
    		l=l1;
    	   if(l1%2==0)
    	    {sum=sum+l1;}
    	 l1=k;
    	 }
    cout<<sum<<endl;
    }
    return 0;
}
