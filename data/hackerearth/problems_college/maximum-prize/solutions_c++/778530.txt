#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

long a[100003];
long n,k;

int main() {
    int t;
    cin >> t;
    while(t--){
        cin >> n >> k;
        if(n == 0){
            cout << "0\n";
        }
        else{
            for(long i = 0 ; i < n ; i++){
                cin >> a[i];
            }
            sort(a,a+n);
            long long prize = 0;
            long i = 0;
            while(k--){
                if(n-1-i < 0){
                    break;
                }
                prize = prize + a[n-1-i];
                i++;
            }
            cout << prize << "\n";
        }
    }
    return 0;
}
