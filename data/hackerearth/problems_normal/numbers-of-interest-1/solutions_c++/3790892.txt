#include <bits/stdc++.h>
using namespace std;
 
int main()
{
   long long int t;
   scanf("%lld",&t);
   while(t--)
   {
   		int x,y,n;
   		scanf("%d%d%d",&x,&y,&n);
   		long long int a[51]={0};
   		for(int i=0;i<x;i++)
   		a[i]=y;
   		for(int i=x;i<n;i++)
   		{
   			
   			for(int j=1;j<=x;j++)
   		{
   			a[i]=a[i]+a[i-j];
   		}}
   		printf("%lld\n",a[n-1]);
   }
}
