#include <iostream>
#include <cstring>
using namespace std;

int main() {
	
	int i,j,T,size;
	char t;
	string in,temp;
	
	cin>>T;
	
	string out[T];
	
	for(i=0;i<T;i++){
		cin>>in;
		size = in.size();
		
		t = in[0];
		temp = in[0];
		
		for(j=1; j<size; j++){
			if(in[j] == t)
				continue;
			else{
				temp = temp + in[j];
				t = in[j];
			}
		}
		
		out[i] = temp;
	}
	
	for(i=0;i<T;i++){
		cout<<out[i]<<endl;
	}
	return 0;
}
