#include<bits/stdc++.h>
using namespace std;

string F(string a, string b){
	int i;
	bool ok;
	for(i=0;i<a.size();i++){
		ok=true;
		for(int j=i;j<a.size();j++){
			if((j-i)>=b.size() || b[j-i]!=a[j]){
				ok=false;
				break;
			}
		}
		if(ok) break;
	}
	for(int j=a.size()-i;j<b.size();j++)
		a+=b[j];
	
	return a;
}

int A[20];
string s[20];
int main(){
	int n,x=0,ans=INT_MAX;
	string s2;
	cin >> n;
	for(int i=0;i<n;i++){
		cin >> s[i];
		x+=s[i].size();
		A[i]=i;
	}
	do{
		s2=s[A[0]];
		for(int i=1;i<n;i++){
			s2=F(s2,s[A[i]]);
		}
		if(s2.size()<ans)
			ans=s2.size();
	}while(next_permutation(A,A+n));
	
	cout << ans << endl;
    return 0;
}
