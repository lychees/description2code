m, n = map(int, raw_input().split())
ar = [int(i) for i in raw_input().split()]
dic = {}
maxCount = 0
maxAge = 0
for i in ar:
    if dic.get(i):
        dic[i] += 1
    else:
        dic[i] = 1
    if dic[i] > maxCount:
        maxCount = dic[i]
        maxAge = i
    elif dic[i] == maxCount and i > maxAge:
        maxAge = i

    print maxAge, maxCount
