#include <iostream>
using namespace std;

bool isprime(int n)
{
	if(n==1)
	return false;
	if(n==2)
	return true;
	if(n%2==0)
	return false;
	for(int i=3;i*i<=n;i+=2)
	if(n%i==0)
	return false;
	
	return true;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(!n)
    	{puts("4");continue;}
    	for(i=n+1;;i++)
    	{
    		if(!isprime(i))
    		break;
    	}
    	cout<<i<<endl;
    }
    return 0;
}
