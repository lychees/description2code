#define MOD 1000000007
int x[3333],y[3333];
int f[3333];
double re[3333];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n;
	double d;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x[i]>>y[i]>>f[i];
		re[i]=-1e38;
	}
	re[0]=f[0];
	d=re[0];
	for(i=1;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			re[i]=max(re[i],re[j]+f[i]-sqrt((long long)(x[i]-x[j])*(x[i]-x[j])+(long long)(y[i]-y[j])*(y[i]-y[j])));
		}
		d=max(re[i],d);
	}
	printf("%.6lf\n",re[n-1]);
//	system("pause");
	return 0;
}
