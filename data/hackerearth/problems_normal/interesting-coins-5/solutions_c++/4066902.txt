using namespace std;
ll a[1001],b[1001];
ll n;
bool check(ll mid)
{
	iloop(i,0,n)
	{
		b[i]=a[i];
	}
	b[0]-=mid;
	if(b[0] < 1)
		b[0] = 1;
	iloop(i,1,n)
	{
		if(b[i]+mid <= b[i-1])
			return false;
		else if(b[i]-mid <= b[i-1]+1 && b[i]+mid >= b[i-1]+1)
			b[i]=b[i-1]+1;
		else
			b[i]-=mid;
	}
	// iloop(i,0,n)
	// {
	// 	cout << b[i] << ' ';
	// }
	// cout << endl;
	return true;
}

int main()
{
	ll t;sc(t);
	while(t--)
	{
		sc(n);
		iloop(i,0,n)
		{
			sc(a[i]);
		}
		// check(3);	
		// check(2);
		ll start=0,final=20000000000,mid;
		while(final > start)
		{
			mid = (start+final)/2;
			if(check(mid))
			{
				final=mid-1;
			}
			else
			{
				start=mid+1;
			}
		}
		// cout << mid << endl;
		if(check(mid-1))
			cout << mid-1 << endl;
		else if(check(mid))
			cout << mid << endl;
		else if(check(mid+1))
			cout << mid+1 << endl;
		else
			cout << mid+2 << endl;
	}
}
