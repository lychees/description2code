#include<iostream>
#include<stdio.h>
long long int dp[213][213];
long long int solve(long long int n,long long int k)
{
if (dp[n][k] >=0)
return dp[n][k];
if (n<k)
return 0;
if (n<2*k)
return 1;
long long int result = 1;
for (int i=k; i<n; i++)
{
result = result + solve(n-i, i);
}
dp[n][k] = result;
return result;
}
int main()
{
long long int n,k,t;
scanf("%lld",&t);
for (int i=0; i<201; i++)
for (int j=0; j<201; j++)
dp[i][j] = -1;
while(t--)
{
scanf("%lld %lld",&n,&k);
long long int i =solve(n,k);
printf("%lld\n",i);
}
return 0;
}
