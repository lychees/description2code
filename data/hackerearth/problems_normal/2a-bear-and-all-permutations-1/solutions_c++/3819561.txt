#include <iostream>
#include <vector>
#include <string>
#include <cassert>
#include <algorithm>
using namespace std;
typedef long long ll;

int main()
{
	vector<string> v = {"1","2","8","68","1504","127792","57140352","258023200384","10151395367145472","3673835865235792306176","13318668301694192513859649536","531680718673514734573555796872790016"};
    int T;
    cin >> T;
    for (int test = 1; test <= T; ++test) {
    	int n,m;
    	cin >> n >> m;
    	assert(n <= v.size());
    	ll res = 0;
    	for (char c : v[n-1]) {
    		res = (10 * res + c - '0') % m;
    	}
    	cout << (res + m) % m << endl;
    }
    return 0;
}
