#include <bits/stdc++.h>

using namespace std;


int main()
{
    int t,i;
	scanf("%d\n",&t);
	long long a,b,count,n;
	while(t--)
	{
		count=0;
		cin>>n;
		a=1;
		b=1;
		while(count<n)
		{
			long long temp=b;
			b=a;
			a=temp+a;
			count+=a-b-1;
		}
		a--;
		while(count!=n)
		{
			a--;
			count--;
		}
		cout<<a<<endl;
	}
	
    return 0;
}
