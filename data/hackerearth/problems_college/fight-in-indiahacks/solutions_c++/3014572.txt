#include <iostream>
#include <string.h>
#define ll long long
using namespace std;
ll dp1[1000001], dp2[1000001];
ll solve(int n, int m1, int m2, int w)
{
	ll ans;
	if(n<m1&&n<m2)
	{
		if(w==1) return 1;
		else return 0;
	}
	if(w==1 && dp2[n]!=-1) return dp2[n];
	if(w==0 && dp1[n]!=-1) return dp1[n];
	if(n>=m1&&n>=m2)
	{
		if(w==0)
		{
			ans=solve(n-m1,m1,m2,1)||solve(n-m2,m1,m2,1);
			dp1[n]=ans;return ans;
		}
		else
		{
			ans=solve(n-m1,m1,m2,0)&&solve(n-m2,m1,m2,0);
			dp2[n]=ans;return ans;
		}
	}
	else if(n>=m1)
	{
		if(w==0)
		{
			ans=solve(n-m1,m1,m2,1);
			dp1[n]=ans;return ans;
		}
		else
		{
			ans=solve(n-m1,m1,m2,0);
			dp2[n]=ans;return ans;
		}
	}
	else
	{
		if(w==0)
		{
			ans=solve(n-m2,m1,m2,1);
			dp1[n]=ans;return ans;
		}
		else
		{
			ans=solve(n-m2,m1,m2,0);
			dp2[n]=ans;return ans;
		}
	}
}
int main()
{
	ll t, n, m1, m2;
	cin>>t;
	while(t--)
	{
		cin>>n>>m1>>m2;
		memset(dp1,-1,sizeof(dp1));
		memset(dp2,-1,sizeof(dp2));
		if(m1==0)
		{
			if(n%m2==0)
				cout<<0<<endl;
			else
				cout<<1<<endl;
		}
		else if(m2==0)
		{
			if(n%m1==0)
				cout<<0<<endl;
			else
				cout<<1<<endl;
		}
		else
			cout<<solve(n,m1,m2,0)<<endl;
	}
	return 0;
}
