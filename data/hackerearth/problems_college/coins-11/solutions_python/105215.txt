limit = 100100
a = [1] * limit    
a[0] = 0     
a[1] = 0
i = 2
while i*i<limit:
	if a[i]:
		j = i*i
		while j<limit:
			a[j] = 0
			j += i
	i += 1

            
t = int(raw_input())
while t:
	n = int(raw_input())
	if n == 3:
		print "3"
		t -= 1
		continue
	if a[n]:
		print "2"
	else:
		print "1"
	t -= 1
