#include <iostream>
#include<map>
#include<vector>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int ans =0;
	const char* p = s.c_str();
	for(int i=0;i<s.length();i++) {
		char a = *(p+i);
		int j =i+1;
		ans++;
		while(*(p+j)!='\0' && *(p+j) == a) {
			ans++;
			j++;
		}
	}
	cout<<ans<<endl;
    return 0;
}
