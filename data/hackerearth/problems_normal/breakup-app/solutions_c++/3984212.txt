#include <ctype.h>
#include <iostream>
#include <sstream>
#include <string>
#include <map>

int main()
{
    std::ios_base::sync_with_stdio(false);

    int n = 0;
    std::cin >> n;
    std::cin.ignore();

    std::map <int, int> m;

    while (n--) {
        std::string s;
        std::getline(std::cin, s);

        int length = s.size();
        std::string temps = "";
        int temp = 0;

        for (int i=0; i<length; ++i) {
            if (std::isdigit(s[i])) {
                temps += s[i];

                if(i+1<length && std::isdigit(s[i+1])) {
                    temps += s[i+1];
                    ++i;
                }

                std::stringstream(temps) >> temp;
                temps.clear();

                if (m.find(temp) != m.end()) {
                    if(s[0] == 'G') {
                        m[temp] += 4;
                    } else {
                        m[temp] += 3;
                    }
                } else {
                    if(s[0] == 'G') {
                        m.insert(std::make_pair(temp, 4));
                    } else {
                        m.insert(std::make_pair(temp, 3));
                    }
                }
            }
        }
        s.clear();
    }

    int date = 0, priority = 0;
    std::map <int, int>::iterator i;
    for (i = m.begin(); i != m.end(); ++i) {
        if (i -> second > priority) {
            priority = i -> second;
            date = i-> first;
        }
    }

    int isMultiple = 0;
    for (i = m.begin(); i != m.end(); ++i) {
        if (i -> second == priority) {
            ++isMultiple;
        }
    }

    if ( (date == 19 || date == 20) && isMultiple <=1) {
        std::cout << "Date\n";
    } else {
        std::cout << "No Date\n";
    }

    return 0;
}
