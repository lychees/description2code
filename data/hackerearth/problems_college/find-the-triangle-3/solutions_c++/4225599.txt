#include <stdio.h>
#include <iostream>
#include <stdlib.h>
#include <algorithm>

using namespace std;

int main()
{
	int t, n;
	cin >> t;
	while(t--)
	{
		int flag = 0;
		cin >> n;
		int *a = (int*) malloc(n*sizeof(int));
		for(int i=0; i<n; i++)
		{
			cin >> a[i];
		}
		sort(a, a+n);
		for(int i=n-1; i>=2; i--)
		{
			if(a[i] < a[i-1] + a[i-2])
			{
				cout << a[i-2] << " " << a[i-1] << " " << a[i] << endl;
				flag = 1;
				break;
			}
		}
		if(flag==0)
		{
			cout << "-1" << endl;
		}
	}
}
