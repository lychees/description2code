#include <iostream>
using namespace std;

int main()
{
	double cx[3];
	double cy[3];
	for (int i=0;i<3;i++)
	cin>>cx[i]>>cy[i];
	
	
	for (int i=0;i<3;i++)
	{
		for(int j=i+1;j<3;j++)
		{
			if (cx[i]>cx[j])
			{
				double temp=cx[i];
				cx[i]=cx[j];
				cx[j]=temp;
				temp=cy[i];
				cy[i]=cy[j];
				cy[j]=temp;
			}
			else if (cx[i]==cx[j])
			{
				if (cy[i]>cy[j])
				{
				double temp=cx[i];
				cx[i]=cx[j];
				cx[j]=temp;
				temp=cy[i];
				cy[i]=cy[j];
				cy[j]=temp;	
				}
			}
		}
	}
	
	double x=cx[0]+cx[1]+cx[2];
	double y=cy[0]+cy[1]+cy[2];
	for (int i=2;i>=0;i--)
	{
		printf("%0.4f %0.4f\n",(x-2*cx[i]),(y-2*cy[i]));
	}
	
    return 0;
}
