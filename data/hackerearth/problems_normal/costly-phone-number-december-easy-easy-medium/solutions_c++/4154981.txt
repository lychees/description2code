#include<bits/stdc++.h>
using namespace std;

int main()
{
	int c[10];
	int l;
	string s;
	int cases;
	cin>>cases;
	while(cases--)
	{
		for(int i=0;i<10;i++)cin>>c[i];
		cin>>l;
		cin>>s;
		bool update;
		while(1)
		{
			update=false;
			for(int i=0;i<10;i++)
			for(int j=0;j<10;j++)
			{
				int d=(i+j)%10;
				if(c[d]>c[i]+c[j])
				{
					update=true;
					c[d]=c[i]+c[j];
				}
			}
			if(!update)break;
		}
		int ans=0;
		for(int i=0;i<l;i++)ans+=c[s[i]-'0'];
		cout<<ans<<endl;
	}
	return 0;
}
