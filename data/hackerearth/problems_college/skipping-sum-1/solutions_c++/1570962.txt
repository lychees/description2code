#include<iostream>
#include<stdio.h>
using namespace std;

long long int sum[100000][10]={0};
long long int a[100000];
int main()
{
    int n,q;


  
   cin>>n>>q;
    for(int i=0;i<n;i++)
   cin>>a[i]; 
   	
   

    for(int i=0;i<=9;i++)
    for(int j=0;j<10;j++)
      for(int k=i;k<n;k=k+j+1)
              sum[i][j]+=a[k];







    for(int i=10;i<n;i++)
       for(int j=0;j<10;j++)
            sum[i][j]=sum[i-(j+1)][j]-a[i-(j+1)];






    // printf("%lld %lld\n",sum[1][0],sum[5][0]);
    while(q--)
    {   long long int l,r,k;
       
       cin>>l>>r>>k;
        l--;
        r--;

        long long int it=((r-l)/k)+1;

        r=l+(it)*k;



        long long int ans=(r<n)?sum[l][k-1]-sum[r][k-1]:sum[l][k-1];


       
        cout<<ans<<endl;

     }








}
