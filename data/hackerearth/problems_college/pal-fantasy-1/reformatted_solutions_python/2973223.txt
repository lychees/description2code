def cp(sa):
    n = len(sa)
    st = ""
    for i in range(n-1, -1, -1):
        st += sa[i]
    return st == sa


t = input()
for c in range(0, t):
    s = raw_input()
    n = len(s)
    if (cp(s)):
        print n
        continue
    mlength = 2*n
    for i in range(n, mlength):
        st = s+s[i-n::-1]
        # print st
        if (cp(st)):
            print len(st)
            break
