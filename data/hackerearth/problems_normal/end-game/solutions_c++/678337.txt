#include <iostream>
#include <cstdlib>

using namespace std;

int main()
{
	int t, n, a, b, c, d, move;
	int moves_left;
	int vert_dist_black;
	int hor_dist_black;
	cin >> t;
	while(t--)
	{
		cin >> n >> a >> b >> c >> d >> move;
		moves_left = abs(n-a);
		vert_dist_black = abs(n-c);
		hor_dist_black = abs(b-d);
		if(move)
		{
			--vert_dist_black;
			--hor_dist_black;
		}
		if( moves_left < hor_dist_black || moves_left < vert_dist_black)
			cout << "White Wins" << endl;
		else
			cout << "Draw" << endl;
	}
    return 0;
}
