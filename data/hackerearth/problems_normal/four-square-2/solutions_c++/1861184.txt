#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int i,T,cnt,N,U;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	cnt=0;
    	U=sqrt(N);
    	for(i=1;i<=U;i++)
    	{
    		if(N%i==0)
    		{
    			if(i%4!=0)
    				cnt+=i;
    				
    			if((N/i)%4!=0 && i*i!=N)
    			cnt+=(N/i);
    		}
    	}
    	cout<<cnt*8<<"\n";
    }
    return 0;
}
