import sys
a = raw_input()
l = len(a)
for i in range(0, l):
    print a[i],
    sys.stdout.softspace = 0
    if i != l-1:
        print '-',
        sys.stdout.softspace = 0
a = a[::-1]
print '\r'
for i in range(0, l):
    print a[i],
    sys.stdout.softspace = 0
    if i != l-1:
        print '-',
        sys.stdout.softspace = 0
