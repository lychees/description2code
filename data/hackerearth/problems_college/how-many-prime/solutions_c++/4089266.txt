#include <iostream>
#include <algorithm>
using namespace std;
typedef unsigned long long ull;
typedef long long ll;

int main()
{
    ull N = 1000000;
    bool isPrime[N+1];
    for (ull i = 2; i <= N; i++)
    {
        isPrime[i] = true;
    }
    for (ull i = 2; i*i <= N; i++)
    {
        if (isPrime[i])
        {
            for (ull j = i; i*j <= N; j++)
            {
                    isPrime[i*j] = false;
            }
        }
    }
    int t;
    cin>>t;
    while(t--)
    {
        ull m,c=0,l,i;
        cin>>m;
        i=m/2;
        while(c==0 && i>=2)
        {
            if(isPrime[i] && isPrime[m-i] && (i!=m-i))
            {
                l=i;
                c++;
            }
            i--;
        }
        if(c!=0)
        {
            cout<<l<<" "<<(m-l)<<endl;
        }
        else
        {
            cout<<"-1 -1"<<endl;
        }
    }
    return 0;
}
