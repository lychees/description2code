#include <iostream>
#include<math.h>
using namespace std;

int main() {
    long long int t;
   cin>>t;
   while(t--)
   {
   	long long int i,t,n,sum=0;
   cin>>n;
    for(i=1;(pow(10,i))<=n;i++)
    {
        sum=sum+(9*pow(10,i-1))*i;
    }
    sum=sum+(n-pow(10,i-1)+1)*i;
    cout<<sum<<endl;
   }
    return 0;
}
