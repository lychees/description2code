
#include <bits/stdc++.h>
using namespace std;
 
int zeroes(int  n)
{
    int count = 0;
    for (int i=5; n/i>=1; i *= 5)
          count += n/i;
    return count;
}
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	printf("%d\n",zeroes(n));
    }
    return 0;
}
