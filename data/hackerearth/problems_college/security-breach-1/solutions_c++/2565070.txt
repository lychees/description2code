#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t;
	char str[20];
	int i,len;
    cin>>t;
    for(t;t>0;t--)
    {
    	cin>>str;
    	len=strlen(str);
    	for(i=0;i<len;i++)
    	{
    		if(str[i]=='2')
    		{
    			cout<<"cde";
    		}
    		else if(str[i]=='7')
    		{
    			cout<<"acf";
    		}
    		else if(str[i]=='6')
    		{
    			cout<<"b3";
    		}
    		else if(str[i]=='9')
    		{
    			cout<<"c6a";
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
