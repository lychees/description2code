#include <stdio.h>
using namespace std;

#define MAXSIZE 100002

int a[MAXSIZE];
long long s[11][MAXSIZE];

long long skipping_sum(int l,int r,int k)
{
	if(r-l < k)
		return a[l];
	
	int mod = (r-l)%k;
	if(k>l)
		return s[k][r-mod];
	return s[k][r-mod] - s[k][l-k];
	
	//while (l <= r)
	//{
	  //  sum = sum + a[L];
	   // l = l + k;
	//}
	//return sum;
}
	
int main()
{
	int n, q, l, r , k;
	scanf("%d %d", &n, &q);
	int i=1;
	a[0] = 0;
	while(i<=n)
	{
		scanf("%d", &a[i]);
		s[1][i] = s[1][i-1]+a[i];
		if(i>2)
			s[2][i] = s[2][i-2]+a[i];
		else
			s[2][i] = a[i];
		if(i>3)
			s[3][i] = s[3][i-3]+a[i];
		else
			s[3][i] = a[i];
		if(i>4)
			s[4][i] = s[4][i-4]+a[i];
		else
			s[4][i] = a[i];
		if(i>5)
			s[5][i] = s[5][i-5]+a[i];
		else
			s[5][i] = a[i];
		if(i>6)
			s[6][i] = s[6][i-6]+a[i];
		else
			s[6][i] = a[i];
		if(i>7)
			s[7][i] = s[7][i-7]+a[i];
		else
			s[7][i] = a[i];
		if(i>8)
			s[8][i] = s[8][i-8]+a[i];
		else
			s[8][i] = a[i];
		if(i>9)
			s[9][i] = s[9][i-9]+a[i];
		else
			s[9][i] = a[i];
		if(i>10)
			s[10][i] = s[10][i-10]+a[i];
		else
			s[10][i] = a[i];
		i++;
	}
	i = 0;
	while(i<q)
	{
		scanf("%d %d %d", &l, &r, &k);
		
		printf("%lld\n", skipping_sum(l, r, k));
		
		i++;
	}
	return 0;
}
