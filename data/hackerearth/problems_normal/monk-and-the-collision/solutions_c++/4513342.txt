#include <iostream>
using namespace std;

int main() {
	int t;cin>>t;
	while(t--)
	{
		int c[10]={0};
		int x;cin>>x;
		while(x--)
		{
			int temp;cin>>temp;
			int z=temp%10;
			c[z]++;
		}
		int sum=0;
		for(int i=0;i<10;i++)
		{
			if(c[i]>=1)
			{
				sum=sum+c[i]-1;
			}
		}
		cout<<sum<<endl;
	}
	return 0;
}
