#include <iostream>
using namespace std;
int K,A,P[100],I,J,T;
float EU;
void factor(int X)
{
    I=2;
    K=0;
    while(X>1)
    {
        if((X%I)==0)
        {
            X=X/I;
            P[K]=I;
            K++;
        }
        else
        {
            I++;
        }
    }
}
int main()
{
    cin>>T;
    for(J=0;J<T;J++)
    {
       cin>>A;
       factor(A);
       if((A==1)||(K==1))
       { 
       	if(A==1)
           cout<<1<<endl;
        else
           cout<<A-1<<endl;
       }
       else
       {
           EU=A*(P[0]-1)/P[0];
           for(I=1;I<K;I++)
           {
              if(P[I]!=P[I-1])
                 EU=EU*(P[I]-1)/P[I];
           }
           cout<<EU<<endl;
       }
    } 
    return 0;
}
