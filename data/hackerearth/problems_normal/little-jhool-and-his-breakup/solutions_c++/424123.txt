#include<iostream>
#include<string>
using namespace std;
int main()
{
	int i,k=0;
	string s;
	char store[4]={'l','o','v','e'};
	cin>>s;
	for(i=0;s[i]!='\0';i++)
	{
		if(s[i]==store[k])
		{  
			k++;
		}
		if(k==4)break;
	}
	if(k==4)
		cout<<"\nI love you, too!";
	else 
		cout<<"\nLet us breakup!";
	return 0;
}
