#include <iostream>
#include <vector>
using namespace std;

void print_vec(vector<int> v){
	for(auto e : v)
		cout<<e<<",";
	cout<<endl;
}
int main()
{
    int n;
    cin>>n;
    int a[100001];
    vector<int>people_rem;
    for(int i = 0 ; i < n ; i++){
    	cin>>a[i];
    	people_rem.push_back(i+1);
    }
    int curr_people = 1;
    int curr_index = 0;
    int next_index = 0;
    while(people_rem.size() > 1){
    	cout<<curr_people<<" ";
    	people_rem.erase(people_rem.begin()+next_index);
    	curr_index = next_index;
    	//cout<<"people_rem.size = "<<people_rem.size()<<endl;
    	//cout<<"people_rem = ";print_vec(people_rem);
    	next_index = (curr_index + (a[curr_people-1]-1)%people_rem.size())%people_rem.size();
    	//cout<<"next_index = "<<next_index<<endl;
    	curr_people = people_rem[next_index];
    }
    cout<<people_rem[0];
    return 0;
}
