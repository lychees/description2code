#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ll t,cur,n,k,i,m,j;
	cin>>n>>t;
	ll a[n+1],p[n+1],b[n+1];
	for(i=1;i<=n;i++)
	cin>>a[i];
	for(i=1;i<=n;i++)
	cin>>p[i];
	vector<bool> visited(n+1,0);
	//visited[1]=1;
	vector<vector<ll> >cycle;
	for(i=1;i<=n;i++)
	{
		if(!visited[i])
		{
			vector<ll> v;
			cur=i;
			while(!visited[cur])
			{	
				visited[cur]=1;
				v.push_back(cur);
				cur=p[cur];		
			}
			cycle.push_back(v);
		}
		
	}
	for(i=0;i<cycle.size();i++)
	{
		m=t%cycle[i].size();
		for(j=0;j<cycle[i].size();j++)
		{
			b[cycle[i][j]]=a[cycle[i][(j+m)%cycle[i].size()]];
		}
	}
	for(i=0;i<n;i++)
	cout<<b[i+1]<<" ";
	cout<<endl;
}
