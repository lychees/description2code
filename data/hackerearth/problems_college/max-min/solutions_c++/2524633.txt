#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,dif=1000000005,i;
    scanf("%d",&t);
    int a[t];
    int k;
    scanf("%d",&k);
    for(i=0;i<t;i++) scanf("%d",&a[i]);
    sort(a,a+t);
    for(i=0;i<t-k;i++)
    {
    	if(a[i+k-1]-a[i]<dif) dif=a[i+k-1]-a[i];
    }
    printf("%d",dif);
    return 0;
}
