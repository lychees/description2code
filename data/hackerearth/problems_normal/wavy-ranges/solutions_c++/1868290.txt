long long count(long long x)
{
	if (x==0)return 0;
	vector<long> v;
	long long X=x;
	
	while(x){v.push_back(x%10);x/=10;}
	
	for (int i=0;i<=v.size();i++)
	 for (int last=0;last<=9;last++)
	  for (int mov=0;mov<=1;mov++)
	   for (int les=0;les<=1;les++)
	    dp[i][last][mov][les]=0;
	
	reverse(v.begin(),v.end());
	
	for (int i=1;i<=v.size();i++)
	 	for (int j=1;j<=9;j++)
	 	 if (i>1||j<=v[0])
	 	 dp[i][j][1][(i>1||j<v[0])]=dp[i][j][0][(i>1||j<v[0])]=1;
	
	for (int i=1;i<v.size();i++)
	 for (int last=0;last<=9;last++)
	  for (int mov=0;mov<=1;mov++)
	   for (int les=0;les<=1;les++)
	    for (int ad=0;ad<=9;ad++)
	    {
	    	if (ad>v[i]&&les==0)continue;
	    	if (ad==last)continue;
	    	if (ad>last&&mov==0)continue;
	    	if (ad<last&&mov==1)continue;
	    	dp[i+1][ad][1^mov][les|(ad<v[i])]+=dp[i][last][mov][les];
	    }
	
	long long res=0;
	for (int last=0;last<=9;last++)
	 for (int les=0;les<=1;les++)
	  for (int mov=0;mov<=1;mov++){
//	  	cout<<last<<" "<<les<<" "<<mov<<" "<<dp[v.size()][last][les][mov]<<endl;
	   res+=dp[v.size()][last][les][mov];}
	for (int i=1;i<=9;i++)
	if (i<=X)--res;
	return res;
}
