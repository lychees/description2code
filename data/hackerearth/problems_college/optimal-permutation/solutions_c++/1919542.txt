#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,n,ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ans=0;
		if(n==1)
		{
			ans=0;
		}
		else
		{
		if(n%2==0)
		{
			ans=ans+(n-1);
			for(int i=n-2;i>=2;i-=2)
			ans=ans+2*i;
		}
		else
		{
			ans=ans+(n-1);
			for(int i=n-2;i>=1;i-=2)
			ans=ans+2*i;
			ans=ans-1;
		}
		}
		cout<<ans<<endl;
	}
	return 0;
}
