#include <iostream>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    for(int i = n + 1; i < m; i++) {
        if( (i % 3 == 0) && (i % 5 == 0)) {
            cout << i;
        } else if( (i % 3 == 0) || (i % 5 == 0)) {
            continue;
        } else {
            cout << i;
        }
        if (i < (m - 1)) {
            cout << ",";
        }
    }
    cout << "\n";
    return 0;
}
