#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int i,j,n;
    cin>>n;
    vector<int>marks(n);
    for(i=0;i<n;i++)
    cin>>marks[i];
    for(i=0;i<n;i++)
    {
    	int rank=0;
    	for(j=0;j<n;j++)
    	{
    		if(marks[j]>marks[i])
    		  rank++;
    	}
    	cout<<rank+1<<" ";
    }
    return 0;
}
