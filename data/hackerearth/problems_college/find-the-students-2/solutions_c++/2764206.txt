#include<bits/stdc++.h>
using namespace std;
main()
{
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sort(arr,arr+n);
	int min=arr[0],count,size=n;
	cout<<n<<endl;
	//cout<<min<<endl;
	while(size!=0)
	{
		for(int i=0;i<n;i++)
		{
			if(arr[i]!=0)
			{
				arr[i]=arr[i]-min;
			}
		}
		for(int j=0;j<n;j++)
		{
			if(arr[j]!=0)
			{
				min=arr[j];
				break;
			}
		}
		int tempsize=0;
		for(int k=0;k<n;k++)
		{
		//	cout<<arr[k]<<" ";
			if(arr[k]!=0)
			tempsize++;
		}
		if(tempsize!=0)
		cout<<tempsize<<endl;
		if(tempsize==0)
		{
			size=0;
			break;
		}
		//cout<<endl;
		//cout<<size<<endl;
	}
}
