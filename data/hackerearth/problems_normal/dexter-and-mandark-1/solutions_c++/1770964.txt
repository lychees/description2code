#include <iostream>
using namespace std;

int T,N,M,res;
int DP[1001][1001];
int modulo = 1000000007;
int det(int a, int b);

int main()
{
    scanf("%d",&T);
    for(int i=0;i<T;i++){
    	scanf("%d%d",&N,&M);
    	res = 0;
    	
    	for(int x=0;x<=N;x++){
    		for(int y=0;y<=N;y++){
    			DP[x][y] = -1;
    		}
    	}
    	
    	for(int j=0;j<=N/(M+1);j++){
    		if(N-j > j)
        		res = (res + det(N-j,j))%modulo;
        }
    	printf("%d\n",res);
    }
    return 0;
}


 int det(int a,int b){
    	if(DP[a][b] != -1)return DP[a][b];    	
    	
    	if(b == 0)return 1;
    	
    	int res = det(a,b-1)%modulo;
    	
    	if((a-1)>=M*b){
    		res = (res + det(a-1,b))%modulo;
    	}
    	
    	
    	DP[a][b] = res;
    	return res;
    }
