#include <iostream>
#include<queue>
#include<utility>
#include<map>
#include<vector>
#define ll long long
using namespace std;

class mycompare {
	
	public:
	bool operator() (pair<ll, ll> p1, pair<ll, ll> p2) {
		if(p1.first != p2.first) {
			return p1.first < p2.first;
		} else {
			return p1.second < p2.second;
		}
	}
};


int main()
{
	int n;
	cin >> n;
	priority_queue<pair<ll, ll>, vector<pair<ll,ll>>, mycompare> q;
	map<ll, ll> m;
	ll id, oldScore, p, l, c, s, newScore;
	while(n--) {
		cin >> id >> oldScore >> p >> l >> c >> s; 
		newScore = p*50 + 5*l + 10*c + 20*s;
		q.push(make_pair(newScore-oldScore, id));
		m.insert(make_pair(id, newScore));
	}
	
	for(int i = 1; i <= 5; i++) {
		ll tid = q.top().second;
		cout << tid << " " << m.find(tid)->second << endl;
		q.pop();
	}
    return 0;
}
