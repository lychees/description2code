#include <bits/stdc++.h>
using namespace std;
int arr[21][21],vis[21][21];
int n;
int temp=0;
int DFS(int x,int y)
{
//	cout<<"JJJ"<<endl;
	if(x==n-1&&y==n-1)
	{
		return 1;
	}
	vis[x][y]=1;
	if(x+1<n&&!vis[x+1][y]&&!arr[x+1][y])
	{
		temp+=DFS(x+1,y);
	}
	if(x>0&&!vis[x-1][y]&&!arr[x-1][y])
	{
	//	vis[x][y]=1;
		temp+=DFS(x-1,y);
	}
	if(y+1<n&&!vis[x][y+1]&&!arr[x][y+1])
	{
	//	vis[x][y]=1;
		temp+=DFS(x,y+1);
	}
	if(y>0&&!vis[x][y-1]&&!arr[x][y-1])
	{
	//	vis[x][y]=1;
		temp+=DFS(x,y-1);
	}
	vis[x][y]=0;
	return 0;
	//return temp;
}
int main() {
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		memset(arr,0,sizeof(arr));memset(vis,0,sizeof(vis));
		temp=0;
		cin>>n;
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				cin>>arr[i][j];
		DFS(0,0);
		//cout<<endl;
		cout<<temp<<endl;
	}
	return 0;
}
