t = int(raw_input())
for cases in range(t):
    n = int(raw_input())
    a = [int(i) for i in raw_input().strip().split()]
    if n == 1:
        print 0
    else:
        a.sort(reverse=True)
        b = [None] * (n - 1)
        b[0] = a[0] + a[1]
        sumTime = b[0]
        for i in range(1, n - 1):
            b[i] = b[i - 1] + a[i + 1]
            sumTime += b[i]
        print sumTime
