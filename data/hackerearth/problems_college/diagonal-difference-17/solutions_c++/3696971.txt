#include <cstdio>
using namespace std;

int main()
{
    int N,pd=0,sd=0,i,j,a[1000][1000];
    scanf("%d",&N);
    for(i=0;i<N;i++)
      for(j=0;j<N;j++)
        scanf("%d",&a[i][j]);
    for(i=0;i<N;i++)
      pd+=a[i][i];
    for(i=0;i<N;i++)
      sd+=a[i][N-i-1];
    if(pd>sd)  
      printf("%d\n",pd-sd);
    else
      printf("%d\n",sd-pd);
}
