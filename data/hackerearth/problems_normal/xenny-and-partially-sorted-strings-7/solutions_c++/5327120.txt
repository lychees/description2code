bool mycmp(string a,string b)
{
	string A=a.substr(0,m);
	string B=b.substr(0,m);
	if(A!=B)
	{
		return A<B;
	}
	else if(A==B) 
	{
		return mp[a]<mp[b];
	}
	
}
