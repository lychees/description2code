#include <iostream>
#include <vector>
using namespace std;
 
int main()
{
  string str;
  cin >> str;
  int n = str.size();
  std::vector<std::vector<int> > dp(n, std::vector<int>(n, 0));
  int rs = 0;
  for (int k = 0; k < n; k++) {
    for (int i = 0; i < n - k; i++) {
      if (k == 0) dp[i][i + k] = 1;
      else if (k == 1) dp[i][i + k] = str[i] == str[i + k] ? 1 : 0;
      else {
        if (dp[i + 1][i + k - 1] && str[i] == str[i + k]) dp[i][i + k] = 1;
      }
      if (dp[i][i + k] == 1) rs++;
    }
  }
  cout << rs <<endl;
}
