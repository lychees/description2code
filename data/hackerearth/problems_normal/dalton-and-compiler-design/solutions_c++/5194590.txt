#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t;
    long long i,a;
    scanf("%lld",&t);
    long long n[t];
    for(i=0;i<t;i++)
    {
        scanf("%lld",&n[i]);
    	if (n[i]%2==0)
    	{
    		a=(n[i]/2);
    		printf("%lld %lld\n",a,a);
    	}
    	if(n[i]%2!=0)
    	{
    		a=((n[i]-1)/2);
    		printf("%lld %lld\n",a,a+1);
    	}
    }
    return 0;
}
