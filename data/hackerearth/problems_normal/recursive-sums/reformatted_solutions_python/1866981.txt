def calc(ans):
    if len(ans) == 1:
        print(ans)
        return
    else:
        temp = 0
        for i in xrange(len(ans)):
            temp += int(ans[i])
        calc(str(temp))


t = int(raw_input())
for k in xrange(t):
    m = int(raw_input())
    ans = 0
    for o in xrange(m):
        l, d = map(int, raw_input().split())
        ans += (l*d)
    calc(str(ans))
