#include <iostream>
using namespace std;

int main() {
	
	int ctr,max;
	string str;
	string fin;
	max = 0;
	while(cin >> str) {

		ctr =0;
		for(int i=0; i<str.length(); i++) {
			if(isdigit(str[i])==true)
				ctr += str[i]-'0';
		}
		if(ctr > max) {
			fin = str;
			max = ctr;
		}
	}
	if(fin.length()>0)
	cout << fin << endl;
	else
	cout << '0' << endl;

	return 0;
}
