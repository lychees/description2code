#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
  	ll i,val,t,n;
  	map<ll,string> m;
  	map<ll,string>::iterator it;
  	map<ll,ll>::iterator it1;
  	map<ll,ll> ind;
  	val=0;
  	i=1;
  	while(val<=1e17)
  	{
  		val=val+i*(i+1)*(i+1);
  		if(i%2)
  			m[val]="Dhruv";
  		else
  			m[val]="Ketan";
  			ind[val]=i;
  			i++;
  	}
  	scanf("%lld",&t);
  	while(t--)
  	{
  		scanf("%lld",&n);
  		if(n<4)
  		{
  			printf("0\n");
  			printf("Draw\n");
  		}
  		else
  		{
  			it=m.lower_bound(n);
  			if((it->first)!=n)
  				it--;	
  			printf("%lld\n%s\n",ind[it->first],(it->second).c_str());
  		}
  	}
  	return 0;
}
