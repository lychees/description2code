#include<iostream>
#include<math.h>
using namespace std;
bool isPrime(long long int n){
	if(n<=1){
		return false;
	}
	if(n==2){
		return true;
	}
	if(n%2==0){
		return false;
	}
	long long int ix=3;
	long long int lim=sqrt(n)+1;
	while(ix<=lim){
		if(n%ix==0){
			return false;
		}
		ix+=2;
	}
	return true;
}

int main()
{
    long long int n;
    cin>>n;
    long long int result=0;
    long long int temp=0;
    if(n>=2)
    { result=1; temp=1;}
    for(long long int i=3;i<=n;i++)
    {
        if(isPrime(i))
        {
            temp*=2;
        }
        result+=temp;
          //cout<<result<<" ";
    }
    cout<<result;
}
