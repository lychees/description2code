#include <iostream>
#include <string>
using namespace std;

int main()
{
	int t,i,n1=0,n2=0,j=0,k=0,c=0,l,m;
	string s1,s2;
	cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>s1>>s2;
    	
    	while(s1[j]!='\0')
    	 { n1++;
    	   j++; }
        while(s2[k]!='\0')
         { n2++;
           k++; }
           
         if(n1==n2)
         {
         	for(l=0;l<n1;l++)
         	{
         		for(m=0;m<n2;m++)
         		{
         			if(s1[l]==s2[m])
         			{
         				s2[m]=0;
         				c++;
         				break;
         			}
         			  
         		}
         	}
         }
         if(c==n1)
           cout<<"Yes"<<endl;
         else
           cout<<"No"<<endl;
         c=n1=n2=j=k=0;
         
    }
    return 0;
}
