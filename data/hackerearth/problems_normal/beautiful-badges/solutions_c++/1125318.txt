#define FILE 0

#define MAX 1111

using namespace std;

int N, C, D, ans;
vector<int> b, y;

int main() {
    
    if(FILE) {
        freopen(INPUT, "r", stdin);
        freopen(OUTPUT, "w", stdout);
    }
    
    scanf("%d", &N);
    
    for(int i = 0; i < N; i++) {
        scanf("%d %d", &C, &D);
        if(C == 1) b.PB(D);
        else y.PB(D);
    }
    
    sort(b.begin(), b.end());
    sort(y.begin(), y.end());
    
    int i = 0, j = 0;
    
    while(i < b.size() && j < y.size()) {
        if(b[i]*b[i]*2 > y[j]*y[j] && y[j]*y[j]*2 > b[i]*b[i]) {
            ans++;
            i++;
            j++;
            continue;
        }
        else if(b[i]*b[i]*2 > y[j]*y[j]) j++;
        else i++;
    }
    
    printf("%d\n", ans);
    
    return 0;
}
