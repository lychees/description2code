#include<bits/stdc++.h>
using namespace std;
 main()
{
	int t;
	long long a,b,count,sum,r;
	scanf("%d",&t);
	
	while(t--)
	{
		count=sum=r=0;
		
		scanf("%lld%lld",&a,&b);
		
		while(a>0||b>0)
		{
			sum=0;
			sum=sum+r+a%10+b%10;
			a=a/10;
			b=b/10;
			r=0;
			
			if(sum>=10)
			{
		    r++;
			count++;
			}
			
		}
		
		if(count==0)
		printf("No carry operation\n");
		else if(count==1)
		printf("1 carry operation\n");
		else
		printf("%lld carry operations\n",count);
	}
	
}
