#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
using namespace std;
vector<ll> a,b;
int main(){
	ll t,n,c,x;
	inp(t);
	while(t--){
		inp(n);
		for(ll i=0;i<n;++i){
			inp(x);
			a.push_back(x);
		}
		for(ll i=0;i<n;++i){
			inp(x);
			b.push_back(x);
		}
		sort(a.rbegin(),a.rend());
		sort(b.begin(),b.end());
		c = 0;
		for(ll i=0;i<n;++i){
			if(a[i]%b[i]==0||b[i]%a[i]==0){
				++c;
			}
		}
		printf("%lld\n",c);
		a.clear();b.clear();
	}
	return 0;
}
