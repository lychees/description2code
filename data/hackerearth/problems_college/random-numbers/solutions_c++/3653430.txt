const int MOD = 1e9 + 7;
int modExp(int a, int n, int p) {
	int ret = 1;
	for (; n; n >>= 1, a = (LL)a * a % p) if (n & 1) ret = (LL)ret * a % p;

	return ret;
}

int main() {

	int n; scanf("%d", &n);
	long long x = 1, y = 1;
	for (int i = 1; i <= n; i++) {
		long long a; scanf("%lld", &a); x = x * (a % (MOD-1)) % (MOD-1);
	}
	for (int i = 1; i <= n; i++) {
		long long a; scanf("%lld", &a); y = y * (a % (MOD-1)) % (MOD-1);
	}

	int ans = (long long)modExp((MOD+1)/2, x+y-1, MOD) * (modExp(2, x, MOD)-1) % MOD;

	printf("%d\n", ans);
	return 0;
}
