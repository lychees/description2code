#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[n];
		int chck[n];
		for(int i=0;i<n;i++){
			cin>>arr[i];
			chck[i]=0;
		}
		int flag = 1;
		int index = 0;
		while(flag){
			int num = arr[index];
			if(num == -1){
				flag = 0;
			}
			if(chck[index] == 1) break;
			chck[index] = 1;
			index = num;
		}
		if(!flag) cout<<"Win\n";
		else cout<<"Lose\n";
	}
    return 0;
}
