#include <map>
#include <iostream>
using namespace std;

int main()
{
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		int x,count=0;
		int mp[10]={0};
		for(int i=0;i<n;i++){
			cin>>x;
			x=x%10;
			if(mp[x]!=0)
				count++;
			else 
				mp[x]=1;
		//	cout<<x<<" ";
		}cout<<count<<endl;
	}
	
    return 0;
}
