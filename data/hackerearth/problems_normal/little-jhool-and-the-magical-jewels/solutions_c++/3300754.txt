#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	set <int> list;
    	string s;
    	cin>>s;
    	list.insert(count(s.begin(),s.end(),'r'));
    	list.insert(count(s.begin(),s.end(),'u'));
    	list.insert(count(s.begin(),s.end(),'b'));
    	list.insert(count(s.begin(),s.end(),'y'));
    	cout<<*list.begin()<<endl;
    }
}
