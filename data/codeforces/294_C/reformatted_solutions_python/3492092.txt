M = 10**9+7


def R():
    return map(int, raw_input().split())


n, m = R()
a = sorted(R())


def P(k):
    return reduce(lambda x, y: x*y % M, [1]+range(1, k+1))


def D(a, b):
    return a*pow(b, M-2, M) % M


t = D(D(P(n-m), P(a[0]-1)), P(n-a[-1]))
for i in range(m-1):
    d = a[i+1]-a[i]-1
    if d:
        t = D(t*pow(2, d-1, M), P(d))
print t
