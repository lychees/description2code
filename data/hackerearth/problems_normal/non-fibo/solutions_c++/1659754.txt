#include <iostream>
using namespace std;
long long int a[1000005];

int main()
{
    int t ;
    cin >> t;
    
	a[0] = 2;
    a[1] = 3;
 	for(int i=2;i<1000005;i++) {
 		a[i] = a[i-1]+a[i-2];
 	}   
    
    while(t--) {
    	long long int n,i;
    	cin >> n;
    	
    	for(i=1;n>0;i++) {
    		n=n-(a[i-1]-1);
    		if(n<=0) {
    			n += a[i-1]-1;
    			cout<<a[i]+n<<endl;
    			break;
    		}
    	}
    	
    }
    
    return 0;
}
