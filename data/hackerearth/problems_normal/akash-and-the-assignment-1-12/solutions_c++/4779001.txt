#include <bits/stdc++.h>
using namespace std;
int H[50005][26];
int main()
{
	int N,Q;
	cin>>N>>Q;
	string S;
	cin>>S;
	for(int i=0;i<26;i++)
		H[0][i]=0;
	for(int i=0;i<S.size();i++)
	{
		int d=S[i]-'a';
		for(int j=0;j<26;j++)
			H[i+1][j]=H[i][j];
		H[i+1][d]=H[i][d]+1;
	}
	while(Q--)
	{
		int K,L,R;
		cin>>L>>R>>K;
		//answer
		if(K>(R-L+1))
			cout<<"Out of range"<<endl;
		else
		{
			int HL[26];
			for(int j=0;j<26;j++)
				HL[j]=H[R][j]-H[L-1][j];
			for(int j=0;j<26;j++)
			{
				K=K-HL[j];
				if(K<=0)
				{
					char ans=j+'a';
					cout<<ans<<endl;
					break;
				}
			}
		}
	}
	return 0;
}
