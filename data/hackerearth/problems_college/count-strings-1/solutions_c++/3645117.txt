#define MOD 1000000007
vector<int> v;
int cnta[27];
int cntb[27];
char A[MAX*10];
char B[MAX*10];
int main()
{
	int T;
	scanf("%d",&T);
	scanf("\n");

//	T = 1;
	while(T--)
	{
		gets(A);
		gets(B);
		memset(cnta,0,sizeof(cnta));
		memset(cntb,0,sizeof(cntb));
		
		int la,lb;
		la = strlen(A);
		lb = strlen(B);
//		printf("%s %s\n",A,B);
		for(int i = 0 ; i < la ; i++)
			cnta[A[i]-'a']++;
		for(int j = 0 ; j < lb ; j++)
			cntb[B[j]-'a']++;
		long long ans = 1;
		
		for(int i = 0 ; i < 26 ; i++)
		{
			if(cntb[i] > cnta[i])
			{
//				printf("setting when %c\n",i+'a');
				ans = 0;
				break;
			}
			else
			{
				while(cntb[i]!=0)
				{
//					printf("in %d\n",cntb[i]);
					ans *= cnta[i];
					cnta[i]--;
					ans %= MOD;
					cntb[i]--;
				}
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
