#include<bits/stdc++.h>
using namespace std;
int BIT[100000][26];
int n;
void update(int x,int val,int pos)
{
	int i;
	for(i=x;i<=n;i+= i & (-i))
	{
		BIT[i][pos]+=val;
	}
}
int query(int x,int pos)
{
	int sum=0;
	for(int i=x;i>0;i-=i & -i)
	{
		sum+=BIT[i][pos];
	}
	return sum;
}
	
int main()
{
	int i,q,count,tmp,r,l,q1;
	string st;
	cin>>n;
	cin >>q;
	cin>>st;
	for(i=0;i<n;i++)
	{
		update(i+1,1,st[i]-'a');
	}
	while(q--) {
	cin>> q1;
	char x;
	if(q1==1)
	{
		cin >> l >> x;
		update(l,1,x-'a');
		update(l,-1,st[l-1]-'a');
		st[l-1]=x;
	}
	if(q1==2)
	{
		cin >>l >>r;
		count=0;
		for(i=0;i<26;i++)
		{
			
			tmp=query(r,i)-query(l-1,i);
			if(tmp%2==1)
				count++;
		}
//		cout <<count<<endl;
		if(count<=1)
			cout <<"yes"<<endl;
		else 
			cout<<"no"<<endl;
	}
	}
	return 0;
}
