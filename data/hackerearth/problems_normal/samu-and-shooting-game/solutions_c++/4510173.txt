double solve(int i, int p){
	if (p >= w) return 1;
	if (i == n) return 0;
	if (dp[i][p] != -1) return dp[i][p];

	double a = (p1 / 100.0) * solve(i+1, p+x) + ((100 - p1) / 100.0) * solve(i+1, p);
	double b = (p2 / 100.0) * solve(i+1, p+y) + ((100 - p2) / 100.0) * solve(i+1, p);
	return dp[i][p] = max(a, b);
}
