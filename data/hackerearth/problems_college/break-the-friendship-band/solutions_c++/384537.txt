#include <stdio.h>
#include <iostream>;
#include <string>

int N;
int count(std::string& beads, int i)
{
	int leftIndex = i - 1;
	int rightIndex = i;
	if (leftIndex < 0) leftIndex += N;
	char right = beads[i];
	char left = beads[leftIndex];
	char curr = right;
	int total = 0;
	while ((curr == right) && (total < N))
	{
		total++;
		rightIndex = (rightIndex + 1) % N;
		curr = beads[rightIndex];
		if (right == 'w' && curr != 'w') right = curr;
		if (curr == 'w') curr = right;
		
	}

	curr = left;
	while ((curr == left) && (total < N))
	{
		total++;
		leftIndex = (leftIndex - 1);
		if (leftIndex < 0) leftIndex += N;
		curr = beads[leftIndex];

		if (left == 'w' && curr != 'w') left = curr;
		if (curr == 'w') curr = left;
	}

	return total;
}
int main()
{
	std::string beads;
	std::cin >> N;
	std::cin >> beads;

	int max = 0;
	for (int i = 0; i < beads.length(); i++)
	{
		int m = count(beads,i);
		if (m > max) max = m;
	}
	std::cout << max << std::endl;
}
