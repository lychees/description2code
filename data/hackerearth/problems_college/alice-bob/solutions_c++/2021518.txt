#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,flag=0;
    cin>>n;
    vector<long long int>v;
    v.resize(n);
    for(int i=0;i<n;i++)
    {
    	cin>>v[i];
    }
    int d=v[1]-v[0];
    if(n>2)
    {
    	for(int i=2;i<n;i++)
    	{
    		if(d!=v[i]-v[i-1])
    		{
    			flag=1;
    		}
    	}
    	if(flag==0)
    	{
    		cout<<"Bob"<<endl;
    	}
    	else
    	{
    		cout<<"Alice"<<endl;
    	}
    }
    else
    {
    	cout<<"Alice"<<endl;
    }
    
    
}
