#include <iostream>
#include <queue>
#include <utility>
using namespace std;
int main()
{
	int t;
	long long int X, M, sum;
	unsigned long long int UM;
	long int i, N, day;
	queue<pair<long, long long>> q;
	pair<long, long long> p;
	cin>>t;
	while(t--) {
		cin>>N;
		sum = 0;
		while(!q.empty()) {
			q.pop();
		}
		for(i=1; i<=N; i++) {
			cin>>X;
			if(X>0)	{
				sum += X;
				q.push(make_pair(i, X));
			}
		}
		cin>>UM;
		M = UM % sum;
		day = 0;
		if(M==0) {
			day = q.back().first;
		}
		while(M > 0) {
			p = q.front();
			q.pop();
			M -= p.second;
			day = p.first;
			q.push(p);
		}
		cout<<day<<endl;
	}
    return 0;
}
