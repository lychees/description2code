#define ss second

#define all(a) a.begin(),a.end()

pair<int, pi>p[102];
int c[55];
bool b[102];

int main(){
	int t; cin >> t;
	while (t--){
		int n, m;
		cin >> n >> m;		
		for (int i = 0; i < n; ++i){
			cin >> p[i].ss.ff >> p[i].ss.ss >> p[i].ff;
		}
		sort(p, p + n);
		reverse(p, p + n);
		for (int i = 0; i < m; ++i){
			cin >> c[i];
		}
		Fill(b, 0);
		for (int i = 0; i < m; ++i){
			int mx = c[i];
			for (int j = 0; j < n; ++j){
				if (i >= p[j].ss.ff && i < p[j].ss.ss){
					if (b[j])continue;
					if (mx > 0) --mx;
					else {
						b[j] = true;
					}
				}
			}
		}
		int ans = 0;
		for (int i = 0; i < n; ++i){
			if (b[i]) continue;
			ans += p[i].ff;
		}
		cout << ans; el;
	}
	return 0;
}
