#include <bits/stdc++.h>
using namespace std;
bool arr[100000010]={0};
vector<int> sieve()
{
	vector<int> primes;
	int n = 100000000;
	for(int i=2;i<=100000000;i++)
	{
		if(arr[i] == 0)
		{
			primes.push_back(i);
			for(int j=2;i*j<=100000000;j++)
				arr[i*j]=1;
		}
	}
	return primes;
	
}

int main()
{
	vector<int> ans = sieve();
    int test,x,y;
    cin >> test;
  	std::vector<int>::iterator low,up;
    while(test--)
    {
    	cin >> x >> y;
    	low = lower_bound(ans.begin(),ans.end(),x);
    	
    	up = lower_bound(ans.begin(),ans.end(),y);
    	int ans;
    	if(*up == y)
    	 ans= up-low+1;
    	 else
    	 ans = up-low;
    	cout << ans<<endl;
    }
    return 0;
}
