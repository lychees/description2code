#include <iostream>
#include <string>
using namespace std;

struct tree
{
	string b;
	int val;
	tree *left,*right;
};

int main()
{
    tree *start,*temp,*tem;
    string a;                             
    int t,n,k,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>k>>m;
        cin>>a;                           
        start=new tree;
        start->b=a;
        start->val=1;
        start->left = start->right = nullptr;
        n--;
        while(n--)
        {
            temp=start;
            cin>>a;                    
            while(1)
            {
                if(a.substr(0,m)<=temp->b.substr(0,m))
                {
                    temp->val++;
                    if(temp->left)
                    temp=temp->left;
                    else
                    {
                        tem=new tree;
                        tem->b=a;
                        tem->val=1;
                        tem->left = tem->right = nullptr;
                        temp->left=tem;
                        break;
                    }
                }
                else
                {
                    if(temp->right)
                    temp=temp->right;
                    else
                    {
                        tem=new tree;
                        tem->b=a;
                        tem->val=1;
                        tem->left = tem->right = nullptr;
                        temp->right=tem;
                        break;
                    }
                }
            }
        }
        temp=start;
            while(1)
            {
                if(temp->val==k)
                {
                    cout<<temp->b<<endl;
                    break;
                }
                else if(temp->val>k)
                {
                    // k--;
                    temp=temp->left;
                }
                else
                {
                    k=k-temp->val;
                    temp=temp->right;
                }
            }
    }
    return 0;
}
