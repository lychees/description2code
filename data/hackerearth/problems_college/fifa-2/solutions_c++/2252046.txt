template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

int k[100005], g[100005], h[100005], x[100005], y[100005];

int main()
{	
	int n;
	
	cin >> n;
	
	for (int i = 1; i <= n; i++) {
		cin >> x[i] >> y[i];
		k[x[i]]++;
	}
	
	for (int i = 1; i <= n; i++) {
		h[i] = n - 1 + k[y[i]];
		g[i] = n - 1 - k[y[i]];
	}
	
	for (int i = 1; i <= n; i++) {
		cout << h[i] << " " << g[i] << "\n";
	}
	
	return 0;	
}
