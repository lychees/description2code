#include <iostream>
#include<memory.h>
using namespace std;
int NN[100+5];
void init(){

memset(NN,0,sizeof(int)*105);
	for(int i=0;i<105/6;i++){

		for(int j=0;j<105/8;j++){
            if(NN[6*i+8*j]!=0)
			NN[6*i+j*8]=min(i+j,NN[6*i+8*j]);
			else
                NN[6*i+j*8]=i+j;
		}

	}



}
int main()
{
    int tt;
    cin>>tt;
    init();
    while(tt--){
    	int n;
    	cin>>n;
    	if(NN[n]!=0)
    	cout<<NN[n]<<endl;
    	else
    		puts("-1");

    }
    return 0;
}
