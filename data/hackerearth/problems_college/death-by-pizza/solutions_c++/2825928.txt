#include<bits/stdc++.h>
using namespace std;
int main()
{
    int N,Q;
    long long T[]={16,9,26,26,1};
    scanf("%d%d",&N,&Q);
    long long A[Q][3];
    long long X[N];
    for(int i=0;i<N;i++)
        scanf("%lld",&X[i]);
    for(int i=0;i<Q;i++)
    {
        for(int j=0;j<3;j++)
            scanf("%lld",&A[i][j]);
    }
    for(int i=0;i<Q;i++)
    {
        if(A[i][0]==1)
        {
            long long u=0;
            for(int j=A[i][1]-1;j<A[i][2];j++)
                u+=X[j];
            printf("%lld/%d\n",u,(A[i][2]-A[i][1]+1));
        }
        else if(A[i][0]==0)
        {
            int j=A[i][1]-1;
            int y=0;
            while(j<A[i][2])
            {
                X[j]+=T[y];
                y++;
                j++;
                y%=5;
            }
        }
    }
    return 0;
}
