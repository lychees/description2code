#include<bits/stdc++.h>
using namespace std;
#define ll int
int dp[1005],n,a[1005];
int solve(int pos)
{
    if(pos >= n)
       return 0;
    if(dp[pos]!=-1)
       return dp[pos];
    int ans=1;
    for(int i=pos + 1;i<n;i++){
        if(a[i]%a[pos]==0){
            ans = max(ans,solve(i) + 1);
        }
    }
    return dp[pos]=ans;
}
int main()
{
    scanf("%d",&n);
    memset(dp,-1,sizeof(dp));
    for(int i=0;i<n;i++)
       scanf("%d",&a[i]);
    sort(a,a+n);
    int k = 0;
    for(int i = 0;i < n;++i)
    	k = max(k,solve(i));
    if(k<=1)
    k=-1;
    cout<<k<<"\n";
    return 0;
}
