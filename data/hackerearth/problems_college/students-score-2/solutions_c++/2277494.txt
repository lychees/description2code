
#include <iostream>
#include <string>

using namespace std;

int main()
{
    long long sum,ma=-1000000000;

    int n;

    cin >> n;

    int a[n];

    for(int i=0;i<n;i++)
        cin >> a[i];

    for(int i=0;i<n;i++)
    {
        sum=0;

        for(int j=i;j<n;j++)
        {
            sum+=a[j];

            if(sum>ma)
                ma=sum;

        }

    }

    if(ma<0)
        ma=0;

    cout << ma;
}
