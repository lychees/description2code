#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	while(1)
    	{
    		if(n%10==n)
    		{
    			cout<<n<<endl;
    			break;
    		}
    		else
    		{
    			int n1=n;
    			int sum=0;
    			while(n1!=0)
    			{
    				sum=n1%10+sum;
    				n1=n1/10;
    			}
    			n=sum;
    		}
    	}
    }
    return 0;
}
