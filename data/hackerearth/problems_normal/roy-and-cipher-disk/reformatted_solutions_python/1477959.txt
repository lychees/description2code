tc = int(input())
while tc > 0:
    inp = raw_input()
    inp = list(inp)
    init = 0
    for i in range(0, len(inp)):
        p = ord(inp[i])
        val = p-97
        if val-init <= 13 and val-init >= -12:
            print val-init,
        else:
            if val-init > 0:
                print val-init-26,
            else:
                print 26+val-init,
        init = val
    print ""
    tc = tc-1
