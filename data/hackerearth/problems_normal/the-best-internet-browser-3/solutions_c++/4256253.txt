#include <iostream>
#include <string.h>
using namespace std;

bool isvowel(char a)
{
	if(a == 'a' || a == 'e' || a == 'i' || a == 'o' || a == 'u')
	return 1;
	else
	return 0;
}

int findratio(char *website, int nbnum)
{
//	int nbnum = 0;
	int jbnum = 0; 
//	nbnum = strlen(website);
	
	for(int i=4; i < nbnum - 4; i++)
	{
		if(!isvowel(website[i]))
		jbnum++;
	}
	
	jbnum += 4;  // for .com
	

	return jbnum;
	
	
}

int main()
{
	int n;
    cin >> n;
    char *website;
    website = new char(200 * sizeof(char));
    int nbnum = 0;
    while( n-- )
    {
    	
    	cin >> website;
    	nbnum = strlen(website);
    	cout << findratio(website,nbnum) << "/" << nbnum<<endl;
    }
    return 0;
}
