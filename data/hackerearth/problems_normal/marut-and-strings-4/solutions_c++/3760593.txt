#include <iostream>
#include <cctype>

using namespace std;

int main() {

	int T;
	cin >> T;
	if(T < 1 || T > 10) {
		cout << "Invalid Test" << endl;
	} else {

		while(T--) {

			string s;
			cin >> s;
			
			if(s.length() > 100) {
				cout << "Invalid Input" << endl;
				continue;
			}

			int downCase = 0, upCase = 0, alpha = 0;

			for(int i = 0; i < s.length(); ++i) {
				if(isalpha(s[i])) {
					alpha++;
					if(isupper(s[i])) upCase++;
					else downCase++;
				}	
			}

			if(alpha == 0) {
				cout << "Invalid Input" << endl;
			} else {
				if(upCase > downCase) cout << downCase << endl;
				else cout << upCase << endl;
			}

		}

	}

}
