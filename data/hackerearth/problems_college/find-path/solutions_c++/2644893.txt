lld bina(lld k,lld n){
    lld low=0,high=n-1,mid;
    while(high-low>1){
        mid=low+(high-low)/2;
        if(A[mid]<k) low=mid;
        else high=mid-1;
    }
    if(A[high]<k) return high;
    else return low;
}
