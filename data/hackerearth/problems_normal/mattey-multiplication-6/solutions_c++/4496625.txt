#include <iostream>
using namespace std;

int noBits(signed long long int no)
{
	int bits = 0;
	while(no)
	{
		no = no&(no-1);
		bits++;
	}
	
	return bits;
}
int main()
{
	signed long long int t,n,m,terms,temp;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		terms = noBits(m);

		for(int i=63;i>=0;i--)
			{
				temp = (signed long long int)1<<i;
				if(temp & m)
				{
					cout<<"("<<n<<"<<"<<i<<") ";
					if(--terms)
					cout<<"+ ";
					
				}
			}
		cout<<endl;
	}
    return 0;
}
