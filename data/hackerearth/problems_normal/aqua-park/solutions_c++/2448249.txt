	 pair<l,l>pp[105];
     pair<l,l>vv[105];

	l check(l val,l ht,l n){
	    l th=-1,ind=-1;
	    for(l p=0;p<n;p++){
            if(ht>=pp[p].first && pp[p].first>=th && ((val<=pp[p].second && val>=vv[p].second)||(val>=pp[p].second && val<=vv[p].second))){
                th=pp[p].first;
                ind=p;
            }
	    }
	   // cout<<ind<<"hell\n";
	    return ind;



	}

              vv[p].first=d;vv[p].second=c;
