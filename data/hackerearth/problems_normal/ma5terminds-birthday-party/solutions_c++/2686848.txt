int prime[1000005];

void sieve()
{
	
	for(int i=2;i<=1000000;i++)
	{
		if(prime[i]==0)
		{
			for(int j=2*i;j<=1000000;j=j+i)
			{
				prime[j]=1;
			}
		}
	}
	prime[1]=1;
	prime[4]=0;
	for(int i=2;i<=1000000;i++)
prime[i]+=prime[i-1];
}
int main()
{
	sieve();
int t,l,r;
scanf("%d",&t);
while(t--)
{
	scanf("%d%d",&l,&r);
