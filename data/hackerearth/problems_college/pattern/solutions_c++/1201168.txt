#include <iostream>
using namespace std;
int main()
{
    int x,n=2;
    
    cin >> x;
    
    for(int i=0;i<x;i++)
    {
     if(i == 0 || i == x-1)
     {
       for(int j=0;j<x;j++)
        cout << "*";
     }
     else
     {
       for(int i=0;i<x-n;i++)
        cout << " ";
        cout << "*";
        n++;
     }
     cout << endl;
    }
    
    return 0;
}
