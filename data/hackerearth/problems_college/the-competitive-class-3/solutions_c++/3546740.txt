#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n, i, j, ans;
	cin>>n;
	vector<int> marks(n);
	for(i = 0; i < n; i++)
	{
		cin>>marks[i];
	}
	for(i = 0; i < n; i++)
	{
		ans=0;
		for(j = 0; j < n; j++)
		{
			if(marks[j] > marks[i] )
				ans++;
		}
		cout<<ans+1<<" ";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
