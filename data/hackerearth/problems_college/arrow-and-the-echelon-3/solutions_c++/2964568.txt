#include <iostream>
using namespace std;

int main()
{  int i,t,j;
    cin>>t;
    while(t--)
    {int n,k,l;
    	cin>>n;
       if(n<3||n==4)
          cout<<"-1\n";
        else if(n==3)
           cout<<"555\n";
         else if(n==5)
            cout<<"33333\n";
       else
       {
       	for(i=0;i<n;i++)
       	  {
       	         k=n-i*5;
       	  		if(k>=0&&k%(3)==0)
       	  		    break;
       	  }
       	  if(i!=n)
       	  {for(j=0;j<k;j++)
       	  cout<<"5";
       	  for(j=0;j<n-k;j++)
       	  cout<<"3";
       	  cout<<"\n";
          }
          else
            cout<<"-1\n";
       }
}
    return 0;
}
