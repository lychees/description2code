LL fact[MAX+5];

void init()
{
    fact[0] = 1ll;
    for(int i = 1; i <= 100000; i++)
        fact[i] = (fact[i - 1] * i) % MD;
}



LL bigmod(LL p,LL e,LL M)
{
    if(e==0)return 1;
    if(e%2==0)
    {
        LL t=bigmod(p,e/2,M);
        return (t*t)%M;
    }
    return (bigmod(p,e-1,M)*p)%M;
}

LL modinverse(LL a,LL M)
{
    return bigmod(a,M-2,M);
}

LL ncr(int n, int r)
{
    return (fact[n] * modinverse( (fact[r] * fact[n - r] ) % MD, MD)) % MD;
}
