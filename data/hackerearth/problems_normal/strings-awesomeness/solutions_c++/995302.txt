#include <iostream>
#include <cstdio>
#include <string>
#include <cctype>

using namespace std;

int target[256];
int curr[256];
int ss[500000];

inline void add(char ch, int& score) {
	if (curr[ch] == target[ch]) score--;
	if (++curr[ch] == target[ch]) score++;
}

inline void del(char ch, int& score) {
	if (curr[ch] == target[ch]) score--;
	if (--curr[ch] == target[ch]) score++;
}

inline int go(int idx) {
	if (idx < 0) return 0;
	return ss[idx];
}

int main()
{
	target['a'] = 2;
	target['c'] = 1;
	target['e'] = 2;
	target['h'] = 2;
	target['k'] = 1;
	target['r'] = 2;
	target['t'] = 1;
	int score = 19;
	
	string s; cin >> s;
	for (int i = 0; i < s.size(); i++) if ('A' <= s[i] && s[i] <= 'Z') s[i] = tolower(s[i]);

	int limit = 11;
	if (s.size() < 11) limit = s.size();
	for (int i = 0; i < limit; i++) add(s[i], score);
	if (score == 26) ss[0] = 1;
	
	for (int a = 1, b = 12; b <= s.size(); a++, b++) {
		ss[a] = ss[a-1];
		del(s[a-1], score);
		add(s[b-1], score);
		if (score == 26) ss[a]++;
	}
	
	int q; cin >> q;
	while (q--) {
		int l, r; scanf("%d%d", &l, &r);
		l--;
		r -= 11;
		if (r < l)
			puts("0");
		else
			printf("%d\n", (go(r)-go(l-1)));
	}
	
    return 0;
}
