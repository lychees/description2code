#!/usr/bin/env python
# -*- coding: utf-8 -*-
import sys


def main():
    testcases = int(sys.stdin.next().rstrip())
    for testcase_num in range(testcases):
        n, m = [int(x) for x in sys.stdin.next().rstrip().split()]
        matrix = []
        for row in range(n):
            matrix.append(list(
                [bool(int(x)) for x in sys.stdin.next().rstrip()]))

        flips = 0
        carry_up = m * [False]
        for row in range(n-1, -1, -1):
            carry_left = False
            for col in range(m-1, -1, -1):
                flip = not(matrix[row][col]) ^ carry_left ^ carry_up[col]
                carry_left = carry_left ^ carry_up[col]
                if flip:
                    flips += 1
                    carry_up[col] = not carry_up[col]
                    carry_left = not carry_left
        print('{}'.format(flips))

if __name__ == '__main__':
    main()
