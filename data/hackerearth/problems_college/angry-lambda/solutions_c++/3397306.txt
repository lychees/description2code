#include <iostream>

int main(int argc, char **argv) {
	int n;
	std::cin >> n;
	while (n--) {
		int N, K;
		std::cin >> N >> K;
		int cnt = 0;
		for(int i = 0; i < N; ++i) {
			int tmp;
			std::cin >> tmp;
			if (tmp <= 0) ++cnt;
		}
		if (cnt >= K)
			std::cout << "NO" << std::endl;
		else
			std::cout << "YES" << std::endl;
	}
	return 0;
}
