

    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int test;
    	cin>>test;
    	while(test--)
    	{
    		string s;
    		cin>>s;
    		//string contains only lowercase charatcers
    		int hash[26]={0};
    		for(int i=0;i<s.size();i++)
    			hash[s[i]-'a']++;
    		int odd=0;
    		for(int i=0;i<26;i++)
    			if(hash[i]&1)
    				odd++;
    		if(odd>1)
    			cout<<"No"<<endl;
    		else
    			cout<<"Yes"<<endl;
    	}
    	return 0;
    }
