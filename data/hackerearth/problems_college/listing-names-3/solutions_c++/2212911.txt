#include<iostream>
#include<map>
#include<algorithm>
#include<string>
using namespace std;
map<string,int>names;
map<string,int>::iterator it;
int main(){
	string str;
	long int testcases;
	cin>>testcases;
	while(testcases--){
		cin>>str;
		names[str]++;
	}
	for(it=names.begin();it!=names.end();it++) cout<<it->first<<" "<<it->second<<endl;
}
