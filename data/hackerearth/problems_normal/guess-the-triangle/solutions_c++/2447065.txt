# include <bits/stdc++.h>
# define ll long long
using namespace std;

struct side
{
	float x1;
	float x2;
};

bool comp(side a, side b)
{
	if(a.x1 != b.x1)
		return a.x1 < b.x1;
	return a.x2 < b.x2;
}
int main()
{
	std::ios::sync_with_stdio(false);
	float x1,x2,x3,y1,y2,y3;

	cin>>x1>>y1>>x2>>y2>>x3>>y3;
	side s[3];

	s[0].x1 = x1 - x2 + x3;
	s[1].x1 = x2 - x3 + x1;
	s[2].x1 = x3 - x1 + x2;


	s[0].x2 = y1 - y2 + y3;
	s[1].x2 = y2 - y3 + y1;
	s[2].x2 = y3 - y1 + y2;

	sort(s, s+3, comp);

	for(int i=0; i<3; i++)
		printf("%.4f %.4f\n", s[i].x1, s[i].x2);
}
/*

	s = 2(x1 + x2 + x3)
	2x1 = a1 + a2
	2x2  = a2 + a3

	a1 - a3 = 2(x1 - x2)
	a1 + a3 = 2(x3)

	2a1 = 2x1 - 2x2 + 2x3
