#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <set>

using namespace std;

void solve() {
	int N, a, b;
	scanf("%d", &N);
	int mx = 0;
	set<int> st;
	int count = 0;
	for (int i = 0; i < N; ++i) {
		scanf("%d %d", &a, &b);
		if (b == 0) {
			count = 0;
		} else {
			if (st.find(a) == st.end()) {
				count++;
				st.insert(a);
			}
			mx = max(mx, count);
		}
	}
	printf("%d\n", mx);
}

int main() {
	int T;
	scanf("%d", &T);
	while(T--) {
		solve();
	}
	return 0;
}
