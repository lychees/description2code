from fractions import gcd
for _ in xrange(input()):
    a, b, c = map(int, raw_input().split())
    d = gcd(a, b)
    if gcd(d, c) != 1:
        print "NO"
    else:
        a = a**2
        b = b**2
        c = c**2
        if a+b == c or a+c == b or b+c == a:
            print "YES"
        else:
            print "NO"
