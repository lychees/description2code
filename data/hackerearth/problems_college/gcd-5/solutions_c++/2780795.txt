#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int i,n;
    	cin>>n;
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=1;i<n;i++)
    	{
    		if(a[i]<a[i-1])
    		 break;
    	}
    	if(i==n)
    	cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
