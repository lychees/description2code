#include<iostream>
#include<string.h>
using namespace std;
int a[5001][5001];
int max(int a,int b){
	return a>b?a:b;
}
int main(){
	string s1,s2;
	int p,q,i,j,m,n;
	cin>>s1>>s2;
	p=s1.length();
	q=s2.length();
	m=p-1;n=q-1;
	
	for(i=0;i<p+1;i++){
		a[i][0]=0;
		
	}
	for(j=0;j<q+1;j++){
		a[0][j]=0;
	}
	for(i=1;i<p+1;i++){
		for(j=1;j<q+1;j++){
			if(s1[i-1]!='a'&&s1[i-1]!='e'&&s1[i-1]!='i'&&s1[i-1]!='o'&&s1[i-1]!='u'&&s1[i-1]==s2[j-1]){
				a[i][j]=1+a[i-1][j-1];
			}
			else{
				a[i][j]=max(a[i-1][j],a[i][j-1]);
			}
		}
	}
	cout<<a[p][q]<<"\n";
	
}
