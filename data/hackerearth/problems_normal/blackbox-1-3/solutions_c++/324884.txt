#include <iostream>
#include<algorithm>
using namespace std;
int hcf(int a,int b)
{
	if(a==0)return b;
else return hcf(b%a,a);
}
int main()
{
    int a[3],t;
    cin>>t;
    while(t--)
    {
    cin>>a[0]>>a[1]>>a[2];
    sort(a,a+3);
    if((a[0]*a[0]+a[1]*a[1]==a[2]*a[2])&&hcf(hcf(a[0],a[1]),a[2])==1)cout<<"YES"<<endl;
    else cout<<"NO"<<endl;
    }
    return 0;
}
