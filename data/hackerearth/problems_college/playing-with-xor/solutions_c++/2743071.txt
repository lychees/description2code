#include<bits/stdc++.h>
using namespace std;

bool dp[2000][20000];
vector<long long> v;

void sieve(long long n)
{
	bool prime[10001] = {false};
	for(int i=2;i<=sqrt(n);i++)
		if(!prime[i])
			for(int j=i*i;j<=n;j+=i)
				prime[j] = true;
	for(int i=2;i<=n;i++)
		if(!prime[i])
			v.push_back(i);
}

int main()
{
	sieve(10000);
	string str;
	long long t,n,ans;
	cin>>t;
	while(t--)
	{
		memset(dp,false,sizeof(dp));
		ans = 0;
		cin>>n;
		long long a[n+1];
		for(int i=1;i<=n;i++)
			cin>>a[i];
		dp[0][0] = true;
		for(int j=1;j<=n;j++)
			for(int i=0;i<10000;i++)
				dp[j][i] = dp[j-1][i^a[j]]|dp[j-1][i];
		
		//for(int i=2;i<=7;i++)
		//	cout<<i<<" "<<dp[n][i]<<"\n";
		
		for(int i=0;i<v.size();i++)
			if(dp[n][v[i]])
				ans++;
		cout<<ans<<"\n";
	}
	return 0;
}
