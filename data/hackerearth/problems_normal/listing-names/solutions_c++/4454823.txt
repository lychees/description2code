#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int t,i;
    cin>>t;
    string *a = new string[t];
    for(i=0;i<t;i++){
    	cin >> a[i];
    }
    sort(a,a+t,less<string>());
    for(i=0;i<t;i++){
    	cout<<a[i];
    	int k=1;
    	while(i<t-1 && a[i]==a[i+1]){
    		++i;
    		++k;
    	}
    	cout<<" "<<k<<"\n";
    }
    return 0;
}
