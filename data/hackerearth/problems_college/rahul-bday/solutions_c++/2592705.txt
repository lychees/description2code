#include <iostream>
using namespace std;
#define ll long long
int main()
{
    ll T,B,W,X,Y,Z;
    cin>>T;
    for(int i=1;i<=T;i++)
    {
    	cin>>B>>W;
    	cin>>X>>Y>>Z;
    	ll ANS=B*X;
    	ll ANS2=(B*Y+B*Z);
    	ANS=min(ANS,ANS2);
    	ll ANS3=W*Y;
    	ANS2=(W*X+W*Z);
    	ANS=ANS+min(ANS2,ANS3);
    	cout<<ANS<<endl;
    }
    return 0;
}
