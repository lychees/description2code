#include <iostream>
#include <stdio.h>
#include <set>
#include <algorithm>

using namespace std;

int main()
{
	int t, n, w1;
	
	scanf("%d", &t);
	
	while(t--)
	{
		scanf("%d", &n);
		
		int w[n-1];
		int h[n];
		
		scanf("%d", &w1);
		for(int i = 0; i<n-1; i++)
			scanf("%d", &w[i]);
			
		for(int i = 0; i<n; i++)
			scanf("%d", &h[i]);
		
		
		sort(w, w+n-1);
		sort(h, h+n);
		 
		int first = w1*h[n-1];
		bool ans = true;

		n--;
		for(int i = 0, j = n-1; i<n && j>=0; i++, j--)
		{	
			int s = w[i]*h[j];
			 
			if(s >= first)
				ans = false;
		}
		
		if(ans)
			printf("YES\n");
		else
			printf("NO\n");
	}
    
    return 0;
}
