#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--){
    	int n,maxi=0; cin>>n;
    	int arr[2*n];
    	for(int i=0;i<2*n;i++) scanf("%d",&arr[i]);
    	sort(arr,arr+2*n);
    	//cout<<arr[0]<<endl;
    	for(int i=0;i<n;i++){
    		if(arr[i]+arr[2*n-1-i]>maxi) maxi=arr[i]+arr[2*n-i-1];
    		//cout<<maxi<<endl;
    	}
    	cout<<maxi<<endl;
    }
    return 0;
}
