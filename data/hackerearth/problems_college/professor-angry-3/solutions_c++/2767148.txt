#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,k,a[10000],i,count=0;
    	scanf("%d %d",&n,&k);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    		if(a[i]<=0)
    		count++;
    	}
    	if(count>=k)
    	printf("NO\n");
    	else
    	printf("YES\n");
    }
    return 0;
}
