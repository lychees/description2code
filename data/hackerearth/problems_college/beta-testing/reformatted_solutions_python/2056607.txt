from __future__ import print_function


def decreasing_order(n, names):
    adict = {}
    for x in xrange(n):
        adict[names[x]] = 0

    for x in xrange(n):
        adict[names[x]] = adict[names[x]] + 1

    alist = []
    length = 0
    for key, value in adict.iteritems():
        alist.append([value, key])
        length += 1
    alist.sort(reverse=True)

    for x in xrange(length):
        alist[x][1], alist[x][0] = alist[x][0], alist[x][1]

    i = 0
    j = 1
    count = 0
    while count != length-1:
        if alist[count][0] == alist[count+1][0]:
            j += 1
            count += 1
        else:
            count += 1
            alist[i:j].sort(reverse=True)
            i = count
            j = count + 1

    for x in xrange(length):
        print(alist[x][0], end=' ')
        print(alist[x][1])


if __name__ == '__main__':
    n = input()
    names = raw_input().strip().split()
    decreasing_order(n, names)
