int MOD = 1000000007;

ll gcd(ll a, ll b) {
	if (b == 0) return abs(a);
	else return gcd(b, a % b);
}

int main() {
	vector<ll> a(101);
	int total;
	for (int A = 1; A <= 100; A++)
		for (int B = A + 1; B <= 100; B++)
			for (int C = B + 1; C <= 100; C++) {
				int d = gcd(A, gcd(B, C));
				a[d]++;
				total++;
			}
	int T; cin >> T;
	while (T--) {
		ll X, Z; scanf("%lld%lld", &X, &Z);
		int cnt = 0;
		for (int i = 1; i <= 100; i++)
			if (abs(Z - X) % i == 0) cnt += a[i];
		printf("%.6f\n", (double)cnt / total);
	}
}
