vpi edge;
int x[100001];
int used[100001];
int find(int a)
{
    if(x[a]==a)
        return a;
    return x[a] = find(x[a]);
}
int unio(int a,int b)
{
    int p1 = find(a);
    int p2 = find(b);
    if(p1==p2)
        return 1;
    x[p1] = p2;
    return 0;
}
int main()
{
    fast_io;
    int n,m,i,a,b;
    edge.pb(mp(0,0));
    cin>>n>>m;
    rep(i,1,n+1)
        x[i] = i;
    rep(i,1,m+1)
        cin>>a>>b,edge.pb(mp(a,b));
    int u=0;
    for(int c=m;c>0;c--)
    {
        if(unio(edge[c].ff,edge[c].ss))
            continue;
        used[c] = 1;   
        u++;
    }
    cout<<m-u<<endl;
    for(i=1;i<=m;i++)
        if(!used[i])
            cout<<i<<endl;
    return 0;
}
