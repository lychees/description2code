#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b){
 if (b==0)
    return a;
  else return  gcd(b,a%b);
 }
 int min(int a,int b){
 if (a>b) return b;
 else return a;}
int main(){
    int t;
    cin>>t;
    map<string,int> table;
    table["Chandu"]=0;
    table["Daspal"]=1;
    while(t--){
        int x,y,c;
        string s;
        cin>>x>>y;
        cin>>s;
        int d=table[s];
        while(x!=0 && y!=0){
        c=gcd(x,y);
        x-=c;
        y-=c;
        d=!d;}
        if (d==0) cout<<"Daspal"<<endl;
        else cout<<"Chandu"<<endl;
    }
    return 0;
}
