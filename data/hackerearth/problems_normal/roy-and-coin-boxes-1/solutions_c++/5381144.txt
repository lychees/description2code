#include <bits/stdc++.h>
using namespace std;
#define ll long long
int has[1000001],ans[1000001];

int main()
{
	int n;
	scanf("%d",&n);
	int a[n+2];
	memset(a,0,sizeof(a));
	int m;
	scanf("%d",&m);
	while(m--)
	{
		int l,r;
		scanf("%d%d",&l,&r);
		a[l-1]++;
		a[r]--;
	}
	has[a[0]]++;
	//cout<<a[0]<<endl;
	for(int i=1;i<n;i++)
	{
		a[i] = a[i-1]+a[i];
		has[a[i]]++;
	}
	ans[0] = has[0];
	for(int i=1;i<n;i++)
	{
		ans[i] = ans[i-1]+has[i];
		//cout<<has[i]<<" "<<ans[i]<<endl;
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int x;
		scanf("%d",&x);
		int answer = n-ans[x-1];
		printf("%d\n",answer);
	}
	return 0;
}
