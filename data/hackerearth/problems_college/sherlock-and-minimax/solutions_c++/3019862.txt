#include <bits/stdc++.h>
using namespace std;
int n;
int arr[109];
int a,b;
int res;

int check(int x)
{
	int ret=INT_MAX;
	for(int i=0;i<n;++i)
	ret=min(ret,abs(x-arr[i]));
	return ret;
	
}

void limits(int x)
{
	if(x<a||x>b) return;
	if(check(x)>check(res))res=x;
}


int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;++i)
	scanf("%d",&arr[i]);
	scanf("%d%d",&a,&b);
	res=a;
	limits(a);
	limits(b);
	for(int i=0;i<n;i++)
	for(int j=i+1;j<n;j++)
	{	limits((arr[i]+arr[j])/2);
		
	}
	printf("%d",res);
	return 0;
}
