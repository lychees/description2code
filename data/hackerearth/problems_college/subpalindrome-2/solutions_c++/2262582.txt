#include <bits/stdc++.h>
using namespace std;
string s;
int dp[1003][1003];
int rec(int i,int j)
{
	if(i>j)
	return 0;
	if(dp[i][j]!=-1)
	return dp[i][j];
	if(i==j)
	return dp[i][j]=1;
	return dp[i][j]=rec(i+1,j)+rec(i,j-1)-rec(i+1,j-1)+(s[i]==s[j])*(rec(i+1,j-1)+1);
}
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
      cin>>s;
      memset(dp,-1,sizeof(dp));
      cout<<rec(0,s.length()-1)<<endl;
    }
  
    return 0;
}
