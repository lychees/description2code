#include <iostream>
using namespace std;

void doCompute(long long int x){
	int c = 0;
	
	while(x>0){
		c+=x&1;
		x=x>>1;
	}
	
	if(c%2==0){
		cout<<"even"<<endl;
	}else{
		cout<<"odd"<<endl;
	}
}

int main()
{
    int t;
    long long int n;
    cin>>t;
    while(t--){
    	cin>>n;
    	doCompute(n);
    }
    return 0;
}
