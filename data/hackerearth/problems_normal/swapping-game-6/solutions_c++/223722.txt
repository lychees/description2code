#include <bits/stdc++.h>

using namespace std;

string func(string in)
{
    string out=in;
    int pos=0;
    int i;
    for(i=0;i<in.length();i+=2)
    {
        out[pos]=in[i];
        pos++;
    }

    if(i==in.length()+1)
    {
        i=in.length()-2;
    }
    else
    {
        i=in.length()-1;
    }

    for(i;i>=0;i-=2)
    {
        out[pos]=in[i];
        pos++;
    }
    //cout<<out;
    return out;

}

int preccomp(string in)
{
    int k=0;
    string tmp=in;
    while(true)
    {
        k++;
        tmp=func(tmp);
        if(tmp==in)
        {
            break;
        }
    }
    return k;
}

int main()
{
    //cout<<preccomp("fiugweuigfewiafwaefveajsvf");
    //func("ctosnet");
    int k;
    string str;
    cin>>k;
    cin>>str;
    k=k%(preccomp(str));

//    cout<<precomp("fiugweuigfewiafwaefveajsvf");

    for(int i=1;i<=k;i++)
    {
        str=func(str);
        //cout<<i<<" "<<str<<endl;
    }
    cout<<str;
    return 0;
}
