#include<stdio.h>
#define mod 1000000007
# define MAX 2000 // assuming we need first 100 rows
long long triangle[MAX + 1][MAX + 1];
 
void makeTriangle()
{
    int i, j;
 
    // initialize the first row
    triangle[0][0] = 1; // C(0, 0) = 1
 
    for(i = 1; i < MAX; i++) 
    {
        triangle[i][0] = 1; // C(i, 0) = 1
        for(j = 1; j <= i; j++) 
        {
            triangle[i][j] = (triangle[i - 1][j - 1]%mod + triangle[i - 1][j]%mod)%mod ;
        }
    }
}
 
long long C(int n, int r)
{
    return triangle[n][r];
}
int main()
{
	int t;
	scanf("%d",&t);
	makeTriangle();
	while(t--)
	{
		int n,i,p;
		scanf("%d%d",&n,&p);
		long long int sum=1;
		for(i=1;i<=p;i++)
		 { sum=(sum%mod+C(n,i)%mod)%mod;
		   
		 }
	    printf("%lld\n",sum);	
	}
	return 0;
}
