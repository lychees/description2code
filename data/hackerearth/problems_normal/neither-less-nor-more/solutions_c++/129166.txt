using namespace std;
typedef long long ll;
typedef unsigned long long ull;

bool a[1000100];
ull cnt,t,cnt1[1000],cnt2[1000],primecnt,left1,right1;
ull prime[100000],primes[10000];
ll k,kfake;
ll ans=-1;

void era(){
for(int i=2;i<=100;i++)
  if(a[i]==0)
  for(int j=i*i;j<=10001;j+=i)
      a[j]=1;
for(int i=2;i<=10001;i++)
  if(a[i]==0)
      prime[++cnt]=i;
return ;
}

ll maxim(ll qw,ll we){
if(qw>=we)
    return qw;
else
    return we;
}

ll minim(ll qw,ll we){
if(qw<=we)
    return qw;
else
    return we;
}


void bin(ll l,ll r){

if(l>r){
    //cout<<"mehemmedv"<<endl;
    return;
    }

ll n=(l+r)/2;
ll say;
ll oo;
ll maxi;
//cout<<l<<" "<<r<<" xa"<<endl;


      for(int i=1;i<=primecnt;i++){
          say=0;
          oo=primes[i];
            cnt2[i]=cnt1[i];
          while(oo<=n){
              say+=n/oo;
              //cout<<say<<" ma "<<i<<endl;
              if(n/oo<primes[i])
                  break;
              oo=oo*primes[i];
          }

          cnt2[i]=say/cnt2[i];
          //cout<<primes[i]<<" "<<say<<" "<<cnt1[i]<<" xa"<<endl;
      }

      maxi=cnt2[1];
      for(int i=1;i<=primecnt;i++)
          maxi=minim(maxi,cnt2[i]);
      //printf("%lld\n",maxi);


        //cout<<maxi<<" "<<left1<<" "<<right1<<endl;
        //cout<<maxi<<" "<<n<<endl;

        if(maxi>=left1 && maxi<=right1){
            ans=maxim(ans,n);
        //cout<<"xa"<<endl;
        }
        //cout<<ans<<" mehemm"<<endl;
        if(maxi>right1)
            bin(l,n-1);
        else
            bin(n+1,r);

}




int main()
{
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);


  era();
  scanf("%d",&t);
  while(t--){
      ans=-1;
      scanf("%lld %lld %lld",&k,&left1,&right1);
      primecnt=0;
      kfake=k;
      for(int i=1;i<=cnt && prime[i]*prime[i]<=kfake;i++)
          if(kfake % prime[i] == 0){
              primes[++primecnt]=prime[i];
              cnt1[primecnt]=0;
              while(kfake % prime[i] == 0)
                  kfake/=prime[i],cnt1[primecnt]++;
              //cout<<prime[i]<<" "<<cnt1[primecnt]<<endl;
          }
      if(kfake>1){
          primes[++primecnt]=kfake,cnt1[primecnt]=1;
          //cout<<kfake<<" "<<1<<endl;
      }

    ll sag=k*(right1+1);

      bin(1,sag);

        cout<<ans<<endl;

  }

  return 0;
}
