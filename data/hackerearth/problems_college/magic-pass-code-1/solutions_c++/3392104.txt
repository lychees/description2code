#include<bits/stdc++.h>
using namespace std;
int sumdig(int num)
{
	int sum=0;
	while(num>0)
	{
		sum+=num%10;
		num/=10;
//		puts("aasdf");
	}
	return sum;
}
int main()
{
	int n;
	int maxi=-0x7fffffff;
	int maxnum=0;

	while(cin>>n)
	{
	//	printf("n is %d\n",n);
		if(sumdig(n)>maxi)
		{
			maxi=sumdig(n);
			maxnum=n;
		}
	}
	cout<<maxnum<<endl;
	return 0;
}
