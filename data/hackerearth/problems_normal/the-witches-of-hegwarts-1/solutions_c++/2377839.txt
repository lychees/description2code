#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#include <queue>
#include <utility>
#include<unordered_set>
#define ll long long int 
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	queue<pair<ll,ll>> q;
    	unordered_set<ll> s;
    	q.push(make_pair(n,0));
    	s.insert(n);
    	while(!q.empty())
    	{
    		ll num=q.front().first;
    		ll count=q.front().second;
    		q.pop();
    		if(num==1)
    		{
    			cout<<count<<"\n";
    			while(!q.empty())
    				q.pop();
    			break;
    		}
    		ll a;
    		a=num/3;
    		if(num%3==0 && s.find(a)==s.end() && a>0)
    		{
    			q.push(make_pair(a,count+1));
    			s.insert(a);
    		}
    		a=num/2;
    		if(num%2==0 && s.find(a)==s.end() && a>0)
    		{
    			q.push(make_pair(a,count+1));
    			s.insert(a);
    		}
    		a=num-1;
    		if(a>0 && s.find(a)==s.end())
    		{
    			q.push(make_pair(a,count+1));
    			s.insert(a);
    		}
    	}
    }
    return 0;
}
