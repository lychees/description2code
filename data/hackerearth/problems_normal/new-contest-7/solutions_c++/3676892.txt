

    #include <cstdio>
    #include <algorithm>
    using namespace std;
    #define MAX 1001
    double p[MAX][MAX];
    int S, N, M, K;
    void print(int l){
        for(int i=0;i<=l;i++){
    		for(int j=0;j<=l;j++)
    			printf("%lf ", p[i][j]);
    		printf("\n");
    	}
    }
    void solve(){
    	for(int i=0;i<MAX;i++)
    		for(int j=0;j<MAX;j++)
    			p[i][j] = 0.0;
    	p[0][0] = 1.0;
    	for(int i=1;i<=N;i++){
    		if(S-M-i+1>0 and S-i+1>0){
    			p[0][i] = p[0][i-1]*((double)(S-M-i+1)/(double)(S-i+1));
    			if(p[0][i]<0.0)
    				p[0][i] = 0.0;
    		}else
    			p[0][i] = 0.0;
    	}
    	for(int i=1;i<=M;i++){
    		for(int j=1;j<=N;j++){
    			//if(M-i+1>0 and S-j+1>0 and S-M-j+i>0 and S-j+1>0){
    				p[i][j] = p[i-1][j-1]*((double)(M-i+1)/(double)(S-j+1)) + p[i][j-1]*((double)(S-M-j+1+i)/(double)(S-j+1));
    				if(p[i][j]<0.0)
    					p[i][j] = 0.0;
    			//}else
    			//	p[i][j] = 0.0;
    		}
    	}
    	//print(N);
    	double ans = 0.0;
    	for(int i=K;i<=M;i++)
    	//for(int i=0;i<K;i++)
    		ans = ans + p[i][N];
    	//printf("%lf\n", 1.0-ans);
    	printf("%lf\n", ans);
    }
    int main(){
    	int T;
    	scanf("%d", &T);
    	while(T--){
    		scanf("%d %d %d %d", &S, &N, &M, &K);
    		solve();
    	}
    }
