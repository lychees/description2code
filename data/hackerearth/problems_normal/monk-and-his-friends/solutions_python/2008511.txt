tc = int(raw_input())
assert(tc>0 and tc<11)
while tc>0:
	tc = tc - 1
	n, m = map(int, raw_input().split())
	assert(n>0 and n<100001)
	assert(m>0 and m<100001)
	k = map(int, raw_input().split())
	for i in k:
		assert(i>=0 and i<=10**12)
	l = []
	for i in xrange(n):
		l.append(k[i])
	l = set(l)
	for i in xrange(n,n+m):
		a = k[i]
		if a in l:
			print "YES"
		else:
			print "NO"
			l.add(k[i])
