#include <cstdio>
#include <algorithm>

using namespace std;

int A[8], n, ans = 0, e1, e2;

void solve(int m) {
    if (m == n) {
        ans++;
        return;
    }

    if (e1 + A[m] <= e2) {
        e1 += A[m];
        solve(m+1);
        e1 -= A[m];
    }

    e2 += A[m];
    solve(m+1);
    e2 -= A[m];
}

int main () {
    int i, sum = 0;

    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        scanf("%d", A+i);
        sum += A[i];
    }

    sort(A, A+n);

    do {
        e1 = e2 = 0;
        solve(0);
    } while(next_permutation(A, A+n));

    printf("%d %d\n", ans, sum);

    if (ans < sum)
        printf("Got no way out!\n");
    else
        printf("We will win!\n");

    return 0;
}
