#include<iostream>
using namespace std;
int gcd(long long int m, long long int n)
{
    long long int r;
    if((m == 0) || (n == 0))
        return 0;
    else if((m < 0) || (n < 0))
        return -1;

    do
    {
        r = m % n;
        if(r == 0)
            break;
        m = n;
        n = r;
    }
    while(true);
    return n;
}


int main()
{
    int t;
    long long int a,b,n;
    cin>>t;
    while(t--)
    {
        int flag = 0;
        long long lcm,f,d,c1,c2,e,q,w;
        cin>>a>>b>>n;
        lcm = (a*b)/gcd(a,b);
        f = lcm/a + lcm/b - 1;
        d = n/f;
        e = n%f;
        c1 = (d*lcm)/a + 1;
        c2 = (d*lcm)/b + 1;
        if(e==0)
            cout<<d*lcm<<"\n";
        else
        {
            while(e--)
            {
                q = c1*a;
                w = c2*b;
                if(q<w)
                {
                    c1++;
                    flag = 1;
                }
                else
                {
                    c2++;
                    flag = 0;
                }
            }
            if(flag==0)
                cout<<w<<"\n";
            else
                cout<<q<<"\n";


        }

    }
    return 0;
}
