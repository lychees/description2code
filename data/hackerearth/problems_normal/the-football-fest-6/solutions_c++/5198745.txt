#include <iostream>
using namespace std;
typedef struct Stackll{
	long data;
	struct Stackll *next;
}stack;
stack *create(){
	return NULL;
}
bool isEmpty(stack *s){
	return s==NULL;
}
void push(stack **s,long data){
	stack *temp = new stack();
	temp->data = data;
	if(isEmpty(*s)){
		temp->next = NULL;
	}
	else
		temp->next = *s;
		*s = temp;
}
long pop(stack **s){
	stack *t = *s;
	long temp = (*s)->data;
	*s = (*s)->next;
	delete t;
	return temp;
}
long top(stack *s){
	return s->data;
}

int main()
{
    stack *s  =create();
    long n,ID,temp,player;
    int t;
    char pass;
    cin>>t;
    while(t--){
    	cin>>n>>ID;
    	push(&s,ID);
    
    	while(n--){
    		
    		cin>>pass;
    		if(pass == 'P'){
    			cin>>ID;
    			push(&s,ID);
    		
    		}
    		else if(pass = 'B'){
    			
    			temp =pop(&s);
    			player = pop(&s);
    			push(&s,temp);
    			push(&s,player);
    			
    		}
    	}
    	cout<<"Player "<<top(s)<<"\n";
    }
    
    return 0;
}
