#include <iostream>
using namespace std;
int main()
{
	float t;
	cin>>t;
	while(t--)
	{
		float d,p;
		cin>>d>>p;
		if(d<0)
		cout<<"0"<<endl;
		else if(d==0 && d*d+4*p==0)
		cout<<"1"<<endl;
		else if(d>0 && d*d+4*p>0 )
		cout<<"4"<<endl;
		else
		cout<<"2"<<endl;
		}
	return 0;
}
