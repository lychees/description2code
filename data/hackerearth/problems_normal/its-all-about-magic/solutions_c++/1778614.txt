#include <iostream>
#include <cstdio>
using namespace std;;
int main(){

	long long int zero=0, one = 1, temp;
	int testCase, M;
	scanf("%d",&testCase);
	for(int i=0; i<testCase; i++){
		zero = 0;
		one=1;
		scanf("%d",&M);
		for(int j=1; j<=M; j++){
			temp = one;
			one = zero;
			zero += temp;
		}
		printf("%lld %lld\n",one, zero);			
		}
	return 0;
}
