#include <iostream>
using namespace std;
int main()
{
   
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int x,p,cnt=0,f=0;
    	scanf("%lld%lld",&x,&p);
    	
    	while(x%p == 0)
    	{
    		f =1;
    		cnt++;
    		x = x/p;
    	}
    	if(f == 1)
    	{
    		printf("%lld ",cnt);
    		if(x == 1)puts("0");
    		else printf("%lld\n",x);
    	}
    	else{
    		printf("%lld %lld\n",cnt,x);
    	}
    	
    }
    return 0;
}
