#include<cstdio>
#include<set>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,x;
		scanf("%d",&n);
		set<int> s;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			s.insert(x);
		}
		int cnt=0;
		set<int> :: iterator ip;
		for(ip=s.begin();ip!=s.end();ip++)
			++cnt;
		printf("%d\n",cnt);
	}
	return 0;
}
