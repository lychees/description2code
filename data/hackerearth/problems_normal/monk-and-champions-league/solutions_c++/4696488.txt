#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
	int i,j,n,m,a,b,c;
	priority_queue<int > q;
	cin>>m>>n;
	for(i=0;i<m;i++)
	{
		cin>>j;
		q.push(j);
	}
	c=0;
	while(true)
	{
		a=q.top();
		q.pop();
		if(!q.empty())
		b=q.top();
		else
		b=0;
		while((n>0)&&(a>=b))
		{
			c+=a;
			a--;
			n--;
		}
		q.push(a);
		if(n<=0)
		{
			break;
		}
	}
	cout <<c<<endl;
    return 0;
}
