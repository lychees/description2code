#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s, p;
    	int j=0;
    	cin>>s>>p;
    	for(int i=0; i<s.length(); i++){
    		if(s[i]==p[j]){
    			j++;
    		}
    		if(j>=p.length())
    			break;
    	}
    	
    	if(j>=p.length())
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
