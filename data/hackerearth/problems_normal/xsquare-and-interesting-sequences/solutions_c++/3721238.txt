ll power(ll a,ll b)
{
	ll res=1;
	while(b)
	{
		if(b%2)
		{
			res=res*a;
			res=res%MAX;
			
		}
		b/=2;
		a=a*a;
		a=a%MAX;
		

	}
	return res;
}
