#include <bits/stdc++.h>
using namespace std;

typedef long long LL;

int main() {
	int t;
	LL a, b;
	
	cin >> t;
	
	while (t--) {
		cin >> a >> b;
	    
	    LL ans = 0;
	    while (a > 1 or b > 1) {
	    	if (a > b) {
	    		ans += a / b;
	    		a %= b;
	    	} else if (b > a) {
	    		ans += b / a;
	    		b %= a;
	    	}
	    }
	    
	    if (ans > 0) ans--;
	    cout << ans << endl;
	}
	
    return 0;
}
