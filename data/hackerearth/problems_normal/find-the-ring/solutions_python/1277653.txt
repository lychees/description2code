n = int(raw_input())
while n > 0:
	ans = 0
	a = raw_input().split()
	a = [int(x) for x in a]
	ind = a[0]
	moves = a[1]
	if moves>0:
		if ind == 0:
			ans = moves%2
		elif ind == 1:
			if moves%2==0:
				ans = 1
			else :
				ans = 0
		else :
			ans = moves%2
	else :
		ans = ind
	print ans
	n = n - 1
