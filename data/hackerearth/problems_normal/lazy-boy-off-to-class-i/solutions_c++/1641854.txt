#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    long long t,i,j,c,s,e;
    vector<long long> arr;
    long long sum;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	for(j=1;j<=7;j++)
    	{
    		scanf("%lld",&c);
	    	for(i=1;i<=c;i++)
	    	{
	    		cin>>s>>e;
	    		arr.push_back(e-s);
	    	}
    	}
    	std::sort(arr.begin(),arr.end());
    	int per=arr.size()/4;
    	for(i=0;i<arr.size()-per;i++)sum+=arr[i];
    	printf("%lld\n",60*60*24*7-sum);
    	arr.clear();
    	
    }
    return 0;
}
