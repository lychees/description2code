#include <bits/stdc++.h>

using namespace std;

#define ll long long

ll modpow(ll a, ll b, ll m)
{
	ll x=1ll, y=a;
	while(b){
		if(b%2)
			x=(x*y)%m;
		y=(y*y)%m;
		b/=2;
	}
	return x;
}

int main()
{

	int t;
	ll n, p, q, ans;
	scanf("%d", &t);
	while(t--){
		scanf("%lld", &n);
		p = modpow(2ll, n, 1000000007);
		ans = (2ll*modpow(p, 1000000005, 1000000007))%1000000007;
		printf("%lld\n", ans);
	}
	return 0;
}
