#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    long int i,j,N;
    long int v1[100001];
    cin>>T;
    while(T--)
    {
        cin>>N;
        int flag=0;
        for(i=1;i<=N;i++)
        {
            cin>>v1[i];
        }
        for(i=1;i<=N;i++)
        {
            if(v1[v1[i]]!=i)
            {
                flag=1;
                break;
            }
        }
        if(flag!=1)
            cout<<"inverse\n";
        else
            cout<<"not inverse\n";
    }
}
