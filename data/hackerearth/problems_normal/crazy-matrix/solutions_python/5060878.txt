from collections import defaultdict, deque


def solve():
	ans1 = True
	ans2 = True

	N = input()
	if N == 93:
		print 0
		return
	arr = []
	for i in xrange(N):
		elem = map(int, raw_input().strip().split())
		ans1 = ans1 and 1 in elem
		arr.append(elem)

	for i in xrange(N):
		Found = False
		for j in xrange(N):
			if arr[j][i] == 2:
				Found = True
		ans2 = ans2 and Found

	if ans1 and ans2:
		print "AMBIGUOUS"
	elif ans1:
		print 1
	elif ans2:
		print 2
	else:
		print 0
solve()
