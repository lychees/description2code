
#include <bits/stdc++.h>
using namespace std;

#ifdef EVIL_DEBUG
#define eprintf(...) fprintf(stderr, __VA_ARGS__)
#else
#define eprintf(...) do{}while(0);
#endif

typedef unsigned long long u64;

int main(int argc, char *argv[]){
    u64 t, a, n, k;
    scanf("%llu", &t);
    while(t--){
        scanf("%llu %llu %llu", &a, &k, &n);
        printf("%llu\n", a+(n-1)*k);
    }
    return 0;
}
