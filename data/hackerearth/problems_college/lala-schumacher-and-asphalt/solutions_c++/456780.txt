#include<cstdio>
#include<algorithm>
#define MAX 100010

using namespace std;
 
int main()
{
	int t;
	int x[MAX],y[MAX],fx,fy,n;
	long long ni,no;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",x+i);
		for(int i=0;i<n;i++)
			scanf("%d",y+i);
		scanf("%d%d%lld%lld",&fx,&fy,&no,&ni);
		long long m,p,t; 
		m=no*(abs(fx)+abs(fy)); 
		for(int i=0;i<n;i++)
		{ 
			p=no*(abs(x[i])+abs(y[i])); 
			t=ni*(abs(x[i]-fx)+abs(y[i]-fy)); 
			if((p+t)<m) 
				m=p+t; 
		} 
		printf("%lld\n",m);
	}
	return 0;
}
