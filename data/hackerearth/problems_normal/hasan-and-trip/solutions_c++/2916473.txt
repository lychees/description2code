#include <iostream>
#include <cmath>
#include <climits>
#include <cstdio>
#include <cfloat>
using namespace std;

struct points
{
	long long x;
	long long y;
	long double hap;
};

long double hp[3005];

int main()
{
    int n;
    cin>>n;
    struct points p[n+1];
    for(int i=0;i<n;i++)
    {
    	cin>>p[i].x>>p[i].y>>p[i].hap;
    }
    hp[0]=p[0].hap;
    for(int i=1;i<n;i++)
    {
    	hp[i]=-1ll<<60;
    	for(int j=0;j<i;j++)
    	{
            long double dist= sqrt((long double)(p[j].y - p[i].y)*(p[j].y - p[i].y)
                      + (p[j].x - p[i].x)*(p[j].x - p[i].x)) ;
    		if(hp[i]<hp[j]+p[i].hap-dist)
    		hp[i]=hp[j]+p[i].hap-dist;
    	}
    }
    cout<<fixed;
    cout.precision(6);
    cout<<hp[n-1]<<endl;
    return 0;
}
