#include <iostream>
using namespace std;

int main()
{
   string a;
   int occ[200] = {0};
   char str[200];
  getline(std::cin,a);
   int len = a.length();
   
   for(int i =0;i<len;i++)
   {
   	
   	occ[a[i]-' ']++;
   	
    str[a[i]- ' '] = a[i];
   }
   int grt=-1, loc=-1;
  for(int i =199;i>=0;i--)
   {
   	if(grt<=occ[i])
   	{
   		grt = occ[i];
   		loc = i;
   	}
   }
   //cout<<occ[1];
   cout<<str[loc]<<' '<<grt;
    return 0;
}
