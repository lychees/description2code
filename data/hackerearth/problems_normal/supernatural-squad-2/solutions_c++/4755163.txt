#include <bits/stdc++.h>
using namespace std;
long long t,n,k,dp[256][256];
long long rec(long long sum, long long curr)
{
	if(sum==n)
	{
		return 1;
	}
	if(n-sum<curr)
	{
		return 0;
	}
	if(dp[sum][curr]!=-1)
	{
		return dp[sum][curr];
	}
	else
	{
		dp[sum][curr]=0;
		for(int i=curr;i<=n-sum;i++)
		{	
			if(sum+i<=n)
			{
				dp[sum][curr]+=rec(sum+i,i);
			}
		}
		return dp[sum][curr];
	}
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n>>k;
		memset(dp,-1,sizeof(dp));
		cout<<rec(0,k)<<endl;
	}
	return 0;
}
