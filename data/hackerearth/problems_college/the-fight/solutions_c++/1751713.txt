#include<iostream>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<stdlib.h>
#include<map>
#include<algorithm>
#include<climits>
#include<set>
#include<vector>
#include<stack>
#include<queue>
#include<list>
 
#define rep(var,i,j) for(var = i ; var <= j ; ++var)
 
typedef long long int ll;
 
using namespace std;
 
int main()
{
	ll n,i,mm,m;
		
	vector<ll> a(1000001,0);
	vector<ll> p(1000001,0);
		
	scanf("%lld",&n);
	
	rep(i,0,n-1)
		{
			scanf("%lld",&mm);
			a[i]=mm;
		}
	
	
	scanf("%lld",&m);
	
	rep(i,1,m)
	{
		ll l,r,k;
		
		scanf("%lld %lld %lld",&l,&r,&k);
		
		p[l-1]+=k;
		p[r]-=k;
	}
	
	rep(i,0,n-1)
	{
		
		printf("%lld\n",a[i]+p[i]);
		
		p[i+1]+=p[i];
	}
 
    return 0;
}
