#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    char str[32];
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	for(int i=0;i<strlen(str);i++)
    	{
    		for(int j=i+1;j<strlen(str);j++)
    		{
    			if(str[i] == str[j])
    				str[j]='#';
    		}
    	}
    	
    	for(int i=0;i<strlen(str);i++)
    	{
    		if(str[i] == '#')
    			continue;
    		cout<<str[i];
    	}
    	
    	cout<<endl;
    }
    return 0;
}
