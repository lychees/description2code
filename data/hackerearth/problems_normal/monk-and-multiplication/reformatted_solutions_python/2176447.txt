n = input()
a = map(int, raw_input().split())
m1 = max(a[0], a[1])
if m1 == a[0]:
    m2 = a[1]
else:
    m2 = a[0]
m3 = 0
ans = [-1, -1]
for i in range(2, n):
    if m1 < a[i]:
        m3 = m2
        m2 = m1
        m1 = a[i]
    elif m2 < a[i]:
        m3 = m2
        m2 = a[i]
    elif m3 < a[i]:
        m3 = a[i]
    ans.append(m2*m1*m3)
for i in range(0, n):
    print ans[i]
