#include <bits/stdc++.h>
using namespace std;
void dfs(int adj[505][505],int v, bool &flag, int vis[], int n)
{
	vis[v] = 1;
	int i;
	for( i = 0 ; i < n ; i++ )
	{
		if(adj[v][i] == 1 && !vis[i])
		{
			dfs(adj,i,flag,vis,n);
		}
		else if(adj[v][i] == 1 && vis[i] == 1)
		{
			flag = 1;
		}
	}
	vis[v] = 2;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n, i, j, adj[505][505];
		scanf("%d",&n);
		for( i = 0 ; i < n ; i++ )
		{
			for( j = 0 ; j < n ; j++ )
			{
				scanf("%d",&adj[i][j]);
			}
		}
		bool flag = 0;
		int vis[505];
		memset(vis, 0, sizeof vis);
		dfs(adj,0,flag,vis,n);
		if(flag)
		{
			printf("Aye\n");
		}
		else
		{
			printf("Nae\n");
		}
	}
    return 0;
}
