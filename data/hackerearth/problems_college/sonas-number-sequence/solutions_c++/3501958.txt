#include<iostream>
#include<algorithm>
#include<string.h>
#include<vector>
using namespace std;

long long a[105];

int main()
{
    long long i,j,n,m,temp,sum=0;
    cin>>n;
    for(i=1;i<=n;i++) {
        cin>>a[i];
    }
    for(i=1;i<=n;i++) {
        temp=i*a[i]-sum;
        cout<<i*a[i]-sum<<" ";
        sum+=temp;
    }
    return 0;
}
