#include <iostream>
#define max(a,b) {a>b?a:b}
using namespace std;

int main()
{
    int t,l,r,i,x,maxx=0;
    cin>>t;
    while(t--)
    {
    	maxx=0;
    	cin>>l>>r;
    	for(i=l;i<=r;i++)
    	{
    		for(int j=l;j<=r;j++)
    		{
    			maxx = max(maxx,(i^j));
    		}
    	}
    	cout<<maxx<<endl;
    }
}
