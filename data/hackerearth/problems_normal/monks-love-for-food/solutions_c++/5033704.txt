#include<iostream>
using namespace std;
void pop(int stack[],int *top);
void push(int stack[],int *top,long int val,int size);

int main()
{
int q,ch,top=-1;
long int val;
cin>>q;
int stack[q];
int size=q;
for(int i=0;i<q;i++)
{ cin>>ch;
  switch(ch)
  { case 1:
  	pop(stack, &top);
  	break;
  	case 2:
  		cin>>val;
  	push(stack,&top,val,size);
  break;
  }
}
	
	return 0;
}

{ *top+=1;
