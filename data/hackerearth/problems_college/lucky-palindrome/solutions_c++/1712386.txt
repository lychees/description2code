#include <bits/stdc++.h>
using namespace std;
bool isPrime(int n){
	if(n<=1)
		return false;
	int val = sqrt(n);
	for(int i=2;i<=val;i++)
		if(n%i==0)
			return false;
	return true;
}
int main()
{
    int t;
    cin>>t;
    string s;
    while(t--) {
    	cin>>s;
    	int low=0 , high=0;
    	int i,j;
    	int l = s.length();
    	for(int k=1;k<l-1;k++) {
    		i = k-1;
    		j = k;
    		while(i>=0 && s[j]!='\0' && s[i] == s[j]){
    			if(high - low <= j-i){
    				high = j;
    				low = i;
    			}
    			i -= 1;
    			j += 1;
    		}
    		i = k-1;
    		j = k+1;
    		while(i>=0 && s[j]!='\0' && s[i] == s[j]){
    			if(high - low <= j-i){
    				high = j;
    				low = i;
    			}
    			i -= 1;
    			j += 1;
    		}
    	}
    	if(isPrime(high-low+1))
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
