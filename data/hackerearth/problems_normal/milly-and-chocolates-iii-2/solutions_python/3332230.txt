t = input()
while t:
	t-=1
	n, k = map(int, raw_input().split())
	arr = []
	for i in range(n):
		tmp = raw_input().split()
		dic = {}
		for j in tmp[1:]:
			dic[j]=1
		arr.append(dic)
	
	ans = n+1
	mx = 1<<n
	for i in range(mx):
		tmp={}
		cnt=0
		for j in range(n):
			if (i&(1<<j)):
				cnt+=1
				for x in arr[j]:
					tmp[x]=1
		if len(tmp)>=k:
			ans = min(ans, cnt)
	if ans==n+1:ans=-1
	print ans
