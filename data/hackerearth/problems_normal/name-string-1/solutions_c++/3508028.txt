#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	long long int a[3],ans;
	cin>>t;
	while(t--)
	{
		for(int i=0;i<3;i++)
		cin>>a[i];
		sort(a,a+3);
		if(a[0]==0 && a[1]==0)
		{
			if(a[2]==0)
			ans=0;
			else
			ans=1;
		}
		else
		{
			if(a[2]!=a[1])
			ans=a[1]*2+1;
			else
			ans=a[1]*2;
		}
		cout<<ans<<endl;
	}
}
