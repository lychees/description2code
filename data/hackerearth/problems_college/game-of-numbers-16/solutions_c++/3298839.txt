#include <iostream>
using namespace std;

int main()
{int t;
	cin>>t;
	while(t){
		t--;
		int q,m,n,count=0;
		
		cin>>q;
        for(n=1;n<=q;n++){
        	for(m=1;m<n;m++){
        		if((m*n)%(m+n)==0){
        			count++;
        			
        		}
        	}
        }	
        cout<<count<<'\n';
	}
    return 0;
}
