#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e2+2;
const int MAXM = 1e4+2;
long long dp[102][MAXM];
long long prefix[102][MAXM];
long long maxi[MAXM];
long long ar[MAXN];
int main() {
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) {
		int x;
		scanf("%d",&x);
		ar[i]=x;
		prefix[i][0]=0;
		for(int j=1;j<=x;j++) {
			int L;
			scanf("%d",&L);
			prefix[i][j]=prefix[i][j-1]+1LL*L;
		}
		dp[i][0]=0;
	}
	dp[0][0]=0;
	for(int i=1;i<=m;i++)
		dp[0][i]=0;
	for(int i=1;i<=n;i++) {
		for(int j=1,p=min(ar[i],(long long)m);j<=p;j++) {
			maxi[j]=-1;
			for(int k=1;k+j<=ar[i]+1;k++)
				maxi[j]=max(maxi[j],prefix[i][j+k-1]-prefix[i][k-1]);
		}
		for(int j=0;j<=m;j++) {
			dp[i][j]=dp[i-1][j];
			for(int k=1,p=min((long long)j,ar[i]);k<=p;k++)
					dp[i][j]=max(dp[i][j],dp[i-1][j-k]+maxi[k]);
		}
	}
	printf("%lld\n",dp[n][m]);
	return 0;
}
