#include <bits/stdc++.h>
using namespace std;
bool prime[10002];
void pre()
{
	int i;
	for(i = 2 ; i <= 10000; i++)
	{prime[i] = 1;
	}
	for(int p =2; p*p <= 10000 ; p++)
	{
		if(prime[p] == 1)
		{
			for(i = p*2 ;i <= 10000; i+= p )
			{
				prime[i] = 0;
			}
		}
	}
}
int main()
{
	pre();
    int n;
    int i;
    cin >> n;
    for(i = 2; i <= n ; i++)
    {
    	if(!prime[i])
    	{cout<<i << " ";
    	}
    }
    return 0;
}
