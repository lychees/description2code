#include<iostream>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--){
		int n;
		cin >> n;
		long long odd = 0, even = 0;
		int o = 0, e = 0, po = 0, pe = 0;		
		for(int i = 1; i <= n; i++){
			int a;
			cin >> a;
			if(a&1){
				o = pe+1;
				e = po;
			}
			else{
				o = po;
				e = pe+1;
			}
			po = o;
			pe = e;
			odd += o;
			even += e;
		}
		if(odd >= even)
			cout << "BOB" << endl;
		else cout << "ANDI" << endl;
	}
	return 0;
}
