#include<bits/stdc++.h>
using namespace std;
int prime(int x)
{
	for(int i=2;i<x/2+1;i++)
	{
		if(x%i==0)
		return 0;
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,a,b,f;
	    cin>>n;
	    if(n%2!=0)
	    {
	    	n=n/2;
	    	a=n,b=n+1;
	    }
	    else
	    {
	    	n=n/2;
	    	a=n-1;
	    	b=n+1;
	    }
	    f=0;
	    while(a>1)
	    {
	    	if(prime(a)&&prime(b))
	    	{
	    		f=1;
	    		break;
	    	}
	    	a--;
	    	b++;
	    }
	    if(f==1)
	    cout<<a<<" "<<b<<endl;
	    else
	    cout<<"-1"<<" "<<"-1"<<endl;
	    
   }
   return 0;
}
