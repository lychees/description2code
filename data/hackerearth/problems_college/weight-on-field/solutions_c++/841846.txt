#include<bits/stdc++.h>
using namespace std;

int main()
 {
 	
 	vector<long long> fibo(60);
 	vector<vector<long long> > ans(60,vector<long long>(60));
 int i,j,t,n;
 	fibo[0]=fibo[1]=1;
 	ans[1][2]=ans[1][1]=1;
 	for(i=2;i<=50;i++)
 	  {
 	   fibo[i]=fibo[i-1]+fibo[i-2];
 	   ans[1][i+1]=fibo[i];
      }
 	  
 	
 	for(i=2;i<=50;i++)
 	 {
 	 	for(j=0;i+j<=50;j++)
 	 	 {
 	 	 	ans[i][i+j]=fibo[i-1]*fibo[j];
 	 	 }
 	 }/*
 	 for(i=1;i<=20;i++)
 	  {
 	  	for(j=1;j<=20;j++)
 	  	  cout<<ans[i][j]<<" ";
 	  	  cout<<endl;
 	  }*/
 	 cin>>t;
 	 while(t--)
 	  {
 	  	cin>>n;
 	  	long long num=0;
 	  	for(i=1;i<=50;i++)
 	  	  num+=ans[i][n];
 	  //	cout<<num<<" ";
 	  	double sum=num*2;
 	  	sum/=fibo[n];
 	  	printf("%.6f\n",sum);
 	  }
 	
 }
