import math
t = input()


def isqrt(n):
    x = 1
    y = (x+n/x)/2
    while abs(y-x) > 1:
        x = y
        y = (x+n/x)/2
    while y**2 > n:
        y -= 1
    return y


for i in xrange(t):
    l, r = map(int, raw_input().split())
    x = isqrt(l)
    if x**2 == l:
        print isqrt(r)-x+1
    else:
        print isqrt(r)-x
