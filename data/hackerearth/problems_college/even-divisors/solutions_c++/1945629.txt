#include <iostream>
#include <math.h>
using namespace std;

int primeFactors(long long int n)
{
	int ans = 0,count; 
    // Print the number of 2s that divide n
    while (n%2 == 0)
    {
        //printf("%d ", 2);
        ans++;
        n = n/2;
    }
 	if(ans == 0) return 0;
    // n must be odd at this point.  So we can skip one element (Note i = i +2)
    for (int i = 3; i <= sqrt(n); i = i+2)
    {
    	count = 0;
        // While i divides n, print i and divide n
        while (n%i == 0)
        {
            //printf("%d ", i);
            count++;
            n = n/i;
        }
        if(count > 0) ans = ans*(count+1);
    }
    // This condition is to handle the case when n is a prime number
    // greater than 2
    if (n > 2){
       //printf ("%d ", n);
       ans = ans *(2);
    }
    return ans;
}

int main()
{
    int t;
    long long int n,temp;
    int ans;
    cin>>t;
    while(t--){
    	ans = 0;
    	temp = 1;
    	cin>>n;
    	cout<<primeFactors(n)<<endl;
    }
    return 0;
}
