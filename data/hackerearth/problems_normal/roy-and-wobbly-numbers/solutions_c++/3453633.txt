#include <iostream>
using namespace std;

int main()
{
	int i, j, t, k, n;
	cin>>t;
	while (t-->0)
	{
		cin>>n>>k;
	
		if (k<9)
		{
			if (k==1)
				k=0;
			for(j =0;j<n;j++)
			{
				if (j%2==0)
					cout<<1;
				else
					cout<<k;
			}
		}
		else if (k > 81)
			cout<<-1;
		else
		{
			i=1;
			while(k>9)
			{	
				k=k-9;
				i=i+1;
			}
			if(k<=i)
				k=k-1;
			for(j =0;j<n;j++)
			{
				if (j%2==0)
					cout<<i;
				else
					cout<<k;
			}
		}
		cout<<"\n";
	}

}
