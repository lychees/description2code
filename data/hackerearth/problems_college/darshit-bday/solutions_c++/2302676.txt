#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int b,w;
		cin>>b>>w;
		ll int x,y,z;
		cin>>x>>y>>z;
		ll int ans=min(min(b*x + w*y, b*x + (z+x)*w),w*y + (z+y)*b);
		cout<<ans<<endl;
	}
	return 0;
}
