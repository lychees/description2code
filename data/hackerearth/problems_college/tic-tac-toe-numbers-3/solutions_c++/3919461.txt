#include <bits/stdc++.h>
using namespace std;

int main () {
	long long int dp[50];
	dp[0] = 0;
	dp[1] = 1;
	for (int i=2; i<26; i++) 
		dp[i] = dp[i-1] + dp[i-2];

	bool mark[1000001];
	for (int i=0; i<=1000000; i++)
		mark[i] = false;

	for (int i=0; i<26; i++) {
		mark[dp[i]] = true;
	}

	int tc;
	scanf("%d",&tc);
	while (tc--) {
		int n;
		scanf("%d",&n);
		long long ans = 0;
		for (int i=0; i<n; i++) {
			if (mark[i]==false)
				ans += i;
		}

//		cout << "Sum is: " << ans << endl;

		int cnt = 0;
		bool flag = true;
		for (int i=2; i<sqrt(ans); i++) {
			if (ans%i==0)
				cnt++;
			if (cnt>0) 
				flag = false;
		}
		if (flag)
			puts("Tic");
		if (ans%2==0)
			puts("Tac");
		if (ans%2 and flag==false)
			puts("Toe");
	}
	return 0;
}
