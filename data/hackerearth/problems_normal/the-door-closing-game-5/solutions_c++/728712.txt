template<typename T>
vector<pair<T,int> > runLength(const T a[], int n) {
	vector<pair<T,int> > r;
	if(n <= 0) return r;
	int cnt = 0;
	for(int i = 0; i <= n; i ++) {
		if(i == n || (i != 0 && a[i] != a[i-1])) {
			r.push_back(make_pair(a[i-1], cnt));
			cnt = 0;
		}
		cnt ++;
	}
	return r;
}


int main() {
	int T;
	scanf("%d", &T);
	char *s = new char[101];
	rep(ii, T) {
		scanf("%s", s);
		vector<pair<char,int> > runs = runLength(s, strlen(s));
		int x = 0;
		if(runs.size() == 1)
			x = 0;
		else
			each(i, runs) if(i->first == '_')
				x ^= i->second;
		puts(x != 0 ? "Alice" : "Bob");
	}
	return 0;
}
