#include <bits/stdc++.h>
using namespace std;
void distance(long long int *x,long long int *y,long long int n)
{
	long long int sumX=0,sumY=0,sum=0,p=0,m=0;
	sort(x,x+n);
	sort(y,y+n);
	for(long long int i=0;i<n;i++)
	{
		sumX=(sumX+(n-(2*i)-1)*x[n-i-1])%1000000007;
	}
	for(long long int i=0;i<n;i++)
	{
		sumY=(sumY+(n-(2*i)-1)*y[n-i-1])%1000000007;
	}
	sum=(sumX+sumY)%1000000007;
	cout<<sum<<endl;
}
int main()
{
	long int n,i=0;
	cin>>n;
	long long int x[n],y[n];
	while(i<n)
	{
		cin>>x[i]>>y[i];
		i++;
	}
	distance(x,y,n);
}
