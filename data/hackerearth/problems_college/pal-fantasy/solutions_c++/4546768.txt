#include <iostream>
#include <algorithm>
using namespace std;

bool rev(string str){
	
	string temp;
	temp = str;
	reverse(str.begin(), str.end());
	if(temp == str){
		return true;
	}
	else{
		return false;
	}
}

int main()
{
    int t;
    string pal;
    
    cin>>t;
    
    while(t--){
    	
    	cin>>pal;
    	
    	int size = pal.size();
    	
    	if(rev(pal)){
    		cout<<pal.size()<<endl;
    		continue;
    	}
    	for(int i=0; i<pal.size(); i++){
    		
    		pal.insert(pal.begin()+size, pal[i]);
    		if(rev(pal)){
    			break;
    		}
    	}
    	
    	cout<<pal.size()<<endl;
    }
    return 0;
}
