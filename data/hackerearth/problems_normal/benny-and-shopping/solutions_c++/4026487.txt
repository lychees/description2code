#include <bits/stdc++.h>
using namespace std;
vector<char> vec;
int main(){
    int t;cin>>t;
    while(t--){
    	char ch[1000];
    	scanf(" %[^\n]s",ch);
    	int i=0;
    	while(ch[i]!='\0') {
			if(ch[i]=='$') {
				vec.push_back(ch[i++]);
				while(ch[i]=='0' || ch[i]==' ') i++;

				while(ch[i]!='\0'){
					if(ch[i]!=' ') {
						int pp = int(ch[i]-'0');
						if(pp>9) break;
						else vec.push_back(ch[i]);
					}
					i++;
				}
				break;
			}
			i++;
    	}
		//cout<<vec.size()<<endl;
		if(vec.size()==1) cout<<vec[0]<<"0";
		else for(int tt=0;tt<vec.size();tt++) cout<<vec[tt];
		cout<<endl;
		fflush(stdout);
		vec.clear();
    }
    return 0;
}
