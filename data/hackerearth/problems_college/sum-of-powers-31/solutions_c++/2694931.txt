#include <iostream>
#include<vector>
#define siz 1000001
using namespace std;
vector<int>v(siz);
int main()
{
	fill(v.begin(),v.end(),0);
	v[1]=1;
	for(long long int i=2;i<siz;i++){
		long long temp=i*i;
		if(temp<v.size() && !v[temp]){
			for(int j=temp;j<v.size();j*=i)v[j]=1;
		}
	}	
   int t;
   cin>>t;
   while(t--){
   	int n;
   	cin>>n;
   	int flag=0;
   	for(int i=0;i<v.size() && n>=i && !flag;i++){
   		if(v[i]){
   			int temp=n-i;
   			if(v[temp])flag=1;
   		}
   	}
   	if(flag)cout<<"Yes\n";
   	else cout<<"No\n";
   }
    return 0;
}
