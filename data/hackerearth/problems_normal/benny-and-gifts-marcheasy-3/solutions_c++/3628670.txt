#include <bits/stdc++.h>
using namespace std;
#define ll long long 
#define fi first 
#define se second
#define pb push_back  
#define mp make_pair
map<pair<ll,ll>, bool >m;
pair<ll , ll > p;

int main()
{
  string s;
  int i , x, y, ans=0;
  x=y=0;
  p=mp(0,0);
  m[p]=true;
  cin>>s;   
  for(i=0;s[i]!='\0';i++)
  {
   if(s[i]=='R')
      ++x;
   if(s[i]=='L')
      --x;
   if(s[i]=='U')
      ++y;
   if(s[i]=='D')  
      --y;
   p=mp(x,y);
  // cout<<p.fi<<" "<<p.se<<endl;
   if(m[p]==false)
      m[p]=true;
   else
      ans++;
  }
  

  cout<<ans;

   return 0;
}
