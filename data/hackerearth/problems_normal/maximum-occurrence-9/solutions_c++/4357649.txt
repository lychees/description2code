#include <iostream>
using namespace std;
#include <string>
#include <map>
#define dbg(x) { cout<<#x<<" : "<<x<<endl;}
int main()
{
	string s;
	getline(cin,s);
	map<char,int> mymap;
	map<char,int>::iterator it;
	int l=s.length();
	for(int i=0; i< l ; i++) {
		it = mymap.find(s[i]);
		if( it != mymap.end() ) {
			int times=mymap.find(s[i])->second;
			times++;
			mymap.erase(s[i]);
			mymap.insert( pair<char,int>(s[i],times) );
		}
		else {
			mymap.insert( pair<char,int>(s[i],1) );
		}
	}
	char max=mymap.begin()->first;
	for( it=mymap.begin(); it!=mymap.end(); it++) {
	//	dbg(it->first);
	//	dbg(it->second);
		if( it->second > mymap.find(max)->second )
			max= it->first;
	}
	cout<<max<<" "<<mymap.find(max)->second<<endl;
	
    //cout << "Hello World!" << endl;
    return 0;
}
