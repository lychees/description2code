#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;

int main()
{
	int tc;
	cin >> tc;


	while(tc--)
	{	
		int d , wt,res;
		res =50;
		cin >> d >> wt;
		if(d > 60)
		{
			cout << "NA" << endl;
		}
		else
		{
			int dc;
			
			dc = 12;
			
			
			res = res + wt*2;	
			
			for(int i=1 ; i <= d; i++)
			{
				if(i>2)
				{
					res = res + dc;
				}
			}
			cout << res << endl;
		}
	}
}
