#include<bits/stdc++.h>
using namespace std;
#define lx long long
int prec[110];
int dig[50004];
void precalc()
{
    for (int i=0;i<=110;i++)
        prec[i] = i % 10;
}
int main()
{
    ios::sync_with_stdio(0);
    //freopen("input.txt","r",stdin);
    precalc();
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,c,n;
        cin>>a>>b>>c>>n;
        dig[0] = a;
        dig[1] =b;
        for (int i=2;i<=n;i++)
            dig[i]=prec[dig[i-1]*c+dig[i-2]];
        if (n==1)
        {
            if (a==0)
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
            continue;
        }
        lx p = 0;
        p = (dig[0]*10+dig[1])%41;
        for (int i=2;i<n;i++)
            p = (p*10+dig[i])%41;
        if (p==0)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
