#include <stdio.h>
using namespace std;
 
void swap (char *x, char *y)
{
    char temp;
    temp = *x;
    *x = *y;
    *y = temp;
}
 
void permute(char *a, int i, int n) 
{
   int j; 
   if (i == n)
     printf("%s\n", a);
   else
   {
        for (j = i; j <= n; j++)
       {
          swap((a+i), (a+j));
          permute(a, i+1, n);
          swap((a+i), (a+j)); 
       }
   }
} 
 
int main()
{
	char str[100];
	gets(str);
	long long int testCases;
	scanf("%lld", &testCases);
	for(long long int c = 0; c <testCases; c++)
	{
		int start, end;
		scanf("%d%d", &start, &end);
		permute(str, start, end);
		printf("\n");
	}
	return 0;
}
