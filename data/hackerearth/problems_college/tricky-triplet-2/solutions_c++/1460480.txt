#include<bits/stdc++.h>
using namespace std;
#define ll long long
inline int inpi( )
{
   int n = 0 , s = 1 ;
    char p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 ;
    while( ( p < '0' || p > '9' ) && p != EOF && p != '-' )
        p = getchar_unlocked( ) ;
    if( p == '-' )
        s = -1 , p = getchar_unlocked( ) ;
    while( p >= '0' && p <= '9' )
    {
        n = ( n << 3 ) + ( n << 1 ) + ( p - '0' ) ;
        p = getchar_unlocked( ) ;
    }
    return n * s ;
}
int main() {
	int n;
	n=inpi();
	int i,j,k,a[n],f,s,t,d;
	ll tot=0;
	for(i=0;i<n;i++) a[i]=inpi();
	for(i=0;i<n;i++) {
		f=a[i];
		for(j=i+1;j<n;j++) {
			s=a[j];
			d=s-f;
			for(k=j+1;k<n;k++) {
				if(a[k]-s==d) tot++;
			}
		}
	}
	printf("%lld\n",tot);
	return 0;
}
