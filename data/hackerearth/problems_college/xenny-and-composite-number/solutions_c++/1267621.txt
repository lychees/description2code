#include<iostream>
#include<queue>
#include<cmath>
using namespace std;
const int MAX = 100010;
int prime[MAX];
void sieve_prime()
{
	int i, j;
	prime[0] = 1;
	prime[1] = 1;
	for (i = 2; i <= sqrt(MAX); i++)
	{
		j = i;
		if (prime[i] == 0)
		{
			while (i*j <= MAX)
			{
				prime[i*j] = 1;
				j++;
			}
		}
	}
}
int main()
{
	int test;
	cin>>test;
	sieve_prime();
	while(test--)
    {
        int n;
        cin>>n;
        if(n==0)
            n=3;
        for(int i=n+1;i<100009;i++)
            if(prime[i]==1)
            {
                cout<<i<<endl;
                break;
            }
    }
	return 0;
}
