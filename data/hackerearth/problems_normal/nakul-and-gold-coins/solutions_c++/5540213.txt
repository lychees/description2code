#include<bits/stdc++.h>
#define assn(n,a,b) assert(n<=b && n>=a)
#define sd(n) scanf("%d",&n)
using namespace std;
#define MAXN 1000009
#define pb push_back
char str[MAXN];
vector<int> ar;
int cnt;
void gen(){
    for(int i=2; i*i<=MAXN; i++)
        if(str[i]==0)
            for(int j=i; j*i<=MAXN; j++)
                str[i*j]=1;
    for(int i=2; i<=MAXN-9; i++)
        if(str[i]==0)ar.pb(i);
}
int main()
{
    gen();
    int t;
    sd(t);
    assn(t,1,300);
    cnt=ar.size();
    while(t--){
        int l,r;
        sd(l),sd(r);
        assn(l,1,100000000);
        assn(r,1,100000000);
        assert(r>=l);
        int ans=0;
        for(int i=0; i<cnt and ar[i]<=r; i++)
        {
            int ll=l/ar[i]+(l%ar[i]>0);
            ll=max(ll,ar[i]+1);
            int rr=r/ar[i];
            if(rr<ll)continue;
            ans += upper_bound(ar.begin(),ar.end(),rr)-lower_bound(ar.begin(),ar.end(),ll);
        }
        cout << ans << endl;
    }
    return 0;
}
