#include<bits/stdc++.h>

using namespace std;

   bool myComparison (const pair<int,string> &a,const pair<int,string> &b)
{
       if(a.first == b.first)
        return a.second < b.second;
       else
	    return a.first > b.first; 
}

int main(){
	vector <pair <int,string> > v1;
     string str;
     int i,n,num;
	 cin >> n;
	 for( i=0;i<n;i++){
	 	cin >> str >> num;
	 	v1.push_back(make_pair(num,str));
	 }	
   
 

    sort(v1.begin(),v1.end(),myComparison);
    for( i=0;i<n;i++){
    	cout<<v1[i].second << " " << v1[i].first <<endl;
	}
	
	return  0;
}
