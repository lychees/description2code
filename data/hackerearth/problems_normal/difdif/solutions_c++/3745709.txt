#include <bits/stdc++.h>
using namespace std;

int bernstein_h(string &A, int mod = 1000000007, int base = 33) {
    int len = A.size();
    long long hval = 0;

    for(int i = 0; i < len; ++i) {
        hval = (hval*base + A[i]) % mod;
    }

    return hval;
}

int main() {
    string T;
    cin >> T;

    map<long long, bool > M;
    int cnt = 0, mod = 1000000007;

    int len = T.length();
    for(int i = 0; i < len; ++i) {
        long long hval = 0;
        for(int j = i; j < len; ++j) {
            hval = (hval * 33 + T[j]);

            if(!M[hval]) {
                M[hval] = 1;
                cnt++;
            }
        }
    }

    cout << cnt << endl;
    return 0;

}
