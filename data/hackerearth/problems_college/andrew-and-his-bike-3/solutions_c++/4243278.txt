#include <bits/stdc++.h>

using namespace std;

vector < int > v[100010];
int vis[100010],dp[100010];

int dfs(int x)
{
    int i,j;

    vis[x]=1;
    dp[x]=1;
    for(i=0;i<v[x].size();i++)
    {
        if(!vis[v[x][i]])
            dp[x]+=dfs(v[x][i]);
    }
    return dp[x];
}

int main()
{
    int n,m,i,x,y;

    //freopen("input.txt","r",stdin);
    cin>>n;

    for(i=0;i<n-1;i++)
    {
        cin>>x>>y;
        v[x].push_back(y);
        v[y].push_back(x);
    }

    dfs(1);

    cout<<dp[1]-1<<" ";
    for(i=2;i<=n;i++)
        cout<<dp[i]<<" ";
}
