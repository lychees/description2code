#include <bits/stdc++.h>
using namespace std;

bool primes[1000010];
vector<int> v[1000010];

void pre(){
	memset(primes, 0, sizeof primes);
	primes[1] = 1;
	primes[0] = 1;
	for(int i = 2; i < 1000010; i++){
		if(primes[i] == 0){
			v[i].push_back(i);
			for(int j = i+i; j < 1000010; j += i){
				primes[j] = 1;
				v[j].push_back(i);
			}
		}
	}
}

int main()
{
    int t;
    cin >> t;
    int a;
    pre();
    while(t--){
    	cin >> a;
    	int n = a;
    	if(v[a][0] != 2){
    		cout << "2^0";
    	}
    	if(v[a][0] < a) {
	    	for(int i = 0; i < v[a].size(); i++){
	    		int cnt = 0;
	    		while(n % v[a][i] == 0){
	    			n /= v[a][i];
	    			cnt++;
	    		}
	    		if(i != 0 || v[a][0] != 2){
	    			cout << "*";
	    		}
	    		cout << v[a][i] << "^" << cnt;
	    	}
    	}
    	cout << endl;
    }
}
