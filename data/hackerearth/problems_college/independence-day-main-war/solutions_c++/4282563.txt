#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    long long int i=0,t,n;
    float x=0,y=0,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	x=0;y=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a>>b;
    		x+=a;
    		y+=b;
    	}
    	printf("(");
    	printf("%.2f",x/n); 
    	printf(" , ");
    	printf("%.2f",y/n);
    	printf(")\n");
    //	cout<<"("<<(x)/n<<" , "<<y/n<<")"<<"\n";
    }
    return 0;
}
