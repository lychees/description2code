#include <iostream>
#include <vector>
using namespace std;

int main()
{
	long long n,q,l,r,sum=0;
	cin>>n;
	vector<int> a(100003);
    for(int i=1;i<=n;i++)
    {
    	cin>>a[i];
    }
    cin>>q;
    for(int j=1;j<=q;j++)
    {
    	sum=0;
    	cin>>l>>r;
    	for(int k=l;k<=r;k++)
    	{
    		sum+=a[k];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
