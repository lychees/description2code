t = int(raw_input())

for i in range(t):
    n = int(raw_input())
    st = raw_input()
    s = st[0]
    count = 0
    for i in range(len(st)):
        if ((i & 1) == 0 and s != st[i]) or ((i & 1) == 1 and s == st[i]):
            count += 1

    print min(count >> 1, n-(count >> 1))
