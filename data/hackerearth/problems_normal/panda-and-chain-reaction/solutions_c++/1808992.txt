#include<bits/stdc++.h>
using namespace std;
#define MOD 1000003
long long a[10000005];
void precomp() {
    a[0]=1;
    a[1]=1;
    a[2]=2;
    for(int i=3;i<1000005;i++) {
        a[i]=(a[i-1]*i)%MOD;
    }
}
int main() {
    int t;cin>>t;
    precomp();
    while(t--) {
        long long n,k;cin>>n>>k;
        if(n>=1000003)
            cout<<0<<endl;
        else {
        cout<<((k%MOD)*(a[n]%MOD))%MOD<<endl;
        }
    }
    return 0 ;
}
