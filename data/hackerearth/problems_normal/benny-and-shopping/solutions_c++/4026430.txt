#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,p;
    scanf("%d",&t);
    for(p=0;p<=t;p++)
    {
    	string str;
    	getline(cin,str);
    	//cout<<str<<endl;
    	int i,l,j=0,k;
    	l = str.length();
    	char temp[1001];
    	for(i=0;i<l;i++)
    	{
    		if(str[i] == '$')
    		{
    			//cout<<"AAA"<<endl;
    			k = i+1;
    			while(((int)str[k]>= 48 && (int)str[k]<= 57)|| (int) str[k] == 32)
    			{
    				//cout<<"BBB"<<endl;
    				temp[j++] = str[k++];
    			}
    			temp[j] = '\0';
    		}
    	}
    	//printf("%s\n",temp);
    	char s[1001];
        k = 0;
        int flag =0;
    	l = strlen(temp);
    	for(j=0;j<l;j++)
    	{
    		if(temp[j] == '0' && flag == 0)
    		  continue;
    		else if((int)temp[j]>= 48 && (int)temp[j]<= 57)
    		{
    			flag = 1;
    			s[k++] = temp[j];
    		}
    	}
    	s[k] = '\0';
    	if(s[0] == '\0' && p>0)
    	  cout<<"$0\n";
    	else if(p>0) 
    	cout<<"$"<<s<<endl;
    }
    return 0;
}
