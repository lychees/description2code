#include <bits/stdc++.h>

using namespace std;

struct circulo
{
	int x, y, r;
};

int main()
{
	int n;
	scanf("%d", &n);
	circulo circ[10];
	for(int i = 0; i < n; i++)
		scanf("%d %d %d", &circ[i].x, &circ[i].y, &circ[i].r);
	int res = 0;
	for(int i = 1 ; i <= 1000; i++)	
		for(int j = 1; j <= 1000; j++)
		{
			int cont = 0;
			for(int k = 0; k < n && cont < 2; k++)
				cont += (hypot(abs(circ[k].x - i), abs(circ[k].y - j)) <= circ[k].r);
			if(cont > 1)
				res++;
		}
	printf("%d\n", res);
}
