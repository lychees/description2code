#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    long int a[n+1][n+1];
    long long int sum = 0,sum2 = 0;

    for (int i = 1;i <= n;i++)
    {
        for (int j = 1; j<= n;j++)
        {
            cin >> a[i][j];
        }
    }

    for (int i = 1;i <= n;i++)
    {
        for (int j = 1;j <= n;j++)
        {
            if (i == j)
                sum += a[i][j];
        }
    }

    long int k = n;
    for (int i = 1;i <= n;i++)
    {
        sum2 += a[i][k];
        k--;
    }

    cout << abs(sum - sum2) << endl;

    return 0;
}
