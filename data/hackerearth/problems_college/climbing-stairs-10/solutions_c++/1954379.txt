#include <iostream>
#include <cmath>
using namespace std;
int main()
{
    long int t, n, sp, spp, w, i;
    cin>>t;
    while(t--) {
    	sp=1;
    	spp=1;
    	cin>>n;
    	if(n==1)
    		cout<<"1";
    	else if (n==2)
    		cout<<"2";
    	else {
	    	for(i=2; i<n+1;i++) {
	    		w=spp+sp;
	    		spp=sp;
	    		sp=w;
	    	}
  			cout<<w;	
    	}
    	cout<<endl;
    }
    return 0;
}
