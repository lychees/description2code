#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long t;
    cin>>t;
    long n,a,b,c,d,move,newd,newd1;
    long pawnmove,kingmove;
    while(t--)
    {
    	cin>>n>>a>>b>>c>>d>>move;
    	pawnmove = (n-a);
    	
    	kingmove = (n-c);
    	
    	newd = d - (n-c);
		newd1 = d + (n-c);
		
		if(newd1>=n)newd1=n;
		
		if(b>=newd && b<=newd1)
		{
			
		}
		else
		{
			kingmove+=min(abs(b-newd),abs(b-newd1));
		}
		
    	if(move==0)
    	{
    		if(pawnmove<kingmove)
    		{
    			cout<<"White Wins\n";
    		}
    		else
    		{
    			cout<<"Draw\n";
    		}
    	}
    	else
    	{
    		if(pawnmove<(kingmove-1))
    		{
    			cout<<"White Wins\n";
    		}
    		else
    		{
    			cout<<"Draw\n";
    		}
    	}
    	
    }
    return 0;
}
