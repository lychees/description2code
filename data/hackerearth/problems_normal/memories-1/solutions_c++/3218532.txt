#include<bits/stdc++.h>

using namespace std;
typedef long long int ll;
typedef vector<ll> vi;
typedef vector<vi> vvi;
typedef pair<ll,ll> ii;
typedef vector<ii> vii;

#define all(x) (x).begin(),(x).end()
#define pb push_back
#define mp make_pair
#define tr(c,it) for(typeof((c).begin()) it=(c).begin();it!=(c).end();it++)
#define rd(x) scanf("%lld",&x)
#define wr(x) printf("%lld\n",x)
#define rep(i,a,b) for(i=a;i<b;i++)
#define pi 3.141592653589793238462643383279
#define f first
#define s second
#define MOD 1000000007     // 10^9+7
#define INF 1000000009     // 10^9+9

ll check[17][17];
ll setbit[17];

int main()
{
   ll n,q,x,y,i,ans,j,m,flag,p,t;
   rd(t);
   while(t--){
   rd(n);
   rd(m);
   memset(check,0,sizeof check);
   while(m--)
   {
     rd(x);
     rd(y);
     check[x][y]=check[y][x]=1;
   }
   ans=0;

   rep(i,1,1<<n)
   {
     memset(setbit,0,sizeof setbit);
     flag=0;
     rep(j,0,n)
     {
       if(i&(1<<j))
        setbit[n-j]=1;
     }
     rep(p,1,16)
      {
        rep(q,1,16)
        {
          if(check[p][q]&&setbit[p]&&setbit[q])
            {
              flag=1;
              break;
            }
        }
        if(flag)
         break;
      }
      if(!flag)
       ans++;
   }
   wr(ans);
   }

   return 0;
}
