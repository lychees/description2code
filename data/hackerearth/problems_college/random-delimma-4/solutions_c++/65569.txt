#include <iostream>
using namespace std;

int main()
{
    int t;
    int n,k;
    double i;
    double ans;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&k,&n);
    	if(k>n)
    	{
    		printf("1.000000000\n");
    	}
    	else
    	{
    		ans=1.0;
    		for(i=n-k+1;i<n;i++)
    		{
    			ans*=i/n;
    		}
    		printf("%0.9lf\n",1.0-ans);
    	}
    }
    return 0;
}
