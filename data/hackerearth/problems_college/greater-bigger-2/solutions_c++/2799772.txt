#include "iostream" 
#include "bits/stdc++.h"

using namespace std;


string lexiSmallestLarge(string s){

	int n  =s.length();
	int firstIndex  = -1;
	for(int i=n-1;i>0;i--){
		if(s[i-1]<s[i])
		{
			firstIndex = i-1;
			break;
		}
	}

	char mini = 'z'+1;
	int ceilIndex = -1;
	if(firstIndex!=-1){

		for(int i = firstIndex+1;i<n;i++){
			if(s[i]>s[firstIndex] && s[i]<mini){
				mini = s[i];
				ceilIndex  = i;
			}
		}
	}

	if(ceilIndex!=-1 && firstIndex!=-1){
		char temp = s[firstIndex];
		s[firstIndex]=s[ceilIndex];
		s[ceilIndex]=temp;
	}
	else return "no answer";
	//cout<<firstIndex<<" "<<ceilIndex<<endl;
//	sort(s+firstIndex,s+n);
	//qsort( s + firstIndex+ 1, size - firstIndex - 1, sizeof(s[0]));
	
	for(int i=firstIndex+1;i<n;i++){
		for(int j=i+1;j<n;j++){
			if(s[i]>s[j]){
				char t= s[i];s[i]=s[j];s[j]=t;
			}
		}
	}
	
	
	return s;

}



int main(){
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		cout<<lexiSmallestLarge(s)<<endl;
	}
}
