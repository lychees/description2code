#include <iostream>
#include <string>

using namespace std ;

int main(int argc, char **argv) {
	string s;
	cin>>s;
	int len  =s.size();
	int sum =0;
	for (int i = 0; i <len; ++i) {
		sum+=(int)s[i]-96;
	}
	cout<<sum<<endl;
}
