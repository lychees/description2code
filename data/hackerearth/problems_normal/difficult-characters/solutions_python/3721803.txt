noOfStrings = int(raw_input())

def countDisplay(string):
	m = []
	for i in range(1, 27):
		m.append(0)
	
	for ch in string:
		m[ord(ch) - 97] += 1 
	ma = {}
	asciiCount = 97
	for c in m:
		if(ma.has_key(c) == False):
			ma[c] = [chr(asciiCount)]
		else:
			ma[c].append(chr(asciiCount))
		asciiCount	+= 1
	output = ""

	for k in sorted(ma.keys()):
		output += ' '.join(sorted(ma[k])[::-1]) + " "
	print output	
	

for c in range(1, noOfStrings + 1):
	countDisplay(raw_input())
