#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k;
    	cin>>n>>k;
    	long long int curr=0,ans=-1,temp;
    	for(int i=0;i<n;i++)
    	{
    		cin>>temp;
    		if(temp<=k) curr+=temp;
    		else
    		{
    			ans=max(ans,curr);
    			curr=0;
    		}
    	}
    	ans=max(ans,curr);
    	cout<<ans<<"\n";
    }
    return 0;
}
