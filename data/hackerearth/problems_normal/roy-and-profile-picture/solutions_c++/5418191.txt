#include <iostream>
using namespace std;

int main()
{
    int N,L,W,H;
    cin>>L;
    cin>>N;
    while(N--)
    {
    	cin>>W>>H;
    	if(W<L || H<L)
    	  cout<<"UPLOAD ANOTHER"<<endl;
    	else if(W>=L && H>=L && H==W) 
    	  cout<<"ACCEPTED"<<endl;
    	else
    	 cout<<"CROP IT"<<endl;
    }
    return 0;
}
