#include <iostream>
using namespace std;

int main()
{
    int n,q,type,a,b,i,ar[100],count;
    cin>>n;
    cin>>q;
    while(q--)
    {
    	cin>>type>>a>>b;
    	count=0;
    	if(type==0)
    	{
    		for(i=a;i<=b;i++)
    		{
    			ar[i]=-1*i;
    		}
    	}
    	if(type==1)
    	{
    		for(i=a;i<=b;i++)
    		{
    		   
    		   if(ar[i]<0)
    			{count=count+1;}
    			else
    			{count=count+0;}
    		}
    	cout<<count<<endl;
    	}
    }
    return 0;
}
