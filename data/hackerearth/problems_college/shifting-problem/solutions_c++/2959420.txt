#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int tc;
    cin>>tc;
    long long int d[1000];
    memset(d, 0, sizeof(d));
    d[0] = 1;
    d[1] = 1;
    for(int i=2; i<1000; i++)
        d[i] = d[i-1] + d[i-2];
    while(tc--)
    {
        int n;
        scanf("%d",&n);
        printf("%lld\n",d[n]);
    }
    return 0;
}
