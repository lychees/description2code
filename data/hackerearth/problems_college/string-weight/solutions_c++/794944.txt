#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
	int t,n;
	char s1[10001],s2[53];
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		int l1,l2,s=0;
		cin>>s1>>n;
		cin.ignore();
		cin.getline(s2,53);
		l1=strlen(s1);
		l2=strlen(s2);
		for(int j=0;j<l2;j++)
		{
			for(int k=0;k<l1;k++)
			{
				if(s2[j]==s1[k])
				{
					s1[k]='*';
				}
			}
		}
		for(int m=0;m<l1;m++)
		{
			if(s1[m]=='*')
			s+=0;
			else
			s+=(s1[m]-96);
		}
		cout<<s<<endl;
	}
	return 0;
}
