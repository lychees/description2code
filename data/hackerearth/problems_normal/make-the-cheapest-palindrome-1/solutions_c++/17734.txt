#include <iostream>
#include <stdlib.h>
#include <string>
#include <string.h>

using namespace std;

int main()
{
	int t,acost,bcost,len;
	int i,j,cost=0;
	cin>>t;
	for(i=0;i<t;i++){
		string x;
		getchar();
		getline(cin,x);
		//cout<<"hello";
		cin>>acost;
		cin>>bcost;
		//cout<<x[1];
		len=x.size();
		//cout<<len;
		//cout<<"i am before 1st for loop";

		//cout<<"i am before 2nd for loop";
		for(j=0;j<(len/2);j++){
			if(len%2!=0){
				cost=-1;
				break;
			}
			//cout<<"i am inside 2nd for loop";
			//cout<<x[j];
			//cout<<x[len-j-1];
			if(x[j]=='/'){
				//cout<<"inside 1st if";
				if(x[len-j-1]=='a')
					cost=cost+acost;
				else if(x[len-j-1]=='b')
					cost=cost+bcost;
				else if(x[len-j-1]=='/'){
					if(acost<bcost)
						cost=cost+acost+acost;
					else
						cost=cost+bcost+bcost;
				}				
			}
			else if(x[len-j-1]=='/'){
				//cout<<"inside else if";
				if(x[j]=='a')
					cost=cost+acost;
				else if(x[j]=='b')
					cost=cost+bcost;
			}
			else if(x[j]!=x[len-j-1]){
				cost=-1;
				break;
			}
		}
		cout<<cost<<"\n";
		cost=0;
	}
	return 0;
}
