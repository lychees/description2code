using namespace std;
typedef long long ll;
//#define DEBUG
#define mod 1000000007
#define pb push_back

int is_prime[10000001];
ll pow2[25];

void calc_pow2(){
    pow2[0] = 1;
    for(int i=1;i<=24;i++){
        pow2[i] = pow2[i-1] + pow2[i-1];
        if(pow2[i]>mod)
            pow2[i]-=mod;
    }
}

void markMultiples(int a){
    int i=2, num;
    while((num=i*a)<=10000000){
        i++;
        is_prime[num]++;
    }
}

void precalc(){
    ll num;
    for(int i=2;i<=10000000;i++){
        if(is_prime[i]==0){
            markMultiples(i);
        }
    }
}

int gcd(int a, int b){
    if(b==0)
        return a;
    return gcd(b, a%b);
}

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int T,a,b;
    precalc();
    calc_pow2();
    scanf("%d", &T);
    while(T--){
        scanf("%d %d", &a, &b);
        int ans = is_prime[gcd(pow2[a]-1,pow2[b]-1)];
        if(gcd(pow2[a]-1,pow2[b]-1)!=1 && ans==0)
            ans++;
        //cout<<gcd(pow2[a]-1,pow2[b]-1)<<endl;
        printf("%d\n", ans);
    }

    return 0;
}
