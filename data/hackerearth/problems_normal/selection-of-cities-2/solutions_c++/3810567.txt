#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long big_power(long long n,long long p);
int main()
{
    long long t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        long long n,ans;
        cin>>n;
        ans=big_power(2,n)-1;
        cout<<ans<<"\n";
    }
    return 0;
}
long long big_power(long long n,long long p)
{
    long long tmp;
    if(p==0)
        return 1;
    tmp=big_power(n,p/2)%mod;
    tmp=(tmp*tmp)%mod;
    if(p%2!=0)
        return (tmp*n)%mod;
    else
        return tmp;
}
