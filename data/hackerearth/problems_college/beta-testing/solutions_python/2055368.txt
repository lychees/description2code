from operator import itemgetter
t = input()
l = raw_input()
l = l.split(' ')
d = {}
t = 0
for i in l:
    t += 1
    if i in d:
        d[i][1] = t
        d[i][0] -= 1
    else:
        d[i] = [-1,t]
l = []
for x in d.keys():
    l.append((d[x][0],d[x][1],x));
l = sorted(l,key=itemgetter(0,1))
for el in l:
    print el[2],-el[0]
