#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int x;
    	cin>>x;
    	int count=0;
    	for(int j=0;j<=x;j++){
    		for(int k=0;k<=x;k++){
    			int c=x-5*j-3*k;
    			if(c>=0){
    				count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
