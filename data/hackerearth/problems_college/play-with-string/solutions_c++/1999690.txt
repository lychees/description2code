#include <iostream>
#include<cstring>
#include<cmath>
using namespace std;

int main() {
	int kase;
		char str[1000];
	cin>>kase;
	while(kase--)
	{
		int flag=0;
		cin>>str;
		int l=strlen(str);
		for(int i=1;i<l;i++)
		{
			
			if(abs(str[i]-str[i-1])!=abs(str[l-i-1]-str[l-1-i+1]))
			{
				flag=1;
				break;
			}	
		}
		if(flag==0)cout<<"Funny"<<endl;
		else cout<<"Not Funny"<<endl;
	}
	
	// your code goes here
	return 0;
}
