#include<bits/stdc++.h>
#define lli long long int
using namespace std;
pair<lli,lli>p[1000009];
lli hd[1000009];
int main()
{
//	ios_base::sync_with_stdio(false);
	int t; scanf("%d",&t);
	while(t--)
	{
		lli n,k,i,j,minx,inp,res;
		scanf("%lld %lld",&n,&k);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&inp);
			p[i]=make_pair(inp,i);
		}
		for(i=0;i<k;i++)scanf("%lld",&hd[i]);
		sort(p,p+n);
		res=0;
		for(i=0,j=0;i<n && j<k;j++)
		{
			minx=INT_MAX;
			while(i<n && p[i].first<hd[j])
			{
				minx=min(minx,p[i].second);
				i++;
			}
			res=minx;
		}
		if(i<n)
		{
			res=INT_MAX;
			while(i<n)
			{
				res=min(res,p[i].second);
				i++;
			}
		}
		printf("%lld\n",res);
	}
	return 0;
}
