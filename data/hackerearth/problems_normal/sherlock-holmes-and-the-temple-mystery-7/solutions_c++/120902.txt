#include <iostream>
using namespace std;
bool v[110][10000]={0};
int main()
{
 int t;
 scanf("%d",&t);
 for(int i=0;i<t;i++)
 {
 	int n;
 	scanf("%d",&n);
 	int a[n];
 	int s=0;
 	for(int j=0;j<n;j++)
 	{
 		scanf("%d",&a[j]);
 		s+=a[j];
 	}
 	if(s%2==0)
 	{
 		s/=2;
 		for(int j=0;j<=n;j++)
 		{
 			for(int k=0;k<=s;k++)
 			{
 				v[j][k]=0;	
 			}
 		}
 		v[0][0]=1;
 		for(int j=1;j<=n;j++)
 		{
 			for(int k=0;k<=s;k++)
 			{
 					if(v[j-1][k]==1)
 					{
 						v[j][k]=1;
 						if(k+a[j-1]<=s)
 						{
 							  v[j][k+a[j-1]]=1;
 						}
 					}
 			}
 		}
 		if(v[n][s]==1)
 		   printf("YES\n");
 		else printf("NO\n");
 	}
 	else  printf("NO\n");
 }
 
 
 
 return 0;
}
