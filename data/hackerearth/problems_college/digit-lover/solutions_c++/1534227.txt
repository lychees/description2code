#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<string.h>
#include<stdio.h>
#define MOD 1000000007
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,sum=0;
		cin>>n; 
		sum=n;
		while(sum>=10)
		{
			int total=0;
			while(sum!=0)
			{
				int a=sum%10;
				total+=a;
				sum/=10;
			}
			sum=total;
		}
		cout<<sum<<endl;
	}
return 0;
}
