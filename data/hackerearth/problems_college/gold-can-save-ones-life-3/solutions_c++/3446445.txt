#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)
 
using namespace std;
 
const int mod = 1000000000;
 
int main(){
	int n,t,k;
	cin >> t;
	vector<int> v(10000),dp(10000);
	while(t--){
		cin >> n >> k;
		FORi(i,0,n){
			cin >> v[i];
			dp[i]=0;
		}
		dp[0]=max(0,v[0]);
		FORi(i,1,n){
			if (i>k)
				dp[i] = dp[i-k-1];
			dp[i] = max(dp[i-1],dp[i]+v[i]);
		}
		cout << dp[n-1] << endl;
	}
    return 0;
}
