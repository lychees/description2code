#include <iostream>
using namespace std;
int count(long long x)
{
	int i ; 
	while(x)
	{
		if(x&1)
		i++ ; 
		x/=2;
	}
	return i ; 
}
int main()
{
    int t ; 
    scanf("%d",&t) ;
    while(t--)
    {
    	long long x,y,temp ; 
    	scanf("%lld %lld",&x,&y) ; 
    	temp = x^y ; 
    	int ans = count(temp) ; 
    	printf("%d\n",ans) ; 
    }
    return 0;
}
