#include <iostream>
#include <algorithm>
using namespace std;

void solve(){
	int x,y,n,d1=0,d2=0,s1=0,s2=0,k;
	cin>>x>>y>>n;
	int a[n];
	for(int i = 0; i<n; i++){
		cin>>a[i];
	}
	sort(a,a+n);
	int i = 0;
	while(s1+a[i]<=x){
		s1 += a[i];
		i++;
		d1++;
		if(i==n) break;
	}
	i = 0;
	while(s2+a[i]<=y){
		s2 += a[i];
		i++;		
		d2++;
		if(i==n) break;
	}
	//cout<<d1<<" "<<d2<<endl;
	if(d1>d2) cout<<"Chef Won"<<endl;
	else if(d1<d2) cout<<"Jhool Won"<<endl;
	else cout<<"Tie"<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--) solve();
    return 0;
}
