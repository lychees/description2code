#include <iostream>
using namespace std;

int main()
{
    int n,i,j,sum,min,flag,array[100][100];
    cin>>n;
    sum=flag=0;
    min=100;
    for(i=0;i<n;i++)
    {
    	cin>>array[i][0];
    	if(array[i][0]%2==0)
    		array[i][1]=0;
    	else
    		array[i][1]=1;
    	sum=sum+array[i][0];
    }
    if(sum%2!=0)
    	cout<<sum;
    else
    {
    	for(i=0;i<n;i++)
    	{
    		if(array[i][0]%2!=0)
    		{
    			if(min>array[i][0])
    			{	min=array[i][0];
    				flag=1;
    			}	
    		}
    		
    	}
    	if(flag==1)
    	{
    		sum=sum-min;
    		cout<<sum;
    	}
    	else
    		cout<<":(";
    }
    return 0;
}
