#include <bits/stdc++.h>

using namespace std;


int main()
{
	int a[100005];
	for(int i = 0; i < 100005; i++) {
		a[i] = 100000000;
	}

    a[2] = a[3] = a[5] = a[7] = 1;
    int x[4] = {2,3,5,7};
    for(int i = 4; i < 100005; i++) {
    	for(int j = 0; j < 4; j++) {
    		if(i - x[j] >= 2) {
    			a[i] = min(a[i], a[i - x[j]] + 1);
    		}
    	}
    }
    int t;
    cin>>t;
    while(t--) {
    	int n;
    	cin>>n;
    	if(a[n] == 100000000) {
    		cout<<-1<<endl;
    	} else {
    		cout<<a[n]<<endl;
    	}
    }
    return 0;
}
