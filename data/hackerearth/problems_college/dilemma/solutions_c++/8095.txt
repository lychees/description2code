void display(vector<int>& v){

    cout<<v[i]<<" ";

}

ull generateAns(int n, int a, int b, int c){
  vector<ull> a1(min(n, a)+1, 1);
  vector<ull> b1(min(n, b)+1, 1);
  vector<ull> c1(min(n, c)+1, 1);
  vector<ull> ans(n+1, 0);
  //display(a1); display(b1); display(c1);
  for(int i=0; i<a1.size(); i++){
    for(int j=0; j<b1.size(); j++){
    if ((i+j)<=n)
      ans[i+j] += a1[i]*b1[j];
    }
  }
  //cout<<"ans ";display(ans);
  vector<ull> ans1(n+1, 0);
  for(int j=0; j<ans.size(); j++){
    for(int k=0; k<c1.size(); k++){
      if (j+k<=n){
        ans1[j+k] += ans[j]*c1[k];
      }
    }
  }
  
  //cout<<"final ans";display(ans1);
  ull summ=0;
  for(int i=0; i<ans.size(); i++){
    summ+=ans1[i];
  }
  return summ;
}




int main(){
  int t, n, a, b, c;
  scanf("%d", &t);
  while(t--){
    scanf("%d %d %d %d", &n, &a, &b, &c);
    ull ans = generateAns(n, a, b, c);
    cout<<ans<<endl; 
  }

}
