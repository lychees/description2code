#include <iostream>
#include <string.h>
using namespace std;


void solve() {
	int n, q;
	int a[2][100005];
	memset(a, 0, sizeof a);
	cin >> n >> q;
	for(int i=0;i<n;i++) {
		int x1, x2, y1, y2;
		cin >> x1 >> y1 >> x2 >> y2;
		if(x1 > x2) {
			x1 = x1^x2;
			x2 = x1^x2;
			x1 = x1^x2;
		}
		if(y1 > y2) {
			y1 = y1^y2;
			y2 = y1^y2;
			y1 = y1^y2;
		}
		a[0][x1] += 1;
		a[0][x2+1] -=1;
		a[1][y1] += 1;
		a[1][y2+1] -=1;
	}
	for(int i=1;i<100005;i++) {
		a[0][i] = a[0][i-1] + a[0][i];
		a[1][i] = a[1][i-1] + a[1][i];
	}
	for(int i=0;i<q;i++) {
		int aa,bb;
		cin >> aa >> bb;
		if(aa==0) {
			cout << a[1][bb] << "\n";
		} else {
			cout << a[0][aa] << "\n";
		}
	}
}

int main()
{
	int t;
	cin >> t;
	while(t--) {
		solve();
	}
    return 0;
}
