#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, a;
    cin>>t;
    while(t--) {
    	a=0;
    	cin>>n;
    	while(n!=0) {
    		a+=(n%10);
    		n=n/10;
    	}
    	if((a==2)||(a==3)||(a==5)||(a==7)||(a==11)||(a==13)||(a==17)||(a==19)||(a==23)||(a==29)||(a==31)||(a==37)||(a==41)||(a==43)||(a==47)||(a==53))
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
