#include<iostream>
using namespace std;
int main()
{
        long int n;
        cin>>n;
        long int a[n],b[n];
        for(long int j=1;j<=n;j++)
        {
            cin>>a[j];
        }
        for(long int j=1;j<=n;j++)
        {
            if(j==1)
            {
             b[j]=a[j];
            }
            else
            {
                b[j]=a[j]*j;
                long long int d=b[j];
                for(long int k=1;k<j;k++)
                {
                    d=d-b[k];
                }
                b[j]=d;
            }
        }
        for(long int j=1;j<=n;j++)
            cout<<b[j]<<" ";
    return 0;
}
