#include <iostream>
using namespace std;

int main()
{
    int t,d,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>d;
    	i=1,j=0;
    	while(j<=d)
    	{
    	j+=i;
    	if(j<d)
    	{
    		i++;	
    	}
    
    	}
    	if(i%3==0)
    	{
    		cout<<"SL";
    	}
    	else if(i%3==1)
    	{
    		cout<<"LB";
    	}
    	else if(i%3==2)
    	{
    		cout<<"BS";
    	}
    	cout<<endl;
    }
    return 0;
}
