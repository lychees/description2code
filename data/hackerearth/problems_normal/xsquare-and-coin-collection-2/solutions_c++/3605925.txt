typedef long long ll;
#include <iostream>
#include <stdio.h>

using namespace std;
ll a[1000007];
int main()
{
    ll t,n,k,sum,maxi,i;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&k);
        for(i=0;i<n;i++)
        scanf("%lld",&a[i]);
        sum=0;
        maxi=0;
        for(i=0;i<n;i++)
        {
            if(a[i]>k)
            sum=0;
            else
            sum+=a[i];
            maxi=max(maxi,sum);
        }
        printf("%lld\n",maxi);
    }
	return 0;
}
