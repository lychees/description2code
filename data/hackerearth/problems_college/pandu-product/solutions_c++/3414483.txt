#include <iostream>
#include <fstream>
#include <cmath>
#include <algorithm>

#define ll long long

using namespace std;

ll n , Pro = 1; 
ll a[1110111];
int main()
{
	cin >> n;                     
	for (int i = 1;i <= n;i ++)
		cin >> a[i];
	sort (a + 1 , a + n + 1);
	ll mx = -(ll)1e18;
	for (int i = 1;i < n;i ++)
		mx = max(mx , a[i] * a[i + 1]);
	cout << mx;	
	return 0;
}
