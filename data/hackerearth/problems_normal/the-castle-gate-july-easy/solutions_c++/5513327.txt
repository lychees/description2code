#include <iostream>
using namespace std;
void count(int n)
{
	int i=1, j=1, cnt=0;
	while (j<=n)
	{
		i=1;
		while (i<j)
		{
			if ((i^j)<=n)
			{
				cnt++;
			}
			i++;
		}
		j++;
	}
	cout<<cnt<<endl;
}
int main()
{
    int m;
    cin>>m;
    int s[m];
    for (int p=0; p<m; p++)
    {
    	cin>>s[p];
    }
    for (int p=0; p<m; p++)
    {
    	count(s[p]);
    }
    return 0;
}
