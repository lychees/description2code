#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	long int l1=a.length()-1;
    	long int c=0;
    	while(l1>=0)
    	{
    		int g1=(a[l1] - '0');
    		int g2=(b[l1] - '0');
    		if((g1+g2)>9) c++;
    		l1--;
    	}
    	if(c==0)
    		cout<<"No carry operation"<<endl;
    	else if(c==1)
    		cout<<c<<" "<<"carry operation"<<endl;
    	else
    		cout<<c<<" "<<"carry operations"<<endl;
    }
    return 0;
}
