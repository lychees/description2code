#include <iostream>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int a[n][k];int tar[n][k];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<k;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    int q;
    cin>>q;
    while(q>0)
    {
    	int ns;
    	cin>>ns;
    	int b[k];
    	for(int l=0;l<ns;l++)
    	{
    		cin>>b[l];
    	}
    int i;int cnt;int f;
    	 for(int i=0;i<n;i++)
    {
    	for(int j=0;j<k;j++)
    	{
    		tar[i][j]=a[i][j];
    	}
    }
   int  m;
     for(int u=0;u<ns;u++)
     {
     	 cnt=0;m=101;f=0;i=0;
     	for(int v=0;v<n;v++)
	     {
	     		if(tar[v][b[u]-1]<m)
	     		{
	     			m=tar[v][b[u]-1];i=v;
	     		}
	     	/*		for(int d=0;d<k;d++ )
	     			{   if(v>0)
	     				tar[v-1][d]=101;
	     			}
	     		}
	     		else if(tar[v][b[u]-1]>m)
	     		{
	     			tar[v][b[u]-1]=101;
	     		}
	     		*/
     	 }
     	 int tmp=tar[i][b[u]-1];
     	 for(int x=0;x<n;x++)
     	 {
     	 	if(tmp==tar[x][b[u]-1])
     	 	cnt++;
     	 	else
     	 	{
     	 		for(int d=0;d<k;d++ )
	     		{   
	     			tar[x][d]=101;
	     		}
     	 		
     	 	}
     	 }
     	if(--cnt==0)
     	{
     		cout<<i+1<<endl;f=1;break;
     	}
     	
     }
    
    if(f==0)
    cout<<i+1<<endl;
     q--;
    }
    return 0;
}
