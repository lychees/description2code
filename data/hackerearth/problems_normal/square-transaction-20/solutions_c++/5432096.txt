#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,i,j,k,q,x;
   cin>>t;
   int arr[t];
   for(i=0;i<t;i++)
   	cin>>arr[i];
   cin>>q;
   while(q--){
   	cin>>x;
   	int flag=0;
   	for(i=0;i<t;i++){
   		if(x-arr[i]<=0){
   			flag=1;
   			break;
   		}
   		x=x-arr[i];
   	}
   	if(flag==1)
   		cout<<i+1<<endl;
   	else
   		cout<<"-1\n";
   }
    return 0;
}
