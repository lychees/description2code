#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	char arr[500] = {0};
    	cin>>arr;
    	int l = strlen(arr);
    	int sum = 0;
    	for(int i =0;i<l;i++)
    	{
    		if(arr[i]!='_')
    		sum = (sum +(arr[i]-'0'))%9;
    	}
    	if(sum == 0)
    	printf("0 or 9\n");
    	else
    	printf("%d\n",9-sum);
    }
    return 0;
}
