#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>
#include <unordered_map>
#include <stack>
using namespace std;

int main()
{
	long long int i,val,n,j,k,count = 0,m,t,hours;
	bool A[n];
	bool B[n];
	cin>> m ;
	for(k=0;k<m;k++)
	{
		cin >> n >> hours;
		for(i=0;i<n;i++)
			cin >> A[i];
		int flag = 1;
		for(t = 0; t<hours ; t++)
		{			
			if(flag == 1)
			{
				for(i=0;i<n;i++)
				{
					if(i == 0 && A[i+1] == 1)
						B[i] = 1;
					else if(i == n-1 && A[i-1] == 1)
						B[i] = 1;
					else if(A[i-1] == 1 && A[i+1] == 1)
						B[i] = 1;
					else
						B[i] = 0;					
				}
				flag = 0;
			}
			else
			{
				for(i=0;i<n;i++)
				{
					if(i == 0 && B[i+1] == 1)
						A[i] = 1;
					else if(i == n-1 && B[i-1] == 1)
						A[i] = 1;
					else if(B[i-1] == 1 && B[i+1] == 1)
						A[i] = 1;
					else
						A[i] = 0;						
				}
				flag = 1;
			}		
			for(i=0;i<n;i++)
			{
				if(A[i] != B[i])
					break;
			}
			if(i==n)
			{
				break;
			}
		}
		if(flag == 1)
		{
			for(i=0;i<n;i++)
				cout << A[i] << " ";
			cout << endl;
		}
		else
		{
			for(i=0;i<n;i++)
				cout << B[i] << " ";
			cout << endl;	
		}
	}
	return 0;
}
