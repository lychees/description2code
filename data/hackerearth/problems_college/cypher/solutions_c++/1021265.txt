#include <iostream>
#define ll long long

using namespace std;

int main()
{
	//cout << (int)'a' << endl;
	//cout << (int)'z' << endl;
	ll n = 0, chu = 0;
	string s = "";
	cin >> n;
	while (n--){
		cin >> s;
		for (ll i = 0; i < s.length(); i++){
			chu = (int)s[i]+i;
			if (chu > 122)
				cout << char( (chu % 122)+96 );
			else
				cout << char( chu );
		}
		cout << endl;
	}
	
	return 0;
}
