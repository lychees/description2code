#include <bits/stdc++.h>
using namespace std;
#define lli long long int
map <char,lli> mp;
map <char,lli> :: iterator it;
long int m=1e9+7;
lli f[1000005],df[1000005];
lli mexp(lli a,lli b)
{
	if(b==0)
	{
		return 1;
	}
	lli t=mexp(a,b/2);
	if(b%2==0)
	{
		return ((t%m)*(t%m))%m;
	}
	else
	{
		return ((t%m)*(t%m)%m*(a%m))%m;
	}
}
void pre()
{
	lli i;
	f[0]=1;
	f[1]=1;
	df[0]=1;df[1]=1;
	for(i=2;i<=1e6;i++)
	{
		f[i]=((f[i-1]%m)*(i%m))%m;
	//	df[i]=mexp(f[i],m-2);
	}
}
int main()
{
    int t;lli i,l,x,ans,w,q;char ch;
    string s;
    pre();
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	l=s.size();
    	for(i=0;i<l;i++)
    	{
    		mp[s[i]]++;
    	}
    	x=mp['a']+mp['e']+mp['u']+mp['i']+mp['o'];
    	if((l-2*x+1)>=0&&x!=0)
    	{
    		w=1;q=1;ans=1;
    		for(i=0;i<26;i++)
    		{
    			ch=i+'a';
    			if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u')
    			{
    				w=((w%m)*(mexp(f[mp[ch]],m-2)%m))%m;
    			}
    			else
    			{
    				q=((q%m)*(mexp(f[mp[ch]],m-2)%m))%m;
    			}
    		}
    		ans=((ans%m)*(f[l-x]%m))%m;
    		ans=((ans%m)*(f[l-x+1]%m))%m;
    		ans=((ans%m)*(mexp(f[l-2*x+1],m-2)%m))%m;
    		ans=((ans%m)*(q%m))%m;
    		ans=((ans%m)*(w%m))%m;
    	}
    	else if(x==0)
    	{
    		ans=f[l];w=1;q=1;
    		for(i=0;i<26;i++)
    		{
    			ch=i+'a';
    			if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u')
    			{
    				w=((w%m)*(mexp(f[mp[ch]],m-2)%m))%m;
    			}
    			else
    			{
    				q=((q%m)*(mexp(f[mp[ch]],m-2)%m))%m;
    			}
    		}
    		ans=((ans%m)*(q%m))%m;
    	}
    	else
    	{
    		ans=-1;
    	}
    	cout<<ans<<endl;
    	mp.clear();
    }
    return 0;
}
