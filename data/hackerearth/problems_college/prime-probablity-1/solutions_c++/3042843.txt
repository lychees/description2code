#include<iostream>
#include<cmath>
long int a[1000001];
using namespace std;
bool prime(int n)
{
	if(n==1)
	return false;
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)return false;
	}
	return true;
}
int solve()
{
	a[0]=0;
	for(int i=1;i<=1000000;i++)
	{
		
		if(prime(i)){a[i]++;}
		a[i]=a[i]+a[i-1];
	}
}
int main()
{
	solve();
	int t;
	cin>>t;
	while(t--)
	{
		int l,r,cnt=0;
		cin>>l>>r;
		float f;
		if(l==r)
		{
			if(prime(l))
			f=1;
			else
			f=0;
		}
		else
		 f=float(a[r]-a[l-1])/float((r-l+1));
		printf("%.6f\n",f);
	}
}
