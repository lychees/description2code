#include <iostream>
using namespace std;

int main()
{
	
    int t,a[100],i;
    cin>>t;
    if(t>100)
    return 0;
    for(i=0;i<t;i++)
    cin>>a[i];
    for(i=0;i<t;i++)
    {
    	
    	 
    	 int rev[10],new1,j=0,l=0,no=0;
    	new1=a[i];
    	while(new1!=0)
    	{
    		rev[j]=new1%10;
    		new1=new1/10;
    		j++;
    		l++;
    		
    		
    	}
    	for(j=0;j<l;j++)
    	{
    		
    		if(rev[j]==1&&rev[j+1]==2){
    		no=-1;
    		break;
    		}
    	}
    	
    	if(a[i]%21==0){
    	cout<<"\nThe streak is broken!";
    	continue;
    	}
    	else if(no==-1)
    	
    
    	{
    		cout<<"\nThe streak is broken!";
    		
    	}
    	else if(no!=-1)
    	cout<<"\nThe streak lives still in our heart!";
    }
    	
    
    return 0;
}
