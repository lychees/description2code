#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i,j,k,l;
	cin>>t;
	while(t--)
	{
		k=0;
		j=0;
     string s,s1;
     cin>>s>>s1>>n;
     l=s.length();
     for(i=0;i<l;i++)
     {
     	if(s[i]!=s1[i])
     	{
     		k++;
     	}
     }
     if(k>n)
     cout<<"No"<<endl;
     else if((n-k)%2!=0)
     {
     	cout<<"No"<<endl;
     }
     else
     cout<<"Yes"<<endl;
	}
    
}
