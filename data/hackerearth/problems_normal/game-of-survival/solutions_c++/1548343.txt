#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	int a[n],b[n];
    	for(int i=0;i<n;i++) scanf("%d",&a[i]);
    	for(int i=0;i<n;i++) scanf("%d",&b[i]);
    	sort(a,a+n);
    	sort(b,b+n);
    	int flag=0;
    	for(int i=0;i<n;i++)
    	{
    		if(b[i]>a[i])
    			{
    				printf("NO\n");
    				flag=1;break;
    				
    			}
    		
    	}
    	if(flag==0)
    	printf("YES\n");
    	
    }
    return 0;
}
