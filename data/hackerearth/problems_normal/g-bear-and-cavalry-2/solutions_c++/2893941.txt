#include <bits/stdc++.h>
using namespace std;
bool cmp(int a, int b){return a>b;}
int main()
{
    int H[43],W[43];
    int T=0,N,i,j,k,ans;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d",&N);
    	for(i=0;i<N;i++)
    	scanf("%d",&H[i]);
    	for(i=0;i<N;i++)
    	scanf("%d",&W[i]);
    	sort(H+1,H+N);
    	sort(W,W+N,cmp);
    	for(i=0;i<N;i++)
    	H[i]*=W[i];
    	ans=H[0];
    	bool flag=true;
    	for(i=1;i<N;i++)
    	{
    		if(H[i]>=ans)
    		{
    			flag=false;
    			break;
    		}
    	}
    	if(flag==true)
    	printf("YES\n");
    	else
    	printf("NO\n");
    	
    	}
    return 0;
}
