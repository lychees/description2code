#include <bits/stdc++.h>
using namespace std;

int prime[1000004];
int main()
{
	prime[1]=0;
    for(int i=2;i<=1000000;i++){
        if(prime[i]==0){
           prime[i]=i;
           for(int j=i+i;j<=1000000;j+=i){
            prime[j]+=i;
           }
        }
    }
	int n,k,cnt=0;
	cin>>n>>k;
	for(int i=1;i<=n;i++){	
		
		if(prime[i]==k)
            cnt++;
	}
    cout<<cnt<<endl;
    return 0;
}
