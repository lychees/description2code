ll hcf(ll a,ll b){
	int rem;
	while(b!=0){
		rem=a%b;
		a=b;
		b=rem;
	}
	return a;
}
int main(){
	int t=scan();
	while(t--){
		ll ar[1000002];
		int n=scan();
		ll ans;
		for(int i=0;i<n;i++){
			scanf("%lld",&ar[i]);
			if(i==0)continue;
			if(i==1){
				ans=hcf(ar[i],ar[i-1]);
				continue;
			}
			ans=hcf(ans,ar[i]);
		}
		ans=ans*n;
		printf("%lld\n",ans);
	}
	return 0;
}
