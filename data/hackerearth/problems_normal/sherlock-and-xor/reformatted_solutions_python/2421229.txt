t = int(raw_input())
while t > 0:
    try:
        n = int(raw_input())
        pm = 0
        nm = 0
        a = map(int, raw_input().split(" "))
        for i in range(0, len(a)):
            if a[i] % 2 == 0:
                pm = pm+1
            else:
                nm = nm+1
        print pm*nm
    except (EOFError):
        break
    t = t-1
