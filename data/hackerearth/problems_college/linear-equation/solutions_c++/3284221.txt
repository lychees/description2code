#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;

        vector<int> A(n);
        for(int i=0; i<n; i++)
        {
            cin>>A[i];
        }

        int supper = accumulate(A.begin(), A.end(),0) / (n-1);

        for(int i=0; i<n; i++)
        {

            cout<<supper-A[i]<<" ";
        }

        cout<<endl;
    }
}
