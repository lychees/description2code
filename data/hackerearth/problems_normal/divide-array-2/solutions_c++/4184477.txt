#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll n;
    cin >> n;
    ll max = 0;
    ll i;
    ll p = 1;
    ll a[n];
    for(i = 0; i < n; i++) {
    	cin >> a[i];
    	if(a[i] > max) {
    		max = a[i];
    	}
    }
    ll q;
    cin >> q;
    ll val;
    while(q--) {
    	cin >> val;
    	if(p <= max) {
    		p = p * val;
    	}
    }
    for(i = 0; i < n; i++) {
    	cout << a[i]/p << " ";
    }
    cout << endl;
    return 0;
}
