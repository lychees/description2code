'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
number_of_iterations = int(raw_input())

for i in range(0, number_of_iterations):
    message = raw_input()
    length = len(message)
    tot = 0
    last = -1
    for j in range(length):
        if (message[j] == 'a' or message[j] == 'z'):
            tot += (length - j) * (j - last)
            last = j

    print tot
