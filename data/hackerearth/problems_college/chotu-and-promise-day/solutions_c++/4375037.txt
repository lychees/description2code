#include <bits/stdc++.h>
using namespace std;
int phi[1000005],a[1000005];
void pre()
{
	int i,j;
	phi[1]=1;
	for(i=2;i<=1000000;i++)
	{
		phi[i]=i;
	}
	for(i=2;i<=1000000;i++)
	{
		if(!a[i])
		{
			phi[i]=i-1;
			for(j=i*2;j<=1000000;j+=i)
			{
				a[j]=1;
				phi[j]/=i;
				phi[j]*=(i-1);
			}
		}
	}
}
int main()
{
	pre();
	int t,n,d;
	cin>>t;
	while(t--)
	{
		cin>>n>>d;
		if(n%d!=0)
		{
			cout<<"0"<<endl;
		}
		else
		{
			cout<<phi[n/d]<<endl;
		}
	}
}
