#include <iostream>
#include <cmath>
using namespace std;
#define MOD 1000000007

bool isPrime(unsigned int n)
{
	if(n==2) return true;
	if(n <= 1 || n % 2 == 0)
		return false;
	unsigned long long int i;
	for(i = 3; i*i <= n; i+=2)
	{
		if(n % i == 0)
			return false;
	}
	return true;
}

int main()
{
    unsigned int i,j,n;
    cin>>n;
    unsigned long long int count = 0;
    if(n==1) cout<<"0\n";
    else if(n==2) cout<<"1\n";
    else {
    	bool isPrime[2000009]={false};
		unsigned int temp=sqrt(2*n-2);
    	for(i=2;i<=temp;i++)
    	{
    		if(!isPrime[i])
    		{
    			for(j=i*i;j<=2*n-2;j+=i)
    				isPrime[j]=true;
    		}
    	}
    	count = (count+3)%MOD; //for 2
    	for(i=3;i<n;i+=2){
    		if(!isPrime[i])
    			count = (count+i+1)%MOD;
    	}
    	i = (n%2)?n:(n+1);
    	j = (n%2)?1:3;
    	for(;i<=(2*n-2);j+=4,i+=2){
    		if(!isPrime[i])
    			count = (count+i-j)%MOD;
    	}
    	cout<<count<<endl;
    }
    
    
    
    
    return 0;
}
