#include<bits/stdc++.h>
using namespace std;
#define  mp make_pair
#define  pb push_back
#define  M 1000000007
typedef long long int ll;
int n;
int a[5002];
int dp[5001][5001];
ll func(int i, int last)
{
	//printf("%d %d\n",i,last);
	if(i>=n)
		return 0;
	if(last==-1)
	{
		return max(1+func(i+1,i),func(i+1,last));
	}
	else
	{
		int &ans=dp[i][last];
		if(ans!=-1)
			return ans;
		if(a[last]<0)
		{
			if(a[i]>0)
			{
				if(abs(a[i])>abs(a[last]))
				return ans=max(1+func(i+1,i),func(i+1,last));
				else
					return ans=func(i+1,last);
			}	
			else
			{
				return ans=func(i+1,last);
			}		
		}
		else
		{
			if(a[i]<0)
			{
				if(abs(a[i])>abs(a[last]))
				return ans=max(1+func(i+1,i),func(i+1,last));
				else
					return ans=func(i+1,last);
			}	
			else
			{
				return ans=func(i+1,last);
			}	
		}
	}
}
int main()
{
	int i,j;
	for(i=0;i<=5000;i++)
		for(j=0;j<=5000;j++)
			dp[i][j]=-1;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	ll ans=func(0,-1);
	cout<<ans<<endl;
	return 0;
}
