T = input()
for t in range(0, T):
    B, W = map(int, raw_input().split())
    X, Y, Z = map(int, raw_input().split())
    total = 0
    if (X < Y):
        total = total + X*B
        if (X+Z < Y):
            total = total + (X+Z)*W
        else:
            total = total + Y*W
    else:
        total = total + Y*W
        if (Y+Z < X):
            total = total + (Y+Z)*B
        else:
            total = total + X*B
    print total
