#include <iostream>
using namespace std;

int main(){
	int t;
	long long int n,i,c,p,j,max;
	cin>>t;
	while(t>0){
		cin>>n;
		c = n;
		c = c-2;
		i=1;
		 p = 2;
		while(c>=0){
			i++;
			p = p*2;
			c = c - (p*i);
		}

		c = -c;
		max = 1;
		if(c !=0){
			j = (p*i)-c;
			c = j<=(i+i-1) ? j  : i+i-1;
			max = c;
			i = i-1;
		}
		
		max = max > i+i-1 ? max : i+i-1;

		cout<<max<<endl;
		t--;
	}

	return 0;
}
