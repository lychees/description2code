#include <stdio.h>

int a[1005][25];
int care[1005],cite;
int lista[1005],nr;

int main()
{
    int n,k;
    scanf("%d %d",&n,&k);

    for (int i=1;i<=n;++i)
    {
        for (int j=1;j<=k;++j)
        {
            scanf("%d",&a[i][j]);
        }
    }

    int q;
    scanf("%d",&q);

    for (int i=1;i<=q;++i)
    {
        int x;
        scanf("%d",&x);

        cite=n;
        for (int i=1;i<=n;++i) care[i]=i;

        for (int j=1;j<=x;++j)
        {
            int col;
            scanf("%d",&col);

            int minim=1000; nr=0;

            for (int z=1;z<=cite;++z)
            {
                if (minim>a[care[z]][col])
                {
                    minim=a[care[z]][col];
                    nr=1;
                    lista[nr]=care[z];
                }else
                if (minim==a[care[z]][col])
                {
                    ++nr;
                    lista[nr]=care[z];
                }
            }

            cite=nr;
            for (int z=1;z<=nr;++z)
                care[z]=lista[z];

        }

        printf("%d\n",care[1]);


    }


    return 0;
}
