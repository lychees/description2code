#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map<string,int> mp;
map<string,int> :: iterator it;
int main()
{ 
	long long max,sum,t;
	string a;
	cin>>t;
    while(t--){
    	cin>>a;
    	mp[a]++;
    }
    max=0;
    for(it=mp.begin();it!=mp.end();it++){
    	if(max<it->second)
    	max=it->second;
    }
    sum=0;
    for(it=mp.begin();it!=mp.end();it++){
    	if(it->second==max){
    		sum=sum+max;
    	}
    }
    cout<<sum;
    return 0;
}
