//Seven-segment display
#include<iostream>
using namespace std;
int main()
{
	char input[100];
	char x;
	int dash = 0;
	cin>>input;
	int i=0;
	while(input[i]!='\0')
	{
		x=input[i];
		switch (x)
		{
			case '0' : dash+=6;break;
			case '1' : dash+=2;break;
			case '2' : dash+=5;break;
			case '3' : dash+=5;break;
			case '4' : dash+=4;break;
			case '5' : dash+=5;break;
			case '6' : dash+=6;break;
			case '7' : dash+=3;break;
			case '8' : dash+=7;break;
			case '9' : dash+=6;break;
			default : break;
		}
	    i++;
	}
	
	cout<<dash;
	return 0;	
	
	
}
