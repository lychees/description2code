#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	int n;
	int i, j=0, max = 0;
	scanf("%d",&n);
	int arr[n];
	for(i=0;i<n;i++) scanf("%d",&arr[i]);
	for(i=1;i<n;i++){
		if(arr[i]-arr[i-1] ==1 || arr[i]-arr[i-1]== -1){ j++;}
		else {
			if(j>max) max=j;
			j=0;
		}
	}
    if(j>max) max=j;
    printf("%d",max+1);
    return 0;
}
