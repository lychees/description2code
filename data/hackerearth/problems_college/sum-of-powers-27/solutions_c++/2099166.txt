#include <stdio.h>
 
int a[1000001];
int main()
{
	long long int i,j;
    int t,n,f;
    a[1]=1;
    for(i=2;i<1000001;i++)
     {
     	if(a[i]!=0)
     	continue;
     	for(j=i*i;j<1000001;j*=i)
     	a[j]=1;
     }
    scanf("%d",&t);
     while(t--)
       {
       	scanf("%d",&n);
       	f=1;
        for(int i=1;i<n;i++)
         {
         	if(a[n-i] && a[i])
         	 {
         	 	f=0;
         	 	printf("Yes\n");
         	 	break;
         	 }
         }
            if(f)
            printf("No\n"); 
       }
       return 0;
}
