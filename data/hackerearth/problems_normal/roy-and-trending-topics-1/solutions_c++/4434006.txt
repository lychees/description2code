#include<bits/stdc++.h>
using namespace std;
#define fore(i,a,b) for(i=a;i<=b;i++)
#define forl(i,a,b) for(i=a;i<b;i++)
#define si(d) scanf("%d",&d)
//#define ss(s) scanf("%s",s)
#define sll(d) scanf("%lld",&d)
#define pin(d) printf("%d\n",d)
#define pln(d) pritf("%lld\n",d)
#define ll long long
#define ull unsigned long long
int main()
{
	int n;
	si(n);
	vector<pair<ll,pair< ll,ll> > > v;
	for (int i=0;i<n;i++)
	{
	ll id,z,p,l,c,s;
	sll(id),sll(z),sll(p),sll(l),sll(c),sll(s);
	ll first=p*50+l*5+c*10+s*20;
	ll new_score=first;
	first-=z;
	v.push_back(make_pair(first,make_pair(id,new_score)));	
	}
		sort(v.begin(),v.begin()+n);
		vector< pair<ll,pair<ll,ll> > > :: iterator it;
		int cnt=0;
		
		for (it=v.begin()+n-1;cnt<5;it--)
		{
			printf("%lld %lld\n",it->second.first,it->second.second);
			cnt++;
		}
	return 0;
}
