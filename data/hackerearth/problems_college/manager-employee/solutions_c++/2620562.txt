#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--) {
		int n;
		cin>>n;
		int a[n+1];
		for(int i = 0; i < n; i++) {
			cin>>a[i];
		}
		sort(a, a + n);
		
		vector<int> ans; 
		for(int i = 0; i < n/2; i++) {
			ans.push_back(a[i] + a[n - i - 1]);
		}

		sort(ans.begin(), ans.end());
		cout<<abs(ans[ans.size() - 1] -  ans[0])<<"\n";
	}
	return 0;
}
