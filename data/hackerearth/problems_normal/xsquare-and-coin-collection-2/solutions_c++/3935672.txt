#include<bits/stdc++.h>
using namespace std;
long long arr[100005];
int main()
{
    long long int t,n,k,i,sum,ans;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=0;i<n;i++)
            cin>>arr[i];
        sum=ans=0;
        for(i=0;i<n;i++)
        {
            if(arr[i]<=k)
                sum+=arr[i];
            else
            {
                ans=max(ans,sum);
                sum=0;
            }
        }
        ans=max(ans,sum);
        cout<<ans<<endl;
    }
    return 0;
}
