#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
  long long int n,temp,ct=0;
  cin>>n;
  long long int a[n];
  long long int b[n];
  long long int i=0,j=0;
  for(i=0;i<n;i++)
  {

      cin>>a[i];
  }
  for(i=0;i<n;i++)
  {

      cin>>b[i];
  }
  j=n-1;
  temp=a[0];
  while(j>0)
  {
    int low=0;
    int high=j-1;
    while(low<=high)
    {
        int mid=((low+high)/2);
        if(b[j]-a[mid]>=0)
        {
          if(ct<(j-mid))
                ct=j-mid;
          high=mid-1;
        }
        else
        {
        	low=mid+1;
        }

    }
    j--;






  }





    	cout<<ct<<endl;



    	t--;

    }
    return 0;
}
