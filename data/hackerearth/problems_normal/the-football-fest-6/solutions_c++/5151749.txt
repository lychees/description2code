#include <iostream>
using namespace std;
void push(int a[],int data);
void pop(int a[]);
int top1=-1;
int top2=-2;
int main()
{
    int t,n,p,q;
    char c[0];
    cin>>t;
    while(t--)
    {
    	 top1=-1;
         top2=-2;
    	cin>>n;
    	int a[n+1];
    	cin>>p;
    	push(a,p);
    	while(n--)
    	{
    		cin>>c[0];
    		if(c[0]=='P')
    		{
    			cin>>q;
    			push(a,q);
    		}
    		else
    		{
    			pop(a);
    		}
    		
    	}
    	cout<<"Player"<<" "<<a[top1]<<endl;
    }
    return 0;
}
void push(int a[],int data)
{
	top1=top1+1;
	a[top1]=data;
	top2=top2+1;
	//	cout<<"Player"<<" "<<a[top1];
}
void pop(int a[])
{
	top1=top1+1;
	a[top1]=a[top2];
	top2=top2+1;
	//	cout<<"Player"<<" "<<a[top1];
}
