

n = int(raw_input())
times = raw_input()
times = times.split()
times = sorted([int(time) for time in times])
solved = 0
minutes = 10
penalty = 0
for i in range(n):
    minutes += times[i]
    if minutes > 720:
        break
    solved += 1
    if minutes > 360:
        penalty += minutes - 360
print str(solved) + ' ' + str(penalty)
