#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
typedef long long ll;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		ll n;
		ll MOD;
		scanf("%lld%lld",&n,&MOD);
		ll t=n/MOD;
		ll sum=0;
		sum+=((MOD*(MOD-1))/2)*t;
		t=n%MOD;
		sum+=(t*(t+1))/2;
		printf("%lld\n",sum);
	}
}
