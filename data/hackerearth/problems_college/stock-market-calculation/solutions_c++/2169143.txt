#include <bits/stdc++.h>
#define ll long long int
#define si(a) scanf("%d", &a)
#define sl(a) scanf("%lld", &a)
#define sc(a) scanf("%c", &a)
#define pb push_back
#define mp make_pair
#define SET(x, a) memset(x, a, sizeof(x))

using namespace std;

ll a[100001];
ll ind[100001];

int main()
{
	ll t,i,j,k,l,w,ww,x,y,z,ans,tot,m,n;
	sl(t);
	while(t--) {
		ans=0;
		sl(n);
		for(i=0;i<n;i++) sl(a[i]);
		ind[n-1]=n-1;
		for(i=n-2;i>=0;i--) {
			if(a[ind[i+1]]>a[i]) ind[i]=ind[i+1];
			else ind[i]=i;
		}
		for(i=0;i<n;i++) {
			if(ind[i]==i) continue;
			ans=ans+a[ind[i]]-a[i];
		}
		cout<<ans<<endl;
	}
	return 0;
}
