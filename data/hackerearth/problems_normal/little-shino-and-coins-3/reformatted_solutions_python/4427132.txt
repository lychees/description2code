if __name__ == "__main__":
    k = int(raw_input(''))
    s = raw_input('')
    count = 0
    for i in range(0, len(s)):
        elements = set(s[i:i+k])
        if (len(elements) == k):
            count += 1
        for j in range(i + k, len(s)):
            elements.add(s[j])
            if (len(elements) == k):
                count += 1
    print count
