#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s;
    cin.ignore();
    while(t--)
    {
    	getline(cin,s);
    	int n=s.length();
    	bool flagx=0;
    	bool flagy=0;
    	int i=0;
    	int c1=0;
   		while(i<n&&s[i]=='Y')
    	{c1++;i++;}
    	while(i<n)
   		{
    		flagx=1;
    		i++;
    		if(s[i-1]=='X')
    		{	
    			while(i<n&s[i]=='X')
    			{c1++;i++;}    			
    		}
   			else
   			{
   				while(i<n&&s[i]=='Y')
   				{c1++;i++;}
    		}	
    	}
    	i=0;
    	int c2=0;
    	while(i<n&&s[i]=='X')
    	{c2++;i++;}
    	while(i<n)
    	{
    		flagy=1;
    		i++;
    		if(s[i-1]=='X')
    		{
   				while(i<n&&s[i]=='X')
   				{c2++;i++;}
   			}
   			else
   			{
    			while(i<n&&s[i]=='Y')
    			{c2++;i++;}
   			}	
    	}
    	if(flagy==1&&flagx==0)
    	cout<<c2<<"\n";
    	else if(flagx==1&&flagy==0)
    	cout<<c1<<"\n";
    	else cout<<min(c1,c2)<<"\n";
    	
    }
}
