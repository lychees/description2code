#include<bits/stdc++.h>
using namespace std;

int main(){

    int t,tt,i,len,j=0,k;
    vector<string> arr;
    vector<string> a;
    map<string,int> m;
    string str;

    cin>>t;
    t = t+1;

    while(t--){
        getline(cin,str);
        arr.push_back(str);
    }

    tt = arr.size();
    
    for(i=0;i<tt;i++){
    
        str = "";
        len = arr[i].length();

        j=0;
        while(j<len){
            if(arr[i][j] !=' '){
                str+=arr[i][j];

            }
            else{
                if(m[str] != 1){
                     a.push_back(str);
                     m[str] = 1;
                }
                str = "";
            }
            j++;
        }
        if(m[str] != 1)
        {
             a.push_back(str);
             m[str] = 1;
        }

        for(k=0;k<a.size();k++){
            cout<<a[k]<<" ";
        }
        cout<<"\n";
        a.clear();
        m.erase(m.begin(),m.end());
    }

    return 0;
}
