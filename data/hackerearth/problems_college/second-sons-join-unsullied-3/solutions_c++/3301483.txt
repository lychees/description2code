#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<limits.h>
#include<vector>
using namespace std;
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
           string A;
           cin>>A;
           int a=0,b=0,cnt=0,cnt1=0;
           for(int i=0;i<A.length();i++)
           {
                   if(A[i]=='U')
                   a++;
                   else
                   b++;
           }
           if(abs(a-b)>1)
           cout<<"-1"<<endl;
           else
           {
              if(a>b)
              {
              	for(int i=0;i<A.length();i++)
              	{
              		if(i%2==0&&A[i]!='U')
              		cnt++;
              		else if(i%2!=0&&A[i]!='S')
              		cnt++;
              		
              	}
              	cout<<cnt/2<<endl;
              }
              else if(b>a)
              {
              		for(int i=0;i<A.length();i++)
              	{
              		if(i%2==0&&A[i]!='S')
              		cnt++;
              		else if(i%2!=0&&A[i]!='U')
              		cnt++;
              		
              	}
              	cout<<cnt/2<<endl;
              }
              else
              {
              	
              	for(int i=0;i<A.length();i++)
              	{
              		if(i%2==0&&A[i]!='S')
              		cnt++;
              		else if(i%2!=0&&A[i]!='U')
              		cnt++;
              		
              	}
              		for(int i=0;i<A.length();i++)
              	{
              		if(i%2==0&&A[i]!='U')
              		cnt1++;
              		else if(i%2!=0&&A[i]!='S')
              		cnt1++;
              		
              	}
              	cout<<min(cnt,cnt1)/2<<endl;
              	
              }
               
                      
            }
               
                
 }
    
return 0; 

}
