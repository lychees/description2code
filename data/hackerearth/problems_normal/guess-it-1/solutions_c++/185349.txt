#include<iostream>
#include<stdio.h>
#include<string.h>
#include<string>
using namespace std;
#define in(t) scanf("%d",&t)
int main()
{
	int t,sum,i,ans;
	in(t);
	string s;
	char num;
	while(t--)
	{
		sum=ans=0;
		cin>>s;
		for(string::iterator it=s.begin(); it!=s.end(); ++it)
		{
			num=*it;
			if(num!='_')
			sum+=(int)(num)-48;
		}
		for(i=1;i<=9;i++)
		{
			sum+=i;
			if(sum%9==0)
			{
			ans=i;
			break;
			}
			else
			sum-=i;
		}
		if(ans==0 || ans==9)
		printf("0 or 9\n");
		else
		printf("%d\n",ans);
	}
	return 0;
	}
