int find_res(int *arr, int index, int k){
	int res= INT_MAX;
    for(int i=index; i<= index+k-1; i++){
        int x1 = 0;
        for(int j=index; j<=index+k-1; j++){
            if(arr[i]>=arr[j]){
                x1 = x1 + 3*(arr[i]-arr[j]);
            }
            else {
                x1 = x1 + 5*(arr[j]-arr[i]);
            }
        }
        if(res > x1) res = x1;
   	}
   	return res;

}
