#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,j;
	cin>>n>>k;
	
	int a[2000][24]={{0}};
	
	for(i=1;i<=n;i++)
	  {
	  	for(j=1;j<=k;j++)
	  	  cin>>a[i][j];
	  }
	int q,m,c;
	cin>>q;
	while(q--)
	 {
	 	cin>>m;
	 	vector<int> ans;
	 	for(i=1;i<=n;i++)
	 	 {
	 	 	ans.push_back(i);
	 	 }
	 	for(int pp=0;pp<m;pp++)
	 	 { int res=INT_MAX;
	 	 	cin>>c;
	 	 	for(i=0;i<ans.size();i++)
	 	 	   res=min(res,a[ans[i]][c]);
	 	 	vector<int> ans2;
	 	 	for(i=0;i<ans.size();i++)
	 	 	   if(res==a[ans[i]][c])
	 	 	      ans2.push_back(ans[i]);
	 	 	ans=ans2;
	 	 	
	 	 }
	 	cout<<ans[0]<<endl;
	 }
}
