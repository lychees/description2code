#include <iostream>
using namespace std;

const int mod = 10000007;

int main()
{
    int n,r; cin>>n>>r;
    n--; r--;
    if(r > n) {cout<<-1; return 0;}
    if(!n) {cout<<1; return 0;}
    int a[n+1],b[n+1];
    a[0] = 1; a[1] = 1;
    for(int i = 2; i <=n ; ++i){
    	b[0] = 1;
    	for(int j = 1; j < i; ++j){
    		b[j] = (a[j-1] + a[j]) % mod;
    	}
    	b[i] = 1;
    	for(int j = 0; j <= i; ++j)
    		a[j] = b[j];
    }
    cout<<a[r];
    return 0;
}
