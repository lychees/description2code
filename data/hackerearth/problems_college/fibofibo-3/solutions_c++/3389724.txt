#include<iostream>
using namespace std;

int cnt0=0,cnt1=0;

int fibo(int n)
{
	if(n==0)
	{
		cnt0++;
		return 0;
	}
	if(n==1)
	{
		cnt1++;
		return 1;
	}
	return fibo(n-1)+fibo(n-2);
}

int main()
{
	int T,N;
	cin>>T;
	while(T--)
	{
		cin>>N;
		fibo(N);
		cout<<cnt0<<" "<<cnt1<<endl;
		cnt0=cnt1=0;
	}
	return 0;
}
