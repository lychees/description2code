#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
#define pb push_back
#define cone __builtin_popcount
int main()
{
	ll freq[100][26];
	memset(freq,0,sizeof(freq));
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<6;j++)
		{
			char x;
			cin>>x;
			freq[i][x-'a']+=1;
		}
	}
	string s;
	cin>>s;
	ll ans=0;
	for(int i=0;i < (1<<n) ;i++)
	{
		if(cone(i)==k)
		{
			vector <int> c;
			int ind=0;
			int copy=i;
			while(copy)
			{
				if(copy&1)
					c.pb(ind);
				ind++;
				copy>>=1;
			}
			sort(c.begin(),c.end());
			do{
				ll ways=1;
				for(int j=0;j<k;j++)
				{
					if(freq[c[j]][s[j]-'a']>0)
					{
						ways*=freq[c[j]][s[j]-'a'];
						ways%=mod;
					}
					else
					{
						ways=0;
						break;
					}
				}
				ans+=ways;
				ans%=mod;
			}while(next_permutation(c.begin(),c.end()));
		}
	}
	cout<<ans<<endl;
	return 0;
}
