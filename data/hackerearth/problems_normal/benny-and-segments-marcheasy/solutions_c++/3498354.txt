#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,l;
		scanf("%d%d",&n,&l);
		set<pair<int,int> >edges;
		set<int>vertices;
		int maxi=0;
		for(int i=0;i<n;i++)
		{
			int a,b;
			scanf("%d%d",&a,&b);
			edges.insert(make_pair(a,b));
			maxi=max(maxi,max(a,b));
			vertices.insert(a);
			vertices.insert(b);
		}
		set<pair<int,int> >:: iterator it,it2;
		int flag=0,cf,cl;
		for(int i=1;i+l<=maxi;i++)
		{
			cf=INT_MAX;
			cl=-1;
			for(it2=edges.begin();it2!=edges.end();it2++)
			{
				if(it2->first>=i && it2->second<=i+l)
				{
					cf=min(cf,it2->first);
					cl=max(cl,it2->second);
				}
			}
			if(cf==i && cl==i+l)
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			printf("No\n");
		else
			printf("Yes\n");
	}
	return 0;
}
