#include <iostream>
using namespace std;
int main()
{
	int t;cin >> t;
	int numb[82],count=10;
	for (int i=1;i<82;i++)
	{
		if (count/10==count%10)
		count++;
		numb[i]=count++;
	}
	while (t--)
	{
		int n,k;cin >> n >> k;
		if (k>81)
		{
			cout << -1 << endl;
			continue;
		}
		for (int i=0;i<n/2;i++)
			cout << numb[k];
		if (n%2==1)
			cout << numb[k]/10;
		cout << endl;
	}
    return 0;
}
