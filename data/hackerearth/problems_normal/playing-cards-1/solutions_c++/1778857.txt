#include <iostream>
#include <bits/stdc++.h>
#define lli long long
#define M 1000000007
using namespace std;
lli cal[1010][1010];
void pre()
{
	cal[0][0] = 1;
	for ( int i = 1; i <= 1002; i++ ) {
		for ( int j = 0; j <= i; j++ ) {
			if ( i == j || j == 0 ) cal[i][j] = 1;
			else cal[i][j] = (cal[i-1][j-1] + cal[i-1][j])%M;
		}
	}
	return;
}
lli C(int x, int y)
{
	if ( y == x ) return 1;
	if ( y == 0 ) return 1;
	if ( y > x ) return 0;
	return cal[x][y];
}
int main()
{
	pre();
	int t,n,m;
	cin >> t;
	assert(t>=1&&t<=10);
	while ( t-- ) {
		cin >> n >> m;
		assert(n>=1&&n<=50);
		assert(m>=1&&m<=1000);
		if ( m > n*52 || (m%2 == 1) ) {
			printf("0\n");
			continue;
		}
		lli ans = 0;
		int tot_fce = 6*n;
		int tot_rnk = 18*n;
		for ( int i = 0; i <= m/2; i++ ) {
            if ( i > tot_fce ) continue;
            if ( m/2-i > tot_rnk ) continue;
            if ( i > tot_rnk ) continue;
            if ( m/2-i > tot_fce ) continue;
            lli val = C(tot_fce,i);
            val = (val*C(tot_rnk,m/2-i))%M;
            val = (val*C(tot_rnk,i))%M;
            val = (val*C(tot_fce,m/2-i))%M;
            ans = (ans + val)%M;
		}
		cout << ans << endl;
	}
	return 0;
}
