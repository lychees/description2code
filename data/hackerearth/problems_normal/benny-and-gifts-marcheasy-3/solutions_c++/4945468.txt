#include <stdio.h>
#include<string.h>
int main()
{
	int i, c = 0, len, x[100000], y[100000], j;
	char a[100000];
	scanf("%s", &a);
	x[-1] = 0;
	y[-1] = 0;
	len = strlen(a);
	for (i = 0; i<len; i++){
		if (a[i] == 'R'){
			x[i] = x[i - 1];
			y[i] = y[i - 1] + 1;
		}
		else if (a[i] == 'L'){
			x[i] = x[i - 1];
			y[i] = y[i - 1] - 1;
		}
		else	if (a[i] == 'U'){
			x[i] = x[i - 1] - 1;
			y[i] = y[i - 1];
		}
		else if (a[i] == 'D'){
			x[i] = x[i - 1] + 1;
			y[i] = y[i - 1];
		}
		for (j = -1; j<i; j++){
			if (x[i] == x[j] && y[i] == y[j]){ c++; break; }
		}

	}
	printf("%d", c);
	return 0;
}
