    #include<iostream>
    #include<math.h>
    using namespace std;
    #define ll long long int
    	
    	int main()
    	{
    	
    	
        ll t;
        cin>>t;
        while(t--)
        {
        ll z,n,m;
        cin>>z>>n>>m;
    	ll count=0;
    	for(ll i=1;i<=sqrt(z);i++){
    	if(z%i==0){
    	if((z/i)<=n && i<=m)
    	count++;
    	if((z/i)<=m && i<=n)
    	count++;
    	if(i*i==z)
    	count--;
    						}
    			
    		}
    		cout<<(count)<<endl;
    		
    		
    	}
    	return 0;
    }
