#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    int aCost, bCost;
    int res[100];
    string pal;
    string fhalf, lhalf;
    for(i=0;i<n;i++)
    {
    	cin>>pal>>aCost>>bCost;
    	
    	fhalf.erase();
    	lhalf.erase();
    	res[i]=0;
    	
    	for(j=0;j<pal.length()/2;j++)
    		fhalf+=pal[j];
    	for(j=pal.length()-1;j>=pal.length()/2;j--)
     		lhalf+=pal[j];
    	
    	for(j=0;j<pal.length()/2;j++)
    	{
    		if(fhalf[j]=='/' && lhalf[j]=='/')
    		{
    			if(bCost<aCost)
    			{
    				lhalf[j] = 'b';
    				fhalf[j] = 'b';
    				res[i] += bCost*2;
    			}
    			else
    			{
    				lhalf[j] = 'a';
    				fhalf[j] = 'a';
    				res[i] += aCost*2;
    			}
    		}
    		else if(fhalf[j]=='/')
    		{
    			fhalf[j] = lhalf[j];
    			res[i] += (lhalf[j]=='a')?aCost:bCost;
    		}
    		else if(lhalf[j] == '/')
    		{
    			lhalf[j]=fhalf[j];
    			res[i] += (fhalf[j]=='a')?aCost:bCost;
    		}
    	}
    	for(j=0;j<pal.length()/2;j++)
   			if(lhalf[j]!=fhalf[j])
    			res[i]=-1;
    			
    }
    for(i=0;i<n;i++)
    	cout<<res[i]<<endl;
}
