#include <stdio.h>
#include <vector>
#define MX_N 100010

using namespace std;

vector<int> arr[MX_N];
int team[MX_N];
int t,n,m,error;

void dfs(int i,int T) {
	if(team[i]) {
		if(team[i] != T) error = 1;
		return;
	}
	team[i] = T;
	for(int j=0;j<arr[i].size();j++) {
		dfs(arr[i][j],-T);
	}
}
int main() {
	int i,j,x,y;
	scanf("%d",&t);
	while(t--) {
		scanf("%d %d",&n,&m);
		for(i=1;i<=n;i++) {
			team[i] = 0;
			while(!arr[i].empty()) arr[i].pop_back();
		}
		for(i=1;i<=m;i++) {
			scanf("%d %d",&x,&y);
			arr[x].push_back(y);
			arr[y].push_back(x);
		}
		error = 0;
		for(i=1;i<=n;i++) {
			if(!team[i]) {
				dfs(i,1);
			}
		}
		if(error) {
			printf("NO\n");
		}else {
			printf("YES\n");
		}
	}
	return 0;
}
