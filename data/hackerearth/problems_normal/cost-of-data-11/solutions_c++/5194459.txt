#include <iostream>
#include<cstdlib>
#include<cstdio>
#include<string>
using namespace std;

struct Node{
	char c;
	struct Node *child[26];
};

int k = 1;

void insert(Node *root, string s){
	for(int i = 0; i < s.length(); i++){
		char c = s[i];
		if(root->child[c-97] == NULL){
			root->child[c- 97] = (Node *)calloc(1, sizeof(Node));
			k++;
		}
		root = root->child[c-97];
	}
}

int main() {
	int n;
	cin>>n;
	Node *root = (Node *)calloc(1, sizeof(Node));
	while(n--){
		string s;
		cin>>s;
		insert(root, s);
	}
	cout<<k<<endl;
	return 0;
}
