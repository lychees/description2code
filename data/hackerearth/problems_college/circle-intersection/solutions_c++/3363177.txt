#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int x1,y1,r1,x2,y2,r2;
    
    cin>>x1>>y1>>r1;
    cin>>x2>>y2>>r2;
    
    int rad_sum=r1+r2;
    int x_sq=pow((x1-x2),2);
    int y_sq=pow((y1-y2),2);
    
    int dist_cent=sqrt(x_sq+y_sq);
    
    if(dist_cent<rad_sum){
    	cout<<"Y";
    }
    else if(dist_cent==rad_sum){
    	cout<<"T";
    }
    else{
    	cout<<"N";
    }
    return 0;
}
