#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>

using namespace std;

int main(){
	
	int t;
	cin>>t;
	
	while(t--){
		int n,flag=0;
		cin>>n;
		
		if(n==1||n==2)
		cout<<"-1"<<endl;
		
		else{
			for(int i=0;i<=n;i++){
				if(i%5==0&&(n-i)%3==0){
					flag=1;
					int a=i;
					int b=n-i;
					
					while(b){
						cout<<"5";
						b--;
					}
					while(a){
						cout<<"3";
						a--;
					}
					cout<<endl;
					break;
				}
				
			}
			if(flag==0){
					cout<<"-1"<<endl;
				}
		}
		
	}




	return 0;
}
