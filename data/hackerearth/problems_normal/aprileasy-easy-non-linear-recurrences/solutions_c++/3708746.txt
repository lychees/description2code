#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007

int main()
{
    long long a, b, c, d, e, f;
    scanf("%lld %lld %lld %lld %lld %lld", &a, &b, &c, &d, &e, &f);
    
    int Q;
    scanf("%d", &Q);
    
    while(Q--)
    {
    	int N;
    	scanf("%d", &N);
    	
    	if(N==0)
    		printf("0\n");
    	else
    		printf("%lld\n", (f + (N-1)*f%MOD*d%MOD)%MOD);
    }
    
    return 0;
}
