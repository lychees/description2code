#include<iostream>
#include<cmath>
#include<map>
#include<string>
#include<cstdlib>
#include<cstdio>
#include<vector>
#include<set>
#include<iomanip>
#include<algorithm>
#include<utility>
#define MOD 1000000007
using namespace std;
typedef long double LD;
typedef long long int LL;
typedef unsigned long long int ULL;

int main()
{
	int n, m;
	cin >> n >> m;
	string name;
	LL p;
	map< string, LL > score;
	LL maxPoint = 0;
	string winner;
	for( int i=0; i<m; ++i )
	{
		cin >> name >> p;
		if( score.find( name ) != score.end() )
		{
			// name found in the map
			score[name] += p;
			if( p > 0 && score[name] > maxPoint )
			{
				maxPoint = score[name];
				winner = name;
			}
		}
		else
		{
			// new name encountered
			score.insert( make_pair( name, p ) );
			if( p > maxPoint )
			{
				maxPoint = p;
				winner = name;
			}
		}
	}
	cout << winner << endl;
	return 0;
}
