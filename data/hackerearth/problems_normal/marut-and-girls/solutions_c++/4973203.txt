#include <iostream>
#include <algorithm>
#include <string>
#include <set>
#include <cstring>
using namespace std;

set<int> cons;

bool check(string s)
{
	int cnt = 0;
	int res = s[0] - '0';
	set<int> tmp = cons;
	
	for (int i = 1; i < s.length(); ++i)
	{
		if (s[i] == ' ')
		{
			tmp.erase(res);
			res = s[i + 1] - '0';
			++i;
			continue;
		}
		
		res = res * 10 + s[i] - '0';
	}
	tmp.erase(res);
	
	return tmp.empty();
}

int main()
{
	ios::sync_with_stdio(false);
    string prop;
    int n, m;
    int qual;
    int cnt = 0;
    
    cin >> m;
    while (m--)
    {
    	cin >> qual;
    	cons.insert(qual);
    }
    
    cin >> n;
    cin.ignore();
    while (n--)
    {
    	getline(cin, prop);
    	cnt += check(prop);
    }
    cout << cnt << endl;
    return 0;
}
