#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll x,a,b,c,d,z=0;
    	cin>>x>>a>>b>>c>>d;
    	if(a>x)
    	{	for(ll q=0;q<=b;++q)
    			for(ll r=0;r<=c;++r)
    				for(ll s=0;s<=d;++s)
    					if((2*q+5*r+10*s)<=x)
    						++z;
    	}
    	else
    	{	for(ll p=0;p<=a;++p)
    			for(ll q=0;q<=b;++q)
    				for(ll r=0;r<=c;++r)
    					for(ll s=0;s<=d;++s)
    						if((p+2*q+5*r+10*s)==x)
    							++z;
    	}
    	cout<<z<<"\n";
    }
    return 0;
}
