#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,s,e;
    cin>>n>>s>>e;
    map<long long int,vector<long long int> > m;
    long long int temp1,temp2;
    for(int i=0;i<n;i++) 
    {
    	cin>>temp1>>temp2;
    	m[temp1-temp2].push_back(temp1+temp2);
    }
	map<long long int,vector<long long int> >::iterator it;
    long long int cleft=s,cright=s,left,right,dist=0;
    for(it=m.begin();it!=m.end();it++)
    {
    	sort(m[it->first].rbegin(),m[it->first].rend());
    	left=it->first;
   // 	cout<<left<<"\n";
    	right=m[it->first][0];
   // 	cout<<right<<"\n";
    	if(right>=e) right=e;
    	if(left>=e) break;
    	if(left<s && right>cright)
    	{
    		dist+=right-cright;
    		cright=right;
    	}
    	else if(left>=cright) 
    	{
    		cleft=left;
    		cright=right;
    		dist+=cright-cleft;
    //		cout<<cleft<<" "<<cright<<" "<<dist<<"\n";
    	}
    	else if(left>=cleft && right>cright)
    	{
    		dist+=right-cright;
    		cright=right;
    		cleft=left;
   // 		cout<<cleft<<" "<<cright<<" "<<dist<<"\n";
    	}
    }
    long long int d=e-s;
    long long int ans=d-dist;
    cout<<ans;
    return 0;
}
