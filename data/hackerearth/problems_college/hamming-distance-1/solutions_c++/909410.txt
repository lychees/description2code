#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{int testcases;
cin>>testcases;
while(testcases-->0)
{
	unsigned long long a,b;
	cin>>a;
	cin>>b;
	int num1[32];
	int num2[32];
	for(int i=0;i<32;i++)
	{
		num1[i]=0;
		num2[i]=0;
	}
	int cou=0;
	while(a!=0)
	{
		num1[cou]=a%2;
		a=a/2;
		cou++;
	}
	cou=0;
    while(b!=0)
	{
		num2[cou]=b%2;
		b=b/2;
		cou++;
	}
	int ans=0;
	for(int i=0;i<32;i++)
	{
		if(num1[i]!=num2[i])
		ans++;
	}
	cout<<ans<<endl;
}
}
