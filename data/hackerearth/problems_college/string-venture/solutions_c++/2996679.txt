#include <iostream>
#include <string>
#include <algorithm>
#include <map>
using namespace std;

int main() {
	string s;

	getline(cin,s);
	int sum=0;
	for(int i = 0; i<s.length(); i++){
		char c = s[i];
		sum += (int) c;
		//cout<<c<<" "<<sum<<endl;
	}
	//cout<<sum<<endl;
	sum = sum/s.length();
	//cout<<sum<<endl;
	if(sum%2==0){
		for(auto it = s.rbegin(); it!=s.rend(); it++){
			cout<<*it;
		}
	} else {
		map<char,int> m;
		for(char c: s){
			if(m.find(c)!=m.end()){
				m[c]++;
			}else
				m[c]=1;
		}
		int max = 0;
		char c=' ';
		for(auto e: m){
			if(e.second >max){
				max = e.second;
				c = e.first;
			}
		}
		cout<<c;
	}
	return 0;
}
