ll power(ll a,ll b)
{
if(!b)
return 1;
if(b==1)
return a;
ll temp=power(a,b/2);
temp=(temp*temp);
if(b&1)
temp=(temp*a);
return temp;
}

int main()
{
	ios::sync_with_stdio( false );
	ll i, n, num;
	ll ans, req, cal = 0;
	cin >> n;
	multiset <ll> s;
	multiset <ll> ::iterator it;
	for( i=0;i<n;i++ )
	{
		cin >> num;
		s.insert( num );
	}
	ans = 0;
	i = 1;
	while( !s.empty() )
	{
		req = i * i;
		it = s.lower_bound( req );
		if(it==s.end())
		{
			it--;
			ans += req - *it;
			s.erase( it );
		}
		else
		{
			if ( (*it) > req )
			{
				if( it != s.begin())
				{
					it--;
					ans += req - *it;
					s.erase( it );
				}
				else
					ans += req;
			}
			else
			{
				ans += req - *it;
				s.erase( it );
			}
		}
		i++;
	}
	cout << ans ;
	return 0;
}
