#include <bits/stdc++.h>
using namespace std;
int a[1000001];
int main()
{
    int t,n,i,*b;
    cin>>t;
    while(t--)
    {
    	memset(a,0,1000001);
    	cin>>n;
    	b=new int [n];
    	for(i=0;i<n;i++)
    	    {
    	    	cin>>b[i];
    	    	a[b[i]]++;
    	    }
    	    for(i=0;i<1000001;i++)
    	    {
    	      	if(a[i]&1)
    	      	{
    	      		cout<<i<<endl;
    	      		break;
    	      	}
    	    }	
    	
    	
    }
    return 0;
}
