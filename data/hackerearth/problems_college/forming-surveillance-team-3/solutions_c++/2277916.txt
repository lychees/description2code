#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	cin >> n;
    	int *A=new int[n];
    	for(int j=0;j<n;j++)
    	{
    		cin >> A[j];
    	}
    	sort(A,A+n);
    	int c=1;
    	for(int j=0;j<n-1;j++)
    	{
    		if(A[j]!=A[j+1])
    		c++;
    	}
    	cout << c << endl;
    }
}
