#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    unsigned long long int p,c;
    unsigned long long int n;
    cin >> t;
    while(t--) {
        p=1;
        c=1;
        scanf("%llu",&n);
        while(p<=n) {
            p*=2;
        }
        p/=2;
        if(p==n) {
            cout<<1<<endl;
            continue;
        }
        n-=p;
        while(n) {
            while(p>n) {
                p/=2;
                }
            c++;
            n=n-p;
        }
    printf("%llu\n",c);
    }
    return 0;
}
