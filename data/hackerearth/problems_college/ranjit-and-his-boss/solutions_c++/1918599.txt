#include <iostream>
#include <unordered_set>
#include <vector>

using namespace std;

typedef vector<int> vi;
#define range(i,a,b) for(auto i=(a);i<b;i++)
#define rep(i,n) range(i,0,n)

#define X 100005

vi ys[X];
unordered_set<int> xs[X];

int main()
{
	int n;
	cin >> n;

	rep(i, n)
	{
		int x, y;
		cin >> x >> y;
		ys[x].push_back(y), xs[y].insert(x);
	}

	int ans = 0;
	rep(x0, X) for(auto y0 : ys[x0])
	{
		if(ys[x0].size() < xs[y0].size())
		{
			for(auto y1 : ys[x0]) if(y1 > y0)
			{
				int x1 = x0 + y1 - y0;
				ans += xs[y0].count(x1) && xs[y1].count(x1);
			}
		}
		else
		{
			for(auto x1 : xs[y0]) if(x1 > x0)
			{
				int y1 = y0 + x1 - x0;
				if(y1 < X) ans += xs[y1].count(x0) && xs[y1].count(x1);
			}
		}
	}

	cout << ans << endl;
	return 0;
}
