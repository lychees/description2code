#include<bits/stdc++.h>
using namespace std;
bool compare(int i,int j)
{
    return i>j;
}
int main()
{
    ios_base::sync_with_stdio(false);
    unsigned long long int k,i;
    cin>>k;
    unsigned long long int v[k];
    memset(v,0,sizeof(v));
    for(i=0;i<k;i++)
    cin>>v[i];
    sort(v,v+k,compare);
    //for(i=0;i<k;i++)
    //cout<<v[i]<<' ';
    unsigned long long int max=0,count=0;
    for(i=0;i<k;i++)
    {
        count++;
        unsigned long long int days=count+v[i];
        if(days>max)
        max=days;
    }
    cout<<max+1<<endl;
    return 0;
}
