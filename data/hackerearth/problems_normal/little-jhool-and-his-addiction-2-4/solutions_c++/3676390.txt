#include<bits/stdc++.h>
using namespace std;


int main () {
	int t;
	cin>>t;
	while (t--) {

		int itm, k;
		cin>>itm>>k;

		vector < int > price;
		price.resize(itm);

		for (int i=0; i<itm; i++) {
			scanf("%d",&price[i]);
		}

		int n = price.size();
		sort(price.begin(),price.end());
		for(int i=0;i<n/2;i++) {
			price[i] += price[n - 1 - i];
		}
		sort(price.begin(), price.begin() + n / 2);

		long long   ans =  price[n / 2 - 1] - price[0];
		cout << ans << endl;

		if (ans>k) {
			cout << "No more girlfriends!" << endl;
		}
		else if (ans==k) {
			cout << "Lucky chap!" << endl;
		}
		else {
			cout << "Chick magnet Jhool!" << endl;
		}
	}
}
