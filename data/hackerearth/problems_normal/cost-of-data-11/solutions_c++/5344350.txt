#include <bits/stdc++.h>
using namespace std;

int node_cnt = 1;

struct trie{
	trie *child[28];
	bool isleaf;
};


struct trie *newTNode()
{
	trie *temp = (trie *)malloc(sizeof(trie));
	for(int i=0;i<28; i++)
	{
		temp->child[i] = NULL;
	}
	temp->isleaf = false;
	return temp;
}

void insert(trie *node, string s, int indi)
{
	if(indi==s.size())
		{
			node->isleaf = true;
			return;
		}
	int c_value = s[indi]-'a';
	if(node->child[c_value] == NULL)
	{
		node->child[c_value] = newTNode();
		node_cnt+=1;
		
	}
	insert(node->child[c_value], s, indi+1);
}



int main()
{
    int t;
    trie *root = newTNode();
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	insert(root, s, 0);
    }
    cout<<node_cnt<<endl;
    return 0;
}
