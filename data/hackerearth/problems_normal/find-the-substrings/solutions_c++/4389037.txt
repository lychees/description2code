	bool operator() (const pii &l, const pii &r) const {
		return l.first < r.first;
	}

const int MAXN = 1e6 + 3;
char s[MAXN];
int n;
char a = 'a', b = 'b', c = 'c';
void solve() {
	int T;
	cin >> T; cin.ignore();
	while (T--) {
		ll res = 0;
		gets(s);
		n = strlen(s);
		int i = 0, j = 0;
		int cnt[150] = {0};
		cnt[s[0]] = 1;
		int tmp;
		while (j < n) {
			while (cnt[a] && cnt[b] && cnt[c] && i < n) { cnt[s[i]]--; i++; continue; }
			tmp = j;
			while (!(cnt[a] && cnt[b] && cnt[c]) && j < n) {
				j++;
				cnt[s[j]]++;
			}

			res += (ll)(j - tmp) * (j - tmp + 1) / 2 + (ll) (j - tmp) * (tmp - i);
		}
		cout << res << endl;
	}
}
