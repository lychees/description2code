#include <iostream>
#include<string>
#include <algorithm>    // std::sort
#include<stdlib.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
    	int n,g[1000],b[1000],gi=0,bi=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>b[bi];
    		++bi;
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>g[gi];
    		++gi;
    	}
    	
    	sort(b,b+bi);
    	sort(g,g+gi);
    	int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		sum=sum+((b[i])*(g[i]));
    	}
    	cout<<sum<<endl;
    	--t;
    }
    return 0;
}
