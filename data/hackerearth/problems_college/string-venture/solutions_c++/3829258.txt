#include <bits/stdc++.h>
using namespace std;
char a[1001],chk[26];
int main()
{
   scanf(" %[^\n]s",a);
   int l=strlen(a);
   int s=0;
   int ma=INT_MIN,mi;
   for(int i=0;i<l;i++)
   {
   	chk[a[i]-97]++;
   }
   for(int i=0;i<l;i++)
   {
   	s+=a[i];
   }
   if((s/l)%2==0)
   {
   	for(int i=l-1;i>=0;i--)
   	printf("%c",a[i]);
   }
   else
   {
   	for(int i=0;i<26;i++)
   	if(chk[i]>ma)
   	{
   	ma=chk[i];
   		mi=i;
   	}
   	printf("%c",mi+97);
   }
}
