#include <iostream>
using namespace std;

int main()
{
    long long int n,c,test,i,j,a[1000005],sum;
    cin>>test;
    while(test--){
    	sum=0;
    	cin>>n>>c;
    	j=0;
    	int f=0;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    		if(f==0)
    		{sum+=a[i];
    		while(sum>c){
    		    sum-=a[j++];
    		}
    		if(sum==c)
    		{
    			f=1;
    			
    		}
    		}
    	}
    	if(f==0){
    		cout<<"NO"<<endl;
    	}
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
