#ifdef WIN32
	#define I64 "%I64d"
#else
	#define I64 "%lld"
#endif

typedef long long ll;

#define f first
#define s second
#define mp make_pair
#define pb push_back
#define all(s) s.begin(), s.end()
#define sz(s) (int(s.size()))
#define fname "a"
#define ms(a,x) memset(a, x, sizeof(a))
#define forit(it,s) for(__typeof(s.begin()) it = s.begin(); it != s.end(); ++it)
#define MAXN 200002

int n;
int a[MAXN];

int main()
{
	#ifdef LOCAL
	freopen(fname".in", "r", stdin);
	freopen(fname".out", "w", stdout);
	#endif

	scanf("%d", &n);
	for (int i = 0; i < n; ++i)
		scanf("%d", &a[i]);

	stack <int> st;
	int ans = 0;
	for (int i = n - 1; i >= 0; --i)
	{
		int t = 0;
		while(!st.empty() && st.top() < a[i]) st.pop(), ++t;
		ans = max(ans, t);
		st.push(a[i]);
	}
	printf("%d\n", ans);

	return 0;
}
