def findK(A,B,C,K):
    low=0
    high=100000
    pos=-1
    while(low<=high):
        mid=(low+high)//2
        poly=A*mid**2+B*mid+C
        if(poly >= K):
            pos = mid
            high = mid-1                
        else:
            low = mid+1
    return pos

        print findK(K[0],K[1],K[2],K[3])
