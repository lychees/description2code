#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T;
	cin>>T;
	char str[1000],str2[1000];
	while(T--)
	{
		int count=0,min=INT_MAX;
		cin>>str>>str2;
		if(strlen(str)==strlen(str2))
		{
			min=0;
			for(int i=0;i<strlen(str2);i++)
			{
				if(str[i]!=str2[i])
				min++;
			}
		}
		else
		{
			for(int i=0;i<=strlen(str)-strlen(str2);i++)
		{
			count=0;
			for(int j=i,k=0;k<strlen(str2);k++,j++)
			{
				if(str[j]!=str2[k])
				count++;
			}
			if(count<min)
			min=count;
			
		}
		}
		cout<<min<<endl;
	}
	return 0;
}
