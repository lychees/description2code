#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,m,k=0;
  cin>>n;
  int a[n];
  for(int i=1;i<=n;i++)
  cin>>a[i];
  for(int i=1;i<n;i++)
  {
    if(a[i]!=0 && a[i+1]!=0)
    {
    m=min(a[i],a[i+1]);
    a[i]=a[i]-m;
    a[i+1]=a[i+1]-m;
   }
  }
  for(int i=1;i<=n;i++)
  {
    if(a[i]!=0)
    {
      k=1;
      break;
    }
  }

  if(k==0)
  cout<<"YES"<<endl;
  else
  cout<<"NO"<<endl;
  return 0;
}
