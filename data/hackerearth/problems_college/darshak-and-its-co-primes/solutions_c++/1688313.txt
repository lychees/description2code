#include <iostream>
using namespace std;
long long int check(long long int a,long long int b)
{
	long long int t;
 while (b != 0) {
    t = b;
    b = a % b;
    a = t;
  }
  return a;
}

int main()
{
    long int t;
    cin>>t;
    while(t--)
    {
    long long int  p,q,f;
    	cin>>p>>q;
    	
    		 f=check(p,q);
    		
    		if(f==1 )
    	      	cout<<"Is a Co-Prime"<<endl;
    		else
    			cout<<"Not a Co-Prime"<<endl;
    		
    	
    	
    }
    return 0;
}
