#include <iostream>
using namespace std;
int c=1;
long long int getr(long long int l)
{
	long long int r=0;
	int rem;
	while(l>0)
    	{
    			rem=l%10;
    			r=r+rem*rem;
    			l=l/10;
    	}
    	if(r/10!=0) getr(r);
    	else if(r==1) cout<<"Case #"<<c++<<":YES"<<endl;
    	else cout<<"Case #"<<c++<<":NO"<<endl;
}
int main()
{
    long long int n;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	getr(n);
    }
    return 0;
}
