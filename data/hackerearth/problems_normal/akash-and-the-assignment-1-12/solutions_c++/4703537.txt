#include <bits/stdc++.h>
using namespace std;
int dp[100005][26];
int main()
{
	int n,q,l,r,k;
	scanf("%d%d",&n,&q);
	string s;
	cin>>s;
	s="0"+s;
	for(int i=1;i<=n;i++)
	{
		int x=s[i]-'a';
		for(int j=0;j<26;j++)
		{
			if(x==j)
			dp[i][j]=dp[i-1][j]+1;
			else
			dp[i][j]=dp[i-1][j];
		}
	}
	while(q--)
	{
		scanf("%d%d%d",&l,&r,&k);
		int sum=0;
		int flag=0;
		char ans;
		for(int j=0;j<26;j++)
		{
			sum=sum+dp[r][j]-dp[l-1][j];
			if(sum>=k)
			{
				flag=1;
				ans=char(j+'a');
				break;
			}
		}
		if(flag==0)
		{
			printf("Out of range\n");
			
		}
		else
		{
			printf("%c\n",ans);
		}
		
	}
	return 0;
	
	
}
