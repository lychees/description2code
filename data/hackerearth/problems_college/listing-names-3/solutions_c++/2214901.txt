#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int i=0;
    vector<string >v;
    string s;
    while(i<n)
    {
    	cin>>s;
    	v.push_back(s);
    	i++;
    	
    }
    sort(v.begin(),v.end());
    int count=0;string s_t=v[0];
    for(int i=1;i<n;i++)
    {
    	if(v[i]==s_t)
    	count++;
    	else
    	{cout<<v[i-1]<<" "<<count+1<<endl;count=0;s_t=v[i];}
    }
    cout<<v[i-1]<<" "<<count+1<<endl;
    return 0;
}
