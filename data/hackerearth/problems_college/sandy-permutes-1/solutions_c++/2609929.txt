#include <iostream>
using namespace std;

int main()
{
   long long int n,k,cnt=0;
   cin >> n >> k;
   long long int a[n];
   for(int i=0;i<n;i++)
   	cin >> a[i];
   	
   	if(n<=k){
   		for(int i=n;i>0;i--)
   			cout << i << " ";	
   	}else{
   		for(int i=0;i<n-1;i++){
   			int max=a[i],ind;
   			for(int j=i+1;j<n;j++){
   				if(a[j]>max){
   					max=a[j];
   					ind=j;
   				}
   			}	
   				if(a[i]!=max){
   					int temp=a[i];
   					a[i]=a[ind];
   					a[ind]=temp;
   					cnt++;
   				}
   				if(cnt==k)
   					break;
   				
   			
   			}
   			for(int i=0;i<n;i++){
   				cout << a[i] << " ";
   			
   		}
   	}
    return 0;
}
