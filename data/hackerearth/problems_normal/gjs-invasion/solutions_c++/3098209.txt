#define ll long long

ll fact(int n){

     if(n==0 || n==1)
     	 return 1;

     ll ans=1;

     for(int i=n;i>=1;--i){

           ans*=i;


     } 	
 
     return ans;
 
}

int main(){

   int t;

   cin>>t;

   while(t--){

      int n;
  
      cin>>n;


      ll ans=0;

      if(n==0 || n==1)
      	printf("%lld\n",ans);

      else{

         for(int k=0;k<=n-2;++k){

               if(k%2==0)
               	 ans+=fact(n)/fact(k)*(3*n+k)*(n-k-1);

               else
               	 ans-=fact(n)/fact(k)*(3*n+k)*(n-k-1);
                        
         }

         ans/=12;

         printf("%lld\n",ans);

      }





   } 

   return 0;
}
