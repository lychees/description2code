#include <bits/stdc++.h>
#define rep(i,x,y) for (int i = (x); i<=(y); i++)
#define repe(i,x,y) for (int i = (x); i < (y);i++)
#define drep(i,x,y) for (int i = (x); i >= (y); i--)
#define mp make_pair
#define pb push_back
#define MAX(a,b) (((a)>(b))?(a):(b))
#define MIN(a,b) (((a)<(b))?(a):(b))
#define MAXN 100
#define sf(n) scanf("%Lf",&n)
#define prf(n) printf("%.2Lf",n)
#define	s(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define pr(n) printf("%d",n)
#define prl(n) printf("%lld",n)
#define endc printf("\n")
#define psp printf(" ")

using namespace std;
typedef long long ll;
typedef long double ld;
typedef pair<int,int> pii;

int n;

pair<pii,ld> A[MAXN];

ld dp[MAXN][MAXN];

void solve() {
	s(n); A[0] = mp(mp(0,0),0.0);
	rep(i,1,n) {
		int x,y; ld p; s(x); s(y); sf(p);
		A[i] = mp(mp(x,y-1),p/100.0);
	}

	sort(A,A+n+1);

	drep(i,n+1,1) {
		rep(j,0,n) {
			if (i > n) { dp[i][j] = 0.0; continue; }
			dp[i][j] = dp[i+1][j];
			int idx = j;
			if (A[i].first.second > A[j].first.second) idx = i;

			if (A[j].first.second < A[i].first.first) {
				dp[i][j] = MAX(dp[i][j], dp[i+1][idx] + A[i].second);
			}
		}
	}

	prf(dp[1][0]); endc;
}

int main() {
	int T; s(T); while(T--) solve();
}
