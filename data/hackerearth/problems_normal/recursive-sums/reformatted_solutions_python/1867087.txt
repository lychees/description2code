import sys
T = int(sys.stdin.readline())
for case in range(T):
    M = int(sys.stdin.readline())
    num = 0
    for x in range(M):
        [i, j] = [int(y) for y in sys.stdin.readline().split()]
        num += i*j
    while len(str(num)) > 1:
        num = sum([int(x) for x in str(num)])
    print num
