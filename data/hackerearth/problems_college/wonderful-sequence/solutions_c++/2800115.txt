#include <iostream>
using namespace std;

int main()
{
	int t; cin>>t;
	while(t--){
		int n; cin>>n; int a[n+1];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		int cnt=0;
		for(int i=0;i<n-1;i++){
			if(a[i]>=a[i+1]) cnt++;
		}
		cout<<cnt<<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
