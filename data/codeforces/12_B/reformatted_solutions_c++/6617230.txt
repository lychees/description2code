#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

int main() {
  string a, b;
  cin >> a >> b;
  sort(a.begin(), a.end());
  if (a.size() != 1)
    if (a[0] == '0')
      swap(a[0], a[1]);
  if (a == b)
    cout << "OK";
  else
    cout << "WRONG_ANSWER";
  return 0;
}
