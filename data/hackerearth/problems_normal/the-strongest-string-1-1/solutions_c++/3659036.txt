#include <bits/stdc++.h>
using namespace std;

int pos = -1; //current position (first position of possible max char found in current daq)
string result = "";
int n;
string s;

char previousmax = 'z'+1;

void daq() {
	while (true) {
		char currmax = 0;
		for(int i = pos+1;i<s.size();i++) {
			if (s[i]>currmax && s[i]<previousmax) {
				currmax=s[i];
				pos = i;
			}
		}
		if (!currmax) {
			return;
		}
		result += currmax;
		
		previousmax = currmax;
	}
}

int main()
{
    cin>>n;
    cin>>s;
    
	daq();
	
	cout<<result;
    return 0;
}
