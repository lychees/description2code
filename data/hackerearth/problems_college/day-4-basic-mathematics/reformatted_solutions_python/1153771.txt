cases = int(input())
battle = 0
completed_day = 0
for i in xrange(cases):
    day = int(raw_input())
    while (completed_day < day):
        battle += 1
        completed_day = (battle*(battle+1))/2

    rem = battle % 3
    battle = 0
    completed_day = 0

    if rem == 0:
        print "SL"
    elif rem == 1:
        print "LB"
    else:
        print "BS"
