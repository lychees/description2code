#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int i,j=0,k=0,l,n,flag;
	cin>>n;
	char a[100][100];
	while(k<n)
	{
	
		cin>>a[k];
		k++;
	}
	k=0;
	while(k<n-1)
	{
		
		for(i=k;i<n-1;i++)
		{
		if(strlen(a[k])==strlen(a[i+1]))
		{
			flag=0;
			for(j=0;j<strlen(a[k]);j++)
			{
				if(a[k][j]!=a[i+1][strlen(a[k])-1-j]){
				flag=1;
				break;
				}
			}
			if(flag==0)
				{
					cout<<strlen(a[k])<<" "<<a[k][strlen(a[k])/2];
					return 0;
				}
		}
	
		}
		k++;
	
	}
  
    return 0;
}
