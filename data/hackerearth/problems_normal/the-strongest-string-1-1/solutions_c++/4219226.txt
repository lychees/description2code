#include <iostream>
#include <string>
using namespace std;
typedef long long ll;
int main()
{
    int map[26]={0};
    ll l;
    cin>>l;
    cin.ignore();
    string s;
    getline(cin,s);
    for(ll i=0;i<s.size();i++) 	map[s[i]-'a']+=1;
    ll f=0;
    for(ll i=25;i>=0;i--)
    {
    	if(map[i]<=0)continue;
    	else
    	{
    		ll j=f;
    		for(;j<s.size();j++)
    		{
    			if((s[j]-'a')==i) 
    			{
    				cout<<s[j];
    				break;
    			}
    			map[s[j]-'a']-=1;
    		}
    		f=j+1;
    	}
    }
    cout<<endl;
    return 0;
}
