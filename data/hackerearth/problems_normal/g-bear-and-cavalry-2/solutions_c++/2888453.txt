#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n, i, flag;
    	cin>>n;
    	int a[n], b[n];
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	cin>>b[i];
    	sort(a+1,a+n);
    	sort(b,b+n,greater<int>());
    	
    	for(i=1;i<n;i++)
    	{
    		flag=1;
    		if((a[i]*b[i])>=(a[0]*b[0]))
    		{
    			flag=0;
    			break;
    		}
    	}
    	if(flag)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
return 0;
}
