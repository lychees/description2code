#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<int> vi;
typedef vector<ii> vii;

const int  MOD=1000000007;
const int  INF= int(1e9);
const int  MAX=100009;
int a[MAX],n,lt,rt;
map<int,int> visited;
int main()
{
	ios_base::sync_with_stdio(false);
    int testCases;
    cin>>testCases;
    while(testCases--) {
        visited.clear();
        cin>>n;
        for(int i=0;i<n;i++) {
            cin>>a[i];
        }
        int res=-INF,j=0;
        for(int i=0;i<n;i++) {
        
            for( ;j<n;j++) {
                if(visited[a[j]]) {
                    break;
                } else {
                    visited[a[j]]++;
                }
            }
            if(j-i > res) {
                lt=i+1;
                rt=j;
                res=j-i;
            }
            visited[a[i]]--;
        }

        cout<<lt<<" "<<rt<<"\n";
    }
	return 0;

}
