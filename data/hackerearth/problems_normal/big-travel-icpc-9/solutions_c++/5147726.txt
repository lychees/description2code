#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <cstdlib>
using namespace std;
typedef long long ll;
#define MOD 1000000007
int main(){
    int n;
    cin >> n;
    vector<ll> x_tab(n);
    vector<ll> y_tab(n);
    
    for (int i = 0; i < n; i++){
        cin >> x_tab[i] >> y_tab[i];
    }
    sort(x_tab.begin(), x_tab.end());
    sort(y_tab.begin(), y_tab.end());
    
    vector<ll> sum_x(n);
    vector<ll> sum_y(n);
    
    for (int i = 1; i < n; i++){
        sum_x[i] = (sum_x[i-1]%MOD + x_tab[i-1]%MOD)%MOD;
        sum_y[i] = (sum_y[i-1]%MOD + y_tab[i-1]%MOD)%MOD;
    }
    ll ans_x = 0;
    ll ans_y = 0;
    for (int i = 1; i < n; i++){
        ans_x = (ans_x%MOD + ((i*x_tab[i])%MOD - sum_x[i] + MOD)%MOD)%MOD;
        ans_y = (ans_y%MOD + ((i*y_tab[i])%MOD - sum_y[i] + MOD)%MOD)%MOD;
    }
    cout << (ans_x + ans_y)%MOD << endl;
}
