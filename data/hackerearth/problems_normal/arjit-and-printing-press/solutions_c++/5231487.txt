#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	string w,r;
    	cin>>w>>r;
    	sort(r.begin(),r.end());
    	int x=0;
    	for(int j=0;j<w.size();j++){
    		if(w[j]>r[x]){
    			w[j]=r[x];
    			x++;
    		}
    		if(x>r.size()-1) break;
    		
    	}
    	cout<<w<<endl;
    	
    }
    return 0;
}
