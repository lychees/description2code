#include<iostream>
#include<vector>
using namespace std;

typedef long long natural;

int main(){
	int t;
	cin >> t;
	while (t--){
		int n;
		cin >> n;
		natural ary[n];
		for (natural i = 0; i < n; i++)
			cin >> ary[i];

		int start, end;
		start = -2;
		end = -2;
		for (int i = 1; i < n; i++){
			if (ary[i - 1] > ary[i]){
				start = i - 1;
				break;
			}
		}
		for (int j = n - 1; j > 0; j--){
			if (ary[j - 1] > ary[j]){
				end = j;
				break;
			}
		}
		for (int i = start+1; i <= end; i++){
			if (ary[i - 1] - ary[i] > 1){
				start = end = -2;
				break;
			}
		}
		cout << start + 1 <<" "<< end + 1 << endl;
	}
	return 0;
}
