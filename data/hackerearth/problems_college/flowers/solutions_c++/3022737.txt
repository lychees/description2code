#include <bits/stdc++.h>

using namespace std;

#define ll long long 

int main()
{
	int n,k;
	cin>>n>>k;
	ll c[n];
	for(int i = 0;i<n;i++)
		cin>>c[i];
	sort(c,c+n);
	int users[k];
	for(int i = 0;i<k;i++)
	users[i] = 1;
	ll sum = 0;
	for(int j = n-1;j>=0;j--)
	{
		sum += users[0]*c[j];
		users[0]++;
		sort(users,users+k);
	}
	cout<<sum;
	return 0;
}
