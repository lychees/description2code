#include <iostream>
using namespace std;

int main()
{
    int a,b;
    cin >> a;
    cin >>b;
    int count =0;
    if(a==b)
    {
    	cout << "wins" << endl;
    	return 0;
    }
    while(a)
    {
    	if(a%10 != b%10)
    	{
    		count++;
    	}
    	a/=10;b/=10;
    }
    if(count!=1)
    	cout << "loses" << endl;
    else
    	cout << "wins" << endl;;
    return 0;
}
