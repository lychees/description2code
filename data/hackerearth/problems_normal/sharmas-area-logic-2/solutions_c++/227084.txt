#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	
   	int n;
   	cin>>n;
   	int minx=200,miny=200;
   	int maxx=-200,maxy=-200;
   	
   	for(int i=0;i<n;i++)
   	{
   		int a,b;
   		scanf("%d%d",&a,&b);
   		minx=min(minx,a);
   		miny=min(miny,b);
   		maxx=max(maxx,a);
   		maxy=max(maxy,b);
   	}
   	
   	int area=(maxx-minx)*(maxy-miny);
   	if(area)
   	  cout<<area<<endl;
   	 else 
   	   cout<<max(maxx-minx,maxy-miny)<<endl;
   	
   	
   }
    return 0;
}
