#include <iostream>
using namespace std;
 
int main() {
	int a[] = {1000,500,100,50,20,10,5,2,1},n=9;
	int t;
	cin>>t;
	while(t--)
	{
		long num,temp,count = 0;
		cin>>num;
		temp = num;
		while(temp != 0)
		{
			for(int i=0;i<n;i++)
			{
				if(a[i] <= temp)
				{
					count += temp / a[i];
					temp = temp % a[i];
					
				}
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
