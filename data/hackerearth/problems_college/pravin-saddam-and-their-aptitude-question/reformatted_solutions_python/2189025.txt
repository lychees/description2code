from math import factorial


def getFactorial():
    facts = {}
    for x in range(1, 19):
        facts[x] = factorial(x)
    return facts


noc = input()
facts = getFactorial()
while noc:
    noc -= 1
    a, b = map(int, raw_input().split())
    sums = 0
    for x in range(a, b+1):
        sums += facts[x]
    print sums % 10
