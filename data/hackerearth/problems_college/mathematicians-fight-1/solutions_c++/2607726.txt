#include <iostream>
#include <cmath>
using namespace std;

int main() {
int t;
cin>>t;
int a[t];
int p2;
int n2;
int n;
for (int var =1; var <=t; ++var) {
cin>>n;
a[var]=1;
if((n && (n-1)))
{
	p2=log2(n);
	n2=pow(2,p2);
	n=n-n2;
	for(int i=0;i<n;i++)
	{
		a[var]+=2;
	}
}

}
for (int var = 1; var <=t; ++var) {
	cout<<a[var]<<"\n";

}
	return 0;
}
