import math

testcases=input()
while testcases:
	t0,t1,t2,v1,v2,d = map(int,raw_input().split())
	
	timea =int(math.ceil(t1+(float(d)/v1)*60))
	timeb=int(math.ceil(t2+(float(d)/v2)*60))
	
	if(t1<t0 and t2<t0):
		print("-1")
	elif(t1>=t0 and t2<t0):
		print(timea)
	elif(t1<t0 and t2>=t0):
		print(timeb)
	else:
		print(min(timea,timeb))
	
	testcases=testcases-1
