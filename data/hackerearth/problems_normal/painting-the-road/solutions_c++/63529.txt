#include <iostream>
using namespace std;
#define INT_MAX 100000000
int main()
{
    	int T;
	cin >> T;
	while(T--){
		int L,K;
		cin >> L >>  K;
		int arr[K][4];
		int i,j;
		for(i=0;i<K;i++){
			for(j=0;j<3;j++){
				cin >> arr[i][j];
			}
		}
		long long int DP[L+1];
		DP[0]=0;
		for(i=1;i<=L;i++){
			DP[i] = INT_MAX;
		}
		for(i=1;i<=L;i++){
			for(j=0;j<K;j++){
				if(i >= arr[j][0] && i <= arr[j][1]){
					DP[i] = min(DP[i],DP[arr[j][0]]+arr[j][2]);
				}
			}
		}
		if(DP[L] >= INT_MAX)
			cout << "-1"<< endl;
		else
			cout << DP[L] << endl;
	}
	return 0;
}
