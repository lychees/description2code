#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int pep[n];
	for(int i=n-1;i>=0;i--)
		cin>>pep[i];
		
	for(int i=0;i<n-1;i++)
	{
		if(pep[i] < pep[i+1])
		{
			int j = i;
			while(pep[j] < pep[j+1])
			{
				int temp = pep[j];
				pep[j] = pep[j+1]-1;
				pep[j+1] = temp+1;
				
				j--;
				if(j < 0)break;
			}
		}
	}
	
	bool win = true;
    for(int i=0;i<n-1;i++)
    {
    	if(pep[i] < pep[i+1])win = false;
    }
    if(win)
    {
    	for(int i=n-1;i>=0;i--)
    		cout<<pep[i]<<" ";
    }
	else
    	cout<<":(";
    return 0;
}
