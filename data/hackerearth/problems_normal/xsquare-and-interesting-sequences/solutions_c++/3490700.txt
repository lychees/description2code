#include <iostream>
using namespace std;
#define MOD 1000000009

int main()
{
	int t;
	long long int n,ans,temp = 26;
	cin>>t;
	while(t--)
	{
		temp = 26;
		cin>>n;
		n = n+1;
		n = n / 2;
		ans = 1 ;
		while(n)
		{
			if(n&1)
			{
				ans = ans*temp ;
				ans = ans%MOD ;
			}
		n = n/2 ;
		temp = temp*temp ;
		temp = temp % MOD ;
		}
		cout<<ans<<endl;
	}	
    return 0;
}
