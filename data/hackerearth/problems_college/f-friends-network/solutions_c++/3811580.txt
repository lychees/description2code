#include<bits/stdc++.h>
using namespace std;
int main()
{
	int tc;
	cin>>tc;
	label :while(tc--)
	{
		bool a[501][501];
		//
		int n;
		cin>>n;
		//
		for(int i=1;i<=n;i++)
			for(int j=1;j<=n;j++)
				cin>>a[i][j];
		
			for(int i=1;i<=n;i++)
				{
					for(int j=1;j<=n;j++)
					{
						if(a[i][j]==1)
						{
							for(int k=1;k<=n;k++)
							{
								if((a[j][k]==1)&&(k!=i))
								{
									if(a[k][i]==1)
									{
										cout<<"Aye"<<endl;
										goto label;
									}
								}
							}
						}
					}
				}
				cout<<"Nae"<<endl;
	}
	return 0;
}
