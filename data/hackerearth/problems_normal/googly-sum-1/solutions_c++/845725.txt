#include<iostream>
#include<stdio.h>
#define si(n) scanf("%d",&n)
#define FOR(i,a,b) for (i = a; i < b; i++)
#define pi(n) printf("%d\n",n)
using namespace std;

const int Maxn = 5005;
const int nil = 200005;
const int Maxm = 400015;
int n;
int a[Maxn];
bool was[Maxm];
int res;
int main()
{

	si(n);
	int j=0,i=0;
	FOR(i,0,n){
		si(a[i]);
		FOR(j,0,i)if(was[a[i]-a[j]+nil])break;
		res+=j<i;
		FOR(j,0,i+1)was[a[i]+a[j]+nil]=true;
	}
	pi(res);
	return 0;
}
