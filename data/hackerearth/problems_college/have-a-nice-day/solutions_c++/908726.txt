#include <iostream>
using namespace std;
#include<algorithm>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	int b[n];
    	for(int i=0;i<n;i++)
    	cin>>b[i];
    	sort(a,a+n);
    	sort(b,b+n);
    	long long s=0;
    	for(int i=0;i<n;i++)
    	s=s+a[i]*b[i];
    	cout<<s<<endl;
    }
}
