#include <iostream>   
using namespace std;    
int a[10]={6,2,5,5,4,5,6,3,7,6};     
int main()
{
	    int t;
	    cin>>t;
	    while(t--)
	    {
		    string s;
		    cin>>s;
		    int sum=0;
		    for(int i=0;i<s.length();i++)
		    {
		    	sum+=(a[s[i]-'0']);
		    }
		    cout<<sum<<endl;
    	}
    return 0;
     
}
