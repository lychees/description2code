#include <iostream>
#include <cstdio>

using namespace std;

int a[100001];
int n, k;

typedef long long int ll;

ll find () {

    ll lo = 1;
    ll hi = 1000000000;
    ll ans = 1000000000;

    while (lo <= hi) {
        ll mid = (lo + hi) / 2;

        int turn = k;
        int last = a[1];
        int last_num = a[1];

        int flag = 0;

        for (int i = 2; i <= n; i++) {
                if (turn == 0) {
                    flag = 1;
                }

                if (a[i] - last > mid) {
                    if (last == last_num) {
                        flag = 1;
                        break;
                    }
                    turn--;
                    last = last_num;
                    i--;
                }
                else if (a[i] - last <= mid) {
                    last_num = a[i];
                }
        }

        if (flag == 1) {
            lo = mid + 1;
        }
        else{
            hi = mid - 1;
            ans = min(ans , mid);
        }

    }

    return ans;
}

int main () {
    int t;
    scanf("%d", &t);

    while (t--) {
        scanf("%d%d", &n, &k);

        for (int i = 1; i <= n; i++) {
            scanf("%d", &a[i]);
        }

        printf("%lld\n", find() ) ;
    }


}
