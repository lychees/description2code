#include <bits/stdc++.h>
using namespace std;
map<string,int>mp;
vector<pair<int,string> >st;
int main()
{
    int n,t;
    string s;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	int i,j,f=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>s;
    		st.push_back(make_pair(s.length(),s));
    	}
    	sort(st.begin(),st.end());
    	for(i=0;i<n;i++)
    	{
    		s = st[i].second;
    		for(j=0;j<s.length();j++)
    		if(mp[s.substr (0,j+1)])
    		break;
    	
    		mp[s]++;
    		
    		if(j!=s.length())
    		f=1;
    	}
    	if(f)
    	cout<<"NO\n";
    	else
    	cout<<"YES\n";
    	mp.clear();
    	st.clear();
    }
    return 0;
}
