#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int arr[10004];
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int sz=0;
        for(int i=0; i<10004; i++)
            arr[i]=0;
        int a,b,t=0,c=0;
        cin>>a;
        b = a/30;
        b++;
        arr[0]=sz=1;
        for(int i=0; i<b; i++)
        {
            for(int j=0; j<sz; j++)
            {
                t = arr[j]*2 + c;
                arr[j]=t%10;
                c = t/10;
            }
            while(c)
            {
                arr[sz++]=c%10;
                c = c/10;
            }
        }
        for(int i=sz-1; i>=0; i--)
            cout<<arr[i];
        cout<<endl;
    }
return 0;
}
