#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i;
    cin>>t;
    while(t--){
    	
    	cin>>n;
    	int wa[n],ho[n];
    	for(i=0;i<n;++i){
    		cin>>wa[i];
    	}
    	
    	for(i=0;i<n;++i){
    		cin>>ho[i];
    	}
    	
    	int maxm  = ho[0],id = 0;
    	for(i=1;i<n;++i){
    		if(ho[i]>maxm){
    			maxm = ho[i];
    			id = i;
    		}
    	}
    	
    	int pro  = wa[0]*maxm;
    	
    	vector<int>wav;
    	vector<int>hov;
    	for(i=1;i<n;++i)
    		wav.push_back(wa[i]);
    	
    	for(i=0;i<n;++i){
    		if(i!=id)
    			hov.push_back(ho[i]);
    	}
    	
    	sort(wav.begin(),wav.end());
    	sort(hov.begin(),hov.end());
    	int f = 1;
    	for(i=0;i<wav.size();++i)
    		if(wav[i]*hov[wav.size()-i-1]>=pro){
    			f = 0;
    			break;
    		}
    	
    	if(f)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
