#include <iostream>
using namespace std;
unsigned long long int a[1000004] = {0};
unsigned long long int fact(unsigned long long int n)
{
	if(n==1)
	{
		a[n] = 1;
		return 1;
	}
	if(a[n]==0)
	{
		a[n] = (n*fact(n-1))%1000003;
		return a[n];
	}
	else
	{
		return a[n];
	}
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	unsigned long long int n,x,k;
    	scanf("%llu %llu",&n,&x);
    	if(n>=1000003)
    	{
    		printf("0\n");
    		continue;
    	}
    	k = fact(n);
    	k=(k*x)%1000003;
    	printf("%llu\n",k);
    }
    return 0;
}
