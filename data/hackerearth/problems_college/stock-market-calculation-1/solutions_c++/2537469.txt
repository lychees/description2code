#define M 100005

# define MOD 1000000007
#define num_of(x) (sizeof (x) / sizeof (*x))
using namespace std;
int a[M];
 
int main()
{
 int t;
 cin>>t;
 while(t--)
 {
  int n;
  cin>>n;
  for(int i=0;i<n;i++)
   cin>>a[i];
  priority_queue<pii> q;
  for(int i=0;i<n;i++)
  {
    q.push(pii(a[i],i));
   // cout<<q.top().first<<endl;
  }
 
  int temp;
  lo shares=0,profit=0;
  int  ind=0;
  while(!q.empty())
  {
   int i;
   temp=q.top().second;
   //cout<<temp<<endl;
   q.pop();
   if(temp<ind)
   continue;
   //cout<<ind<<endl;
   for( i=ind;i<temp;i++)
   {
     shares++;
     profit-=a[i];
 
   }
   if(i==temp)
   {
    profit+=shares*a[i];
    shares=0;
    ind=temp+1;
   }
 
  }
  cout<<profit<<endl;
 }
 return 0;
}
