#define GOD 1000000007
 
ll fast_expo( ll base, ll powr)
{
  ll i ,j , k , ans=1;
 
  while(powr)
  {
    if(powr&1)
      ans=((ans%GOD)*(base%GOD))%GOD;
    base=((base%GOD)*(base%GOD))%GOD;
    powr>>=1;
  }
  return (ans%GOD);
 
}
 
 
int main()
{
  ll  prod ,cnt, n , i, j , k , a[100005], x ,y;
  ll Q, q;

  prod=1;
  cnt=0;
