#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
   ll t;
   cin>>t;
   while(t>0)
   {
   	t--;
   	ll size,i,j;
   	cin>>size;
   	vector<ll> v;
   	for(i=0;i<size;i++)
   	{
   		ll t;
   		cin>>t;
   		v.push_back(t);
   	}
   	ll c=0;
   	for(i=0;i<size;i++)
   	for(j=i+1;j<size;j++)
   	{
   		if((v[i]+v[j])%2==0 && v[i]!=v[j])
   		c++;
   	}
   	cout<<c<<'\n';
   }
    return 0;
}
