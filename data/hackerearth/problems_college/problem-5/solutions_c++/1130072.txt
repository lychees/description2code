#include <iostream>
#include<map>
#include<vector>
#include<algorithm>

using namespace std;


int main()
{

    long long int a,n;
    cin>>a>>n;

    long long int sum=0;
    long long int previous=a;

    for (int i=0; i<n; i++)
    {

        sum = sum + previous;
        previous = previous*a;

    }

    cout << sum<<endl;
}
