#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,w,x,y,z;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>w>>x>>y>>z;
		int lef = min(max(w, x), max(y, z));
		int righ = max(min(w, x), min(y, z));
		if (lef>righ)
		{
			cout << "Nothing" << endl;
		}
		else
		{
			if (lef==righ)
			{
				cout << "Point" << endl;
			}
			else
			{
				cout << "Line" << endl;
			}
		}
    	
    }
    return 0;
}
