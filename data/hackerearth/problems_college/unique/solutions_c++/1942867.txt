#include<bits/stdc++.h>
using namespace std;
main()
{
	std::ios_base::sync_with_stdio(false);
    int t;
    long long int n,i,z,count;
    cin>>t;
    while(t--)
    {
    	map<long long int,long long int> a;
    	cin>>n;
    	count=0;
    	for(i=1;i<=n;i++)
    	{
    		cin>>z;
    		if(a[z]==0)
    		{
    			count++;
    			a[z]=1;
			}
		}
		cout<<count<<endl;
	}
}
