#include<iostream>
#include<string.h>
#include<math.h>
using namespace std;

int main()
{
    int T,T1,len,root,i,j,k;
    char arr[200];
    char mat[14][14];
    cin>>T;
    T1=T;
    while(T--+1)
    {
        gets(arr);
        len=strlen(arr);
        root=sqrt(len);
        k=0;
        for(i=0;i<root;i++)
        {
            for(j=0;j<root;j++)
            {
                mat[i][j]=arr[k++];

            }
        }
        for(i=0;i<root;i++)
        {
            for(j=0;j<root;j++)
            {
            	if(T1!=T)
                cout<<mat[j][i];

            }
        }
       // cout<<root;
       if(T1!=T)
        cout<<endl;
    }
    return 0;
}
