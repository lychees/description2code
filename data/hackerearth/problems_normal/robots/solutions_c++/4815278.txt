#include <iostream>
#include<algorithm>
using namespace std;
#define mod 1000000007
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		long long int res=1;
		for(int i=0;i<n;i++)
		{
			res=((res%mod)*(a[i]-i)%(mod))%mod;
		}
		cout<<res<<endl;
	}
    return 0;
}
