#define _CRT_SECURE_NO_WARNINGS
#include <iostream>
#include <string>

using namespace std;


int main()
{
	int i, Q, L1, R1, L2, R2;
	string str;

#ifdef _MSC_VER
	freopen("Text.txt", "r", stdin);
#endif

	cin >> str;
	cin >> Q;

	for (i = 0; i < Q; i++)
	{
		cin >> L1 >> R1;
		cin >> L2 >> R2;

		if (str.compare(L1 - 1, R1 - L1 + 1, str, L2 - 1, R2 - L2 + 1) == 0)
		{
			cout << "Yes" << endl;
		}
		else
		{
			cout << "No" << endl;
		}

	}

	return 0;
}
