#include<iostream>
#include<string>
#include<vector>
#include<cmath>
#include<cstring>
#include<set>
#include<map>
#include<queue>
#include<algorithm>
#include<cstdio>
#include<iomanip>
using namespace std;

int a[100005];
int main()
{
	int t,n,k;
	for(int i=1 ; i<=100001 ; i++)
		for(int j=0 ; (1<<j)<=i ; j++){
			bool x=i&(1<<j);
			bool y=i&1<<(j+1);
			bool z=i&1<<(j+2);
			a[i]+= x&&z&&!y;
		}
	scanf("%d",&t);
	while(t--){
		int ans=0;
		scanf("%d%d",&n,&k);
		for(int i=1; i<=n ; i++) ans+=a[i]>=k;
		printf("%d\n",ans);
	}
	return 0;
}
