#include<bits/stdc++.h>
#define ull unsigned long long
#define MAX 100000
using namespace std;

//ull summ[MAX];

int main()
{
	int t,val,q;
	cin >> t;
	ull a[t];
	ull target;

	cin >> a[0];
	for(int i=1; i<t; i++)
	{
		cin >> val;
		a[i]=a[i-1]+val;
	}
	cin >> q;
	
	while(q--)
	{
		cin >> target;
		int i=0;
		while(true)
		{
			if(a[i]<target)
				i++;
			else if (a[i]>=target)
			{
				int val = i+1;
				if(val>t)
					val=-1;
				cout<<val<<"\n";
				break;
			}
		}
	}
}
