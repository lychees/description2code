#include <iostream>
#include <string.h>
#define MAX 100002
using namespace std;

long long proces(long long input[], long N, long long X)
{
    long long count = 0, temp;
    for(int i = 1; i < N; i++)
    {
        if(input[i] <= input[i-1])
        {
            temp = input[i-1] - input[i];
            count += ((temp/X) + 1);
            input[i] += (X*(temp/X + 1));
        }
    }
    return count;
}

int main()
{
    int T;
    long N;
    long long input[MAX];
    long long X;
    cin >> T;
    while(T--)
    {
        cin >> N >> X;
        for(int i = 0; i < N; i++)
            cin >> input[i];
        cout << proces(input, N, X) << endl;
    }
    return 0;
}
