#include<bits/stdc++.h>
#include<iostream>
#define dl(x) std::cout << #x"=" << x << std::endl;
#define dt(x) std::cout << #x"=" << x << "\t";
using namespace std;

int max(int a, int b)
{
    return (a > b) ? a : b;
}



int main(int argc, char const *argv[])
{
	int t;
	scanf("%d",&t);
	for (int tk = 0; tk < t; ++tk)
	{
		int n,k;
		scanf("%d %d",&n,&k);
		int a[100005],b[100005];
		for (int i = 0; i < n; ++i)
		{
			scanf("%d",&a[i]);
		}
		for (int i = 0; i < n; ++i)
		{
			scanf("%d",&b[i]);
		}
		int ans=0;
		for (int i = 0; i < n; ++i)
		{
			ans = max(ans, ((k/a[i])*b[i]));
		}
		printf("%d\n", ans);
	}
	return 0;
}
