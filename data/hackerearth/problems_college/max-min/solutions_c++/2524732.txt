#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	int n,k,l;
	scanf("%d",&n);
	scanf("%d",&k);
	vector<int> x;
	for(int i=0 ; i<n; i++)
	{
		cin>>l;
		x.push_back(l);
	}
	sort(x.begin(),x.end());
	int min=INT_MAX;
	int d;
	for(int i=k-1 ; i<n ; i++)
	{
		d=x[i]-x[i-k+1];
		if(min>d)
		{
			min=d;
		}
	}
	printf("%d",min);
	return 0;
}
