#include <iostream>
#include <stdio.h>
#define DIVI 1000000007

using namespace std;

typedef long long int Lint;

Lint  Raise( Lint a, Lint b ) {
	
	Lint r, v, s;
	
	if( b == 0 )
		return 1;
	else {
		
		if( ( b & 0x1 ) == 0 ) {
			r = Raise( a, b / 2 );
			s = ( ( r % DIVI ) * ( r % DIVI ) ) % DIVI;
		}
		else {
			r = Raise( a, ( b - 1 ) / 2 );
			v = ( ( r % DIVI ) * ( r % DIVI ) ) % DIVI;
			s = ( ( a % DIVI ) * ( v % DIVI ) ) % DIVI;
		}
		return s;
	}
}
int main()
{
	Lint t, n, v, v2, v3;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld", &n );
		v = Raise( 2, n - 1 );
		v2 = ( ( v % DIVI ) * 3 ) % DIVI;
		
		if( n & 0x1 )
			v3 = ( v2 + 1000000006 ) % DIVI;
		else
			v3 = ( v2 + 1 ) % DIVI;
		
		printf("%lld\n", v3 );
	}
	return 0;
}
