#include<iostream> 
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	for(int k=0;k<t;k++)
	{
		cin>>n;
		if(n%3==0&&n%5==0)
		{
                    while(n--)
				cout<<"5";
			cout<<endl;
		}
		else if(n%3==0)
		{
			 while(n--)
				cout<<"5";
			cout<<endl;
		}
		else if(n==5)
		{
			cout<<"33333"<<endl;
		}
		else
		{
			int x=0,y=0,flag=0;
			for(int i=3;i<=n;i+=3)
			{
				if((n-i)%5==0)
				{
					flag=1;
					x=i;
					y=n-i;
				}
			}
			while(x--)
				cout<<"5";
			while(y--)
				cout<<"3";
			if(flag==0)
				cout<<"-1";
			cout<<endl;
	
			
		}
		
	}
	
}
