#include<stdio.h>

int max(int arr[], int l, int r)
{
    int m = arr[l];
    int ind = l;
    for(int i=l+1; i<r; i++)
    {
        if(arr[i]>m)
        {
            m = arr[i];
            ind = i;
        }
    }
    return ind;
}

// function to swap two elements
void swap(int *a, int *b)
{
    int temp = *a;
    *a = *b;
    *b = temp;
}

// function to get the largest permutation.
void largest_permutation(int arr[], int n, int k)
{
    for(int i=0; i<n; i++)
    {
        int pos = max(arr,i,n); // find the position of maximum element

        if(k>0 && pos!=i)
        {
            swap(&arr[pos], &arr[i]); // swap the maximum element with ith index element
            k--;
        }
    }
}

{
    int n ;
