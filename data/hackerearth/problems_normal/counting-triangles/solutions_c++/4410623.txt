#include <stdio.h>
#include <algorithm>
#include <map>
#include <vector>

using namespace std;

int main(){
	int size;
	scanf("%d", &size);

	map < vector<int>, int> val;
	while(size--){
		int a, b, c;
		scanf("%d %d %d", &a, &b, &c);
		vector<int> nums = {a, b, c};
		sort(nums.begin(), nums.end());
		if(val[nums])
			val[nums] += 1;
		else
			val[nums] = 1;
	}

	int count = 0;
	for(auto i : val){
		if(i.second == 1)
			count += 1;
	}
	printf("%d", count);
}
