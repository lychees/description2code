#include <bits/stdc++.h>
using namespace std;

vector <vector <int> >h;
int main()
{
    h.resize(26);
    int n;
    cin>>n;
    string s;
    cin>>s;
    for(int i = 0;i < s.length();i++)
    {
        int idx = s[i] - 'a';
        h[idx].push_back(i);
    }

    int prev = -1;
    string ans = "";
    for(int i = 25;i >= 0;i--)
    {
        for(int j = 0;j < h[i].size();j++)
        {
            if(h[i][j]>prev)
            {
                prev = h[i][j];
                ans += (char)(i + 'a');
                break;
            }
        }
    }
    cout<<ans<<endl;
    return 0;
}
