#include <iostream>
using namespace std;

long long int x[1000001];
long long int y[1000001];
long long int f[1000001];
long long int g[1000001];
void setXY(){
	for(int i=2; i<=500000; i++) {
		for(int j=i; j<=1000000; j+=i){
			if(!x[j]){
				x[j] = i;
			}
			y[j] += i;
		}
	}
}
void setFG(){
	setXY();
	f[0] = 0, g[0] = 0;
	for(int i=1; i<1000000; i++){
		f[i] = f[i-1] + x[i];
		g[i] = g[i-1] + y[i]-i;
	}
}
void printArray(long long int *arr, int n){
	for(int i=0; i<n; i++) {
		cout << arr[i] << " ";
	}
	cout << endl;
}
int main()
{
    int T, N;
    setFG();
    cin >> T;
    while(T--){
    	cin >> N;
    	cout << (f[N]+g[N])%N << endl;
    }
    return 0;
}
