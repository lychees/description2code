def sieve(n):
    is_prime = [True] * (n+1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, n+1):
        if is_prime[i]:
            for j in range(i+i, n+1, i):
                is_prime[j] = False
    return is_prime

n = int(raw_input())
is_prime = sieve(n+n)
total = 0
for i in range(n):
    if is_prime[i]:
        total += i+1
nn1 = n+n-1
for i in range(n, nn1):
    if is_prime[i]:
        total += nn1 - i

print total % 1000000007
