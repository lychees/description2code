#include <iostream>
#include <iomanip>
#include<cmath>
using namespace std;

int main()
{
    int t;
    double h,d,v;
    cin>>t;
    double cot;
    while(t--)
    {
       cin>>h>>d>>v;
       cot= (sqrt(d*d-h*h)*v)/h;
    
       printf("%.9f",cot);
       cout<<endl;
    }
    return 0;
    
}
