#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b;
        cin>>a>>b;
        int arr[a];
        arr[0]=a-1;
        for(int i=1;i<a;i++)
        {
            if (i%2==0)
            {
                int j=(i+2)/2;
                arr[i]=a-j;
            }
            else
            {
                arr[i]=(i/2);
            }
        }
        for(int i=0;i<a;i++)
        {
            if(arr[i]==b)
            {
                cout<<i<<endl;
                break;
            }
        }
    }
    return 0;
}
