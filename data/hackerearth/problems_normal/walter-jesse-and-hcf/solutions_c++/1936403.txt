#define ll long long 
#define pb push_back
#define chal(a,n) for(long long i=a;i<=n;i++)
int main() {
    int t; cin>>t; 
    while (t--)
    {
    	ll  x,y,k,c;
    
    	cin>>x>>y>>k;
    	c = __gcd(x,y);
    	vector <ll> v;
    	for(long long  j=1;j*j<=c;++j)
    	{
    		if(c%j==0)
    		{
    			v.pb(j);
    			if(j*j != c)
    			v.pb(c/j);
    		}
    	}
    	sort(v.begin(),v.end());
    	if(k<=v.size())
    	cout<<v[v.size()-k]<<"\n";
    	else

    }

}
