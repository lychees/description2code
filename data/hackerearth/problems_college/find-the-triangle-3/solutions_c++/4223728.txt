#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
	int t;
	cin>>t;
	while(t--){
    int n;
    cin>>n;
    vector<long long> l(n);
    for(int k=0;k<n;k++)
        cin>>l[k];
    sort(l.begin(),l.end());
    int i,j,k;
    bool flag = true;
    for(i = l.size()-1;i>=0;i--){
        for(j=i-1;j>=0;j--){
            for(k=j-1;k>=0;k--){
                if(l[j]+l[k]>l[i]){
                    flag = false;
                    cout<<l[k]<<" "<<l[j]<<" "<<l[i]<<endl;
                    break;
                }
            }
            if(!flag)
                break;
        }
        if(!flag)
            break;
    }
    if(flag)
        cout<<-1<<endl;
	}
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
