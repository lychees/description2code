#include <bits/stdc++.h>

using namespace std;

int main()
{
  long int t,n,q,i,maxi,mini,x;
  cin>>t;
  while(t--)
  {
      maxi=0;
      mini=INT_MAX;
      cin>>n>>q;
      for(i=0;i<n;i++)
      {
          cin>>x;
          if(x>maxi)
            maxi=x;
          if(x<mini)
            mini=x;
      }
      cout<<q*(maxi-mini)<<"\n";
  }
    return 0;
}
