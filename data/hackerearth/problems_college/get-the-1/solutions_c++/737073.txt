#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int ans[t];
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		int a=n;
		while(!(a==37 || a==58 || a==89 || a==145 || a==42 || a==20 || a==4 || a==16 || a==1))
		{
			int sum=0,rem,p=a;
			while(p>0)
			{
				rem=p%10;
				p=p/10;
				sum+=rem*rem;
			}
			a=sum;
			if(a==37 || a==58 || a==89 || a==145 || a==42 || a==20 || a==4 || a==16)
			 {
			 	ans[i]=0;
			 	break;
			 }
			else if(a==1)
			 {
			 	ans[i]=1;
			 	break;
			 }
		}
	}
	for(int i=0;i<t;i++)
	{
		if(ans[i]==0)
			cout<<"Case #"<<i+1<<":NO"<<endl;
		else
			cout<<"Case #"<<i+1<<":YES"<<endl;
	}
}
