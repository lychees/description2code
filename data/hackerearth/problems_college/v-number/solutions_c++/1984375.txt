#include <iostream>
using namespace std;

int main()
{int t;
scanf("%d",&t);
while(t--)
{
	string s;
	int s1=0,s2=0;
	cin>>s;
	int len=s.length();
	for(int i=0;i<len/2;i++)
	s1+=(s[i]-48);
	for(int i=len/2;i<len;i++)
	s2+=(s[i]-48);
	if(s1==s2)
	printf("YES\n");
	else
	printf("NO\n");
}
    
    return 0;
}
