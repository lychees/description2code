#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
    ll t,n,d,a,k;
    int arr[]={0,-1,1,1,2,1,2};
    cin>>t;
    while(t--){
    	cin>>n;
    	
    	if(n==1)
    	{
    		cout<<"-1"<<endl;
    		continue;
    	}
    	d=n%7;
    	a=n/7;
    	if(d==1){
    		a--;
    		k=2;
    	}
    	else
    	k=arr[d];
    	cout<<k+a<<endl;
    	
    }
    
    
  return 0;  
    
}
