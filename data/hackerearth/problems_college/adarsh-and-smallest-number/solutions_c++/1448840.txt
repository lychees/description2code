#include <bits/stdc++.h>
using namespace std;
#define For(i,a,b) for(int i=a;i<b;i++)
#define mp make_pair
#define pb push_back
#define mod 1000000007
#define eps 0.00000001

long long n,ans[500005],t,sieve[500005];

int main( ){
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    //ios::sync_with_stdio(0);
    For(i,0,500005) ans[i]=1;
    memset(sieve,0,sizeof(sieve));
    For(i,2,500005) if(sieve[i]==0) for(int j=i+i;j<500005;j+=i) sieve[j]=1;
    For(i,2,500005) if(sieve[i]==0)
    {
        for(long long j=i;j<500005;j*=i)
        {
            ans[j]*=i;
            ans[j]%=mod;
        }
        //ans[i]*=ans[i-1];
        //ans[i]%=mod;
    }
    For(i,2,500005)
    {
        ans[i]*=ans[i-1];
        ans[i]%=mod;
    }
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<ans[n]<<endl;
    }
    return 0;
}
