s = raw_input()
d = {}
N, K, L, Q = map(int, s.split())
for i in range(N):
	nd = raw_input()
	fin = nd.split()
	name = fin[0]
	fin.remove(name)
	attr = tuple(int(x) for x in fin)
	d[attr] = name

for j in range(Q):
	attr = raw_input()
	attr = tuple([int(x) for x in attr.split()])
	if attr in d:
		print d[attr]
	else:
		print "You cant fool me :P"
