t = int(input())
while t:
    t -= 1
    n = int(input())
    x = map(int, raw_input().split())
    f = 0
    a = ([])

    for i in range(0, n):
        a.append(0)
    k = 0

    if x[0] == -1:
        f = 1

    while a[k] == 0 and x[k] != -1 and x[k] < n:
        a[k] = 1
        k = x[k]
        if x[k] == -1:
            f = 1

    if f == 1:
        print "Win"
    else:
        print "Lose"
