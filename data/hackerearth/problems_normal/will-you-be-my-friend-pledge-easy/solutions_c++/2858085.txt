#include <iostream>
using namespace std;
int gcd(int max, int child)
{
	if(child==0)
		return max;
	else gcd(child,max%child);
}
int main()
{
    int max, n, c=0;
    cin>>max>>n;
    while(n--)
    {
    	int child, result;
    	cin>>child;
    	result=gcd(max,child);
    	if(result!=1)
    	c++;
    }
    cout<<c;
    	
}
