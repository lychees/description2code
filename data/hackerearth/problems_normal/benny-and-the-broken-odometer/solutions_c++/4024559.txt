inline int nxt(){

	scanf("%d", &n);

}

void solve(){
	int n = nxt();
	int m = 0;
	int c = 1;
	while (n > 0){
		int d = n % 10;
		n /= 10;
		if (d > 3)
			d--;
		m += d * c;
		c *= 9;
	}
	cout << m << "\n";
}

	while (t--){
		solve();
