for i in xrange(input()):
    N, M = map(int, raw_input().split())
    if M == 0:
        print 0
    elif N <= 1:
        print 1
    elif N == 2:
        print 2
    elif N == 3:
        if M < 2:
            print 0
        else:
            print int("720"[-M:])
    elif N == 4:
        if M < 5:
            print 0
        else:
            print int("620448401733239439360000"[-M:])
    else:
        print 0
