int main()
{
    int n,temp,temp1,check,count;
    int cnt;
    int a[15];
    int i,j,k;
    int lo=0,hi;
    scanf("%d",&n);
while(n!=0)
{    count=0;
     for(i=1;i<n;i++)
     {  temp=i;//n
        check=0;j=0;
        while(temp!=0)
        {   a[j++]=temp%10;
            temp=temp/10;
            cnt=j;
        }
       


        lo=0;
        for(k=cnt;k>lo;lo++)
        {   hi=k;
            while(hi>lo)
            {   temp1=0;
                for(j=hi;j>lo;j--)
                {   temp1=temp1*10+a[j-1];
                }hi--;
             

                if(temp1%cnt==0)
                {   check++;
             
                }
                if(check>1)
                {   break;
                }
            }
            if(check>1)
            {  
                break;
            }
        }
        if(check==1)
        {  
            count++;
        }

     }
       printf("%d\n",count);
        scanf("%d",&n);

}
}
