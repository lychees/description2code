#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
typedef set<string> ss;
typedef vector<int> vs;
typedef map<int,char> msi;
typedef pair<int,int> pa;
typedef long long int ll;

ll cur,n,ans,pre,vis[102],i;
char s[10004];
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		n = strlen(s);
    	cur = 1;
    	ans= 1;
    	pre;
    	memset(vis,-1,sizeof(vis));
 		vis[s[0]] = 0;
 		
   		for (i = 1; i < n; i++)
    	{
        	pre=vis[s[i]];
 			if (pre== -1 || (i - cur)>pre)
            cur++;
        	else
        	{
            	if (cur > ans)
                ans = cur;
 				cur= i - pre;
       		}
 			vis[s[i]] = i;
    	}
       cout<<max(ans,cur)<<"\n";
	}
	return 0;
}
