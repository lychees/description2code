using namespace std;
int t;
int r[1002][1002];
int a;
int b;
int main(){
	scanf("%d", &t);
	while (t--){
		scanf("%d%d", &a, &b);
		int countt = 0;
		for (int i = 0; i < a; i++){
			for (int j = 0; j < b; j++){
				r[i][j] = 0;
				if (i - 2 >= 0){
					if (r[i - 2][j]){
						continue;
					}
				}
				if (j - 2 >= 0){
					if (r[i][j - 2]){
						continue;
					}
				}
				r[i][j] = 1;
				countt++;
			}
		}
		cout << countt << endl;
	}
	return 0;
}
