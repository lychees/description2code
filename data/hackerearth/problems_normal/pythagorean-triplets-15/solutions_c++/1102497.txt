#include <iostream>
#include <cmath>
using namespace std;
int gcd(int a, int b)
{
	if(b==0)
	 return a;
	else
	 return gcd(b,a%b);
}
int main()
{
    int t,n,i,j,k;
    long int a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int count=0;
    	for(i=2;i<=(int)(sqrt(n-1)+1);i++)
    	{
    		for(j=1;j<i;j++)
    		{
    		
    			if((gcd(i,j))==1 && ((i-j)%2) && ((i*i+j*j)<=n))
    			{
    					k=0;
    					a=i*i-j*j;
    					b=2*i*j;
    					c=i*i+j*j;
    					while((++k)*c<=n)
    						count++;
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
