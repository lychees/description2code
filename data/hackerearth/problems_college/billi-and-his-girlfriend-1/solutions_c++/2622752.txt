#include <iostream>
#include <math.h>
using namespace std;


long long int  poweronentiation(long long int a, long long int b, long long int mod)
{
	if(b==0)
		return 1;
	long long int res = 1;
	while(b)
	{
		if(b%2==1)
		{
			res = res *a;
			res = res%mod;
		}
		a = (a*a)%mod;
		b = b/2;
	}
	return res%mod;
}

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		long long int a,b,k;
   		cin>>a>>b>>k;
   		
   		k = pow(10,k);
   		
   		long long int res = poweronentiation(a,b,k);
   		long long int ans =0;
   		while(res)
   		{
   			ans += res%10;
   			res = res/10;
   		}
   		cout<<ans<<endl;
   }
    return 0;
}
