#include<bits/stdc++.h>
using namespace std;

int 
gcd ( int a, int b )
{
  if ( a==0 ) return b;
  return gcd ( b%a, a );
}
long long int etf(long long int n) 
     { 
       long long int result = n; 
       for(long long int i = 2;i*i <= n;i++) 
       { 
         if (n % i == 0) result -= result / i; 
         while (n % i == 0) n /= i; 
       } 
       if (n > 1) result -= result / n; 
       return result; 
     } 

int main()
{
	
	long long int t;
	scanf("%lld",&t);
	while(t--){
		//memset(flag,true,sizeof(flag));
		long long int a,b,ans=0;
		scanf("%lld%lld",&a,&b);
		a = gcd(a,b);
		for(long long int i = 1; i*i <= a; i++)
		{
			if(a%i == 0 )
			{
				ans += etf(i);
			}
			if(a%(a/i) == 0 && i*i != a)
			{
				ans += etf(a/i);
			}
			
		}
		printf("%lld\n",ans);
	}
	
}
