#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	bool flag=true;
	string s;
	cin>>s;
	int index1;
	for(int i=s.size()-2;i>=0;i--)
	{
		if(s[i]<s[i+1])
		{
		index1=i;
			flag=false;	
		break;
	
		}
		
	}
	if(flag==true)
	{
		cout<<"no answer"<<endl;
		continue;
	}
	char small=index1;
	
	int ans=index1+1;
	for(int i=index1+1;i<s.size();i++)
	{
		if(s[i]>s[small] && s[i]<s[ans])
		{
			ans=i;
		}
		
	}
	//cout<<small<<ans;
	char tmp;
	tmp=s[small];
	s[small]=s[ans];
	s[ans]=tmp;
	int len=s.size();
	sort(s.begin()+small+1,s.begin()+len);
	cout<<s<<endl;
}
}
