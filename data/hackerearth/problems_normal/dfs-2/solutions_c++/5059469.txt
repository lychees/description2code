#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
	long a[100000];
	int lo,hi,i,n,q,l,r;
	cin >> n;
	for (i = 0; i < n; i++)
	 cin >> a[i];
	cin >> q;
	sort(a,a+n);
	while (q--)
	{
		cin >> l >> r;
		lo = lower_bound(a,a+n,l) - a;
		hi = upper_bound(a,a+n,r) - a;
		cout << hi - lo << "\n";
	}
    return 0;
}
