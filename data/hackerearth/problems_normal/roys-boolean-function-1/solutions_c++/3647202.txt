#include <iostream>
using namespace std;

bool gcd(int i, int n){
	if (n==0){
		if (i==1)
			return true;
		else
			return false;
	}
	else
		gcd(n, i%n);
}
bool isPrime(int x){
	if(x==1)
		return false;
	else if(x==2)
		return true;
	else if (x%2==0)
		return false;
	else{
		bool check = true;
		for(int i =3;i<x/i;i+=2){
			if(x%i==0){
				check= false;
				break;
			}
		}
		return check;
	}
}

int main()
{
    int t;
    cin>>t;
    while(t){
    	t--;
    	int n;
    	cin >>n;
    	int count =0;
    	for (int i=1;i<n;i++){
    		if (gcd(i,n))
    			count++;
    	}
    	if (isPrime(count))
    		cout<<"TRUE\n";
    	else
    		cout<<"FALSE\n";
    }
    return 0;
}
