#include <iostream>
#include<stdio.h>
using namespace std;
 
int main()
{ 
    long long int no_testcases,i=0,length,breadth,height,m;double result; 
    cin>>no_testcases;
    while(i<no_testcases){
    	cin>>length>>breadth>>height;
    	m = (length>breadth)?((breadth>height)?height:breadth):((length>height)?height:length);
    	result = m/2.0;
    	printf("%.1f\n",result);
        i++;
    } 
    return 0;
}
