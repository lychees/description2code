mod = 1000003
fact = []
fact.append(1)
for i in range(1, 1000003):
    fact.append((i*fact[i-1]) % mod)

test = int(input())
while test:
    test -= 1
    sub = map(int, raw_input().split())
    sub[1] %= 1000003
    if sub[0] > 1000003:
        print 0
    else:
        print(fact[sub[0]]*sub[1]) % mod
