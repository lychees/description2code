t = int(raw_input())
for p in xrange(t):
    n = int(raw_input())
    a = list(map(int, raw_input().split()))
    cnt1 = 0
    for i in xrange(n):
        if a[i] < 0:
            cnt1 += 1
    if n-cnt1 == cnt1:
        print(cnt1)
    else:
        if cnt1 > n-cnt1:
            print(cnt1)
        else:
            print(cnt1+1)
