#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int T, R, G, B, i;
    scanf(" %d %d %d %d",&T,&R,&G,&B);
    int * red = new int [1000001];
    int * green = new int [1000001];
    int * blue = new int [1000001];
    for(i=0;i<1000000;i++)
    {
        red[i] = 0;
        green[i] = 0;
        blue[i] = 0;
    }
//    int red[1000001] = {0};
//    int green[1000001] = {0};
//    int blue[1000001] = {0};
    int colors [8] = {0};
    int r=0,rp=0,g=0,gp=0,b=0,bp=0;
    for(i=0;i<T;i++)
    {
        if(i >= (R + 2*R*r) && i < (2*R + 2*R*r))
        {
            red[i] = 1;
            rp += 1;
            if (rp%R == 0)
                r += 1;
        }
        if(i >= (G + 2*G*g) && i < (2*G + 2*G*g))
        {
            green[i] = 1;
            gp += 1;
            if (gp%G == 0)
                g += 1;
        }
        if(i >= (B + 2*B*b) && i < (2*B + 2*B*b))
        {
            blue[i] = 1;
            bp += 1;
            if (bp%B == 0)
                b += 1;
        }
    }
    for(i=0;i<T;i++)
    {
        if (red[i] == 1 and green[i] == 0 and blue[i] == 0)  //red
            colors[0] += 1;
        else if (red[i] == 0 and green[i] == 1 and blue[i] == 0) //green
            colors[1] += 1;
        else if (red[i] == 0 and green[i] == 0 and blue[i] == 1) //blue
            colors[2] += 1;
        else if (red[i] == 1 and green[i] == 1 and blue[i] == 0) //yellow
            colors[3] += 1;
        else if (red[i] == 0 and green[i] == 1 and blue[i] == 1) //cyan
            colors[4] += 1;
        else if (red[i] == 1 and green[i] == 0 and blue[i] == 1) //magenta
            colors[5] += 1;
        else if (red[i] == 1 and green[i] == 1 and blue[i] == 1) //white
            colors[6] += 1;
        else                                                      //black
            colors[7] += 1;
    }
    for(i=0;i<8;i++)
    {
        printf("%d ",colors[i]);
    }
    printf("\n");
    return 0;
}
