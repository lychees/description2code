#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,T,n,p,u,r,s,i;
	scanf("%d",&T);
	for(t=1;t<=T;t++)
	{
		scanf("%d %d %d %d %d",&n,&p,&u,&r,&s);
		for(i=0;i<n;i++)
		{
			if(p+r<=u)
				p=p+r;
			else if(p-s>=0)
				p=p-s;
			else
				break;
		}
		printf("Case #%d: %d\n",t,p);
	}
	return 0;
}
