#include<iostream>
using namespace std;
int arr[1000];
int gcd(int a, int b)
{
	if (b == 0)
		return a;
	gcd(b, a%b);
}
int main()
{
	int t, n,res=0;
	cin >> t;
	while (t--)
	{
		cin >> n;
		for (int i = 0; i < n; i++)
		{
			cin >> arr[i];
			if (arr[i]>res)
				res = arr[i];
		}
		for (int i = 0; i < n; i++)
		{
			res = gcd(res, arr[i]);
		}
		cout << res << endl;
	}
	return 0;
}
