#include <bits/stdc++.h>
#define LL long long 
using namespace std;

int main()
{
	LL i,j,k,freq,q;
	char c;
	string str,sstr;
    cin>>str;
    map<char,LL> m,m2;
    for(i=0;i<=str.length()-1;)
    {
    	c=str[i];
    	j=i+1;
    	freq=0;
    	while(str[j]>='0' && str[j]<='9' && j<=str.length()-1) 
    	{
    		freq=freq*10+(int)(str[j]-48);
    		j++;
    	}
    	m[c]+=freq;
    	i=j;
    }
    freq=0;
    for(auto itr=m.begin();itr!=m.end();itr++)
    {
        //cout<<itr->first<<" "<<itr->second<<endl;
        freq+=itr->second;
        m2[itr->first]=freq;
    }
    cin>>q;
    for(i=0;i<=q-1;i++)
    {
        cin>>k;
        for(auto itr=m2.begin();itr!=m2.end();itr++)
        {
            if(itr->second>=k) 
            {
                cout<<itr->first<<endl;
                break;
            }
            
        }
        auto itr=--m2.end();
        if(itr->second<k) cout<<"-1\n";
    }
    return 0;
}
