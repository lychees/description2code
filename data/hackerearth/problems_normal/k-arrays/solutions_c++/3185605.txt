#include <bits/stdc++.h>
using namespace std;
#define ll int64_t
#define mod 1000000007

ll fast_exp(ll base, ll exp)
{
    ll res=1;
    while(exp>0)
    {
       if(exp%2==1) res=(res*base)%mod;
       base=(base*base)%mod;
       exp/=2;
    }
    return res;
}

int main()
{
    //freopen("x.txt","r",stdin);
    ll t,n,m,a[20][20];
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	for(int i=0;i<m;i++)
    	{
    		for(int j=0;j<n;j++) cin>>a[i][j];
    	}
    	ll x=0;
    	ll y=1e12;
    	ll len[2];
    	len[0]=len[1]=1;
    	for(int i=1;i<(1<<n);i++)
    	{
    		vector< ll > v;
    		for(int j=0;j<n;j++)
    		{
    			if(i&(1<<j))
    			{
    			    v.push_back(j);
    			}
    		}
    		ll sz=v.size();
    		ll p=1;
    		for(int j=0;j<sz;j++)
    		{
    			ll sum=0;
    			for(int l=0;l<m;l++) sum+=a[l][v[j]];
    			p=(p*sum)%mod;
    		}
    		p=(p*fast_exp(sz,mod-2))%mod;
    		if(p>x)
            {
                x=p;
                len[0]=sz;
            }
            if(p<y)
            {
                y=p;
                len[1]=sz;
            }
    	}
    	cout<<(x^y)<<" "<<(len[0]^len[1])<<endl;
    }
    return 0;
}
