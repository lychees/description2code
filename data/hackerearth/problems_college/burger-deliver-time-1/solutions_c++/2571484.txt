#include<iostream>
using namespace std;
int partition(int a[],int as[],int low,int high)
{
    int i,j,temp,key;
    key=a[low];
    i=low;
    j=high+1;
    while(i<=j)
    {
        do
            i++;
        while(key>=a[i]);
        do
            j--;
        while(a[j]>key);
        if(i<j)
        {
            temp=a[i];
            a[i]=a[j];
            a[j]=temp;
            temp=as[i];
            as[i]=as[j];
            as[j]=temp;
        }
    }
    temp=a[low];
    a[low]=a[j];
    a[j]=temp;
    temp=as[low];
    as[low]=as[j];
    as[j]=temp;
    return j;
}
void qs(int a[],int as[],int low,int high)
{
    int mid;
    if(low<high)
    {
        mid=partition(a,as,low,high);
        qs(a,as,low,mid-1);
        qs(a,as,mid+1,high);
    }
}
main()
{
    int n,a[1000][2],arr[1000],as[1000];
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i][0]>>a[i][1];
        arr[i]=a[i][0]+a[i][1];
        as[i]=i;
    }
    qs(arr,as,0,n-1);

    for(int i=0;i<n;i++)
    {
        if(arr[i]==arr[i+1])
        {

        	int j=i;
        	while(i+1<n&&arr[i]==arr[i+1])
        		i++;


        	qs(as,arr,j,i);
        	for(int k=j;k<=i;k++)
        	cout<<as[k]+1<<" ";
        	goto en;
        }
        cout<<as[i]+1<<" ";
        en:;
}
}
