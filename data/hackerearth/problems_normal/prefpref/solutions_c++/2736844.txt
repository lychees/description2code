#include<bits/stdc++.h>
using namespace std;
#define S(x) scanf("%d",&x)
#define P(x) printf("%d",x)

#define MOD 1000000007
#define MAX 1000000010
#define ll long long
#define mp make_pair
#define pb push_back
typedef pair<int,int> pp;
string s, t  ;
void solve()
{
	int i , j ,ans=0 ; 
	int n = s.size();
	int n1 = t.size();
	i= 0, j =0 ; 
	while(i<n && j<n1)
	{
		if(s[i]==t[j])
		{
			ans++;
			i++;j++; 
		}
		else
		{
			j++;
		}
	}
	cout<<ans<<endl;
}
int main()
{
	cin>>s>>t;
	solve(); 
	return 0;
}
