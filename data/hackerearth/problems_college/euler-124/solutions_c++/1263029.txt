#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	int primes[100000]={};
	int p=0;
	int v[100001]={};
	long long aa[100001]={};
	for(int i=0;i<100001;i++)
		aa[i]=1;
	
	for(int i=2;i<=100000;i++)
	{
		if(v[i]==0)
		{
			primes[p++]=i;
			for(int j=2*i;j<=100000;j+=i)
				v[j]=1;
			
		}
	}
	
	pair <int,int> ans[100001]={};
	
	int a[501];
	int h=100000;
	for(int i=0;i<t;i++)	
	{
		cin>>a[i];
		h=max(h,a[i]);
	}
	for(int i=0;i<p;i++)
	{
	//	cout<<primes[i]<<" ";
		for(int j=primes[i];j<=h;j+=primes[i])
		{
	//		if(j==99998)
	//			cout<<primes[i]<<" haah";
			aa[j]=aa[j]*primes[i];
		}
	}
	for(int i=1;i<=h;i++)
	{
	//	cout<<i<<" "<<aa[i]<<endl;
		ans[i-1].first=aa[i];
		ans[i-1].second=i;
	//	cout<<i<<" "<<ans1<<endl;
	}
//	cout<<aa[100000]<<endl;
	sort(ans,ans+h);
	for(int i=0;i<t;i++)
		cout<<ans[a[i]-1].second<<endl;
		
	
	return 0;
}
