# print 'Hello World!'

num = int(raw_input())

for _ in range(num):
	word = raw_input()
	word_map = {}
	word_map[word[0]] = 0
	doubleStr = False
	for letter in word[1:]:
		if letter in word_map:
			print "Yes"
			doubleStr = True
			break
		else:
			word_map[letter] = 0
	if not doubleStr:
		print "No"
