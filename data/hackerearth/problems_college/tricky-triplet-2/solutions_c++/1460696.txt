#include<stdio.h>
#include<stdlib.h>
int main()
{
	int n, count[30003]={0}, distance[30003]={0}, index, val, mRange, i;
	unsigned long long ans = 0ull;
	scanf("%d", &n);
	for (index = 0; index < n; index++)
	{
		scanf("%d", &val);
		count[val]++;
		ans += distance[val];
		mRange = (val < 15001)?val:(30002 - val);
		for (i = 1; i <= mRange; i++)
		{
			distance[val + i] += count[val - i];
			distance[val - i] += count[val + i];
		}
	}
	for (index = 0; index < 30003; index++)
	{
		ans+=((unsigned long long)count[index]*((unsigned long long)count[index]-1)*((unsigned long long)count[index]-2))/6ull;
	}
	printf("%llu\n", ans);
	return 0;
}
