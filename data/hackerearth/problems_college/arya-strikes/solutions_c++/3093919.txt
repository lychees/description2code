
#include<iostream>
#include<algorithm>
using namespace std;

int arya[51][51];

int findout(int ,int ,int );

int bt(int ,int ,int );

int main(){
	
    int k,n,no_testcases;
        
        cin>>no_testcases;
	
        for(k=0;k<no_testcases;k++){
          
            for(int h=0;h<55;h++)
                
                for(int g=0;g<55;g++)
                    
                    arya[h][g]=0;
		
		cin>>n;
                
		 for(int h=0;h<n;h++)
                
                     for(int g=0;g<n;g++)
                       
                    
                       cin>>arya[h][g];
		
                 if(!bt(0,0,n))
                     
                     cout<<"NOT POSSIBLE"<<endl;
                 
                 else
			
                     cout<<"POSSIBLE"<<endl;
		 
	}
	return 0;
} 

int bt(int x,int y,int n){
	
    if (y==n-1 && x==n-1 )
            
            return 1;
	
        if(findout(x,y,n)){
		
            if(bt(x+1,y,n)) 
		
                    return 1;
	
                if(bt(x,y+1,n))
		
                    return 1;
		
                else return 0; 
	}
        
	return 0; 
}

int findout(int x,int y,int n){
	
    if(! (x>=0 && y>=0 && x<n && y<n && arya[x][y]==1))
            
            return 0;
	
        else 
            
            return 1;
}
