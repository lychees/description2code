#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define MOD 1000000007

int main()
{
	#ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int T,N;
    ll K;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d%lld",&N,&K);
        vector<ll> A(N),PrefixSum(N);
        for(int i = 0; i < N; i++) scanf("%lld",&A[i]);
        PrefixSum[0] = A[0];
        for(int i = 1; i < N; i++) PrefixSum[i] = PrefixSum[i-1] + A[i];
        int left = 0,right = 0;
        vector<ll> ans;
        map<int,int> M;
        while(left < N)
        {
            ll a = (left - 1 >= 0 ? PrefixSum[left - 1] : 0LL);
            while(right < N && PrefixSum[right] - a <= K) right++;
            int n = right - left;
            M[n]++;
            left++;
        }
        /***
        for(int i = 1; i <= N; i++)
        {
            cout << M[i] << " ";
        }
        cout << "\n";
        ***/
        for(int i = N; i >= 1; i--)
        {
            if(M[i] && M[i+1])
            {
                ans.push_back(M[i] + M[i+1]);
                M[i] += M[i+1];
            }
            else if(M[i]) ans.push_back(M[i]);
            else ans.push_back(0);
        }
        reverse(ans.begin(),ans.end());
        for(vector<ll>::iterator it = ans.begin(); it < ans.end(); it++)
        {
            cout << (*it) << " ";
        }
        cout << "\n";
    }
    #ifndef ONLINE_JUDGE
    cerr << "Time elapsed: " << 1.0 * clock() / CLOCKS_PER_SEC << " s.\n";
    #endif
    return 0;
}
