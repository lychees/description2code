#include<iostream>
#include <cstdio>
#include <iomanip>
using namespace std;

int main()
{
    int t,i,j;
    double a=1.0,p=1.0,s[2005],n1,r;

    cin >> t;

    for(i=0;i<t;i++)
    {
                    cin >> n1;
                    cin >> r;
                    a = 6.0/(n1*(n1+1)*(2*n1+1));
                    s[i] = a * (n1*n1*(n1+1)*(n1+1))/(4*r);
    }
    for(i = 0;i<t;i++)
    {
          p = p *s[i];
    }

   // cout <<setprecision(4) << p << endl;
    printf("%.4lf\n", p);
    return 0;
}
