#include<bits/stdc++.h>

using namespace std;

int funct(long long int Z , long long int M , long long int N)
{
	long long int n1,n2,ways = 0;
	int sq = sqrt(Z);
	for(n1=1;n1<=sq;n1++){
		n2=Z/n1;
		if((n2*n1)==Z){
            if(n2<=M){
                ways++;
            }
            if(n2<=N && n1!=n2){
                ways++;
            }
        }
	}
	return ways;
}

int main()
{
	int T;
	long long int M,Z,N;
	cin>>T;
	while(T--){
		cin>>Z>>N>>M;
		cout<<funct(Z,M,N)<<endl;
	}
	return 0;
}
