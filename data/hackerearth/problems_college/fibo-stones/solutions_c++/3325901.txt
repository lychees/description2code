#include<bits/stdc++.h>
using namespace std;

const long long N=105;

long long fibo[N];

void preCompute()
{
  fibo[1]=1;
  fibo[2]=1;
  for(int i=3;i<N;i++) {
    fibo[i]=fibo[i-1]+fibo[i-2];
  }
}
int main() {
 
  long long n,m,x,y;
  long long ans=0;
  preCompute();
  cin>>n>>m;
  map<long long,long long> mp1;
  map<long long,long long> mp2;
  map<long long,long long> :: iterator it;
  for(int i=1;i<=n;i++) {
    mp1[fibo[i]]++;
  }
  for(int i=1;i<=m;i++) {
    cin>>x;
    mp2[x]++;
  }
 
  for(it=mp1.begin();it!=mp1.end();it++) {
    ans+=it->first*(it->second-mp2[it->first]);
  }
  cout<<ans<<endl;
  
}
