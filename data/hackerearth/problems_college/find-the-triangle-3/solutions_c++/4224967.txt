#include <bits/stdc++.h>
using namespace std;
vector<int>v;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
	int n,temp;
	scanf("%d",&n);
	v.clear();
	for(int i=0;i<n;i++)
	{
		scanf("%d",&temp);
		v.push_back(temp);
	}
	sort(v.begin(),v.end());
	int flag=1;
	for(int i=n-1;i>=2;i--)
	{
		if(v[i]<v[i-1]+v[i-2])
		{
			printf("%d %d %d\n",v[i-2],v[i-1],v[i]);
			flag=0;
			break;
		}
	}
	if(flag)
	printf("-1\n");
	}
    return 0;
}
