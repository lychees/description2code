#include <iostream>
using namespace std;
int sigma (int i){
	return (i+1) * i/2;
}

int main(){
	int tC, count, battle, day;
	cin >> tC;
	while(tC--){
		count = 1;
		battle = 1;
		cin >> day;
		while(count < day){
			battle++;
			count = sigma(battle);
		}
		if(battle%3 == 0)
			cout << "SL" << endl;
		else if (battle % 3 == 1)
			cout << "LB" << endl;
		else
			cout << "BS" << endl;
		
	}
    return 0;
}
