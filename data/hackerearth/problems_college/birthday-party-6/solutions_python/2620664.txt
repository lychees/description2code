	M, N = int(M), int(N)
	if N >= M + 2:
		print -1
	else:
		res_num = pow(math.factorial(M),2) * (M+1)
		if N >= M + 1:
			den = 1
		else:
			den= math.factorial(M + 1 -N)
		res = res_num / den
		if res > 1000000007:
			res = res % 1000000007
