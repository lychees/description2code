#include<iostream>
#include<vector>
#define MIN(a,b)  ((a)<(b)?(a):(b))
using namespace std;
int main()
 {
 	int T;
 	cin>>T;
 	while(T--)
 	 {
 	 	int N;
 	 	cin>>N;
 	 	vector<vector<int> > shops(N,vector<int>(3));
 	 	for(int i=0;i<N;i++)
 	 	 cin>>shops[i][0]>>shops[i][1]>>shops[i][2];
 	 	for(int i=1;i<N;i++)
 	 	 {
 	 	   shops[i][0]=shops[i][0]+MIN(shops[i-1][1],shops[i-1][2]);
		   shops[i][1]=shops[i][1]+MIN(shops[i-1][0],shops[i-1][2]);
		   shops[i][2]=shops[i][2]+MIN(shops[i-1][1],shops[i-1][0]);	
		   }
	   cout<<MIN(shops[N-1][0],MIN(shops[N-1][1],shops[N-1][2]))<<endl;
	  }
	return 0;
 }
