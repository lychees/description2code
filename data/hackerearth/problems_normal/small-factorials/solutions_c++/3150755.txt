#include <iostream>
using namespace std;

int main()
{
    int tc,n;
    int a[200];
    cin>>tc;
    while(tc--)
    {
    	cin>>n;
    	for(int i=0;i<200;i++)
    	  a[i]=-1;
    	a[0]=1;
    	for(int i=n;i>1;i--)
    	{
    		int j=0;
    		while(a[j] != -1)
    		{
    			a[j]=a[j]*i;
    			j++;
    		}
    		int t=a[0];
    		j=0;
    	 while(t || a[j]!= -1)
    	     {
    	     	if(j==0) {
    	     		a[j]=t%10;
    	     		t=t/10;
    	     		j++;
    	     	}
    	     	else{
    	     		  if(a[j]==-1) a[j]=0;
    	     		  t=t+a[j];
    	     		  a[j]=t%10;
    	     		  t=t/10;
    	     		  j++;
    	     	    }
    	     
    	     }
    	}
    	for(int i=199;i>=0;i--)
    	  if(a[i] != -1) cout<<a[i];
    	  cout<<endl;
    }
    return 0;
}
