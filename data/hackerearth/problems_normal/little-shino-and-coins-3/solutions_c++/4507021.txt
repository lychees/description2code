#include <iostream>
#include <string>
using namespace std;
int main()
{
    int c;
    cin>>c;
    string s;
    cin>>s;
    int a=0;
    for(int i = 0;i<s.size();i++)
    {
        bool b[26];
        int sC=0;
        for(int l=0;l<26;l++)
        {
            b[l]=false;
        }
        for(int j =i;j<s.size();j++)
        {
            if(b[s[j]-'a'] == false)
               sC++;
            b[s[j]-'a'] = true;
            if(sC == c)
            {
                a++;
            }
        }
    }
    cout<<a<<endl;
}
