#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <iostream>
using namespace std;

long long int count(char s[1000001],char x,char y)
{
  long long int c = 0,c2=0,i;
  long long int sc = 0,sc2=0;

  for(i = 0;i < strlen(s);i++)
  {
     if(s[i] == x) c++;
     if(s[i] == y) sc += c;
     
     if(s[i]==y) c2++;
     if(s[i]==x) sc2+=c2;
  }
  return sc+sc2;
}


int main()
{
	long long int l;
	scanf("%lld",&l);
	
		char s[1000001];
		long long int q;
		
		cin >> s;
		
		cin >> q;
		
		long long int a[26][26];
		for(long long int i=0;i<26;i++)
		{
			for(long long int j=0;j<26;j++)
			{
				a[i][j]=-1;
			}
		}
	
		while(q-- > 0)
		{
			char x,y;
			
			cin >> x;
			cin >> y;
			int u,v;
			u=(int)x-97;
			v=(int)y-97;
			
			if(a[u][v]!=-1)
			{
				printf("%lld\n",a[u][v]);
			}
			
			
			else
			{
				if(x!=y)
				{
					a[u][v]=count(s,x,y);
					a[v][u]=a[u][v];
				}
				
				else
				{
					long long int z=0;
					for(long long int i=0;i<strlen(s);i++)
					{
							if(s[i]==x)
							{
								z++;
							}
					}
					
					a[u][v]=(z*(z-1))/2;
					
					a[v][u]=a[u][v];
				}
				
				printf("%lld\n",a[u][v]);
		
			}
		}
		
	
	return 0;
}
