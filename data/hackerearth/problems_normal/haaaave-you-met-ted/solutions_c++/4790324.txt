#include <iostream>
#include <climits>
using namespace std;

int main()
{
	int t, i;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	long long int n, j, m, i, min=LLONG_MAX, count=0;
    	cin>>n;
    	if(n==1)
    	{
    		cout<<"1\n";
    	}
    	else
    	{
    		for(j=0;j<n;j++)
    		{
    			count=0;
		    	cin>>m;
		    	while(m)
	    		{
	    			m&=m-1;
	    			count++;
	    		}
	    		if(count<min)
	    		{
	    			min=count;
	    		}
    		}
    		cout<<min<<"\n";	
    	}	
    }
    return 0;
}
