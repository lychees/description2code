#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
struct point{
	int x,y;
};
bool sort_p(const point &a, const point &b)
{
	if(a.y<b.y)
		return true;
	else if(a.y>b.y)
		return false;
	else if(a.x<b.x)
		return true;
	else
		return false;
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int t[n],d[n];
    for(int i=0;i<n;i++)
    	cin>>t[i]>>d[i];
    int a[n];
    for(int i=0;i<n;i++)
    {
    	a[i]=t[i]+d[i];
    	//cout<<t[i]<<" "<<d[i]<<endl;
    }
    vector<point> points;
    for(int i=0;i<n;i++)
    {
    	point p;
    	p.x = i+1;
    	p.y = a[i];
    	//cout<<p.x<<" "<<p.y<<" "<<a[i]<<endl;
    	points.push_back(p);
    }
    sort(points.begin(),points.end(),sort_p);
    for(int i=0;i<n;i++)
    {
    	cout<<points.at(i).x<<" ";
    }
    return 0;
}
