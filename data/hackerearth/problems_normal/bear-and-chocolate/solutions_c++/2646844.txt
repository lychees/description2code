#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int T;
    cin>>T;
    
    while(T--){
       bool result=false;
       int cherry=0;
    int n;
    	cin>>n;
    	char s[n][n];
    	
    	for(int i=0;i<n;i++){
    	       for(int j=0;j<n;j++)
    	          cin>>s[i][j];
    	}
    	
    
       
       
      for(int i=0;i<n;i++){
       	    for(int j=0;j<n;j++){
       	    	  if(s[i][j]=='#'){
       	    	       cherry++;
       	    	     }
       	    }
       }
     //  cout<<"Number of cherry::"<<cherry<<endl;
    if(cherry %2 == 0){
     int row=0,col=0;
     for(int i=0;i<n;i++){
     	 for(int j=0;j<n;j++){
     	 	 if(s[i][j]=='#')
     	 	   row++;
     	 }
     	if((row*2)==cherry){
     		result = true;
     		break;
     	}
     else if((row*2)>cherry)
    	 break;
     	 
     }
    
    for(int i=0;i<n;i++){
     	 for(int j=0;j<n;j++){
     	 	 if(s[j][i]=='#')
     	 	   col++;
     	 }
     	if((col*2)==cherry){
     		result = true;
     		break;
     	}
     else if((col*2)>cherry)
     break;
     	 	
     }
    
    }
    
    if (cherry %2 != 0)
        result = false; 
      if(result == true )
           cout<<"YES"<<endl;
      else
          cout<<"NO"<<endl;
    }
    
    return 0;
}
