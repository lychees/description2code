#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int i,t,n,m,sum,cnt;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        long long a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        sum=0;
        cnt=0;
        for(i=n-1;i>=0;i--)
        {
            if(sum+a[i]>=m)
            {
                sum+=a[i];
                cnt++;
                break;
            }
            sum+=a[i];
            cnt++;
        }
        if(sum<m)
        	cnt=-1;
        cout<<cnt<<"\n";
    }
    return 0;
}
