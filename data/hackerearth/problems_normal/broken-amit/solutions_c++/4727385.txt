

    #include <bits/stdc++.h>
    #define INF 100000000000000LL
    using namespace std;
    int n;
    long long A[22];
    long long B[22];
    bool vis[20][1<<20];
    long long dp[20][1<<20];
    long long f(int idx, int mask)
    {
    	if ( idx == n ) return 0;
    	if ( vis[idx][mask] ) return dp[idx][mask];
    	vis[idx][mask] = true;
    	long long ans = -INF;
    	for ( int i = 0; i < n; i++ ) {
    		if ( mask&(1<<i) ) continue;
    		long long val = 0,val1 = 0;
    		val = (long long)idx*(long long)B[idx];
    		val = val*(long long)i*(long long)(i-1);
    		val /= 2;
    		val = (long long)(i)*A[idx]-val;
    		val1 = A[idx] - B[idx]*(long long)idx*(long long)i;
    		val1 = val1*(long long)(n-i);
    		val += val1;
    		ans = max(ans, val + f(idx+1,mask|(1<<i)));
    	}
    	dp[idx][mask] = ans;
    	return ans;
    }
    int main()
    {
    	cin >> n;
    	for ( int i = 0; i < n; i++ ) {
    		cin >> A[i];
    	}
    	for ( int i = 0; i < n; i++ ) {
    		cin >> B[i];
    	}
    	long long ans = f(0,0);
    	cout << ans << endl;
    	return 0;
    }
