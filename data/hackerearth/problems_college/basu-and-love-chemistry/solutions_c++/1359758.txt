#include <cstdio>
#include <cstring>
#include <algorithm>
#define MOD 1000000007
#define MX 200005
using namespace std;
typedef long long LL;
LL P[MX], S[MX], ans;
char buf[MX];

int main() {
	scanf("%s", buf);
	int N = strlen(buf), i;
	P[0] = 1;
	S[0] = 1;
	for (i = 1; i < N; i++) {
		P[i] = P[i - 1] * 10 % MOD;
		S[i] = (S[i - 1] + P[i]) % MOD;
	}
	
	for (i = 0; i < N; i++) {
		ans += (i + 1) * S[N - i - 1] * (buf[i] - '0') % MOD;
	}
	printf("%d\n", int(ans % MOD));
	return 0;
}
