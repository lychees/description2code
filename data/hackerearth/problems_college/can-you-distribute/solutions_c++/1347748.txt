#include <iostream>
using namespace std;
int main() {
	int n,r;
	cin>>n>>r;
	if(r>n)
	  cout<<"-1"<<endl;
	else
	 {
	 	long long ans1=1;
	 	if(r<(n/2)){
	 	for(long long int i=n-r+1;i<=n-1;i++)
	 	  ans1=(ans1*i);
	 	//cout<<ans1;
	 	long long ans2=1;
	 	for(long long int j=1;j<=r-1;j++)
	 	   ans2=(ans2*j);
	 	   //cout<<ans2;
	 	 long long ans=(ans1/ans2)%10000007;
	 	 cout<<ans<<endl;
	 	}
	 	else
	 	{
	 	 for(long long int i=r;i<=n-1;i++)
	 	  ans1=(ans1*i);
	 	//cout<<ans1<<endl;;
	 	long long ans2=1;
	 	for(long long int j=1;j<=n-r;j++)
	 	   ans2=(ans2*j);
	 	   //cout<<ans2<<endl;;
	 	 long long ans=(ans1/ans2)%10000007;
	 	 cout<<ans<<endl;
	 	}
	 	    
	 	
	 }
	// your code goes here
	return 0;
}
