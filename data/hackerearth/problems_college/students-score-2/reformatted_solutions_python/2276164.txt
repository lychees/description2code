def maxSubArraySum(s, size):
    max_so_far = 0
    max_ending_here = 0
    s = map(int, s.split())
    for i in range(size):
        max_ending_here = max_ending_here + s[i]
        if (max_ending_here < 0):
            max_ending_here = 0
        if (max_so_far < max_ending_here):
            max_so_far = max_ending_here
    return max_so_far


n = int(raw_input())
s = raw_input()
print maxSubArraySum(s, n)
