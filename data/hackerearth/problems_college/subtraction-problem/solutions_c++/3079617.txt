#include<iostream>
using namespace std;
main()
{
    int a,b,sub,temp=0,cnt=0,sum=0;
    cin>>a;
    cin>>b;
    sub=a-b;
    
    while(sub>0)
    {
        temp=temp*10+sub%10;
        sub=sub/10;
        if(temp>=0&&temp<9&&cnt==0)
        {
            temp=temp+1;
            cnt++;
        }
        else if(temp==9)
        {
            temp=temp-1;
            cnt++;
        }
    }
    while(temp>0)
    {
        sum=sum*10+temp%10;
        temp=temp/10;
    }
    cout<<sum<<endl;
}
