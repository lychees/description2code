import math
noOfTestCases = int(raw_input())


def decode(message):
    lenOfString = len(message)
    squareRoot = int(math.sqrt(lenOfString))
    newString = ""
    outerStart = 1
    while (outerStart <= squareRoot):
        innerStart = outerStart - 1
        counter = 1
        while (counter <= squareRoot):
            # print innerStart
            newString += message[innerStart]
            innerStart += squareRoot
            counter += 1
        outerStart += 1

    print newString


for caseNo in range(1, noOfTestCases + 1):
    decode(raw_input())
