#include <iostream>
using namespace std;
long long fun(long long a, long long b) ;
    

int main()
{
    int t;
    long long int a,b,ans;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	ans=fun(a,b);
    	cout<<ans<<"\n";
    }
    return 0;
}
long long fun(long long a, long long b)
{
    long long x = a^b;
    long long s = x>>1;

    while (s) 
    {
        x = x|s;
        s >>= 1;
    }
    return a&b&~x;
}
