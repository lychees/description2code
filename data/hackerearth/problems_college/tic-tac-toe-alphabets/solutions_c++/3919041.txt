#include<iostream>
#include<string.h>
using namespace std;
void insertion_sort (char arr[], int length)
{
	 	int j;
	 	char temp;
		
	for (int i = 0; i < length; i++)
	{
		j = i;
		
		while (j > 0 && arr[j] < arr[j-1])
		{
			  temp = arr[j];
			  arr[j] = arr[j-1];
			  arr[j-1] = temp;
			  j--;
		}
	}
	
}
int main()
{
	int n=0;
	char str[10000];
	cin>>n;
	for(int i=0;i<=n;i++)
	{
		gets(str);
		int l1=strlen(str);
		insertion_sort(str,l1);
		cout<<str<<endl;
	}
}
