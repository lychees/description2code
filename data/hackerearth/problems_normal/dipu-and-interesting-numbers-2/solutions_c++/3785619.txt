#include <bits/stdc++.h>

#define MAX 1000005
#define INF 1111111111

typedef long long ll;
typedef unsigned long long llu;

using namespace std;

ll __sqrt(ll lo, ll hi, ll x)
{
    ll mid=(lo+hi)/2;
    if(lo>hi)return mid;
    if((mid*mid)==x)return mid;
    if((mid*mid)>x)__sqrt(lo,mid-1,x);
    else if((mid*mid)<x)__sqrt(mid+1,hi,x);
}

int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    ll t,n,m;
    scanf("%lld",&t);
    for(ll i=1;i<=t;i++){
        scanf("%lld %lld",&n,&m);
        ll res=__sqrt(0,1e9,m)-__sqrt(0,1e9,n-1);
        printf("%lld\n",res);
    }
    return 0;
}
