#include<bits/stdc++.h>
using namespace std;

struct data{
	double x;
	int ind;
};

bool comp(const struct data &a,const struct data &b) {
	if(a.x < b.x) {
		return true;
	} else if(a.x == b.x) {
		if(a.ind < b.ind) {
			return true;
		} else {
			return false;
		}
	}
	return false;
}

int main() {
	int n,m;
	cin>>n>>m;
	struct data a[m+1];
	double ans = (double)(m+1)/(double)2;
	for(int i = 1 ; i <= m; i++) {
		a[i-1].ind = i;
		a[i-1].x = abs(ans-double(i));
	}
	sort(a,a+m,comp);
    for(int i = 0 ; i < n; i++) {
    	cout<<a[i%m].ind<<"\n";
    }
	return 0;
}
