#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
int n, arr[100008];
int main()
{
    int t;
    scanf ("%d", &t);
    while (t--) {
    	int n;
    	scanf ("%d", &n);
    	for (int i = 0; i < n; i++) {
    		scanf ("%d", &arr[i]);
    	}
    	sort (arr, arr + n);
    	long long ans = 0;
    	for (int i = 0; i < n; i++) {
    		ans += abs (arr[i] - arr[n / 2]);
    	}
    	printf ("%lld\n", ans);
    }
    return 0;
    	
    	
}
