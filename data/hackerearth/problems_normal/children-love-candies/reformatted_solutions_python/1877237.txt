from __future__ import division
q = int(raw_input())
while q > 0:
    q -= 1
    n, t = map(int, raw_input().split())
    c = n
    for i in range(t):
        if (c % 2 == 0):
            x = c/2
        else:
            x = (c+1)/2
        c -= x
        x = (c//4)
        c -= x
    print int(c)
