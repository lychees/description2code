T = int(raw_input())
X = [None for i in range(2*(10**6))]
j = 1
for i in range(1, 10**6+4):
    if X[i] is None:
        X[i] = i + j
        X[i+j] = i
        j += 1
# print X[:10]
for i in range(T):
    N1, N2 = [int(i) for i in raw_input().split()]
    if N1 > N2:
        temp = N2
        N2 = N1
        N1 = temp
    if N2 == 0:
        print "Don't Play"
    elif N1 != N2 and N1 != 0 and X[N1] == N2:
        print "Don't Play"
    else:
        print "Play"
