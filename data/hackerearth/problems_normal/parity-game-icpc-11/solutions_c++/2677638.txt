#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
typedef long long LL;
typedef pair<int, int> PII;

int n, q;
unsigned a[60000][13] = {};

int main() {
    scanf("%d%d", &n, &q);
    REP(i, q) {
        int from, to;
        scanf("%d%d", &from, &to), --from, --to;
        for (int j = from; j <= to; ++j) {
            a[i][j >> 5] |= 1 << (j & 31);
        }
    }
    if (q < n) {
        printf("0\n");
        return 0;
    }
    int ans = 1234567;
    REP(i, n) {
        int cnt = 0;
        for (int j = i; j < q; ++j) if (a[j][i >> 5] & (1 << (i & 31))) {
            if (++cnt == 1) {
                REP(k, 13) swap(a[i][k], a[j][k]);
            } else {
                REP(k, 13) a[j][k] ^= a[i][k];
            }
        }
        if (cnt == 0) {
            printf("0\n");
            return 0;
        }
        ans = min(ans, cnt);
    }
    printf("%d\n", ans);
    return 0;
}
