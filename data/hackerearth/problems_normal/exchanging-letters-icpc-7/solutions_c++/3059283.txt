#include <iostream>
#include <vector>
using namespace std;
long long int mod=1000000007;
vector<vector<long long int> > v1;
int main()
{
    long long int n,k,t,p1,p2,sum=0;
    v1.resize(2000+1);
    for(p1=0;p1<=2000;p1++)
    {
    	v1[p1].resize(2000,0);
    	v1[p1][0]=1;
    }
    for(p1=1;p1<=2000;p1++)
    {
    	for(p2=1;p2<=p1;p2++)
    		v1[p1][p2]=(v1[p1-1][p2]%mod+v1[p1-1][p2-1]%mod)%mod;
    }
    cin>>n>>k>>t;
    for(p1=0;p1<k;p1++)
    {
    	cin>>p2;
    	sum+=p2;
    }
   // p2=0;
    //cout<<sum<<"\n";
    //cout<<sum-k<<"-->"<<sum+k;
    if(t<sum-k)
    	cout<<v1[0][1];
    else if(t>sum+k)
    	cout<<v1[0][1];
    else if((t-sum+k)%2==0)
    {
    	p1=(t-sum+k)/2;
    	//p2=k-p1;
    	cout<<v1[k][p1];
    }
    else
    {
    	cout<<0;
    }
    return 0;
}
