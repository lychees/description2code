#include <iostream>
#include <string>
using namespace std;

int main()
{
    string str;

	cin>>str;

	int len = str.length();
	int pairNo = 0;

	cin>>pairNo;

	for (int i = 0; i < pairNo; i++)
	{
		long a1=0;
		long a2=0;

		cin>>a1;
		cin>>a2;
		a1=a1%(len);
			
		if(a1==0){
				a1=len;
		}
			
		a2=a2%(len);
			
		if(a2==0){
				a2=len;
		}
		//cout<<a1<<" "<<a2<<endl;
		if(str.at(a1-1)==str.at(a2-1)){
			cout<<"Yes"<<endl;
		}
		else{
			cout<<"No"<<endl;
		}



	}
	

    return 0;
}
