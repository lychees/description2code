#include<iostream>
using namespace std;

bool dp[102][100002];
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);

    int n;
    cin>>n;

    int arr[n+1],sum=0;
    for(int i=1;i<=n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }

    for(int i=0;i<=sum;i++)
        dp[0][i]=0;

    for(int i=0;i<=n;i++)
        dp[i][0]=1;

    for(int i=1;i<=n;i++)
        for(int j=1;j<=sum;j++)
        {
            if(arr[i]>j)
                dp[i][j]=dp[i-1][j];

            else
                dp[i][j]=max(dp[i-1][j],dp[i-1][j-arr[i]]);
        }

    int t;
    cin>>t;

    while(t--)
    {
        int z;
        cin>>z;

        if(z>sum)
            cout<<"No\n";

        else
        {
            if(dp[n][z])
                cout<<"Yes\n";

            else
                cout<<"No\n";
        }
    }
}
