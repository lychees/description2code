#include <iostream>
using namespace std;
int heap[1000000];

void restore_up(int i){
	
	int value=heap[i];
	while(i!=1 && value > heap[i/2]){
		heap[i]=heap[i/2];
		i/=2;
	}
	heap[i]=value;
}

void restore_down(int m){
	--heap[1];
	int value=heap[1];
	int i=1,j=2*i;
	
	while(j<=m ) 
	{
		if((j<m) && (heap[j] < heap[j+1]))
			j++;
		if(heap[j] < heap[j/2])  //checking parent
			break;
		
		heap[j/2]=heap[j];     //child data to parent
		j=j*2;
		
		
	}
	heap[j/2]=value;
}
int main()
{
    int m,n;
    cin>>m;
    cin>>n;
    for(int i=1;i<=m;i++){
    	cin>>heap[i];
    	restore_up(i);
    	
    }
    long long int sum=0;
    while(n>0){
    	
    	sum=sum+heap[1];
    	restore_down(m);
    	n--;
    }
   cout<<sum;
    return 0;
}
