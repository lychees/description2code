#include <iostream>
#include <algorithm>
using namespace std;
long long int a[10000000],b[10000000];
int main() {
long long 	int t;
long long 	int p,q,k;
	cin>>t;
long long 	int y=1;
	while(y<=t)
	{
		cin>>p>>q>>k;
		
	long long 	int i=0,j=0;
		a[0]=0;
		while(a[i]<=k)
		{
			i++;
		long long 	int s=1;
			for(int u=1;u<=p;u++)
			s=s*i;
			a[i]=s+i;
		}
		b[0]=k;
		while(b[j]>=0)
		{
			j++;
			long long 	int s=1;
			for(int u=1;u<=q;u++)
			s=s*j;
			b[j]=k-(s);
		}
		
		sort(b,b+j);
	long long 	int ans=0;
		for(int kk=0;kk<i;kk++)
		{
	long long 		int low=lower_bound(b,b+j,a[kk])-b;
	long long 	int high=upper_bound(b,b+j,a[kk])-b;
		ans+=(high-low);
		}
		cout<<"Case #"<<y<<": "<<ans<<endl;
		y++;
		
	}
	return 0;
}
