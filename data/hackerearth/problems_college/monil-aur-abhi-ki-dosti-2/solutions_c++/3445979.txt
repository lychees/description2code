#include <bits/stdc++.h>
 
#define LLI long long int
#define LD long double
#define PB push_back
#define MP make_pair
#define FORi(i, a, b) for(int i = a; i < b ; ++i)
#define FORd(i, a, b) for(int i = a; i > b ; --i)
 
using namespace std;
 
const int mod = 1000000000;
 
int main(){
	int t;
	string s,p;
	bool isp;
	int a[26];
	cin >> t;
	while(t--){
		cin >> p >> s;
		FORi(i,0,26)
			a[i]=0;
		FORi(i,0,p.size())
			a[p[i]-'a']++;
		FORi(i,0,s.size()){
				a[s[i]-'a']--;
				if (i>=p.size())
					a[s[i-p.size()]-'a']++;
				isp = 1;
				FORi(i,0,26){
					if (a[i]!=0){
						isp = 0;
						break;
					}
				}
				if (isp){
					cout << "YES" << endl;
					break;
				}
		}
		if (!isp)
			cout << "NO" << endl;
	}
    return 0;
}
