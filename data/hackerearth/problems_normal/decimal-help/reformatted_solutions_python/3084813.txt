def decimal(x, y, r):
    count = 0
    remainder = x % y
    quotient = 0
    while count < r and remainder != 0:
        remainder = remainder*10
        quotient = remainder/y
        remainder = remainder % y
        count += 1
    if count < r:
        print 0
    else:
        print quotient


for t in xrange(input()):
    x = [int(i) for i in raw_input().split()]
    decimal(x[0], x[1], x[2])
