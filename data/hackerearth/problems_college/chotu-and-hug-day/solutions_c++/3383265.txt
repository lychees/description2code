#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t ,n, len;
	scanf("%d",&t);
	while(t--)
	{
	scanf("%d %d",&n ,&len);
	int a[n], i , j , k,m;
	for(i=0; i<n ;i++)
	scanf("%d",&a[i]);
	
	sort(a,a+n);
	if(a[n-1]<len)
	{
	printf("%d\n",(len-a[n-1]));
	continue;
	}
	
	for(i=0; i<n ;i++)
	{
		
		if((len-a[i])==0)
		{
		printf("%d\n",(len-a[i]));
		break;
		}
		if((len-a[i])<0 && i!=0)
		{
			m=(a[i]-len);
			k=(len-a[--i]);
		if(m==k)
		{
			printf("%d\n",m);
			break;
		}
			if(m>k)
		{
			printf("%d\n",k);
			break;
		}
			else
		{
			printf("%d\n",m);
			break;
		}
		}
			if((len-a[i])<0 && i==0)
		{
			printf("%d\n",(a[i]-len));
			break;
		}
		
	}
	}
	}
