#include <iostream>
using namespace std;

int top = -1;

bool isEmpty(){
	if(top == -1)
		return true;
	else
		return false;
}

int pop(int stack[]){
	if(!isEmpty()){
			return stack[top--];
	}
	else
		return 0;
}

void push(int stack[],int rating){
	stack[++top]=rating;
}

int main()
{
    int persons,rating,totalRating=0;
    int stack[persons];
    cin>>persons;
    while(persons--){
    	cin>>rating;
    	if(rating == 0)
    		totalRating -= pop(stack);
    	else{
    		push(stack,rating);
    		totalRating += rating;
    	}
    		
    }
    cout<<totalRating;
    return 0;
}
