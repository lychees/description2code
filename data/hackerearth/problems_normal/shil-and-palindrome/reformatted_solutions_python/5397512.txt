s = raw_input()
alphabet = [chr(i + ord('a')) for i in xrange(26)]
f = dict(zip(alphabet, [0]*26))
for c in s:
    f[c] += 1
output = ''
for c in alphabet:
    while f[c] > 1:
        output += c
        f[c] -= 2
mid = []
for k, v in f.iteritems():
    if v == 1:
        mid.append(k)
if len(mid) > 1:
    print -1
else:
    print output + ''.join(mid) + output[::-1]
