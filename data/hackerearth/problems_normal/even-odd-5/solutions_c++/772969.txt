#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	long int r,c;
    	cin >> r >> c;
    	
	
		long int n_odd=0,sum_odd=0;
		long int n_even=0,sum_even=0;
	   if(r<=c){

    		if(r%2==0){
    			n_even = (r-2)/2+1;
    			n_odd = (r-2)/2+1;
    		}else{
    			n_even = (r-3)/2+1;
    			n_odd = (r-1)/2+1;
    		}
    //    		cout << n_even << " " << n_odd << endl;
    		
    		sum_even = n_even*(n_even+1);
    		sum_odd = n_odd*n_odd;
        	cout << sum_even << " " << sum_odd << endl;  
        	}else{

                if(c%2==0){
                    n_even = (c-2)/2+1;
                    n_odd = (c-2)/2+1;
                }else{
                    n_even = (c-3)/2+1;
                    n_odd = (c-1)/2+1;
                }
                // cout << n_even << " " << n_odd << endl;
                                           
                sum_even = n_even*(n_even+1);
                sum_odd = n_odd*n_odd;
                // cout << sum_even << " " << sum_odd << endl;
                
                if(c&1 && r&1 || !(c&1) && !(r&1)){
                    // cout << " 1 " << endl;
                    n_odd = n_even = (r-c)/2;
                    // cout << n_even << endl;

                    sum_even += n_even*c;
                    sum_odd += n_odd*c;
                }

                if(c&1 && !(r&1)){
                    // cout << " 2 " << endl;
                    n_even = (r-c)/2+1;
                    n_odd = n_even-1;
                    sum_even += n_even*c;
                    sum_odd += n_odd*c;
                }

                if(!(c&1) && r&1){
                    // cout << " 3 " << endl;
                    n_odd = (r-c)/2+1;
                    n_even = n_odd-1;
                    // cout << n_odd << endl;
                    sum_even += n_even*c;
                    sum_odd += n_odd*c;

                }

        		cout << sum_even << " " << sum_odd << endl;
        	}
    }
    return 0;
}
