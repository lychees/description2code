#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int power1(long long int a,long long int b)
{
	long long int res=1;
	while(b)
	{
		if(b&1) res = res * a % mod;
		a = a * a % mod;
		b /= 2; 
	}
	return res;
}
int main()
{
	int t;
	cin >> t;
	while( t-- ){
	long long int n; cin >> n;	
	long long int res = power1(2, n-1);
	res = power1(res,mod-2);
	cout << res << endl;
   }
   return 0;
}
