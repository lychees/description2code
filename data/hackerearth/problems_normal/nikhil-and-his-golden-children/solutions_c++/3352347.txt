#include <bits/stdc++.h>

typedef long long int ll;
#define mod 1000000007
#define S(x) scanf("%lld",&x)
using namespace std;

ll powr (ll a, ll b)
{
    if (b == 0)
        return 1;
    ll x = powr(a, b/2);
    if (b % 2 == 0)
        return (x*x)%mod;
    else
        return (((x*x)%mod)*a)%mod;
}

int main()
{
    ll t,n,m,r,i,x,ans;
    S(t);
    while(t--) {
        S(n);
        S(m);
        ans = 1;
        ll sum = 0;
        for(i = 0; i < n ; i++) {
            S(x);
            sum += x;
        }
        m -= sum;
        if(m < 0) {
            printf("0\n");
            continue;
        } else if(m == 0) {
            printf("1\n");
            continue;
        }
        swap(n,m);
        n = n+m-1;
        r = m-1;
        r = min(n-r,r);
        for(i = 1; i <= r; i++) {
            ans *= (n*powr(i,mod-2))%mod;
            ans %= mod;
            n--;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
