#include <iostream>
#include<cassert>
using namespace std;

int main()
{
    int Q;
    cin>>Q;
    long long N;
    int T;
    while(Q--)
    {
    	cin>>N>>T;
    	assert(N>=0&&N<=10000000000000000LL);
    	assert(T>=0&&T<=50);
    	while(T--)
    	{
    		if(N%2==0)
    		{
    			N=N-N/2;
    		}
    		else
    		{
    			N=N-(N+1)/2;
    		}
    		N=N-N/4;
    	}
    	cout<<N<<endl;
    }
    return 0;
}
