#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,m,n,i,j,x;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&m,&n);
    	int rem[101];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&rem[i]);
    	}
    	sort(rem,rem+n);
    	
    	bool dp[10005];
    	for(i=0;i<=m;i++)
    	dp[i]=0;
    	for(i=1;i<=m;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			if(i-rem[j]>=0)
    			{
    				if(dp[i-rem[j]]==0)
    				{
    				dp[i]=1;
    				break;
    				}
    			}
    			else break;
    		}
    	}
    	if(dp[m])
    	printf("Little Deepu\n");
    	else printf("Kate\n");
    }
    return 0;
}
