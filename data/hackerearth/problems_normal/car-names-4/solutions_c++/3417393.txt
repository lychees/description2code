#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
main()
{
    int a,end=0,y;
    string str;
    cin>>a;
    while(a>0)
    {
        if(end > 0) cout<<endl; end++;
        cin>>str;
        vector<char> dis;
        if(str.size()%3 != 0) {cout<<"Not OK"; goto fin;}
        else
        {
            for(int x=0; x<str.size(); x++)
            {
                char let = str[x];
                for(y=x+1; y<str.size()/3; y++)
                {
                    if(str.size()/3 == 1) continue;
                    if(let != str[y]) {cout<<"Not OK"; goto fin;}
                }
                if(count(dis.begin(), dis.end(), str[y-1]) == 0) dis.push_back(str[y-1]);
            }
        }
        if(dis.size() < 3) {cout<<"Not OK"; goto fin;}
        cout<<"OK";
        fin:
        a--;
    }
}
