#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int T;
    cin >> T;
    while (T){
    	int N;
    	cin >> N;
    	int len;
    	cin >> len;
    	vector<int> den(len);
    	int i,j;
    	for (j=0; j<len; j++){
    		cin >> den[j];
       	}
       	vector<int> min(N+1,999);
       	min[0] = 0;
       	for (i=1; i<=N; i++){
       		for(j=0; j<len; j++){
       			int temp = i - den[j];
       			min[i] = (temp >=0 && (min[temp]+1) < min[i]) ? (min[temp]+1) : min[i];	
       				
       		}
       	}
       	if (min[N]>=999) min[N] = -1;
    	T--;
    //	for(i=0;i<=N;i++)
    //		cout << min[i] <<'k'<< endl;
    	cout << min[N]<<endl;
    }
    return 0;
}
