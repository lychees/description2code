#define f first

using namespace std;

typedef pair<int,int> pr;
typedef long long ll;

ll  A[11][11][100002];
int B[100002];

int main()
{
	int n,q,i,j,k,l,r;
	ll ans; 
		
	scanf("%d %d",&n,&q);


	for(i = 1;i <= n;i++)
	{
		scanf("%d",&B[i]);
	}
	
	for(i = 1; i <= 10;i++)
	{	
		for(j = 1;j <= i;j++)
		{
			for(k = j;k <= n;k =k+i)
			{
				A[i][j][k] = B[k];  
			
			}
			for(k = 2;k <= n;k++)
			{
				A[i][j][k] =  A[i][j][k-1] + A[i][j][k];
				
			}
		}		
	}

	for(i = 1;i <= q;i++)
	{
		scanf("%d %d %d",&l,&r,&k);
		j = l%k;
		if(j == 0)
			j = k;
		ans = 	A[k][j][r] - A[k][j][l-1];
		printf("%lld\n",ans);
	}
	
	return 0;
}
