#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;
long long R[1000001],K[1000001];
int main()
{
    int t;
    int m,n;
    scanf("%d",&t);
    while(t--){
    	string s;
    	scanf("%d%d",&m,&n);
    	cin>>s;
    	int N = s.length();
    	memset(R,0,sizeof(R));
    	memset(K,0,sizeof(K));
    	long long tmp = 0,ans = 0;
    	R[0] = 1;
    	for(int i=0;i<N;i++){
    		if (s[i]=='R') tmp++;
    		if (tmp>=m) ans += R[tmp-m];
    		R[tmp]++;
    	}
    	//for(int i=0;i<=tmp;i++) cout << R[i] << " ";
    	//cout << endl;
    	cout << ans << " ";    
    	tmp = 0,ans = 0;K[0] = 1;
    	for(int i=0;i<N;i++){
    		if (s[i]=='K') tmp++;
    		if (tmp>=n) ans += K[tmp-n];
    		K[tmp]++;
    	}
    	//for(int i=0;i<=tmp;i++) cout << K[i] << " ";
    	//cout << endl;
    	cout << ans << "\n";
    }
    	
    return 0;
}
