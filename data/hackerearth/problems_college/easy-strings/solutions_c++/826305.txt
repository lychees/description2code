#include <iostream>
#define ll long long

using namespace std;

int main()
{
    ll n = 0, tc = 0;
    cin >> n;
    while (n--){
    	cin >> tc;
    	
    	if (tc == 1){
    		cout << 1 << endl;
    		continue;
    	}
    	
    	if (tc%2 == 0)
    		cout << tc << endl;
    	else
    		cout << tc - 1 << endl;
    }
    return 0;
}
