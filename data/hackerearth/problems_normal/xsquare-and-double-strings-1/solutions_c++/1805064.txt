#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int arr[26]={0};
    	for(int i=0;i<s.size();i++){
    		arr[s[i]-'a']++;
    	}
    	int f = 0;
    	for(int i=0;i<26;i++){
    		if(arr[i]>=2){
    			cout<<"Yes\n";
    			f = 1;
    			break;
    		}
    	}
    	
    	if(!f) cout<<"No\n";
    	
    }
    return 0;
}
