#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define M 1000000
 
int c[M+5];

 
int main() {
  ios_base::sync_with_stdio(0);
  int t;
  cin>>t;
  while(t-->0) {
    int a,b;
    cin>>a>>b;
    int d1,d2,d3,d4;
    cin>>d1>>d2>>d3>>d4;
    queue<int> s;
    for(int i=1; i<=M; i++) {
      c[i] = 1e9;
    }
    c[a] = 0;
    s.push(a);
    bool f= 0;
    while(!s.empty()) {
      int x = s.front();
      s.pop();
      if(x==b) {
        cout<<c[x]<<endl;
        f = 1;
        break;
      }
      int z = x*d1;
      if(z>=2 && z<=M && c[z]>c[x]+1) {
        c[z] = c[x]+1;
        s.push(z);
      }
      z = x/d2;
      if(z>=2 && z<=M && c[z]>c[x]+1) {
        c[z] = c[x]+1;
        s.push(z);
      }
      z = x+d3;
      if(z>=2 && z<=M && c[z]>c[x]+1) {
        c[z] = c[x]+1;
        s.push(z);
      }
      z = x-d4;
      if(z>=2 && z<=M && c[z]>c[x]+1) {
        c[z] = c[x]+1;
        s.push(z);
      }
    }
    if(!f) {
      cout<<-1<<endl;
    }
  }

  return 0;
}
