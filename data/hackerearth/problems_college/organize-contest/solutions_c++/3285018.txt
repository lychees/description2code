#include <iostream>
#include<stdio.h>
#include<map>
using namespace std;

int main() {
	int n;
	scanf("%d",&n);
	string s[1005];
	int input[1005];
	for(int i=0;i<n;i++){
		cin>>s[i]>>input[i];
	}
	int ans[1005]={0};
	multimap<std::string,int>my;
	multimap<int,std::string>reverse;
	int d=0;
	int first[1005];
	int last[1005];
	for(int i=0;i<n;i++){
		if(my.count(s[i])<=0){
			my.insert(std::pair<std::string,int>(s[i],d++));
			last[d-1]=i;
			first[d-1]=input[i];
			reverse.insert(std::pair<int,std::string>(d-1,s[i]));
		}
		else{
			int e=my.find(s[i])->second;
			last[e]=i;
		}
	}
	for(int i=0;i<n;i++){
		int a=my.find(s[i])->second;
		ans[a]=ans[a]+input[i];
	}
	int index=-1;
	int max;
	for(int i=0;i<d;i++){
		if(i==0){
			max=ans[i];
			index=i;
			continue;
		}
		if(ans[i]>max){
			index=i;
			max=ans[i];
		}
		else if(ans[i]==max){
			//string chk=reverse.find(index)->second;
			if(last[index]>last[i])index=i;
		}
	}
	string anss=reverse.find(index)->second;
	cout<<anss;
	return 0;
}
