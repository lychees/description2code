#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define PII pair<ll, ll>
#define f first
#define s second
#define F(i,a,b) for(ll i = (ll)(a); i <= (ll)(b); i++)
#define RF(i,a,b) for(ll i = (ll)(a); i >= (ll)(b); i--)
#define inf LLONG_MAX
#define mod 1000000007
#define MAXN 100005
#define pb(x) push_back(x)

ll n, ans, x;
bitset < 2005 > mat[2005];
int main() 
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	F(i,1,n)
	{
		F(j,1,n)
		{
			cin>>x;
			mat[i][j]=x;
		}
	}
	F(i,1,n)
	{
		F(j,i+1,n)
		{
			ll cnt=(mat[i] & mat[j]).count();
			if(cnt>0)
				ans+=cnt*(cnt-1)/2;
		}
	}
	ans/=2;
	cout<<ans<<endl;
	return 0;
}
