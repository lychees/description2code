#include <iostream>
using namespace std;

int main()
{
	int t;
	int n,k,inp;
	cin>>t;
	while(t--){
		long long max_so_far=0,curr_max=0;
		cin>>n>>k;
		for(int i=0;i<n;i++){
			cin>>inp;
			if(inp>k){
				curr_max=0;
			}
			else{
				curr_max+=inp;
				max_so_far=max(curr_max,max_so_far);
			}
		}
		cout<<max_so_far<<endl;
	}
    return 0;
}
