#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int m=0, number=0;
		long res = 0, temp = 0, count = 0;
		cin >> m;
		while (m--)
		{
			cin >> count >> number;
			res += (count*number);
			temp = res;
			do
			{
				res = res / 10 + (res % 10);
			} while (temp /= 10);
		}
		cout << res << endl;
	}
	return 0;
}
