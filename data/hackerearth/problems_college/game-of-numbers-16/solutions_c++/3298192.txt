#include<iostream>
using namespace std;
int main()
{
    int i,t,q,j,count;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>q;
        for(i=2;i<=q;i++)
            for(j=i+1;j<=q;j++)
            if((i*j)%(i+j)==0)
            {//cout<<i<<" "<<j<<"\n";
                count++;}
        cout<<count<<"\n";
    }
    return 0;
}
