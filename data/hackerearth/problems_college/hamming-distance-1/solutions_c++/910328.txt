#include <iostream>
using namespace std;

int main()
{
	int t, count;
	unsigned long x, y;
	cin >> t;
	while(t>0)
	{
		count = 0;
		t--;
		cin >> x;
		cin >> y;
		while (x>0 || y>0)
		{
			if(x%2 != y%2)
			{
				count++;
			}
			x = x/2;
			y = y/2;
		}
	    cout << count << endl;
	}
    return 0;
}
