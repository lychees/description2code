#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    char m[2000001],ne[2000001];
    int cur=0;
    cin>>m;
    for(int i=0;i<n;i++){
    	if(m[i]!='w')
    	   {
    	   	ne[cur++]=m[i];
    	   	continue;
    	   }
    	   ne[cur++]='v';
    	   ne[cur++]='v';
    }
    int ans2=cur;
    int c=0,sum=0;
    for(int i=0;i<cur;i++){
    	if(ne[i]=='v'){
    		sum++;
    	} else
    	{
    		c+=sum/2;
    		sum=0;
    	}
    }
    c+=sum/2;
    printf("%d %d\n",ans2-c,ans2);
    return 0;
}
