#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;
long long primes[10];
long long Count(long long a , long long b , int primecount){
  long long final = 0 ;
  for(int i = 1 ; i <= (pow(2,primecount)) - 1 ; i++){
  	    int bitcount = 0 ;
  	    long long prod = 1;
  	    //printf("i ==> %d\n",i);
  	    for(int j = 0 ; j < primecount ; j++){
  	    	if((i & (int)pow(2,j)) > 0){
  	    	     
  	    	     bitcount++;
  	    	     prod = prod * primes[j];
  	    	     if(prod > 1000000000)
  	    	          prod = 0 ;
  	    	}
  	    }
  	    if(prod > 0){
  	    	if(bitcount % 2 == 0){
  	    		final -= ((b / prod) - ((a - 1) / prod));
  	    	} else {
  	    		final += ((b / prod) - ((a - 1) / prod));
  	    	}
  	    }
  	    //printf("final ==> %d\n",final);
  }
  return final;
}
int main()
{
    int k , t;
    scanf("%d%d",&k,&t);
    for(int i = 0 ; i < k ; i++){
    	long long  x ;
    	scanf("%lld",&x);
    	primes[i] = x;
    }
    while(t-- > 0){
    	long long  a , b ;
    	scanf("%lld%lld",&a,&b);
    	printf("%lld\n",Count(a,b,k));
    }
    return 0;
}
