const int N = 1e4+3;
int dp[N][12];
int n,p;
int rec(int pos,int cnt) {
  if(cnt>=p) return 0;
  if(pos==n && cnt<p)
    return 1;
  if(dp[pos][cnt]!=-1)
    return dp[pos][cnt];
  int ans = 0;
  if(pos==0) {
    ans+=rec(pos+1,0);
    if(ans>=MOD) ans-=MOD;
  }
  else {
    ans+=rec(pos+1,cnt+1);
    if(ans>=MOD) ans-=MOD;
    ans+=rec(pos+1,0);
    if(ans>=MOD)
      ans-=MOD;
  }
  return dp[pos][cnt]=ans;
}
int main() {
  int t;
  sci(t);
  while(t--) {
    M(dp,-1);
    sci(n);sci(p);
    cout<<rec(0,0)<<endl;
  }
}
