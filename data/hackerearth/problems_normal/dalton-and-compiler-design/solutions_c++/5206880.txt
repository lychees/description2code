#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int t = 0;
    cin>>t;
    for(int i =0;i<t;i++){
        int s = 0;
        cin >> s;
        float c = 0;
        c = s/2.0;
        if(floor(c)==c){
            cout<<c<<" "<<c<<endl;
        }
        if(floor(c)!=c){
            if(ceil(c)==floor(c)){
                cout<<floor(c)<<" "<<floor(c)+1<<endl;
            }
            else{
                cout<<floor(c)<<" "<<ceil(c)<<endl;
            }
        }
    }
}
