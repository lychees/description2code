#include <stdio.h>
int main()
{
    long long int cases;
    scanf("%lld",&cases);
    while(cases>0){
    	long int inputs,temp=0,even=0,odd=0;
    	scanf("%lld",&inputs);
    	long int arr[inputs];
    	for(temp;temp<inputs;temp++){
    		scanf("%lld",&arr[temp]);
    		if(arr[temp]%2==0){
    			even++;
    		}
    		else{
    			odd++;
    		}
    		}
    	printf("%lld\n",even*odd);
    	cases --;
    	
    		}
    		return 0;
    	}
