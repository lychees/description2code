#include <bits/stdc++.h>
using namespace std;

int dp[1000010];

int factorize(int num){
	int temp=num;
	int ans=0;
	while(num%2==0){
		ans++;
		num/=2;
	}
	for(int i=3;i*i<=num;i+=2){
		while(num%i==0){
			ans++;
			num/=i;
		}
	}
	if(num>1){
		ans++;
	}
	dp[temp]=ans;
	return 0;
}



int pre(){
	for(int i=1;i<=1000000;i++){
		factorize(i);
	}
	for(int i=2;i<=1000000;i++){
		dp[i]=dp[i]+dp[i-1];
	}

	return 0;
}


int main()
{
	pre();
    int t;
    int a,b;
    scanf("%d",&t);
    while(t-->0){
    	scanf("%d%d",&a,&b);
   		printf("%d\n",dp[b]-dp[a-1]);
    }
    return 0;
}
