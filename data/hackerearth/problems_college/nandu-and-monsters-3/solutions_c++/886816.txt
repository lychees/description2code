#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
long long ar[123456];
int main() 
{
	long long n,i,j,q,x,y;
	string s;
	char c;
	scanf("%lld %lld",&n,&q);
	cin>>s;
	for(i=1;i<=n;i++)
	{
		if(s[i-1]=='0')
		ar[i]=1;
	}

	while(q--)
	{
		scanf("%lld%lld",&x,&y);
		if(ar[x]*ar[y]==0)
		{printf("No\n");continue;}
	
		while(1)
		{
			if(y==x)
			{printf("Yes\n");y=-1;break;}
			if(ar[y/2]!=0)
			{y/=2;}
			else
			break;
			if(y==0)
			if(y==x)
			{printf("Yes\n");y=-1;break;}
			else
			break;
		}
		
		if(y==-1)
		continue;
		while(1)
		{
			if(y==x)
			{printf("Yes\n");y=-1;break;}
			if(ar[x/2]!=0)
			{x/=2;}
			else
			break;
			if(x==0)
			if(y==x)
			{printf("Yes\n");y=-1;break;}
			else
			break;
		}
		
		if(y!=x&&y!=-1)
		{printf("No\n");continue;}
	}
	return 0;
}
