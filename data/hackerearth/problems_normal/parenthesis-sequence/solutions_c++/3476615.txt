#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	string s;
	while(test--)
	{
		cin>>s;
		int l=s.length();
		int i,t=0,f=0,b=0;
		for(i=0;i<l;i++)
		{
			if(s.at(i)==')')
			t--;
			else
			t++;
			if(t<0)
			{
				f++;
				t=0;
			}
		}
		if(t>0)
		b+=t;
		for(i=0;i<f;i++)
		cout<<'(';
		cout<<s;
		for(i=0;i<b;i++)
		cout<<')';
		cout<<endl;
	}
	return 0;
}
