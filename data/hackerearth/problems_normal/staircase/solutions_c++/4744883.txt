#include <iostream>
using namespace std;
long long int jack_stair(long long int n)
{
	long long int dp[n+1],i;
	dp[0]=0;
	dp[1]=1;dp[2]=2;dp[3]=4;
	for(i=4;i<=n;i++)
	{
		dp[i]=dp[i-1]+dp[i-2]+dp[i-3];
	}
	return dp[n];
}
long long int jil_stair(long long int n)
{
	long long int dp[n+1],i;
	dp[0]=0;
	dp[1]=1;dp[2]=2;
	for(i=3;i<=n;i++)
	{
		dp[i]=dp[i-1]+dp[i-2];
	}
	return dp[n];
}
int main()
{
    long long int jac,jil,i,n,j;
    cin>>n;
    if(n<0)
    {
    	cout<<"ERROR"<<endl;
    	return 0;
    }
    jac=jack_stair(n);
    jil=jil_stair(n);
    cout<<"Jack-"<<jac<<endl;
    cout<<"Jill-"<<jil<<endl;
    return 0;
}
