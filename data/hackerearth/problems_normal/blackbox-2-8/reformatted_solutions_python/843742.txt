t = int(input())
while (t > 0):
    k = long(input())

    def factorial(n):
        if n == 0:
            return (1)
        else:
            return reduce(lambda x, y: x*y, range(1, n+1, 1))

    def sum_digits(n):
        return sum([int(i) for i in str(n)])

    sumfac = sum_digits(factorial(k))
    print("%s") % (sumfac)
    t = t-1
