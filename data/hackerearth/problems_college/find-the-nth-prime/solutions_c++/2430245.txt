#include <iostream>
#include <math.h>
using namespace std;

long long arr[100000]={0};

bool isprime(long n){
                if(n==1)
                        return 0;
               if(n==2 || n==3)
                        return 1;
                if(n%2==0 || n%3==0)
                        return 0;
                int i=5;
                int w = 2;
                while(i*i <=n){
                        if(n%i==0)
                                return 0;
                        i+=w;
                        w = 6-w;
                }
                return 1;
}

void saveprime(){
	int k=0;
	//long long arr[100000]={0};
	for(long long i=0;i<1000000;i++){
		if(isprime(i)){
			arr[k] = i;
			k++;
		}
	}
	//for(long long i=0;i<k;i++)
	//	cout << arr[i]<<endl;
}

int main(){
	int t;
	saveprime();
