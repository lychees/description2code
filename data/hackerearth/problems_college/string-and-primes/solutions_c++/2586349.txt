#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	int i,j,n,p,k,flag=0,flag1=1;
    	string str;
    	cin>>str; n=str.size();
    	for(i=0;i<n;i++)
    	{
    		flag=0;
    		p=str[i];
    		for(j=2;j<=sqrt(p);j++)
    		{
    			if(p%j==0)
    				{
    					flag=1; break;
    				}
    		}
    		if(flag==1)
    		{
    			cout<<str[i];
    			flag1=0;
    		}
    	}
    	if(flag1==1)
    		cout<<"NULL";
    	cout<<endl;
    
    }
   
}
