#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tt;
    for (cin >> tt; tt > 0; tt--)
    {
    	int n, m, k;
    	cin >> k >> n >> m;
    	int ans = 0;
    	int x = 0;
    	for (int i = 0; i < n; i++)
    	{
    		if (x < k)
    		{
    			int d = k - x;
    			int e = (d - 1) / m + 1;
    			x += e * m;
    		}
    		if (x > k)
    			ans++;
			x -= k;
    	}
    	cout << ans << endl;
    }
    
    return 0;
}
