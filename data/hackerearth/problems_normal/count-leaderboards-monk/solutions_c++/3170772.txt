#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
ll c[1001][1001];
ll f[1001];
ll dp[1001];
ll solve(int n,int k)
{
	int i,j;
	f[0]=1;
	for(i=1;i<=n;i++)
	{
		f[i]=(i*f[i-1])%mod;
	}
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(j==0 || i==j)
			{
				c[i][j]=1;
			}
			else
			c[i][j]=(c[i-1][j] + c[i-1][j-1])%mod;
		}
	}
	dp[0]=1;
	for(i=1;i<=n;i++)
	{
		for(j=0;j<i;j++)
		{
			ll ways=(c[i-1][j]*f[j+1])%mod;
			ways=(ways*dp[i-j-1])%mod;
			dp[i]+=ways;
			dp[i]%=mod;
		}
	}
	return (dp[n]*c[n][k])%mod;
}
int main()
{
	int n,k;
	cin >> n>>k;
	ll ans=solve(n,k);
	cout << ans << endl;
	return 0;
	
}
