#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	
	int ntc,n,d,c,i;
	scanf("%d",&ntc);
	while(ntc--)
	{
		
		scanf("%d%d",&n,&d);
		int arr[n];
		for(i=0;i<n;i++)
		scanf("%d",&arr[i]);
		sort(arr,arr+n);
		c=0;
		for(i=0;i<n-1;)
		{
			if(arr[i+1]-arr[i]<=d)
			{
				c++;
				i=i+2;
			}
			else
			
			i++;
		}
		
		printf("%d\n",c);
		
		
	}
    return 0;
}
