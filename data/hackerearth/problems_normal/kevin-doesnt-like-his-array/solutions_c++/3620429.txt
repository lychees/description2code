#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
#include<algorithm>
using namespace std;
int a[100001],b[100001],x,y,ans;
priority_queue<int> Q;
void sol()
{
    int i,n;
    cin>>n;
    for(i=0;i<n;i++) cin>>a[i];
    memset(b,0,sizeof(b));
    for(i=0;i<n;i++) b[a[i]]++;
    for(i=0;i<=100000;i++)
    {
        if(n%2==0&&b[i]>n/2) {cout<<"-1"<<endl;return ;}
        if(n%2==1&&b[i]>n/2+1) {cout<<"-1"<<endl;return ;}
    }
    memset(b,0,sizeof(b));
    for(i=0;i<n-1;i++)  if(a[i]==a[i+1]) b[a[i]]++;
    for(i=0;i<=100000;i++) if(b[i]>0) Q.push(b[i]);
    while(!Q.empty())
    {
        ans++;
        x=Q.top();Q.pop();
        if(Q.empty())
        {
            ans+=x-1;
            break;
        }
        y=Q.top();Q.pop();
        x-=1;
        y-=1;
        if(x>0) Q.push(x);
        if(y>0) Q.push(y);
    }
    cout<<ans<<endl;
}
int main()
{
     sol();
    return 0;
}
