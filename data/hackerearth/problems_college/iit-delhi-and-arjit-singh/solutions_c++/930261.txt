const double pi=acos(-1.0);
#define MAXN 1000004
LL phi[MAXN + 1], prime[MAXN/10], sz=0;
vector<bool> mark(MAXN + 1);

int main()
{
    
   phi[1] = 1; 
   for (LL i = 2; i <= MAXN; i++ )
   {
    	if(!mark[i])
		{
        	phi[i] = i-1;
        	prime[sz++]= i;
    	}
    	for (LL j=0; j<sz && prime[j]*i <= MAXN; j++ )
		{
        	mark[prime[j]*i]=1;
        	if(i%prime[j]==0)
			{
            	LL ll = 0;LL xx = i;
            	while(xx%prime[j]==0)
            	{
                	xx/=prime[j];
                	ll++;         
            	}
            	LL mm = 1;
            	for(LL k=0;k<ll;k++)mm*=prime[j];
            	phi[i*prime[j]] = phi[xx]*mm*(prime[j]-1);
            	break;
    		}
        	else 
				phi[i*prime[j]] = phi[i]*(prime[j]-1 );
    	}
	}
	for(LL i=2;i<=MAXN;i++)
		phi[i]+=phi[i-1];
   	int t;
   	scanf("%d",&t);
   	while(t--)
   	{
    	LL n,m;
    	cin>>n>>m;
    	printf("%lld\n",phi[n/m]);             
   	} 
   	return 0;    
}
