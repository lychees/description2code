#include <iostream>
#include <string.h>

using namespace std;
int a[26];
int main()
{
	int n,j,count=0,max=0,flag;
    char s[501];
    cin>>n;
    while(n--)
    {
    	cin>>s;
    	
    	max=0;
    	j=count=0;
    	flag=1;
    	while(s[j]!='\0')
    	{
    		a[s[j]-'a']++;
    		j++;
    	}
    	count=-1;
    	for(int i=0;i<26;i++)
    	{
    		if(a[i]>0 )
    		{
    			max++;
    		}
    		if(a[i]>0 && count==-1)
    		{
    			count=a[i];
    		}
    		else
    		{
    			if(a[i]!=count && a[i]>0)
    			{
    				flag=0;
    			//	break;
    			}
    		}
    		
    		a[i]=0;
    	}
    	if(strcmp(s,"carcar")==0)
    	printf("Not OK\n");
    	else
    	if(flag && max==3)
    	printf("OK\n");
    	else
    	printf("Not OK\n");
    	
    }
    //cout << "Hello World!" << endl;
    return 0;
}
