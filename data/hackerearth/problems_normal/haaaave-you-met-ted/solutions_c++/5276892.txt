#include <iostream>
#include<algorithm>
#include<climits>

using namespace std;

#define llu  long long int

int calc( llu n )
{
	int count = 0;
	
	while( n )
	 {
	 	 count ++;
	 	 
	 	 n = n&(n-1);
	 } 
	
	return count;
}

int main()
{
	llu t;

	cin >> t;
	
	for(llu test_case=1 ; test_case<=t; test_case++ )
	 {
	 	 llu n ;
	 	
	 	 cin >> n ;
	 	 
	 	 llu mini = LLONG_MAX;
	 	 
	 	 for( llu i=1; i<=n ; i++)
	 	 {
	 	 	llu temp;
	 	 	
	 	 	cin >> temp ;
	 	 	
	 	 	llu x =  calc( temp );
	 	 	
	 	 	if( mini > x )
	 	 	 {
	 	 	    mini = x;	
	 	 	 }
	 	 }
		 
		 cout << mini << "\n"; 		 
	 }
	 
	 return 0;
}
