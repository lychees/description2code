t=int(raw_input())
while t>0:
	n=int(raw_input())
	p=raw_input().split()
	i=0
	j=n-1
	m_c=0
	r_c=0
	m_sum=0
	r_sum=0
	while i<j:
		r_c+=1
		r_sum+=int(p[j])
		while m_sum<2*r_sum and i<j:
			m_c+=1
			m_sum+=int(p[i])
			i+=1
		if m_sum!=r_sum:
			i-=1
			m_c-=1
			m_sum-=int(p[i])
		j-=1
	if i==j:
		m_c+=1
	print m_c,r_c
	t-=1
