#include <iostream>
#include <bits/stdc++.h>
#define MOD 1000000007
#define ll unsigned long long
using namespace std;

ll power (ll a, ll b) {
	ll ans = 1;
	while (b) {
		if (b & 1) {
			ans = (ans * a) % MOD;
		}
		a = (a * a) % MOD;
		b>>=1;
	}
	return ans%MOD;
}

int main()
{
    //cout << "Hello World!" << endl;
    ll n,a[100005],b[100005],q,id,e,fact=1,c;
    c=0;
    cin>>n;
    for(ll i=1;i<=n;i++){
    cin>>a[i];
    if(a[i]==0)
    c++;
    else
    fact=(fact*a[i])%MOD;
    }
	cin>>q;
    while(q--){
    	ll qu;
    	cin>>qu;
    	if(qu==0){
    		cin>>id>>e;
   			if(a[id]==0 && e==0)
   			continue;
   			else if(e==0){
   				c++;
   				fact=(fact*power(a[id],MOD-2))%MOD;
   			}
   			else if(a[id]==0){
   				c--;
   				fact=(fact*e)%MOD;
   			}
   			else{
   				fact=(fact*power(a[id],MOD-2))%MOD;
   				fact=(fact*e)%MOD;
   			}
   			a[id]=e;
    	}
    	else{
    		cin>>id;
    		if(c==0)
    		{
    			cout<<(fact*power(a[id],MOD-2))%MOD<<endl;
    		}
    		else if(c==1){
    			if(a[id]==0)
    			cout<<fact<<endl;
    			else
    			cout<<0<<endl;
    		}
    		else
    		{
    			cout<<0<<endl;
    		}
    	}
    }
    return 0;
}
