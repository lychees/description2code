#include <bits/stdc++.h>
#define mod 1000000007
#define lli long long int

using namespace std;
int n,m;

void dfs(vector<int> adj[],int u, vector<bool>& visited,lli &max, lli &maxcount, vector<lli>& arr){
	visited[u]=true;
	//for(int i=0;i<n;i++){
	for(int v: adj[u] ){
		if(!visited[v]){
			
			if(arr[v]>max){
				max=arr[v];
				maxcount=1;
			}
			else if(arr[v]==max)
				maxcount++;
			
			dfs(adj,v,visited,max,maxcount,arr);
		}
	}
}

int main()
{
	int u,v;
	scanf("%d",&n);
	scanf("%d",&m);
	vector<lli> arr(n);
	vector<bool> visited(n,false);
	vector<int> adj[n];
	//,vector<int>(n,0));
	
	for(int i=0;i<n;i++){
		scanf("%lld",&arr[i]);
	}
	
	for(int i=0;i<m;i++){
		scanf("%d %d",&u,&v);
		adj[u-1].push_back(v-1);
		adj[v-1].push_back(u-1);
	}
	
	lli ret=1;
	lli maxcount,max;
	for(int i=0;i<n;i++){
		if(!visited[i]){
			max=arr[i];
			maxcount=1;
			dfs(adj,i,visited,max,maxcount,arr);
			ret=(ret*maxcount)%mod;
		}
	}
	
	printf("%lld\n",ret);
	return 0;
	
}
