#include <bits/stdc++.h>

#define MAX 1000005
#define INF 1111111111

typedef long long ll;
typedef unsigned long long llu;

using namespace std;

llu arr[100005];

int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    llu t,n,x;
    scanf("%llu",&t);
    for(llu i=1;i<=t;i++){
        scanf("%llu %llu",&n,&x);
        for(llu j=0;j<n;j++)scanf("%llu",&arr[j]);
        llu res=0;
        for(llu j=0;j<n-1;j++){
            if(arr[j]>=arr[j+1]){
                llu dif=arr[j]-arr[j+1];
                llu cnt=(dif/x)+1;
                arr[j+1]+=(x*cnt);
                res+=cnt;
            }
        }
        printf("%llu\n",res);
    }
    return 0;
}
