#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main()
{
    int T;
    cin>>T;
    int N;
    while(T--){
    	cin>>N;
    	double R=0;
    	double r;
    	double a;
    	for(int i=0;i<N;i++){
    		cin>>a;
    		r = a/(4*3.1416);
    		R = R + sqrt(r);
    	}
    	double A = 4*3.1416*R*R;
    	R = 0;
    	cout << fixed << setprecision(4);
    	cout<<A<<"\n";
    }
    return 0;
}
