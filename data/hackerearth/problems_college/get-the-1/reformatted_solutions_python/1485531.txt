t = input()
for i in range(0, t):
    n = input()
    S = 0
    while (n > 9):
        while (n != 0):
            rem = n % 10
            S += pow(rem, 2)
            n = n/10
        n = S
        S = 0
    if n == 1:
        print("Case #"+str(i+1)+":YES")
    else:
        print("Case #"+str(i+1)+":NO")
