#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>
#include <string.h>
#include <map>
#include <stdlib.h>
#include <vector>
using namespace std;
 
#define ll long long
#define gc getchar
 
#define FOR(i,n) for(int i=0;i<n;i++)
#define scanint(n) scanf ("%d",&n)
 
 

int main(){
	int t,n,m,x1,x2,y1,y2,p;
	scanf("%d",&t);
	FOR(i,t){
		scanf("%d",&n);
		scanf("%d",&m);
		scanf("%d",&x1);
		scanf("%d",&y1);
		scanf("%d",&x2);
		scanf("%d",&y2);
		scanf("%d",&p);
		int ans1 = abs( x1- x2);
		ans1  = min(ans1, abs(n-x1+x2));
		ans1  = min(ans1, abs(n-x2+x1));
		int ans2 = abs(y1-y2);
		ans2  = min(ans2, abs(m-y1+y2));
		ans2  = min(ans2, abs(m-y2+y1));
		int ans = (ans1+ans2)*p;
		cout <<ans<<endl;
		if(ans<=1000){
			cout << "You saved the group."<<endl;			
		}
		else{
			cout << "Let go of the group." <<endl;
		}
	}
	return 0;
}
