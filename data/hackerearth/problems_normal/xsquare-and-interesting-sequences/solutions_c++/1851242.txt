

    #include <bits/stdc++.h>
    #define lli long long
    #define M 1000000009
    using namespace std;
    lli fast_pow(lli a, lli b)
    {
    	lli res = 1;
    	while ( b > 0 ) {
    		if ( b&1 ) res = (res*a)%M;
    		a = (a*a)%M;
    		b >>= 1;
    	}
    	return res;
    }
    int main()
    {
    	int t;
    	lli n;
    	cin >> t;
    	while ( t-- ) {
    		cin >> n;
    		cout << fast_pow(26LL,n/2+n%2) << endl;
    	}
    	return 0;
    }
