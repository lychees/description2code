#include <iostream>
#include <vector>
#include <cstdlib>
#define N 4000000
using namespace std;
typedef unsigned long long int ull;

ull index[N+2]={0};
int main()
{
	ull n,s=0;
	cin>>n;
	for(ull i=2;i<n;i++)
	{
           if(index[i]==0)
			{
				for(ull j=2*i;j<n;j+=i)
				index[j]=1;
				s+=i;
			}
    }
	                      
	cout<<s<<endl;
	return 0;
}
