#include <bits/stdc++.h>
using namespace std;
#define MAXN 1000010

int dp[MAXN];
void pre(){
	//2 3 5 7
	for(int i=0; i<MAXN; ++i) dp[i] = -1;
	dp[0] = 0;
	for(int i=2; i<MAXN; ++i){
		int mn = 1e9+7;
		if(i > 1 && dp[i-2] != -1) mn = min(mn, dp[i-2]);
		if(i > 2 && dp[i-3] != -1) mn = min(mn, dp[i-3]);
		if(i > 4 && dp[i-5] != -1) mn = min(mn, dp[i-5]);
		if(i > 6 && dp[i-7] != -1) mn = min(mn, dp[i-7]);
		if(mn != 1e9+7) dp[i] = mn+1;
	}
}

int main()
{
	int t;
	cin >> t;
	pre();
	while(t--){
		int n;
		cin >> n;
		cout << dp[n] << endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
