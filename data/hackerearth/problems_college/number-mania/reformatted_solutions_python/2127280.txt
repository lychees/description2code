from math import sqrt


def fact(n):
    if n == 1 or n == 0:
        return 1
    else:
        return n * fact(n-1)


def factors(num):
    final = set()
    count = 0
    for i in xrange(1, int(sqrt(num))+1):
        if num % i == 0:
            count += 1
            if count > 1:
                return 0

    return 1
