def F(x,y):
	if(x==0):
		return (y + 1)%1000
	if(str(x)+"-"+str(y) in d):
		return d[str(x)+"-"+str(y)]
	if x > 0 and y == 0:
		d[str(x)+"-"+str(y)] = F(x - 1, 1) % 1000
		return d[str(x)+"-"+str(y)]
	if x > 0 and y > 0:
		d[str(x)+"-"+str(y)] = F(x - 1, F(x, y - 1)) 
		return F(x - 1, F(x, y - 1)) 
x,y = map(int,raw_input().split())
d = dict()
ans = F(x,y)
if ans<10:
	print "00"+str(ans)
elif ans<100:
	print "0"+str(ans)
else:
	print ans
