#include <iostream>
#include <stdio.h>
#include <string>
#include <string.h>

using namespace std;

int h[256];

int main()
{
    string s;
    getline(cin,s);
    int ln=s.size(),ma=0,k=0;
    for(int i=0;i<ln;++i)
    h[s[i]]++;
    for(int i=0;i<256;++i)
    if(h[i]>ma){
        ma=h[i];
        k=i;
    }
    cout<<(char)k<<" "<<ma;
}
