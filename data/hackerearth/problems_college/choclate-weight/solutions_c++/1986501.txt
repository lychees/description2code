#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
typedef long long int ll;
int main()
{
    ll n,x,i,j,z,s,d;
    cin>>n;
    ll a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    cin>>x;
    bool f;
    for(i=0;i<x;i++)
    {
        cin>>z;
        for(j=0;j<n;j++)
        {
            if(a[j]==z)
             {
                 f=true;
                 break;
             }
            else
            {
                d=abs(a[j]-z);
                if(d<=a[j])
                  f=binary_search(a,a+j,d);
                else
                  f=binary_search(a+j+1,a+n,d);
                if(f)
                  break;
            }
        }
        if(f)
            cout<<"I Like You\n";
        else
            cout<<"I Hate You\n";
    }
    return 0;
}
