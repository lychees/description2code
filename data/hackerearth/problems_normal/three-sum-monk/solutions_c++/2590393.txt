#include <bits/stdc++.h>
#define lli long long

using namespace std;

int A[502];
vector <int> v;

void pre()
{
	for ( int i = 1;; i++ ) {
		if ( i*i*i > 6000 ) break;
		v.push_back(i*i*i);
	}
	return;
}

//Code for FAST INPUT :)
template <typename T>
inline void fi(T *a)
{
    register char c=0;
    while (c<33) c=getchar_unlocked();
    *a=0;
    int tmp = 0;
    while (c>33)
    {
        if ( c == 45 ) tmp = 1;
        else *a=*a*10+c-'0';
        c=getchar_unlocked();
    }
    if ( tmp == 1 ) *a = 0-(*a);
}

int main()
{
	v.clear();
	pre();
	int t;
	lli n,num,den;
	fi(&t);
	assert(t >= 1 && t <= 40);
	while ( t-- ) {
		fi(&n);
		assert(n >= 1 && n <= 500);
		den = n*(n-1)*(n-2);
		den /= 6;
		num = 0;
		for ( int i = 0; i < n; i++ ) fi(&A[i]), assert(A[i] >= 1 && A[i] <= 2000);
		sort(A, A+n);
		for ( int i = 0; i < v.size(); i++ ) {
			for ( int fs = 0; fs < n-2; fs++ ) {
				if ( v[i]-A[fs] < A[fs+1] ) break;
				for ( int sc = fs+1; sc < n-1; sc++ ) {
					if ( v[i] - A[fs] - A[sc] < A[sc+1] ) break;
					int st = lower_bound(A+sc+1, A+n, v[i]-A[fs]-A[sc]) - A;
					if ( st == n || A[st] > v[i]-A[fs]-A[sc] ) continue;
					num += upper_bound(A+sc+1, A+n, v[i]-A[fs]-A[sc]) - A - st;
				}
			}
		}
		double ans = (double)num/(double)den;
		printf("%0.9lf\n", ans);
	}
	return 0;
}
