#include <bits/stdc++.h>
#include<map>
using namespace std;
unordered_multimap<int,int> m;
int n,k,a,h,flag;
int main()
{
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {cin>>a;
     m.insert(pair<int,int>(a,i));
    
    }
 
    unordered_multimap<int,int>::iterator it,ij;
    for(it=m.begin();it!=m.end();it++)
    { h=k-it->first;
      ij=m.find(h);
      if(ij!=m.end()&&ij->second!=it->second)
     { flag=1;
      break;
     }
    }
    if(flag)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    
     
    return 0;
}
