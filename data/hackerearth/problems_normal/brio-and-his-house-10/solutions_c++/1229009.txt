#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int T;
   cin>>T;
   
   while(T--)
   {
   	double a,b,c;
   	cin>>a>>b>>c;
   	double area;
   	
   	
   	double s=((a+b+c)/2);
   	
   	double abc=a*b*c;
   	s=sqrt((s*(s-a)*(s-b)*(s-c)));
   	s=4*s;
   	s=abc/s;
   	area=((3.1415)*s*s);
   	cout<<setprecision(4)<<fixed;
   	cout<<area<<endl;
   	
   	
   	
   	
   	
   }
    return 0;
}
