#include <iostream>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	string s;
   	while(t--){
   		cin>>s;
   		int a[26];
   		for(int i=0;i<26;i++) a[i]=0;
   		int n=s.size();
   		bool q=false;
   		for(int i=0;i<n;i++) a[s[i]-'a']++;
   		for(int i=1;i<26;i++){
   			if(a[i]==a[i-1]){
   				q=true;
   			}
   			else{
   				q=false;
   				break;
   			}
   		}
   		if(q) cout<<"Yes\n";
   		else cout<<"No\n";
   	}
    return 0;
}
