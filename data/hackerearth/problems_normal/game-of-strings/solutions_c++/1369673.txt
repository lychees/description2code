#include <iostream>
#include<cstring>
#include<stdio.h>
using namespace std;
char X[10000], Y[10000];
int L[5005][5005];
int max(int a, int b);

int lcs( int m, int n )
{
   
   int i, j;
  
   for (i=0; i<=m; i++)
   {
     for (j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         L[i][j] = 0;
  
       else if (X[i-1] == Y[j-1]&&X[i-1]!='a'&&X[i-1]!='u'&&X[i-1]!='e'&&X[i-1]!='i'&&X[i-1]!='o'&&Y[j-1]!='a'&&Y[j-1]!='e'&&Y[j-1]!='i'&&Y[j-1]!='o'&&Y[j-1]!='u'&&X[i-1]!='A'&&X[i-1]!='U'&&X[i-1]!='E'&&X[i-1]!='I'&&X[i-1]!='O'&&Y[j-1]!='A'&&Y[j-1]!='E'&&Y[j-1]!='I'&&Y[j-1]!='O'&&Y[j-1]!='U')
         L[i][j] = L[i-1][j-1] + 1;
  
       else
         L[i][j] = max(L[i-1][j], L[i][j-1]);
     }
   }
    
   
   return L[m][n];
}
  

int max(int a, int b)
{
    return (a > b)? a : b;
}
int main() {
	//scanf("%s %s", X, Y);
	cin>>X;
	cin>>Y;
	cout<<lcs(strlen(X), strlen(Y))<<endl;
	return 0;
}
