#include<bits/stdc++.h>
using namespace std;
int a[] = {6,2,5,5,4,5,6,3,7,6};
int main()
{
	unsigned long long int t,i,result;
	char num[100001];
	cin>>t;
	while(t--)
	{
		cin>>num;
		result =0;
		i=0;
		while(num[i]!='\0')
		{
			result += a[num[i]-'0'];
			i++;
		}
		cout<<result<<endl;
	}
	return 0;
}
