#include <iostream>
using namespace std;

int main()
{
	int n;
    cin>>n;
    int arr[n],x,i,j,k,flag=0;
    for(i=0;i<n;i++)
    cin>>arr[i];
    cin>>x;
    for(i=0;i<=n-3;i++)
    {
    	for(j=i+1;j<=n-2;j++)
    	{
    		for(k=j+1;k<=n-1;k++)
    		{
    			if(arr[i]+arr[j]+arr[k]==x)
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    		break;
    	}
    	if(flag==1)
    	break;
    }
    if(flag==1)
    cout<<"True";
    else
    cout<<"False";
    cin>>arr[i];
    return 0;
}
