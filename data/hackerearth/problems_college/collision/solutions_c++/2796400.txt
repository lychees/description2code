#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	map<int, int> m;
    	int n;
    	cin >> n;
    	int i, j, ans=0;
    	for(i=0;i<n;i++)
    	{
    		cin >> j;
    		if(m[j%10]!=0)
    		ans++;
    		else
    		m[j%10]=1;
    	}
    	cout << ans << endl;
    	m.clear();
    }
    return 0;
}
