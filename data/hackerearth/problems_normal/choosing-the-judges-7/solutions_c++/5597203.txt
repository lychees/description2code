#include <iostream>
#include<climits>;
using namespace std;
int maxScore(long long a[],int n){
	long long A[n];
	long long Max(INT_MIN);
	long long x,y;
	for(int i=n-1;i>=0;i--){
		if(i+1>=n)
		x = 0;
		else
		x = A[i+1];
		if(i+2>=n)
		y = 0;
		else
		y = A[i+2];
		
		A[i] = max(a[i]+y,x);
		if(A[i]>Max)
		Max = A[i];
	}
	return Max;
}
int main()
{
    int t,n;
    cin>>t;
    int k(1);
    while(t--){
    	cin>>n;
    	long long a[n];
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    	}
    	cout<<"Case "<<k++<<": "<<maxScore(a,n)<<endl;
    }
    return 0;
}
