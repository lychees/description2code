#include <iostream>
#include <map>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    long long int a,b,c,d;
    for(int i = 0; i < t; i++)
    {
            cin >> a >> b >> c >> d;
            map<int ,long long int> m;
            
            long long int cu = 0;
            while(a%2 == 0) {
                      a = a/2;
                      cu++;
            }
            m[2] = b*cu;
            cu = 0;
            while(c%2 == 0) {
                      c = c/2;
                      cu++;
            }
            m[2] = m[2] - (d*cu);
            cu = 0;
            if(m[2] < 0) {
                    cout << "Not divisible" << endl;
                    continue;
            }
            
            int yy = 0;
            for(int j = 3; j <= sqrt(a); j = j + 2) {
                    while(a%j == 0) {
                              a = a/j;
                              cu++;
                    }
                    m[j] = b*cu;
                    cu = 0;
                    while(c%j == 0) {
                              c = c/j;
                              cu++;
                    }
                    m[j] = m[j] - (d*cu);
                    cu = 0;
                    if(m[j] < 0) {
                            cout << "Not divisible" << endl;
                            yy = 1;
                            break;
                    }     
            }
            
            if(yy == 0) {
                  if(a > 2) {
                       m[a] = 1;
                  }
                  if(c == 1 || a%c == 0) {
                       cout << "Divisible" << endl;
                  }else {
                        cout << "Not divisible" << endl;
                  } 
            }
            
    }
    return 0;
}
