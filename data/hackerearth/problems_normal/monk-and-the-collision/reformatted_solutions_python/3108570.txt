t = input()
for j in range(t):
    ans = 0
    n = input()
    s = []
    a = map(int, raw_input().split())
    for i in range(10):
        s.append([])
    for i in range(n):
        index = a[i] % 10
        if len(s[index]) != 0:
            ans += 1
            continue
        s[index].append(a[i])
    print ans
