#include <cmath>
#include <iostream>
using namespace std;

int main()
{
    int t,a,b;
    scanf("%d",&t);
    while(t--)
    {
    	a = b = 0;
    	string x;
    	cin>>x;
    	for(int i=0;i<8;i++)
    	{
    		a*=2;a+=x[i];
    	}
    	cin>>x;
    	for(int i=0;i<8;i++)
    	{
    		b*=2;b+=x[i];
    	}
    	if(pow(a,1.0/a) <= pow(b,1.0/b)) printf("The Hound\n");
    	else printf("The Mountain\n");
    }
}
