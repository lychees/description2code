#include<iostream>
#include<cstring>
#include<stdio.h>
using namespace std;

int main()
{
	int T=0;
	scanf("%d",&T);
    while(T--){
    		int N=0;
    		scanf("%d",&N);
    		unsigned int arr[N];
    		memset(arr,0,sizeof(arr));
    		unsigned long long int sum =0;
    		for(int i =0;i<N;i++){
    			unsigned int element =0;
    			scanf("%d",&element);	
    			arr[i] =element;
    			sum +=element;
    		}
    		
    		long long int M;
    		scanf("%lld",&M);
    		M = (M-1) % sum;
    		M +=1;
    		int i=0;
    		for(i=0;i<N;i++){
    			M -=arr[i];
    			if(M<=0)
    				break;
    		}
    		printf("%d\n",i+1);
    }
    return 0;
}
