#include<stdio.h>
#include<string.h>
void sort(char ch1[]);
int main()
{

	scanf("%d",&n);
	char ch[n][100];
	int i,j,count=0;
	for(i=0;i<n;i++)
	scanf("%s",&ch[i]);
	char temp1[100];
	char temp2[100];
	for(i=0;i<n;i++)
	{   if(strcmp(ch[i],"N"))
	   {
		for(j=i+1;j<n;j++)
		{    
		  strcpy(temp1,ch[i]);
		  strcpy(temp2,ch[j]);
		  sort(temp1);
		  sort(temp2);
			if((ch[i][0]==ch[j][0])&&(ch[i][strlen(ch[i])-1]==(ch[j][strlen(ch[j])-1])&&!strcmp(temp1,temp2)))
			  strcpy(ch[i],"N");

			  
		}
	  }
	}
	for(i=0;i<n;i++)
	{
		if(strcmp(ch[i],"N"))
		count++;
	}
	printf("%d\n",count);
	return 0;	
}
void sort(char ch1[])
{
  int i,j;
  char temp;
  for(i=0;i<strlen(ch1);i++)
  {
   for(j=i+1;j<strlen(ch1);j++)
   { 
    if(ch1[i]>ch1[j])
     {
      temp=ch1[i];
      ch1[i]=ch1[j];
      ch1[j]=temp;
     }
   }
}
}
