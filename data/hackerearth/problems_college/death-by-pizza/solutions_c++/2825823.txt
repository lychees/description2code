#include <iostream>
using namespace std;
const int MAX = 1000008;
int BIT[MAX];
int a[MAX];
int n;
int update(int x, int val)
{
    for(;x <= n;x += x&-x)
        BIT[x] += val;
}
int query(int x)
{
    int sum = 0;
    for(;x>0;x -= x&-x)
        sum += BIT[x];
    return sum;
}
int main(void){
    int pizza[]={16,9,26,26,1};
    int q,x,l,r;
    cin>>n>>q;
    for(int i=1;i<=n;i++){cin>>a[i];
        update(i,a[i]);}
        //for(int i=0;i<=n;i++)
        //cout<<BIT[i]<<endl;
    
    while(q--)
    {
        cin>>x>>l>>r;
        
        if(x)
        {
            cout<<query(r)-query(l-1)<<"/"<<(r-l+1)<<endl;
        }
        else{
            int val=0;
            for(int i=0;i<=r-l;i++) update(l+i,pizza[i%5]);
                //cout<<val<<endl;
               
        }
    }
}
