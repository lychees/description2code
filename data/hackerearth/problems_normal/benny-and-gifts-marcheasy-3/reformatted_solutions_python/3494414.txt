string = raw_input().strip()
points_visited = set()
x, y, res = 0, 0, 0
points_visited.add((0, 0))

# print x, y
for ins in string:
    if ins == 'R':
        y += 1
    elif ins == 'L':
        y -= 1
    elif ins == 'U':
        x -= 1
    elif ins == 'D':
        x += 1
    if (x, y) not in points_visited:
        points_visited.add((x, y))
        # print '%d %d is snowy' % (x, y)
    else:
        res += 1
        # print '%d %d is icy' % (x, y)
    # print x, y
print res
