#include<stdio.h>

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int x,y,a=1,b=1,count=0;
		scanf("%lld %lld",&x,&y);
		while(x>1 && y>1)
		{
			if(x<y)
		{
			x=x+y;
			y=x-y;
			x=x-y;
		}
		count=count+x/y;
		x=x%y;
		}
		
		if(x==1 && y>1)
		{
			count=count+(y-1);
		}
		else if(y==1 && x>1)
		{
			count=count+(x-1);
		}
		printf("%lld\n",count);
	}

}
