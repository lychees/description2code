#include <iostream>
using namespace std;

int main()
{
	int a[100000],n,ch,c,great,small,q;
	unsigned int num;
	n=0;
	great=0;
	cin>>q;
	for(int k=0; k<q; k++)
	{
		cin>>ch;
		if(ch==1)
		{
			cin>>num;
			a[n++]=num;
		}
		else if(ch==2)
		{
			c=0;
			cin>>num;
			for(int i=0; i<n; i++)
			{
				if(a[i]==num)
				{
					c++;
					for(int j=i; j<n; j++)
					{
						a[j]=a[j+1];
					}
					n--;
					if(c==1)
					{
						break;
					}
				}
			}
			if(c==0)
			{
			 cout<<"-1\n";
			}
		}
		else if(ch==3)
		{
			if(n==0)
			{
				cout<<"-1\n";
			}
			else
			{
				great=a[0];
				for(int i=1; i<n; i++)
				{
					if(a[i]>great)	
				 	great=a[i];
				}
				cout<<great<<"\n";
			}
		}
		else if(ch==4)
		{
			if(n==0)
			{
				cout<<"-1\n";
			}
			else
			{
				small=a[0];
				for(int i=1;i<n;i++)
				{
					if(a[i]<small)
					 small=a[i];
				}
				cout<<small<<"\n";
			}
		}
	}
}
