#include <iostream>
using namespace std;

int main()
{
	long long t,i,red,green,blue,yellow,cyan,magenta,white,black,R,G,B;
	bool r,g,b;
	red=green=blue=yellow=cyan=magenta=white=black=0;
	r=g=b=true;
	cin>>t>>R>>G>>B;
	for(i=0;i<t;i++)
	{
		if(i%R==0)
			r=!r;
		if(i%G==0)
			g=!g;
		if(i%B==0)
			b=!b;
		if(r&&g&&b)
			white++;
		else if(r&&!g&&!b)
			red++;
		else if(!r&&g&&!b)
			green++;
		else if(!r&&!g&&b)
			blue++;
		else if(!r&&!g&&!b)
			black++;
		else if(r&&g&&!b)
			yellow++;
		else if(r&&!g&&b)
			magenta++;
		else if(!r&&g&&b)
			cyan++;
	}
	cout<<red<<" "<<green<<" "<<blue<<" "<<yellow<<" "<<cyan<<" "<<magenta<<" "<<white<<" "<<black;
    return 0;
}
