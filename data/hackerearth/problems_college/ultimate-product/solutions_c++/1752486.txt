#include<bits/stdc++.h>
using namespace std;
typedef long long int LL;
#define FOR(i,a,b) for(int i=a;i<b;i++)
#define VI vector<int>
#define pb push_back
#define MOD 1000000007
int visited[1000001]={0};
VI g[100001];
void dfs(int src,int *sum)
{
	visited[src]=1;
	(*sum)+=1;
	int l=g[src].size();
	FOR(i,0,l)
	{
		if(visited[g[src][i]]==0)
		dfs(g[src][i],sum);
	}
}
LL foo(int a);
int main()
{
	ios_base::sync_with_stdio(false);
	int n,m;
	cin>>n>>m;
	FOR(i,0,100001)
		visited[i]=0;
	FOR(i,0,m)
	{
		int a,b;
		cin>>a>>b;
		g[a].pb(b);
		g[b].pb(a);
	}
	LL ans=1;
	FOR(i,1,n+1)
	{
		int sum=0;
		if(visited[i]==0)
		dfs(i,&sum);
		ans*=foo(sum);
		ans%=MOD;
	}
	cout<<ans<<endl;
	return 0;
}
LL foo(int a)
{
	LL sum=1;
	while(a>=1)
	{
		sum*=a;
		a--;
		sum%=MOD;
	}
	return sum;
}
