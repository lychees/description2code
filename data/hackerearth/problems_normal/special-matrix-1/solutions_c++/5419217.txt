using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    int T;
    cin>>T;
    while(T--)
    {
    cin>>n;
    char a[n][n],posi=0,posj=0;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		char ch;
    		cin>>ch;
    		if(ch=='\*')
    		{
    			//cout<<"i= "<<i<<" And j="<<j;
    			posi=i;
    			posj=j;
    		}
    	}
    }
    //cout<<"positions are:\n";
   // cout<<posi<<" abc "<<posj<<endl;
    int x=n/2;
   // cout<<"x = "<<x<<endl;
    cout<<abs(x-posi)+abs(x-posj)<<"\n";
    }

}
