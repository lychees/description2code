T = input()

def calculate(n):
	res = [[0]*(n+1),[0]*(n+1)]
	res[0][0] = 0
	res[1][0] = 0
	res[0][1] = 1
	res[1][1] = 1
	res[0][2] = 2
	res[1][2] = 1
	res[0][3] = 3
	res[1][3] = 2
	for i in range(4,n+1):
		res[0][i] = res[0][i-1] + res[1][i-1]
		res[1][i] = res[0][i-1]
	
	return res[0][n] + res[1][n]

while(T>0):
	n = input()
	if(n == 1):
		print(4)
	elif(n == 2):
		print(9)
	elif(n == 3):
		print(25)
	else:
		res = calculate(n)
		print(res**2)
	T = T-1
