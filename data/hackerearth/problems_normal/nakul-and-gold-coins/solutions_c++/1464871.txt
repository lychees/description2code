#include<bits/stdc++.h>
using namespace std;
int a[1000002];
vector<long long int> v;
vector<int> v1;
vector<int>::iterator it,it1;
void sieve()
{
    v1.reserve(100000);
    long long int i,j;
    for(i=2;i<1000001;++i)
    {
        if(!a[i])
        {
            v.push_back(i);
            for(j=i*i;j<1000001;j+=i)
            {
                a[j]=1;
            }
        }
    }
    v1.reserve(100000);
    for(i=0;i<v.size();++i)
    {
        for(j=i+1;j<v.size();++j)
        {
            if(v[i]*v[j]<=100000000)
            v1.push_back(v[i]*v[j]);
            else
                break;
        }
    }
    sort(v1.begin(),v1.end());
}
int main()
{
    sieve();
    int t,s,n;
    cin>>t;
    while(t--)
    {
        cin>>s>>n;
        it= lower_bound(v1.begin(), v1.end(), s);
        it1= upper_bound(v1.begin(), v1.end(), n);
        cout<< distance(it, it1)<<endl;
    }
    return 0;
}
