#include <iostream>
using namespace std;

int main(){


int t;
cin>>t;

while(t--){
	
	string str;
	cin>>str;
	
	int length=str.size();
	int counter=0;
	
	for(int i=0; i<length; i++){
	if(str[i]=='A' || str[i]=='E' || str[i]=='I' || str[i]=='O' || str[i]=='U' ||
	str[i]=='a' || str[i]=='e' || str[i]=='i'|| str[i]=='o' || str[i]=='u')	
	cout<<str[i];
	
	else
	
	if(str[i]!='A' || str[i]!='E' || str[i]!='I' || str[i]!='O' || str[i]!='U' ||
	str[i]!='a' || str[i]!='e' || str[i]!='i' || str[i]!='o' || str[i]!='u')
	counter++;
	}
	
	if(counter==length)
	cout<<"No"<<endl;
	else
	cout<<endl;
}
return 0;	
}
