#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
	{
		int t;
		cin>>t;
		while(t--){
		int maxx = 0;
		int n,maxi = 0;
		cin>>n;
		vector<int>arr(n);
		for(int i  =0;i<n;i++)cin>>arr[i];
		vector<int>dp;
		dp.push_back(arr[0]);
		for(int i = 1;i<n;i++)
		{
			if(arr[i]>dp.back() && arr[i]>i)dp.push_back(arr[i]);
			else if(arr[i]>i)
			{
				int l = 0;
				int r = dp.size()-1;
				int mid;
				int indx = -1;
				while(l<=r)
				{
					mid = (l+r)/2;
					if(arr[i]>dp[mid])
					{
						l = mid+1;
					}
					else if(arr[i]<dp[mid])
					{
						indx = mid;
						r = mid-1;
					}
					else {
						indx = mid;
						break;
					};
				}
				dp[indx] = arr[i];
				
			}
		}
		cout<<n-dp.size()<<endl;
		}
		
	    return 0;
	}
