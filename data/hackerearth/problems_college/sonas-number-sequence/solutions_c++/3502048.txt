#include <iostream>

using namespace std;
int main()
{
   
   long long int n;
   cin>>n;
   long long int a[n],b[n],i,j;
   for(i=0;i<n;i++)
     cin>>b[i];
     a[0]=b[0];
     cout<<a[0]<<" ";
   for(i=1;i<n;i++)
   {
   	a[i]=((i+1)*b[i])-(i*b[i-1]);
    cout<<a[i]<<" ";
   }
   cout<<"\n";
   return 0;
}
