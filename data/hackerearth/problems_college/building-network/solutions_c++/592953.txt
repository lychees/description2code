#include <iostream>
#define ll long long
using namespace std;

inline void scanint(int* x)
{
	register char c = getchar_unlocked();
	*x = 0;
	for(; (c<48)||(c>57);c = getchar_unlocked());
	for(; (c>47)&&(c<58);c = getchar_unlocked())
		*x = (int)((((*x)<<1) + ((*x)<<3)) + c - 48);
}


inline void scanl(ll* x)
{
	register char c = getchar_unlocked();
	*x = 0;
	for(; (c<48)||(c>57);c = getchar_unlocked());
	for(; (c>47)&&(c<58);c = getchar_unlocked())
		*x = (ll)((((*x)<<1) + ((*x)<<3)) + c - 48);
}


int t,n;


ll sum,mi,ma,r;

int main()
{
    scanint(&t);
    
    while(t--)
    {
    	scanint(&n);
    	sum=0;mi=100000000000000;
    	
    	for(int i=0;i<n;i++){
