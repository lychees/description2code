#include <iostream>
using namespace std;

long long f(long long x,long long n)
{
	long long res = 1;
	for(long long i=2;i*i<=x && x*2>n;i++)
	if(x%i==0){
		long long w = 1;
		while(x%i==0)
		{
			x/=i;
			w*=i;
		}
		res = max(res,w);
	};
	
	res = max(res,x);
	return res;
}
int main()
{
   	long long n;
	cin>>n;
	long long x=n/2+1;
	while(f(x,n)*2<=n)++x;
	cout<<x;
    return 0;
}
