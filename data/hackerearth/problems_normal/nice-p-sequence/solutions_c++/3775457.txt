#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
void inv  (long int a, long int b,long int *x,long int *y)
{
    if (a==0)
    {
        *x=0; *y=1; return;
    }
    inv((b%a), a, x, y);
   long int z=*x; long int zz=*y;
    *x=zz-(b/a)*z;
    *y=z;
}
void sol()
{
    long a,n,p,x,y;
    cin>>a>>p>>n;
    if(__gcd(a,p)!=1)
    {
        cout<<"-1"<<endl;
        return ;
    }
    else
    {
        x=0;y=0;
        inv(a,p,&x,&y);
        x=(x+p)%p;
        if(n%2==0)
            cout<<(x+a)*(n/2)<<endl;
        else
            cout<<(x*(n/2)+a*(n/2+1))<<endl;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)sol();
    return 0;
}
