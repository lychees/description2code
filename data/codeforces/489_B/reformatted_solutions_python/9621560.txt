b = int(raw_input())
boys = map(int, raw_input().split())

g = int(raw_input())
girls = sorted(map(int, raw_input().split()))

boys.sort()

res = 0

for i in range(b):
    for j in range(g):
        if abs(boys[i]-girls[j]) <= 1:
            girls[j] = 1000000
            res += 1
            break


print res
