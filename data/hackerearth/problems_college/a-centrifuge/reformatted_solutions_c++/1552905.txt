#include <iostream>
#include <math.h>
using namespace std;

int main() {
  int n, m, t, ans;
  cin >> t;
  while (t--) {
    cin >> m >> n;
    if (n == 0) {
      cout << "0\n";
    } else if (m >= n) {
      cout << "10\n";
    } else {
      n = 2 * n;
      if (n % m == 0)
        ans = n / m;
      else
        ans = n / m + 1;
      cout << ans * 5 << endl;
    }
  }
}
