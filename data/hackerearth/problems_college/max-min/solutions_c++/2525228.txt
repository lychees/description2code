#include <bits/stdc++.h>
using namespace std;

int main()
{
  
   int n;
   cin>>n;
   int k;
   cin>>k;
   int a[n];
   for(int i=0;i<n;i++)
   cin>>a[i];
   sort(a,a+n);
   int min=INT_MAX;
   for(int i=0;i<=n-k;i++)
   {
   	if(a[i+k-1]-a[i]<min)min=a[i+k-1]-a[i];
   }
   if(k==1)cout<<a[0]<<endl;
   else cout<<min<<endl;
   
   return 0;
}
