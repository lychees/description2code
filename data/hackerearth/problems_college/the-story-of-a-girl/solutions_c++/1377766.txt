#include <iostream>
using namespace std;
#define mod 1000000007
typedef long long ll;
void fib(int n, ll&x, ll&y){
    if(n==1){
        x = y = 1;
        return ;
    }
     
    if(n&1){
        fib(n-1, y, x);
        y=(y+x)%mod;
    }else{
        ll a, b;
        fib(n>>1, a, b);
        y = (a*a+b*b)%mod;
        x = (a*b + a*(b-a+mod))%mod;
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long n;
    	cin>>n;
    	ll x,y;
    	fib(n,x,y);
    	cout << x << "\n";
    }
    return 0;
}
