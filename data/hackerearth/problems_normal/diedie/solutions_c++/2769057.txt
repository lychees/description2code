#include <cstdio>

using namespace std;

#define MOD 1000000007

int mod_pow(long long a, long long b){
  long long ret = 1;

  while(b){
    if(b & 1) ret = ret * a % MOD;
    a = a * a % MOD;
    b >>= 1;
  }

  return ret;
}

int main(){
  int T;
  long long N;

  scanf("%d",&T);

  while(T--){
    scanf("%lld",&N);
    printf("%d\n",mod_pow(mod_pow(2,N - 1),MOD - 2));
  }

  return 0;
}
