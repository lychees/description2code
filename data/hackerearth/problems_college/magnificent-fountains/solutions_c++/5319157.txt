#include <iostream>
#include<stdio.h>
#include<bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>
using namespace std;
namespace mp = boost::multiprecision;
int main()
{
    int t;
    cin>>t;
    while(t--){
     int n;
     mp::cpp_int lcm=1,arr[10000];
      int i;
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>arr[i];
    	 for(i=0;i<n;i++){
    	 	lcm=(lcm*arr[i])/(__gcd(lcm,arr[i]));
    	 }
    	 cout<<lcm%1000000007<<"\n";
    }
    return 0;
}
