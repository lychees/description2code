#include <iostream>
using namespace std;

int main()
{	int t;
	cin>>t;
	while(t--)
	{	int r,f;
		cin>>r;
		int a,b,c,d;
		cin>>a>>b>>c>>d;
		if(a>r)
		{	f=0;
			for(int y=0;y<=b;y++)
			{	for(int z=0;z<=c;z++)
				{	for(int k=0;k<=d;k++)
					{	if(2*y+5*z+10*k<=r)
							f++;
					}
				}
			}
			
		}
		else
		{	f=0;
			for(int x=0;x<=a;x++)
			{	for(int y=0;y<=b;y++)
				{	for(int z=0;z<=c;z++)
					{	for(int k=0;k<=d;k++)
						{	if(x+2*y+5*z+10*k==r)
							f++;
						}
					}
				}
			}
	    }
		cout<<f<<"\n";
	}
}
