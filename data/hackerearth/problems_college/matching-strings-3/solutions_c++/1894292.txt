#include <iostream>
using namespace std;

int min(int a,int b)
{
	if(a<b)
	  return a;
	else
      return b; 
}

int main()
{
	int t,ctr,len1,len2,i;
    cin>>t;
    while(t--)
    {
	   string str1,str2;
       ctr=0;	
       cin>>str1;
       cin>>str2;
       len1 = str1.size();
       len2 = str2.size();
       for(i=0;i < min(len1,len2);i++)
       {
       	   if(str1[i] == str2[i])
       	      ctr++;
       }
       cout<<ctr<<endl;
    }
    return 0;
}
