#include <iostream>
#include <utility>
#include <set>

#include <unordered_map>

using namespace std;

int main()
{
   	int test ;
   	scanf("%d",&test) ; 
   	
   	while(test--)
   	{
   		string str ; 
   		cin >> str ;
   		
   		int len = str.length() ;
   		
   		unordered_map <int,char> mp ;
   		
   		for(int i = 0 ; i < len ; i++)
   		{
   			mp[str[i]]++ ; 
   		}
   		
   		set<pair<int,char>> st ;
   		
   		for(unordered_map<int,char> :: iterator it = mp.begin() ; it != mp.end() ; it++)
   		{
   			st.insert(make_pair(it->second,it->first)) ;
   		}
   		
   		for(set<pair<int,char>> :: iterator it = st.begin() ; it != st.end() ; it++)
   		{
   			int temp  =  (*it).first ;
   			while(temp--)
   				printf("%c",(*it).second) ;
   		}
   		
   		
   		printf("\n") ;
   	}
   	
   	
    return 0;
}
