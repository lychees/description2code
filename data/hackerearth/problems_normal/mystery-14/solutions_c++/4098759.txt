#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n,a;
    	cin >> n;
    	long long int sum =0;
    	while(n--)
    	{
    		cin >> a;
    		if(a%2 == 1)
    		{
    			sum = sum + a;
    		}
    	}
    	cout << sum << endl;
    }
    return 0;
}
