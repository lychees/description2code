int get(int i, ll p)
{
    if(i == 0) return S[0][p];
    if(p >= 2ll*L[i-1]) return S[i][p-L[i-1]*2];
    if(p >= L[i-1]) return get(i-1,L[i-1]*2-p-1);
    return get(i-1,p);
}
