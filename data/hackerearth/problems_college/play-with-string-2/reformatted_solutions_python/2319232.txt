t = int(raw_input())
while t:
    i = 0
    c = 0
    try:
        s = str(raw_input())

        r = s[::-1]
        for i in range(1, len(s)):
            if abs(ord(s[i])-ord(s[i-1])) == abs(ord(r[i])-ord(r[i-1])):
                c = c+1

        if c == len(s)-1:
            print("Funny")
        else:
            print("Not Funny")

    except (EOFError):
        break
