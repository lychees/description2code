

#include<bits/stdc++.h>
using namespace std;

int main(){
	
	int t;
	cin>>t;
	while(t--){
		
		string s;
		cin>>s;	
		int cnt = 1;
		for( int i=4; i<s.length(); i++ ){
			switch(s[i]){
				case 'a':
				case 'e':
				case 'i':
				case 'o':
				case 'u':
					break;
				default:
					cnt++;
			}
			//cout<<cnt<<endl;
		}
		cout<<cnt<<"/"<<s.length()<<endl;
	}
}
