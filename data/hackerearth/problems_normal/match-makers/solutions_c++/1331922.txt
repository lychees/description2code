#include<algorithm>
#include<iostream>

using namespace std;

int main(void)
{
    int t;
    cin>>t;
    for(int i=0; i<t; i++)
    {
        int n;
        cin>>n;
        int b[n], g[n];
        for(int j=0; j<n; j++)
            cin>>g[j];
        for(int j=0; j<n; j++)
            cin>>b[j];
        sort(b, b+n);
        sort(g, g+n);

        int c = 0;

        for(int j=0; j<n;j++)
        {
            if(((b[j] % g[n-j-1]) == 0) || ((g[n-j-1] % b[j]) == 0))
                c++;
        }

        cout<<c<<endl;
    }
    return 0;
}
