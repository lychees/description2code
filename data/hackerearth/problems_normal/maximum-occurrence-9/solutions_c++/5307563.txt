#include<bits/stdc++.h>
using namespace std;
int main()
{
    char s[1001];
    gets(s);
    int max=0,pos=0;
    vector<int> v(10001,0);
    for(int i=0;i<strlen(s);i++)
    {
        v[s[i]]++;// b/c char, no string so no -'a'
        if(v[s[i]]>max){
            max=v[s[i]];
        pos=i;
    }
}
if(s[pos]=='a')
    s[pos]=s[pos]-32;
cout<<s[pos]<<" "<<max;
return 0;
}
