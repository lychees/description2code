#include <iostream>
using namespace std;

int main()
{
  int t;
  cin>>t;
  while(t--){
  	int n,m,i=0;
  	cin>>n;
  	while(1){i=0;
  		m=n;
  		while(m>0){
  			if(m%2==1&&i>=0)i++;
  			else if(i>0) i--;
  			m=m/2;
  		if(i==2) m=-1;	 
  		}
  	if(m==0) break;	
  	else n++;
  	}
  	printf("%d\n",n);
  }
   return 0;
}
