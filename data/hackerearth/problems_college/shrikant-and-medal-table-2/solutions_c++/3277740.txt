void reverser(int s,int e)
{
	pair <ll,string > pe[150];

//	cout << s << e<< endl;
	for(int i =s;i <= e;i++)
	{
		pe[i-s].first = p[i].first;
		pe[i-s].second = p[i].second;
	}	

	ll k = e-s;
	for(int i =s;i <= e;i++)
	{
		 p[i].first = pe[k].first;
		p[i].second = pe[k].second;
		k--;
	}
}
