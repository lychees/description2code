#include <iostream>
#include <algorithm>

using namespace std;
long long unsigned gcd (long long unsigned a,long long unsigned b )
{
    if(b==0) return a;
    else return gcd(b,a%b);
}
int main()
{
    long t;
    cin>>t;
    while(t--)
    {
        long long unsigned a[3];
        cin>>a[0]>>a[1]>>a[2];
        sort(a,a+3);
        if(((a[0]+a[1])>a[2])&&(((a[1]*a[1]) + (a[0]*a[0]))==(a[2]*a[2])) )
        {
            long d=0;
            for(long i=0;i<3;i++)
            {
                d=gcd(d,a[i]);
            }
            if(d==1)
                {
                    cout<<"YES\n";
                }
            else
                cout<<"NO\n";
        }
        else cout<<"NO\n";
    }
    return 0;
}
