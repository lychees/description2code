#include <stdio.h>

#define mod 1000000009

long long int power (long long int base,long long int  exp)
{
	long long int res=1;
	while(exp>0)
	{
		if(exp%2==1)
		{
			res=(res*base)%mod;
			
		}
		
		base=(base*base)%mod;
		exp/=2;
	}
	return res%mod;
}

int main ()
{
	long long int t, n, an, k;
	
	scanf ("%lld", &t);
	
	while (t--) {
		scanf ("%lld", &n);
		an = ((power (10, n) - power (8, n)) * power (2, mod - 2)) % mod;
		if (an < 0) {
			an = an + mod;
		}
		printf ("%lld\n", an);
	}
	
	return 0;
}
