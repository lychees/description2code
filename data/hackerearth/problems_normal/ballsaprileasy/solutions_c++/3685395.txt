#include<iostream>
#include<algorithm>


using namespace std;

long long a[1000005];

int main(){

    long long n,i,ans=0,x=1,temp=1;
    cin>>n;
    for(i=1;i<=n;i++){
        cin>>a[i];
    }
    sort(a+1,a+n+1);
    for(i=1;i<=n;i++){
        if(i!=1){
            if(a[i]<=a[i-1])ans+=a[i-1]-a[i]+1,a[i]=a[i-1]+1;
            ans-=x;
        }
        while(x<a[i]){
            ans+=x;
            temp++;
            x=temp*temp;
        }
        ans+=(x-a[i]);
        a[i]=x;
    }
    cout<<ans;
    return 0;




}
