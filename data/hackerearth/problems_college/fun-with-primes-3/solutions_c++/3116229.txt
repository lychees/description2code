#include <iostream>
using namespace std;
long int sum_of_digits(long int);
long int isprime(long int);
int main()
{
    long int n,test,*res,i;
    cin>>test;
    res= new long int [test];
    for(i=0;i<test;i++)
	{
		cin>>n;
		res[i]=isprime(sum_of_digits(n));
	}
	for(i=0;i<test;i++)
	{
		if(res[i])
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
    return 0;
}
long int sum_of_digits(long int n)
{
	long int sum=0;
	while(n!=0)
	{
		sum+=n%10;
		n=n/10;
	}
	return sum;
}
long int isprime(long int n)
{
	int i;
	if(n==0||n==1)
	{
		return 0;
	}
	else if(n==2||n==3||n==5)
	{
		return 1;
	}
	else
	{
		for(i=2;i<=n/2;i++)
		{
			if(n%i==0)
			  return 0;
		}
		return 1;
	}
}
