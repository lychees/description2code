#include <stdio.h>
#include <set>
#include <cstdlib>
char a[1000001];
using namespace std;
int main()
{
   int t;
   long long int i,p,ans;
   scanf("%d",&t);
   while(t--){
		scanf("%s",&a);
		set<char>s;
		for(i = 0; a[i]; i++){
			s.insert(a[i]);
		}
		p = s.size();
		ans = (p * (p + 1))/2;
		printf("%lld\n",ans);
   }
   system("pause");
   return 0;
}
