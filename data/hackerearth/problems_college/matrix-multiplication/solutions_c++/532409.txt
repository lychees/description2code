#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int x1,y1,x2,y2;
        cin>>x1>>y1>>x2>>y2;

        int res = 0;

        if(y1==x2)
            res++;
        if(x1==y2)
            res++;

        cout<<res<<endl;
    }
}
