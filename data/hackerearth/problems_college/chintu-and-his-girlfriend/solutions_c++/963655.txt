#include <bits/stdc++.h>

using namespace std;

bool vis[102][32];

int main()
{
	int n,k;
	string s;
	cin >> n >> k;
	for ( int i = 0; i < n; i++ ) {
		cin >> s;
		for ( int j = 0; j < s.size(); j++ ) vis[i][s[j]-97] = true;
	}
	int cnt = 0;
	for ( int j = 0; j < 26; j++ ) {
		for ( int i = 0; i < n; i++ ) {
			if ( !vis[i][j] ) goto p1;
		}	
		cnt++;
		p1: { }
	}
	if ( cnt >= k ) cout << ":-)" << endl;
	else cout << ":-(" << endl;
	return 0;
}
