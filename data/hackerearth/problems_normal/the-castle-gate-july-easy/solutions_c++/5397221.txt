#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int c=0;

    int a[t];

    for(int i=0;i!=t;i++){
        cin>>a[i];
    }

    for(int i=0 ; i < t ; i++){
        for(int j=1 ; j<=a[i] ; j++){
            for(int k = j+1 ; k<=a[i] ; k++){

            if((j^k)<=a[i]){

                c++;
            }
    }

    }
    cout<<c<<endl;
    c=0;
    }
    return 0;
}
