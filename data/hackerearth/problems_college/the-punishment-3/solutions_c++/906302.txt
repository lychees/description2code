#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int N,K;
		cin>>N>>K;
		vector<int> A(N+1);
		A[N]=1;
		if(K>N-K)
		{
			K=N-K;
		}
		A[N]=0;
		for(int i=N-1;i>=K;i--)
		{
			A[i]=9;
		}
		A[N-K]--;
		A[0]=1;
		for(int i=N-1;i>=0;i--)
		{
			cout<<A[i];
		}
		cout<<endl;

	}
	return 0;
}
