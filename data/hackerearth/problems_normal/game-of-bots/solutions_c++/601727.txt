#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
string A,B;
int T;
int main()
{
	cin>>T;
	while(T--)
	{
		cin>>A>>B;
		vector<int> A1,A2,B1,B2;
		for(int i=0;A[i];i++)
		if(A[i]=='A')
		{
			A1.push_back(i);
		}
		else if(A[i]=='B')
		{
			B1.push_back(i);
		}
		
		for(int i=0;B[i];i++)
		if(B[i]=='A')
		{
			A2.push_back(i);
		}
		else if(B[i]=='B')B2.push_back(i);
		
		if(A1.size()!=A2.size()||B1.size()!=B2.size())cout<<"No\n";
		else
		{
			int n=A1.size(),m=B1.size();
			bool flag=true;
			for(int i=0;i<n;i++)
			if(A1[i]<A2[i])
			{
				flag=false;
				break;
			}
			
			if(flag)
			{
				for(int i=0;i<m;i++)
				if(B1[i]>B2[i])
				{
					flag=false;
					break;
				}
			}
			flag?cout<<"Yes\n":cout<<"No\n";
		}
	}
	return 0;
}
