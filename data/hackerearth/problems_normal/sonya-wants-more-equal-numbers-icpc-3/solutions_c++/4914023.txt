const int N = 1e5+5;
int a[N];
int f[N];
int req;
int n;
int dp[300];
int rec(int pos) {
  if(pos > n ) {
    return 0;
  }
  if(dp[pos]!=-1) return dp[pos];
  int ans = 0;
  ans = max(ans, rec(pos+1));
  int sum = 0;
  for(int i = pos ; i <= n ; i++) {
    sum += a[i];
    if(sum > req) break;
    if(sum == req) {
      ans = max(ans,1 + rec(i+1));
    }
  }
  return dp[pos] = ans;
}
bool cmp( pii a , pii  b ) {
  if( a.fr == b.fr ) return a.se < b.se;
  return a.fr > b.fr;
}
int main() {
  cin >> n ;
  for(int i = 1 ; i<= n ; i++) {
    sci(a[i]);
  }
  for(int i = 1 ; i<=  n ; i++) {
    int sum = 0;
    for(int j = i ; j<=n ; j++) {
      sum += a[j];
      f[sum] = 1;
    }
  }
  vector< pii >  ans;
  for(int i = 1 ; i<= 100000 ;i++) {
    if(f[i]) {
      M(dp,-1);
      req = i;
      int temp = rec(1);
      ans.pb(mp(temp,i));
    }
  }
  sort(ans.begin(),ans.end(),cmp);
  int mx = ans[0].fr;
  cout<<mx<<endl;
  for(int i = 0 ; i < ans.size() ; i++) {
    if(ans[i].fr < mx) {
      break;
    }
    cout<<ans[i].se<<" ";
  }
  cout<<endl;
}
