#include <bits/stdc++.h>
using namespace std;

long long gcd(long long a,long long b)
{
    if(b > a)
        swap(a,b);
    while(b!=0)
    {
        int temp = a;
        a = b;
        b = temp%b;
    }
    return a;
}
int main()
{
   long long n;
   cin>>n;
   long long x[n];
   long long y[n];
   for(long long i = 0;i< n;i++)
   {
       cin>>x[i]>>y[i];
   }
   long long sum= 0;
   pair<long long,long long> p[n];
   for(long long i = 1; i  <= n;i++)
   {
       sum+=x[i-1]*y[i%n] - y[i-1]*x[i%n];
   }
    sum = abs(sum/2);
    sum+=1;
    long long bound = 0;
   for(long long i = 1;i<= n;i++)
   {
       long long a = abs(x[i%n] - x[i-1]);
       long long b = abs(y[i%n] -y[i-1]);
       bound+=gcd(a,b);

   }
   long long ans = sum - (bound/2);
    cout<<ans<<endl;

}
