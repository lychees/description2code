t=int(raw_input())
i=1
b=0
while t:
    x=long(raw_input())
    while x>=10:
        b=0
        while x>0:
            a=x%10
            b=(a*a)+b
            x=x/10
        x=b
    if x == 1:
        print "Case #"+str(i)+":YES"
    else:
        print "Case #"+str(i)+":NO"
    t=t-1
    i=i+1
