vector<int> Prefix(string str)
{
	int len = str.size();
	vector<int> vec(len, 0);
	for(int i = 1; i < len; ++i) {
		int j = vec[i - 1];
		while(j > 0 && str[i] != str[j])
			j = vec[j - 1];
		if(str[i] == str[j])
			vec[i] = j + 1;
	}
	
	return vec;
}
