#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int l1=0,l2=0;
    	int p=1;
    	int count=0;
    	if(s.length()==1)
    	{
    		int i=0;
    		if(s[i]=='a' || s[i]=='e' || s[i]=='o'||s[i]=='i'||s[i]=='u'||s[i]=='y')
    			cout<<"easy"<<endl;
    		else
    			cout<<"hard"<<endl;
    	}
    	else
    	{
    	//check if consecutive
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='a' || s[i]=='e' || s[i]=='o'||s[i]=='i'||s[i]=='u'||s[i]=='y')
    		{
    			l1++;
    			count=0;
    		}
    		else
    		{
    			count++;
    			l2++;
    		}
    		if(count==3)
    		{
    			p=0;
    			break;
    		}
    	}
    	if(p==0)
    		cout<<"hard"<<endl;
    	else if(l2>l1)
    		cout<<"hard"<<endl;
    	else
    		cout<<"easy"<<endl;
    	}
    }
    return 0;
}
