#include<iostream>


using namespace std;

long maximumXOR(unsigned long long l, unsigned long long r) {
    unsigned long long q = l ^ r, a = 1;
    while(q){
        q /= 2;
        a <<= 1;
    }
    return --a;
}


int main()
{
	long T;
	
	cin>>T;
	
	for(long i = 0;i < T; i++)
	{
		unsigned long long n;

		cin>>n;

		if(n>1){
		unsigned long long k =maximumXOR(0, n-1);
		
		cout<<k<<" "<< (n-(k+1)/2)*2<<endl;
		}
		else
			cout<<"0 1"<<endl;
	}	

	return 0;
}
