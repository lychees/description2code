n = input()
name = []
counter = []
alpha = []
for i in range(0, n):
    book = raw_input()
    bo = list(book)
    name.append(bo[0])

for i in range(0, len(name)):
    if name[i] not in alpha:
        counter.append(name.count(name[i]))
        alpha.append(name[i])

shelves = 0
for i in counter:
    if (i % 10 != 0):
        shelves = shelves+(int(i/10)+1)
    else:
        shelves = shelves+(int(i/10))

print(shelves)
