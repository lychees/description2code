#include<bits/stdc++.h>
#define ll long long
using namespace std;
 
int main() {
    int N;
    cin>>N;
    int ans=0;
    for(int i=0;i<N;i++)
    {
        string A;
        cin>>A;
        int  n= A.size();
        stack<char> st;
        //st.clear();
        for(int j=0;j<n;j++)
        {
            if(st.empty())
            {
                st.push(A[j]);
            }
            else
            {
                if(st.top()==A[j])
                st.pop();
                else
                st.push(A[j]);
            }
            //cout<<st.size()<<endl;
        }
        if(st.empty())
        ans++;
    }
    cout<<ans;
    }
