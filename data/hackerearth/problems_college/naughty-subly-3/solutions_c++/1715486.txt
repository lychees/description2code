#include <iostream>
using namespace std;

int main()
{
    int a[100000];
    int n;
    cin>>n;
    int count=0;
    int i=0,j=0;
    int b[1000];
    int k=0;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	for(j=0;j<i;j++)
    	{
    	
    		if(a[i]==a[j])
    		{
    			count++;
    			break;
    		}
    			
    	}
    	if(j==i)
    		b[k++]=a[i];
    }
    cout<<"\n"<<n-k;
    return 0;
}
