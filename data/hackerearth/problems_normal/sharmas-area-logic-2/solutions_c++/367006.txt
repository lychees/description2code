#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
    int t;scanf("%d",&t);
    while(t--){
        int n,x,y;
        scanf("%d",&n);
        int minx=200,miny=200,maxx=-200,maxy=-200;
        for(int i=0;i<n;i++){
            scanf("%d%d",&x,&y);
            minx=min(minx,x);miny=min(miny,y);
            maxx=max(maxx,x);maxy=max(maxy,y);
        }    
        int area=(maxx-minx)*(maxy-miny);
        if(area==0){area=(maxx-minx)+(maxy-miny);}
        printf("%d\n",area);
    }    
    return 0;
}
