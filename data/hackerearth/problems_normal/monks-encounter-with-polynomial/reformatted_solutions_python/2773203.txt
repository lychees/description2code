def calceq(x, a, b, c):
    return (a*(x**2))+(b*x)+c


for _ in xrange(int(raw_input())):
    a, b, c, k = map(int, raw_input().split())
    l = 0
    u = 10**5
    pos = 0
    while l <= u:
        mid = (l+u)/2
        if calceq(mid, a, b, c) >= k:
            u = mid-1
            pos = mid
        else:
            l = mid+1
    print pos
