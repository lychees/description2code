#include<iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;
int n;
char mat[100][100];

int horizontal()
{
	for(int i=0;i<n/2;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(mat[i][j]!=mat[n-1-i][j])
			return 0;
		}
	}
	return 1;
}
int vertical()
{
	for(int i=0;i<n/2;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(mat[j][i]!=mat[j][n-i-1])
			return 0;
		}
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	
		cin>>n;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				cin>>mat[i][j];
			}
		}
		int hori=horizontal();
		int vert=vertical();
		if(hori&&vert)
		cout<<"BOTH"<<endl;
		else if(hori)
		cout<<"HORIZONTAL"<<endl;
		else if(vert)
		cout<<"VERTICAL"<<endl;
		else
		cout<<"NO"<<endl;
	}
}
