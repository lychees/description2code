#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,i;
    int n,k;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	if(n<=5)
    	cout<<pow(10,n)-pow(10,k)-pow(10,n-k)+1<<endl;
    	else
    	{int a[100000]={0};
    	a[n]=1;
    	for(i=k;i<n;i++)
    	a[i]=9;
    	if(n>=2*k)
    	{
    		a[n-k]-=1;
    	}
    	else
    	{
    		if(n<2*k)
    		
    		{
    			for(i=n-k;;i++)
    			{
    				if(a[i]>0)
    				{a[i]-=1;
    				break;}
    				else
    				a[i]=9;
    			}
    		}
    	}
    	a[0]=1;
    	
    	for(i=n-1;i>=0;i--)
    	cout<<a[i];
    	cout<<endl;
    	}
    
    }
    
    return 0;
}
