#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, k;
	cin >> n >> k;
	map<int, vector<int>> m;
	int temp1, temp2;
	for (int i = 0; i<n; i++)
	{
		cin >> temp1;
		temp2 = temp1%k;
		m[temp2].push_back(temp1);
	}
	map<int, vector<int>>::reverse_iterator rit;
	for (rit = m.rbegin(); rit != m.rend(); rit++)
	{
		sort(m[rit->first].begin(), m[rit->first].end());
		int len = m[rit->first].size();
		for (int j = 0; j<len; j++)
		{
			cout << m[rit->first][j] << " ";
		}
	}
	return 0;
}
