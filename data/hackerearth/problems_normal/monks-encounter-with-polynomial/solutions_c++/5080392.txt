#include<bits/stdc++.h>
using namespace std;
#define MIN -0.5

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int a,b,c,k,p;
    	cin>>a>>b>>c>>k;
    	p=c;
    	c=c-k;
    	long int det=b*b-4*a*c;
    	long int x0,x1;
    	x0=(-b-sqrt(det))/(2*a);
    	x1=(-b+sqrt(det))/(2*a);
    	if(x0>x1)
    	{
    		if(x0*x0*a+b*x0+p>=k)
    		{
    			if(x0<MIN)
    				cout<<0<<endl;
    			else
    				cout<<x0<<endl;
    		}
    		else
    		{
    			if((x0+1)<MIN)
    				cout<<0<<endl;
    			else
    				cout<<(x0+1)<<endl;
    		}
    	}
    	else
    	{
    		if(x1*x1*a+b*x1+p>=k)
    		{
    			if(x1<MIN)
    			cout<<0<<endl;
    			else
    			cout<<x1<<endl;
    		}
    		else
    		{
    			if((x1+1)<MIN)
    			cout<<0<<endl;
    			else
    			cout<<(x1+1)<<endl;
    		}
    	}
    }
}
