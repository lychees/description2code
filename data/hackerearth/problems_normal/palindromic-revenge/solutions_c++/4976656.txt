#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    int cs;
    string st;
    cin >> cs;
    for (int i = 0; i < cs; ++i)
    {
        int ans = 0;
        cin >> st;
        for (int i = 0, j = st.size() - 1; i < st.size()/2 ; ++i, --j)
        {
            ans += abs(st[i]-st[j]);
        }
        cout << ans << endl;
    }

    return 0;
}
