#include<bits/stdc++.h>
using namespace std;
vector<int> vt;
bool cmp(pair<int,int> a, pair<int,int> b)
{
		if(a.first < b.first) return true;
		else if(a.first > b.first) return false;
		else
		{
				if(a.second < b.second) return true;
				else return false;
		}
}
int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				vt.clear();
				int n;
				cin>>n;
				int tmp;
				int i;
				for(i=0;i<n;++i)
				{
						cin>>tmp;
						vt.push_back(tmp);
				}
				int p,q;
				cin>>p>>q;
				vector<pair<int,int> > ans;
				vector<int>::iterator it,ip;
				for(ip = vt.begin(); ip !=vt.end(); ++ip)
						if(__gcd(*ip,p) != 1) ans.push_back(make_pair(abs(p-*ip),*ip));
				if(ans.size() < q) cout<<-1<<endl;
				else
				{
						vector<pair<int,int> >::iterator ips;
						sort(ans.begin(),ans.end());
						int cnt = 0;
								vt.clear();
						for(ips = ans.begin(); ips!=ans.end(); ++ips)
						{
								vt.push_back(ips->second);
								cnt++;
								if(cnt == q) break;
						}
						sort(vt.begin(),vt.end());
						for(it=vt.begin();it!=vt.end();++it)
								cout<<*it<<" ";
						cout<<endl;
				}

		}
		return 0;
}
