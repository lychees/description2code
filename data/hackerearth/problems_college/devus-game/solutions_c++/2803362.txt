#include <cstdio>
#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int T,N;
    cin >> T;
    while(T--)
    {
        cin >> N;
        vector<int> input;
        vector<bool> visited;
        input.resize(N); visited.resize(N);

        for(int i = 0; i < N; ++i)
        {
            cin >> input[i];
            visited[i] = false;
        }

        int i = 0;
        bool win = true;
        while(input[i] != -1)
        {
            i = input[i];
            if(visited[i] == true)
            {
                win = false;
                break;
            }
            else
            {
                visited[i] = true;
            }
        }
        win ? printf("Win\n") : printf("Lose\n");
    }
    return 0;
}
