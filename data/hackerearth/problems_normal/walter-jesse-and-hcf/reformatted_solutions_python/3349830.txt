from fractions import gcd
import math


def kFactorEven(a, maxA, curK, gcd, lowerList):
    if a >= maxA:
        if a == maxA and gcd % a == 0:
            lowerList.append(a)
        if curK > len(lowerList):
            return -1
        else:
            return lowerList[len(lowerList) - curK]
    if gcd % a == 0:
        if curK - 1 == 0:
            return gcd / a
        else:
            lowerList.append(a)
            return kFactorEven(a+1, maxA, curK-1, gcd, lowerList)
    else:
        return kFactorEven(a+1, maxA, curK, gcd, lowerList)


def kFactorOdd(a, maxA, curK, gcd, lowerList):
    if a >= maxA:
        if a == maxA and gcd % a == 0:
            lowerList.append(a)
        if curK > len(lowerList):
            return -1
        else:
            return lowerList[len(lowerList) - curK]
    if gcd % a == 0:
        if curK - 1 == 0:
            return gcd / a
        else:
            lowerList.append(a)
            return kFactorOdd(a+2, maxA, curK-1, gcd, lowerList)
    else:
        return kFactorOdd(a+2, maxA, curK, gcd, lowerList)

for _ in xrange(T):
