int dist(int r,int c)
{
//	cout<<r<<" "<<c<<endl;
	if(count[r][c]!=-1)
		return count[r][c];
	if(r==k&&c==k)
	{
		if(a[r][c]=='1')
	{
		flag=1;
		return 1;
		
	}
		else
			return 0;
	}
	int c1,c2,f1=0,f2=0;
	c1=c2=1;
	if((r+1)<=k)
	{
		if(a[r+1][c]!='0')
			c1+=dist(r+1,c);
		else
			c1=0;
	}
	if((c+1)<=k)
	{
		if(a[r][c+1]!='0')
			c2+=dist(r,c+1);
		else
			c2=0;
	}
/*	if(((c+1)<=n)&&((r+1)<=m)&&(a[r+1][c+1]!='#'))
	{
		flag=3;
		if(a[r][c]=='*')
			c3++;
		c3+=dist(r+1,c+1);
	}
*/
	c1=max(c1,c2);
	count[r][c]=c1;
	return count[r][c];
}
