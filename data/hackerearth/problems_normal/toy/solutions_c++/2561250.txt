#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,a[100001],i,j,ans,x,cnt;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
   // memset(cnt,0,sizeof(cnt));
    ans=0;
    sort(a,a+n);
    for(i=0;i<n;)
    {
    	ans++;
    	cnt=0;
    	x=a[i]+4;
    	for(j=i+1;j<n;j++)
    	{
    		if(a[j]<=x)
    		{
    			cnt++;
    		}
    	}
    	i=i+cnt+1;
    	
    }
    cout<<ans<<endl;
    return 0;
}
