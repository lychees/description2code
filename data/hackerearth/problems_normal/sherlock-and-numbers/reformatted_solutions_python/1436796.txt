t = input()
for ____ in xrange(t):
    n, k, p = map(int, raw_input().split())
    l = map(int, raw_input().split())
    l = sorted(l)
    x = 1
    if (n - k) < p:
        print -1
    else:
        for i, j in enumerate(l):
            if p > (j - x):
                p = p - j + x
                x = j + 1
            else:
                break
        print x + p - 1
