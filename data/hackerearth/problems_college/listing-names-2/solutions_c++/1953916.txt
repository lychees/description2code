#include <iostream>
#include<map>
using namespace std;

int main()
{  map<string,int> map1;
   map<string , int>::iterator it;  
   int t;cin>>t;
   string s;
   while(t--)
   { cin>>s;
    it=map1.find(s);
    if(it==map1.end())
       map1[s]=1; 
    else
      (it->second)++;
     	
   }
   for(it=map1.begin();it!=map1.end();it++)
     cout<<it->first<<" "<<it->second<<endl;
    	
    
    return 0;
}
