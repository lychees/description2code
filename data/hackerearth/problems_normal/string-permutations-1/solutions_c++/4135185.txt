#include<iostream>
#include<map>
using namespace std;
int main()
{
	int t;cin>>t;
	int max=-1;int c=1;
	map<string,int>str;
	while(t--)
	{
		string s;
		cin>>s;
		str[s]++;
		if(str[s]>max)
		{
			max=str[s];
			c=1;
		}
		else if(str[s]==max)
		{
			c++;
		}
	}
	cout<<c*max;
}
