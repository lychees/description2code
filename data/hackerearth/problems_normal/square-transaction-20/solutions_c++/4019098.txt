#include<iostream>
using namespace std;
#define ull long long int
int main()
{
    ios::sync_with_stdio(false);
    ull t,q,i,n,low,high;
    cin>>t;
    ull arr[t+1];
    for(i=0;i<t;i++)
        cin>>arr[i];
    for(i=1;i<t;i++)
        arr[i]+=arr[i-1];
    cin>>q;
    while(q--)
    {
        cin>>n;
        low=0;high=t-1;
        if(n>arr[high])
            cout<<"-1\n";
        else{
            while(low<=high)
            {
                ull mid=(low+high)>>1;
                if(low==high && arr[low]<n){
                    cout<<low+2<<"\n";
                    break;
                }
                else if(low==high && arr[low]>n){
                    cout<<low+1<<"\n";
                    break;
                }
                else if(arr[mid]==n){
                    cout<<mid+1<<"\n";
                    break;
                }
                else if(arr[mid]>n)
                    high=mid;
                else
                    low=mid+1;
            }
        }
    }
    return 0;
}
