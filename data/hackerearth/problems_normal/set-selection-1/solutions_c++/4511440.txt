#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 2000000

int dp[MAX];
int main()
{
  int n;
  cin>>n;int i,j;
  int arr[n+1];
  dp[0]=1;
  for(i=1;i<=n;i++)
  {
    cin>>arr[i];
  }

  for(i=1;i<=n;i++)
  {
    for(j=MAX-1-arr[i];j>=0;j--)
    {
      if (dp[j]==1)
      dp[j+arr[i]]=1;
    }
  }
  int test;cin>>test;int t;
  int z;
  for(t=0;t<test;t++)
  {
    cin>>z;
    if (dp[z]==1)
    cout<<"Yes\n";
    else
    cout<<"No\n";
  }

return 0;
}//main
