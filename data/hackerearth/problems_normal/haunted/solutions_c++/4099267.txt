int main(){

    #ifdef LOCAL_MACHINE
    freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
    #endif // LOCAL_MACHINE
    int n,m;
    int max_awards=0;
    cin>>n>>m;
    priority_queue<int> awards;
    unordered_map<int,int> total;
    while(n--){
        cin>>m;
        total[m]+=1;
        if(max_awards<total[m]){
            max_awards=total[m];
            cout<<m<<" "<<max_awards<<endl;
            awards=priority_queue<int>();
            awards.push(m);
        }
        else{
            if(max_awards==total[m]){
                awards.push(m);
            }
            cout<<awards.top()<<" "<<max_awards<<endl;
        }
    }
}
