#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
    int i,t,n,a[100001];
    cin>>t;
    while(t--)
    {
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    for(i=n-1;i>=2;i--)
    {
    if((a[i]+a[i-1]>a[i-2])&&(a[i-1]+a[i-2]>a[i])&&(a[i]+a[i-2]>a[i-1]))
    {
    cout<<a[i-2]<<" "<<a[i-1]<<" "<<a[i]<<endl;
    break;
    }
    }
    if(i<2)
    cout<<"-1"<<endl;
    }
    return 0;
}
