#include <iostream>
#include<vector>
#include<stdlib.h>
using namespace std;
#define min(x,y) (x<y?x:y)
int n;
vector<unsigned long long >res;
unsigned long long cal(int x,int y)
{
	unsigned long long c=0;
	res.clear();
	res.push_back(abs(x-1)+abs(y-1));
	res.push_back(abs(x-1)+abs(n-y));
	res.push_back(abs(n-x)+abs(n-y));
	res.push_back(abs(n-x)+abs(y-1));
	c=res[0];
	for(int i=1;i<4;i++)
	c=min(res[i],c);
	return c;
}
int main()
{
	int i,k;
	unsigned long long ans=0;

	cin>>arr[i];
	for(i=0;i<k;i++)
	cin>>brr[i];
	for(i=0;i<k;i++)
	ans+=cal(arr[i],brr[i]);
