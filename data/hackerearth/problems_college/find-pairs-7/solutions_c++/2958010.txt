#include <iostream>
#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		unordered_map<int ,int>m;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			m[a[i]]++;
		}
		long long ctr=n;
		for(auto i:m)
		{
			if(i.second>1)
			{
				ctr+=(long long)(i.second)*(i.second-1)/2;
			}
		}
		cout<<ctr<<endl;
	}
//    cout << "Hello World!" << endl;
    return 0;
}
