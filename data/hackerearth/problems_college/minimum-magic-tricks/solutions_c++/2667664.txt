using namespace std;
ll ans[1000005];
ll power(ll a,ll b)
{
	if(b==0)
	{
		return 1;
	}
	ll x = power(a,b/2);
	if(b%2)
	{
		return (x*x*a)%mod;
	}
	return (x*x)%mod;
}
int main() 
{
	ios_base::sync_with_stdio(0);
	ll t;
	cin>>t;
	ans[1] = 1;
	ans[2] = 1;
	for(ll i=3;i<=10004;i++)
	{
		ans[i] = (ans[i-2] + power(2,i-2))%mod;
	}
	while(t--)
	{
		ll n;
		cin>>n;
		if(n==1)
		{
			cout<<"1\n";
			continue;
		}
		cout<<ans[n]<<"\n";
	}
	return 0;
}
