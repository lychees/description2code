map<int,int> primeFactorize(int N){

	map<int,int> ret;
	int expo = 0;
	int NN = sqrt(N)+1;

	while(N%2 == 0){
		expo++;
		N /= 2;
	}

   if(expo)
      ret[2] = expo;
   //pi(N);

   for(int i = 3; i <= NN; i += 2){

        expo = 0;
		while( N%i == 0 ){
		  expo++;
		  N /= i;
		}

        if(expo)
          ret[i] = expo;
	}

	if(N>1)
        ret[N] = 1;

	return ret;
}

bool isWinning(map<int,int> Primo, int N){
   //pi(Primo);
   if(N == 1)
      return true;
   if(Primo.size() == 1){
      //puts("I am here");
      return true;
   }

   for(map<int,int>::iterator itr = Primo.begin(); itr != Primo.end(); itr++){

      int XX = itr->second;

      for(int i = 1; i <= XX; ++i){
        int pw = int ( pow(itr->first,i) );
        //pi(pw);
        itr->second = itr->second-i;
        map<int,int> pr = Primo;
        if(itr->second == 0){
            pr.erase(itr->first);
        }
        if(!isWinning(pr,N/pw))
           return true;
        itr->second = itr->second+i;
      }
   }
   return false;
}
