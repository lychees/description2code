#include<iostream>
#include<bits/stdc++.h>
#define all(a) a.begin(),a.end()
using namespace std;

int a[101];
int main(){
	int t,n,i,k,j,cost;
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++){
			scanf("%d",&a[i]);
		}
		int ans=1e9;
		for(i=1;i<=100;i++){
			vector<int>v;
			for(j=0;j<n;j++){
				if(i>=a[j]){
				cost=(i-a[j])*3;
				}
				else{
					cost=(a[j]-i)*5;
				}
				v.push_back(cost);
			}
			sort(all(v));
			cost=0;
			for(j=0;j<k;j++){
				cost+=v[j];
			}
			ans=min(ans,cost);
		}
		printf("%d\n",ans);
	}

	return 0;
}
