#include <iostream>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    
    while(t--) {
    	
    	int i;
    	int nr, ng;
    	cin >> nr >> ng;	
 
    	int wealth[nr];
    	
    	for(i=0;i<nr;++i)
    		cin >> wealth[i];
    		
    	int h = 0, myvalue = wealth[0];
    	
    	sort(wealth, wealth+nr);
    	
    	for(i=nr-1;i>=0;--i)
    		if(myvalue == wealth[i])
    			break;
    //	i += 1;
    	
    		while(i <= nr-1){
    		
    			if(i+ng > nr-1)		
    				break; 
    			else if(wealth[i+ng] > myvalue)
    				++h;
    				
    			i = i+ng;
    		
    		}
    		cout << h << endl;
   }
    return 0;
}
