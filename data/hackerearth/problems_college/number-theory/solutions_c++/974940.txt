#include <iostream>
#include<cmath>
using namespace std;

int main()
{
 int t;
 cin>>t;
 while(t--)
  {
  	int64_t n,a[10]={0},cnt=0,i;
  	cin>>n;
  	int64_t x=n;
  	while(x>0)
  	 {
  	  if(a[x%10]==0)
  	   a[x%10]=1;
  	   x=x/10;
  	 }
  	 if(n==1)
  	  {cout<<1<<"\n";
  	  continue;}
  	  int64_t tmp=sqrt(n);
  	for(i=1;i<=tmp;i++)
  	 {
  	 	if(n%i==0)
  	 	 {
  	 	 	int64_t y=i;
  	 	 	while(y>0)
  	         {
  	         if(a[y%10]==1)
  	             {cnt++;break;}
  	             y=y/10;
  	        }
  	        
  	        y=n/i;
  	        if(i==y)
  	        continue;
  	        while(y>0)
  	         {
  	         if(a[y%10]==1)
  	             {cnt++;break;}
  	             y=y/10;
  	        }
  	 	 }
  	 }
  	 cout<<cnt<<"\n";
  }
}
