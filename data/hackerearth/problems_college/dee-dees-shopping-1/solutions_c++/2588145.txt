#include<bits/stdc++.h>
using namespace std;
#define ll long long 

int dp(int n)
{
    if(n==0) return 0;
    if(n<0) return 100000000;
    return min(1+dp(n-8), 1+dp(n-6));
}

int main()
{
    
    int t; cin>>t;
    
    while(t--)
    {
        int n; cin>>n;
        int temp=dp(n);
        if(temp>=100000000) cout<<"-1\n";
        else cout<<temp<<"\n";
    }
    return 0;
}
