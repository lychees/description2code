#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define MAX 1000000
#define INF 100000000
int dp[1000000+1];
int a[]={2,3,5,7};
int minx(int a,int b){
	if(a>b){
		return b;
	}
	return a;
}
int main()
{
	int i,j;
    for(i=0;i<=MAX;i++){
    	dp[i]=INF;
    }
    int t,max=0;
    cin>>t;
    int b[t];
    for(i=0;i<t;i++){
    	cin>>b[i];
    	if(b[i]>max){
    		max=b[i];
    	}
    }
    dp[2]=1;
    dp[3]=1;
    dp[5]=1;
    dp[7]=1;
    for(i=4;i<=max;i++){
    	if(i!=5 && i!=7){
    		for(j=0;j<4;j++){
    			if(i-a[j]>=0){
    				dp[i]=minx(dp[i],dp[i-a[j]]+1);
    			}
    		}	
    	}
    }
    for(i=0;i<=max;i++){
    	if(dp[i]==INF){
    		dp[i]=-1;
    	}
    }
    for(i=0;i<t;i++){
    	cout<<dp[b[i]]<<endl;
    }
    return 0;
}
