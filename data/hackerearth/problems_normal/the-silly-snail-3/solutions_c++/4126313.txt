#include<iostream>
#include<cstdio>
#include<cstdlib>
using namespace std;
int l[100001],r[100001];

void preorder(long int x)
{
	cout << x << " ";
	if(l[x]!=-1 && l[x]!=0)
	preorder(l[x]);
	if(r[x]!=-1 && r[x]!=0)
	preorder(r[x]);
}

int main()
{
	int t;
	cin >> t;
	long int n,x,y,z;
	while(t--)
	{
		cin >> n;
		for(int i=0;i<100001;i++)
		{
			l[i]=-1;
			r[i]=-1;
		}
		for(int i=0;i<n;i++)
		{
			cin >> x >> y >> z;
			l[x]=y;
			r[x]=z;
		}
		preorder(1);
		cout << endl;
	}
	return 0;
}
