    void mul(ll a[100][100],ll b[100][100]){
        int i,j,k;
        f(i,0,2)
            f(j,0,2)
                c[i][j]=0;
        f(k,0,2){
            f(i,0,2){
                f(j,0,2){
                    c[i][j]=(c[i][j]+(a[i][k]*b[k][j])%MOD)%MOD;
                }
            }
        }
    }
     
    void cal(){
        int i,j;
        ans[0][0]=1;ans[0][1]=0;ans[1][0]=0;ans[1][1]=1;
        z[0][0]=1;z[0][1]=1;z[1][0]=1;z[1][1]=0;
        while(n>0){
            if(n&1){
                mul(ans,z);
                f(i,0,2)
                    f(j,0,2)
                        ans[i][j]=c[i][j];
                n--;
            }
            else{
                mul(z,z);
                f(i,0,2)
                    f(j,0,2)
                        z[i][j]=c[i][j];
                n>>=1;
            }
        }
    }
