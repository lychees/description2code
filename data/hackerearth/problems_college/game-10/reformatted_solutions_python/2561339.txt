import sys

for __ in range(input()):
    a, b = map(int, sys.stdin.readline().split())
    if b < (a/2):
        ans, k = 1, 0
        while k != b:
            ans += 2
            k += 1
    else:
        ans, k = 0, a-1
        while k != b:
            ans += 2
            k -= 1
    print ans
