# print 'Hello World!'


def prime(n):
    for i in range(2, n):
        if n % i == 0:
            return 0
    return 1


t = input()
while t > 0:
    t -= 1
    s = raw_input()
    p = ""
    for i in s:
        if prime(ord(i)) == 0:
            p += i
    if len(p) > 0:
        print p
    else:
        print"NULL"
