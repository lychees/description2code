#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int main()
{
    int n,k,l=0,j;
    scanf("%d%d",&n,&k);
    int b[27]={0};
    for(j=1;j<=n;j++)
    {
    	
    	char a[200]={0};
    	scanf("%s",a);
    	int s;
    	s=strlen(a);
    	for(int i=0;i<s;i++)
    	{
    		if(b[a[i]-97]==j-1)
    		{
    			b[a[i]-97]=j;
    		}
    	
    	}
    	
    }
    l=0;
    for(int i=0;i<=26;i++)
    {
    	if(b[i]==n)
    	{
    		l=1;
    	}
    //	printf("%d ",b[i]);
    }
    if(l)
    	{
    		printf(":-)\n");
    	}
    	else
    	printf(":-(\n");
	return 0;
}
