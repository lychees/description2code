#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;cin>>t;
	while(t--){
		int n, k;cin>>n>>k;
		int a[1010]={0};
		for(int i=0; i<n; i++){
			string str="";cin>>str;
			for(int j=0; j<k; j++){
				if(str[j] == '1')
					a[i]|=(1<<j);
			}
		}
		int ans=INT_MAX;
		for(int i=0; i<(1<<k); i++){
			int j, cnt=__builtin_popcount(i);
			for(j=0; j<n; j++)
				if(!(a[j]&i))break;
			if(j == n)
				ans=min(ans, cnt);
		}
		cout<<ans<<endl;
	}
	return 0;
}
