#include <iostream>
typedef long long ll;
using namespace std;
ll n,q,a,c,b[1000000];
ll bc(ll n,ll k)
{ll l=1;
 ll h=n;
 ll m;
 while(l<h)
 { m=(l+h)/2;
 if(b[m]<k)
 l=m+1;
 else if(b[m]>k)
 h=m-1;
 else
 return m;
 }
 return l;
}
int main()
{
    cin>>n;
    b[0]=0;
    for(int i=1;i<=n;i++)
    {cin>>a;
      b[i]=b[i-1]+a;
    }
    cin>>q;
    while(q--)
    {cin>>c;
     ll g=bc(n,c);
     if(c>b[n])
     cout<<-1<<endl;
     else if(b[g]<c)
     cout<<g+1<<endl;
     else
     cout<<g<<endl;
    }
      
    
    return 0;
}
