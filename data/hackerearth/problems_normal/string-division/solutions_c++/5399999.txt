#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	
    string s;
    cin>>s;
    int n=s.size();
    string first,second,third,fourth;
    int i,j,k;
    int flag=0;
    set<string> temp;
    if(n>=10){
        cout<<"YES";
        cout<<endl;
        continue;
    }
    else{
    	temp.clear();
    	for(i=1;i<n;i++){
    		
    		for(j=i+1;j<n;j++){
    			
    			for(k=j+1;k<n;k++){
    				
    				first=s.substr(0,i);
    				second=s.substr(i,j-i);
    				third=s.substr(j,k-j);
    				fourth=s.substr(k,n-k);
    				
    				temp.insert(first);
    				temp.insert(second);
    				temp.insert(third);
    				temp.insert(fourth);
    				if(temp.size()==4){
    					cout<<"YES";
    					cout<<endl;
    				  flag=1;
    				  
    					break;}
    					
    				temp.clear();	
    				}
    				
    				if(temp.size()==4){
    				
    					break;}
    				
    				
    			}
    			
    			if(temp.size()==4){
    					
    					break;}
    			
    		}
    		
    		
    		
    	}
    	
    	
    	
    
    
    
    if(flag==0){
    	cout<<"NO";
    	cout<<endl;
    	
    }
	
}

    }
