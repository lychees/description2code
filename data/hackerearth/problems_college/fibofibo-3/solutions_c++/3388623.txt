#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int z[n],o[n];
    	z[0]=1;
    	o[0]=0;
    	z[1]=0;
    	o[1]=1;
    	for(int i=2;i<=n;i++)
         {
         	z[i]=z[i-1]+z[i-2];
         	o[i]=o[i-1]+o[i-2];
         }
    	cout<<z[n]<<" "<<o[n]<<endl;
    }
    return 0;
}
