#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i;
    long long sum=0;
    long long sum1=0;
    cin>>n;
    int a[n];
    for(i=0; i<n; i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    for(i=0; i<n/2; i++)
    {
        sum=sum+(a[n-i-1]-a[i])+(a[n-i-1]-a[i+1]);
        sum1=sum1+(a[n-i-1]-a[i])+(a[n-i-2]-a[i]);
    }
    cout<<max(sum,sum1)<<endl;
    return 0;
}
