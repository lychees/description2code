#include <iostream>
using namespace std;

int main()
{
	char hr[2],hg[2],hb[2];
	char hexa[] = "0123456789UVWXYZ";
	int tc,r,g,b;
	cin>>tc;
	int i =1;
	while(tc--){
		cin>>r>>g>>b;
		hr[1] = hexa[r%16];
		r/=16;
		hr[0] = hexa[r%16];

		hg[1] = hexa[g%16];
		g/=16;
		hg[0] = hexa[g%16];

		hb[1] = hexa[b%16];
		b/=16;
		hb[0] = hexa[b%16];

		cout<<"Case "<<i<<": "<<hr[0]<<hr[1]<<" "<<hg[0]<<hg[1]<<" "<<hb[0]<<hb[1]<<endl;
		i++;
	}
    return 0;
}
