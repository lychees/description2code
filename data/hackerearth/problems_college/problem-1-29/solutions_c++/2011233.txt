#include <iostream>
using namespace std;

int solve(int i , int j,int n,int a[105][105])
{
	if(i==n-1 && j==n-1 && a[i][j]==1)
		return 1;
	//else if(i==n-1 && j==n-1 && a[i][j]!=0)
	//	return 0;
	
	if(a[i][j]==0)
		return 0;
	
	if(i+1>n || j+1>n)
		return 0;
	
	return solve(i+1,j,n,a)||solve(i,j+1,n,a);
	
}
int main()
{
	int testcase;
	scanf("%d",&testcase);
	while(testcase--)
	{
		int n,a[105][105];
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				scanf("%d",&a[i][j]);
				
		if(solve(0,0,n,a)==0)
			cout<<"NOT POSSIBLE\n";
		else
			cout<<"POSSIBLE\n";
	}
	return 0;
}
