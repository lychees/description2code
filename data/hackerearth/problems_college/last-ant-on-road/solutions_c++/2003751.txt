int calc(vector <int> v,int n)
{

    int ansp=0,ansn=0;
    for(int i=0;i<n;i++)
    {
        if(v[i]==-1)
            ansn++;
            else
                ansp++;

    }
 //   cout<<ansn<<"   +"<<ansp<<endl;
    if(ansp>ansn)
    {
        return ansn+1;

    }
    else
        return ansn;

}
int main()
{
int tt;
cin>>tt;
while(tt--)
{
    int n;
    cin>>n;
    vector <int> v;
    for(int i=0;i<n;i++)
    {
        int a;
        cin>>a;
        v.push_back(a);
    }

    int ans=calc(v,n);
    cout<<ans<<endl;
}

return 0;
}
