#include <iostream>
using namespace std;

int determine(string s)
{
	int len = s.length();
	int countVowels = 0;
	int consonantCount = 0;
	int consecutive = 0;
	for(int i=0;i<len;i++)
	{
		if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i' || s[i] == 'o' || s[i] == 'u' || s[i] == 'y')
		{
			countVowels = countVowels + 1;
			consecutive = 0;
		}
		else
		{
			consonantCount = consonantCount + 1;
			consecutive = consecutive + 1;
		}
		
		if(consecutive == 3)
		{
			break;
		}
	}
	
	if(consecutive == 3 || consonantCount > countVowels)
	{
		return 0;
	}
	else
	{
		return 1;
	}
}


int main()
{
    int testcase;
    cin>>testcase;
    for(int i=0;i<testcase;i++)
    {
    	string text;
    	cin>>text;
    	
    	int det = determine(text);
    	if(det == 0)
    	{
    		cout<<"hard"<<endl;
    	}
    	else
    	{
    		cout<<"easy"<<endl;
    	}
    }
    return 0;
}
