#include <iostream>
using namespace std;

int gcd(int a,int b){
	if(a%b==0) return b;
	else{
		return gcd(b,a%b);
	}
}


int foo(int n){
	int ret=0;
	for(int i=1;i<=n;i++){
		if(gcd(n,i)==1) ret++;
	}
	return ret;
}

int SpecialSum(int m){
	int ret=0;
	for(int i=1;i<m;i++){
		if(m%i==0) ret=ret+foo(i);
	}
	return ret;
}
