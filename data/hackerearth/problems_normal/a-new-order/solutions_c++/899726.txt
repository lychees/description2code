#include <iostream>
using namespace std;


	int edge[26][26];
	int check[26];
	int out[26];
	
	string order[26];

int main()
{
	int n;
	cin>>n;
	
	string words[n];
		
	for(int i=0;i<n;i++)
	 cin>>words[i];
	 
    for(int i=0;i<n;i++)
	  {	
	  	for(int j=0;j<words[i].length();j++)
	    {
	  	  check[words[i][j]-'a']=1;
	    }
	  	
	  }
	  
	 
	 int j;
	 
	for(int i=0;i<(n-1);i++)
	 {
		j=0;
		
		while(j<words[i].length() && j<words[i+1].length())
		{
			if(words[i][j]!=words[i+1][j])
		    { 
		    	edge[words[i+1][j]-'a'][words[i][j]-'a']=1;
		    	break; 
		    } 
		    j++;
		}
		
	 }
		int left=0;
		
		for(int i=0;i<26;i++)
		{
			out[i]=0;
			
			for(int j=0;j<26;j++)
			{
				if(edge[i][j]==1)
				 out[i]++;
			}
			
			if(check[i]==1) left++;
		}
		
		int rank=0;
		
		while(left>0)
		{
			string temp="";
			
			for(int i=0;i<26;i++)
			{
				if(out[i]==0 && check[i]==1)
				 {temp+=char(i+'a'); left--;  out[i]=-1;}
				 
			}
			
			order[rank]=temp;
			rank++;
			
			
			for(int i=0;i<temp.length();i++)
			{
			    for(int j=0;j<26;j++)
			    {
			    	if(edge[j][temp[i]-'a']==1 && out[j]>0)
			    	 out[j]--;
			    	
			    }
			}
			
			
		}
		
		for(int i=0;i<26;i++)
		{
			if(order[i]=="") break;
			else
			cout<<order[i]<<"\n";
			
		}
		
		 return 0;
		
	}
