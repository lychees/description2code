#include <iostream>
using namespace std;

string str1,str2,str;

int check()
{
	int i,n=str.size();
	for(i=0;i<n;i++)
		if(str[i]!=str[n-i-1])
			return 0;
	return 1;		
}

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	cin>>str1>>str2;
    	str=str1+str2;
    	if(check())
    	{
    		cout<<"YES\n";
    		continue;
    	}
    	str.clear();
    	str=str2+str1;
    	if(check())
    	{
    		cout<<"YES\n";
    		continue;
    	}
    	cout<<"NO\n";
    }
    return 0;
}
