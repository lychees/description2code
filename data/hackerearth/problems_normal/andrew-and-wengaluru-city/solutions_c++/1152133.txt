#include<iostream>
using namespace std;
int main()
{
        int t,n,i,max_ind;
    long long y = 1000000007,c,maxi,prev_max;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long a[n];
        c = 0;
        max_ind = maxi = prev_max = 0;
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<n;i++)
            if(a[i]>maxi)
            {
                maxi = a[i];
                max_ind = i;
            }
        for(i=0;i<max_ind;i++)
        {
         if(a[i]>prev_max)
            prev_max = a[i];
         c = (c%y)+(prev_max-a[i])%y;
        }
        prev_max=0;
        for(i=n-1;i>max_ind;i--)
        {
            if(a[i]>prev_max)
                prev_max=a[i];
           c = (c%y)+(prev_max-a[i])%y;
        }
        cout<<c<<endl;
    }
 return 0;   
}
