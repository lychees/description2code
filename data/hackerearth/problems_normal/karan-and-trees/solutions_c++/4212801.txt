#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	long long a,b,temp;
    	cin>>a>>b;
    	if(b>a)
    	{
    		temp = a;
    		a = b;
    		b = temp;
    	}
    	while(a>b)
    	{
    		a = a/2;
    		if(b>a)
    		{
    			temp = a;
    			a = b;
    			b = temp;
    		}
    	}
    	cout<<a<<endl;
    }
}
