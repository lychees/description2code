#include <bits/stdc++.h>

using namespace std;

long long A[502][502];
long long dp[502][502];
int n,m;

bool valid(long long mid)
{
    dp[0][0] = mid;
    for ( int i = 1; i < m; i++ ) {
        dp[0][i] = -1;
        if ( dp[0][i-1] != -1 ) dp[0][i] = dp[0][i-1] + A[0][i];
        if ( dp[0][i] <= 0LL ) dp[0][i] = -1;
    }
    for ( int i = 1; i < n; i++ ) {
        dp[i][0] = -1;
        if ( dp[i-1][0] != -1 ) dp[i][0] = dp[i-1][0] + A[i][0];
        if ( dp[i][0] <= 0LL ) dp[i][0] = -1;
    }
    for ( int i = 1; i < n; i++ ) {
        for ( int j = 1; j < m; j++ ) {
            dp[i][j] = -1;
            if ( dp[i-1][j] != -1 ) dp[i][j] = dp[i-1][j] + A[i][j];
            if ( dp[i][j] <= 0LL ) dp[i][j] = -1;
            if ( dp[i][j-1] != -1 ) dp[i][j] = max(dp[i][j], dp[i][j-1] + A[i][j]);
            if ( dp[i][j] <= 0 ) dp[i][j] = -1;
        }
    }
    if ( dp[n-1][m-1] <= 0LL ) return false;
    return true;
}

int main()
{

    int t;
    long long l,r,mid,ans;
    scanf("%d", &t);
    for ( int tc = 1; tc <= t; tc++ ) {
        scanf("%d%d", &n, &m);
        for ( int i = 0; i < n; i++ ) {
            for ( int j = 0; j < m; j++ ) scanf("%lld", &A[i][j]);
        }
        l = 1, r = 250000002LL;
        while ( l <= r ) {
            mid = (l+r)>>1;
            if ( valid(mid) ) {
                ans = mid;
                r = mid-1;
            }
            else l = mid+1;
        }
        cout << ans << endl;
    }
    return 0;
}
