#include <bits/stdc++.h>
using namespace std;
#define gc() getchar_unlocked()
inline long long int read_int()
{
    char c = gc();
    while(c<'0' || c>'9')
    c = gc();
    long long int ret = 0;
    while(c>='0' && c<='9')
    {
    ret = (ret<<3) + (ret<<1) + c - 48;
    c = gc();
    }
    return ret;
}
int main()
{
	long long int t,l,b,h,min;
	double val;
	t = read_int();
	while(t--)
	{
		l = read_int();
		b = read_int();
		h = read_int();
		if(l<b && l<h)
		{
			min = l;
		}
		else if(b<h)
		{
			min = b;
		}
		else
		{
			min = h;
		}
		val = min/2.0;
		printf("%.1f\n",val);
	}
	return 0;
}
