#include<iostream>
using namespace std;
#define lld long long int
lld findMaxValue(lld weight[],lld values[],lld items[],int n,int capacity) {
   
   lld knapsack[capacity+1];
   knapsack[0] = 0;
   items[0] = -1;
   int i,j;
   for(j=1;j<=capacity;j++) {
      items[j] = items[j-1];
      
      lld max = knapsack[j-1];
      for(i=0;i<n;i++) {
         lld x = j-weight[i];
         if(x >= 0 && (knapsack[x] + values[i]) > max) {
            max = knapsack[x] + values[i];
            items[j] = i;
         }
         knapsack[j] = max;
      }
   }
   return knapsack[capacity];
}

		lld weight[l];
	   lld values[l];
		for(int i=0; i<l; ++i)
			scanf("%lld", &values[i]), weight[i] = (lld)i+1;
	   int capacity = l; // capacity of the knapsack  
	   int size = l;
	   
	   lld items[capacity+1];
	   lld val = findMaxValue(weight,values,items,size,capacity);
	   printf("%lld\n", val);
	}

}
