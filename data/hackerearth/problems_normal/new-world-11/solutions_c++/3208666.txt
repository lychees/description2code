#include <bits/stdc++.h>

using namespace std;

int x[1234567];

int main() {
  int tt;
  scanf("%d", &tt);
  while (tt--) {
    int n, k;
    scanf("%d %d", &n, &k);
    int low = 0;
    for (int i = 0; i < n; i++) {
      scanf("%d", x + i);
    }
    for (int i = 0; i < n - 1; i++) {
      low = max(low, x[i + 1] - x[i]);
    }
    int high = x[n - 1] - x[0];
    while (low < high) {
      int mid = (low + high) >> 1;
      int pr = 0;
      int jumps = 0;
      for (int i = 1; i < n; i++) {
        if (x[i] - x[pr] > mid) {
          pr = i - 1;
          jumps++;
        }
      }
      jumps++;
      if (jumps <= k) {
        high = mid;
      } else {
        low = mid + 1;
      }
    }
    printf("%d\n", low);
  }
  return 0;
}
