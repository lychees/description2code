#include <iostream>
#include<vector>
using namespace std;
#include<cstdio>
int main()
{
	int t,k,a,b,count,prime[10];
	long long result;
	scanf("%d %d",&k,&t);
	for(int i=0;i<k;i++)	scanf("%d",prime+i);
	while(t--){
		result=0;
		scanf("%d %d",&a,&b);
		int na,nb;
		for(int i=1;i<(1<<k);i++){
			na=a-1,nb=b,count=0;
			for(int j=0;j<k;j++){
				if(i&(1<<j)){
					count++;
					na/=prime[j];
					nb/=prime[j];
				}
			}
			if(count&1)	result+=(nb-na);
			else	result-=(nb-na);
		}
		printf("%lld\n",result);
	}
    return 0;
}
