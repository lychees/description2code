#include <bits/stdc++.h>
using namespace std;


int func (int a[], int n)
{
    int sum = 0;
    int i, j;
   
    for (i = 0; i < n; i++)
      sum += a[i];
     
    if (sum%2 != 0)  
       return 0;
   
    int p[sum/2+1][n+1];
     
    
    for (i = 0; i <= n; i++)
      p[0][i] = 1;
       
    for (i = 1; i <= sum/2; i++)
      p[i][0] = 0;     
      
     
     for (i = 1; i <= sum/2; i++)  
     {
       for (j = 1; j <= n; j++)  
       {
         p[i][j] = p[i][j-1];
         if (i >= a[j-1])
           p[i][j] = p[i][j] || p[i - a[j-1]][j-1];
       }        
     }    
      
    
      
     return p[sum/2][n];
}     

int main() {
	
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,a[200];
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		
		if(func(a,n)==1)
		printf("YES\n");
		
		else
		printf("NO\n");
		
		
		
		
	}
	return 0;
}
