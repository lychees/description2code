#include <bits/stdc++.h>
using namespace std;
typedef pair <int,int> p;
int main()
{
    string s;
    cin>>s;
    map < p , int > m;
    pair < int ,int > l;
 l=make_pair(0,0);
m[l]=1;
int x=0,y=0,c=0;
    for(int i=0;i<s.length();i++)
{
    if(s[i]=='U')
    {
        y=y-1;
        l=make_pair(x,y);
     if(m[l]==1)
     {
         c++;
     }
     else
        m[l]=1;
    }
    if(s[i]=='D')
    {
        y=y+1;
        l=make_pair(x,y);
     if(m[l]==1)
     {
         c++;
     }
     else
        m[l]=1;
    }
    if(s[i]=='L')
    {
        x=x-1;
        l=make_pair(x,y);
     if(m[l]==1)
     {
         c++;
     }
     else
        m[l]=1;
    }
    if(s[i]=='R')
    {
        x=x+1;
        l=make_pair(x,y);
     if(m[l]==1)
     {
         c++;
     }
     else
        m[l]=1;
    }

}
printf("%d\n",c);
    return 0;
}
