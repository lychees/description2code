#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d\n",&t);
	while (t --) {
		char s[1000];
		scanf("%[^\n]s",s);
		//gets(s);
		//getchar();
		int len = strlen(s);
		int ans = 0;
		for (int i = 0; i < len; i++) {
			ans += s[i];
		}
		ans = ans / len;
		if (ans & 1) {
			reverse(s,s+strlen(s));
			printf("%s\n",s);
		} else {
			printf("%s\n",s);
		}
	}
	
	return 0;
}
