m=raw_input()
for i in range(len(m)):
    if m[i]==' ':
        x,y=int(m[:i]),int(m[i+1:len(m)])
a,b=0,0
hcf=1
if x>y:
    a,b=x,y
else:
    a,b=y,x
while True:
    rem=a%b
    if rem==1:
        break
    if a%b==0:
        hcf=b
        break
    else:
        a=b
        b=rem
n1=x/hcf
n2=y/hcf
print str(n1)+" "+str(n2)
