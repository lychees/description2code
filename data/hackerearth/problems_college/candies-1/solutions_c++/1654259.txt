#include<iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll x,k,s=0;
        k=n;
        while(k--)
        {
            cin>>x;
            s+=x;
        }
        if(s%n==0)
        {
            cout<<"YES\n";
            cout<<s/n<<" "<<0;
        }
        else
        {
            cout<<"NO\n";
            cout<<s/n<<" "<<s%n;
        }
        cout<<'\n';
    }
    return 0;
}
