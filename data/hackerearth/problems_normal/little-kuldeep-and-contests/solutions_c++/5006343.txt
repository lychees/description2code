

    #include <bits/stdc++.h>
    using namespace std;
    int n;
    pair<int, int> a[128];
    int main()
    {
    	int i, j;
    	int h1, h2, m1, m2;
    	bool flag;
    	scanf("%d", &n);
    	for(i=1;i<=n;i++)
    	{
    		scanf("%d:%d-%d:%d", &h1, &m1, &h2, &m2);
    		a[i].first=h1*60+m1;
    		a[i].second=h2*60+m2;
    	}
    	flag=false;
    	for(i=1;i<=n && !flag;i++)
    	{
    		for(j=i+1;j<=n && !flag;j++)
    		{
    			if(a[i].second<=a[j].first || a[j].second<=a[i].first);
    			else flag=true;
    		}
    	}
    	if (flag==false)
    		printf("Who needs a moderator?\n");
    	else
    		printf("Will need a moderator!\n");
    	return 0;
    }
