#include <iostream>
using namespace std;

int main()
{
	int n;
   	cin>>n;
   	int b[n];
   	//int a[n];
   	int sum=0;
   	for(int i=0;i<n;i++)
   	cin>>b[i];
   	for(int i=0;i<n;i++)
   	{
   		
   		if(i==0)
   		{cout<<b[i]<<" ";
   			sum=b[i];
   		}
   		else
   		{
   		int ans=b[i];
   		ans=ans*(i+1);
   		ans-=sum;
   		sum+=ans;
   		cout<<ans<<" ";	
   		}
   	}
    return 0;
}
