#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll n;
int t;
int sod(ll s)
{
	if(s==0)
	return 0;
	return (s%10)+sod(s/10);
}

int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll x=1;
		int f=0;
		x=ll(sqrt(n))-82;
		while(x<=(ll(sqrt(n))+1))
		{
			if(x>=1)
			{
			if(x*(x+sod(x))==n)
			{
				f=1;
				break ;
			}
			}
			x++;
		}
		if(f)
		cout<<x<<endl;
		else
		cout<<"-1"<<endl;
	}
    return 0;
}
