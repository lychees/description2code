#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    long long i=0,n=0, j=0, k=0, l=0;
    
    cin>>n;
    
    vector<long long> str(n,0),cal(n,0);
    
    for(i=0;i<n;i++)
    {
    	cin>>str[i];
    }
    
    for(i=0;i<n;i++)
    {
    	cin>>cal[i];
    }
    
    sort(str.begin(),str.end());
    sort(cal.begin(),cal.end());
    
    for(i=0;i<n;i++)
    {
    	j+=cal[i]*str[i];
    }
    
    cout<<j<<endl;
}
