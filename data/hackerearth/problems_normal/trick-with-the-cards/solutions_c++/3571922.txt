#include<bits/stdc++.h>
using namespace std;
main()
{
     int t;
    cin>>t;

    while(t--)
    {
         long int n;

        cin>>n;
        if(n==0)
        {
            cout<<"0"<<"\n";
        }
        else if(n<=3 && n>0)
        {
            cout<<"1"<<"\n";
        }
        else if(n>3 && n%3==0)
        {
            n=n/3;
            cout<<n<<"\n";
        }
        else{

            cout<<n<<"\n";

        }
    }


    }
