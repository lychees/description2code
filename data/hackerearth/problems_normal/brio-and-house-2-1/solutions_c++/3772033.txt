int main(){
	int n,x1,y1,x2,y2,r;
	cin>>n;
	while(n--){
		cin>>x1>>y1>>x2>>y2>>r;
		int a=(y2-y1);
		int b=(x2-x1);
		int c=x2*y1-y2*x1;
		c=abs(c);
		double tmp=c*1.0;
		double den=sqrt(a*a+b*b);
		if(den!=0)
		tmp=tmp/den;
		if(tmp>r){
			printf("SAFE\n");
		}
		else if(tmp<r){
			printf("REPLANNING\n");
		}
		else{
			printf("JUST MISSED\n");
		}
	}
	return 0;
}
