int f(int a[], int n) {
	vi l(n, 1);
	l[0] = 1;

	int maxL = 1;
	F(i, 1, n) {
		F(j, 0, i) {
			if(a[i] > a[j]) {
				l[i] = max(l[i], 1 + l[j]);
				maxL = max(maxL, l[i]);
			}
		}
	}

	return maxL;
}

void b(int n) {
	stack<int > stk;
	while(n >= 2) {
		stk.push(n % 2);
		n /= 2;
	}

	stk.push(n);
	while(!stk.empty()) {cout << stk.top(); stk.pop();}
	cout << endl;
}

int main() {
	int n;
	sc(n);
	
	int a[n];
	F(i, 0, n) sc(a[i]);

	int l = f(a, n);
	b(l);
	return 0;
}
