#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,k=0,l=0,s=0,m=0,i,j;
    string a;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cin>>a;
        for(i=0;i<n;i++)
        {
            if(a[i]=='0')
                l++;
        }
        k=n-l;
        for(i=0;i<l;i++)
        {
            if(a[i]=='1')
            {
                s++;
            }
        }
        for(i=0;i<k;i++)
        {
            if(a[i]=='0')
            {
                m++;
            }
        }

        j=min(s,m);
        cout<<j<<endl;
        s=0;
        k=0;
        l=0;
        m=0;
    }
    return 0;
}
