#include <iostream>
#include <map>

using namespace std;

int main()
{
    int n,i=0;
    cin >> n;
    string name;
    map<string,int> mp;
    for (int i=0; i< n; i++) {
       	cin >> name;
    	if (mp.count(name) == 1) {
    		mp[name]++;
    	}
    	else {
    		mp.insert(make_pair(name,1));
    	}
    }
    
    for (map<string,int>::iterator i=mp.begin(); i != mp.end(); ++i) {
    	cout << i->first << " " << i->second << endl;
    }
    
    
    
    return 0;
}
