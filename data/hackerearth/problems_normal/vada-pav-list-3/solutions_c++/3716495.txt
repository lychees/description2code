#include <bits/stdc++.h>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    
    int n;
	string s;
	set<string> hash;
	
    cin >> n;
    
    for (int i = 0; i < n; i++) {
    	cin >> s;
    	hash.insert(s);
    }
    
    cout << hash.size() << endl;
    
    for (auto it = hash.begin(); it != hash.end(); it++)
    	cout << *it << '\n';
    
    return 0;
}
