#include <bits/stdc++.h>
using namespace std;
struct edge{
    int u,v,c;
};
bool compare(const edge &a,const edge &b){
    return a.c<b.c;
}
int sz[100001],arr[100001];
int root(int a){
    while(arr[a]!=a){
        a=arr[a];
    }
    return a;
}
void un(int a,int b){
    int x=root(a);
    int y=root(b);
    if(sz[x]>sz[y]){
        arr[y]=x;
        sz[x]+=sz[y];
    }
    else{
          arr[x]=y;
        sz[y]+=sz[x];
    }
}

           // cout<<i<<endl;
