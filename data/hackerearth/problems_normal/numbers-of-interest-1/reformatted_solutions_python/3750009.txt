t = int(input())
for t0 in xrange(t):
    X, Y, N = raw_input().strip().split(' ')
    X = int(X)
    Y = int(Y)
    N = int(N)
    li = []
    for i in xrange(X):
        li.append(Y)
    t = N-X
    for i in xrange(t):
        tt = i+X
        s = 0
        for j in range(i, tt):
            s += li[j]
        li.append(s)
    print li[N-1]
