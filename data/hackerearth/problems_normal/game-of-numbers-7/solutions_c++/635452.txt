#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

typedef long long ll;

int main() {
int T, X, Y;
ll i, j, sq;

scanf("%d", &T);
while (T--) {
scanf("%d %d", &X, &Y);
vector<bool> special(Y-X+1, true);
for (i = 2, sq = i*i; sq <= Y; i++, sq = i*i)
for (j = (sq - (X % sq)) % sq; j <= Y-X; j += sq)
special[(int)j] = false;
//count how many true is there in special vector
printf("%d\n", (count(special.begin(), special.end(), true)));
}

return 0;
}
