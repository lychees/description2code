#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n,i,j,a;
    	cin>>n;
    	string str;
    	vector<pair<long int,string> >v;
    	for(i=0;i<n;i++)
    	{
    		cin>>a;
    		cin>>str;
    		v.push_back(make_pair(a,str));
    	}
    	sort(v.begin(),v.end());
    	cin>>j;
    	string str1;
    	str1=v[j-1].second;
    	cout<<str1<<endl;
    }
}
