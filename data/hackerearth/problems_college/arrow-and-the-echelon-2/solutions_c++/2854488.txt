#include<iostream>
//#include<conio.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		long int n,i;
		long int m,r,flag=0;
		cin>>n;
 	   if(n==1||n==2)
		cout<<"-1"<<endl;
		if(n%3==0)
		{
			for(i=0;i<n;i++)
			{
				s=s+'5';
			}
			cout<<s<<endl;
		}
		else
		{
			m=n/3;
		while(m!=0)
		{
			--m;
			r=n-3*(m);
			if(r%5==0)
			{
				for(i=0;i<n-r;i++)
				{
					s=s+'5';
				}
				for(i=0;i<r;i++)
				{
					s=s+'3';
				}
				cout<<s<<endl;
				flag=1;
				if(flag==1)
				{
					break;
				}
			}
		   if(m==0)
		   {
				cout<<"-1"<<endl;
		   }
		}
	}
		
		
		
		
	}
	//getch();
}
