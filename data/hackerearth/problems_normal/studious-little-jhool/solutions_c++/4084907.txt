#include<iostream>
#include<algorithm>

using namespace std;

int main(){

    int t;
    cin >> t;
    while(t--) {
        long long n,a[200]={0},i;
        cin >> n; a[10] = a[12] = 1;
        for(i=10;i<=n;i++) {
            if(a[i] > 0) {
                if(a[i + 10] == 0) a[i + 10] = a[i] + 1;
                else a[i + 10] = min(a[i + 10],a[i] + 1);
                if(a[i + 12] == 0) a[i + 12] = a[i] + 1;
                else a[i + 12] = min(a[i + 12],a[i] + 1);
            }
        }
        if(a[n] > 0) cout << a[n] << "\n";
        else cout << "-1\n";
    }





}
