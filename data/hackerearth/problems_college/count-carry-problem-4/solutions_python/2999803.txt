t=input()
while t>0:
	t-=1
	n1,n2=map(int,raw_input().split())
	c=0
	while (n1%10) and (n2%10):
    		a= n1%10
    		b= n2%10
    		if (a+b>9):
        		c+=1
    		n1/=10
    		n2/=10
	if c==0:
	    print "No carry operation"
	elif c==1:
    		print "1 carry operation"
	else:
    		print "{0} carry operations".format(c)
