#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    char a[10][5]={'_','0',' ',' ',' ',
                   '.',',','?','!','1',
                   'a','b','c','2',' ',
                   'd','e','f','3',' ',
                   'g','h','i','4',' ',
                   'j','k','l','5',' ',
                   'm','n','o','6',' ',
                   'p','q','r','s','7',
                   't','u','v','8',' ',
                   'w','x','y','z','9',
    };
    while(t--)
     {
     	int x,y,z;
     	cin>>x>>y>>z;
     	int k=1;
     	char m[100];
     	cin.ignore();
     	gets(m);
     //	puts(m);
     	int i=0;
     	int pos,val,temp=0;
     	while(m[i]!='\0')
     	 {
     	 	 char q=m[i];
     	 	int l;
     	 	for(int j=0;j<10;j++)
     	 	 {
     	 	 	for(l=0;l<5;l++)
     	 	 	 {
     	 	 	 	if(q==a[j][l])
     	 	 	 	  {
     	 	 	 	  	pos=j;
     	 	 	 	  	val=l;
     	 	 	 	  }
     	 	 	 }
     	 	 	 
     	 	 }
     	 	// cout<<pos<<" "<<val;
     	 	 if(pos!=k)
     	 	  {
     	 	  	temp+=x;
     	 	  	k=pos;
     	 	  }
     	 	 temp+=((val+1)*y);
     	 	 if(temp<=z)
     	 	  cout<<m[i];
     	 	 else break;
     	 	 i++;
     	 }
     	 cout<<"\n";
     }
}
