#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	int t;
	ll n,s,a,b;
	cin>>t;
	while(t--)
	{
		cin>>n;
		s=sqrt(n);
		if(s*s==n)
			cout<<"YES\n";
		else
		{
			a=(ll)(s+1)*(s+1)-n;
			b=(ll)n-(s*s);
			if(a<b)
				cout<<"+"<<a<<"\n";
			else
				cout<<"-"<<b<<"\n";
		}
	}
	return 0;
}
