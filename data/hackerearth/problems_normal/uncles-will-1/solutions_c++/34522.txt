#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int a,b,c,i,arr[300],diff[300],temp,tlen,j,q,t,rem;

    cin>>t;
    for(q=0; q<t; q++)
    {
        cin>>a>>b>>c;
        for(i=0; i<c; i++)
        cin>>arr[i];
        if(a>=c)
        {
            cout<<c<<endl;
            continue;
        }
    sort(arr,arr+c);
    for(i=1; i<c; i++)
    {
        diff[i-1]=arr[i]-arr[i-1];
    }
    temp=c-1;
    sort(diff, diff+temp);
    tlen=a;
    rem=c-a;
    for(i=0; i<rem; i++)
    {
        tlen=tlen+diff[i];
    }
    cout<<tlen<<endl;
    }
    return 0;
}
