#include <iostream>
#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
string s1,s2;
ll i,j,n,a[26],b[26],co;
int main()
{
    cin>>n;
    while(n--)
    {
    	for(ll i=0;i<26;i++)
    	a[i]=0;
    	
    	for(ll i=0;i<26;i++)
    	b[i]=0;
    	
    	cin>>s1;
    	for(ll i=0;i<s1.length();i++)
    	a[s1[i]-'a']++;
    	
    	cin>>s2;
    	for(ll i=0;i<s2.length();i++)
    	b[s2[i]-'a']++;
    	
    	
    	for(ll i=0;i<26;i++)
    	if(a[i]>0 && b[i]>0)
    	{
    		co=1;
    		break;
    	}
    	
    	if(co==1)
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    	co=0;
    }
    
    
    return 0;
}
