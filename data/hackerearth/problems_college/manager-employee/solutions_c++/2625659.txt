#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;int i;
    	scanf("%lld",&n);
    	long long a[n];
    	for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    	sort(a,a+n);
    	long long s1=INT_MIN,s2=INT_MAX;
    	for(i=0;i<n/2;i++)
    	{
    		long long s=a[i]+a[n-i-1];
    		if(s>s1)
    		s1=s;
    		if(s<s2)
    		s2=s;
    	}
    	s1=abs(s1-s2);
    	printf("%lld\n",s1);
    }
    return 0;
}
