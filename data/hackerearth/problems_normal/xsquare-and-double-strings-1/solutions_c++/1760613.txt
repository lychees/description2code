#include<iostream>
#include<stdio.h>
#include<string.h>
#include<cstring>
using namespace std ;

int main()
    { int t ;
    int count[27] ;
      cin>>t;
      while(t--)
       { memset(count , 0 , sizeof(count));
         string s ;
         int flag = 0 ;
         cin>>s ;
         for(int i = 0;i<s.size();i++)
            count[s[i]-'a']++ ;
         for(int i =0;i<27;i++)
           {if(count[i] >=2)
             { printf("Yes\n");
               flag = 2 ;
                break ;
             }
           }
         if(flag == 0)
           printf("No\n");
        }
       
       return 0 ;
    }
