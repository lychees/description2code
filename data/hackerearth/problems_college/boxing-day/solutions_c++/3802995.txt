const int mod = 1e9 + 7;
 
int ADD(int a, int b, int m = mod) {
    int s = a;
    s += b;
    if( s >= m )
      s -= m;
    return s;
}
 
int MUL(int a, int b, int m = mod) {
    return (1LL * a * b % m);
}
 
int power(int a, int b, int m = mod) {
    int res = 1;
    while( b ) {
        if( b & 1 ) {
            res = 1LL * res * a % m;
        }
        a = 1LL * a * a % m;
        b /= 2;
    }
    return res;
}
 
ll nC2(ll x) {
    return ( x * ( x - 1 ) / 2 );
}
