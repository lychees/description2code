int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,m;
	    scanf("%lld%lld",&n,&m);
	    vector<ll> v[n+1];
	    for(i=0;i<m;i++)
	    {
		    ll a,b;
		    scanf("%lld%lld",&a,&b);
		    v[a].push_back(b);
	    	v[b].push_back(a);
	    }
	    ll distance[n+1];
	    memset(distance,-1,sizeof distance);
		queue<ll> q;
		q.push(1);
		distance[1]=0;
		while(!q.empty())
		{
			ll curr=q.front();
			q.pop();
			for(i=0;i<v[curr].size();i++)
			{
				ll temp=v[curr][i];
				if(distance[temp]==-1)
				{
					distance[temp]=distance[curr]+1;
					q.push(temp);
				}
			}
		}
		printf("%lld\n",distance[n]);
		}
	return 0;
}
