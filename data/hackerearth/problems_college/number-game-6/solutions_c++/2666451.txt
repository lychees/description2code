#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

vector<ll>res;

int main() {
  int st=1;
  while(true) {
    ll ml=1ll*st*(st+1)*(st+1);
    if(st>1) {
      ml+=res[st-2];
    }
    if(ml<=1e17) {
      res.push_back(ml);
      st++;
    } else {
      break;
    }
  }
  int tt;
  cin>>tt;
  while(tt--) {
    ll n;
    cin>>n;
    int idx=upper_bound(res.begin(),res.end(),n)-res.begin()-1;
    if(idx==-1) {
      puts("0\012Draw");
    } else {
      printf("%d\012",idx+1);
      if((idx+1)&1) {
        puts("Dhruv");
      } else {
        puts("Ketan");
      }
    }
  }
  return 0;
}
