#include <bits/stdc++.h>
using namespace std;
bool srt_val(pair<string, int> a, pair<string, int> b){
	return (a.second >= b.second) ? true : false;//(a.first < b.first);
}
int main()
{
    map<string, int> m;
    int t;
    string s;
    cin >> t;
    while(t--)
	{
		cin >> s;
		m[s]++;
	}	
	vector<pair<string, int>> mc(m.begin(), m.end());
	sort(mc.begin(), mc.end(), srt_val);
	for(vector<pair<string, int>> :: iterator it = mc.begin(); it != mc.end(); it++){
		cout << it->first << " " << it->second << endl;
	}
	return 0;
}
