const int MAXN = 111111;
ll arr[MAXN];

int up_bound(int lo,int hi,ll val) {
	while (lo <= hi) {
		int mid = (lo + hi) / 2;
		if(arr[mid] > val) hi = mid - 1;
		else lo = mid + 1;
	}
	return hi;
}

int main() {
	ios_base::sync_with_stdio(false);

    int t; cin >> t;
	while (t --) {
		int n,k,p; cin >> n >> k >> p;
		for (int i = 0; i < n; i ++) cin >> arr[i];

		sort(arr,arr + n);

		bool flag = false;
		for (int i = 0; i < n; i ++) {
			int up = up_bound(0,n - 1,arr[i] + p + p);
			int num = up - i + 1;
			if(num >= k) {
				flag = true;
				break;
			}
		}

		if(flag) puts("NO");
		else puts("YES");
	}
	
	return 0;
}
