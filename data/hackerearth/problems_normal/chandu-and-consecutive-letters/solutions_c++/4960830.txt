#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	string res="";
    	int i=0;
    	while(i<str.size())
    	{
    		char ch=str.at(i++);
    		while(i<str.size() && str[i]==ch)
    			i++;
    		res+=ch;
    	}
    	cout<<res<<endl;
    }
    return 0;
    
}
