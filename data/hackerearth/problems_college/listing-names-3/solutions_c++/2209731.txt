#include<bits/stdc++.h>
using namespace std;
char str[100][100];
int n;
void swap(char *a,char *b)
{
	char temp[1000];
	strcpy(temp,a);
	strcpy(a,b);
	strcpy(b,temp);
}
void sort()
{
	for(int i=0;i<n-1;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(strcmp(str[i],str[j])>0)
			{
				swap(str[i],str[j]);
			}
		}
	}
}
int main()
{
	int count=1,j;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>str[i];
	sort();
	for(int i=0;i<n;count=1)
	{
		for(j=i+1;j<n;j++)
		{
			if(strcmp(str[i],str[j])==0)
			{
				count++;
				//cout<<"sa"<<endl;
			}
			else
			{
				
				break;
			}
		}
		cout<<str[i]<<" "<<count<<endl;
		i=j;
	}
	
}
