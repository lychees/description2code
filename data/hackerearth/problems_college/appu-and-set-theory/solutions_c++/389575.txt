#include<iostream>
using namespace std;
typedef long long LL;

LL fi(LL b,LL p,LL m)
{
  LL ret=1;
  while(p)
    {
        if(p&1)
            ret=(ret*b)%m;
        b=(b*b)%m;
        p=p/2;
     
    }
   return ret;        
}

int main()
{
LL n;
cin>>n;
LL ans=0;
LL d=1000000000+7;

while(n--)
{
LL x;
cin>>x;          
LL tem=fi(2,x,d);
ans=((ans*(1+tem))%d+tem)%d ;          
}    
cout<<ans<<endl;
return 0;
}
