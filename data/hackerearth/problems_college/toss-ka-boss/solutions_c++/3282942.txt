#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char a[1000000];
		scanf("%s",a);
		long long int size = strlen(a);
		long long int sum = 0;
		long long int temp = 0,i;
		for(i = 0; i < size; i++)
		{
			if(a[i]=='H')
			{
				temp++;
				sum = sum + temp;
			}
			if(a[i]=='T')
			{
				temp = 0;
			}
		}
		cout<<sum<<endl;
	}
	return 0;
}
