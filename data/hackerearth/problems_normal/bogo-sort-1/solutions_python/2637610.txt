numbers = []
for _ in range(int(input())):
    numbers.append(int(raw_input()))
plus = []
minus = []
zero = []
for number in numbers:
    if number > 0:
        plus.append(number)
    elif number < 0:
        minus.append(number)
    else:
        zero.append(number)

print('\n'.join(str(i) for i in sorted(minus,reverse=False)))
print('\n'.join(str(i) for i in sorted(plus,reverse=False)))
