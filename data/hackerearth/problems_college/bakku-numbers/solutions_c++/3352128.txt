#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main() 
{
    ll p1,p2;
    cin>>p1>>p2;
    ll t,l,r;
    cin>>t;
    while(t--) {
    	cin>>l>>r;
    	ll x,y,z;
    	x = r/p1 - (l-1)/p1;
    	y = r/p2 - (l-1)/p2;
    	z = r/(p1*p2) - (l-1)/(p1*p2);
    	x += y-z;
    	double ans = x;
    	ans /= (double)(r-l+1);
    	printf("%.6lf\n",ans);
    }
	return 0;
}
