#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    int n,i;
    int a[1000000] = {0};
    a[1] = 1;
    for( i = 2; i <= 1000;i++){
        for(int j = 2;pow(i,j) <= 1000000;j++){
            int pos = pow(i,j);
            a[pos] = 1;
        }
    }
    cin>>t;
    while(t--){
        cin>>n;
        for( i = 1;i <= n/2 ; i++){
            if(a[i] && a[n-i]) {
                break;
            }
        }
        if (i<=n/2) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        }
return 0;       
}
