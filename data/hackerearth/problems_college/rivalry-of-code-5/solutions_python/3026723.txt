for _ in range(input()):
	disrupted = raw_input()
	actual = raw_input()
	dlen = len(disrupted)
	alen = len(actual)
	mincount = []
	if alen >= dlen:
		for index in range(alen):
			if index + dlen <= alen:
				count = 0
				incr = 0
				for pos in range(index, index + dlen):
					if disrupted[incr] != actual[pos]:
						count += 1
					incr += 1
				mincount.append(count)
	else:
		for index in range(dlen):
			if index + alen <= dlen:
				count = 0
				incr = 0
				for pos in range(index, index + alen):
					if disrupted[pos] != actual[incr]:
						count += 1
					incr += 1
				mincount.append(count)
	print min(mincount)
