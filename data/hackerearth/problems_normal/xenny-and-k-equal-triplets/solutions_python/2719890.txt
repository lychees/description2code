import sys
from fractions import gcd

T = eval(raw_input())
while T > 0:
	T -= 1
	N, K = map(lambda x:int(x), sys.stdin.readline().split())
	count = len(filter(lambda x: int(x) == K, sys.stdin.readline().split()))
	temp1 = N * (N-1) * (N-2) / 6
	temp2 = count * (count-1) * (count-2) / 6
	commondiv = gcd(temp1, temp2)
	numerator = temp2 / commondiv
	denominator = temp1 / commondiv
	print "{}/{}".format(numerator, denominator)
