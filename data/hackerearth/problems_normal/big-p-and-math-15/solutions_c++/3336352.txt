#include<bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;

int main()
{
    string a,b;
    cin>>a>>b;
    string a1 =a; string a2=a;
    string b1=b; string b2=b;
    for(int i=0, sz=a1.size();i<sz;i++)
    {
        if(a1[i]=='6')
            a1[i]='5';
    }
    for(int i=0, sz=b1.size();i<sz;i++)
    {
        if(b1[i]=='6')
            b1[i]='5';
    }
    for(int i=0, sz=a2.size();i<sz;i++)
    {
        if(a2[i]=='5')
            a2[i]='6';
    }
    for(int i=0, sz=b2.size();i<sz;i++)
    {
        if(b2[i]=='5')
            b2[i]='6';
    }
    cout<<atoi(a1.c_str())+atoi(b1.c_str())<<" "<<atoi(a2.c_str())+atoi(b2.c_str())<<'\n';
    return 0;
}
