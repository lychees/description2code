    return mk(a.f || a.s, a.s - max(a.f, a.s));
  if (f.find(a) != f.end())
    return f[a];
  ll ans = 0, p = 1;
  while (p <= a.s / 10)
    p *= 10;
  pr r = dfs(mk(max(a.f, a.s / p), a.s % p));
  pr b = dfs(mk(a.f, a.s / p * p + r.s));
  return f[a] = mk(r.f + b.f, b.s);
}
