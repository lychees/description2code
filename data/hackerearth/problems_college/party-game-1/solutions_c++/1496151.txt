#include <iostream>
#include <algorithm>
typedef long long int LL; using namespace std;

int main() { LL t,n,x,y,a,b; cin>>t; while(t--) {

    cin>>a>>b>>n;

    LL p[n+1];

    for(x=0;x<n;x++)
    cin>>p[x];

    LL sa=0,f=0;

    sort(p,p+n);
    x=0;
    while(x<n)
    {
        sa+=p[x];
        if(sa>a || sa>b)
        {
            f=1;
            break;
        }
        x++;
    }

    if(f==1)
    {

        if(sa>a && sa>b)
        cout<<"Tie\n";

    }else

}
