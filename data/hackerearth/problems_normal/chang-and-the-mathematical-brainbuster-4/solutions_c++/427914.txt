#include <iostream>
using namespace std;

int main()
{
	int N,i,j,*arr,mincount,max3timecount,tc=10000;
	cin >> N;
	arr=(int *)malloc(sizeof(int)*N);
	for(i=0;i<N;i++)
		cin >> arr[i];
	for(i=0;i<N;i++)
	{
		mincount=0;
		max3timecount=0;
		for(j=0;j<N;j++)
		{
			if(arr[i]>arr[j])
				mincount++;	
			if((arr[i]*3)<arr[j])
				max3timecount++;		
		}		
		if(tc>(max3timecount+mincount))
			tc=max3timecount+mincount;
	}
		cout << tc;
    
    return 0;
}
