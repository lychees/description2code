#include <iostream>
#include<cstdio>
#include<set>
using namespace std;

int main()
{
	set<string> friendList;
	int t;
	scanf("%d", &t);
	while(t--) {
		int n;
		scanf("%d", &n);
		string name;
		for(int i = 0; i < n; i++) {
			cin >> name;
			friendList.insert(name);
		}
		set<string>::iterator it;
		for(it = friendList.begin(); it != friendList.end(); it++) {
			cout << *it << endl;
		}
		friendList.clear();
	}
    return 0;
}
