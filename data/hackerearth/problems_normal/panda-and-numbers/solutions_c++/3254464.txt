#include<bits/stdc++.h>
#define ll long long int
#define l int
#define tr(i,n) for(i=0;i<n;i++)
#define tr1(i,n) for(i=1;i<=n;i++)
#define sc(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define p(n) printf("%d",n)
#define pl(n) printf("%lld",n)
using namespace std;
l a[1000005]={0};
l b[1000005];
l solve(l n)
{
	if(a[n]==1)
	return 1;
	if(n<=0)
	return 0;
	if(b[n]>0)
	return b[n];
	l k=n,z=0;
	while(n>0)
	{
		l m=n%10;
		n=n/10;
		if(m==0)
		continue;
		if(k-m*m<0)
		continue;
		 z=b[k-m*m]=solve(k-m*m);
		 if(z)
		 break;
		 
	}
	return z;
}
	
main()
{
	a[1]=1;a[4]=1;a[27]=1;a[256]=1;a[3125]=1;a[46656]=1;a[823543]=1;
	l t,n;
	sc(t);
	memset(b,0,sizeof(b));
	while(t--)
	{
		sc(n);
		if(solve(n)!=0)
		printf("Yes\n");
		else
		printf("No\n");
	}
}
