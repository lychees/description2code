#include <iostream>
#include <algorithm>
using namespace std;

long long fun(long long arr[],long long n)
{
	long long i,j,sum,s;
	s=0;
	sum=0;
	for(i=0;i<n;i++)
		sum+=arr[i];
		for(i=0;i<n;i++)
		{
			sum-=arr[i];
			if(sum==s)
    			return i;
    		s+=arr[i];
		}
			return -1;
}
int main()
{
    long long t,n,i;
    long long arr[1000001];
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>arr[i];
    	if(fun(arr,n)==-1)
  			cout<<"NO\n";
  		else
  			cout<<"YES\n";
    	
    }

}
