def sortThem(p, q, r):
    data = sorted([p, q, r])
    return data[0], data[1], data[2]


T = int(raw_input())

for i in range(T):
    p, q, r = raw_input().split()
    p = int(p)
    q = int(q)
    r = int(r)
    p, q, r = sortThem(p, q, r)
    a, b, c = p, q, r
    a = c - 1
    countOne = a - b
    a, b, c = p, q, r
    c = a + 1
    countTwo = b - c

    if countOne >= countTwo:
        print countOne
    else:
        print countTwo
