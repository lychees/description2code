#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
#include<memory>
#include<iterator>
#include<math.h>

using namespace std;

int main()
{
    int t,n;
    cin>>t;while(t--)
    {
        cin>>n;
        vector<int> a(n);
        for(int i=0;i<n;i++)
            cin>>a[i];
        sort(a.begin(),a.end());
        for(int i=0;i<n;i++)
        {
            cout<<a[i]<<" ";
        }
        cout<<endl;
    }
return 0;
}
