#include<iostream>
#include<map>
using namespace std;
int main(){
    int t,n;
    cin>>t;
    map<int ,int > mp;
    while(t--){
            cin>>n;
            int temp;
            for(int i=0;i<n;i++){
                cin>>temp;
                mp[temp]++;
            }
            map<int ,int >::iterator itr=mp.begin();
            int max=-1;
            while(itr!=mp.end()){
                    if(max<itr->second){
                        max=itr->second;
                    }
                itr++;
            }
            map<int ,int >::reverse_iterator ritr =mp.rbegin();
            while(ritr!=mp.rend()){
                    if(max==ritr->second){
                        cout<<ritr->first<<" ";
                    }
                ritr++;
            }
            cout<<endl;
            mp.clear();
    }
return 0;
}
