#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,minm=10000000,maxm=-1;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	for(int i=0;i<n;i++)
    	{
    		maxm=max(maxm,a[i]+a[n-1-i]);
    		minm=min(minm,a[i]+a[n-1-i]);
    	}
    		cout<<maxm-minm<<"\n";
	    	if(maxm-minm==k)
	    		cout<<"Lucky chap!\n";
	    	else
	    	{
	    		if(maxm-minm<k)
	    			cout<<"Chick magnet Jhool!\n";
	    		else
	    			cout<<"No more girlfriends!\n";
	    	}
    }
    return 0;
}
