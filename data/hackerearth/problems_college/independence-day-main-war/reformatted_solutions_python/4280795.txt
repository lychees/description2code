t = int(raw_input())
for ii in range(t):
    lst1 = list()
    lst2 = list()
    n = int(raw_input())
    for i in range(n):
        line = raw_input().split()
        lst1.append(float(line[0]))
        lst2.append(float(line[1]))
    a = sum(lst1)/n
    b = sum(lst2)/n
    print("(%.2f" % a), (", %.2f)" % b)
