from collections import Counter
t = int(raw_input())
while t:
    a = list(raw_input())
    a = [x for x in a if x != " "]
    b = list(raw_input())
    b = [x for x in b if x != " "]
    c1 = Counter(a)
    c2 = Counter(b)
    diff1 = c1-c2
    diff2 = c2-c1
    if (len(list(diff1)) > 0 and len(list(diff2)) == 0):
        print "You win some."
    elif (len(list(diff1)) == 0 and len(list(diff2)) > 0):
        print "You lose some."
    else:
        print "You draw some."
    t = t-1
