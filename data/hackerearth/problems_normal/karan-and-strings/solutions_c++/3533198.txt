#include <bits/stdc++.h>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
    int t; cin>>t;
    while(t--) {
    	string s, c = "";
    	cin>>s;
    	int i, j, n = s.length();
    	c += s[0];
    	for(i = 1; i < n; i++) {
    		if(s[i] == s[i-1]) continue;
    		else c+= s[i];
    	}
    	cout<<c<<endl;
    }
    return 0;
}
