#include <iostream>
#define ll long long
using namespace std;

int main()
{
   ll t,q,m,n,i,j;
   cin>>t;
   while(t--)
   {
   	ll res = 0,prod,sum;
   	cin>>q;
   	for(i=1;i<=q;i++)
   	{
   		m = i;
   		for(j=i+1;j<=q;j++)
   		{
   			n = j;
   			prod = m*n;
   			sum = m+n;
   			if((prod%sum)==0)
   			{
   				res++;
   			}
   		}
   	}
   	
   	cout<<res<<endl;
   	
   }
}
