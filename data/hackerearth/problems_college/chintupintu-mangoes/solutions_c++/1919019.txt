#include<bits/stdc++.h>
using namespace std;
#define ll long long
void sieve()
{
	int i;	
}
int main()
{
    int t,x,h,d,c,i,j,z;
    scanf("%d",&t);
    while(t--) {
    	scanf("%d%d%d",&x,&h,&d);
        c = 0;
        if(d == 1) {
        	c = 1;
        	goto go;
        }
        z = sqrt(h);
        for(i = 1; i <= z; i++) {
        	if(h%i == 0) {
        		j = h/i;
        		if(i > 1 && j >= d) {
        			c++;
        			break;
        		}
        		else if(j > 1 && i >= d) {
        			c++;
        			break;
        		}
        	}
        }
        go:
        if(c) {
        	puts(x%2?"Chintu":"Pintu");
        }
        else {
        	puts("Pintu");
        }
    }
}
