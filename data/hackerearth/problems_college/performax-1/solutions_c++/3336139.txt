#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i;
    cin>>n;
    int arr[n],diff = 0,count = 0;
    int r,h = 0;
    cin>>r;
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    for(i=1;i<n;i++)
    {
        if(arr[i]<arr[i-1])
        {
            diff = arr[i-1]-arr[i];
            h = diff/r;
            h++;
            count+=h;
            arr[i] = arr[i] + (h*r);
        }
        else if(arr[i]==arr[i-1])
        {
            count++;
            arr[i] = arr[i]+r;
        }
    }
    cout<<count<<endl;
    return 0;
}
