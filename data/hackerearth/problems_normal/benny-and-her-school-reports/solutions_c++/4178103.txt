#include <iostream>
using namespace std;

int main()
{
    long long t,n,m,x;
    cin >> t;
    while(t--){
    	cin >> n >> m >> x;
    	long long sum = 0,grade;
    	for(long long i = 0; i < n; i++){
    		cin >> grade;
    		sum += grade;
    	}
    	grade = x*(n+1) - sum;
    	if(grade > m){
    		cout << "Impossible" << endl;
    		continue;
    	}
    	if(grade <= 0){
    		cout << 1 << endl;
    		continue;
    	}
    	cout << grade << endl;
    }
    return 0;
}
