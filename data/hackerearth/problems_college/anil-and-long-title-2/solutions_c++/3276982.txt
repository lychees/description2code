#include <iostream>
using namespace std;

int main()
{
	string s;
    cin >> s;
    unsigned int i;
    for( i=0;i<s.size();i++){
     if(s[i]>='a' && s[i]<='z'){
     	s[i]=s[i]+('A'-'a');
     	break;
     }
    }
    i++; 
    for(;i<s.size();i++){
    	if(s[i]=='_' && i+1<s.size() && s[i+1]>='a' && s[i+1]<='z')
    	 s[i+1]=s[i+1]+('A'-'a');
    }
    cout << s;
    return 0;
}
