#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
#include<limits.h>
int main()
{
	long long int i,j,n,l,t,s,k;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		long long int num[n+9],sum[n+9];
		for(i=0;i<n;i++)
		scanf("%lld",&num[i]);
		sort(num,num+n);
		s=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&sum[i]);
		}
		sort(sum,sum+n);
		for(i=0;i<n;i++)
		s+=num[i]*sum[i];
		printf("%lld\n",s);
	}
	return 0;
}
