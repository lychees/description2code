#include<stdio.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
unsigned long long int countSetBitsUtil(unsigned long long int x)
{
    if (x <= 0)
        return 0;
    return (x %2 == 0? 0: 1) + countSetBitsUtil (x/2);
}
int main(){
	long long int t;
	cin>>t;
	while(t--){
	unsigned long long int a,b,c;
		cin>>a>>b;
		c=a^b;
		
	
	if(countSetBitsUtil(c)%2 == 0){
		cout<<"YES"<<"\n";
	}
	else{
		cout<<"NO"<<"\n";
	}
}
return 0;
}
