#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,i,j,ans;
	char s[100006];
	scanf("%d",&t);
	while (t --) {
		scanf("%s",s);
		ans = 0;
		int len = strlen(s);
		for (i = 0; i < len; i++) {
			ans += s[i]-48;
		}
		int p = ans;
		if (p / 10 == 0) {
			printf("%d\n",p);
			continue;
		}
		while (1) {
			ans = 0;
			while (p != 0) {
				ans = ans+p%10;
				p/=10;
			}
			if (ans /10 == 0) {
				break;
			} else {
				p = ans;
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
