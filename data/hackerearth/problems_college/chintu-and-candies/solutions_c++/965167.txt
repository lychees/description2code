#include <iostream>
using namespace std;

int totient(int n)
{
    int i, ans = n;
    
    for (i=2;i*i<=n;i++)
    {
        if (n%i==0)
            ans-=ans/i;
            
        while(n%i==0)
            n/=i;
    }    
    
    if (n!=1)
        ans-=ans/n;
    return ans;    
}        

int main()
{
	ios_base::sync_with_stdio(false);
    int t,n,x,k,i;
    cin>>t;
    bool flag;
    while(t--)
    {
    	cin>>n>>x>>k;
    	flag = false;
    	for(i=n;i>x;i--)
    	{
    		if(totient(i)==k)
    		{
    			flag = true;
    			break;
    		}
    	}
    	
    	if(flag)
    	{
    		cout<<i<<"\n";
    	}
    	else
    	{
    		cout<<-1<<"\n";
    	}
    	
    }
    return 0;
}
