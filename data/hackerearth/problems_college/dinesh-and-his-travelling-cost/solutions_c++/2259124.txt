#define ff first

typedef pair <int,int> pii;

const int N = 555;

vector <pii> g[N];
int dis[N];

void dijkstra (int so) {
  set <pii> st;
  st.insert (mp (0,so));
  dis [so] = 0;
  while (!st.empty()) {
    set <pii>::iterator it = st.begin();
    st.erase (it);
    pii pp = *it;
    int u = pp.ss;
    int udis = pp.ff;
    for (pii vv: g [u]) {
      int v = vv.ff;
      int uv = vv.ss;
      if (dis[v] > udis + uv) {
        st.erase (mp (dis[v], v)); 
        dis[v] = udis + uv;
        st.insert (mp (dis[v], v));
      }
    }
  }
}

    cin >> u >> v >> w;
    g [u].pb ({v, w});
    g [v].pb ({u, w});
  }
  int so, Quora;
  cin >> so >> Quora;
  for (int i = 0; i <= 500; ++i) {
    dis[i] = (1 << 30);
  }
  dijkstra (so);

    }
  }
}
