#include<bits/stdc++.h>
using namespace std;
stack<int> ss;
int main() {
int tc,num1,num2;
cin>>tc;
while(tc--) {
cin>>num1>>num2;
int n1 = num1;
int n2 = num2;
while(n1) {
ss.push(n1%10);
n1 /= 10;
}
int k = 0;
int r1 = 0,r2=0;
while(!ss.empty()) {
r1 += (pow(10,k++) * ss.top());
ss.pop();
}
//cout<<r1<<endl;
while(n2) {
ss.push(n2%10);
n2 /= 10;
}
k = 0;
while(!ss.empty()) {
r2 += (pow(10,k++) * ss.top());
ss.pop();
}
//cout<<r2<<endl;
int ll = r1+r2;
while(ll) {
ss.push(ll%10);
ll /= 10;
}
k = 0;
int lll = 0;
while(!ss.empty()) {
lll += (pow(10,k++) * ss.top());
ss.pop();
}
cout<<lll<<endl;
 }
return 0;
}
