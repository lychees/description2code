#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
  int n;
  cin>>n;int i;
  int arr[n];
  ll dp[n];
  for(i=0;i<n;i++)
  {
    cin>>arr[i];
  }

  dp[0]=1;
  for(i=1;i<n;i++)
  {
    if (arr[i]>arr[i-1])
    dp[i]=dp[i-1]+1;
    else
    dp[i]=1;
  }

  for(i=n-2;i>=0;i--)
  {
    if (arr[i]>arr[i+1] && dp[i]<=dp[i+1])
    {
      dp[i]=dp[i+1]+1;
    }
  }
  ll result=0;
  for(i=0;i<n;i++)
  {
    result+=dp[i];
  }
  cout<<result<<"\n";

return 0;
}//main
