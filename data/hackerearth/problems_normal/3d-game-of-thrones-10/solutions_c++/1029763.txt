using namespace std;


long ntest, test, n, cnt, z, x;
long a[1000];

int main()
{
    cin >> ntest;
    for (test = 1; test <= ntest; test++)
    {
        cin >> n;
        n = n * 3;
        cnt = 0;
        z = 0;
        for (int i = 1; i <= n; i++) 
        {
            cin >> x;
            if (x == 1) cnt++;
            z = z ^ x;
        }   
        
        if (cnt == n) 
        {
           if (n % 2 == 0) cout << "Lannister" << endl;
           else cout << "Stark" << endl;
        }
        else 
        {    
             if (z > 0) cout << "Stark" << endl;
             else cout << "Lannister" << endl;
        }     
    }

    return 0;
}
