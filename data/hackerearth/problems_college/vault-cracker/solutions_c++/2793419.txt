#include <bits/stdc++.h>
using namespace std;
 int ans[5001][5001]={0};
 
 string s1,s2;
 int l1,l2;
 
 int check(int in1,int in2){
     
     if(in1==l1 &&in2==l2)return 1;
     if(in2==l2)return 0;
     if(in1==l1)return 1;
     if(ans[in1][in2]!=-1)return ans[in1][in2];
     int t1=check(in1,in2+1);
     int t2=0;
     if(s1[in1]==s2[in2])t2=check(in1+1,in2+1);
     int an=t1+t2;
     ans[in1][in2]=an;
     return an;
     
     
 }
 
int main()
{  cin>>s1>>s2;
l1=s1.size();
l2=s2.size();

for(int i=0;i<l1;i++)
for(int j=0;j<l2;j++)ans[i][j]=-1;
cout<<check(0,0);
    
    return 0;
}
