#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	for(int i=0;i<t;i++)
   	{
   		string a,b;
   		cin>>a>>b;
   		int out=0;
   		for(int j=0;j<=a.size();j++)
   		{
   			string cur = a.substr(0,j) + b + a.substr(j);
   			//cout<<"checking: "<<cur;
   			string rev = cur;
   			std::reverse(rev.begin(), rev.end());
   			if(rev==cur)
   				out++;
   		}
   		cout<<out<<endl;
   	}
    return 0;
}
