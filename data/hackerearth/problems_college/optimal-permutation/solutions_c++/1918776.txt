#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int n,t,s,temp,i,j,k;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1)
    	cout<<"0"<<endl;
    	else if(n==2)
    	cout<<"1"<<endl;
    	else if(n==3)
    	cout<<"3"<<endl;
    	else
    	{
    		s=n/2;
    	temp=((n-2)*(2+n-1))/2;
    	s=s+temp;
    	cout<<s<<endl;
    	}
    	
    }
    return 0;
}
