#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n, k;
    	cin >> n >> k;
    	vector<int> v;
    	int i, j;
    	for(i=0;i<n;i++)
    	{
    		cin >> j;
    		int count = 0;
    		while(j)
    		{
    			j &= (j-1);
    			count++;
    		}
    		v.push_back(count);
    	}
    	sort(v.begin(),v.end());
    	reverse(v.begin(), v.end());
    	int sum = 0;
    	for(i=0;i<k;i++)
    	{
    		sum+=v[i];
    	}
    	cout << sum << endl;
    }
    return 0;
}
