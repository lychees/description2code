#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>

#include <vector>

#define INF 2147000000

#define LLPRIME 9223372036854775783
#define INTPRIME 2147483647

using namespace std;

int T;
vector <string> V;

string str;

int main()
{
    ios_base::sync_with_stdio(0);

    cin >> T;
    for (int i = 1; i <= T; i++) {
        cin >> str;
        V.push_back(str);
    }

    sort(V.begin(), V.end());
    V.push_back("$$$");

    int rep = 1;
    for (int i = 0; i < V.size() - 1; i++) {
        rep = 1;
        while (V[i + 1] == V[i]) {
            i++;
            rep++;
        }
        cout << V[i] << " " << rep << endl;
    }

}
