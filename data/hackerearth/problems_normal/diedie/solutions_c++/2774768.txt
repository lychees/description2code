#include <iostream>
#define lli long long unsigned int
using namespace std;
lli modP = 1000000007;

lli output(lli b, lli e, lli m){
	lli ans = 1;
	while(e>0){
		while(e%2 == 0){
			b *=b;
			e/=2;
			b%=m;
		}		
		ans *= b;
		e--;
		ans %= m;
	}
	return ans;
}

int main()
{
	ios::sync_with_stdio(false);
    int t;
    cin >> t;
    while(t--){
    	lli n;
    	cin >> n;
    	n--;
    	n %= (modP-1);
    	cout << output(2, n*(modP-2), modP) << endl;
    	
    }
    return 0;
}
