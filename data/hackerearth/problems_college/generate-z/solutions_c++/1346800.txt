#include <cstdio>
#include <iostream>
using namespace std;

int main(){
	int n,i,j;
	scanf("%d",&n);

	for(i=1;i<=n;++i)
		printf("*");
	printf("\n");

	for(i=2;i<n;++i){
		for(j=1;j<=(n-i);++j)
			printf(" ");
		printf("*\n");
	}
	for(i=1;i<=n;++i)
		printf("*");
		

	return 0;
}
