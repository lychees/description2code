noOfTestCases = int(raw_input())


def maxRadius(l, b, h):

    no = min(l, b, h) / 2
    intNo = int(no)
    decimal = int((no - intNo) * 10)

    print str(intNo) + "." + str(decimal)


for no in range(1, noOfTestCases + 1):
    l, b, h = raw_input().split(" ")
    l, b, h = float(l), float(b), float(h)
    maxRadius(l, b, h)
