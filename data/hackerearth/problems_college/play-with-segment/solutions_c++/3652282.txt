ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
int main()
    {
		int i,t,n,ans,start,mini,maxi,val;
		sc("%d",&t);
		while(t--)
		{
			sc("%d",&n);
			vi v(n);
			for(i=0;i<n;i++)
				sc("%d",&(v[i]));
			ans=1;
			start=0;
			mini=maxi=v[0];
			vi m(1000001);
			m[v[0]]=0;
			for(i=1;i<n;i++)
			{
				val=v[i];
					if(val>maxi)
					{
						maxi=val;
						if(val-mini>1)
						{
							//cout<<"himini "<<i<<"\n";
							ans=max(ans,i-m[mini]);
							start=m[mini]+1;
							mini=v[start];
						}
					}
					else
					{
						if(val<mini)
						{
							mini=val;
							if(maxi-val>1)
							{
								//cout<<"himaxi"<<"\n";
								ans=max(ans,i-m[maxi]);
								start=m[maxi]+1;
								maxi=v[start];
							}
						}
						else
						{
							//cout<<"hi"<<"\n";
							ans=max(ans,i-start+1);
						}
					}
					//cout<<"ans="<<ans<<" start="<<start<<"\n";
				m[val]=i;
			}
			ans=max(ans,n-start);
			pf("%d\n",ans);
		}
    return 0;
}
