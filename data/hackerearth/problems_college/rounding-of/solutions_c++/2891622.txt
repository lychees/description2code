#include <iostream>
using namespace std;

int main()
{
	long  long int t,n,b,i,j;
	double ans,p,q;
	cin>>t;
	while(t--){
		cin>>n>>b;
		i=0;
		p=n;
		q=b;
		p=p/q;
		i=n/q;
		if((p-i)>=0.5)
		i++;
	//	cout<<"i is "<<i<<endl;
		ans=i*b;
		cout<<ans<<endl;
	}
  //  cout << "Hello World!" << endl;
    return 0;
}
