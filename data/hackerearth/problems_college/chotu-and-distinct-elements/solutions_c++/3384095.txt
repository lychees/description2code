#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t ,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int a[n] ,i , j=0,k, m ,b[n];
		for(i=0; i<n ;i++)
		scanf("%d",&a[i]);
		
		sort(a,a+n);
		if(a[0]==a[n-1])
		{
			printf("%d\n",a[0]);
			continue;
		}
			
		for(i=0;i<n-1;i++)
		{
			if(a[i]==a[i+1])
			a[i]=0;
		}
		j=0;
		for(i=0;i<n;i++)
		{
			if(a[i]!=0)
			b[j++]=a[i];
		}
		for(i=j-1 ,m=0; i>=0,m<i; i-- ,m++)
		{
		printf("%d %d ",b[i],b[m]);
		}
			if((j-1)%2==0)
			{
		k=(j-1)/2;
		printf("%d",b[k]);
	}
		printf("\n");
	}
	return 0;
}
