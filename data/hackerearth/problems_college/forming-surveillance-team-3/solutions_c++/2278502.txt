#include <iostream>
using namespace std;

int main()
{
    long t,m,i,j,k,c;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>m;
    	long a[m],b[m];
    	for(j=0;j<m;j++)
    	{
    		cin>>a[j];
    		b[m]=0;
    	}
    	c=0;
    	for(k=0;k<m;k++)
    	{
    		if(c==0)
    		{
    			b[c]=a[k];
    			c++;
    		}
    		else{
	    		for(j=0;j<c;j++)
	    		{
	    			if(a[k]==b[j])
	    				break;
	    		}
	    		if(j==c)
	    		{
	    			b[c]=a[k];
	    			c++;
	    		}
    		}
    	}
    	cout<<c<<endl;
    }
}
