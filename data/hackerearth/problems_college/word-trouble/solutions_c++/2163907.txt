

#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    int i;
    int l1;
    int l2;
    string s1,s2;
    int c;
    cin>>t;
    while(t--)
    {
    	cin>>s1>>s2;
    	l1=s1.size();
    	l2=s2.size();
    	if(s1==s2)
    	{
    		cout<<"Exact\n";
    	}
    	else
    	{
    		c=0;
    		if(s1.size()<s2.size())
    		{
    			for(i=0 ; i< l1;i++)
    			{
    			if(s1[i]!=s2[i])
    			c=c+1;
			}
    			c=c+s2.size()-s1.size();
    		}
    		else
    		{
    			for(i=0; i<l2 ;i++)
    			{
    			if(s1[i]!=s2[i])
    			c=c+1;
			}
    			c=c+s1.size()-s2.size();
    		}
    		cout<<"Non Exact "<<c<<endl;
    		
    	}
    }
    return 0;
}
