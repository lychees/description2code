#include<iostream>

using namespace std;

unsigned long long int test,n,sum,i,x,dv,rem,previous,total,j,maxi;

unsigned long long int power(unsigned long long int x)
{
	unsigned long long int y=1;

	while(x--)
		y*=2;

	return y;
}

int main()
{

	cin>>test;
	while(test--)
	{
		cin>>n;

		sum=0;
		previous=0;
		i=1;
		while(true)
		{

			sum+=(i*power(i));

			if(sum>=n)
				break;
			
			previous=sum;
			i++;
		}

		if(i>1)
			maxi=(2*(i-1))-1;
		else
			maxi=0;

		n-=previous;
		
		dv=(n-1)/i;
		rem=(n-1)%i;


		if(dv>1)
		{
			maxi=(2*i)-1;
		}
		else if(((dv*i)+rem+1)>maxi)
		{
			maxi=(dv*i)+rem+1;

			if((dv==1)&&(rem==(i-1)))
				maxi-=1;
		}
		cout<<maxi<<endl;

	}		

	return 0;
}
