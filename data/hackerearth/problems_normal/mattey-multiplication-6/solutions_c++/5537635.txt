#include <iostream>
#include<vector>
using namespace std;

int main()
{
    long T,N,M;
    
    cin >> T;
    while(T--){
    	vector<long> list={};
    	cin >> N;
    	cin >> M;
    	
    	while(M>1 ){
    		
    		int i=0;
    		long tmp=1;
    		long pre=tmp;
    		while(true){
    			pre=tmp;
    			tmp = tmp<<1;
    			if(tmp>M){
    				break;
    			}
    			else{
    				i++;
    			}
    		}
    		list.push_back(i);
    		//cout << i << endl;
    		M = M- pre;
    	}
    	//cout << M << endl;
    	if(M==1)
    	    list.push_back(0);
    	int count = list.size();
    	int i = 0;
    	while(i<count){
    		if(i!=0)
    			cout << " + " ;
    		cout << "(" << N << "<<" << list.at(i) << ")" ;
    		i++;
    	}
    	cout << endl;
    }
    return 0;
}
