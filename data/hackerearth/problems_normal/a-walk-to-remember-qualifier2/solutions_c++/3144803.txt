#include<bits/stdc++.h>
using namespace std;
#define f first
#define s second
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define ll long long
vector<int>comp[200011],adj[200011];
int disc[200011],mark[200011],low[200011],stmem[200011],compno[200011];
stack<int>st;
int tym,comps;
void dfs(int u)
{
    disc[u]=tym;
    tym++;
    low[u]=disc[u];
    st.push(u);
    stmem[u]=1;
    for(int i=0;i<adj[u].size();i++)
    {
        int k=adj[u][i];
        if(disc[k]==0)
        {
            dfs(k);
            low[u]=min(low[u],low[k]);
        }
        else if(stmem[k]==1)
            low[u]=min(low[u],disc[k]);
    }
//    printf("%d\n",u);
//    for(int i=0;i<5;i++)
//        printf("%d ",stmem[i]);
//    printf("\n");
    if(low[u]==disc[u])
    {
        while(!(st.top()==u))
        {
            comp[comps].pb(st.top());
            compno[st.top()]=comps;
            stmem[st.top()]=0;
            st.pop();
        }
        comp[comps].pb(u);
        compno[u]=comps;
        comps++;
        stmem[u]=0;
        st.pop();
    }
}
int main()
{
    int i,j,n,m;
    scanf("%d %d",&n,&m);
    for(i=0;i<m;i++)
    {
        int u,v;
        scanf("%d %d",&u,&v);
        u--;
        v--;
        if(u==v)
            mark[u]=1;
        else
            adj[u].pb(v);
    }
    tym=1;
    comps=0;
    for(i=0;i<n;i++)
        if(disc[i]==0)
            dfs(i);
//    for(i=0;i<n;i++)
//        printf("%d ",compno[i]);
//    printf("\n");
//    for(i=0;i<comps;i++)
//    {
//        for(j=0;j<comp[i].size();j++)
//            printf("%d ",comp[i][j]);
//        printf("\n");
//    }
    for(i=0;i<n;i++)
    {
        if(mark[i]==1||comp[compno[i]].size()>1)
            printf("1 ");
        else
            printf("0 ");
    }
    return 0;
}
