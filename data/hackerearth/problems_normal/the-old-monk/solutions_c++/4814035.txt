#include<bits/stdc++.h>
using namespace std;
long long a[100005],b[100005];
int main()
{
    long long t,n,i,lef,mid,ryt,maxi,pos;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>b[i];
        }
        maxi=-1;
        for(i=0;i<n;i++)
        {
            lef=0;
            ryt=i;
            pos=LLONG_MAX;
            while(lef<ryt)
            {
                mid=(lef+ryt)/2;
                if(a[mid]<=b[i])
                {
                    maxi=max(maxi,i-mid);
                    ryt=mid;
                }
                else
                    lef=mid+1;
            }
        }
        if(maxi==-1)
            cout<<0<<"\n";
        else
        cout<<maxi<<"\n";
    }
    return 0;
}
