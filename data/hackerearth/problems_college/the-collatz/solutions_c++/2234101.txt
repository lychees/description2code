#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int a;
    	cin>>a;
    	int even = 0;
    	int odd = 0;
    	int steps = 0;
    	
    	while(a/2!=0)
    	{
    		if(a%2==1)
    		{
    			odd += a;
    			a =  a*3+1;
    		}
    		
    		else
    		{
    			even += a;
    			a /= 2;
    		}
    		
    		
    		
    		steps++;
    	}
    	odd++;
    	//cout<<steps<<even<<odd;
    	if((even%steps)<(odd%steps))
    	{
    		cout<<"Odd Rules\n";
    	}
    	else if((even%steps)>(odd%steps))
    	{
    		cout<<"Even Rules\n";
    	}
    	else
    	{
    		cout<<"Tie\n";
    	}
    }
    return 0;
}
