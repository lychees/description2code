#include <bits/stdc++.h>
using namespace std;

int main()
{
   int test;
   cin>>test;
   while(test--)
   {
      long int length,i,j,count=0;
      cin>>length;
      long int arr[length];
      for(i=0;i<length;i++)
         cin>>arr[i];
      for(i=0;i<length-1;i++)
         for(j=i+1;j<length;j++)
         {
            if((arr[i]+arr[j])%3==0)
               count++;
         }
      cout<<count<<endl;
      count=0;
   }
   return 0;
}
