#include <iostream>
#include<stdio.h>
#include<algorithm>
long long t,x,y,n,c1,c2,sum,a[1000000],i;
using namespace std;

int main()
{
scanf("%lld",&t);
while(t--){
   scanf("%lld %lld %lld",&x,&y,&n);
   for(i=0;i<n;i++){
   	scanf("%lld",&a[i]);
   }
   std::sort(a,a+n);
   sum=0;c1=0;c2=0;
   for(i=0;i<n;i++){
   	sum=sum+a[i];
   	if(sum>x&&sum>y)break;
   	if(sum<=x)c1++;
   	if(sum<=y)c2++;
   }
   if(c1>c2)printf("Raghu Won\n");
   else if(c1<c2)printf("Sayan Won\n");
   else printf("Tie\n");
	
}
    return 0;
}
