'''

s = 1
t = int(raw_input())
for i in range(0, t, 1):
    line = raw_input()
    one, two, three = line.split()
    a = int(one)
    b = int(two)
    c = int(three)
    temps = s
    count = 1
    while temps < c:
        start = a * temps
        end = b * temps
        if end >= c:
            temps = end
        else:
            if count % 2 == 0:
                temps = end
            else:
                temps = start
        count += 1
    if (count - 1) % 2 == 0:
        print "Bob"
    else:
        print "Adam"
