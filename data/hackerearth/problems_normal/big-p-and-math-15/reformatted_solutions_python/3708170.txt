x = raw_input()
maxs = ""
mins = ""
for i in range(len(x)):
    if (x[i] == '6'):
        mins = mins+'5'
        maxs = maxs+'6'
    elif (x[i] == '5'):
        mins = mins+'5'
        maxs = maxs+'6'
    else:
        mins = mins+x[i]
        maxs = maxs+x[i]
arr1 = map(int, mins.strip().split(' '))
arr2 = map(int, maxs.strip().split(' '))
sumx = []
sumx.append(sum(arr1))
sumx.append(sum(arr2))
print sumx[0], sumx[1]
