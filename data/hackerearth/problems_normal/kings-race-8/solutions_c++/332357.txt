#include <bits/stdc++.h>
using namespace std;
struct st{
    int jmp;
    int id;
    int pas;
}arr[1000015];

bool cmp(st a,st b){
    return a.jmp<b.jmp ;
}

int hur[1000015] ;

int main(){
    int T,n,k,i,j,q,r;
    scanf("%d",&T);
    while(T--){
        scanf("%d %d",&n,&k) ;
        for(i=0;i<n;i++){
            scanf("%d",&arr[i].jmp) ;
            arr[i].id = i;
            arr[i].pas = -1;
        }
        for(i=0;i<k;i++){
            scanf("%d",&hur[i]);
        }
        sort(arr,arr+n,cmp) ;
        j = 0 ;
        q = 0;
        r = 0;
        for(i=0;i<n;i++){
            while(hur[j]<=arr[i].jmp && j<k){
                j++;
            }
            arr[i].pas = j;
            if(j==q){
                r = min(r,arr[i].id) ;
            }
            else if(j>q){
                q = j;
                r = arr[i].id ;
            }
        }
        cout<<r<<endl;
    }
}
