#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long n,c,ans,i,sq,s;
    while(t--)
    {
    	cin>>n;
    	vector<long> piles;
    	sq = sqrt(n);
    	for(i=2;i<=sq;i++)
    	{
    		if(n%i==0)
    		{
    			c = 0;
    			while(n%i==0)
    			{
    				c++;
    				n/=i;
    			}
    			piles.push_back(c);
    		}
    	}
    	
    	if(n!=1)
    	{
    		piles.push_back(1);
    	}

    	ans = piles[0];
    	s = piles.size();
    	for(i=1;i<s;i++)ans = (ans^piles[i]);
    	
    	if(ans)
    	{
    		cout<<"Printf{}\n";
    	}
    	else
    	{
    		cout<<"Pappu Bhujia\n";
    	}
    	
    }
    return 0;
}
