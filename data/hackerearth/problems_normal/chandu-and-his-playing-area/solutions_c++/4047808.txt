#include<iostream>
using namespace std;

void fun(long x)
{
    long area;
    if(x%4==0)
    {
        area=(x/4)*(x/4);
        cout<<area<<endl;
    }
    else
    {
        area=(x/4)*(x/4+1);
        cout<<area<<endl;
    }
}

int main()
{
    long T,N,i;
    cin>>T;
    for(i=0;i<T;i++)
    {
        cin>>N;
        if(N%2==0)
            fun(N);
        else
            fun(N-1);
    }
    return 0;
}
