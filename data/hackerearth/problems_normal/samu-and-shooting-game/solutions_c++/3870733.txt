#include <bits/stdc++.h>
using namespace std;

int x,y,p1,p2,n,w,ts,tc=0;
double sv[1001][10010],p,q;
int vs[1001][10010];

double go(int i,int t){
	if(i>=n){
		if(t>=w) return 1.0;
		return 0.0;
	}
	if(vs[i][t]==tc) return sv[i][t];
	vs[i][t]=tc;
	
	double a=p*go(i+1,t+x)+(1-p)*go(i+1,t);
	double b=q*go(i+1,t+y)+(1-q)*go(i+1,t);
	
	sv[i][t]=max(a,b);
	
	return sv[i][t];
}
int main(){
	scanf("%d",&ts);
	while(ts--){
		tc++;
		scanf("%d%d%d%d%d%d",&x,&y,&n,&w,&p1,&p2);
		p=(double)p1/100.0;
		q=(double)p2/100.0;
		printf("%.6lf\n",go(0,0)*100);
	}
	return 0;
}
