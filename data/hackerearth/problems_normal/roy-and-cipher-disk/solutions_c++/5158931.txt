#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int ans[s.length()];
        int first = 97;
        for(int i=0;i<s.length();i++){
            int num;
            num = s[i] - first;
            if(num>13){
                num=num-26;
            }
            if(num<=-13){
                num = (num+26);
            }
            first = s[i];
            ans[i] = num;
        }
        for(int i=0;i<s.length();i++){
            if(i==s.length()-1){
                cout<<ans[i]<<endl;
            }
            else{
                cout<<ans[i]<<" ";
            }
        }
    }
}
