#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	long long int i,len;
    	i=0;
    	while(s[i]!='\0')
    	    i++;
    	len=i;    
        if(len%2==0)
        {
        	len/=2;
        	long long int sum=0;
        	i=0;
        	while(i<len)
        	{
        		sum+=(s[i]-48);
        		i++;
        	}
        	while(s[i]!='\0')
        	{
        		sum-=(s[i]-48);
        		i++;
        	}
        	if(sum==0)
        	   cout<<"YES"<<endl;
        	else
        	   cout<<"NO"<<endl;
        }
        else
           cout<<"NO"<<endl;
    }
    return 0;
}
