#include<iostream>

using namespace std;

int palindrome[1000];
char arr[1000];
int main() {
	int testCase,len;
	cin >> testCase;
	while (testCase--)
	{
		
		cin >> arr;
		len = 1;
		for (int i = 1; arr[i] !='\0' ; i++)
		{
			palindrome[i-1] = abs(arr[i] - arr[i - 1]); 
			len++;
		}
		int i = 0,flag=0;
		len--;
		while (i < len-1 ) {
			if (palindrome[i] == palindrome[len-1])
			{
				len--;
				i++;
				continue;
			}
			else
				flag = 1;
			break;
		}
		if (!flag) {
			cout << "Funny\n";
		}
		else {
			cout << "Not Funny\n";
		}
	}

	return 0;
}
