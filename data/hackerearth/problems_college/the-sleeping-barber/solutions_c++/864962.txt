#include <bits/stdc++.h>
using namespace std;

int main()
{
	int d;
	cin>>d;
	while(d--)
	{
		int i,g;
		cin>>g;
		long long m,G[g+1];
		
		unsigned long long sum=0;
		cin>>m;
		for(i=0;i<g;i++)
		cin>>G[i];
		sort(G,G+g);
		for(i=0;i<g;i++)
		{
			sum+=G[i];
			if(sum>m)
			break;
		}
		if(i==g)
		cout<<g<<endl;
		else
		cout<<i<<endl;
	}
    return 0;
}
