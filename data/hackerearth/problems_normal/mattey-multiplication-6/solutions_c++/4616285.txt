#include <iostream>
#include<vector>
using namespace std;
vector <int> ans;
int main()
{
    long long t,n,m;
    cin >>t;
    while(t--)
    {
    	ans.clear();
    	cin >>n>>m;
    	int bit=0;
    	while(m)
    	{
	    	if(m%2!=0)
	    	{
	    		ans.push_back(bit);
	    	}
	    	m=m/2;
	    	bit++;
    	}
    	 cout <<"("<<n<<"<<"<<ans[ans.size()-1]<<")";
    	for(int i = ans.size() - 2; i >= 0; i--){       //traverse in a reverse order

          
            cout <<" + ("<<n<<"<<"<<ans[i]<<")";
        }
        cout <<endl;
    }
    return 0;
}
