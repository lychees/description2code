#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll n,w,wi[1005],val[1005];
ll dp[1005][1005];
ll func(ll tn, ll tw)
{
    if(tn==0||tw==0)
        return 0;
    if(dp[tn-1][tw]!=-1)
        return dp[tn-1][tw];
    if(wi[tn-1]>tw)
        return func(tn-1,tw);
    return dp[tn-1][tw]=max((val[tn-1]+func(tn-1,tw-wi[tn-1])),func(tn-1,tw));
}
main()
{
    ll t,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>w;
        for(i=0;i<n;i++)
            cin>>wi[i];
        for(i=0;i<n;i++)
            cin>>val[i];
        memset(dp,-1,sizeof(dp));
        cout<<func(n,w)<<endl;
    }
    return 0;
}
