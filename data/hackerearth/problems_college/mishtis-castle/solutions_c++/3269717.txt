#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int dp[101][101];
int dp1[101][101];

int main()
{
    int t,n,i,j,k,c;
    cin>>t;
    while(t--)
    {
    	memset(dp1,0,sizeof(dp1));
    	memset(dp,0,sizeof(dp));
    	cin>>n>>c;
    	for(i=1;i<=n;i++)
    		for(j=1;j<=n;j++)
    			cin>>dp[i][j];
    			
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			dp[i][j]=dp[i][j]+max(dp[i-1][j],dp[i][j-1]);
    		}
    	}
    	
    	if(dp[n][n]>=c)
    		cout<<dp[n][n]-c<<endl;
    	else
    		cout<<"-1\n";
    }
    return 0;
}
