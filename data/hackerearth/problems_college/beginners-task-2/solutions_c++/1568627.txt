#include <iostream>
using namespace std;
typedef long int l;

int main(){
	ios::sync_with_stdio(false);
	int T=0;
	int N;
	l X;
	
	cin >> T;
	
	while(T--)
	{
	 N=0;
	 cin >> N;
	 l sum=0;
	 int num=0;
	 while(N--)
	 {
	  X=0;
	  cin >> X;
	  sum+=X;
	  num++;
	 }
	 cout << sum/num << endl;
	}
	
	return 0;
}
