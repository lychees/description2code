#include <bits/stdc++.h>
using namespace std;
int a[110];
int dp[110][100005];
int fun(int index,int sum,int n)
{
if(sum==0)
return true;
if(index==n)
return false;
if(dp[index][sum]!=-1)
return dp[index][sum];
if(a[index]<=sum)
dp[index][sum]=fun(index+1,sum-a[index],n)||fun(index+1,sum,n);
else
dp[index][sum]=fun(index+1,sum,n);
return dp[index][sum];
}

int main()
{
int n,x;
scanf("%d",&n);
for(int i=0;i<n;i++)
scanf("%d",&a[i]);
int q;
scanf("%d",&q);
memset(dp,-1,sizeof dp);
while(q--)
{
scanf("%d",&x);
if(x>100000)
{
cout<<"No\n";
continue;
}
int res=fun(0,x,n);
if(res)
cout<<"Yes\n";
else
cout<<"No\n";
}
return 0;
}
