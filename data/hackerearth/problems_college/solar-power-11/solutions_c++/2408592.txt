#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
typedef long long int ll;
#define mod 1000000007
ll arr[1005][1005];
int main()
{
	ll i,j,n,k,cnt=0,s,d,t;
    char c1[2];
    scanf("%lld",&t);
    while(t--)
    {
        memset(arr,0,sizeof(arr));
        cin>>n>>k;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                arr[i][j]=k;
            }
        }
        scanf("%lld",&s);
        while(s--)
        {
            cin>>c1;
            if(c1[0]=='m')
            {

                ll a,b,c,d;
                scanf("%lld %lld %lld %lld",&a,&b,&c,&d);
                cnt=0;
                for(int r=a;r<=c;r++)
                {
                    for(int col=b;col<=d;col++)
                    {
                           // printf("k\n");
                            cnt+=arr[r][col];
                    }
                }
                printf("%lld\n",cnt);
            }
            else
            {
                ll a,b,c;
                scanf("%lld %lld %lld",&a,&b,&c);
                arr[a][b]=(arr[a][b]%mod*c%mod)%mod;
            }
        }

    }
    return 0;
}
