lists=[_ for _ in raw_input().split()]
init=97
dicit={}
for key_add in range(26):
	key=init+key_add
	dicit.update({chr(key):lists[key_add]})
T=int(raw_input())
tot_m=0
tot_d=0
for _ in range(0,T):
	lists=list(raw_input())
	sum_m=0
	for i in lists:
		sum_m=sum_m+int(dicit[i])
	lists=list(raw_input())
	sum_d=0
	for i in lists:
		sum_d=sum_d+int(dicit[i])
	tot_d=tot_d+sum_d
	tot_m=tot_m+sum_m
if(tot_m==tot_d):
	print "Draw"
else:
	if(tot_m>tot_d):
		print "Marut"
	else:
		print "Devil"
