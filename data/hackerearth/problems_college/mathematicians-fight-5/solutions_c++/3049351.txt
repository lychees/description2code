#include <bits/stdc++.h>

using namespace std;

map<int , int> mp;

int main()
{
	int n,t,x;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		int x = 1;
		int count = pow(2,x);
		while(count < n) {
			++x;
			count = pow(2,x);
		}
		if(count == n) {
			cout << "1" << endl;
		} else {
			count = pow(2,x-1);
			x = n - count;
			cout << 1+(x*2) << endl;
		}
	}
	return 0;
}
