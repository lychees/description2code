#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>

using namespace std;


int main()
{
	
    char str[100];
   	cin >> str;
    
    std::vector<char> myvector (str, str+strlen(str)); 
    std::sort (myvector.begin(), myvector.end());

    
    for (auto it = std::begin(myvector); it!=std::end(myvector); ++it)
    std::cout << *it;
    
}
