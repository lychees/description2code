#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,i;
	long long int total=0;
	cin>>n;
	int arr[n];
	for(i = 0; i < n; i++) {
		cin>>arr[i];
	}
	sort(arr,arr+n);
	int arr1[n];
	int j,m;
	j = 0;
	m = n-1;
	int index = 0;
	for(i = 0; i < n; i++) {
		if(i%2 == 0) {
			arr1[index++] = arr[j++];
		}
		else {
			arr1[index++] = arr[m--];
		}
		if(i) {
			total += abs(arr1[i]-arr1[i-1]);
		}
	}
	long long int answer=0;
	j = 0;
	m = n-1;
	index = 0;
	for(i = 0; i < n; i++) {
		if(i%2 == 0) {
			arr1[index++] = arr[m--];
		}
		else {
			arr1[index++] = arr[j++];
		}
		if(i) {
			answer += abs(arr1[i]-arr1[i-1]);
		}
	}
	total = max(total,answer);
	cout<<total<<endl;
	return 0;
}
