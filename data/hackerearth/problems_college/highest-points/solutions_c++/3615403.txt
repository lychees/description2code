#define MOD 1000000007

using LD = long double;
using LL = long long int;
using ULL = unsigned long long int;

{

	std::cin >> n;
	auto points = std::set<LL> ();
	while (n--)
	{
		auto w = 0LL, t = 0LL, l = 0LL;
		std::cin >> w >> t >> l;
		points.insert((3 * w) + t - l);
	}
	std::cout << *points.rbegin() << std::endl;
	return 0;
}
