#include <iostream>
using namespace std;

void ways(int n,int *dp)
{
	
	int i,x,y,z;
	
	dp[0]=1;
	
	for(i=1;i<=n;i++)
	{
		x=(i-1<0)?0:dp[i-1];
		y=(i-3<0)?0:dp[i-3];
		z=(i-4<0)?0:dp[i-4];
		
		dp[i]=(x+y+z)%10000007;
	}
	
	
}
int main()
{
    int tno,i,max;
    
    cin>>tno;
    
    int *N=new int[tno];
    
    for(i=0;i<tno;i++)
    	cin>>N[i];
    	
    max=0;
    for(i=0;i<tno;i++)
    	if(max<N[i])
    		max= N[i];
    		
    int *dp=new int[max+1];
    
    ways(max,dp);
    
    for(i=0;i<tno;i++)
    	cout<<dp[N[i]]<<endl;
    		
    
    return 0;
}
