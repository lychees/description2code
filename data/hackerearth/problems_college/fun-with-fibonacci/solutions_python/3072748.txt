aa = int(raw_input())
outt=[]
for x in range(aa):
    b= int(raw_input())
    if b==0 or b==1:
        outt.append(0)
    else:
        l=[0,1]
        while(l[-1]<=b):
            l.append(l[-1]+l[-2])

    s=[j for j in l if j<=b if j%2==0]
    outt.append(sum(s))
for k in outt:
    print k
