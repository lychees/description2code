#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char str[100005];
		scanf("%s",str);
		int l=strlen(str)-1;
		int mark=l;
		bool found=false;
		for (int i = 0; i < l; i++)
        {
            char cI = str[i];
            char cJ = str[l];
            if (cI == cJ)
            {
                found = true;
                l--;
                if(mark > i)
                    mark = i;
            }
            else
            {
                if (found)
                {
                    found = false;
                    i--;
                }
                l = strlen(str) - 1;
                mark = l;
            }
		}
		printf("%d\n",mark+strlen(str));
	}
    return 0;
}
