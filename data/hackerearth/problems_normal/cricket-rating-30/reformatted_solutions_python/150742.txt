n = int(raw_input())
if n <= 0:
    print '0'
else:
    inp = raw_input()
    results = inp.split()
    results = list(map(int, results))

    rating = 0
    tot = 0

    for i in range(len(results)):
        tot += results[i]
        if tot < 0:
            tot = 0
        if tot > rating:
            rating = tot
    print(rating)
