#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int palindrome(string a,string b)
{
	int k=0;
	int f=0;
	int a1,b1;
	a1=a.length();
	b1=b.length();
	if(a1 != b1)
		return 0;
	else
	{
		for(int i=a1-1; i>=0; i--)
		{
			if(a.at(i)==b.at(k))
			{
				k++;
			}
			else
			{	f=1;	
				break;
			}
		}
		if(!f)
				return 1;
	}
	return 0;
}

int main()
{
    int n;
    int x;
    cin>>n;
    string s[n];
    
    for(int i=0; i<n; i++)
    {	
    		cin>>s[i];
   	}
    for(int i=0; i<n-1; i++)
    {
    	for(int j=i+1; j<n; j++)
    	{
    		x=palindrome(s[i],s[j]);
    		if(x)
    			break;
    	}
    	if(x)
    	{	cout<<s[i].length()<<" "<<s[i].at((s[i].length()/2))<<endl;
    		break;
    	}
    	
    }
    
    return 0;
}
