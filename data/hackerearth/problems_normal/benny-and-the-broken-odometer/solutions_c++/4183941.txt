#include <iostream>
using namespace std;

int main()
{
    int t,n,i,digit,ans;
    cin >> t;
    while(t--){
    	cin >> n;
    	ans = 0;
    	i = 1;
    	while(n){
    		digit = n%10;
    		n /= 10;
    		if(digit > 3){
    			digit--;
    		}
    		ans += digit*i;
    		i *= 9;
    	}
    	cout << ans << endl;
    }
    return 0;
}
