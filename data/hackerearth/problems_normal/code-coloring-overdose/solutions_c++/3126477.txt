#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k,m;
    scanf("%d %d %d",&n,&k,&m);
    int arr[n];
    unordered_map<int, int> v;
    for(int i=0;i<n;i++)
    {
    	int a;
    	scanf("%d",&a);
    	v[a]+=1;
    }
    
    long long dp[m+1];
    memset(dp,0,sizeof(dp));
    dp[0]=1;
    unordered_map<int, int > :: iterator it=v.begin(),it2=v.end();
    while(it!=it2)
    {
    	long long prev=dp[0];
    	//cout<<it->first<<" "<<it->second<<endl;
    	for(int x=1;x<=m;x++)
    	{
    		long long temp=dp[x];
    		long long value=1;
    		
    		for(int i=1;i<=it->second;i++)
    		value=(value*2)%1000000007;
    		
    		
    		
    		dp[x]=(dp[x]+((value-1)*prev)%1000000007)%1000000007;
    		prev=temp;
    	}
    	it++;
    	
    }
    long long sum=0;
    	for(int i=0;i<=m;i++)
    	sum=(sum+dp[i])%1000000007;
    cout<<sum<<endl;;
    
    return 0;
}
