#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

vector<int> inp;

int main(){
    int n,k,x;
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++){
        scanf("%d",&x);
        inp.pb(x);
    }
    sort(inp.rbegin(),inp.rend());
    lli ans  = 0;
    for(int i=0;i<n;i++){
        int m = i/k + 1;
        ans += (long long)m*inp[i];
    }
    printf("%lld\n",ans);
    return 0;
}
