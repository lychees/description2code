#include <iostream>
using namespace std;

int sum_dig(int n)
{
	int sum = 0;
	while(n > 0) {
		sum += n % 10;
		n = n/10;
	}
	return sum;
}

int main()
{
    int n;
    int sum = 0, max_sum = 0;
    while((scanf("%d",&n)) > 0) {
    	int out = sum_dig(n);
    	if(out >= sum) {
    		sum = out;
    		max_sum = n;
    	}
    }
    printf("%d\n", max_sum);
    return 0;
}
