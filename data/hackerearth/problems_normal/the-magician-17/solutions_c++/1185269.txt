#include <cstdio>

int main(){
	int T, n, k, res, l, r, tot;
	
	scanf("%d", &T);
	
	while (T--){
		
		scanf("%d%d", &n, &k);
		
		if (n < 3){
			printf("%d\n", k);
			continue;
		}
		if (n == 3){
			printf("%d\n", k == 1 ? 1 : 5 - k);
			continue;
		}

		if (k == n){
			l = 1; r = 1;
			res = 2; tot = 3;
		}
		else{
			r = n - k;
			res = 1;
			l = 0;
			tot = r + 1;
		}
		
		while (tot + r < n){
			res = 2;
			r = r * 2 + l;
			l = 1;
			tot = r + l + 1;
		}
		
		printf("%d\n", res + (n - tot)* 2);
	}
	
	return 0;
}
