#include<bits/stdc++.h>
#define opt ios_base::sync_with_stdio(false);cin.tie(NULL);
#define li long int
#define lli long long int
#define rep(i,a,b) for(i=a;i<b;i++)
#define repi(i,v) for(i=v.begin();i!=v.end();i++)
#define vi vector<int>
#define vli vector<li>
#define vlli vector<lli>
#define pb push_back
#define pob pop_back
#define all(v) v.begin(),v.end()
#define sz size()
#define lb lower_bound
#define ub upper_bound
#define elif else if
#define mp make_pair
#define vpp vector<pair<li,li> >
#define F first
#define S second
#define mod 1000000007
#define itr iterator
using namespace std;
int main(){
    opt
    int t;
    cin>>t;
    while(t--){
        int n,i,j,k=0;
        cin>>n;
        vli v;
        rep(i,0,n){
            li x;
            cin>>x;
            v.pb(x);
        }
        i=0,j=v.sz-1;
        while(i<j){
            li chk=2*v[j],sum=v[i];
            while(sum<=chk){
                if(i+1==j)
                    break;
                sum+=v[++i];
            }
            v[i]=(sum-chk);
            j--;
        }
        cout<<i+1<<" "<<n-i-1<<"\n";
    }
    return 0;
}
