#include <iostream>
#include <bits/stdc++.h>
using namespace std;
set <int> s;
set <int> :: reverse_iterator rit;
map <int,int> mp;
map <int,int> :: iterator it;
int main()
{
    int n,i,x;
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	cin>>x;
    	mp[x]++;
    }
    for(it=mp.begin();it!=mp.end();it++)
   	{
   		if(it->second==1)
   		{
   			s.insert(it->first);
   		}
   	}
   	if(s.empty())
   	{
   		cout<<"-1"<<endl;
   	}
   	else
   	{
   		rit=s.rbegin();
   		cout<<*rit<<endl;
   	}
    return 0;
}
