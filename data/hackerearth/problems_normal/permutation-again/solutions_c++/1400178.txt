#include<cstdio>
#include<iostream>
using namespace std;
long long a[100001];
main()
{
	a[1]=1;
a[2]=1;
	for(int i=3; i<=100000; i++)
	{
		if(i%2!=0)
			a[i]=i-1+a[i-1];
		else
			a[i]=i+a[i-1];
	}
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n;
		scanf("%d", &n);
		printf("%lld\n", a[n]);
	}
}
