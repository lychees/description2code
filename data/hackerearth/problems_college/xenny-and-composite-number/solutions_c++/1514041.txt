#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t,n,i,j,m;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	i=n+1;
    	m=1;
    	while(i>n)
    	{
    		for(j=2;j<=sqrt(i);j++)
    		{
    			if(i%j==0)
    			{m=0;break;}
    			else 
    			{m=1;}
    		}
    	if(m==0)
    	{cout<<i<<endl;break;}
    	if(m==1)
    	{i++;}
    	}
    }
    return 0;
}
