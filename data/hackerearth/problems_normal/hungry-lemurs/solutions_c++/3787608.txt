#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    
 
	int nbananas, klemurs, i, j, calc, output;
 
	cin >> nbananas >> klemurs;
 
    output = 1e9; // a large value
 
    for (j = 2*klemurs; j >= 1; j--)
    {
        i = nbananas - (nbananas % j);
    
        output = min(output, abs(nbananas - i) + abs(klemurs - j));
        i += j;
        output = min(output, abs(nbananas - i) + abs(klemurs - j));
    }
 
    cout << output;
 
    return 0;
}
