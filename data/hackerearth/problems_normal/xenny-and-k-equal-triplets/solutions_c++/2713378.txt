#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    unsigned long long int i,T,N,K,ctr_K,el,num,den,n1,n2,min,i1;
    
    scanf("%llu",&T);
    while(T--){
    	ctr_K=0;
    
      	scanf("%llu",&N);
     
    	scanf("%llu",&K);
    	
    	for(i=0;i<N;i++){
    	   scanf("%llu",&el);
    	    if(el==K)
    	     { ctr_K++; }
    	}
      
    	if(ctr_K<3)
    		printf("0/1\n");
    		
   
    	else
    	{
    		num=ctr_K*(ctr_K-1)*(ctr_K-2);
    		den=N*(N-1)*(N-2);
    		
               n1=num;n2=den;
            /*  while(n1!=n2){
                   if(n1>=n2-1)
                      n1=n1-n2;
                   else
                     n2=n2-n1;
                 }
              */
              min=(n1>n2)?n2:n1; /* minimum value is stored in variable min */
  for(i1=min;i1>=1;--i1)
  {
      if(n1%i1==0 && n2%i1==0)
      {
          //printf("HCF of %d and %d is %d", num1, num2,i);
           
           break;
      }
  }
         n1=i1;
				
    		num=num/n1;
    		den=den/n1;
    
    		printf("%llu/%llu\n",num,den);
     }
    	
    }
    return 0;
}
