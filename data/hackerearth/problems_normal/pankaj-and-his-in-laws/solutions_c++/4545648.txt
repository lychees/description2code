    #include <iostream>
    using namespace std;
     
    int main()
    {
        int n,a[100],b[100],i,j,c,d,p,x[20],temp,l;
        cin>>n;
        for(i=0;i<n;i++)
        cin>>a[i];
         for(i=0;i<n;i++)
          b[i]=1;
        for(i=1;i<n;i++)
        {
            for(j=0;j<i;j++)
            {
                if(a[j]<a[i])
                {
                    c=b[i];
                    d=b[j]+1;
                    if(c>=d)
                        b[i]=c;
                    else
                        b[i]=d;
                }
            }
        }
        p=b[0];
        for(i=1;i<n;i++)
        {
            if(p<b[i])
            {
                p=b[i];
            }
        }
        temp=p;
        for(j=0;temp!=0;j++)
        {
        	x[j]=temp%2;
        	temp=temp/2;
        }
        l=j;
        for(j=l-1;j>=0;j--)
        cout<<x[j];
        cout<<"\n";
    }
