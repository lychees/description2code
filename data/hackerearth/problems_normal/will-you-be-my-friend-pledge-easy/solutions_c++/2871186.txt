#include <bits/stdc++.h>
using namespace std;

int GCD(int A, int B) {
    if(B==0)
        return A;
    else
        return GCD(B, A % B);
   }
int main()
{
   int m,n,count=0;
   int a;
   cin>>m;
   cin>>n;
   for(int i=0;i<n;i++)
   {
   	cin>>a;
   	if(GCD(m,a)>1)
   	count++;
   }
   	cout<<count;
    return 0;
}
