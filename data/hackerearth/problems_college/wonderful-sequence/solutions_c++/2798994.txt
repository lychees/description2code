#include <bits/stdc++.h>
using namespace std;
static int a[100001],inc[100001];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n;
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%d",&a[i]);
    		inc[i]=0;
    	}
    	int ma=0;
    	for(int i=1;i<n;i++){
    		for(int j=0;j<i;j++){
    			if(a[j]<a[i]){
    				inc[i]=max(inc[i],inc[j]+1);
    				ma=max(ma,inc[i]);
    			}
    		}
    	}
    	printf("%d\n",n-ma-1);
    }
    return 0;
}
