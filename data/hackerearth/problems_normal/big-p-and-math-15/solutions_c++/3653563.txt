#include <iostream>
#include <stdlib.h>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
	char a[100001];
	char b[100001];
	
	cin>>a>>b;
	//cout<<atoi(a);
	int max=0,min=1000001;
	int temp=atoi(a)+atoi(b);
	max=temp;
	min=temp;
	//printf("%d\n",temp);
	int i=0;
	int lena=strlen(a);
	while(a[i]!='\0')
	{
		if(a[i]=='5')
		{
			max+=pow(10,lena-i-1);
		}
		if(a[i]=='6')
		{
			min-=pow(10,lena-i-1);
		}
		i++;
	}
	i=0;
	int lenb=strlen(b);
	while(b[i]!='\0')
	{
		if(b[i]=='5')
		{
			max+=pow(10,lenb-i-1);
		}
		if(b[i]=='6')
		{
			min-=pow(10,lenb-i-1);
			//printf("%d\n",pow(10,abs(lenb-i-1)));
		}
		i++;
	}
	cout<<min<<" "<<max<<endl;
   // cout << "Hello World!" << endl;
    return 0;
}
