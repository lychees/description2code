#include <iostream>
#include <bits/stdc++.h>
#define pi 3.141592653589793
using namespace std;

int main()
{
    int t;
    cin>>t;
    double v;
    
    while(t-->0){
    	cin>>v;
    	double r3 = (v)/(2.0*pi);
    	double r = cbrt(r3);
    	double h = v/(pi*pow(r,2));
    	//cout<<r<<":"<<h<<endl;
    	double area = 2*pi*r*h + 2*pi*r*r;
    	printf("%.9lf\n",area);
    }
    return 0;
}
