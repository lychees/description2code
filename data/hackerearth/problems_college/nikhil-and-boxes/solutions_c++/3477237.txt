#include<iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int y;
    scanf("%d", &y);
    while(y--)
    {
    	long long int l,w,h;
    	cin>>l>>w>>h;
    	if(l>w && w<h)
    	{
    		l=w;
    	}
    	else if(l>h && w>h)
    	{
    		l=h;
    	}
    	
    	double r;
    	r = l/2.0;
    	printf("%0.1f\n",r);
    }
    return 0;
}
