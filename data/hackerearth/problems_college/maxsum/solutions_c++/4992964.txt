#include <bits/stdc++.h>
using namespace std;

int main(){
	int t;
	cin>>t;
	int n;
	int arr[10001];
	int take,leave,tp;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>arr[i];

		take=arr[0];
		leave=0;
		for(int i=1;i<n;i++){
			tp=max(take,leave);
			take=leave+arr[i];
			leave=tp;
		}
		cout<<max(take,leave)<<endl;
	}	
	return 0;
}
