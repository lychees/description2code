#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n,h;
    	scanf("%d%d",&n,&h);
    	int a[n];
    	int min=10010;
    	for(int i=0;i<n;i++){
    		scanf("%d",&a[i]);
    		if(min>abs(h-a[i])) min=abs(h-a[i]);
    	}
    	printf("%d\n",min);
    }
    return 0;
}
