#include <iostream>
using namespace std;

int t;
unsigned long long int num;
unsigned long long int steps;

int main()
{
	cin>>t;
	while(t--){
		cin>>num;
		steps=0;
		
		start:
			if(num==1){
				cout<<steps<<endl;
				continue;
			}
			while(num%2!=0 && num%3!=0){
				++steps;
				--num;
			}
			while(num%3==0){
				num/=3;
				++steps;
			}
			
			while(num%2==0){
				num/=2;
				++steps;
			}
			
			goto start;
			
		}
	return 0;
}
