    #include <iostream>
    
    using namespace std;
     
     
    void fun()
    {
    	int n;
    	cin >> n;
    	int a[n], ans=0;
    	for(int l=0; l<n;l++) {
            cin >> a[l];
        }
        for(int i=0;i<n-1;i++) {
            for(int j=i+1; j<n;j++) {
                if((a[i]+a[j])%2==0 && a[i]!=a[j]) {
                        ans=ans+1;
                }
            }
        }
        cout << ans<<endl;
        return;
    }
    
    int main()
    {
        int t;
        cin >> t;
        while (t--)
        	fun();
        return 0;
    }
