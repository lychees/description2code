#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long pass,fail,x,y;
		cin>>pass>>fail;
		
		x = (pass+fail)/2 ,y = pass-x;
		if(x<0 || y<0 || ((pass+fail)%2==1))
			cout<<"impossible"<<endl;
		else if(x<y)
    		cout << y <<" "<<x<< endl;
    	else
    		cout <<x <<" "<<y<< endl;
	}
    return 0;
}
