#include <iostream>
#include <string>

using namespace std;

int power(int a, unsigned long long b) {
    if(b == 0) return 1;
    if(b == 1) return a%10;
    if(b%2 == 0) return ((power(a, b/2))*(power(a, b/2)))%10;
    else return (a*(power(a, b/2))*(power(a, b/2)))%10;
}

int solve(void) {
    string a;
    cin >> a;
    unsigned long long b;
    cin >> b;
    int x = a.size();
    x = a[x-1] - '0';
    return power(x, b);
}


int main(void) {
    int t;
    cin >> t;
    while(t--) {
        cout << solve() << endl;
    }
    return 0;
}
