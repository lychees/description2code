#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long n,k;
        cin>>n>>k;
        long a[n];
        for(int i=0;i<n;i++)
        cin>>a[i];
        long l[n],r[n];
        memset(l,-1,sizeof(l));
        memset(r,-1,sizeof(r));
        for(long i=1;i<n;i++)
        {
            long j=i-1;
            while(l[j]!=-1 && a[j]>a[i])
            j=l[j];
            if(a[j]<a[i])
            l[i]=j;
            else
            l[i]=-1;
        }
        for(long i=n-2;i>=0;i--)
        {
            long j=i+1;
            while(r[j]!=-1 && a[j]>a[i])
            j=r[j];
            if(a[j]<a[i])
            r[i]=j;
            else
            r[i]=-1;
        }
        long long s=0;
        for(long i=0;i<n;i++)
        {
            long r1=n+1;
            if((i-k)>=0)
            {
                long j=i-k;
                while(l[j]!=-1 && a[j]>a[i])
                {
                    j=l[j];
                }
                if(a[j]<a[i])
                r1=min(r1,i-j);
            }
            if((i+k)<n)
            {
                long j=i+k;
                while(r[j]!=-1 && a[j]>a[i])
                {
                    j=r[j];
                }
                if(a[j]<a[i])
                r1=min(r1,j-i);
            }
            if(r1!=n+1)
            s+=r1;
        }
        cout<<s<<endl;
    }
    return 0;
}
