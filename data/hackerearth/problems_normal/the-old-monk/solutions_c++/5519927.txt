#include <iostream>
#include<climits>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
        long int *arri=new long int[n];
    	long int *arrj=new long int[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>arri[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin>>arrj[i];
    	}
    	int min1=0;
    	for(int i=0;i<n;i++)
    	{
    		int j=i;
    		while(j>=i && arrj[j]>=arri[i] && j<n)
    		{
    			j++;
    		}
    		int r=j-i-1;
    		if(r>min1)
    		min1=r;
    	}
    	cout<<min1<<endl;
    }
    return 0;
}
