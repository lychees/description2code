#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int len = str.length();
    	int i = 1;
    	string b = "";
    	b+=str[0];
    	while(i<len)
    	{
    		if(str[i]!=str[i-1]) b+=str[i];
    		i++;
    	}
    	cout<<b<<endl;
    }
}
