N, M = map(int, raw_input().split())
divs = map(int, raw_input().split())
cols = [i-1 for i in map(int, raw_input().split())]

i = 0
for num in xrange(N):
    divider = divs[cols[num]]
    i = (i/divider+1)*divider  # next possible
    while 0 in [i % divs[j] for j in xrange(cols[num]+1, M)]:
        i += divider
print i
