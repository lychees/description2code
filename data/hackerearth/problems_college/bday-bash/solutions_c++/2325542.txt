#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long c,sp,bf,f;
    	cin>>c>>sp>>bf>>f;
    	sp*=100;
    	bf*=50;
    	f*=25;
    	long long spr=sp,bfr=bf,fr=f;
    	long long sum=spr+bfr+fr;
    	while(sum<c){
    		if(sum+sp<=c){
    			sum+=sp;
    			spr+=sp;
    		}
    		if(sum+bf<=c){
    			sum+=bf;
    			bfr+=bf;
    		}
    		if(sum+f<=c){
    			sum+=f;
    			fr+=f;
    		}else
    			break;
    	}
    	if(sum>c||sum<c)
    		cout<<"No\n";
    	else cout<<spr<<" "<<bfr<<" "<<fr<<endl;
    }
    return 0;
}
