for i in range(input()):
    n=input()
    res=0
    flag=0
    while True:
    	res += (n%10)**2
    	n=n/10
    	if n==0:
    		if res==1:
    			flag=1
    			break
    		elif res<10:
    			break
    		n=res
    		res=0
    	if n>0:
    		continue
    if flag==1:
    	print 'Case #'+str(i+1)+':YES'
    else:
    	print 'Case #'+str(i+1)+':NO'
