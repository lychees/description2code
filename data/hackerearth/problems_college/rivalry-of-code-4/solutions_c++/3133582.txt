#include <iostream>
using namespace std;
int main()
{
	int tc,i,j;
	cin>>tc;
	string s,d;
	while(tc--)
	{
		cin>>d>>s;
		if(s.length()<d.length())
		{
			string a;
			a=s;
			s=d;
			d=a;
		}
		int ans=51;
		for(i=0;i<s.length()-d.length()+1;i++)
		{
			int c=0;
			for(j=0;j<d.length();j++)
			if(d[j]!=s[i+j])
			c++;
			if(c<ans)ans=c;
		}
		cout<<ans<<endl;
	}
	return 0;
	}
