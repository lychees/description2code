#include<iostream>
#define ll long long int

using namespace std;
int main()
{
	ll t,n,a,b;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n==5)
		cout<<3<<endl;
		if(n==6)
		cout<<5<<endl;
		if(n==7)
		cout<<7<<endl;
		for(a=8,b=16;b>0;a=b,b=b*2)
		{
			//cout<<a<<" "<<b<<endl;
			if(n>=a&&n<b)
			{
				cout<<(n-a)*2+1<<endl;
				break;
			}
			
			
		}
		
	}
	return 0;
}
