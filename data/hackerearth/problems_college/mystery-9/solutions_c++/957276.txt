#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t,n;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		
		if(n<=31)
			cout<<"1";
		else if(n<=59)
			cout<<"2";
		else if(n<=90)
			cout<<"3";
		else if(n<=120)
			cout<<"4";
		else if(n<=151)
			cout<<"5";
		else if(n<=181)
			cout<<"6";
		else if(n<=212)
			cout<<"7";
		else if(n<=243)
			cout<<"8";
		else if(n<=273)
			cout<<"9";
		else if(n<=304)
			cout<<"10";
		else if(n<=334)
			cout<<"11";
		else
			cout<<"12";
		cout<<endl;
	}
	return 0;
}
