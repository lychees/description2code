#include<iostream>
#include<algorithm>
#include<vector>
#include<string.h>
#include<map>
using namespace std;

map< pair< int,int > , int >mp;
vector< pair<int,int> >v;

int main()
{
  int i,j,n,m,x,y,temp,ans=0;
  cin>>n;
  for(i=1;i<=n;i++) {
    cin>>x>>y;
    v.push_back(make_pair(x,y));
    mp[make_pair(x,y)]++;
  }
  sort(v.begin(),v.end());
  for(i=0;i<v.size();i++) {
    for(j=i+1;j<v.size();j++) {
      if(v[i].first!=v[j].first) break;
      temp=v[j].second-v[i].second;
      if(mp.find(make_pair(v[i].first+temp,v[i].second))!=mp.end() && mp.find(make_pair(v[j].first+temp,v[j].second))!=mp.end()) ans++;
    }
  }
  cout<<ans;
//  system("pause");
  return 0;
}
