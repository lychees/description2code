#include<bits/stdc++.h>
using namespace std;
#define ll long long 

int main()
{
    int t; cin>>t;
   
    while(t--)
    {
        ll b, w; cin>>b>>w;
        ll x, y, z; cin>>x>>y>>z;
        if(abs(x-y)>z)
        {
            if(x>y)
                cout<<y*w+b*(y+z)<<"\n";
            else
               cout<<x*b+w*(x+z)<<"\n";
        }
        else cout<<b*x+y*w<<"\n";
        
    }
    return 0;
}
