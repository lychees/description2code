#include <iostream>
using namespace std;

int main()
{
    int t,i=1,num,st=-1,en=-1,max2=0,max1=0;
    int arr[100001]={0};
    cin>>t;

    while(i<=t)
    {
    	cin>>num;
    	arr[i]=num;
    	i++;
    }
    max2=0;
    for(i=1;i<=t;i++)
    {
    	if(st==-1 )
            {
                st=1;
                max2++;
            }
        else if(st==1  && (arr[i]-arr[i-1]==1||arr[i]-arr[i-1]==-1))
        {
            max2++;

        }
        else if(st==1  && (arr[i]-arr[i-1]!=1  && arr[i]-arr[i-1]!=-1))
        {
            st=1;
            if(max1<max2)
                max1=max2;
                max2=1;

        }

    }
    if(max1<max2)
        max1=max2;
    cout<<max1;
    return 0;
}
