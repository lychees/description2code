#include <stdio.h>
#include<iostream>
using namespace std;
int main()
{
    long long a,b;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%Ld%Ld",&a,&b);
    	if(a+1==b)
    	{
    		printf("%Ld\n",a&b);
    		continue;
    	}
    	if(b%2==0)
    	b-=1;
    	printf("%Ld\n",b&(b-1));
    }
    return 0;
}
