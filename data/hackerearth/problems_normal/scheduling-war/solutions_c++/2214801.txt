#include <bits/stdc++.h>
using namespace std;
int a[100000];
int main()
{
    int t, s1, s2, x, num1, dif, pos, i, s, t1;
    cin>>t;
    s = s1 = dif = 0;
    for(i = 0 ; i < t; i++)
    {
    	cin>>a[i];
    	s += a[i];
    }
   
    cin>>t1;
    while(t1--)
    {
    	cin>>x;
    s2 = 0;
    for(i = 0 ; i < x; i++)
    {
    	cin>>pos;
    	s2 += a[pos-1];
    }
    cin>>num1;
    s1 += num1;
  
    if(s2 < num1)
    {
    	dif += num1-s2;
      }
    }
    if( s1 > s)
    cout<<s1-s<<endl;
    else
    cout<<dif<<endl;
    
    return 0;
}
