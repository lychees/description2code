#include <iostream>
using namespace std;
int gcd(int a,int b){
	if(b==0){
		return a;
	}
	return gcd(b,a%b);
}
int main()
{
    int n,k,size,maxsize,i,temp;
    cin>>n>>k;
    int a[n];
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    if(a[0]>=k){
    	size=1;
    }
    else{
    	size=0;
    }
    maxsize=0;
    temp=a[0];
    for(i=1;i<n;i++){
    	temp=gcd(temp,a[i]);
    	if(temp>=k){
    		size++;
    	}
    	else{
    		temp=a[i];
    		if(size>maxsize){
    			maxsize=size;
    		}
    		if(a[i]>=k){
    			size=1;
    		}
    		else{
    			size=0;
    		}
    	}
    }
    maxsize=maxsize>size?maxsize:size;
    cout<<maxsize;
    return 0;
}
