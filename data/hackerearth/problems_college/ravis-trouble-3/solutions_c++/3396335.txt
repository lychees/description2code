#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		string p,q;
		cin>>p>>q;
		int l=q.length();
		int k=p.length();
		int index=0;
		int counter=0;
		if(k>=l)
		{
		for(int i=0;i<l && index<k;i++)
		{
			while(index<k)
			{
				if(q.at(i)==p.at(index))
				{
					counter++;
		            index++;
		            break;
				}
				index++;
			}
			
		}
		}
		if(counter==l)
		{
			cout<<"YES"<<endl;
			
		}
		else
		cout<<"NO"<<endl;
	}
}
