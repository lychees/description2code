#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int *a = new int[n];
		for(int i=0;i<n;i++)
			a[i] = i+1;
		
		do
		{
			for(int i=0;i<n;i++)
				cout<<a[i];
			cout<<" ";
		}while(next_permutation(a,a+n));
		cout<<endl;
	}
}
