#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000000007LL
#define get getchar_unlocked

int main()
{
    //freopen("in10.txt", "r", stdin);
    //freopen("out10.txt", "w", stdout);
    int t;
    s(t);
    while (t--) {
    int n, i, j, x, cnt, k;
    s(n);
    int mrk[n+1], a[n+1], ans = 0;
    memset(a, -1, sizeof(a));
    for (i = 1; i <= n; ++i) {
        for (j = 1; j <= n; ++j)
            mrk[j] = 0;
        for (j = 1; j <= n; ++j) {
            s(x);
            ++mrk[x];
        }
        cnt = 0, k = -1;
        for (j = 1; j <= n; ++j) {
            if (mrk[j] >= 2) {
                ++cnt;
                k = j;
            }
        }
        assert(cnt <= 1);
        if (cnt) {
            a[i] = k;
            ++ans;
        }
    }
    assert(ans == n-2);
    for (i = 1, j = n-1; i <= n; ++i) {
        if (a[i] == -1)
            a[i] = j++;
        printf("%d ", a[i]);
    }
    printf("\n");
    }
    return 0;
}
