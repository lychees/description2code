#include <iostream>
using namespace std;

int main()
{
    int t, n, count;
    cin >> t;
    int array[1000001];
    while(t--)
    {
    	count = 0;
    	cin >> n;
    	for (int i = 0; i < n; i++)
    		cin >> array[i];
    	for (int i = 0; i < n; i++) {
    		for (int j = n-1; j >= i; j--) {
    			if (array[j] == array[i])
    				count ++;
    		}
    	}
    	cout << count << endl;
    }
    return 0;
}
