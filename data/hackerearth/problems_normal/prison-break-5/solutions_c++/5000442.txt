#include <iostream>

using namespace std;
int n,count1=0;
int arr[20][20];
bool visited[20][20];
int x[4]={0,1,-1,0},y[4]={1,0,0,-1};
void dfs(int j,int k)
{
    for(int i=0;i<4;i++)
    {
        if((j+x[i]>=0&&j+x[i]<n)&&(k+y[i]>=0&&k+y[i]<n)&&(!visited[j+x[i]][k+y[i]]))
        {
            if(arr[j+x[i]][k+y[i]]==0)
            {
                visited[j+x[i]][k+y[i]]=true;
               // cout<<j+x[i]<<" "<<k+y[i]<<" "<<arr[j+x[i]][k+y[i]]<<endl;
                dfs(j+x[i],k+y[i]);
            }
        }
    }
    visited[j][k]=false;
    if(j==n-1&&k==n-1)
        count1++;
}


int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n;
        for(int j=0;j<n;j++)
        {
            for(int k=0;k<n;k++)
            {
                cin>>arr[j][k];
                visited[j][k]=false;
            }
        }
        visited[0][0]=true;
        dfs(0,0);
        cout<<count1<<endl;
        count1=0;
    }
}
