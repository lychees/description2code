int binary_search(int a[],int x,int lo,int hi){
	while(hi>lo){
		int mid=lo+(hi-lo)/2;
		if(a[mid]==x) return mid;
		else if(a[mid]<x) lo=mid+1;
		else hi=mid-1;
	}
	return lo;
}
