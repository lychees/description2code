#include<iostream>
using namespace std;

int main()
{
	char arr[1001][1001];
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		for(int i=0;i<n;i++)
			for(int j=0;j<m;j++)cin>>arr[i][j];
		int cnt=0;
		for(int i=0;i<n-1;i++)
			for(int j=0;j<m-1;j++)
			{
				if(arr[i][j]=='/' and arr[i][j+1] == '\\' and arr[i+1][j]=='\\' and arr[i+1][j+1]=='/')cnt++;
			}
		cout<<cnt<<endl;
	}
	return 0;
}
