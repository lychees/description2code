int n,k;
ll dp[1<<16][17];
ll func(int i,int mask,int last)
{
	int j;
	if(i==n){
		for(j=0;j<n;j++){
			if((mask>>(j))&1){
				break;
			}
		}
		if(abs(3*last-2*(j+1))<=k)
		return 1;
		return 0;
	}
	if(i==1){
		ll ans=0;
		for(j=0;j<n;j++){
			mask^=(1<<j);
			ans+=func(i+1,mask,j+1);
			mask^=(1<<j);
		}
		return ans;
	}
	if(dp[mask][last]!=-1)
	return dp[mask][last];
	ll ans=0;
	for(j=0;j<n;j++){
		if((mask>>(j))&1){
			if(abs(3*last-2*(j+1))<=k){
				mask^=(1<<j);
				ans+=func(i+1,mask,j+1);
				mask^=(1<<j);
			}
		}
	}
	dp[mask][last]=ans;
	return ans;
}
