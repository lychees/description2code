#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long sum,n,x,y,p,q,a[100000],t,i,j;
    cin>>t;
    while(t--)
    {sum=0,p=0,q=0;
    	cin>>x>>y>>n;
    	for(i=0; i<n; i++)
    	{cin>>a[i];
    	}
    	sort(a,a+n);
    	for(i=0;i<n;i++)
    	{sum+=a[i];
    	 if(sum<=x)p+=1;
    	if(sum<=y)q+=1;
    	}
    	if(p>q)cout<<"Chef Won\n";
    	else if(p<q)cout<<"Jhool Won\n";
    	else cout<<"Tie\n";
    }
    return 0;
}
