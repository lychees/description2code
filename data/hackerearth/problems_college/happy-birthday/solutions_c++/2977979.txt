#include<bits/stdc++.h>
using namespace std;
long long int b,c,d,t,n,i,j,k,a[1000];
int main()
{
    scanf("%lld",&n);
    b=0;c=0;d=0;t=0;
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&a[i]);
    	if(a[i]==1)
    	b++;
    	else if(a[i]==2)
    	c++;
    	else if(a[i]==3)
    	d++;
    	else
    	t++;
    }
    k=0;
   // printf("%lld %lld %lld %lld\n",b,c,d,t);
    if(c%2==0)
    {
    	t=t+c/2;
    	c=0;
    }
    else
    {
    	t=t+c/2;
    	c=1;
    }
    while((b>0)&&(d>0))
    {
    	b--;d--;
    	t++;
    }
    if((c==1)&&(b>=2))
    {
    	t++;
    	c=0;b=0;
    }
    else if((c==1)&&(b==1))
    {
    	k++;
    	c=0;b=0;
    }
    while(b>0)
    {
    	b=b-4;
    	k++;
    }
    //printf("%lld %lld %lld %lld\n",b,c,d,t);
    k=k+t+d+c;
    printf("%lld\n",k);
    return 0;
}
