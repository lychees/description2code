using namespace std;
int mat[2][2]={{1,1},{1,0}};
int res[2][2],pw[2][2],d[2][2];
void mul(int a[2][2],int b[2][2],int c[2][2]){
    for(int i=0;i<2;i++) for(int j=0;j<2;j++){
        d[i][j]=0;
        for(int k=0;k<2;k++) d[i][j] += ((LL)a[i][k]*b[k][j])%MOD;
        d[i][j]%=MOD;
    }
    for(int i=0;i<2;i++) for(int j=0;j<2;j++) c[i][j]=d[i][j];
}
void raise(LL power){
    for(int i=0;i<2;i++) for(int j=0;j<2;j++){
        res[i][j]=(i==j);
        pw[i][j]=mat[i][j];
    }
    while(power){
        if(power&1) mul(res,pw,res);
        mul(pw,pw,pw);
        power>>=1;
    }
}
int main(){
    int T;
    LL N;
    scanf("%d",&T);
    while(T--){
        scanf("%lld",&N);
        if(N<=2) puts("1");
        else{
            raise(N-2);
            printf("%d\n",(res[0][0]+res[0][1])%MOD);
        }
    }
    return 0;
}
