from math import sqrt

kases = int(raw_input())
for kase in range(kases):
	entry = raw_input()
	all_values = entry.split(' ')
	z = int(all_values[0])
	m = int(all_values[1])
	n = int(all_values[2])
	
	z_sqrt = int(sqrt(z)) + 1
	factors_list = {}
	counter = 0
	
	for x in range(1, z_sqrt):
		if z % x == 0:
			factors_list[x] = z // x
	
	for x in factors_list:
		if x <= m and factors_list[x] <= n:
			counter += 1
		if x <= n and factors_list[x] <= m and x != factors_list[x]:
			counter += 1
	
	print counter
