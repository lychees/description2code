#include<bits/stdc++.h>
using namespace std;
bool cmp(const pair<long int,long int>&i, const pair<long int,long int>&j)
{
    if(i.second!=j.second) return i.second<j.second;
    else
        return i.first<j.first;
}

int main()
{
        int i,j,k,m,n,t,a,b;
        vector<pair<int,int> > vp;
        cin>>n;
        for(i=0;i<n;i++)
        {
                cin>>a>>b;
                vp.push_back(make_pair(i+1,a+b));
        }
        sort(vp.begin(),vp.end(),cmp);
        for(i=0;i<n;i++) cout<<vp[i].first<<" ";
        return 0;
}
