#include<bits/stdc++.h> 
#define ll long long 
#define mylimit 1000001
using namespace std;
int main()
{
	set<ll int> s;
	ll int a[1000010];
	memset(a,0,sizeof(a));
	ll int i,j,t;
	a[1]=1;
	s.insert(1);
	for(i=2;i<mylimit;i++)
	{
		t =i;
		for(j=2;;j++)
		{
			t*= i;
			if(t<mylimit)
			{
				a[t]=1;
				s.insert(t);
			}
			else
			{
			break;
		       }
		}
	}
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int x;
		cin>>x;
		int flag =0;
		set<ll int>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			i = *it;
			if((x-i)>=0 && a[x-i]==1)
			{
				flag =1;
				break;
			}
		}
		if(flag)
		{
			cout<<"Yes"<<"\n";
		}
		else
		{
			cout<<"No"<<"\n";
		}
	}
	return 0;
}
