#include<iostream>
using namespace std;
#include<math.h>
int hashi[10000001];
int multi[10000001];
int main()
{
    int p=2,k=1;
    while(p*k<=10000000)
        hashi[p*k++]++;
    for(int j=3;j*j<=10000000;j+=2)
    {
        if(hashi[j]==0)
        {
            int p=j,k=1;
        while(p*k<=10000000)
        hashi[p*k++]++;
        }
    }
    for(int i=2;i<=10000000;i++)
        if(hashi[i]==0)
        hashi[i]=1;
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    multi[hashi[i]]++;
    int m;
    cin>>m;
    for(int i=0;i<m;i++)
    {
        int k;
        cin>>k;
       if(k>32)
       cout<<0<<endl;
       else
        cout<<multi[k]<<endl;
    }
}
