t = int(raw_input())
while t > 0:
    a, b = [int(x) for x in raw_input().split()]
    if a > 60:
        print("NA")
    else:
        r = 50

        if a > 2:
            r += (a-2)*12
        if b > 0:
            r += (b*2)
        print(r)
    t = t-1
