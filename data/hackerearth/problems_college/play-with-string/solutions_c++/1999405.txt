#include <iostream>
using namespace std;

int main()
{
    int t;
    for(cin>>t;t>0;t--)
    {
    	string str;
    	cin>>str;
    	int j=str.length()-1;
    	for(int i=0;i<str.length()-2;i++)
    	{
    		if(abs(str[i+1]-str[i])!=abs(str[j-1]-str[j]))
    		break;
    		j--;
    	}
    	if(j==1)
    	cout<<"Funny"<<endl;
    	else
    	cout<<"Not Funny"<<endl;
    }
    return 0;
}
