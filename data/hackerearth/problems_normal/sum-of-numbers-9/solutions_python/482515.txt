
# Read input from stdin and provide input before running code

test = int(raw_input())
def f(l, s):
    if s == 0: 
#    	print "Yes"
    	return True
    if l == []:
 #   	print "No"
    	return False
    return f(l[1:], s) or f(l[1:], s - l[0])

for i in range(0,test):
	n = int(input())
	m = str(raw_input())
	m = m.split(" ")
	for k in range(0,n):
		m[k]=int(m[k])
	s=int(input())
	if f(m, s) :
		print "YES"
	else:
		print "NO"
