#include <iostream>
using namespace std;
int isprime[5000002]={0};
void prime()
{
	int i=2,j;
isprime[0]=1;
isprime[1]=1;
for(i=2;i*i<=1999999;i++)
{
   if(isprime[i]==1)
      continue;
   for(j=i*2;j<=1999999;j+=i)
   {
      isprime[j]=1; 
   }
}
	}
int main()
{ios_base::sync_with_stdio(false);
   long long int t,m,n,sum[1000002];
   prime();
   for(int i=0;i<=1000001;i++)
   {if(i!=0)
   	{if(isprime[i]==0)
   sum[i]=sum[i-1]+i;
   else
   sum[i]=sum[i-1];
   	}
   }
   
   cin>>t;
   while(t--)
   {
   	cin>>m>>n;
   	if(sum[n]-sum[m-1]<=1999999)
   {	if(isprime[sum[n]-sum[m-1]]==0)
   	cout<<"YES\n";
   	else
   	cout<<"NO\n";}
   	else
   	
   	{if((sum[n]-sum[m-1])%2==0||(sum[n]-sum[m-1])%3==0||(sum[n]-sum[m-1])%5==0||(sum[n]-sum[m-1])%7==0)
   		cout<<"NO\n";
   		else
   		cout<<"YES\n";
   	}
   }
   
    return 0;
}
