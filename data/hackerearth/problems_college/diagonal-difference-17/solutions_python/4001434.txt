n=input()
prim=[]
seco=[]
arr=[]
k=0

while k<n:
	
	arr=[]
	arr=list(map(int,raw_input().split()))
	rarr=arr[::-1]
	
	prim.append(arr[k])
	
	seco.append(rarr[k])
	
	k=k+1



addprim=0
addsec=0

for item in prim:
	addprim=addprim+item
	
for key in seco:
	addsec=addsec+key
	
print abs(addprim - addsec)
