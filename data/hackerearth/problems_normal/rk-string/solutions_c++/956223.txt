#include <iostream>
using namespace std;

string str;
int n,m,x,y,a[1000010],b[1000010],l;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	cin >> n >> m;
    	cin >> str;
    	l = str.size();
    	
    	x=y=0;
    	for(int i=0;i<l;i++) {
    		if(str[i] == 'R') a[x++] = i;
    		else b[y++] = i;
    	}
    	
    	long long an1,an2;
    	an1 = an2 = 0;
    	if(n) {
    		for(int i=0;i+n-1<x;i++) {
	    		long long t1 = (a[i]);
	    		if(i-1 >= 0) t1 -= a[i-1];
	    		else t1++;
	    		long long t2 = -1*a[i+n-1];
	    		if(i+n < x) t2 = t2 + a[i+n];
	    		else t2 += l;
	    		an1 += t1*t2;
	    		//cout << a[i] << " " << a[i+n-1] << " " << an1 << " " << t1 << " " << t2 << endl;
    		}
    	} else {
    		for(int i=0;i<x;i++) {
    			long long t1 = a[i];
    			if(i-1>=0) t1 -= a[i-1];
    			else t1++;
    			t1--;
    			an1 += (t1*(t1+1))/2;
    		}
    		long long t1 = l-1;
    		if(x-1 >=0) t1 -= a[x-1];
    		else t1++;
    		an1 += (t1*(t1+1))/2;
    	}
    	cout << an1 << " ";
    	if(m) {
    		for(int i=0;i+m-1<y;i++) {
    			long long t1 = b[i];
    			if(i-1 >=0) t1 -= b[i-1];
    			else t1++;
    			long long t2 = -1*b[i+m-1];
    			if(i+m < y) t2 = t2 + b[i+m];
    			else t2 += l;
    			an2 += t1*t2;
    		}
    	} else {
    		for(int i=0;i<y;i++) {
    			long long t1 = b[i];
    			if(i-1>=0) t1 -= b[i-1];
    			else t1++;
    			t1--;
    			an2 += (t1*(t1+1))/2;
    		}
    		long long t1 = l-1;
    		if(y-1>=0) t1 -= b[y-1];
    		else t1++;
    		an2 += (t1*(t1+1))/2;
    	}
    	cout << an2 << endl;
    }
}
