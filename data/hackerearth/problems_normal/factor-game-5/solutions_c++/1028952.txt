	using namespace std;
#include <bits/stdc++.h>

#define FOR(i,a,b) for(int i = (a), _b= (b); i<= _b; ++i)

const int N = 200000;

int nt[N], IsCal[N], Cal[N], n;
int ri(){ int x; scanf("%d", &x); return x;}
void init(){
	nt[1] = 1;
	for( int i = 1; i * i <= 100000; ++i){
		if ( nt[i] == 0){
			int j = i;
			while( j < 100000){
				j += i;
				nt[j] = i;
			}
		}
	}
	int p = 0;
	FOR(i,1,100000) if( nt[i] == 0) {
		nt[i] = i;
		++p;
	}
	
	Cal[1] = 0;
	IsCal[1] = 1;

	FOR(i,2,100000){
		int x = i;

		while(nt[x] > 1){
			if( IsCal[x] ){
				Cal[i] += Cal[x];
				break;
			}
			int p = nt[x];
			Cal[i] ++;
			x /= p;
		}
		IsCal[x] = 1;
	}
}

int main(){
	
	//freopen("inp","r",stdin);

	init();
	n = ri();
	int ret = 0;
	FOR(i,1,n){
		int x = ri();
		ret ^= Cal[x];
	}
	if (ret != 0) printf("ALICE\n"); else printf("BOB\n");
	return 0;
}
