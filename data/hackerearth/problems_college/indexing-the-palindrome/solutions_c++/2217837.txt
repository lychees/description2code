    #include <bits/stdc++.h>
    using namespace std;
    int T;
    string str;
    bool check(string s){
    	for(int i =0; i < (s.size())/2; i++){
    		char a= s[i];
    		char b = s[s.size()-i-1];
    		if(a != b ){return false;}
    	}
    	return true;}
    int main()
    {
    	scanf("%d\n",&T);
    	while(T--){
    		cin >> str;
    		if( check(str) ){ printf("-1\n");}
    		else{
    			int i;
    			for(i= 0; i < str.size(); i++){
    			if( check(str.substr(0,i) + str.substr(i+1))){break;}}
    			printf("%d\n",i);
    		}
    	}
        return 0;
    }
