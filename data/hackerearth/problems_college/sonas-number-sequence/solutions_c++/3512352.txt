#include <iostream>
using namespace std;

int main()
{
	int s;
	cin>>s;
	int a[s],d[s],q[s];
	for(int i=0;i<s;i++)
	{
		cin>>a[i];
	}
	d[0]=a[0];q[0]=a[0];
	cout<<d[0]<<" ";
	for(int j=1;j<s;j++)
	{
	   d[j]=(j+1)*a[j]-d[j-1];
	   q[j]=d[j]+d[j-1];
	   cout<<q[j]-q[j-1]<<" ";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
