#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int a[105][137];
int c[133];
int main()
{
    string tmp;
    int t,n,i,j,max,store;
    cin >> t;

    while(t--){
            memset(c,0,sizeof(c));
            memset(a,0,sizeof(a));
        cin >> n;
        for (i = 1; i <= n; i++){
            cin >> tmp;
            for (j = 0; j < tmp.size();j++){
                a[i][tmp[j]] = 1;
            }
        }
        j = 0;
        for (j = 97; j <= 130; j++)
        for (i = 1; i <= n; i++){
            c[j] = c[j] + a[i][j];
        }
    max = 0;

    for (i = 97; i <= 130; i++){
        if (c[i] > max){
            max = c[i];
            store = i;
        }
    }
    printf("%c\n", store);
    }
    return 0;
}
