#include<bits/stdc++.h>
using namespace std;
long int a[1000005];
int main()
{
    long int n,k;
    scanf("%ld%ld",&n,&k);
    for(int i=0;i<n;i++)
    {
    	scanf("%ld",&a[i]);
	}
sort(a,a+n);
int j=n-1;
int i=0;
bool sum=false;
while(i<j)
{
if((a[i]+a[j])==k)
{
	sum=true;
	break;
}
if((a[i]+a[j])<k)
{
	i++;
}
else
{
	j--;
}
	
}
if(sum==true)
{
	printf("YES\n");
}
else
{
	printf("NO\n");
}

    return 0;
}
