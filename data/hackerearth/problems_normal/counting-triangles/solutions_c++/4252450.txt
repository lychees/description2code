#include <iostream>
#include <map>
using namespace std;
typedef long long int ll;
int main()
{
    ll n;
    cin >> n;
    map <ll, map<ll, map<ll, ll> > > m;
    ll k = 0;
    while(n--) {
    	ll a, b, c, st, en, mi;
    	cin >> a >> b >> c;
    	mi = a + b + c;
    	st = min(a, min(b, c));
    	en = max(a, max(b, c));
    	mi = mi - st - en;
    	m[mi][st][en]++;
    	if(m[mi][st][en] == 1) {
    		k++;
    	}
    	if(m[mi][st][en] == 2) {
    		k--;	
    	}
    }
    cout << k << endl;
    return 0;
}
