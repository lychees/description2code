#include <iostream>
using namespace std;

#define p 1000000007

long long modexp(long long x){
	long long ans = 1;
	long long y = p-2;
	while(y){
		if(y&1){
			ans = (ans*x)%p;
		}
		x = (x*x)%p;
		y >>= 1;
	}
	return ans;
}

int main()
{
    int n,q,id,type;
    cin >> n;
    long long a[n],value,product = 1;
    int count = 0;
    for(int i = 0; i < n; i++){
    	cin >> a[i];
    	if(a[i] == 0){
    		count++;
    		continue;
    	}
    	product = (a[i]*product)%p;
    }
    cin >> q;
    while(q--){
    	cin >> type >> id;
    	id--;
    	if(type == 0){
    		cin >> value;
    		if(a[id] == value){
    			continue;
    		}
    		if(a[id] == 0 && value != 0){
    			a[id] = value;
    			count--;
    			product = (product*value)%p;
    			continue;
    		}
    		if(value == 0){
    			count++;
    			product = (product*modexp(a[id]))%p;
    			a[id] = 0;
    			continue;
    		}
    		product = (product*modexp(a[id]))%p;
    		a[id] = value;
    		product = (product*value)%p;
    		continue;
    	}
    	if((count == 1 && a[id] != 0) || (count > 1)){
    		cout << 0 << endl;
    		continue;
    	}
    	if(count == 1){
    		cout << product << endl;
    		continue;
    	}
    	cout << (product*modexp(a[id]))%p << endl;
    }
    return 0;
}
