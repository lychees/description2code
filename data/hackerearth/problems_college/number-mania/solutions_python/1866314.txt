inpt = input()
def isPrime(n):
	for i in range(2, int(n**.5)+1):
		if n % i == 0:
			return False
	return True
	
def factorial(n):
	res = 1
	for i in range(2, n+1):
		res *= i
	return res
	
for i in range(2, inpt+1):
	if isPrime(i):
		print factorial(i)
