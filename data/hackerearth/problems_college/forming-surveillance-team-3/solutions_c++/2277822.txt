#include <iostream>
#include <algorithm>
#define max 100000
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,a[max],i,j,un=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(j=0;j<n;j++)
    	{
    		if(a[j]==a[j+1])
    		continue;
    		else
    		un++;
    	}
    cout<<un<<endl;
    }
    return 0;
}
