#include <iostream>
#include<stack>

using namespace std;

int main()
{
    int t,n=1,c,a,b;
    char arr[]={'U','V','W','X','Y','Z'};
    cin>>t;
    
    while(t--)
    {
    	stack<int> s;
    	cout<<"Case "<<n++<<": ";
    	for(int i=0;i<3;i++)
    	{	cin>>c;
    		a=c;
			do
			{
				b=a%16;
				s.push(b);
				a=a/16;
			}while(a!=0);
			if(c<16)
				s.push(0);
		
	
		while(!s.empty())
    		{ if(s.top()<10)
    			cout<<s.top();
    			else
    			cout<<arr[s.top()-10];
    			s.pop();
    		}
    		cout<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
