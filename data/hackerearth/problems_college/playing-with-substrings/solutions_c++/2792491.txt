#include<bits/stdc++.h>
using namespace std;


int main()
{
	string str1,str2,z;
	int a,b,i,j,ans,x;
    cin>>str1>>str2;
	a=str1.length();
    b=str2.length();
    set<string>abc;
    ans=0;
    for(i=0;i<a;i++)
    {
        for(j=i;j<a;j++)
        {
            z=str1.substr(i,j-i+1);
            if(abc.find(z)==abc.end())
            {
                abc.insert(z);
                x=z.find(str2);
                if(x!=string::npos)
                  ans++;
            }
        }
	}
    cout<<ans;
    return 0;
}
