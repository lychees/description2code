#!/usr/bin/env python
# -*- coding: UTF-8 -*-
import time
import sys
import io
import re
import math
start = time.clock()
ans = 0
(n, d) = map(int, raw_input().split())
l = sorted([int(x) for x in raw_input().split()])
for i in range(n):
    j = i
    while True:
        j += 1
        if j > n-1:
            break
        elif l[j]-l[i] > d:
            break
        elif l[j]-l[i] <= d:
            ans += 2
print ans
