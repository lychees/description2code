#include <bits/stdc++.h>
using namespace std;

int main()
{
    int S, Q;
    scanf("%d %d", &S, &Q);
    
    set<int> X;
    
    while(Q--)
    {
    	int A;
    	scanf("%d", &A);
    	
    	int ans = __gcd(S, A);
    	
    	if(X.find(ans)!=X.end())
    		printf("-1\n");
    	else
    	{
    		X.insert(ans);
    		printf("%d\n", ans);
    	}
    }
    
    return 0;
}
