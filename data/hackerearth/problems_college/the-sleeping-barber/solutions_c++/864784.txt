#include <iostream>
using namespace std;

long long int partition(long long int a[3000], long long int p, long long int r)
{
	long long int x, down, up, temp;
	x=a[p];
	down=p;
	up=r;
	while(down<up)
	{
		while(a[down]<=x && down<r)
			down++;
		while(a[up]>x)
			up--;
		if(down<up)
		{
			temp=a[up];
			a[up]=a[down];
			a[down]=temp;
		}
	}
	temp=a[up];
	a[up]=a[p];
	a[p]=temp;
	return up;
}

void quicksort(long long int a[3000], long long int p, long long int r)
{
	if(p<r)
	{
		long long int q=partition(a,p,r);
		quicksort(a,p,q-1);
		quicksort(a,q+1,r);
	}
}

int main()
{
   	long long int d, g, m, a[3000], i, sum; 
    cin>>d;
    while(d--)
    {
    	cin>>g>>m;
    	for(i=0;i<g;i++)
    		cin>>a[i];
    	quicksort(a,0,g-1);
    /*	for(i=0;i<g;i++)
    		cout<<a[i]<<" ";*/
    	sum=0;
    	for(i=0;i<g;i++)
    	{
    		sum+=a[i];
    		if(sum>m)
    			{
    				break;
    			}
    		
    	}
    	cout<<i<<endl;
    }
}
