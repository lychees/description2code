	#include <bits/stdc++.h>
	using namespace std;
	
	int main() {
	    int T, N, i;
	    map <char, char> H;
	    string A[100], m;
	    cin >> T;
	    while(T--) {
	    	cin >> m;
	    	cin >> N;
	    	for(i=0; i<N; i++)
	    		cin >> A[i];
	    	for(i=0; i<26; i++)
	    		H[m[i]] = 'a'+i;
	        for(i=0; i<N; i++) {
	            for(int j=0; j< A[i].size(); j++)
	                A[i][j] = H[A[i][j]];
	        }
	        sort(A,A+N);
	        for(i=0; i<N; i++)  {
	            for(int j=0; j<A[i].size(); j++)
	                cout << m[A[i][j]-'a'];
	            cout<<endl;
	        }
	    }
	}
