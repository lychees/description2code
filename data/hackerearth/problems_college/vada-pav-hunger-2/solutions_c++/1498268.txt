#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define pb push_back
vector<int>adj[1005];
int ma=0;
bool visit[1005];
void dfs(int s,int ans)
{
visit[s]=true;
ans++;
ma=max(ma,ans);
for(int i=0;i<adj[s].size();i++)
{
    if(!visit[adj[s][i]])
    dfs(adj[s][i],ans);
}
}
int main()
{
int n;
sd(n);
memset(visit,false,sizeof(false));
for(int i=1;i<=n-1;i++)
{
    int x,y;
    sd(x);
    sd(y);
    adj[x].pb(y);
    adj[y].pb(x);
}
dfs(1,0);
cout<<2*(n-1)-(ma-1);
return 0;
}
