#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--){
    	int n;cin>>n;
    	int A[n];
    	int i;
    	for(i=0;i<n;i++)cin>>A[i];
    	int x=100,y=0,ans;
    	for(i=0;i<n;i++)
    	if(A[i]<x)x=A[i];
    	for(i=0;i<n;i++)
    	if(A[i]>y)y=A[i];
    	ans=2*(y-x);
    	cout<<ans<<endl;
    }
}
