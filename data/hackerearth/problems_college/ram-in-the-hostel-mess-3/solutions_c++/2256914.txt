#include <bits/stdc++.h>
using namespace std;

#define LL long long

int n;
LL dp[112345][2];

LL func(int pos,int ty)
{
	if(pos == n)
		return 1;
	LL &ret = dp[pos][ty];
	if(ret != -1)
		return ret;
	if(ty == 1)
		return ret = func(pos+1,0);
	else
		return ret = func(pos+1,0) + func(pos+1,1);	
}

int main()
{
	int t;
	LL v;
	cin >> t;
	while(t--)
	{
		cin >> n;
		memset(dp,-1,sizeof(dp));
		v = func(1,0) + func(1,1);
		cout << v*v << endl;
	}
    return 0;
}
