#include <iostream>

using namespace std;

int main()
{
   // cout << "Hello world!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
       int n ;
       cin>>n;
       int a[n];
       for(int i=0;i<n;i++)
        cin>>a[i];
        int c=0;
       for(int i=0;i<n;i++)
       {
           for(int j=i+1;j<n;j++)
           {
               if(((a[i]+a[j])%2==0)&&(a[i]!=a[j]))
                c++;
           }
       }
       cout<<c<<endl;
   }
    return 0;
}
