#include <iostream>
using namespace std;

int main()
{
    long long int z,n;
	cin>>z;
	for(long long int j=0;j<z;j++)
    	{
    		cin>>n;
    	long long int a[n];
    	for(long long int j=0;j<n;j++)
    	{
    		cin>>a[j];
    	}
    	long long int x=0,y=0,xsum=0,ysum=0,m=0,o=n-1;
    	while(m<=o)
    	{
    		if(xsum<=ysum)
    		{
    			x+=1;
    			xsum+=a[m];
    			m++;
    		}
    		else
    		{
    			y+=1;
    			ysum+=2*a[o];
    			o--;
    		}
    		
    		
    	}
    	cout<<x<<" "<<y<<"\n";
    	}

    return 0;	
}
