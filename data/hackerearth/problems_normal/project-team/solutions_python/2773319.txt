def find(inputlist,i,j,maxi,mini):
    if i>j:
        
        if(mini>10000):
            return 0
        return int(maxi-mini)
    else:
       
        temp=inputlist[i]+inputlist[j]
        if temp>maxi:
            maxi=temp
        elif temp<mini:
            mini=temp
        i=i+1
        j=j-1
        return find(inputlist,i,j,maxi,mini)
