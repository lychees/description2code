#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a[t];
    for(int i=0;i<t;i++)
    {
    	cin>>a[i];
    }
    int arr[t+1][110];
    int m=0;
    for(int x=0;x<=t;x++)
    {
    for(int l=0;l<=109;l++)
    {
    	arr[x][l]=0;
    }
    }
    for(int j=1;j<=t;j++)
    {
    	for(int k=1;k<110;k++)
    	{
    		if(k==a[m])
    		{
    			arr[j][k]=arr[j-1][k]+1;
    			m++;
    		}
    		else
    		arr[j][k]=arr[j-1][k];
    	}
    }
    int q;
    cin>>q;
    while(q--)
    {
    	int x,y,z;
    	cin>>x>>y>>z;
    	if(x==0)
    	cout<<arr[y+1][z]<<"\n";
    	else
    	cout<<arr[y+1][z]-arr[x+1][z]<<"\n";
    }
    
}
