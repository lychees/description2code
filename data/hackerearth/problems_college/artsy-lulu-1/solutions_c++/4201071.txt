#include <bits/stdc++.h>
using namespace std;

int main()
{	
	char a;
	while(cin>>a){
		int h;
		cin>>h;
		
		if(h<2){
			cout<<"INVALID"<<endl;
		}
		else if(a<='Z' && a>='A' && a+h-1>'Z'){
			cout<<"INVALID"<<endl;
		}
		else if(a<='z' && a>='a' && a+h-1>'z'){
			cout<<"INVALID"<<endl;
		}
		else{
			for(int i=0;i<h;i++){
				for(int j=0;j<i+1;j++){
					printf("%c ",a+i);
				}
				cout<<endl;
			}
		}
		cout<<endl;
	}
    return 0;
}
