#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5+10;
int pri[maxn],q,n,k,m,ans;
priority_queue<int,vector<int>,greater<int> > Q;
int main(){

    scanf("%d",&q);
    while (q--){
        scanf("%d%d",&n,&k);
        m=0;
        for (int i=2; 1LL*i*i<=1LL*k; i++)
            if (k%i==0){
                while (k%i==0){
                    pri[++m]=i;
                    k/=i;
                }
            }
        if (k>1) pri[++m]=k;
        while (!Q.empty()) Q.pop();
        for (int i=1; i<=n; i++) Q.push(1);
        for (int u,i=m; i>=1; i--){
            u=Q.top();Q.pop();
            u*=pri[i];
            Q.push(u);
        }
        ans=0;
        while (!Q.empty()){
            ans+=Q.top();
            Q.pop();
        }
        printf("%d\n",ans);
    }
}
