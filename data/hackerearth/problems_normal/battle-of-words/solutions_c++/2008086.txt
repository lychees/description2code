#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
    	string stra,strb;
    	getline(cin,stra);
    	getline(cin,strb);
    	int na,nb;
    	na=stra.size();
    	nb=strb.size();
    	int i;
    	int a[26]={0};
    	int b[26]={0};
    	for(i=0;i<na;i++)
    	{
    		a[stra[i]-'a']++;
    	}
    	for(i=0;i<nb;i++)
    	{
    		b[strb[i]-'a']++;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(a[i]==b[i])
    		{
    			a[i]=0;
    			b[i]=0;
    		}
    		else if(a[i]>b[i])
    		{
    			a[i]=a[i]-b[i];
    			b[i]=0;
    		}
    		else
    		{
    			b[i]=b[i]-a[i];
    			a[i]=0;
    		}
    	}
    	int flaga=0,flagb=0;
    	for(i=0;i<26;i++)
    	{
    		if(a[i]!=0)
    		{
    			flaga=1;
    			break;
    		}
    	}
    	for(i=0;i<26;i++)
    	{
    		if(b[i]!=0)
    		{
    			flagb=1;
    			break;
    		}
    	}
    	if(flaga==0&&flagb==1)
    	{
    		printf("You lose some.\n");
    	}
    	else if(flaga==1&&flagb==0)
    	{
    		printf("You win some.\n");
    	}
    	else if(flaga==1&&flagb==1)
    	{
    		printf("You draw some.\n");	
    	}
    }
}
