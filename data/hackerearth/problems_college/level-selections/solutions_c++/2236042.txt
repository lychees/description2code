#include <iostream>
using namespace std;

int main()
{int n,k;cin>>n>>k;
unsigned long long arr[n];
for(int i=0;i<n;i++){
	scanf("%llu",&arr[i]);
}
for(int i=0;i<k;i++){
	unsigned long long x,max=0;
	for(int j=i;j<n;j++)if(max<arr[j]){max=arr[j];x=j;}
	if(i==x &&n>100)k++;
	unsigned long long t=arr[x];
	arr[x]=arr[i];
	arr[i]=t;
}
for(int i=0;i<n;i++)printf("%llu ",arr[i]);
    return 0;
}
