#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    vector<int>v;
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	for(int i=1;i<=n;i++)
    	{
    		v.push_back(i);
    	}
    	do
    	{
    		for(int i=0;i<n;i++)
    		cout<<v[i];
    		cout<<" ";
    	}
    	while(next_permutation(v.begin(),v.end()));
    	cout<<endl;
    	v.clear();
    }
    
}
