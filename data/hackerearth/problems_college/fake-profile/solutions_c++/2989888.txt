#include <iostream>
using namespace std;

int main()
{
	string s;
    cin>>s;
    int len=s.size();
    int count=len;
    for(int i=0;i<len;i++)
    {
    		for(int j=i+1;j<len;j++)
    		{
    			if(s[i]==s[j])
    			count--;
    		}
    }
    if(count%2==0)
    cout<<"CHAT WITH HER!";
    else
    cout<<"IGNORE HIM!";
    return 0;
}
