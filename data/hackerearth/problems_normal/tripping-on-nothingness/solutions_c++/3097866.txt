#include <iostream>
#include <set>

using namespace std;

int gcd(int a, int b) {
	if(b == 0) return a;
	else return gcd(b, a % b);
}

int main()
{
    int S, Q; cin >> S >> Q;
    set<int> ans;
    
    while(Q--) {
    	int A; cin >> A;
    	int g = gcd(A, S);
    	
    	if(ans.count(g) > 0) cout << -1 << endl;
    	else {
    		cout << g << endl;
    		ans.insert(g);
    	}
    }
    return 0;
}
