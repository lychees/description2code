#include <stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
int main()
{
   int arr[1000000]={0};
   arr[1]=1;
   for(int i=2;i<=1000;i++)
   {
     for(int j=2;pow(i,j)<=1000000;j++)
   {  int pos=pow(i,j);
     arr[pos]=1;
   }}
   int t,n,flag=0;
   cin>>t;
   while(t--)
   {flag=0;
     cin>>n;
     
     
     for(int i=1;i<=n/2;i++)
     {
     if(arr[i]&&arr[n-i])
     {cout<<"Yes"<<"\n";
     flag=1;
     break;
     }
     
     }
     if(flag==0)
     cout<<"No"<<"\n";
   }
   
}
