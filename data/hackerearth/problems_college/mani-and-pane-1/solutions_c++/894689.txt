#define N 1000002

LL c[N];

inline LL mod(LL x){
    if(x < 0) return -x;
    return x;
}

inline void Solve(){
    LL x, y, i;
    sld(x); sld(y);
    LL d = mod(x - y), ans = 0;
    for(i = 1; i <= 100; i++){
        if(d % i == 0) ans += c[i];
    }
    LL den = ( 100 * 99 * 98 ) / 6;
    LD out = (LD)ans / (LD)den;
    printf("%0.6LF\n", out);
    return;
}

inline LL gcd(LL x, LL y){
    if(y > x){
        return gcd(y, x);
    }
    if(y == 0) return x;
    return gcd(y, x % y);
}

inline void Pre(){
    LL i, j, k, g;
    for(i = 1; i <= 100; i++){
        c[i] = 0;
    }
    for(i = 1; i <= 100; i++){
        for(j = i + 1; j <= 100; j++){
            for(k = j + 1; k <= 100; k++){
                g = gcd(i, gcd(j, k));
                c[g]++;
            }
        }
    }
}
