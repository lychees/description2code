#include<stdio.h>
#include<iostream>
#define max 10000001
#define pb(i) push_back(i)
#define lli long long int
int mini(int a,int b){return a<b?a:b;}
int maxi(int a,int b){return a>b?a:b;}
using namespace std;
int main()
{
 int t;
 scanf("%d",&t);
 while(t--){
 int n,i,j,k;int x[1009],y[1009],minx=max,maxx=-max,miny=max,maxy=-max;
 scanf("%d",&n);
 for(i=0;i<n;i++){
 	scanf("%d %d",&x[i],&y[i]);
 minx=mini(minx,x[i]);maxx=maxi(maxx,x[i]); miny=mini(miny,y[i]);maxy=maxi(maxy,y[i]);

 }
 if(maxx==minx&&maxy==miny)
 	printf("0\n");
 else if(maxx==minx)
 printf("%d\n",maxy-miny);
 else if(miny==maxy)
 printf("%d\n",maxx-minx);
 else
 printf("%d\n",(maxx-minx)*(maxy-miny));
 }
 
 return 0;
}
