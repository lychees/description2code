#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char ch[500];
    int k,n,l;
    cin>>ch>>n;
   	l=strlen(ch);
   	if(l>n)
   	{
   		k=l;
   		l=l-n;
   		cout<<l<<"\n";
   		for(int i=0;i<k;i++)
   		{
   			cout<<ch[i];
   		}
   	}
   	else{
   		cout<<"0"<<endl;
   		for(int i=0;i<k;i++)
   		{
   			cout<<ch[i];
   		}
   	}
    return 0;
}
