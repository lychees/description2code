#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a[1001]={};
    int i,j,n,res;
    a[0]=a[1]=1;
    for(i=2;i<=sqrt(1000);i++)
    {
        if(a[i]==0)
        {
            for(j=2*i;j<=1000;j=j+i)
                a[j]=1;
        }
    }
    int t;
    cin>>t;
    while(t--)
    {
        res=0;
        cin>>n;
        while(n--)
        {
            cin>>i;
            if(a[i]==0)
                res++;
        }
        cout<<res<<endl;
    }
    return 0;
}
