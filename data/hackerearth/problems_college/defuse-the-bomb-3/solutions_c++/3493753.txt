#include <bits/stdc++.h>
#define ll long long int

using namespace std;

void swap(int *a, int *b) {
    int t = *a;
    *a = *b;
    *b = t;
}

int main()
{
    int t;
    ll a, b;

    cin >> t;
    while (t--) {
        cin >> a >> b;
        int lb = 0;
        int la = 0;
        ll tmpa = a;
        ll tmpb = b;
        vector<int> va;

        while (tmpa) {
            va.push_back(tmpa % 10);
            tmpa /= 10;
            la++;
        }
        while (tmpb) {
            tmpb /= 10;
            lb++;
        }

        reverse(va.begin(), va.end());
        vector<ll> v;
        for (int j = la - 1; j - lb + 1 >= 0; j--) {
            ll sum = 0;
            int k = j;
            int tmp = lb;
            int u = 0;
            while (tmp > 0) {
                sum += pow(10, u) * va[k];
                k--;
                tmp--;
                u++;
            }
            v.push_back(sum * b);
        }

        for (int i = v.size() - 1; i >= 0; i--) {
            cout << v[i] << endl;
        }
        v.clear();
        va.clear();
    }
    return 0;
}
