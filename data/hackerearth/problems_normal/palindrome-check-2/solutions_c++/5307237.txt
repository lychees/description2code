#include <iostream>
using namespace std;

int main()
{

	char S[100];
	static int count=0;
	int i=0;
	bool isPal = false;

	cin>>S;

	while (S[i] != '\0') {
		count++;
		i++;
	}
	for(i=0; i < count, count>=0; i++, count--)
	{
		if(S[i] == S[count-1])
		{
			isPal = true;
		}
		else{
			isPal = false;
			break;
		}
	}

	if(isPal)
		cout<<"YES";
	else
		cout<<"NO";
    return 0;
}
