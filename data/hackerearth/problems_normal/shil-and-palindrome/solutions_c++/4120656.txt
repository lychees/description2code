#include <iostream>
using namespace std;

int main()
{
	char s[100005];
	int a[27]={0},i,n,x=0,j,k;
	cin>>s;
	for(n=0;s[n]!='\0';n++)
	{
		a[s[n]-'a']++;
	}
	//cout<<n<<endl;
	for(i=0;i<27;i++)
	{
		//cout<<a[i]<<" ";
		if(a[i]%2!=0)
			x++;
	}
	//cout<<endl<<x<<endl;
	if(n%2==0)
	{
		if(x)
			cout<<"-1"<<endl;
		else
		{
			j=0;k=n-1;
			for(i=0;i<27;i++)
			{
				while(a[i])
				{
					s[j]='a'+i;
					s[k]='a'+i;
					j++;
					k--;
					a[i]-=2;
				}
			}
			cout<<s<<endl;
		}
	}
	else
	{
		if(x>1)
			cout<<"-1"<<endl;
		else
		{
			for(i=0;i<27;i++)
			{
				if(a[i]%2!=0)
				{
					x=i;
					break;
				}
			}
			//cout<<x<<endl;
			a[x]--;
			j=0;k=n-1;
			for(i=0;i<27;i++)
			{
				while(a[i])
				{
					s[j]='a'+i;
					s[k]='a'+i;
					j++;
					k--;
					a[i]-=2;
				}
			}
			s[k]='a'+x;
			cout<<s<<endl;
		}
	}
    return 0;
}
