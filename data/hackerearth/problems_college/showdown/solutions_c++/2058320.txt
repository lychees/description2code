#include <bits/stdc++.h>

using namespace std;

int A[2002];
int B[2002];
int dp[2002][2002];
bool vis[2002][2002];
int n;

int f(int idx1, int idx2)
{
	if ( idx1 == n || idx2 == n ) return 0;
	if ( vis[idx1][idx2] ) return dp[idx1][idx2];
	vis[idx1][idx2] = true;
	int ans = f(idx1+1,idx2);
	if ( A[idx1] > B[idx2] ) ans = max(ans, 100 + f(idx1+1,idx2+1));
	else if ( A[idx1] == B[idx2] ) ans = max(ans, 50 + f(idx1+1,idx2+1));
	dp[idx1][idx2] = ans;
	return ans;
}

int main()
{
	cin >> n;
	for ( int i = 0; i < n; i++ ) cin >> A[i];
	for ( int i = 0; i < n; i++ ) cin >> B[i];
	sort(A, A+n);
	sort(B, B+n);
	int ans = f(0,0);
	cout << ans << endl;
	return 0;
}
