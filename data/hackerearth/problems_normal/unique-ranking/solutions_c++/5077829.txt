#include <iostream>
#include <algorithm>
#include <vector>
#include <climits>
using namespace std;
vector<int>si[21];
bool vis[10001];
vector<int>::iterator it;
int n;
void ini()
{
	for(int i=0;i<n;i++)
	vis[i]=false;
}
int main()
{
    int k, a;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<k;j++)
    	{
    		cin>>a;
    		si[j].push_back(a);
    	}
    }
    int q;
    cin>>q;
    while(q--)
    {	
    	ini();
    	int c,b,minm,cnt,d,ind=0;
    	cin>>c;
    	while(c--)
    	{
    		cin>>b;
    		b--;
    		minm=INT_MAX;
    		for(int i=0;i<n;i++)
    		{	
    			if(vis[i]==false)
    			if(si[b][i]<minm)
    			{
    				minm=si[b][i];
    				ind=i;
    			}
    		}
    		cnt=count(si[b].begin(),si[b].end(),minm);
    		if(cnt!=1)
    		{
    			for(int i=0;i<n;i++)
    			{
    				if(si[b][i]!=minm)
    				vis[i]=true;
    				//else
    				//vis[i]=false;
    			}
    			
    			
    		}
    		if(c==0)
    		{
    			cout<<(ind+1)<<"\n";
    			break;
    		}
    		else
    		if(cnt==1)
    		{
    			while(c--)
    			cin>>d;
    			cout<<(ind+1)<<"\n";
    			break;
    		}
    	}
    }
    return 0;
}
