#include<iostream>
using namespace std;
main()
{
    int t,d,wt,tot;
    cin>>t;
    while(t--)
    {
        cin>>d;
        cin>>wt;
        if(d>60)
        {
            cout<<"NA"<<endl;
            continue;
        }
        if(d<=2)
        {
        	tot=50+wt*2;
        	cout<<tot<<endl;
        	continue;
        }
        tot=50+(d-2)*12+wt*2;
        cout<<tot<<endl;
    }
}
