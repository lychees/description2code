#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,m;
    	cin>>n>>m;
    	char a[n][m];
    	for(int i=0;i<n;i++){
    		for(int j=0;j<m;j++){
    			cin>>a[i][j];
    		}
    	}
    	int count=0;
    	
    	for(int i=0;i<n-1;i++){
    		for(int j=0;j<m-1;j++){
    			if(a[i][j]=='\/' && a[i][j+1]=='\\' && a[i+1][j]=='\\'&&a[i+1][j+1]=='\/')
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
