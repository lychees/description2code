n = int(raw_input())
max5 = [(-10**9-11, 0, 0) for _ in range(5)]
for _ in range(n):
    di, z, p, l, c, s = map(int, raw_input().split())
    nz = p * 50 + l * 5 + c * 10 + s * 20
    dz = nz - z
    min5 = min(max5)
    if (dz, di, nz) > min5:
        i = max5.index(min5)
        max5[i] = (dz, di, nz)
for x, y, z in sorted(max5, reverse=1):
    print y, z
