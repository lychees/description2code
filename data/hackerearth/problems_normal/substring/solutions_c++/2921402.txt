#include <iostream>
using namespace std;

int main()
{
    // cout << "Hello World!" << endl;
   int t,i,count;
   long long s;
   cin>>t;
   char a[100005];
   while(t--){
   		cin>>a;
   		count=s=0;
   		for(i=0;a[i]!='\0';++i){
   			if(a[i]=='A'||a[i]=='a'||a[i]=='b'||a[i]=='B'||a[i]=='C'||a[i]=='c')count++;
   			else {s+=(count*(count+1))/2;count=0;}
   		}
   		if(count!=0) s+=(count*(count+1))/2;
   		cout<<s<<"\n";
   }
   return 0;
}
