int main() {
	int T;
	scanf("%d", &T);
	rep(ii, T) {
		unsigned A, B;
		scanf("%u%u", &A, &B);
		unsigned ans = 0;
		bool lt = false;
		for(int i = 31; i >= 0; -- i) {
			int a = A >> i & 1, b = B >> i & 1;
//			cerr << i << ": " << a << ", " << b << ", " << lt << endl;
			if(!lt && a == 1 && b == 1)
				ans |= 1U << i;
			lt |= a < b;
		}
		printf("%u\n", ans);
	}
	return 0;
}
