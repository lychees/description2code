#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str1,str2;
		cin>>str1>>str2;
		string str3 = str1 + str2;
		string str = str3;
		reverse(str3.begin(),str3.end());
		if(str3 == str)
		{
			printf("YES\n");
			continue;
		}
		str3 = str2 + str1;
		str = str3;
		reverse(str3.begin(),str3.end());
		if(str3 == str)
		{
			printf("YES\n");
			continue;
		}
		cout<<"NO"<<"\n";
	}
}
