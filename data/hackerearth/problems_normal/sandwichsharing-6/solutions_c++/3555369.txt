#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false) ; 
    int t;
    cin >> t ;
    while(t--){
    	long long int n,p ; 
    	cin >> n >> p ; 
    	while(n != 3 && p > 0){
    		if(n%2 == 0 )n =n/2 +1 ;
    		else {
    			n = n/2  + 2 ; 
    		}
    		p--; 
    	}
    	cout << n << endl;
    }
    return 0;
}
