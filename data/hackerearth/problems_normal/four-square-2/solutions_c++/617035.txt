#include <bits/stdc++.h>
using namespace std;
main()
{
    int a; scanf("%d", &a);
    for (int g=0;g<a;g++)
    {
        long long b;scanf("%lld", &b);//ans=0; 
        long long ans=0; 
        for (int y=1; y<=sqrt(b); y++)
        {if (b%y) continue;
        	if (y==sqrt(b))
        	{
        		if (y%4)
        		{
        			ans+=y; 
        		}
        	}
        	else
        	{
        		if (y%4) ans+=y;
        		if ((b/y)%4) ans+=b/y; 
        	}
        }
        printf("%lld\n", 8*ans);
    }
    return 0; 
}
