#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */  
    long p,q,l,r,d,temp;long long sqr;int f=0;
    cin>>p>>q;
    for(;p<=q;p++)
        {
        sqr=p*p;d=1;temp=p;
        while(temp=temp/10)d++;
        l=sqr/pow(10,d);
        r=sqr-l*pow(10,d);
        if(l+r==p){cout<<p<<" ";f++;}
        
    
    }
    if(!f)cout<<"INVALID RANGE";
    return 0;
}
