    #include<iostream>
    using namespace std;
    int main(){
    	string s;
    	cin>>s;
    	long long int a,b;
    	int t,l;
    	l=s.length();
    	cin>>t;
    	while(t--){
    		cin>>a>>b;
    		if(a>l){
    		a=a%l;
    		if(a==0)
    		a=l;
    		}
    		if(b>l){
    		b=b%l;
    		if(b==0)
    		b=l;
    		}
    		if(s[a-1]==s[b-1])
    		cout<<"Yes";
    		else
    		cout<<"No";
    		cout<<endl;
    	}
    	return 0;
    }
