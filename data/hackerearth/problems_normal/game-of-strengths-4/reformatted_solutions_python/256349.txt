T = int(raw_input())
for t in range(T):
    N = int(raw_input())
    inp = raw_input().split(' ')
    strengths = []
    for n in range(N):
        strengths.append(int(inp[n]))

    strengths.sort()
    sumofstrengths = 0
    total = 0
    for num in strengths:
        total = total + num

    for i in range(N):
        total = total - strengths[i]
        sumofstrengths = sumofstrengths + total - ((N-i-1) * strengths[i])

    maxs = strengths.pop()
    absolutepower = sumofstrengths * maxs
    print absolutepower % 1000000007
