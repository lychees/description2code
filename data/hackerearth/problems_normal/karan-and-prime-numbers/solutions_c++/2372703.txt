#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t, a,b,sum,i,j,flag;
    cin>>t;
    while(t--)
    { sum=0;
    	scanf("%d%d",&a,&b);
    	if(a==1)
    	sum=-1;
    	for(i=a;i<=b;i++)
    	{flag=0;
    		for(j=2;j<=sqrt(i);j++)
    		{
    			if(i%j==0)
    			{ flag++; break;}
    		}
    		if(!flag)
    		 sum+=i; 
    	}
    	printf("%d\n",sum);
    	
    }
    return 0;
}
