

#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n ,res = 0, sum = 0, sum1;

    cin >> n;

    for(int i = 0; i < n; i++){

        cin >> sum1;

        if(sum <= 0){               

            if(sum1 < 0) continue;      
            sum = sum1;                 

        } else sum += sum1;             

        res = max(res , sum);           
    }

    cout << res << "\n";

    return 0;
}
