#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
    long long s,n,i,davy,will;
    int t;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	s = 0;
    	long long a[n],dp[n+2];
    	dp[n] = dp[n+1] = 0;
    	for(i=0;i<n;i++)cin>>a[i];
    	
    	for(i=n-1;i>=0;i--)
    	{
    		s+=a[i];
    		dp[i] = max(s-dp[i+1],s-dp[i+2]);
    	}
    	davy = dp[0];
    	will = s-dp[0];
    	
    	if(davy==will)
    	{
    		cout<<"DRAW";
    	}
    	else if(davy>will)
    	{
    		cout<<"LOSE";
    	}
    	else
    	{
    		cout<<"WIN";
    	}
    	cout<<"\n";
    }
    
    return 0;
}
