#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n];
    	for(int i=0;i<n;i++)	cin>>a[i];
    	sort(a,a+n);
    	int mini=INT_MAX;
		for(int i=0;i<n;i++)
		{
			int	l=i-1,h=i+1,temp=k-1,res=0,tot=0,tot1=0;
			while(temp!=0)
			{
				if(l>=0)	tot=(3*(a[i]-a[l]));
				else	tot=INT_MAX;
				if(h<n)	tot1=(5*(a[h]-a[i]));
				else	tot1=INT_MAX;
				if(tot<tot1)
				{
					res+=tot;
					temp--;
					l--;
				}
				else
				{
					res+=tot1;
					h++;
					temp--;
				}
			}
			if(res<mini)	mini=res;
		}

    	cout<<mini<<endl;
    }
    return 0;
}
