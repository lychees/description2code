#include<bits/stdc++.h>
using namespace std;
double a[51][10001],dice[51][10001];
int main()
{
	int T,count,i,j,t;
	int M,N,K;
	long long int dem;
	
	double result;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d%d%d",&M,&N,&K);
		for(i=0;i<51;i++)
		{
			for(j=0;j<10001;j++)
			{
				a[i][j] =0;
				dice[i][j] = 0;
			}
		}
		for (int j = 1; j <= N && j <= K; j++)
            dice[1][j] = (double)1/N;
            
        int i,j;
	    for (i = 2; i <= M; i++)  //m throws
	    {
	        for (j = 1; j <= K; j++)
	        {
	             for (int k = 1; k <= N && k < j; k++)
	             {
	                    dice[i][j] = (dice [i][j]+dice[i-1][j-k]);
	             }
	             dice[i][j] /= N;
	        }
	    }  
	    result = dice[M][K];
		count = 0;
		if(result != 0)
		{
			while(result < 1 )
			{
				result*= 10;
				count++;
			}
		}
			printf("%.3f %d\n",result,count);     
	}
	return 0;
}
