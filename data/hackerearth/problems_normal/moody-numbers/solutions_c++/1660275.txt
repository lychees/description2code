
#include<bits/stdc++.h>
using namespace std;

vector<int> arr(200,-1);
int f(long long n);

int digSum(long long n)
{
    int sum=0,temp=0;
    while(n)
    {
        temp=n%10;
        sum+=temp;
        n=n/10;
    }

    return sum;
}
int main()
{
    int t,i;
    long long n;

    arr[1]=1;
    arr[2]=1;
    arr[3]=0;
    arr[4]=1;

    for(i=5;i<=163;i++)
        if(arr[i]==-1)
            arr[i]=f(i);


    /*for(i=1;i<=20;i++)
    {
        cout<<i<<","<<arr[i]<<endl;
    }*/
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        if(n<163)
        {
            if(arr[n])
                printf("YES\n");
            else
                printf("NO\n");
            continue;
        }
        n=n*n;
        n=digSum(n);

        if(arr[n])
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}

int f(long long n)
{
    //cout<<"n="<<n<<endl;
    if((n==1)||(n==4))
        return arr[n]=1;

    if(arr[n]!=-1)
        return arr[n];

    int prev=n;
    n*=n;
    n=digSum(n);

    if((n==1)||(n==4))
        return arr[prev]=1;

    //cout<<"prev="<<prev<<",n="<<n<<endl;
    if(n<=prev)
    {
        if(arr[n]==1)
            return arr[prev]=arr[n];
        return arr[prev]=0;
    }

    return arr[prev]=f(n);
}
