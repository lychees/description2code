#include <iostream>
using namespace std;

int main()
{
 	int n;
 	cin>>n;
 	int arr[n][n];
 	long long s1,s2;
 	s1=0,s2=0;
 	for (int i = 0; i < n; i++) {
 		for (int j = 0; j < n; j++) {
 			cin>>arr[i][j];
 			if(i == j )
 				s1 += arr[i][j];
 			if( i+j == n-1)
 				s2 += arr[i][j];
 		}
 	}
    cout << abs(s1-s2) << endl;
    return 0;
}
