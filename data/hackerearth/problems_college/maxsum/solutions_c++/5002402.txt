#include <iostream>
using namespace std;
int a[10005];
int main()
{
    int t,n;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	for(int i=0;i<n;++i) cin >> a[i];
    	int old,inc,exc;
    	inc = a[0];
    	exc = 0;
    	for(int i=1;i<n;++i)
    	{
    		old = inc;
    		inc = max(inc,exc + a[i]);
    		exc = old;
    	}
    	cout << max(inc,exc) << endl;
    }
    return 0;
}
