#include <iostream>
using namespace std;
#define ll long long
long  long fa(long long a)
{
	ll i, j ,k ,cnt=0;
	
	for(i=1;i*i<=a;i++)
	{
		if(a%i==0)
		{
			if(i%4){cnt+=i;}
			
			if(i*i!=a)
			 {
			 	if((a/i)%4)
			 	 {cnt+=(a/i);}
			 }
		}
	}
	//if(a%4)cnt++; 
	return cnt;
}
