#include <iostream>
using namespace std;

int print(char *string, int start,int end)
{
	char temp;
	if(start==end)
	{
		cout<<string<<endl;
		return 0;
	}
	for(int j=start;j<=end;j++)
	{
		temp=*(string+start);
		*(string+start)=*(string+j);
		*(string+j)=temp;
		print(string,start+1,end);
		temp=*(string+start);
		*(string+start)=*(string+j);
		*(string+j)=temp;
	}
	return 0;
}

int main()
{
    char string[102];
    int n,start,end;
    cin>>string;
    cin>>n;
    while(n--)
    {
    	cin>>start>>end;
    	print(string,start,end);
    	cout<<endl;
    }
    return 0;
}
