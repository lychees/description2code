#include <iostream>
using namespace std;

int main()
{   
    int t;
    cin>>t;
    while(t--)
    {
    	int l,u;
    	cin>>l>>u;
    	int r=0;

	    for(int i=l;i<=u;i++)
	    {
	        for(int j=i;j<=u;j++)
	        {
	           r=max(r,i^j);
	        }
	    }
	    cout<<r<<endl;
    }
    return 0;
    
}
