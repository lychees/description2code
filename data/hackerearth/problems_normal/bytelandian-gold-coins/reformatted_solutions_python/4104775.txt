from functools import wraps
for _ in xrange(10):
    try:
        n = int(raw_input())
    except Exception:
        break

    def memo(f):
        cache = {}

        @wraps(f)
        def wrap(*args):
            if args not in cache:
                cache[args] = f(*args)
            return cache[args]
        return wrap

    @memo
    def sol(n):
        if n < 2:
            return n
        else:
            return max(n, sol(n//2)+sol(n//3)+sol(n//4))
    print sol(n)
