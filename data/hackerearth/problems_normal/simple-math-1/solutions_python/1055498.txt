import sys
n,r=map(int,sys.stdin.readline().split());
def fac(n):
	if(n<=1):
		return 1
	else:
		return n*fac(n-1)
if(n>=r):
	print (fac(n-1)/(fac(n-r)*fac(r-1)))%10000007
else:
	print -1
