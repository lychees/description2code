#include <iostream>
using namespace std;

void divisors(int N) {
    int count = 0;
    for( int i = 1;i * i <=N;++i ) {
         if( N % i == 0) {
         if( i * i == N )
                     count++;
                 else                                                     
                     count += 2;
      }
    }
    if(count<4)
    {
    	cout<<"NO"<<endl;
    }
    else
    {
    	cout<<"YES"<<endl;
    }
}

int main()
{
    int n,x,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	divisors(x);
    }
    
    return 0;
}
