#include<iostream>
using namespace std;
int main()
{
	int t,i,j,flag;
	cin>>t;
	string s;
	while(t--)
	{
		flag=0;
		cin>>s;
		int len = s.size();
		int a[len];
		for(i=0;i<len;i++)
		{
			a[i]=s[i]-48;
		}
		
		
		for(i=0;i<=len/2;i++)
		{
			if(a[i]==0||a[i]==1||a[i]==8)
			{
				if(a[i]!=a[len-i-1])
				{
					flag=1;
					break;
				}
			}
			else if(a[i]!=0&&a[i]!=1&&a[i]!=8)
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		{
			cout<<"NO"<<endl;
		}
		else
		{
			cout<<"YES"<<endl;
		}
	}
}
