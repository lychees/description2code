#include <iostream>
using namespace std;
int main()
{
    int t,N,R,D,x,v,ans;
    cin>>t;
    while(t--)
    {
        cin>>N>>D>>R;
        x=(N%D)*10;
        while(R--)
        {
           ans=(x/D);
           v=(x%D);
           x=v*10;
        }
        cout<<ans<<endl;
    }
    return 0;
}
