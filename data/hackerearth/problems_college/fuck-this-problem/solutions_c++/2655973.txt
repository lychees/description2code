#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    char str[100000];
    int n;
    gets(str);
    cin>>n;
    //cin>>n;
    //cout<<str<<endl;
    //cout<<n;
    int count = 0;
    for(int i=0;i<strlen(str);i++)
    {
    	if(str[i] == '\0') break;
    	cout<<str[i];
    	count++;
    	if(count%n == 0)	cout<<"\n";
    }
    return 0;
}
