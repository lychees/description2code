#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
    
    int n,l,j,sum;
    
    string s;
    
    scanf("%d", &n);
    
    for (int i = 0; i < n; i++) {
    	cin >> s;
    	l = s.size();
    	sum = 0;
    	int a[l];
    	
    	for (j = 0; j < l; j++) {
    		if (s[j] == 'A' || s[j] == 'a' || s[j] == 'B' || s[j] == 'b' || s[j] == 'C' || s[j] == 'c') {
    			a[j] = 1;
    		} else {
    			a[j] = 0;
    		}
    	}
    	sum += a[0];
    	for (j = 1; j < l; j++) {
    		if (s[j] == 'A' || s[j] == 'a' || s[j] == 'B' || s[j] == 'b' || s[j] == 'C' || s[j] == 'c') {
    			a[j] = a[j] + a[j-1];
    			sum += a[j];
    		}
    	}
    	
    	printf("%d\n", sum);
    	
    	
    }
    
    return 0;
}
