#include<bits/stdc++.h>
using namespace std;

#define LL long long

int mod = 1e9 + 7;
int n,k,inp[102];
int answ;
LL totsum;

int dp[100005][102];

int solve(int sum,int pos)
{
	if(sum < 0)
		return 0;
	if(pos == -1 && sum == 0)
		return 1;
	else if(pos == -1 && sum != 0)
		return 0;

	int &ret = dp[sum][pos];		
	if(ret != -1)
		return ret;

	return ret = (solve(sum,pos-1) + solve(sum-inp[pos],pos-1))%mod;
}

int main()
{
	scanf("%d %d",&n,&k);
	totsum = 0;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&inp[i]);
		totsum += inp[i];
	}

	answ = 0;
	memset(dp,-1,sizeof(dp));
	for(int i=0;i<=k-1;i++)
	{
		if(i > totsum/2)
			break;
		answ = (answ + solve(i,n-1))%mod;
		if(i != totsum - i)	
			answ = (answ + solve(i,n-1))%mod;
	}
	
	int tot = 1;
    	for (int i = 0; i < n; i++) 
        	tot = (2 * tot)%mod;

	int ans = (tot - answ + mod)%mod;
	cout << ans << endl;
     
	return 0;
}
