#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[3009];
int main()
{
	ll n,k;
	cin>>n;
	for(ll i=1;i<=n;++i)
	cin>>a[i];
	sort(a+1,a+n+1);
	bool flag=false;
	cin>>k;
	for(ll  i=1;i<=n;++i)
	{
		ll  start=i+1;
		ll  end=n;
		while(start<end)
		{
			ll  tempsum=a[start]+a[end];
			if(k-a[i]>tempsum)
			{
				++start;
			}
			else if(k-a[i]<tempsum)
			{
				--end;
			}
			else if(k-a[i]==tempsum) 
			{
				printf("YES");
				return 0;
			}
		}
	}
	puts("NO");
	return 0;
	
}
