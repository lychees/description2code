#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
lli a[1000005];
void sieve()
{
	lli i,j;
	a[1]=1;
	a[0]=1;
	for(i=2;i<=1000000;i++)
	{
		if(!a[i])
		{
			a[i]=i;
			for(j=i*i;j<=1000000;j+=i)
			{
				if(a[j]==0)
				{
					a[j]=i;
				}
			}
		}
	}
}
int main()
{
    ios_base::sync_with_stdio(false);
    sieve();
    lli n,x;
    scanf("%lld",&n);
    while(n--)
    {
    	scanf("%lld",&x);
    	printf("%lld\n",x-a[x]);
    }
    return 0;
}
