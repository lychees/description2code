#include<cstdio>
#include<math.h>
using namespace std;
int main()
{
	int ntc; scanf("%d",&ntc);
	while(ntc--)
	{
		
		long long N; scanf("%lld",&N);
		N = N*2;
	    long long ans = sqrt(N);
	    long long tmp = ans * (ans +1 );
	    while(tmp < N) 
	    {
	    	ans++;
	    	 tmp = ans * (ans +1 );
	    //	printf("ans =  %lld and tmp = %lld  N = %lld   res = %d\n",ans,tmp,N,tmp <= N);
	    	
	    }
	
	    printf("%lld\n",ans);
	}
}
