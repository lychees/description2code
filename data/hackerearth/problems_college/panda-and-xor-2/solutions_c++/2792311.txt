#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int t;cin>>t;
    while(t--)
    {
    	int a,b,count=0;cin>>a>>b;
    	int x=a^b;
    	while(x)
    	{
    		if(x&1)
    			count++;
    		x>>=1;
    	}
    	if(count&1)
    		cout<<"NO\n";
    	else
    		cout<<"YES\n";
    }
    return 0;
}
