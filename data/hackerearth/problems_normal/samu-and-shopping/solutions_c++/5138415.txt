#define MAXN 100005
#define MDL 1000000007
 
void fast_io() { cin.tie(0); ios::sync_with_stdio(false); }
 
ll powmod(ll a,ll b) {ll res=1;a%=MDL;for(;b;b>>=1){if(b&1)res=res*a%MDL;a=a*a%MDL;}return res;}
ll gcd(ll a,ll b) {return b==0?a:gcd(b,a%b);}
ll lcm(ll a,ll b) {return a*(b/gcd(a,b));}
 
int main() {
fast_io();
int T=1;
r(T);

 int n,x[MAXN],y[MAXN],z[MAXN],dp[MAXN][3];

 FOR(tt,0,T)
 {
  r(n);
  reset(dp);
  FOR(i,1,n+1)
  {
    r(x[i]);r(y[i]);r(z[i]);
  }
  FOR(i,1,n+1)
  {
    dp[i][0]=x[i]+min(dp[i-1][1],dp[i-1][2]);
    dp[i][1]=y[i]+min(dp[i-1][0],dp[i-1][2]);
    dp[i][2]=z[i]+min(dp[i-1][0],dp[i-1][1]);
  }
  wl(min(dp[n][0],min(dp[n][1],dp[n][2])));

  cerr << "case " << (tt+1) << " " << clock() << " ms" << endl;
 }
 
 return 0;
}
