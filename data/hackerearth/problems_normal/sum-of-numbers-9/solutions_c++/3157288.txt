#include<bits/stdc++.h>
using namespace std;
#define ll long long int
vector<ll> arr(20);
int main()
{
    ll test,i,j,num,ans,a,b,given_num;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld",&num);
        ll total=(1<<num);
        ans=0;
        vector<ll> sum(total,0);
        for(ll i=0;i<num;i++)
            scanf("%lld",&arr[i]);
        scanf("%lld",&given_num);
        bool flag=false;
        for(ll i=0;i<(total);i++)
        {
            for(ll j=0;j<num;j++)
            {
                if((i)&(1<<j))
                   sum[i]+=arr[j];
            }
            if(sum[i]==given_num)
            {
                flag=true;
                break;
            }
        }
        if(flag)
            printf("YES\n");
        else
            printf("NO\n");
    }
}
