#include <iostream>
#include <string>

using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);

	int T, M;
	string A;

	cin >> T;

	while (T--)
	{
		cin >> A >> M;
		
		long long int N = 0;

		for (int i = 0; i < A.length(); i++)
		{
			N = 10*N + (A[i] - '0');
			N = N % M;
		}

		cout << N << endl;
	}

	return 0;
}
