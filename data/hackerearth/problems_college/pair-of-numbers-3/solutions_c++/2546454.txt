#include <bits/stdc++.h>
using namespace std;
typedef unsigned int uint;

int main()
{
   uint t;
   cin>>t;
   ++t;
   
   while(--t)
   {
   	uint n,ans=0;
   	cin>>n;
   	for(uint i=2;i!=n;++i)
   	{
   		for(uint j=i+1;j!=(n+1);++j)
   		{
   			if(i*j%(i+j)==0)
   			  ++ans;
   		}
   	}
   	cout<<ans<<endl;
   }
   return 0;
}
