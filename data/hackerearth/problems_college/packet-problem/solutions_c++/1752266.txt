#include<iostream>
#include<map>
using namespace std;
map<long int,long int>A;
long int fibo(long int n){
	if(A[n]!=0)
	return A[n];
	else if(n==0)
	return 0;
	else if(n==1)
	return 1;
	else{
	 long int s=fibo(n-1)+fibo(n-2);
	 A[n]=s;
	 return s;}
}
main(){
	std::ios_base::sync_with_stdio(false);
	long int t,m;
	A[0]=0;A[1]=1;
	cin>>t;
	while(t--){
		cin>>m;
	cout<<fibo(m-1)<<" "<<fibo(m)<<endl;
         }
}
