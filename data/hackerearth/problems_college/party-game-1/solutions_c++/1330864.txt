    #include <iostream>
    #include <algorithm>
    using namespace std;
    int main(){
    int a, b, dish[10000], testCase;
    int n;
    cin >> testCase;
    while(testCase--){
    cin >> a >> b >> n;
    for(int i = 0; i < n; i++)
    cin >> dish[i];
    sort(dish, dish+n);
    dish[n] = 1;
    int kCalA = 0, kCalB = 0, i = 0, countA = 0, countB = 0;
    while(i < n){
    if(kCalA+dish[i] <= a){
    kCalA += dish[i];
    countA++;
    }
    if(kCalB+dish[i] <= b){
    kCalB += dish[i];
    countB++;
    }
    i++;
    }
    if(countA > countB)
    cout << "Chef Won\n";
    else if(countA < countB)
    cout << "Jhool Won\n";
    else
    cout << "Tie\n";
    }
    return 0;
    }
