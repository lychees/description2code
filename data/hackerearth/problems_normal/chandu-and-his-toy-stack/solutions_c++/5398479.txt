#include <iostream>
#include <stdio.h>
#include<algorithm>
using namespace std;
int main()
{
   long long int N,X,Y,T;

   scanf("%d",&T);
   while(T--)
   {
   scanf(" %lld %lld %lld",&N,&X,&Y);
   long long int a[N],b[N];
   for(long long int i=0;i<N;i++)
   scanf("%lld %lld",&a[i],&b[i]);

   sort(a,a+N);
   sort(b,b+N);
   long long int sum=0;
   for(long long int k=0;k<N;k++)
   {  if(a[k]>b[k])
       {
       sum=sum+(a[k]-b[k])*Y;
       }
       else if(b[k]>a[k])
       {  
         sum=sum+(b[k]-a[k])*X;
        }
       
   }
   cout<<sum<<endl;
   }

    return 0;
}
