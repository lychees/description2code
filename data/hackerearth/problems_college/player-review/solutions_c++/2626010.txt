#include <iostream>
using namespace std;

int main()
{
    
    long long int n;
    cin>>n;
    int a[n],i,sum=0,res=0;
    for(i=0;i<n;i++){
    	cin>>a[i];
    }
    for(i=0;i<n;i++)
    {
    	sum = sum +a[i];
    	if(sum<0)
    	{
    		sum =0;
    	}
    	res = max(res,sum);
    }
    cout<<res<<"\n";
    return 0;
}
