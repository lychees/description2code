#include <iostream>
#include <cstdio>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while (t--) {
		int n;
		scanf("%d",&n);
		long long int a[n];
		long long int l[n];
		for (int i = 0; i < n; i++) {
			scanf("%lld",&a[i]);
		}
		for (int i = 0; i < n; i++) {
			scanf("%lld",&l[i]);
		}
		for (int i = 0; i < n-1; i++) {
			for (int j = i+1; j < n; j++) {
				if (a[j] < a[i]) {
					long long int temp = a[i];
					a[i] = a[j];
					a[j] = temp;
					temp = l[i];
					l[i] = l[j];
					l[j] = temp;
				}
			}
		}
		long long int le = 1;
		long long int r = n-1;
		long long int energy = 0;
		int i = 0;
		while (i != n-1) {
			long long int temp = a[i]+l[i];
			temp = a[i+1]-temp;
			long long int temp1 = temp*r;
			temp = temp*le;
			if (temp <= temp1) {
				le++;
				energy += temp;
				r--;
			}
			else {
				r--;
				energy += temp1;
				le++;
			}
			i++;
		}
		printf("%lld\n",energy);
	}
	return 0;
}
