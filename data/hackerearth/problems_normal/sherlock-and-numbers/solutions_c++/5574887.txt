#include <iostream>
typedef long long ll;
 ll a[1000000],n,k,p,t,i;
using namespace std;
int main()
{
	cin>>t;
	while(t--){
		cin>>n>>k>>p;
		for(i=1;i<=k;i++){
			cin>>a[i];
			if(a[i]<=p) p++;
		}
		
		if(p<=n)
		cout<<p<< endl;
		else 
		cout<<"-1"<<endl;
	}
    
    return 0;
}
