#include <iostream>
using namespace std;

char test(bool r1,bool g1,bool b1){
	if(r1==false && g1==false && b1==false) return 'B';
	if(r1==true && g1==false && b1==false) return 'r';
	if(r1==false && g1==true && b1==false) return 'g';
	if(r1==false && g1==false && b1==true) return 'b';
	if(r1==true && g1==true && b1==false) return 'y';
	if(r1==true && g1==false && b1==true) return 'm';
	if(r1==false && g1==true && b1==true) return 'c';
	if(r1==true && g1==true && b1==true) return 'w';
}
int main()
{
    int t,r,g,b,a=0,r2=0,g2=0,b2=0,c=0,y=0,m=0,w=0,b3=0;
    char c1;
    cin>>t>>r>>g>>b;
    bool r1=false,g1=false,b1=false;
    while(a<t){
    	if(a%r==0 && (a/r)%2==0){
    		r1=false;
    	}
    	if(a%r==0 && (a/r)%2==1){
    		r1=true;
    	}
    	if(a%g==0 && (a/g)%2==0){
    		g1=false;
    	}
    	if(a%g==0 && (a/g)%2==1){
    		g1=true;
    	}
    	if(a%b==0 && (a/b)%2==0){
    		b1=false;
    	}
    	if(a%b==0 && (a/b)%2==1){
    		b1=true;
    	}
    	c1=test(r1,g1,b1);
    	if(c1=='B') b3++;
    	if(c1=='r') r2++;
    	if(c1=='g') g2++;
    	if(c1=='b') b2++;
    	if(c1=='c') c++;
    	if(c1=='y') y++;
    	if(c1=='m') m++;
    	if(c1=='w') w++;
    	
    	a++;
    }
    cout << r2 << " "<<g2<<" "<<b2<<" "<<y<<" "<<c<<" "<<m<<" "<<w<<" "<<b3<<endl;
}
