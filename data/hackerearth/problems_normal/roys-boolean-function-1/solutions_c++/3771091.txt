#include <iostream>
#include <math.h>
using namespace std;

int gcd(int a,int b)
{
	if(a==0)
		return b;
	return gcd(b%a,a);
}

bool isprime(int n)
{
	if(n<2)
	return false;
	else if(n%2==0 && n!=2)
	return false;
	else
	{
		for(int i=3;i<=sqrt(n);i+=2)
		{
			if(n%i==0)
				return false;
		}
		return true;
		
	}
	
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int count=0;
    	for(int i=1;i<n;i++)
    	{
    		if(gcd(i,n)==1)
    		{
    			count++;
    		}
    	}
    	bool var=isprime(count);
    	if(var==true)
    		cout<<"TRUE\n";
    	else
    		cout<<"FALSE\n";
    	
    	
    }
    return 0;
}
