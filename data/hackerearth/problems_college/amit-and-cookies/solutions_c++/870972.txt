#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int a[n];
    
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    
    int sum=0,i=0;
    
    for(i=0;i<n;i++)
    {
    	sum=sum+a[i];
    	if(sum>=k)
    		break;
    }
    cout<<(i)<<endl;
    return 0;
}
