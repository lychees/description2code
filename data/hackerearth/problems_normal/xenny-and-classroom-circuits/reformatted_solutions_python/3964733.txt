
T = int(input())
for i in range(0, T):
    N = int(input())
    line = str(raw_input())
    pos = 0
    out_of_pos_count = 0
    for c in line:
        if pos % 2 == 0:
            if c != 'B':
                out_of_pos_count += 1
        pos += 1

    if out_of_pos_count > int(N/2):
        out_of_pos_count = N - out_of_pos_count
    print(int(out_of_pos_count))
