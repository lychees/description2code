#include <cstdio>
#include <algorithm>

const uint MAX = 32*1000000 + 1;
const uint BD = 11;
const uint MAXANS = 2000;
uint arr[MAX + 1];

uint ans(uint n, uint b){
	if (b >= BD) return MAXANS;
	else if (n < MAX) return arr[n];

	uint t = ans(n - 1, b  + 1);
	if(n % 2 == 0) t = std::min(t, ans(n/2, b  + 1));
	if(n % 3 == 0) t = std::min(t, ans(n/3, b  + 1));
	return 1 + t;
}

int main()
{
	arr[1] = 0; arr[2] = 1; arr[3] = 1;
    for(uint i = 4; i < MAX; ++ i){
    	uint t = arr[i - 1];
    	if(i % 2 == 0) t = std::min(t, arr[i/2]);
    	if(i % 3 == 0) t = std::min(t, arr[i/3]);
    	arr[i] = t + 1;
    }
    int T; scanf("%d", &T);
    while(T-->0){
    	uint n; scanf("%u", &n);
    	printf("%u\n", ans(n, 0));
    }
    return 0;
}
