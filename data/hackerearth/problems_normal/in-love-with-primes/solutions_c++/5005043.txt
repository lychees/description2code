#include <bits/stdc++.h>
using namespace std;
bool prime(int n)
{
	if(n==1) return false;
	if(n==2) return true;
	for(int i = 2;i<=sqrt(n);i++)
	{
		if(n%i==0) return false;
	}
	return true;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	bool f = true;
    	map<int,int> m;
    	for(int i = 1;i<n;i++)
    	{
    		if(prime(i))
    		{
    		    m[i]++;
    			if(m[n-i]>0)
    			{
    				f = false;
    				cout<<"Deepa"<<endl;
    				break;
    			}
    		}
    	}
    	if(f==true)
    	{
    		cout<<"Arjit"<<endl;	
    	}
    	m.clear();
    }
    return 0;
}
