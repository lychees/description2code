#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int arr[n];
    	int i=0;
    	for(i=0;i<n;++i)
    		scanf("%d",&arr[i]);
    	sort(arr,arr+n);
    	i=0;
    	int j=n-1;
    	int mx=arr[i]+arr[j];
    	int mn = mx;
    	++i;
    	--j;
    	while(i<j)
    	{
    		if((arr[i]+arr[j])>mx)mx=arr[i]+arr[j];
    		else if((arr[i]+arr[j])<mn)mn=arr[i]+arr[j];
    		++i;
    		--j;
    			
    	}
    	cout<<mx-mn<<endl;
    		
    }
}
