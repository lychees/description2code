#include <iostream>
#include<math.h>
#define LL long long
using namespace std;
LL mat[1025][1025];
LL count=1;

int abs(int a)
{
	if(a<0)
	return -a;
	else
	return a; 
}

void fill(int a, int b, int c, int d)
{
	LL mid1,mid2;
	if(abs(a-c)<=1 && abs(b-d)<=1)
	{
		mat[a][b]=count++;
		mat[a][b+1]=count++;
		mat[a+1][b]=count++;
		mat[a+1][b+1]=count++;
	}
	else
	{
		mid1=(a+c)/2;
		mid2=(b+d)/2;
		fill(a,b,mid1,mid2);
		fill(a,mid2+1,mid1,d);
		fill(mid1+1,b,c,mid2);
		fill(mid1+1,mid2+1,c,d);
	}
}

int main()
{
  int n,max,i,j;
  cin>>n;
  max=pow(2,n);
  fill(1,1,max,max);
  for(i=1;i<=max;i++)
  {
  	for(j=1;j<=max;j++)
  	{
  		cout<<mat[i][j]<<" ";
  	}
  	cout<<endl;
  }
}
