#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				int k;
				cin>>k;
				string s;
				cin>>s;
				int leng=s.length();
				int value[1000007];
				value[0]=0;
				for(int i=1;i<=leng;i++)
					{
						value[i]=value[i-1]+(s[i-1]-97+1);
					}
				int start=0;
				int end=1;
				int count=0;
				while(end<=leng)
					{
						if(value[end]-value[start]<k)
							end++;
						else if(value[end]-value[start]>k)
							start++;
						else
							{
								count++;
								start++;
								end++;
							}
					}
				cout<<count<<endl;
			}
		return 0;
	}
