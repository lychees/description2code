int main()
{
    ios::sync_with_stdio(false);
     const int sieveSize = 100000;
     int lp[sieveSize+1];
     memset(lp,0,sizeof(lp));
     vector<int> primes;
      
     for (int i=2; i<=sieveSize; ++i) {
         if (lp[i] == 0) {
             lp[i] = i;
             primes.push_back (i);
         }
         for (int j=0; j<(int)primes.size() && primes[j]<=lp[i] && i*primes[j]<=sieveSize; ++j)
             lp[i * primes[j]] = primes[j];
     }
     int n;
     cin>>n;
     for(int i=2;i<=n;i++)
        if (!binary_search(primes.begin(),primes.end(),i)) cout<<i<<endl;
    return 0;
}
