#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

string rec(int n,int k)
{
	int i;
	string r;

	for(i=1;i<k;i++)
		r+='9';
	r+='8';

	for(i=n-k-k;i;i--)
		r+='9';

	for(i=1;i<k;i++)
		r+='0';
	r+='1';

	return r;
}

int main()
{
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		k=min(k,n-k);
		cout<<rec(n,k)<<endl;
	}
	return 0;
}
