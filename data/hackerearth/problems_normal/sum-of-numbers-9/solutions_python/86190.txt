from itertools import *
def hasSubset(elems,nelems,ssum):
		for i in xrange(nelems+1):
				for s in combinations(elems,i):
						if ssum == sum(s):
								print "YES"
								return
		print "NO"
		return

def main():
		ntests = int(raw_input())
		for _ in xrange(ntests):
				nelems = int(raw_input())
				elems = map(int,raw_input().split())
				ssum = int(raw_input())
				hasSubset(elems,nelems,ssum)

main()
