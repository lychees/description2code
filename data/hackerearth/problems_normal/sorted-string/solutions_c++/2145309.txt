#include<bits/stdc++.h>
using namespace std;

int main() {
    
    int i,j,k,n;
    char s[110];
    cin >> n; 
    while(n--) {
       vector <pair<int,int> > v(26);
       for(i=0;i<26;i++) {
       	  v[i].first=0;
       	  v[i].second=i;
       }
       scanf("%s",s);
       j=strlen(s);
       for(i=0;i<j;i++) {
          v[s[i]-'a'].first++;
       }

       sort(v.begin(),v.end());

       for(i=0;i<26;i++) {
       	char c=v[i].second+'a';
       	 for(j=0;j<v[i].first;j++) {

       	 	printf("%c",c);
       	 }
       }
       printf("\n");
    }


	return 0;
}
