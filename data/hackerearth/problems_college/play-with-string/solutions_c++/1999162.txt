#include <iostream>
#include<cstdio>
#include<string.h>
#include<stdlib.h>
using namespace std;

int main()
{

    int t,k=0,i,n,flag=0;
    char a[100001],b[100001];
    scanf("%d",&t);
    while(t--)
    {
        k=0;
        flag=0;
        scanf("%s",a);
        n=strlen(a);
        for(i=n-1;i>=0;i--)
            b[k++]=a[i];

        for(i=1;i<(n-1);i++)
        {
            if(abs(a[i]-a[i-1])==abs(b[i]-b[i-1]))
                flag=1;
            else
            {
                flag=0;
                break;
            }
        }
        if(flag)
            printf("Funny\n");
        else
            printf("Not Funny\n");
    }
    return 0;
}
