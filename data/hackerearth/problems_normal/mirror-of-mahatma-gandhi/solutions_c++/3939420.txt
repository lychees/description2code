#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i,flag;
	string s;
	scanf("%d",&t);
	
	while(t--){
		flag=0;
		getchar();
		cin>>s;
		
		for(i=0;i<s.size();i++){
			if(s[i]=='2' ||s[i]=='3' ||s[i]=='4' ||s[i]=='5' ||s[i]=='6' ||s[i]=='7' ||s[i]=='9'){
				flag=1;
				break;
			}
		}
		n=s.size();
		if(flag==0){
			for(i=0;i<n/2;i++){
				if(s[i]!=s[n-i-1]){
					flag=1;
					break;
				}
			}
		}
		
		if(flag==1){
			printf("NO\n");
		}
		else{
			printf("YES\n");
		}
	}
	return 0;
}
