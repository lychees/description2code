#include <iostream>
using namespace std;

int main()
{

    unsigned int i,j,co,a[1000],n,k=1;
    a[0]=2;
    for(i=3;i<=7919;i=i+2)
    {
    	co=0;
    	for(j=2;j<i/2;j++)
    	{
    		if(i%j==0)
    	{
    		co=1;
    		break;
    	}
    	}
    	if(co==0)
    	{
    		a[k]=i;
    		k++;
    	}

    }
    
    cin>>n;
    cout<<a[n-1];
}
