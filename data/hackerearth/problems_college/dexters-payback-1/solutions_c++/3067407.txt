#include <algorithm> 
#include <iostream> 
#include <math.h>  
using namespace std; 

 
#define Max(x,y) ((x)>(y)?(x):(y)) 
#define Min(b,a) ((b)<(a)?(b):(a)) 
 
typedef long long LL; 
 
int ms1(int w, int h) { 
    return (w * h + 1) / 2; 
    } 
   
  int MaxStones(int width, int height) { 
   
    return ms1(width/2, height/2) + 
      ms1((width+1)/2, (height)/2) + 
      ms1((width+1)/2, (height+1)/2) + 
      ms1((width+0)/2, (height+1)/2); 
    } 
  
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int weigth,heigth;
        cin>>weigth>>heigth;
        LL answer = MaxStones(weigth,heigth);
        cout<<answer<<endl;
    }
    return 0;
}
