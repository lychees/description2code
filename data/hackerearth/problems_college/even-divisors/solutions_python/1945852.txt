def factors(n):
    return set(x for tup in ([i, n//i] 
                for i in range(1, int(n**0.5)+1) if n % i == 0) for x in tup)
 
t=int(input())
for i in range ( t ):
	num=int(input())
	fact = (factors(num))
	ctr = 0
	for i in fact:
		#print i

			ctr += 1
