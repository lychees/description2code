#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int i,j,k,l;
    string a;
    cin>>a;
    l=a.length();
    long long int b[l];
    memset(b,0,sizeof b);
    k=(int)a[l-1]-48;
    if(k%2==0){
    	b[l-1]++;
    }
    for(i=l-2;i>=0;i--){
    	k=(int)a[i]-48;
    	if(k%2==0){
    		b[i]++;
    	}
    	b[i]+=b[i+1];
    }
    for(i=0;i<l;i++){
    	cout<<b[i]<<" ";
    }
    return 0;
}
