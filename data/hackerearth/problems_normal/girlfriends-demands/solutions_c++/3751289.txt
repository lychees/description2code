#include <iostream>
#include <cassert>
using namespace std;

int main()
{
   string str;
	long int q,i;
	long long int a, b,len;
	cin >> str;
	len = str.length();
	assert(1 <= len & len <= 100000);
	cin >> q;
	assert(1 <= q & q <= 100000);

	for (i= 1; i <= q; i++)
	{
		cin >> a;
		assert(1 <= a & a <= 1000000000000000000);
		cin >> b;
		assert(1 <= b & b <= 1000000000000000000);
		a = ((a % len) == 0) ? len : (a % len);
		b = ((b % len) == 0) ? len : (b % len);
		
		if (str[a-1] == str[b-1])
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
    return 0;
}
