#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main()
{
	ll t,n,a,b,arr[112345],i,j;
	for(cin>>t;t;t--)
	{
		memset(arr,0,sizeof(arr));
		cin>>a>>b>>n;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		sort(arr,arr+n);
		ll rag=0,say=0;
		ll r=0,s=0;
		for(i=0;i<n;i++)
		{
			rag+=arr[i];
			say+=arr[i];
			if(rag<=a)
			{
				r++;
			}
			if(say<=b)
			{
				s++;
			}
			else
			{
				break;
			}
		}
		if(r<s)
		{
			cout<<"Sayan Won"<<endl;
		}
		else if(r==s)
		{
			cout<<"Tie"<<endl;
		}
		else
		{
			cout<<"Raghu Won"<<endl;
		}
	}
	return 0;
}
