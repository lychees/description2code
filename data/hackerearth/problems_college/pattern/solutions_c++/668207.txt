#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
    cout<<"\n";
    for(int i=1;i<=n;i++)
    {
    	if(i==1||i==n)
    	{
    		for(int j=1;j<=n;j++)
    		cout<<"*";
    		cout<<"\n";
    	}
    	else
    	{
    		for(int j=n;j>i;j--)
    		cout<<" ";
    		cout<<"*\n";
    	}
    }
    return 0;
}
