#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    	map<string,int>m;
    cin>>t;
    while(t--)
    	{
    			string s;
    			cin>>s;
    		
    			m[s]++;
    	}
    vector<pair<string,int> > v(m.size());
copy(m.begin(), m.end(), v.begin());
sort(v.begin(),v.end());
for(int i=0;i<v.size();++i)
	cout<<v[i].first<<" "<<v[i].second<<endl;
return 0;
}
