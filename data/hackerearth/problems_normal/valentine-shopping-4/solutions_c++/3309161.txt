#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int i,j,n,m,t,s,y;float dis,k;
    int a[1000][3]={0};
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d %d",&n,&m);
   	for(i=0;i<n;i++)
   	{
   		for(j=0;j<m;j++)
   		{
   			scanf("%d %d %d",&a[j][0],&a[j][1],&a[j][2]);
   		}
   		
   		k=100;s=0;
   		for(j=0;j<m;j++)
   		{dis=100;
   		  for(y=0;y<3;y++)
   		  {
   		  	if(a[j][y]==100){dis=0;}
   		  	else if(a[j][y]>0){dis=dis-dis*((float)a[j][y]/(float)100);}
   		  }//printf("%f dis",dis);
   		  if(dis<k){k=dis;s=j+1;}
   		}
   		printf("%d ",s);
   	}
   	printf("\n");
   }
    return 0;
}
