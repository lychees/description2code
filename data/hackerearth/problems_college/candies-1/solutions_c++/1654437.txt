#include <iostream>
using namespace std;

int main()
{
	long long t;
	cin>>t;
            for (long i = 0; i < t; i++)
            {
                long long x;cin>>x;
                long long int sum = 0;
                
                for (int j = 0; j < x; j++)
                {
                   long long int y;
                   cin>>y;
                    sum += y;
                }
                if (sum % x == 0)
                {
                   cout<<"YES\n"<<sum / x<<" 0"<<endl;
                }
                else
                {
                    cout<<"NO\n"<<sum / x<<" "<<sum % x<<endl;
                }
            }
	
    
    return 0;
}
