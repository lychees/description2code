ii solve(int b, int g, char in)
{
	int n = b+g;
	s[0] = in;
	int j = 1;
	rep(i,0,n){
			if(g == 0){
				while(b){s[j++] = 'b'; b--;}
				break;
			}
			if(b == 0){
				while(g){s[j++] = 'g'; g--;}
				break;
			}
			if(i & 1){
				if(s[j-1] == 'b'){s[j++] = 'b'; b--;}
				else{s[j++] = 'g'; g--;}
			}
			else{
				if(s[j-1] == 'b'){s[j++] = 'g'; g--;}
				else{s[j++] = 'b'; b--;}
			}
		}
		int cb = 0, cg = 0;
	rep(i,0,j-1){
		//printf("%c ",s[i]);
		if(s[i] != s[i+1]) cg++;
		else cb++;
	}
	return ii(cb,cg);
}
