#include <bits/stdc++.h>
#define ll long long int

using namespace std;
 
vector <pair<ll,ll> >v; 
ll memo[201][201];

ll sum(ll i,ll j)
{

	if( i < 0) {
		return 0;
	}
	if ( memo[i][j] != -1 ) {
		return memo[i][j];
	}
	ll ans1 ,ans2;
	ans1 =0;
	if (v[i].second < v[j].second && v[i].first < v[j].first) {
		ans1 = v[i].second+sum(i-1,i);
	}
	ans2 = sum(i-1,j);
	if ( ans1 > ans2 ) {
		 return memo[i][j] =ans1;
	}
	 return memo[i][j] = ans2;

} 
	
int main()
{
	ll t,n,i,j,k,ans,a,b;
	
	scanf("%lld",&t);
	while (t--) {
		scanf("%lld",&n);
		v.clear();
		for ( i = 0; i <=n ;i++) {
			for ( j =0;  j <= n; j++) {
				memo[i][j] = -1;
			}
		}
		for ( i = 0; i < n; i++) {
			scanf("%lld%lld",&a,&b);
			v.push_back(make_pair(a,b));
		
		}
		sort(v.begin(),v.end());
		v.push_back(make_pair(LONG_MAX,LONG_MAX));
		ans = sum(n-1,n);
		printf("%lld\n",ans);
		
	}
	return 0;
}
