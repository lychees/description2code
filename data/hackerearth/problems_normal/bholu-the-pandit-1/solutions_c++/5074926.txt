#include<iostream>
#define mod 1000000007 
using namespace std;
long int gcd(long int a,long int b)
{
	if(a%b==0)
	return b;
	return gcd(b,a%b);
}
int main()
{
	int t;
	long int a,b,c,s1,vol;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		s1=gcd(gcd(a,b),c);
		vol=(a/s1*b/s1)%mod;
		vol=vol*(c/s1)%mod;
		cout<<s1<<" "<<vol<<endl;
	}
	
	return 0;
}
