#include<bits/stdc++.h>
using namespace std;

const int N=1010;

int a[N];

int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) {
      scanf("%d",a+i);
    }
    sort(a,a+n);
    int lr,sm;
    scanf("%d%d",&lr,&sm);
    printf("%d\012",a[sm-1]+a[n-lr]);
  }
  return 0;
}
