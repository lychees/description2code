#include <bits/stdc++.h>
using namespace std;

int n;
int arr[100005];
bool can[100005];


int main() {
	scanf("%d", &n);
	for(int i = 1; i <= n; i++)
		scanf("%d", &arr[i]);
	int pos = 1;
	stack <int> S;
	
	for(int i = 1; i <= n; i++) {
		while(!S.empty() && S.top() < arr[i]) S.pop();
		if(S.size() == 0) can[i] = true;
		S.push(arr[i]);
	}
	while(!S.empty()) S.pop();
	for(int i = n; i >= 1; i--) {
		while(!S.empty() && S.top() < arr[i]) S.pop();
		if(S.size() == 0) can[i] = true;
		S.push(arr[i]);
	}
	for(int i = 1; i <= n; i++)
		if(can[i])
			printf("%d ", i);
}
