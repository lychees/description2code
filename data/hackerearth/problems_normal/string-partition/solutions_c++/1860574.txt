#include<iostream>
#include<string.h>

using namespace std;

int main()
{
    int t,n,k,z,i,j;
    cin>>t;
    char str[101];
    long long num,fromto[101][101],dp[101][101],tmp,r;
    while(t--)
    {
        cin>>r>>z;
        cin>>str;
        n = strlen(str);
        for(i = 0;i<n;i++)
        {
            num = 0;
            for(j =i;(j<n);j++)
            {
                num = num*10 + (str[j]-'0');
                if(num > z)
                    break;
                fromto[i][j] = num;
            }
            for(;j<n;j++)
                fromto[i][j] = -1;
        }
        i = 0;
        for(j = 0;j<n;j++)
            dp[i][j] = fromto[0][j];
        for(i= 1;i<r;i++)
        {
            for(j= 0;j<n;j++)
            {
                dp[i][j] = -1;
            }
        }
        for(i = 1;i<r;i++)
        {
            for(j = 0;j<n;j++)
            {
                for(k = 0;k<j;k++)
                {
                    if((fromto[k+1][j]!= -1)&&(dp[i-1][k] != -1))
                    {
                        tmp = dp[i-1][k]+fromto[k+1][j];
                        if(tmp>dp[i][j])
                            dp[i][j] = tmp;
                    }
                }
            }
        }
        cout<<"\n"<<dp[r-1][n-1];



    }
    return 0;
}
