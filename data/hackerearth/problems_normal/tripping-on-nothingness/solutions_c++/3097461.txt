#include<bits/stdc++.h>
using namespace std;

int main()
 {
     map<int,int> mp;
     int s,q;
    cin>>s>>q;
    while(q--)
     {
       int a;
       cin>>a;

       int ans=__gcd(s,a);
        if(mp[ans]==0)
           {
           cout<<ans<<endl;
           mp[ans]++;
           }
           else
           cout<<-1<<endl;
     }
 }
