using namespace std;
typedef long long int LL;

#define fr(i,a,b) for(int i = a; i < b; i++)
#define frll(i,a,b) for(LL i = a; i < b; i++)
#define s(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define p(x) printf("%d\n",x)
#define pll(x) printf("%lld\n",x)

#define MAX 10001

struct arr{
	int l;
	int d;
};

struct arr arr_v[17];
int compare(struct arr arr1, struct arr arr2){
	return (arr1.d > arr2.d);
}

int main(){
    int t; 
    int n;
    s(n);
	LL sum_l = 0, sum_d = 0;
	fr(i, 0, n){
		s(arr_v[i].l);
		sum_l += arr_v[i].l;
	}	
	fr(i, 0, n){
		s(arr_v[i].d);
		arr_v[i].d *= i;
		sum_d += arr_v[i].d;
	}
	sort(arr_v, arr_v+n, compare);
	
	LL ans = sum_l;
	LL sub_d = 0;
	
	fr(i, 0, n-1){
		sub_d += arr_v[i].d;
		ans += sum_l - (sum_d - sub_d);
		sum_l -= (sum_d - sub_d);
	}
	pll(ans);
    return 0;
}
