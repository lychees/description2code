def count(arr, target):
    add = target
    for item in arr:
        if item <= add:
            add += 1
    return add


for _ in xrange(input()):
    numbers, removed, smallest = map(int, raw_input().split())
    removed_numbers = map(int, raw_input().split())
    if smallest > numbers-removed:
        print -1
    else:
        index = count(removed_numbers, smallest)
        print index
