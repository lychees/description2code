#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,c,z;
	cin>>t;
	stack<int> s;
	while(t--)
	{
		int n,id;
		cin>>n>>id;
		while(!s.empty())
				s.pop();
		s.push(id);
		for(int i=0;i<n;i++)
		{
			char query;
			cin>>query;
		if(query=='B')
			{
				int x=s.top();
				s.pop();
	
				int y=s.top();
				s.pop();
				       
				s.push(x);
				s.push(y);
				continue;
			}
	
			cin>>c;
			s.push(c);
		}
			
		z=s.top();
		cout<<"Player "<<z<<"\n";
		
	}
	return 0;
}
