#include<iostream>
using namespace std;
#define MOD 1000000007 
int main(){
   int t;
   scanf("%d",&t);
   while(t--) {
     int n,i;
     scanf("%d",&n);
     long long int ans=1;
     for(i=0;i<n-1;i++)
        ans=(ans<<1)%MOD;
     ans=(ans*(n*(n+1))/2)%MOD;
     printf("%lld\n",ans);
   }
   return 0;
}
