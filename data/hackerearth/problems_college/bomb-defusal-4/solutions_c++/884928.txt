#include <iostream>
#include<math.h>
using namespace std;

int main()
{
   int t,p,a,b,u,v;
   cin>>t;
   while(t--)
   {
   	cin>>p>>a>>b>>u>>v;
   	if(a==u&&b==v)
   	cout<<"Counter-Terrorists Win !"<<endl;
   	else if((a-b==u-v || a+b==u+v) && p>1)
   	cout<<"Counter-Terrorists Win !"<<endl;
  else
   	{	if(abs(u+v+a-b)%2==0 || abs((u+v-a-b))%2==0)
   		{
   			if(p>2)
   			cout<<"Counter-Terrorists Win !"<<endl;
   			else
   			cout<<"Terrorists Win !"<<endl;
   		}
   		else
   		cout<<"Terrorists Win !"<<endl;
   	}
   }
    return 0;
}
