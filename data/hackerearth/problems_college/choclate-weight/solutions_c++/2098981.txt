#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<long int> vec;
int main()
{
    long int n,p,j,x,i;
    int flag;
    cin>>n;
    while(n--)
    {   cin>>x;
        vec.push_back(x);
    }
    sort(vec.begin(),vec.end());
    cin>>p;
    while(p--)
    {
        cin>>x;
        if(count(vec.begin(),vec.end(),x)>0)
            {flag=1;cout<<"I Like You\n";}
        else{

            for(j=0;j<vec.size();++j)
            {
                if(vec.at(j)<=x)
                continue;
                else break;
            }
            for(i=0;i<j;++i)
            {
                if((x-vec.at(i)!=vec.at(i))&&count(vec.begin(),vec.begin()+j,x-vec.at(i))>0)
                {cout<<"I Like You\n";flag=1;
                break;
                }
                else if((x-vec.at(i)==vec.at(i))&&count(vec.begin(),vec.begin()+j,vec.at(i))>=2)
                {cout<<"I Like You\n";flag=1;
                break;
                }
                else flag=0;

            }

        }
        if(flag!=1)
        cout<<"I Hate You\n";
    }



return 0;

}
