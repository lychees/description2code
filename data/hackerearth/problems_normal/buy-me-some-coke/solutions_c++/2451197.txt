typedef unsigned long long int uint64;
typedef long long int int64;
  
int tot=0;
int c;
int dp[155][155][55];
int vis[155][155][55];
int timer=1;
int go(int coke,int n5,int n10){
	
	if(coke==c){
		return 0;
	}
	
	if(vis[coke][n5][n10]==timer)
	return dp[coke][n5][n10];
	
	vis[coke][n5][n10]=timer;
	
	int n1=tot-coke*8;
	n1-=(5*n5+10*n10);
	
	int ret=1e9;
	// 1*8
	if(n1>=8)
	ret=min(ret,go(coke+1,n5,n10)+8);
	// 1*5 + 3*1
	if(n5>=1 && n1>=3)
	ret=min(ret,4+go(coke+1,n5-1,n10));
	// 2*5
	if(n5>=2)
	ret=min(ret,2+go(coke+1,n5-2,n10));
	// 1*10
	if(n10>=1)
	ret=min(ret,1+go(coke+1,n5,n10-1));
	// 1*10 + 3*1
	if(n10>=1&& n1>=3)
	ret=min(ret,4+go(coke+1,n5+1,n10-1));
	
	dp[coke][n5][n10]=ret;
	return ret;
}
int main(){
	int t,n1,n5,n10;
	cin>>t;
	while(t--){
		timer++;
		cin>>c>>n1>>n5>>n10;
		tot=n1+5*n5+10*n10;
		if(tot<8*c){
			printf("-1\n");
			continue;
		}
		
		cout<<go(0,n5,n10)<<endl;
	}
	return 0;
}
