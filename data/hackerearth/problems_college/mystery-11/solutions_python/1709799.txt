from sys import stdin
tc = raw_input()
def sod(x):
	if x == 1:
		return 2
	elif x == 2 or x == 3 or x == 5 :
		return 5
	elif x == 4 :
		return 4
	elif x == 6 or x == 9:
		return 6
	elif x == 7:
		return 3
	elif x == 8:
		return 7
	else:
		return 6

for i in stdin:
	ans = 0
	for j in i:
		try:
			j = int(j)
			ans += sod(j)
		except ValueError:
			pass
	print ans
