#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	int arr[10001];
	arr[1]=arr[3]=arr[4]=1;
	arr[2]=0;
	for(int i=5;i<10001;i++){
		if(arr[i-1]&&arr[i-4]){
			arr[i]=0;
		}
		else{
			arr[i]=1;
		}
	}
	while(t--){
		int n;
		cin>>n;
		if(arr[n])
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
}
