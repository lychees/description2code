#include <bits/stdc++.h>
using namespace std;


int main()
{

   int t,i;
   char s[1001];
   cin>>t;
   while(t--)
   {   int flag=0;
       int count=0;
       map<char,int> m;
       map<char,int> :: iterator it;
       cin>>s;
       for(i=0;s[i]!='\0';i++)
       {
           {
             m[s[i]]++;
           }
       }
       for(it=m.begin();it!=m.end();++it)
       {
           if(((it->second)%2)!=0)
            {count++;
            flag=1;}
       }
       if(flag)
       cout<<count-1<<endl;
       else cout<<count<<endl;
   }
    return 0;
}
