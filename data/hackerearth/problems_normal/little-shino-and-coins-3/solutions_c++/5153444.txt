#include <iostream>
#include<unordered_set>
using namespace std;

int main()
{
    int K;
    cin>>K;
    string str;
    cin>>str;
    long long len=str.length();
    long long i,j;
    long long count=0;
    	unordered_set<char> s;
    for(i=0;i<(len-K)+1;i++)
    {
    
    	s.insert(str[i]);
    		if(s.size()==K)
    				 count++;
    	for(j=i+1;j<len;j++)
    	{
    		s.insert(str[j]);
    			if(s.size()==K)
    				 count++;
    	}
  
    	  s.erase(s.begin(),s.end());
    }
    cout<<count;
    return 0;
}
