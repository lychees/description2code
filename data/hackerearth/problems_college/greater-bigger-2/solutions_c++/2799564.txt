#include <iostream>
#include<algorithm>
using namespace std;

int main()
{  int t;
   cin>>t;
    while(t--){
    	char a[105];
    	cin>>a;
    	int n=0;
    	while(a[n]){
    		n++;
    	}
    	int f=0;
    	char temp;
    	 for(int i=n-1;i>0;i--){
    	 	
    	 	if(a[i]>a[i-1]){
    	 		f=1;
    	 		int k;int m=a[i-1];
    	 		for(int j=i;j<=n-1;j++){
    	 			if(a[j]>a[i-1]){
    	 				m=min(m,(int)a[j]);
    	 				k=j;
    	 			}
    	 		}
    	 		temp=a[k];
    	 		a[k]=a[i-1];
    	 		a[i-1]=temp;
    	 		char s1[103];
    	 		int c=0;
    	 		
    	 		for( int j=i;j<n;j++){c++;
    	 			s1[j-(i)]=a[j];
    	 		}
    	 		sort(s1,s1+c);
    	 		for(int j=i;j<n;j++)
    	 		{
    	 			a[j]=s1[j-(i)];
    	 		}
    	 		break;
    	 	}
    	 }
    	 if(f){
    	 	cout<<a<<endl;
    	 	
    	 }
    	 else{
    	 	cout<<"no answer"<<endl;
    	 }
    	
    }
    //cout << "Hello World!" << endl;
    return 0;
}
