#include <bits/stdc++.h>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int count = 0;

	for(int i = 0; i < s.length(); i++)
	{
		if(count == 0)
		{
			if(s[i] == 'l')
				count++;
		}
		if(count == 1)
		{
			if(s[i] == 'o')
				count++;
		}
		if(count == 2)
		{
			if(s[i] == 'v')
				count++;
		}
		if(count == 3)
		{
			if(s[i] == 'e')
				count++;
		}
	}
	if(count == 4)
		cout<<"I love you, too!";
	else
		cout<<"Let us breakup!";

	return 0;
}
