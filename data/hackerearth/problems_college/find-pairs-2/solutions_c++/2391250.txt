#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,i,j,arr[100000],n;
   cin>>t;
   while(t--)
   {
   	int count=0;
   	cin>>n;
   	for(i=0;i<n;i++)
   	cin>>arr[i];
   	for(i=0;i<n;i++)
   	{
   	 for(j=i;j<n;j++)
   	 if(arr[i]==arr[j] && j>=i)
   	 count++;
   	}
   	cout<<count<<endl;
   }
    return 0;
}
