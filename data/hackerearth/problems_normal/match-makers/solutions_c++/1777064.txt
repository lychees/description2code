#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int tc;
    long long int i,n,r;

    cin>>tc;
    while(tc--)
    {
        r=0;
        cin>>n;
        long long int boys[n],girls[n];
        for(i=0;i<n;i++)
            {
                cin>>boys[i];
            }

        for(i=0;i<n;i++)
            {
                cin>>girls[i];
            }
        sort(boys,boys+n,greater<int>());
        sort(girls,girls+n);

        for(i=0;i<n;i++)
        {   
            if(boys[i]%girls[i]==0 || girls[i]%boys[i]==0)
            {
                r++;

            }

        }
        cout<<r<<endl;
    }
    return 0;
}
