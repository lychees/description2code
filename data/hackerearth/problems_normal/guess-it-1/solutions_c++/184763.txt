int main()
{
    ios_base::sync_with_stdio(0);
    int n,len,i,sum;
    string num;
    cin>>n;
    while(n--)
    {
        cin>>num;
        sum=0;
        len=num.length();
        for(i=0;i<len;++i)
        {
            if(num[i]!='_')
                sum+= (num[i]-'0');
        }
        if(sum%9==0) cout<<"0 or 9\n";
        else
        {
            cout<<9-(sum%9)<<"\n";
        }
    }
    return 0;
}
