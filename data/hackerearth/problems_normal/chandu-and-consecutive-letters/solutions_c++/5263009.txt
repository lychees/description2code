#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    string s;
    int tc;
    cin>>tc;
    while(tc--) {
    	cin>>s;
    	char current = ' ';
    	for(int i=0;i<s.size();i++) {
    		if(s[i] != current) {
    			cout<<s[i];
    			current = s[i];
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
