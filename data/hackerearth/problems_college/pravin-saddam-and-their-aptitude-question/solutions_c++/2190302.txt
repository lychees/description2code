#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b,sum=0;
    	cin>>a>>b;
    	for(int i=a;i<=b;i++)
    	{
    		if(i>4)
    		break;
    		if(i<=2)
    		sum=sum+i;
    		if(i==3)
    		sum=sum+6;
    		if(i==4)
    		sum=sum+4;
    	}
    	cout<<(sum%10)<<endl;
    }
    return 0;
}
