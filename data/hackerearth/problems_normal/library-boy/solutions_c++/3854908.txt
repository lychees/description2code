#include <iostream>

using namespace std;

int v[26];
int n;

int main()
{
    cin >> n;
    int i;
    for(i=0;i<n;i++)
    {
    	string x;
    	cin >> x;
    	v[x[0]-'a']++;
    }
    int ans=0;
    for(i=0;i<26;i++)
    {
    	if(!v[i]) continue;
    	ans+=(v[i]/10)+(v[i]%10?1:0);
    }
    cout << ans << endl;
    return 0;
}
