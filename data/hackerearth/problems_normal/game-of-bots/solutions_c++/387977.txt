#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char ini[1000000],fin[1000000];
		cin>>ini>>fin;
		int len=strlen(ini),flag=0,k=0;
		for(int i=0;i<len;i++)
		{
			if(ini[i]=='A'||ini[i]=='B')
			{
			//	cout<<ini[i]<<endl;
				for(int j=k;j<len;j++)
				{
					if(fin[j]=='A'||fin[j]=='B')
					{
			//			cout<<fin[j]<<endl<<i<<endl<<j<<endl;
						k=j+1;
						if(ini[i]!=fin[j])
						{
							flag=1;
			//				cout<<"hi1";
							break;
						}
						else if(j>i&&ini[i]!='B'&&flag==0)
						{
							flag=1;
			//				cout<<"hi2";
							break;
						}
						else if(j<i&&ini[i]!='A'&&flag==0)
						{
							flag=1;
			//				cout<<"hi3";
							break;
						}
						else
						break;
					}
					if(flag==1)
					break;
				}
			}
			
		}
		if(flag==1)
		{
			cout<<"No"<<endl;
		}
		else
		{
			cout<<"Yes"<<endl;
		}
	}
	return 0;
}
