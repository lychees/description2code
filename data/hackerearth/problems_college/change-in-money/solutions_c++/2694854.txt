#include<cstdio>
#include<algorithm>
#include<iostream>
#include<vector>
#include<cstring>
using namespace std;
long long int coin(long long int a[],long long int m,long int n)
{
	long long int tbl[n+1],i,j;
	memset(tbl,0,sizeof(tbl));
	tbl[0]=1;
	for(i=0;i<m;i++)
	{
		for(j = a[i];j<=n;j++)
		{
			tbl[j]=tbl[j] + tbl[j-a[i]];
		}
	}
	return tbl[n];
}
int main()
{

	long long int n,m;
	cin>>n>>m;
	long long int a[m];
	long long int i;
	for(i=0;i<m;i++)
	{
		cin>>a[i];
	}
      long long int p= coin(a,m,n);
      cout<<p<<"\n";

}
