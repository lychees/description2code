#include <iostream>
using namespace std;

int main()
{
    int T,N,D;
    cin>>T;
    while(T--)
    {
    	int Dis=0;
    	int trav=0;
    	cin>>N>>D;
    	while(N)
    	{
    		
    		
    		if(trav<6)
    		{
    			Dis+=2;
    			N--;
    			trav+=2;
    		}
    		if(trav==6)
    		{
    			trav=0;
    			N+=2;
    		}
    		
    	}
    	if(Dis>=D){cout<<"Yes\n";}
    	else
    	{
    		cout<<"No\n";
    	}
    }
    return 0;
}
