#include <iostream>
#include <ctype.h>
using namespace std;

bool validate_username(string email, int s, int e){
	int i = s-1;
	
	while(++i<=e)
		if(!(isalnum(email[i]) || email[i]=='_'))
			return false;
	
	return true;
}


bool validate_hostname1(string email, int s, int e){
	int i = s-1;bool flag = false;
	
	while(++i<=e)
		if(isalpha(email[i])) flag = true;
		else if(isdigit(email[i]));
		else if (email[i]=='-');
		else return false;
		
	if(flag) return true;
	else return false;
}

bool validate_hostname2(string email, int s, int e){
	int i = s-1;
	
	while(++i<e)
		if(islower(email[i]));
		else return false;
	
	return true;
}

bool validate(string email){
	int s,e,i;
	
	for(i=0; email[i]!='@';++i);
		

	if(!(i>=1 && i<=16))	
		return false;
	else if(validate_username(email,0,i-1)){
		
		s = i+1;
		for(i=s; email[i]!='.'; ++i);
		
		if(!((i-s)>=1 &&(i-s)<=16))	
			return false;
		
		else if(validate_hostname1(email,s,i-1)){
			if(!((email.length() - i-1)>=1 && (email.length() - i-1)<=10))	
				return false;
			else
				return validate_hostname2(email,i+1,email.length());
		}
	}else return false;
}

int main()
{
    string email;
    int T;
    cin>>T;
    while(T--){
    	cin>>email;
    
    	if(validate(email))	cout<<"YES"<<endl;
    	else	cout<<"NO"<<endl;
    }
}
