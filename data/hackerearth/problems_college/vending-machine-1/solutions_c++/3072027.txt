#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
vector<int> s;
void check(int n)
{
    if(n-1)
    {
        s.push_back(s[0]);
        s.push_back(s[0]);
        s.erase(s.begin());
        check(n-1);
    }
}
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=5;i++)
        s.push_back(i);
    check(n);
    switch(s[0])
    {
    case 1:
        cout<<"Sheldon";
        break;
    case 2:
        cout<<"Leonard";
        break;
    case 3:
        cout<<"Penny";
        break;
    case 4:
        cout<<"Rajesh";
        break;
    case 5:
        cout<<"Howard";
        break;
    }
    return 0;
}
