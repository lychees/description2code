for _ in range(int(raw_input())):
    ans = 0
    t = {}
    for __ in range(int(raw_input())):
        a, b = map(int, raw_input().split())
        if a == b:
            continue
        if b in t and t[b]:
            t[b] -= 1
        else:
            ans += 1
        if a in t:
            t[a] += 1
        else:
            t[a] = 1
    print ans
