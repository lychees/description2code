#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	string a;
    	int b[26]={0};
    	
    	for(int j=0;j<n;j++)
    	{
    		cin>>a;
    		
    		for(int k=0;k<26;k++)
    		{
    			if(a.find(char(97+k))!=-1)
    				b[k]++;
    		}
    	}

		int max=b[0];
		int index=0;
		for(int k=1;k<26;k++)
   		{
    		if(b[k]>max)
    		{
    			index=k;
    			max=b[k];
    		}
    	}
    	
    	for(int k=0;k<26;k++)
   		{
    		if(b[k]==max)
    		{
    			index=k;
    			break;
    		}
    	}
    	
    	cout<<char(97+index)<<endl;
    	
   	}
    return 0;
}
