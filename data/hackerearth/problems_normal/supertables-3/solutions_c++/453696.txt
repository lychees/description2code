#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int calculateGcd(int a,int b){
    int t;
    while(b!=0){
        t=b;
        b=a%b;
        a=t;
    }
    return a;
}

int main()
{
    int T,a,b,i;
    long long gcd,lcm,l,r,m,sol,val;
    cin >> T;
    while(T--)
    {
   cin >> a >> b >> i;
		gcd=calculateGcd(a,b);
		lcm = (a*b)/gcd;
		l = 1;
		r = 100000000000000000LL;
		while ( l <= r ) {
			m = (l+r)/2;
			val = (m/a + m/b - m/lcm);
			if ( val == i ) {
                sol = max((m/a)*a, (m/b)*b);
				cout << sol << endl;
				break;
			}
			else if ( val < i ) {
				l = m+1;
			}
			else {
				r = m-1;
			}
		}

    }
    return 0;
}
