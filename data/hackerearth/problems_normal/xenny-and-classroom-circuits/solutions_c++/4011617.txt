#include<bits/stdc++.h>
using namespace std;
int main()
{
     int  t;
     cin>>t;
     while(t--)
     {
           int n;
           cin>>n;
           string s;
           cin>>s;


           int count1=0;
           int count2=0;
           for(int i=0;i<s.length();i++)
           {
                  if(i%2!=0)
                  {
                        if(s[i]!='B')
                        {
                             count1++;
                        }
                        if(s[i]!='G')
                        {
                          count2++;
                        }
                  }
                  else
                  {
                        if(s[i]!='G')
                        {
                             count1++;
                        }
                        if(s[i]!='B')
                        {
                             count2++;
                        }
                  }
           }

           int ans=min(count1/2,count2/2);
           cout<<ans<<endl;


     }
     return 0;

}
