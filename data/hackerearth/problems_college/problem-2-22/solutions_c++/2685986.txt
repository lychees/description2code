#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t,r,x,y,a,b,i,flag;
    int l;
    cin>>t;
    while(t--){
    	i=0;
    	 	flag=0;
    	cin>>r>>x>>y;
    	for(i=0;i<4;i++)
    	{
    		cin>>a>>b;
    		l = (x-a)*(x-a)+(y-b)*(y-b);
    		//cout<<l<<"\n";
    		if(l>r*r)
    		flag=1;
    	}
    	if(flag==0)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
