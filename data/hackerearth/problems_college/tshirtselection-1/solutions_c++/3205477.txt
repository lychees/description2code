#include <iostream>
//#include<conio.h>
#include<map>
using namespace std;
#define magic 1000000007
typedef unsigned long long int A;

map<A, int> mymap; 
map<A, int>::iterator  iter;
void solve()
{
  mymap.clear();
  int n ;
  cin>>n;
  
  A x;
  
  int currlen = 0;
  int max = 0;
  
  int currst = 1;
  
  int ansst;

 
  for(int i = 1; i<= n ; ++i)
   {
          cin>>x;
          
          iter = mymap.find(x);
          if(iter == mymap.end() || iter->second < currst)
          {
                  ++currlen;                  
          }
          else
          {
              if(currlen > max)
               {
                    ansst = currst; 
                    max = currlen;
               }
               
              int oldpos = iter->second;
              currst = oldpos+1;
              currlen = i -currst+1;
                        
          }
          
         
          
          mymap[x] = i;
   }
   
  if(currlen > max)
          {
                    ansst = currst; 
                    max = currlen;
          }
  cout<<ansst<< " " <<ansst+max-1<<"\n";
  

}
     

int main()
{
    
  int t;
  cin>>t;
  
  for(int i = 0 ;i<t; ++i)
  solve();
  
  //getch();
  return 0;
}
