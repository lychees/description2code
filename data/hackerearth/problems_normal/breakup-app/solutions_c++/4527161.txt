#include <bits/stdc++.h>
using namespace std;
int isDigit(char c){
	if(c>48 && c<58)
		return 1;
		else 
		return 0;
}
int main()
{
    int n;
    cin>>n;
    char chat[1001];
    int i,j;
    int a[31];
    	for(i=1;i<=31;i++){
    		a[i]=0;
    	}
    int weight;
    while(n--){
    	cin.clear();
        cin.ignore();
    	cin.getline(chat, sizeof(chat));
    	if(chat[0]=='G')
    	weight=2;
    	else
    	weight=1;
    	for(i=3;chat[i]!='\0';++i){
    		if(isDigit(chat[i]) && isDigit(chat[i+1])){
    		a[atoi(&chat[i])]+=weight;
    		i++;}
    		else if(isDigit(chat[i]) && !isDigit(chat[i+1]))
    		a[atoi(&chat[i])]+=weight;
    	}
    }
    int max=0;
    int d=0;
    for(i=1;i<=31;i++){
    	if(a[i]>max){
    	max=a[i];
    	d=i;}
    }
    if(d==19 || d==20)
    cout<<"Date"<<endl;
    else
    cout<<"No Date"<<endl;
    
    return 0;
}
