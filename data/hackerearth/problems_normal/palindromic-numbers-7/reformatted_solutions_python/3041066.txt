def check(num):
    rev = 0
    tmp = num
    while (tmp > 0):
        rem = tmp % 10
        rev = 10*rev+rem
        tmp = tmp/10
    if num == rev:
        return True
    else:
        return False


n = int(raw_input())
# print n
for i in range(n):
    a, b = map(int, raw_input().split())
    # print a,b
    j = a
    count = 0
    while j <= b:
        if check(j):
            count = count+1
        j = j+1
    print count
