#include <iostream>
using namespace std;

int main()
{
    long long int n,t,x,y;
    long long int m,a[50],i,j,c=0;
    
    cin>>t;
    while(t--)
    {
    	for(i=0;i<50;i++)
    	a[i]=0;
    	cin>>x>>y>>n;
    	for(i=0;i<x;i++)
    	a[i]=y;
    	for(i=x;i<n;i++)
    	{
    		for(j=1;j<=x;j++)
    		a[i]+=a[i-j];
    	}
    	cout<<a[n-1]<<"\n";
    	
    }
}
