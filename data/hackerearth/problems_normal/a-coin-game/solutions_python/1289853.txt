def prog():
	m = 10**6
	m+=1
	re = [-1]*m
	count = 0
	for i in xrange(m):
		if(re[i]==-1):
			x = i+count
			re[i] = x
			if(x<m):
				re[x] = i
			count+=1
	t = input()
	while(t):
		t-=1
		inp = raw_input().split()
		a = int(inp[0])
		b = int(inp[1])
		if(re[a]==b):
			print "Don't Play"
		else:
			print "Play"
prog()
