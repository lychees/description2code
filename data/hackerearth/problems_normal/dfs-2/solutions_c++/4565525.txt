typedef unsigned long long ull;

const int N = 300000;
const int INF = (int)1e7;
const double eps = 1e-6;
const long long MOD = 1e9 + 7;
const double PI = M_PI;

int n,q;
ll a[N];

void solve(){
	
	scanf("%d", &n);
	for(int i = 0; i < n; ++i)
		scanf("%lld", a + i);
	sort(a, a + n);
	scanf("%d", &q);
	while(q--){
		ll low,high;
		scanf("%lld%lld", &low, &high);
		if(a[0] > high || a[n - 1] < low){
			printf("0\n");
		}
		else{
			int l = -1; int r = n - 1;
			while(r - l > 1){
				int m = (l + r) >> 1;
				if(a[m] >= low)
					r = m;
				else
					l = m;
			}
			int L = r;
			l = 0; r = n;
			while(r - l > 1){
				int m = (l + r) >> 1;
				if(a[m] <= high)
					l = m;
				else
					r = m;
			}
			int R = l;
			int ans = R - L + 1;
			printf("%d\n", ans); 
		}
	}
	
}

int main(){
	std::ios::sync_with_stdio(false);
	//freopen("input.in","r",stdin);
	//freopen("output.out","w",stdout);
	
	int t = 1;
	//scanf("%d", &t);
	while(t--){
		solve();
	}	
	return 0;
}
