#include <bits/stdc++.h>
using namespace std;

int N;

int main()
{
	int T;
	scanf("%d",&T);
	for (int t=0;t<T;t++) {
		scanf("%d",&N);
		vector<int> a;
		for (int i=0;i<N;i++) {
			int x;
			scanf("%d",&x);
			a.push_back(x);
		}
		int cur = 0;
		for (int i=0;i<N;i++) {
			int x;
			scanf("%d",&x);
			cur^=x-1;
			cur^=a[i]-x;
		}
		printf("%s\n",(cur==0) ? "Second" : "First");
	}
}
