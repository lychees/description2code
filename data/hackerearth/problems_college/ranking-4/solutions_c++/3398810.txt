#include <iostream>
using namespace std;
#define ll long long
#include <vector>
#include <algorithm>

int main()
{	ll t;
	cin>>t;
	while(t--)
	{	ll n,q,i,fi,la,mi,c;
		cin>>n;
		vector<ll> z,y;
		for(i=0;i<n;++i)
		{	scanf("%lld",&q);
			z.push_back(q);
			y.push_back(q);
		}
		sort(z.begin(),z.end());
		for(i=0;i<n;++i)
		{	fi=0;
			la=n-1;
			mi=(fi+la)/2;
			while(fi<=la) 
			{	if(z[mi]<y[i])
         		fi=mi+1;    
      			else if(z[mi]==y[i]) 
      			{	c=1;
      				while(z[mi+c]==y[i])
      					++c;
					if(z[mi+1]!=y[i])
						c=1;
      				printf("%lld ",n-mi-c+1);
         			break;
      			}
      			else
         			la=mi-1;
 				mi=(fi+la)/2;
   			}
		}
		printf("\n");
	}
    return 0;
}
