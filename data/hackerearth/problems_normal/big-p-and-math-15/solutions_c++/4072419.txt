#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    ll A,B,max[1000000],min[1000000],i=0,j=0,k=0,l=0,c1=0,c2=0,p=0,q=0;
    int a[1000000],b[1000000],c[1000000],d[1000000];
    cin>>A>>B;
    ll m = A, n = B;
    while(m)
    {
    	j=m % 10;
    	if(j==5)
    		{b[i] = 6;
    		a[i] =j;
    		}
    	else if(j==6)
    		{a[i] = 5;
    		b[i] =j;}
    	else
    		{a[i] = j;
    		b[i] =j;}
    	m/=10;
    	i++;
    }
    while(n)
    {
    	k=n%10;
    	if(k==5)
    		{d[l] = 6;
    		c[l] =k;}
    	else if(k==6)
    		{c[l] = 5;
    		d[l] =k;}
    	else{
    		c[l] = k;
    		d[l] = k;
    		}
    	n = n/10;
    	l++;
    }
    //cout<<i<<l;
/*    for(ll f =0;f<i;f++)
    	cout<<a[f]<<" "<<b[f];
    for(ll f=0;f<l;f++)
    	cout<<c[f]<<" "<<d[f];*/
    if(i<l)
    {
    	for(ll h=i;h<=l;h++)
    	{
    		a[h]=0;
    		b[h]=0;
    	}
    	for(ll g=0;g<l;g++)
    	{
    		ll sum = a[g]+c[g]+c1;
    		c1 = sum/10;
    		min[p++]=sum%10;
    	}
    	if(c1)
    		min[p++] = c1;
    	for(ll g=0;g<l;g++)
    	{
    		ll sum = b[g] + d[g] + c2;
    		c2 = sum/10;
    		max[q++]=sum%10;
    	}
    	if(c2)
    		max[q++] = c2;
    	for(ll g=p-1;g>=0;g--)
    		cout<<min[g];
    	cout<<" ";
    	for(ll g=q-1;g>=0;g--)
    		cout<<max[g];
    }
    else{
    	for(ll h = l;h<i;h++)
    	{
    		c[h]=0;
    		d[h]=0;
    	}
    	for(ll g=0;g<i;g++)
    	{
    		ll sum = a[g] + c[g] + c1;
    		c1 = sum/10;
    		min[p++]=sum%10;
    	}
    	if(c1)
    		min[p++] = c1;
    	for(ll g=0;g<i;g++)
    	{
    		ll sum = b[g] + d[g] + c2;
    		c2 =sum/10;
    		max[q++]=sum%10;
    	}
    	if(c2)
    		max[q++] = c2;
    	for(ll g = p-1; g>=0;g--)
    		cout<<min[g];
    	cout<<" ";
    	for(ll g =q-1;g>=0;g--)
    		cout<<max[g];
    }
    
    return 0;
}
