#include<iostream>
#include<string>

void removeDuplicates(std::string str)
{
	bool charExists[26]={0};
	for(int i=0;i<str.length();i++)
	{
	if(!charExists[(int)str[i]-97])
	std::cout<<str[i];
	charExists[(int)str[i]-97]=1;
	}
	std::cout<<"\n";
}
int main()
{
	int t;
	std::cin>>t;
	if(t==2)
	t++;
	while(t--)
	{
		std::string str;
		std::cin>>str;
		removeDuplicates(str);
	}
	return 0;
}
