#include<bits/stdc++.h>

using namespace std;

bool comp(const pair<int , string>& a, const pair<int , string>& b)
{
    return a.first > b.first;
}

int main()
{
    int flag=0;
    vector<pair<string, int> >pii;
    vector<pair<int,string> > fii;
    int n,val;
    string str;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {   flag=0;
        cin>>str>>val;
       vector<pair<string,int> >:: iterator iter=pii.begin();
       vector<pair<int,string> >:: iterator it=fii.begin();
            while(it!=fii.end())
            {
                string s1;
                s1=it->second;
                if(s1 == str)
                    break;
                else it++;
            }
            while(iter!=pii.end())
            {
                string s1;
                s1=iter->first;
                if(s1 == str)
                    {  flag=1; break;}
                else iter++;
            }
            if(flag==1){ //cout<<"sdsd\n";
            val += iter->second;
            fii.erase(it);
            pii.erase(iter);
            }


        pii.push_back(make_pair(str,val));
        fii.push_back(make_pair(val,str));
        sort(fii.begin(),fii.end(),comp);
    }

    //cout<<pii.size()<<" "<<fii.size()<<endl;
    vector<pair<int,string> >:: iterator itr = fii.begin();

    cout<<itr->second<<endl;
    return 0;
}
