
#include <iostream>
using namespace std;

int main()
{long int n,arr[100];
long int siz,index=0,r;

cin>>n;
cin>>r;

for(int i=0;i<n;i++)
arr[i]=i+1;
siz=n;

while(siz!=1)
{
	index=((index+r)%siz)-1;


  if(index==-1)
    index=siz-1;

	siz--;

	for(long int i=index;i<siz;i++)
	arr[i]=arr[i+1];

	if(index==siz)
	index=0;

}

cout <<arr[0];

    return 0;
}
