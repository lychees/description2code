
t = int(raw_input())

for _ in range(t):
    n = int(raw_input())

    s = []
    a = []

    for i in range(n):
        data = int(raw_input())
        s.append(data)
    for i in range(n):
        data = int(raw_input())
        a.append(data)

    s.sort()
    a.sort()

    for i in range(n):
        if s[i] < a[i]:
            break

    if i == n-1:
        print("YES")
    else:
        print("NO")
