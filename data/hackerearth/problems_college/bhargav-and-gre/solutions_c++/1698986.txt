#include <stdio.h>
#include <iostream>
using namespace std;
int main ()
{
int n,syn=0,ant=0;
string reference,words[100];
cin>>reference;	
cin>>n;
for (int i=0;i<n;i++)
{
cin>>words[i];
if (words[i][0]==reference[0]+3 && words[i].length()==reference.length())
   ant++;
if (words[i][0]==reference[0] && words[i].length()==reference.length() )
   syn++;
}
cout<<syn<<" "<<ant;
}
