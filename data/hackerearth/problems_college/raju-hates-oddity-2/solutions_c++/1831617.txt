#include <iostream>
#include <cstdio>
#include <string.h>
#include <math.h>
#include <cstdlib>
#include <vector>
#include <map>
#define ll long long int

using namespace std;

int main()
{
	ll t;
	
	cin >> t;
	
	ll arr[1000];
	arr[1] = 1;
	arr[2] = 2;
	
	for (ll i = 3; i <= 50; i++) {
		arr[i] = arr[i - 1] + arr[i - 2];
	}
	
	while (t--) {
		ll n;
		cin >> n;
		
		cout << arr[n] << endl;
		
		
	}
	
	return 0;
}
