#include <iostream>
#include <list>
using namespace std;
int g[1001][1001]={0};
int luck[1001]={1000000};
int visited[1005]={0};
void bfs(int s,int n)
{
	list<int> l;
	l.push_back(s);
	visited[s]=1;
	while(!l.empty())
	{
		int a=l.front();
		l.pop_front();
	//cout<<a<<" ";
		for(int i=0;i<n;i++)
		{
			//	cout<<g[a][i]<<" ";
			if(visited[i]==0&&g[a][i]==1)
			{
					visited[i]=1;
	//	cout<<i<<" ";
				luck[i]=min(luck[i],luck[a]+1);l.push_back(i);
					//	cout<<luck[i]<<" "<<endl;
			}
		}
	//	cout<<endl;
	}
}
int main() {
	int n,m;
	cin>>n>>m;
while(m--)
{
	int a,b;
	cin>>a>>b;
	g[a][b]=1;
	g[b][a]=1;
}

luck[0]=0;
visited[0]=0;
for(int i=1;i<n;i++)
{luck[i]=1000000;
visited[i]=0;
}

for(int i=0;i<n;i++)
{
if(visited[i]==0)
{
	bfs(i,n);
}
}
for(int i=1;i<n;i++)
cout<<luck[i]<<endl;
	
	
	return 0;
}
