#include <iostream>
#include<utility>
#include<map>
#include<queue>
using namespace std;
#define ll long long
int main()
{
   ll n;
   cin>>n;
   pair<ll,ll> p;
   priority_queue< pair<ll,ll> > mp;
  	map<ll,ll> m;
  	map<ll,ll>::iterator it;
   for(ll i=0;i<n;i++)
   {
   		ll id,old,po,l,c,s,neww;
   		cin>>id>>old>>po>>l>>c>>s;
   		neww=(po*50)+(l*5)+(c*10)+(s*20);
   		ll change=neww-old;
   		p=make_pair(change,id);
   		mp.push(p);  
   		m[id]=neww;
   }
   for(ll i=0;i<5;i++)
   {
   		p=mp.top();
   		it=m.find(p.second);
   		cout<<p.second<<" "<<it->second<<endl;
   		mp.pop();
   }
   
}
