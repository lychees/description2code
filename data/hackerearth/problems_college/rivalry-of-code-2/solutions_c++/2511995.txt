#include <iostream>
#include<climits>
using namespace std;

int main() {

int t; //Number of test case

cin>>t;

   while(t--)      //Loop till the test case exists
{

    string x,y;   //Accepting strings

        cin>>x;

            cin>>y;

            int ct=0;     //declaring count variable

            int maxi=INT_MAX;

            if(x.size()==y.size())   
    {

               for(int i=0;i<x.size();i++)  
        {

                        if(x[i]!=y[i])

                            ct++;                   
        }

                       maxi=ct;
    }

           else
    {

                    if(x.size()>y.size())      
        {

                        string temp=x;

                           x=y;

                           y=temp;

        }

                    for(int i=0;i<=(y.size()-x.size());i++)
        {

                              ct=0;

                            for(int j=0;j<x.size();j++)
            {


                                if(x[j]!=y[j+i])


                                    ++ct;
            }

                           if(ct<maxi)

                                maxi=ct;
        }
    }

           cout<<maxi<<"\n";
}

return 0;

}
