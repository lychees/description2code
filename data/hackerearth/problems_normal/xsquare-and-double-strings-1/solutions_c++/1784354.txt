#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t,flag=0,len,i,j;
	char str[101];
  cin>>t;
  while(t--)
  {
  	flag=0;
  	cin>>str;
  	int len=strlen(str);
  	for(i=0;i<len;i++)
  	{
  		
  		for(j=i+1;j<len;j++)
  		{
  			if(str[i]==str[j])
  			{
  				flag=1;
  				break;
  			}
  		}
  	}
  	
  	if(flag==1)
  	{
  		cout<<"Yes"<<"\n";
  	}
  	else
  	{
  		cout<<"No"<<"\n";
  	}
  	
  }
  return 0;
}
