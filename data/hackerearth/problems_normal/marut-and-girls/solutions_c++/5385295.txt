#include<bits/stdc++.h>
using namespace std;
main()
{
int n;
cin>>n;
int ar[100002];
for(int i=1;i<=10001;i++)
ar[i]=0;
for(int i=0;i<n;i++)
{ int a;
cin>>a;
ar[a]++;
}
int m;
cin>>m;
int count=0;

for(int i=0;i<=m;i++)
{  char str[100002];
   gets(str);
   int num=0;
   int ans=0;
   for(int i=0;i<strlen(str);i++)
   {
     if(str[i]==' ')
      {
        if(ar[num]>0)
        ans++;
        num=0;
      }
     else
     {
       int a=str[i];
       num=num*10+(a-48);
     }

   }
   if(ar[num]>0)
    ans++;
if(ans==n)
  count++;
}
cout<<count;

}
