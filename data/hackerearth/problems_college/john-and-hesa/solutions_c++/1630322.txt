#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
        {
        int x,p;
        cin>>x>>p;
        int n=0;
        while(x!=1 && x%p==0)
            {
            n++;
            if(x==p)
                {
                x=0;
                break;
            }
            x=x/p;
        }
        cout<<n<<" "<<x<<"\n";
    }
    return 0;
}
