#include <bits/stdc++.h>
using namespace std;
typedef long long int lli;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int n,e,v;
    	scanf("%d %d %d", &n, &e, &v);
    	lli drinks=0;
    	for(int i=0;i<n;i++){
    		int enemy;
    		scanf("%d", &enemy);
    		//cout<<" Energy before fight : "<<e<<endl;
    		if(e<=enemy){
    			int portion = 1+(enemy-e)/v;
    			//int power = portion;//==0?1:portion;
    			//cout<<"portion= "<<portion<<" , need drinks = "<<power<<endl;
    			drinks+=portion;
    			e+=v*portion;
    		}
    		e-=enemy;
    		//cout<<" Energy after fight : "<<e<<endl;
    	}
    	cout<<drinks<<endl;
    }
    return 0;
}
