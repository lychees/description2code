int is_prime(int x)
{
    if(x < 2) return 0;
    else if(x == 2) return 1;
    else if(x % 2 == 0) return 0;
    for(int i = 3; 1LL*i*i <= x; i++)
        if(x%i == 0)
            return 0;
    return 1;
}
int main()
{
    int t;
    cin(t);
    while(t--)
    {
        int p , q , s;
        cin(p);
        cin(q);
        cin(s);
        if(p%2 == 0)
            printf("Havanese\n");
        else
        {
           // p %= 2;
            if(q<2*s)
                printf("Havanese\n");
            else
            {
                int win = 0;
                if(q%s == 0)
                    win = 1;
                else
                {
                    for(int div = 2; div <= 40000 && win == 0; div++)
                    {
                        if(q%div) continue;
                        int f = q/div;
                        if(f < s) continue;
                        if(f < (2*s)) win = 1;
                        if(is_prime(f)) win = 1;
                    }
                }
                    if(win)
                        printf("Bernard\n");
                    else
                        printf("Havanese\n");
            }
        }
    }
    return 0;
}
