pa = []

n = int(raw_input())
ans = 0


def isprime(x):
    it = 2
    while it*it <= x:
        if x % it == 0:
            return 0
        it += 1
    return 1


for i in range(2, n+1):
    if isprime(i):
        pa.append([i, 1])
    else:
        for j in pa:
            if i % j[0] == 0:
                j[1] += 1
    tmp = 1
    for j in pa:
        tmp *= 2
    ans += tmp

print ans/2
