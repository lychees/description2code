def sosod(val):
        return sum(int(c) ** 2 for c in str(val))
num=input()

for i in range(1,num+1):
        n=input()
        list1=[]
        for j in range(0,10):
                list1.append(sosod(n))
                n=sosod(n)
        if (1 in list1) or (10 in list1) or (13 in list1) or (19 in list1) or (82 in list1) or (100 in list1):
                print "Case #%d:YES" %i
        else:
                print "Case #%d:NO" %i
