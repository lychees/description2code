#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long int t,i;
   	map <string,int> my;
   	string a;
   	
   	cin>>t;
   	for(i=0;i<t;i++){
   		cin>>a;
   		if(my.find(a)==my.end())
   			my[a]=1;
   		else my[a]++;
   	}
   	
   	map <string,int>::iterator it;
   	for(it = my.begin();it!=my.end() ;it++){
   		cout<<(*it).first<<" "<<(*it).second<<endl;
   	}
   	
    return 0;
}
