#include<iostream>
#include<algorithm>
#include<map>
#include<vector>
using namespace std;
bool comp (const pair<int,int>& i1 ,const pair<int,int>& i2){
	if(i1.first==i2.first){
		return i1.second>i2.second;
	}
   else{
   return i1.first<i2.first;
   }

}

int main(){
	int cases,i;
	cin>>cases;
	string str;
	while(cases--){
	cin>>str;
	vector<int> arr(26,0);
	for(i=0;i<str.length();i++){
	arr[str[i]-'a']=arr[str[i]-'a']+1;	
	}	
    
   	
	
	
	vector< pair<int,int> > xrr;
	for(i=0;i<26;i++){
	xrr.push_back(std::make_pair(arr[i],i));	
	}
	
	sort(xrr.begin(),xrr.end(),comp);
	
	for(i=0;i<26;i++){
	cout<<char('a'+xrr[i].second)<<" ";	
	}
	
	cout<<endl;
	
}
	
	return 0;
}
