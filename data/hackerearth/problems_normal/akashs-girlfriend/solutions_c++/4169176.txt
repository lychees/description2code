#include <iostream>
using namespace std;

using ull = unsigned long long;
constexpr ull mod = 1000000009ull;
constexpr ull i2 = 500000005ull;

ull powm(ull a, ull n) {
	ull res = 1;
	while (n) {
		if (n % 2)
			res = (res * a) % mod;
		a = (a * a) % mod;
		n /= 2;
	}
	return res;
}

int main() {
	int t;
	cin >> t;
	for (int i = 0; i < t; ++i) {
		ull n;
		cin >> n;
		ull res = (i2 * (powm(10, n) + (mod - powm(8, n)))) % mod;
		cout << res << '\n';
	}
    return 0;
}
