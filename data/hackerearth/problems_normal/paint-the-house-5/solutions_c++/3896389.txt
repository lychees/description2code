

#include <iostream>
#include <algorithm> 
using namespace std;

struct Worker
{
	int start;
	int cost;
	int speed;
};


bool compareTo(Worker a , Worker b)
{
	if(a.start==b.start)
	{
		if(a.speed == b.speed)
		 return a.cost < b.cost;
	
	return (a.speed > b.speed) ;
	}		
	return a.start<b.start;
}

int main()
{
	
	int n;
	
	long long int area=0 , area_covered=0;
   cin>>n>>area;
    
   Worker w[n];
   
   for(int i=0;i<n;i++)
    cin>>w[i].start>>w[i].cost>>w[i].speed;
    
	sort(w,w+n,compareTo);
	
	int bestspeed = 0;
	long long int cost = 0;
	
	for(int i=0;i<n;i++)
	{
		area_covered += bestspeed*(w[i].start - w[i-1].start) * 1LL;
		
		if(area_covered >= area)
			break;
		
		if(w[i].speed > bestspeed)
		{
			bestspeed = w[i].speed;
			cost += w[i].cost;
		}
		
		
	}
	
   cout<<cost;
   
  
  

  
    return 0;
}
