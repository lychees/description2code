#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	unsigned long long int n,k,i,l,m;
    	cin>>n;
    	if(n==1 || n==2)
    	cout<<1<<endl;
    	else
    	{
    		i=1;
    		while(true)
    		{
    			k = 2 + pow(2,i+1)*(i-1) + i + (i-1);
    			l = 2 + pow(2,i)*(i-2) + 2*(i-1)+1;
    			if(n<=k && n>=l)
    			{
    				m = 2*i -1;
    				break;
    			}
    			else if(n==k+1)
    			{
    				m = 2*i - 1 + n - k;
    				break;
    			}
    			//else continue;
    			i++;
    		}
    		cout<<m<<endl;
    	}
    	
    }
    return 0;
}
