#include<bits/stdc++.h>
using namespace std;
const long long NEG = (long long) 1e16 + 5;
int n,k;
vector <int> vec;
long long dp[10005];


long long solve(int idx) {
    if (idx >= n) return 0;
    if (dp[idx] != -1) return dp[idx];
    long long tmp = -NEG;
    tmp= max (tmp, solve(idx + 1));
    tmp = max (tmp, vec[idx] + solve (idx + k + 1));
    return dp[idx] = tmp;
}

int main () 
{
    int t,x;
   	cin>>t;
    while (t--)
	{
        scanf ("%d %d", &n, &k);
        memset(dp, -1, sizeof(dp));
        vec.clear();
        for (int i = 0; i < n; i++) {
          
            scanf("%d",&x);
            vec.push_back(x);
        }
        long long res;
        res=solve(0);
        printf("%lld\n",res);
    }

    return 0;
}
