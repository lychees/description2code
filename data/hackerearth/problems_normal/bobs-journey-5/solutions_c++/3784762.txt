#include <iostream>
#include <string>
#include <vector>
#include <set>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	vector<string> v;
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++){
    		string s;
    		cin>>s;
    		v.push_back(s);
    	}
    	vector<char> v1;
    	for(int i=0;i<v.size();i++)
    	v1.push_back(v[i][0]);
    	//for(int i=0;i<v1.size();i++)
    	//cout<<v1[i]<<' ';
    	//cout<<endl;
    	set<char> v2(v1.begin(),v1.end());
    	if(v1.size()==v2.size())
    	cout<<"YES"<<endl;
    	else
    	cout<<"NO"<<endl;
    }
    return 0;
}
