#include<bits/stdc++.h>
using namespace std;
const int N=1e6;
int a[N+4],b[N+4],bit[N+3];
int n;
int query(int at)
{
    int sum=0;
    for(;at>=1;at-=at&(-at))
        sum+=bit[at];
    return sum;
}
inline void upd(int at,int by)
{
    for(;at<=n;at+=at&(-at))
        bit[at]+=by;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
        scanf("%d",a+i);
        b[i]=a[i];
    }
    sort(b,b+n);
    for(int i=0;i<n;++i)
    {
        int idx=int(lower_bound(b,b+n,a[i])-b); 
        a[i]=idx+1;
    }
    int ans=0;
    for(int i=0;i<n;++i)
    {
        ans=query(a[i]-1);
        printf("%d ",ans);
        upd(a[i],1);
    }
    puts("");
return 0;
}
