#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,k;
	cin>>n>>k;
	int arr[n];
	for(int i=0;i<n;i++)
	{
	cin>>arr[i];
	}
	sort(arr,arr+n);
	int j=k-1;
	int count[100000]={0};
	int m=0;
	for(int i=0;i<n-1;i++)
	{
		count[m]=arr[j]-arr[i];
		m++;
		j++;
		if(j==n)
		break;
	}
	int min=*min_element(count,count+m);
	cout<<min<<endl;
	
}
