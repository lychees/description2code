#include<bits/stdc++.h>
using namespace std;
#define max 1005
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k,p,i,j;
		scanf("%d%d",&n,&k);
		int r[max]={0},c[max]={0},sr[max],sc[max];
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{	scanf("%d",&p);
				r[i]+=p;
				c[j]+=p;
				//printf("%d %d\n",r[i],c[j]);
			}
		}
		/*sort(r,r+n);
		sort(c,c+n);
		sr[0]=r[0];
		sc[0]=c[0];*/
		for(i=1;i<=k;i++)
		{
			sort(r,r+n);
			int minr=r[0];
			
			sr[i]=sr[i-1]+minr;
			r[0]+=n;
			//printf("%d\n",sr[i]);
		}
		for(i=1;i<=k;i++)
		{
			sort(c,c+n);
			int minc=c[0];
			
			sc[i]=sc[i-1]+minc;
			c[0]+=n;
			//printf("%d\n",sc[i]);
		}
		int ans=INT_MAX;
		for(i=0;i<=k;i++)
		{
			ans=min(ans,sr[i]+sc[k-i]+i*(k-i));
		}
		printf("%d\n",ans);
	}
}
