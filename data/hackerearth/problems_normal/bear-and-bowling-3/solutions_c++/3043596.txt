#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
long long power(long long n, long long m)
{
    if(m==0) return 1;
    long long x=power(n,m/2);
    if(m%2==0) return (x*x)%MOD;
    else return (((x*x)%MOD)*n)%MOD;
}
int main()
{
	int n;
	scanf("%d",&n);
	long long a[n];
	int i,j;
	long long ans=0;
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	for(i=0;i<n;i++)
	{
		long long x=power(2,n-i-1);
		long long y;
		if(i==0)
		y=1;
		else
		y=(power(2,i-1)*(i+2))%MOD;
		x=(x*y)%MOD;
		x=(x*a[i])%MOD;
		ans=(ans+x)%MOD;
	}
	printf("%lld\n",ans);
	return 0;
}
