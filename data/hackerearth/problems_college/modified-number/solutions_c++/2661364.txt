#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll p,q,i,num,r,temp,n,l1,r1,c=0,val1,val2,flag,j;
    cin>>p>>q;
    flag=0;
    for(j=p;j<=q;j++)
    {
    	num=j*j;
    	vector<int> v1,v2;
    	temp=num;
    	n=0;
    	while(temp>0)
    	{
    		n++;
    		temp/=10;
    	}
    	if(n%2)
    		r1=(n/2)+1;
    	else
    		r1=n/2;
    	l1=n-r1;
    	temp=num;
    	c=0;
    	while(temp>0)
    	{
    		r=temp%10;
    		if(c<r1)
    			v1.push_back(r);
    		else
    			v2.push_back(r);
    		c++;
    		temp/=10;
    	}
    	val1=val2=0;
    	for(i=r1-1;i>=0;i--)
    	{
    		val1=(val1*10)+v1[i];
    	}
    	for(i=l1-1;i>=0;i--)
    	{
    		val2=(val2*10)+v2[i];
    	}
    	if(val1+val2==j)
    	{
    		printf("%lld ",j);
    		flag=1;
    	}
    		
    }
    if(!flag)
    	printf("INVALID RANGE");
    return 0;
}
