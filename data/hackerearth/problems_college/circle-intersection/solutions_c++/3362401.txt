#include <bits/stdc++.h>
using namespace std;

int main()
{
    double x1,y1,x2,y2,r1,r2,val;
    cin>>x1>>y1>>r1;
    cin>>x2>>y2>>r2;
    val=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
    if(val==r1+r2)
    {
    	printf("T");
    }
    else if(val>r1+r2)
    	printf("N");
    else
    	printf("Y");
    return 0;
}
