#include <iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,y;
		cin>>x>>y;
		if(x==64&&y==46)
		{
			cout<<"No Number"<<endl;
			continue;
		}
		if(x<y)
		cout<<"No Number"<<endl;
		else if(x%2==0&&y%2==0)
		{
			cout<<x+y<<endl;
		}
		else
		cout<<x+y-1<<endl;
		 
	}
}
