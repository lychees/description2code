#include <bits/stdc++.h>
#define ll long long
using namespace std;
int mod=(1e9)+7;
ll func(ll n,ll b)
{
	ll z=1;
	while(b>0)
	{
		if(b%2)
			z=(z*n)%mod;
		n=(n*n)%mod;
		b/=2;
	}return z;
}
int main()
{
    int t;cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
    	ll z=func(2,n);
    	cout<<--z<<endl;
    }
    return 0;
}
