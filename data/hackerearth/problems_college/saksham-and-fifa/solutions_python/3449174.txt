#print 'Hello World!'
t=input()
ma,i=0,0
while t>0:
	t-=1
	m,n=map(int,raw_input().split())
	if m>ma:
		ma=m
		i=n
if i>0:
	print"Left Corner"
elif i<0:
	print"Right Corner"
else:
	print"Either Corner"
