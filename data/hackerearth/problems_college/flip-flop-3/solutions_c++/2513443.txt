#include<iostream>
using namespace std;
#include<string.h>
int main()
{
	int t,i,n;
	long int ctr;
	scanf("%d",&t);
	while(t--)
	{
		ctr=0;
		char s[100000];
		scanf("%s",s);
		for(i=0;s[i]!='\0';i++)
		{
			if(s[i]==s[i+1])
			ctr++;			
		}
		printf("%ld\n",ctr);
	}
		
return 0;
}
