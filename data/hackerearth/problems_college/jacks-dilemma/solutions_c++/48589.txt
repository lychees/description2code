#include "iostream"
#include "cstdio"
#include "string"

using namespace std;

int main(int argc, char const *argv[])
{
	int t,n,j,l,lift,stair,downTime,upTime;
	char dir;
	scanf("%d",&t);
	while(t>0)
	{
		scanf("%d",&n);
		scanf("%d",&j);
		scanf("%d",&l);
		cin>>dir;
		stair=j*10;
		if (dir=='U')
		{
			upTime=(n-l)*5 + 10;
			downTime=(n-j)*5 + 10 + j*5 + 10;
			if (j==n)
			{
				downTime-=10;
			}
			lift=downTime + upTime;
		
		}
		else if (dir=='D')
		{
			if (j<=l)
			{
				downTime=(l-j)*5 + 10 + j*5 + 10;
				lift=downTime;
			}
			else
			{
				downTime=l*5 + 10;
				upTime = n*10 + 10 + 10 + 10 ;
				lift=downTime+ upTime;
			}
			
		}
		if (stair<lift)
		{
			printf("Stairs %d\n", stair);
		}
		else
			printf("Lift %d\n", lift);
		t--;
	}
	return 0;
}
