    #include <iostream>
    using namespace std;
    int main()
    {
	    long long int t,n,x,odd,count,even;
	    cin>>t;
	    while(t--)
	    {
		    cin>>n;
		    odd=0ll;
		    even=0ll;
		    for(long long int i=0;i<n;i++)
		    {
			    cin>>x;
			    if(x%2!=0)
			    odd+=1;
			    else
			    even+=1;
		    }
		    count=odd*even;
		    cout<<count<<endl;
	    }
   		 return 0;
    }
