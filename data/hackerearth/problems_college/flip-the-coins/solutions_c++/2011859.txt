#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int q;
    cin>>q;
    int type,a,b;
    char v[101];
    for(int i=0;i<101;i++)
    {
    	v[i]='0';
    }
    int count=0;
    while(q--)
    {
    	count=0;
    	cin>>type;
    	if(type==0)
    	{
    		cin>>a>>b;
    		for(int i=a;i<=b;i++)
    		{
    			v[i]='1';
    		}
    	}
    	else
    	{
    		cin>>a>>b;
    		for(int i=a;i<=b;i++)
    		{
    			if(v[i]=='1')
    			{
    				count++;
    			}
    		}
    		cout<<count<<endl;
    	}
    	
    }
    
}
