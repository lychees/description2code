#include<iostream>
using namespace std;
int arr[26];
int main()
{
	string str;
	int test;
	cin>>test;
	while(test--)
	{
		for(int i=0;i<26;i++)
		arr[i]=0;
		cin>>str;
		int x=str.length();
		for(int i=0;i<x;i++)
		arr[str[i]-'a']++;
		for(int i=0;i<26;i++)
		{
			while(arr[i]>1)
			arr[i]-=2;
		}
		int count=0;
		for(int i=0;i<26;i++)
		{
			if(arr[i])
			count++;
		}
		int ans=0;
		for(int i=0;i<26;i++)
		{
			ans+=arr[i];
		}
		if(ans>0)
		ans--;
		cout<<ans<<endl;
	}
}
