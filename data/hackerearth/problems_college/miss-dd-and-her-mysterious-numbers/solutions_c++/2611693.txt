#include <bits/stdc++.h>
using namespace std;
int hasha[10000005];
int seive[10000005];
int main()
{
    memset(hasha,0,sizeof(hasha));
    int n,q,k;
    scanf("%d%d",&n,&q);
    for(int i=2;i<=n;i+=2)
    {
        seive[i]=1;
    }
    for(int i=3;i<=n;i+=2)
    {
      if(!seive[i])
        {
        for(int j=i;j<=n;j+=i)
            seive[j]++;
        }
     }
  for( int i=2; i <= n; ++i )  {
      hasha[seive[i]]++;
  }
    while(q--)
    {
    	scanf("%d",&k);
    	if(k>n)
    	printf("0\n");
    	else
    	printf("%d\n",hasha[k]);
    }
    return 0;
}
