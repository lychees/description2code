#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	priority_queue<ll int,vector<ll int>,greater<ll int> > p;
	p.push(4);p.push(7);
	ll int t;
	vector<ll int> v;
	while(1)
	{
		t=p.top();
		p.pop();
		v.push_back(t);
		double x = ((double)(LONG_MAX - 4))/10.0;
		if(t>x)
		break;
		p.push(t*10+4);
		p.push(t*10+7);
	}
	//~ cout<<v.size()<<endl;
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int x,y,ans;
		cin>>x>>y;
		vector<ll int>::iterator it1,it2;
		it1=lower_bound(v.begin(),v.end(),x);
		if(it1==v.end())
		ans=0;
		else
		{
			if(*it1>y)
			ans=0;
			else
			{
				it2=upper_bound(v.begin(),v.end(),y);
				ans=it2-it1;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
