#include <iostream>
#define M 1000000007
using namespace std;
 
int main()
{
    long t,a,b,c,d,i,n,m;
    cin>>t;
    long long arr[1005][1005];
    for (int i = 0; i <1005; i++)
  		for (int j = 0; j < 1005; j++)
  			arr[i][j] = 1;
	for(i=1;i<1005;i++)
	{
		for(int j=1;j<1005;j++)
		{
			arr[i][j]=((arr[i-1][j]+arr[i-1][j-1])%M+arr[i][j-1])%M;
		}
	}
    while(t--)
    {
    	cin>>a>>b>>c>>d;    
    	cout<<arr[c-a][d-b]<<endl;
    }
    return 0;
}
