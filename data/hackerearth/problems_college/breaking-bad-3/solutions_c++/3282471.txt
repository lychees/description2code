#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

vector<int> V;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		int n;
		cin >> n;
		V.clear();
		for(int i = 0;i < n;i++) {
			int val;
			cin >> val;
			V.push_back(val);
		}
		sort(V.begin(), V.end());
		long long ans = 0ll;
		for(int i = 0;i < V.size();i+=2) {
			ans += (long long) V[i] * V[i+1];
		}
		cout << ans << "\n";
	}
    return 0;
}
