#include<iostream>
#include<algorithm>
# include<stdio.h>
# include<math.h>

using namespace std;

int main()
{
    int n,sum=0,i,j,k,a[4];
    scanf("%d",&n);

    for(i=0;i<3;i++)
        scanf("%d",&a[i]);
    sort(a,a+3);

    for(i=0;i<=n/a[0];i++)
    {
        for(j=0;j<=n/a[1];j++)
        {
            for(k=0;k<=n/a[2];k++)
                if(i*a[0]+j*a[1]+k*a[2]==n)
                    sum++;
        }
    }
    
    printf("%d\n",sum);
return 0;
}
