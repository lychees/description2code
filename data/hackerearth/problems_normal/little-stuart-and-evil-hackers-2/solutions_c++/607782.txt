#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000006
#define ll long long int
#define ull unsigned long long int
#define scan(x) scanf("%d",&x)
#define scanll(x) scanf("%lld",&x)
#define print(x) printf("%d\n",x)
#define printll(x) printf("%lld\n",x)

ll fact[200002];

void pre_process()
{
	fact[0]=1;
	for(int i=1;i<=200001;i++)
	{
		fact[i]=(fact[i-1]*i)%MOD;
	}
}

ll mpow(ll a, ll n)
{ll ret=1;ll b=a;while(n) {if(n&1)
ret=(ret*b)%MOD;b=(b*b)%MOD;n>>=1;}
return (ll)ret;}

ll modInverse(ll a){
    return mpow(a,500000001);
}

int main()
{
	int t;
	cin>>t;
	ll n,m,a,x,ans;
	pre_process();

	while(t--){
		scanll(m);
		scanll(n);

		a=(fact[2*n]*modInverse(fact[n]))%MOD;
		a=(a*modInverse(fact[n]))%MOD;		
		ans=mpow(a,m);
		printll(ans);

	}
	
	return 0;
}
