#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    long long int test_cases, K,M, N,cost,reqd;
   
    cin>>test_cases;
    while(test_cases>0)
    {
    	 cost=0;
    	 reqd=0;
    	cin>>N>>K>>M;
    	for(long long int j=1;j<=N;j++)
    	{
    		cost+=K*(j*j*j);	
    	}
    	if (cost>M){
    		reqd=cost-M;
    	}
    	cout<<reqd<<endl;
    	test_cases--;
    }
    return 0;
}
