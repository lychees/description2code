LL getpow(LL a,LL b,LL mod){
    LL ret=1LL;
    LL inter=a;
    while(b){
        if(b&1){
            ret *= inter;
            ret %= mod;
        }
        inter *= inter;
        inter %= mod;
        b = b>>1;
    }
    return ret;
}
