#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
int ans[500005],dp[27];
int main()
{
	int i,j,n;
	//freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
	int a,h,e,r,c,k,t;
	char s[500005];
	ss(s);
	n=strlen(s);
	for(i=0;i<n;i++)
        s[i]=tolower(s[i]);
    dp[(int)(s[0]-'a')]++;
    for(i=1;i<n;i++)
    {
        if(i<10){
        dp[(int)(s[i]-'a')]++;
        ans[i]=0;}

        else
        {
            dp[(int)(s[i]-'a')]++;
            if(i!=10)
            dp[(int)(s[i-11]-'a')]--;
            a=dp[(int)('a'-'a')];
            h=dp[(int)('h'-'a')];
            e=dp[(int)('e'-'a')];
            r=dp[(int)('r'-'a')];
            t=dp[(int)('t'-'a')];
            c=dp[(int)('c'-'a')];
            k=dp[(int)('k'-'a')];
            ans[i]=ans[i-1];
            if(a==2&&h==2&&e==2&&r==2&&t==1&&c==1&&k==1)
                ans[i]++;
        }
        //printf("%d\n",ans[i]);
    }
    int q,l;
    sd(q);
    while(q--)
    {
        sd(l);
        sd(r);
        printf("%d\n",max(0,ans[r-1]-ans[min(l+8,n-1)]));
    }
	return 0;
}
