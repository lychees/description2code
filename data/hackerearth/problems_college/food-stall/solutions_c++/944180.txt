#include<iostream>
#include<algorithm>
#include<vector>
#include<stdio.h>
using namespace std;
#define Max 100000
int main()
{
    //freopen("DATA.dat", "r", stdin);
    int t;
    cin>>t;
    while(t--){
        int n, i; long long d, count=0;
        cin>>n>>d;
        vector <long long> p(Max);
        for(i=0; i<n; i++)
            cin>>p[i];
        sort(p.begin(), p.begin()+n);
        for(i=0; i<n; i+=2)
            if(p[i+1]-p[i]<=d)
                count++;
        cout<<count<<endl;
    }
    return 0;
}
