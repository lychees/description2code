#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar//_unlocked

ll a[100005];
ll l[100005];
ll r[100005];

int main()
{
    ll t, n, k, i, j;
    ll ans;
    sl(t);
    while (t--) {
        sl(n);
        sl(k);
        a[0] = LONG_LONG_MAX;
        a[n+1] = LONG_LONG_MAX;
        for (i = 1; i <= n; ++i)
            sl(a[i]);
        for (i = 1; i <= n; ++i) {
            j = i-1;
            while (a[i] > a[j])
                j = l[j];
            l[i] = j;
            //cout << j << " ";
        }
        //cout << endl;
        for (i = n; i >= 1; --i) {
            j = i+1;
            while (a[i] >= a[j])
                j = r[j];
            r[i] = j;
        }
        ans = 0;
        for (i = 1; i <= n; ++i) {
            if (a[i] > k)
                ans += (r[i] - i) * 1LL * (i - l[i]);
        }
        printf("%lld\n", ans);
    }
    return 0;
}
