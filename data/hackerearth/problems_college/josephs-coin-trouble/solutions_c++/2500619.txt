void solve()
{
	int n;
	vector<int> a;
	cin>>n;
	string s;
	getline(cin, s);
	getline(cin, s);
	int cnt = 0, st = 0;
	for(int i=0;i<s.size();i++) {
		if(s[i] == ' ') {
			a.push_back(atoi(s.substr(st, i-st).c_str()));
			st = i+1;
		}
	}
	a.push_back(atoi(s.substr(st).c_str()));
	sort(a.begin(), a.end());
	//for(int i=0;i<a.size();i++)
	//	cout<<a[i]<<" ";
	//cout<<"\n";
	vector<ll> dp(n+1, 0);
	dp[0] = 1;
	for(int i=0;i<a.size();i++) {
		for(int j=a[i];j<=n;j++)
			dp[j] += dp[j-a[i]];
			
	}
	cout<<dp[n]<<"\n";
}
