#include <bits/stdc++.h>
using namespace std;

int main(void) {
    
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    
    int T = 0;
    cin >> T;
    while(T--) {
    	string S;
    	cin >> S;
    
    	int count = 0;
    	for(int i = 0; i < S.length(); ) {
    		if(S[i] == 'D' || S[i] == 'B')
    			++i;
    		else if(S[i] =='S')
    			i += 2;
    		else if(S[i + 1] == 'B')
    			i += 2;
    		else if(S[i + 2] == 'B' || S[i + 1] == 'S')
    			i += 3;
    		else if(i > 0 && S[i - 1] == 'S')
    			i += 1;
    		else
    			++count, ++i;
    	}
    	cout << count << "\n";
    }
    
    return 0;
}
