#include <iostream>
#include<algorithm>
#include<stdlib.h>
#include<cstdio>
#include<string.h>
using namespace std;
int compare(const void *a,const void *b)
{
	 return ( *(int*)a - *(int*)b );
}

int main()
{
	char str[100];
	cin>>str;
	int arr[100];
	int l=strlen(str);
	for(int i=0;i<l;i++)
	{
		arr[i]=str[i]-'a';
	}
	qsort(arr,l,sizeof(int),compare);
	//cout<<arr[str.size()]<<endl;
	for(int i=0;i<l;i++)
	{
	char ch=arr[i]+'a';
	cout<<ch;
	}
	cout<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
