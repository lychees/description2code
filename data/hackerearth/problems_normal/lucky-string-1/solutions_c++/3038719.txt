#include <iostream>
#define ll long long int

using namespace std ;

int main()
{
	long int t;
	ll n;
	cin >> t;
	while(t--)
	{
		ll twop = 1, pow = 0;
		ll k = twop*pow , prev, quo, rem, shift ;		
		cin >> n;
		
		while(n>k){
			prev = k ;
			twop = twop << 1; 
			pow++ ;
			k += twop*pow ;
		}
	
		n = n - prev ;
		rem = n%pow ;
		
		if(rem == 0){
			quo = (n/pow) - 1 ;
		}else{
			quo = n/pow ;
			shift = (pow-rem) ;
			quo = quo >> shift ;
		}
				
		if(quo&1)
			cout << "Earth\n" ;
		else
			cout << "Hacker\n" ;
		
	}
}
