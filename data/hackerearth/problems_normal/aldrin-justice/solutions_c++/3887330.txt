

    #include <bits/stdc++.h>
    using namespace std;
    int main ()
    {
    	int tc;
    	cin >> tc;
    	while (tc--)
    	{
    		int a, b, c, d;
    		cin >> a >> b >> c >> d;
    		int lef = min(max(a, b), max(c, d));
    		int righ = max(min(a, b), min(c, d));
    		if (lef>righ)
    		{
    			cout << "Nothing" << endl;
    		}
    		else
    		{
    			if (lef==righ)
    			{
    				cout << "Point" << endl;
    			}
    			else
    			{
    				cout << "Line" << endl;
    			}
    		}
    	}
    	return 0;
    }
