#include <iostream>

using namespace std;

void readInt(int &a) {
	register int c;
	while(((c = getchar_unlocked()) < '0' || c > '9') && c != '-');
	a = c-'0';
	while((c = getchar_unlocked()) >= '0' && c <= '9') {
		a = (a << 1)+(a << 3)+(c-'0');
	}
}

int main(void) {
	unsigned long long ans1[91] = {0};
	unsigned long long ans0[91] = {0};
	ans1[0] = 1;
	ans0[0] = 0;
	for(int i = 1; i < 91; i++) {
		ans0[i] = ans1[i-1]+ans0[i-1];
		ans1[i] = ans0[i-1];
	}
	
	int t, n;
	readInt(t);
	for(int i = 0; i < t; i++) {
		readInt(n);
		cout << ans1[n] << ' ' << ans0[n] << endl;
	}
	return 0;
}
