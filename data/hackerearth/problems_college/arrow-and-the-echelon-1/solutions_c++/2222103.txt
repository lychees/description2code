#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int l;
		scanf("%lld",&l);
		ll int i,ox=-1,oy=-1;
		ll int x,y;
		for(x=0;x<=l;x++)
		{
			y=(l-x);
			if((x%5)==0 && (y%3)==0)
			{
				if(y>oy)
				{
					oy=y;
					ox=x;
				}
			}
		}
		string s="";
		if(ox==-1 && oy==-1)
		s="-1";
		else
		{
			for(i=0;i<oy;i++)
			s+='5';
			for(i=0;i<ox;i++)
			s+='3';
		}
		cout<<s<<endl;
	}
	return 0;
}
