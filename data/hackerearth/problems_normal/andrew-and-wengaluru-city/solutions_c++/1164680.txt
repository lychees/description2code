#include <iostream>
using namespace std;

int main()
{
	int testCases;
	int arrSize;
	int *arr;
	cin>>testCases;
	for(int i=1;i<=testCases;i++)
	{
		cin>>arrSize;
		arr=new int[arrSize];
		for(int j=0;j<arrSize;j++)
		{
			cin>>arr[j];
		}
		unsigned long long int totWater=0;
	for(int j=0;j<arrSize;j++)
		{
			unsigned long long int sofar=0;
			do
			{
				//cout<<arr[j]<<endl;
				int found=0;
				int max=0;
				for(int k=j+1;k<arrSize;k++)
				{
					if(arr[j]<=arr[k])
					{
						/*sofar+=k-j-1;
						arr[j]=arr[j]-1;*/
						found=1;
						j=k-1;
						break;
					}
					else
					{
						sofar+=arr[j]-arr[k];
						if(max<arr[k])
							max=arr[k];
					}
				}
				if(!found)
				{
					sofar=0;
					arr[j]=max;
				}
				else
					break;
			}while(arr[j]>0);
			totWater+=sofar;
		}
		delete[] arr;
		cout<<totWater%1000000007<<endl;
	}

    return 0;
}
