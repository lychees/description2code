#include <iostream>
#include<stdlib.h>
using namespace std;
long long int maxi(long long int a,long long int b)
{
	if(a>b)
	return a;
	else
	return b;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int x,y,z;
    	cin>>x>>y>>z;
    	long long int k=maxi(abs(y-x),abs(z-y));
    	cout<<k-1<<endl;
    }
    return 0;
}
