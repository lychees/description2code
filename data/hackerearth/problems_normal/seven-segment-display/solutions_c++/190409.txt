#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t,n,ans;
	int arr[10];
	char str[100010];
	arr[0]=6;
	arr[1]=2;
	arr[2]=5;
	arr[3]=5;
	arr[4]=4;
	arr[5]=5;
	arr[6]=6;
	arr[7]=3;
	arr[8]=7;
	arr[9]=6;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",str);
		ans=0;
		n=strlen(str);
		for (i=0;i<n;i++)
		{
			ans+=arr[str[i]-'0'];
		}
		printf("%d\n",ans);
	}
	return 0;
}
