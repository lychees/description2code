a = [0 for i in xrange(105)]
a[0] = 1
a[1] = 1
for i in xrange(2, 100):
	a[i] = a[i - 1] + a[i - 2]

n, m = map(int, raw_input().split())
ans = 0
x = map(int, raw_input().split())
a = a[:n]
for y in x:
	idx = -100
	for i in xrange(len(a)):
		if a[i] == y:
			idx = i
			break
	if idx != -100:
		del a[idx]
