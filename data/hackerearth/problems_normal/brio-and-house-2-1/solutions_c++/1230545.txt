#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int x1,y1,x2,y2;
        int r;
        scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
        scanf("%d",&r);
        int d=((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
        ll res=ll(r)*r*d;
        int D=x1*y2-x2*y1;
        res=res-ll(D)*D;
        if(res<0) {
            printf("SAFE\012");
        } else if(res==0) {
            printf("JUST MISSED\012");
        } else {
            printf("REPLANNING\012");
        }
    }
    return 0;
}
