#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t, prefix_sum, n, a[100005], l , r, flag;
	scanf("%lld", &t);
	while(t--)
	{
		flag = 0;
		prefix_sum = 0;
		scanf("%lld", &n);
		for(int i = 0; i < n; i++)
		{
			scanf("%lld", &a[i]);
			prefix_sum += a[i];
		}

		l = 0;
		r = prefix_sum - a[0];
		for(int i = 1; i < n; i++)
		{
			if(l == r)
			{
				flag = 1;
				printf("YES\n");
				break;
			}

		//	cout << i << ' ' << l << ' ' << r << endl;
			l += a[i - 1];
			r -= a[i];

		}

		if(!flag)
			printf("NO\n");


	}	
}
