

    #include <cstdio>
    #include <algorithm>
    #include <cmath>
    #include <cassert>
    using namespace std;
    int ar[15];
    bool check(int mask, int s, int all){
    int sum, j;
    for(int mask1 =((mask-1)& mask);mask1;mask1 = ((mask1-1) & mask)){
    int t = mask1;
    sum = 0;
    j = 0;
    while(t){
    if(t&1){
    sum = sum + ar[j];
    }
    t = t>>1;
    j++;
    }
    if(sum<=s and sum>0){
    int s3 = all - sum-s;
    if(s3<=s and s3>0)
    return true;
    }
    }
    return false;
    }
    int main(){
    int N, s1, s2, s3=0, tmp, ans = 10000000;
    scanf("%d", &N);
    assert(N<=12);
    for(int i=0;i<N;i++){
    scanf("%d", &ar[i]);
    assert(ar[i]>0 and ar[i]<=100);
    s3 = s3 + ar[i];
    }
    if(N<3)
    assert(false);
    int j;
    for(int i=1;i<(1<<N);i++){
    s1 = s3, j = 0;
    tmp = i;
    while(tmp){
    if(tmp&1){
    s1 = s1 - ar[j];
    }
    tmp = tmp >> 1;
    j++;
    }
    if(check(i, s1, s3)){
    ans = min(ans, s1);
    }
    }
    if(ans<s3)
    printf("%d\n", ans);
    else{
    //printf("%d %d\n", ans, s3);
    assert(false);
    }
    return 0;
    }
