#include<iostream>
#include<string.h>
#include<algorithm>
#include<vector>
#include<limits.h>
using namespace std;
long long int absi(long long int a)
{
 if(a>0)
 return a;
 else
 return (-1)*a;
}
long long int p[100005],q[100005];
int main() 
{
	// your code goes here
	long long int i,j,m,n,t,x,y,at,bt,min,temp;
	scanf("%lld",&t);
	while(t--)
	{
	   scanf("%lld",&n);
	   for(j=1;j<=n;j++)
	   scanf("%lld",&p[j]); 
	   for(j=1;j<=n;j++)
	   scanf("%lld",&q[j]);
	   scanf("%lld%lld",&x,&y); 
	   scanf("%lld%lld",&at,&bt); 
	   temp=0;
	   temp+=at*absi(x);
	   temp+=at*absi(y);
	   min=temp;
	   for(i=1;i<=n;i++)
	   {
	   	temp=0;
	   	temp+=at*absi(p[i]);
	   	temp+=at*absi(q[i]);
	   	temp+=bt*absi(p[i]-x);
	   	temp+=bt*absi(q[i]-y);
	   	if(temp<min)
	   	min=temp;
	   }
	   printf("%lld\n",min);
	}
	return 0;
}
