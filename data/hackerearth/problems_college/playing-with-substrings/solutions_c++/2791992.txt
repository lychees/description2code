#include <bits/stdc++.h>
using namespace std;

int main()
{int count=0;
   string s1,s2;
   map<string,int> my;
   cin>>s1>>s2;
  int l1=s1.size(),l2=s2.size();
  for(int i=0;i<l1;i++){
      for(int j=i;j<l1;j++){
          string s(s1.begin()+i,s1.begin()+j+1);
          size_t found = s.find(s2);
          
          if (found!=string::npos &&my[s]!=1){my[s]=1;count++;}
          
      }
  }
  cout<<count;
    return 0;
}
