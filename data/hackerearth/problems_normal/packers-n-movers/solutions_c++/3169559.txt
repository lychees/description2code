using namespace std;
int arr[200];
int myfunc(int n,int max,int k)
{
    int low=max,high=0;
    for(int i=0;i<n;i++)
    high=high+arr[i];
    while(low<high)
    {
        int mid=low+(high-low)/2;
        int current_sum=0,count=1;
        for(int i=0;i<n;i++)
        {
            if(current_sum+arr[i]<=mid)
                current_sum+=arr[i];
            else
                {
                    current_sum=arr[i];
                    count++;
                    
                }
        }
        if(count<=k)
            high=mid;
        else
            low=mid+1;
        
    }
    return low;
    
}
int main()
{

 int m,n;
 scanf("%d%d",&m,&n);
 int mx=-99;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&arr[i]);
        mx=max(arr[i],mx);
        
    }
    int ans=myfunc(n,mx,m);
    printf("%d",ans);

    return 0;
}
