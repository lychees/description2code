ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}
bool comp(const pp& a,const pp& b)
{
	if(a.first^b.first)
		return(a.first<b.first);
	return(a.second>b.second);
}
