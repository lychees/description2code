using namespace std;

int main()
{
	int n;
	cin>>n;
	for(int i=1; i<=n; i++){
		string a,b;
		cin>>a;
		cin>>b;
		int j=0; int count=0;
		while(j<a.length() && j<b.length()){
			if(a.at(j)==b.at(j))
				count++;
			j++;
		}
		cout<<count<<endl;
	}
	return 0;
}
