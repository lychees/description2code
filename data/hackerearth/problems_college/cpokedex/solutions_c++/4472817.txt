#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
    ll n,i,j,k,l,S,L,sf,lf,t;
    cin>>t;
    while(t--){
    cin>>n;
    ll a[n];
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    map<ll,ll>m;
    for(i=0;i<n;i++){
        m[a[i]]++;
    }
    set<ll>s;
    for(map<ll,ll>::iterator it=m.begin();it!=m.end();it++){
        s.insert(it->second);
        //cout<<it->second;
    }
    /*for(set<int>::iterator it1=s.begin();it1!=s.end();it1++){
        cout<<*it1<<endl;
    }*/
    if(s.size()>2){
        cout<<"NO"<<endl;
    }
    else if(s.size()==2){
        ll b[2];
        k=0;
        for(set<ll>::iterator it=s.begin();it!=s.end();it++){
            b[k]=*it;
            k++;
        }
        S=min(b[0],b[1]);
        L=max(b[0],b[1]);
        sf=0;
        lf=0;
        for(map<ll,ll>::iterator it1=m.begin();it1!=m.end();it1++){
            if((it1->second)==S){
                sf++;
            }
            else if((it1->second)==L){
                lf++;
            }
        }
        if(sf==1){
                if(S!=1) //if the small element in set is not 'one' like in case A={1,1,2,2,2,3,3,3} here S=2(freq of 1)
                {cout<<"NO"<<endl;} //A={1,1,2,2,2,3,3,3}
                else{
                    cout<<"YES"<<endl;// A={9,10,10,10}
                }
            }
            else if((lf==1 || sf==1 )&& (L-S)==1 ){
                    cout<<"YES"<<endl;//if frequency of any element is one we will directly delete that element and get the result A={9,9,10,10,11} B={9,9,10,10,11,11,11}                cout<<"YES"<<endl;
            }
            else{
                cout<<"NO"<<endl;//A={1,1,2,2,3,3,4,5}
            }
        }
        else{
            cout<<"YES"<<endl;
        }
    }
}
