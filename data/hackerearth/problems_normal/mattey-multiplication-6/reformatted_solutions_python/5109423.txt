
def mul(n, m):
    c = 0
    l = []
    while m != 0:
        if (m & 01):
            l.append(c)
        n <<= 1
        m >>= 1
        c = c+1
    return l


T = int(raw_input())
a = []
for t in range(0, T):
    a = [int(z) for z in raw_input().split()]
    n = a[0]
    m = a[1]
    l = []
    st = ""
    l = mul(n, m)
    l.reverse()
    for i in range(0, len(l)-1):
        st = st+("(%d<<%d) + " % (n, l[i]))
    st = st+("(%d<<%d)" % (n, l[-1]))
    print st
