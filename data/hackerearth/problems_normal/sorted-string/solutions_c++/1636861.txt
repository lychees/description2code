#include <iostream>
#include<algorithm>
using namespace std;

struct s
{
	char c;
	int cnt;
} arr[255];

bool cmp(struct s a,struct s b)
{
	if(a.cnt==b.cnt)
	return a.c<b.c;
	return a.cnt<b.cnt;
}
int main()
{
    int tc,i,j;
    char str[1000];
    cin>>tc;
    while(tc--)
    {
    	cin>>str;
    	for(i=0;i<255;i++)
    	{
    		arr[i].cnt=0;
    		arr[i].c='a';
    	}
    	for(i=0;str[i];i++)
    	{
    		arr[str[i]].c=str[i];
    		arr[str[i]].cnt++;
    	}
    	sort(arr,arr+255,cmp);
    	for(i=0;i<255;i++)
    	{
    		if(arr[i].cnt!=0)
    		{
    			//cout<<arr[i].c<<" "<<arr[i].cnt<<endl;
    			for(j=0;j<arr[i].cnt;j++)
    			{
    				cout<<arr[i].c;
    			}
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
