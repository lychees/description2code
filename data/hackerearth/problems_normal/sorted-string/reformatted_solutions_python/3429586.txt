T = int(raw_input())

for _ in xrange(T):
    s = raw_input()
    d1 = {}
    for ch in s:
        d1[ch] = d1.get(ch, 0) + 1

    d2 = {}
    for k in d1:
        v = d1[k]
        l = d2.get(v, [])
        l.append(k)
        d2[v] = l

    for k in d2:
        d2[k].sort()

    keys = sorted(d2)
    st = ""
    for k in keys:
        v = d2[k]
        for item in v:
            st += k * item

    print(st)
