#include <iostream>
#include <unordered_set>
using namespace std;

int main()
{
  int t=0;
  int k=0;
  long long n;
  long long m;
  long long i;
  long long j;
  long long val;
	
  unordered_set<long long> s;

  cin >> t;
  for(k=0;k<t;k++)
  {
  	 cin >> n >> m;
  	 for(i=0;i<n;i++)
  	 {
  	 	cin >> val;
  	 	s.insert(val);
  	 }
  	 for(j=0;j<m;j++)
  	 {
  	 	cin >> val;
  	 	if(s.find(val) !=s.end())
  	 	{
  			cout << "YES" << "\n";	 		
  	 	}
  	 	else
  	 	{
  	 		cout << "NO" << "\n";
  	 		s.insert(val);
  	 	}
  	 }
  	 s.erase(s.begin(),s.end());
  }
  return 0;   
}
