#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d",&n);
		ans = 0;
		while (n != 1) {
			if (n % 3 == 0) {
				ans++;
				n = n / 3;
			} else if (n % 2 == 0) {
				ans++;
				n = n / 2;
			} else if (n == 1) {
				break;
			}
			else if ((n-1) % 3 == 0) {
				ans++;
				n--;
				if (n == 1) {
					break;
				}
				n = n / 3;
				ans++;
			} else if ((n-1) % 2 == 0) {
				ans++;
				n--;
				if (n==1) {
					break;
				}
				n = n/2;
				ans++;
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
