#include <stdio.h>
#include <string>
#include <iostream>
#include <math.h>
#include <map>

using namespace std;

void DragRacing()
{
	string p1;
	string p2;
	map<string, string> racemap;
	std::map<string, string>::iterator it;

	int n;
	cin >> n;

	int po = pow(2.0, n);

	for (int i = 0; i < po-1; i++)
	{
		cin >> p1;
		cin >> p2;

		racemap.insert(pair<string, string>(p2, p1));
	}


	it = racemap.find(p2);

	while (it != racemap.end())
	{
		p2 = it->second;
		it = racemap.find(p2);
	}

	cout << p2 << endl;

}

int main()
{
	DragRacing();
}
