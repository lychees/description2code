from collections import Counter
n = raw_input()
n = int(n)
s = []

for a in xrange(0, n):
    s.append(raw_input())


for b in s:

    l = list(b)

    for n, i in enumerate(l):
        j = ord(i)
        l[n] = j

    l.sort()

    l.sort(key=Counter(l).get)

    for n, i in enumerate(l):
        j1 = chr(int(i))
        l[n] = j1

    print "".join(l)
