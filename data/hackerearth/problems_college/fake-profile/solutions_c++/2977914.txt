#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char a[100];
    cin >> a;
    int t,i,j;
    t = strlen(a);
    for(i=0;i<strlen(a)-1;i++)
    {
    	for(j=i+1;j<strlen(a);j++)
    	{
    		if(a[j] == a[i])
    			t--;
    	}
    }
    if(t%2 == 0)
    	cout << "CHAT WITH HER!" << endl;
    else
    	cout << "IGNORE HIM!" << endl;
    return 0;
}
