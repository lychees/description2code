#include <iostream>
#include<math.h>
using namespace std;

int parity(long long N)
{
	int temp=1;
	while(N!=0)
	{
		temp=temp*(-1);
		N=N&(N-1);
	}
	return temp;
}


int main()
{
    int T;
    long long N,i,temp,count;;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	count=parity(N);
    	if(count==1)
    		cout<<"even"<<endl;
    	else
    		cout<<"odd"<<endl;
    }
    return 0;
}
