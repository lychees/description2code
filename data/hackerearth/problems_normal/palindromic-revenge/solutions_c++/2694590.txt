#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j;
    long long int sum;
    scanf("%d\n",&t);
    while(t--)
    {
    	sum=0;
    	char a[100000];
    	scanf("%s",a);
    	i=0,j=strlen(a)-1;
    	while(1)
    	{
    		if(i>=j)
    		break;
    		sum+=abs(a[i]-a[j]);
    		i++;
    		j--;
    		
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
