#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	cin.ignore();
	while(t--){
		string s;
		getline(cin,s);
	//	cout<<s;
		int l=s.length();
		int no=0;
		for(int i=0;i<l;i++){
			if(s[i]>='0' && s[i]<='9'){
				no+=s[i]-48;
			}
		}
		vector<string> arr;
		string s1="";
		for(int i=0;i<l;i++){
			if(s[i]>='0' && s[i]<='9'){
				if(i==l-1){
					arr.push_back(s1);
				}
				continue;
			}
			if(s[i]==' '){
				arr.push_back(s1);
				s1="";
				continue;
			}
			s1+=s[i];
			if(i==l-1){
				arr.push_back(s1);
			}
		}
		sort(arr.begin(),arr.end());
		vector<string>::iterator it=arr.begin();
		while(it!=arr.end()){
			string p=*it;
			cout<<p<<" ";
			no=no-p.length();
			if(no>=0)cout<<no<<" ";
			else cout<<-no<<" ";
			it++;
		}
		cout<<endl;
	}
	return 0;
}
