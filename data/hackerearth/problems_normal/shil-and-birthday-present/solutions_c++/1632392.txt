#include<stdio.h>
#include<unordered_map>
#include<stdlib.h>
using namespace std;
int main(){
    int N;
    scanf("%d", &N);
    unordered_map<int, int> m;
    int e;
    long long count = 0;
    for( int i = 0; i < N; i++){
        scanf("%d", &e);
        if(m.find(e) == m.end()){
            m[e] = 1;
            count++;
        }
    }
    printf("%lld", count * (count - 1)/2);
}
