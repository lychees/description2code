#include<bits/stdc++.h>
using namespace std;
vector<int>a(1<<20);
int main()
{

	int n;
	while(scanf("%d",&n)==1)
	{
		for(int i=1;i<=n;++i)
			cin>>a[i];
		bool flag=false;
		for(int i=n;i>=1;--i)
		{
			if((a[i]&1)==0)
			{
				printf("%d ",a[i]);
				flag=true;
			}
		}
		if(flag==false)printf("None to display");
		printf("\n");

	}


	return 0;
}
