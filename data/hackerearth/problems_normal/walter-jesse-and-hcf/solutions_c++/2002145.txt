#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#define ll long long
using namespace std;

ll gcd(ll a, ll b)

{
    if(b>a) swap(a,b);
    if(b==0) return a;
    return gcd(b,a%b);
}

int main()
{
    ll days;
    cin >> days;
    while(days--)
    {
       ll s,e,index;
        cin >> s >> e>>index;
         vector<ll> factor;
         ll hcf = gcd(s,e);
        
         for(ll i=1;i<=sqrt(hcf);i++)
         {
             if(hcf%i==0)
         {
             factor.push_back(i);
             if(i*i != hcf) factor.push_back(hcf/i);
         }
         }
         sort(factor.begin(),factor.end());
         
         if(index > factor.size())
            cout << "No crymeth today" << endl;
         else
            cout << factor[factor.size() - index] << endl;
    }
}
