# include <bits/stdc++.h>

using namespace std;

int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int m, n;
        scanf("%d %d", &m, &n);
        vector<pair<int, int> > a(m);
        for (int i = 0; i < m; ++i) {
            scanf("%d %d", &a[i].first, &a[i].second);
        }
        sort(a.begin(), a.end());

        vector<pair<int, int> > b(n);

        for (int i = 0; i < n; ++i) {
            scanf("%d %d", &b[i].first, &b[i].second);
        }

        sort(b.begin(), b.end());
        vector<bool> va(m);
        vector<bool> vb(n);
        
        int k, l;
        scanf("%d %d", &k, &l);
        int x = l;

        int i = 0;
        int j = 0;
        while (i < m && j < n && l > 0) {
            if (a[i].first < b[j].first) ++i;
            else if (a[i].first > b[j].first) ++j;
            else {
                if (a[i].second < b[j].second) ++i;
                else if (a[i].second > b[j].second) ++j;
                else {
                    --l;
                    va[i] = 1;
                    vb[j] = 1;
                    ++i, ++j;
                }
            }
        }
        k = max(0, k - (x - l));

        i = 0;
        j = 0;
        while (i < m && j < n && k > 0) {
            while (va[i]) ++i;
            while (vb[j]) ++j;
            if (a[i].first < b[j].first) ++i;
            else if (a[i].first > b[j].first) ++j;
            else {
                --k;
                ++i, ++j;
            }
        }

        if (!k) {
            if (!l) cout << "Great\n";
            else cout << "Good\n";
        }
        else cout << ":(\n";
    }
}
