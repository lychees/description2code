#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,s=0,a=0;
    string x,y;
    cin>>x;
    cin>>n;
    while(n--){
    	cin>>y;
    	if(x.size()==y.size()){
    		if(x[0]==y[0]){
    			s++;
    		}else if(y[0]-x[0]==3){
    			a++;
    		}
    	}
    }
    cout<<s<<" "<<a<<endl;
    return 0;
}
