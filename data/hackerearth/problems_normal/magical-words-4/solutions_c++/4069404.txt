#include <bits/stdc++.h>
using namespace std;
#define ll long long int
string s;

int main()
{
	ll t,i,n,k,a,b;
	cin >> t;
	while(t--){
		cin >> s;
		n=s.size();
		k=0;
		//odd palindrome
		for(i=0;i<n;i++){
			for(a=i,b=i;a>=0 && b<n;a--,b++){
				if(s[a]==s[b]) k+=(b-a+1)*(b-a+1);
				else break;
			}
		}
		//even palindrome
		for(i=0;i<n-1;i++){
			for(a=i,b=i+1;a>=0 && b<n;a--,b++){
				if(s[a]==s[b]) k+=(b-a+1)*(b-a+1);
				else break;
			}
		}
		cout << k << endl;
	}
    return 0;
}
