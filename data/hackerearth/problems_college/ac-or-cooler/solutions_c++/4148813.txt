#include <iostream>
using namespace std;

int main()
{
    long n,m,i,j,s=0,b,c=0,a[1001];
    for(i=0;i<=1000;i++)
 	a[i]=0;
 	cin>>n>>m;
 	for(i=1;i<=2*n;i++){
 	cin>>b;a[b]++;
 	}
 	for(i=1;i<=1000;i++){
 		if(s+i>=m||c==n)
    	break;
    	for(j=1;j<=a[i];j++){
    	if(s+i>=m||c==n)
    	break;
    	s=s+i;
    	c++;
 		}
 	}
 	cout<<c;
    return 0;
}
