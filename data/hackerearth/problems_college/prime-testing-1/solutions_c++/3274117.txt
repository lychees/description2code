//sieve of eratosthenes
#include<iostream>
#define max 1001
using namespace std;

int main()
{
	
	bool hash[max];
	for(int i=1;i<=max;i++)
		hash[i]=true;
	for(int i=2;i*i<=max;i++)
		for(int j=i*2;j<=max;j+=i)
			if(hash[j])
				hash[j]=false;
	/*for(int i=2;i<=max;i++)
		if(hash[i])
			cout<<i<<endl;*/
			hash[1]=false;
			hash[0]=false;
	int t;
	cin>>t;
	while(t--)
	{
		long n,c=0;
		cin>>n;
		while(n--)
		{	int a;
			cin>>a;
			if(hash[a])
				c++;
			
		}
		cout<<c<<endl;
	
	}
	return 0;
	
}
