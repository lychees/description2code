#include <bits/stdc++.h>
using namespace std;
#define scan(x) scanf("%d",&x)
#define print(x) printf("%d",x)
#define MOD 1000000007
#define scanl(x) scanf("%lld",&x)
#define printl(x) printf("%lld",x)
int know[100001];
bool vis[100001];
long long int maxi,maxcount;
void dfs(int i,int n,map<int,vector<int> > &m)
{
	vis[i]=1;
	if(know[i]>maxi)
	{
		maxcount=1;
		maxi=know[i];
	}
	else if(know[i]==maxi)
	{
		maxcount++;
	}
	vector<int>::iterator it;
	for(it=m[i].begin();it!=m[i].end();it++)
	{
		if(!vis[*it])
		{
			dfs(*it,n,m);
		}
	}
}
 
 
int main()
{
	int n,p;
	scan(n);
	scan(p);
	for(int i=1;i<=n;i++)
	{
		scan(know[i]);
		vis[i]=0;
	}
	map<int,vector<int> > m;
	int t1,t2;
	for(int i=0;i<p;i++)
	{
		scan(t1);
		scan(t2);
		m[t1].push_back(t2);
		m[t2].push_back(t1);
	}
	long long int ans=1;
	for(int i=1;i<=n;i++)
	{
		if(!vis[i])
		{
			maxcount=0;
			maxi=-1;
			dfs(i,n,m);
			ans=(ans*maxcount)%MOD;
		}
	}
	cout<<ans;
    return 0;
}
