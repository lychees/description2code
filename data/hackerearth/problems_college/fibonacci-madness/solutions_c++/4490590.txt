#include <iostream>
using namespace std;

int main()
{	long long int fib[10000];
	long long	int T,N;
   	cin>>T;
   	for(long long int i=0;i<10000;++i)
   		fib[i]=4;
    fib[0]=0;
    fib[1]=1;
    for(long long int i=0;i<T;++i)
    {
    	 cin>>N;
    	 for(long long int j=2;j<N;++j)
    	 	{if(fib[j]==4)
    	 	fib[j]=fib[j-1]+fib[j-2];
    	 	}
    	 for(long long int j=N-1;j>=0;--j)
    	 cout<<fib[j]<<" ";
    	 cout<<endl;
    }
}
