import math
t = int(raw_input())
for i in range(1, t+1):
    n = int(raw_input())
    flag = 0
    while flag != 1:
        sum1 = 0
        while n != 0:
            sum1 += pow(n % 10, 2)
            n /= 10
        if sum1 == 1:
            print('Case #'+str(i)+':YES')
            flag = 1
        elif sum1 < 10 and sum1 != 1:
            print('Case #'+str(i)+':NO')
            flag = 1
        else:
            flag = 0
        n = sum1
