	#include<iostream>
	using namespace std;
	int main()
	{
	    int T;
	    cin>>T;
	    int a[10001],dp[10001];
	    while (T--)
	    {
		int N;
		cin>>N;
		for (int i=0;i<N;i++)
		    cin>>a[i];
		for (int i=0;i<10001;i++) {
		    dp[i]=0;
		}
		dp[0]=1;
		for (int i=0;i<N;i++)
		{
		    for (int j=10001;j>=0;j--)
		    {
			if (dp[j])
			    dp[j+a[i]]=1;
		    }
		}
		int ct=0;
		for (int j=0;j<10001;j++)
		    if (dp[j])
			ct++;
		cout<<ct<<endl;
	    }
	    return 0;
	}
