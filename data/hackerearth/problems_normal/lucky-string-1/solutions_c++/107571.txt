inline void inp( int &n ) {
        n=0; int ch = getcx(); int sign = 1;
        while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
        while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
		n = n * sign;
}
int evallen(LL n){
	LL digs = 1;
	while(n > 0){
		n -= digs * (1LL<<digs);
		digs++;
	}
	return digs - 1;
}
LL dign(LL n, int digs){
	//if(digs <= 1) return n;
	FOR(i, 1, digs){
		n -= LL(i) * (1LL<<i);
	}
	return n;
}

		LL digno = dign(n, digs);
		LL num = (digno + digs-1) / digs, w = digno % digs;
