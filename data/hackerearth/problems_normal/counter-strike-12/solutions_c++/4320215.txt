#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		int n,m,d;
		cin>>n>>m>>d;
		int x11[1005],y12[1005],x21[1005],y21[1005];
		for(int j=0;j<n;j++){
			cin>>x11[j]>>y12[j];
		}
		for(int j=0;j<m;j++){
			cin>>x21[j]>>y21[j];
		}
		int distance;
		int ans=0;
		for(int j=0;j<m;j++){
			for(int k=0;k<n;k++){
				distance = abs(x11[k]-x21[j]) + abs(y12[k] - y21[j]);
			
				if(distance <= d){
					ans++;
					break;
				}
			}
			
		}
		if(ans > m / 2 + 1) cout << "YES\n";    
        else cout << "NO\n";
		
	}
	return 0;
}
