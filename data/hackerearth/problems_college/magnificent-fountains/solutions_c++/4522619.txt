#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll p[10005];
vector<ll> prime;
void sieve() {
	p[0] = 1;
	p[1] = 1;
	for(int i=2;i<10005;i++) {
		if(!p[i])
		for(int j=2;j*i<10005;j++) {
			p[i*j] = 1;
		}
	}
	for(int i=0;i<10005;i++) {
		if(!p[i]) {
			prime.push_back(i);
		}
	}
}
ll poo(ll x, ll b) {
	if(b == 0) {
		return 1;
	}
	ll temp = poo(x, b/2);
	temp = temp*temp;
	if(b & 1) {
		temp = temp * x;
	}
	return temp;
}
vector<ll> v[1005];
int main() {
    ll t;
    sieve();
    cin>>t;
    while(t--) {
    	for(int i=0;i<1005;i++) {
    		v[i].clear();
    	}
    	ll n;
    	cin>>n;
    	ll a[n];
    	for(int i=0;i<n;i++) {
    		cin>>a[i];
    	}
    	ll ans = 1;
    	for(int i=0;i<prime.size();i++) {
    		ll pr = prime[i];
    		ll mp = 0;
    		for(int j=0;j<n;j++) {
    			ll ct = 0, temp = a[j];
    			while(temp % pr == 0) {
    				temp /= pr;
    				ct ++;
    			}
    			mp = max(mp, ct);
    		}
    		ans = (ans * poo(pr, mp)) % mod;
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
