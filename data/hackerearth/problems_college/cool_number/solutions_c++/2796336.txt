#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int i,j,n,k;
    	cin >> n >> k;
    	int ans = 0;
    	for(i=1;i<=n;i++)
    	{
    		j = i; int count=0;
    		while(j)
    		{
    			if(j%8==5)
    			 count++;
    			 j = j>>1;
    		}
    		if(count>=k)
    		ans++;
    	}
    	cout << ans << endl;
    }
    return 0;
}
