#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t,i;
    string a,b,c;
    unsigned long long int x,y;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
	    y=stoi(b);
	    for(i=0;i<=(a.length()-b.length());i++)
	    {
	    	c=a.substr(i,b.length());
	    	x=stoi(c);
	    	cout<<(x*y)<<"\n";
	    	//cout<<c<<"\n";
	    }
    }
    return 0;
}
