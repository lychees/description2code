#include <bits/stdc++.h>
#define ll long long
#define ull unsigned long long
using namespace std;

struct BIT 
{
	int val[1<<17];
	void init()
	{
		memset(val,0,sizeof(val));
	}
	
	int query(int pos)
	{
		int sum = 0;
		for(; pos > 0; pos -= pos & -pos)
			sum += val[pos];
		return sum;
	}
	
	void update(int pos, int incr)
	{
		for(; pos < 1<<17; pos += pos & -pos)
			val[pos] += incr;
	}
} B;


int main()
{
    int n, q;
    cin >> n >> q;
    for(int i=1;i<=n;i++)
    {
		int x;
		cin >> x;
		B.update(i,x);
	}
	while(q--)
	{
		int arr[5] = {16,9,26,26,1};
		int t,l,r;
		cin >> t >> l >> r;
		if(t == 1)
		{
			if(l == 1)
			{
				cout << B.query(r) << "/" << r-l+1 << "\n";
			}
			else
			{
				cout << B.query(r) - B.query(l-1) << "/" << r-l+1 << "\n";
			}
		}
		else
		{
			int temp = 0;
			for(int i=l;i<=r;i++)
			{
				B.update(i,arr[temp%5]);
				temp++;
			}
		}
		
	}
    
}
