#include <iostream>
#include <cstdio>

#define MAXN 3000000

using namespace std;

int N;

void read()
{
	scanf("%d", &N);
}

int used[MAXN];

void solve()
{
	int i, j;
	for(j = 0, i = 0; j < N; i++)
		if(used[i] >= 2)
			j++;

	printf("%d\n", i - 1);

}

void genNumbs()
{
	for(int i = 1; i <= 100; i++)
		for(int j = i + 1; j <= 100; j++)
			used[i * i * i + j * j * j]++;
}

int main()
{
	genNumbs();
	
	int T;
	
	scanf("%d", &T);
	for(int i = 0; i < T; i++)
	{
		read();
		solve();
	}	
	return 0;
}
