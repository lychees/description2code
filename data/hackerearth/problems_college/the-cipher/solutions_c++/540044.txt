#include<stdio.h>
#include<vector>
using namespace std;
int main()
{
    int t,n,i,x;
    scanf("%d",&t);
    //vector<long>vec(0,0);
    while(t--)
    {
         vector<long>vec(0,0);
        long n;
        scanf("%ld",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&x);
            vec.push_back(x);
        }
        for(i=0;i<n;i++)
        {
            char c;
            c=char(255-vec[i]);
            printf("%c",c);
        }
        printf("\n");
    }
    return 0;
}
