#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		long long int tempn=n;
		long long int sumdigits=0,dig;
		while(tempn)
		{
			dig=tempn%10;
			sumdigits=sumdigits+dig;
			tempn=tempn/10;
		}
		int flag=0;
		
		for(long long int i=2;i<=sumdigits/2;i++)
		{
			if(sumdigits%i==0)
			{
				flag=1;
				break;
			}
		}
		if(flag==1 || sumdigits==1)
		cout<<"NO"<<endl;
		else if(sumdigits==2 || sumdigits==3 || sumdigits==5)
		cout<<"YES"<<endl;
		else 
		cout<<"YES"<<endl;
	}
    return 0;
}
