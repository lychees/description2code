#include <bits/stdc++.h>
#define pb push_back
using namespace std;
string s;int i;
vector<int>pos[26];
int main()
{
    cin>>s;
    int n=s.size();
    for(i=0;i<n;i++)
       {
           //cout<<s[i]-'a'<<" ";
           pos[s[i]-'a'].pb(i);
       }
       int q;
       cin>>q;
       while(q--)
       {
           cin>>s;
           int flag=0,prev=-1;
           for(i=0;i<s.size();i++)
           {
               //cout<<s[i]-'a'<<" ";
              int idx =upper_bound(pos[s[i]-'a'].begin(),pos[s[i]-'a'].end(), prev)- pos[s[i]-'a'].begin();;
              //cout<<idx<<endl;
              if(idx==pos[s[i]-'a'].size())
              {
    			flag = 1;
    			break;
    		  }
             prev=pos[s[i]-'a'][idx];
           }
        if(flag)
    		cout << "No\n";
    	else
    		cout << "Yes\n";
       }
}
