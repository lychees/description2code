#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	bool p[26];
	while ( t-- ) {
		string s;
		cin >> s;
		int n = (int)s.size();
		int m;
		memset(p, false, sizeof(p));
		cin >> m;
		while ( m-- ) {
			char x;
			cin >> x;
			p[x-'a'] = true;
		}
		long long ans = 0;
		for ( int i = 0; i < n; i++ ) {
			if ( p[s[i]-'a'] ) continue;
			ans += (s[i]-'a')+1;
		}
		cout << ans << endl;
		
	}
	return 0;
}
