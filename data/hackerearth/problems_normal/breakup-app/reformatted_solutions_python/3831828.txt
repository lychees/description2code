D = {}
A = [0]*31
N = int(raw_input())
while N > 0:
    N -= 1
    S = (raw_input())+" "
    W = 2 if 'G:' in S else 1

    for i in range(1, 31):
        if " "+str(i)+" " in S:
            A[i] += W

M = max(A)
C = 0
for i in A:
    C = C+1 if i == M else C
if C == 1 and (A[19] == M or A[20] == M):
    print "Date"
else:
    print "No Date"
