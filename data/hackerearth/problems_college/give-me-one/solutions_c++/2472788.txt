typedef long long int ll;
const int mx = 2e7 + 5;
using namespace std;
int dp[mx];
int t, n;
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
//    freopen("data.in","r",stdin);
    dp[1] = 0;
    dp[2] = 1;

    for(int i = 3 ; i < mx; i++){
        if(i%2==0&&i%3!=0)
            dp[i]=1+min(dp[i/2],dp[i-1]);
        else if(i%3==0&&i%2!=0)
            dp[i]=1+min(dp[i/3],dp[i-1]);
        else if(i%2==0&&i%3==0)
            dp[i]=1+min(dp[i/2],min(dp[i/3],dp[i-1]));
        else
            dp[i]=1+dp[i-1];
    }
    cin >> t;
    while(t--){
        cin >> n;
        cout << dp[n] << endl ;
    }
    return 0;
}
