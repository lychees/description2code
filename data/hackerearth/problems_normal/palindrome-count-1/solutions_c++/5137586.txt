#include <iostream>
#include <bits/stdc++.h>
using namespace std;


//Checking if string is palindrome or not?
bool isPalindrome(string &s) {
	
    string t = s;
    reverse (t.begin (), t.end ());
    return s == t;
}

int main() {
	
	char s[50];    
	scanf("%s", s);
	
	int n = strlen(s);
	
	int dp[n + 1][n + 1];
	dp[0][0] = 1;
	
	for(int i = 1; i < n; i++)
		dp[0][i] = 0;
		
	for(int i = 1; i < n; i++) 
		dp[1][0] = 0;
	
	for(int i = 1; i <= n; i++) {
		
		for(int j = i; j <= n; j++) {
			
			string s1 = string(s + i - 1, j - i + 1);
		
			if(i == j)
				dp[i][j] = dp[i - 1][j - 1] + 1;
				
			else if(isPalindrome(s1)) {
				
				dp[i][i] = max(dp[i - 1][j], dp[i][j - 1]) + 1;
				dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]) + 1;
			}
				
			else 
				dp[i][j] = max(dp[i - 1][j], dp[i][j - 1]);
		}
	}
	
	printf("%d", dp[n][n] - 1);
	
	return 0;
}
