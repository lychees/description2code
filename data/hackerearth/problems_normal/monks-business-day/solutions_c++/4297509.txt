#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n, m;
    	cin >> n >> m;
    	
    	vector<pair<pair<int, int>, int>> e(m + 1);
    	
    	for(int i = 0; i < m; i++)
    	{
    		cin >> e[i].first.first >> e[i].first.second >> (e[i].second);
    	}
    	for(int i = 0; i < m; i++)
    	{
    		e[i].second = - e[i].second;
    	}
    	
    	vector<int> dist(n + 1);
    	
    	for(int i = 1; i <= n; i++)
    		dist[i] = INT_MAX;
    	dist[1] = 0;
    	
    	for (int i = 1; i <= n - 1; i++)
		{
		    for (int j = 0; j < m; j++)
		    {
		        int u = e[j].first.first;
		        int v = e[j].first.second;
		        int weight = e[j].second;
		        if (dist[u] != INT_MAX && dist[u] + weight < dist[v])
		            dist[v] = dist[u] + weight;
		    }
		}
		
		int flag = 0;
	    for (int j = 0; j < m; j++)
	    {
	        int u = e[j].first.first;
	        int v = e[j].first.second;
	        int weight = e[j].second;
	        if (dist[u] != INT_MAX && dist[u] + weight < dist[v])
	        {
	        	flag = 1;    	
	        }
	    }
		
    	
    	if(flag)
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    	
    }
}
