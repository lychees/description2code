#include <iostream>
using namespace std;
int f[26];
int main()
{
    char str[10001];
    int t,i,j,ones=0,odds=0;
    cin>>t;
    while(t--){
    	cin>>str;
    	for(i=0;i<26;i++)
    		f[i]=0;
    	for(i=0;str[i];i++){
    		f[str[i]-'a']++;
    	}
    	ones=0;
    	odds=0;
    	for(i=0;i<26;i++){
    		if(f[i]&1){
    			if(f[i]==1)
    				ones++;
    			else
    				odds++;
    		}
    	}
		if(ones==0 && odds==0)
			cout<<"0"<<endl;
    	else if(ones==1)
    		cout<<odds<<endl;
    	else if(ones==0)
    		cout<<odds-1<<endl;
    	else if(ones>=1)
    		cout<<odds+ones-1<<endl;
    }
    return 0;
}
