#include<stdio.h>
#include<math.h>
using namespace std;
typedef long long int lli;

lli a[1000005];
lli g[1000005];

void Preprocess(){
    for(lli i=2;i<=1000;i++){
        for(lli j=2;i*j<=1000000;j++){
            lli x=i*j;
            if(i<sqrt(x))
                g[x]+=(i+j);
            else if(i==sqrt(x))
                g[x]+=i;
            if(a[x]==0)
                a[x]=i;
        }
    }


     for(lli i=2;i<=1000000;i++)
        if(a[i]==0)
            a[i]=i;

     for(lli i=2;i<=1000000;i++)
        a[i]=a[i-1]+a[i];

     for(lli i=1;i<=1000000;i++)
           g[i]+=1;

     g[1]=0;
     for(lli i=2;i<=1000000;i++)
        g[i]=(g[i-1]+g[i]);

     for(lli i=1;i<=1000000;i++)
        a[i]=(a[i]+g[i])%i;
}

int main(){
    int t,n;
    Preprocess();
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        printf("%d\n",a[n]);
    }
    return 0;
}
