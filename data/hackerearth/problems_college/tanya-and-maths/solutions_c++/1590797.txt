using namespace std;

int fi(int n)
{
       int result = n;
       for(int i=2;i*i <= n;i++)
       {
         if (n % i == 0) result -= result / i;
         while (n % i == 0) n /= i;
       }
       if (n > 1) result -= result / n;
       return result;
}

int a,b;
void read()
{
    cin >> a >> b;
}

void solve()
{
    int sum = 0;
    int mn = min(a,b);
    int mx = max(a,b);
    for(int i = 1;i<=mn;++i)
    {
        if(a%i == 0 && b%i == 0)
        {
            sum += fi(i);
        }
    }
    cout << sum << endl;
}

    {
        read();
        solve();
