#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int i,T,c=0;
    cin>>T;
   
    
    unsigned long long  num;
    unsigned int max=(int)3*pow(10,8);
    for(i=0;i<T;i++)
   {
   	scanf("%llu",&num);
   	//cin>>num;
     if(num<max)
     c++;
   }
    cout<<c;
    
    
}
