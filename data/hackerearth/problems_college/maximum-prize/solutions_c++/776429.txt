#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int u=0;u<t;u++)
    {
    	  int n,k;
    	  cin>>n;
    	  cin>>k;
    	  int a[n];
    	  for(int i=0;i<n;i++)
    	     cin>>a[i];
    	  sort(a,a+n);
    	  long long int sum=0;
    	  int i=n-1;
    	  while(k>0 && i>=0)
    	  {
    	  	   sum+=a[i];
    	  	   i--;
    	  	   k--;
    	  }
    	  cout<<sum<<"\n";
    }
    return 0;
}
