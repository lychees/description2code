#include <bits/stdc++.h>
using namespace std;

int t;

long long n;
long long x, y;

long long now;

long long sum;

long long ar(long long pocz, long long kon, long long ile)
{
    return (pocz+kon)*ile/2;
}

long long f(long long a, long long b)

int main()
{
    srand(42);
    scanf("%d", &t);
    while(t--)
    {
        scanf("%lld%lld%lld", &x, &y, &n);
        sum=x+y;
        n-=2;
        while(n)
        {
            if (n<=10)
            {
                now=abs(x-y);
                sum+=now;
                x=y;
                y=now;
                n--;
                continue;
            }
            if (x>=y && rand()&1)
            {
                long long dz=min(f(x, y), (n/3)*2);

                dz/=2;

                sum+=ar(y, y, dz);
                sum+=ar(x-2*y, x-dz*2*y, dz);
                sum+=ar(x-y, x-dz*2*y+y, dz);

                x-=dz*2*y;

                n-=3*dz;

                continue;
            }
            now=abs(x-y);
            sum+=now;
            x=y;
            y=now;
            n--;
        }
        printf("%lld\n", sum);
    }
