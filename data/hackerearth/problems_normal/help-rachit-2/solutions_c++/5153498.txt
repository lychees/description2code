#include <iostream>
#include<bits/stdc++.h>

using namespace std;
const int N=10002;
int leftup[N],rightup[N],leftdown[N],rightdown[N];

bool isprime(int number)
{	for(int i=2;i*i<=number;i++)
	{	if(number%i==0)return false;
	}
	return true;
}

int main()
{	int a=1;
	int start=4;
	leftup[1]=0;
	
	for(int i=2;i<=5000;i++)
	{	int number=a+start;
		start+=8;
		a=number;
		if(isprime(number))
		{	leftup[i]=1;
		}
		
		//cout<<number<<endl;
	
	}
	a=1;
	start=8;
	
	for(int i=2;i<=110;i++)
	{	
	int number=a+start;
		start+=8;
		a=number;
		if(isprime(number))
		{	leftdown[i]=1;
		}
		//cout<<number<<endl;
	
	}
	a=1;
	start=6;
	for(int i=2;i<=5000;i++)
	{	
	int number=a+start;
		start+=8;
		a=number;
		if(isprime(number))
		{	rightdown[i]=1;
		}
		//cout<<number<<endl;

	}
	a=1;
	start=2;
	
for(int i=2;i<=5000;i++)
	{	
	int number=a+start;
		start+=8;
		a=number;
		if(isprime(number))
		{	rightup[i]=1;
		}
		//cout<<number<<endl;

	}
	for(int i=1;i<=5000;i++)
	{	leftup[i]+=leftup[i-1];
		leftdown[i]+=leftdown[i-1];
		rightup[i]+=rightup[i-1];
		rightdown[i]+=rightdown[i-1];
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{	int n;
		scanf("%d",&n);
		int copy=n	;
	
		n/=2;
	
		int number=leftup[n+1]+leftdown[n+1]+rightup[n+1]+rightdown[n+1];
		double ans=(double)1.0*(1.0*number/(2*copy-1));
		ans*=100.0;
		cout<<fixed<<setprecision(6)<<ans<<endl;

	}
		
}
