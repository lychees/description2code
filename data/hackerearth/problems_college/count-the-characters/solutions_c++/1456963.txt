#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,i,k=0,count=0,m;
    cin>>t;
    m = t;
    int a[26] = {0};
    while(t--)
    {
    	 string s;
    	 cin>>s;
    	 int y = s.size();
    	
    	 for(i=0;i<y;i++)
    	    {
    	    	if(a[s[i] - 'a'] == k)
    	    	  a[s[i] - 'a']++;
    	    }
    	  k++;  
    	 
    }
    
    for(i=0;i<26;i++)
     {
      if(a[i] == m )
     	   count++;
   //  cout<<a[i]<< " ";
     }
     cout<<count;
    return 0;
}
