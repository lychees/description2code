#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    int i,k,t,k1,k2;
    scanf("%d",&t);
    char a[500];
    char b[500];
    while(t--)
    {
        k=0;
        scanf(" %s",&a);
        scanf("%s",&b);
        k1=strlen(a);
        k2=strlen(b);
        k1=k1<k2?k1:k2;
        for(i=0;i<k1;i++)
        {
            if(a[i]==b[i])
            {
                k++;
            }
        }
        printf("%d\n",k);
        for(i=0;i<k2;i++)
        {
            b[i]='\0';
        }
    }
    return 0;
}
