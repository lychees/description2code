#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int n,N,count=0;
    cin>>n;
	
	for(int k=1;k<=n;k++)
    {
    	cin>>N;
    	for(int i=1;i<=N;i++)
    	{	
    		for(int j=i+1;j<=N;j++)
    		{
				if((i^j)<=N)
				{
					count++;
				}
    		}
    	}
    	cout<<count<<"\n";
    	count = 0;
    }
    
    return 0;
}
