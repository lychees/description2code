#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    cin >> t ;
    
    while(t--)
    {
        long long n, m, x, num, sum = 0, y ;
        cin >> n >> m >> x ;

        num = x*(n+1) ;

        for(int i=0 ; i<n ; i++)
        {
            cin >> y ;
            sum += y ;
        }

        if( sum >= num ) cout << 1 << endl;
        else
        {
            if( num - sum <= m ) cout << num-sum << endl;
            else cout << "Impossible\n";
        }
    }

}
