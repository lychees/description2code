#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int n,num;
    
    scanf("%d", &n);
    
    for (int i = 0; i < n; i++) {
    	scanf("%d", &num);
    	
    	int count = 0;
    	
    	while (num > 0) {
    		if (num % 2 == 1) {
    			count++;
    		}
    		num = num / 2;
    	}
    	
    	if (count % 2 == 0) {
    		printf("even\n");
    	} else {
    		printf("odd\n");
    	}
    	
    }
    
    return 0;
}
