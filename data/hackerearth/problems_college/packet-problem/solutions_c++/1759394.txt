#include <iostream>
using namespace std;

int main()
{
	long long int zero[92], one[92];
	zero[0] = 0;
	one[0] = 1;
	for (int incr = 1; incr < 91; incr++) {
		one[incr] = zero[incr - 1];
		zero[incr] = zero[incr - 1] + one[incr - 1];
	}
	int test, inp;
	cin >> test;
	while(test--) {
		cin >> inp;
		cout<< one[inp] << " " << zero[inp] << endl;
	}
    return 0;
}
