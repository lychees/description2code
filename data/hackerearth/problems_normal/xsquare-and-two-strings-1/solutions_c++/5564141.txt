#include<bits/stdc++.h>
using namespace std;
int main()
{
    char s1[100001],s2[100001];
    bool b1[26],b2[26];
    unsigned int t,i,j;
    for(i=0;i<26;i++)
        b1[i]=b2[i]=false;
    scanf("%u",&t);
    while(t--)
    {   for(i=0;i<26;i++)
            b1[i]=b2[i]=false;
        int flag=0;
        scanf("%s",s1);
        scanf("%s",s2);

        for(i=97;i<=122;i++)
        {
            for(j=0;j<strlen(s1);j++)
            {
                if(i==s1[j])
                {
                    b1[i-97]=true;
                    break;
                }
            }
            for(j=0;j<strlen(s2);j++)
            {
                if(i==s2[j])
                {
                    b2[i-97]=true;
                    break;
                }
            }
            if(b1[i-97] && b2[i-97]) break;
        }
        for(i=0;i<26;i++)
        {
            if(b1[i] && b2[i])
            {flag=1; break;}
        }
        if(flag)
            printf("Yes\n");
        else
            printf("No\n");
    }

    return 0;
}
