#include<iostream>
#include<vector>

using namespace std;

typedef long long ll;

int main()
{
    int t;
    cin>>t;

    while(t!=0)
    {
       int n;
       cin>>n;

       ll val[n+1];

       for(int i=n; i>0; i--)
        {
            cin >> val[i];
        }

       ll pre[n+1];

       pre[0]=0;

       for(int i=1; i<=n; i++)
            pre[i]=pre[i-1]+val[i];

       ll dp[n];

        dp[0]=0;
        dp[1]=val[1];
        dp[2]=dp[1]+val[2];
        dp[3]=dp[2]+val[3];
        for(int i=4; i<=n; i++)
        {
            dp[i]=max(-1ll,pre[i-1]-dp[i-1]+val[i]);
            dp[i]=max(dp[i],pre[i-2]-dp[i-2]+val[i]+val[i-1]);
            dp[i]=max(dp[i],pre[i-3]-dp[i-3]+val[i]+val[i-1]+val[i-2]);
        }
        cout << dp[n] << endl;

       t--;
    }
}
