bool pode(int a, int b) {
  if (a > b) swap(a,b);
  return (2*a*a > b*b);
}

int main() {
    //freopen("in","r",stdin);
    int n,a,b;
    vi v1,v2;
    scanf("%d",&n);
    while(n--) {
      scanf("%d %d",&a,&b);
      if (a == 1) {
        v1.push_back(b);
      } else {
        v2.push_back(b);
      }
    }
    sort(v1.begin(),v1.end());
    sort(v2.begin(),v2.end());
    int cont = 0;
    int j = 0;
    for(int i = 0; i < v1.size(); i++) {
      int k = j;
      while(k < v2.size()) {
        if (pode(v1[i],v2[k])) {
          cont++;
          j = k+1;
          break;
        }
        k++;
      }
    }
    printf("%d\n",cont);



    return 0;
}
