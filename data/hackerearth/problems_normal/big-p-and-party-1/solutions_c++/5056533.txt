#include <iostream>
#include <vector>
#include <cstring>
#include <queue>
#include <algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,e;
    cin>>n>>e;
    vector <int> grid[n];
    for(int i=0;i<e;i++){
    	int x,y;
    	cin>>x>>y;
    	grid[x].push_back(y);
    	grid[y].push_back(x);
    }
    int lucky[n];
    memset(lucky,-1,sizeof(lucky));
    queue <int> q;
    q.push(0);
    lucky[0]=0;
    while(!q.empty()){
    	int s=q.front();
    	q.pop();
    	for(int j=0;j<grid[s].size();j++){
    		int t=grid[s][j];
    		if (lucky[t]==-1){
    			lucky[t]=0;
    			lucky[t]+=(lucky[s]+1);
    			q.push(t);}
    	}
    }
    for(int z=1;z<n;z++){
    	cout<<lucky[z]<<endl;
    }
    
    return 0;
}
