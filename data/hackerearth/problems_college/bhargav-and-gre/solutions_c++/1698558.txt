#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char arr[100] , a[100][100];
    int n , j = 0 , s = 0 , an = 0;

    cin >> arr;
    cin >> n;

    for(int i = 0 ; i < n ; i++)
    {
    	cin >> a[i];
    }

    for(int i = 0 ; i < n ; i++)
    {
    	if(((arr[j] - a[i][j]) == 0) && (strlen(arr) == strlen(a[i])))
    	{
    		s = s + 1;
    	}

    	else if(((a[i][j] - arr[j]) == 3) && (strlen(arr) == strlen(a[i])))
    	{
    		an = an + 1;
    	}
    }

    cout << s << " " << an;
}
