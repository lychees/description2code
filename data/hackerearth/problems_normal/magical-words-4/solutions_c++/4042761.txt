#include<bits/stdc++.h>
using namespace std;
int dp[1002][1002];
int main()
{
    int t;
    char str[10002];
    cin>>t;
    while(t--)
    {
    scanf("%s",str);
    int n=strlen(str);
    int i,j;
    for(i=0;i<=n-1;i++)
    {
    	for(j=0;j<=n-1;j++)
    {
    	dp[i][j]=0;
    }
    }
    for(i=n-1;i>=0;i--)
    {
        for(j=i;j<n;j++)
        {
            if(i==j)
                dp[i][j]=1;
            else
            {
                if(str[i]==str[j])
                {
                    if(i+1>j-1)
                    {
                        dp[i][j]=2;
                    }
                    else
                    {
                        if(dp[i+1][j-1]!=0)
                        dp[i][j]=2+dp[i+1][j-1];

                    }
                }
                else
                    dp[i][j]=0;
            }

        }
    }
    long long sum=0;
    for(i=0;i<n;i++)
    {
        for(j=i;j<n;j++)
        {
            sum=sum+dp[i][j]*dp[i][j];
        }
    }
    cout<<sum<<endl;
    }
    return 0;

}
