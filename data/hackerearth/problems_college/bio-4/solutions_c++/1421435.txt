#include <iostream>
#include <vector>
#include <algorithm>

using std::cout;
using std::cin;
using std::vector;
using std::string;

typedef long long ll;

ll n_time(const string& lng, const string& srt) {
  auto n_l = lng.size();
  auto n_s = srt.size();

  int n_min = n_s;
  
  for (size_t lo = 0; lo<n_l; lo++) {
    auto hi = lo + n_s;
    if (hi>n_l) break;

    int nd = 0;
    for (auto i = lo; i<hi; i++) {
      if (lng[i] != srt[i-lo]) nd++;
    }
    if (nd<n_min) n_min = nd;
  }
  return n_min;
}

int main() {
  int T;
  cin >> T;

  while (T--) {

    string target;
    cin >> target;
    
    string destin;
    cin >> destin;

    cout << n_time(target, destin) << '\n';
    
  }
}
