#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int a[3000005];
vector <int> v;
void sieve()
{
	long int i,j;
	for(i=2;i<=3000000;i++)
	{
		if(!a[i])
		{
			v.push_back(i);
			for(j=i*i;j<=3000000;j+=i)
			{
				a[j]=1;
			}
		}
	}
}
int main()
{
    sieve();
    int n,i;
    long int m=1e9+7;
    long long int ans=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	ans+=((upper_bound(v.begin(),v.end(),n-1+i)-v.begin())-(lower_bound(v.begin(),v.end(),i)-v.begin()));
    	ans%=m;
    }
    ans%=m;
    cout<<ans<<endl;
    return 0;
}
