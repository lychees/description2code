#include <iostream>
#include<cmath>
#define ll long long int
using namespace std;

ll gcd(ll a,ll b,ll &x,ll &y)
{
	if(a==0){
		x=0,y=1;
		return b;
	}
	ll x1,y1;
	ll d=gcd(b%a,a,x1,y1);
	x=y1-(b/a)*x1;
	y=x1;
	return d;
}

bool check(ll a,ll b,ll c,ll &x0,ll &y0,ll &g)
{
	g=gcd(abs(a),abs(b),x0,y0);
		if(c%g!=0){
		return false;
		}
		x0*=c/g;
		y0*=c/g;
		if(a<0) {
			x0*=-1;
		}
		if(b<0){	
			y0*=-1;
		}
	return true;
}

int main()
{
   ll h,u,d,t,a,b,e,f,g;
   ll ans,ans1,ans2,m;
    cin>>t;
    while(t--){
    	cin>>h>>u>>d;
    	bool c = check(u,0-d,h,a,b,e);
    	
    	if(c == false){
    		cout<<"-1"<<endl;
    		continue;
    	}
    	f = u/e;
    	g = d/e;
    	ans = max(abs(a)/g,abs(b)/f) + 1;
    	a = a + ans*g;
    	b = b + ans*f;
    	ans1 = a/g;
    	ans2 = b/f;
    	m = min(ans1,ans2);
    	a = a - m*g;
    	b = b - m*f;
    	cout<<a+b<<endl;
    	
    }
    return 0;
}
