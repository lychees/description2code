#include <iostream>
#include <set>
using namespace std;

int main() {
	int k;
	cin >> k;

	set<string> S;
	auto itr = S.begin();
	while (k--) {
		string str;
		cin >> str;

		S.insert(str);
	}

	int words;
	cin >> words;
	string result = "";

	for (int i = 0; i < words; ++i) {
		string str;
		cin >> str;

		itr = S.find(str);
		if (itr == S.end()) { // not found
			char c = str[0];
			c = toupper(c);
	
			result = result + c + ".";
		}
	}
	result.pop_back();
	cout << result << endl;
	return 0;
}
