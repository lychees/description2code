#include<bits/stdc++.h>
using namespace std;
int main()
{
	string ch;
	cin>>ch;
	int i,j,count=0,flag=0;
	for(i=0;i<ch.length();i++)
	{
		for(j=1;j<ch.length()-i+1;j++)
		{   flag=0;
			string h=ch.substr(i,j);
			for(int y=0;y<h.length()-1;y++)
			{
				if(h[y]!=h[y+1]) {
					flag=1;break;
				}
			}
			if(!flag) count++;
		}
	}
	cout<<count<<endl;
}
