#include <iostream>
#include <cstdlib>
#include<cstring>
#include<climits>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	int N,K;    	
      	scanf("%d %d",&N,&K);
      	//printf("%d %d\n",N,K);
      	char record[N][K];
      	//memset(record,,sizeof(record));
      	int i=0,j=0;
      	for(i=0;i<N;i++)
      	{
      		//char string[K+1];
      		scanf("%s",record[i]);      		
      		
      	}
      
        int	min=INT_MAX,temp_sum,k=0;
        char min_char;
        for(j=0;j<K;j++)
      	{
      		temp_sum=0,min_char=-1,min=INT_MAX;
      	for(k=1;k<=26;k++)
      	{
      		 temp_sum=0;
      		 char e1=(char)(k+96);
      		 
      	  	for(i=0;i<N;i++)
      		{
      		 temp_sum = temp_sum+abs(record[i][j]-e1);	
      		}	
      		if(temp_sum<min)
      		{
      			min=temp_sum;
      			min_char=e1;
      		}
      	}
      	printf("%c",min_char);
      	}
      	printf("\n");
    }
    return 0;
}
