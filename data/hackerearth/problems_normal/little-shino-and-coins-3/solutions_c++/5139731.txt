#include <iostream>
using namespace std;

int main()
{
	int k;
	string st;
	cin>>k>>st;
	int n = st.size();
	int i;
	int a[27];
	int co=0;
	for(i=0;i<n;i++)
	{
		int j;
		for(j=0;j<27;j++)
			a[j]=0;
		int dis=0;
		for(j=i;j<n;j++)
		{
			if(a[st[j]-'a']==0)
			{
				dis++;
				a[st[j]-'a']++;
			}
			if(dis==k)
				co++;
			
		}
	}
    cout << co<< endl;
    return 0;
}
