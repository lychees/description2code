'''
Sum of primes in a range
'''

LIMIT = 10**6
NUMBERS = [True] * LIMIT
NUMBERS[0] = NUMBERS[1] = False

for (i, isprime) in enumerate(NUMBERS):
    if isprime:
        # starting the non-prime marking at the prime's square (i*i) instead of its double.
        # Mark factors non-prime
        for number in xrange(i*i, LIMIT, i):
            NUMBERS[number] = False

if __name__ == '__main__':
    for _ in range(int(raw_input())):
        a, b = map(int, raw_input().split())
        print sum(a + num for num, isPrime in enumerate(NUMBERS[a:b + 1]) if isPrime)
