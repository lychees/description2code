#include <iostream>
using namespace std;

int main() {
	long long int t,n,x,i,sum,temp;
	cin>>t;
	while(t--){
	    cin>>n>>x;
	    sum=0;
	    long long int a[n];
	    for(i=0;i<n;i++){
	        cin>>a[i];
	    }
	    for(i=1;i<n;i++){
	        if(a[i-1]>=a[i]){
	            temp = a[i-1]-a[i];
	            temp = (temp/x)+1;
	            sum+=temp;
	            a[i] += temp*x;
	        }
	    }
	    cout<<sum<<endl;
	}
	
	return 0;
}
