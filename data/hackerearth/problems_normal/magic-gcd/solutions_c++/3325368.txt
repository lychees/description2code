#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	//cout<<a<<" "<<b<<endl;
	if(b==0)
		return a;
	return gcd(b,a%b);
}
int main()
{
	ios_base::sync_with_stdio(false);
	int t,n1,n2,k;
	cin>>t;

	while(t--)
	{
		cin>>n1>>n2>>k;
		long max=0;
		for(int i=n1-k;i<=n1+k;i++)
		{
			for(int j=n2-k;j<=n2+k;j++)
			{
				if(abs(n1-i)+abs(n2-j)<=k||abs(n2-i)+abs(n1-j)<=k)
				{
					int temp=gcd(i,j);
					//cout<<temp<<" "<<i<<" "<<j<<endl;
					max<temp ? max=temp:1;
				}
			}
		}
		cout<<max<<endl;
	}
	return 0;
}
