n_tests = input()
for tc in range(n_tests):
    inpstr = raw_input()
    chmap = [0] * 26
    found = False
    for ch in inpstr:
        chmap[ord(ch) - ord('a')] += 1
        if chmap[ord(ch) - ord('a')] >= 2:
            print 'Yes'
            found = True
            break

    if not found:
        print 'No'
