#include<bits/stdc++.h>

using namespace std;

int const N = 1e2+10;

int t, n, m;

int dp[N][N], a[N][N];

int solve(int i, int j){
  if(dp[i][j] == -1){
   int ret = 0;
   if(i+1 <= n && a[i][j] < a[i+1][j]) ret = max(ret,solve(i+1,j)+1);
   if(j+1 <= m && a[i][j] < a[i][j+1]) ret = max(ret,solve(i,j+1)+1);
   dp[i][j] = ret; 
  }
  return dp[i][j];
}

int main(){
  scanf("%d",&t);
  while(t--){
    scanf("%d%d",&n,&m);
    for(int i = 1; i <= n; i++)
     for(int j = 1; j <= m; j++)
      scanf("%d",&a[i][j]);
     memset(dp,-1,sizeof(dp));
     printf("%d\n",1+solve(1,1));
  }
}
