#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    int invalid=0,zero=0,T=0,size=0,alpha=0;
    int uc=0,lc=0;
    cin>>t;
    if(t>10 || t<1)	{
    	cout<<"Invalid Test"<<endl;
    }
    else{
    	 while(t--){
    	string s;
    	cin>>s;
    	
    	if(s.size()>=101 || s.size()<1)	size=1;
    	invalid=zero=uc=lc=T=alpha=0;
    	for(int i=0;i<s.size();i++){
    		if(s[i]>=65 && s[i]<=90){
    			uc++; alpha=1;
    		}	
    		else if(s[i]>=97 && s[i]<=122)	{
    			lc++;alpha=1;
    		}
    		else if(s[i]>=48 && s[i]<=57)	zero=1;
    		else invalid = 1;
    		
    	}
    	 
    
    	if( size==1 || alpha==0)	cout<<"Invalid Input"<<endl;
    	else if(invalid == 1 && alpha == 1) cout<<"0"<<endl;
    	 
    	else cout<<(uc>lc?lc:uc)<<endl;
    	
    	 
    	
    }
    }
   
    return 0;
}
