#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main()
{
	int t;
	string str;
	cin >> t;
	while(t--)
	{
		cin >> str;
		int n = str.size();
		vector<int> a(n);
		vector<int> b(n);
		vector<int> c(n);
		a[0] = b[0] = c[0] = 0;
		if(str[0] == 'a')
			a[0] = 1;
		else if(str[0] == 'b')
			b[0] = 1;
		else 
			c[0] = 1;
			
		for(int i=1; i<n; i++)
		{
			a[i] = a[i-1];
			b[i] = b[i-1];
			c[i] = c[i-1];
			if(str[i] == 'a')
				a[i]++;
			else if(str[i] == 'b')
				b[i]++;
			else 
				c[i]++;
		}
		long long int res = 0;
		int ra, rb, rc;
		int idx = 0, len, i=0;
		while(i<n)
		{
			ra = a[i];
			rb = b[i];
			rc = c[i];
			if(idx > 0)
			{
				ra -= a[idx-1];
				rb -= b[idx-1];
				rc -= c[idx-1];
			}
			if(ra <=0 or rb <= 0 or rc <= 0)
			{
				res += (i-idx +1);
				i++;
			}
			else 
				idx++;
		}
	
		cout << res << endl;
	}
    return 0;
}
