#include <iostream>
using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    	long long int l=0,b=0,h=0,ans=0;
    	scanf("%lld%lld%lld",&l,&b,&h);
    	ans=min(l,min(b,h));
    	if(ans&1)
    	printf("%lld.5\n",ans/2);
    	else
    	printf("%lld.0\n",ans/2);
    }
    return 0;
}
