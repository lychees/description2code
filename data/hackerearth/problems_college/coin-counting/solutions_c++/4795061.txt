#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int c=0;
    	for(int i=0;i<=n/5;i++)
    	{
    		for(int j=0;j<=n/3;j++)
    		{
    			for(int k=0;k<=n;k++)
    			{
    				if(i*5+j*3+k==n)
    				{
    					c++;
    					break;
    				}
    				else
    				if(i*5+j*3+k>n)
    				break;
    			}
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
