#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int tc;
   cin>>tc;
   while(tc--)
   {
   	long long a,b,d,e;
   	cin>>a>>b;
   	long long c=0;
   	while(a>0&&b>0)
   	{
   	     d=a%10;a=a/10;
   	     e=b%10;b=b/10;
   	     if(d+e>9)
   	     {
   	     	c++;a++;
   	     }
   	}
   	if(c==0)cout<<"No carry operation\n";
   	else if(c==1)cout<<"1 carry operation\n";
   	else cout<<c<<" carry operations\n";
   }
    return 0;
}
