#include <iostream>
#include <cassert>
#include <string>
using namespace std;

int main()
{
    int t,i,j;
    cin>>t;
    cin.ignore();
    assert(t>=1 && t<=100);
    string *s = new string[t];
    char ch;
    for(i=0;i<t;i++)
    {
    	getline(cin,s[i]);
    }
    for(i=0;i<t;i++)
    {
    	if(s[i].find("2")==-1 && s[i].find("3")==-1 && s[i].find("4")==-1 && s[i].find("5")==-1 && s[i].find("6")==-1 && s[i].find("7")==-1 && s[i].find("9")==-1)
    	{
    		for(j=0;j<s[i].length();j++)
    		{
    			if(s[i][j]!=s[i][s[i].length()-1-j])
    			{
    				cout<<"NO"<<endl;
    				break;
    			}
    		}
    		if(j==s[i].length())
    			cout<<"YES"<<endl;
    	}
    	else
    		cout<<"NO"<<endl;
    }
    return 0;
}
