#include<bits/stdc++.h>
using namespace std;
int n,a[25][25],b[25][25],c;
void dfs(int i,int j)
{
    if(i==n-1&&j==n-1){
        c++;
        return;
    }
    else
    {
        b[i][j]=1;
        if(j<n-1&&b[i][j+1]==0&&a[i][j+1]==0)
        {
            dfs(i,j+1);
        }
        if((i+1)<=n-1 && b[i+1][j]==0&&a[i+1][j]==0){
            dfs(i+1,j);
        }
        if(i>0&&b[i-1][j]==0&&a[i-1][j]==0){
            dfs(i-1,j);
        }
        if(j>0&&b[i][j-1]==0&&a[i][j-1]==0){
            dfs(i,j-1);
        }
        b[i][j]=0;
        return;
    }
}
int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
        c=0;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++){
                scanf("%d",&a[i][j]);
                b[i][j]=0;
            }
        }
        dfs(0,0);
        printf("%d\n",c);
    }
    return 0;
}
