#include <iostream>
using namespace std;

int main()
{
	int t,flag;
	long long int k;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		flag=0;
		k=1;
		for(int i=0;i<s.size()/2;i++)
		{
			if(s[i]!=s[s.size()-i-1])
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
		cout << "Palindrome" << endl;
		else
		{
			for(int i=0;i<s.size();i++)
			{
				k=k*(s[i]-'a'+1);
			}
			cout << k << endl;
		}
	}
    
    return 0;
}
