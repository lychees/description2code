#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long

ll ans[1000005][11];

int main()
{
    for(ll i=0;i<10;i++) ans[1][i]=1;
    for(ll i=2;i<1000003;i++) 
    {
    	//ans[i][0]=0;
    	for(ll j=0;j<10;j++)
    	{
    		if(j==0) ans[i][j]=ans[i-1][j+1]%mod;
    		else if(j==9) ans[i][j]=ans[i-1][j-1]%mod;
    		else ans[i][j]=(ans[i-1][j-1] + ans[i-1][j+1])%mod;
    	}
    }
    ll t;
    cin>>t;
    while(t--)
    {
    	ll a;
    	cin>>a;
    	if(a==1) { cout<<"10\n"; continue; }
    	ll res=0;
    	for(ll i=1;i<10;i++) res=(res + ans[a][i])%mod;
    	cout<<res<<"\n";
    }
    return 0;
}
