#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int T, i, j;
    string s, t;
    cin>>T;
    while(T--)
    {
    	cin>>s>>t;
    	int n=s.length();
    	int carry=0, cnt=0;
    	for(i=n-1; i>=0; --i)
    	{
    		if((s[i]-'0')+(t[i]-'0')+carry>9)
    		{
    			cnt++;
    			carry=1;
    		}
    	}
    	if(cnt==0)
    		cout<<"No carry operation"<<endl;
    	else if(cnt==1)
    		cout<<cnt<<" carry operation"<<endl;
    	else
    		cout<<cnt<<" carry operations"<<endl;
    	s.clear();
    	t.clear();
    }
    return 0;
}
