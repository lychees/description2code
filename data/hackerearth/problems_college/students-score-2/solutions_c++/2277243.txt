#include <bits/stdc++.h>

using namespace std;

int main()
{
	int i,j,k,temp,n,sum =0,max =0;
	
	cin >> n;
	
	int a[n];
	
	for(i=0; i<n;i++)
		cin >> a[i];
		
	for(i =0; i<n; i++){
		sum = sum + a[i];
		
		if(sum < 0)
			sum =0;
			
		if(max < sum)
			max = sum;
		
	}
	
	cout << max << endl;
}
