#include<iostream>
#include<algorithm>
using namespace std;

int getGCD(int x, int y)
{
	if (x < 0) 
		x = -x;
	swap(x, y);
	while (x != 0) {
        int t = x;
        x = y % x;
        y = t;
    }
    return y;
}

int main()
{
    int T, N, gcd, num, den;
   	cin >> T;
    while (T--) {
    	cin >> N;
    	N %= 360;
    	gcd = getGCD(N, 180);
		num = N / gcd;
    	den = 180 / gcd;
    	
    	if (num != 1)
			cout << num;
    	cout << "$";
    	
    	if (N && den != 1) 
    		cout << "/" << den;
    	cout << "\n";
   }
   return 0;
}
