#include<iostream>
using namespace std;
long long a[1000001][11]={0};
#define f_in(st) freopen(st,"r",stdin);
#define f_out(st) freopen(st,"w",stdout);
#define M 1000000007
int main()
{
	//f_in("in11.txt");
	//f_out("out11.txt");
	long long i,j;
	for(i=0;i<=9;i++)
	{
		a[1][i]=1;
	}
	for(i=2;i<1000001;i++)
	{
		for(j=0;j<=9;j++)
		{
			if(j==0)
			{
				a[i][j]=a[i-1][j+1]%M;
			}
			else
			{
				a[i][j]=((a[i-1][j-1])%M+(a[i-1][j+1])%M)%M;
			}
		}
	}
	long long t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long sum=0;
		if(n==1)
		{
			cout<<"10"<<endl;
		}
		else 
		{
			for(i=1;i<=9;i++)
		    {
		    	sum=(sum%M+a[n][i]%M)%M;
		    }
		    cout<<sum<<endl;
		}
	}
	return 0;
}
