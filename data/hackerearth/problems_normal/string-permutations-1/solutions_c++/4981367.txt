#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin >> n;
    vector<string> s(n, "");
    vector<int> cnt;
    for(int i = 0 ; i < n ; i++){
    	cin >> s[i];
    }
    sort(s.begin(), s.end());
    int cont = 1;
    for(int i = 1; i < n ; i++){
    	if(s[i] == s[i - 1])cont++;
    	else{
    		cnt.push_back(cont);
    		cont = 1;
    	}
    }
    cnt.push_back(cont);
    sort(cnt.begin(), cnt.end());
    //for(int i = 0 ; i < cnt.size(); i++)cout << cnt[i] << " ";
    cout << count(cnt.begin(), cnt.end(), cnt.back()) * cnt.back() << "\n";
    
    return 0;
}
