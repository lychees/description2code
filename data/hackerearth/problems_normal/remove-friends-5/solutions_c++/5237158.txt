#include <bits/stdc++.h>
#include <vector>

int main()
{
	int t,n,k,v;
	std::cin>>t;
	while(t--)
	{
		std::cin>>n>>k;
		std::list<int> f;
		std::list<int>::iterator it,next,l;
		for(int i=0;i<n;i++){
			std::cin>>v;
			f.push_back(v);
		}
		//for(int i=0;i<f.size()-1 && k>0;i++)
		for(it=f.begin();it!=f.end() && k>0;it++)
		{
			l=f.begin();
			next=it;
			next++;
			if(*it < *(next))
			{
				f.erase(it);
				k--;
				if(it==l)
					it=f.begin();
				else
				{
					it--;
					it--;
					
				}
			}
		}
		if(k>0)
		{
			for(int i=0;i<k;i++)
				f.pop_back();
		}
		for(it=f.begin();it!=f.end();it++)
			std::cout<<*it<<" ";
		std::cout<<"\n";
	}
	return 0;
}
