#include <bits/stdc++.h>
using namespace std;

int main()
{
	int k,n;
    cin>>n>>k;
    
	int prime[10001];
	
	prime[0]=prime[1]=-1;
	for(int i=2;i<10001;++i){
		if(!prime[i]){
			for(int j=i*i;j<10001;j+=i){
				prime[j]=-1;
			}
		}
	}
	for(int i=10000;i>=0;--i){
		if(prime[i]){
			if(i+k<=10000 && prime[i+k]!=-1){
				prime[i]=1+prime[i+k];
			}
		}
	}
	
    int arr[n][n];
    for(int i=0;i<n;++i)
    	for(int j=0;j<n;++j)
    		cin>>arr[i][j];
    	
    
    int cnt=0,min_res=INT_MAX,res_x,res_y;
    bool flag;
    for(int i=1;i<n-1;++i){
    	flag=true;
    	for(int j=1;j<n-1;++j){
    		if(prime[arr[i][j]]==-1||prime[arr[i-1][j-1]]==-1||prime[arr[i-1][j+1]]==-1||
    		prime[arr[i+1][j-1]]==-1||prime[arr[i+1][j+1]]==-1){
    			continue;
    		}
    		cnt=(prime[arr[i][j]]+prime[arr[i-1][j-1]]+prime[arr[i-1][j+1]]+
    		prime[arr[i+1][j-1]]+prime[arr[i+1][j+1]]);
    		if(cnt<min_res){
    			min_res=cnt;
    			res_x=i;
    			res_y=j;
    		}
    	}
    }
    if(min_res==INT_MAX)
    	cout<<"no\n";
    else{
    	cout<<"yes\n";
    	cout<<min_res<<endl<<res_x+1<<" "<<res_y+1<<endl;
    }
    
    return 0;
}
