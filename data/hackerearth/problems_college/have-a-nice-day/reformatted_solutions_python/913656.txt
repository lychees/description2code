t = input()
while t:
    t = t-1
    n = input()
    man = map(int, raw_input().split())
    wom = map(int, raw_input().split())
    man.sort()
    wom.sort()
    sumup = 0
    for x in range(0, n):
        sumup = sumup + (man[x]*wom[x])
    print sumup
