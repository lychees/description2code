#include <bits/stdc++.h>
#define ll long long

using namespace std;

vector <ll> vec,vec1;

ll swapfunc (ll a[], ll n, ll j) {
    ll pos = j, maxi = a[j];
    for (int i = j; i < n; i++) {
        if (a[i] > maxi) {
            maxi = a[i];
            pos = i;
        }
    }

    ll tmp;
    if (maxi != a[j]) {
        tmp = a[j];
        a[j] = a[pos];
        a[pos] = tmp;
        return 1;
    }

    return 0;
}

int main() {
    ll n, k, i, j;
    cin >> n >> k;
    ll a[n];

    for (i = 0; i < n; i++)
        cin >> a[i];
	
	int arr1[n];
	for(int i=0;i<n;i++)
	   arr1[i]=n-i;
	
	if(k >= n) {
	   for(int i=0;i<n;i++)
	   cout<<arr1[i]<<" ";
	   cout<<endl;
	   exit(0);
	}
    j = 0;
    while (k && j < n) {
        if(swapfunc(a, n, j)) {
            k--;
        }
        j++;
    }

    for (i = 0; i < n; i++)
        cout << a[i] << " ";

    return 0;
}
