#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int a,b,c;
    while(t--)
    {
    	cin>>a>>b;
    	int ans = 0;
    	c = 0;
    	while(a>0||b>0)
    	{
    		c = (a%10) + (b%10) + c;
    		if(c>=10)
    		{
    		  ans++;
    		}
    		c/=10;
    		a/=10;
    		b/=10;
    		
    	}
    	
    	if(ans==0)
    	{
    		printf("No carry operation\n");
    		continue;
    	}
        if(ans==1)
        {
        	printf("1 carry operation\n");
        	continue;
        }
        printf("%d carry operations\n",ans);
    }
    return 0;
}
