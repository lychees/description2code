#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int ar[n];
    for(int i=0; i<n; i++)
    {
        scanf("%d",&ar[i]);
    }
    int flag=0;
    sort(ar,ar+n);
    for(int i=0; i<n; i++)
    {
        if(ar[i]%2==1)
        {
            flag++;
        }
    }
    if(flag==0)
        printf(":(\n");
    else
    {
        int sum=0;
        for(int i=0; i<n; i++)
            sum+=ar[i];
        if(flag%2==1)
            printf("%d\n",sum);
        else
        {
            int temp;
            for(int i=0;i<n;i++)
            {
                if(ar[i]%2==1)
                {
                    temp=ar[i];
                    break;
                }
            }
            sum-=temp;
            printf("%d\n",sum);
        }


    }
}
