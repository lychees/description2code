#include<bits/stdc++.h>

using namespace std;

int main()
{
	char s[200005];
    cin>>s;
    int l = strlen(s);
    long long a = s[0]-'0', b=s[0]-'0';
    for(int i=1;i<l;i++)
    {
        b=b*10+(i+1)*(s[i]-'0');
        b=b%1000000007;
        a=(a+b)%1000000007;
    }
    cout<<a<<endl;
    return 0;
}
