void swap(int a[], int i, int j)
{
	for (int k =j;k>i;k--)
	{
		
		int temp = a[k];
		a[k]=a[k-1];
		a[k-1]=temp;
	}
}


void f(int a[], int s,int e, int k)
{
	if(s>=e || k==0)
		
	return;
	
	
	
	int i =s;
	
	for ( ;i<=e && i<=s+k;i++)
		if(a[i]<a[s]) break;
	if(i==e+1 || i==s+k+1)
	{
		f(a,s+1,e,k);return ;
	}
	int minind=i;
		
	for(;i<=e&&i<=s+k;i++)
	{
		if(a[i]<a[minind]) minind=i;
	}
		
		
	
	
		swap(a,s,minind);
		
		f(a,s+1,e,k-( minind-s ));
		
}
