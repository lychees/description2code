
#include<bits/stdc++.h>
typedef long long int ll;

int main()
{
	int t;
	scanf("%d",&t);
	ll n;
	while(t--)
	{
		scanf("%lld",&n);
		if(n==1)
		{
			printf("Hacker\n");continue;
		}
		if(n==2)
		{
			printf("Earth\n");continue;
		}
		ll sum=2;ll p=1;
		while(sum<=n)
		{
			p++;
			sum+=(1LL<<p)*p;
		}
		sum-=(1LL<<p)*p;
		ll rem=n-sum;
		if(rem==0)
		{
			printf("Earth\n");
			continue;
		}
		ll pos1=rem/p;
		ll pos2=rem%p;
		if(pos2==0) 
		{
			if( (pos1-1) &1LL )
			{
				printf("Earth\n");
			}
			else
			{
				printf("Hacker\n");
			}
			continue;
		}
		if(pos1 &(1LL<<(p-pos2)) )
		{
			printf("Earth\n");
		}
		else
		{
			printf("Hacker\n");
		}
	}
	return 0;
}
