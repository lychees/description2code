#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	   {
	   	long int a,max=0,max1=0;
	   	for(int j=0;j<3;j++)
	   	    {
	   	    	cin>>a;
	   	    	if(a>max)
	   	    	  {
	   	    	  	max1=max;
	   	    	  	max=a;
	   	    	  }
	   	    	 else if(a>max1)
	   	    	 max1=a;
	   	    }
	   	 if(max1<max)
	   	 cout<<(2*max1)+1;
	   	 else if(max1==max)
	   	 cout<<max+max1;
	   	 else if(max1==0&&max1==0)
	   	  cout<<0;
	   	 else
	   	 cout<<1;
	   	 cout<<endl;
	   }
    //cout << "Hello World!" << endl;
    return 0;
}
