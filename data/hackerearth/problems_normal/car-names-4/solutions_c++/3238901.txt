#include <iostream>
using namespace std;

int main()
{
    int t,count,n1,n2,n3,i;
    char name[501];
    cin>>t;
    while(t--)
    {
    	count=0;
    	i=0;
    	n1=1;
    	n2=1;
    	n3=1;
    	cin>>name;
    	while(name[i+1]==name[i])
    	{
    		n1++;
    		i++;
    	}
    	i++;
    	while(name[i+1]==name[i])
    	{
    		n2++;
    		i++;
    	}
    	if(n1!=n2)
    	{
    		cout<<"Not OK"<<endl;
    		continue;
    	}
    	i++;
    	while(name[i+1]==name[i])
    	{
    		n3++;
    		i++;
    	}
    	if(n2!=n3)
    	{
    		cout<<"Not OK"<<endl;
    		continue;
    	}
    	i++;
    	if(name[n1-1]==name[n1+n2+n3-1])
    		cout<<"Not OK"<<endl;
    	else if(name[i]!='\0')
    		cout<<"Not OK"<<endl;
    	else
    		cout<<"OK"<<endl;
    }
    return 0;
}
