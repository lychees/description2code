#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int mpow(long long a,long long b){
long long ret=1;
while(b){
    if(b&1){ret=ret*a; ret%=mod;}
    a*=a;
    a%=mod;
    b>>=1;
}
return ret;
}
int main(){
int t;
cin>>t;
while(t--){
        vector<int> v;
    int n;
    cin>>n;
for(int i=0;i<n;i++){
    int num;
    cin>>num;
    v.push_back(num);
}
if(n==1&&v[0]==1) {cout<<"1"<<endl; continue;}
int primes[10001];
memset(primes,0,sizeof(primes));
for(int i=0;i<n;i++){
    //consider v[i]
    int count=0;
    while(v[i]%2==0){
        v[i]>>=1;
        count++;
    }
    primes[2]=max(primes[2],count);
    count=0;
    for(int k=3;k<=sqrt(v[i]);k+=2){
        while(v[i]%k==0){
            count++;
            v[i]/=k;
        }
        primes[k]=max(primes[k],count);
        count=0;
    }
    if(v[i]>2) primes[v[i]]=max(primes[v[i]],1);

}
long long ans=1;
for(int i=2;i<=10000;i++){
    if(primes[i]) ans*=mpow(i,primes[i]);
    if(ans>mod) ans%=mod;
}
cout<<ans<<endl;


}
}
