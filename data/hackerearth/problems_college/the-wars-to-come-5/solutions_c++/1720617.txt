#include <iostream>
using namespace std;

int main()
{
    int N,wildling[1000],S,sum=0,large,count=0,pos;
    //Enter size of army
    cin>>N;
    //Enter strength of army
    for(int i=0;i<N;++i)
    	cin>>wildling[i];
    //Enter total strength
    cin>>S;
    //Finding minimum number
    for(int i=0;i<N;++i)
    {
    	large=wildling[i];
    	for(int j=i+1;j<N;++j)
    		if(wildling[j]>large)
    		{
    			large=wildling[j];
    			pos=j;
    		}
    	wildling[pos]=0;
    	sum+=large;
    	count++;
    	if(sum>=S)
    		break;
    }
    //Display the Value
    cout<<count;
    return 0;
}
