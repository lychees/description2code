#include <iostream>
#include <algorithm>
using namespace std;
/////////
/////////
int main() {
	int T,N,count1,i;
	////////////////
	cin>>T;
	int a[150];
	
	while(T--){
		count1=0;
		cin>>N;
		for(i=0;i<N;i++)
			cin>>a[i];///////////
		
		sort(a,a+N);
		
		for(i=0;i<N;i++){
			if(i==N-1){
				count1++;//////////
				break;
			}
////////////
			if(a[i+1]==a[i]+1){
				
				i++;
			}
			count1++;
		}//////////////
		
		cout<<count1<<endl;
	}
	///////////////////
	return 0;
}
