int shuffle(string S) { 
	int count[26];
	memset(count, 0, sizeof(count));
	int l = S.size();
	for(int i = 0; i < l; i++){
	    count[S[i] - 'a']++;
	}
	
	int odds = 0;
	
	for(int i = 0; i < 26; i++){
	//	cout<<count[i]<<endl;
	    if(count[i] % 2 == 1){
	        odds++;
	    }
	}
	
	return odds;
}
