#include <iostream>
#include <cassert>
using namespace std;

int main()
{
   int n,i;
	long int arr[100000], max1 = 0, max2 = 0, min1 = 0, min2 = 0,result,temp;
	cin >> n;
	assert(2 <= n & n <= 100000);
	for (i = 0; i < n; i++)
	{
		cin >> arr[i];
		assert(-1000000000 <= arr[i] & arr[i] <= 1000000000);
	}
	max1 = min1= arr[0];
	max2 = min2 = arr[1];

	if (max2 > max1)
	{
		temp = max1;
		max1 = max2;
		max2 = temp;
	}
	if (min2 < min1)
	{
		temp = min1;
		min1 = min2;
		min2 = temp;
	}
	for ( i = 2; i < n; i++)
	{
		if (arr[i] >= max1)
		{
			max2 = max1;
			max1 = arr[i];
		}
		else if(arr[i]>max2)
		{
			max2 = arr[i];
		}
		

		if (arr[i] <= min1)
		{
			min2 = min1;
			min1 = arr[i];
		}
		else if (arr[i] < min2)
		{
			min2 = arr[i];
		}
		
	}

	result = ((max1*max2)> (min1*min2) ? (max1*max2) : (min1*min2));
	cout << result;
    return 0;
}
