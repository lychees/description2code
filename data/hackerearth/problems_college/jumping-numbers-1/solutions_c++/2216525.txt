#include<bits/stdc++.h>
using namespace std;

#define REP(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define ll long long
#define ull unsigned ll
#define mod 4294967143
ll dp[300][15][15];
int main()
{
    for(int i=0;i<=10;i++)
    dp[1][i][i] =1;
    for(int d=2;d<=264;d++)
    {
        for(int i=1;i<=10;i++)
        {
            for(int j=0;j<=10;j++)
            {
                dp[d][i][j] = dp[d-1][i][j-1] + dp[d-1][i][j+1];
                if(dp[d][i][j]>=mod)
                while(dp[d][i][j]>=mod)
                dp[d][i][j] -= mod;
            }
        }
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int ans=0;
        for(int i=1;i<=10;i++)
        {
            ans  = (ans + dp[n][i][i+1] + dp[n][i][i-1])%mod;
        }
        printf("%d\n",ans);
    }
    return 0;
}
