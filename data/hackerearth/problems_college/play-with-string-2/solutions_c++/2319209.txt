#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
char s[100000],r[100000];
cin>>s;
memset(r,0,sizeof(r));
for(int i=0;i<strlen(s);i++)
r[strlen(s)-i-1]=s[i];
int f=0;
for(int i=1;i<strlen(s);i++)
{
                int p,q;
                p=abs(s[i]-s[i-1]);
                q=abs(r[i]-r[i-1]);
                if(p!=q)
                                f=1;
}
if(f==0)
                cout<<"Funny"<<endl;
if(f==1)
                cout<<"Not Funny"<<endl;
}
return 0;
}
