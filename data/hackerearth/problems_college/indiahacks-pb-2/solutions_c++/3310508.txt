#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    char sol[50];
    cin >> n;
    char a[n][50];

    for (int i = 0;i < n;i++)
        cin >> a[i];

    //char ans[50];
    for (int i = 0;i <n;i++)
    {
    	int k = 0;

    	for (int z = strlen(a[i])-1;z >= 0;z--)
    	{
    		sol[k++] = a[i][z];
    	}

        for (int j = i+1;j < n;j++)
        {
            if (strcmp(a[j],sol) == 0)
            {
                cout << strlen(a[i]) << " ";
                cout << a[i][strlen(a[i]) / 2] << endl;
                break;
            }
        }
    }

    return 0;
}
