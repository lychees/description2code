#include<iostream>
#include<stdio.h>
#include<math.h>

using namespace std;

#define ll long long int

ll power(ll a,ll b,ll m)
{
	ll res=1;

	while(b>0)
	{
		if(b%2==0)
		{
			b=b/2;
			a=(a*a)%m;
		}
		else
		{
			res*=a;
			res=(res%m);
			b--;
		}
	}
	return res;

}

int main()
{
	ll i,n,k,a,b,t;
	scanf("%lld",&t);

	while(t--)
	{
		scanf("%lld %lld %lld",&a,&b,&k);

		n=power(a,b,pow(10,k));
		k=0;
		while(n!=0)
		{
			k+=(n%10);
			n/=10;
		}
		printf("%lld\n",k);
	}
	return 0;
}
