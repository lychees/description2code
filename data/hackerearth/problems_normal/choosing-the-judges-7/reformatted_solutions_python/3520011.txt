t = input()
for m in xrange(t):
    n = input()
    # print 'Case '+str(m+1)+': '+maxi[n-1]
    l = map(int, raw_input().split())
    if n == 0:
        print 'Case '+str(m+1)+': '+'0'
    else:
        maxi = [0]*n
        maxi[0] = l[0]
        if n >= 2:
            maxi[1] = max(l[0], l[1])

        for i in xrange(2, n):
            maxi[i] = max(l[i]+maxi[i-2], maxi[i-1])
            # print 'Case '+str(m+1)+': '+str(maxi[n-1])

        print 'Case '+str(m+1)+': '+str(maxi[n-1])
