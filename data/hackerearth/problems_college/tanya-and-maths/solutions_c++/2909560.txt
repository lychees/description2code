#include <iostream>
#include<cstdio>
using namespace std;
int fn(int a,int b)
{
	int temp;
	while(b!=0)
        {
            temp=b;
            b=a%b;
            a=temp;
        }
     return a;
}
int main()
{
    int t,a,b;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d",&a,&b);
    	printf("%d\n",fn(a,b));
    }
    return 0;
}
