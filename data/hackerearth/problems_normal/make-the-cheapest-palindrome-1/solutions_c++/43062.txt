#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
  	int t,i,j;
  	cin>>t;
  	while(t--)
  	{
  		int a,b,ans=0;
  		bool flag=true;
  		string s;
  		cin>>s;
  		cin>>a>>b;
  		i=0,j=s.size()-1;
  		while(i<j)
  		{
  			if(s[i]==s[j])
  			{
  				if(s[i]=='/')
  				ans+=(2*min(a,b));
  			}
  			else
  			{
  				if((s[i]=='a' && s[j]=='b') || (s[i]=='b' && s[j]=='a'))
  				flag=false;
  				else if(s[i]=='a' || s[j]=='a')
  				ans+=a;
  				else if(s[i]=='b' || s[j]=='b')
  				ans+=b;
  				else
  				ans+=min(a,b);
  			}
  			if(!flag)
  			break;
  			i++,j--;
  		}
  		if(!flag)
  		cout<<"-1"<<endl;
  		else
  		cout<<ans<<endl;
  	}
    return 0;
}
