#include <bits/stdc++.h>

using namespace std;

int main()
{
    string s,part="",converted = "";
    locale settings;
    getline(cin,s);
    map<string,int>m;
    vector<string>v;

    for(int i=0; i<s.length(); i++)
    {


        if(s[i] != ' ') part += s.substr(i,1);
        else if(s[i] == ' ')
        {
            for(int j = 0; j < part.size(); ++j)
            {
                converted += (toupper(part[j], settings));
            }
            part = "";
            m[converted]++;
            if(m[converted] == 1) v.push_back(converted);
            converted="";
        }



    }

    for(int j = 0; j < part.size(); ++j)
    {
        converted += (toupper(part[j], settings));
    }

    part = "";
    m[converted]++;
    if(m[converted] == 1) v.push_back(converted);



    for(int i=0; i<v.size(); i++)
    {
		 cout << v[i] << " " << m[v[i]] << endl;
    }






    return 0;
}
