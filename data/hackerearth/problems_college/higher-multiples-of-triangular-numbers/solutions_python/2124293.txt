from math import sqrt
def factors(n):
	final = set()
	for i in xrange(1,int(sqrt(n))+1):
		if n%i == 0:
			final.add(i)
			final.add(n/i)
	return len(final)

for i in range(input()):
	num = input()
	diff = 3
	start = 3
	ans = start
	while factors(start) <= num:
		start += diff
		diff += 1
		ans = start
	print ans
