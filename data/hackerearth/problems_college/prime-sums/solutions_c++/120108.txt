#include<iostream>
#include<math.h>
#include<vector>
using namespace std;
int main()
{
long long n,sum=0;
scanf("%lld",&n);
vector<bool>po(n+1,true);
long long limit=sqrt(n);
po[0]=po[1]=false;

for(long long i=2;i<=limit;i++)
{
for(long long j=i*i;j<=n;j+=i)
po[j]=false;
}

for(long long k=2;k<n;k++)
{
if(po[k])
{
sum+=k;
}}
cout<<sum<<"\n";
}
