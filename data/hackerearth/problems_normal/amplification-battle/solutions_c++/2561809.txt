#include <bits/stdc++.h>
using namespace std;

int main()
{
    long t;
    scanf("%ld",&t);
    while(t--)
    {
    	long long a,b,c;
    	scanf("%lld%lld%lld",&a,&b,&c);
    	long long ans,count=1;
    	ans=ceil((double)c/b);
    	while(ans>1)
    	{
    		if(count%2!=0)
    		   ans=ceil((double)ans/a);
    		else
    		   ans=ceil((double)ans/b);
    		count++;
            //cout<<count<<" "<<ans<<endl;
    	}
    	if(count%2==0)
    	   cout<<"Bob\n";
    	else
    	   cout<<"Adam\n";
    }
    return 0;
}
