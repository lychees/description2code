#include <iostream>
using namespace std;

unsigned int gcd (unsigned int a, unsigned int b)
    {
      unsigned int x;
      while (b)
        {
          x = a % b;
          a = b;
          b = x;
        }
      return a;
    }
 
void checkPrime(int no)
{
	int i;
	if(no==1)
	{
		cout<<"FALSE"<<endl;
		return;
		
	}
	for(i=2;i<no;i++)
	{
		if(no%i == 0)
		{
			cout<<"FALSE"<<endl;
			return;
		}
	}
	cout<<"TRUE"<<endl;
	
}

int main()
{
	int t,i,n,cnt;
	cin>>t;
	while(t--)
	{
		cnt=0;
		cin>>n;
		if(n==1)
		{
			cout<<"FALSE"<<endl;
			continue;
		}
		for(i=1;i<n;i++)
		{
			if(gcd(n,i)==1)
				cnt++;
		}
	//	cout<<cnt<<endl;
		checkPrime(cnt);
	}
    return 0;
}
