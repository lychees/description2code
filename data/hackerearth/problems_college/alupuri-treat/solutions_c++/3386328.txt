#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	int m[101][101];

	cin>>n;
	for(int i = 0;i<n;i++){
		for(int j =0;j<n;j++){
			cin>>m[i][j];
		}
	}
	int f = 0;
	int s = 0;
	for(int i = 0,j=0;i<n,j<n;i++,j++){
		f = f + m[i][j];
	}
	for(int i = n-1,j=0;i>=0,j<n;i--,j++)
		s += m[i][j];

	cout<<abs(f-s)<<endl;
	return 0;
}
