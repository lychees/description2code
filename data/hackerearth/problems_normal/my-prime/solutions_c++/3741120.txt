#include <iostream>
using namespace std;

int main()
{
	int n,i,count,j;
	cin>>n;
	int * a=new int[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
		count=0;
		for(j=0;j<n;j++)
		{
			if(i!=j)
			{
				if(a[i]%a[j]==0)
				count++;
			}
		}
		if(count==0)
		 cout << a[i] << " ";
	}
   delete a;
    return 0;
}
