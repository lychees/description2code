int main()
{
	int t;
	string s;
	cin >> t;
	while(t--)
	{
		cin >> s;
		int l = s.size();

		int st = 0;
		int cnt1 = 0;
		for(int i=0;i<l;i++)
		{
			if(st == 0 && s[i] == 'X')
			{
				cnt1++;
				st = (st + 1)%2;	
			}
			else if(st == 0 && s[i] == 'Y')
				continue;
			if(st == 1 && s[i] == 'Y')
			{
				cnt1++;
				st = (st + 1)%2;
			}
			else if(st == 1 && s[i] == 'X')
				continue;
		}
		int op = l - cnt1;
		st = 1;
		cnt1 = 0;
		for(int i=0;i<l;i++)
		{
			if(st == 0 && s[i] == 'X')
			{
				cnt1++;
				st = (st + 1)%2;	
			}
			else if(st == 0 && s[i] == 'Y')
				continue;
			if(st == 1 && s[i] == 'Y')
			{
				cnt1++;
				st = (st + 1)%2;
			}
			else if(st == 1 && s[i] == 'X')
				continue;
		}
		op = min(op,l-cnt1);
		cout << op << endl;
	}

	return 0;
}
