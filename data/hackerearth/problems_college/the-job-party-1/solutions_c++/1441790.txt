#include <bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--) {
    	long long int n,k1;
    	scanf("%lld%lld",&n,&k1);
    	k1++;
    if(k1==1)
    printf("0\n");
    else if(k1%2==0) {
    		printf("%lld\n",(k1/2)*(k1-1));
    	}
    	else
    	printf("%lld\n",(k1)*((k1-1)/2));
    }
    return 0;
}
