#include <iostream>
#define MAX 100
using namespace std;

int main()
{
    int n,i,j;
    int b[MAX],a[MAX],sum=0;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>b[i];
    }
    a[0] = b[0];
    for(j=1;j<n;j++)
    {
    	sum+=a[j-1];
    	a[j]=b[j]*(j+1)-(sum);
    }
    for(int k=0;k<n;k++)
    {
    	cout<<a[k]<<" ";
    }
    return 0;
}
