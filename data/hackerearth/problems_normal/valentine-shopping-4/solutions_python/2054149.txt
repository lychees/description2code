t = input()
for tc in xrange(t):
	n,m = map(int,raw_input().split(" "))
	pura_final = []
	for x in xrange(n):
		final = []
		for y in xrange(m):
			total = 1000000
			s = raw_input()
			mynums = [int(i) for i in s.split(" ")]
			for ele in mynums:
				total-=((total*ele)/100)
			final.append(total)
		pura_final.append(final.index(min(final))+1)
		pura_final = map(str,pura_final)
	print ' '.join(pura_final)
