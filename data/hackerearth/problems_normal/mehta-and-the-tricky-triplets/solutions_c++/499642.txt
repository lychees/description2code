void hasDigit(long long a,bool &p,bool &q,bool &r,bool &s){
	int d;
	
	while(a){
		d=a%10;
		if(d==2)		p=true;
		else if(d==3)	q=true;
		else if(d==5)	r=true;
		else if(d==7)	s=true;
		a/=10;
	}
	
	return;
}

int main(){
	bool a[100010][4];
	long long i,x,n;
	bool p,q,r,s;
	
	cin>>n;
	
	for(i=0;i<n;i++){
		scanf("%lld",&x);
		p=q=r=s=false;
		hasDigit(x,p,q,r,s);
		a[i][0]=p;a[i][1]=q;a[i][2]=r;a[i][3]=s;
	}
	
	long long t,e[4],f[4][4],g[4][4][4],all=0;

	for(int j=0;j<4;j++){
		for(i=t=0;i<n;i++)
			if(a[i][j]){
				t++;			
			}
			
		if(t>=3)
			e[j]=( ((t)*(t-1)*(t-2))/6 );		
		else
			e[j]=0;
	}
	
	for(int j=0;j<4;j++){
		for(int k=j+1;k<4;k++){
			for(i=t=0;i<n;i++)
				if(a[i][j] && a[i][k]){
					t++;			
				}
			
			if(t>=3)
				f[j][k]=( ((t)*(t-1)*(t-2))/6 );		
			else
				f[j][k]=0;
		}
	}
	
	for(int j=0;j<4;j++){
		for(int k=j+1;k<4;k++){
			for(int l=k+1;l<4;l++){
				for(i=t=0;i<n;i++)
					if(a[i][j] && a[i][k] && a[i][l]){
						t++;			
					}
			
				if(t>=3)
					g[j][k][l]=( ((t)*(t-1)*(t-2))/6 );		
				else
					g[j][k][l]=0;
			}
		}
	}
	
	for(i=t=0;i<n;i++)
		if(a[i][0] && a[i][1] && a[i][2] && a[i][3]){
			t++;			
		}
			
	if(t>=3)
		all=( ((t)*(t-1)*(t-2))/6 );		
	else
		all=0;
	
	cout<<(e[0] + e[1] + e[2]+ e[3] - f[0][1] - f[0][2] - f[0][3] - f[1][2] -f[1][3] - f[2][3] + g[0][1][2] + g[0][1][3] + g[0][2][3] + g[1][2][3] - all    );
	
	return 0;
}
