import math
sqr=int(math.sqrt(10**6+100))
prm =[ 0 for i in range(10**6+100)]
def prim():

	for c in range(2,sqr,1):

		if prm[c]==0:

			for d in range(c*c,10**6+12,c):

				prm[d]=1


for i in range(int(raw_input())):
	a, b=map(int, raw_input().split())
	prim()
	cnt=0
	if a>2:
		print 0

	else:

		for k in range(3,b+1,1):

			if prm[k]==0 and k>=a:
				cnt+=1


		print cnt
