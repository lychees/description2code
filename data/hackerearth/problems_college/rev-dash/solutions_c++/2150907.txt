#include<iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

void reverse_string(char *t)
{
    int l, c;
    char *e, s;

    l = strlen(t);
    e = t + l - 1;

    for (c = 0; c < l/2; c++)
    {
        s  = *t;
        *t = *e;
        *e = s;
        t++;
        e--;
    }

}

void add_dash(char *s)
{
    int l,c,i=1;
    char z[100];
    l = strlen(s);
    z[0] = s[0];
    z[1] = '-';
    for(c = 2 ; c < (2*l)-1 ; c++)
    {
        if(c%2==0)
        {
            z[c] = s[i];
            i += 1;
        }
        else
        {
            z[c] = '-';
        }
    }
    z[c] = '\0';
    cout<<z<<"\n";
    reverse_string(z);
    cout<<z;
}

int main()
{
    char a[100];
    scanf(" %[^\n]",a);
    add_dash(a);
    return 0;
}
