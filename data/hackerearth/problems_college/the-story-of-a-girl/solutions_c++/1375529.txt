#include<bits/stdc++.h>
#define rep(i,n) for(i=0;i<n;i++)
void mul(long int f[][2],long int m[][2])
{
    long int i,j,k,res[2][2];
    memset(res,0,sizeof(res));
    rep(i,2) rep(j,2) rep(k,2)
    res[i][j]=(res[i][j]+f[i][k]*m[k][j])%1000000007;

    rep(i,2) rep(j,2)
    f[i][j]=res[i][j];
}
void power(long int f[][2],long int n)
{
    if(n==0 || n==1)
        return ;
    long int m[2][2]={{1,1},{1,0}};
    power(f,n/2);
    mul(f,f);
    if(n%2!=0)
        mul(f,m);
}
long int fib(long int n)
{
    if(n==0)
        return 0;
    long int f[2][2]={{1,1},{1,0}};
    power(f,n-1);
    return f[0][0]%1000000007;
}
int main()
{
    long int tc,n;
    scanf("%ld",&tc);
    while(tc--)
    {
        scanf("%ld",&n);
        printf("%ld\n",fib(n)%1000000007);
    }
    return 0;
}
