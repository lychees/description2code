t = input()
i = 1
while i <= t:
    str = raw_input()
    list = str.split()
    x1 = list[0]
    y1 = list[1]
    x2 = list[2]
    y2 = list[3]
    if y1 != x2 and y2 != x1:
        print "0"
    elif y1 == x2 and y2 == x1:
        print "2"
    elif y1 == x2 or y2 == x1:
        print "1"
    i = i + 1
