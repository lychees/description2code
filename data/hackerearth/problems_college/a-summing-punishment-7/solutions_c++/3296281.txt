#include <iostream>
using namespace std;

int main()
{
    int n, t, i, j, k, sum;
    cin>>n;
    int ar[n];
    for(i=0;i<n;i++){
    	cin>>ar[i];
    }
    cin>>t;

    for(k=0;k<t;k++){
    	cin>>i;
    	cin>>j;
    	sum=0;i--;j--;
    	for(;i<=j;i++){
    		sum+=ar[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
