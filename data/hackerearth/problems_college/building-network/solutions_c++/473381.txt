#include <iostream>
using namespace std;
unsigned long long ar[1005];


int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	int n;
    	cin>>n;
    	int i=0;
    	while(i<n)
    	{
    		cin>>ar[i];
    		i++;
    	}
    	i=0; unsigned long long sum=0,mn=-1;
    	while(i<n)
    	{
    		sum=sum+ar[i];
    		mn=min(mn,ar[i]);
    		i++;
    	}
    	sum=sum+(n-2)*mn;
    	cout<<sum<<endl;
    	t--;
    }
}
