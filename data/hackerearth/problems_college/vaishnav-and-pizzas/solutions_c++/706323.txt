#include <bits/stdc++.h>
using namespace std;
long long dp[10005];
static unsigned int phi(unsigned int x) {
unsigned int ret = 1,i,pow;
for (i = 2; x != 1; i++) {
pow = 1;
while (!(x%i)) {
x /= i;
pow *= i;
}
ret *= (pow - (pow/i));
}
return ret;
}
void pre(){
	dp[0]=0;
	dp[1]=0;
	dp[2]=1;
	for(int i=3;i<=10000;i++)
		dp[i]=dp[i-1]+phi(i);
}
int main(){
	pre();
	int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		cout<<dp[n]<<endl;
	}
	return 0;
}
