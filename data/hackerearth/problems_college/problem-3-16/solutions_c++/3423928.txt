int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin >> t;
	for(int i=0;i<t;i++)
	{
		int n, k;
		cin >> n >> k;
		vector<int> mj5(k + 1, 0);
		for(int j=0;j<n;j++)
		{
			int mjj;
			cin >> mjj;
			mj5[mjj-1]++;
		}
		int ans=INT_MAX;
		for(int kk=0;kk<k;kk++)
		{
			if(mj5[kk]!=0)
			{
				int hh = n-mj5[kk];
				if(hh<ans)
					ans=hh;
			}
		}
		cout << ans << endl;
	}
   return 0;
}
