#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--) {
		int n,min=50,i,c;
		unsigned long long s,p;
		scanf("%d",&n);
		for(i=1;i<=n;i++) {
			c=0;
			scanf("%llu",&s);
			p=s;
			while(p!=0) {
				p=p&(p-1);
				c++;
			}
			if(c<min)
				min=c;
		}
		cout<<min<<endl;
	}
	return 0;
}
