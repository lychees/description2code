#include <iostream>
using namespace std;

int main()
{
    long long a[100001],i,x,c,y;
    int tc;
    cin>>tc;
    a[0]=0;
    a[1]=1;
    for(i=2;i<100001;i++)
    {
    a[i]=a[i-1]+a[i-2];	
    }
    while(tc--)
    {c=0;
    	cin>>x>>y;
    	for(i=0;i<100001;i++)
    	{
    		if(a[i]>y)
    		break;
    		if(a[i]>=x && a[i]<=y)
    		c++;
    	}
    	if(c%2==0)
    	cout<<"VALID\n";
    	else
    	cout<<"INVALID\n";
    }
    return 0;
}
