#include <bits/stdc++.h>
using namespace std;
#define ll long long
const int MOD = 1e9 +7;
ll pw(ll a,ll b,ll m)
{
    ll r = 1;
    while(b)
    {
        if(b&1)
            r = (r*a)%m;
        b >>= 1;
        a = (a*a)%m;
    }
    return r;
}
ll dm[101],c[101][101];

int main()
{
	int n,i,l,r,x,j;
	cin>>n>>l>>r;
	dm[0]=1;
	dm[1]=0;
	for(i=2;i<=n;i++){
		dm[i] = ((i-1)*(dm[i-1]+dm[i-2]))%MOD;
	}
	for(i=0;i<=n;i++){
		for(j=0;j<=i;j++){
			if(j==0 || i==j)
				c[i][j]=1;
			else
				c[i][j] = (c[i-1][j] + c[i-1][j-1])%MOD;
		}
	}
	ll p=0,q=0;
	for(i=0;i<=n;i++){
		ll ways= (c[n][i]*dm[i])%MOD;
		if(i>=l && i<=r){
			p+=ways;
			p%=MOD;
		}
		q+=ways;
		q%=MOD;
	}
	ll ans = (p* pw(q,MOD-2,MOD))%MOD;
	cout<<ans<<endl;
    return 0;
}
