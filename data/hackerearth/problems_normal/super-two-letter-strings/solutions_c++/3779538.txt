#include <bits/stdc++.h>
using namespace std;
int dp[100007];
const int mod=1000000007;
int main()
{
	int t;
	cin >> t;
	while(t--) {
    int n,k;
	    cin>>n>>k;
	    memset(dp,0,sizeof(dp));
		dp[0]=1;
		for(int i=1;i<=n;i++){
			for(int j=i-1;(j>=0)&&(i-j<=k);j--){
				dp[i]+=dp[j];
				dp[i]=(dp[i])%mod;
			}
		}
		cout << dp[n] << endl;
	}
    return 0;
}
