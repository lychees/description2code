ll poison[10000005],n ;
void sieve () {
	ll i,j ;
	for (i=2;i<=10005;i++)
	{   if (poison[i]==0)
	    {  for (j=2*i;j<10000005;j+=i)
	          poison[j]=1 ;
	    }
	}
}
int main () {
	ll i,x,y ;
	cin(n) ;
	sieve() ;
	for (i=0;i<n;i++)
	{  cin(x) ;
	   y=sqrt(x) ;
	   if (poison[x]==0||(y*y==x&&poison[y]==0))
	     couts("NO\n") ;
	   else
	     couts("YES\n") ;  
	}  
	return 0 ;
}
