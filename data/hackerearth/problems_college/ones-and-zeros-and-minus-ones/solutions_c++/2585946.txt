#include<iostream>
using namespace std;
int arr[100001];
int main()
{
	int n;
	cin >> n;
	for (int l = 0; l<n; l++)
	{
		int count = 0, countleft = 0;
		int x, size;
		cin >> size;
		for (int i = 0; i<size; i++)
		{
			cin >> x;
			if (x == (-1))
			{
				arr[count] += countleft;
				if (count != 0)
					arr[count - 1] += countleft;
				countleft = 0;
				count++;
			}
			else if (x == 1)
			{
				countleft++;
			}
		}
		int max = arr[0];
		for (int i = 0; i<count; i++)
		{
			if (arr[i]>max)
				max = arr[i];
			arr[i] = 0;
		}
		printf("%d/1\n", max);
	}
	return 0;
}
