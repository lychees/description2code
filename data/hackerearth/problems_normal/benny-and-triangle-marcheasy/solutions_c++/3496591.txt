#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;

int main()
{
	double x,y,p;
    cin>>y>>x>>p;
    double a = x*y/2;
    //cout<<a<<endl;
    double al = ((p)/100)*a;
    //cout<<al<<endl;
    double xl = x-sqrt(2*al*(x/y));
    //printf("%.2f\n",xl);
    cout << std::fixed << std::setprecision(2) << xl;
}
