#include <iostream>
#include <vector>


using namespace std;

int main()
{
    int n,m;
    cin>>n>>m;
    vector<int> c;
    
    while(m--){
    	int x;
    
    	cin>>x;
    	
    	c.push_back(x);
    }
    vector<int> pn;
    while(n--){
    	int x;
    	cin>>x;
    	pn.push_back(x);
    }
    
    int last =0;
    int p;
    int x=0;
    while(x<pn.size()){
    
    	int i= pn[x]-1;
    	p=last/c[i] + 1;
    	
    	last= c[i]*p;
    	int j=i+1;
    	while(j<c.size()){
    		if(last%c[j]==0){
    			last+=c[i];
    			j=i+1;
    		}
    		else j++;
    	}
    	x++;
    }
    cout<<last;
    return 0;
}
