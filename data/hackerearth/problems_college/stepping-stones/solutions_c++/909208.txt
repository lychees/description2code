    register int c = gx();
    x = 0;
    for(;(c<48 || c>57);c = gx());
    for(;c>47 && c<58;c = gx())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
}
i64 n,a,b,kk,dp[5005][5005],sum[50005],ans;
int main()
{
	//freopen("aain.txt","r",stdin);
	//freopen("aaout.txt","w",stdout);
	ios_base::sync_with_stdio(false);
	while(cin>>n>>a>>b>>kk){
		memset(dp,0,sizeof dp);
		memset(sum,0,sizeof sum);
		if(a<b){
			dp[a][0]=1,sum[0]=0;
			for(int k=1;k<kk+1;k++){
				for(int i=1;i<b;i++){
					sum[i]=(sum[i-1]+dp[i][k-1])%mod;
					//DBG(i);DBG(sum[i]);
				}
				for(int i=1;i<b;i++){
					//cout<<sum[b-1]<<" "<<sum[i]<<" "<<sum[i-1]<<" "<<sum[2*i-b>=0?2*i-b:0]<<endl;
					dp[i][k]=(sum[(i+b-1)/2]-sum[i]+sum[i-1]+mod)%mod;
					//DBG(i);DBG(dp[i][k]);
				}			
			}
			ans=0;
			for(int i=1;i<b;i++)
			ans=(ans+dp[i][kk])%mod;
			cout<<ans<<endl;
		}
		else if(b<a){
			dp[a][0]=1,sum[n+1]=0;
			for(int k=1;k<kk+1;k++){
				for(int i=n;i>=b;i--){
					sum[i]=(sum[i+1]+dp[i][k-1])%mod;
				}
				for(int i=b+1;i<n+1;i++){
					dp[i][k]=(sum[i+1]+sum[(i+b+2)/2]-sum[i]+mod)%mod;
				}			
			}
			ans=0;
			for(int i=b+1;i<n+1;i++)
			ans=(ans+dp[i][kk])%mod;
			cout<<ans<<endl;
		}
	}			
	return 0;
}
