#include <iostream>
using namespace std;

int main()
{
    long unsigned int t,n,i,c,a[1000000];
    cin>>t;
    while(t)
    {
    	c=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++)
    	{
    		if(a[a[i]-1]!=i+1)
    		{
    			cout<<"not inverse"<<"\n";
    			break;
    		}
    		c++;
    	}
    	if(c==n)
    	cout<<"inverse"<<"\n";
    	t--;
    }
    return 0;
}
