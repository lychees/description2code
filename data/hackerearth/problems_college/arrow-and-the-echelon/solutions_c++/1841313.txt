#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	cin >> n;
    	int cnt1 = n;
    	int flag = 0;
    	while(cnt1 >= 0) {
    		if((cnt1 % 3 == 0) && ((n - cnt1) % 5 == 0)) {
    			for(int i = 0; i < cnt1; i++) {
    				printf("5");
    			}
    			for(int i = 0; i < n - cnt1; i++) {
    				printf("3");
    			}
    			printf("\n");
    			flag = 1;
    			break;
    		}
    		cnt1 -= 5;
    	}
    	if(!flag) {
    		puts("-1");
    	}
    }
    return 0;
}
