#include<stdio.h>
#include<string.h>
#define max(a,b) (a>b?a:b)
bool check(long long int a)
{
    if(a>1000000000000LL)
        return false;
    return true;
}
int main()
{
    int t,i,j,k;
    scanf("%d",&t);
    long long int pow[15]={0};
    pow[0]=1;
    for(i=1;i<=13;i++)
    {
        pow[i]=10*pow[i-1];
    }
    while(t--)
    {
        char s[25];
        scanf("%s",s);
        int n=strlen(s);
        for(i=0;i<n;i++)
            s[i]-='0';
        long long int dp[25][25]={0},ans=0;
        for(i=0;i<n;i++)
        {
            for(j=i;j<n;j++)
            {
            	if(i==j)
            		dp[i][j]=s[i];
            	else
                	dp[i][j]=s[j]+10*dp[i][j-1];
                if(dp[i][j]>1000000000000LL)
                    dp[i][j]=1000000000001LL;
                //printf("i=%d j=%d dp=%lld s=%d\n",i,j,dp[i][j],s[j]);
            }
            if(s[i]==0)
            {
                for(j=i+1;j<n;j++)
                {
                    dp[i][j]=1000000000001LL;
                }
            }
        }
        bool flag=false;
        for(i=0;i+3<n;i++)
        {
            for(j=i+1;j+2<n;j++)
            {
                for(k=j+1;k+1<n;k++)
                {
                    if(check(dp[0][i]) && check(dp[i+1][j]) && check(dp[j+1][k]) && check(dp[k+1][n-1]))
                    {
                        //printf("i=%d j=%d k=%d\n",i,j,k);
                        ans=max(ans,dp[0][i]+dp[i+1][j]+dp[j+1][k]+dp[k+1][n-1]);
                        flag=true;
                    }
                }
            }
        }
        if(flag)
            printf("%lld\n",ans);
        else
            printf("unlucky\n");
    }
    return 0;
}
