#include<iostream>
using namespace std;
 int main()
 {
 	
 	string s;
 	
 	  cin>>s;
	 	int l1=s.length();
	 	int l=0,o=0,v=0,e=0;
	 	
	 	for(int i=0;i<l1;i++)
	 	{
	 
	 		if(s[i]=='l')
			 {
 				l=1;
 			
 			 }
 			 else if(l==1&&s[i]=='o')
 			 {
 			        o=1;
 			        
 			   }
 			  else if(o==1&&s[i]=='v')
 			  {
			        v=1;
			        
			        }
			        
			  else if(v==1&&s[i]=='e')
			  {
			        e=1;	
			        
			   }
 			      
	 		
	 	}
	
	 	if(l==1&&o==1&&v==1&&e==1)
	 	   cout<<"I love you, too!" <<endl;
	 	else
		    cout<<"Let us breakup!" <<endl;   
	 
 	return 0;
 }
