first = 0
second = 0
third = 0
n = input()
list = [int(j) for j in raw_input().split()]
for i in range(0, n):
    curr = list[i]
    if i == 0:
        first = curr
    else:
        if curr >= first:
            third = second
            second = first
            first = curr
        else:
            if curr < first and curr >= second:
                third = second
                second = curr
            else:
                if curr < second and curr >= third:
                    third = curr
    if i < 2:
        print(-1)
    else:
        ans = first*second*third
        print(ans)
        ans = 0
