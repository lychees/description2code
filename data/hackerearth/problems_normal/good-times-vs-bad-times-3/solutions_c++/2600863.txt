#include <bits/stdc++.h>
using namespace std;

int res[24][60][60][2];
int good=1,bad=0;

void solve(){
	int lastgood=0,lastbad=0;
	for(int h=23;h>=0;--h){
		for(int m=59;m>=0;--m){
			for(int s=59;s>=0;--s){
				bool isgood=true;
				for(int i=2;i<=59;++i){
					if(h%i==0 && m%i==0 && s%i==0){
						isgood=false;
						break;
					}
				}
				isgood?lastgood++:lastbad++;
				res[h][m][s][bad]=lastbad;
				res[h][m][s][good]=lastgood;
			}
		}
	}
}

int main()
{
    int t,h,m,s;
    solve();
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d%d",&h,&m,&s);
    	int bad_res=res[h][m][s][bad],good_res=res[h][m][s][good];
    	if(good_res && bad_res){
    		int d=__gcd(good_res,bad_res);
    		good_res/=d;
    		bad_res/=d;
    	}
    	printf("%d:%d\n",bad_res,good_res);
    }
    return 0;
}
