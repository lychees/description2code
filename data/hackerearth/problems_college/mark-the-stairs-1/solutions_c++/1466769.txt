#include <iostream>
using namespace std;

int main()
{
    int t,x,y,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>x>>y;
    	if(x%2==0 && y%2==0 && x>=y && (x-y)<=2)
    	cout<<(x+y)<<"\n";
    	else if(x%2!=0 && y%2!=0 && x>=y && (x-y)<=2)
    	cout<<(x+y-1)<<"\n";
    	else
    	cout<<"No Number\n";
    }
    return 0;
}
