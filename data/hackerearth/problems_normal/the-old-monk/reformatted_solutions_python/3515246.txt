for _ in range(int(raw_input())):
    n = int(raw_input())
    a = map(int, list(raw_input().split()))
    b = map(int, list(raw_input().split()))
    max1 = 0
    for i in range(n):
        low = 0
        high = n-1
        pos = 0
        while low <= high:
            mid = low + (high - low) / 2
            if b[mid] >= a[i]:
                pos = mid
                low = mid + 1
            else:
                high = mid - 1
        max1 = max(max1, pos-i)
    print max1
