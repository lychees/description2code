#include <iostream>
using namespace std;

int gcd_iter(int u, int v) {
  int t;
  while (v) {
    t = u; 
    u = v; 
    v = t % v;
  }
  return u < 0 ? -u : u; /* abs(u) */
}

int main(){
long long int x,y,z;
long long int d,i;
long int t;
cin>>t;
while(t--)
{
	cin>>x>>y>>z;

	d = gcd_iter( x, y);
	d = gcd_iter( d, z);
	if(d>1)
		cout<<"NO"<<endl;
	else
	{
		if(y*y+z*z==x*x||x*x+y*y==z*z||x*x+z*z==y*y)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
}
	return 0;
}
