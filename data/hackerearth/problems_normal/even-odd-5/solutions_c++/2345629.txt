#include <iostream>
using namespace std;

int main()
{
    int t;
	cin>>t;
	while(t--)
	{
	   long long  int r,c;
	    cin>>r>>c;
	    long long int se=0,so=0;
	    if(r<=c)
	    {
	    	if(r%2==0)
	    	{
	    		int nums=r/2;
	    		se+=(nums*(2+r))/2;
	    		so=(nums*r)/2;
	    	}
	    	else
	    	{
	    		int nums=r/2+1;
	    		se+=((nums-1)*(1+r))/2;
	    		so=(nums*(1+r))/2;	
	    	}
	    }
	    else
	    {
	    	if(c%2==0)
	    	{
	    		int nums=c/2;
	    		se+=(nums*(2+c))/2;
	    		so=(nums*c)/2;
	    	}
	    	else
	    	{
	    		int nums=c/2+1;
	    		se+=((nums-1)*(1+c))/2;
	    		so=(nums*(1+c))/2;	
	    	}
	    	if((r-c)%2==0)
	    	{
	    	se+=((r-c)/2+(r-c)%2)*c;
	    	so+=((r-c)/2+(r-c)%2)*c;
	    	}
	    	else if(c%2==0)
	    	{
	    		se+=((r-c)/2)*c;
	    		so+=((r-c)/2+(r-c)%2)*c;
	    	}
	    	else 
	    	{
	    		so+=((r-c)/2)*c;
	    		se+=((r-c)/2+(r-c)%2)*c;
	    	}
	    }
	    cout<<se<<' '<<so<<endl;
	}
    return 0;
}
