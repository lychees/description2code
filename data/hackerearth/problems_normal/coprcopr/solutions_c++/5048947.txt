#include <bits/stdc++.h>

#define pb push_back
#define fr first
#define se second
#define mp make_pair
#define memo(a,b) memset(a,b,sizeof(a))
#define MAX 10000001

using namespace std;
typedef long long ll;
ll n,t,v,s,x,y,ans,a[MAX],ph[MAX];
bool f;

int main()
{
  ios::sync_with_stdio(false);
  cin.tie(0);
  cout.tie(0);
  for(ll i = 2;i*i<MAX;++i){
    if(!a[i]){
      for(ll j = i*i;j<MAX;j+=i)
	a[j] = 1;
    }
  }
  for(ll i = 1;i<MAX;++i)
    ph[i] = i;
  for(ll i = 2;i<MAX;++i){
    if(!a[i]){
      for(ll j = i;j<MAX;j+=i)
	ph[j]-=(ph[j]/i);
    }
  }
  cin>>t;
  while(t--){
    cin>>n;
    ans = 0;
    for(ll i = 1;i<=n;++i)
      ans+=ph[i];
    cout<<ans<<endl;
  }
}
