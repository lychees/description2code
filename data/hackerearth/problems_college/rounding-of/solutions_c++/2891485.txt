#include<bits/stdc++.h>
using namespace std;

long long int mod_pow(long long int a, long long int b,long long int mod)
{
	long long int ans=1;
	while(b)
	{
		int r=b%2;
	
		if(r!=0)
		ans=(ans*a)%mod;
		
		a=(a*a)%mod;
		
		b/=2;
	}	
	return ans;
}

long long gcdf(long long a,long long b)
{
	if(a==0)
	return b;
	return gcdf(b%a,a);	
}

long long min1(long long a,long long b)
{
	return a>b?b:a; 	
}

long long max1(long long a,long long b)
{
	return a<b?b:a; 	
}

struct node{
	string s;
	long long int a1,a2;
};

bool comp(struct node less,struct node more)
{
	return less.a1<more.a1;	
}

long long int fact(long long int n)
{
	long long int ans=n,mod=1000000007;
	for(int i=1;i<=n-1;i++)
	{
		ans=(ans*i)%mod;
	}
	
	return ans;
}

int main() {
int t;
cin>>t;

for(int j=1;j<=t;j++)
{
	int n,b;
	cin>>n>>b;
	
	if(n%b==0)
	cout<<n<<"\n";
	else
	{
		int a1=n/b,a2=n/b+1;
		
		int ans1=n-(a1*b);
		int ans2=(a2*b)-n;
		
		if(ans1<ans2)
		cout<<a1*b<<"\n";
		else
		cout<<a2*b<<"\n";
	}
}
return 0;
}
