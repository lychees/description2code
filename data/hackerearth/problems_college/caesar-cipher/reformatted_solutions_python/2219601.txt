import sys
for t in range(input()):
    n = int(raw_input())
    s = raw_input()
    for i in range(0, len(s), 1):
        ch = s[i]
        num = ord(ch)
        if num >= 65 and num <= 90:
            num = num+n
            while num > 90:
                num = num-26
        if num >= 97 and num <= 122:
            num = num+n
            while num > 122:
                num = num-26
        sys.stdout.write(chr(num))
    print
