t = int(raw_input())

for case in xrange(t):
    s = raw_input()
    nums = ""
    numming = False
    numbers = map(str, range(10))
    for char in s:
        if char == "$":
            numming = True
            continue

        if numming and (char in numbers):
            nums += char

        if numming and (char not in numbers) and (char != " "):
            break

    print("$"+str(int(nums)))
