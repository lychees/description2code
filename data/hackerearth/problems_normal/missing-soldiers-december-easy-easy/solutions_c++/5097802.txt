#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;
const int inf = 1e9;

pair <int, int> pa[N];

int main() {
	int x, y, d, n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d %d %d", &x, &y, &d);
		pa[i] = make_pair(x, x + d);
	}
	sort(pa, pa + n);
	int res = 0, maxv = -inf;
	for (int i = 0; i < n; i++) {
		if (pa[i].first > maxv) {
			res += pa[i].second - pa[i].first + 1;
		} else if (pa[i].second > maxv) {
			res += pa[i].second - maxv;
		}
		maxv = max(maxv, pa[i].second);
	}
	printf("%d\n", res);
	return 0;
}
