#include <iostream>
using namespace std;


int main()
{
    int n,m,k;
    scanf("%d",&m);
    while(m--)
    {
        scanf("%d",&n);
        string s;
        for(int j=n;j>=0;j--)
        {
            if(j%3==0 && (n-j)%5==0)
            {
                s="";
                for(int k=0;k<j;k++)
                    s+='5';
                for(int k=0;k<n-j;k++)
                    s+='3';
                break;
            }
        }
        if(s=="")
            cout<<"-1\n";
        else
            cout<<s<<endl;
    }
    return 0;
}
