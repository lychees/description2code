#include <bits/stdc++.h>
using namespace std;

long long N;

bool F(long long x) {
	if(x >= N) return true;
	if((x * (x + 1)) >> 1 >= N) return true;
	return false;
}

int main()
{
   int T;
   scanf("%d", &T);
   
   while(T--) {
   		scanf("%lld", &N);
   		long long low = 1;
   		long long high = 1e10;
   		while(high - low >= 2) {
   			long long mid = (low + high) >> 1;
   			if(F(mid)) {
   				high = mid;
   			} else {
   				low = mid;
   			}
   		}
   		
   		long long Ans;
   		for(long long i = low; i <= high; ++i) {
   			if(F(i)) {
   				Ans = i;
   				break;
   			}
   		}
   		printf("%lld\n", Ans);
   }
    return 0;
}
