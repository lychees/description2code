    #include <bits/stdc++.h>
    using namespace std;
    #define lim 1000002
    long long A[lim],B[lim];
    int main()
    {
    	long long T,i,j,k,count,ans,high;
    	cin>>T;
    	cin>>high;
    	A[0]=high;
    	for(i=1;i<T;i++){
    		cin>>A[i];
    		high=max(high,A[i]);
    	}
    	B[0]=1;
    	B[1]=2;
    	for(i=2;i<=high;i++) B[i]=B[i-1]+B[i-2];
    	for(i=0;i<T;i++){
    		cout<<B[A[i]]*B[A[i]]<<endl;
    	}
        return 0;
    }
