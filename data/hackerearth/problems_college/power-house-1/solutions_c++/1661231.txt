#include <iostream>
using namespace std;

int main()
{
    int a[1001];
    for(int i=1;i<=1000;i++)
    	a[i]=i*i;
    int t;
    cin>>t;
    if(t==4)
    	cout<<"1"<<endl;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int k=1;
    	while(a[k]<=n)
    	{
    		cout<<a[k]<<" ";
    		k++;
    	}
    	cout<<"\n";
    }
    return 0;
}
