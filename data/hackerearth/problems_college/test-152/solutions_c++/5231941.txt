#include <iostream>
#include<cstring>
#include<map>
using namespace std;

int main()
{
    long n;

    map<char,int> freq;
    cin>>n;
    long len;
    string str;
    for(int p=0;p<n;p++)
    {
        cin>>len;
        cin>>str;
         for(int i=0;i<len;i++)
        {
            freq[str[i]]=0;

        }
        for(int i=0;i<len;i++)
        {
            freq[str[i]]=freq[str[i]]+1;

        }
        map<char,int>::iterator it = freq.begin();
        long large=it->second;
        while(it!=freq.end())
        {
            if(it->second>large)
                large=it->second;
            it++;
        }
        if(large%2==0)
        {
            cout<<"Yes"<<endl;

        }
        else
           {
            cout<<"No"<<endl;
           }
    }
    return 0;
}
