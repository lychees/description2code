#include <iostream>
using namespace std;
#include<bits/stdc++.h>
vector<int> v;
bitset<1000000002> b;
#define LL long long
void pre(){
	LL num=2;
	while(num*num<=1000000000){
		v.push_back(num*num);
		LL temp = num*num;
		int j = temp;
		while(temp<=1000000000){
			b[temp] = 1;
			temp+=j;
		}
		num++;
	}
}
int main()
{
	int t;
	pre();
	cin >> t;
	while(t--){
		int L,R;
		cin >> L >> R;
		int ans = 0;
		for(int i=L;i<=R;i++){
			if(!b[i])
			  ans++;
		}
		cout<<ans<<endl;
	}
}
