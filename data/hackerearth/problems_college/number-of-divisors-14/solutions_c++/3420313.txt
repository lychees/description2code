#include<bits/stdc++.h>
using namespace std;
#define max 10000000
int lowprime[max];
vector<int>v;
void seive(){
    for(int i=2;i<max;i++){
        if(lowprime[i]==0){
        	v.push_back(i);
            for(int j=i;j<max;j+=i){
                if(!lowprime[j])
                	lowprime[j]=1;
            }
        }
    }
}
int main(){
	seive();
	//printf("v=%d\n",v.size());
	int t;
	scanf("%d",&t);
	while(t--){
		unsigned long long int n;
		scanf("%llu",&n);
		unsigned long long int i=0,d=1;
		
		while(n>1 && i<v.size()){
			
			unsigned long long int e=0;
			while(n%v[i]==0){
				n/=v[i];
				e++;
			}
			if(e>0)
				d*=(e+1);
			i++;
		}
		if(n>1)
			d*=2;
		printf("%llu\n",d);
	}
}
