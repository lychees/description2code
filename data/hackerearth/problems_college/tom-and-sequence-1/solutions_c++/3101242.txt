#include <bits/stdc++.h>

using namespace std;

long long int dp[100009];

int main()
{
    dp[0] = 1;
    dp[1] = 2;
    dp[2] = 3;
    long long int count = 3;
    unsigned long long int index;
    
    for (index = 4; index <= 1300000; index++)
    {
         int k = 0;
         for (int j = 2;j <= sqrt(index);j++)
         {
         	if (index % j == 0)
         		{
         			k = 1;
         			break;
         		}
         }
         
         if (k == 0)
         {
         	dp[count++] = index;
         }
    }

    int tc;
    cin >> tc;

    while(tc--)
    {
        long int n;
        cin >> n;
		//cout << count << "   " << index << "   " << dp[100008] << endl << endl;
        unsigned long long ans = 1;
        for (long int i = 1;i <= n;i++)
        {
            ans = (ans * dp[i]) %1000000007 ;
            //cout << dp[i] << "  ";
        }

        cout << ans << endl;
    }

    return 0;
}
