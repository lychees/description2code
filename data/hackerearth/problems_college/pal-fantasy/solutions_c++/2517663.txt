#include <bits/stdc++.h>
using namespace std;
inline bool isPalin(string& s)
{
	return equal(s.begin(),s.begin()+s.size()/2,s.rbegin());
}

int main()
{
    int t;
    cin >> t;
    
    while(t--){
    	string s;
    	cin >> s;
    	int len = s.size();
    	for(int i=0;i<len;i++)
    	{
    		for(int j=i-1;j>=0;j--)
    		{
    			s+= s[j];
    		}
    		if(isPalin(s))
    			break;
    		else
    			for(int j=0;j<i;j++)
    				s.pop_back();
    		
    	}
    	cout << s.size() << "\n";
    }
    return 0;
}
