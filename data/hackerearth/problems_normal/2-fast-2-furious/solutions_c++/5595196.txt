#include<iostream>
using namespace std;
int main()
{
	long n,i;
	long long int deff=0,diff=0,max=-2147483648,mix=-2147483648;
	cin>>n;
	long int a[n],b[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	{
	cin>>b[i];
	}
	for(i=0;i<n-1;i++)
	{
		deff=a[i]-a[i+1];
		if(deff<0)
		deff=-deff;
		if(deff>max)
		max=deff;
    }
    for(i=0;i<n-1;i++)
	{
		diff=b[i]-b[i+1];
		if(diff<0)
		diff=-diff;
		if(diff>mix)
		mix=diff;
    }
    if(mix>max)
    cout<<"Brian"<<endl<<mix;
    else
    cout<<"Dom"<<endl<<max;
return 0;
}
