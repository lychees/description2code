n,k,l,q = map(int,raw_input().split())
di = dict()
for _ in range(n):
	ss = raw_input().split()
	din = ss[0]
	del(ss[0])
	ss=map(int,ss)
	ans = 0 
	for i in ss:
		ans=100*ans+i
	di[ans] = din
for _ in range(q):
	ss = raw_input().split()
	ss=map(int,ss)
	ans = 0 
	for i in ss:
		ans=100*ans+i
	if ans in di: print(di[ans])
	else : print("You cant fool me :P")
