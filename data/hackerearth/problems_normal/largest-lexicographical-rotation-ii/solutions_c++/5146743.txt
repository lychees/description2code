{
    cin>>s;
    int n=s.length();
    temp=s+s;
    int i=0,j=1,l=0;
    while(i+l<n)
    {
        if(temp[i+l]==temp[j+l]) ++l;
        else
        {
            if(temp[i+l]<temp[j+l])
            {
                i=max(i+l,j);
                j=i+1;
                l=0;
            }
            else
            {
                j=j+l+1;
                l=0;
            }
        }
    }
    cout<<temp.substr(i,n)<<endl;
}
