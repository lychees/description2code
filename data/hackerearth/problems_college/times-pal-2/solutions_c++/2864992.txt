#include<bits/stdc++.h>
using namespace std;
bool ispail(string s)
{
	int l=s.size();
	int mid=l/2;
	for(int i=0;i<=mid;i++)
	{
		if(s[i]!=s[l-i-1]) return 0;
	}
return 1;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int n=a.size();
    	int ans=0;
    	for(int i=0;i<=n;i++)
    	{
    		string s=a.substr(0,i)+b+a.substr(i,n);
    		if(ispail(s)) ans++;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
