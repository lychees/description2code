#include <iostream>
#include <cmath>
#include <cstring>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	string s,r;
   	
   	while(t--){
   		int flag = 0;
   		s.clear();
   		r.clear();
   		cin>>s;
   		
   		int l = s.length();
   		for(int i =l-1;i>=0;i--){
   			r = r + s[i];   			
   		}
   		for(int i = 1;i<l;i++){
   			if(abs((s[i]-'0')-(s[i-1]-'0')) == abs((r[i]-'0')-(r[i-1]-'0'))){
   				flag = 1;
   			}else {
   				flag = 0;
   				break;
   			} 
   		}
   		if(flag == 1)
   			cout<<"Funny\n";
   		else cout<<"Not Funny\n";
   	}
    return 0;
}
