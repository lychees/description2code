#include <bits/stdc++.h>
using namespace std;
#define lli long long int
#define rep(i,n) for(int i=0;i<n;i++)
bool isnum(char ch){
	return '0'<=ch && ch<='9';
}
bool ischar(char ch){
	return 'a'<=ch && ch<='z';
}
int find(lli a[],lli k){
	int i;
	for( i=0;i<26;i++){
		k-=a[i];
		if(k<=0)return i;
	}
	return -1;
}
void solve(){
	string ss;
    cin>>ss;
    lli sz=ss.size();
    lli a[26]={0};
   for(lli i=0;i<sz;i++){
    	if(ischar(ss[i])){
    		char ch=ss[i];
    		lli v=0;i++;
    		while(i<sz && isnum(ss[i])){
    			v=v*10+ss[i]-'0';
    			i++;
    		//	cout<<"B";
    		}
    	//	cout<<ch<<" "<<v<<"\n";
    		a[ch-'a']+=v;
    		i--;
    	}
    }
    //rep(i,26)cout<<a[i]<<" ";
    lli q,k;
    cin>>q;
    while(q--){
    	cin>>k;
    	if(find(a,k)==-1)cout<<"-1\n";
    	else cout<<(char)(find(a,k)+'a')<<"\n";
    }
}
int main(){
    solve();
    return 0;
}
