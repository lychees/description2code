int a[25],flag,n,k;
void go(int idx,int val){
	if(flag==1)
	return ;
	if(val==k){
		flag=1;
		return;
	}
	if(idx>=n)
	return ;
	if(val==-1e9)
	go(idx+1,a[idx]);
	else
	go(idx+1,val+a[idx]);
	go(idx+1,val);
}
int main(){
	int t,i;
	cin>>t;
	while(t--){
		flag=0;
		cin>>n;
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		cin>>k;
		if(k==0){
			cout<<"YES"<<endl;
			continue;
		}
		sort(a,a+n);
		go(0,-1e9);
		if(flag==1)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	return 0;
}
