#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
        long long n,x,y,a,r,i,maxx,minn;
        unsigned long long sum=0;
        cin>>x>>y>>n;
        sum=x+y;
        for(i=3;i<=n;i++){
                maxx=max(x,y);
                minn=min(x,y);
                r=maxx-minn;
                sum+=r;
                if(r==0)
                    break;
                x=y;
                y=r;
        }
        if(i<n){
            a=n-i;
            sum+= 2*x*(a/3) + (a%3)*x;
        }
        cout<<sum<<endl;
    }
}
