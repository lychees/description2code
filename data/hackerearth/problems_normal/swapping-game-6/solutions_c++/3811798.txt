#include <iostream>
#include <string>
using namespace std;

bool issame(int *a, int size)
{
	for(int i=0;i<size;i++)
	{
		if(a[i]!=i)
			return false;
	}
	return true;
}

void swap(int *a, int size)
{
	int b[1000];
	b[0]=a[0];
	int start=1, end=size-1;
	for(int i=1;i<size;i++)
	{
		if(i%2==1)
			b[i]=a[end--];
		else
			b[i]=a[start++];
	}
	for(int i=0;i<size;i++)
		a[i]=b[i];
}

void swap2(int *a, int size)
{
	int b[1000];
	int start=0, end=size-1;
	for(int i=0;i<size;i++)
	{
		if(i%2==0)
			b[start++]=a[i];
		else
			b[end--]=a[i];
	}
	for(int i=0;i<size;i++)
		a[i]=b[i];
}

int main()
{
    int n;
    cin>>n;
    string a;
    cin>>a;
    int b[1000];
    for(int i=0;i<(a.length());i++)
    	b[i]=i;
    int count=1;
   	swap(b, a.length());
   	while(!issame(b, a.length()))
   	{
   		count++;
   		swap(b, a.length());
   	}
   	n=n%count;
   	for(int i=0;i<(a.length());i++)
    	b[i]=i;
   	while(n--)
   		swap2(b, a.length());
   	for(int i=0;i<(a.length());i++)
   		cout<<a.at(b[i]);
    return 0;
}
