#print 'Hello World!'
import sys
n=input()
s=[int(i) for i in raw_input().split()]
t=input()
s.sort()
 
for i in range(n):
	x=t-s[i]
	l=0
	r=n-1
	while(l<r):
		if s[l]+s[r]==x:
			print "YES"
			sys.exit(0)
		elif s[l]+s[r]>x:
			r-=1
		else:
			l+=1
else:
	print "NO"
