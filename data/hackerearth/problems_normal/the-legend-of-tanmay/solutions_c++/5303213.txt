#include <bits/stdc++.h>
using namespace std;
void solve()
{
	int n,a[32],neg=0;
	long long ansmx,ansmn;
	bool mlt=false;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]<0)
		{
			neg++;
		}
	}
	if(n==1)
	{
		cout<<a[0]<<" "<<a[0]<<endl;
		return;
	}
	sort(a,a+n);
	if(!neg)
	{
		ansmn=a[0];
		ansmx=1;
		for(int i=0;i<n;i++)
		{
			if(a[i])
			{
				mlt=true;
				ansmx*=a[i];
			}
		}
		if(!mlt)
		{
			ansmx=0;
		}
		cout<<ansmx<<" "<<ansmn<<endl;
	}
	else
	{
		
		ansmx=1;
		bool x=false,y=false;
		for(int i=0;i<neg-neg%2;i++)
		{
			ansmx=ansmx*a[i];
			x=true;
		}
		for(int i=neg;i<n;i++)
		{
			if(a[i]!=0)
			{
				y=true;
				ansmx*=a[i];
			}
		}
		if(!x && !y)
		{
			ansmx=0;
		}
		ansmn=1;
		for(int i=0;i<neg-(neg+1)%2;i++)
		{
			ansmn*=a[i];
		}
		for(int i=neg;i<n;i++)
		{
			if(a[i]!=0)
			{
				ansmn*=a[i];
			}
		}
		
		cout<<ansmx<<" "<<ansmn<<endl;
	}
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	long long t;
	cin>>t;
	while(t--)
	{
		solve();
	}
	return 0;
}
