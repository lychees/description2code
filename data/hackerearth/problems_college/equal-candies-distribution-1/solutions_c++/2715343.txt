#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,k,ans,p,s;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int num[n+9];
		for(i=0;i<n;i++)
		scanf("%d",&num[i]);
		sort(num,num+n);
		i=1;
		k=0;
		int count[5];
		count[0]=0;
		count[1]=1;
		count[2]=1;
		count[3]=2;
		count[4]=2;
		p=0;
		while(i<n)
		{
			if(num[i]!=num[i-1])
			{
				s=num[i]-num[i-1];
				p+=s;
				if(s>=5)
				{
			k+=s/5+count[s%5];	
				}
				else if(s>=2&&s<5)
				{
					k+=s/2+count[s%2];	
				}	
				else
				{
					k+=1;
				}
				num[i-1]+=p;
				num[i+1]+=p;
			}
			i++;
			
		}
		printf("%d\n",k);
		
	}
	return 0;
}
