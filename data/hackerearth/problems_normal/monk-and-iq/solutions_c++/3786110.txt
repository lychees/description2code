#include <iostream>
#include <queue>
#include <vector>
using namespace std;


int main()
{
	int c,p,n,i,x;
	int courses[100001][3];
	pair <long,int> z;
	priority_queue <pair<long,int>,vector<pair<long,int> >,greater<pair<long,int> > >pq;
	cin>>c>>p>>n;
	for(i=1;i<=n;i++){
		cin>>x;
		courses[i][0]=1;
		courses[i][1]=x;
		courses[i][2]=0;
		pq.push(make_pair(x,i));
	}
	for(;i<=c;i++){
		courses[i][0]=0;
		courses[i][1]=0;
		courses[i][2]=0;
		pq.push(make_pair(0,i));
	}
	while(p--){
		if(scanf("%d",&x)==EOF)
			x=0;
		//scanf("%d",&x);
		z=pq.top();
		pq.pop();
		courses[z.second][0]+=1;
		courses[z.second][2]=courses[z.second][1];
		courses[z.second][1]=x;
		pq.push(make_pair(courses[z.second][0]*1L*(courses[z.second][1]+courses[z.second][2]),z.second));
    	cout << z.second <<" ";
	}
    return 0;
}
