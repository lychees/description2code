#include <iostream>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[n];
    	int i ;
    	for(i=0;i<n;i++)
    	{
    		
    		cin>>arr[i];
    		
    		
    	}
    	int count=0;
    	for(i=0;i<n;i++)
    	{
    		for(int j=i;j<n;j++)
    		{
    			if((arr[i]+arr[j])%3==0 && i!=j)
    			{
    				count++;
    				
    			}
    			
    		}
    		
    		
    	}
    	cout<<count<<"\n";
    	
    	
    }
    
    
    
    return 0;

	
	
}
