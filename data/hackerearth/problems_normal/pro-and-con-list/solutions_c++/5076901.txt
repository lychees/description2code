#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		priority_queue <long long> pq;
		int n;
		scanf("%d", &n);
		long long res = 0;
		for (int i = 0; i < n; i++) {
			int x, y;
			scanf("%d %d", &x, &y);
			pq.push((long long) x + y);
			res -= y;
		}
		res += pq.top();
		pq.pop();
		res += pq.top();
		printf("%lld\n", res);
	}
	return 0;
}
