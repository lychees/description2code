#include <iostream>
using namespace std;

int main()
{
 
        long int N, X, Y , a;
        cin >> N;
        
        for (long int  i = 0; i < N; i++) {
         
          cin >> X >> Y;
          long count = 0;
          long  maxAns = 0;;
          	for(long int  j = 0; j< X;j++){
          		cin >> a;
          		if (a <= Y)
          			count += a;
          			else
          			{
          				maxAns = count >= maxAns ? count : maxAns; 
          				count = 0;
          			}
          	}
          maxAns = count >= maxAns ? count : maxAns; 
          cout << maxAns << endl;
          
        }
}
