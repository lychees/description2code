using namespace std;

int n;
int dp[55][10000];
int tans;
int ar[55];

int fun(int i,int dif,int sk)
{
	if(i==n)
		return dif;
	int &res=dp[i][dif];
	if(res!=-1)
		return res;
	if(i==sk)
		res=min( fun(i+1,dif,sk), fun(i+1,dif,sk) );
	else
		res=min( fun(i+1,dif+ar[i],sk), fun(i+1,abs(dif-ar[i]),sk) );
	//cout<<i<<" "<<res<<endl;
	return res;
}

int main()
{
	int t,i,j,k,ans=INF,pos;
	in(t);
	while(t--)
	{
		in(n);
		loop(n) in(ar[i]);
		int sum=0;
		loop(n)
			sum+=ar[i];
		ans=INF;
		for(i=0;i<n;i++)
		{
			for(j=0;j<=n;j++)
				for(k=0;k<=sum;k++)
					dp[j][k]=-1;
			//memset(dp,-1,sizeof(dp));
			tans=fun(0,0,i);
			//cout<<tans<<endl;
			if(tans<ans)
			{
				ans=tans;
				pos=i;
			}
			else if(tans==ans && ar[i]<ar[pos])
			{
				pos=i;
			}
		}
		out2(ans,ar[pos]);
	}
	return 0;
}
