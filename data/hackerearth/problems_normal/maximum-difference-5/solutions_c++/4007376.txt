#include<iostream>
#include<algorithm>

using namespace std;

bool myfunction (int i,int j) { return (i>j); }


int main(){
    int N, *A, *B;
    long long int sum1=0, sum2=0;
    cin>>N;
    A = new int[N];
    B = new int[N];
    for(int i=0; i<N; ++i)
        cin>>A[i];
    sort(A, A+N, myfunction);
    for(int i=0; i<(N+1)/2; ++i)
        B[ 2*i ] = A[i];
    for(int i=(N+1)/2; i<N; ++i)
        B[ (N-1-i)*2 + 1 ] = A[i];
    for(int i=1; i<N; ++i)
        sum1 += abs(B[i]-B[i-1]);

    sort(A, A+N);
    for(int i=0; i<(N+1)/2; ++i)
        B[ 2*i ] = A[i];
    for(int i=(N+1)/2; i<N; ++i)
        B[ (N-1-i)*2 + 1 ] = A[i];
    for(int i=1; i<N; ++i)
        sum2 += abs(B[i]-B[i-1]);
    /*for(int i=0; i<N; ++i)
        cout<<B[i]<<" ";*/
    cout<< ((sum1>sum2)?sum1:sum2);
    return 0;
}
