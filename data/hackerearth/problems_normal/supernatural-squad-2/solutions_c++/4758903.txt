#include <iostream>
#include<string.h>
using namespace std;

long long int dp[200 + 1][200 + 1]; 
int k;
long long int solve(int n, int m)
{
    if(n == 0) return 1;
    if(m == 0) return 0;
    if(n < 0) return 0;
    if(n < k || m < k) return 0;
    if(dp[n][m] == -1)
    {
        dp[n][m] = solve(n, m - 1) + solve(n - m, m);
    }
    return dp[n][m];
}

int main()
{
    
    int t;
    cin >> t;

    while(t--)
    {
        long long int n,i,j;
        cin >> n >> k;
        for(i=0;i<=n;i++){
        	for(j=0;j<=n;j++){
        		dp[i][j]=-1;
        	}
        }
        long long int res = solve(n, n);
        cout << res << endl;
    }
    return 0;
}
