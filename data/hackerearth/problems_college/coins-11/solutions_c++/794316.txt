#include <iostream>
#include<math.h>
using namespace std;

int isPrime(long number) {
        long s = (long)sqrt(number) + 1;
        for (long i = 2; i < s; i++) {
            if (number % i == 0) {
                // number is perfectly divisible - no prime
                return 0;
            }
        }
        return 1;
    }

int main()
{
    long t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n==1||n==0)
    	{
    		cout<<"1\n";
    		continue;
    	}
    	if(n==3)
    	{	cout<<"3\n";
    		continue;
    	}	
    	if(isPrime(n))
    		cout<<"2\n";
    	else
    		cout<<"1\n";
    }
}
