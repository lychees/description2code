#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    long long int a[1000002];
    a[0]=0;
    a[1]=1;
    a[2]=2;
    for(int i=3;i<=1000000;i++)
    {
    	a[i]=a[i-1]+a[i-2];
    }
    while(t--)
    {
    	long long int n;
    	scanf("%lld",&n);
    	long long int sum=0;
    	for(int i=2;a[i]<=n;i=i+3)
    	{
    		sum=sum+a[i];
    	}
    	printf("%lld\n",sum);
    }
}
