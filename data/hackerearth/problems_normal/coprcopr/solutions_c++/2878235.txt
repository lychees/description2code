#include<bits/stdc++.h>
using namespace std;
#define MAX 10000000
#define ll long long

ll lp[MAX+9];
ll phi[MAX+9];
ll sum[MAX+9];

void solve(){
    lp[0]=lp[1]=1;
    for(int i=2;i*i<MAX;i+=1){
        if(lp[i]==0){
            lp[i]=i;
            for(int j=i+i;j<MAX;j+=i){
                if(lp[j]==0) lp[j]=i;
            }
        }
    }

    phi[1]=1;
    sum[1]=1;
    for(int i=2;i<MAX+1;i+=1){
    	if(lp[i]==0)phi[i]=i-1;
    	else{
    		if(lp[i]==lp[i/lp[i]])
    			phi[i] = phi[i/lp[i]]*lp[i];
    		else
    			phi[i] = phi[i/lp[i]]*(lp[i]-1);
    	}
    	sum[i]=phi[i]+sum[i-1];
    }
}

int main(){
    int t;
    cin>>t;
    solve();
    while(t--){
        ll n;
        cin>>n;
        cout<<sum[n]<<endl;
    }
}
