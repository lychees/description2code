#include <iostream>
#include <map>
using namespace std;

long long N;
map<long long, long long> dp;

long long solve(long long n)
{
    if(n==0 || n==1) return n;
    if(dp[n]) return dp[n];

    long long ans=n;
    ans=max(ans,solve(n/2)+solve(n/3)+solve(n/4));
    return dp[n]=ans;

}
int main()
{
    while(cin.good())
    {
    	cin>>N;
    	cout<<solve(N)<<endl;
    }
    return 0;
}
