import math
test = int(input())
for t in range(test):
    n = int(input())
    x = {}
    while n % 2 == 0:
        x[2] = x.get(2, 0)+1
        n = n//2
    i = 3
    while i <= math.sqrt(n):
        while n % i == 0:
            x[i] = x.get(i, 0)+1
            n = n//i
        i += 2
    if n > 2:
        x[n] = x.get(n, 0)+1
    p = 1
    for i in x.keys():
        p = p*(x[i]+1)
    p = p-1
    print(p)
