#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	long int a,b;
   	cin>>a>>b;
   	
   	long int res=0;
   	while(a>0 && b>0)
   	{
   		if(a>b)
   		{
   			res=res+a/b;
   			a=a%b;
   			
   		}
   		else
   		{
   			res=res+b/a;
   			b=b%a;
   		}
   		
   	}
   	if(res>0) res--;
   	cout<<res<<endl;
   }
   return 0;
}
