#include<iostream>
#include<math.h>
using namespace std;

int prime(int x)
{
    if (x == 0 || x == 1)
        return 0;
    for (int i = 2; i <= sqrt(x); i++)
        if (x%i == 0)
            return 0;
    return 1;

}

int main()
{
    int a, b, n, t, c = 0, m;
    int ret[100];
    cin >> n ;
    cin>>t;
    for (a = 0; a < t; a++)
    {
        c = 0;
        cin >> m;
        for (b = 1; b <= m; b++)
            if (prime(b))
                c++;
        ret[a] = c;
    }
    for (a = 0; a < t; a++)
    {
        cout << ret[a] << endl;
    }
    return 0;
}
