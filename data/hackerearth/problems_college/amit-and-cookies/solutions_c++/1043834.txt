#include<iostream>
#include<vector>
#include<algorithm>
#define c cin
#define o cout
using namespace std;
int main(){
	int t,k;
	c>>t>>k;
	int i;
	vector<int>v(t);
	for(i=0; i<t; i++)
		c>>v[i];
	sort(v.begin(), v.end());
	int ct=0, acc=0;
	bool chk=true;
	for(i=0; i<v.size()&&chk; i++){
		if(acc+v[i]<=k){
			acc+=v[i];
			ct++;
		}
		else
			chk=false;
	}
	o<<ct<<"\n";
	return 0;
}
