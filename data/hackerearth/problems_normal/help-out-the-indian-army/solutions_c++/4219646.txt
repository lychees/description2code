#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <list>
#include <iterator>
#include <stdio.h>
#define fr(i,s,n) for(int i=s;i<n;i++)
#define rfr(i,n,s) for(int i=n;i>=s;i--)
#define lld long long int
#define pb(x) push_back(x)
#define mp(x,y) make_pair(x,y)
using namespace std;

int main()
{
	lld n,s,e,x,p,ans=0,curr_end;
	cin>>n>>s>>e;
	vector<pair<lld,lld> >v;
	fr(i,0,n)
	{
		cin>>x>>p;
		v.pb(mp(x-p,x+p));
	}
	sort(v.begin(),v.end());
//	fr(i,0,n) cout<<v[i].first<<" "<<v[i].second<<endl;
	curr_end=s;
	if(v[0].first>s&&v[0].first<e)
	ans=(lld)(ans+v[0].first-s);
	curr_end=min(max(curr_end,v[0].second),e);
	//cout<<ans<<endl;
	fr(i,1,n)
	{
		if(v[i].first>=e) break;
		if(v[i].first>curr_end)
		{
			//cout<<v[i].first<<" "<<curr_end<<endl;
			ans=(lld)(ans+v[i].first-curr_end);
		}
		//	cout<<ans<<endl;
			curr_end=min(max(curr_end,v[i].second),e);
			
	}
	
    cout<<ans+e-curr_end;
    return 0;
}
