#include<bits/stdc++.h>
using namespace std;

#define max_n 1000004
bool isprime[max_n + 1];

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	vector<long long> v;
	for(int i=2;i<=sqrt(max_n);i++)
		if(!isprime[i])
			for(int j=i*i;j<=max_n;j+=i)
				isprime[j] = true;
	
	v.push_back(1);
	for(int i=2;i<=max_n;i++)
		if(!isprime[i])
			v.push_back(i);
	long long t,n,num,ans;
	cin>>t;
	while(t--)
	{
		ans = 0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>num;
			if(num == 1)
				ans--;
			ans+=*upper_bound(v.begin(),v.end(),num);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
