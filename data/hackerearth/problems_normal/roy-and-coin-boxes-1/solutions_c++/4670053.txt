#include <iostream>

using namespace std;
int segTreeSize = 0;

//int* createSegTree(int size)
int main(){
    struct _ { ios_base::Init i; _() { cin.sync_with_stdio(0); cin.tie(0);cout.tie(0); } } _;

    int N=0; //Number of coin boxes
    int M=0; //Number of days
    cin>>N>>M;
    int *coinBox = new int[N+1];
    for(int i=0; i<=N ;i++)
        coinBox[i] = 0;
    long long int currentSum = 0;
    while(M--){
        int lindex, rindex;
        cin>>lindex>>rindex;
        coinBox[lindex] += 1;
        coinBox[rindex+1] -= 1;
    }
    int *countArr = new int[N+1];
    for(int i=0; i<N; i++){
        countArr[i] = 0;
    }
    for(int i=1; i<=N; i++){
        currentSum += coinBox[i];
        countArr[currentSum] += 1;
    }
    for(int i=1; i<=N; i++){
        countArr[i] += countArr[i-1];
    }
    int Q=0; //Number of queries
    cin>>Q;
    while(Q--){
        int minCoins=0;
        cin>>minCoins;
        cout<<countArr[N] - countArr[minCoins-1]<<endl;
    }
    return 0;
}
