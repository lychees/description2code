#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int T,N;
    cin>>T;
    while(T--) {
    	cin>>N;
    	vector<int> A(N);
    	for(int i=0;i<N;i++) {
    		cin>>A[i];
    	}
    	sort(A.begin(),A.end());
    	vector<int> B(N);
    	for(int i=0;i<N;i++) {
    		cin>>B[i];
    	}
    	sort(B.begin(),B.end());
    	long long ans=0;
    	for(int i=0;i<N;i++) {
    		ans+=A[i]*B[i];
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
