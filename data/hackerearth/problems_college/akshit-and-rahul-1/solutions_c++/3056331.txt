#include <iostream>
#include<math.h>
#include<stdio.h>
using namespace std;

 typedef long long ll;
 double golden = (sqrt(5.0) + 1.0) / 2.0;
 ll main2(int A1, int A2, int B1, int B2)
 { 
    	ll ans = 0;
    	for(int A=A1;A<=A2;A++){
    		int x = (int)(A * golden) + 1;
    		x = max(x,B1);
    		if(x <= B2) ans += B2 - x + 1;
    	}
    	return ans;
    }

int main()
{
    int t;
    cin>>t;
    long long a,b;
    for(int i=1;i<=t;i++)
    {
    	    int A1,A2,B1,B2;
    		scanf("%d%d%d%d",&A1,&A2,&B1,&B2);
    		ll ans = main2(A1,A2,B1,B2) + main2(B1,B2,A1,A2);
    		cout<<"Case #"<<i<<": "<<ans<<endl;
    	
    }
    return 0;
}
