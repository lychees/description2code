#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	float p1,p2,p3,q1,q2,q3;
	pair<float,float> point[3];
	int i;
	cin>>p1>>q1>>p2>>q2>>p3>>q3;
	point[0].first = p1+p2-p3;
	point[1].first = p1-p2+p3;
	point[2].first = p2-p1+p3;
	point[0].second = q1+q2-q3;
	point[1].second = q1-q2+q3;
	point[2].second = q2-q1+q3;
	sort(point,point+3);
	
	for(i=0;i<3;++i)
		printf("%.4f %.4f\n",point[i].first,point[i].second);
	
    return 0;
}
