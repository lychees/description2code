#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    long int A[100],B[100],i,j,k,M;
    A[1]=0;B[1]=1;
    A[2]=1;B[2]=1;
    for(i=3;i<=90;i++)
    {
        A[i]=A[i-1]+A[i-2];
        B[i]=B[i-1]+B[i-2];
    }
    cin>>T;
    while(T--)
    {
        cin>>M;
        cout<<A[M]<<" "<<B[M]<<"\n";
    }
}
