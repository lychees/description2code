#include<iostream>
using namespace std;
typedef unsigned long long LL;
int main(){
LL T,num,wait,cost;
cin>>T;
while(T--){
cin>>num>>wait;
cost=0;
if(num<3){
	cost+=50;
	cost+=wait*2;
	cout<<cost<<endl;	
}
else if(num>60)
	cout<<"NA"<<endl;
else{
	num-=2;
	cost=12*num+50;
	cost+=wait*2;
	cout<<cost<<endl;
}
}
return 0;
}
