using namespace std;
#define MAX 1002*1002
map<int, int> mp;
#define MOD 1000000007
long long int k[MAX];
int main(){
	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++){
		for (int j = 1; j <= m; j++){
			mp[i^j]++;
		}
	}
	k[0] = 1;
	for (int i = 1; i < MAX; i++){
		k[i] = k[i - 1];
		k[i] *= (long long int)(i);
		k[i] %= MOD;
	}
	long long int ans = 1;
	map<int, int>::iterator ite;
	for (ite = mp.begin(); ite != mp.end(); ite++){
		ans *= k[(*ite).second];
		ans %= MOD;
	}
	printf("%lld\n", ans);
	return 0;
}
