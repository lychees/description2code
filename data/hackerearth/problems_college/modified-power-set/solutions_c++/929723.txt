#include<bits/stdc++.h>
using namespace std;
#define temp(s) (s).begin(),(s).end()
#define func(s) sort(temp(s)),(s).resize(unique(temp(s))-(s).begin())
int main(){
	long long int t,k;
	scanf("%lld",&t);
	while(t--){
		string s;
		cin>>s;
		func(s);
		k=s.length();
		printf("%lld\n",k+(k*(k-1))/2LL);
	}
	return 0;
}
