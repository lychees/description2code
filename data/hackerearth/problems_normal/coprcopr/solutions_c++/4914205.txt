const int N = 10000000;
vector<ll> lp(N + 1);
vector<ll> phi(N + 1);
vector<ll> sum(N + 1);
vector<ll> pr;

  int main()
  {
  	phi[1] = 1;
    for (int i = 2; i <= N; ++i)
    {
        if (lp[i] == 0)
        {
            lp[i] = i;
            phi[i] = i - 1;
            pr.push_back(i);
        }
        else
        {
            //Calculating phi
            if (lp[i] == lp[i / lp[i]])
                phi[i] = phi[i / lp[i]] * lp[i];
            else
                phi[i] = phi[i / lp[i]] * (lp[i] - 1);
        }
        for (int j = 0; j < (int)pr.size() && pr[j] <= lp[i] && i * pr[j] <= N; ++j)
            lp[i * pr[j]] = pr[j];
    }
  	
  	/*f1(i,1,60)
  		cout<<"phi["<<i<<"] = "<<phi[i]<<endl;*/
    f0(i,1,N){
    	sum[i] = sum[i-1]+phi[i];
	}
	T(){
		int n;
		cin>>n;
		cout<<sum[n]<<endl;
	}
  	return 0;
  }
