#include <cstdio>
#include <cstring>

using namespace std;

bool isPalin(char * p, int length)
{
	char * q;
	
	q = p + length - 1;
	
	while (q > p) {
		if (*q != *p) {
			return false;
		}
		--q;
		++p;
	}
	
	return true;
}

int main()
{	
	int t;
	
	scanf("%d", &t);
	
	long long int length, x, i, j, k, l;
	char a[111];
	
	while (t--) {
		scanf("%s", a);
		length = strlen(a);
		
		if (isPalin(a, length)) {
			printf("%lld\n", length);
			continue;
		}
		
		i = 0;
		
		
		while (a[i] != '\0') {
			
			j = i;
			k = i + 1;
			
			if (a[k] == '\0') {
				x = j;
				break;
			}
			
			while (a[k] != '\0') {
				if (a[k] != a[j]) {
					break;
				}
				
				++k;
				--j;
			}
			
			if (a[k] == '\0') {
				x = j + 1;
				break;
			}
			
			j = i;
			k = i;
			while (a[k] != '\0') {
				if (a[k] != a[j]) {
					break;
				}
				
				++k;
				--j;
			}
			
			if (a[k] == '\0') {
				x = j + 1;
				break;
			}
			
			++i;
		}
		
		if (a[i] == '\0') {
			x = length;
		}
		//printf("i = %lld, j = %lld, k = %lld, x = %lld\n", i, j, k, x);
		printf("%lld\n", length + x);
	}
	
	return 0;
}
