#include <iostream>
using namespace std;

int main()
{
	int testCases;
    cin >> testCases;
    int i=0;
    for(;i<testCases;i++){
    	int items;int shops;
    	cin >>items;
    	cin >> shops;
    	int j=0;
    	double min=100;int minIn=0;
    	for(;j<items;j++){
    		int k=0;
    		
    		for(;k<shops;k++){
    			int l=0;double res=1.0;	
    			for(;l<3;l++){
    				double dis;
    				cin >>dis;
    				dis=(dis/100);
    				res=res*(1-dis);
    			}
    		//	cout<<"hhh------"<<res<<endl;
    			if(res<min){
    				min=res;
    				minIn=k;
    			}
    		}
    		cout << (minIn+1)<<" ";
    		minIn=-1;min=100;
    	}
    	cout <<endl;
    }
}
