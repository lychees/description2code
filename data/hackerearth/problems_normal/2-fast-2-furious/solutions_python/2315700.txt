check_points = raw_input()
check_points = int(check_points)
dom_speeds = [int(elem) for elem in raw_input().split(' ')]
brian_speeds = [int(elem) for elem in raw_input().split(' ')]

dom_max = 0
brian_max = 0
for i in xrange(check_points-1):
    diff_1 = abs(dom_speeds[i] - dom_speeds[i+1])
    diff_2 = abs(brian_speeds[i] - brian_speeds[i+1])
    if diff_1 > dom_max:
        dom_max = diff_1
    if diff_2 > brian_max:
        brian_max = diff_2

if dom_max > brian_max:
    print "Dom"
    print dom_max
elif brian_max > dom_max:
    print "Brian"
    print brian_max
else:
    print "Tie"
    print brian_max
