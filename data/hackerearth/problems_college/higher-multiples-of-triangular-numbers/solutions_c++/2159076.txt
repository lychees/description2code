#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    vector<long long int> v;
    long long int i,j,k,n,u,sqt,cnt;
    for(i=1;i<=10000000;i++)
    {
        k=(i*(i+1))/2;
        v.push_back(k);
    }
    cin>>T;
    while(T--)
    {
        cin>>n;
        for(i=0;i<v.size();i++)
        {
            u=v[i];
            cnt=0;
            sqt=sqrt(u);
            for(j=1;j<=sqt;j++)
            {
                if(u%j==0)
                {
                    if(j*j!=u)
                        cnt+=2;
                    else
                        cnt+=1;
                }
            }
            if(cnt>n)
            {
                cout<<u<<"\n";
                break;
            }
        }
    }
}
