#include<iostream>
using namespace std;
int main()
{
    string s;
    int m;
    cin>>m;
    int count=0;
    for(int i=0;i<m;i++)
    {
        cin>>s;
        int length=s.length();
        char stack[length];
        stack[0]=s[0];
        int top=0;
        for(int i=1;i<length;i++)
        {
            if(stack[top]==s[i])
            {
                top--;
            }
            else
            {
                stack[++top]=s[i];
            }
        }
         if(top<0)
            count++;
    }
    cout<<count<<"\n";
}
