void Adjust(int a[], int i, int n)
{
    int j = 2 * i, item = a[i];
    while(j <= n)
    {
        if((j < n) && (a[j] > a[j+1]))
            j++;
        if(item < a[j])
            break;
        a[j/2] = a[j];
        j = 2 * j;
    }
    a[j/2] = item;
}

void Heapify(int a[], int n)
{
    for(int i = n/2; i >= 1; i--)
        Adjust(a, i, n);
}
