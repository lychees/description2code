//constants
const int mxn=1e5+5;
const int MOD=1e9+7;
using namespace std;
ll A[mxn][3];
ll num[mxn];
int main()
{
	int t=1;
	sc1(t);
	while(t--)
	{
		memset(num,0,sizeof(num));
		int n,m;
		cin>>n>>m;
		for(int i=1;i<=m;++i)
		{
			cin>>A[i][0]>>A[i][1]>>A[i][2];
		}
		int q;
		sc1(q);
		ll x[100005];
		memset(x,0,sizeof(x));
		while(q--)
		{
			int l,r;
			cin>>l>>r;
			x[l]++;
			x[r+1]--;
		}
		for(int i=1;i<=m;++i)
		{
			x[i]+=x[i-1];
		//	cout<<x[i]<<" ";
		}
		
		for(int i=1;i<=m;++i)
		{
			num[A[i][0]]+=x[i]*A[i][2];
			num[A[i][1]+1]-=x[i]*A[i][2];
		}
		for(int i=1;i<=n;++i)
		{
			num[i]+=num[i-1];
			cout<<num[i]<<" ";	
		}	
		pn();
	}
	return 0;
}
