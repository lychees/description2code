def bell_seq_generator():
    bell = [1]
    N = 0
    while N<=1000:
        yield bell[N]%MOD
        x = bell[-1]
        bell.append(0)
        N += 1
        for j in xrange(0, N):
            x, bell[j] = (bell[j]+x)%MOD, x%MOD
        bell[N] = x

MOD=1000000007
if __name__ == '__main__':
	x=list(bell_seq_generator())
	for i in range(int(raw_input())):
		print x[int(raw_input())-1]
