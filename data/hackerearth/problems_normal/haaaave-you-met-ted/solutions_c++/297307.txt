#include<iostream>
#include<climits>
#include<cstdio>
using namespace std;
typedef long long ll;
ll fun(ll x)
{
    ll c = 0;
    while(x){
        int p = x%2;
        if(p)c++;
        x = x/2;
    }
    return c;
}
int main()
{
    int t;ll i,maxele,a[100002],n;
    cin >> t;
    while(t--){
        scanf("%lld",&n);
        for(i=0;i<n;i++)scanf("%lld",&a[i]);

        maxele = INT_MAX;
        for(i=0;i<n;i++){
            maxele = min(maxele,fun(a[i]));
            if(maxele == 1){
                break;
            }
        }
        printf("%lld\n",maxele);
    }
    return 0;
}
