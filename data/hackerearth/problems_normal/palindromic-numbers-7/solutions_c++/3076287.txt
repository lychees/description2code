#include<bits/stdc++.h>
using namespace std;
main()
{
	int a,b,t,i,x;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int r,s=0,j,n=0,y;
		cin>>a;
		cin>>b;
		for(j=a;j<=b;++j)
		{
			s=0;
			x=j;y=j;
			while(x>0)
			{
			r=x%10;
			s=s*10+r;
			x=x/10;
			}
			if(s==y)
			n++;
		
		
		}	
		
		cout<<n<<endl;
	}return 0;
}
