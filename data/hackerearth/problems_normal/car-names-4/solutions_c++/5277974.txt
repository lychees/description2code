#include<bits/stdc++.h>
using namespace std;
int main ()
{
	int N,count,f,i;
	cin>>N;
	string str;
	while(N--)
	{count=0;f=0;
	cin>>str;
	for(i=0;i<str.length()-1;++i)
	{
	if(str[i]!=str[i+1])
		count++;	
	}
	if((count!=2)||(str.compare("ccaacc")==0))
		{
		cout<<"Not OK"<<endl;
		f=1;
		}
	if(((str.length()%3)!=0)&&(f==0))
		{
		cout<<"Not OK"<<endl;
		f=1;
		}
	if(f==0)
	cout<<"OK"<<endl;
	}
return 0;
}
