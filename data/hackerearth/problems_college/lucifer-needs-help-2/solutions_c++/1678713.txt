using namespace std;
long long int maxi=1000007;
long long int tree[1000007]={0};
long long int a[1000007]={};
void update(long long int i, long long int x)
{
	while (i <= maxi) {
		tree[i] = tree[i] + x;
		i = i + (i & -i);
	}
}
long long int read(long long int i)
{
	long long int s = 0;
	while (i > 0) {
		s = s + tree[i];
		i = i - (i & -i); 
	}
	return s;
}
int main()
{

	long long int n;
	scanf("%lld",&n);
	
	for(long long int i=1;i<=n;i++){
		scanf("%lld",&a[i]);
		
	}
	
	long long int m;
	cin>>m;
	for(long long int i=0;i<m;i++){
		long long int x,y,z;
		scanf("%lld%lld%lld",&x,&y,&z);
		//x--;
		
		update(x,z);
//		cout<<read(1)<<endl;
		update(y+1,-z);
	}
	for(long long int i=1;i<=n;i++){
		long long int aa=read(i);
		printf("%lld\n",aa+a[i]);
	}
	
	return 0;
}
