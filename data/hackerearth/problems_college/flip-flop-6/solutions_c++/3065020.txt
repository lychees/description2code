#include<iostream>
#include<string.h>
using namespace std;
 char s[1000000];
int main(){
 int n,i,j,sum;
 
 cin>>n;
 while(n--){
       cin>>s;
       j=1;
       int l=strlen(s);
       for(i=1;i<l;++i){
          if(s[i-1]!=s[i])
           j++;
       }
       cout<<l-j<<endl;
 }
 return 0;
}
