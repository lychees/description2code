#include <iostream>
#include<vector>
#include<set>
#include<cstdio>
using namespace std;

int main()
{
    long long int n,count=0;
    int a;
    
    scanf("%lld",&n);

    vector<bool> v(1000001,0);
    //set<int> s;
    for(int i=0;i<n;i++)
      {
      
      	scanf("%d",&a);
      	if(v[a])
      	count++;
      	else
      	v[a]=1;
      	//s.insert(a);
      	
      }
      n-=count;
    long long int ans= (n*(n-1))/2;
    
    cout<<ans<<"\n";
    
    return 0;
}
