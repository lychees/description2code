#include<bits/stdc++.h>
int a[1000000],b[100000];
int main() {
	int t,i,c,n;
	scanf("%d",&t);
	while(t--) {
		c=0;
		for(i=0;i<10005;i++)b[i]=0;
		scanf("%d",&n);
		for(i=0;i<n;i++) {
			scanf("%d",&a[i]);
			b[a[i]]=1;
		}
		for(i=1;i<10001;i++) {
			if(b[i])c++;
		}
		printf("%d\n",c);
	}
	return 0;
}
