#include<bits/stdc++.h>
using namespace std;
int main()
{
    string tmp,str,ptr;
    int i,j,ans=0;
    map <string,int> mp;
    cin>>str>>ptr;
    for(i=0;i<str.size();i++)
    {
        for(j=1;j<=str.size()-i;j++)
        {
            tmp=str.substr(i,j);
            if(mp[tmp]==0)
            {
                mp[tmp]++;
                if(tmp.size()>=ptr.size())
                for(int k=0;k<=tmp.size()-ptr.size();k++)
                {
                    if(ptr==tmp.substr(k,ptr.size()))
                    {
                        ans++;
                        break;
                    }
                }
            }
        }
    }
    cout<<ans;
}
