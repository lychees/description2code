using namespace std;
vector< int >v[maxn];
int subsz[maxn];
int vis[maxn];
void dfs(int s)
{
	subsz[s]=1;
	vis[s]=1;
	for(int i=0;i<sz(v[s]);i++)
	{
		if(!vis[v[s][i]])
		{
			dfs(v[s][i]);
			subsz[s]+=subsz[v[s][i]];
		}
	}
}

	{
		cin>>a>>b;
		v[a].pb(b);
		v[b].pb(a);
	}
	dfs(1);

		else
			cout<<subsz[i]<<" ";
	}
	return 0;
}
