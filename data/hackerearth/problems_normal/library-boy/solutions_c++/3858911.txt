#include<bits/stdc++.h>
#include<string.h>
using namespace std;
int main()
{
	int n,a[255],w,c=0;
	for(int i=0;i<255;i++)
	{
		a[i]=0;
	}
	cin>>n;
	string s;
	while(n--)
	{
		cin>>s;
		w=s[0];
		a[w]++;
		
		
	}

for(int i=0;i<255;i++)
	{
	
		if((a[i]>0)&&(a[i]<=10))
		c++;
		else if(a[i]>10)
		{
		if((a[i]%10)!=0)
		c=c+(a[i]/10)+1;
		else
		c=c+(a[i]/10);
	}
}
cout<<c;
	return 0;
}
