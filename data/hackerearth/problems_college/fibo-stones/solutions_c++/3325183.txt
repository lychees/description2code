#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,m,i ;
    cin>>n>>m ;
   long  int arr[m],sum=0,sumf=2,f=1,s=1,t;
    if(n<3&&n<m)
    return 0;
    for(i=2;i<n;i++)
    { 
    	t=f+s ;
    	sumf+=t ;
    	f=s ;
    	s=t ;
    	
    }
   
    
    for(i=0;i<m;i++)
    {cin>>arr[i] ;
    sum+=arr[i] ;
    }
    
    cout<<sumf-sum ;
    return 0;
}
