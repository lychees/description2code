#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int binarysearch(int a[],int start,int end,int key){
	if(key<0){
		return 0;
	}
	else{
		int mid=(start+end)/2;
		int diff = end-start;
		if(start<=end){
			//int diff = end-start;
			if(diff==1){
				//cout<<key<<endl;
				if(a[start]==key||a[end]==key)
					return 1;
				else
					return 0;
			}
			else
			{
				if(a[mid]==key){
					return 1;
				}
				else if(a[mid]<key){
					return binarysearch(a,mid+1,end,key);
				}
				else
					return binarysearch(a,start+1,mid-1,key);
			}
		}
		else
			return 0;
	}
}
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    int sum=0;
    cin>>sum;
    sort(a,a+n);
    int sumofmax3=0;
    for(int i=n-1;i>n-1-3;--i){
    	sumofmax3+=a[i];
    }
    //cout<<sumofmax3<<endl;
    if(sum>sumofmax3){
    	cout<<"False\n";
    	return 0;
    }
    if(sum==sumofmax3){ 
    	cout<<"True\n";
    	return 0;
    }
    else{
    	int i=0,flag=0,j=n-1;
    	for(;i<(n-2)/2;){
    		//cout<<(sum-(a[i]+a[n-i-1]))<<endl;
    		if(binarysearch(a,i+1,j-1,(sum-(a[i]+a[j])))){
    			cout<<"True\n";
    			return 0;
    		}	
    		flag^=1;
    		if(flag==1){
    			i++;
    		}
    		else{
    			j--;
    		}
    		
    	}
    	i=0;
    	for(;j>=i+2;--j){
    		//cout<<(sum-(a[i]+a[n-i-1]))<<endl;
    		if(binarysearch(a,i+1,j-1,(sum-(a[i]+a[j])))){
    			cout<<"True\n";
    			return 0;
    		}	
    		
    	}
    	j=n-1;
    	for(;i<=j-2;++i){
    		//cout<<(sum-(a[i]+a[n-i-1]))<<endl;
    		if(binarysearch(a,i+1,j-1,(sum-(a[i]+a[j])))){
    			cout<<"True\n";
    			return 0;
    		}	
    		
    	}
    	cout<<"False\n";
    }
    return 0; 
}
