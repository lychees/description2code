#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

#define in(n) scanf("%d",&n)
#define MOD 1000000007
#define MAXN 100005


int main()
{
//	freopen("input.txt","r",stdin);
	int n,i,l,r,q,a;
	int sum[100004];
	in(n);
	sum[0]=0;
	for(i=1;i<=n;i++)
	{
		in(a);
		sum[i]=sum[i-1]+a;
	}
	in(q);
	for(i=0;i<q;i++)
	{
		in(l);
		in(r);
		cout<<sum[r]-sum[l-1]<<endl;
	}
	return 0;
}
