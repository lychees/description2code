#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;
void bubble(int *a,int n)
{
    int i,k,temp;
    for(i=0;i<n-1;i++)
        for(k=0;k<n-i-1;k++)
        if(a[k]>a[k+1])
    {
        temp=a[k];
        a[k]=a[k+1];
        a[k+1]=temp;
    }
}
int main()
{
    int n,*a,i,diff=0;
    cin>>n;
    a=(int*)malloc(n*sizeof(int));
    for(i=0;i<n;i++)
        cin>>a[i];
    bubble(a,n);
    if(abs(a[0]-a[1])>abs(a[0]-a[2]))
        diff=a[0]-a[1];
    else
        diff=abs(a[0]-a[2]);
    for(i=0;i<n-2;i++)
    {
        if(diff<abs(a[i]-a[i+2]))
            diff=abs(a[i]-a[i+2]);
    }
    if(diff<abs(a[n-1]-a[n-2]))
        diff=abs(a[n-1]-a[n-2]);
    if(diff<abs(a[n-1]-a[n-3]))
        diff=abs(a[n-1]-a[n-3]);
        cout<<diff;
    return 0;
}
