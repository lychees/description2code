#include <iostream>
using namespace std;
long long int gcd(long long int a,long long int b)
{
  	while(b!=0)
  	{
  		long long int t=a%b;
  		a=b;
  		b=t;
  	}
  	return a;
}
int main()
{
    ////cout << "Hello World!" << endl;
    long int t;
    long long int a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	if(a==1 || b==1)
    	cout<<"Is a Co-Prime"<<endl;
    		else
    		{
    	long long int k=gcd(a,b);
    	if(k==1)
    	cout<<"Is a Co-Prime"<<endl;
    	else
    	cout<<"Not a Co-Prime"<<endl;
    		}
    }
    return 0;
}
