#include<iostream>

using namespace std;

int main(){

    int t,start,end,aCost,bCost,ans;
    string input;
    cin>>t;
    //cin.ignore(256,' ');
    while(t--){
        int flag=0;
        cin>>input;
        cin>>aCost>>bCost;
        //cin.ignore(256,' ');
        ans=0;
        start=0 , end=input.length()-1;
        while(start<end){
            if(input[start] != input[end]){
                if(input[start]=='/'||input[end]=='/'){
                    if(input[start]=='a' || input[end]=='a')
                        ans+=aCost;
                    else
                        ans+=bCost;
                }
                else
                    flag=-1;

            }
            else if(input[start]=='/')
                ans+=2*min(aCost,bCost);

            start++ , end--;
        }

        if(flag==-1)
            cout<<-1<<endl;
        else cout<<ans<<endl;
    }
    return 0;
}
