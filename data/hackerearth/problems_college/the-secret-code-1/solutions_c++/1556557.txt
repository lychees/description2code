    #include<bits/stdc++.h>
    #define ll long long
    using namespace std;
     
    vector<ll> a;
    void sieve()
    {
    ll x=1;
    int i=0;
    a.push_back(3);
    a.push_back(5);
     
     
    while(x<=1000000009)
    {
    x=a[i]*10+3;
    a.push_back(x);
    x=a[i]*10+5;
    a.push_back(x);
    i++;
    }
    }
    int main()
    {
    int t,i,j;
    ll l,r;
    sieve();
    sort(a.begin(),a.end());
    cin>>t;
    while(t--)
    {
    i=0;
    cin>>l>>r;
    if(r<a[0] || l>a[a.size()-1] || l>r)
    {
    cout<<"0\n";
    continue;
    }
    while(i<a.size() && l>a[i])
    {
    i++;
    }
    j=a.size()-1;
    while(j>=0 && r<a[j])
    {
    j--;
    }
    if(j<i)
    cout<<"0\n";
    else
    {
    	int ans = upper_bound(a.begin(), a.end(), r) - lower_bound(a.begin(), a.end(), l-1);
    	cout<<ans<<endl;
    }
    }
    }
