#!/usr/bin/python

import sys

def compare(x, y):
    if x[0] > y[0]:
        return -1
    if x[0] < y[0]:
        return 1
    if x[1] > y[1]:
        return 1
    return -1


T = input()
for i in range(T):
    arr = [(0, each) for each in range(97, 123)]
    word = raw_input().strip()
    for each in word:
        val = ord(each)
        pos = val - 97
        c, d = arr[pos]
        arr[pos] = (c+1, d)
    arr.sort(cmp=compare, reverse=True)
    for each in arr:
        sys.stdout.write("%s " % (chr(each[1])))
    print
