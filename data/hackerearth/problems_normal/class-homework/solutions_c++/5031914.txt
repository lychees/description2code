#include <bits/stdc++.h>

using namespace std;

#define ll long long int

ll num[25][15];

ll f(string& s,ll a,ll b)
{
    ll res,len=b-a+1;
    if(len>13)
        return -1;
    if(s[a]=='0')
    {
        if( len==1 )
            return 0;
        return -1;
    }
    res=num[a][len];
    if( res>1000000000000 )
        return -1;
    return res;
}

ll maketwo(string& s,ll a,ll b)
{
    ll res=-1,i,one,two;
    for(i=a;i<=(b-1);i++)
    {
        one=f(s,a,i);
        two=f(s,i+1,b);
        if( one!=-1 && two!=-1 )
            res=max(res,one+two);
    }
    return res;
}

ll maketwopairs(string& s,ll a,ll b,ll l,ll r)
{
    ll one=maketwo(s,a,b);
    ll two=maketwo(s,l,r);
    if( one!=-1 && two!=-1 )
        return one+two;
    return -1;
}

ll solve(string& s,ll l,ll r)
{
    ll res,i,len=r-l+1;
    if( len<4 )
        return -1;
    res=-1;
    for(i=l+1;i<=(r-2);i++)
        res=max(res,maketwopairs(s,l,i,i+1,r));
    return res;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string s;
    ll res,t,i,j,len;
    cin>>t;
    while( t-- )
    {
        cin>>s;
        len=s.length();

        for(i=0;i<len;i++)
            num[i][0]=0;

        for(i=0;i<len;i++)
        {
            for(j=1;j<=13;j++)
            {
                if( (j+i-1)>(len-1) )
                    continue;
                num[i][j]=num[i][j-1]*10+(s[j+i-1]-'0');
            }
        }

        res=solve(s,0,len-1);
        if( res<0 )
            cout<<"unlucky\n";
        else
            cout<<res<<"\n";
    }

    return 0;
}
