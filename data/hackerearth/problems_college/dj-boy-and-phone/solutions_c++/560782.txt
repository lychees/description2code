#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>
#define mod 1000000009
using namespace std;
long long dp[10007][11];
vector<int> v[12];
void pre()
{
    for(int i=0;i<9;i++)
    v[i].clear();
    memset(dp,0,sizeof(dp));
    v[0].push_back(0);v[0].push_back(8);
    v[1].push_back(2);v[1].push_back(4);v[1].push_back(1);
    v[2].push_back(1);v[2].push_back(3);v[2].push_back(5);v[2].push_back(2);
    v[3].push_back(3);v[3].push_back(2);v[3].push_back(6);
    v[4].push_back(4);v[4].push_back(1);v[4].push_back(5);v[4].push_back(7);
    v[5].push_back(5);v[5].push_back(2);v[5].push_back(4);v[5].push_back(6);v[5].push_back(8);
    v[6].push_back(6);v[6].push_back(5);v[6].push_back(3);v[6].push_back(9);
    v[7].push_back(7);v[7].push_back(4);v[7].push_back(8);
    v[8].push_back(8);v[8].push_back(0);v[8].push_back(5);v[8].push_back(9);v[8].push_back(7);
    v[9].push_back(9);v[9].push_back(6);v[9].push_back(8);
    for(int i=0;i<10;i++)
        dp[1][i]=1;
    for(int i=2;i<=10000;i++)
    {
        for(int j=0;j<10;j++)
        {
                for(int k=0;k<v[j].size();k++)
                dp[i][j]=(dp[i][j]+dp[i-1][v[j][k]])%mod;
        }
    }
}
int main()
{
    pre();
    int t;
    cin>>t;
    while(t--)
    {
        long long ans=0;
        int n;
        cin>>n;
        for(int i=0;i<10;i++)
        ans=(ans+dp[n][i])%mod;
        cout<<ans<<endl;
    }
    return 0;
}
