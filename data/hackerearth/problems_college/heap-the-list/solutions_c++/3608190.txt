#include<bits/stdc++.h>
using namespace std;

struct compare
{
	
	bool operator()(const int& l,const int& r)
	{
		
		return l>r;
	}

};

int main()
{   priority_queue<int,vector<int>, compare > pq;
    int n=100;
	while(n--)
	{
		long int x;
		cin>>x;
		pq.push(x);
		
	}
	
	
	cout<<pq.top();
	
	
	return (0);
	
}
