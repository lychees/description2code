ll power(ll b,ll e)
{
	ll x=1;
	while(e>0)
	{
		if(e&1)
			x=x*b;
		b=b*b;
		e=e/2;
	}
	return x;
}  
ll gcd(ll a,ll b)
{
    if(b==0)
          return a;
    return gcd(b,a%b);
}
ll mult(ll A,ll B)
{
	if ( B == 0 ) return 0;
	ll u = mult(A, B/2);
	ll res;
	if ( B%2 == 0 ) 
		res = u + u;
	else
		res = u + u + A;
	while ( res >= mod ) res -= mod;
	return res;
}

ll recur(ll l,ll u,ll count)
{
	if(count==2)
	{
		if(l==(u-1)) 
			return s[l]-48;
		else
		{
			if(u-l==13)
			{
				ll p=0;
				for(int j=l;j<u;j++)
					p=p*10+s[j]-48;
				if(p==1000000000000)
					return p;
				else return -1;
			}
			if(u-l>12) return -1;
			if(s[l]=='0') return -1;
			else
			{
				ll p=0;
				for(int j=l;j<u;j++)
					p=p*10+s[j]-48;
				return p;
			}
		}
	}
	if(count==0)
	{
		ll ans=-1;
		
		for(int i=l+1;i<u-2;i++)
		{
			ll o1=recur(l,i+1,count+1); if(o1==-1) continue;
			ll o2=recur(i+1,u,count+1); if(o2==-1) continue;
			ans=max(ans,o1+o2);
		}
		return ans;
	}
	if(count==1)
	{
		ll p1=-1;
		for(int i=l;i<u-1;i++)
		{
			ll o1=recur(l,i+1,count+1); if(o1==-1) continue;
			ll o2=recur(i+1,u,count+1); if(o2==-1) continue;
			p1=max(o1+o2,p1);
		}
		return p1;
	}
}
