#include <stdio.h>
#include <cstring>
using namespace std;

char message[101];
int main(){
	int i;
	gets (message);
	int n=strlen(message);
	char fl = 'k';
	char fo = 'k';
	char fv = 'k';
	char fe = 'k';
	for(i=0; i<n; i++){
		if (message[i]=='l')
			fl='l';
		if (message[i]=='o' && fl=='l')
			fo='o';
		if (message[i]=='v' && fo=='o')
			fv='v';
		if (message[i]=='e' && fv=='v')
			fe='e';
	}
	
	if (fe=='e')
		printf ("I love you, too!\n");
	else
		printf ("Let us breakup!\n");
    return 0;
}
