#include <iostream>
#include <set>
using namespace std;

typedef long long LL;

int main()
{
  int T;
  cin >> T;
  while(T--) {
    LL L, R, K;
    LL ans = 0;
    cin >> L >> R >> K;
    LL sel = L;
    while(sel <= R) {
      ans ++;
      sel = (sel*K)+1;
    }
    cout << ans << endl;
  }
  return 0;
}
