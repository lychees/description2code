#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int h;
   	int m;
   	cin>>h>>m;
   	h=h%12;
   	
   	
   	
   	int angleh=.5*(60*h+m);
   	int minh=6*m;
   	
   	
   	
   	int res=abs(angleh-minh);
   	//int res=(0.5*(60*h-11*m));
   
   	if(res>180)
   	res=360-res;
   	cout<<res<<endl;
   }
    return 0;
}
