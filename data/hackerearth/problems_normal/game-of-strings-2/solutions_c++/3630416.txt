#include <bits/stdc++.h>
using namespace std;
char s[1000004];
int f[1000005];
int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int k;
        scanf("%s %d", s, &k);
        int n = strlen(s);
        set<int> isGood;
        for (int i = 1; i < n; ++i) {
            int j = f[i - 1];
            while(j and s[i] != s[j])
                j = f[j - 1];
            j += s[i] == s[j];
            if (i < k)
                isGood.insert(j);
            f[i] = j;
        }        
        int len = f[n - 1];
        bool found = false;
        while (len) {
            if(isGood.find(len) != isGood.end()) {
                printf("%*.*s\n", len, len, s);
                found = true;
                break;
            }
            len = f[len - 1];
        }
        if(!found)
            printf("Puchi is a cheat!\n");
    }     
}
