#include <iostream>
#include <cstdlib>
#include <cmath>

using namespace std;


long sum(long n)
{
	long num_dig=0,num=n;
   	long sum_dig=0,j=1;
   	int i;
   	while(n!=0)
   	{
   		n=n/10;
   		num_dig++;
   	}
   	for(i=1;i<num_dig;i++)
   	{
   		sum_dig+=(i*9*j);
   		j=j*10;
   	}
   	sum_dig+=((num-j+1)*i);
   	return sum_dig;
 
}
 
int main()
{
   int t;
   scanf("%d",&t);
  
   while(t--)
   {long n;
   scanf("%ld",&n);
   
   	printf("%ld\n",sum(n));
   }
   return 0;
 
}
