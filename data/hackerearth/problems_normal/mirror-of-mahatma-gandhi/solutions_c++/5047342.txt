#include <iostream>
#include <cstring>
#include <algorithm>
using namespace std;

int main()
{
    int t,start,end;
    cin>>t;
    while(t!=0)
    {
    	string str;
    	int l,j=0;
    	cin>>str;
    	l=str.length();
    	bool flag=0;
    	for(int i=0; i<l; i++)
    	{
    		
    		if(str[i]!='0' &&  str[i]!='1' && str[i]!='8' )
    			{
    				flag=1;
    				break;
    			}
    	}
    	if(flag==1)
    		cout<<"NO\n";
    	else
    	{
    		 flag=0, start=0 , end=str.size()-1;
            while(start<end)
            {
                if(str[start]!=str[end])
                {
                    flag=1;
                    break;
                }
                start++, end--;
            }
            if(flag==0) cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
    	}	
    		
    	t--;
    }
    return 0;
}
