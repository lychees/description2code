int main(){
   cin.sync_with_stdio(false);
   list< int > lis ;
                  int N;

                  vector< int > v(N);

                    sort(v.begin(),v.end());
             vector< int > vv;
             int st = 0;
             int en = N - 1;
             int val = 0;
             int Ans = 0;
             int  pre = v[0];
             while( val <= N ){
              val++;
              if(val & 1){
              Ans += abs(v[st] - pre);
              pre = v[st];
              st++;
              }
              else {
              Ans += abs(v[en] - pre);
              pre = v[en];
              en--;
              }
             }
             int Ans1 = 0;
              pre = v[N-1];
             val = 0;
             st = 0;
             en = N-1;
             while( val <= N ){
              val++;
              if(val % 2 == 0){
              Ans1 += abs(v[st] - pre);
              pre = v[st];
              st++;
              }
              else {
              Ans1 += abs(v[en] - pre);
              pre = v[en];
              en--;
              }
             }
             Ans = max(Ans , Ans1);
             cout << Ans << endl;
}
