#include<bits/stdc++.h>
using namespace std;
int main()
{

    {
        string s,t;
        cin>>s>>t;
        int n1=s.length();
        int n2=t.length();
        set<string>p;
        int i,j,ans=0;
        for(i=0;i<n1;i++)
        {
            for(j=i;j<n1;j++)
            {
                string z=s.substr(i,j-i+1);
                if(p.find(z)==p.end())
                {
                	p.insert(z);
                int x=z.find(t);
                if(x!=string::npos)
                    ans++;
                }
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
