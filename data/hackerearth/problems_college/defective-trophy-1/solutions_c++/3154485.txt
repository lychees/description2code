#include <bits/stdc++.h>
using namespace std;
 
int solve(int n)
{
int a=3,b=1;
if(n<=1)return 0;
if(n==2)return 1;
while(n>a)
{
 b++;
 a*=3;
 
}
 
return b;
 
}
 
int main()
 
{
   
   int t;
   cin>>t;
   while(t--)
   {
   	int a;
   	cin>>a;
   	cout<<solve(a)<<endl;
   	
   }
   
   return 0;
}
