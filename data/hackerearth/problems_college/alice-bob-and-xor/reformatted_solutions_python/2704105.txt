def xr(N):
    if (N % 2) == 0:
        if (N % 4) == 0:
            r = N
        else:
            r = N+1
    else:
        if (N % 4) == 1:
            r = 1
        else:
            r = 0
    return r


for _ in range(int(raw_input())):
    a, b = map(int, raw_input().split())
    x, y = xr(a-1), xr(b)
    print x ^ y
