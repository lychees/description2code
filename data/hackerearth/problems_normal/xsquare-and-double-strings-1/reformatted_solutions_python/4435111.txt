N = input()
dict = {}
for i in range(0, N):
    count = 0
    for k in dict.keys():
        dict[k] = 0
    string = raw_input()
    for j in range(0, len(string)):
        if string[j] in dict.keys():
            dict[string[j]] += 1
        else:
            dict[string[j]] = 1
    for j in dict.values():
        if j > 1:
            count = 1
            print "Yes"
            break
    if count == 0:
        print "No"
