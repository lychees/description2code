#include<iostream>
#include<algorithm>
#include<climits>
#include<string.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char s[102];
		scanf("%s",s);
		int len;
		len=strlen(s);
		int i;
		bool ispalindrome=true;
		for(i = 0; i<len/2; i++)
		{
			if(s[i] != s[len - 1 -i])
			{
				ispalindrome = false;	
				break;
			}
		}
		if(ispalindrome==false)
		printf("-1\n");
		else
		{
			if(len%2==0)
			printf("%d\n",s[len/2-1]);
			else
			printf("%d\n",s[(len/2)]);
		}
	}
	return 0;
}
