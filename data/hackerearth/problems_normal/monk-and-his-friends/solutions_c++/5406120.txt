#include <iostream>
#include<vector>
#include<algorithm>
#include<set>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int a=0;a<t;a++){
    	int n ,m ;
    	scanf("%d %d",&n,&m);
    	set<long long int> s1;
    	for(int i=0;i<n;i++){
    		long long temp; scanf("%lld",&temp);
    		s1.insert(temp);
    	}
    	for(int i=0;i<m;i++){
    		long long temp;scanf("%lld",&temp);
    		if(s1.find(temp)!=s1.end()){
    			printf("YES \n");
    		}
    		else{
    			printf("NO \n");
    			s1.insert(temp);
    		}
    		
    	}
    	
    }
    
}
