#include <cstdio>

#include <string.h>

#define ll long long int
#define maxN 1000000
#define maxVal 100000000
#define minVal -100000000
#define mod 1000000007LL

#define gcd(a,b) __gcd(a,b)

using namespace std;

char a[maxN+5],p[3];

int main()
{
    int n,q,c=0,i;
    char x;
    scanf("%d%d%s",&n,&q,a);
    for(i=1;i<n;i++)
    {
    	if(a[i]==a[i-1]&&a[i]=='.')
    		c++;
    }
    while(q--)
    {
    	scanf("%d%s",&i,p);
    	x=p[0];
    	i--;
    	if(a[i]!=x)
    	{
    		if(x=='.')
    		{
    			//value may increase
    			if((i-1)>=0&&a[i-1]=='.')
    				c++;
    			if((i+1)<n&&a[i+1]=='.')
    				c++;
    		}
    		else
    		{
    			//value may decrease
    			if(a[i]=='.')
    			{
    				if((i-1)>=0&&a[i-1]=='.')
    					c--;
    				if((i+1)<n&&a[i+1]=='.')
    					c--;
    			}
    		}
    		a[i]=x;
    	}
    	printf("%d\n",c);
    }
    return 0;
}
