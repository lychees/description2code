#include<bits/stdc++.h>
using namespace std;
bool prime(int n)
{
	int i;
	if(n==2)
	return true;
	if(n==0||n%2==0)
	return false;
	else
	{
		for(i=3;i<=sqrt(n);i++)
		{
			if(n%i==0)
			return false;
		}
	}
	return true;
}
int main()
{
	int i,j,k,n;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		if(!prime(i))
		cout<<i<<endl;
	}
	return 0;
}
