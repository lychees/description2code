#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int N,K,a[100001]={0},b[100001]={0};
    	scanf("%d %d",&N,&K);
    	for(int i=0;i<N;i++)
    	{
    		scanf("%d",&a[i]);
    		b[a[i]]++;
    	}
    	long long int sum=0;
    	for(int i=100000;i>=0&&K>0;i--)
    	{
    		if(b[i]!=0)
    		{
    			while(b[i]&&K>0)
    			{
    				sum=sum+i;
    				K--;
    				b[i]--;
    			}
    		}
    	}
    	printf("%lld\n",sum);
    }
}
