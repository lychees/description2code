#include<bits/stdc++.h>
using namespace std;
int main(){

  cin.sync_with_stdio(false);
   int t;
   cin >> t;
   while(t--){
    
     string x;
     cin >> x;
     int ans = 0;
     for(int i = 1 ; i < x.size() ; ++i ) 
       if(x[i]==x[i-1])
       ans++;
       
       cout << ans << endl;
   
   }

}
