#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int total=0;
        int n,k;
        int a;
        cin>>n>>k;
        while(n--)
        {
            cin>>a;
            if(a<=0)
                total++;
        }
        if(total<k)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
        
    }
return 0;
}
