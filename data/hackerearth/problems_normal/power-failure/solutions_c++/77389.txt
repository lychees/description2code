#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
#define mod 1000000007
#define LL long long
#define MAXX 110000
int array[MAXX],sum[MAXX];
void precompute(){
	array[1]=1;
	for(int i=2;i<MAXX;i++){
		if(array[i]==0){
			int j=2*i;
			while(j<MAXX){
				array[j]=1;
				j+=i;
			}
		}
	}
	for(int i=1;i<MAXX;i++){
		sum[i]=sum[i-1]+array[i];
	}
}
int main(){
	int tst;
	scanf("%d",&tst);
	precompute();
	while(tst--){
		int n,m;
		scanf("%d%d",&n,&m);
		vector<int>vecc(n);
		for(int i=0;i<n;i++)scanf("%d",&vecc[i]);
		sort(vecc.begin(),vecc.end());
		reverse(vecc.begin(),vecc.end());
		LL ans=1;
		int initial_val=sum[m];
		//cout<<initial_val<<endl;
		for(int i=0;i<n;i++){
			ans=ans*(initial_val-sum[vecc[i]-1]);
			//cout<<ans<<endl;
			if(ans>=mod)ans%=mod;
			initial_val--;
		}
		printf("%d\n",ans);
	}
	return 0;

}
