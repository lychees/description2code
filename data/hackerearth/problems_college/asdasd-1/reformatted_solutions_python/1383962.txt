def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)  # supposing you want multiple factors repeated
            n //= d
        d += 1
    if n > 1:
        primfac.append(n)
    return primfac


def f(x):
    return pow(2, x)-1


for i in range(int(raw_input())):
    a, b = map(int, raw_input().split())
    x = f(a)
    y = f(b)
    A = primes(x)
    B = primes(y)
    c = set(A) & set(B)
    c = list(c)
    print len(c)
