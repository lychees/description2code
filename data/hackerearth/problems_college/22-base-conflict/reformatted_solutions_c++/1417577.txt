#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll find(string str, ll k) {
  int n = str.length();
  ll po = 1;
  ll ans = 0;
  for (int i = n - 1; i >= 0; i--) {
    ll x = 0;
    if (str[i] >= 48 && str[i] <= 57) {
      x = str[i] - 48;
    } else if (str[i] >= 'A' && str[i] <= 'F') {
      x = str[i] - 'A' + 10;
    } else
      return -1;
    if (x >= k)
      return -1;
    ans += (po * x);
    po = po * k;
  }
  return ans;
}
int main() {
  int n;
  scanf("%d", &n);
  string a[n];
  for (int i = 0; i < n; i++) {
    cin >> a[i];
  }
  ll sum;
  scanf("%lld", &sum);
  for (ll k = 2; k <= 16; k++) {
    ll cnt = 0;
    for (int i = 0; i < n; i++) {
      ll x = find(a[i], k);
      if (x == -1) {
        cnt = -1;
        break;
      }
      cnt += x;
    }
    if (cnt == sum) {
      printf("%lld\n", k);
      return 0;
    }
  }
}
