#include<iostream>
using namespace std;
int main()
{
	int t,n,ctr1,ctr2,change,temp,x=-1;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ctr1=n;ctr2=0,change=n;
		while(ctr1)
		{
			if((ctr1%8==0)&&(ctr2%6==0))
			{
     		 temp=((ctr1/8)+(ctr2/6));
			 if(temp<change)
			 change=temp;
		    }
			ctr2++;
			ctr1--;
		}
		if(change==n)
		cout<<x<<"\n";
		else
		cout<<change<<"\n";
	}
}
