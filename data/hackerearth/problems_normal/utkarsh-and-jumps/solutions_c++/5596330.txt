#include <iostream>
using namespace std;

int main(){
    int N, P;
    cin >> N >> P;
    double p = P / 100.0;
    double dp[N + 10];
    dp[0] = 1;
    for(int i = 1;i <= N;++i){
    	dp[i] = 0;
    }
    for(int i = 0;i <= N;++i){
    	dp[i + 2] += dp[i] * p;
    	dp[i + 3] += dp[i] * (1 - p);
    	//cout << dp[i] << "\n";
    }
    printf("%.6lf\n", dp[N]);
    return 0;
}
