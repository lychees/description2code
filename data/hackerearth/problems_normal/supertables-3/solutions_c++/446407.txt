#include<iostream>
using namespace std;
long int GCD(long int a,long int b){
	long int c;
	while(a!=0){
		c = a; 
		a = b%a;  
		b = c;
	}return b;
}
void input(){
	long int a=0,b=0,n=0;
	cin>>a>>b>>n;
	long int lcm=((a*b)/GCD(a,b));
	long int x1=lcm/a;
	long int x2=lcm/b;
	long int total=x1+x2-1;
	long int no=n/total;
	long int ans=no*lcm;
	long int mod=n%total;
	long int countA=1,countB=1;
	if(no!=0){
		countA=x1*no+1;
		countB=x2*no+1;
	}
	for(long int i=0;i<mod;i++){
		if((countA*a)<(countB*b)){
			ans=(countA*a);
			countA=countA+1;
		}
		else if((countA*a)>(countB*b)){
			ans=(countB*b);
			countB=countB+1;
		}
		else{
			cout<<"NOT POSSIBLE";
		}
	}
	cout<<ans<<"\n";
}
int main(){
	long int total=0;
	cin>>total;
	for(long int i=0;i<total;i++)
		input();
	return 0;
}
