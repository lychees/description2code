#include<iostream>
using namespace std;

int a[100]={0};

int main(void)
{
  int T, N;
  cin>>T;
  while(T--){
    int mino = 1, mine = 0, maxo = 1, maxe = 0, p, ans=0;
    cin>>N;
    for(int i=0; i<N; i++){
      cin>>a[i];      
    }
    cin>>p;
    p--;
    for(int i=0; i<N; i++){
      if(i%2){
	if(a[mino]>a[i])
	  mino = i;
	if(a[maxo]<a[i])
	  maxo = i;
      }
      else{
	if(a[mine]>a[i])
	  mine = i;
	if(a[maxe]<a[i])
	  maxe = i;
      }
      //      cout<<p<<" : p"<<endl;
      if(a[maxe] - a[mine] > p){
	if(maxe>mine){
	  ans = maxe/2+1;
	}
	else
	  ans = mine/2+1;
	break;
      }
      else if(a[maxo] - a[mino] > p){
	if(maxo>mino)
	  ans = maxo/2+2;
	else
	  ans = mino/2+2;
	break;
      }
      else if(a[maxo] - a[mine] > p){
	if(maxo>mine)
	  ans = maxo/2+2;
	else
	  ans = mine/2+2;
	break;
      }
      else if(a[maxe] - a[mino] > p){
	if(maxe>mino)
	  ans = maxe/2+2;
	else
	  ans = mino/2+2;
	break;
      }
    }
    if(ans)
      cout<<ans<<endl;
    else
      cout<<N<<endl;
  }
  return 0;
}
