#include<stdio.h>
#include<string.h>
char arr[100000]={NULL};
int main()
{
int test;
scanf("%d",&test);
int count=0,i,len;
while(test--)
{
    count=0;
scanf("%s",arr);
len=strlen(arr);
for(i=0;i<len;i++)
{
if(arr[i]=='B')
{
if(i>=1)
arr[i-1]='1';
if(i>=2)
arr[i-2]='1';
}
else if(arr[i]=='S')
{
if(i>=1)
arr[i-1]='1';
if(i+1<=len-1&&arr[i+1]=='-')
arr[i+1]='1';
}
}
for(i=0;i<len;i++)
{
if(arr[i]=='-')
    count++;
}
printf("%d\n",count);
}
return 0;
}
