#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int a[4];
    	for(int i=0;i<4;i++)
    	a[i]=0;
    	for(int j=0;j<s.size();j++)
    	{
    		if(s[j]-'r'==0)
    		a[0]++;
    		else if(s[j]-'u'==0)
    		a[1]++;
    		else if(s[j]-'b'==0)
    		a[2]++;
    		else if(s[j]-'y'==0)
    		a[3]++;
    	}
    	int min=a[0];
    	for(int k=0;k<4;k++)
    	{
    		if(a[k]<min)
    		min=a[k];
    	}
    	cout<<min<<endl;
    }
    return 0;
}
