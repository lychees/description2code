nq = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
for i in range(nq[1]):
    query = map(int, raw_input().split())
    summ = 0
    if query[0] == 1:
        summ = sum(a[query[1]-1:query[2]:2] + b[query[1]:query[2]:2])
    else:
        summ = sum(b[query[1]-1:query[2]:2] + a[query[1]:query[2]:2])
    print summ
