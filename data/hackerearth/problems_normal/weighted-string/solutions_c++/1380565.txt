#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int k,init=0,sum=0,ans=0;
    	string s;
    	cin>>k>>s;
    	for(int i=0;i<s.length();i++){
    		sum+=s[i]-'a'+1;
    		while(sum>k&&init<=i){
    			sum-=s[init]-'a'+1;
    			init++;
    		}
    		if(sum==k)
    		ans++;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
