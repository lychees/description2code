#include <iostream>
using namespace std;

int total;

int calculate(int num)
{
	total = total +num;
	if (num != 0)
	{
		calculate ((num-1));
	}
}

int main()
{
	int T;
	cin >> T;
	
	for (int x; x<T; x++)
	{
		int num;
		cin >> num;
		calculate(num);
		cout << total << "\n";
		total =0;
	}
	
    return 0;
}
