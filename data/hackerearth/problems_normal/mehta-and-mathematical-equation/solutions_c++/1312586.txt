#include<bits/stdc++.h>
using namespace std;

const int md=1e9+7;
typedef long long ll;

int len;
string str;
ll dp[67][3][2];

void conv(ll x) {
    str="";
    while(x) {
        str+=(char)(x%2+48);
        x>>=1;
    }
    reverse(str.begin(),str.end());
    len=str.size();
    return;
}
ll solve(int idx,int carry,int flag) {
    if(idx==-1) {
        if(carry==0&&flag) {
            return 1;
        }
        return 0;
    }
    ll& res=dp[idx][carry][flag];
    if(res!=-1) {
        return res;
    }
    res=0;
    for(int i=0;i<2;i++) {
        for(int j=0;j<2;j++) {
            for(int k=0;k<2;k++) {
                int sum=((i+j+k)==3);
                if((i+j+k+carry)%2==str[idx]-48) {
                    res=(res+solve(idx-1,(i+j+k+carry)/2,flag|sum))%md;
                }
            }
        }
    }
    return dp[idx][carry][flag]=res;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        ll n;
        scanf("%lld",&n);
        conv(n);
        memset(dp,-1,sizeof(dp));
        printf("%lld\012",solve(len-1,0,0));
    }
    return 0;
}
