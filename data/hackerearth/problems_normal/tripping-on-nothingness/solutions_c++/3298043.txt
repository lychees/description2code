#include <iostream>
#include <map>
using namespace std;

int gcd(int a, int b){
	if(a == 0){
		return b;
	}
	return gcd(b%a,a);
}

int main()
{
    int s,q,a,temp;
    map<int,int>h;
    cin >> s >> q;
    while(q--){
    	cin >> a;
    	temp = gcd(a,s);
    	if(h[temp]){
    		cout << -1 << endl;
    	}
    	else{
    		h[temp] = -1;
    		cout << temp << endl;
    	}
    }
    return 0;
}
