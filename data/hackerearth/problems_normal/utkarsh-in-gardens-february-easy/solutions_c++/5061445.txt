#include <iostream>
#include<algorithm>
#include<map>
#include<vector>
#include<stack>
using namespace std;

int a[2005][2005];
int mark[2005]={};
vector<int> node[2005];
long long int ans=0;
int n;


int main()
{
	ios::sync_with_stdio(false);
	int i,j,k,c;

	cin>>n;
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		{
			cin>>a[i][j];
			if(a[i][j]==1)
				node[i].push_back(j);
		}
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			c=0;
			for(k=0;k<node[i].size();k++)
				if(a[j][node[i][k]]==1)
					c++;
			ans+=(c*(c-1))/2;
		}
	}
	
	cout<<ans/2<<endl;
	
    return 0;
}
