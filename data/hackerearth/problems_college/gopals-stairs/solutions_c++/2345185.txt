#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	long long int f[100000],i,n;
          f[0]=1;
          f[1]=1;
          f[2]=2;
          for(i=3;i<=100000;i++)
          {
                    f[i]=(f[i-3]+f[i-2]+f[i-1])%1000000007;
          }
          int t;
          scanf("%d",&t);
          while(t--)
          {
          scanf("%lld",&n);
          printf("%lld\n",f[n]);
          }
          return 0;
}
