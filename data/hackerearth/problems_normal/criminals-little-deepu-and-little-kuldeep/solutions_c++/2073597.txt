#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	cin>>n;
    	long a[100005],c=1,k=1;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	for(i=0;i<n-1;i++)
    	if(a[i]==a[i+1])
    	c++;
    	else          
        {
          if (c > k)
        k = c; 
          c = 1;      
        }
        if(c>k) k=c;
    	
    	cout<<k<<endl;
    	
    }
    return 0;
}
