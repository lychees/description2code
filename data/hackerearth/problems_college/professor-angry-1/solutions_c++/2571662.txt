#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,k,arr[1005],i,j,count;
    cin>>t;
    while(t--){
    int flag=0;
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>arr[i];
    sort(arr,arr+n);
    for(i=0;i<=arr[n-1];i++)
    {
        count=0;
        for(j=0;j<n;j++)
        {
            if(arr[j]<=i)
                count++;
        }
        if(count<k)
            {flag=1;cout<<"YES\n";break;}
    }
    if(flag==0)
        cout<<"NO\n";
    }
    return 0;
}
