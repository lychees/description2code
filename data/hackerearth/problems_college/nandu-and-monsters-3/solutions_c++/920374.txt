#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    
    int n,q;
    scanf("%d%d",&n,&q);
    map<int,bool> m;
    string s;
    cin>>s;
    for(int i=1;i<=n;i++){
    	if(s[i-1] == '1'){
    		m[i] = true;
    	}
    }
    while(q--){
      	    int i,j;
    	    scanf("%d%d",&i,&j);
    	
	    	bool f = false;
	    	
	    	if(m[i] || m[j]){
	    			f = true;
	    	}
	    	
	    	while(i != j){
	    		
	    		
	    		
	    		while(i > j){
	    			if(m[i]){
	    				f = true;
	    			}
	    			i/=2;
	    		}
	    		
	    		while(j > i){
	    			if(m[j]){
	    				f = true;
	    			}
	    			j/=2;
	    		}
	    		
	    		if(m[i] || m[j]){
	    			f = true;
	    		}
	    		
	    	}
	    	if(f){
	    		cout<<"No\n";
	    	}
	    	else{
	    		cout<<"Yes\n";
	    	}
	    	
	    	
    	}
    
    
    return 0;
}
