testcases_count = raw_input()
for num in range (1 , int(testcases_count) + 1):
    case = raw_input()
    values = case.split(' ')
    val_arr = []
    friends_count = int(values[0].strip())
    estimate = int(values[1].strip())
    d = 0
    yes_flag = False
    for c in range(1, friends_count + 1):
        d += 1
        cost = int(raw_input())
        val_arr.append(cost)
        arr_sum = sum(val_arr)
        #print arr_sum
        if arr_sum == estimate:
            yes_flag = True
            break
        elif arr_sum > estimate:
            while sum(val_arr) > estimate:
                val_arr.pop(0)
            if sum(val_arr) == estimate:
                yes_flag = True
                break
        
        #print c , "NO"
    while d < friends_count:
        d += 1
        raw_input()
    if yes_flag == True:
        print "YES"
    else:
        print "NO"
