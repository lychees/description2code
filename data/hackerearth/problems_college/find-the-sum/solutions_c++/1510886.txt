#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll N = 1000001;
const ll INF = 2009000999;
const ll md =1000*1000*1000+7;
ll t,n,l,r;
ll f(ll x)
{
    if (x==0)
        return 0;
    if (x==1)
        return 1;
    ll n=(sqrt(double(4+8*x))-1)*0.5;
    return (n/2+n%2)*(n/2+n%2)+((x-(n*(n+1)/2))%2)*(n+1);
}
int main()
{
    scanf("%lld",&t);
    while (t--)
    {
        scanf("%lld%lld",&l,&r);
        printf("%lld\n",f(r)-f(l-1));
    }
}
