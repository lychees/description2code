    using namespace std;
    void costmat(int mat[1001][1001],int cost[1001][1001],int i,int j,int ccost,int x,int y,int m,int n)
    {
    	if(i<0 || j<0 || i>=m || j>=n ||(i>x && j>y))
    	   return ;
    	ccost=ccost+mat[i][j];
    	// Check if cost is minimum than previous.
    	if(ccost<cost[i][j] || cost[i][j]==-1)
    	{
    		cost[i][j]=ccost;
    		// Can go in any direction.
    		costmat(mat,cost,i,j+1,ccost,x,y,m,n);
    		costmat(mat,cost,i+1,j,ccost,x,y,m,n);
    		costmat(mat,cost,i,j-1,ccost,x,y,m,n);
    		costmat(mat,cost,i-1,j,ccost,x,y,m,n);
    	}
    }
    int main()
    {
       int test,m,n,i,j,mat[1001][1001],cost[1001][1001],x,y,t;
       cin>>test;
       while(test--)
       {
       	   cin>>m>>n;
       	   for(i = 0 ; i < m ; i++)
       	   {
       	   	 for(j = 0 ; j < n ;j++)
       	   	 {
       	   	 	  cin>>mat[i][j];
       	   	 	  cost[i][j]=-1;
       	   	 }
       	   }
       	   cin>>x>>y>>t;
       	   costmat(mat,cost,0,0,0,x-1,y-1,m,n);
       	   //cout<<cost[x-1][y-1]<<endl<<t;
       	   if(t>=cost[x-1][y-1])
       	   {
       	   	  cout<<"YES\n"<<t-cost[x-1][y-1]<<endl;
       	   }
       	   else
       	    cout<<"NO\n";
       }
       
        return 0;
    }
