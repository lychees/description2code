#include <iostream>
#include <vector>
using namespace std;

typedef long long ll;

ll mod = 1000000007;

int main()
{
	int BIG = 100001;
	vector<ll> A(BIG, 0);
	vector<ll> B(BIG, 0);
	vector<ll> F(BIG, 0);
	vector<ll> G(BIG, 0);
	
	A[1] = 1;
	B[1] = 1;
	
	for(int i=2;i<=BIG;i++){
		A[i] = A[i-1] + B[i-1];
		B[i] = A[i-1];
		F[i] = F[i-1] + G[i-1];
		G[i] = F[i-1] + B[i-1];
		
		A[i]%=mod;
		B[i]%=mod;
		F[i]%=mod;
		G[i]%=mod;
	}
	
	int T;
	cin >> T;
	for(int t=0;t<T;t++){
		int N;
		cin >> N;
		cout << (F[N]+G[N])%mod << endl;
	}
    return 0;
}
