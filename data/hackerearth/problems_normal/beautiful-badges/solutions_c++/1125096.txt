#include <stdio.h>
#include <iostream>

#include <string.h>

#include <vector>

#define w1 while(1)
#define pint(n) printf("%d\n",n)
#define fill(n,m) memset(n,m,sizeof(n))

using namespace std;

vector <float> azul;
vector <float> amar;

int used1[100005],used2[100005];

int main()
{
    int i,n,c=0,id,l,j;
    float diag = sqrt(2);
    scanf("%d",&n);
    for(i=0;i<n;i++) {
         scanf("%d %d",&id,&l);
         if(id==1) azul.push_back(l);
         else amar.push_back(l);
    }
    
    sort(azul.begin(),azul.end());
    sort(amar.begin(),amar.end());
    int k=0;
    for(i=0;i<azul.size();i++) {
         if(used1[i] == 1) continue;
         for(k=(lower_bound(amar.begin(),amar.end(),azul[i]/diag)-amar.begin());k<amar.size();k++){
              if(used2[k] == 1) continue;
              if(azul[i]*diag <= amar[k]) break;
              if(azul[i] < amar[k]*diag) {
                         c++;
                         used1[i]++;
                         used2[k]++;
                         //printf("%f with %f %d %d\n",azul[i],amar[k],used1[i],used2[k]);
              }
              if(used1[i]==1) break;
         }
    }
    printf("%d\n",c);
    return 0;
}
