#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		long long int n,arr[100005];
		cin >> n;
		for(int i=0;i<n;i++)
			cin >> arr[i];
		sort(arr,arr+n);
		long long int msum=0,mside1=0,mside2=0,mside3=0,flag=0;
		for(int i=n-1;i>1;i--)
		{
			int k=n-3;
			for(int j=n-2;j>0;j--)
			{
				while(k>=0 && arr[i]>=arr[j]+arr[k])
					k--;
				if(k>=0)
				{
					cout << arr[k] << ' ' << arr[j] << ' ' << arr[i] << '\n';
					flag=1;
					break;
				}
			}
			if(flag)
				break;
		}
		if(!flag)
			cout << "-1\n";

	}
	return 0;
}
