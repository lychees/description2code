#define MOD 1000000007LL
using namespace std;
long long facto[20000001];
long long int inv(long long int x) {
long long int M=MOD;
long long int a = 1, b = x;
while (b != 1) {
long long int c = M / b;
a *= c; a %= M;
b *= c; b %= M;
if (b > M/2) { a = M - a; b = M - b; }
}
return a;
}
long long fact(int x)
	{
	if(x<=1000000)
		return facto[x];
	long long ans=facto[1000000];
	for(int i=1000000+1;i<=x;i++)
		ans=(ans*i)%MOD;
	return ans;
	}
long long int nCr(long int n, long int p) {
long long int num=1,den=1;
int i;
for(i=0;i<p;i++)
	num=(num*(n-i))%MOD;
for(i=0;i<p;i++)
	den=(den*(i+1))%MOD;
//cout<<num<<" "<<den<<endl;
return (num * inv(den)) % MOD;
}
