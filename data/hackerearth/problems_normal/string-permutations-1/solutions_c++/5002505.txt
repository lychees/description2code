#include <iostream>
#include <map>

using namespace std;

int main()
{
	int t, count, max;
	string str;
	map<string, int> mp;	
	
    cin>>t;
    while(t--)
    {
    	cin>>str;
    	if(mp.find(str) != mp.end())
    	{
    		mp[str]++;
    	}
    	else
    	{
    		mp.emplace(str, 1);
    	}
    }
    auto it = mp.begin();
	max = it->second;
	count = 1;
	it++;
	while(it != mp.end())
	{
		if(it->second > max)
		{
			max = it->second;
			count = 1;
		}
		else if(it->second == max)
		{
			count++;
		}
		it++;
	}
	cout<<max*count<<endl;
    return 0;
}
