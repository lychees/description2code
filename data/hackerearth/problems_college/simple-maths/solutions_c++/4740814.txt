#include <iostream>
using namespace std;

long long int lc(long int num1,long int num2)
{
	int a, b, temp, gcd, lcm;
    a = num1;
    b = num2;
    for (; b > 0;) 
    {
            temp = b;
            b = a%b;
            a = temp;
    }
    gcd = a;
    lcm = (num1*num2)/gcd;
    return lcm;
}

int main()
{
    int T;
    cin>>T;
    long long int n,i;
    long int a,b,cnt;
    while(T--)
    {
    	cin>>n>>a>>b;
    	if(a==0||b==0)
    	{
    		cout<<"-1\n";
    		continue;
    	}
    	cnt=0;
    	if(a%b==0)
    		cnt+=(n/b);
    	else if(b%a==0)
    		cnt+=(n/a);
    	else
    	{
    		cnt+=(n/a);
    		cnt+=(n/b);
    		cnt-=(n/(lc(a,b)));
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
