    forn (i, n)
        scanf("%d", &a[i]), p[i] = i, res *= (i + 1), sum += a[i];
    int ans = 0;
    forn (qq, res)
    {
        forn (mask, (1 << n))
        {
            int cnt1 = 0, cnt2 = 0, flag = 1;
            forn (i, n)
            {
                if ((mask >> i) & 1)
                {
                    cnt1 += a[p[i]];
                    if (cnt1 > cnt2)
                        flag = 0;    
                }
                else
                    cnt2 += a[p[i]];
            }
            ans += flag;
        }
        next_permutation(p, p + n);
    }
    printf("%d %d\n", ans, sum);
    if (sum >= ans)
        puts("Got no way out!");
    else
        puts("We will win!");
    return 0;
}
