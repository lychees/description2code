for i in range(int(raw_input())):
    a=raw_input().split();
    x=a[0]
    y=a[1]
    r=0
    for j in range(len(x)):
        a1=int(x[j])
        a2=int(y[j])
        if(a1+a2>9):
            r=r+1
    if(r==0):
        print "No carry operation"
    elif(r==1):
        print "1 carry operation"
    else:
        print str(r)+" carry operations"
