#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
	char str[4000];
	int t,i;
	long long int sum=0,x,tempsum;
	scanf("%d",&t);
	while(t--){
		scanf("%s",str);
		tempsum=0;
		for(i=0;str[i]!='\0';i++)
		{
			tempsum+=(int)(str[i]-'0');
		}
		while(tempsum>=10){
			x=0;
			while(tempsum){
				x+=tempsum%10;
				tempsum/=10;
			}
			tempsum=x;
		}
		sum+=tempsum;
	
	}
	while(sum>=10){
		x=0;
		while(sum){
			x+=sum%10;
			sum/=10;
		}
		sum=x;
	}
	printf("%lld\n",sum);
    return 0;
}
