#include <iostream>
using namespace std;

int main()
{
    int t,i,c,j,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int ar[n];
    	for(i=0;i<n;i++)
    		cin>>ar[i];
    	c=0;
    	for(i=0;i<n;i++)
    		for(j=i;j<n;j++)
    			if(ar[i]==ar[j])
    				c++;
    	cout<<c<<endl;
    }
    return 0;
}
