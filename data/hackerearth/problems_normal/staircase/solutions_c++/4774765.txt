#include<bits/stdc++.h>
using namespace std;
int dp[1000005];
int rec(int n){
	if(dp[n]!=-1){
		return dp[n];
	}
	else if(n<=2){
		return n;
	}
	else if(n==3){
		return 4;
	}
	else{
		dp[n]=((rec(n-3))+rec((n-2))+rec(n-1));
		return dp[n];
	}
}
int rrec(int n){
	if(dp[n]!=-1){
		return dp[n];
	}
	else if(n<=2){
		return n;
	}
	else{
		dp[n]=((rrec(n-1))+(rec(n-2)));
		return dp[n];
	}
}
int main(){
	int n;cin>>n;
	if(n<=0){
		printf("ERROR\n");
	}
	else{
		memset(dp,-1,sizeof(dp));
		int ja=rec(n);
		memset(dp,-1,sizeof(dp));
		int ji=rrec(n);
		printf("Jack-%d\n",ja);
		printf("Jill-%d\n",ji);
	}
	return 0;
}
