#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
 
long list[10000];
long Dp[10000];
 
long f (long i, long j)
{
	if (i > j) // Base condition
		return 0;
 
	long &ret = Dp[i];
	if (ret != -1)
		return ret;
 
	ret = max(list[i] + f(i + 2, j), f(i + 1, j));
	return ret;
}
 
int main()
{
	long T, N;
 
	cin >> T;
	while(T--)
	{
		cin >> N;
		memset(Dp, -1, sizeof(Dp));
 
		for (long n = 0; n < N; n++)
		{
			cin >> list[n];
		}
 
		cout << f(0, N - 1) << endl;
	}
	return 0;
}
