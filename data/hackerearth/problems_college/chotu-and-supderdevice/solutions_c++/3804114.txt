#include <iostream>
using namespace std;
int ans (int x, int n)
{
	if (x == n) {
		return 0;
	} else {
		if (x*2 == n) {
			return 1;
		} else if (x*2 < n) {
			int i, j;
			i = ans (x*2, n) + 1;
			j = ans (x - 1, n) + 1;
			cout << i << j << endl;
			if (i <= j) {
				return i;
			} else {
				return j;
			}
		} else {
			return ans (x - 1, n) + 1;
		}
	}
}
int main()
{
	int i, t, x, n;
	cin >> t;
	for ( ; t; --t) {
		cin >> x >> n;
		if (x == n) {
			cout << "0" << endl;
		} else if (x > n) {
			cout << x - n << endl;
		} else {
			i = ans (x, n);
			cout << i << endl;
		}
	}
	return 0;
}
