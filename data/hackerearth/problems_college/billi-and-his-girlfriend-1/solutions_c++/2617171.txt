#include<bits/stdc++.h>
#define lli long long int

using namespace std;
lli fast_exp(lli base, lli exp,lli MOD) {
    lli res=1;
    while(exp>0) {
       if(exp%2==1) res=(res*base)%MOD;
       base=(base*base)%MOD;
       exp/=2;
    }
    return res%MOD;
}

int main()
{
    int t = 0 ;
    scanf("%d",&t);

    while(t--)
    {
        lli A=0,B=0,K=0,ans=0;
        scanf("%lld %lld %lld",&A,&B,&K);
        lli modulo = pow(10,K);

        lli result = fast_exp(A,B,modulo);

        //printf("%lld\n",result);


        while(result)
        {
            lli rem = result%10;
            ans +=rem;
            result /=10;
        }
        printf("%lld\n",ans);

    }
return 0;
}
