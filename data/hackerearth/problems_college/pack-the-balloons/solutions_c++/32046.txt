#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    unsigned long long len,wid,hgt,min,ans;
    int t,i;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        scanf("%lld%lld%lld",&len,&wid,&hgt);
        if(len<=wid&&len<=hgt)
        min=len;
        else
        if(wid<=len&&wid<=hgt)
        min=wid;
        else
        min=hgt;
        ans=min/2;
        if(min%2!=0)
        printf("%lld.5\n",ans);
        else
        printf("%lld.0\n",ans);
    }
return 0;
}
