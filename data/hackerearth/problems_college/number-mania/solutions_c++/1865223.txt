#include <bits/stdc++.h>

using namespace std;
int p[110];
int res[200];

int fact(int i)
{
    int k=1,j;
    int prod,carry;
    while(k<=i)
    {
        carry=0;
        for(j=0;j<200;j++)
        {
            prod=res[j]*k+carry;
            res[j]=prod%10;
            carry=prod/10;
            //cout<<res[j];
        }
        k++;
    }
//    return res;
}
int main()
{

        int a[4]={2,3,5,7};
        int i=1,j,n;
    p[1]=1;
    for(i=0;i<4;i++)
    {
        for(j=a[i]+1;j<=100;j++)
        {
            if(j%a[i]==0 && p[j]==0)
                {p[j]=1;
                //cout<<j<<endl;;
                }
        }
    }

    cin>>n;
    int k;
    for(i=2;i<=n;i++)
    {
        if(p[i]==0)
       {
            memset(res,0,200*(sizeof(int)));
        res[0]=1;
        fact(i);
        for(j=200;j>=0;j--){
            if(res[j]!=0)
                break;
        }
        for(k=j;k>=0;k--)
        {
            cout<<res[k];
        }
        cout<<endl;
       }
    }

}
