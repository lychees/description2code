#include<iostream>
#define mod 1000000007
using namespace std;
typedef long long ll;
ll fact(int i)
{
	if(i==1)
	return 1;
	else return (i*(fact(i-1)%mod))%mod;
}
int main()
{
	int i,j,k=1,count[4096]={0},n,m,t;
	ll ans;
	cin>>t;
	while(t--)
	{
	cin>>n>>m;
	ans=1;
	for(i=1;i<=4096;i++)
		count[i]=0;
	
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			k=i|j;
			count[k]++;
		}
	}
	for(i=1;i<4096;i++)
	{
		if(count[i]>1)
		{
			ans=((ans%mod)*(fact(count[i])%mod))%mod;
		}
	}
	cout<<ans<<endl;
	}
	return 0;
}
