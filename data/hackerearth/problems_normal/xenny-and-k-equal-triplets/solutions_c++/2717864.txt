#include <stdio.h>
#include <iostream>
using namespace std;

long long cal_gcd(long long a,long long b) {
	
	while(b!=0) {
		long long tmp = b;
		b = a%b;
		a = tmp;
	}
	return a;
}

int main()
{
	int test;
	scanf("%d",&test);
    
    while(test--) {
    	int n;
    	long k;
    	scanf("%d %ld",&n,&k);
    	
    	int cnt_k=0;
    	long tmp;
    	for(int i=0; i<n; i++) {
    		scanf("%ld",&tmp);
    		if(tmp==k) cnt_k++;
    	}
    	
    	if(cnt_k==n) {
    		printf("1\n");
    		continue;
    	}
    	
    	long long num = ((long long)(cnt_k) * (cnt_k-1) * (cnt_k-2))/6;
    	long long den = ((long long)(n) *(n-1) * (n-2))/6;
    	
    	long long gcd = cal_gcd(num,den);
    	
    	num/=gcd;
    	den/=gcd;
    	
    	printf("%lld/%lld\n",num,den);
    	
    }
    return 0;
}
