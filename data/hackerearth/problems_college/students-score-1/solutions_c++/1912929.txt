#include<bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	int a[n];
	int maxendinghere[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
		maxendinghere[i]=0;
	}
	
	int j=0;
	int max=0;
	for(int i=0;i<n;i++){
		
		j+=a[i];
		if(j>0){
			maxendinghere[i]=j;
		}
		else{
			j=0;
		}
		if(max<maxendinghere[i]){
			max=maxendinghere[i];
		}
	}
	cout<<max<<endl;
}
