bool Hash[27][27][27][27][27];
int main()
{
    ll n,i,c=0;
    char a[6];
    cin>>n;
    while(n--)
    {
        cin>>a;
        int l=strlen(a);
        switch(l)
        {
            case 1: if(Hash[a[0]-'a'+1][0][0][0][0]==0)
                        c++;
                    Hash[a[0]-'a'+1][0][0][0][0]=1;
                    break;
            case 2: if(Hash[a[0]-'a'+1][a[1]-'a'+1][0][0][0]==0)
                        c++;
                    Hash[a[0]-'a'+1][a[1]-'a'+1][0][0][0]=1;
                    break;
            case 3: if(Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][0][0]==0)
                        c++;
                    Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][0][0]=1;
                    break;
            case 4: if(Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][a[3]-'a'+1][0]==0)
                        c++;
                    Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][a[3]-'a'+1][0]=1;
                    break;
            case 5: if(Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][a[3]-'a'+1][a[4]-'a'+1]==0)
                        c++;
                    Hash[a[0]-'a'+1][a[1]-'a'+1][a[2]-'a'+1][a[3]-'a'+1][a[4]-'a'+1]=1;
                    break;
        }
    }
    cout<<c<<endl;
    for(int v=0;v<27;v++)
        for(int w=0;w<27;w++)
            for(int x=0;x<27;x++)
                for(int y=0;y<27;y++)
                    for(int z=0;z<27;z++)
                        {
                            if(Hash[v][w][x][y][z]==1)
                            {
                                cout<<char(v+'a'-1);
                                if(w)
                                    cout<<char(w+'a'-1);
                                if(x)
                                    cout<<char(x+'a'-1);
                                if(y)
                                    cout<<char(y+'a'-1);
                                if(z)
                                    cout<<char(z+'a'-1);
                                cout<<endl;
                            }
                        }

}
