import sys


def compare(x, y):
    if x[1] > y[1] or (x[1] == y[1] and x[0] < y[0]):
        return -1
    return 1


N = int(sys.stdin.readline())
stud = []
for i in xrange(N):
    line = sys.stdin.readline().split()
    stud.append((line[0], int(line[1])))

stud = sorted(stud, cmp=compare)
# print stud
for tup in stud:
    print tup[0], tup[1]
