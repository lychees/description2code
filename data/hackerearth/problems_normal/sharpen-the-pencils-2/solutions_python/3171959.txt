tc=int(raw_input())
while tc:
    tc-=1
    N=int(raw_input())
    A=map(int,raw_input().split())

    i=0
    j=N-1
    Madhav_length=0
    Riya_length=0
    while (i<=j):
        if Madhav_length <= 2*Riya_length:
            Madhav_length+=A[i]
            i+=1
        else:
            Riya_length+=A[j]
            j-=1
    print i,N-i
