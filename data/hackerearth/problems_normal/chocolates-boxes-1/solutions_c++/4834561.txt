#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,num,i,ans=0,a[100001]={0},check;
		cin>>n;
		while(n--)
		{
			cin>>num;
			a[num]++;
		}
		cin>>check;
		for(i=1;i<=check;i++)
		{
			if(a[i])
			{
				ll req=check-i;
				
				if(req==i)
				{
					ans+=(a[i]*(a[i]-1))/2;
				}
				else
				ans+=(a[i]*a[req]);
				a[req]=0;
				
			}
		}
		cout<<ans<<"\n";
	}
}
