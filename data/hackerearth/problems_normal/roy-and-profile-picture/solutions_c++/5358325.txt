#include <iostream>
using namespace std;

int main()
{
	int l,n,w,h,i;
	cin >> l;
	cin >> n;
	i=1;
	while(i<=n)
	{
	
		cin >> w; 
		cin >> h;
		if(w==h && w>=l )
		 cout << "ACCEPTED"<< endl;
		 else 
		 {
		 	if(w<l || h<l)
		     cout << "UPLOAD ANOTHER"<< endl;
		   else
		    cout << "CROP IT"<< endl;
		 }
	i++;
		
	}
	
    
    return 0;
}
