#include <iostream>
#include <string>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int n = s.length(),k=0;
    for(int i=0;i<n;i++)
        if(s[i]%2==0) k++;
    for(int i=0;i<n;i++)
        if(s[i]%2==0)
        {
            cout<<k<<" ";
            k--;
        }
        else
            cout<<k<<" ";
}
