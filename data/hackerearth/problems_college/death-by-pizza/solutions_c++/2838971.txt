#include<bits/stdc++.h>
using namespace std;
int gec(int a,int b)
{
    int c=1;
    while(a!=0)
    {
        c=b;
        b=a%b;
        if(b==0)
            return c;
        a=c;
    }
}

{
    int q,t,n,i,m,qq,x,y,z;

        cin>>s[i];
    while(q--)
    {
        cin>>x>>y>>z;
        switch (x)
        {
       case 1:
           m=0;
        for(i=y-1;i<z;i++)
            m=m+s[i];
            qq=z-y+1;
            int qp;
        qp=gec(m,qq);
        cout<<m<<"\/"<<qq<<endl;
         break;
        case 0:
          int c=0;
        for(i=y-1;i<z;i++)
        {
            s[i]+=a[c++];

        }
        break;
        }
    }

}
