#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n;
		scanf("%lld",&n);
		long long int i,a[n];
		for( i = 0 ; i <n;  i++)
		{
			scanf("%lld",&a[i]);
		}
		sort(a,a+n);
		long long int sum = 0;
		for( i = 0; i < n-1; i = i+2)
		{
			sum = sum + a[i] * a[i+1];
		}
		cout<<sum<<endl;
	}
	return 0;
}
