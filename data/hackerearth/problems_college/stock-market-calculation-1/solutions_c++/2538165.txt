#include <iostream>
using namespace std;

int max(int a[],int st,int n)
{
	if(st<n-1)
	{int ans=st;
	for(int i=st;i<n;i++)
	if(a[ans]<a[i])ans=i;
	return ans;
	}
	return n;
}
int sumtilli(int a[],int st,int m)
{
	int sum=0;
	for(int i=st;i<m;i++)
	sum+=a[i];
	int k=m-st;
	return (a[m]*k-sum);
}
int q(int a[],int n)
{
	int res=0;
	int st=0;
	int maxi=max(a,st,n);
	//	cout<<"smaxi = "<<maxi<<endl;
	while(maxi<n)
	{
		res+=sumtilli(a,st,maxi);
	//	cout<<res<<" "<<endl;
		st=maxi+1;
	//	cout<<"st = "<<st<<endl;
		maxi=max(a,st,n);
	//	cout<<"smaxi = "<<maxi<<endl;

	}
		
	return res;
	
}
