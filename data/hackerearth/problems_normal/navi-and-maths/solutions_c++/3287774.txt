#include <bits/stdc++.h>
#include <iostream>
using namespace std;
#define mod 1000000007
long long  poow(long long  a, long long  b, long long  MOD) {
long long  x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
 
long long  modInverse(long long a, long long m) {
    return poow(a,m-2,m);
}
int main()
{
	ios::sync_with_stdio(false) ; 
	int t ;
	cin >> t ;
	int k = 1 ;
	while(t--){
		int n ; 
		cin >> n; long long  a[n+1] ; 
		for(int i=0;i<n;i++)cin >> a[i] ; 
	//	long long sum[n+1] ,mul[n+1];
		bool b[n];
		for(int i=0;i<n;i++)b[i] = false ;
		b[0] = true ;
		long long ans = 0 ; 
		for(long long  i=0;i<pow(2,n);i++){
			long long sum =0 ; 
			long long mul = 1; 
				for(int j=0;j<n;j++){
					if(b[j]){
						sum += a[j] ; 
						mul *= a[j] ; 
						mul %= mod ; 
					}
				}
			//	cout << sum << " " << mul << endl;
				if(sum ==0 )continue ; 
				mul*= modInverse(sum,mod) ; 
				mul %= mod; 
				ans = max(mul,ans) ; 
				for(int j=0;j<n;j++){
					if(b[j])b[j] = false; 
					else {
						b[j] = true ;
						break ; 
					}
				}
		}
		cout <<"Case #"<<k<<": "<< ans << endl;k++;
}
    return 0;
}
