#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	string s;
    	cin>>s;
    		long long mul=1LL;
    	int i,len=s.length();
    	for(i=0;i<s.length();i++)
    	{
    		if(s[i]==s[len-i-1])
    	;
    		else
    		break;
    		mul*=(s[i]-96);
    	}
    	if(i==s.length())
    	cout<<"Palindrome\n";
    	else
    	{
    	
    		for(int j=i;j<s.length();j++)
    		mul*=(s[j]-96);
    		cout<<mul<<"\n";
    	}
    }
    return 0;
}
