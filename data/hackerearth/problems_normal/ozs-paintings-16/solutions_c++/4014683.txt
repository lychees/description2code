#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int x,y,z,ans=0;
		cin>>x>>y>>z;
		for(int i=0;i<3;i++){
			int xrem = x-i;
			int yrem = y-i;
			int zrem = z-i;
			if(xrem<0 || yrem<0 || zrem<0)	continue;
			ans = max(ans,i+xrem/3+yrem/3+zrem/3);
		}
		cout<<ans<<endl;
	}

    return 0;
}
