#include<iostream>
#include<cstring>
using namespace std;
int arr[27];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		for(int i=0;i<26;i++)
            arr[i]=0;
		int c=0;
		int l=str.length();
		for(int i=0;i<l;i++)
		{
			if(arr[str[i]-97]==0)
                c++;
			arr[str[i]-97]++;
		}
		if(l<4)
		{
			cout<<"NO"<<endl;
			continue;
		}
		else if(l==4)
		{
			if(c==4)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
		else if(l==5)
		{
			if(c>=3)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
		else if(l==6)
		{
			if(c<2)
			cout<<"NO"<<endl;
			else if(c==2)
			{
				int flag=0;
				for(int i=0;i<26;i++)
				{
					if(arr[i]==3)
					{
						flag=1;
						break;
					}
				}
				if(flag==1)
				cout<<"YES"<<endl;
				else
				cout<<"NO"<<endl;
			}
			else
			cout<<"YES"<<endl;
		}
		else if(l<10)
		{
			if(c>=2)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}
		else
		cout<<"YES"<<endl;
	}
}
