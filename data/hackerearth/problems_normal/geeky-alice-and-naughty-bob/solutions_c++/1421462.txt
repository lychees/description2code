#include<iostream>
using namespace std;
int sum[1000006];
int main()
{
    int i,a,b,t;
    sum[0]=0;
    sum[1]=1;
    sum[2]=3;
    sum[3]=9;
    sum[4]=15;
    sum[5]=18;
    for(i=6;i<1000006;i++)
    {
        sum[i]=sum[i-1]+9;
    }
    cin>>t;
    while(t--)
    {
    cin>>a>>b;
    if(a!=0)
    {
    cout<<sum[b]-sum[a-1]<<endl;
    }
    else
    cout<<sum[b]-sum[a]<<endl;
    }
}
