#include <iostream>
using namespace std;

bool solved[10000000];


int main()
{
	int t, n, s, r, i;
	cin>>t;
		while(t)
	{
		cin>>n;
		int streak = 0, maxStreak = 0;
		
		for(i=0;i<1000000;i++)
			solved[i] = false;
		while(n)
		{
			cin>>s>>r;

			if(solved[s] == false)
			{
				if(r==1)
				{
					solved[s] = true;
					streak++;
				}
				else
				{
				
					streak = 0;
				}
				if(streak>maxStreak)
				{
					maxStreak = streak;
				}
			}
			else
			{
				if(r == 0)
				{
					streak = 0;
				}
			}
			
			n--;
		}
		cout<<maxStreak<<endl;
		t--;
	}
	return 0;
}
