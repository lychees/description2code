#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,p,x,y,u,v;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);

        if(x==0||y==0)
        {
            if(abs(u-v)%2!=0)
            {
                printf("Terrorists Win !\n");
            }
            else
            {
                int ans;
                if(x==u&&y==v) ans=0;
                else if(x==u||y==v) ans=1;
                else if(u-x==v-y) ans=1;
                else ans=2;
                if(ans<p) printf("Counter-Terrorists Win !\n");
                else printf("Terrorists Win !\n");
            }
        }
        else
        {
            if(abs(x-y)%2!=0)
        {
            if(abs(u-v)%2==0)
            {
                printf("Terrorists Win !\n");
            }
            else
            {
                int ans;

                if(x==u&&y==v) ans=0;
                else if(x==u||y==v) ans=1;
                else if(u%x==v%y) ans=1;
                else ans=2;
                if(ans<p) printf("Counter-Terrorists Win !\n");
                else printf("Terrorists Win !\n");
            }
        }
        else
        {
            if(abs(u-v)%2!=0)
            {
                printf("Terrorists Win !\n");
            }
            else
            {
                int ans;
                if(x==u&&y==v) ans=0;
                else if(x==u||y==v) ans=1;
                else if(u%x==v%y) ans=1;
                else ans=2;
                if(ans<p) printf("Counter-Terrorists Win !\n");
                else printf("Terrorists Win !\n");
            }
        }
        }
    }
    return 0;
}
