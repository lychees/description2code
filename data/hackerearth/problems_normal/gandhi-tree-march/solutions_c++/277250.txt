string s;
int n,pos;
vector<char> l[10001],r[10001];
void build(int c){
	if(pos>=n)
	return;
	if(s[pos]!='.'){
		if(c>=0)
		r[c].pb(s[pos]);
		else
		l[-c].pb(s[pos]);
		pos+=2;
		build(c-1);
		build(c+1);
	}
	pos++;
	return;
}
int main(){
	int t,c,i;
	cin>>t;
	while(t--){
		cin>>c>>s;
		n=s.length();
		for(i=0;i<=n;i++){
			l[i].clear();
			r[i].clear();
		}
		pos=0;
		build(0);
		if(c>=0){
			if(r[c].empty())
			cout<<"Common Gandhijee!"<<endl;
			else{
				sort(all(r[c]));
				for(i=0;i<r[c].size();i++)
				cout<<r[c][i];
				cout<<endl;
			}
		}
		else{
			if(l[-c].empty())
			cout<<"Common Gandhijee!"<<endl;
			else{
				sort(all(l[-c]));
				for(i=0;i<l[-c].size();i++)
				cout<<l[-c][i];
				cout<<endl;
			}
		}
	}
	return 0;
}
