#include<bits/stdc++.h>
using namespace std;
#define lli long long int
lli rev(lli n){
	lli temp=0;//=n;
	while(n>0){
		temp=temp*10+n%10;
		n/=10;
	}
	return temp;
}
int main(){
	int t;
	lli a,b;
	cin>>t;
	while(t--){
		scanf("%lld%lld",&a,&b);//cin>>a>>b;
		//cout<<rev(rev(a)+rev(b))<<endl;
		printf("%lld\n",rev(rev(a)+rev(b)));
	}
	
	return 0;
}
