#include<bits/stdc++.h>
using namespace std;

int main(){
	long long int t; cin>>t;
	while(t--){
		long long int n; cin>>n;
		long long int *a = new long long int[n];
		bool possible = false;
		for(long long int i=0;i<n;i++){
			cin>>a[i];
			if(a[i]==-1)
				possible = true;
		}
		
		if(possible){
			long long int temp = a[0];
			long long int flags[n];
			memset(flags,0,sizeof(flags));
			flags[0] = 1;
			while(temp!=-1 && flags[temp]==0){
				//cout<<"TEMP : "<<temp<<endl;
				cout.flush();
				flags[temp]=1;
				temp = a[temp];
				if(flags[temp]==1){
					cout<<"Lose";
					goto END;
				}
				//else{
				//	flags[temp]=1;
				//	cout<<"IN HERE: "<<temp<<endl;
				//	cout.flush();
				//}
			}
			cout<<"Win";
		}
		else{
			cout<<"Lose";
		}
		END:
			cout<<endl;
	}

	return 0;
}
