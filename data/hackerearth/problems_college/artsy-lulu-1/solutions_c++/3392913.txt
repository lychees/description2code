bool print(char c, int h) {
	int i, j;
	
	if (h < 2) return 0;
	if (islower(c) && c - 'a' + h > 26) return 0;
	if (isupper(c) && c - 'A' + h > 26) return 0;
	for (i = 1; i <= h; i++) {
		for (j = 1; j <= i; j++) printf("%c ", c);
		puts("");
		c++;
	}
	return 1;
}
