#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define maxn 10005
vector<int> g[maxn],size(maxn,0),mark(maxn,0);
int ans,k,n,x;
void dfs(int v,int p){
	size[v]=1;
	for(auto u:g[v]){
		if(u!=p){
			dfs(u,v);
			if(mark[u]==0)size[v]+=size[u];
		}
	}
	
	if(size[v]>=k){
		mark[v]=1;
		ans++;
	}
	
}

int main()
{
    cin>>k>>n;
    for(int i=2;i<=n;i++){
    	cin>>x;
    	g[x].pb(i);
    	g[i].pb(x);	
    }
    
    dfs(1,0);
    
    cout<<ans<<endl;
    
    return 0;
}
