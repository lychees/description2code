#include <stdio.h>
#include <math.h>
#include <iostream>

using namespace std;


static int a[1024][1024];

void printMatrix(int n)
{
	a[0][0] = 1;
	a[0][1] = 2;
	a[1][0] = 3;
	a[1][1] = 4;


	for (int i = 2; i <= n; i++)
	{
		int p = pow(2.0, i - 1);
		int s = p*p;

		for (int i = 0; i < p; i++)
		{
			for (int j = 0; j < p; j++)
			{
				a[i][j+p] = a[i][j] + s;
				a[i + p][j] = a[i][j] + 2 * s;
				a[i + p][j + p] = a[i][j] + 3 * s;
			}
		}
	}
}


int main()
{
	int n;
	cin >> n;

	printMatrix(n);

	int p = pow(2.0, n);

	for (int i = 0; i < p; i++)
	{
		for (int j = 0; j < p; j++)
		{
			printf("%d ", a[i][j]);
		}

		printf("\n");
	}
}
