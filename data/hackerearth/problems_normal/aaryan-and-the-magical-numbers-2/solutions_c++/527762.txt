#include<cstdio>
using namespace std;
long long int arr[30];
long long int countf(long long int n)
{
    if(n<2)
        return n;
    if(n==2||n==3)
      return 1;
    if(n==4||n==5)
      return 2;
    if(n==6||n==7)
      return 3;
    if(n>7&&n<10)
       return n-4;
    long long int po=1;
    long long int tmp=n;
    int len=0;
    while(true)
    {
       tmp/=10;
       po=po*10;
       len++;
       if(tmp-10<0)
         break;
    }
    long long int msd=tmp;
    if(msd!=2&&msd!=3&&msd!=5&&msd!=7)
      return countf(msd)*arr[len]+countf(msd)+countf(n%po);
    else
      return countf(msd*po-1);
}
