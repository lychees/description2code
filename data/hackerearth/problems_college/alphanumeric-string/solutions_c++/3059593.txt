#include <bits/stdc++.h>
using namespace std;
vector<string> v,v2;
void solve()
{
	v.clear();v2.clear();
	string str;
		string str2="";
		getline(cin,str);	
		for(int i=0;i<str.size();i++)
		{
			if(str[i]==' ')
			{
				v.push_back(str2);
				str2="";
			}
			else
			{
				str2=str2+str[i];
			}
		}
		if(str2.size()>0)
			v.push_back(str2);
		str2="";
		int temp=0;
		for(int i=0;i<v.size();i++)
		{
			str2="";
			for(int j=0;j<v[i].size();j++)
			{
				if(v[i][j]>='0'&&v[i][j]<='9')
				{
					temp=temp+v[i][j]-'0';	
				}
				else
				{
					str2=str2+v[i][j];
				}
			}
			v2.push_back(str2);
		}
		sort(v2.begin(),v2.end());
		for(int i=0;i<v2.size();i++)
		{
			temp=temp-v2[i].size();	
			int temp2=(temp>0)?(temp):(-temp);
			cout<<v2[i]<<" "<<temp2<<" ";
		}
}
int main()
{
	int t;
	scanf("%d ",&t);
	while(t--)
	{
		solve();
		cout<<"\n";
	}
    return 0;
}
