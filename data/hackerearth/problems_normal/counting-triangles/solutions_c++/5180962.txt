#include <iostream>
#include <map>
#include <array>
#include <algorithm>
#include <cstdio>

using namespace std;
int main(){
    long n;
    array<long long, 3> arr;
    map<array<long long , 3>, long> mapp;
    scanf("%ld",&n);
    long cnt=0;
    while(n--){
        scanf("%lld %lld %lld", &arr[0],&arr[1],&arr[2]);
        sort(arr.begin(),arr.end());
        mapp[arr]++;
    }
    for(auto i = mapp.begin();i!=mapp.end();i++){
        if(i->second==1L){
            cnt++;
        }
    }
    printf("%ld",cnt);
}
