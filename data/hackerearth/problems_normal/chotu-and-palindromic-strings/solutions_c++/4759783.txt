#include <iostream>
using namespace std;

string str1,str2;

bool check(string str)
{
	int i,n=str.size();
	for(i=0;i<n;i++)
		if(str[i]!=str[n-i-1])
			return false;
	return true;		
}

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	cin>>str1>>str2;
    	if(check(str1+str2) || check(str2+str1))
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
