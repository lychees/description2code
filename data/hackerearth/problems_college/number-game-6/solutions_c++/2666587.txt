#include<bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	map<long long,long long> pos;
	long long sum = 0,i=1.,t,n;
	vector<unsigned long long> v;
	while(sum<=10e17)
	{
		sum+=i*(i+1)*(i+1);
		v.push_back(sum);
		pos[sum] = i;
		i++;
	}
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long ans = pos[*upper_bound(v.begin(),v.end(),n)]-1;
		cout<<ans<<"\n";
		if(ans == 0)
			cout<<"Draw\n";
		else if(ans%2 == 0)
			cout<<"Ketan\n";
		else
			cout<<"Dhruv\n";
	}
	return 0;
}
