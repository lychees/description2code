#include<bits/stdc++.h>
#define pii pair<int,int>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll unsigned long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%lld",&(n))
#define pf(n) printf("%lld\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int dx[]={0,1,0,-1};
int dy[]={1,0,-1,0};
int co=0,n,m,q;
int arr[1005][1005];
void dfs(int i,int j)
{
    arr[i][j]=-1;
    co--;
    for(int k=0;k<4;k++)
    {
        int a=i+dx[k];
        int b=j+dy[k];
        if(a>=0 && a<n && b>=0 && b<m && arr[a][b]==1)
            dfs(a,b);
    }
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cin>>n>>m>>q;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            cin>>arr[i][j];
            if(arr[i][j]==1)
                co++;
        }
    }

    while(q--)
    {
        int a,b;
        cin>>a>>b;
        a--,b--;
        if(arr[a][b]==-1)
            cout<<co<<"\n";
        else if(arr[a][b]==0)
            cout<<co<<"\n";
        else
        {
        dfs(a,b);
        cout<<co<<"\n";
        }
    }

}
