#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long n,a[10005],k[10005],count=0;
    	cin>>n;
    	for(long long i=0;i<n;i++)
    	cin>>a[i];
    	for(long long i=0;i<n-1;i++)
    	{
		k[i]=a[i+1]-a[i];
		if(k[i]<1)
			count++;
	}
	cout<<count<<endl;
	
    }
}
