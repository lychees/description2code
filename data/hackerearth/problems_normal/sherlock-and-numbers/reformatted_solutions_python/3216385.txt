for _ in range(int(raw_input())):
    n, k, p = map(int, raw_input().split())
    a = sorted(map(int, raw_input().split()))
    if p < a[0]:
        ans = p
    elif p > n - k:
        ans = -1
    else:
        ans = p
        for i in a:
            if i <= ans:
                ans += 1
            else:
                break
    print ans
