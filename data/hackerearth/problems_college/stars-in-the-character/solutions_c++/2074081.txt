#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
long i,t,n,ln;
char a[1000000];
scanf("%ld",&t);
while(t--)
{
   long int count=0,countx=0;;
   scanf("%s",a);
   for(i=0; a[i]!='\0'; i++)
    {
        if( a[i]=='A' || a[i]=='R'|| a[i]=='P' || a[i]=='O'
        || a[i]=='D' || a[i]=='Q')
         count++;
        else if(a[i]=='B')
           count+=2;
         a[i]+=32;
         if(a[i]=='a' || a[i]=='b' || a[i]=='d' ||
         a[i]=='e' || a[i]=='o' || a[i]=='g' || a[i]=='p' ||
         a[i]=='q')
           countx++;
    }
   if(count-countx==-5)
     cout<<-3<<endl;
    else
   printf("%ld\n",count-countx);
}
return 0;
}
