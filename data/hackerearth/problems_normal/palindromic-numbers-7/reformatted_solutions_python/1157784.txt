T = int(raw_input())
while T:
    count = 0
    A, B = raw_input().split()
    A, B = int(A), int(B)
    for i in range(A, B+1):
        reverse = 0
        temp = i
        while (temp > 0):
            x = temp % 10
            temp = temp/10
            reverse = reverse*10+x
        if (reverse == i):
            count += 1
    print(count)
    T -= 1
