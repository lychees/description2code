#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[110];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,k;
	cin>>n>>k;
	ll int i,mc=1,cur=0,ans=0;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n,greater<ll int>());
	for(i=0;i<n;i++)
	{
		ans+=a[i]*mc;
		cur=(cur+1)%k;
		if(cur==0)
		mc++;
	}
	cout<<ans<<endl;
	return 0;
}
