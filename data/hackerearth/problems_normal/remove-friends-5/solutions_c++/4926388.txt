#include <iostream>
#include <list>
using namespace std;

int main()
{
    int t,n,k,temp;
    cin>>t;
    while(t--)
    {
        list<int>a;
        cin>>n;
        cin>>k;
        for(int i=0;i<n;++i)
        {
            cin>>temp;
            a.push_back(temp);
        }
        int i=0;
        list<int>::iterator iterator=a.begin();
        list<int>::iterator j=a.begin();
        j++;
        while(i!=k&&iterator!=--a.end())
        {
            if(*iterator<*(j))
            {
            	i++;
                a.erase(iterator--);
                while(i!=k&&iterator!=--a.begin()&&(*iterator<*j))
                {
                	a.erase(iterator--);
                	i++;
                }
  				iterator=j;
                j++;
            }
            else
            {
                ++iterator;
                j++;
            }
        }
        while(i!=k)
        {
            a.pop_back();
            i++;
        }
        for(iterator=a.begin();iterator!=a.end();++iterator)
        {
            cout<<*iterator<<" ";
        }
        cout<<endl;
 
    }
    return 0;
}
