#include <iostream>

#include <string>

using namespace std;

int misses(const string& S, char c) {
   int res = 0;
   for (int i = 0; i < (int) S.size(); i += 2)
      if (S[i] != c)
         ++res;
   return res;
}

int main(int argc, char* argv[]) {
   ios_base::sync_with_stdio(false); 
   cin.tie(NULL);

   int TC;
   cin >> TC;
   for (int tc = 1; tc <= TC; ++tc) {
      int N;
      cin >> N;
      string S;
      cin >> S;
      int res = min( misses(S, 'B'), misses(S, 'G') );
      cout << res << '\n';
   }

   return 0;
}
