#include <iostream>
using namespace std;

int main()
{
	unsigned int t;
	string s;
	cin>>t;
	for(unsigned int i=0;i<t;i++){
		cin>>s;
		char current_pos = 'a';
		for(unsigned int j=0;j<s.length();j++){
		
			int clockwise = ( ('z' - current_pos) + (s[j] - 'a') + 1) % 26;
			int anticlockwise = ( ('z' - s[j]) + (current_pos - 'a') + 1) % 26;

			if(clockwise > anticlockwise){
				cout<<-anticlockwise<<" ";
			}else{
				cout<<clockwise<<" ";
			}
			current_pos  = s[j];
		}
		cout<<endl;
	}
	return 0;
}
