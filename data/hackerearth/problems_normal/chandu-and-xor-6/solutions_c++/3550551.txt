#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m;
    cin>>m>>n;
    int a[m][n];
    for(int i=0;i<m;i++)
    {
    	for(int j=0;j<n;j++)
    	cin>>a[i][j];
    }
    int row[m],col[n];
    for(int i=0;i<m;i++)
    {
    	row[i]=0;
    	for(int j=0;j<n;j++)
    	row[i]+=a[i][j];
    }
    for(int j=0;j<n;j++)
    {
    	col[j]=0;
    	for(int i=0;i<m;i++)
    	col[j]+=a[i][j];
    }
    int ans=0;
    for(int i=0;i<m;i++)
    {
    	for(int j=0;j<n;j++)
    	ans=max(ans,row[i]^col[j]);
    }
    cout<<ans;
    return 0;
}
