#include <bits/stdc++.h>
using namespace std;
long long ar[1000005],mx=1000000;
void seive(){
    ar[0]=0;
    ar[1]=2;
    ar[2]=3;
    for(int i=3;i<=mx;i++)
    ar[i]=ar[i-1]+ar[i-2];
}
int main() {
	seive();
	int t;
	cin>>t;
	while(t--){
	    long long n;
	    cin>>n;
	    cout<<ar[n]*ar[n]<<endl;
	}
	return 0;
}
