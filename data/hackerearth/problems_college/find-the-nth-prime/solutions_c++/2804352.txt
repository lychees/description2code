#include <bits/stdc++.h>
using namespace std;

void markMultiples (bool arr[], int a, int n) {
    int num;
    for (int i = 2; (num  = i * a) <= n; i++)
        arr[num-1] = 1;
}

void sieve(int n, int data[]) {
	int pos = 0;
    if (n >= 2) {
        bool arr[n];
        memset(arr, 0, sizeof(arr));
        for (long long int i = 1; i < n; i++) {
            if (arr[i] == 0) {
                data[pos] = i+1;
                pos++;
                markMultiples(arr, i+1, n);
            }
        }
    }
}

int main () {
    int data[1100];
    memset(data, 0, sizeof(data));
    sieve(8000, data);
    int n;
    scanf("%d", &n);
    printf("%d", data[n-1]);
}
