#include <iostream>
#include <stdio.h>
#include <algorithm>

using namespace std;

const int MAX = 1e7+2;

int prime[MAX];
int phi[MAX];
long long ans[MAX];

void primes()
{
	fill(prime, prime+MAX, 0);
	
	prime[1] = 1;
	for(int i = 2; i<MAX; i++)
	{
		if(prime[i] == 0)
		{
			for(int j = i+i; j <MAX; j+=i)
				prime[j] = i;
		}
	}
}

void phii()
{ 
	phi[1] = 1;
	for(int i = 2; i < MAX; i++)
	{
		int k = i;
		int v = prime[k];
		if(v == 0)
			phi[i] = k-1;
		else
		{
			while(k % v == 0) 
				k /= v;
				
			phi[i] = phi[k] * (i/k)/v*(v-1);
		}
	}
}
int main()
{
	primes();
	phii();
	
	ans[0] = 0;
	for(int i = 1; i < MAX; i++)
		ans[i] = ans[i-1]+phi[i];
	
	int t, n;
	
	scanf("%d", &t);
	
	while(t--)
	{
		scanf("%d", &n);
		printf("%lld\n", ans[n]);
	}
	
    return 0;
}
