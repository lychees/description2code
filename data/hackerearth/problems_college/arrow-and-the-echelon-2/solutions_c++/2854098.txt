#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	ll t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
	
		ll no3=-1;
		ll no5=-1;
		for(i=0;i<=n;i++)
		{
			if((n-i)%3==0 && i%5==0)
			{
				no3=i;
				no5=n-i;
				break;
			}
		}
		if(no3==-1 || no5==-1)
		{
			cout<<"-1"<<endl;
		}
		else
		{
			for(i=0;i<no5;i++)
			{
				cout<<"5";
			}
			for(i=0;i<no3;i++)
			{
				cout<<"3";;
			}
			cout<<endl;
		}
	}
}
