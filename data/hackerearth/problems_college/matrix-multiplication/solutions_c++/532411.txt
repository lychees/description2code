LL mod = 1000000007;

LL BigMod(LL base,LL exp)
{
      if(exp == 0) return 1;
      LL x = BigMod(base,exp/2);
      x = (x*x)%mod;
      if(exp%2) x = (x*base)%mod;
      return x;
}
