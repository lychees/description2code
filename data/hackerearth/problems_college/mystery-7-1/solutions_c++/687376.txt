#include <iostream>
using namespace std;

int main()
{
    int t, i;
    char str[100];
    scanf("%d", &t);
    while(t>0)
    {
    	scanf("%s", &str);
    	for(i=0; str[i]!='\0'; i++)
    	{
    		if((int)str[i]==115)
    		printf("7");
    		else if((int)str[i]==118)
    		printf("8");
    		else if((int)str[i]==121 || (int)str[i]==122)
    		printf("9");
    		else
    		printf("%d", ((int)str[i]-91)/3);
    	}
    		printf("\n");
    	t--;
    }
    return 0;
}
