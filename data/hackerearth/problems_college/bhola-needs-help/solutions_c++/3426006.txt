#include <iostream>
#include <bits/stdc++.h>
#include <algorithm>
#include<stdio.h>
using namespace std;
vector<long long int>v;
int main()
{
	long long int a,b,i,k;
	scanf("%lld %lld",&a,&b);
	k=__gcd(a,b);
	for(i=1;i*i<=k;i++)
	{
		if(k%i==0)
		{
			v.push_back(i);
			if(i*i!=k)
				v.push_back(k/i);
		}
	}
	sort(v.begin(),v.end());
	int q;
	long long int l,r;
	scanf("%d",&q);
	while(q--)
	{
		scanf("%lld %lld",&l,&r);
		i=upper_bound(v.begin(),v.end(),r)-v.begin();
		if(i!=0)
		{
			if(v[i-1]>=l && v[i-1]<=r)
				printf("%lld\n",v[i-1]);
			else
				printf("-1\n");
		}
		else
			printf("-1\n");
	}
	return 0;
}
