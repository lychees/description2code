#include <iostream>
#include <stdio.h>
#include <string>

using namespace std;

int main()
{
	
	int t, n, k, i, d;
	char ch;
	
	cin >> t;
	
	while( t-- ) {
		
		string s;
		cin >> s;
		
		n = s.length();
		
		for(i = 0, k = 0;i < n;i++, k++ ) {
			d = s[ i ] - 'a';
			d = ( d + k ) % 26;
			ch = (char)( 'a' + d );
			cout << ch;
		}
		cout << endl;
	}
	return 0;
}
