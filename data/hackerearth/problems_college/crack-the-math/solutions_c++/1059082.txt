#include <iostream>
#include <string>
using namespace std;

int main()
{
    string a, b;
    int t;
    cin >> t;
    while (t--)
    {
    	cin >> a;
    	cin >> b;
    	for (int i = 0; i < a.length(); ++i)
    	{
    		if (a[i] == b[i]) cout << 0;
    		else cout << 1;
    	}
    	cout << endl;
    }
    
    return 0;
}
