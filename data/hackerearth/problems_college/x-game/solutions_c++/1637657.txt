#include <iostream>
using namespace std;

#define mod 1000000007
#define ll long long

ll power(ll a,ll x){

	if(!x)
		return 1;
	ll temp = power(a,x/2);
	temp=(temp*temp)%mod;
	if(x%2==0)
		return temp;
	else
		return (a*temp)%mod;
}

int main(){
	ll test;
	cin>>test;
	while(test--){
		ll n;
		cin>>n;
		ll val,result=0;
		for(ll i=0;i<n;i++){
			cin>>val;
			result |= val;
		}
		result= (result*power(2,n-1))%mod;
		cout<<result<<endl;
	}
}
