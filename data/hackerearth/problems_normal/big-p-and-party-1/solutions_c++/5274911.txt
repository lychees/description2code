#include<bits/stdc++.h>
using namespace std;
struct node{
	int x,y;
};
int a,b;
vector <int> v[1001];
int lucky[1001];
bool vis[1001];
int main()
{
	cin>>a>>b;
	while(b--)
	{
		int c,d;
		cin>>c>>d;
		v[c].push_back(d);
		v[d].push_back(c);
	}
	memset(lucky,-1,sizeof(lucky));
	lucky[0]=0;
	memset(vis,false,sizeof(vis));
	vis[0]=true;
	queue <node> q;
	node temp;
	temp.x=0,temp.y=0;
	q.push(temp);
	while(!q.empty())
	{
		node temp1=q.front();
		lucky[temp1.x]=temp1.y;
		q.pop();
		for(int i=0;i<v[temp1.x].size();i++)
		{
			if(vis[v[temp1.x][i]]==false)
			{
				vis[v[temp1.x][i]]=true;
				node temp2;
				temp2.x=v[temp1.x][i],temp2.y=temp1.y+1;
				q.push(temp2);
			}
		}
	}
	for(int i=1;i<a;i++)
	cout<<lucky[i]<<endl;
}
