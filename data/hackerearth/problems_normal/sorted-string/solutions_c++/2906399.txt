#include <bits/stdc++.h>
using namespace std;
vector<int> v(26,0);
bool cmp(char c,char d){
	if(v[c-'a']==v[d-'a']){
		return c<d;
	}
	
		return v[c-'a']<v[d-'a'];
	
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		for(int i=0;i<26;i++){
			v[i]=0;
		}
		for(int i=0;i<s.length();i++){
			v[s[i]-'a']++;
		}
		sort(s.begin(),s.end(),cmp);
		cout<<s<<endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
