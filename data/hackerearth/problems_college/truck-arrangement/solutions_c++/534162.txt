#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
	pair<int ,int> p[10002];
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int i;
		for(i=0;i<n;i++)
		{
			scanf("%d",&p[i].first);
			p[i].second =i;
		}
		sort(p,p+n);
		int ans1,ans=0;
		for(i=0;i<n;i++)
		{
			
			ans1=p[i].second -i;
			if(ans1<0)
			ans1=-ans1;
			if(ans1>ans)
			ans=ans1;
			
		}
		printf("%d\n",ans);
	}
	return 0;
}
