#include <bits/stdc++.h>
using namespace std;
int a[100100]={0};
map<int,int> mp;
class todo
{
	int dolo;
	int polo;
	public:
	todo (int d=1, int p=1): dolo(d), polo(p){}
	friend bool operator <(const todo& a, const todo& b)
	{
		if(a.polo<b.polo)
			return true;
		else if(a.polo==b.polo)
			{
				if(a.dolo<b.dolo)
					return true;
				else
					return false;
			}
		return false;
	}
	friend ostream& operator <<(ostream& os, const todo& a)
	{
		os<<a.dolo<<" "<<a.polo<<endl;
	}
};

int main()
{
	priority_queue<todo> todol;
	int n,m,k=0;
	cin>>n>>m;
	//int b[m];

	for(int i=0;i<n;i++)
	{
		int q;
		cin>>q;
		if(mp.count(q)==0)
		{
			mp[q]=k;
			a[k]++;
			k++;
		}
		else
		{
			a[mp[q]]++;
		}
	//	cout<<q<<"HIII"<<b[q]<<endl;
		todol.push(todo(q,a[mp[q]]));
		cout<<todol.top();
	}
    return 0;
}
