#include<iostream>
using namespace std;
int main(){
	int n,rem1,rem2;
	cin>>n;
	long int a1[n],a2[n];
	long int count=0;
	
	for(int i=0;i<n;i++)
	cin>>a1[i]>>a2[i];
	for(int i=0;i<n;i++)
	{	count=0;
		
		while(a1[i]>0 || a2[i]>0)
		{
			rem1=a1[i]%10;
			rem2=a2[i]%10;
			if(rem1+rem2>9)
			count=count+1;
			a1[i]=a1[i]/10;
			a2[i]=a2[i]/10;
		}
		if(count==0)
		cout<<"No carry operation"<<endl;
		if(count==1)
		cout<<"1 carry operation"<<endl;
		if(count>1)
		cout<<count<<" carry operations"<<endl;
		
	}
	return 0;
}
