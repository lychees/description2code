#include <iostream>
#include <string>
using namespace std;

int main()
{
   string arr[100];
   int n;
   cin>>n;
   char ch;
   int length;
   string temp;
   for(int i=0;i<n;i++)
   {
   	cin>>arr[i];
   	temp.clear();
   	
   	for(int j=arr[i].size()-1;j>=0;j--)
   	temp+=arr[i][j];
   	
   	
   	for(int j=0;j<i;j++)
   	{
   		//cout<<arr[j].size()<<" "<<temp.size()<<"\n";
   		if(arr[j].compare(temp)==0)
   		{
   			length=arr[j].size();
   			ch=arr[j][length/2];
   			break;
   		}
   	}
   }
   
   cout<<length<<" "<<ch;
    return 0;
}
