using namespace std;
int V;
int G[300][300] ;
bool isBicolourable( int src)
{

    int col[V];
    for (int i = 0; i < V; ++i)
        col[i] = -1;


    col[src] = 1;


    queue <int> q;
    q.push(src);


    while (!q.empty())
    {

        int u = q.front();
        q.pop();


        for (int v = 0; v < V; ++v)
        {

            if (G[u][v] && col[v] == -1)
            {

                col[v] = 1 - col[u];
                q.push(v);
            }


            else if (G[u][v] && col[v] == col[u])
                return false;
        }
    }

    return true;
}


int main()
{
    int t,l,r,n,e,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        V=n;
        for(i=0;i<=n;i++)
            for(j=0;j<=n;j++)
                G[i][j]=0;
        cin>>e;
        for(i=0;i<e;i++)
        {
            cin>>l>>r;
            G[l][r]=G[r][l]=1;
        }
        if(isBicolourable(0))
        {
            cout<<"BICOLORABLE.\n";
        }
        else cout<<"NOT BICOLORABLE.\n";
    }

    return 0;
}
