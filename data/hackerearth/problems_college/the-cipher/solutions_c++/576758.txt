#include<bits/stdc++.h>
using namespace std;


char convert(int n)
{
    char ch=255;
	int mult=1;
	while(n)
	{
		ch-=(mult*(n&1));
		n/=2;
		mult*=2;
	}
	return ch; 
}
int main()
{
	int t;
	scanf(" %d",&t);
	assert(t>=1 && t<=1000);
	char ch;
	while(t--)
	{
		int n,num;
		scanf(" %d",&n);
		assert(n>=1 && n<=10000);
		while(n--)
		{
        		scanf(" %d",&num);
			ch=convert(num);
			printf("%c",ch);
		}
		printf("\n");
	}
	return 0; 
}
