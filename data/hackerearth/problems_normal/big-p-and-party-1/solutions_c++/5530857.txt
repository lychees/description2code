#include <bits/stdc++.h>
 
using namespace std;
 
vector<int> graph[112345];
int n, m, a, b, c,x, visited[112345],parent[112345];
 
int main () {
	int dis[112345]={-1};
	cin >> n >> m;
	for(int i = 0; i < m; i += 1) {
		cin >> a >> b;
		graph[a].push_back(b);
		graph[b].push_back(a);
	}
	queue<int> q;
	q.push(0);
	parent[0] = 0;
	dis[0] = 0;
	 while(!q.empty())
    {
        int u=q.front();
        q.pop();
        for(int i=0;i<graph[u].size();i++)
        {
            int v= graph[u][i];
            if(!visited[v])
            {
                visited[v]=1;
                dis[v]=dis[u]+1;
                q.push(v);
            }
        }
    }
	 
	for(int i=1;i<n;i++)
		cout<<dis[i]<<endl;
	
}
