#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>
using namespace std;

int main()
{
    int n;
    float p;
    long long *h, h1;
    scanf("%d",&n);
   	h = new long long[n];
    for(int i = 0; i<n; ++i)
    	scanf("%lld",&h[i]);
    sort(h,h+n);
    for(int i = 0; i<n-1; ++i)
    {
    	h1 = h[i+1] / h[i];
    	p = log2(h1);
    	if(floor(p) != p || h[i+1] % h[i] != 0 )
    	{
    		printf("NO");
    		return 0;
    	}
    }
    printf("YES");
    return 0;
}
