#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	int n,k;
    	scanf("%d%d",&n,&k);
    	int cnt = 0;
    	for(int i = 0; i < n; i++) {
    		int x;
    		scanf("%d",&x);
    		cnt += x == k;
    	}
    	
    	long long num = cnt * 1LL * (cnt - 1) * 1LL * (cnt - 2);
    	long long den = n * 1LL * (n - 1) * 1LL * (n - 2);
    	
    	long long g = __gcd(num, den);
    	printf("%lld/%lld\n",num/g, den/g);
    }
    return 0;
}
