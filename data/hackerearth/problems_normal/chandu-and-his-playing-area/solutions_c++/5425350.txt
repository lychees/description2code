#include <bits/stdc++.h>

using namespace std;

int main()
{

    long long t , n;

    cin >> t;

    while(t--){

        cin >> n;

        long long len = n / 4;                  // length
        long long width = len + (n % 4) / 2;    // width
        cout << len * width << "\n";
    }

    return 0;
}
