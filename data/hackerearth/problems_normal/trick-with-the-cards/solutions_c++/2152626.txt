#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	long long a;
    	cin>>a;
    	if(a == 0) {
    		cout<<"0\n";
    	} else if(a<3) {
    		cout<<"1\n";
    	} else if(a%3==0) {
    		cout<<a/3<<"\n";
    	} else {
    		cout<<a<<"\n";
    	}
    }
    return 0;
}
