#include <iostream>
using namespace std ;
#include <string.h>
int main()
{
    int t ;
    cin >> t ;
    while ( t--)
    {
        char a[100000] ;
        cin >> a ;
        int l = strlen(a) ;
        char b[100000] ;

        int k    = 0; ;
        int i ;
        for ( i = l-1 ; i >= 0  ; i-- )
        {
            b[k] = a[i] ;
            k = k+1 ;
        }
      //  cout << a <<endl;
      // cout << b <<endl;
    int r = 0;
     for ( i = 0 ; i<l-1 ; i++)
     {
     int q = ( a[i+1] - a[i]) ;
     int w = (b[i+1] - b[i]) ;
    // cout << "q -->" << q <<endl;
     // cout << "w -->" << w <<endl;
       if ( q < 0)
       {
        q = -q ;
       }
       if( w < 0)
       {
         w = -w ;
       }
       if ( q != w )
       {
         r = 1 ;
         break ;
       }


     }
     if ( r == 1 )
     {
        cout << "Not Funny" <<endl;
     }
     else{

       cout << "Funny" <<endl;
     }

    }
    return  0 ;
}
