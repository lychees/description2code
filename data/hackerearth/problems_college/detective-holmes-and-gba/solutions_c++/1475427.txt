#include <iostream>
using namespace std;

int main()
{
    int t,n,m,s;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m>>s;
    	if(m%n==0)
    	{
    		if(n==0)
    		cout<<n<<endl;
    		else
    		cout<<s-1<<endl;
    	}
    	else
    	{
    		int ans=((m%n)+s-1)%n;
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
