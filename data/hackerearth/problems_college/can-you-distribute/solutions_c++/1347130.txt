#include<bits/stdc++.h>
#define mod 10000007
using namespace std;
int C[500][500];
int b(int n, int k)
{

    int i, j;


    for (i = 0; i <= n; i++)
    {
        for (j = 0; j <= min(i, k); j++)
        {

            if (j == 0 || j == i)
                C[i][j] = 1;


            else
                C[i][j] = (C[i-1][j-1]%mod + C[i-1][j]%mod)%mod;
        }
    }

    return C[n][k];
}
