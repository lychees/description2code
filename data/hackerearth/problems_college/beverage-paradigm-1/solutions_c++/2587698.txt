#include<stdio.h>
#include<stdbool.h>
bool numbers(int arr[],int arr_size,int sum)
{
    int i,k,j,l,r;
    for(i=0;i<arr_size-2;i++)
    {
       for(j=i+1;j<arr_size;j++)
       {
           for(k=j+1;k<arr_size;k++)
           {
               if(arr[i]+arr[j]+arr[k]==sum)
               {
                   printf("True\n");
                   return true;
               }
           }
       }
    }
    printf("False\n");
    return false;
}
int main()
{
        int sum,n;
        scanf("%d",&n);
        int arr[n];
        for(int i=0;i<n;i++)
        scanf("%d",&arr[i]);
        scanf("%d",&sum);
        int arr_size=sizeof(arr)/sizeof(arr[0]);
        numbers(arr,arr_size,sum);
    return 0;
}
