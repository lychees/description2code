#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	int x[t],y[t],r[t];
	for(int i=0;i<t;i++){
		cin>>x[i]>>y[i]>>r[i];
	}
	float d;
	int a[1000][1000]={0};
	for(int i=0;i<1000;i++){
		for(int j=0;j<1000;j++){
			for(int k=0;k<t;k++){
				d=sqrt(pow(((i+1)-x[k]),2)+pow(((j+1)-y[k]),2));
				if(d<=r[k]) a[i][j]++;
				if(a[i][j]>=2) break;
			}
		}
	} int count=0;
	for(int i=0;i<1000;i++){
		for(int j=0;j<1000;j++){
			if(a[i][j]>=2) count++;
		}
	}
	cout<<count<<endl;
}
