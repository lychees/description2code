#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	if(n==0)
    		cout<<"0"<<endl;
    	else if(n<3)
    		cout<<"1"<<endl;
    	else if(n%3==0)
    		cout<<n/3<<endl;
    	else
    		cout<<n<<endl;
    }
    return 0;
}
