#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i,j,k,n,m,W;
    cin>>t;
    int K[105][10005];
    int wt[105],val[105];
    while(t--)
    {
    	cin>>n>>W;
       for(i=1;i<=n;i++)
       cin>>val[i]>>wt[i];
       memset(K,0,sizeof(K));
    for (i = 1; i <= n; i++)
       for (int w = 0; w <= W; w++)
         if (wt[i] <= w)
           	   K[i][w] = max(val[i]+K[i-1][w-wt[i]],  K[i-1][w]);
         else
               K[i][w] = K[i-1][w];
           
    int mini=0 ;     
    for(i=0;i<=W;i++)
   		{
   			if(K[n][mini]-mini<K[n][i]-i)
   				mini=i;
   		}
       
   
   cout<<mini<<" "<<K[n][mini]+W-mini<<endl;
    }
}
