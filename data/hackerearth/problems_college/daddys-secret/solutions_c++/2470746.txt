#include <bits/stdc++.h>

#define ll long long
#define pb push_back
#define mp make_pair
#define ld double

using namespace std;

const int nm = 5010;

int n, m;
pair<int, ld> a[nm];
int f[nm];

int main() {
#ifdef LOCAL
	freopen("input.txt", "r", stdin);
#endif
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &n, &m);
		for (int i = 1; i <= n; ++i)
			scanf("%d%lf", &a[i].first, &a[i].second);
		int res = 0;
		for (int i = 1; i <= n; ++i) {
			f[i] = 1;
			for (int j = 1; j < i; ++j)
				if (a[j].first <= a[i].first)
					f[i] = max(f[i], f[j] + 1);
			res = max(res, f[i]);
		}
		printf("%d\n", n - res);
	}
#ifdef LOCAL
	printf("%.3lf s\n", (double) clock() / (double) CLOCKS_PER_SEC);
#endif
}
