using namespace std;
bool prime[1000005];
bool visited[1000005];
void sieve()
{
    prime[0]=true;
    prime[1]=true;
    for(int i=2;i<=sqrt(1000004);i++)
    {
        if(prime[i]==false)
        {
            for(int j=i*i;j<=1000004;j+=i)
                prime[j]=true;
        }
    }
}
int main()
{

    sieve();
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int flag=0;
        int ans_i=-1000000;
        int ans_j=1000000;
        for(int i=1;i<=n;i++)
        {
            if(prime[i]==false && prime[n-i]==false && i!=(n-i))
            {

                if((ans_j-ans_i)>=(abs(n-2*i)))
                {
                    flag=1;
                    ans_j=max(n-i,i);
                    ans_i=min(n-i,i);
                }

            }
        }
        if(flag==0)
        cout<<-1<<" "<<-1<<endl;
        else
        cout<<ans_i<<" "<<ans_j<<endl;

    }
}
