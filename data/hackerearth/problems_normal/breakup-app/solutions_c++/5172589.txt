#include <iostream>
#include <string>
#include <sstream>
using namespace std;

int main()
{
    int n,prm=0,snd=0,weight;
    cin >> n;
    string str;
    for(int i=0;i<n;i++){
    	getline(cin,str);
    	if(str[0]=='M')
    		weight=1;
    	else
    		weight=2;
    	stringstream ss(str);
    	string temp;
    	int num;
    	while(getline(ss,temp,' ')){
    		if(stringstream(temp)>>num)
   			{
   				if(num==19 || num==20)
   					prm+=weight;
   				else
   					snd+=weight;
   			}
    	}
    }
    if(prm>snd)
    	cout << "Date";
    else
    	cout << "No Date";
    return 0;
}
