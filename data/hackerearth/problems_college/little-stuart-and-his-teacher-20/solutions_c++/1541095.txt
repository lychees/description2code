#include <bits/stdc++.h>

using namespace std;
const int MAX = 1000005;
bool prime[MAX];
long long toti[MAX], totiS[MAX];

void s()
{
    int i, j;
    for(i = 0;i < MAX;++i)
        toti[i] = i;
    for(i = 2;i < MAX;++i)
    {
        if(prime[i] == false)
        {
            toti[i] = i-1;
            for(j = 2*i;j < MAX;j += i)
                prime[j] = true, toti[j] -= (toti[j] / i);
        }
    }
    totiS[0] = 0;
    toti[1] = 0;
    for(i = 1;i < MAX;++i)
        totiS[i] = totiS[i-1] + (i - toti[i] - 1);
}

int main()
{
    int t, n;
    s();
    for(cin >> t;t--;)
    {
        cin >> n;
        cout << totiS[n] << endl;
    }
    return 0;
}
