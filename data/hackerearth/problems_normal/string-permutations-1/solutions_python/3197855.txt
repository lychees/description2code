from collections import Counter
from collections import defaultdict
t=int(raw_input())
li=[]
count=0
for _ in range(t):
    x=raw_input().strip()
    li.append(x)
counts=Counter(li)
temp=counts.values()

m=max(temp)
for i in temp:
    if i==m:
        count+=1

ans=m*count
print ans
