#include <bits/stdc++.h>

using namespace std;

int t , n , k;
string arr [500 + 10];
int dp [500 + 10][ (1 << 10) + 10];

int rec (int index, int msk){

    if(index == n) return 0;

    int &st = dp[index][msk];
    if(st != -1) return st;

    int ans = 1e9;

    for(int i = 0; i < arr[index].size(); i++){

        if(arr[index][i] == '0') continue;

        if(msk & (1 << i)) ans = min (ans , rec(index + 1, msk));
        else ans = min(ans, 1 + rec(index + 1, msk | (1 << i)));
    }

    return st = ans;
}


int main()
{
    ios::sync_with_stdio(false);cin.tie(0);

    cin >>       t         ;

    while(t--){

        memset(dp , -1, sizeof dp);

        cin >> n >> k;

        for(int i = 0; i < n; i++) cin >> arr[i];

        cout << rec(0 , 0) << "\n";
    }

    return 0;
}
