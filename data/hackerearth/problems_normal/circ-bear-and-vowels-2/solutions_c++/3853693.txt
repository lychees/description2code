#include <bits/stdc++.h>
#include<sstream>
using namespace std;


int main() {
 int t;
 cin>>t;
 while(t--){
   string s;
   cin>>s;
   int n = s.length();
   bool f = true;
   int v = 0,c = 0,temp = 0;
   for(int i=0;i<n;i++){
       if(s[i] == 'a' || s[i] == 'e' || s[i] == 'i'|| s[i] == 'o'|| s[i]=='u' || s[i]=='y'){
           ++v;
           temp = 0;
       }
       else{
           ++c;
           ++temp;
       }
       if(temp == 3){
           f = false;
           break;
       }
   }
   //cout<<c<<" "<<v<<"\n";
   if(!f || (c > v)){
       cout<<"hard\n";
   }
   else{
       cout<<"easy\n";
   }
   
 }
 return 0;
}
