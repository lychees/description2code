//Supertables
#include <iostream>
#include <bits/stl_algo.h>

typedef long long LL;

LL nthElemInSuperTable(LL a, LL b, LL n)
{
	LL min = 1;
	LL max = n * std::max(a, b);	
	LL lcm = (a * b) / std::__gcd(a, b);
	
	while (min <= max)
	{
		LL mid = (min + max) / 2;
		LL factors = mid / a + mid / b - mid / lcm;
		
		if (n == factors)
		{
			return std::max((mid / a) * a, (mid / b) * b); 
		}
		else if (factors > n)
		{
			max = mid - 1;
		}
		else
		{
			min = mid + 1;
		}
	}
	
	return -1;
}

int main()
{
	LL t, a, b, n;
	std::cin >> t;
	
	while (t--)
	{
		std::cin >> a >> b  >> n;
		std::cout << nthElemInSuperTable(a, b, n) << std::endl;
	}
	
	return 0;
}
