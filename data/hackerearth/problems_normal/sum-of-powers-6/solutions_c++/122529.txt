#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int k,n,i,a[1000000],x,y=0;
    
    for(i=0;i<1000000;i++)
    a[i]=0;
    
    a[1] = 1;
    
    for(i = 2; i <= 1000;i++)
    {
        for(int j = 2;pow(i,j) <= 1000000;j++)
        {
            x = pow(i,j);
            a[x] = 1;
        }
    }
    cin>>k;
    while(k--){
        cin>>n;
        y=0;
        for(int i = 1;i <= n/2 ; i++){
            if(a[i] && a[n-i]) {
                y=1;
                break;
            }
        }
        if (y) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        }
return 0;       
}
