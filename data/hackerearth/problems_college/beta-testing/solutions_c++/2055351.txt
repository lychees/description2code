#include <iostream>
#include <string>
#include <map>
#include <stdlib.h>
#include <algorithm>
using namespace std;
struct aman{
string s;
int count;
};

bool compare(aman a, aman b)
{
	return (a.count>b.count);
}

int main()
{
	int n;
	cin>>n;
	string a[n];
	map<string, int> M,S;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		M[a[i]]++;
	}
	aman A[n+1];
	int j=0;
	for(int i=0;i<n;i++)
	{
		if(!S[a[i]])
		{
			A[j].s=a[i];
			A[j].count=M[a[i]];
			j++;
			S[a[i]]++;
		}
	}
	sort(A,A+j,compare);
	for(int i=0;i<j;i++)
		cout<<A[i].s<<" "<<A[i].count<<endl;
	
	return 0;
}
