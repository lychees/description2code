#include <vector>
#include <utility>
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
        int a, b, c, d;
        vector<pair<int,int>> ZA;

        cin >> a >> d;
        for (b = 0; b < a; ++b) {
                cin >> c;
                ZA.push_back(make_pair(0, c));
        }
        for (b = 0; b < a; ++b)
                cin >> ZA[b].first;
        sort(ZA.rbegin(), ZA.rend());
        for (b = 1; b < ZA.size(); ++b)
                ZA[b].second += ZA[b - 1].second;
        while (d--) {
                cin >> c;
                cout << ZA[c - 1].second << endl;
        }

        return 0;
}
