#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int T, t;
    long long N, log2NLong, max = 0, frequency = 0, temp;
    double log2N;
    
    cin>>T;
    
    for(t = 0; t < T; t++)	{
    	cin>>N;
    	if(N == 1)	{
    		max = 0;				frequency = 1;
    	}
    	else if(N == 2)	{
    		max = 1;				frequency = 2;
    	}
    	else	{
    		log2N = log(N) / log(2);
    		log2NLong = (long long)log2N;
    		temp = (long long)pow(2, log2NLong);
    		if(temp < N)	{
    			max = (long long)pow(2, log2NLong + 1) - 1;
    			frequency = (N - temp) * 2;
    		}
    		else if(temp == N)	{
    			max = temp - 1;
    			frequency = (N - (temp/2)) * 2;
    		}
    	}
    	cout<<max<<" "<<frequency<<endl;
    }

    return 0;
}
