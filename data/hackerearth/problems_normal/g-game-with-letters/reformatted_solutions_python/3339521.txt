'''

T = int(input())
for tc in xrange(T):
    n, S = raw_input().strip().split()
    n = int(n)
    input_word_score = {}
    team_score = []
    for letter in S:
        if letter in input_word_score:
            input_word_score[letter] += 1
        else:
            input_word_score[letter] = 1
    # print(input_word_score)
    for team_index in xrange(1, n+1):
        team = raw_input()
        score = 0
        for member in team:
            if member in input_word_score:
                score += input_word_score[member]
        team_score.append((score, 20-len(team), n-team_index))

    final_sorted_team = sorted(team_score, reverse=True)

    # print(final_sorted_team)
    print(n-final_sorted_team[0][2])
