#include <iostream>
#include <bits/stdc++.h>

using namespace std;
int a[1000005];
int cum[1000005];
void pre()
{
	a[0] = 0;
	a[1]  = 1;
	for(int i = 2;i < 1000005; i++){
		int temp = a[i-1]*i;
		int sum1 = temp;
		while(sum1 > 9){
			int sum = 0;
			temp = sum1;
		while(temp!= 0){
			int tmp = temp%10;
			sum = sum + tmp;
			temp = temp/10;
		}
		sum1 = sum;
	}
	a[i] = sum1;
	}
	cum[0] = 0;
	for(int i = 1; i < 1000005; i++){
		cum[i] = cum[i-1] + a[i];
	//	cout << cum[i] << endl;
	}
}
int main()
{
    pre();

    int test;
  	cin >> test;
  for(int i = 0; i < test; i++){
    	int a, b;
    	cin >>a >> b;
    	if(a != 0)
    		cout << cum[b] - cum[a-1] << endl; 
    	else
    		cout << cum[b] << endl;
    }
    return 0;
}
