#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;

int main()
{	int t, min;
	string str;
	cin>>t;
	int ct[26];
	while (t-- > 0 )
	{	cin>>str;
		memset(ct,0,sizeof(ct));
		for ( int i=0; i<str.length(); i++ )
			ct[str[i]-'a']++;
		min=ct['r'-'a'];
		if ( ct['u'-'a'] < min )
			min=ct['u'-'a'];
		if ( ct['b'-'a'] < min )
			min=ct['b'-'a'];
		if ( ct['y'-'a'] < min )
			min=ct['y'-'a'];
		cout<<min<<endl;
	}
	return 0;
}
