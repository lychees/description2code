#include <iostream>
#include <vector>
#include <set>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int T=0;
	
	cin >> T;
	
	vector<int> fr;
	vector<int> temp;
	
	while(T--)
	{
	 int N=0;
	 int M=0;
	 
	 cin >> N >> M;
	 
	 fr.push_back(N);
	 temp.push_back(N);
	 
	 while(M--)
	 {
	  int P=0;
	  cin >> P;
	  temp.push_back(P);
	 }
 	}
	
	set<int> fin(temp.begin(),temp.end());
	set<int>::iterator it;
	 
	int count=0;
	 
	for(it=fin.begin();it!=fin.end();it++)
	{
	 for(int i=0;i<fr.size();i++)
	  if(fr[i] == *it)
	   count++;
    }
	
	cout << fin.size()-count << endl;
	
	fr.clear();
	temp.clear();
	
	return 0;
}
