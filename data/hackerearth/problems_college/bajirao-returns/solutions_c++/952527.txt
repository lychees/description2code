#include<stdio.h>
int ar[1001][1001];
int dp[101][101];
int p[101][101];
void recur(int n,int i,int j)
{
    if(j<1 || j>n)
        return;
    ar[i][j]=1;
    if(ar[i][2*j]==0)
    recur(n,i,2*j);
    if(ar[i][j/2]==0)
    recur(n,i,j/2);
}
int main()
{
    int n,i,j,k,m;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            scanf("%d",&p[i][j]);
            
            dp[i][j]=0;
            if(i==1)
                dp[i][j]=p[i][j];
        }
        recur(n,i,i);
    }
    for(i=2;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            for(k=1;k<=n;k++)
            {
                if(ar[k][j])
                {
                    if(dp[i-1][k]+p[i][j]>dp[i][j])
                        dp[i][j]=dp[i-1][k]+p[i][j];
                }
            }
        }
    }
    int ans=dp[n][1];
    for(j=2;j<=n;j++)
    {
        if(dp[n][j]>ans)
            ans=dp[n][j];
    }
    printf("%d\n",ans);
}
