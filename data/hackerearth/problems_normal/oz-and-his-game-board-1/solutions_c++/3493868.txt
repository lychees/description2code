#include<iostream>
using namespace std;
int main()
{
	int t,a,b,c;
	long long int n,m,ans;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		cin>>a>>b>>c;
		if(m>=n)
		{
			if(a+b-2*c>0)
			{
				ans=(a+b)*(n)+b*(m-n);
			}
			else
			{
				ans=2*c*n+b*(m-n);
			}
		}
		else
		{
			if(a+b-2*c>0)
			{
				ans=(a+b)*(m)+a*(n-m);
			}
			else
			{
				ans=2*c*m+a*(n-m);
			}
		}
		cout<<ans<<endl;
	}
	
}
