#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n;
    	int x;
    	cin>>n;
    	int y;
    	int p=n/2;
    	string s[n];
    	int j;
    	int k;
    	for(j=0;j<n;j++)
    	{  
    		cin>>s[j];
    	
    		
    	}
    	for(j=0;j<n;j++)
    	{
    		for(k=0;k<n;k++)
    		{
    			if(s[j][k]=='*')
    			{x=j;
    			y=k;
    			
    			break;
    				
    			}
    			
    		}
    	}
    	int q=abs(x-p)+abs(y-p);
    	cout<<q<<endl;
    }
}
