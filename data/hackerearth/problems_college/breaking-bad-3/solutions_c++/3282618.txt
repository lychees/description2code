#include <bits/stdc++.h>

using namespace std;

int n;
long long arr[2*100005];

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	int i;
    	for(i=0;i<n;i++) cin >> arr[i];
    	sort(arr,arr+n);
    	reverse(arr,arr+n);
    	long long ans=0;
    	for(i=0;i<n;i+=2)
    	{
    		ans+=(arr[i]*arr[i+1]);
    	}
    	cout << ans << endl;
    }
    return 0;
}
