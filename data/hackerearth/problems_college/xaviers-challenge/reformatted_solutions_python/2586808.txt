x = input()
for i in range(0, x):
    y = list(raw_input())
    s = 0
    for j in range(0, len(y)):
        s += ord(y[j])
    s = s/len(y)
    z = ""
    if s % 2 == 0:
        for j in range(0, len(y)):
            z += y[j]
        print z
    else:
        for j in range(1, len(y)+1):
            z += y[-j]
        print z
