#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n;
		scanf("%d", &n);
		int arr[n];
		for(int i=0; i<n; i++)
			scanf("%d", &arr[i]);
		sort(arr, arr+n);
		int i=0; 
		int j=n-1;
		while(i<=j)
		{
			if(arr[i]==arr[i-1])
				i++;
			else if(arr[j]==arr[j+1])
				j--;
			else if(arr[i]==arr[j])
			{
				printf("%d ", arr[i]);
				i++;
				j--;
			}
			else
			{
				printf("%d %d ", arr[j], arr[i]);
				i++;
				j--;
			}
		}
		printf("\n");
	}
	return 0;
}
