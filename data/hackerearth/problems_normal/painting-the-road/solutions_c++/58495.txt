#include <iostream>
#include <cstdio>
#define MAX 300000
#define s(n) scanf("%lld", &n)
#define lli long long int
using namespace std;

int main(){
    lli a[513], b[513], c[513], ans[257], t, l, k, costm, cost;
    s(t);
    while(t--){
        s(l); s(k);
        for(int i = 0; i < k; i++){
            s(a[i]); s(b[i]); s(c[i]);
        }
        for(int i = 1; i <= l; i++){
            costm = MAX;
            for(int j = 0; j < k; j++){
                if(a[j] >= i || b[j] < i) continue;
                lli cost = c[j];
                if(a[j]>=1) cost += ans[a[j]];
                if(cost<costm)
                    costm = cost;
            }
            if(costm == MAX){
                printf("-1\n");
                break;
            }
            ans[i] = costm;
        }
        if(costm != MAX)
            printf("%lld\n", ans[l]);
    }
    return 0;
}
