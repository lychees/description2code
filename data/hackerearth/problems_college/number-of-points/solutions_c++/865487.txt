#include <bits/stdc++.h>
using namespace std;

long long gcd(long long a,long long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		long long x,y,xx,yy;
		cin>>x>>y>>xx>>yy;
		
		long long a=(long long)abs(x-xx);
		long long b=(long long)abs(y-yy);
		if(a==0&&b==0)
		cout<<0<<endl;
		else
		{
		long long ans=gcd(a,b)-1;
		cout<<ans<<endl;
		}
	}
	return 0;
}
