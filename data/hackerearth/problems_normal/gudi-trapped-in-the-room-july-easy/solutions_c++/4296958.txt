string rot(string n)
{
   string snum = n;
   string tnum;

   tnum.resize(snum.length());

   int len = snum.length();
   for(int i=0;i<len;i++)
     tnum[(i+h)%len]=snum[i];

   //cout<<"rotation: " <<tnum<<endl;
   return tnum;
}
string add(string n)
{
   string snum=n;

   for(int i=1;i<snum.length();i+=2)
   {
       int nm= snum[i]-'0';
       nm = (nm+a)%10;
       snum[i]= nm+'0';
   }
  // cout<<"addition: "<<snum<<endl;
   return snum;
}



void rec(string n)
{
   if(vis[n])return;
   vis[n]=true;

   if(n<res)res=n;

   rec(rot(n));
   rec(add(n));
}

int main(void)
{
   int t;
   cin>>t;

   while(t--)
   {
        res="999999";
        vis.clear();
        cin>>n>>a>>h;
        rec(n);
        cout<<res<<endl;
   }

  return 0;
}
