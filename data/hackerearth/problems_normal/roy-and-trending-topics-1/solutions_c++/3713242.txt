#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<pair<long long int,long long int> > pq;
    map<long long int,long long int> mp;
    long long int id[n],z[n],p[n],l[n],c[n],s[n],nz[n],incr;
    for(int i=0;i<n;i++){
    	cin>>id[i]>>z[i]>>p[i]>>l[i]>>c[i]>>s[i];
    	nz[i] = 50*p[i] + 5*l[i] + 10*c[i] + 20*s[i];
    	incr = nz[i] - z[i];
    	mp[id[i]] = nz[i];
    	pq.push_back(make_pair(incr,id[i]));
    }
    make_heap(pq.begin(),pq.end());
    //int len = min(n,5);
    for(long long int i=0;i<5;i++){
    	cout<<pq.front().second<<" "<<mp[pq.front().second]<<endl;
    	pop_heap(pq.begin(),pq.end());
    	pq.pop_back();
    }
    return 0;
}
