#include <iostream>
#include<algorithm>
#include<stdio.h>

using namespace std;



long long int max_moves(long long int a[],long long int n)
{
	long long int k1=a[1]-a[0]-1;
	long long int k2=a[2]-a[1]-1;
	
	if(k1<0 && k2<0)
	  return 0;
	
	if(k1>k2)
	  return k1;
	else return k2;
	
}

int main()
{
	long long int t,a[3],n=3;
	cin>>t;
	while(t--)
	{
		cin>>a[0]>>a[1]>>a[2];
		sort(a,a+n);
		
		cout<<max_moves(a,n)<<endl;
		
	}
    
    return 0;
}
