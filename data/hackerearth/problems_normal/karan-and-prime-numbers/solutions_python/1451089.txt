def is_prime(n):
    if n==1:
         return False
    if n % 2 == 0 and n > 2:
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True
def add(f,s):
    sum=0
    for nos in range(f,s+1):
       
        if is_prime(nos):
            sum+=nos
    return sum
