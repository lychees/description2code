#include <iostream>
using namespace std;
int sum[1001]={0};
void preComp(){
	int i,A[2600],j,d,temp,c,tsum,k;
	for(i=0;i<2600;++i)
		A[i] = -1;
	A[1]=1;
	sum[0]=1;
	for(i=1;i<=1000;++i){
		j=1;
		tsum=c=0;
		while(A[j] > -1){
			temp=(A[j]*i)+c;
			A[j]=temp%10;
			c=temp/10;
			++j;
		}
		while(c){
			A[j++]=c%10;
			c/=10;
		}
		d=j-1;
		for(k=d;k>0;--k)
			tsum+=A[k];
		sum[i]=tsum;
	}
		
}
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int t,n;
	preComp();
	cin>>t;
	while(t--){
		cin>>n;
		cout<<sum[n]<<"\n";
	}
	return 0;
}
