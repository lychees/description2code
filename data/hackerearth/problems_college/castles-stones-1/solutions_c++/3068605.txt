#include<bits/stdc++.h>
using namespace std;
int  main(){
    int tt;
    cin>>tt;
    while(tt--){
        int m,n;
        cin>>m>>n;
        int arr[1007];
        for(int i=0;i<n;i++){
            cin>>arr[i];
        }
        sort(arr,arr+n);
        int ans=0;
        int i=0,j=n-1;
        for (;i<=j;){
        	if(arr[j]>m)
        		--j;
            else if(arr[i]+arr[j]>m){
                ans+=1;
                j--;
            }else {
        			ans++;
        			i++;
        			j--;
            			
            }
        }
       
        cout<<ans<<endl;



        }
    }
