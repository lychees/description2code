#include<iostream>
#include <sstream> 

using namespace std;
int arr[9000001];
int res[9000001]={0};
long long solve(int numberOfElements,int Sum)
{
 res[0]=1;
  	for (int i = 0; i < numberOfElements; ++i)
 	{

  	  for (int j = arr[i]; j < Sum+1; ++j)
  	  {
  	  	/* code */
  	  	res[j]+=res[j-arr[i]]; 
  	  //	cout<<res[j]<<endl;
  	  }
 	 }
 	 return res[Sum];

}
int main()
{

   int numberOfElements,Sum;
   string elements;
   stringstream ss;
    cin>>Sum;
    getline(cin, elements);
    elements="";
    getline(cin, elements);

    ss << elements;
    numberOfElements=0;
     while(!ss.eof())
    {
      ss>>arr[numberOfElements];
      //cout<<arr[numberOfElements]<<endl;
      numberOfElements++;
    }
    
   
    cout<<solve(numberOfElements,Sum)<<endl;
	return 0;
}
