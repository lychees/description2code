#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long arr[50] = {0}, num = 2;
	for(int i = 1; i < 50; i++)
	{
		arr[i] = arr[i-1] + i * num;
		num *= 2;
	}
    ios::sync_with_stdio(0);
    int t; cin>>t;
    while(t--)
    {
    	long long n; cin>>n;
    	bool flag=0;
    	int index = lower_bound(arr,arr+50,n)-arr;
    	if(arr[index] == n)
    	{
    		flag = 1;
    	}
    	else
    	{
    		n -= arr[index-1];
    		if(n%index == 0)
    		{
    			n = n/index-1;
    			if(n%2) {flag = 1;}
    		}
    		else
    		{
    			long long temp = n/index;
    			n %= index;
    			if(bitset<64>(temp)[index-n]) {flag = 1;}
    		}
    	}
    	cout<<(flag?"Earth":"Hacker")<<endl;
    }
    return 0;
}
