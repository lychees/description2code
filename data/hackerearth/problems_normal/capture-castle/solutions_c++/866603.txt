void calculateCost(int mat[][MAX],int cost[][MAX], int i,int j,int M,int N,int currcost,int x,int y){

	if(cost[i][j] == -1 || cost[i][j] > currcost) {
		cost[i][j]  = currcost;
		calculateCost(mat,cost,i+1,j,M,N,currcost,x,y);
		calculateCost(mat,cost,i,j+1,M,N,currcost,x,y);
		calculateCost(mat,cost,i-1,j,M,N,currcost,x,y);
		calculateCost(mat,cost,i,j-1,M,N,currcost,x,y);
	}
}
