#include <iostream>
#include<cmath>
#define li long int
using namespace std;

int main()
{
    li n;
    cin>>n;
    li i=5,c=0;
    while(n>0)
    {
    	n=n-i;
    	i*=2;
    	c++;
    }
    n+=i/2;
    
    li x=pow(2,c-1);
   	
   	li a=n/x;
   	//cout<<n<<" "<<c<<" "<<a<<" "<<x;
   	if(n%x==0)
   	;
   	else
   	a++;
   	switch(a)
   	{
   		case 1:cout<<"Sheldon"<<endl;break;
   		case 2:cout<<"Leonard"<<endl;break;
   		case 3:cout<<"Penny"<<endl;break;
   		case 4:cout<<"Rajesh"<<endl;break;
   		case 5:cout<<"Howard"<<endl;break;
   	}
    
    return 0;
}
