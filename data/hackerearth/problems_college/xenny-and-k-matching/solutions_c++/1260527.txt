#include<bits/stdc++.h>
using namespace std;
#define MAX 1000007
int a[MAX];

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
    int l=0, r=k;
    int ans=0;
    while(r < n)
    {
        int x = a[l];
        int y = a[r];
        int z = abs(x-y);
        int w = abs(x+y);
        if(z%k==0 || w%k==0) ans++;
        r++;
        l++;
    }
    printf("%d\n",ans);
}
