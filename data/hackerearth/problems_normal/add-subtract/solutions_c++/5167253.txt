#define fastIO ios::sync_with_stdio(false);cin.tie(0)

#define pii(x) printf("%d\n",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld\n",x)
#define pf printf
#define present_set_map(container, element) (container.find(element) != container.end()) 
#define present_vec(container, element) (find(all(container),element) != container.end())

using namespace std;

int main()
{
    fastIO;
    ll tc,n,ans,i,j,k,val,t;
    cin>>tc;
    while(tc--)
    {
        ll mini=1e18;
        ll mini_index=-1;
        cin>>n>>k;
        ll arr[101];
        for(i=0;i<n;i++)
            cin>>arr[i];
        for(val=0;val<=100;val++)
        {
            ll c=0;
            vector<ll>v(n);
            for(i=0;i<n;i++)
            {
                if(arr[i]>val)
                    v[i]=(arr[i]-val)*5;
                else
                    v[i]=(val-arr[i])*3;
            }
            sort(all(v));
            for(i=0;i<k;i++)
                c+=v[i];
            mini=min(mini,c);
        }
        cout<<mini<<"\n";
    }
}
