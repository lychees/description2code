#include <iostream>
#include <cstdio>

using namespace std;
unsigned long long sum (int);
unsigned long long N[8];

int main () {
    int t, n, p, q, r, i, s;
    unsigned long long count=0;
    scanf ("%d", &t);
    while (t--) {
        scanf ("%d %d %d %d",&n ,&p ,&q ,&r);
        for (i = 0; i < 8; i++) {
            N[i] = 0;
        }
        N[0] = sum (n);
        s = n-p-1;
        if (s >= 0) {
            N[1] = sum (s);
        }
        s = n-q-1;
        if (s >= 0) {
            N[2] = sum (s);
        }
        s = n-r-1;
        if (s >= 0) {
            N[3] = sum (s);
        }
        s = n-p-q-2;
        if (s >= 0) {
            N[4] = sum (s);
        }
        s = n-q-r-2;
        if (s >= 0) {
            N[5] = sum (s);
        }
        s = n-r-p-2;
        if (s >= 0) {
            N[6] = sum (s);
        }
        s = n-p-q-r-3;
        if (s >= 0) {
            N[7] = sum (s);
        }
        
        count = (N[0]+N[4]+N[5]+N[6])-(N[1]+N[2]+N[3]+N[7]);    
        printf ("%lld\n", count);
    }
    
    return 0;
}

unsigned long long sum (int i) {
    int n;
    unsigned long long s=0;
    for (n = 0; n <= i; n++) {
        s = s+(((1+n)*(2+n))/2);
    }
    return (s);
}
