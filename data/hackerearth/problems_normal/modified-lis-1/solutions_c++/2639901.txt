#include<bits/stdc++.h>
using namespace std;
 
#define sd(mark) scanf("%d",&mark)
#define slld(mark) scanf("%lld",&mark)
#define clr(mark) memset(mark,0,sizeof(mark))
#define F first
#define S second
#define MP make_pair
#define PB push_back
#define pii pair<int,int>
#define ll long long
#define mod 1000000007
int bit[2][100010];
ll cnt[2][100010];
int a[100010];
void update(int i,int val,int f,ll c)
{
	while(i<=100001)
	{
		if(bit[f][i]==val)
			cnt[f][i]=(cnt[f][i]+c)%mod;
		else if(bit[f][i]<val)
			cnt[f][i]=c;
		bit[f][i]=max(bit[f][i],val);
		i=i+(i&(-i));
	}
}
pair<int,ll> query(int i,int f)
{
	int ret=0;
	int c=0;
	while(i)
	{
		if(ret==bit[f][i])
			c=(c+cnt[f][i])%mod;
		else if(ret<bit[f][i])
			c=cnt[f][i];
		ret=max(ret,bit[f][i]);
		i=i-(i&(-i));
	}
	return MP(ret,c);
}
int main()
{
	int n,i,maxx=0;
	for(i=0;i<=100000;++i)
		cnt[0][i]=cnt[1][i]=bit[0][i]=bit[1][i]=0;
	update(1,0,0,1);
	update(1,0,1,1);
	sd(n);
	for(i=0;i<n;++i)
	{
		sd(a[i]);
		bool f=0;
		if(a[i]<0)
		{
			f=1;
			a[i]=-a[i];
		}
		pair<int,ll> p=query(a[i],1-f);
		int val=1+p.F;
		maxx=max(maxx,val);
		int c=p.S;
		update(a[i]+1,val,f,c);
	}
	pair<int,ll> p0=query(100001,0);
	pair<int,ll> p1=query(100001,1);
	maxx=max(p0.F,p1.F);
	if(p0.F==p1.F)
		printf("%d %lld\n",maxx,(p0.S+p1.S)%mod);
	else if(p0.F>p1.F)
		printf("%d %lld\n",maxx,p0.S);
	else
		printf("%d %lld\n",maxx,p1.S);
}
