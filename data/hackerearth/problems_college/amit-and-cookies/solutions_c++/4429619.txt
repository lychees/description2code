#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    priority_queue<int> q;
    int n,k,i,j,num;
    cin>>n>>k;
    for(i=1;i<=n;i++)
    {
        cin>>num;
        q.push(-num);
    }
    int cnt=0;
    int sum=0;
    while(!q.empty()&&sum<k)
    {
        if((sum-q.top())>k)
            break;
        sum=sum-q.top();
        q.pop();
        cnt++;
    }
    cout<<cnt<<endl;

}
