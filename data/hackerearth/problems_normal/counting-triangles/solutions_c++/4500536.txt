#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
bool comp(pair<l1,pair<l1,l1> >  p1,pair<l1,pair<l1,l1> >  p2)
{
 if(p1.first!=p2.first)
 return p1.first<p2.first;
 else if(p1.second.first != p2.second.first)
 return p1.second.first < p2.second.first;
 else
 return p1.second.second< p2.second.second;
}
int main()
{
	l1 n,a,b,c,i,count=0;
	cin>>n;
	pair<l1,pair<l1,l1> > p;
	l1 num[3];
	vector< pair<l1,pair <l1,l1> > > vec;
	for(i=1;i<=n;i++)
	{
		scanf("%lld%lld%lld",&a,&b,&c);
		num[0]=a;
		num[1]=b;
		num[2]=c;
		sort(num,num+3);
		p=make_pair(num[0],make_pair(num[1],num[2]));
		vec.push_back(p);
	}
	stable_sort(vec.begin(),vec.end(),comp);
 	l1	f=0;
	for(i=1;i<vec.size();i++)
	{
		if(vec[i]!=vec[i-1]&&vec[i]!=vec[i+1]&&(i+1)<vec.size())
		count++;
		
	}
	if(vec[0]!=vec[1]&&vec.size()>=2)
	count++;
	if(vec.size()>2)
	if(vec[vec.size()-1]!=vec[vec.size()-2])
	count++;
	
	cout<<count<<endl;
}
