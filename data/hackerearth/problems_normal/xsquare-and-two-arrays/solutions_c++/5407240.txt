#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int N, Q;
    vector<int> A, B;
    cin >> N >> Q;
    for(int i = 0;i<N;i++)
    {
    	int x;
    	cin >> x;
    	A.push_back(x);

    }
    for(int i = 0;i<N;i++)
    {
    	int x;
    	cin >> x;
    	B.push_back(x);

    }
    while(Q--)
    {
    	int c, L, R;
    	long sum = 0;

    	cin >> c >> L >> R;
    	if(c== 1){
         for(int i=L-1; i<R; i+=2)
    	 {sum = sum + A[i];
     	
         }
         for(int i=L; i<R; i+=2)
   		 {
   		 	sum = sum + B[i];
   		 
   		 }
    	}
        else if(c == 2){
         for(int i=L-1; i<R ; i+=2)
    	 {
    	     sum = sum + B[i];
            
         }
         for(int i=L; i<R; i+=2)
   		 {
   			sum = sum + A[i];
   		
   		 }
        }
        cout << sum << endl;
    }
    return 0;
}
