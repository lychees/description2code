#include <iostream>

using namespace std;
#define LL long long
LL freq[100005];
LL arr[100005];
LL dp[100005][2];
int main() {int n;cin>>n;
for(int i=1;i<=n;++i)
{cin>>arr[i];freq[arr[i]]++;}
if(freq[1])dp[1][1]=freq[1];

for(int i=2;i<=100000;++i)
{
    dp[i][1]=max(max(dp[i-2][1],dp[i-2][0]),dp[i-1][0])+freq[i]*i;
    dp[i][0]=max(dp[i-1][1],dp[i-1][0]);
}
    cout<<max(dp[100000][1],dp[100000][0]);
    return 0;
}
