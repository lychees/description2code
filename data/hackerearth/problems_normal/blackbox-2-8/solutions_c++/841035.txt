#include <iostream>
#include <vector>
using namespace std;

vector<vector<int> > all;

void PR() {
	vector<int> arr;
	arr.push_back(1);
	all.push_back(arr);
	int sz=1; // highest index+1;
	int x;
	for(int i=2;i<=1000;i++) {
		int carry=0;
		for(int j=0;j<sz;j++) {
			x=arr[j]*i+carry;
			arr[j]=x%10;
			carry=x/10;
		}
		while(carry) {
			//cout<<carry<<":::"<<sz<<":::"<<carry%10<<endl;
			arr.push_back(carry%10);
			carry/=10;
			sz++;
		}
		all.push_back(arr);
	}
	
}


int main()
{
	PR();
	int T,N;
	cin>>T;
	while(T--) {
		cin>>N;
		if(N==0) {cout<<"1"<<endl; continue;}
		int sum=0;
		for(int i=0;i<all[N-1].size();i++) {
			sum+=all[N-1][i];
			//cout<<":"<<all[N-1][i];
		}
		//cout<<endl;
		cout<<sum<<endl;
	}
    return 0;
}
