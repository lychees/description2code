#include<stdio.h>
#include<bits/stdc++.h>
#include<algorithm>
#define MAX 10000001
using namespace std;
vector<long int> v[15];
long int r[MAX]={0};
int prime(void)
{
	long int i;
	for(i=2;i<MAX;i++)
	{
		if(i>2)
		v[r[i-1]].push_back(i-1);
		if(r[i])
		continue;
		for(long int k=1;k*i<MAX;k++)
		r[k*i]++;
	}
	v[r[MAX-1]].push_back(MAX-1);
	return 0;
}
int main()
{
	int t;
	prime();
	scanf("%d",&t);
	while(t--)
	{
		long int a,b,k,am;
		scanf("%ld %ld %ld",&a,&b,&k);
		if(k>=9)
		{
			printf("0\n");
			continue;
		}
		long int tem=lower_bound(v[k].begin(),v[k].end(),a)-v[k].begin();
		long int temp=upper_bound(v[k].begin(),v[k].end(),b)-v[k].begin();
		am=temp-tem;
		printf("%ld\n",am);
	}
	return 0;
}
