#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,p;
    	scanf("%d%d",&n,&p);
    	long int temp;
    	long int flag=0,count=0;
    	for(int i=0;i<n;i++)
    	{
	    	scanf("%ld",&temp);
	    	if(temp>p)
	    		count++;
	    	if(temp==p)
	    		flag=1;
    	}
    	if(flag==1)
    	printf("YES %d\n",count);
    	else
   		printf("NO 0\n");
    }
}
