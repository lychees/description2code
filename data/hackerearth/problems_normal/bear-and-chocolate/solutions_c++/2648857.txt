#include <bits/stdc++.h>
#define ll long long int
#define ull unsigned long long int
using namespace std;

int main()
{
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		int a[n][n];
		int cherries_above = 0, cherries = 0;
		char c;
		for(int i = 0 ; i < n ; i++)
		{
			for(int j = 0 ; j < n ; j++)
			{
				cin >> c;
				if(c == '#')
				{
					a[i][j] = 1;
					cherries += 1;
				}
				else
					a[i][j] = 0;
			}
		}
		if(cherries % 2 == 1)
		{
			cout << "NO" << endl;
			continue;
		}
		bool flag = false;
		
		for(int i = 0 ; i < n ; i++)
		{
			for(int j = 0 ; j < n ; j++)
				cherries_above += a[i][j];
			if(cherries_above == cherries/2)
			{
				flag = true;
				break;
			}
		}
		if(flag)
		{
			cout << "YES" << endl;
		}
		else
		{
			cherries_above = 0;
			for(int i = 0 ; i < n ; i++)
			{
				for(int j = 0 ; j < n ; j++)
					cherries_above += a[j][i];
				if(cherries_above == cherries/2)
				{
					flag = true;
					break;
				}
			}
			if(flag)
				cout << "YES" << endl;
			else
				cout << "NO" << endl;
		}
		
	}
	return 0;
}
