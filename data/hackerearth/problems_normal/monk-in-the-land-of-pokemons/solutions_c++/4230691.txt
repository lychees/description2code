#include <iostream>
#include<unordered_map>
using namespace std;

int main()
{
    int test;
    cin>>test;
    for(test;test>0;test--){
    	int N;
    	cin>>N;
    	int count=0;
    	unordered_map<int,int> food;
    	
    	for(int i=0;i<N;i++){
    		int A,B;
    		cin>>A>>B;
    		if(food.find(A)==food.end()){
    			pair<int,int> p(A,1);
    			food.insert(p);
    		}
    		else{
    			food[A]++;
    		}
    		if(food.find(B)==food.end()){
    			count++;
    		}
    		else{
    			int num=food[B];
    			if(num==0)
    			count++;
    			else{
    				food[B]--;
    			}
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
