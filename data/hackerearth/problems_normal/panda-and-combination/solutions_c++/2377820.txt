#include <iostream>
using namespace std;

int main()
{
  unsigned long long int t,n,m;
  
   cin>>t;
   while(t--)
   {
   	cin>>n>>m;
   	 if(m==1)
   	  cout<<n<<endl;
   	 else
   	 {  long long ans=0;
   	 	while(n)
   	 	{
   	 		ans+=n%m;
   	 		n/=m;
   	 	}
   	 	
   	 cout<<ans<<endl;	
   	 }
   	   
   	
   }
    return 0;
}
