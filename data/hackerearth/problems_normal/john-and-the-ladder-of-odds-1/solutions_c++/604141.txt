#include <iostream>
#include <cstdio>

using namespace std;

#define mod 1000000009
#define getcx getchar_unlocked

long long int scan()
{
    long long int n=0;
    char ch = getcx();
    long long int sign=1;
    while( ch < '0' || ch > '9' ) {
        if(ch=='-')
            sign = -1;
        ch=getcx();
    }
    while( ch >= '0' && ch <= '9' )
        n = (n<<3)+(n<<1) + ch-'0',
    ch=getcx();
    return(n * sign);
}

long long int fib(long long int n) {
	long long int i, j, k, h, t;
	i = h = 1;
	j = k = 0;
	while (n > 0) {
		if (n%2 == 1) {
			t = (j*h)%mod;
			j = (i*h + j*k + t)%mod;
			i = (i*k + t)%mod;
		}
		t = (h*h)%mod;
		h = (2*k*h + t)%mod;
		k = (k*k + t)%mod;
		n = n/2;
	}
	return j;
}

int main()
{
    long long int t = scan(), n;
    while (t--) {
        n = scan();
        printf("%lld\n", (fib(n)*fib(n+1))%mod);
    }
    return 0;
}
