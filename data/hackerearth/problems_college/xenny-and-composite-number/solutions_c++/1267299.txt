#include <string.h>
#define ll long long
using namespace std;


void s(long long  &x)
{register int c = getchar_unlocked();x = 0;for(;(c<48 || c>57);c=getchar_unlocked());for(;c>47&&c<58;c=getchar_unlocked()){x=(x<<1)+(x<<3)+c-48;}}

bool prime(long long x)
{
	bool ans=true;
	if(x==1)
	{
		ans=false;
	}
	else if(x==2)
	{
		ans=true;
	}
	else
	{
		for(int i=2;i<=sqrt(x);i++)
		{
			if(x%i==0)
			{
				ans=false;
				break;
			}
		}
	}
	return ans;
}


int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n;
		s(n);
		if(n==1 || n==0)
		{
			printf("4\n");
		}
		else if(n%2)
		{
			printf("%lld\n",(n+1));
		}
		else
		{
			if(prime(n+1))
			{
				printf("%lld\n",(n+2));
			}
			else
			{
				printf("%lld\n",(n+1));
			}
		}
	}
}
