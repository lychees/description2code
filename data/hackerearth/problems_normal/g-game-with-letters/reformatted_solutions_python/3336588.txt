from collections import Counter as cnt

for i in range(input()):
    st = raw_input().split()
    n = int(st[0])
    s = cnt(st[1])
    ans = []
    for j in range(n):
        ans.append(raw_input())
    fn = []
    for j in ans:
        fn.append(sum(s[k] for k in j))
    # print fn
    m = max(fn)
    l = float('inf')
    for j in range(n):
        if fn[j] == m:
            if len(ans[j]) < l:
                aa = j+1
                l = len(ans[j])
    print aa
