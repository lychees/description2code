#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t, l;
    char s[30];
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>s;
    	l=strlen(s);
    	for(int i=0;i<l;i++)
    	{
    		if(s[i]==s[i+1])
    		{
    			for(int j=i;j<l;j++)
    				s[j]=s[j+1];
    			l--;
    			i--;
    		}
    	}
    	for(int i=0;i<l;i++)
    		cout<<s[i];
    	cout<<endl;
    }
    
    return 0;
}
