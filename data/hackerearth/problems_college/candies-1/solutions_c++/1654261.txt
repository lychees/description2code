#include <iostream>
using namespace std;

int main() {
	int t,n;
	long long a[1001];
	cin>>t;
	while(t--){
		cin>>n;
		long long sum=0,flag=0;
		for(int i=0;i<n;i++)
		cin>>a[i];
		
		for(int i=0;i<n;i++)
		sum+=a[i];
		
		if(sum%n==0)
		flag=1;
		
		long long int student=sum/n;
		long long int teacher=sum-(n*student);
		
		if(flag==1)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
		cout<<student<<" "<<teacher<<endl;
		
	}
	return 0;
}
