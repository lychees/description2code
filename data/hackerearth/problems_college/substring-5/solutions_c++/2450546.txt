#include<sstream>
#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<vector>
using namespace std;
int main()
{
          string a;
          cin>>a;
          int l=a.length();
          int m=1,sum=l;
          for(int i=0;i<l;i++)
          {
                    if(a[i]==a[i+1])
                    {
                              sum+=m;
                              m++;
                    }
                    else
                              m=1;
          }
          cout<<sum;
          return 0;
}
