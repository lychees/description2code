#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--) {
    	long long int n,q;
    	cin>>n>>q;
    	vector<long long int> v(n);
    	long long int R = 0;
    	for(int i=0;i<n;i++)
    		cin>>v[i];
    	sort(v.begin(),v.end());
    	for(int i=1;i<n;i++) {
    		R = R + (v[i]-v[i-1])*q;
    	}
    	cout<<R<<endl;
    }
    return 0;
}
