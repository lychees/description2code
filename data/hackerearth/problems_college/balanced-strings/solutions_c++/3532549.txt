#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	string str;
    	cin >> str;
    	int n=str.length();
    	int c[26]={0};
    	for(int i=0;i<n;i++)
    	{
    		c[str[i]-97]++;

    	}
    	int k=c[0];
    	int flag=0;
    	for(int i=0;i<26;i++)
    	{
    		if(c[i]!=k)
    		{
    			flag=1;
    			//cout << i << endl;
    		}

    	}
    	if(flag==1)
    	cout << "No"  << endl;
    	else
    	cout << "Yes" << endl;

    }
    return 0;
}
