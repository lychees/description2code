#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int i,a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    int q,temp,sum,flag;
    cin>>q;
    while(q--)
    {
    
    	cin>>temp;
    	//cout<<temp<<endl;
    	for(i=0;i<n-1;i++)
    	{	
    		sum=0,flag=0;
       		for(int k=i+1;k<n;k++)
	    	{
	    		sum=0;
	    		sum+=a[i];
    			if(sum==temp)
    			{
    				flag=1;
    				break;
    			}
	    		else
	    		{
	    			sum+=a[k];
	    			if(sum==temp)
	    			{
	    				flag=1;
	    				break;
	    			}
	    		}
	    		//cout<<sum<<endl;
	      	}
	      	if(flag==1)
	      	{
	      		break;
	      	}
    	}
       	if(flag==1)
    	cout<<"I Like You"<<endl;
    	else
    	cout<<"I Hate You"<<endl;
    }
}
