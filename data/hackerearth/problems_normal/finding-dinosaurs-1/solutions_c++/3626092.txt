#include <iostream>
#include <map>
using namespace std;

#define DINO_NAME_LEN 10
char DinoNames[100000][DINO_NAME_LEN+1];

int main()
{
	int N, K, L, Q;
	int n, k, atr;
	unsigned int dkey;
	std::map<unsigned int,char*> dinoMap;
	std::map<unsigned int,char*>::iterator dinoIt;


	cin>>N>>K>>L>>Q;
	for(n=0; n<N; n++)
	{
		cin>>DinoNames[n];
		DinoNames[n][DINO_NAME_LEN]='\0';
		dkey=0;
		for(k=0; k<K; k++)
		{
			cin>>atr;
			dkey = dkey*100 + (atr+1);
		}
		dinoMap.insert(std::pair<unsigned int,char*>(dkey, DinoNames[n]));
	}
	for(n=0; n<Q; n++)
	{
		dkey=0;
		for(k=0; k<K; k++)
		{
			cin>>atr;
			dkey = dkey*100 + (atr+1);
		}
		dinoIt = dinoMap.find(dkey);
		if(dinoIt == dinoMap.end())
			cout<<"You cant fool me :P"<<endl;
		else
			cout<<dinoIt->second<<endl;
	}

    return 0;
}
