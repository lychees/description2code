#include <stdio.h>
#include<iostream>
using namespace std;
long long int greatestCommonDivisor(long long int m, long long int n)
{
    long long int r;

    /* Check For Proper Input */
    if((m == 0) || (n == 0))
        return 0;
    else if((m < 0) || (n < 0))
        return -1;

    do
    {
        r = m % n;
        if(r == 0)
            break;
        m = n;
        n = r;
    }
    while(true);

    return n;
}

int main(void)
{
    long long int num1, num2,t;
    cin>>t;
    while(t--)
    {
    	cin>>num1>>num2;
    	long long int gcd = greatestCommonDivisor(num1, num2); 

    //printf("The GCD of %d and %d is %d\n", num1, num2, gcd);
	if(gcd==1)
	printf("Is a Co-Prime\n");
	else
	printf("Not a Co-Prime\n");
	

    }
    return 0;
}
