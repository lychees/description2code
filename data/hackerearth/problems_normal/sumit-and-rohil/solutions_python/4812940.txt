size=int(raw_input())
myDict=dict()
while size>0:
    name=raw_input()
    key="".join(sorted(name))+name[0]+name[-1]
    if key not in myDict:
        myDict[key]=list()
        myDict[key].append(name)
    else:
        myDict[key].append(name)
    size-=1
