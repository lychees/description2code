#include<iostream>
#include<cstring>
using namespace std;
bool palin(string s1,int l)
{
	//cout<<l;
	int i,j;
	i=0;
	j=l-1;
	while(i<=j)
	{
		if(s1[i]!=s1[j])
			return false;
		i++;
		j--;
	}
	return true;
}
int main()
{
	int t,count,i,l;
	string s1,s2,s3;
	cin>>t;
	while(t--)
	{
		s3="";
		count=0;
		cin>>s1>>s2;
		l=s1.size();
		int l2=s2.size();
		for(i=0;i<=l;i++)
		{
			s3=s1.substr(0,i);
			s3+=s2+s1.substr(i,l);
			if(palin(s3,l+l2))
				count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
