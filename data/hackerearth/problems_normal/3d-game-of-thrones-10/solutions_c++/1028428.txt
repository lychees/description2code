#include <cstdio>

int A[52][52][52], T, N, res;
int i, j, k;

int main(void){
	for (int i = 0; i <= 50; i++)
	    for (int j = 0; j <= 50; j++)
	        for (int k = 0; k <= 50; k++){
	        	A[i][j][k] = 0;
	        	for (int l = 0; l < i; l++) if (A[l][j][k] == 0) A[i][j][k] = 1;
	        	for (int l = 0; l < j; l++) if (A[i][l][k] == 0) A[i][j][k] = 1;
	        	for (int l = 0; l < k; l++) if (A[i][j][l] == 0) A[i][j][k] = 1;
	        }
	scanf("%d", &T);
	while (T--){
		scanf("%d", &N);
		for (int l = 0; l < N; l++){
			scanf("%d %d %d", &i, &j, &k);
			if (l == 0) res = i^j^k;
			else res = res^i^j^k;
		}
		if (res) printf("Stark\n");
		else printf("Lannister\n");
	}
	return 0;
}
