#include <bits/stdc++.h>
using namespace std ;
#define MAX 1000000000

bool chk(int n){
    int k ;
    while(n){
        k = n%10;
        n /= 10 ;
        if(k&1) return false ;
    }
    return true ;
}

vector<int>v ;

int main(){
    v.push_back(0) ; v.push_back(2) ;
    v.push_back(4) ; v.push_back(6) ; v.push_back(8) ;

    int sz = 5, i , u = 1 ;
    while(v[sz-1] < MAX){
        for(i = 0 ; i < 9 ; i+=2){
            v.push_back(v[u]*10+i) ;
            sz++;
        }
        u++ ;
    }

    int T , n , x , y  , z ;
    scanf("%d" , &T);
    while(T--){
        scanf("%d" , &n);
        if(chk(n)){
            printf("Unlimited Power\n") ;
        }
        else{
            x = v[upper_bound(v.begin(),v.end(),n) - v.begin()] ;
            y = v[upper_bound(v.begin(),v.end(),n) - v.begin() - 1] ;
            x = x - n ;
            y = n - y ;
            z = __gcd(x,y) ;
            x /= z ;
            y /= z ;
            printf("%d/%d\n",x,y) ;
        }
    }
}
