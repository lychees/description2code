int rec(int a[], int n,int curr, int x) {
    priority_queue<int>q;

    for(int i=0;i<n;i++) {
        q.push(a[i]);
    }

    q.push(0);
    int time=0;

    while(1) {
        int u = q.top();

        if(u == 0) {
            break;
        }
        q.pop();

        int v = q.top();

        if(v == 0) {
            time+=u;
            break;
        }

        q.pop();

        time+=v;
        u-=v;
        q.push(u);
    }

    if(time<=x) {
        return 1;
    } else {
        return 0;
    }
}
