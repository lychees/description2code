#include<bits/stdc++.h>
using namespace std;
long long gcd(long long a, long long b)
{
	
    if(b==0)
 	return a;
	else
	return gcd(b,a%b);
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long e,a,b,c,d,f;
		long long flag=0LL;
		cin>>a>>b>>c;
		flag=gcd(gcd(a,b),c);
	//	cout<<flag;
		
		d=a*a + b*b;
		e=b*b+c*c;
		f=a*a+c*c;
		if((d==c*c || e==a*a || f==b*b)&&flag==1)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
