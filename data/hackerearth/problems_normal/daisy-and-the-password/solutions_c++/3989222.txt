#include <iostream>
#include <string>
#include<cstring>
using namespace std;

int main()
{
	string s1,s2;
	int n,count,flag,l1,l2,i,j;
	cin>>n;
	for(int k=1;k<=n;k++)
	{count=0;
		cin>>s1;
	l1=2*(s1.size());
	cin>>s2;
	l2=s2.size();
	s1=s1+s1;
	for(i=1;i<=l1;i++)
	{
		flag=0;
		for(j=1;j<=l2;j++)
		{
			if(s1[i]==s2[j])
			{
				count++;
				s2[j]=0;
				flag=1;
			}
			if(flag==1) break;
		}
	}
	if(l1==l2 && count==l1) cout<<"Possible"<<endl;
	else cout<<"Impossible"<<endl;
	}
    return 0;
}
