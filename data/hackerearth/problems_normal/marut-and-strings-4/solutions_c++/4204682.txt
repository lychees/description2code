#include <iostream>
using namespace std;

int main()
{
    string s;
    int t;
    cin>>t;
    if(t<1||t>10)
    {
    cout<<"Invalid Test";
    exit(0);
}

    while(t--)
    {
    	cin>>s;
    	if(s.length()>100)
    	{
		cout<<"Invalid Input";
		exit(0);
	}
    	int count1=0;
    	int count2=0;
    	int count=0;
    	for(int i=0;i<s.length();i++)
    	{
    		  	 	
    	 if(s[i]>='a'&&s[i]<='z')
    	{
    	//	cout<<count1;
			count1++;
		}
	
     	else if(s[i]>='A'&&s[i]<='Z')
		{
			//cout<<count2;
			count2++;
		}
		else if(!(s[i]>='a'&&s[i]<='z')||!(s[i]>='A'&&s[i]<='Z'))
    		{
    		 /*   cout<<"hello"<<endl;
				cout<<count;*/
				count++;
    }
	}
	if(count==s.length())
    	cout<<"Invalid Input"<<endl;
	else if(count1<count2)
	cout<<count1<<endl;
	 
    	else
    	cout<<count2<<endl;
    	
    }
    return 0;
}
