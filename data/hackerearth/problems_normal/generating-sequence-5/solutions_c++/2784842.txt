#include <iostream>
using namespace std;

int main()
{
	long long T,g,N;
    cin>>T;
    while(T--)
    {
    	cin>>g>>N;
    	if(g == 1)
    	{
    		int num = 2;
    		while(N > 1)
    		{
    			cout<<num<<" ";
    			N--;
    		}
    		cout<<++num;
    		cout<<endl;
    	}
    	else
    	{
    		while(N > 1)
    		{
    			cout<<g*2<<" ";
    			N--;
    		}
    		cout<<g*3<<endl;
    	}
    }
    return 0;
}
