#include <iostream>
#include<math.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int N;
    	cin>>N;
    	
    	int k,count=0,i,j,a,b,c;
    	for (i = 2; i <=(sqrt(N-1) + 1); i++)
          for (j = 1; j < i; j++)
              if ((gcd(i,j) == 1) && ((i-j) % 2) && ((i*i + j*j) <= N))
              {
                    k = 0;
                    a = i * i - j * j;
                    b = 2 * i * j;
                    c = i * i + j * j;
                    while ((++k) * c <= N)
                      count++;
              }
    	cout<<count<<"\n";
    }
    return 0;
}
