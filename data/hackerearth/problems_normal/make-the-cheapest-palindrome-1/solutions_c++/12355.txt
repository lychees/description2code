#include <iostream>
#include <string>
using namespace std;
int main()
{
	int num;
	string s;
	int a, b;
	cin>>num;
	for (int t=0; t<num; t++)
	{
		cin>>s;
		cin>>a;
		cin>>b;
		int ret = 0;
		bool fl = 0;
		for(int i=0,j=s.size()-1;i<j;i++,j--) 
		{ 
			if(s[i]=='b'&&s[j]=='/') ret+=b; 
			else if(s[i]=='/'&&s[j]=='b') ret+=b; 
			else if(s[i]=='a'&&s[j]=='/') ret+=a; 
			else if(s[i]=='/'&&s[j]=='a') ret+=a; 
			else if(s[i]=='/'&&s[j]=='/') 
			{ 
			    int mn=min(b,a); 
			    ret+=2*mn; 
			} 
			else if(s[i]==s[j]) ; 
			else {fl=1; break;} 
		} 
		if(fl) {
			cout<<-1<<endl; 
		} else {
			cout<<ret<<endl; 
		}
	}
    return 0;
}
