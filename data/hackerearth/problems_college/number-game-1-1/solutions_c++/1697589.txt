#include<iostream>
using namespace std;
int main()
{
	int tc=0;
	cin>>tc;
	while(tc-->0)
	{
		long int n=0;
		cin>>n;
		int count=0;
		while(n!=1)
		{
			if(n%3==0)
			{
				n=n/3;
			}
			else if(n%2==0)
			{
				n=n/2;
			}
			else
			{
				n=n-1;
			}
			count++;
		}
		cout<<count<<endl;
	}
	return 0;
}
