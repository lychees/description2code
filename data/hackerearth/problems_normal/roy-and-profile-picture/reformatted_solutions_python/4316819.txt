L = int(input())
N = int(input())
for i in range(N):
    ip = raw_input()
    IP = ip.split()
    if (int(IP[0]) < L or int(IP[1]) < L):
        print "UPLOAD ANOTHER"
    elif (int(IP[0]) >= L and int(IP[1]) >= L):
        if (int(IP[0]) == int(IP[1])):
            print "ACCEPTED"
        else:
            print "CROP IT"
