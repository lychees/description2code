#include <iostream>
#include <cstdio>
using namespace std;
char finalstr[200000];
int main()
{
	int t,len,i,j,k,cnt,l;
	char str[50];
	scanf("%d",&t);
	while(t--){
		scanf("%s",str);
		len=0;
		j=0;
		for(i=0;str[i]!='\0';i++)
		{
			if(str[i]>='0'&&str[i]<='9'){
				cnt=str[i]-'0';
				for(k=0;k<j;k++)
				{
					for(l=0;l<cnt-1;l++){
						finalstr[j*(l+1)+k]=finalstr[k];
					}
				}
				j=cnt*j;
			}
			else{
				finalstr[j]=str[i];
				j++;
			}
		}
		finalstr[j]='\0';
		printf("%s\n",finalstr);
	}
    return 0;
}
