#include <iostream>
using namespace std;


int F(int N, int P)
{
	if(P==0)
		return N;
	if(N<=3)
	    return 3;
	if(N%2==0){
	 	N=(N/2)+1;P--;}
	else{
		N=((N+1)/2)+1;P--;}
	if(P==0)
		return N;
	if(N<=3)
	    return 3;
	else
		return F(N,P);;
}

int main()
{
    int Test;
    scanf("%d",&Test);
    while(Test--)
    {
    	int N,P;
    	scanf("%d%d",&N,&P);
    	int Res=F(N,P);
    	printf("%d\n",Res);
    }
    return 0;
}
