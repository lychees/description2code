#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,i,q,x,y,min=0,max=0,c,f=0;
	cin>>n>>q;
	char a[123456];
	scanf("%s",a);
	int b[133333];
	for(i=1;i<=n;i++)
	b[i]=a[i-1]-48;
	for(i=0;i<q;i++){
		f=0;
		cin>>x>>y;
	
				if(x>=y){
			max=x;
			min=y;
		}
		else {
			min=x;
			max=y;
		}
		c=(floor(log(max)/log(2))-floor(log(min)/log(2)));
		max=max/((int)pow(2,c));
	//	cout<<c<<endl;
		while(max!=min){
			max/=2;
			min/=2;
		}
		while(x!=max){
			if(b[x]==1){
				f=1;
				break;
			}
			x=x/2;
			
		}
			while(y!=max){
			if(b[y]==1){
				f=1;
				//cout<<b[y]<<"k";
				break;
			}
			y=y/2;
			
		}
		if(x==y&&b[x]==1){
		printf("No\n");
		continue;
		}
		if(x==y&&b[x]==0){
			printf("Yes\n");
			continue;
		}
		if(f==1)
		printf("No\n");
		else
		printf("Yes\n");
	}
	return 0;
}
