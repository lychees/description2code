#include <iostream>
using namespace std;
int co1[100005][26],co2[100005][26];
char a[100005],b[100005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%s%s",&a,&b);
    	int n,m;
    	for(n=0;a[n];n++);
    	for(m=0;b[m];m++);
    	int i,j;
    	for(i=1;i<=n;i++)
    	{
    		for(j=0;j<26;j++)
    		co1[i][j] = co1[i-1][j];
    		co1[i][a[i-1]-'a']++;
    	}
    		for(i=1;i<=m;i++)
    	{
    		for(j=0;j<26;j++)
    		co2[i][j] = co2[i-1][j];
    		co2[i][b[i-1]-'a']++;
    	}
    	int found = 0;
    	for(i=1;i<=m;i++)
    	{
    		j = i+n-1;
    		if(j>m)break;
    	//	printf("from %d to %d\n",i,j);
    		int k;
    		int valid = 1;
    		for(k=0;k<26;k++){
    			int cc1,cc2;
    			cc1 = co1[n][k] - co1[0][k];
    			cc2 = co2[j][k] - co2[i-1][k];
    			if(cc1||cc2)
    			{
    			//	printf("%c is present in %d in first and %d in second\n",k+'a',cc1,cc2);
    			}
    		if(co2[j][k]-co2[i-1][k]!=co1[n][k]-co1[0][k]){
    			valid=0;
    			break;
    		}
    		}
    		
    		if(valid){
    			found=1;
    			break;
    		}
    	}
    	if(found)printf("YES\n");
    	else printf("NO\n");
    }
    return 0;
}
