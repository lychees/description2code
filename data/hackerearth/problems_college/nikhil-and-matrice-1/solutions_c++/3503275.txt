#include<bits/stdc++.h>
#define ll long long 
using namespace std;
int main()
{
    //	freopen("in.txt","r",stdin);
   int t;
   cin>>t;
   while(t--)
   
  {
  	int n;
  	cin>>n;
  	int arr[n+1][n+1];
    int c=1;
   int top=0,down=n-1,left=0,right=n-1;
    while(true)
     {
         // Print top row
         for(int j = left; j <= right; ++j) arr[top][j]=c++;
         top++;
         if(top > down || left > right) break;
         //Print the rightmost column
         for(int i = top; i <= down; ++i) arr[i][right]=c++;
         right--;
         if(top > down || left > right) break;
         //Print the bottom row
         for(int j = right; j >= left; --j)arr[down][j]=c++; ;
         down--;
         if(top > down || left > right) break;
         //Print the leftmost column
         for(int i = down; i >= top; --i) arr[i][left]=c++ ;
         left++;
         if(top > down || left > right) break;
     }
 for(int i=0;i<n;i++)
 {
 	for(int j=0;j<n;j++)
 	 cout<<arr[i][j];
 	  cout<<endl;
 }
 cout<<endl;
  }
   		return 0;
}
