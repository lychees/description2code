#include <iostream>
using namespace std;

int ans[50000001];

int f(int x) {
	if (x % 2 == 0) return ans[x >> 1];
	return (ans[(x - 1) >> 1] + 2) % x;
}

int main()
{
	int bef = 0;
	for (int i = 2; i <= 100000000; ++i) {
		int cur = (bef + 2) % i;
		if (i % 2 == 0) ans[i >> 1] = cur;
		bef = cur;
	}
    int t, x;
    cin >> t;
    while (t--) {
    	cin >> x;
    	cout << f(x) + 1 << '\n';
    }
    return 0;
}
