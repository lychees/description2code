#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,x;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int v[n],ar[n];
		for(i=0;i<n;i++)
            cin>>v[i],ar[i]=v[i];
		sort(ar,ar+n);
		int s,e=0;
		bool cnd=false;
		for(i=1;i<n;i++)
        {
            if(!cnd && v[i]<v[i-1])
            s=i,cnd=true;
            if(cnd && v[i]>v[i-1])
            {
                e=i;
                break;
            }
        }
        if(cnd && !e)
            e=n,cnd=true;
        if(cnd)
        reverse(v+s-1,v+e);
        for(i=0;i<n && cnd;i++)
        if(v[i]!=ar[i])
            cnd=false;
        if(cnd)
            cout<<s<<" "<<e<<endl;
        else
            cout<<-1<<" "<<-1<<endl;
	}
    return 0;
}
