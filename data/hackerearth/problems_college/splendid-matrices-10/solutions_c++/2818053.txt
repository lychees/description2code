#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

vector<vector<int> > board;

void build(int n, int row, int col)
{
	static int curr=1;
	if(n==1)
	{
		board[row][col]=curr++;
		board[row][col+1]=curr++;
		board[row+1][col]=curr++;
		board[row+1][col+1]=curr++;
	}
	else
	{
		int p=pow(2,n-1);
		build(n-1, row, col);
		build(n-1, row, col+p);
		build(n-1, row+p, col);
		build(n-1, row+p, col+p);
	}
	return;
}

int main()
{
    int n;
    cin>>n;
    long long int p=pow(2,n);
    for(int i=0; i<p; i++)
    {
    	vector<int> temp(p,0);
    	board.push_back(temp);
    }
    build(n,0,0);
    for(int i=0; i<p; i++)
    {
    	for(int j=0; j<p; j++)
    	{
    		cout<<board[i][j]<<' ';
    	}
    	cout<<endl;
    }
    return 0;
}
