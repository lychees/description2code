n = int(raw_input())
s = raw_input()
s = s.split(' ')
t = [0 for i in range(n)]
for i in range(n):
    t[i] = int(s[i])

t = sorted(set(t))
if len(t) <= 2:
    print "NO"
else:
    for i in range(2, len(t)):
        if t[i - 1] == t[i] - 1 and t[i - 2] == t[i] - 2:
            print "YES"
            break
    else:
        print "NO"
