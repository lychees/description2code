def sieve(n):
	n = n+1
	sieve = [True]*n
	m = int(n**0.5)+1
	d = 2
	for i in xrange(4,n,2):
		sieve[i] = False
	for i in xrange(3,m,2):
		for j in xrange(i*i, n, i):
			sieve[j] = False
	sieve[0] = sieve[1] = False
	return sieve

ary = sieve(10**7)
