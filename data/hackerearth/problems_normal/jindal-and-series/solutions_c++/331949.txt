#include<iostream>
#define MOD 10000007
using namespace std;
int main(){
    long long t,n,possible[1000000];
    cin>>t;
    possible[0]=possible[1]=1;possible[2]=2;possible[3]=4;
    for(long long i=4;i<=1000000;i++){
        possible[i]=(possible[i-1]+possible[i-4]+possible[i-3])%MOD;
    }    
    while(t--){
        cin>>n;
        cout<<possible[n-1]<<endl;
    }    
    return 0;
}
