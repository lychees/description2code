#include <iostream>
using namespace std;

int palin(string s,int a)
{   int l=s.length(),i;
    for(i=a;i<l;++i)
        if(s[i]!=s[l-i-1+a])
            return 0;
    return 1;
}

int main()
{   int n;
    cin>>n;
    while(n--)
    {   string s;
        int l,k,i;
        cin>>s;
        l=s.length();
        for(i=0;i<l;++i)
            if(palin(s,i))
            {   cout<<l+i;
                k=1;
                break;
            }
        if(k==0)
            cout<<2*l-1;
        cout<<"\n";
    }
    return 0;
}
