#include<bits/stdc++.h>
using namespace std;
#define vi vector<int>
int main()
{
	int i, t, n, val, j;
	char ch;
	string s;
	cin >> t;
	while( t--)
	{
		cin >> s;
		n = s.length();
		vi v(26);
		for( i=0;i<n;i++ )
		{
			v[s[i]-'A']++;
		}
		for( i=0;i<26;i++ )
		{
			val = i+65;
			ch =(char)val;
			for( j=1;j<=v[i];j++ )
			{
				cout << ch;
			}
		}
		cout << "\n";
	}
	return 0;
}
