#include <iostream>
using namespace std;

int main()
{
    long long int t,result=0;
    cin>>t;
    while(t--){
    	long long int A[5];
    	result=0;
    	for(int i=0; i<5; i++){
    		cin>>A[i];
    	}
    	for(int k=A[0];k<=A[1];k++){
			if(k+A[4]<=A[2])
				result=k;
			//cout<<result;
		}
    	if(result!=0)
    		cout<<result<<endl;
    	else
    		cout<<"STAY IN LONDON"<<endl;
    }
    return 0;
}
