#include <iostream>
#include<cstring>
using namespace std;

int main()
{
	
	long int n,i=0,j,k,count=0,count1=0;
	string s,s1,s2,s3,s4;
	cin>>n;
	while(n--)
	{
		count1=0;
	  cin>>s;
	  s3=s;
	  j=s.length();
	  cin>>s1;
	  for(i=0;i<=j;i++)
	  {                                                   
	  s2=s.insert(i,s1);           
     for(k=0;k<s2.length();k++)
     {
       s4[k]=s2[s2.length()-k-1];	
	 }
     count=0;
	 for(k=0;k<s2.length();k++)
	 {
	 	if(s4[k]==s2[k])
	 	{
	 		count++;
		 }
     }
     if(count==s2.length())
     {
     	count1++;
	 }
	 
	  
	  
	  s=s3;
	  
	  }
	  cout<<count1<<endl;
	}
}
