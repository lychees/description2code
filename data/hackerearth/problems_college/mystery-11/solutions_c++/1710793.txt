#define vll vector<long long int>

typedef long long int ll;

using namespace std;

ll solve(string s)
{
    ll ans=0,i;
    for(i=0;s[i]!='\0';i++)
    {
        switch(s[i])
        {
            case '0':ans+=6;break;
            case '1':ans+=2;break;
            case '2':ans+=5;break;
            case '3':ans+=5;break;
            case '4':ans+=4;break;
            case '5':ans+=5;break;
            case '6':ans+=6;break;
            case '7':ans+=3;break;
            case '8':ans+=7;break;
            case '9':ans+=6;break;
        }
    }

    return ans;
}

int main()
{
    ll t;
    scanf("%lld",&t);

    while(t--)
    {
        char s[10000];
        cin>>s;

        printf("%lld\n",solve(s));
    }

    return 0;
}
