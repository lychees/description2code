#include <iostream>
using namespace std;
# define PR 1000000007
int main()
{
   int t;
   cin>>t;
   long long n;
   string k;
   for (int i=0;i<t;i++)
    { cin>>n;
    cin>>k;
    long long sum=0;
    for (int j=0;j<n;j++)
    {
    	sum=((sum*10)%PR+k[j]-'0')%PR;
    }
    	
    	cout<<sum<<endl;
    }
    return 0;
}
