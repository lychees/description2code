#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin >> t;
   string  s;
   while(t--){
   		cin >> s;
   		int ans = 0;
   		int count = 0;
   		for(int i = 0;i < s.size();i++){
   			if(s[i] == 'A'||s[i]=='a'||s[i]=='B'||s[i]=='b'||s[i]=='C'||s[i] =='c'){
   				count++;
   			}else{
   				if(count > 1)ans += (count*(count+1))/2;
   				else if(count == 1)ans ++;
   				count = 0;
   			}
   		}
   		if(count > 1)ans += (count*(count+1))/2;
   		else if(count == 1)ans++;
   		cout<<ans<<endl;
   }
    return 0;
}
