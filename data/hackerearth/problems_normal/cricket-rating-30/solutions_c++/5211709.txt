#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a;
    int max=0;
    int sum=0;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	if(sum+a>0)
    	{
    		sum+=a;
    	}
    	else if(sum+a<0)
    	{
    		sum=0;
    	}
    	if(sum>max)
    	{
    		max = sum;
    	}
    }
    cout<<max;
    return 0;
}
