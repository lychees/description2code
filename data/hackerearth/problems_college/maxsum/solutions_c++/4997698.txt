#include <iostream>

using namespace std;
int M[10005];
int A[10005];
int maxi(int a,int b)
{
    if (a>b)
        return a;
    else
    return b;
}
int main()
{
    int T,i,n;
    cin>>T;
    //T=1;
    while(T--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>A[i];
        }

        {
            M[0]=A[0];
            M[1]=max(A[1],A[0]);
            for(i=2;i<n;i++)
            {
               if((i-2)<0)
               {
                   M[i]=maxi(A[i],M[i-1]);
               }
               else
                M[i]=maxi(M[i-2]+A[i],M[i-1]);
            }
            cout<<maxi(M[n-1],M[n-2])<<endl;

        }


    }


    return 0;
}
