#include <bits/stdc++.h>
using namespace std;

#define FOR(i, a, b) for (int (i) = (a); (i) < (b); (i)++)
#define REP(i, n) FOR(i, 0, n)

typedef long long LL;

LL solve(LL k) {
	if (k == 0) return 0;
	LL t;
	for (t = 1; t * (t + 1) / 2 <= k; t++); t--;
	LL ans = 0;
	int diff = k - t * (t + 1) / 2;
	if (k - t * (t + 1) / 2 != 0) t++;
	
	//cout << "here " << t << " " << diff << endl;
	
	if (diff == 0) {
		if (t % 2 != 0) ans += t;
	} else if (diff % 2 != 0) ans += t;
	ans += (t / 2) * (t / 2);
	return ans;
}

int main() {
	int T;
	scanf("%d", &T);
	while (T--) {
		LL a, b;
		scanf("%lld %lld", &a, &b);
		LL ans = solve(b) - solve(a - 1);
		printf("%lld\n", ans);
	}
	return 0;
}
