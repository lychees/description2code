#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> finance;
    vector<int> marki;
    for(int i=0; i<n; i++)
    {
        int k;
        cin>>k;
        finance.push_back(k);
    }
    for(int i=0; i<n; i++)
    {
        int k;
        cin>>k;
        marki.push_back(k);
    }

    sort(finance.begin(),finance.end(),greater<int>());
    sort(marki.begin(),marki.end(),greater<int>());

    int mi,fi=0;
    int wins=0,draws=0;
    for(int i=0; i<marki.size(); i++)
        if(finance[0]>marki[i])
        {
            mi = i;
            break;
        }
    while(fi<finance.size() && mi<marki.size())
    {
        if(finance[fi] == marki[mi])
        {
            fi++;
            mi++;
            draws++;
        }
        else if(finance[fi] > marki[mi])
        {
            fi++;
            mi++;
            wins++;
        }
        else
            mi++;
    }

    cout<<(wins*100)+(draws*50);

    return 0;
}
