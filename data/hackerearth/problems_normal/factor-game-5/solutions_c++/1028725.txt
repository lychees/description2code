#include <iostream>

#define MAXN 100005

using namespace std;

int n, r, a[MAXN], p[MAXN];

int main() {
  //freopen("", "r", stdin);
  //freopen("", "w", stdout);
  //ios_base :: sync_with_stdio(0);

  for (int i = 2; i < MAXN; i++)
    if (p[i] == 0)
      for (int j = i; j < MAXN; j += i) p[j] = i;
  cin >> n;
  r = 0;
  for (int i = 0; i < n; i++) {
    int x;
    cin >> x;
    int d = 0;
    while (x > 1) {
      x /= p[x];
      d++;
    }
    r ^= d;
  }

  if (r)
    cout << "ALICE";
  else
    cout << "BOB";
}
