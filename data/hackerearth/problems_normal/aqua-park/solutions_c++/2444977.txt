#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=100;
pair<int,int>lft[mx];
pair<int,int>rht[mx];

int main(){
   int n;
   int x;
   cin>>x>>n;
   for(int i=0;i<n;i++){
      cin>>lft[i].first>>lft[i].second;
      cin>>rht[i].first>>rht[i].second;
      if(lft[i]>rht[i])swap(lft[i],rht[i]);
   }
   double h=1e7;
   while(true){
      int idx=-1;
      double best=-1.0;
      for(int i=0;i<n;i++){
         if(lft[i].first<=x && x<=rht[i].first){
            //(x-lft.x)/dy=(r.x-l.x)/(r.y-l.y)
            double num=(x-lft[i].first)*(rht[i].second-lft[i].second);
            double den=rht[i].first-lft[i].first;
            double hi=lft[i].second+num/den;
            if(hi<h && hi>best){
               best=hi;
               idx=i;
            }
         }
      }
      if(idx==-1)break;
      if(lft[idx].second<rht[idx].second)x=lft[idx].first;
      else x=rht[idx].first;
      h=best;
   }
   printf("%d\n",x);
   return 0;
}
