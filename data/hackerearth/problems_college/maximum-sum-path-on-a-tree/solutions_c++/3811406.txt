#include<bits/stdc++.h>
#define scd(x) scanf("%d",&x)
#define prd(x) printf("%d",x)
#define sclld(x) scanf("%lld",&x)
#define prlld(x) printf("%lld",x)
#define f_in(x) freopen(x,"r",stdin)
#define f_out(x) freopen(x,"w",stdout)

using namespace std;

typedef long long int llt;
vector< vector<int> > graph;
vector<llt> fee;
int n;
llt ans;
llt travel(int i,int parent)
{
    if(graph[i].size() == 1 && graph[i][0] == parent)
    {
        ans = max(ans,fee[i]);
        return fee[i];
    }
    int len = graph[i].size();
    llt first=0,second=0,temp;
    for(int j = 0 ; j< len ; j++)
    {
        if(graph[i][j]==parent)
            continue;
        temp = travel(graph[i][j],i);
        if(temp >= first)
        {
            second = first;
            first = temp;
        }else if(temp > second)
        {
            second = temp;
        }
    }
    ans = max(ans,fee[i] + first + second );
    return first + fee[i];
}

int main()
{
    ios::sync_with_stdio(false);
    int t,x,y;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ans = 0;
        fee.assign(n+1,0);
        graph.resize(n+1);
        for(int i = 1 ; i <= n ; i++)
            cin>>fee[i];
        for(int i = 1 ; i<n; i++)
        {
            cin>>x>>y;
            graph[x].push_back(y);
            graph[y].push_back(x);
        }
        travel(1,-1);
        cout<<ans<<"\n";
        fee.clear();
        graph.clear();
    }

    return 0;
}
