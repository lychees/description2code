int a[N],pr[N],ne[N];

    for(i=1;i<=n;i++){

        mx=max(mx,a[i]);
        pr[i]=max(a[i],pr[i-1]);
    }
    for(i=n;i>=1;i--)ne[i]=max(ne[i+1],a[i]);
    int len=0,pos=0;
    for(i=1;i<=n;i++)if(a[i]==mx){
        j=i+1;k=1;
        while(j<=n&&a[j]==a[i])j++,k++;
        if(k>len)len=k,pos=i;
        i=j;
    }
    int mi=a[pos],l=pos,r=pos;

    for(i=1;i<n;i++){
        if(pr[l-1]>ne[r+1]||(pr[l-1]==ne[r+1]&&a[l-1]>=a[r+1])){
            mi=min(mi,a[--l]);
