#include <iostream>
using namespace std;

int current = -1;
const int size = 100000;
void push(int[],int,int);
void pop(int[]);

int main()
{
    int T=0,N=0,ID=0,st[size]={0};
    char choice;
    cin >> T;
    for(int i = 0; i < T; i++)
    {
    	 current = -1;
        cin >> N >> ID; // first guy passes to the others id
        st[++current] = ID;
        for(int k = 0; k < N; k++)
        {
            cin >> choice;

            switch(choice)
            {
            case 'P':
                push(st,ID,size);
                break;
            case 'B':
                pop(st);
                break;
            }
        }
        cout << "Player " << st[current] << endl;
    }
    return 0;
}


void push(int a[size],int id,int n)
{
    if(current < n - 1)
    {
    cin >> id;
    a[++current] = id;
    }
}


void pop(int a[size])
{
    if(current > 0)
    {
        int temp=0;
        temp = a[current];
        a[current] = a[current-1];
        a[current - 1] = temp;

    }
}
