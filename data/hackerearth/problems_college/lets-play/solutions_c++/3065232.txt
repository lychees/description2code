typedef long long ll;

const int N = (int)1e5 + 123;
const int INF = (int)2e9 + 123;
const double eps = 1e-6;
int n, k;
struct az {
	int x, y, res;
}d[N];
pair<int, int> a[N];
int calc(int v)
{
	return a[v].F * a[v].S;
}
int main()
{
	ios_base::sync_with_stdio(0);cin.tie(NULL);
//	freopen(fname".in","r",stdin);
//	freopen(fname".out","w",stdout);
	cin >> n >> k;
	for (int i = 1; i <= n;i ++)
	{
		cin >> a[i].F >> a[i].S;
		d[i].res = INF;
	}
	d[n + 1].res = INF;
	d[1].res = calc(1);
	d[1].x = a[1].F;
	d[1].y = a[1].S;
	for (int i = 2; i <= n + 1;i ++)
	{
		for (int j = i - 1; j >= max(0, i - k); j --)
		{
			if (d[i].res > max(a[i].F, d[j].x) * max(a[i].S, d[j].y))
			{
				d[i].res = max(a[i].F, d[j].x) * max(a[i].S, d[j].y);
				d[i].x = max(a[i].F, d[j].x);
				d[i].y = max(a[i].S, d[j].y);
			}
		}
	}
	cout << d[n + 1].res << "\n";
 	return 0;
}
