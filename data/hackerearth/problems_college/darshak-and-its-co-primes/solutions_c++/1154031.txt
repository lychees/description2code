#include<iostream>
using namespace std;
long long gcd(long long a,long long b)
{

    if(b == 0) {
            return a;
    }
    else {
        return gcd(b, a % b);
    }
}


int main()
{
	long long t,p,q,k;
	cin>>t;
	while(t!=0)
	{
		cin>>p>>q;
		if(p>q)
		k = gcd(p,q);
		else
		k = gcd(q,p);
		if(k!=1)
		cout<<"Not a Co-Prime";
		else
		cout<<"Is a Co-Prime";
		cout<<"\n";
		t--;
	}
	return 0;
}
