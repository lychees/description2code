#include <iostream>
#include<cstring>
using namespace std;
#define MAX 1001
int main() {
	// your code goes here
	int a[26],i,t;
	cin>>t;
	while(t-->0)
	{
	char s[MAX],t[MAX];
	for(i=0;i<26;++i)
	a[i]=0;
	
	cin>>s>>t;
	int c=0;
	int flag=0;
	for(i=0;i<strlen(t);++i)
	{
		++a[t[i]-97];
	}
	
	
	for(i=0;i<strlen(s);++i)
	{
		if(a[s[i]-97]==0)
		{
			flag=1;
			break;
		}
		else
		{
		--a[s[i]-97];
		}
	}
	
	
	if(!flag)
	{
	for(i=0;i<26;++i)
	{
		if(a[i]!=0)
		++c;
	}
	cout<<c<<endl;
	}
	else
	cout<<"-1"<<endl;
	}
	return 0;
}
