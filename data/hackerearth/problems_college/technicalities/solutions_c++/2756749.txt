using namespace std;
const long long MOD = 1000000007;
char s[1000007];
int main()
{
//	freopen("input.txt", "r", stdin);
//	freopen("output.txt", "w", stdout);
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++)
	{
		int n;
		scanf("%d", &n);
		scanf("%s", s);
		long long ans = 0;
		for (int i = 0; i < n; i++)
		{
			ans = (10 * ans) + s[i] - '0';
			ans %= MOD;
		}
		printf("%lld\n", ans);
	}
}
