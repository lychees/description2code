#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,k,p,arr[100005]={0},ans=0;
		scanf("%d%d%d",&n,&k,&p);
		for(int i=1;i<=k;i++)
		scanf("%d",&arr[i]);
		sort(arr+1,arr+k+1);
		for(int i=1;i<=k;i++)
		{
			if(p==0)
			break;
			if(arr[i]-arr[i-1] > p)
			{
				ans+=p;
				p=0;
				break;
			}
			else
			{
				p-=(arr[i]-arr[i-1]-1);
				ans=arr[i];
			}
		}
		if(n-arr[k] >= p)
		{
		ans+=p;
		p-=(n-arr[k]);
		}
		if(p>0)
		printf("-1\n");
		else
		printf("%d\n",ans);
	}
return 0;
}
