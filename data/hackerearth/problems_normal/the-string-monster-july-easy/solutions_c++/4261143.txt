#include<bits/stdc++.h>
typedef long long ll;
ll mod =1000000007;
using namespace std;
int n;
int curr[26];
char str[40][40];
char key[1000];
int has[35][26];
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
bool func(int i)
{
	int j;
	if(i==n)
		return false;
	int fl=0;
	if(func(i+1))
		return true;
	for(j=0;j<26;j++)
	{	
		curr[j]=curr[j]-has[i][j];
		if(curr[j]<0)
		{
			while(j!=-1)
			{
				curr[j]+=has[i][j];
				j--;
			}	
			return false;
		}
	}
	for(j=0;j<26;j++)
	{
		if(curr[j]!=0)
		{	
			bool sol= func(i+1);
			for(j=0;j<26;j++)
			{
				curr[j]=curr[j]+has[i][j];
					
			}
			return sol;
		}
	}
	return true;

}
int main()
{
	int i,j,fl,test;
	scanf("%d",&test);
	while(test--)
	{	memset(curr,0,sizeof(curr));
		memset(has,0,sizeof(has));
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%s",str[i]);
			for(j=0;str[i][j];j++)
				has[i][str[i][j]-'a']++;
		}
		scanf("%s",key);
		for(i=0;key[i];i++)
		{
			curr[key[i]-'a']++;
		}
		if(func(0))
		{
			printf("YES\n");
		}
		else
			printf("NO\n");
	}
	return 0;
}
