#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		unsigned long long n,sum=0,temp;
   		cin>>n;
   		while(n>0)
   		{
   			sum+=n%10;
   			n=n/10;
   		}
   		//cout<<sum<<endl;
   		while(sum>9)
   		{
   			temp=sum;
   			sum=0;
   			while(temp>0)
   			{
   				sum+=temp%10;
   				temp/=10;
   			}
   			//cout<<" sum : "<<sum<<endl;
   			if(sum<=9)
   			break;
   		}
   		cout<<sum<<endl;
   	}
    return 0;
}
