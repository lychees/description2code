#include <bits/stdc++.h>
using namespace std;

int main()
{
	
	long tc,x,y;
	cin>>tc;
	while(tc--)
	{
		long n;
		cin>>n;
		map<long,long>hash;
		long curr=0,max=0;
		while(n--)
		{
			cin>>x>>y;
			if(y==0)
			{
				curr=0;
			}
			else
			{
				if(hash.find(x)!=hash.end())
				{
					continue;
				}
				else
				{
					hash[x]=y;
					curr++;
					if(curr>max)
					max=curr;
				}
			}
		}
		cout<<max<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
