#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
#define debug(x) cerr<<">value ("<<#x<<") : "<<x<<endl;

const double pi=2*acos(0.0);
double l,r;

double area(double r, double l)
{
    double h=sqrt(l*l-r*r);
    return pi*r*r*h/(double)3.0;
}

double poor(double R)
{
    if(r==R)
    {
        return pi*r*r*l;
    }
    bool flag=0;
    if(R<r)
    {
        swap(R,r);
        flag=1;
    }
    double x=r*l;
    x=x/(R-r);
    double ans=area(R,l+x)-area(r,x);
    if(flag)
        swap(R,r);
    return ans;
}

int main()
{
//    ios_base::sync_with_stdio(false);
//    cin.tie(NULL);
    cin>>r>>l;
    double lr = max(0.0,r-l), rr= r+l, EPS = 0.00000001;
    for(int i=0;i<600;i++)
    {
        double m1 = lr + (rr - lr) / (double)3.0;
        double m2 = rr - (rr - lr) / (double)3.0;
        if (poor (m1) < poor (m2)) // f - convex function
            lr = m1;
        else
            rr = m2;
    }
    cout<<fixed<<setprecision(2)<<poor(rr)<<endl;
    return 0;
}
