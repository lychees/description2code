#include <bits/stdc++.h>
using namespace std;
#define ll long long
const ll mod=1e9+7;
ll n,k,dp[10][1<<10];
char dice[12][8],str[15];
ll solve(int pos,int mask)
{
    if(pos==k)
        return 1;
    if(dp[pos][mask]!=-1)
        return dp[pos][mask];
    ll ret=0;
    for(int i=0;i<n;i++)
    {
        if(mask&(1<<i))
            continue;
        ll cnt=0;
        for(int j=0;j<6;j++)
        {
            if(dice[i][j]==str[pos])
                cnt++;
        }
        ret+=cnt*solve(pos+1,mask|1<<i);
        ret%=mod;
    }
    dp[pos][mask]=ret;
    return ret;
}
int main()
{
    memset(dp,-1,sizeof(dp));
    cin>>n>>k;
    for(int i=0;i<n;i++)
        for(int j=0;j<6;j++)
            cin>>dice[i][j];
    cin>>str;
    cout<<solve(0,0)<<endl;
    return 0;
}
