#include <iostream>
#include <string>

using namespace std;

int main()
{
    string s;
    while(getline(cin,s)) {
    	size_t com_pos = s.find("//");
    	size_t val_pos = s.find("->");
    	if(com_pos != string::npos) {
    		size_t val_pos = 0;
    		while((val_pos = s.find("->",val_pos,2)) != string::npos) {
    			if(val_pos < com_pos) {
    				s.replace(val_pos,2,".");
    				com_pos--;
    				val_pos++;
    			}
    			else
    				break;
    		}
    		
    	}
    	else
    	{
    		size_t val_pos = 0;
    		while((val_pos = s.find("->",val_pos,2)) != string::npos) {
    			s.replace(val_pos,2,".");
    		}
    	}
    	cout << s << endl;
    }
    return 0;
}
