#include <iostream>
#include <queue>
#include <algorithm>
using namespace std;

int main()
{
	long x[3],n,i=0,temp,p=1;
	cin>>n;
	cin>>x[0]>>x[1]>>x[2];
	sort(x,x+3);
	cout<<-1<<endl<<-1<<endl;
	p=x[0]*x[1]*x[2];
	cout<<p<<endl;
	for(i=3;i<n;i++)
	{
		cin>>temp;
		bool flag=false;
		if(temp>x[2])
		{
			x[0]=x[1];
			x[1]=x[2];
			x[2]=temp;
			flag=true;
		}
		else if(temp>x[1])
		{
			x[0]=x[1];
			x[1]=temp;
			flag=true;
		}
		else if(temp>x[0])
		{
			x[0]=temp;
			flag=true;
		}	
		if(flag)
			p=x[0]*x[1]*x[2];
		cout<<p<<endl;
	}
    return 0;
}
