#include <bits/stdc++.h>
using namespace std;

int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

int lcm(int a,int b)
{
	int r;
r=gcd(a,b);
return(a*b/r);
}

int main()
{
	int a[21],x,j,val;
	long long int p,q;
  int n,i,t;
  long long int ans;
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
  	scanf("%d",&a[i]);
  }
  scanf("%d",&t);
  while(t--)
  {
  	ans=0;
scanf("%lld %lld",&p,&q);
 for(i=1;i<(1<<n);i++)
 {
 	x=0; val=1;
 	for(j=0;j<n;j++)
 	{
 		if(i&(1<<j))
 		{
 		x++;
 		val=lcm(val,a[j]);
 	//	cout<<"val"<<val<<"\n";
 		}
 	}
 	if(x%2==1)
 	{
 		ans=ans+((q/val)-(p-1)/val);
 	}
 	else
 	ans=ans-((q/val)-(p-1)/val);
 }
 cout<<ans<<"\n";
  }
    return 0;
}
