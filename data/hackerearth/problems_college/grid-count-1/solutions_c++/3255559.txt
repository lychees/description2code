#include <iostream>
#include <cmath>
#include <map>
#define maxNum 1000000000 
using namespace std;
map<long long int,bool>store;

void allCubes(){
	int m=cbrt(maxNum);
	for(int i=1;i<=m;++i){
		store[(i*i*i)]=true;
	}
}

int main()
{
	int n,k,cnt=0;
	allCubes();
	scanf("%d%d",&n,&k);
	int arr[n][n];
	int strip[n][n];
	for(int i=0;i<n;++i){
		for(int j=0;j<n;++j){
			scanf("%d",&arr[i][j]);
		}
	}
	
	for(int j=0;j<n;++j){
		int sum=0;
		for(int i=0;i<k;++i){
			sum+=arr[i][j];
		}
		strip[0][j]=sum;
		
		for (int i=1; i<n-k+1; i++){
            sum += (arr[i+k-1][j] - arr[i-1][j]);
            strip[i][j] = sum;
       }
		
		
	}
	
	
	
	for(int i=0;i<n-k+1;++i){
		int sum=0;
		for(int j=0;j<k;++j){
			sum+=strip[i][j];
		}
		if(store.find(sum)!=store.end()){
			cnt++;
		}
		
		for(int j=1;j<n-k+1;++j){
			sum+= (strip[i][j+k-1] - strip[i][j-1]);
		
			if(store.find(sum)!=store.end()){
				cnt++;
			}
		}
		
	}
	
	
	cout<<cnt<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
