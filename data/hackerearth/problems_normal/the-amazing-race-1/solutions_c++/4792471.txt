#include <iostream>
#include <stack>
using namespace std;

int main()
{
	int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;
    	stack<pair<int,int> > s;
    	long long int al[n],ar[n],a[n];
    	al[0] = 0;
    	for (int i = 0; i < n; i++) {
    		// cout << "hw" << endl;
    		cin >> a[i];
    		if (!s.empty()) {
    			while(!s.empty() && (s.top().first < a[i]) )s.pop();
    			al[i] = i-((s.empty())?0:s.top().second);
    			// cout << al[i]<< endl;
    		}
    		
    		s.push(make_pair(a[i],i));
    	}
    	
    	stack<pair<int,int> >().swap(s);
    	ar[n-1]=0;    	;
    	for (int i = n-1; i >= 0; i--) {
    		if (!s.empty()) {
    			while(!s.empty() && (s.top().first < a[i]) )s.pop();
    			ar[i] = ((s.empty())?n-1:s.top().second)-i;
    		}    			
    		s.push(make_pair(a[i],i));
    	}
    	long long int v,mv=0,mi=1;
    	for (int i = 0; i < n; i++) {
    		// cout << "("<<al[i] << " " << ar[i] << ")";
    		v = ((ar[i]+al[i])*(i+1))%(1000000007);
    		if (v > mv) {
    			mv = v;
    			mi = i+1;
    		}
    	}
    	// cout << endl;
    	cout << mi << endl;
    }
    return 0;
}
