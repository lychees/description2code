#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
#include<vector>
#include<set>

#define ll long long
#define ull unsigned long long
#define MAX 4005000

using namespace std;

int primes[MAX+2];
bool mark[MAX+2];

void sieve()
{
    int ind=0;
    for(int i=2;i<MAX;i++)
    {
        if(!mark[i])
        {
            primes[ind++]=i;
            for(int j=i;j<MAX;j+=i)
                mark[j]=1;
        }
    }
}

int main()
{
    int n;
    sieve();
    scanf("%d",&n);
    int i=0;
    unsigned long long sum=0;
    while(primes[i]<n)
    {
        sum+=primes[i++];
    }
    printf("%llu\n",sum);
    return 0;
}
