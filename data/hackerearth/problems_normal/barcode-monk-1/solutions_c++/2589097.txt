const int maxN = 100000 + 5;
const int mod = 1000*1000*1000 + 7;
const int base = 37;
const int SQ = 317;
const int maxL = 17;

bool can[maxN];
int dp[maxN];

main()
{
    ios::sync_with_stdio(0); cin.tie(0); 
    int n, k; cin >> n >> k;

    can[0] = 1;
    rep(i,0,k)
    {
	  int x; cin >> x;
	  rep(j,x,n+1)
		can[j] |= can[j-x];
    }
    
    vector<int> vv;
    rep(i,1,n+1) if(can[i])
	  vv.push_back(i);

    dp[0] = 1;
    rep(i,1,n+1)
    {
	  dp[i] = dp[i-1];
	  rep(j,0,vv.size())
		if(i > vv[j])
		    dp[i] = (dp[i] + dp[i-vv[j]-1]) % mod;
		else if(i == vv[j])
		    dp[i] = (dp[i] + 1) % mod;
    }
    

    cout << dp[n] << endl;
    return 0;
}
