#include<iostream>
using namespace std;
int numberOfDivisor(int N){
	int i,count = 0;
	for(i = 1; i * i < N;i++){
		if(N%i == 0){
			count++;
			if(i != (N/i)){
				count++;
			}
		}
	}
	return count;
}

int main(){
	
	int numBeer,integer;
	cin>>numBeer;
	for(int i = 0 ; i < numBeer ; i++){
		cin>>integer;
		if(numberOfDivisor(integer) < 4){
			cout<<"NO"<<endl;
		}
		else
			cout<<"YES"<<endl;
	}
	return 0;
}
