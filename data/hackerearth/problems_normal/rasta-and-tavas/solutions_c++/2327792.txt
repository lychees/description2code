using namespace std;

typedef pair<int, int> pii;

const double eps = 1e-9;
const double pi = acos(-1.0);

const int maxn = (int)1e6 + 10;

int d[maxn];
int p[maxn];

int main() {

	int n,k; cin >> n >> k;

	for (int i = 2; i < maxn; i++) {
		if (p[i]) continue;
		for (int j = i; j < maxn; j += i)
			p[j] = i;
	}

	for (int i = 2; i < maxn; i++) {
		int x = i;
		while(x % p[i] == 0) x /= p[i];
		d[i] = d[x] + p[i];
	}

	int res = 0;

	for (int i = 1; i <= n; i++) if (d[i] == k) {
		res++;
	}

	cout << res << endl;

	return 0;
}
