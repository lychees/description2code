#include <bits/stdc++.h>

using namespace std;

int main()
{
   int k=0,i,li,ri,t;string s;
   cin>>s;
   cin>>t;
   while (t--)
   {
       k=0;
       cin>>li>>ri;
       for (i=li-1;i<ri-1;i++)
       {
           if (s[i]==s[i+1])
            k+=1;
       }
       cout<<k<<endl;
   }
    return 0;
}
