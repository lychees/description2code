#include <bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define ft first
#define sd second
#define mem(a, v) memset(a, v, sizeof(a))
typedef long long ll;
typedef vector<int> VI;
typedef pair<int, int> PII;
typedef vector<VI> matrix;
const ll mod = 1000000007LL;

ll arr[1000100], ans[1000100];

int main()
{
    //freopen("input.txt", "r", stdin);
    //freopen("output.txt", "w", stdout);
    int t;
    cin>>t;
    while(t--){
        int n;
        ll k;
        cin>>n>>k;
        for(int i=1; i<=n; i++){
            scanf("%lld", &arr[i]);
            ans[i] = 0;
        }
        int idx = 1;
        ll cur = 0;
        for(int i=1; i<=n; i++){
            cur += arr[i];
            while(cur > k){
                cur -= arr[idx];
                idx++;
            }
            if(i >= idx){
                ans[1] += 1;
                ans[i - idx + 2] -= 1;
            }
        }
        for(int i=1; i<=n; i++){
            ans[i] += ans[i-1];
        }
        for(int i=1; i<=n; i++){
            printf("%lld ", ans[i]);
        }
        printf("\n");
    }
    return 0;
}
