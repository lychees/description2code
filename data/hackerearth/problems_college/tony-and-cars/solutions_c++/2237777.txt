#include<iostream>
#include<math.h>
using namespace std;

int main()
{
	int test;
	int long long num,temp;
	int long long total = 0;
	int long long digit =0;
	cin>>test;
	
	while(test--)
	{
		total =0;
		digit =0 ;
		cin>>num;
		temp = num;
		while(temp > 0)
		{
			digit++;
			temp = temp/10;
		}
		
		
		for(int i =1;i<digit;i++)
		{
			
			total += i*9*pow(10,i-1);
			
		
		}
		total += digit*(num-pow(10,digit-1)+1);
		
		
		cout<<total<<endl;
	}
}
