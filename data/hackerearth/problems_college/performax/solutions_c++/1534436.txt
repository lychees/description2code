#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long n,d,ans=0;
    cin>>n>>d;
    long a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    for(int i=1;i<n;i++)
    {
    	if(a[i]<=a[i-1])
    	{
    		long temp = ((a[i-1]-a[i])/d+1);
    		a[i] += temp*d;
    		ans += temp;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
