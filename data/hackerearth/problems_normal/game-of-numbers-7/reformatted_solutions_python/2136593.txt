T = int(raw_input())
while T > 0:
    X, Y = map(int, raw_input().split())
    sp = [True for x in range(Y-X+1)]
    z = 2
    ans = 0
    while z * z <= Y:
        gap = int(z * z)
        base = X + (gap - X % gap) % gap
        k = base
        while k <= Y:
            sp[k-X] = False
            k = k + gap
        z = z + 1
    ans = sp.count(True)
    print ans
    T = T - 1
