#include <bits/stdc++.h>
using namespace std;

int main()
{
    int points[26];
    for(int i=0;i<26;i++)
    	cin>>points[i];
   	int M = 0;
   	int D = 0;
   	int Q;
   	cin>>Q;
   	while(Q--)
   	{
   		string a,b;
   		cin>>a;
   		cin>>b;
   		int v1[26];
   		int v2[26];
   		memset(v1,0,sizeof(v1));
   		memset(v2,0,sizeof(v2));
   		for(int i=0;i<a.size();i++)
   			{
   				v1[a[i]-'a']++;		
   			}
   		for(int i=0;i<b.size();i++)
   			{
   				v2[b[i]-'a']++;
   			}
   		for(int i=0;i<26;i++)
   		{
   			if(v1[i] && v2[i])
   			{
   				int m = min(v1[i],v2[i]);
   				v1[i] -= m;
   				v2[i] -= m;
   			}
   		}
   		for(int i=0;i<26;i++)
   		{
   			M += v1[i]*points[i];
   			D += v2[i]*points[i];
   		}
   	}
   	if(M == D)
   		cout<<"Draw"<<endl;
   	else if(M > D)
   		cout<<"Marut"<<endl;
  	else
  		cout<<"Devil"<<endl;
    return 0;
}
