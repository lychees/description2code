#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    
    cin>>tc;
    while(tc--)
    {
        int x=1;
        string str;
        cin>>str;
        int l=str.length();
      //  cout<<l<<endl;
        if(l%3!=0)
        {
            x=0;
            
        }
        else if((str[0]==str[l/3])||(str[l/3]==str[2*l/3])||(str[0]==str[2*l/3]))
        {
        	x=0;
        
        }
        else
        {
        for(int i=0;i<=(l/3)-2;i++)
        {
            if(str[i]!=str[i+1])
            {
            x=0;
            break;
            }
        }
       // cout<<x<<" ";
        for(int i=l/3;i<=(2l/3)-2;i++)
        {
            if(str[i]!=str[i+1])
            {
            x=0;
            break;
            }
        }
      //  cout<<x<<" ";
        for(int i=(2*l/3);i<=(l-2);i++)
        {
            if(str[i]!=str[i+1])
            {
            x=0;
            break;
            }
        }
        }
       
        if(x==0)
        {
        cout<<"Not OK"<<endl;   
        }
        if(x==1)
        {
            cout<<"OK"<<endl;
        }
    }
}
