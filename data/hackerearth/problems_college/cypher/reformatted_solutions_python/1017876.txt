def decipher(s):
    text = ""
    for i in xrange(len(s)):
        if ord(s[i]) + i > 122:
            text += (chr(ord(s[i])+i - 26))
        else:
            text += (chr(i+ord(s[i])))
    return text


t = int(raw_input())

for i in xrange(t):
    s = raw_input()
    print decipher(s)
