#include <iostream>
using namespace std;

int main()
{   int t;
    cin>>t;
    while(t--)
    {
    	long long int n,x,y,ul1,ul2,ll1,ll2,res;
    	cin>>n;
    	cin>>x;
    	cin>>y;
    	ul1 = x+1;
    	ul2 = n;
    	ll1 = n-y;
    	ll2 = n;
    	if(ll1 <=ul2  && ll1>=ul1)
    	{
    		res = (ul2-ll1+1);
    	}
    	else if(ll1<ul1)
    	{
    		res =(ul2-ul1+1); 
    	}
    	cout<<res<<"\n";
    	
    }
    
    return 0;
}
