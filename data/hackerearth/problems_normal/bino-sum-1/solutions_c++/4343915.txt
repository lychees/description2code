#include <iostream>
using namespace std;
#define ll long long
ll f[1005],dp[1005][1005],mod=1e9+7;

ll bigMod(ll a, ll b)
{
    ll x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod; 
        b /= 2;
    }
    return x;
}

int main()
{
    ll t,n,i,j,k;
    f[0]=1;
    for(i=1;i<=1000;i++)
		f[i]=(i*f[i-1])%mod;
		
    for(i=0;i<=1000;i++)
    {
    	for(j=0;j<=i;j++)
    	{
    		if(j==0)
    			dp[i][j]=1;
    		else
    			dp[i][j]=(dp[i][j-1]%mod+(f[i]*((bigMod(f[j], mod-2) * bigMod(f[i-j], mod-2)) % mod)) % mod)%mod;
    	}
    }
    
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld%lld",&n,&k);
    	printf("%lld\n",dp[n][k]);
    }
    
    return 0;
}
