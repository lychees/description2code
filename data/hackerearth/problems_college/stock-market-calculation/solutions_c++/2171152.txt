#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t, n, i, x, max;
	long long unsigned int sum;
	vector <int> a;
	cin>>t;
	while(t--) {
		sum=0;
		cin>>n;
		for(i=0;i<n;i++) {
			cin>>x;
			a.push_back(x);
		}
		a.push_back(0);
		i=0;
		z:
		max=*max_element(a.begin()+i, a.end());
		while(a[i++]!=max)
			sum=sum+(max-a[i-1]);
		if(i!=n)
			goto z;
		cout<<sum<<endl;a.clear();
	}
    return 0;
}
