#include <algorithm>
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
  string p;
  cin >> p;
  int n;
  cin >> n;
  vector<string> all(n);
  int i;
  for (i = 0; i < n; i++)
    cin >> all[i];
  sort(all.begin(), all.end());
  for (i = 0; i < n; i++)
    if (all[i].substr(0, p.size()) == p) {
      p = all[i];
      break;
    }
  cout << p;
}
