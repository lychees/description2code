def isPrime(n):
    for i in range(2, n):
        if n % i == 0:
            return False

    return True if n != 1 else False


def gcd(a, b):
    while b:
        a, b = b, a % b
    return a


def getNumberOfNumbers(n):
    number = 0
    for i in range(1, n+1):
        if gcd(i, n) == 1:
            number += 1

    return number


T = int(raw_input())

for test in range(T):
    N = int(raw_input())

    if isPrime(getNumberOfNumbers(N)):
        print("Correct")
    else:
        print("Incorrect")
