#include <iostream>
using namespace std;

int main()
{
	long long n,i,j,k,t,sum,count,C[100002];
	string A[100002],S,T;
	cin>>n;
	for(i=0;i<=n+1;i++) C[i]=1;
	k=0;
	for(i=0;i<n;i++){
		cin>>A[k];
		T=A[k];
		for(j=i-1;j>=0;j--){
			if(T<A[j]){
				S=A[j+1];
				A[j+1]=A[j];
				A[j]=S;
				t=C[j+1];
				C[j+1]=C[j];
				C[j]=t;
			}
			else if(T>A[j]) break;
			else if(T==A[j]){
				C[j]++;
				for(j=j+1;j<i;j++){
					S=A[j+1];
					A[j+1]=A[j];
					A[j]=S;
					t=C[j+1];
					C[j+1]=C[j];
					C[j]=t;
				}
				k--;
				break;
			}
		}
		k++;
		C[k]=1;
	}
	for(i=0;i<k;i++){
		cout<<A[i]<<" "<<C[i]<<endl;
	}
    return 0;
}
