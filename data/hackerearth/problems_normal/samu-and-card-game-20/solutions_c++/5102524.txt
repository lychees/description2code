#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int initialx,initialy;
    int n,m,k,count;
    int t,x,y;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	initialx=1;
    	initialy=1;
    	count=0;
    	cin>>n>>m;
    	cin>>k;
    	for(int j=0;j<k;j++)
    	{
    		cin>>x>>y;
    		while((initialx+x<=n&&initialy+y<=m)&&(initialx+x>=1&&initialy+y>=1))
    		{
    			initialx=initialx+x;
    			initialy=initialy+y;
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
