ll x_pow_y(l x, l y){
    if(y==0)
        return 1;
    ll temp;
    temp = x_pow_y(x,y/2);
    if(y%2==0)
        return temp*temp;
    else
        return x*temp*temp;
}

l gcd(l a,l b)
{
  if(a==0)
    return b;
  if(b==0)
    return a;
  return gcd(b,a%b);
}

// function to get index of smallest number just greater than the given number----------------
l binarySearch( vector<l>& arr, l t)
{
  l n=arr.size();
  l i=0,j=n-1;
    while(i<=j)
    {
        l mid = (i+j)/2;
        if(t < arr[mid])
          j=mid-1;
        else
          i=mid+1;
    }
    return i;
}

int isPowerOfTwo (unsigned l x)
{
  return ((x != 0) && ((x & (~x + 1)) == x));
}
