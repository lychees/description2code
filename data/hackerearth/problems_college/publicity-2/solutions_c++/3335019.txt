#include<bits/stdc++.h>
#define MAX 100005
using namespace std;

list<int>*adj;

void dfs(int v,int visited[])
{
	visited[v]=1;
	list<int>::iterator it;
	for(it=adj[v].begin();it!=adj[v].end();it++)
		if(visited[*it]==0)
			dfs(*it,visited);
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,edges,u,v,ans=0;
		scanf("%d",&n);
		adj=new list<int>[MAX];
		int visited[MAX]={0};
		scanf("%d",&edges);
		while(edges--)
		{
			scanf("%d %d",&u,&v);
			adj[u].push_back(v);
			adj[v].push_back(u);
		}
		for(int i=0;i<n;i++)
			if(visited[i]==0)
			{
				dfs(i,visited);
				ans++;
			}
	printf("%d\n",ans);
	}
}
