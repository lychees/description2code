#include <bits/stdc++.h>

using namespace std;

int A, B;
long long C;
int dp[1001];

void _main()
{
    memset(dp, 0, sizeof dp);
    scanf("%d%d%lld", &A, &B, &C);
    int parity=0;
    while(C>1)
    {
        C=(C+B-1)/B;
        swap(A, B);
        parity^=1;
    }
    if(parity==0)
        printf("Bob\n");
    else
        printf("Adam\n");
}

int main()
{
    int TEST;
    scanf("%d", &TEST);
    while(TEST--)
        _main();
    return 0;
}
