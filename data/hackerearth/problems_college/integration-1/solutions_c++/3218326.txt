#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll i,j,k,n,p,m,r,u,s;
    int t;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n>>p>>u>>r>>s;
        m=p;
        while(n!=0)
        {
            if(p+r<=u)
            {
                p+=r;
                n--;
            }
            else
            {
                p=p-s;
            if(p<0)
            {
                p=m;
                break;
            }
                n--;
            }
        }
        cout<<"Case #"<<i+1<<": "<<p<<endl;

    }
    return 0;
}
