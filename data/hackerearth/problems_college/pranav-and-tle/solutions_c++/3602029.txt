#include <bits/stdc++.h>
using namespace std;
int main() {
   int t, i;
   char num[1000000];  
   long m;
   scanf("%d", &t);
   getchar();
   while(t--) {
      scanf("%s %ld", num, &m);
      long ans = 0;
      for(i = 0 ; num[i] != '\0' ; i++) {
         ans = ( ans * 10 + num[i] - 48 ) % m;
      }   
      printf("%ld\n", ans); 
   }
   return 0;
}
