#include<bits/stdc++.h>
using namespace std;
int a[100001];
int leftsearch(int  x,int  l,int  r)
{
	
	if(r>=l)
	{
		int mid=(l+r) >>1;
		if(a[mid]==x && a[mid-1]<x) return mid;
		if(a[mid]>x && a[mid-1]<x) return mid;
		if(a[mid]<x) return leftsearch(x,mid+1,r);
		return leftsearch(x,l,mid-1);
	}
	
	return -1;
}
int rightsearch(int x,int l,int r)
{
	if(r>=l)
	{
		int mid=(l+r)/2;
		
		if(a[mid]==x && a[mid+1]>x) return mid;
		if(a[mid]<x && a[mid+1]>x) return mid;
		if(a[mid]>x) return rightsearch(x,l,mid-1);
		 return rightsearch(x,mid+1,r);
	}
	return -1;
}
int main()
{
	ios::sync_with_stdio(false);
	int t,T;
	cin>>t;
	for(int i=1;i<=t;i++)
	{
		cin>>a[i];
	}
	sort(a+1,a+t+1);
	a[0]=0;
	a[t+1]= INT_MAX; 
	cin>>T;
	while(T--)
	{
	  int l,r;
	  cin>>l>>r;
	  
	  	cout<<rightsearch(r,0,t+1)-leftsearch(l,0,t+1)+1<<endl;
	}
}
