for _ in range(input()):
    n, m = map(int, raw_input().split())
    a = []
    diamonds = 0
    for i in range(n):
       a.append(raw_input().split())
    for i in range(n-1):
        for j in range(m-1):
            if a[i][j]=="/" and a[i+1][j+1]=="/" and a[i+1][j]=="\\" and a[i][j+1]=="\\":
                diamonds += 1
    print diamonds
