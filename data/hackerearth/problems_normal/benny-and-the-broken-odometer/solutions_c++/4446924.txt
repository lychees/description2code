#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T,x,tosub,real;
    string s;
    cin >> T;
    while(T--){
    	cin >> s;
    	tosub=0,real=0;
    	int t=0,power9,minus;
    	for(int i=s.length()-1;i>=0;i--){
    		x=(int)(s[i]-'0');
    		//cout << "print digit: "<< x << endl;
    		power9=pow(9,t);
    		minus = pow(10,t)-power9;
    		if(i==s.length() && x>3){
    			tosub+=1;
    		}
    		else if(x>3){
    			tosub+= power9 + x*minus;
    		}
    		else{
    			tosub+= x*minus;
    		}
    		real+=x*pow(10,t);
    		//cout << "real: " << real;
    		t++;
    	}
    	cout << (real-tosub) << endl;
    }
    return 0;
}
