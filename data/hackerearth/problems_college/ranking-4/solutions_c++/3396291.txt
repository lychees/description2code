ii arr[1<<20];
int ranks[1<<20];
int main()
{
	//freopen("input.txt","r",stdin);

	fast_io;
	int t;cin >> t;
	while(t--)
	{
		int n;cin >> n;
		forall(i,0,n)
		{
			cin >> arr[i].first;
			arr[i].second=i;
		}
		sort(arr,arr+n);
		reverse(arr,arr+n);
		int it=1;
		forall(i,0,n)
		{
			int cur=arr[i].first;
			int cnt=0;
			while(i<n)
			{
				if(cur!=arr[i].first)break;
				ranks[arr[i].second]=it;
				i++;
				cnt++;
			}
			i--;
			it+=cnt;
		}

		cout << endl;
		SET(ranks,0);
	}
