#include<iostream>

using namespace std;

int main()
{
	int *query, *price;
	int qCounter = 0, pCounter = 0, size=0;
	
	cin >> size;
	query = new int[size]();
	price = new int[size]();
	
	while(qCounter < size)
	{
		cin >> query[qCounter];

		if (query[qCounter] == 1)
		{
			if (price[pCounter-1] <= 0)
				query[qCounter] = 0;
			else
			{
				query[qCounter] = price[pCounter-1];
				price[pCounter-1] = NULL;
				pCounter--;
			}
		}
		else if (query[qCounter] == 2)
		{
			cin >> price[pCounter];
			pCounter++;
		}
		qCounter++;
	}
	
	for (int counter = 0; counter < size; counter++)
	{
		if (query[counter] != 2)
		{
			if (query[counter] == 0)
				cout << "No Food" << endl;
			else
				cout << query[counter] << endl;
		}
	}



}
