tc = int(raw_input())
for z in xrange(0, tc):
    N, X = raw_input().split()
    n = int(N)
    x = int(X)
    a = [int(i) for i in raw_input().split()]
    count = 0
    for i in range(len(a)):
        if (i == len(a)-1):
            break
        if a[i] >= a[i+1]:
            diff = a[i]-a[i+1]
            c = (diff/x)+1
            a[i+1] += c*x
            count = count+c
    print count
