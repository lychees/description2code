int main()
{

	int a[10001]={0,1,0,1,1,0,1,0};
	for(i=7;i<10001;i++)
	{
		l=i-1;
		if(a[l]==0)
		{
			a[i]=1;
		}
		else
		{
			l=i-4;
			if(a[l]==0)
			a[i]=1;

			a[i]=0;
		}
	}
	while(t--)
	{
		int n;
		scanf("%d",&n);
		if(a[n])
		printf("Yes\n");
		else
		printf("No\n");
	}
    return 0;
}
