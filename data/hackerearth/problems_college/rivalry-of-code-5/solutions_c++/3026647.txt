#include <iostream>
#include <limits.h>
using namespace std;
int main()
{
	int t, c, i, j, l, min;
	string dis, s;
    cin>>t;
    while(t--)
    {
    	min=INT_MAX;
    	c=0;
    	cin>>dis;
    	cin>>s;
    	if(s.length()<dis.length())
    		swap(s,dis);
    	l=dis.length();
    	for(i=0;i<=s.length()-l;i++)
    	{
    		c=0;
    		for(j=i;j<(i+l);j++)
    		{
    			if(dis[j-i]!=s[j])
    				c++;
    		}
    		if(c<min)
    			min=c;
    	}
    	cout<<min<<endl;
    }
    return 0;
}
