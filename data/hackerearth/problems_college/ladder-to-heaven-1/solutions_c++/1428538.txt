#include <iostream>
using namespace std;

int main() 
{             
    long long int T;
	 scanf("%lld",&T);  
	while(T--)
	{
		long long int x=1,y=2,sum,limit;     
    	long long int evensum=2;
		scanf("%lld",&limit);
   		while( (x+y)<limit )
		{
       		 sum=x+y;
       		 x=y;
       		 y=sum;
       		 if (sum%2==0)
           	 evensum+=sum;
   		 }
   			 printf("%lld\n",evensum);
	}
    return 0;
}
