#include <iostream>
#include <stack>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int i,j,t,n,num ,back;
    char pk;
    cin>>t;
    stack <int> s;
    for(i=0;i<t;i++){
    		cin>>n;
    		cin>>num;
    		s.push(num);
    		for(j=0;j<n;j++){
    			cin>>pk;
    			if(pk=='P'){
    				back=0;
    				cin>>num;
    				s.push(num);
    			}
    			else if(back%2==1){	
    				s.push(num);
    				back++;
    			}
    			else{
    				num=s.top();
    				s.pop();
    				back++;
    			}
    		}
    		num=s.top();
    		cout<<"Player "<<num<<endl;
    } 
 
    
    return 0;
}
