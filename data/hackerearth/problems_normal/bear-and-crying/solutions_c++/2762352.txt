#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    string s;
    cin>>s;
   long long int n,i,semi=0,count=0,finalvalue=0;
    n=s.length();
    for(i=0;i<n;i++)
    {
        if(s[i]==';')
           {
               semi++;
            finalvalue=(finalvalue+count)%mod;}
        else
        {
            count=(2*count+semi)%mod;
        }

    }
    cout<<finalvalue;
}
