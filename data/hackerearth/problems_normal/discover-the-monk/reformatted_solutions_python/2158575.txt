def binarySearch(alist, item):
    first = 0
    last = len(alist)-1
    found = False

    while first <= last and not found:
        midpoint = (first + last)//2
        if alist[midpoint] == item:
            found = True
        else:
            if item < alist[midpoint]:
                last = midpoint-1
            else:
                first = midpoint+1

    return found


a = raw_input().split()
c = int(a[1])
b = sorted([int(i) for i in raw_input().split()])

for k in range(c):
    a = int(raw_input())
    if binarySearch(b, a):
        print "YES"
    else:
        print "NO"
