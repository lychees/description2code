#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;

int main()
{int n,t;

cin>>n;
	int max1[n+1],max2[n+1],A[n+1];
	for(int i = 1;i<=n;i++)
	{cin>>A[i];}
max1[1]=A[1];
max2[n]=A[n];
	for(int i= 2;i<=n;i++)
	{max1[i]=max(max1[i-1],A[i]);}
for(int i = n-1;i>0;i--)
{max2[i]=max(max2[i+1],A[i]);}
for(int i=1;i<=n;i++)
{if(max1[i]==A[i]||max2[i]==A[i])
{cout<<i<<" ";}}
	

    return 0;
}
