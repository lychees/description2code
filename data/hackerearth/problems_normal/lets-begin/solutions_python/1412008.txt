T = int(raw_input())

for i in range(T):
	n = int(raw_input())
	if n == 0 or n == 1:
		print -1
	elif n%7 == 0:
		print n/7
	elif n%7 == 1:
		print (n-1)/7 + 1
	elif n%7 == 2:
		print (n-2)/7 + 1
	elif n%7 == 3:
		print (n-3)/7 + 1
	elif n%7 == 4:
		print (n-4)/7 + 2
	elif n%7 == 5:
		print (n-5)/7 + 1
	elif n%7 == 6:
		print (n-6)/7 + 2
