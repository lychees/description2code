#include <iostream>
using namespace std;

int main()
{
   long long int t,n;
   cin>>t;
   while(t--)
  {
  	long long int i, a,s=0;
  	cin>>n;
  	for(i=0;i<n;i++)
  	{
  	cin>>a;
  	s+=a;
  	}
  	if(s%n==0)
  	{
  	cout<<"YES"<<endl;
  	cout<<s/n<<" "<<0<<endl;
  	}
  	else
  	{
  	cout<<"NO"<<endl;
  	cout<<s/n<<" "<<s%n<<endl;
  	}
  }
}
