#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define MODN 1000000007
#define vi vector<int>
#define vvi vector<vi>
#define pb push_back
#define forn(i,n) for( i = 0; i < n; i++ )
#define s(n) scanf("%d",&n);

ll modfun(ll a,ll b,ll m)
{
    ll ans = 1;
    while (b > 0)
    {
            if (b & 1)
            {
               a=a%m;
               ans = (ans * a)%m;
               ans=ans%m;
            }
        b=b>>1;
        a=a%m;
        a = (a*a)%m;
        a=a%m;
    }
    return ans%m;
}

int main()
{
	ll i, t, n, m, ans;
	cin >> t;

	while( t-- )
	{
		cin >> n;
		if( n < 3 ) ans = 0;
		else{
			ans = ( modfun(3,n,MODN) % MODN) - ( ( 3 * modfun(2,n,MODN) ) % MODN ) + 3;
			ans = (ans + MODN) % MODN;
		}
		cout << ans <<endl;
	 }
    return 0;
}
