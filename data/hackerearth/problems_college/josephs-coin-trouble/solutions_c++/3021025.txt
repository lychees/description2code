#include <bits/stdc++.h>
using namespace std;

int dp[9000005];

int main()
{
    long long int n, arr[4];
    cin>>n;
    
    int m=0;
    while(cin>>arr[m++]);
    sort(&arr[0], &arr[m]);
    
    memset(dp, 0, sizeof(dp));
    dp[0] = 1;
    
    for(long int i=0; i<m; i++)
    	for(int j=1; j<=n; j++)
    	{
    		if(j-arr[i]<0)
    			continue;
    		dp[j] += dp[j-arr[i]];
    	}
    cout<<dp[n]<<endl;
    
    return 0;
}
