#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	string s;
	int o,t;//occurrences & time completion
}node;
bool my(node a,node b)
{
	if(a.o==b.o)
	{
		if(a.t<b.t)
		return true;
	}
	else if(a.o>b.o)
	return true;
	return false;
}
int main()
{
	int n;
	scanf("%d",&n);
	int i,j,k;
	string s;
	map<string,int> m;
	string names[n];
	for(i=0;i<n;i++)
	{
		cin>>s;
		names[i]=s;
		m[s]+=1;
	}
	node a[n];
	int mc=0;
	map<string,int>::iterator it,it2;
	for(it=m.begin();it!=m.end();it++)
	{
		a[mc].s=it->first;
		a[mc].o=it->second;
		mc++;
	}
	for(i=0;i<mc;i++)
	{
		for(j=n-1;j>=0;j--)
		{
			if(names[j]==a[i].s)
			{
				k=j;
				break;
			}
		}
		a[i].t=j;
	}
	sort(a,a+mc,my);
	for(i=0;i<mc;i++)
	{
		cout<<a[i].s<<" "<<a[i].o<<endl;
	}
	return 0;
}
