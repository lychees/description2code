#include <iostream>
using namespace std;

int main()
{
    int t,o,k,n[1000]={0},p;
    cin>>t;
    while(t--)
    {
    	int n[1000]={0};
    	cin>>o>>k;
    	for(int i=0;i<k;i++)
    	{
    		cin>>p;
    		n[p]++;
    	}
    	for(int i=1;i<=o;i++)
    	{
    		if(n[i]%2!=0)
    		{
    			cout<<i<<endl;
    			
    		}
    		
    	}
    }
    return 0;
}
