#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
ll ans[1415][1415];
int main()
{
	int n;
	scanf("%d",&n);
	vector<ll>ar,tmp;
	ll tmp1;
	for(int i=0;i<n;++i)
	{
		scanf("%lld",&tmp1);
		ar.push_back(tmp1);
	}
	for(int i=0;i<n;++i)
	{
		priority_queue<int>lt;
		priority_queue<int>rt;
		lt.push(ar[i]);
		ans[i][i]=ar[i];
		for(int j=i+1;j<n;++j)
		{
			if((j-i)%2)
			{
				if(lt.top()<=ar[j])
					rt.push(-ar[j]);
				else
				{
					rt.push(-lt.top());
					lt.pop();
					lt.push(ar[j]);
				}
				ans[i][j]=lt.top();
			}
			else
			{
				if(-rt.top()>=ar[j])
					lt.push(ar[j]);
				else
				{
					lt.push(-rt.top());
					rt.pop();
					rt.push(-ar[j]);
				}
				ans[i][j]=lt.top();
			}
		}
	}
	ll q;
	scanf("%lld",&q);
	ll l,r;
	while(q--)
	{
		scanf("%lld %lld",&l,&r);
		printf("%lld\n",ans[l-1][r-1]);
	}
	return 0;
}
