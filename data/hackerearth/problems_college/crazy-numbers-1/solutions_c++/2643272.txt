#include <iostream>
#include <string>
#include <cmath>
#include <vector>
using namespace std;

int main()
{		
	int a;
	long int b;
	cin >>a;
	while(a--)
	{
		cin >>b;
		if(b==0)
			cout<<0<<endl;
		else if(b==1)
			cout<<10<<endl;
		else
		{
		
		vector<long long int> one = {0,1,1,2,2,2,2,2,2,2,1,0};
		for(int i =2;i<b;i++)
			{
				vector<long long int> two(12,0);
				for(int j= 1;j<=10;j++)
					two[j]=(one[j-1]+one[j+1])%1000000007;
				one = two;
				
			}
			long long int ans =0;
		for(int i =1;i<=10;i++)
			{
				ans = (ans+one[i])%1000000007;
			}
		cout <<ans<<endl;
		
		}
	}
   // cout << "Hello World!" << endl;
    return 0;
}
