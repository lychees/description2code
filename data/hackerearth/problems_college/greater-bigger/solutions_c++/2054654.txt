#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    string s, y;
    int t, i, j, flag, p, l;
    char x, m;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	flag=0;
    	p=200;
    	l=s.size();
    	//cout<<l<<s[1]<<s[0];
    	m='z'+1;
    	for(i=l-1; i>=1; i--)
    	{
    		if(s[i]>s[i-1])
    		{
    		flag=1;
    		for(j=i; j<l; j++)
    		{
    			if(s[j]<m and s[j]>s[i-1])
    			{
    				m=s[j];
    				p=j;
    			}
    		}
    		//cout<<p;
    		x=s[i-1];
    		s[i-1]=s[p];
    		s[p]=x;
    		//cout<<s[0];
    		y=s.substr(i);
    		//cout<<y<<endl;
    		sort(y.begin(), y.end());
    		for(j=0; j<i; j++)
                cout<<s[j];
    		cout<<y<<endl;
    		//cout<<y<<endl;
    		break;
    		}
    	}
    	if(flag==0)
    		cout<<"no answer\n";

    }

    return 0;
}
