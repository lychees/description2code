#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
#define ll long long

int main()
{
	ll t;
	cin >> t;
	
	while (t--)
	{
		vector<ll> value(3, 0);
		cin >> value[0] >> value[1] >> value[2];

		sort(value.begin(), value.end());

		ll diff = value[1] - value[0];
		ll diff2 = value[2] - value[1];

		if (diff > diff2)
			cout << diff - 1 << '\n';
		else
			cout << diff2 - 1 << '\n';
	}

	return 0;
}
