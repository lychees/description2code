#include <iostream>
#define M 1000000007
using namespace std;
typedef long long ll;
int main()
{
    ll t,n,k;
    cin>>t;
    while(t--){
    	
    	cin >>n>>k;
    	ll res =1;
    	for(int i=n;i>=1;i--){
   			res = (res*k)%M;
   			k--;
    		
    	}
   		cout << res << endl;
    }
    return 0;
}
