#include<iostream>
#include<cstdio>
#include<cstring>
#define mod 1000000007
using namespace std;
int n;
int dp[100004][3][3][3];
int solve(int idx,int curr,int prev,int count){
    if(idx == n){ 
        if((count == 1 && !(prev == 0 && curr == 0)) || (prev == 0 && curr == 0 && count == 0)) 
            return 1;
        return 0;
    }   
    if(dp[idx][curr][prev][count] != -1) 
        return dp[idx][curr][prev][count];
    int ct = count;
    if(curr == 0 && prev ==0)
        ct = count + 1;
    if(ct > 1)
        return 0;
    return dp[idx][curr][prev][count] = (solve(idx+1,0,curr,ct) + solve(idx+1,1,curr,ct)) % mod;    
}

int main(){
    int t;
    cin>>t;
    while(t--){
        cin>>n;
        memset(dp,-1,sizeof(dp));
        cout<<solve(0,2,2,0)<<endl;
    }   
    return 0;
}
