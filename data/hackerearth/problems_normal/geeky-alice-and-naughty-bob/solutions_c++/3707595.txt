#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long  t,a,b,ans;
    int arr[]={0,1,3,9,15,18};
    cin>>t;
    while(t--){
    	ans=0;
    	cin>>a>>b;
    	if(a<6)
    		ans-=arr[a-1];
    	else
    		ans-=(18+(a-1-5)*9);
    	if(b<6)
    		ans+=arr[b];
    	else
    		ans+=(18+(b-5)*9);
    	
    	cout<<ans<<endl;
    }
    
    return 0;
}
