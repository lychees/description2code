#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mp make_pair
#define mod 1000000007
#define gc getchar_unlocked
using namespace std;

int dp[5002][5002];
char a[5002],b[5002];
bool isvowel(char a)
{
    if(a=='a' || a=='e' || a=='i' || a=='o' || a=='u')return true;
    return false;
}
int maa(int a,int b,int c)
{
    int ret=max(a,b);
    ret=max(ret,c);
    return ret;
}
void solve(int m,int n)
{
   for(int i=1;i<=m;i++)
   {
       for(int j=1;j<=n;j++)
       {
           int val1,val2,val3;
    if(a[i-1]==b[j-1] && !isvowel(a[i-1]))val1=dp[i-1][j-1]+1;
    else val1=dp[i-1][j-1];
    val2=dp[i][j-1];
    val3=dp[i-1][j];
    dp[i][j]=maa(val1,val2,val3);
       }
   }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    //memset(dp,-1,sizeof(dp));
    cin>>a>>b;
    int m=strlen(a); int n=(strlen(b));
    solve(m,n);
    cout<<dp[m][n];
	return 0; 
}
