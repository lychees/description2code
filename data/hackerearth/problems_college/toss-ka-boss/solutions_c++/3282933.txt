#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string s;
    long long int i,l,count,ans;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	l=s.length();
    	count=0;
    	ans=0;
    	for(i=0;i<l;i++)
    	{
    		if(s[i]=='H')
    		{
    			count++;
    		}
    		else
    		{
    			ans+=(count*(count+1))/2;
    			count=0;
    		}
    		
    	}
    		ans+=(count*(count+1))/2;
    		cout<<ans<<"\n";
    	
    }
    return 0;
}
