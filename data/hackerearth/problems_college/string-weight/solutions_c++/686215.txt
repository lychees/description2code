typedef long long int LLD;
typedef unsigned long long int LLU;

using namespace std;

int main(){
	
	int n, t, cnt[26], sum;
	string s, ch;
	
	scanf("%d", &t);
	while(t--){
		cin >> s;
		fill(cnt, cnt + 26, 0);
		for(int i=0;i<s.length();i++)
			cnt[s[i] - 97]++;
		scanf("%d", &n);
		while(n--){
			cin >> ch;
			cnt[ch[0] - 97] = 0;
		}
		sum = 0;
		for(int i=0;i<26;i++)
			sum += ((i+1) * cnt[i]);
		printf("%d\n", sum);
	}
	
    return 0;
}
