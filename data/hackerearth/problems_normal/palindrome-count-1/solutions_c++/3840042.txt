#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int dp[100][100];
int main()
{
  char a[100];
  cin>>a;
  for(int i=0;i<strlen(a);i++)
  {
      dp[i][i]=1;
  }
  for(int i=0;i<strlen(a)-1;i++)
  {
      if(a[i]==a[i+1])dp[i][i+1]=1;
  }
  for(int i=3;i<=strlen(a);i++)
  {
      for(int j=0;j<strlen(a)-i+1;j++)
      {
           if(a[j]==a[j+i-1]  && dp[j+1][j+i-1-1]==1)
            dp[j][j+i-1]=1;
      }
  }
  int ans=0;
  for(int i=0;i<strlen(a);i++)
   {   for(int j=0;j<strlen(a);j++)
     if(dp[i][j]==1)ans++;
  
}
cout<<ans;

 return 0;
}
