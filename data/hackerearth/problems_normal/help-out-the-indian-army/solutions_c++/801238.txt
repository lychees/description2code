#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cerr<<#x<<":"<<x<<"\n"
#define f first 
#define s second
#define ll long long
ll n,a,b,i,j,z,mx,x,y,w;
pair<ll,ll> A[100001];
int main()
{
  cin>>n>>a>>b;
  for(i=0;i<n;i++)
  {
    cin>>x>>y;
    A[i]=make_pair(x-y,x+y);
    w=max(w,x+y);
  }
  sort(A,A+n);
  z=max(min(A[0].f,b)-a,0ll);
  z+=max(b-max(w,a),0ll);
  for(i=0;i<n;)
  {
    for(j=i,mx=A[i].s;i<n && A[i].f<=mx;i++)
      mx=max(mx,A[i].s);
    if(i<n)
      z+=max(min(A[i].f,b)-max(mx,a),0ll);
  }
  cout<<z<<"\n";
	return 0;
}
