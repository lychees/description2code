T = int(input())
for t in range(0, T):
    integers = map(int, raw_input().split())
    N = integers[0]
    M = integers[1]
    A = integers[2]
    B = integers[3]
    C = integers[4]

    if N < M:
        k = N
    else:
        k = M

    if 2*C < A+B:
        k = 0

    cost = N*A + M*B + k*(2*C-A-B)

    print cost
