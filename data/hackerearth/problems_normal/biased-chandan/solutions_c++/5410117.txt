#include <iostream>
#include <stack>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    stack<int> s;
    
    int n,a,sum=0;
    
    cin>>n;
    
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	if(a==0 && i!=0)
    	{
    		s.pop();
    	}
    	else
    		s.push(a);
    }
    
    while(!s.empty())
    {
    	sum+=s.top();
    	s.pop();
    }
    
    cout<<sum<<endl;
    
    return 0;
}
