#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int arr[1000000];
int main()
{
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;++i)
    {
    	cin>>arr[i];
    }
    int ans=0;
    for(int i=0;i<n-k;++i)
    {
    	if(abs(arr[i]-arr[i+k])%k==0 || (arr[i]+arr[i+k])%k==0) ans++;
    }
    cout<<ans<<endl;
    return 0;
}
