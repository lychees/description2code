#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <functional>
#include <algorithm>

using namespace std;

int dp[2000007];
 
int main()
{  int T,N,m1,m2,i;
    cin>>T;
    dp[0]=0;
   
    while(T--)
    {
    	cin>>N>>m1>>m2;
      if(m1==1||m2==1)
       dp[1]=1;
       else
       dp[1]=0;
      for (int i=1;i<=N;i++)
       {
       	if(i>=m1&&!dp[i-m1])
       	   dp[i]=1;
       	else if (i>=m2&&!dp[i-m2])
       	   dp[i]=1;
       	else dp[i]=0;
       }
       cout<<dp[N]<<"\n";
    }
    return 0;
}
