#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    int i,l;
    char ans;
    string s;
    cin>>n;
    map<string,int> a;
    map<string,int>::iterator it;
    for(i=0;i<n;i++)
    {
    	cin>>s;
    	l=s.length();
    	if(l%2!=0)
    	{
    		it=a.find(s);
    		if(it!=a.end())
    		{
    			ans=s[l/2];
    			break;
    		}
    		reverse(s.begin(),s.end());
    		a[s]=1;
    	}
    	
    }
    cout<<l<<" "<<ans;
    return 0;
}
