#include <bits/stdc++.h>
#define ll long long
using namespace std;
int belongs (ll int num,ll int range)
{
	if(num>=1 && num<=range)
	return 1;
	return 0;
}
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int z,m,n;
		scanf("%lld%lld%lld",&z,&m,&n);
		ll int i;
		ll int x,y;
		ll int ans=0;
		for(i=1;i*i<=z;i++)
		{
			if((z%i)==0)//i is a factor
			{
				x=i;
				y=z/i;
				if(belongs(x,m) && belongs(y,n))
				{
					ans++;
				}
				if(x!=y && belongs(x,n) && belongs(y,m))
				{
					ans++;
				}
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
