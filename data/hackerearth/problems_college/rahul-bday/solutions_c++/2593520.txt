#include <iostream>
using namespace std;
long long int findmax(long long int x,long long int y)
{
	return (x>y?x:y);
}

int main()
{
   long long int test,b,w;
   long long int x,y,z,cost1;
   cin>>test;
   while(test--)
   {
   	cost1=0;
   	cin>>b>>w;
   	cin>>x>>y>>z;
   	long long int max1=findmax(x,y);
   	
   	if(max1==x&&(y+z)<x)
   	{
   		cost1=b*(y+z)+w*y;
   	} 	
   	else if(max1==y&&(x+z)<y)
   	{
   		cost1=b*x+w*(x+z);
   	}
   	else
   	cost1=b*x+w*y;
   	
	cout<<cost1<<endl;
   }
    return 0;
}
