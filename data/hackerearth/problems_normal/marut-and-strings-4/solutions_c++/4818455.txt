#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    char s[100];
    int t,i,n=0,u=0,l=0,len=0;
    cin>>t;
    if(t>10||t<=0)
     cout<<"\nInvalid Test";
    else
    {
     while(t--)
     {
       cin>>s;
       u=l=n=0;
       len=strlen(s);
       if(len>100)
        cout<<"\nInvalid Input";
       else{ 
       for(i=0;s[i]!='\0';i++)
       {
	    if(isalpha(s[i]))
	    {
		 if(islower(s[i]))
		  l++;
		 else if(isupper(s[i]))
		  u++;
    	}
	   else
	     n++;
      }
      if(n==len)
      cout<<"\nInvalid Input";
      if(u==len || l==len)
       cout<<"\n0"; 
      else if(n!=len)
       {
	    if(u<l)
	     cout<<"\n"<<u;
	    else
	    cout<<"\n"<<l;
       }  
       }
     } 
    }  
   return 0;
}
