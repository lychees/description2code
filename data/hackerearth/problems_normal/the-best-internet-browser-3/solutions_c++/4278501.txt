#include<iostream>
#include<vector>
#include<string>
#include<cstdio>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int size1,size2,count=0;
		string a,s;
		cin>>a;
	    size1=a.size();	
		s=a.substr(4,a.size()-8);
		for(int x=0;x<s.size();x++)
		{
			if(s[x]=='a'||s[x]=='A'||s[x]=='e'||s[x]=='E'||s[x]=='i'||s[x]=='I'||s[x]=='o'||s[x]=='O'||s[x]=='u'||s[x]=='U')
			count++;
		}
			size2=count+4;
			cout<<size1-size2<<"/"<<size1<<endl;
	}
	
}
