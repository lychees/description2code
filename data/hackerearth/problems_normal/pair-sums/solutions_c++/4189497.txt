#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}

main(){

long long int n,sum,i,j,flag,ans;
flag=0;
cin>>n>>sum;
long long int a[n];
for(i=0;i<n;i++){cin>>a[i];}
qsort(a, n, sizeof(long long int), cmpfunc);
i=0;j=n-1;

while(i<j){
        ans=a[i]+a[j];
if(ans==sum){flag=1;break;}
if(ans<sum){i++;}
else{j--;}


}
if(flag){cout<<"YES";}
else{cout<<"NO";}




}
