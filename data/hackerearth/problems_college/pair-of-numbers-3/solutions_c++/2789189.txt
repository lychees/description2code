#include <iostream>
using namespace std;
int main()
{
	int t,i,j;
    //cout << "Hello World!" << endl;
   cin>>t;
   while(t--)
   {
   	int n,count=0;
   	cin>>n;
   	for(i=2;i<n;i++)
   	{
   		for(j=i+1;j<=n;j++)
   		{
   			if((i*j)%(i+j)==0)
   			count++;
   		}
   	}
   	cout<<count<<endl;
   }
   return 0;
}
