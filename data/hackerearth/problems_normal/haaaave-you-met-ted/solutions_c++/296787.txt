	std::ios::sync_with_stdio(false);
	int t; cin >> t;
	while (t--){
		int n; cin >> n; ll val, ans=INT64_MAX;
		for (int i = 0; i < n; i++){
			cin >> val;
			int s = 0;
			while (val>0){
				int r = val & 1;
				if (r)s++;
				val >>= 1;
			}
			if (s < ans)ans = s;
		}
		cout << ans << endl;
	}
