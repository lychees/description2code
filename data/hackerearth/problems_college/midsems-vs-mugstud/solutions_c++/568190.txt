#include <iostream>
using namespace std;
int n=0;

int main()
{
	int t;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a[n][n][n];
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++)
    			for(int k=0;k<n;k++)
    				scanf("%d", &a[i][j][k]);
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++){
    			for(int k=0;k<n;k++){
    				printf("%d ", (a[i][j][k] - (i?a[i-1][j][k]:0) - (j?a[i][j-1][k]:0) - (k?a[i][j][k-1]:0) + ((i&&j)?a[i-1][j-1][k]:0) + ((i&&k)?a[i-1][j][k-1]:0) + ((k&&j)?a[i][j-1][k-1]:0) -((i&&j&&k)?a[i-1][j-1][k-1]:0)));
    			}
    			printf("\n");
    		}
    
    }
    return 0;
}
