#include <bits/stdc++.h>
#define MAX 105
#define inf 10000000000000000
#define ll long long
#define MOD 1000000007
#define s(x) scanf("%d", &x);

using namespace std;

int dp[2][3000005][2];
int a[201];
int b[201];

int main() {
    int n,q;
    cin >> n;

    for ( int i = 1; i <= n; i++ )
        cin >> a[i];
    for ( int i = 1; i <= n; i++ )
        cin >> b[i];
    cin >> q;
    dp[0][0][0] = 1;

    for ( int i = 1; i <= n; i++ ) {
        memset(dp[i%2], 0, sizeof(dp[i%2]));

        for ( int j = 0; j <= 300000; j++ ) {
            for ( int k = 0; k < 2; k++ ) {
                if ( j == 0 && k == 1 )
                    continue;
                if ( !dp[(i-1)%2][j][k] )
                    continue;
                    
                int actual = j, new_diff;
                
                if ( k == 1 ) 
                    actual = -j;

                new_diff = actual + a[i];
                dp[i%2][abs(new_diff)][new_diff < 0] += dp[(i-1)%2][j][k];

                dp[i%2][abs(new_diff)][new_diff < 0] %= MOD;

                new_diff = actual - b[i];
                dp[i%2][abs(new_diff)][new_diff < 0] += dp[(i-1)%2][j][k];

                dp[i%2][abs(new_diff)][new_diff < 0] %= MOD;

                new_diff = actual + a[i] - b[i];
                dp[i%2][abs(new_diff)][new_diff < 0] += dp[(i-1)%2][j][k];

                dp[i%2][abs(new_diff)][new_diff < 0] %= MOD;

                new_diff = actual;
                dp[i%2][abs(new_diff)][new_diff < 0] += dp[(i-1)%2][j][k];

                dp[i%2][abs(new_diff)][new_diff < 0] %= MOD;
            }
        }
    }
    int ans = 0;
    for ( int j = 0; j <= min(q,300000); j++ ) {
        for ( int k = 0; k < 2; k++ ) {
            ans += dp[n%2][j][k];
            ans%=MOD;
        }
    }

    cout << ans << endl;
    return 0;

}
