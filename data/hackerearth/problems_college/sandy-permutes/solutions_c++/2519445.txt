using namespace std;
#define ll long long
ll int gcd(ll int a,ll int b){if(a==0)return b;if(b==0)return a;if(a==1||b==1)return 1;
if(a==b)return a;if(a>b)return gcd(b,a%b);else return gcd(a,b%a);}
int main()
{
	ll int n,k,i,j=0;
	cin>>n>>k;
	ll int a[n];
	for(i=0;i<n;i++) cin>>a[i];
	while(k--){
		ll int z,temp,max=0;
		for(i=j;i<n;i++){
			if(a[i]>max){
				max = a[i];
				z = i;
			}
		}
	//	cout<<max<<"\n"<<endl;
		temp = a[z];
		a[z] = a[j];
		a[j] = temp;
		if(z==j) k++;
		j++;
		if(j==n) break;
	}
	for(i=0;i<n;i++) cout<<a[i]<<" ";
	cout<<endl;
	return 0;
}
