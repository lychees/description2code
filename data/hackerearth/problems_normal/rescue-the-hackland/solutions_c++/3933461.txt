#include <iostream>
using namespace std;
 
int
main ()
{
  int tc = 0;
      long long l, r;
      int k;
  cin>>tc;
  while (tc--)
    {
      cin >> l;
      cin >> r;
      cin >> k;
      long long num = l;
      long long count = 0;
      while (num <= r)
	{
	  num = (num * k) + 1;
	  count++;
	}
      cout << count << endl;
    }
}
