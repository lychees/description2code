using namespace std;
long int fib[100];
int bs(int l,int h,long int key){
    if(l < h){
        int m = l+(h-l)/2;
        if(key == fib[m])
            return m;
        else if(key > fib[m])
            return bs(m+1,h,key);
        else
            return bs(l,m,key);
    }
    else if(l == h){
        if(key <= fib[l])
            return l;
    }
}
int main() {
    int t,start,count;
    long int n,x,i=2,y,num=2;
    fib[0] = 1;
    fib[1] = 1;
    while(num <= 100000){
        fib[i] = fib[i-1] + fib[i-2];
        num = fib[i];
        i++;
    }
    scanf("%d",&t);
    while(t--){
    	scanf("%ld%ld",&x,&y);
    	if(y < x)
            printf("VALID\n");
        else{
                if(x == 1)
                    count = 1;
                else
                    count =0;
            start = bs(0,i-1,x);


            while(fib[start] <= y){
                count++;
                start++;
            }
           if(count%2 == 0)
                printf("VALID\n");
           else
            printf("INVALID\n");
        }
    }
    return 0;
}
