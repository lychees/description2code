#include <iostream>
using namespace std;
long long int rev(long long int a)
{
	long long int r=0;
	while(a!=0)
	{
		r=r*10;
		r=r+a%10;
		a=a/10;
	}
	return r;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int a,b,x=0;
    	cin>>a>>b;
    	x=rev(a)+rev(b);
    	cout<<rev(x)<<endl;
    }
    return 0;
}
