from sys import stdin
import math
fa = [1]
hi = 500005
mod = 10**9 + 7
for i in xrange(1, hi):
    fa.append((fa[-1]*i) % mod)


def ncr(n, r):
    num = fa[n]
    den = (fa[r] * fa[n-r]) % mod
    den = pow(den, mod-2, mod)
    return (num * den) % mod


t = int(stdin.readline())
for _ in xrange(t):
    n, r = map(int, stdin.readline().split())
    a = map(int, stdin.readline().split())
    print ncr(r-sum(a) + n-1, n-1)
