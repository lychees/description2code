#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000009
#define MAX 100005
#define ll long long int
ll modexp(ll a, ll b){ll res=1;while(b){if(b&1)res=(res*a)%MOD;a=(a*a)%MOD;b/=2;}return res;}
ll modinv(ll n){return modexp(n,MOD-2);}
ll arr[MAX];
int main()
{
        int i;
        ll n;
        ll tn , te; 
        ll ans;
        int t;
        ll to = modinv(2);
        cin>>t;
        while(t--)
        {
                cin>>n;
                tn = modexp(10,n-1);
                te = modexp(8,n-1);
                ans = (5*tn - 4*te)%MOD;
                //ans = (ans*to)%MOD;
                if(ans<0) ans+=MOD;
                cout<<ans<<endl;
        }
}
