#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int sum=0;
    vector<int>s(n);
    for(int i=0;i<n;i++)
    {
    	cin>>s[i];
    	sum +=s[i];
    }
    int count=0;
    for(int i=n-1;i>=0;i--)
    {
    	if(s[i]==0)
    	{
    		count++;
    	}
    	else
    	{
    		if(count>0)
    		{
    			sum -=s[i];
    			count--;
    		}
    		
    	}
    }
    	
    cout<<sum<<endl;
    return 0;
}
