T = int(raw_input())

for test in range(T):
	N = int(raw_input())
	array = []
	for submission in range(N):
		array.append([int(x) for x in raw_input().split()])
	
	streak = 0	
	maximum = 0
	used = {}
	for elem in array:
		submission = elem[0]
		result = elem[1]
		if submission not in used:
			if result == 1:
				streak += 1
				used[submission] = 0
		if result == 0:
			if streak > maximum:
				maximum = streak
			streak = 0
	if streak > maximum:
		maximum = streak
	
	print(maximum)
