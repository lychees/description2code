struct FenwickTree {
	typedef int T;
	vector<T> v;
	FenwickTree(int n): v(n, 0) {}
	void add(int i, T x) {
		for(; i < (int)v.size(); i |= i+1) v[i] += x;
	}
	T sum(int i) {	//[0, i)
		T r = 0;
		for(-- i; i >= 0; i = (i & (i+1)) - 1) r += v[i];
		return r;
	}
	T sum(int left, int right) {	//[left, right)
		return sum(right) - sum(left);
	}
};


int main() {
	int T;

		int N;
		scanf("%d", &N);
		vector<int> A(N);
		rep(i, N) scanf("%d", &A[i]);
		int first;
		scanf("%d", &first);
		vector<int> values = A;
		sort(all(values));
		values.erase(unique(all(values)), values.end());
		rep(i, N)
			A[i] = lower_bound(all(values), A[i]) - values.begin();
		int X = values.size();
		long long cnt1 = 0, cnt2 = 0;
		FenwickTree ftL(X), ftR(X);
		rep(i, N) ftR.add(A[i], +1);
		rep(i, N) {
			ftR.add(A[i], -1);
//			cerr << i << ": " << ftL.sum(A[i]) << ", " << ftR.sum(A[i]) << endl;

			cnt1 += (long long)ftL.sum(A[i]) * ftR.sum(A[i]);
			cnt2 += (long long)(i - ftL.sum(A[i]+1)) * ((N - 1 - i) - ftR.sum(A[i]+1));
			ftL.add(A[i], +1);
		}
//		cerr << cnt1 << ", " << cnt2 << endl;
		puts((cnt2 < cnt1) != (first != 0) ? "Ananya" : "Sarika");

	}
