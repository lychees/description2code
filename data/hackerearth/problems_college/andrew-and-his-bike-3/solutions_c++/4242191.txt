#include <bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
vi a[100005];

int cdesc[100005];

bool visited[100005];
int dfs(int s){
    int i,sz,c=1;
    visited[s]=true;
    sz=a[s].size();
    
    for(i=0;i<sz;i++){
        if(!visited[a[s][i]]){
           c=c+dfs(a[s][i]);
        }
    }
    cdesc[s]=c;
    return c;
    
}
int main() {
	int n,i,u,v;
	cin>>n;
	for(i=0;i<n-1;i++){
	    cin>>u>>v;
	    a[u].push_back(v);
	    a[v].push_back(u);
	}
	//memset(count,0,sizeof count);
	memset(visited,false,sizeof visited);
	dfs(1);
	cout<<cdesc[1]-1<<" ";
	for(i=2;i<=n;i++){
	    cout<<cdesc[i]<<" ";
	}
	return 0;
}
