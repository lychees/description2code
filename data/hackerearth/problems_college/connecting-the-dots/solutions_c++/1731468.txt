#include<bits/stdc++.h>
using namespace std;
int main() {
	int n,m;
	scanf("%d%d",&n,&m);
	int a[n][n];
	memset(a,0,sizeof(a));
	int i,j;
	int x,y,c;
	for(i=0;i<m;i++) {
		scanf("%d%d%d",&x,&y,&c);
		a[x-1][y-1] = c;
	//	a[y-1][x-1] = c;
	}
	
	queue<int>q;
	int visited[n];
	memset(visited,0,sizeof(visited));
	visited[0] = 1;
	int ans[n];
	memset(ans,0,sizeof(ans));
	q.push(0);
	q.push(-1);
	while(!q.empty()) {
		int u = q.front();
		q.pop();
		int v = q.front();
		q.pop();
		//printf("%d %d\n",u,v);
		for(i=0;i<n;i++) {
			if((v==-1||a[u][i]!=v)&&visited[i]==0&&a[u][i]!=0) {
			
				ans[i] = ans[u] + 1;
					
				visited[i] = 1;
				q.push(i);
				q.push(a[u][i]);
			}
		}
	}
	if(ans[n-1]!=0)
	printf("%d\n",ans[n-1]);
	else
	printf("-1\n");
}
