#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;

    	for(int i=0;i<n;i++)

    	for(int i=1;i<n;i++)

		cout<<temp<<endl;
    }
    return 0;
}
