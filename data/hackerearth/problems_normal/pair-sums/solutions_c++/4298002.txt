#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,k;scanf("%d%d",&n,&k);
    int a[n],i,c[2*n+6];
    for(i=0;i<n;i++)	scanf("%d",&a[i]),c[i]=0;
    memset(c,0,sizeof c);
    for(i=0;i<n;i++){
    	if(c[k-a[i]])	{cout<<"YES"<<endl;return 0;}
    	c[a[i]]=1;
    }
    cout<<"NO"<<endl;
    return 0;
}
