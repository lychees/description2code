#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
    int test;
    cin>>test;
    while(test--){
    	long long int n,a[100000]={0};
		cin>>n;
		if(n==1)
		{
			cout<<"4"<<endl;
			continue;
		}
		a[0]=0;
		a[1]=2;
		a[2]=3;
		for(int i=3;i<10000;i++)
		a[i]=a[i-1]+a[i-2];
		cout<<a[n]*a[n]<<endl;
    }
    return 0;
}
