#include<bits/stdc++.h>
#define LL long long
#define mod  1000000007
using namespace std;


LL bmod(LL a,LL b)
{
        if(b==0)
        {
             return 1;
        }
        else if(b%2==0)
        {
             LL k=bmod(a,b/2)%mod;
             return(k*k)%mod;
        }
        else if(b%2!=0)
        {
              LL k=a%mod;
              LL kk=bmod(a,b-1)%mod;

              return(k*kk)%mod;
        }
}


int main()
{
      int t;
      cin>>t;
      while(t--)
      {
           LL n;
           cin>>n;
           cout<<bmod(2,n)-1<<endl;

      }


      return 0;

}
