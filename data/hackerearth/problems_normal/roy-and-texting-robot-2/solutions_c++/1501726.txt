#include<bits/stdc++.h>

using namespace std;
int a[256],b[256];
void pre()
{
    char ch;
    int z=1;
    for(ch='1';ch<='9';ch++)
    {
        a[ch]=4;
        b[ch]=z++;
    }

    a['0']=2;
    a['1']=5;
    a['7']=5;
    a['9']=5;


    int x=1;
    z=2;
    for(ch='a';ch<='o';ch++)
    {
        a[ch]=x++;
        b[ch]=z;
        if(x==4)
        {
            x=1;
            z++;
        }

    }

    a[ch]=1;
     b[ch++]=7;
      a[ch]=2;
       b[ch++]=7;
    a[ch]=3;
     b[ch++]=7;
      a[ch]=4;
       b[ch++]=7;

    a[ch]=1;
    b[ch++]=8;
     a[ch]=2;
      b[ch++]=8;
      a[ch]=3;
       b[ch++]=8;

    a[ch]=1;
     b[ch++]=9;
      a[ch]=2;
       b[ch++]=9;
    a[ch]=3;
     b[ch++]=9;
      a[ch]=4;
       b[ch++]=9;


    a['.']=1;
     a[',']=2;
      a['?']=3;
       a['!']=4;

    b['.']=1;
     b[',']=1;
      b['?']=1;
       b['!']=1;


     a['_']=1;
     b['_']=0;


}
int main()
{
    int t,i,ans,j;
    char s[1005];

    cin>>t;
    pre();
    while(t--)
    {
        cin>>s;

        if(b[s[0]]!=1)
            ans=1;
        else
        ans=0;
        i=1;
        ans+=a[s[0]];
        while(s[i])
        {
            if(b[s[i]]!=b[s[i-1]])
                ans++;
            ans+=a[s[i]];
            i++;
        }

        cout<<ans<<'\n';
    }
    return 0;
}
