#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
map <string, int> mp;

int main()
{
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		mp.clear();
		int flag = 0;
		
		while (n--) {
		string s;
		cin >> s;
		
		if (mp[s] >=  1 ) {
				flag = 1;
				//break;
			}
			
			
		int sz = s.size();
		string t = "";
		for (int i = 0; i < sz; i++) {
			t = t + s[i];
			mp[t]++;
		}
		
		}
		
		if (flag == 1)
			cout << "NO"<<endl;
		else
			cout <<"YES"<<endl;
	}
	return 0;
}
