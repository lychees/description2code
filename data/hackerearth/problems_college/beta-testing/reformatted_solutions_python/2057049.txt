n = int(raw_input().strip())
dc = dict()
s = raw_input().strip().split()
for el in s:
    if el in dc:
        dc[el] += 1
    else:
        dc[el] = 1
l = [(key, value) for (key, value) in sorted(dc.items(), reverse=True)]
for tup in l:
    print(tup[0]+" " + str(tup[1]))
