#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int B,W;
    	cin>>B>>W;
    	long int x,y,z;
    	cin>>x>>y>>z;
    	long long int res1=(x*B)+(y*W);
    	if(x>z)
    	{
    		if((y+z)<x)
    		{
    			x=y+z;
    		}
    	}
    	if(y>z)
    	{
    		if((x+z)<y)
    		{
    			y=x+z;
    		}
    	}
    	long long int res2=(x*B)+(y*W);
    	if(res1<=res2)
    	{
    		cout<<res1<<endl;
    	}
    	else
    	{
    		cout<<res2<<endl;
    	}
    }
}
