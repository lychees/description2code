#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int a,b;
    cin>>a>>b;
    char c[a][b];
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            cin>>c[i][j];
    char d[b];
    for(int i=0;i<b;i++)
        d[i]=c[0][i];
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            if(d[j]>c[i][j])
                d[j]=c[i][j];
        }
    }
    int f=0;
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            if(d[j]==c[i][j])
            {
                f++;break;
            }
        }
    }
    cout<<f<<endl;
    return 0;

}
