#include<cstdio>
#include<queue>
using namespace std;
double ar[20][20];
double ans[300000];
int f[300000];
queue<int> q;
int main()
{
int N; scanf("%d",&N);
for(int i=0;i<N;i++)
for(int j=0;j<N;j++)
scanf("%lf",&ar[i][j]);
int s= 1<<N; s--;
f[s]=1; q.push(s);
ans[s]=1.0;
while(q.size())
{
int t=q.front(); q.pop();
int cc=0;
for(int i=0;i<N;i++)
{
int tt= t & (1<<i); if(tt) cc++;
}

int mp = cc * (cc -1); mp/=2;
double pp= 1.0/mp;
for(int i=0;i<N;i++)
{
int tt= t & (1<<i); if(tt==0) continue;
for(int j=0;j<i;j++)
{
tt= t & (1<<j); if(tt==0) continue;
int x=(1<<i) ^ t ;
if(f[x]==0)
q.push(x);
f[x]=1;
ans[x] += ar[j][i]*ans[t]*pp;


x=(1<<j) ^ t ;
if(f[x]==0)
q.push(x);
f[x]=1;
ans[x]+=ar[i][j]*ans[t]*pp;


}
}


}
for(int i=0;i<N;i++)
{
int t= 1<<i;
printf("%0.5lf ",ans[t]);
}

}
