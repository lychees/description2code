#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
struct candy{
	int id;
	int sw;
	int cost;
};
int d[100007];
struct candy cam[100007];
int cmp(struct candy p,struct candy q)
{
	if(p.sw==q.sw)
	{
		if(p.cost==q.cost)
		{
			return p.id<q.id;
		}
		return p.cost<q.cost;
	}
	return p.sw>q.sw;
}
int main()
{
	int m,p;
	scanf("%d %d",&m,&p);
	int i;
	for(i=0;i<m;i++)
		scanf("%d %d %d",&cam[i].id,&cam[i].sw,&cam[i].cost);
	sort(cam,cam+m,cmp);
	for(i=0;i<m;i++)
	d[cam[i].id]=i+1;
	for(i=0;i<p;i++)
	{
		int tem;
		scanf("%d",&tem);
		printf("%d\n",d[tem]);
	}
	return 0;
}
