#include <iostream>
using namespace std;

int main()
{
   string S;
   int t;
   cin>>t;
   while(t--)
   {
   	cin>>S;
   	int i,j;
   	for(i=0,j=S.length()-1;i<j;i++,j--)
   	 {
   	 	if(S[i]!=S[j])
   	 	 {
   	 	 	if(S[i]==S[j-1])
   	 	 	cout<<j<<endl;
   	 	 	else
   	 	 	cout<<i<<endl;
   	 	 	break;
   	 	 }
   	 }
   	 if(i>=j)
   	  cout<<"-1"<<endl;
   }
    return 0;
}
