#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long n, count=0;
    cin>>n;
    long long a[n], i, s, sum=0;
    for(i=0;i<n;++i)
    	cin>>a[i];
    sort(a,a+n);
    cin>>s;
    for(i=n-1;i>=0;--i){
	    count++;	
    	sum+=a[i];
    	if(sum>=s)
    		break;
    	
    }
    
    cout<<count<<endl;
    
    
    return 0;
}
