#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i;
		scanf("%d",&n);
		int a[300];
		int arr_size = 1,carry =  0,prod = 1;
		a[0] = 1;
		for(int x = 1; x <= n; x++ )
		{
			for( i = 0; i < arr_size; i++ )
			{
				prod=a[i]*x + carry;
				carry = prod/10;
				a[i] = prod%10;
			}
			while(carry)
			{
				a[i++] = carry%10;
				carry = carry/10;
				arr_size++;
			}
		}
		for(int i = arr_size-1 ; i>=0; i-- )
		{
			printf("%d",a[i]);
		}
		printf("\n");
	}
}
