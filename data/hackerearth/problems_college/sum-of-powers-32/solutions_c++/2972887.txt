#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[1000005]={0};
int main()
{
    
    
    for(ll i=1 ;i<=1000;i++){
        for(ll p1=2;pow(i,p1)<=1000000;p1++){
            for(ll j=1;j<=1000;j++){
                for(ll p2=2;pow(i,p1)+pow(j,p2)<=1000000;p2++){
                    if(pow(i,p1)+pow(j,p2)<=1000000)a[(ll)(pow(i,p1)+pow(j,p2))]=1;
                    //cout<<(int)pow(i,p1)<<' '<<pow(j,p2)<<endl;
                    if(j==0 ||j==1)break;
                }
            } if(i==0 ||i==1)break;
        }
    }    
    
    int q;
    cin>>q;
    while(q--){
        
        ll n;
        cin>>n;
        if(n==0){cout<<"Yes"<<endl;continue;}
        if(a[n]==1)cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        
    }
    
    return 0;
}
