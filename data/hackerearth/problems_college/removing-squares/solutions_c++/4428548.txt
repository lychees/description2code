const int mod = 1e9 + 7;

int ans[100100];
inline int best(int n) {
    if(!n) return n;
    if(ans[n]) return ans[n];
    
    ans[n] = n;
    for(ll i = 1; i * i <= n; i++)
        ans[n] = min(ans[n], best(n - i * i) + 1);
    return ans[n];
}
