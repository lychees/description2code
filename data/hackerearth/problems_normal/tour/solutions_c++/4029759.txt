#include <iostream>
#include <map>
using namespace std;

int main() 
{
	ios_base::sync_with_stdio(false);
	char city[12];
	int id,i,j,p,dist[1000][1000],n;
	long long int ans;
	cin>>n;
	string s;
	ans=0;
	map<string,int> m;
	for(i=0;i<n;i++)
	{
		cin>>s;
		m[s]=i;
	}
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			cin>>dist[i][j];
	cin>>p;
	i=0;
	while(p--)
	{
		cin>>s;
		j=m[s];
		ans+=dist[i][j];
		i=j;
	}
	cout<<ans;
	
	return 0;
}
