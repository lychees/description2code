#include <iostream>
#include <vector>
using namespace std;
int flag;
vector<int> hour(vector<int>&v2)
{
 vector<int> v3(v2.size(),0);
 if(v2!=v3)
 {
  if(v2[1]==1)
  {
   v3[0]=1;
  }
  for(int i=1;i<=v2.size()-2;i++)
  {
   if(v2[i-1]==1 && v2[i+1]==1)
   {
    v3[i]=1;
   }
  }
  if(v2[v2.size()-2]==1)
  {
   v3[v2.size()-1]=1;
  }
 }
 else
 {
  flag++;
 }
 return v3;
}
extern int flag;
int main()
{
  int t;

  cout<<"";
  cin>>t;
  while(t--)
  {
   int n,input;
   long long int m;
   vector<int>v;
   cin>>n;
   cin>>m;
   while(n--)
   {
    cin>>input;
    v.push_back(input);
   }
   flag=0;
   while(m--)
   {
    v=hour(v);
    if(flag!=0)
    {
     goto THERE;
    }
   }
   THERE:
   for(int i=0;i<v.size();i++)
   {
    cout<<v[i]<<" ";
   }
   cout<<"\n";
  }
  return 0;
}
