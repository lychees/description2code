#include <iostream>
#include<algorithm>
using namespace std;
 
int main()
{
	int t;
	cin >> t;
	while(t--){
		int maxlps = 0;
    string s;
    int d;
    cin >> s >> d;
    int n = s.length();
    
    int i,j,lps[n];
    lps[0]=0;
    for(i=1;i<n;i++){
    	j = lps[i-1];
    	while(j>0 && s[i]!=s[j])j=lps[j-1];
    	if(s[i]==s[j])j++;
    	lps[i]=j;
    	if(i<d) maxlps = max(maxlps,lps[i]);
    }
    int k = lps[n-1];
    while(k>maxlps && k>0) k = lps[k-1];
    if(k==0) cout << "Puchi is a cheat!" << endl;
    else {
    	for(i=0;i<k;i++) cout << s[i];
    	cout << endl;
    }
	}
	
    return 0;
}
