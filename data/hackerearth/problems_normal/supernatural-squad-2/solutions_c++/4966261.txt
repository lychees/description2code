#include <iostream>
using namespace std;
long int cache[205][205];

long int solve(int n,int m,int k){
	if(n==0) return 1;
	if(n<0) return 0;
	if(m==0) return 0;
	if(n<k||m<k) return 0;
	if(cache[n][m]==-1){
		cache[n][m]=solve(n,m-1,k)+solve(n-m,m,k);
	}
	return cache[n][m];
}


int main()
{
	
    int t;
    cin>>t;
    while (t--){
    	
    	int n,k;
    	cin>>n>>k;
    	for(int i=0;i<=n;i++)
    	for(int j=0;j<=n;j++)
    	cache[i][j]=-1;
    	cout<<solve(n,n,k)<<"\n";
    }
    cout<<endl;
    return 0;
}
