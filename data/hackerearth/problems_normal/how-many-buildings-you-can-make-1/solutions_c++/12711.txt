#include <iostream>
#include <set>

using namespace std;

int main()
{
	int testcase;
	cin >> testcase;

	int n;
	set <pair<int, int> > rectset;

	while (testcase--)
	{
		cin >> n;

		for (int i = 1; i <= n; ++i)
		{
			for (int j = 1; j <= n/i; ++j)
			{
				if (i*j <= n)
				{
					if (i < j)
					{
						rectset.insert(make_pair(i, j));
					}
					else
					{
						rectset.insert(make_pair(j, i));
					}
				}
			}
		}

		cout << rectset.size() << endl;
		rectset.clear();
	}
}
