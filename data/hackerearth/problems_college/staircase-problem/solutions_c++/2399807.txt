#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n==1 || n==2)
    	{
    		cout<<1<<endl;
    		continue;
    	}
    	int temp=2;
    	int i=1;
    	int j=2;
    	while(temp<n)
    	{
    		i++;
    		temp=temp+(++j);
    	}
    	cout<<i<<endl;
    }
}
