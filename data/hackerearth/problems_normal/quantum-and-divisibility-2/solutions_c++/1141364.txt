map<int,LL> prime(int n)
{
  	map<int,LL>r;
  	LL cnt=0;          
  	for(int i=2;i<=sqrt(n);i++)
  	{
    	cnt=0;      
    	if(n%i==0)
    	{
      		while(n%i==0)
      		{
        		cnt++;
        		n/=i;             
      		}          
    	}  
    	r[i]=cnt;      
  	}
  	if(n>1)
  	{
    	r[n]=1;       
  	}        
  	return r;    
}

    	map<int,LL> x=prime(a),y=prime(c);
    	map<int,LL>::iterator p;
    	for(p=y.begin();p!=y.end();p++)
    	{
      		if((p->second)*d > x[p->first]*b)
      		{
        		flag=0;                 
      		}                               
    	}
