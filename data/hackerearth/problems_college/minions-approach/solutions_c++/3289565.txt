int sieve[1000005];
void fi()
{
	sieve[0] = 1;
	sieve[1] = 1;
	for(int i = 2; i <= 100000; i++) {
		if(sieve[i] == 0) {
			for(int j = i*2; j <= 100000; j+=i) {
				sieve[j] = 1;
			}
		}
	}
}
int main()
{
	fi();
	int t,n;
	S(t);
	while(t--) {
		S(n);
		int count = 1;
		for(int i = 2; i <= n; i++) {
			if(__gcd(i, n) == 1) count++;
		}
		if(sieve[count] == 0) cout << "Correct" << endl;
		else cout << "Incorrect" << endl;
	}
	return 0;
}
