from fractions import gcd


def modifySieve(n):
    for i in xrange(2, n):
        if not prime[i]:
            for j in xrange(i * 2, n, i):
                prime[j] += 1


prime = [0]*1000000
modifySieve(1000000)


for chk in xrange(input()):
    a, b = map(int, raw_input().split())
    x = a*b/gcd(a, b)
    if prime[x] > 1 and prime[prime[x]] == 0:
        print "Yes"
    else:
        print "No"
