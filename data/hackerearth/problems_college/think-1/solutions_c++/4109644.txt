#include<iostream>
using namespace std;
int reverse(int n){
    int x=0;
    while(n){
        x=(x*10)+ n%10;
        n=n/10;
    }
    return x;
}
int square(int x){
    return x*x;
}
int main(){
int t;
cin>>t;
for(int i=0;i<t;i++){
    int n;
    cin>>n;
    if(square(n)==reverse(square(reverse(n)))&& reverse(n)!=n){
        cout<<reverse(n)<<endl;
    }
    else
        cout<<"No"<<endl;
}
}
