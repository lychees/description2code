#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

int n, ci, di, res;
vector <int> azul, amarelo;
int main() {
	scanf("%d", & n);
	azul.clear();
	amarelo.clear();
	while(n--) {
		scanf("%d %d", & ci, & di);
		if(ci == 1) {
			azul.push_back(di);
		} else {
			amarelo.push_back(di);
		}
	}
	res = 0;
	sort(azul.begin(), azul.end());
	sort(amarelo.begin(), amarelo.end());
	for(int i = 0, j = 0; i < (int)(azul.size()) && j < (int)(amarelo.size());) {
		if((azul[i] * azul[i]) + (azul[i] * azul[i]) > (amarelo[j] * amarelo[j]) && (amarelo[j] * amarelo[j]) + (amarelo[j] * amarelo[j]) > (azul[i] * azul[i])) {
			res++;
			i++;
			j++;
		} else if(azul[i] < amarelo[j]) {
			i++;
		} else {
			j++;
		}
	}
	printf("%d\n", res);
}
