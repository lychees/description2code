#include<cstdio>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int w[100005];
map<int,int> wins;
int main()
{
    int n,m,i,cur_max;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
        cin>>w[i];
        wins[w[i]]=0;

    }
    cur_max=w[0];
    wins[cur_max]=1;
    cout<<cur_max<<" "<<"1"<<endl;
    for(i=1;i<n;i++)
    {
        wins[w[i]]++;
        int cur_wins=wins[w[i]];
        int max_wins = wins[cur_max];
        if(cur_wins>max_wins)
        {
            cur_max=w[i];
        }
        else if(cur_wins==max_wins)
        {
            cur_max = max(w[i],cur_max);
        }
        cout<<cur_max<<" "<<wins[cur_max]<<endl;
    }


return 0;
}
