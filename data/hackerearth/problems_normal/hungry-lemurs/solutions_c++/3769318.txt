#include <iostream>
using namespace std;

int abs(int a) {
	return (a > 0) ? a : -a;
}

int min(int a, int b) {
	return (a > b) ? b : a;	
}

int main()
{
    int k, n;
    cin >> n >> k;
    
    if (n%k == 0) {
    	cout << 0;
    	return 0;
    }
    
 	int a, b;
 	int m = n;
 	
 	for (int i = 1; i <= k + n; i++) {
 		a = n - (n/i)*i;
 		b = ((n/i)+1)*i - n;
 		m = min(m, min(a,b) + abs(k-i));
 	}
 	
 	cout << m;
    
    return 0;
}
