#include<bits/stdc++.h>
int main()
{
	long long t,n,p,u,r,s,i=1;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld %lld %lld %lld",&n,&p,&u,&r,&s);
		while(n--)
		{
			if(p+r<=u)
			p=p+r;
			else if(p-s>=0)
			p=p-s;
			else;			
		}printf("Case #%lld: %lld\n",i,p);
		i++;
	}return 0;
}
