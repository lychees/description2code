#include <iostream>
using namespace std ;
int main()
{
    long long n ;
    cin >> n;
    long long  a[n] ;
    long long  b[n]  ;
    long long  i ;
    for ( i = 0; i < n ; i++)
    {
        cin >> b[i]  ;
    }
    for ( i = 0 ; i< n ; i++)
    {
        a[i] = b[i]*(i+1) ;
    }
   
    long long  c[1000] ;
    c[0] = a[0] ;

    for ( i = 1; i< n ; i++)
    {

       c[i] = a[i] -a[i-1];

    }
    for ( i = 0; i< n ; i++)
    {
        cout << c[i] << " " ;
    }
    cout << endl;

    return  0;

}
