T = int(raw_input())
def nc2(n):
	return (n*(n-1))/2
for _ in xrange(T):
	N = int(raw_input())
	A = map(int,raw_input().split())
	oc = 0
	ec = 0
	d = dict()
	for x in A:
		if x % 2 == 0:
			ec += 1
		else:
			oc += 1
		if x not in d:
			d[x] = 1
		else:
			d[x] += 1
	#print oc,ec,d
	ans = nc2(oc) + nc2(ec)
	for k in d.keys():
		ans -= nc2(d[k])
	print ans
