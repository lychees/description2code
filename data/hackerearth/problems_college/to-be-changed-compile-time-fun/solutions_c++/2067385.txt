#include <iostream>
#include <algorithm>
using namespace std;

int main(){
    int n,i,j,k,ma,mark,check,sum=0;
    cin>>n;
    int num[n];
    for(i=0;i<n;i++)
        cin>>num[i];
    for(i=0;i<n-2;i++){
        ma = num[0]*num[2];
        mark=1;
        for(j=1;j<=n-2-i;j++){
            check = num[j-1]*num[j+1];
            if(ma < check){
                ma = check;
                mark=j;
            }
        }
        sum+=ma;
        for(k=mark;k<n-1-i;k++)
            num[k] = num[k+1];
    }
    cout<<sum<<endl;
}
