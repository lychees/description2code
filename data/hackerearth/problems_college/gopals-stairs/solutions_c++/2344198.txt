#include <iostream>
using namespace std;
#define mod 1000000007
int main() {
	int a[100005]={0};
	int t,n,i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		a[0]=a[1]=1;
		a[2]=2;
		a[3]=4;
		for(i=4;i<=n;i++)
		{
			if(a[i]==0)
			a[i]=((a[i-1]%mod+a[i-2]%mod)%mod+a[i-3]%mod)%mod;
		}
		printf("%d\n",a[n]);
	}
	
	return 0;
}
