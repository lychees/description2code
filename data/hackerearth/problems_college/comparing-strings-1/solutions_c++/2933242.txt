#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
bool check(string a,string b)
{
	if( b.find(a) != std::string::npos ){
		return true;
	}//if( b.substr(a) )return True;
	
	return false;
}
bool fun(string a,string b)
{
	string s = "";
	for(int i = 0;i< a.size();i++){
		s = "";
		for(int j = i;j < a.size(); j++){
			s += a[j];
			if( check(s,b) )return true;
		}
	}
	return false;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string a,b;
    	bool f;
    	cin >> a;cin >> b;
    	if( a.size() <= b.size() ){
    		f = fun( a, b );
    	}else{
    		f = fun( b, a );
    	}
    	if(f)cout << "YES\n";
    	else cout << "NO\n";
    }
    return 0;
}
