#include <bits/stdc++.h>
using namespace std;
long long dp[ 10005],a[10005];
int main()
{
   int t,n;
   cin>>t;
   while(t--){ 
   	memset(dp,0,sizeof(dp));
   	cin>>n;
   	for(int i=1;i<=n;++i)
   	cin>>a[i];
   	dp[1]=a[1];
   	dp[2]=max(a[1],a[2]);
   	for(int i=3;i<=n;++i){ 
   		dp[i]=max(dp[i-1],dp[i-2]+a[i]);
   		//cout<<dp[i-2]+a[i]<<" ";
   	}
   	cout<<dp[n]<<endl;
   }
    return 0;
}
