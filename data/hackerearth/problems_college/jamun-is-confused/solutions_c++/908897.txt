#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int gcd(int a,int b)
{
if(b==0)
return a;
else
return gcd(b,a%b);
}
int main()
{
	int t;
	cin>>t;
	int k=1;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		int t=gcd(a[0],a[1]);
		for(int i=2;i<n;i++)
		t=gcd(t,a[i]);
		cout<<t<<endl;
	}
}
