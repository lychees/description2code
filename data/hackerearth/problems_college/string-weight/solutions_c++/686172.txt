#include <iostream>
#include <map>
#define mci map<char,int>
#define c cin
#define o cout
#define el endl
using namespace std;

string let="abcdefghijklmnopqrstuvwxyz";
mci weight;

void set(){
	for(int i=0;i<26;i++) weight[let[i]]=i+1;
}

int main(){
	set();
	int t; c>>t;
	while(t--){
		string in; c>>in;
		int tmp=0, len=in.length();
		for(int i=0;i<len;i++){
			tmp+=weight[in[i]];
		}
		int skip; c>>skip;
		char skipped;
		while(skip--){
			c>>skipped;
			for(int i=0;i<len;i++){
				if(in[i]==skipped) tmp-=weight[skipped];
			}
		}
		o<<tmp<<el;
	}
}
