import math


def is_prime(n):
    for i in range(2, int(math.sqrt(n))+1):
        if n % i == 0:
            n = 0
            break
    return n


t = int(raw_input())
for i in range(0, t):
    s = raw_input()
    s = s.split()
    sum = 0
    if int(s[1]) < 2:
        sum = 0
    if int(s[0]) < 2:
        s[0] = "2"
    for j in range(int(s[0]), int(s[1])+1):
        sum = sum + is_prime(j)
    print sum
