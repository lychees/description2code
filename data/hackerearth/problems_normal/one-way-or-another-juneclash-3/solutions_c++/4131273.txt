#include <iostream>
using namespace std;

int main()
{
    int t,n ;
    cin >> t;
    
    while(t--){
    	bool flag = true;
    	cin >> n;
    	long long x,y,z;
    	cin >> x >> y >> z;
        bool pr = ((x+y)%2 == (z%2));
        for(int i=1;i<n;i++){
        	cin >> x >> y >> z;
        	if(pr != ((x+y)%2 == (z%2)))
        		flag = false;
        }
        if(flag) cout << "Yes" << endl;
        else cout << "No" << endl;
    	
    }
    return 0;
}
