{

        {
            cout<<-1<<"\n";
         continue;
        }
        if(n==3 && a[0]+a[1]<=a[2])
        {
            cout<<-1<<"\n";

        }int f=0;

            }

        {
            cout<<a[j]<<" "<<q<<" "<<p<<"\n";
        }

            cout<<-1<<"\n";
        }
    }
return 0;
}
