for i in range(int(raw_input())):
    a,b=map(int,raw_input().split())
    a=bin(a)[2:].zfill(32)
    b=bin(b)[2:].zfill(32)
    #print a,b
    a=list(a)
    b=list(b)
    j=0
    cnt=0
    while j<len(a):
        if a[j] == b[j]:
            cnt=cnt+0
        else:
            cnt=cnt+1
        j=j+1
    print cnt
