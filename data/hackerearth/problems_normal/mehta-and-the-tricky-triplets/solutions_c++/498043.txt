#include<stdio.h>
#include<cstring>
#include<string>
#include<iostream>
using namespace std;
typedef long long int ll;
ll a[100002];
int num[16];
bool two_p[100002],three_p[100002],five_p[100002],seven_p[100002];
ll choose(ll n)
{
	return (n*(n-1)*(n-2))/6;
}
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		two_p[i]=three_p[i]=five_p[i]=seven_p[i]=false;
		while(a[i]>0)
		{
			if(a[i]%10==2)
			{
				two_p[i]=true;
			}
			if(a[i]%10==3)
			{
				three_p[i]=true;
			}
			if(a[i]%10==5)
			{
				five_p[i]=true;
			}
			if(a[i]%10==7)
			{
				seven_p[i]=true;
			}
			a[i]/=10;
		}
	}
	for(int i=0;i<n;i++)
	{
		num[0]+=two_p[i];
		num[1]+=three_p[i];
		num[2]+=five_p[i];
		num[3]+=seven_p[i];
		num[4]+=(two_p[i] && three_p[i]);
		num[5]+=(two_p[i] && five_p[i]);
		num[6]+=(two_p[i] && seven_p[i]);
		num[7]+=(three_p[i] && five_p[i]);
		num[8]+=(three_p[i] && seven_p[i]);
		num[9]+=(five_p[i] && seven_p[i]);
		num[10]+=(two_p[i] && three_p[i] && five_p[i]);
		num[11]+=(two_p[i] && three_p[i] && seven_p[i]);
		num[12]+=(two_p[i] && five_p[i] && seven_p[i]);
		num[13]+=(three_p[i] && five_p[i] && seven_p[i]);
		num[14]+=(two_p[i] && three_p[i] && five_p[i] && seven_p[i]);
	}
	ll ans=0;
	ans=ans+choose(num[0])+choose(num[1])+choose(num[2])+choose(num[3])-choose(num[4])-choose(num[5])-choose(num[6])-choose(num[7])-choose(num[8])-choose(num[9])+choose(num[10])+choose(num[11])+choose(num[12])+choose(num[13])-choose(num[14]);
	cout<<ans<<endl;
}
