#include <iostream>
#include<stdio.h>
#include<cstring>
using namespace std;
string s;
int check(int l,int r){
  while(l<r){
  	if(s[l]!=s[r]) return 0;
  	l++;
  	r--;
  }
  return 1;
}
int main()
{
	int t;
    scanf("%d",&t);
    while(t--){
    cin>>s;
    int len=s.length();
  
    int maxi=0,i;
    int l,r;
    for(i=0;i<len;i++){
   		r=len-1,l=i;
   	    int  flag = check(l,r);
   	    if(flag)
   	    maxi = max(maxi,r-l+1);
    }
  
   printf("%d\n",len+len-maxi);
    
    }
    return 0;
}
