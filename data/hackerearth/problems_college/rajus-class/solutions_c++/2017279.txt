#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		priority_queue<int,vector<int>,greater<int> > mypq;  
		map<int,string>mymap;
		int n;
		scanf("%d",&n);
		while(n--)
		{
			int temp;
			string t;
			cin>>temp>>t;
			mymap[temp]=t;
			mypq.push(temp);
		}
		int rollnum;
		scanf("%d",&rollnum);
		rollnum--;
		while(rollnum--)
		{
			mypq.pop();
		}
		cout<<mymap[mypq.top()]<<endl;
	}
}
