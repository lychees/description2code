#include<stdio.h>
int main()
{
    int n,q,a,b,c,i,count;
    scanf("%d%d",&n,&q);
    int coin[n];
    for(i=0;i<n;i++)
    	coin[i]=0;
    while(q--)
    {
    	count=0;
    	scanf("%d%d%d",&c,&a,&b);
    	if(c==0)
    	{
    	   	for(i=a;i<=b;i++)
    			coin[i]=1;
    	}
    	else
    	{
    		for(i=a;i<=b;i++)
    		{
    			if(coin[i]==1)
    				count++;
    		}
    		printf("%d\n",count);
    	}
    }
    return 0;
}
