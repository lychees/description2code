#include <queue>
#include <iostream>
#include <vector>
using namespace std;
int main()
{

    int x,y;
    cin >> x >> y;
    priority_queue<int> a;

    while(x--)
    {

        int q;
        cin >> q;
        a.push(q);

    }
    int t=0;
    while(y--)
    {
        int w=a.top();
        t+=w;
        w--;
        a.pop();
        a.push(w);

    }
    cout << t << endl;
}
