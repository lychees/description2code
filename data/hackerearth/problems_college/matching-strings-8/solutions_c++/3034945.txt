#include <iostream>
#include <string>
#include <math.h>

using namespace std;
int T,i,j,k,n,m;
string a,b;
int main()
{   cin>>T;
    while(T--)
    {
       cin>>a>>b;
       n=a.length();
       m=b.length();
       
       if(a[0]=='\0' || b[0]=='\0')
          cout<<0<<endl;
     else
     {
       k=0;i=0;j=0;
       while(i<n && j<m)
       {
       	  if(a[i]==b[j])
       	     k++;
       	  i++;j++;   
       	  
       }
       cout<<k<<endl;
     }
     a[0]='\0';
     b[0]='\0';
    }
    
    return 0;
}
