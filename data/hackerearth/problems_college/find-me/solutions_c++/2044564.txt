#include<iostream>
#define ll long long int
using namespace std;
int main()
{
ll t;
cin>>t;
while(t--)
{
ll a,b;
cin>>a>>b;
ll s[a],c=0,count=0;
for(int i=0;i<a;i++)
{
cin>>s[i];
if(s[i]==b)
c=1;
if(s[i]>b)
count++;
}
if(c==1)
cout<<"YES "<<count<<endl;
else
cout<<"NO 0"<<endl;
}
return 0;
}
