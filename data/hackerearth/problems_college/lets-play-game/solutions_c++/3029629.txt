#include <iostream>
using namespace std;

#define lld long long int

int main()
{
    //printf("Hello World!\n");
    
    lld tc, n, t, i;
    
    cin >> tc;
    
    while (tc--) {
    	cin >> n >> t;
    	lld timep[n], profit[n];
    	for (i = 0; i < n; i++) {
    		cin >> timep[i];
    		timep[i] = t / timep[i];
    	}
    	lld sum = 0;
    	for (i = 0; i < n; i++) {
    		cin >> profit[i];
    		sum = max(timep[i] * profit[i], sum);
    	}
    	
    	cout << sum << endl;
    	
    }
    
    return 0;
}
