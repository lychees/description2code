#include <iostream>

int main()
{
	std::string s;
	int n, max, min, last = 0;
	std::cin >> n >> s;
	max = n;
	min = 0;
	for (int i = 0; i < n; i++) {
		if (s[i] == 'w') {
			max++;
			last += 2;
		}
		if (s[i] == 'v') {
			last += 1;
		}
		if (s[i] !='w' && s[i] != 'v') {
			min++;
			min += last - last / 2;
			last = 0;
		}
		if (i == n - 1) {
			min += last - last / 2;
		}
	}
	std::cout << min << " " << max;
}
