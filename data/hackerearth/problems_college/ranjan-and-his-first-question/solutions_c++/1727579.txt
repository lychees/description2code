#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,m;
		scanf("%lld %lld",&n,&m);
		long long int first = n/m;
		long long int second = n%m;
		long long int ans = 0;
		ans += first*((m*(m-1))/2);
		ans += (second*(second+1))/2;
		printf("%lld\n", ans);
	}
	return 0;
}
