arr = [0.000000]*1000010


def main():
    n, t = map(int, raw_input().split())
    p = float(t)/float(100)
    arr[0] = 1.000000
    arr[2] = p
    arr[3] = 1.000000 - p
    for i in range(4, n+1):
        arr[i] += (arr[i-2]*p)
        arr[i] += (arr[i-3]*(1 - p))
    print("%.6f" % arr[n])


main()
