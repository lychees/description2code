#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int n,i,k,j;
	cin>>n>>k;
	int  a[n];
	int p=0;
	for(i=0,j=n-1;j>=i;j--,i++)
	{
		a[p++]=j;
		a[p++]=i;
	}
	int c=0;
	for(i=0;i<n;i++)
	{
		if(a[i]==k)
		{
			break;
		}
		c++;
	}
	
	cout<<c<<"\n";

	}
}
