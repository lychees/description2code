#include<iostream>
#include<stdlib.h>
using namespace std;
main()
{
    long long int n,l,r,i,j,q,*a,*ans,t=0,s;
    cin>>n;
    a=(long long int*)calloc(n,sizeof(long long int));
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    cin>>q;
    ans=(long long int*)calloc(q,sizeof(long long int));
    t=0;
    for(j=1;j<=q;j++)
    {
        cin>>l>>r;
        s=0;
        for(i=l;i<=r;i++)
        {
            s=s+a[i];
        }
        ans[t]=s;
        t++;
    }
    t--;
    for(j=0;j<=t;j++)
    {
        cout<<ans[j]<<endl;
    }
}
