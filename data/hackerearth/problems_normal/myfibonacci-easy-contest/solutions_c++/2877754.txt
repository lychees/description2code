#include<iostream>
using namespace std;
int main()
{
	int n,prev,sum=0,num;
	cin>>prev;
	cin>>num;
	cin>>n;
	int i=3;
	if(n==1)
	{
		cout<<prev;
	}
	else
	{
		if(n==2)
	{
		cout<<num;
	}
	
	else
	{
	while(i<=n)
	{
	 sum=num+prev;
	 prev=num;
	 num=sum;
	 i++;
	}
	cout<<sum;
	}
	}
	return 0;
}
