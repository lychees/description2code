#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);

using namespace std;
int check7(int n){
    int t;
    while(n){
        t=n%10;
        if(t==7)return 1;
        n=n/10;
    }
    return 0;
}
int A[1000001];
void preprocess()
{
    A[1]=0;A[2]=0;A[3]=0;
    
    for(int i=2;i<=1000;i++)
    {
        for(int j=2*i;j<=1000000;j+=i)
            A[j]=1;
    }
}
int main(int argc, char** argv) {

    int t;
    preprocess();
    cin>>t;
    while(t--){
        int m,n,count=0;
        cin>>m>>n;
        for(int i=m;i<=n;i++){
        //    if(A[i]==1)cout<<i<<endl;
            if(A[i]==1 && check7(i)==1)count++;
        }
        if(count)printf("%d\n",count);
        else cout<<"-1"<<endl;
    }
    return 0;
}
