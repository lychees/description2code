#include<bits/stdc++.h>
using namespace std;

#define ll long long 
bool compare(pair<ll,ll> a,pair<ll,ll> b)
{
	return a.first<b.first;
}
int main()
{
	ios::sync_with_stdio(false);
	ll t,n,q,a,b,f,s,j,k,i;
	
	
	cin>>t;
	while(t--)
	{
		pair<ll,ll>p;
	vector<pair<ll,ll > >vec;
	   vec.clear();
		cin>>n>>q;
	
		for(int i=1;i<=n;i++)
		{
			cin>>a>>b;
		   p=make_pair(a,b);
		   vec.push_back(p);
			
		}
		ll sum[1000]={0};
		sort(vec.begin(),vec.end(),compare);
		sum[1]=vec[0].first;
		sum[2]=vec[0].second;
		j=3;
		for(int i=1;i<vec.size();i++)
		{
			a=vec[i].first;
			b=vec[i].second;
			if(a>=sum[j-2]&&a<=sum[j-1])
			{
				if(b>sum[j-1])
			    sum[j-1]=b;
			}
			else if(a>sum[j-1])
			{
				sum[j++]=0;
				sum[j++]=a;
				sum[j++]=b;
			}
			
		}
		sum[j++]=0;
		while(q--)
		{
			cin>>k;
			f=0;
			for( i=1;i<j;i=i+3)
			{
	            f=0;
				s=sum[i+1]-sum[i]+1;
				if(s>=k)
				{
					
					cout<<sum[i]+k-1<<endl;
					f=1;
					break;
				}
				k=k-s; 
			}
			if(f==0) cout<<"-1\n";
		}
	}
		
return 0;
}
