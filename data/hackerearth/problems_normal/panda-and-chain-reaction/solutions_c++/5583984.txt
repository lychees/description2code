#include <iostream>
#include <iomanip>
#include <limits>
#include <cstdio>
#include <cstdlib>
#include <vector>

using namespace std;

typedef unsigned long long ll;

ll M=1000003;

ll fact[1000003];

void Factorial(){
	ll res=1;
	ll N,i;

	fact[0]=1;
	fact[1]=1;

	for(i=2;i<=1000002;++i)
	{
		res=( (i%M) * (res) )%M;
		fact[i]=res;
	}

	return ;
}

int main() {
	ll noOfInputs;
	ll N,X;
	
	int i=0;
	cin>>noOfInputs;
	Factorial();
	
	while(noOfInputs != 0 )
	{
		cin>>N;
		cin>>X;
		if( N >= M )
		{
			cout<<0<<endl;
			--noOfInputs;
			continue;
		}
		cout<< ((X%M) * fact[N])%M <<endl;
		--noOfInputs;
	}

	getchar();
	getchar();
	return 0;
}
