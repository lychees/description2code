#include<iostream>
using namespace std;
int main(){
int T,hr,min;
int angle_hr,angle_min,temp_angle;
cin>>T;
while(T>0){
cin>>hr>>min;
if(hr==12)
	angle_hr=0.5*min;
else
	angle_hr=(hr*30)+(0.5*min);
angle_min=min*6;
if(angle_hr>angle_min)
	temp_angle=angle_hr-angle_min;
else
	temp_angle=angle_min-angle_hr;
if(temp_angle<(360-temp_angle))
	cout<<temp_angle<<endl;
else{
	temp_angle=360-temp_angle;
	cout<<temp_angle<<endl;
}
T--;
}
return 0;
}
