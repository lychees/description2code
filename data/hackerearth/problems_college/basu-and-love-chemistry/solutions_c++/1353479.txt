#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
char str[200010];
int main()
{
cin>>str;
long long ans = str[0] - 48,ans2 = str[0]-48,l = strlen(str);
for(long long i = 1; i < l; i++)
{
long long h = (str[i] - 48);
ans = (ans*10 + h*(i+1))%MOD;
ans2+=ans;
ans2 = ans2%MOD;
}
cout<<ans2<<endl;
}
