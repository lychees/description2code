#include <iostream>
using namespace std;

int main()
{
	int n,t;
	int dp[100001];
	dp[1]=dp[3]=dp[4]=1;
	dp[2]=0;
	for(int i=5;i<100000;i++)
	{
		if(dp[i-4]==1&&dp[i-1]==1)
		dp[i]=0;
		else
		dp[i]=1;
	}
	
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(dp[n]==1)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    }
    return 0;
}
