#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    while(t--){
    	long int n,k,i,count=0,a;
    	
    	cin>>n>>k;
    	for(i=0;i<n;i++){
    		cin>>a;
    		if(a<=0)count++;
    	}
    	
    	if(k>count) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
