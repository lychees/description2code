	n,k=map(int,raw_input().split())
	a=map(int,raw_input().split())
	pos=[False]*(k+1)
	c=[False]*(k+1)
	for x in a:
		y=x
		while y<=k:
			c[y]=True
			y+=x
	
	p=list(c)
	p[0]=True
	i=0
	while i<=k:
		if (c[i]):
			j=i
			while j<=k:
				if (c[i] and c[j] and i+j<=k):
					p[i+j]=True
				j+=1
		i+=1
	
	i=k
	while i>=0:
		if (p[i]==True):
