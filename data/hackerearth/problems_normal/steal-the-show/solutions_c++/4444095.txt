#include<iostream>
#include<cmath>
using namespace std;
int main()
{int sum=0,ceo,coo,cto,members[20],n,temp,m,val[30],j;
 cin>>ceo>>coo>>cto;
 cin>>n;
 for(int i=0;i<n;i++)
  cin>>members[i];
 for(int i=0;i<n;i++)
  {for(j=i;j<n;j++)
    {if(members[i]>members[j])
      {temp=members[i];
       members[i]=members[j];
       members[j]=temp;
      }
    }
  }
 m=n+3;
 j=0;
 for(int i=0;i<m-1;i++)
  {if(i==0)
    {if(cto<coo)
      {val[0]=cto;
       val[m-1]=coo;
      }
     else
      {val[0]=coo;
       val[m-1]=cto;
      }
    }
   else if(i!=(m-1)/2)
    {val[i]=members[j];
     j++;
    }
   else if(i==(m-1)/2)
    {if(m%2==1)
      val[i]=ceo;
     else
      {if(abs(ceo-val[i-1])<abs(members[j]-val[i-1]))
        val[i]=ceo;
       else
        {val[i]=members[j];
         members[j]=ceo;
        }
      }
    }
  }
 for(int i=1;i<m;i++)
  sum+=abs(val[i]-val[i-1]);
 cout<<sum<<endl;
 return 0;
  
}
