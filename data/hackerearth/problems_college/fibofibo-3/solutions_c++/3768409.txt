#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n==0)cout<<1<<" "<<0<<"\n";
    	else if(n==1)cout<<0<<" "<<1<<"\n";
    	else
    	{
    		int dp0[50],dp1[50];
    		dp0[2] = 1;dp0[3]=1;
    		dp1[2] =1;dp1[3] = 2;
    		for(int i=4;i<=n;i++)dp0[i]=dp0[i-1]+dp0[i-2];
    		for(int i=4;i<=n;i++)dp1[i] = dp1[i-1] + dp1[i-2];
    		cout<<dp0[n]<< " " << dp1[n] << "\n";
    	}
    }
    return 0;
}
