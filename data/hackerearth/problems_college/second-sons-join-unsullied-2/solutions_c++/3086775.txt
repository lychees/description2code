
#include<bits/stdc++.h>
using namespace std;

string s1;
int main()
{
    int n,sum;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        sum=0;
        cin>>s1;
        int l=s1.length();
        for(int i=0;i<l;i++)
            if(s1[i]=='U')
                sum++;
            else
                sum--;

    if(abs(sum)>=2)
    {
        printf("-1\n");
        continue;
    }
    if(l%2) //length is odd
    {
        char x,y;
        if(sum>0)
        {
            x='U';
            y='S';
        }
    else
    {

            x='S';
            y='U';
    }
    int counts=0;
    for(int i=0;i<l;i++)
    {
        if(i%2)
        {
            if(s1[i]!=y)
                counts++;
        }
        else
        {
            if(s1[i]!=x)
                counts++;
        }
    }
    printf("%d\n",counts/2);
    }
else
{ char x,y;
    int xx=0;
    x='U';
    y='S';
    for(int i=0;i<l;i++)
    {
        if(i%2)
        {
            if(s1[i]!=y)
                xx++;
        }
        else
        {
            if(s1[i]!=x)
                xx++;
        }
    }
    swap(x,y);
    int yy=0;
    for(int i=0;i<l;i++)
    {
        if(i%2)
        {
            if(s1[i]!=y)
                yy++;
        }
        else
        {
            if(s1[i]!=x)
                yy++;
        }
    }
    printf("%d\n",min(xx,yy)/2);
}
    }

 return 0;
}
