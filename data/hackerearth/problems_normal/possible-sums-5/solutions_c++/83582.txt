#include <cstdio>
#include <vector>
#include <set>

using namespace std;

int main() {

	int t,n,a;
	set<int> sums;
	vector<int> nsums;
	scanf("%d", &t);

	while(t--) {
		scanf("%d", &n);
		sums.clear();
		sums.insert(0);
		while(n--) {
			scanf("%d", &a);
			nsums.clear();
			for(set<int>::iterator it=sums.begin(); it!=sums.end(); ++it) nsums.push_back(*it+a);
			for(vector<int>::iterator it=nsums.begin(); it!=nsums.end(); ++it) sums.insert(*it);
		}
		printf("%u\n", (unsigned int)sums.size());
	}
	return 0;
}
