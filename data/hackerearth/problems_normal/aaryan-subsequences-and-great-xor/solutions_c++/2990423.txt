#include<set>
#include <iostream>
using namespace std;

int main()
{set<int>s;
	int n;
	cin>>n;
	int x;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		s.insert(x);
	}
	
	set<int>::iterator it;
	it=s.begin();
	int t=1;
	for(it=s.begin();it!=s.end();it++)
	{
		t=t|*it;
	}
	
	cout<<t<<endl;
    //cout << "Hello World!" << endl;
    return 0;
}
