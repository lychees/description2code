#include <iostream>
#define ll long long
using namespace std;


int main()
{
    ll int t;
    cin>>t;
    
    while(t--)
    {
    	ll int x;
    	cin>>x;
    	ll int zero=0;
    	if(x<5)
    		cout<<0<<endl;
    	else
    	{
    		long y=5;
    		while(y<=x)
    		{
    			zero=zero+x/y;
    			y*=5;
    		}
    		cout<<zero<<endl;
    	}
    }
    
    return 0;
}
