#include <iostream>
using namespace std;

int main()
{
    int tc,i,flag; char str[101];
    cin>>tc;
    while(tc--)
    {
    	cin>>str;
    	i=flag=0;
    	while(str[i+1]!='\0')
    	{
    		if(str[i]==str[i+1])
    		{
    			flag=1; break;
    		}
    		i++;
    	}
    	if(flag==1)
    		cout<<"SLAP"<<"\n";
    	else
    		cout<<"KISS"<<"\n";
    }
    return 0;
}
