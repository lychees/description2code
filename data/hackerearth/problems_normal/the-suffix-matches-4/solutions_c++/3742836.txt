#include<bits/stdc++.h>
#include<string>
using namespace std;
#define ll long long int
vector<ll> Z(200001);//
void getZarr(string str)//
{
    reverse(str.begin(),str.end());
    int n = str.length();
    int L, R, k;
    L = R = 0;
    Z[0]=n;
    //cout<<Z[0]<<" ";
    for (int i = 1; i < n; ++i)
    {
        if (i > R)
        {
            L = R = i;
            while (R<n && str[R-L] == str[R])
                R++;
            Z[i] = R-L;
            R--;
        }
        else
        {
            k = i-L;
            if (Z[k] < R-i+1)
                 Z[i] = Z[k];
            else
            {
                L = i;
                while (R<n && str[R-L] == str[R])
                    R++;
                Z[i] = R-L;
                R--;
            }
        }
        //cout<<Z[i]<<" ";
    }
}
