#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)

using namespace std;

int main()
{
	ll i, j, k, n, m;
	gt(n);
	ll a[n];
	for(i = 0; i < n; i++) 
		gt(a[i]);
	sort(a, a + n);
	ll ct = 0, t = 0;
	for(i = n - 1; i >= 0; i--) {
		if(i == n - 1) {
			ct += a[i] + 1;
			t = a[i];
		}
		else {
			t--;
			if(t < a[i]) {
				ct += a[i] - t;
				t = a[i];
			}
		}
	}
	cout << ct + 1 << endl;
	return 0;
}
