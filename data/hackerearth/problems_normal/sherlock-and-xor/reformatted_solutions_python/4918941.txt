T = input()


while T > 0:
    T -= 1
    lenl = input()
    l = map(int, raw_input().split(" "))
    E = 0
    O = 0
    P = 0
    while lenl > 0:
        lenl -= 1
        x = l.pop()
        if x % 2 == 0:
            # Even
            P += O
            E += 1
        else:
            # Odd
            P += E
            O += 1
    print P
