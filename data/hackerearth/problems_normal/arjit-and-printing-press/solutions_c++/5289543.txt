#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    long long t,i,j;
    string w,r;

    cin>>t;
    while(t--)
    {
        cin>>w>>r;
        sort(r.begin(),r.end());
        i=0; j=0;
        while(i<w.size()&&j<r.size())
        {
            if(w[i]>r[j])
            {
                w[i]=r[j];
                i++;
                j++;
            }
            else
            {
                i++;
            }
        }
        cout<<w<<endl;
    }
    return 0;
}
