int joseph(int n)
{
	if (n == 1 || n == 2){
		return 1;
	}
	if (n == 3){

		return 3;
	}
	if (n % 2 == 0){
		return  2 * joseph(n / 2) - 1;
	}
	else{
		return 2 * joseph((n - 1) / 2) + 1;
	}


}
