#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
map<ll,int> m;

int main()
{
    int n,mm;
    cin>>n>>mm;
    ll a[n+1];
    for(int i=0;i<n;i++) cin>>a[i];
    cout<<a[0]<<" "<<1<<"\n";
    if(m.count(a[0])==0) m[a[0]]=1;
    ll age=a[0],cnt=1;
    for(int i=1;i<n;i++){
    	if(m.count(a[i])==0) m[a[i]]=1;
    	else m[a[i]]++;
    	if(m[a[i]]==cnt){
    		if(a[i]>age){
    			age=a[i];
    			
    		}
    	}
    	else if(m[a[i]]>cnt){
    		age=a[i];
    		cnt=m[a[i]];
    		
    	}
    	cout<<age<<" "<<cnt<<"\n";
    	
    }
    
}
