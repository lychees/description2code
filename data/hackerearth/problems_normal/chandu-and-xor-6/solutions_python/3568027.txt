a,b= map(int,raw_input().split(" "))
z=[]
k=[]
l=[]
xor=[]
p=[]
for i in range (0,a):
    z.append(map(int,raw_input().split()))
l=map(list,zip(*z))
#print z
#print l

for i in range (0,b):
    s=0
    for j in range (0,a):
        s=s+z[j][i]
    p.append(s)
#print p

for i in range (0,a):
    s=0
    for j in range (0,b):
        s=s+l[j][i]
    k.append(s)
#print k

for i in k:
    for j in p:
        a=i^j
        xor.append(a)

print max(xor)
