def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, pow(a, 1, b))

def make_coprime_filter(X):
    def coprime_filter(A):
        return gcd(A, X) == 1
    return coprime_filter

T = int(raw_input())
for i in range(T):
    _, X = map(int, raw_input().split())
    As = map(int, raw_input().split())
    co_filter = make_coprime_filter(X)
    r = filter(co_filter, As)
    print " ".join(map(str, r))
