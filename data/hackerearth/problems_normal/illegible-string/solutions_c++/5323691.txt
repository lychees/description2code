#include<iostream>
using namespace std;
int main()
{
	int a;
	string str,tmp="",tmp2="";
	cin>>a;
	cin>>str;
	for(int x=0; x<a; x++)
	{
		if(str[x] == 'w') tmp += "vv";
		else tmp += str[x];
	}
	for(int x=0; x<tmp.size(); x++)
	{
		if(x != tmp.size()-1 &&tmp[x] == 'v' && tmp[x+1] == 'v') {tmp2 += "w"; x++;}
		else tmp2 += tmp[x];
	}
	cout<<tmp2.size()<<' '<<tmp.size();
	return 0;
}
