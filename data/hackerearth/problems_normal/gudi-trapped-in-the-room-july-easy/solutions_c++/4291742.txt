#include <iostream>
using namespace std;
#include<set>
void dfs(string str,std::set<string> &s,int a,int h)
{
	s.insert(str);
	string next=str;
	for(int i=0;i<str.length();i++)
	{
		//next[(i+h)%str.length()]=str[i];
		next[i]=str[(str.length()-h%str.length()+i)%str.length()];
	}
	if(s.find(next)==s.end())
	dfs(next,s,a,h);
	
	next=str;
	for(int i=0;i<next.length();i++)
	{
		if((i+1)%2==0)
		next[i]=(next[i]-'0'+a)%10 + '0';
	}
	if(s.find(next)==s.end())
	dfs(next,s,a,h);
}

int main() 
{
	int t,a,h;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s>>a>>h;
		std::set<string> se;
		dfs(s,se,a,h);
		//for(auto i:se)
		//cout<<i<<" ";
		//cout<<endl;
		cout<<*(se.begin())<<endl;
	}
	// your code goes here
	return 0;
}
