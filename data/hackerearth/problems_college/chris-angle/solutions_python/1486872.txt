from fractions import gcd
t=int(raw_input())
while(t):
	angle=int(raw_input())
	if angle>=360:
		angle=angle-360*(angle//360)
	g=gcd(angle,180)
	rad=180
	
	while(g!=1):
		angle=angle/g
			
		rad/=g
		g=gcd(angle,rad)
	
	if angle==1 and rad!=1:
		print "$"+"/"+str(rad)
	elif angle!=1 and rad==1:
		print str(angle)+"$"
	elif angle==1 and rad==1:
		print "$"
	else:
		print str(angle)+"$"+"/"+str(rad)		
		
	t-=1
