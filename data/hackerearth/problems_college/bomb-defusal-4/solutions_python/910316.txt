t = input()
for T in xrange(t):
	P,x,y,u,v = [int(x) for x in raw_input().split()]
	if x == u and y == v:
		print 'Counter-Terrorists Win !'
	elif abs(u-x) == abs(y-v) and P>1:
		print 'Counter-Terrorists Win !'
	elif (abs(u-x)+abs(v-y)) % 2 == 0 and P> 2:
		print 'Counter-Terrorists Win !'
	else:
		print 'Terrorists Win !'
