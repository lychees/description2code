using namespace std;

typedef unsigned long long UInt64;

const int N = 1111;
const UInt64 Mod = ( UInt64 )( 1e9 + 7 );

int main()
{
	int T;
	cin >> T;
	while( T-- ){
		
		char door[N];

		cin >> door;

		int cnt = 0, ans = 0, flag = 0, l = strlen( door );
		
		for( int i = 0; i < l; ++i ){
			if( door[i] == '_' ){
				cnt++;
			}
			else{
				ans = ans^cnt;
				flag = 1;
				cnt = 0;
			}
		}
		if( flag && cnt )		ans = ans^cnt;

		if( ans )	cout << "Alice\n";
		else   cout << "Bob\n";

	}
	return 0;
}
