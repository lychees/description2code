import math
import sys

input = []
days = []
limit = 0
i = 0
for line in sys.stdin:
    if limit == 0:
        limit = int(line)
    else:
        if i % 2 == 1:
            input.append(line)
        else:
            days.append(int(line))

    i = i+1
    if i == (limit*2)+1:
        break

for i in range(0, limit):
    if (days[i]*(days[i]+1)/2) % 2 == 1:
        print(input[i].count('0'))
    else:
        print(input[i].count('1'))
