n=int(raw_input())
dom=map(int,raw_input().split())
brian=map(int,raw_input().split())
d=[]
b=[]
for i in range(n-1):
    d.append(abs(dom[i+1]-dom[i]))
    b.append(abs(brian[i+1]-brian[i]))
if max(d)==max(b):
    print "Tie"
elif max(d)>max(b):
    print "Dom"
    print max(d)
elif max(d)<max(b):
    print "Brian"
    print max(b)
