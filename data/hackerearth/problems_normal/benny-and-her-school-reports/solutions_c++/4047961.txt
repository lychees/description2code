#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,m,x;
		scanf("%lld%lld%lld",&n,&m,&x);
		ll p,sum=0;
		for(ll i=0;i<n;i++)
		{
			scanf("%lld",&p);
			sum+=p;
		}
		ll res=x*(n+1)-sum;
		if(res>m)
		printf("Impossible\n");
		else
		{
			if(res>1)
			printf("%lld\n",res);
			else
			printf("1\n");
		}
	}
	return 0;
}
