#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	double a,b,c,r,result=0;
    	cin>>a>>b>>c;
    	r=(a*b*c)/sqrt((a+b+c)*(a+b-c)*(b+c-a)*(c+a-b));
         result=3.1415*r*r;
        cout << std::fixed;
        cout << setprecision(4);
    	cout<<result<<endl;
    }
    return 0;
}
