#include <bits/stdc++.h>
using namespace std;
#define pii pair<long long,long long>
typedef long long ll;

map<string,string>mymap;

int main()  {
    int n,i,val;
    string str,str1,str2,tmp;
    cin>>n;
    val = 1<<n;
    val--;
    for(i=0;i<val;++i)
    {
        cin>>str1>>str2;
        tmp = str2;
        mymap[str2] = str1;
        if(mymap.find(str1)==mymap.end())
            mymap[str1] = str1;
    }
    str = tmp;
    while(str!=mymap[str])
    {
        str = mymap[str];
    }
    cout<<str<<"\n";
    return 0;
}
