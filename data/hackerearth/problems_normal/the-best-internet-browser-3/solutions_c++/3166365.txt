#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;
int main()
{
    int t,l,i,s = 0;
    cin>>t;
    char arr[200];
    while(t)
    {
        cin>>arr;
        l = strlen(arr);
        for(i = 4;i<l-3;i++)
        {
           if(arr[i]!='a'&&arr[i]!='e'&&arr[i]!='o'&&arr[i]!='i'&&arr[i]!='u')
           {
               s += 1;
           }
        }

            if(arr[l-4]=='c')
            {
                s += 3;
            }
            else
            {
                for(i = l-4;i<l;i++)
                {

             if(arr[i]!='a'&&arr[i]!='e'&&arr[i]!='o'&&arr[i]!='i'&&arr[i]!='u')
            {
                s += 1;
            }
                }
        }
        cout<<s<<"/";
        cout<<l<<endl;
        t--;
        s = 0;
    }
    return 0;
}
