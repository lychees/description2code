#include <iostream>
using namespace std;

#include<stdio.h>
long long int mod = 1000000007;
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
  {
     int n,i;
     scanf("%d",&n);
     long long int ans=1,sum=0;
     for(i=0;i<n-1;i++){
        ans=ans<<1;
        
        ans = (ans % mod);
     }
        for(i = 1 ;i <=n;i++) {
        	sum = ((sum %mod) + ((ans%mod)*(i%mod))%mod)%mod;
        }
      //  ans = ((ans%mod) * (sum%mod))%mod;
        cout<<sum<<endl;
      
   }
   return 0;
}
