#include<iostream>
using namespace std;

int main()

{
    long long int temp=1,t,x,y,sum=0,j=1;
    cin>>t;
    while(t--)
    {
    	sum=0;
        cin>>x>>y;
        for(int i=x;i<=y;i++)
        {
            temp=1;
            for(int j=1;j<=i;j++)
            {
             temp=temp*j;
            }
            sum+=temp;
        }
        cout<<sum%10<<endl;
    }
    return 0;
}
