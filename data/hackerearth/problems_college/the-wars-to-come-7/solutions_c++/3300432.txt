#include<iostream>
using namespace std;
int main()
{
	int t,strength,ws[1000],temp;
	cin>>t;
	for( int i=0; i<t;i++)
		cin>>ws[i];
	cin>>strength;
	for( int i=0; i<t; i++)
		for(int j=0;j<t-1;j++)
			if( ws[j]<ws[j+1])
			{
				temp=ws[j];
				ws[j]=ws[j+1];
				ws[j+1]=temp;
			}
	int count,sum=0;
	for( int i=0; i<t;i++)
	{
		if( sum >= strength)
			break;
		sum+=ws[i];
		count++;
	}
	cout<<count;
	return 0;
}
