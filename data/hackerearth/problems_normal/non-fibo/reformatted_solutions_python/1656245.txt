fibo = []
fibo.append(1)
fibo.append(2)
for i in range(2, 77):
    t = fibo[-1]+fibo[-2]
    if t not in fibo:
        fibo.append(t)


def p(N):
    temp = len([x for x in fibo if x <= N])
    n = N + temp
    while (temp > 0):
        temp = len([x for x in fibo if x <= n and x > N])
        N = n
        n += temp
    print n


for t in range(int(raw_input())):
    N = int(raw_input())
    p(N)
