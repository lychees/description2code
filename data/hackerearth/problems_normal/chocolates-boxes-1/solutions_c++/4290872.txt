#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<int>v(n);
		for(int i=0;i<n;i++)
		cin>>v[i];
		int k;
		cin>>k;
		map<int,int>mp;
		for(int i=0;i<n;i++)
		mp[v[i]]++;
		int c=0;
		for(int i=0;i<n;i++)
		{
			if(mp[k-v[i]])
			c+=mp[k-v[i]];
			
			if(k-v[i]==v[i])
			c--;
		}
		
		cout<<c/2<<endl;
		mp.clear();
		v.clear();
	}
    //cout << "Hello World!" << endl;
    return 0;
}
