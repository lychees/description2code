n = int(raw_input())
c = [0 for i in xrange(30)]
for i in xrange(0, n):
    stri = raw_input()
    no = [int(s) for s in stri.split() if s.isdigit()]
    # print no
    if "M:" in stri:
        for j in no:
            c[j-1] += 1
    if "G:" in stri:
        for j in no:
            c[j-1] += 2
d = sorted(c, reverse=True)
ma = max(c)
# print ma
if d[0] == d[1]:
    print "No Date"
elif c[18] == ma or c[19] == ma:
    print "Date"
else:
    print "No Date"
