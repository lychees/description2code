#include <bits/stdc++.h>
using namespace std;
vector <string> m; 
int main()
{
    string a; cin >> a;int c;cin>>c;
    for (int g=0;g<c;g++)
    {
    	string b; cin >> b;
    	if  (b.length()<a.length()) continue;int r=0; 
    	for (int g=0; g<a.length(); g++)
    	{
    		if (a[g]!=b[g])
    		{
    			r=1; break;
    		}
    	}
    	if (r==0)
    	{
    		m.push_back(b);  
    	}
    }sort(m.begin(), m.end());
    if (m.size()==0)
    {
    	cout << a; return 0; 
    }cout << m[0];return 0;
}
