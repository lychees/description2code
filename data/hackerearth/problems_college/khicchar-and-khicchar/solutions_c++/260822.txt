#include<iostream>
using namespace std;
#include<stdio.h>

int gcd(int x,int y)
{
        if (y == 0)
                return x;
        else if (x == 0)
                return y;
        else
                return gcd(y,x%y);
}

int gcdLarge(int a, char *b, int c)
{
        int val = c, i = 0, y;

        if(b[i] == '\0')
                return gcd(a, c);

        while(val < a && b[i])
        {
                val = val*10 + b[i]-'0';

                i++;
        }

        if(b[i])
                return gcdLarge(a, b+i, val % a);
        else
                return gcd(a, val);
}


int main()
{
        int t, e;

        char a[260],b[260],c[260],d[260];

        scanf("%d", &t);

        while(t--)
        {
                scanf("%s\n%s\n%s\n%s%d", a, b,c,d,&e);


                e=gcdLarge(e, a, 0);
                e=gcdLarge(e, b, 0);
                e=gcdLarge(e, c, 0);
                e=gcdLarge(e, d, 0);

                cout<<e<<endl;
        }

        return 0;
}
