#include <iostream>
#include<algorithm>
using namespace std;
long long a[100000];
int main()
{
  int t;
  cin>>t;
  int n,i,j;
  long long temp,max;
  while(t--)
  {
  	cin>>n;
  	temp=max=1;
  	for(i=0;i<n;i++)
  	  cin>>a[i];
  	sort(a,a+n);
    for(i=1;i<n;i++)
      {
      	if(a[i]==a[i-1])
      	{
      		temp++;
      		if(temp>max)
      		 max=temp;
      	}
        else
          temp=1;
      }
   cout<<max<<endl; 
  }
    return 0;
}
