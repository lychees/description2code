#include<iostream>

using namespace std;

long long dp[10000009],seive[10000009];

void precompute()
{
    dp[1]=0;
    int i;
    for(i=2;i*i<=10000000;i++)
    {
        for(int j=i*2;j<=10000000;j+=i)
            seive[j]=1;
        if(seive[i]==0)
            dp[i]=dp[i-1]+i;
        else
            dp[i]=dp[i-1];
    }
    for(;i<=10000000;i++)
    {
        if(seive[i]==0)
            dp[i]=dp[i-1]+i;
        else
            dp[i]=dp[i-1];
    }

}

int main()
{

    precompute();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x,y;
        scanf("%d%d",&x,&y);

        printf("%lld\n",dp[y]-dp[x-1]);
    }
    return 0;
}
