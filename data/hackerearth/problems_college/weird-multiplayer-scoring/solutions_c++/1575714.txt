#include <cstdio>
#include<map>
#include<iostream>
#include<cstring>
#include<limits.h>
using namespace std;
int main()
{
    int n,m,pt;
    long long max=0;
    map<string,long long> h;
    scanf("%d%d",&n,&m);
    string str,pl;
    for(int i=0;i<m;i++){
    cin>>str>>pt;
    h[str]+=pt;
    if(h[str]>max){max=h[str];pl=str;}
    }
    cout<<pl<<"\n";
    return 0;
}
