#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int t,a,b,c,k,ans,x;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c>>k;
    	for(ans=0;;ans++){
    		x=a*ans*ans+b*ans+c;
    		if(x>=k){
    			break;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
