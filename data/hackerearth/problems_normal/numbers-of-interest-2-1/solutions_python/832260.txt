arr = [0]*100005
f = [0]*100005
ans = [0]*100005
def mark(a):
	i = 2
	while i*a<=100002:
		num = i*a
		if arr[num]==0:
			arr[num] += a
		i += 1
def sieve():
	for i in xrange(2,100002):
		if arr[i]==0:
			arr[i] = arr[i-1]+i
			mark(i)
		else:
			arr[i] += arr[i-1]

sieve()
for i in xrange(1,100002//2+1):
	for j in xrange(i+i,100002,i):
		f[j] += i
for i in xrange(2,100002):
	ans[i] = ans[i-1]+f[i]

t = input()
for i in xrange(t):
	n = input()
	print (ans[n]+arr[n])%n
