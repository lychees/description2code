int main()
{

    {
        int n;

        FOR(i,n)

        }

        S.insert(0);

        {
            int x = ar[i];
            vector<int> V;
            for(set<int>::iterator it = S.begin(); it!=S.end();it++)

                int y = *it + x;
                V.pb(y);
            }

            }
        }

            printf("NO\n");
