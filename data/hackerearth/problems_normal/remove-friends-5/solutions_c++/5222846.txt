#include <iostream>
#include <stack>
#include <queue>
using namespace std;
 
int main()
{
    int testcases;
    cin >> testcases;
    for(int t=0; t<testcases; t++){
    	int n, k;
    	cin >> n >> k;
    	stack<int> s;
    	stack<int> reverse;
    	
    	int popularity;
    	cin >> popularity;
    	s.push(popularity);
    	for(int i=1; i<n; i++){
    		int popularity;
    		cin >> popularity;
    		while(!s.empty() && (k > 0) && (s.top() < popularity)){
    			s.pop();
    			k--;
    		}
    		s.push(popularity);
    	}
    	
    	while(!s.empty()){
    		reverse.push(s.top());
    		s.pop();
    	}
    	
    	while(!reverse.empty()){
    		cout << reverse.top() << " ";
    		reverse.pop();
    	}
    	cout << endl;
    }
    return 0;
}
