#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t ;
	while(t--)
	{
	long long n,k,m,x, p[1005] , mn = 1e18 ;
	cin >> n >> k >> m ;
	
	for(int i=0 ; i<n ; i++)
	{
		cin >> p[i] ;
	}
	
	int ans = 0 ;
	
	for(int i=0 ; i<n ; i++)
	{
		long long sum = p[i]*k ;
		sum += (p[i]-1)*m ;
		
		for(int j=0 ; j<p[i] ; j++)
		{
			cin >> x ;
			sum += x;
		}
		
		if( sum < mn ) mn = sum , ans = i+1 ;
	}
	
	cout << ans << " " << mn << endl;
}
}
