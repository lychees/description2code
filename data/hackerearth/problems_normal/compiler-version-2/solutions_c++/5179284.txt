#include <bits/stdc++.h>
using namespace std;

int main()
{
   	string str;
   	while(getline(cin,str))
   	{
   		if(str.length()==0) break;
   		int i=0;
   		while(str[i]!='\0')
   		{
   			if(str[i]=='/' and str[i+1]=='/')
   			{
   				//i-=1;
   				break;
   			}
   			if(str[i]=='-' and str[i+1]=='>')
   			{
   				cout<<".";
   				i+=1;
   			}
   			else cout<<str[i];
   			i+=1;
   		}
   		while(str[i]!='\0')
   		{
   			cout<<str[i];
   			i++;
   		}
   		
   		cout<<endl;
   	}
    return 0;
}
