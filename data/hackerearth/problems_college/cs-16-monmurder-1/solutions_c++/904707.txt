#include<bits/stdc++.h>
using namespace std;
int pn[1000000],arr[1000000];
int main()
{
	int i,j,x,y,cnt,n,p,t,ans;
	memset(pn,0,sizeof(pn));
	pn[0]=pn[1]=1;
	for(i=2;i<sqrt(1000000);++i)
	{
		cnt=2;
		for(j=i*cnt;j<1000000;j=i*cnt)
		{
			pn[j]=1;
			++cnt;
		}
	}
	cin>>t;
	while(t--)
	{
	cin>>n;
	p=0;
	for(i=2;i<=n;++i)
	{
		cnt=0;
		if(pn[i]==0)
		{
			while(n%i==0)
			{
				++cnt;
				n=n/i;
			}
			if(cnt)
			{
				arr[p]=cnt;
				++p;
			}
		}
	}
	if(p==1)
	ans=1;
	else if(p>1){ans=(arr[0]^arr[1]);
	 for(j=2;j<p;++j)ans=(ans^arr[j]);}
	 if(ans==0) cout<<"Pappu Bhujia\n";else cout<<"Printf{}\n";
	}
	return 0;  
}
