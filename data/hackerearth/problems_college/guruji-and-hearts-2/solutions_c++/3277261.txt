#include <iostream>
#include <string>
#include <string.h>
#include<stdio.h>

using namespace std;

int main()
{
    char str[100];
    int i=0, cnt=0;
    cin>>str;

    for(;i<strlen(str)-1; i++)
        {
            if(str[i]=='<' && str[i+1]=='3')
                cnt++;
        }

    cout<<cnt;

    return 0;
}
