#include <iostream>
using namespace std;

int main()
{   int t,n;
    cin>>t;
    while(t--)
    {   
    	cin>>n;
    	
    	int a[n],b[n];
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	    b[i]=0;
    	}
    	
    	int j=0;
    	while(j!=-1)
    	{
    		j=a[j];
    		if(b[j]==-1){
    			cout<<"Lose"<<endl;
    			break;
    		}
    		else
    		{
    		b[j]=-1;
    		
    		}
    		
    	}
    	
    	if(j==-1)
    	cout<<"Win"<<endl;
    	
    	
    	
    }
    return 0;
}
