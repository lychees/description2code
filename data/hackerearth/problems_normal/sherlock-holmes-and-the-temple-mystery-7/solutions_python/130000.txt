testcase = input()
for i in range(testcase):
    itemno = input()
    costs = raw_input().split()
    for i in range(itemno):
        costs[i] = int(costs[i])
    s = sum(costs)
    if s%2 == 0:
        print "YES"
    else:
        print "NO"
