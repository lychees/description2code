int gcd(int a,int b)
{ 
 if(a==0)
  return 0;
 if(b==0)
  return a;
 return gcd(b,a%b);
}
int max(int a,int b){return (a>b)?a:b;}
int min(int a,int b){return (a<b)?a:b;}
int main()
{
 int x,y;
 int t,i,n;
 cin>>t;
 while(t--)
 {
  cin>>n;
  cin>>x;
  for(i=1;i<n;i++)
   cin>>y,x=gcd(max(x,y),min(x,y));
  if(x==1)
   cout<<"YES"<<endl;
  else
   cout<<"NO"<<endl;
 }
 return 0;
}
