#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	cin >> T;

	int dirChangeTime = 10;
	int reachTime = 5;
	int dropOrPickupTime = 10;

	while(T--)
	{
		int n, J, L;
		char c;

		cin >> n >> J >> L >> c;

		int rearchByStairs = J * 10;
		int reachByLift = 0;
		int goToJthFloorTime = 0;

		if(c == 'U')
		{
			goToJthFloorTime = (n - L) * reachTime + dirChangeTime + (n - J)*reachTime;
			reachByLift = goToJthFloorTime + J * reachTime + dropOrPickupTime;
			
			if(n != J)
				reachByLift += dropOrPickupTime;
		}

		else if(c == 'D')
		{
			if(L < J)
				goToJthFloorTime = L * reachTime + dirChangeTime + J * reachTime;
				
			if(L >= J)
				goToJthFloorTime = (L - J) * reachTime;
				
			reachByLift = goToJthFloorTime + dropOrPickupTime + J * reachTime + dropOrPickupTime;
		}

		if(rearchByStairs < reachByLift)
			cout << "Stairs" << " " << rearchByStairs << endl;
		else
			cout << "Lift" << " " << reachByLift << endl;
	}

	return 0;
}
