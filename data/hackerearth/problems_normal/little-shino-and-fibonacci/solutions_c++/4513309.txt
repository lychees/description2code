#include <bits/stdc++.h>
#define ll long long
#define mod 10000
#define Mod 1000000007
using namespace std;
ll fib[100001];
ll solve(ll n){
 if(!n) return 0LL;
 ll total = n/15000LL;
 total = ((total%Mod * fib[15000])%Mod + fib[n % 15000])%Mod;
 return total;
}
int main() {
    cin.sync_with_stdio(false);
     fib[2] = 1;
     for(int i = 3 ; i <= 15000; ++i )  fib[i] = ( fib[i - 1] + fib[i - 2])%mod;
     for(int i = 3 ; i <= 15000 ; ++i )  fib[i] = (fib[i] + fib[i - 1]);
    int T;
    cin >> T;
    while( T-- ){
      ll l , r;
      cin >> l >> r;
      ll right = solve(r);
      ll left  = solve(l - 1);
       right -= left;
       right = (right + Mod)%Mod;
       cout << right << endl;
    }
    return 0;
}
