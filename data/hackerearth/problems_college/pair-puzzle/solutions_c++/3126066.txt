#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n,k,i,j;
    	cin>>n>>k;
    	long long int a[k];
    	a[0]=n/k;
    	for(i=1;i<k;i++)
    	{
    		a[i]=(int)(n-i)/k+1;
    	}
    	if(k%2==0)
    	{
    	long long int sum=0;
    	sum=sum+(a[0]*(a[0]-1))/2;
    	for(i=1;i<(float)k/2;i++)
    	{
    		sum=sum+a[i]*a[k-i];
    	}
    	sum=sum+(a[k/2]*(a[k/2]-1))/2;
    	cout<<sum<<endl;
    	}
    	else
    	{
    		long long int sum=0;
    		sum=sum+(a[0]*(a[0]-1))/2;
    		for(i=1;i<(float)k/2;i++)
    		{
    			sum=sum+a[i]*a[k-i];
    		}
    		cout<<sum<<endl;
    	}
    	
    }
    return 0;
}
