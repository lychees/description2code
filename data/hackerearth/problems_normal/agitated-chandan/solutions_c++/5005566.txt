#include<bits/stdc++.h>
#define repi(a,b,c) for(int i=a;i<b;i++)
using namespace std;

int t,n,dis[100005];
bool visited[100005];
vector< vector< pair<int,int> > > v;


void bfs(int a)
{
    queue<int>  q;
    q.push(a);
    dis[a]=0;
    visited[a]=true;
    int top;
    while(!q.empty())
    {
        top=q.front();
        q.pop();

        for(vector< pair<int,int> >::iterator it=v[top].begin();it!=v[top].end();it++)
        {
            if(visited[it->first]==false)
            {
                dis[it->first]= dis[top] + it->second;
                q.push(it->first);visited[it->first]=true;
            }
        }
    }

}

{

            v[b].push_back(make_pair(a,c));
