#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
	unsigned long long n,a[100000],i,cnt[100000]={},j,max=0;
	scanf("%llu",&n);
	for(i=0;i<n;i++)
	{
		scanf("%llu",&a[i]);
	
	}
	j=0;
	for(i=1;i<n;i++)
	{
		if((a[i]-a[i-1])==1||a[i]-a[i-1]==-1)
		cnt[j]++;
		else
		j++;
	}
	for(i=0;i<j+1;i++)
	if(cnt[i]>max)
	max=cnt[i];
    printf("%llu\n",max+1);
    return 0;
}
