int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	unsigned long long n;
    	cin>>n;
    	unsigned long long sum=0;
    	bool f=0;
    	for(int i=1;i<=63;++i){
    		unsigned long long ans=0;
    		ans=((unsigned long long )1<<i);
    		//cout<<ans<<endl;
    		for(int j=0;j<i;++j){
    			unsigned long long temp=ans|((unsigned long long )1<<j);
    			//cout<<temp<<endl;
    			if(temp<=n){
    				sum=(sum%MOD+temp%MOD)%MOD;
    			}
    			else{
    				f=1;
    				break;
    			}
    		}
    		if(f){
    			break;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
