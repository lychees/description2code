noOfTestCases = int(raw_input())


def totalScore(string):
    prev = 0
    score = 0
    for s in string:
        if (s == 'H'):
            score += prev + 1
            prev += 1
        else:
            prev = 0
    return score


for caseNo in range(1, noOfTestCases + 1):
    print totalScore(raw_input())
