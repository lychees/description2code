        #include<bits/stdc++.h>
        using namespace std;
         
        #define lld long long int
        #define gx getchar
        #define pb push_back
        #define in insert
        #define mp make_pair
        #define inf 2000000000
        #define mod 1000000007
        #define m 10000002

         
        void scanint(lld &x)
        {
            register int c = gx();
            x = 0;
            for(;(c<48 || c>57);c = gx());
            for(;c>47 && c<58;c = gx())
            {
                x = (x<<1) + (x<<3) + c - 48;
            }
        }
        
        lld a[10000002];
        lld power[61];
        void solve()
        {
        	lld i,j;
        	for(i=1;i<m;i++) a[i]=0;
        	for(i=2;i<m;i++)
        	{
        		if(a[i]==0) 
        		{
        			a[i]++;
        			for(j=2*i;j<m;j=j+i) a[j]++;
        		}
        	}
        	power[0]=1;
        	for(i=1;i<61;i++) power[i]=power[i-1]<<1;
        }
         
        int main()
        {
        	ios_base::sync_with_stdio(0);
        	lld n,sum,x,i;
        	sum=0;
        	solve();
        	scanf("%lld",&n);
        	for(i=1;i<=n;i++)
        	{
        		scanf("%lld",&x);
        		sum+=power[a[x]];
        	}
        	printf("%lld\n",sum);
    	return 0;
        }
