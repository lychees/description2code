#include <bits/stdc++.h>
using namespace std;
int main() {
	ios :: sync_with_stdio(0);
    int t;cin>>t;map <string, int> H;map <string,int>::reverse_iterator it;
    string str;    
    
    while(t--){  
    	cin>>str;
        if(H.find(str)==H.end())
                H[str] = 1;  
        else             
                H[str]++;
    } 
    for(it = H.rbegin();it!=H.rend();it++)
    {
        cout<<it->first<<" "<<it->second<<endl;
    }
}
