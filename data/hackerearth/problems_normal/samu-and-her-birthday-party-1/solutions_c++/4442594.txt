using namespace std;

string likes[501];

int main()
{
	int T;
	cin >> T;
	while(T--) {
		int N, K;
		cin >> N >> K;
		for(int i = 0; i < N; i++)
			cin >> likes[i];
		
		int output = INT_MAX;
		for(int i = 0; i < 1<<K; i++)
		{
			bool ans[501] = {0};
			bool totAns = true;
			int cnt = 0;
			for(int j = 0; j < K; j++)
			{
				if (i & (1<<j))
				{
					for(int k = 0; k < N; k++)
						if (likes[k][j] == '1')
							ans[k] = true;
					cnt++;
				}
			}
			for(int j = 0; j < N; j++)
				if (!ans[j])
					totAns = false;
			if (totAns)
				output = min(output, cnt);
		}
		cout << output << endl;
	}
	return 0;
}
