using namespace std;

#define ll long long
#define vt vector

bool mysort(pair<ll,ll> a,pair<ll,ll> b)
{
	if(a.first!=b.first)
		return a.first>b.first;
	else
		return a.second<b.second;
}

int main()
{
	ll t,n,k;
	cin>>t;
	while(t--)
	{
		scanf("%lld %lld",&n,&k);
		vector<pair<ll,ll> > p(n);
		vector<ll> hurd(k);
		for(ll i=0;i<n;i++)
		{
			scanf("%lld",&p[i].first);
			p[i].second=i;
		}
		for(ll i=0;i<k;i++)
		{
			scanf("%lld",&hurd[i]);
		}
		sort(p.begin(),p.end(),mysort);
		ll j=0,mx=INT_MIN,i=0,pid=n+1;
		while(j<k&&p[0].first>=hurd[j])
		{
				mx = max(hurd[j],mx);
				j++;
				pid = min(pid,p[0].second);
		}
		if(mx==INT_MIN)
		{
			cout << 0 << endl;
		}
		else
		{
			for(ll i=1;i<n;i++)
			{
				if(p[i].first>=mx)
				{
					pid = min(pid,p[i].second);
				}
				else
					break;
			}
			if(pid>=n)
				pid=0;
			cout << pid << endl;
		}
	}
	return 0;
}
