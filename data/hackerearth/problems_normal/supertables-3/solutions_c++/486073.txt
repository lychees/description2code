using namespace std; using ll = long long;
ll a, b, k,l;
ll gcd(ll a, ll b){
	while (b > 0){
		a = a%b; b = a^b; a ^= b; b ^= a;
	}
	return a;
}
ll lcm(ll a, ll b){
	return (a*b) / gcd(a, b);
}
ll f(ll x){
	return (x / a) + (x / b) - (x / l);
}
int main(){
	//freopen("in.txt", "r", stdin);
	std::ios::sync_with_stdio(false); int t; si(t); 
	while (t--){
		sll(a); sll(b); sll(k);
		l = lcm(a, b);
		ll low = 1, high = max(a, b)*k; ll ans;
		while (1){
			ll mid = (low + high) >> 1;
			ll fmid = f(mid);
			if (fmid < k){
				low = mid + 1;
			}
			else if (fmid>k){
				high = mid - 1;
			}
			else {
				if (mid%a == 0 || mid%b == 0){
					ans = mid; break;
				}
				else{
					high = mid - 1;
				}
			}
		}
		pll(ans);
	}
	return 0;
}
