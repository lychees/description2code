#include<iostream>
#include<cstring>

using namespace std;

int main()
{
    int t,i,m,sum,j;
    string s;
    char c;

    cin>>t;
    cin.ignore();

    while(t--)
    {
        getline(cin,s);

        m=s.size();
        sum=0;

        for(i=0;i<m;i++)
            sum+=(int)s[i];

        sum=(sum/m);
        if((sum%2)!=0)
        {
            i=0;
            j=m-1;

            while(i<j)
            {
                c=s[i];
                s[i]=s[j];
                s[j]=c;
                i++;
                j--;
            }
        }
        cout<<s<<"\n";
    }

    return 0;
}
