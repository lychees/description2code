#include<bits/stdc++.h>
using namespace std;

int cnt(unsigned long long int n)
{
    int cnt=0;
    while(n)
    {
        n=n&(n-1);
        cnt++;
    }
    return cnt;
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        unsigned long long int m,p;
        cin >> m >> p;
        cout << cnt(m^p) << endl;
    }
    return 0;
}
