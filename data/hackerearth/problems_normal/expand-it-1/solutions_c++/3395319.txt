#include<iostream>
#include<string>
//#include<>

using namespace std;


 int main(){
	long int arr[26];
	
	for(long int i=0;i<26;i++){
		arr[i]=0;
	}
	
	string s;
	cin>>s;
	long int q;
	cin>>q;
	
	for(long int i=0;i<s.length();){
		char currChar = s[i];
		i++;
		long int inc = 0;
		while(s[i]>=48 && s[i]<=57){
			inc = inc*10 + s[i]-48;
			i++;
		}
		arr[currChar-97] += inc;
	}
	
	/*
	for(long int i=0;i<26;i++){
		cout<<(char)(i+97)<<" "<<arr[i]<<endl;
	}*/
	
	
	
	
	//return 0;
	
	while(q--){
		long int k;
		cin>>k;
			
		for(long int i=0;i<26;i++){
			k-=arr[i];
			if(k<=0){
				cout<<(char)(i+97)<<"\n";
				break;
			}
			if(i==25){
				cout<<"-1\n";
			}
		}
	}
	return 0;
}
