#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
char a[100005],b[100005];
int main()
{



    long long t,i,k,j;
    scanf("%lld",&t);
    while(t--)
    {k=0;
        scanf("%s%s",a,b);
         for(i=0;i<strlen(a);++i)
         {
             for(j=0;j<strlen(b);++j)
             {
                 if(a[i]==b[j])
                 {
                     k=1;
                     break;
                 }
             }
             if(k==1)
             {
                 break;
             }
         }
         if(k==1)
         {
             printf("YES\n");
         }
         else
         {
             printf("NO\n");
         }
    }
    return 0;
}
