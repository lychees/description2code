#include <iostream>
using namespace std;

void FindAndPrintSeq(long g, int n) {
	
	for( int i = 2 ; i <= n ; i++) {
		cout<<g*2<<" ";
	}
	cout<<g*(3);
	cout<<endl;
}

int main()
{
    int T;
    cin>>T;
    for (int i = 0 ; i < T ; i++) {
    	int n;
    	long g;
    	cin>>g>>n;
    	FindAndPrintSeq(g,n);
    }
    return 0;
}
