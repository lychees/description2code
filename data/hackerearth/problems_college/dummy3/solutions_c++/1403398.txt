#include <cstdio>
#include <iostream>
using namespace std;

#define MAX 1000

int arr[MAX][MAX];
int ans[MAX];


int main(){
	int i,j,n,m;

	scanf("%d",&n);

	for(i=0;i<n;++i)
		for(j=0;j<n;++j)
			scanf("%d",&arr[i][j]);

	for(i=0;i<n;++i)
		ans[i] = arr[i][n-1];

	for(i=n-2;i>=0;--i){
 	   	ans[0] += arr[0][i];
    	for(j=1;j<n;j++)
        	ans[j] = min(ans[j - 1]+arr[j][i],ans[j]+arr[j][i]);


        for(j=n-2;j>= 0;--j)
            ans[j] = min(ans[j], ans[j+1] + arr[j][i]);
	}

    m = ans[0];
    for(i=1;i<n;++i)
        if(m>ans[i])
            m = ans[i];
    printf("%d\n",m);
	return 0;
}
