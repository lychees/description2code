
t= int(raw_input())
temp = 40000000000000000
arr=[]
curr =1
prev=1
while curr<=temp:
	prev=curr+prev
	if (prev %2==0):
		arr.append(prev)
		curr, prev=prev,curr
	else:
		curr,prev=prev,curr
		
		
	
	
for _ in xrange(t):
	n= int(raw_input())
	i=0
	fans=0
	while True:
		if(arr[i]<=n):
			fans=fans+arr[i]
			i=i+1
		else:
			break
	print (fans)
