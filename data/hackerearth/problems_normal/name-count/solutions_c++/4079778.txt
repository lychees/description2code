#define MOD 1000000007
#include <iostream>
using namespace std;
long long int modu(long long int x,long long int y){
	if(y==0)
	return 1;
	if(y%2==1)
	return (x*modu((x*x)%MOD,y/2))%MOD;
	else
	return modu((x*x)%MOD,y/2);
}
long long int fact(long long int x){
	if(x==0)
	return 1;
	else
	return (x*fact(x-1))%MOD;
}
