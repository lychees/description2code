#include <iostream>
#include <stdio.h>
#include <map>

using namespace std;
const long MOD = 1000000007;

map<long, long>ans;

long calculate(long n)
{
	if(ans.count(n))
		return ans[n];
	long a = n/2;
	if(n % 2 == 0)
		return ans[n] = (calculate(a)*calculate(a) + calculate(a-1)*calculate(a-1)) % MOD;
	else
		return ans[n] = (calculate(a)*calculate(a+1) + calculate(a-1)*calculate(a)) % MOD;
}

int main()
{
	ans[0] = 1;
	ans[1] = 1;
	
	int t;
	long n;
	
 	scanf("%d", &t);
	
 	while(t--)
	{
		scanf("%ld", &n);
		printf("%ld\n",calculate(n-1));
	}
 
    return 0;
}
