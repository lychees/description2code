#include<iostream>
using namespace std;
int main()
{
    int tc,m,p,i,n = 0;
    int q[100],w[100];
    cin>>tc;
    m = tc;
    tc = 1;
    do
    {
    	char web[200];
    	cin>>web;
    	i=4;
    	n=0;
    	while(web[i] != '.')
    	{
    		if(web[i] == 'a' || web[i] == 'e' || web[i] =='i' || web[i] =='o' || web[i] =='u') 
    		{
    			n = n;
    		}
    		else n=n+1;
    		i++;
    	}
    	i = i+4;
    	n = n+4;
    	q[tc] = n;
    	w[tc] = i;
    	tc++;
    }while(tc <= m);
    for(p=1;p<=m;p++)
    {
    	cout<<q[p]<<"/"<<w[p]<<endl;
    }
    return 0;
}
