T = int(raw_input())
M = 201
A = [[[0 for _ in xrange(M)] for _ in xrange(M)] for _ in xrange(M)]

for N in xrange(M):
    for K in xrange(M):
        A[N][K][1] = 1 if N >= K else 0
for N in xrange(1, M):
    for K in xrange(1, M):
        for S in xrange(2, N/K + 1):
            i = K
            while i*(S-1) <= N-i:
                A[N][K][S] += A[N-i][i][S-1]
                i += 1
for _ in xrange(T):
    N, K = map(int, raw_input().split())
    print sum(A[N][K])
