#include <iostream>
#include<algorithm>
using namespace std;
struct st
{
	long long s,e;
	int sco;
};
st q[55];
long long dp[55];
bool cmp(st a,st b)
{
	return a.e<b.e;
}
void solve(int n)
{
	sort(q,q+n,cmp);
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		dp[i]=q[i].sco;
		for(int j=0;j<i;j++)
		{
			if(q[j].e<=q[i].s)
				dp[i]=max(dp[i],q[i].sco+dp[j]);
		}
		ans=max(ans,dp[i]);
	}
	printf("%.2f\n",(double)ans/100);
}
int main()
{
    int t,n;
    cin>>t;
    while(t>0)
    {
    	t--;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		long long x,y;
    		int z;
    		cin>>x>>y>>z;
    		st u;
    		u.s=x;u.e=y;u.sco=z;
    		q[i]=u;
    	}
    	solve(n);
    }
    return 0;
}
