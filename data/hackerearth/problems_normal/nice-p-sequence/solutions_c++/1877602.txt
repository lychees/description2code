#include <iostream>
#define ll long long int
using namespace std;
void EE(ll a, ll b, ll& x, ll& y)
{
    if(a%b == 0)
    {
        x=0;
        y=1;
        return;
    }
    EE(b,a%b,x,y);
    int temp = x;
    x = y;
    y = temp - y*(a/b);
}
ll inverse(int a, int m)
{
    ll x,y;
    EE(a,m,x,y);
    if(x<0) x += m;
    return x;
}
ll gcd(ll x,ll y)
{
	ll r=1;
	while(r)
	{
		r = x%y;
		x = y;
		y = r;
	}
	return x;
}
int main()
{
    ll t,a,n,p,b,sum;
    cin>>t;
    while(t--)
    {
    	cin>>a>>p>>n;
    	if(gcd(a,p)!=1)
    		cout<<-1<<endl;
    	else
    	{
    		b = inverse(a,p);
    		sum = (n/2) * (a+b);
    		if(n&1)
    			sum += a;
    		cout<<sum<<endl;
    	}
    }
    return 0;
}
