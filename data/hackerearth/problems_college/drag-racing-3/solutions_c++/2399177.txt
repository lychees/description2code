#include <iostream>
#include <set>
#include <map>
#include <string>
#include <cmath>
#include <utility>
using namespace std;


map <string,bool> win;
map <string,bool> loss;


int main()
{
    int N,no_of_drivers;	
    scanf("%d",&N);
    
    no_of_drivers = pow(2,N) -1;
    
    string s1[no_of_drivers],s2[no_of_drivers];
    
    for(int i=0;i<no_of_drivers;i++)
    {
    	
    	cin>>s1[i]>>s2[i];	
      	
		if(!loss[s1[i]])
    		win[s1[i]]=true;
    	
    	
    	if(win[s2[i]])
    		win.erase(s2[i]);
		
		loss[s2[i]] = true;
		
    }
    
    
    for (map <string,bool>::iterator it=win.begin(); it!=win.end(); ++it)
    {
    	if(it->second)
    	{
    		cout <<it->first<<"\n";
    	}
    }	
    	
    return 0;
    
}
