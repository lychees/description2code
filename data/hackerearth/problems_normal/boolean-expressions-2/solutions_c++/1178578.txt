char s[505], op[505];
int n;

#define MOD 1000000009
ill dp[305][305][2];

int f(int l,int r,int what) {

	if ( l == r ) {
		if ( s[l]-'0' == what ) return 1;
		return 0;
	}

	ill &res = dp[l][r][what];
	if ( res != -1 ) return res;
	res = 0;

	F(i,l,r) {

		vector<bool> g;
		if ( op[i] == 'a' ) {
			g.pb(0);g.pb(0);g.pb(0);g.pb(1);
		}
		if ( op[i] == 'o' ) {
			g.pb(0);g.pb(1);g.pb(1);g.pb(1);	
		}
		if ( op[i] == 'x' ) {
			g.pb(0);g.pb(1);g.pb(1);g.pb(0);
		}

		F(j,0,2) {
			F(k,0,2) {
				int x = j*2 + k;
				if ( g[x] == what ) {
					res += (((ill)f(l,i,j)*(ill)f(i+1,r,k))%MOD);
					if ( res >= MOD ) res %= MOD;
				}	
			}
		}

	}

	return res;

}


void ans() {

	F(i,0,305) F(j,0,305) F(k,0,2) dp[i][j][k] = 0;

	F(len,0,n) {

		F(l,0,n) {
			int r = l+len;
			if ( r >= n ) continue;

			F(what,0,2) {

				ill &res = dp[l][r][what];
				res = 0;
				if ( l == r ) {
					if ( s[l]-'0' == what ) res = 1;
					continue;
				}

				F(i,l,r) {

					vector<bool> g;
					if ( op[i] == 'a' ) {
						g.pb(0);g.pb(0);g.pb(0);g.pb(1);
					}
					if ( op[i] == 'o' ) {
						g.pb(0);g.pb(1);g.pb(1);g.pb(1);	
					}
					if ( op[i] == 'x' ) {
						g.pb(0);g.pb(1);g.pb(1);g.pb(0);
					}

					F(j,0,2) {
						F(k,0,2) {
							int x = j*2 + k;
							if ( g[x] == what ) {
								res += dp[l][i][j]*dp[i+1][r][k];
							}	
						}
					}

					res %= MOD;

				}

			} 

		}

	}


}

    }
}
