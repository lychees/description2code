#include <iostream>
#include <vector>
#include <stdio.h>
#include <algorithm>
#include <cmath>
using namespace std;


int main() {
    
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n,i,j,sum=0,sum1=0,sum2=0,sumsum=0;
        scanf("%lld",&n);
        vector < int > a;
        for(i=0;i<n;i++) 
        {
            scanf("%lld",&j);
            a.push_back(j);
            sum=sum+abs(a[i]);  // calculating maximum value of B
        }

        sort(a.begin(),a.end());

        // for maximizing A we do the following
        if(n%2==0){   for(i=n-1;i>=n/2;i--) sum1=(sum1+a[i]); /* if n is even we need n/2 elements on even index and n/2         elements on odd index, adding last n/2 maximum elements and subtracting it with n/2 smaller elemets we get sumsum*/
                       for(i=0;i<n/2;i++) sum2=(sum2+a[i]);
                       sumsum=(sum1-sum2)%1000000007; }
        else
        {              for(i=n-1;i>n/2;i--) sum1=(sum1+a[i]); /* if n is odd  we need n/2+1 elements on odd index and n/2 elements in even index, adding last n/2 maximum elements and subtracting it with n/2 +1 smaller elemets we get sumsum*/
                       for(i=0;i<=n/2;i++) sum2=(sum2+a[i]);
                       sumsum=(sum1-sum2)%1000000007; } 
        sum=sum%1000000007; // maximum B
        sumsum=sumsum%1000000007; // maximum A
        long long int ans=(sumsum*sum)%1000000007; // maximum C
        printf("%lld\n",ans);   
    }
    return 0;
}
