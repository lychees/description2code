for _ in range(input()):
    z = int(raw_input())
    a = map(int, raw_input().split())
    a = sorted(a)
    if z % 2 == 0:
        m = (a[(z/2)] + a[(z/2)-1]) / 2
    else:
        m = a[(z/2)]
    b = sum(a) / z
    p = 0
    r = 0
    for i in a:
        p += abs(i-m)
        r += abs(i-b)
    print min(p, r)
