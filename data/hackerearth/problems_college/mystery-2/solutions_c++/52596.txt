#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    int a,b;
    int value;
    int ans=1;
    int temp;
    for(int i=0;i<T;i++){
        ans=0;
    	value=0;
    	temp=1;
    	cin>>a>>b;
    	while(a/temp!=0){
    		temp=temp*b;
    		value++;
    	}
    	temp=temp/b;
    	while(a!=0){
    		ans= ans*10+a/temp;
    		a=a%temp;
    		temp=temp/b;
    	}
    	while(temp!=0){
    		ans=ans*10;
    		temp=temp/b;
    	}
    	cout<<ans<<endl;
    }
}
