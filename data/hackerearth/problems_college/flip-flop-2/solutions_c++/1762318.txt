#include <iostream>
#include <cstring>
using namespace std;

int main()
{
  
  int n;
  char *str = new char[100000];
  cin >> n;
  
  for(int i=0; i< n; i++) {
  	cin >> str;
  	int count =0;
  	for(int j=0; j < strlen(str)-1; j++) {
  		if (str[j] == str[j+1]) {
  			count++;
  		}
  	}
  	cout << count << endl;  	
  	str[0] = '\0';
  }
  return 0;
}
