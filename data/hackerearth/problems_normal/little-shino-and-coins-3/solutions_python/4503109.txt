num=input('')
str1=raw_input('')
str1=list(str1)
i=0
str_count=0
while i<=len(str1)-num:
    c=0
    li=[]
    j=i
    while c<=num and j<len(str1):
        if str1[j] not in li:
            li.append(str1[j])
            c=c+1
        j=j+1
        if c==num:
            str_count=str_count+1
    i=i+1
print str_count
