#include<bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    int t,i,j,n,len;
    char str[55];
    cin>>t;
    while(t--)
    {
    	cin>>n>>str;
    	len=strlen(str);
    	for(i=0;i<len;i++)
    	{
    		j=str[i]-65-n;
    		if(j<0)
    			j=j+26;
    		str[i]=(char)(j+65);
    	}
    	cout<<str<<endl;
    }
    return 0;
}
