#include <iostream>
using namespace std;

int main()
{
	unsigned long long int T,X,Y,N,i,j,arr[50],temp;
	cin>>T;
	while(T>0)
	{
		cin>>X>>Y>>N;
		for(i=0;i<X;i++)
		{
			arr[i]=Y;
		}
		for(i=X;i<N;i++)
		{
			temp=0;
			for(j=0;j<X;j++)
			{
				temp=temp+arr[i-j-1];
			}
			arr[i]=temp;
		}
		cout<<arr[N-1]<<endl;
		T--;
	}
    return 0;
}
