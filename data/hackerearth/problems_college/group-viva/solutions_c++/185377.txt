#include <bits/stdc++.h>

using namespace std;
long long int gcd(long long int a, long long int b)		{
	b = b%a;
    if(b==0)	{
       return a;
    }
    else
      gcd(b,a);
}
int main() {
 
	int t;
    cin>>t;
    
	while(t--)	{
    	long long int n,r,c=0,d=0,h;
    	cin>>n>>r;
        
         for(long long int i=2; i<=(long long int)pow(r,0.5); i++ )	{
			if((r-1)%i==0)	{
               h = r/i;
               if(h==i)
                  c++;
            else	
				c+=2;
         }
      }
        c = c+2;
        c = n-c;
        d = gcd(c,n);
     c = c/d; n= n/d;
    cout<<c<<"/"<<n<<"\n";
   }
   return 0;
}
