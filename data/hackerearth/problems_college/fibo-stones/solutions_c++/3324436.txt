#include<stdio.h>
int main()
{
	long long int fb[86],sum=0;
	fb[2]=fb[1]=1;
	for(int i=3;i<85;i++)
	fb[i]=fb[i-1]+fb[i-2];
	int n,m,i;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
	sum=sum+fb[i];
	for(i=0;i<m;i++)
	{
		long long int k;
		scanf("%lld",&k);
		sum=sum-k;
	}
	printf("%lld",sum);
	return 0;
}
