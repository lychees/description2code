#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n][3];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<3;j++)
			{
				cin>>a[i][j];
			}
		}
		
		for(int i=n-2;i>=0;i--)
		{
			for(int j=0;j<=2;j++)
			{
				int p,q;
				if(j==0)
				{
					p=a[i+1][j+1];
					q=a[i+1][j+2];
				}
				if(j==1)
				{
					p=a[i+1][j-1];
					q=a[i+1][j+1];
				}
				if(j==2)
				{
					p=a[i+1][j-2];
					q=a[i+1][j-1];
				}
				a[i][j]=a[i][j]+min(p,q);
				//cout<<a[i][j]<<" ";
			}
		}
		int kp=min(a[0][0],a[0][1]);
		cout<<min(kp,a[0][2])<<"\n";
		
		
	}
}
