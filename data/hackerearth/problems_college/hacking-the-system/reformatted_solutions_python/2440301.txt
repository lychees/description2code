def main():
    s = raw_input()
    n = raw_input()
    l1 = len(s)
    l2 = len(n)
    if l1 != l2:
        print "loses"
    else:
        c = 0
        i = 0
        while i < l1:
            if s[i] != n[i]:
                c = c+1
            i = i+1
        if c <= 1:
            print "wins"
        else:
            print "loses"


if __name__ == "__main__":
    main()
