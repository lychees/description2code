#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int pos;
string s;
string a[20005];
void process(int b){
	if(s[pos]!='.'){
		if(abs(b)<=10000)
			a[b+10000]+=(s[pos]);
		pos+=2;
		process(b-1);
		process(b+1);
	}
	pos++;
}
int main()
{
    int t;cin>>t;
    while(t--){
    	int c;cin>>c;
    	cin>>s;
    	pos=0;
    	for(int i=0;i<20005;i++)	a[i].clear();
    	process(0);
    	sort(a[c+10000].begin(),a[c+10000].end());
    	if(a[c+10000].size()){
    		cout<<a[c+10000]<<endl;
    	}
    	else{
    		cout<<"Common Gandhijee!"<<endl;
    	}
    }
    return 0;
}
