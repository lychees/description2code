#include<bits/stdc++.h>
using namespace std;

int dp[510][510]={0};

void pre(){

	for(int i=0; i<=500; i++){
		dp[1][i]=i;
		dp[0][i]=0;
		dp[i][1]=1;
		dp[i][0]=0;
	}
	for(int i=2; i<=500; i++)
	for(int j=2; j<=500; j++){
		dp[i][j]=INT_MAX;	
		for(int k=1; k<=j; k++)	
			dp[i][j]=min(1+max(dp[i-1][k-1], dp[i][j-k]), dp[i][j]);
	}
}
int main(){
	pre();
	int t;cin>>t;
	while(t--){
		int n, k;	
		cin>>n>>k;
		cout<<dp[k][n]<<endl;
	}
	return 0;
}
