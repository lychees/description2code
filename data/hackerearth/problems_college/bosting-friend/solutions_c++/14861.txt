#include<stdio.h>
#include<cmath>

long long gcd(long long a,long long b)
{
   if(b==0)
   return a;
   else
   return gcd(b,a%b);
}
int main()
{
    int t,pg,pd,x;
	long long n,v1,a1;
	scanf("%d",&t);
	x=t;
	while(t--)
	{
	   scanf("%lld%d%d",&n,&pd,&pg);
       if(pd!=100 && pg==100)
	   printf("#%d:Broken\n",x-t);
	   else if(pd!=0 && pg==0)
	   printf("#%d:Broken\n",x-t);
	   else{
	       v1=gcd(pd,100);
		   a1=100/v1;
		   if(a1>n) printf("#%d:Broken\n",x-t);
		   else printf("#%d:Possible\n",x-t);
		   }
    }
	return 0;
}
