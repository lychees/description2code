#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,rev1,rev2;
    scanf("%d",&t);
    while (t --) {
    	scanf("%d",&n);
    	int p = n;
    	rev1 = 0;
    	while (p != 0) {
    		rev1 = rev1 * 10 + p % 10;
    		p = p / 10;
    	}
    	if (n == rev1) {
    		printf("No\n");
    		continue;
    	}
    	n = n * n;
    	p = rev1 * rev1;
    	int ans = rev1;
    	rev1 = 0;
    	while (p != 0) {
    		rev1 = rev1 * 10 + p % 10;
    		p = p / 10;
    	}
    	if (n == rev1) {
    		printf("%d\n",ans);
    	} else {
    		printf("No\n");
    	}
    	
    }
    
    return 0;
}
