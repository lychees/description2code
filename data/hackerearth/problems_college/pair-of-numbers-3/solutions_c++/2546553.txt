#include <iostream>
using namespace std;

int main()
{
   long long int t,cnt,N,i,j;  
      cin>>t;
      while(t--)  
      {  
           cnt=0;  
           cin>>N;
           for(i=1;i<=N;i++)  
           {  
                for(j=i+1;j<=N;j++)  
                {  
                     if((i*j)%(i+j)==0)  
                     cnt++;  
                }  
           }  
           cout<<cnt<<endl;
      }
    return 0;
}
