#include <iostream>
#include<vector>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t>0)
	{
		long long int n;
		cin>>n;
		vector<long long int> vec;
		vec.resize(n);
		long long int sum=0;
		for(int i=0;i<n;i++)
		{
			cin>>vec[i];
			sum+=vec[i];
		}
		long long int m;
		cin>>m;
		m=m%sum;
		if(m==0 && sum!=0)
		{
			for(int i=n-1;i>=0;i--)
			{
				if(vec[i]!=0)
				{
					cout<<i+1<<endl;
					break;
				}
			}
		}
		else
		{
			long long int tempsum=0;
			for(int i=0;i<n;i++)
			{
				tempsum+=vec[i];
				if(tempsum>=m)
				{
					cout<<i+1<<endl;
					break;
				}
			}
		}
		t--;
			
	}
	
}
