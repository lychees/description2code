#include <iostream>
#include<string.h>
using namespace std;
bool check(int a[],int b[]) {
	for(int i=0;i<26;i++){
		if(a[i]!=b[i]) return false;
	}
	return true;
}
int main()
{
    int t,i;
    cin>>t;
    while(t--) {
    	char patt[1001];
    	char text[100002];
    	scanf("%s%s",patt,text);
    	int a[26],b[26];
    	for(i=0;i<26;i++){
    		a[i]=0;
    		b[i]=0;
    	}
    	for(i=0;patt[i]!='\0';i++){
    		a[patt[i]-'a']++;
    		b[text[i]-'a']++;
    	}
    	int n=strlen(text);
    	int m=strlen(patt);
    	int flag=0;
    	if(check(a,b)) {
    		cout<<"YES\n";
    		continue;
    	}
    	for(i=strlen(patt);i<n;i++) {
    		b[text[i-m]-'a']--;
    		b[text[i]-'a']++;
    		if(check(a,b)) {
    		cout<<"YES\n";
    		flag=1;
    		break;
    		}
    	}
    	if(flag==0) cout<<"NO\n";
    }
}
