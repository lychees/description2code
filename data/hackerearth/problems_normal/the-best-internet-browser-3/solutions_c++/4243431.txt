#include <iostream>
#include <string>
using namespace std;
int main()
{
    //cout << "Hello World!" << endl;
    int t; cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int x=0;
    	int l=s.size();
    	for(int i=4;i<l-4;i++)
    	{
    		if(s[i]!='a'&&s[i]!='e'&&s[i]!='i'&&s[i]!='o'&&s[i]!='u')
    			x++;
    	}
    	x+=4;
    	cout<<x<<"/"<<l<<endl;
    }
    return 0;
}
