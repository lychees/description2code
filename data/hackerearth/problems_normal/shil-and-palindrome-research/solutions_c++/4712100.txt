#include <iostream>
using namespace std;
int type,l,r,idx,n,q;
char val;
string s;
int BIT[27][100009];
void update(int no,int idx,int val)
{
	while(idx<=n)
	{
		BIT[no][idx]+=val;
		idx+=(idx&(-idx));
	}
}
int get(int no,int idx)
{
	int sum=0;
	while(idx>0)
	{
		sum+=BIT[no][idx];
		idx = idx&(idx-1);
	}
	return sum;
}
int main()
{
	int i;
    cin>>n>>q;
    cin>>s;
    for(i=0;i<n;i++)
    {
    	update(s[i]-'a',i+1,1);
    }
    while(q--)
    {
    	cin>>type;
    	if(type==1)
    	{
    		cin>>idx;
    		cin>>val;
    		update(s[idx-1]-'a',idx,-1);
    		update(val-'a',idx,1);
    		s[idx-1]=val;
    		//cout << type << " " << idx << " " << val << " Done!\n";
    	}
    	else
    	{
    		cin>>l>>r;
    		int count=0;
    		for(i=0;i<26;i++)
    		{
    			int x=get(i,r);
    			int y=get(i,l-1);
    			if(l==1) y=0;
    			if(x<y) 
    			{
    				int z=1/0;
    			}
    			if((x-y)%2==1)
    			{
    				if(count>0) 
    				{
    					cout << "no\n";
    					count++;
    					break;
    				}
    				count++;
    			}
    		}
    		if(count<=1)
    		{
    			cout << "yes\n";
    		}
    	}
    }
    return 0;
}
