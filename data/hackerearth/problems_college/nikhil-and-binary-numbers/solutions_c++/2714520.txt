#include<iostream>
using namespace std;
int main()
{
	int t,n,a[100005],count,total,i;
	cin>>t;
	while(t--)
	{
		count=0;
		total=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			total=total^a[i];
		}
		for(i=0;i<n;i++)
		{
			if((total^a[i])==0)
			{
				count++;
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
