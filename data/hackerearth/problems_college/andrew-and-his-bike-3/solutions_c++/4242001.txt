#include <bits/stdc++.h>

using namespace std;

vector<int> v[100010];
bool visited[100010] = {0};
int count1[100010];

int dfs(int cur){
	visited[cur] = true;
	int len = v[cur].size();
	int ans = 0;
	for(int i = 0; i < len; i++){
		if (!visited[v[cur][i]]){
			ans += dfs(v[cur][i]);
		}
	}
	if (cur != 0)
		ans++;
	count1[cur] = ans;
	return ans;
}

int main()
{
    int n;
    cin >> n;
    
    for(int i = 0; i < n; i++){
    	int x, y;
    	scanf("%d%d", &x, &y);
    	v[x-1].push_back(y-1);
    	v[y-1].push_back(x-1);
    }
    int x = dfs(0);
    for(int i = 0; i < n; i++){
    	cout << count1[i] << " ";
    }
    
}
