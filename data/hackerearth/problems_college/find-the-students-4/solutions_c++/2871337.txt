#include <iostream>
using namespace std;

void find_count(int arr[],int len)
{
	int min=arr[0];
	
	int count=0;
	
	for(int i=1;i<len;i++)
	{	
			if(arr[i]<min){
				min = arr[i];
			}
	}
		
	while(1){
			
		int new_min=-1;
		count=0;
		
		for(int i=0;i<len;i++)
		{	
			if(arr[i]!=0){
				count++;
				arr[i]-=min;
				if(arr[i]!=0){
					if(new_min==-1){
						new_min=arr[i];
					}else if(arr[i]<new_min){
						new_min=arr[i];
					}	
				}	
			}
		}
		
		cout<<count<<endl;
				
		if(new_min == -1){
			break;
		}
		
		min = new_min;
		
	}
	
	
}


int main()
{
	
	int N;
	cin>>N;
	
	int arr[N];
	
	for(int i=0;i<N;i++){
		cin>>arr[i];
	}
	
	find_count(arr,N);
	
    return 0;
}
