#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--) {
    	string a, b;
    	cin >> a >> b;
    	int alen = a.length();
    	int blen = b.length();
    	if (alen == 0 || blen == 0) {
    		cout << "0" << endl;
    	}
    	int i = 0;
    	int count = 0;
    	while (i < alen) {
    		string line = a.substr(0, i);
    		line += b;
			line += a.substr (i, alen-i);
			int len = line.length();
			int j = 0;
			while (j < len/2) {
				if (line[j] != line[len-1-j]) {
					break;
				}
				j++;
			}
			if (j >= len/2) {
				count++;
			}
    		i++;
    	}
    	string line = a;
    	line += b;
    	int len = line.length();
		int j = 0;
		while (j < len/2) {
			if (line[j] != line[len-1-j]) {
				break;
			}
			j++;
		}
		if (j >= len/2) {
			count++;
		}
    	cout << count << endl;
	}
    return 0;
}
