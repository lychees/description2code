#include <iostream>
using namespace std;

int main()
{
    int n,i,k,ans,max;
    cin>>n>>k;
    int a;
    int c[1000001];
    for(i=0;i<n;i++)
    { 
           cin>>a;
           c[a]++;
    }
    max=0;
    for(i=k;i>=0;i--)
    {      
    	if(c[i]>=max)
    	{
    		max=c[i];
    		ans=i;
    	}
    }
    cout<<ans;
    return 0;
}
