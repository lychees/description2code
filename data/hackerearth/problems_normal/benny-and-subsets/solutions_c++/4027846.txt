#include <cstdio>
#include <algorithm>
using namespace std;

const int Maxb = 20;
const int mod = 10000007;

int t;
char S[Maxb + 1];
int n;
int mask[Maxb];
int ways;

int Solve(int x)
{
	for (int i = 0; i < Maxb; i++)
		if (x & 1 << i)
			if (mask[i] != -1) x ^= mask[i];
			else return 0;
	return ways;
}

int main()
{
	{
		int gmask;
		scanf("%d%d",&n,&gmask);
		fill(mask, mask + Maxb, -1); ways = 1;
		while (n--) {
                int cur;
			scanf("%d", &cur);

			int i;
			for (i = 0; i < Maxb; i++)
				if (cur & 1 << i)
					if (mask[i] != -1) cur ^= mask[i];
					else { mask[i] = cur; break; }
			if (i == Maxb) ways = 2 * ways % mod;
		}
		printf("%d\n", Solve(gmask));
	}
	return 0;
}
