#include <algorithm>
#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main() {
  int n, a, b;
  cin >> n >> a >> b;
  int ans[110] = {0};
  for (int i = 0; i < a; i++) {
    int x;
    cin >> x;
    ans[x] = 1;
  }
  for (int i = 1; i <= n; i++) {
    if (ans[i] == 0) {
      ans[i] = 2;
    }
    cout << ans[i] << ' ';
  }
  return 0;
}
