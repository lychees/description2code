#include <iostream>
#include <algorithm>

#define mod 1000000007
using namespace std;

int main()
{
	int n;
	int t;
	long long m[1005];
	long long just[1005];
	cin >> t;
	
	while (t--) {
		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> m[i];
		}
		
		sort(m, m+n);
		 
		long long sol = 1;
		for (int i = 0; i < n; i++) {
			sol *= (m[i] - i);
			sol %= mod;
		}
		cout << sol << endl;
	}
    return 0;
}
