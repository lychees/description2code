import sys
t=input()

while t>0:
    s=sys.stdin.readline().strip().split()
    s=map(int,s)
    i,n=s
    if i==0:
        if n%2==0:
            print 0
        else:
            print 1
    if i==1:
        if n%2==0:
            print 1
        else:
            print 0
    if i==2:
        if n==0:
            print 2
        else:
          if n%2==0:
            print 0
          else:
            print 1
    t=t-1
