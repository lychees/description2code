#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int n,i;
    cin>>n;
    long long int a[n],s;
    for(i=0;i<n;++i)
    {
    	cin>>a[i];
    }
    cin>>s;
    long long int sum=0,ctr=0;
    sort(a,a+n);
    for(i=n-1;i>=0;--i)
    {
    	sum+=a[i];
    	++ctr;
    	if(sum>=s)
    	break;
    }
    cout<<ctr<<"\n";
    return 0;
}
