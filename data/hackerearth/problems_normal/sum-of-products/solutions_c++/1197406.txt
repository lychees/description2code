#include <iostream>
#include <math.h>
using namespace std;
#define mod 1000000007

unsigned long long int sum(unsigned long long int j,unsigned long long int i)
{
	unsigned long long int jj = (j*(j+1))/2;
	unsigned long long int ii = (i*(i+1))/2;
	return (jj-ii) ;
}
int main()
{
    int T;
    unsigned long long int N;
    cin>>T;
    
    while(T--)
    {
    	unsigned long long int ans = 0;
    	cin>>N;
    	if(N<=10)
    	{
    		for(unsigned long long int i=1;i<=N;i++)
    		{
    			ans += ((N/i) * i)%mod;
    		}
    		cout<<ans<<endl;
    		continue;
    	}
    	unsigned long long int temp = (unsigned long long int)sqrt(N);
    	unsigned long long int mul=0;
    //	cout<<sum(1000000000,1000000000-2)<<endl;
    	unsigned long long int last=0;
    	for(unsigned long long int i=1;i<=temp;i++)
    	{
    		mul = (N/i);
    		mul = mul*i;
    		ans+= mul;
    		if(i==temp && last!=0)
    		{
    		//	cout<<last<<endl;
    			if(last == (i+1))
    			{
    		//		cout<<"here"<<endl;
    				
    			}
    			else
    			{
    				ans+= (((sum(N/i,N/(i+1))%mod) * i)%mod);
    				ans%=mod;
    			}
    			continue;
    		}
    		else
    		{
    			
    		mul = ((sum(N/i,N/(i+1))%mod) * i)%mod;
    		last = (N/(i+1)) + 1;
    		//cout<<last<<end
    		}
    		
    		ans+=(mul);
    		ans%=mod;
    	//	cout<<ans<<endl;
    	}
    	cout<<ans%mod<<endl;
    }
    return 0;
}
