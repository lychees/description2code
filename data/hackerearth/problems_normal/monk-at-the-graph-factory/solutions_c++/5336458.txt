#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int flag=1;
    int k;
    int sum=0;
    int z=n;
    while(z--)
    {    cin>>k;
        sum=sum+k;

    }
    if(sum!=2*(n-1))
        flag=0;
    if(flag)
        cout<<"Yes"<<endl;
    else
        cout<<"No"<<endl;





}
