#include <iostream>
using namespace std;

bool d2b(int n)
{
	int sum=0,r=0,i=0,b[16];
	while(n!=0)
	{
		r=n%2;
		b[i]=r;
		if(i>0)
		{
			if(b[i]*b[i-1]==1)
			return true;
		}
		i++;
		n/=2;
	}
	return false;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
  {
  	int n;
  	cin>>n;
  	while(d2b(n))
  	{
  		n++;
  	}
  	cout<<n<<endl;
  }
    return 0;
}
