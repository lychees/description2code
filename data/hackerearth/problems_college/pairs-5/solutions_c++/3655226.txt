#define MAX 12

vector<int> v;

bool isPossible(int l, int n, int cnt, int prev) {
	if(cnt == 2) {
		return !l;
	}

	int x = (1<<n);
	bool result = false;

	for(int i=0;i<x;i++) {
		if(i & ((x-1) ^ l)) continue;

		int value = 0;
		FOR(j, 0, n) {
			if(i & (1<<j)) {
				value += v[j];
			}
		}
		if((prev >= value) 
			&& isPossible(((x-1) ^ i)^((x-1) ^ l), n, cnt+1, value)) {
			result = true;
		}
			
	}
	return result;
}


int func(int n) {
	int result = INT_MAX;
	int x = (1<<n);

	for(int i=0;i<x;i++) {
		int value = 0;
		FOR(j, 0, n) {
			if(i & (1<<j)) {
				value += v[j];
			}
		}
		if(isPossible((x-1) ^ i, n, 0, value)) result = min(result, value);
	}
	return result;
}


int main() {

	int n, x;
	

	sc1(n);
	FOR(i, 0, n) {
		sc1(x);
		v.push_back(x);
	}

	pr1(func(n));

	return 0;
}
