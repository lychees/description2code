#include<bits/stdc++.h>
using namespace std;
int a[5000002];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        int t_odd=0,t_even=0,o_diff=0,e_diff=0,even=0,odd=0;
        for(int i=0;i<n;i++)
        {
            if(a[i]%2==0)
            {
                t_odd=odd+o_diff;
                t_even=even+e_diff+1;
                o_diff=t_odd-odd;
                e_diff=t_even-even;
                even=t_even;
                odd=t_odd;
            }
            else
            {
                t_odd=odd+e_diff+1;
                t_even=even+o_diff;
                o_diff=t_odd-odd;
                e_diff=t_even-even;
                even=t_even;
                odd=t_odd;
            }
        }
       // cout<<odd<<endl;
       // cout<<even<<endl;
        if(even>odd)
            cout<<"ANDI"<<endl;
        else
            cout<<"BOB"<<endl;
    }
    return 0;
}
