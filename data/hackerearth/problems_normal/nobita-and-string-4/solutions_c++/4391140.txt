#include<iostream>
#include<vector>
#include<string>

using namespace std;


int main()
{
	int t;
	cin >> t;

	cin.ignore();
	for (int i = 0; i < t; i++)
	{
		string input;
		getline(cin, input);
		for (int index = input.length()-1; index >= 0; index--)
		{
			int count = 0;
			while ((index >= 0) && (input[index] != ' ') )
			{
				count++;
				index--;
			}

			cout<<input.substr(index+1, count)<<" ";
		}
		cout << endl;
	}

	return 0;
}
