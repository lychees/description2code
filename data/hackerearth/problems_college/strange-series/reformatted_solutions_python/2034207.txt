def ncr(n, r):
    r = min(n, n-r)
    num = 1
    den = 1
    for i in range(r):
        num = num*n
        n -= 1
        den = den*r
        r -= 1
    return num//den


t = int(raw_input())
for _ in range(t):
    a = int(raw_input())
    if a == 1:
        print("1")
    else:

        sum = 0
        for k in range(a):
            sum = sum + ncr(a, k)
        print(sum)
