#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t,n,j,sum;
    
    scanf("%d", &t);
    
    for (int i = 0; i < t; i++) {
    	scanf("%d", &n);
    	
    	sum = 0;
    	j = 1;
    	
    	while (sum < n) {
    		sum += j;
    		j++;
    	}
    	
    	j--;
    	
    	if (j % 3 == 0) {
    		printf("SL\n");
    	} else if (j % 3 == 1) {
    		printf("LB\n");
    	} else {
    		printf("BS\n");
    	}
    	
    }
    
    return 0;
}
