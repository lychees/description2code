#include <iostream>
using namespace std;
typedef unsigned long long ll;

int main(){
   
   ll test,a,b,x,y;
   cin>>test;
   while(test--){
      
      cin>>a>>b;
      while(x!=a+1)  
       {   x=a;
           y=x++;
           a=a&y++;
       }
       cout<<a<<endl;
   }
   
    
    return 0;
}
