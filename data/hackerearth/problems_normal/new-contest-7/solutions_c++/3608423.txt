#include <bits/stdc++.h>
#define PB push_back
#define LL long long

using namespace std;
const int MAXN = 1007;
int main()
{
    double ans;
    int s,n,m,k,i,j,T;
    double dp[MAXN][MAXN];
    cin>>T;
    while(T--)
    {

        cin>>s>>n>>m>>k;
        memset(dp,0,sizeof(dp));
        dp[0][0] = 1.0;
        for (i=1;i<=n;i++)
        {
            for (j=0;j<=min(i,m);j++)
            {
                if (!j)
                {
                    dp[i][j] = (dp[i-1][j]*(s-m-(i-1-j)))/(s-i+1);
                }
                else
                dp[i][j] = (dp[i-1][j]*(s-m-(i-1-j)))/(s-i+1)+(dp[i-1][j-1]*(m-j+1))/(s-i+1);
            }
        }
        ans = 0.0;
        for (i=k;i<=m;i++) ans+=dp[n][i];
        printf("%.6f\n",ans);
    }
    return 0;
}
