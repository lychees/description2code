t = int(raw_input())
while t > 0:
    b, w = map(int, raw_input().split())
    x, y, z = map(int, raw_input().split())
    num1 = 0
    num = b*x+w*y
    if x < y:
        num1 += (b+w)*x+w*z
    elif x > y:
        num1 += (b+w)*y+b*z
    else:
        num1 += num
    if (num <= num1):
        print num
    else:
        print num1
    t -= 1
