#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
    	long long cont=0;
   long long a,b;
   cin>>a>>b;
   int cry=0;
   while(a>0&&b>0)
   {
   	if((a%10+b%10+cry)>=10)
   	{
   		cry=(a%10+b%10+cry)/10;
   		cont++;
   	}
   	else
   	{
   		cry=0;
   	}
   	a=a/10;
   	b=b/10;
   }
   if(cont==1)
   cout<<cont<<" "<<"carry operation"<<endl;
   else if(cont>1)
    cout<<cont<<" "<<"carry operations"<<endl;
   else
   cout<<"No carry operation"<<endl;
    }
    return 0;
}
