#include<bits/stdc++.h>
using namespace std;

int main()
{
    string word,t="";
    stack<string> st;
    int n;
    cin>>n;
    getchar();
    while(n--)
    {
        getline(cin,word);
        for(int i=0;word[i];i++)
        {
            if(word[i]==' ')
            {
                st.push(t);
                t="";
            }
            else
            {
                t+=word[i];
            }
        }
        st.push(t);
        t="";
        if(st.size()>=1)
        {
            cout<<st.top();
            st.pop();
        }
        while(!st.empty())
        {
            cout<<' '<<st.top();
            st.pop();
        }
        cout<<endl;
    }
    return 0;
}
