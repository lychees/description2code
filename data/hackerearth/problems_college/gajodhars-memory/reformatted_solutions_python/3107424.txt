# print 'Hello World!'
from collections import Counter
t = input()
while t > 0:
    t -= 1
    n = input()
    l = []
    while n > 0:
        n -= 1
        s = raw_input()
        l.append(s[0])
    c = Counter(l)
    k = 0
    for i in c:
        if c[i] == 1:
            k += 1
    print k
