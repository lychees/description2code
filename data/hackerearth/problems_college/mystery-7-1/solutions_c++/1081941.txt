#include <iostream>
#define c cin
#define o cout
using namespace std;
int cel(char x);
int i;
int main(){
	int t;
	c>>t;
	while(t--){
		string s;
		c>>s;
		for(i=0; i<s.length(); i++){
			o<<cel(s[i]);
		}
		o<<"\n";
	}
    return 0;
}
