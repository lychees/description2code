#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int q;
    cin>>q;
    
    while(q--)
    {
    	int ch,i,j,x,chief[10]={0};
    	
    	cin>>ch;
    	for(i=0;i<ch;i++)
    	{
    		cin>>x;
    		chief[x]=1;
    	}
    	
    	int soldier,count=0,sol;
    	
    	for(soldier=1;soldier<=n;soldier++)
    	{      sol=soldier;
    		while(sol>0)
    		{
    			if(chief[sol%10]==1)
    			   {
    			   	count++;
    			   	break;
    			   }
    			   sol/=10;
    		}
    	}	
    		cout<<count<<endl;
    		count=0;
    }
    }
