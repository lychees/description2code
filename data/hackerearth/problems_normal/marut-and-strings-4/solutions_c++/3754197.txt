#include <iostream>
using namespace std;
 
int main()
{
    int cases,l,small,capital;
    string s;
    cin>>cases;
    if(cases<1||cases>10){
		cout<<"Invalid Test\n";
		cases=0;
    }
    while(cases--){
    	small=0;
    	capital=0;
    	cin>>s;
    	l=s.size();
    	for(int i=0;i<l;i++){
    		if(s[i]>64&&s[i]<91)
    		capital++;
    		else if(s[i]>96&&s[i]<123)
    		small++;
    	}
    	if(small==0 &&capital==0||l>100)
    		cout<<"Invalid Input\n";
    	else if(small==0||capital==0)
    		cout<<0<<"\n";
    	else if(small>capital)
    		cout<<capital<<"\n";
    	else 
    		cout<<small<<"\n";
    }
    return 0;
}
