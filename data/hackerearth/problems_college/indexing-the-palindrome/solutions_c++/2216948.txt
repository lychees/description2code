int isPalin(string str,int length)
{
    int i,j;
    for(i=0,j=length-1;i<j;i++,j--)
    {
        if(str[i]!=str[j])
            return 0;
    }
    return 1;
}
int isPalinRemove(char *str,int length,int index)
{
    string sd="";
    int i;
    for(i=0;i<length;i++)
    {
        if(i!=index)
            sd+=str[i];
    }
    return isPalin(sd,length-1);
}
