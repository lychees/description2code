#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
    ll a,b;
    cin>>a>>b;
    string s;
    cin>>s;
    int k=0;
    long long ans=a;
    long long finans=0;
    long long size=s.size();
    while(k<size)
    {
    	if((s[k]-'0')%2==1)
    	{
    		finans+=ans;
    		finans%=b;
    	}
    	ans=ans*ans;
    	ans%=b;
    	k++;
    	
    }
    cout<<finans;
    return 0;
}
