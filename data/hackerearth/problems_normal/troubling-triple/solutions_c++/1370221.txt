#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
const int MAX = 2001;

struct Ele{
    int id;
    ll val;
};

Ele a[MAX];
Ele b[MAX*MAX];
int f[MAX];
int n, k, m, res;

int IncCmp(Ele a, Ele b){
    return a.val < b.val;
}

int DecCmp(Ele a, Ele b){
    return a.val > b.val;
}

void set(int id, int val){
    while (id <= n){
        f[id] += val;
        id += id & (-id);
    }
}

int get(int id){
    int ret = 0;
    while (id > 0){
        ret += f[id];
        id -= id & (-id);
    }
    return ret;
}

int main()
{
    Ele tmp;
    scanf("%d%d", &n, &k);
    for (int i = 1; i <= n; i++){
        scanf("%lld", &a[i].val);
        a[i].id = i;
        for (int j = 1; j < i; j++){
            b[m].val = 1LL*a[i].val*a[j].val;
            b[m].id = i;
            m++;
        }
    }
    sort(a + 1, a + n + 1, DecCmp);;
    sort(b, b + m, IncCmp);
    for (int j = 0, i = 1; i <= n; i++){
        while (j < m && b[j].val*a[i].val <= k){
            set(b[j].id, 1);
            j++;
        }
        res += get(a[i].id - 1);
    }
    printf("%d", res);
    return 0;
}
