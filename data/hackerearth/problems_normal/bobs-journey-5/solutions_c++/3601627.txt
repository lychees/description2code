#include<iostream>
#include<string>
#include<map>
using namespace std;
int main(){
	int t, n, count, flag=0;
	map<char, int> m;
	cin>>t;
	string place;
	while(t--){
		for(int i=0; i<26; i++)
			m[static_cast<char>(i+97)]=0;
		cin>>n;
		count=0;flag=0;
		while(n--){
			cin>>place;	
			if(m[place.at(0)]==0)
				m[place.at(0)]=1;
			else{
				flag=1;
			}
		}
		if(flag==0)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
