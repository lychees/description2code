#include <iostream>
#include <vector>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
    long long int i,j,t,min,max,diff,base;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld %lld",&min,&max);
		diff=max-min;
		vector<bool> mark(max - min + 1, true);
		for(i=2;i*i<=max;i++){
			j=i*i;
			base=min+((j-min % j)%j);
 			for (long long k = base; k <= max; k += j)
					mark[(int)(k - min)] = false;
		}
		cout << (count(mark.begin(), mark.end(), true)) << endl;
		
	}
	
}
