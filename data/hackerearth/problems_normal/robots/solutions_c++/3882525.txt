#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
long long int t,n,ans,i;
cin>>t;
while(t--){
cin>>n;
vector<long long int> arr(n);
for(i=0;i<n;i++){
	cin>>arr[i];
}
	
sort(arr.begin(),arr.end());	
ans=1;
for(i=0;i<n;i++){
ans=(ans*(arr[i]-i))%1000000007	;
	
}	
	
cout<<ans<<endl;	
}	

	
	
	
	
	
	return 0;
}
