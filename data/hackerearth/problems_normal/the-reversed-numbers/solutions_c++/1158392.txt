#include <iostream>
using namespace std;
int RevNum(long long n)
{
    long long m=0,temp=0;
    while(n)
    {
        temp = n%10;
        n /= 10;
        m = m*10 + temp;
    }
    return m;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        long long a,b;
        cin>>a>>b;
        long long c = RevNum(RevNum(a) + RevNum(b));
        cout<<c<<endl;
    }
}
