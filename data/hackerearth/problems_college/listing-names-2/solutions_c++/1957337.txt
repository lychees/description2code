#include <iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    map <string,int> dict;
    vector<string> val;
    int k=0;
    for(int i=0;i<tc;i++)
    {
        string s;
        cin>>s;
        if(!dict[s])
        {
            dict[s]=1;
            val.push_back(s);
            k++;
        }
        else
            dict[s]++;

    }
    sort(val.begin(),val.end());
    map<string,int>::iterator it;
    for(int i=0;i<k;i++)
    {
        it=dict.find(val[i]);
        cout<<val[i]<<" "<<it->second<<endl;
    }
}
