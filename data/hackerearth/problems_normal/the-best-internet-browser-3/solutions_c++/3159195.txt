#include<bits/stdc++.h>
using namespace std;
int main()
{
	string str;
	int test,i;
	cin>>test;
	for(i=0;i<test;i++)
	{
		cin>>str;

		int j=4;
		int count=0;
		while(str[j]!='.')
		{
			if(str[j]!='a' && str[j]!='e' && str[j]!='i' && str[j]!='o' && str[j]!='u')
			{
				count++;
			
			}
			j++;
		}
		count+=4;

		cout<<count<<"/"<<str.length()<<endl;
	}
	return 0;
}
