
#include<bits/stdc++.h>
using namespace std;
double dp[4000];


double  cal (double x1,double  y1,double  x2,double  y2)
{
        double  l=(x2-x1)*(x2-x1);
        double  ll=(y2-y1)*(y2-y1);

        return sqrt(l+ll);
}

int main()
{
      int n;
      cin>>n;
       double  x[n],y[n],f[n];

      for(int i=0;i<n;i++)
      {
            dp[i]=-INFINITY;
      }
     for(int i=0;i<n;i++)
     {
         cin>>x[i]>>y[i]>>f[i];
     }


    dp[0]=f[0];
     for(int i=0;i<n;i++)
     {

          for(int j=i+1;j<n;j++)
          {
                   double  d=cal(x[i],y[i],x[j],y[j]);
                   double fun=f[j]+dp[i];

                //   cout<<d<<" "<<fun<<endl;
            //       cout<<fun-d<<endl;
                   dp[j]=max(dp[j],(fun-d));


                 //  cout<<dp[j]<<endl;

          }

     }
     /*
     for(int i=0;i<n;i++)
     {
          cout<<dp[i]<<" ";
     }
     printf("\n");
     */
      cout<<fixed;
      cout.precision(6);
      cout<<dp[n-1]<<endl;

     return 0;

}
