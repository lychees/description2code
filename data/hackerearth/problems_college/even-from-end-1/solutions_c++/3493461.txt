#include <iostream>
using namespace std;

int main()
{
	 
	 int n;
	 int i;
	 int arr[100],flag=0;
    while(cin>>n){
       
       
       
       for(i=0;i<n;i++)
       {
       	
       	 cin>>arr[i];
       }
       for(i=n-1;i>=0;i--)
       {
         
            if(arr[i]%2==0 )
       	   {    
       	   	    
       	       flag=1;
       	       cout<<arr[i]<<" ";
       	   
       	  }
       }
       if(flag!=1)
       {  
       	cout<<"None to display";
       	
       }
       flag=0;
       cout<<endl;
}
     return 0;
}
