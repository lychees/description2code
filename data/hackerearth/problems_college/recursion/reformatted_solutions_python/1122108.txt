from itertools import permutations
t = int(raw_input())
while (t):
    t -= 1
    a = ""
    n = int(raw_input())
    for i in range(1, n+1):
        a += str(i)
    perms = [''.join(p) for p in permutations(a)]
    set(perms)
    perms.sort()
    a = ""
    for i in perms:
        a += i+" "
    a.strip()
    print a
