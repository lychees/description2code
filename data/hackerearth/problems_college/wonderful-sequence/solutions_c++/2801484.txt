#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   long int n,count;
   count=0;
   cin>>n;
   
   long long int arr[n];
   for(long long int i=0;i<n;i++)
   {
   	cin>>arr[i];
   }
 for(long long int p=0;p<n-1;p++)
  {
  	if(arr[p+1]<=arr[p])
  	 count++;
  
  }
  cout<<count<<"\n";
}
}
