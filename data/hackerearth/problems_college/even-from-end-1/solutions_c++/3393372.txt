#include<iostream>
using namespace std;

bool findEven(int array[], int size1){
	bool ret = false;
	for(int i = 0; i < size1; i++){
		if(array[i] %2 == 0){
			ret = true;
			break;
			
		}
		else{
			ret = false;
		}
	}
	
	return ret;
}

int main(){
	int size = 0;
	while(cin >> size){
		int arr[size];
		for(int i = 0; i < size; i++){
			cin >> arr[i];
		}
		
		if(findEven(arr, size) == true){
			for(int i = size - 1; i >= 0; i--){
				if(arr[i] % 2 == 0){
					cout << arr[i] << " ";
				}
			}
			cout << endl;
		}else{
			cout << "None to display" << endl;
			
		}
	}
}
