#include <bits/stdc++.h>
using namespace std;
#define MAX 100001
bool a[MAX];
int main()
{
    long long int i;
    for(i=0;i<MAX;i++)
    {
    	a[i]=true;
    }
    a[0]=false;
    a[1]=false;
    long long int j;
    for(i=2;i<MAX;i++)
    {
    	if(a[i])
    	{
    		for(j=i*i;j<MAX;j=j+i)
    		{
    			a[j]=false;
    		}
    	}
    }
    int t;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	if(n==3)
    	{
    		cout<<"3"<<endl;
    	}
    	else if(a[n])
    	{
    		cout<<"2"<<endl;
    	}
    	else 
    	cout<<"1"<<endl;
    }
    return 0;
}
