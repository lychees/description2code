#include <bits/stdc++.h>
using namespace std;

int n;
int cost[100005][3];

int minCost(int idx, int flag){
	if(idx>=n) return 0;
	int ans = 1e4;
	for(int i=0; i<3; i++){
		if(i!=flag){
			ans = min(ans, cost[idx][i]+minCost(idx+1, i));
		}
	}
	return ans;
}

int main(){
    int t;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(int i=0; i<n; i++){
    		for(int j=0; j<3; j++){
    			cin>>cost[i][j];
    		}
    	}
		int ans = minCost(0, -1);
    	cout<<ans<<endl;
    }
    return 0;
}
