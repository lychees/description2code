#include <iostream>
#include <cmath>
using namespace std;

bool mons[100001];
int main()
{
    int n,q,i,j;
    string line;
    cin>>n>>q;
    cin>>line;
    for(int i=0;i<n;++i)
    {
    	if(line[i]=='1') mons[i+1]=true;
    }
    while(q--)
    {
    	bool ans=false;
    	cin>>i>>j;
    	if(!mons[i] && !mons[j])
    	{
    		int level1=log(i)/log(2);
    		int level2=log(j)/log(2);
    		//cout<<"levels: "<<level1<<level2<<endl;
    		bool flag=false;
    		while(level2>level1)
    		{
    			j=j/2;
    			level2--;
    			if(mons[j])
    			{
    				flag=true;
    				break;
    			}
    		}
    		while(level1>level2)
    		{
    			i=i/2;
    			level1--;
    			if(mons[i])
    			{
    				flag=true;
    				break;
    			}
    		}
    		while(i!=j)
    		{
    			i=i/2;
    			j=j/2;
    			level1--;
    			level2--;
    			if(mons[i] || mons[j])
    			{
    				flag=true;
    				break;
    			}
    		}
    		ans=!flag;
    	}
    	if(ans) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;
    	
    }
    return 0;
}
