#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,m,i,j,max;
	scanf("%d%d",&n,&m);
	char a[n][m+1];int c[m];
	for(i=0;i<n;i++)
	{
		scanf("%s",a[i]);
	}
	for(i=0;i<m;i++)
	{max=0;
		for(j=0;j<n;j++)
		{
			if(a[j][i]>=max)
			{
				max=a[j][i];
				c[i]=j;
			}
		}
	}
	sort(c,c+m);
	int ans=1;
	for(i=0;i<m-1;i++)
	{
		if(c[i]!=c[i+1])
		ans++;
	}
	printf("%d",ans);
}
