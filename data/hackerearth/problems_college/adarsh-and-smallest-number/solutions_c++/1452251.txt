#include<iostream>
using namespace std;
#define M 500001
#define MOD 1000000007
int main()
{
    long long a[M];
    int i,j;
    for(i=0;i<M;i++)
	    	a[i] = i;

    for(i=2;i<M;i++)
		for(j=2;i*j < M-1;j++)
			a[i*j] /= a[i];
	
	for(i=2;i<M;i++)
		a[i] = (a[i]*a[i-1]%MOD)%MOD;
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<a[n]%MOD<<endl;
	}
	return 0;
}
