#include<iostream>
using namespace std;
#define size 10000000
int main()
{
	long long j=0,k=0,r1,t,r,g,b,i,red[size],blue[size],green[size],rt=0,bt=0,gt=0,yt=0,ct=0,mt=0,wt=0,blt=0;
	cin>>t>>r>>g>>b;
	while(j<t)
	{
		r1=0;
		while(r1<r)
		{
		red[j++]=k;
		r1++;
		}
		if(k==0)
		k=1;
		else if(k==1)
		k=0;
		
	}
	k=0;
	j=0;
	while(j<t)
	{
		r1=0;
		while(r1<g)
		{
		green[j++]=k;
		r1++;
		}
		if(k==0)
		k=1;
		else if(k==1)
		k=0;
		
	}
	k=0;
	j=0;
	while(j<t)
	{
		r1=0;
		while(r1<b)
		{
		blue[j++]=k;
		r1++;
		}
		if(k==0)
		k=1;
		else if(k==1)
		k=0;
		
	}
	
	for(i=0; i<t; i++)
	{
		if(red[i]==0 && blue[i]==0 && green[i]==0)
		blt++;
		else if(red[i]==1 && blue[i]==1 && green[i]==1)
		wt++;
		else if(red[i]==1 && blue[i]==0 && green[i]==0)
		rt++;
		else if(red[i]==0 && blue[i]==1 && green[i]==0)
		bt++;
		else if(red[i]==0 && blue[i]==0 && green[i]==1)
		gt++;
		else if(red[i]==1 && blue[i]==0 && green[i]==1)
		yt++;
		else if(red[i]==1 && blue[i]==1 && green[i]==0)
		mt++;
		else if(red[i]==0 && blue[i]==1 && green[i]==1)
		ct++;
	}
	
	cout<<rt<<" "<<gt<<" "<<bt<<" "<<yt<<" "<<ct<<" "<<mt<<" "<<wt<<" "<<blt<<endl;
	
	
	

}
