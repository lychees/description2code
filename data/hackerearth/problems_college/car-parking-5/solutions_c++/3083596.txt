inline ll scan(){
    char c = getchar_unlocked();
    register ll x = 0;
    bool neg=false;
    if(c=='-'){
        neg=true;
    }
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(neg==false) return x;
    else if(neg==true) return -1*x;
}

int main(){
    ll t;
    cin>>t;
    while(t--){
        string s;
        ll i,j,k,len,l,p=0,q,x,y;
        cin>>s;
        len=s.size();
        bool ans[len+10];
        reset(ans,true);
        for(i=0;i<len;i++){
            if(s[i]=='B'){
                j=i;
                while(j>=max(i-2,0)){
                    ans[j]=false;j--;
                }
            }else if(s[i]=='S'){
                ans[i]=ans[i+1]=false;
                if(i>0){ans[i-1]=false;}
            }else if(s[i]=='D'){
                ans[i]=false;
            }
        }
        for(i=0;i<len;i++){
            if(ans[i]==true){
                p++;
            }
        }
        cout<<p<<endl;
    }
    return 0;
}
