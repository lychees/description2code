using namespace std;

#define ll unsigned long long

ll N[100007];

int main()
{
	int t; cin >> t;
	while (t--) {
		ll n; cin >> n;
		for (int i = 0; i < n; ++i) {
			cin >> N[i];
		}
		long long m; cin >> m;
		ll s = accumulate(N, N + n, (ll)0);

		if (s >= m) {
			ll i = 0;
			while (m > 0) {
				m -= N[i];
				++i;
			}
			cout << i << endl;
		}
		else {
			ll years = m / s - 1;
			m -= years * s;
			ll i = 0;
			while (m > 0) {
				m -= N[i % n];
				++i;
			}
			cout << i % n << endl;
		}
	}
	return 0;
}
