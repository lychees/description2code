#include <iostream>
#include <string>
#include <algorithm>
#include <bits/stdc++.h>
typedef long long int int64;
typedef unsigned long long int uint64;
typedef unsigned long int uint32;

using namespace std;

int main()
{
    uint32 n;
    uint32 * cats;
    uint32 * food;
    uint64 sum = 0;
    cin >> n;
    cats = new uint32[n];
    food = new uint32[n];
    for(uint32 i = 0; i < n; i++) cin >> cats[i];
    for(uint32 i = 0; i < n; i++) cin >> food[i];
    sort(cats, cats + n);
    sort(food, food + n);
    for(uint32 i = 0; i < n; i++) sum += cats[i] * food[i];
    cout << sum << endl;
    delete(cats);
    delete(food);
    return 0;
}
