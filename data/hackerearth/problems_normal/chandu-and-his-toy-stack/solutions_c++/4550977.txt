#include <bits/stdc++.h>
using namespace std;

int main() {
	long long t;
	cin >> t;

	while(t--) {
	long long n, x, y;
	cin >> n >> x >> y;

	long long a[n], b[n];
	for(long long i = 0; i < n; ++i) cin >> a[i] >> b[i];

	sort(a, a + n); sort(b, b + n);
	
	long long c = 0;
	for(long long i = 0; i < n; ++i) {
		if(a[i] < b[i]) {
			c += (b[i] - a[i]) * x;
		} else if(a[i] > b[i]) {
			c += (a[i] - b[i]) * y;
		}
	}

	cout << c << endl;
	}
	return 0;
}
