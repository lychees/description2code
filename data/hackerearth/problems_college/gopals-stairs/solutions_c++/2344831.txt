#include <iostream>
using namespace std;
unsigned  arr[100005];
#define m (1000000000+7)
int no_of_ways(int n)
{
	arr[1]=1,arr[2]=2,arr[3]=4;
	if(arr[n]>0)
	return arr[n];
	for(int i=4;i<=n;i++)
	{
		arr[i]=((arr[i-1]%m+arr[i-2]%m)%m+arr[i-3]%m)%m;
	}
	return arr[n];
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int ans=no_of_ways(n);
    	cout<<ans<<endl;
    }
    return 0;
}
