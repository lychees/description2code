#include <iostream>
using namespace std;

int n, q, a[100005], b[100005];
long long preA[100005], preB[100005];

int main(int argc, const char * argv[]) {
    cin >> n >> q;
    for(int i = 0; i < n; i++)
        cin >> a[i];
    for(int i = 0; i < n; i++)
        cin >> b[i];
    
    preA[0] = 0;
    long long sum = 0;
    for(int i = 1; i <= n; i++) {
        if(i % 2)
            sum += a[i - 1];
        else
            sum += b[i - 1];
        preA[i] = sum;
    }
    
    preB[0] = 0;
    sum = 0;
    for(int i = 1; i <= n; i++) {
        if(i % 2)
            sum += b[i - 1];
        else
            sum += a[i - 1];
        preB[i] = sum;
    }
    
    while(q--) {
        int c, x, y;
        cin >> c >> x >> y;
        if(c == 1) {
            if(x % 2)
                cout << preA[y] - preA[x - 1] << endl;
            else
                cout << preB[y] - preB[x - 1] << endl;
        }
        
        else {
            if(x % 2)
                cout << preB[y] - preB[x - 1] << endl;
            else
                cout << preA[y] - preA[x - 1] << endl;
        }
    }
    return 0;
}
