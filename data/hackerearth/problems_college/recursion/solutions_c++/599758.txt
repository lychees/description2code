#include<bits/stdc++.h>
using namespace std;
void permute(string& list, int level, vector<string>& v)
{
    if(level == list.size()){
        v.push_back(list);
        return;
    }
    for(int i=level;list[i];++i){
        swap(list[level], list[i]);
        permute(list, level + 1, v);
        swap(list[level], list[i]);
    }
}

int main()
{
    string anagrama[]={"1","12","123","1234","12345","123456","1234567","12345678","123456789"};
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n;
        cin>>n;
        vector<string> v;
        permute(anagrama[n-1], 0, v);
        sort(v.begin(), v.end());
        copy(v.begin(), v.end(), ostream_iterator<string>(cout, " "));
        cout<<endl;
        v.clear();
    }
    return 0;
}
