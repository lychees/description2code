l=int(raw_input(''))
n=int(raw_input(''))
ws=[]

for i in range (0,n):
	data=raw_input('')
	data=data.split(' ')
	ws.append([int(numeric_string) for numeric_string in data])


for ls in ws:
	d1=ls[0]
	d2=ls[1]

	if d1<l or d2<l:
		print "UPLOAD ANOTHER"

	else:
	    if d1==d2:
	        print "ACCEPTED"
	    else:
	        print "CROP IT"
