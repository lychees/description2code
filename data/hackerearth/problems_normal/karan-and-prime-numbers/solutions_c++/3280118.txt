#include <iostream>
using namespace std;
int  checkprime(int N) {
    int count = 0;
    for( int i = 1;i * i <=N;++i ) {
         if( N % i == 0) {
         if( i * i == N )
                     count++;
                 else                                                     // i < sqrt(N) and (N / i) > sqrt(N)
                     count += 2;
      }
    }
    if(count == 2)
        return 1;
    else
        return 0;
}
int main()
{
    int t,a,b;
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	int sum=0;
    	for(int i=a;i<=b;i++){
    		if(checkprime(i)==1)
    		sum+=i;
    	}
    	cout<<sum<<endl;
    }
    
    return 0;
}
