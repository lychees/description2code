#include<bits/stdc++.h>
using namespace std;

bool compare(pair<long long int, long long int> p1, pair<long long int, long long int> p2)
{
    if(p1.first==p2.first)
        return p1.second<p2.second;
    return p1.first<p2.first;
}


int main()
	{
		int t;
		cin>>t;
		while(t--)
			{
				long long int N,A,B,C,p,q;
				scanf("%lld %lld %lld %lld",&N,&A,&B,&C);
				vector<pair<long long int,long long int>>v;
				for(int i=0;i<N;++i)
					{
						scanf("%lld %lld",&p,&q);
						v.push_back(make_pair(p,1));
						v.push_back(make_pair(q,2));
					}
				sort(v.begin(),v.end(),compare);
				long long int ans=A*N;
				long long int cnt=A*N;
				for(int i=0;i<2*N;++i)
					{
						if(v[i].second==1)
							cnt=cnt+B-A;
						else
							cnt=cnt+C-B;
						ans=max(ans,cnt);
					}
				cout<<ans<<endl;
			}
		return 0;
	}
