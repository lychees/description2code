#include <iostream>
using namespace std;

int main()
{
	int n;
    cin >> n;
    int arr [n];
    for (int i = 0; i < n; i++)
    	cin >> arr [i];
    int max_overall = 0, max_here = 0;
    for (int i = 0; i < n; i++)
    {
    	max_here += arr [i];
    	if (max_here > 0)
    	{
    		if (max_overall < max_here)
    			max_overall = max_here;
    	}
    	else 
    		max_here = 0;
    }
    cout << max_overall << endl;
    return 0;
}
