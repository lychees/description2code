#include<iostream>
#include<cmath>
#include<cstdio>
#include<cstring>
#define LL long long
using namespace std;
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int n;
        scanf("%d",&n);
        int xMin=100,xMax=-100,yMin=200,yMax=-200;
        int i;
        int x,y;
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&x,&y);
            if(x>xMax) xMax=x;
            if(x<xMin) xMin=x;

            if(y>yMax) yMax=y;
            if(y<yMin) yMin=y;
        }
        int ans;
        int xDiff,yDiff;
        xDiff=abs(xMax-xMin);
        yDiff=abs(yMax-yMin);
        if(xDiff==0) ans=yDiff;
        else if(yDiff==0) ans=xDiff;
        else ans=xDiff*yDiff;
        printf("%d\n",ans);
    }
    return 0;
}
