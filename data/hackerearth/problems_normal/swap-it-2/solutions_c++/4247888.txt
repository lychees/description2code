#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k,x;
        scanf("%d%d",&n,&k);
        vector<int> v;
        v.clear();
        for(int i=0;i<n;i++)
        {
            scanf("%d",&x);
            v.push_back(x);
        }
        int temp,minIn;
        for(int i=0;i<n;i++)
        {
            minIn=i;
            if(k==0)
                break;
            else{
            for(int j=i+1;j<n;j++)
            {
                if((j-i)>k)
                    break;
                if(v[j]<=v[minIn])
                    minIn=j;
            }
            temp=v[minIn];
            for(int j=minIn;j>i;j--)
            {
                v[j]=v[j-1];k--;
            }
            v[i]=temp;
            }
        }
        for(int i=0;i<n;i++)
            cout<<v[i]<<" ";
        printf("\n");
    }
    return 0;
}
