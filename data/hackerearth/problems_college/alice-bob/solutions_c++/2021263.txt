#define rep(i, a, b) for (int i = (a); i <= (b); ++i)

#define clr( x , y ) memset(x,y,sizeof(x))

typedef long long lint;
int a[110],n;
 
void init()
{
    scanf("%d",&n);
    rep(i,1,n) scanf("%d",&a[i]);
    sort(a+1,a+n+1);
}
 
int gcd(int x ,int y)
{
    if ( !y ) return x;
    return gcd( y , x % y );
}
 
void work()
{
    int ans=a[1];
    rep(i,2,n) ans = gcd( ans , a[i] );
    if ( ( a[n] / ans - n ) & 1 ) printf("Alice");else printf("Bob");
}
 
int main()
{
    init();
    work();
    return 0;
}
