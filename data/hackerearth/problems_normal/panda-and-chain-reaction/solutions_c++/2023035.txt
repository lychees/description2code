#include <bits/stdc++.h>
using namespace std;
#define mod 1000003
typedef long long ll;

ll fact[mod+1];

void func()
{
fact[0]=1;
for(int i=1;i<=mod;i++)
{
	fact[i]=(fact[i-1]*(ll)i)%mod;
}
}

int main()
{
   int  t;
   cin>>t;
   func();
   while(t--)
   {
   	ll n,x;
   	cin>>n>>x;
   	if(n>=mod)
   	cout<<0<<endl;
   	else
   	{
   		x=x%mod;
   		cout<<(x*fact[n])%mod<<endl;
   	}
   }
   return 0;
}
