//char strn[405]={'/0'};
vector<int> ans(1000001,-1);
int process()
{
	
	int	d=0;
	for(int i=0;i<1000001;++i)
	{
		
		if(ans[i]==-1)
		{
			ans[i]=i+d;
			if(i+d<1000001)
			{
				ans[i+d]=i;
			}
			else
				break;
			++d;
		}
		
		
	}
	
	return 0;
}

int main()
{
	
	int t=0;
	long long n1,n2;
	cin>>t;
	process();
	while(t--)
	{
		cin>>n1>>n2;
		if(n1==0 && n2==0)
			cout<<"Don't Play\n";
		else if(n1==n2)
			cout<<"Play"<<endl;
		else if(ans[n1]==n2)
			cout<<"Don't Play"<<endl;
		else
			cout<<"Play\n";
	}
	return 0;
}
