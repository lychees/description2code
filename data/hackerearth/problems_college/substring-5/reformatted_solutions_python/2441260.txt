def totalsubs(a):
    l = len(a)
    return ((l+1)*l)/2


text = raw_input()
subs = 0
l = len(text)
arr = []
tmp = text[0]
s = ''
for a in range(l):
    if tmp != text[a]:
        arr.append(s)
        s = ''
    tmp = text[a]
    s += tmp
if s != '':
    arr.append(s)
for a in arr:
    subs += totalsubs(a)
print subs
