#include <iostream>
# include <math.h>
using namespace std;
int gcd ( int a, int b )
{
  int c;
  while ( a != 0 ) {
     c = a; a = b%a;  b = c;
  }
  return b;
}
inline bool IsPrime( int number )
{
 if ( ( (!(number & 1)) && number != 2 ) || (number < 2) || (number % 3 == 0 && number != 3) )
  return (false);

 for( int k = 1; 36*k*k-12*k < number;++k)
  if ( (number % (6*k+1) == 0) || (number % (6*k-1) == 0) )
   return (false);
  return true;
 }

int main()
{
    int N;
    int test_case;
    cin >> test_case;
    while(test_case--)
    {
    cin>>N;
    int prime_value;
    int count_prime=1;
    for(int i=2;i<N;i++)
    {
        prime_value = gcd(i,N);
        if(prime_value==1)
            count_prime++;
    }
    if(IsPrime(count_prime))
        cout << "TRUE"<<"\n";
    else
        cout << "FALSE"<<"\n";
    }
    return 0;
}
