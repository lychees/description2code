import math


def is_prime(n):
    if (n % 2 == 0 and n > 2) or n < 2:
        return False
    return all(n % i for i in range(3, int(math.sqrt(n)) + 1, 2))


tc = int(raw_input())
for i in range(0, tc):
    count = 0
    s = raw_input()
    n = map(int, s.split())
    for j in range(n[0], n[1] + 1):
        y = is_prime(j)
        if y:
            count = count + j
    print count
