#include <iostream>
using namespace std;

int main()
{
    long long n,q,x,i,j=1,k;
    cin>>n;
    long long a[1000005];
    for(i=1;i<=n;i++)
    {
    	cin>>x;
    	k=j-1;
    	while(j<=(x+k))
    	{
    		a[j]=i;
    		j++;
    	}
    }
    cin>>q;
    while(q--)
    {
    	cin>>x;
    	cout<<a[x]<<endl;
    }
}
