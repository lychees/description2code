#include<bits/stdc++.h>
#define LL long long
#define mod 1000000007
using namespace std;


LL bmod(LL a,LL b)
{
      if(b==0)
      {
            return 1;
      }
      else if(b%2==0){
          LL k=bmod(a,b/2)%mod;
          return(k*k)%mod;
      }
      else if(b%2!=0)
      {
           LL k=a%mod;
           LL kk=bmod(a,b-1)%mod;
           return(k%mod*kk%mod)%mod;
      }
}

int  main()
{
     int n;
     cin>>n;
     LL a[n];
     LL k=-1;
      for(int i=0;i<n;i++)
      {
            cin>>a[i];
            if(k==-1)
            {
                 k=a[i];
            }
            else
            {
                 k^=a[i];
            }
      }
      if(k==0)
      {
           cout<<(bmod(2,n-1)%mod-1)%mod<<endl;
      }
      else
      {
            cout<<0<<endl;
      }
      return 0;

}
