#include <iostream>
#include <cstdio>
using namespace std;

long long int a[1000001] = {0};

int main()
{
        for (int i = 1; i <= 1000000; i++) {
                for (int j = 5; j <= i; j *= 5) {
                        a[i] += i/j;
                }
                a[i] += a[i-1];
        }

    int t,a2,b;
    long long int count;

    scanf("%d", &t);

    for (int i = 0; i < t; i++) {
        scanf("%d%d", &a2, &b);

        count = a[b] - a[a2-1];

        printf("%lld\n", count);
    }


    return 0;
}
