#include <iostream>
using namespace std;

int main()
{
    int M,A[100][100],dig1=0,dig2=0,i,j;
    cin>>M;
    for(i=0;i<M;++i)
    {
    	for(j=0;j<M;++j)
    	{
    		cin>>A[i][j];
    	}
    }
    for(i=0;i<M;++i)
    for(j=0;j<M;++j)
    {
    	if(i==j)
    	dig1+=A[i][j];
    	if(j==(M-i-1))
    	dig2+=A[i][j];
    }
    if((dig1-dig2)>0)
    cout<<dig1-dig2;
    else
    cout<<dig2-dig1;
    return 0;
}
