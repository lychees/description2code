#include <iostream>
#include <cstdio>
using namespace std;
#define N 5000000
int a[N];
long long o[N], e[N];
int main()
{
   int T, n;   
   cin>>T;
	for(int k=0; k<T;k++)
	{
	    cin>>n;
	    long long sumO=0LL, sumE=0LL;
	    for(int j=0;j<n;j++){
	    	scanf("%d", &a[j]);
	    	
	    }
	    if(a[n-1]%2LL == 0LL){
	    	e[n-1] = 1;
	    	o[n-1] = 0;
	    }
	    else{
	    	e[n-1] = 0;
	    	o[n-1] = 1;
	    }
	    sumO = o[n-1];
	    sumE = e[n-1];
	    for(int i=n-2; i>=0;i--){
	    	if(a[i]%2LL == 0LL){
	    		o[i]= o[i+1];
	    		e[i] = e[i+1] + 1;
	    		
	    	}
	    	else{
	    		e[i]= o[i+1];
	    		o[i] = e[i+1] + 1;
	    	}
	    	sumO += o[i];
	    	sumE += e[i];
	    }
	    if(sumE > sumO){
	    	printf("ANDI\n");
	    }
	    else{
	    	printf("BOB\n");
	    }    
	    
	}
    return 0;
}
