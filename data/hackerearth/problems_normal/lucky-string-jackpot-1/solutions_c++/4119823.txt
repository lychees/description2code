#include <iostream>
using namespace std;
#define ll long long int
int funk(ll n)
{
		if(n<=3) return 1;
		ll one = 2;
		ll sum = 2;
		int cnt = 1;
		ll odd = 1;
		ll end = sum+odd;
		ll pend = 2;
		while(n > end)
		{
				one<<=1;
				cnt++;
				sum += one*cnt;
				odd+=2;
				pend = end;
				end = sum+odd;
				//cout<<"end="<<end<<" pend="<<pend<<" odd="<<odd<<endl;
		}
		if(n == pend+1) return odd-1;
		return odd;
}
int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				ll n;
				cin>>n;
				cout<<funk(n)<<endl;
		}
		return 0;
}
