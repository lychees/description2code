#include <iostream>
#include<iomanip>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    for(int i = 0; i<tc; i++)
    {
    	int N;
    	cin>>N;
    	double pi;
    	double op = 0;
    	int j;
    	for(j=0; j<N; j++)
    	{
    		std::cin>>pi;
    		op += pi;
    	}
       	std::cout<< std::fixed;
    	std::cout<< std::setprecision(8) << op/j << '\n';
    	
    }
    return 0;
}
