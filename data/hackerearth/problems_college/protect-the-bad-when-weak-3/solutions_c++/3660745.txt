#include<bits/stdc++.h>
#define pii pair<int,int>
#define fi first
#define se second
#define mp make_pair
#define vpi vector< pii >
#define pb push_back
#define ll long long int
#define mod 1000000007
#define gcd(a,b) __gcd(a,b)
#define sf(n) scanf("%d",&(n))
#define pf(n) printf("%d\n",(n))
#define setbits(n) __builtin_popcount((n))
#define pq priority_queue
using namespace std;
int dx[]={0,0,-1,1};
int dy[]={-1,1,0,0};
bool vis[105][105];
int arr[105][105];
int ans=0;
int n;
void func(int x,int y)
{
	if(x==n-1 && y==n-1)
		{
			ans++;
			return ;
		}
		vis[x][y]=true;
	for(int i=0;i<4;i++)
	{
		int k1=x+dx[i];
		int k2=y+dy[i];
		if(vis[k1][k2]==true)
			continue;
		if(k1>=n || k2>=n ||k1<0 || k2<0 || arr[k1][k2]==1)
			continue;
		func(k1,k2);
		vis[k1][k2]=false;
	}
}
int main()
{
	sf(n);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			sf(arr[i][j]);
		}
	}
	func(0,0);
	pf(ans);
}
