#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while (t--) {
    	int n, k;
    	cin >> n >> k;
    	
    	vector<int> v(n);
    	for (int i = 0; i < n; i++) {
    		int cur = 0;
    		for (int j = 0; j < k; j++) {
    			char c;
    			cin >> c;
    			cur <<= 1;
    			if (c == '1')
    				cur += 1;
    		}	
    		v[i] = cur;
    	}
    	
    	int ans = k;
    	for (int i = 1; i < (1 << k); i++) {
    		bool fl = true;
    		
    		for (auto t : v) {
    			if (!(t & i)) {
    				fl = false;
    				break;
    			}
    		}
    		
    		if (fl)
    			ans = min(ans, __builtin_popcount(i));
    	}
    	cout << ans << endl;
    }
}
