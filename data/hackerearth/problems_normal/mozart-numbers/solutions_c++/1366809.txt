#include <bits/stdc++.h>
using namespace std;

int main () {
	int tc;
	scanf("%d",&tc);
	while (tc--) {
		int a, b;
		scanf("%d%d",&a,&b);
		assert(a<=b);
		int ans = 0;
	
		if (b<7)
			ans = 0;
		if (a>7 and b<13) 
			ans = 0;
		if (a>0 and a<=7 and b<13 and b>6)
			ans = 8;
		if (a>7 and b>=13) // { (8, 13)
			ans = 144;
		if (a<=7 and b>=13)
			ans = 152;
		if (a>13) 
			ans = 0;	
		printf("%d\n",ans);
	}
	return 0;
}
