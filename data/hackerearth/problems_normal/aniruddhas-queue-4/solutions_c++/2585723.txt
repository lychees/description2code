#include<iostream>
using namespace std;
main()
{
int n; int ar[100000]; long long int m,a; int p;
int t;
cin>>t;
for(int z=0;z<t;z++)
{
cin>>n;
for(int i=0;i<n;i++)
cin>>ar[i];
cin>>m;
a=0;
for(int i=0;i<n;i++)
    a=a+ar[i];
m=m%a;
a=0;
p=0;
if(m==0)
{
 for(int i=n-1;i>=0;i--)
  if(ar[i]!=0)
    {
    p=i+1;
    break;
    }
}
else
{
for(int i=0;i<n;i++)
{
a=a+ar[i];
p=i+1;
if(a>=m)
break;
}
}
cout<<p<<endl;
}
}
