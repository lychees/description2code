#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    long long int t,b,x,y,z,w,ans;
    cin>>t;
    while(t--)  {
        cin>>b>>w;
        cin>>x>>y>>z;
        ans=min(b*x,b*(y+z))+min(w*y,w*(x+z));
        cout<<ans<<"\n";
    }
    return 0;
}
