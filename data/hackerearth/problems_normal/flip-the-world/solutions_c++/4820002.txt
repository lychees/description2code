#include <bits/stdc++.h>
using namespace std;
int arr[100][100];
void flip(int n, int m)
{
	for(int i=0;i<=n;i++)
	{
		for(int j=0;j<=m;j++)
		{
			arr[i][j]=arr[i][j]^1;
		}
	}
}


int main()
{
    int t;
   	cin>>t;
   	while(t--)
   	{
   	int count=0;
   	int n,m;
   	cin>>n>>m;
   	char c;
   	for(int i=0;i<n;i++)
   	{
   		for(int j=0;j<m;j++)
   		{
   			cin>>c;
   			arr[i][j]=c-'0';
   		}
   	}
   	for(int i=n-1;i>=0;i--)
   	{
   		for(int j=m-1;j>=0;j--)
   		{
   			if(arr[i][j]==0)
   			{
   				flip(i,j);
   			/*	for(int k=0;k<n;k++)
   				{
   					for(int l=0;l<m;l++)
   					{
   						cout<<arr[i][j]<<" ";
   					}
   					cout<<"\n";
   				}*/
   				count++;
   			}
   		}
   	}
   	cout<<count<<"\n";
   	}
    return 0;
}
