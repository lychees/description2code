#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int l = s.length();
    vector <string> subs;
    long long int ans = 0;
    for (int i = 0; i < l; i++)
        for (int j = i + 1; j <= l; j++)
            subs.push_back (s.substr (i, j - i));
    for(int i=0;i<subs.size();i++){
    	string t = subs[i];
	    reverse (t.begin (), t.end ());
	    if(t == subs[i]) ans++;
    }
    cout<<ans;
    return 0;
}
