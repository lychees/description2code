using namespace std;

LL dp[16][(1<<16)];
LL n,m;
LL ar[100];

LL func(LL ind, LL prev, LL mask) {
           if(ind == n)
           return 0;
           
           if(dp[prev][mask] != -1)
           return dp[prev][mask];
           
           LL l , i;
           
           l = 0;
           
           for(i=0;i<n;i++) {
                            if((mask&(1<<i)) == 0) {
                                             l = max(l,(ind+1)*abs(ar[prev]-ar[i]) + func(ind+1,i,(mask|(1<<i))));
                            }
           }
           
           
           return dp[prev][mask] = l;
}

int main()
{
    LL i,j,k,l,t;
    
    cin >> t;
    
    while(t--) {
               cin >> n;
               memset(dp,-1,sizeof dp);
               
               for(i=0;i<n;i++) {
                                cin >> ar[i];
               }
               
               l = 0;
               
               for(i=0;i<n;i++) {
                                l = max(l,func(1,i,(1<<i)));
               }
               
               cout << l << endl;
               
    }
    
     
    return 0;
}
