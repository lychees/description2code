    #include <iostream>
    using namespace std;
     
    int main()
    {
       int t=1,n,test;
       int a[100001];
       cin>>test;
       while(t<=test)
       {
       	
       	cin>>n;
       	for(int i=0;i<n;i++)
       	{
       	cin>>a[i];
       	}
       int dp[n+1];
       dp[0]=a[0];
       dp[1]=max(a[0],a[1]);
       for(int i=2;i<n;i++)
       {
       	dp[i]=max((dp[i-2]+a[i]),dp[i-1]);
       }
       	cout<<"Case "<<t<<": "<<dp[n-1]<<endl;
       	t++;
       }
        return 0;
    }
