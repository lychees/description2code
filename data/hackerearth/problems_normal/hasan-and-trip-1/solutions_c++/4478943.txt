double dist(double x1, double y1, double x2, double y2)
{
	return sqrt((x2 - x1)*(x2 - x1) + (y2 - y1)*(y2 - y1));
}
int main()
{
	double  X[3005], Y[3005], F[3005];
	int i,j,n;
	cin >> n;
	for (i = 0; i < n; i++)
	{
		cin >> X[i] >> Y[i] >> F[i];
	}
	double dp[3005];
	dp[0] = F[0];
	for (i = 1; i < n; i++)
	{
		dp[i] = -1e9;
		for (j = 0; j < i; j++)
		{
			dp[i] = max(dp[i],dp[j]- dist(X[i], Y[i], X[j], Y[j]));
		}
		dp[i] += F[i];
	}
