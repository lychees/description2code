#include <bits/stdc++.h>
using namespace std;

long numArray[100005];
long long cumSum[100005][10];

int main()
{
    long long numTotal,queries;
    cin>>numTotal>>queries;

    for(long i=0; i<numTotal; i++)
    {
        cin>>numArray[i];
        if(i==0)cumSum[i][0]=numArray[i];
        else
            cumSum[i][0]=cumSum[i-1][0]+numArray[i];
    }

    for(long i=0; i<numTotal; i++)
    {
        for(int p=2; p<=10; p++)
        {
            if(i-p<0)
                cumSum[i][p-1]=numArray[i];
            else
                cumSum[i][p-1]=cumSum[i-p][p-1]+numArray[i];
        }
    }

    while(queries--)
    {
        long l,r,k;
        cin>>l>>r>>k;

        if(l-k-1>=0)
            cout<<cumSum[(r-(r-l)%k)-1][k-1]-cumSum[l-k-1][k-1]<<endl;
        else
            cout<<cumSum[(r-(r-l)%k)-1][k-1]<<endl;
    }
    return 0;
}
