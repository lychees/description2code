#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll int n,tym;
ll int dp[1001][1001];
int main()
{
    int t;
    cin>>t;
    while(t--) {
    	ll int sum=0,i,j;
    	cin>>n>>tym;
    	memset(dp,0,sizeof(dp));
    	dp[1][1]=1;
    	for(i=2;i<=n;i++) {
    		for(j=1;j<=i;j++) {
    			if(i<n) {
    			if(j>=tym*(i-j)) {
    				dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%mod;
    			}
    			else
    			dp[i][j]=(dp[i-1][j-1])%mod;
    			}
    			else {
    				if(j>=tym*(i-j) && j>(i-j)) {
    				dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%mod;
    			}
    			else
    			dp[i][j]=(dp[i-1][j-1])%mod;
    			}
    		}
    	}
    	for(j=1;j<=n;j++) {
    		//cout<<dp[n][j]<<" ";
    		if(j>=tym*(n-j)) {
    			sum=(sum+dp[n][j])%mod;
    		}
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
