#include <iostream>
#define max(a,b)    (a>=b?a:b)
#define min(a,b)    (a<b?a:b)
using namespace std;

int main()
{
    int x,n,m,i,mi1,ma1,mi2,ma2,ans;
    cin>>n>>m;
    mi1=mi2=100000000;
    ma1=ma2=-1;
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	mi1=min(mi1,x);
    	ma1=max(ma1,x);
    }
    for(i=0;i<n;i++)
    {
    	cin>>x;
    	mi2=min(mi2,x);
    	ma2=max(ma2,x);
    }
    ans=max(2*mi1,ma1);
    if(ans<mi2 && 2*mi1<=ans)
    	cout<<ans;
    else cout<<"-1";
}
