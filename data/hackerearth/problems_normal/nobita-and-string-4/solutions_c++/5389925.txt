#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t,y,z,i,l;
    char a[100001];
    cin>>t;
    while(t--)
    {
    	scanf(" %[^\n]s",a);
    //	gets(a);
    //	cin>>a;
    	l=strlen(a);
    	z=l-1;
    	y=l-1;
    	while(y>=0)
    	{
    		if(a[y]==' ')
    		{
    			for(i=y+1;i<=z;i++)
    			cout<<a[i];
    			cout<<" ";
    			z=y-1;
    		}
    		y--;
    	}
    	for(i=y+1;i<=z;i++)
    			cout<<a[i];
    	cout<<endl;
    }
    return 0;
}
