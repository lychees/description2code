#include <iostream>
using namespace std;
typedef long long ll;
ll ma = 1000000007;
ll fa[300005];
ll po(ll a, ll b)
{
	if(b==0)return 1;
	ll x = po(a,b/2);
	x = (x*x)%ma;
	if(b%2==1)x = (x*a)%ma;
	return x;
}
ll ncr(ll n,ll r)
{
	if(r<0)return 0;
	if(n < r)return 0;
	if(n==r)return 1;
	if(r==0)return 1;
	ll num = fa[n];
	ll sec = (fa[r] * fa[n-r])%ma;
	sec = po(sec,ma-2);
	return (num*sec)%ma;
}
int main()
{

int n,m,c,i,t;
for(n=1;n<=300001;n++)
{
	fa[n] = (n*fa[n-1])%ma;
}

while(t--)
{
	scanf("%d%d%d",&n,&m,&c);
	int bar = m+1;
	ll way = ncr(bar,c);
//	printf("to choose %d conn from %d way are %lld\n",c,bar,way);
	int rem = n-m;
//	way = (way*fa[c])%ma;
	int bar2 = c-1;
	rem = rem - c;
	int total = rem + bar2;
//	printf("we need %d bars to convert %d balls in %d compponent\n",bar2,rem,c);
	ll way2 = ncr(total, bar2);
	if(rem==0 && c==0)way2 = 1;
//	if(rem > 0)

//	else way2 = 0;
	printf("%lld\n",(way*way2)%ma);

}
