#include<bits/stdc++.h>
using namespace std ;
 int main(){
 int t;
cin>>t;
while(t--){

    string str1 , str2;
    cin>>str1>>str2;
    int s1= str1.length();
    int s2 = str2.length();
    int i=0;
    int x;
    if(s1>s2) x=s2;
    else x=s1;
    int score=0;
    while(i<x){
        if(str1[i]==str2[i]) score++;
        i++;
    }
    cout<<score<<endl;

}

 }
