import math
noT = raw_input()
for _ in xrange(int(noT)):
    L, R = map(int, raw_input().split(" "))
    LN = math.sqrt(L)
    if int(LN) * int(LN) == L:
        LN = int(LN)
    else:
        LN = int(LN)+1

    RN = int(math.sqrt(R))
    if RN * RN > R:
        RN -= 1
    print int(math.ceil(RN+1-LN))
