#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
 
int main()
{
   int t;
	cin >>  t;
	vector<int> val(t);
	for(int i=0;i<t;i++){
		cin >> val[i];
	}
	sort(val.begin(), val.end());
	int sub = 0;
	for(int i=0;i<t;i++){
		val[i] -= sub;
		cout << val[i] << " ";
		sub++;
	}
	cout << endl;
    return 0;
}
