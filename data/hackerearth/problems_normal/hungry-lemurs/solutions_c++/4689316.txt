    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
    long int n,k,p,s,i;
    cin>>n>>k;
    p=min(n,abs(n-k));
    for(i=2;i<=n;i++)
    {
    s=n/i;
    p=min(p,abs(s-k)+abs(n-i*s));
    s++;
    p=min(p,abs(s-k)+abs(n-i*s));
    }
    cout<<p;
    return 0;
    }
