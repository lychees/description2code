#include<bits/stdc++.h>
using namespace std;
#define fl(a1,b1) for(int i=a1;i<b1;i++)
#define sc(a1) scanf("%lld",&a1)
#define pr(a1) printf("%lld",a1)
#define ll long long
#define p() printf("\n")
int main()
{
    int i,j,f,k;
    ll l,d,n,c,len,ans;
    string str[5004];
    sc(l);sc(d);sc(n);
    for(i=0;i<d;i++)
        cin>>str[i];
    while(n--)
    {
        c=0;
        string a;
        cin>>a;
        len=a.length();
        for(i=0;i<d;i++)
        {
            ans=0;
            k=0;
            for(j=0;j<len;j++)
            {
                if(a[j]=='(')
                {
                    f=0;
                    while(a[j]!=')')
                    {
                        if(a[j]==str[i][k])
                         {
                             ans++;
                             f=1;
                         }
                         j++;
                    }
                    k++;
                    if(f==0)break;
                }
                else if(a[j]==str[i][k])
                {
                    ans++;
                    k++;
                }
                else
                {
                    break;
                }
            }
            if(l==ans)
                c++;
        }
        pr(c);
        p();
    }
    return 0;
}
