#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int fib[100000];
int main()
{
    //cout << "Hello World!" << endl;
    long long int  t,n;
    cin >> t;
    fib[0]=1;
    fib[1]=1;
    fib[2]=2;
    fib[3]=4;
    for(int i=4; i <=100000; i++)
    {
    	fib[i]=((fib[i-1] % mod) + (fib[i-2] % mod) + (fib[i-3] % mod)) % mod;
    }
    while(t--)
    {
    	cin >> n;
    	cout << fib[n] <<"\n";
    }
    return 0;
}
