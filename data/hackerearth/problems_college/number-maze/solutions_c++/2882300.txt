#include <iostream>
using namespace std;

int main()
{
    long long n,m,i,f=0;
    cin>>n>>m;
    for(i=n+1;i<m;i++)
    {
    	if(i%5==0&&i%3==0)
    	{
    		if(f)
    		cout<<","<<i;
    		else
    		cout<<i;
    		f=1;
    	}
    	else
    	if(i%5!=0&&i%3!=0)
    	{
    		if(f)
    		cout<<","<<i;
    		else
    		
    	cout<<i;
    	f=1;
    	}
    	
    }
    return 0;
}
