#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int i,j,count=0,len1,len2,len;
    	char c1[25],c2[25];
    	scanf("%s%s",c1,c2);
    	len1=strlen(c1);
    	len2=strlen(c2);
    	count=abs(len1-len2);
    	if(len1>len2)
    		len=len2;
    	else
    		len=len1;
    	for(i=0;i<len;i++)
    	{
    		if(c1[i]!=c2[i])
    		{
    			count++;
    		}
    	}
    	if(count==0)
    		printf("Exact\n");
    	else
    		printf("Non Exact %d\n",count);
    }
    return 0;
}
