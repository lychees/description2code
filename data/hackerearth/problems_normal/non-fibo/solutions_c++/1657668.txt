#include <iostream>
#include <stdio.h>

using namespace std;
#define ll  long long 
ll fib[100],noof=3;
ll fun(ll n)
{
	ll i=1,j=0;
	while(1)
	{
		//cout<<fib[i]<<" ";
		j += (fib[i+1] - fib[i] - 1);
		//if(j == n)
		//	return fib[i+1]-1;
		if(j >= n)
		{
			return fib[i+1] - (j-n) - 1;
		}
		i++;
	}
}
int main()
{
    ll t,n;
    fib[1] = 1;
    fib[2] = 2;
    ll lim = 10000000000000000;
    //cout<<lim<<endl;
    while(1)
    {
    	fib[noof] = fib[noof-1] + fib[noof-2];
    	//cout<<fib[noof]<<endl;
    	if(fib[noof] > lim)
    		break;
    	noof++;
    }
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<fun(n)<<endl;
    }
    return 0;
}
