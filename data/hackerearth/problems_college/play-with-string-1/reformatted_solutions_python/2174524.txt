T = input()
for _ in range(T):
    s = raw_input()
    r = list(reversed(s))
    # print s,r
    lasts = s[0]
    lastr = r[0]
    ok = 1
    for i in range(1, len(s)):
        ns = s[i]
        nr = r[i]
        if abs(ord(nr)-ord(lastr)) != abs(ord(ns)-ord(lasts)):
            ok = 0
        lasts = ns
        lastr = nr
    if ok:
        print "Funny"
    else:
        print "Not Funny"
