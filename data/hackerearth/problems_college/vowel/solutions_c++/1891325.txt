#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s1;
    vector <char> s2;
    while(t--)
    {
    	cin>>s1;
    	int j=0;
    	for(int i=0;i<s1.length();i++)
    	{
    		if(s1[i]=='a' || s1[i]=='e' ||s1[i]=='i'||s1[i]=='o'||s1[i]=='u'||s1[i]=='A'||s1[i]=='E'||s1[i]=='I'||s1[i]=='O'||s1[i]=='U')
    		{
    			s2.push_back(s1[i]);
    			j++;
    		}
    	}
    	if(j==0)
    	{
    		cout<<"No"<<endl;
    	}
    	else
    	{
    		for(int k=0;k<j;k++)
   		 	{
    			cout<<s2[k];
    		}
    		cout<<endl;
    	}
    	s2.clear();
    }
    return 0;
}
