#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,k,l,r,count=0;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	r=n%8;
    	l=n/8;
    	if(r==6)
    	count=l+1;
    	else
    	if(l==0)
    	count=-1;
    	else
    	{
    		if(r==0)
    		count=l;
    		else
    		if(r%2!=0)
    		count=-1;
    		else
    		{
    			count=l+1;
    		}
    		
    	}
    	cout<<count<<endl;
    	count=0;
    }
    return 0;
}
