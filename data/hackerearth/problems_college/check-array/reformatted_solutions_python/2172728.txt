def foo(n, l):
    i = 0
    left = 0
    right = sum(l[1:])
    over = False
    while not over:
        if left == right:
            return "YES"

        left = left + l[i]
        i = i + 1
        if i == n - 1:
            right = 0
            over = True

        right = right - l[i]

    return "NO"


if __name__ == '__main__':
    tc = int(raw_input())
    while tc > 0:
        tc = tc - 1
        n = int(raw_input())
        l = map(int, raw_input().split())
        print foo(n, l)
