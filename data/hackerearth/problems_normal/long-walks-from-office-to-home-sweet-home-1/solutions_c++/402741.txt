#include<iostream>
#include<cmath>

#include<algorithm>

#include<stack>
#define PB push_back
#define MP make_pair
#define LL long long int
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
#define M 1000000007
using namespace std;

LL a[51][51] = {0};

vector<int > v;

int n;
void binaryconv(LL i)
{
	stack<int > s;
	while(1)
	{
		s.push(i % 2);
		i /= 2;
		if(i == 0)
		break;
	}
	v.clear();
	while(!s.empty())
	{
		v.PB(s.top());
		s.pop();
	}
	return;
}

void mult()
{
	int i,j,k;
	LL ans[51][51];
	LL s;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			s = 0;
			for(k=0;k<n;k++)
			{
				s += a[i][k] * a[k][j];
				s %= M;
			}
			ans[i][j] = s;
		}
	}
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
	a[i][j] = ans[i][j];
	return;
}





int main()
{
	int t,i,j,U,V;
	LL ans[51][51],ans1[51][51];
	LL x,y,z;
	LL s,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k>>U>>V;
		
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		ans[i][j] = 0;
		
		for(i=0;i<n;i++)
		ans[i][i] = 1;
		
		binaryconv(k);
		/*for(i=v.size() - 1;i>=0;i--)
		cout<<v[i];
		cout<<endl;
		*/
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				scl(a[i][j]);
			}
		}
		
		
	
		for(i=v.size() - 1;i>=0;i--)
		{
			if(v[i] == 1)
			{
				
				for(x=0;x<n;x++)
				{
					for(y=0;y<n;y++)
					{
						s = 0;
						for(z=0;z<n;z++)
						{
							s += ans[x][z] * a[z][y];
							s %= M;
						}
						ans1[x][y] = s;
					}
				}
				for(x=0;x<n;x++)
				for(y=0;y<n;y++)
				ans[x][y] = ans1[x][y];
				
			}
			mult();	
		}
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		if(ans[i][j] < 0)
		ans[i][j] += M;
		
		printf("%lld\n",ans[U][V]);
	}
	
	return 0;
}
