nums = [ i for i in xrange(1000001) ]
ls = [ 0 for _ in xrange(1000001) ]

def pre():
	nums[1] = 0
	for i in xrange(2, 1000001):
		if nums[i] != 1:
			for j in xrange(i, 1000001, i):
				count = 0
				while True:
					if nums[j]%i == 0:
						nums[j] /= i
						count += 1
					else:
						break
				ls[j] += count
			
if __name__ == '__main__':
	pre()
	for i in xrange(1, len(ls)):
		ls[i] = ls[i] + ls[i-1]
	for _ in xrange(input()):
		a, b = map(int, raw_input().split(" "))
		print ls[b]-ls[a-1]
