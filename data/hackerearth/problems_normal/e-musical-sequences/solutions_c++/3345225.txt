#include <iostream>
#include <cstdio>
#include <memory.h>
#include <vector>

using namespace std;


void solve() {
	int n, m;
	long long z;
	cin >> n >> m >> z;
	vector<long long> a(n + 1);
	for (int i = 0; i < n; ++i) cin >> a[i];
	for (int i = 0; i < n; ++i) {
		a[n] = a[i] - a[n];
	}
	
	a[n] = (a[n] % m + m) % m;

	if (n % 2 == 1) {
		z %= n + 1;
		cout << a[z] << endl;
	} else {
		int k = (z / (n + 1)) & 1;
		
		if (k) k = -1;
		else k = 1;
		
		z %= n + 1;
		
		cout << (a[z] * k + m) % m << endl;
	}
}

int main() {
	ios_base::sync_with_stdio(false);
	int t; cin >> t;
	while (t --> 0)
		solve();
	return 0;
}
