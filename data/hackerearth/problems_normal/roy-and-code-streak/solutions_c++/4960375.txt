#include<iostream>
#include<map>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,n;
	cin>>n;
	map<int,int> m;
	int count=0;
	int max=0;
	for(i=0;i<n;i++)
	{
		int pid,st;
		cin>>pid>>st;
		if(st!=0 && m.find(pid)==m.end())
		{
			m[pid]=1;
			count++;
		}
		else if(st==0)
		{
			if(count>max)
			max=count;
			count=0;
		}
		
	}
	if(count>max)
	max=count;
	cout<<max<<endl;
	}
}
