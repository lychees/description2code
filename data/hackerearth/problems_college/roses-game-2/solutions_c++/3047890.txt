#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int a[n];
    	bool visited[n];
    	bool flag=true;
    	for(int i=0;i<n;i++){
    		cin>>a[i];
    		visited[i]=false;
    	}
    //	cout<<a[n-1]<<"\n";
    	int i=0;
    	while(a[i]!=-1){
    		if(visited[i]==false)
    		visited[i]=true;
    		else{
    		flag=false;
    		break;
    		}
    		int temp=a[i];
    		i=temp;
    	}
    	if(flag)
    	cout<<"Win\n";
    	else
    	cout<<"Lose\n";
    }
    return 0;
}
