#define FNAME ""

const int MOD = 1e9 + 7;
const int MAXN = 1e6 + 5;

int fact[MAXN + 5];

int main() 
{
#ifdef LOCAL    
    freopen(FNAME".in", "r", stdin);
    freopen(FNAME".out", "w", stdout); 
#endif    
    fact[0] = 1;
    forab (i, 1, MAXN)
        fact[i] = (fact[i - 1] * 1ll * i) % MOD;
    int n, m;
    vi v;
    scanf("%d%d", &n, &m);
    forn (i, n)
        forn (j, m)
            v.pb((i + 1) ^ (j + 1));
    sort(all(v));
    v.pb(-1);
    int cur = 0, ans = 1;
    for (int i = 0; i < sz(v); i++)
    {
        if (i != 0 && v[i] != v[i - 1])
        {
            ans = (ans * 1ll * fact[cur]) % MOD;
            cur = 0;
        }
        cur++;
    }
    printf("%d\n", ans);
    return 0;
}
