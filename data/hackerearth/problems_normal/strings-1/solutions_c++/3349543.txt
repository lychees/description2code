#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string s1,s2;
    while(t--)
    {
    	cin>>s1>>s2;
    	if(s1==s2)
    	cout<<"YES\n";
    	else
    	{
    		if(s1.size()!=1||s2.size()!=1)
    		cout<<"NO\n";
    		else
    		{
    			if((s1[0]=='2'&&s2[0]=='4')||(s1[0]=='4'&&s2[0]=='2'))
    			cout<<"YES\n";
    			else
    			cout<<"NO\n";
    		}
    	}
    }
    return 0;
}
