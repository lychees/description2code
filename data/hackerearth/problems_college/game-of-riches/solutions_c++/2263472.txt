#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    int n,m;
    while(test--)
    {
        scanf("%d%d",&n,&m);
        int x,myid;
        priority_queue<int> pq;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&x);
            if(i==0)
             myid=x;
            pq.push(x);
        }
        int count=0;
        while(!pq.empty())
        {
            x=pq.top();
            if(x==myid)
                break;
            else
            count++;
            pq.pop();
        }
        if(!pq.empty())
        {
            pq.pop();
            count++;
        }
        int count1=0;
        while(count>m)
        {
            count=count-m;
            count1++;
        }
    cout<<count1<<endl;
    }
    return 0;
}
