#include <iostream>
using namespace std;

int main()
{
    string s, d;
    cin >> s;
    int t;
    cin >>t;
    int syn, ant;
    syn = 0, ant = 0;
    while(t--){	
    	cin >>d;
    	if(d.size() == s.size()){
    		if(d[0] == s[0]) syn++;
    		else if(char(d[0]-3) == s[0]) ant++;
    	}
    }

    cout << syn << " " << ant <<endl;
    return 0;
}
