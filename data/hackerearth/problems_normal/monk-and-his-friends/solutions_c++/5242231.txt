#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    int T;
    cin>>T;
    while(T--)
    {
    	int N,M;
    	cin>>N>>M;
    	unordered_map<long long int,long long int> st;
    	for(int i=0;i<N;i++)
    	{
    		long long int x;
    		cin>>x;
    		st.insert(make_pair(x,1));
    	}
    	for(int i=0;i<M;i++)
    	{
    		long long int x;
    		cin>>x;
    		
    		if(st.find(x)==st.end())
    		{
    			st.insert(make_pair(x,1));
    			cout<<"NO\n";
    		}
    		else
    		{
    			cout<<"YES\n";
    		}
    		
    	}
    }
    return 0;
}
