#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int N;
    	long long a,b;
    	cin>>a>>b>>N;
    	long long cals[N];
    	
    	for(int i = 0 ; i < N; i++)
    	{
    		cin>>cals[i];

    	}
    	sort(cals, cals+N);
    	
    	long long aT = 0 ;
    	long long aCount = 0;
    	long long bCount = 0;
    	long long bT =0;
    	
    	for(int i = 0 ; i<N; i++)
    	{
    		if(aT + cals[i]  <= a)
    		{
    			aT +=cals[i];
    			aCount++;
    		}
    		else break;
    	}
    	
    	for(int i = 0 ; i<N; i++)
    	{
    		if(bT + cals[i]  <= b)
    		{
    			bT +=cals[i];
    			bCount++;
    		}
    		else break;
    	}
    	
    	if(aCount>bCount)
    		cout<<"Raghu Won"<<endl;
    	else if(aCount< bCount)
    	cout<<"Sayan Won"<<endl;
    	else cout<<"Tie"<<endl;
    }
    return 0;
}
