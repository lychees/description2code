#include<bits/stdc++.h>
using namespace std;
char w[1000000],a[1000000];
int main()
{
int t,sy,ay;
cin>>w>>t;
sy=0,ay=0;
while(t--)
{
 cin>>a;
 if(w[0]==a[0] && strlen(w)==strlen(a))
    sy++;
 else if(w[0]+3==a[0]&& strlen(w)==strlen(a))
      ay++;
}
cout<<sy<<" "<<ay<<endl;
return 0;
}
