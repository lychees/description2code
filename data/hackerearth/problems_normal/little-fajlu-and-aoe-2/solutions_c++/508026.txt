#include <bits/stdc++.h>
using namespace std;


int main(int argc, char const *argv[])
{
	int t,food,gold,sam,cha,pal,power,i,j,k;
	cin >> t;
	while(t--)
	{
		int temp=-1;
		cin >> food >> gold >> sam >> pal >> cha >> power ;


		for(i=0;i<=food/100;i++)
		{
			for(j=0;j<=(food-100*i)/125 && j<=gold/50;j++ )

			{
				for(k=0;k<=(food-100*i-125*j)/50 && k<=(gold-50*j)/100; k++)
				{
					if(temp < sam*i+pal*j+cha*k)
						temp = sam*i+pal*j+cha*k;
				}
			}
		}

		if(temp<power)
			cout << "-1" << endl;
		else
			cout << temp-power << endl;
	}
	return 0;
}
