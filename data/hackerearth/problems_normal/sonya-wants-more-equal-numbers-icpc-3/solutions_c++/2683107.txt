#include<bits/stdc++.h>
#define ll long long
#define rep(i,x,y) for(i=x;i<y;i++)
#define pii pair<int,int>
#define pll pair<ll,ll>
#define ff first
#define ss second
#define pb push_back
#define mp make_pair
#define debug(x) cerr<<#x<<" "<<x<<endl

using namespace std;

set<pii>s[100010];
set<pii>::iterator it;
vector<int>ans;
int a[100010],sum[100010];
int main()
{
    int i,j,t,n,ct,en,mx=0;
    cin>>n;
    rep(i,1,n+1)
    cin>>a[i];

    sum[0]=0;
    rep(i,1,n+1)
    {
        sum[i]=sum[i-1]+a[i];
        s[sum[i]].insert(mp(i,1));
        //debug(sum[i]);
    }
    for(i=1;i<=n;i++)
    {
        for(j=i+1;j<=n;j++)
        {
            sum[j]-=a[i];
   //         debug(sum[j]);
            s[sum[j]].insert(mp(j,i+1));
        }
    }

    ans.clear();
    for(i=1;i<=100000;i++)
    {
        if(s[i].size()==0)
            continue;
        ct=1;
        it=s[i].begin();
        en=it->first;
        it++;
        for(;it!=s[i].end();it++)
        {
            if(it->second > en)
            {
                ct++;
                en=it->first;
            }
        }
        if(ct>mx)
        {
            mx=ct;
            ans.clear();
            ans.pb(i);
        }
        else if(ct==mx)
        {
            ans.pb(i);
        }
    }

    sort(ans.begin(),ans.end());
    cout<<mx<<endl;
    for(i=0;i<ans.size();i++)
        cout<<ans[i]<<" ";
    cout<<endl;

    return 0;
}
