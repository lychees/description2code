#include <cstdlib>
#include <iostream>
#include <cstdio>

using namespace std;

/*
 * 
 */
int main(int argc, char** argv) {

    int A[10][4]={0,0,0,0,
                  1,1,1,1,
                  2,4,8,6,
                  3,9,7,1,
                  4,6,4,6,
                  5,5,5,5,
                  6,6,6,6,
                  7,9,3,1,
                  8,4,2,6,
                  9,1,9,1};
    
    long long B,E;
    int T;
    cin>>T;
    while(T--){
        scanf("%lld%lld",&B,&E);
        int unit_B = B%10;
        int unit_E = (E-1)%4;
        printf("%d\n",A[unit_B][unit_E]);
    }
    return 0;
}
