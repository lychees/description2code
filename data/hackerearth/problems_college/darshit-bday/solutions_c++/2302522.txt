#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	long long b,w,x,y,z;
    	cin >> b >> w >> x >> y >> z;
    	long long c=x*b + y*w;;
    	if(z>x+y)
    	{
    		c=x*b + y*w;
    	}
    	else if(y>x+z)
    	{
    		c=x*(b+w) + z*w;
    	}
    	else if(x>y+z)
    	{
    		c=y*(b+w) + z*b;
    	}
    	cout << c << endl;
    }
}
