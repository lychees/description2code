t = int(input())
while t:
    t -= 1
    s = raw_input()
    l = len(s)
    c = 0
    h = 0
    for i in range(0, l):
        if s[i] == 'A' or s[i] == 'a' or s[i] == 'B' or s[i] == 'b' or s[i] == 'C' or s[i] == 'c':
            h = h+1
        else:
            c = c+((h*(h+1)/2))
            h = 0
    c = c+((h*(h+1)/2))
    print c
