#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		ll n;
		cin>>n;
		string s;
		cin>>s;
		ll ans=0;
		ll sum=1;
		for(ll i=1;i<n;i++){
			
			if(s[i]>=s[i-1]){
				sum++;
			}else{
				ans+=((sum)*(sum+1))/2;
				sum=1;
			}
		}
		ans+=((sum)*(sum+1))/2;
		cout<<ans<<endl;
	}
	return 0;
}
