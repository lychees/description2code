t = input()
while t > 0:
    t = t-1
    c = 0
    d = 0
    count = 0
    a, b = map(int, raw_input().split())
    while a != 0 or b != 0:
        c = a % 10
        d = b % 10
        if (c+d) > 9:
            count += 1
        a = a/10
        b = b/10
    if count == 0:
        print "No carry operation"
    elif count == 1:
        print count, "carry operation"
    else:
        print count, "carry operations"
