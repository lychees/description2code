int n;

int b1=39,b2=37;
pii ghs(int hs[])
{
	ll m1=0,m2=0;int i;
	for (i=0;i<26;i++)
	{
		m1=(b1*m1+hs[i])%MOD;
		m2=(b2*m2+hs[i])%MOD;
	}
	return mp((int)m1,(int)m2);
}

		{
			int hs[26]={0};
			for (j=n/2;j<n;j++)
			{
				int x=j-n/2;
				if (msk&(1<<x))

					for (k=0;k<lad[j].size();k++)
