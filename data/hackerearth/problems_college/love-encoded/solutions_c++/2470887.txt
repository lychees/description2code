#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
	int T,N;
	string a;
	cin>>T;
	while(T--)
	{
		cin>>N;
		N=N%26;
		cin>>a;
		int k;
		for(int i=0;i<a.length();i++)
		{
			k=int(a[i]);
			k-=N;
			if(k<65)
			{
				a[i]=a[i]+26-N;

			}
			else
			{
				a[i]-=N;
			}
			
		}
		cout<<a<<endl;
	}
	return 0;
}
