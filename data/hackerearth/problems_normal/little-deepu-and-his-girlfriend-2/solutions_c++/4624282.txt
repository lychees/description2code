#include <bits/stdc++.h>
using namespace std;

    	int dp[100000 + 2];
    	int a[100005];
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
    	int m, n;
		scanf("%d%d", &m, &n);
		for ( int i = 0; i < n; i++ )
		scanf("%d", &a[i]);
    	sort(a, a+ n);
    	for(int i = 0; i <= m; i++)
    	{
    		dp[i] = 1;
    		for(int j = 1; j <= n && i >= a[j - 1]; j++)
    		{
    				if(i >= a[j - 1] && dp[i - a[j - 1]] == 1)
    				{	dp[i] = 0; break;}
    			
    		}
    	}
    	cout << ((dp[m] != 0) ? "Kate" : "Little Deepu") << "\n";

    }
    return 0;
}
