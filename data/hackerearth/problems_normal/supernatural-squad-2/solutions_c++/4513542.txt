#include<bits/stdc++.h>
using namespace std;
#define f(i,a,b) for(nt i=(nt)(a);i<=(nt)(b);i++)
typedef long long int nt;
nt k,dp[205][205];
nt fun(nt n,nt pre)
{nt i;
if(n==0)return 1;
if(n<=k-1)return 0;
if(dp[n][pre]!=-1)return dp[n][pre];
	nt count=0;
	for(i=pre;i>=k;i--)
	count+=fun(n-i,i);
	dp[n][pre]=count;
	return count;
}
int main()
{nt t,n;
cin>>t;
while(t--)
{memset(dp,-1,sizeof(dp));
	cin>>n>>k;
	if(k>n)cout<<"0"<<endl;
	else cout<<fun(n,n)<<endl;
}
	return 0;
	
}
