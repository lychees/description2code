typedef long long int lli;

void sc(const lli& a){scanf("%lld",&a);}
void pr(const int& a){printf("%d",a);}
void pr(const lli& a){printf("%lld",a);}
int modpow(int a, int b, int mod)
{
    lli res=1;
    while(b>0)
    {
        if(b&1) res=(res*a)%mod;
        a=(a*a)%mod;
        b=b/2;
    }
    return res%mod;
} 
int arr[100005];
 
int main()
{
    int t,n,i;
    sc(t);
    lli cost=0;
    while(t--)
    {
        sc(n);
        cost=0;
        for(i=0;i<n;i++)
            sc(arr[i]);
        for(i=0;i<n-1;i++)
            cost+=max(arr[i],arr[i+1]);
        pr(cost);line();
    }
}
