#include <iostream>
#include<stdio.h>
using namespace std;
#define primemax 1000000
typedef long long ll;
ll arr[50][50];
ll res[50][50];
ll temp[50][50];
ll m,n;
#define M 1000000007
void print()
{
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
            cout<<temp[i][j];
        cout<<endl;
    }
}
void calc()
{
    for(int i=0;i<m;i++)
        for(int j=0;j<m;j++)
            if(i==j)
            res[i][j]=1;

    while(n>0)
    {
            if(n&1)
            {
                for(int i=0;i<m;i++)
                    for(int j=0;j<m;j++)
                    {
                        ll val=0;
                        for(int k=0;k<m;k++)
                            val=(val+res[i][k]*arr[k][j])%M;
                        temp[i][j]=val;
                    }
             //       cout<<"temp"<<endl;
           // print();
                for(int i=0;i<m;i++)
                    for(int j=0;j<m;j++)
                        res[i][j]=temp[i][j];
            }
            for(int i=0;i<m;i++)
                    for(int j=0;j<m;j++)
                    {
                        ll val=0;
                        for(int k=0;k<m;k++)
                            val=(val+arr[i][k]*arr[k][j])%M;
                        temp[i][j]=val;
                    }
            for(int i=0;i<m;i++)
                for(int j=0;j<m;j++)
                    arr[i][j]=temp[i][j];
            n>>=1;

    }
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
            cout<<res[i][j]<<" ";
        cout<<endl;
    }
}
int main()
{
    //cout << "Hello World!" << endl;
   // freopen("in.txt","r",stdin);
    ll t;
    cin>>t;
    while(t--)
    {

        cin>>m>>n;
        for(int i=0;i<m;i++)
            for(int j=0;j<m;j++)
                cin>>arr[i][j];
        calc();
        for(int i=0;i<m;i++)
            for(int j=0;j<m;j++)
                arr[i][j]=temp[i][j]=res[i][j]=0;
    }
    return 0;
}
