#include <iostream>
#include <vector>
using namespace std;
//vector<int> vec[1000000];
long int con[1000000];

void init(){
	for(long int i=0;i<1000000;i++){
		con[i] = 0;
	}
}
int main()
{
    long int t,n,a,b,i,count;
    cin>>t;
    while(t--){
    	cin>>n;
    	count = 0;
    	init();
    	for(i=0;i<n;i++){
    		cin>>a>>b;
    		con[a] = con[a]+1;
    		if(con[b]>0){
    			con[b]--;
    		}
    		else{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
