#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define s(n) scanf("%d", &n);
#define sll(n) scanf("%lld", &n);
#define MOD 1000000007
#define mp make_pair
#define pb push_back
 
  
int main()
{
	int n,a,tmp,b,c,x,y,t,i,j,l,r;
	s(t);
	while(t--)
	{
		l=r=0;
		s(n);
		s(a)s(b)s(c);
		for(i=1;i<=n;i++)
		{
			s(x)s(y);
			tmp=a*x+b*y+c;
			if(tmp<0)
				l++;
			else
			if(tmp>0)
				r++;
		}
		printf("%d\n", max(l,r));
	}
	
	
	return 0;
}
