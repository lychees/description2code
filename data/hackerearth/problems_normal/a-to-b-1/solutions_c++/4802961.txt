#include<bits/stdc++.h>
using namespace std;
typedef  long long int lu;
#define  od 1000000007

lu modi(lu n,lu x)
{
	if(x==0)
	return 1;
	lu ans=modi(n,x/2)%od;
	ans=(ans*ans)%od;
	if(x%2!=0)
	return (ans*n)%od;
	else 
	return ans;
}

int main()
{
	lu n,q,i,v,id,sw,getr,ans,zq=0,nukha;
	scanf("%lld",&n);
	lu ar[n];
	ans=1;
	zq=0;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&ar[i]);
		if(ar[i]!=0)
		ans=(ans*ar[i])%od;
		else zq++;
	}
	
	scanf("%lld",&q);
	while(q--)
	{
		scanf("%lld",&sw);
		if(sw==1)
		{
			scanf("%lld",&id);
			id--;
			if(ar[id]==0&&zq>1)
			printf("0\n");
			else if(ar[id]==0&&zq==1)
			printf("%lld\n",ans);
			else if(ar[id]!=0&&zq>0)
			printf("0\n");
			else if(ar[id]!=0&&zq==0)
			{
				getr=modi(ar[id],1000000005);
				nukha=(ans*getr)%od;
				printf("%lld\n",nukha);
			}
		}
		else if(sw==0)
		{
			
			
			scanf("%lld%lld",&id,&v);
			id--;
			if(ar[id]==0&&v==0)
			continue;
			else if(ar[id]!=0&&v==0)
			{
				zq++;
				getr=modi(ar[id],1000000005);
				
				ans=(ans*getr)%od;
				ar[id]=0;
			}
			else if(ar[id]==0&&v!=0)
			{
				zq--;
				ar[id]=v;
				ans=(ans*v)%od;
			}
			else if(ar[id]!=0&&v!=0)
			{   
				getr=modi(ar[id],1000000005);
				ans=(((ans*v)%od)*getr)%od;
				ar[id]=v;
			}
		}
	}
  return 0;
}
