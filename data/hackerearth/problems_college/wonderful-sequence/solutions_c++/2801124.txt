#include <iostream>

using namespace std;

int main()
{
    int t, n;
    
    cin>>t;
    
    while(t--)
    {
    	cin>>n;
    	
    	if(n == 0) 
    		cout<<"0"<<endl;
    	else
    	{
    		int ans = 0;
    		int sequence[n];
    		
	    	cin>>sequence[0];
	    	
	    	for(int i = 1; i<n; i++)
	    	{
	    		cin>>sequence[i];
	    		if(sequence[i] <= sequence[i-1])
	    			ans++;
	    	}
	    	cout<<ans<<endl;
    	}
    	
    	
    }
    
    return 0;
}
