#include<bits/stdc++.h>
using namespace std;
int main(){
    int arr[27][27][27][27][27];
    memset(arr,0,sizeof(arr));
    int t;
    scanf("%d",&t);
    int ans = 0;
    
    while(t--){
        char s[6];
        scanf("%s",s);
        int l = strlen(s);
        int a = 0,b=0,c=0,d=0,e=0,cnt = 0;
        for(int i = 0 ; i < l;i++){
            if(i == 0)
                a = s[i]-'a' + 1 ;
            else if(i == 1)
                b = s[i]-'a' + 1;
            else if(i == 2)
                c = s[i]-'a' + 1;
            else if(i == 3)
                d = s[i]-'a' + 1;
            else if(i == 4)
                e = s[i]-'a' + 1;
        }
        if(!arr[a][b][c][d][e])
         {
         	ans++;
         	arr[a][b][c][d][e]++;
         }
    }
   
	printf("%d\n",ans);
    for(int i = 0 ; i < 27; i++){
        for(int j = 0 ; j < 27 ;j++){
            for(int k = 0 ; k < 27 ;k++){
                for(int l = 0 ;l < 27 ;l++){
                    for(int m = 0 ; m < 27 ;m++){
                        if(arr[i][j][k][l][m]){
                        	if(i)
                            printf("%c",i+'a'-1);
                            if(j)
                            printf("%c",j+'a'-1);
                            if(k)
                            printf("%c",k+'a'-1);
                            if(l)
                            printf("%c",l+'a'-1);
                            if(m)
                            printf("%c",m+'a'-1);
                            printf("\n");
                        }
                    }
                }
            }
        }
    }

    return 0;
}
