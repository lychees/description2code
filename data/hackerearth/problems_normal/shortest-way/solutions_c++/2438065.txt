#include <iostream>
#include <cmath>
using namespace std;

int main() {	
	int t;
	long long a,b;
	cin >> t;
	long long count=0;
	long long temp;
	while(t--) {
		cin >> a >> b;
		count = 0;
		while(true) {
			if(a==1 || b==1) {
				if(a==1) {
					count += b-1;
					break;
				} else {
					count += a-1;
					break;
				}
			} else if(a>b) {
				count += a/b;
				a = a%b;
			} else {
				count += b/a;
				b = b%a;
			}
		}
		cout << count << endl;
	}
	return 0;
}
