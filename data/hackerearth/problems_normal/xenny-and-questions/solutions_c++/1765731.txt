#include <iostream>
#include <string>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <vector>
#include <stack>

using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
    int n,k,mx=-999;
    scanf("%d%d",&n,&k);
    int hash[100005]={0};
    for(int i=0;i<n;i++)
    {
        int temp;
        scanf("%d",&temp);
        mx=max(mx,temp);
        hash[temp]++;
    }
    int i=mx;
    if(k==0 || mx==0 || n==0){
    printf("0\n");
        continue;}
   long long res=0;
    while(1)
    {
        if(hash[i]!=0)
        {
            int tm=hash[i];
            for(int I=0;I<tm;I++)
            {
                res=res+i;
                k--;
                if(k==0)
                break;
            }
            
        }
        if(k==0 || i==0)
            break;
        i--;
        
    }
    printf("%lld\n",res);
}
    return 0;
}
