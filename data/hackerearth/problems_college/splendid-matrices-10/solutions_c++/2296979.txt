#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int n,k=1;
	cin>>n;
	int size=pow(2,n);
	int SplendidMatrix(int **,int,int,int,int);
	int** ar = new int*[size];
	for(int i = 0; i < size; ++i)
    	ar[i] = new int[size];
    for(int i=0;i<size;i++)
    	for(int j=0;j<size;j++)
    		ar[i][j]=0;
    k=SplendidMatrix(ar,0,0,size,k);
    for(int i=0;i<size;i++){
    	for(int j=0;j<size;j++)
    		cout << ar[i][j]<<" ";
    	cout<<"\n";
    }
    return 0;
}
int SplendidMatrix(int **ar,int left,int right,int size,int k){
	if(size==2){
		ar[left][right]=k++;
		ar[left][right+1]=k++;
		ar[left+1][right]=k++;
		ar[left+1][right+1]=k++;
		return k;
	}
	else{
		size/=2;
		k=SplendidMatrix(ar,left,right,size,k);
		k=SplendidMatrix(ar,left,right+size,size,k);
		k=SplendidMatrix(ar,left+size,right,size,k);
		k=SplendidMatrix(ar,left+size,right+size,size,k);
		return k;
	}
	
}
