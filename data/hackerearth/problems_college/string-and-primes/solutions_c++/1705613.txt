#include<iostream>
using namespace std;
#include<string.h>
#include<math.h>
bool isprime(int num)
{
	int i;
	if(num==1) return false;
	if(num==2) return true;
	if(!(num&1)) return false;      ////even value bitch!!!!!!!
	for(i=3;i<=(int)sqrt(num);i++)
	{
	if(num%i==0)
	{
		return false;
	}
	}
	
}
int main()
{
	int t,i,ct=1,k=0;
	char str[100001];
	char res[100001];
	cin>>t;
	while(t--)
	{
		k=0;
		ct=1;
	cin>>str;
	int len=strlen(str);
	for(i=0;i<len;i++)
	{
	int t=str[i];
	if(!(isprime(t)))
	{
	res[k++]=str[i];	
	}
	res[k]='\0';
	}
	if(k==0)
	{
		cout<<"NULL";
	}
	else
	{
		cout<<res;
	}

cout<<"\n";
	}
    return 0;
}
