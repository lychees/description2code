#include <bits/stdc++.h>
using namespace std;


int main()
{
  
    int t;
    cin>>t;
    while(t--)
    {
        long long int b,w,x,y,z;
        cin>>b>>w>>x>>y>>z;
        long long int ansb=0,answ=0;
        if(x>y)
         {
            answ=y*w;
            ansb=min(b*x,b*(y+z));
         }
        else if(x<y)
            {
          
            ansb=x*b;
            answ=min(w*y,w*(x+z));
         }else
            {
            
            ansb=b*x;
            answ=w*y;
        }
        cout<<ansb+answ<<endl;
            
     }
        
        
        
        
    
    return 0;
}
