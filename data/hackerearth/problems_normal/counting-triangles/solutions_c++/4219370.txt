#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
