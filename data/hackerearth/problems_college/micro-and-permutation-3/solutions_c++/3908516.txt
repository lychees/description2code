#include <bits/stdc++.h>
 
using namespace std;
 
int myCompare(string X, string Y){
	string XY = X.append(Y);
	string YX = Y.append(X);
	return XY.compare(YX) > 0 ? 1 : 0;
}
 
int main()
{
	int n;
	cin >> n;
	vector<string > a;
	string t;
	for(int i =0;i< n;i++){
	  cin >> t;
	  a.push_back(t);
		
	}
	sort(a.begin(),a.end(),myCompare);
    for(int i=0;i<n;i++){
    	cout<< a[i];
    }
    cout << endl;
    return 0;
}
