#include <iostream>
using namespace std;

int main()
{
   int t,n;
   string s1,s2;
   cin>>t;
   while(t--)
   {
   	cin>>s1>>s2>>n;
   	int count=0;
   	for(int i=0;i<s1.size();i++)
   	   if(s1[i]!=s2[i]) count++;
    if(count<=n && (n-count)%2==0)	 cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
   }
}
