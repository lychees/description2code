#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    string s;
    map <string, int > m;
    while(n--) {
    	cin >> s;
    	if(m.find(s) != m.end()) m[s]++;
    	else m[s] = 1;
    }
    map < string, int > :: iterator it;
    for(it=m.begin();it!=m.end();it++) {
    	cout << it->first << ' ' << it->second << endl;
    }
    return 0;
}
