#include <iostream>
#include<ctype.h>
using namespace std;

int main()
{
    string str;
    cin>>str;
    for(int i = 0; i < str.length()-1; i++)
      {
            if (i == 0)
              str[i] = toupper(str[i]);
            if (str[i] == '_')
              str[i + 1] = toupper(str[i + 1]);
      }
    
    cout<<str;
    return 0;
}
