#include<iostream>
using namespace std;
#include<stdio.h>
#include<math.h>
#include<stdlib.h>
long long a[1000006]={0};
int main()
 {
 int t,n,i,j;
 cin>>t;
 n=t;
 for(i=0;i<n;i++)
   {
   	cin>>a[i];
   }
   long long sum=0;
   for(i=0;i<n;i++)
     {for(j=i+1;j<n;j++)
        {sum = sum | (a[i]^a[j]);
        }
     }
    cout<<sum<<endl;;
}
