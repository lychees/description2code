#include<bits/stdc++.h>
using namespace std;
#define s(x) scanf("%d",&x);
#define sl(x) scanf("%lld",&x);
#define p(x) printf("%d",x);
#define pl(x) printf("%lld",x);
#define fr(i,a,n) for(i=a;i<n;i++)
#define ln() printf("\n");
typedef long long ll;
ll mod=1000000007;
int main()
{
    int t;
    s(t);
    while(t--)
    {
        ll n,i,j,k,tmp,maxx=0;
        sl(n)sl(k);
        ll a[n],dp[n];
        for(i=0;i<n;i++)
            sl(a[i]);
        for(i=0;i<n;i++)
        {
            if(i-k>0)
            {
                dp[i]=max(maxx,a[i]+dp[i-k-1]);
                maxx=dp[i];
            }
            else
            {
                dp[i]=max(maxx,a[i]);
                maxx=dp[i];
            }
        }
        printf("%lld ",dp[n-1]);
        ln();
    }

    return 0;
}
