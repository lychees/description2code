#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	double arr[1005], out[1005];
	double prob, ans = 0;
	int r, n;
	for(int i = 0; i < t; i++) {
		scanf("%d %d",&r,&n);
		double sum = 0, sum1 = 0;
		for(int j = 1; j <= r; j++) {
			arr[j] = (double)j/(double)n;
			sum += (arr[j]*arr[j]);
		}
		for(int j = 1; j <= r; j++) {
			prob = (arr[j]*arr[j])/sum;
			sum1 += prob*arr[j];
		}
	//	cout << sum1 << endl;	
		out[i] = sum1;
	}
	ans = 1;
	for(int i= 0; i < t; i++) {
		ans *= out[i];
	}
	printf("%.4lf\n",ans);
	return 0;
}
