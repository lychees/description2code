#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n;
cin>>n;
int arr[n];
int i;
for(i=0;i<n;i++)
{
	cin>>arr[i];
}
int x;
cin>>x;
int flag = 0;
sort(arr,arr+n);

for(i=0;i<n-2 && flag == 0  ;i++)
{
	int l = i+1;
	int r = n-1;
	while(l<r && flag == 0 )
	{
	if(arr[i] + arr[l] + arr[r] == x)
	{
		cout<<"True";
	    flag = 1;
	}
	else if(arr[i] + arr[l] + arr[r] < x)
        {
        l++;
	}
	else{
        r--;
	}
	}
}
if( flag == 0 )
{
    cout<<"False";
}


}
