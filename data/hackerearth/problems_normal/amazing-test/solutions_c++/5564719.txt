#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,x,sum=0;;
		cin>>n>>x;
		int arr[n+1],ans[n+1][x+1];
		ans[0][0]=0;
		for(int i=1;i<=n;i++){cin>>arr[i];
							  sum+=arr[i];
	                         ans[i][0]=0;          }
	     for(int i=0;i<=x;i++)ans[0][i]=0;                    
		for(int i=1;i<=n;i++){
			for(int j=1;j<=x;j++){
				ans[i][j]=ans[i-1][j];
				if(j>=arr[i]){ans[i][j]=max(ans[i][j],ans[i-1][j-arr[i]]+arr[i]);
			
				}
				
			}
		}
		
      if(sum-ans[n][x]>x)cout<<"NO\n";
      else cout<<"YES\n";
	}
}
