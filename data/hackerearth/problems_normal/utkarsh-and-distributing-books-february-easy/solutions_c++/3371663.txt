#include <iostream>
using namespace std;

int main()
{
	int n,t,max=1000000,k,sum=0;
	cin>>n;
	for(int i=0;i<n;i++ )
	{	max=1000000;
		sum=0;
		int k;
		cin>>k;
		for(int j=0;j<k;j++)
		{
			cin>>t;
			if (t<max)
			 max=t;
			sum+=t-1;
			
		}
		if(max<0)
		max=0;
	cout<<max-1<<" "<<sum<<endl;
	}
    return 0;
}
