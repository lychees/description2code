#include <bits/stdc++.h>
using namespace std;
const int maxn = 100001;
bool arr[maxn];
void siv(){
	for(int i=0;i<=sqrt(maxn);i++){
		if(arr[i] == false){
			for(int j = i*i; j<maxn;j +=i){
				arr[j] = true;
			}
		}
	}
}
int main()
{
	int t; cin>>t;
	while(t--){
		int n; cin>>n;int flag = 0;
		for(int i=2;i<=n/2;i++){
			if(arr[i] == false && arr[n-i] == false){
				flag = 1; break;
			}
		}
		flag ? puts("Deepa") : puts("Arjit");
	}
	
	
	return 0;
}
