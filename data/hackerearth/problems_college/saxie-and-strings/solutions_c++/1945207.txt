#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s,s1;
    while(t--)
    {
    	cin>>s;
   		s1.assign(s);
   		reverse(s.begin(),s.end());
   		if(s1.compare(s)==0)
   		{
    		cout<<"YES"<<endl;
   		}
   		else
   		{
   			cout<<"NO"<<endl;    
    	}
    	s1.clear();
    	s.clear();
    }
}
