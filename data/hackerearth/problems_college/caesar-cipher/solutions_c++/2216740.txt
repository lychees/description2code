#include <bits/stdc++.h>
using namespace std;

int main() 
{
	// your code goes here
	int test; cin>>test;
	while(test--)
	{
		int k; cin>>k;
		k=k%26;
		string s; cin>>s;
		for(int i =0;i<s.size();i++)
		{
			if(s[i]>='a'&&s[i]<='z')
			{
				int l = s[i] - 'a';
				int kk=l+k;
				if(kk>25)kk=kk-26;
				//cout<<k<<endl;
				printf("%c",'a'+kk);
			}
			if(s[i]>='A'&&s[i]<='Z')
			{
				int l = s[i] - 'A';
				int kk=l+k;
				if(kk>25)kk=kk-26;
				//cout<<k<<endl;
				printf("%c",'A'+kk);
			}
		}
		cout<<endl;
	}
	return 0;
}
