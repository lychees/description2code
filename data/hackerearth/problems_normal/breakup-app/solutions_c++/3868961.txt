#include<iostream>
#include<string>
#include<algorithm>
#include<climits>
using namespace std;
int main(){
	int t;
	cin>>t;
	cin.ignore(INT_MAX,'\n');
	int arr[31]={0};
	int max=INT_MIN;
	while(t--){
		string str;
		getline(cin,str);
		int lim=str.size();
		char key=str[0];
		 
		for(int i=0;i<lim;i++){
			int num=0;
			if(str[i]>='1'&&str[i]<='9'){
				num=num*10+str[i]-48;
				i++;
				if(str[i]>='0'&&str[i]<='9')
					num=num*10+str[i]-48;
				if(key=='G')
					arr[num]+=2;
				else
					arr[num]+=1;
				if(max<arr[num])
					max=arr[num];
			}
		}
		}
		if(arr[19]==max||arr[20]==max){
			cout<<"Date";
		}
		else{
			cout<<"No Date";
		}
	}
