#include <bits/stdc++.h>
using namespace std;
#define ll

ll int func(ll int x,ll int y,ll int c){
    if (x==y&&x!=1) return -1;
    if (x==1&&y==1) return c;
    if (x%2==0&&y%2==0) return -1;
    ll int b,s;
    if (x>y){
        b=x;
        s=y;
    }
    else{
        b=y;
        s=x;
    }
    return func(b-s,s,c+1);
}

ll int f(ll int x){
    ll int m=x;
    for (int i=1;i<x/2+1;i++){
        ll int temp=func(x-i,i,0);
        if (temp==-1) continue;
        temp++;
        m=min(temp,m);
    }
    return m;
}

int main(){
    ll int t,l;
    cin >> t;
    while (t--){
        cin >> l;
        if (l!=1)
        cout << f(l) << endl;
        else cout << 0 << endl;
    }
    return 0;
}
