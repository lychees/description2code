#include <iostream>
#include <string.h>
using namespace std;


class num{
public:
	char n[200];
	int dgt;
	void mul(int a){
		int m,c=0,r,i;
		for(i=0;i<dgt;i++){
			m=n[i]-48;
			r=m*a+c;
			n[i]=r%10+48;
			c=r/10;
		}
		while(c>0){
			n[i++]=c%10+48;
			c=c/10;
			++dgt;
		}
	}
	void print(){
		for(int i=dgt-1;i>=0;i--){
			printf("%c",n[i]);
		}
	}
};

int main()
{
    
    num rst;
	char t,n,i;
	scanf("%d",&t);
	for(i=0;i<t;i++){
		scanf("%d",&n);
		rst.dgt=1;
		memset(rst.n, '0', 200 * sizeof(char));
		rst.n[0]='1';
		while(n){
			rst.mul(n--);
		}
		rst.print();
		printf("\n");
	}
	return 0;
}
