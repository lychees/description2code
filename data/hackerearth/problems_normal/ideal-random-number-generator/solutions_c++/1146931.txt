#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	long long int A,B,C,P,Q,x;
	cin>>A>>B>>C;
	
	P=C*C;
	Q=2*A*B;
	if(A+B<=C)
	{
		P=Q=1;
	}
	else
	{
		if(A<C) P-=(C-A)*(C-A);
		if(B<C) P-=(C-B)*(C-B);
	x=__gcd(P,Q);
	P/=x;
	Q/=x;
	}
    cout <<P<<"/"<<Q<< endl;
    return 0;
}
