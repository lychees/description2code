#include<bits/stdc++.h>
using namespace std;
int convert(int x)
{
	int an=0,b=1;
	while(x!=0)
	{
		if(x%10==1)
			an+=b;
		x/=10;
		b*=2;
	}
	return an;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int x,y,a,b;
		scanf("%d %d",&x,&y);
		a=convert(x);
		b=convert(y);
		double n=a*1.0*log2(b);
		double m=b*1.0*log2(a);
		double o=min(m,n);
		if(o==m)
			printf("The Hound\n");
		else
			printf("The Mountain\n");
	}
	return 0;
}
