typedef long long int uli;
using namespace std;

const int mx = 300;
char b[mx][mx];
int n;

void draw(int r,int c,int dr,int dc,char ch){    
    int cr,cc=c;
    for(int len=n;len>0;len-=2){
        cr = r;
        int c0 = cc;
        for(int i=0;i<len;i++){
            b[cr][cc] = ch;
            cr+=dr;
            cc+=dc;
        }
        cc = c0+2*dc; 
    }
}
void solve(){   
    for(int i=0;i<mx;i++)
        for(int j=0;j<mx;j++)
            b[i][j] = '.';
    draw(n-1,0,-1,1,'/');
    draw(n-1,n+n-1,-1,-1,'\\');
    draw(n,2,1,1,'\\');
    draw(n,n+n-1+2,1,-1,'/');

}
void prnt(){
    for(int i=0;i<mx;i++){
        int lst = -1;
        for(int j=0;j<mx;j++){
            if(b[i][j]!='.') lst = j;
        }
        for(int j=0;j<=lst;j++){
            if(b[i][j]=='.')printf(" ");
            else  printf("%c",b[i][j]);
        }
        if(lst==-1)break;
        puts("");
    }
}
int main(){
    //    freopen("data.in","r",stdin);
//    freopen("data.out","w",stdout);
    scanf("%d",&n);
    solve();   
    prnt();

    n++;
    solve();
    prnt();
    return 0;
}
