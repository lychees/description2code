#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define MOD 1000000007
#define PI 3.14159265359
#define ff first
#define ss second

using namespace std;

typedef long long ll;
typedef vector<ll> vi;
typedef pair<int,int> pi;
typedef vector<pi > vpi;

ll a[10002];

int main() {
    
    ios:: sync_with_stdio(0);
    cin.tie(NULL),cout.tie(NULL);
    int T;
    cin>>T;
    while(T--){
        int n,k,i,j;
        cin>>n>>k;
        for(i=1;i<=n;i++){
            cin>>a[i];
        }
        vi dp(n+1),maxv(n+1);
        maxv[0]=INT_MIN;
        for(i=1;i<=k+1;i++){
            dp[i]=a[i];
            maxv[i]=max(dp[i],maxv[i-1]);
        }
        for(j=k+2;j<=n;j++){
            if(maxv[j-k-1]<=0)
                dp[j]=a[j];
            else
                dp[j]=a[j]+maxv[j-k-1];
            maxv[j]=max(dp[j],maxv[j-1]);    
        }
        ll mv=INT_MIN;
        for(i=1;i<=n;i++){
            mv=max(mv,dp[i]);
            //cout<<dp[i]<<endl;
        }
        if(mv>0)
            cout<<mv<<endl;
        else
            cout<<0<<endl;
    }
	return 0;
}
