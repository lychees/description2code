#include <iostream>
using namespace std;
void swap(int *a,int *b)
{
	int temp=*a;
	*a=*b;
	*b=temp;
}
int main()
{
    int n;
    cin>>n;
    int a[n];
    int ev=0,odd=0;
    for(int i=0;i<n;i++)
    {cin>>a[i];
    if(a[i]%2)
    odd++;
    else
    ev++;
    }
    for(int i=1;i<n;i++)
    {
    	int j=i-1;
    	int temp=i;
    	if(!(a[i]%2))
    	{
    		while((j>=0)&&((a[j]>a[temp])||(a[j]%2)))
    		{
    			swap(&a[temp],&a[j]);
    			temp=j;
    			j--;
    		}
    	}
    	else
    	{
    		while((j>=0)&&(a[j]>a[temp])&&((a[j]%2)))
    		{
    			
    			swap(&a[temp],&a[j]);
    			temp=j;
    			j--;
    		}
    	}
    }
    int sum=0,once=1;
    for(int i=0;i<n;i++)
    {
    	cout<<a[i]<<" ";
    	if(ev>0)
    	{
    		sum+=a[i];
    		ev--;
    	}
    	else
    	{
    		sum+=a[i];
    		odd--;
    	}
    	if((ev==0)&&(once))
    	{
    	cout<<sum<<" ";
    	once=0;
    	sum=0;
    	}
    	if(odd==0)
    	{
    	cout<<sum<<" ";
    	sum=0;
    	}
    	
    }
    return 0;
}
