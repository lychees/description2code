#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int> val[100001];
int main(){
	for(int i=0;i<=100000;i++)val[i].first=1,val[i].second=i;
	for(int i=2;i<=100000;i++){
		int v=1,x=i;
		for(int j=2;j<=317;j++){
			if(x%j==0)v*=j;
			while(x%j==0)x/=j;
		}
		v*=x;
		val[i].first=v;
	}
	sort(val,val+100001);
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		cout<<val[n].second<<endl;
	}
	return 0;
}
