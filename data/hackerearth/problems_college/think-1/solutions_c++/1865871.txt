#include <iostream>
using namespace std;

int rev(int x)
{
	int temp=0;
	while(x>0)
	{
		int n=x%10;
		temp=temp*10+n;
		x/=10;
	}
	return temp;
	}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int k;
		cin>>k;
		int z=rev(k);
		int sqz=z*z;
		int sqk=k*k;
		int reva=rev(sqz);
		if(z!=k && sqk==reva)
		cout<<z<<endl;
		else cout<<"No"<<endl;
	}
	
    
    
}
