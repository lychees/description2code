#include <iostream>
#include <bits/stdc++.h>
using namespace std;
map <string,int> arr;
map <string,int>::iterator it;
string st[11111];
string temp;
string conv(string s){
	string ret="";
	int i,k=s.size();
	for(i=0;i<k;i++){
		if((int)(s[i])>96)ret+=(char)(s[i]-32);
		else ret+=s[i];
	}
	return ret;
}
int main()
{
    int i,k=0;
    while(!feof(stdin)){
    	cin>>temp;
    	temp=conv(temp);
    	st[k++]=temp;
    	arr[temp]++;
    }
    for(i=0;i<k;i++){
    	if(arr[st[i]])cout<<st[i]<<" "<<arr[st[i]]<<endl;
    	arr[st[i]]=0;
    }
    return 0;
}
