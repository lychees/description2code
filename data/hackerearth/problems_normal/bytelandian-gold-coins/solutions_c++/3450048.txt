#include <iostream>
#include <map>
using namespace std;

typedef unsigned long long UINT64;

map<UINT64, UINT64> gMap;

UINT64 f(UINT64 a)
{
	if(gMap.find(a) != gMap.end())
	{
		return gMap[a];
	}
	if(a == 0)
	{
		return 0;
	}
	UINT64 nVal = f(a/4) + f(a/3) + f(a/2);
	nVal = max(nVal, a);
	gMap[a] = nVal;
	return nVal;
}

int main()
{
	UINT64 num;
	cin >> num;
	cout << f(num) <<"\n";
}
