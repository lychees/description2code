#include <iostream>
#include <stdio.h>
using namespace std;

long long alpha[26];

int main()
{
    int T;
    scanf("%d", &T);
    while(T--) {
    	long long N;
    	scanf("%lld", &N);
    	for(int i=0; i<26; i++)
    		scanf("%lld", &alpha[i]);
    	
    	long long x = (12*N)/26;
    	long long y = (12*N)%26;
    	long long val = 0;
    	for(int i=0; i<26; i++) {
    		val += alpha[i]*x + (x*(x-1))/2;
    		alpha[i] += x;
    	}
    	for(int i=0; i<y; i++) {
    		val += alpha[i]++;
    	}
    	printf("%lld\n", val);
    }
    return 0;
}
