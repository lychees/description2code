#include <iostream>
using namespace std;
long long int arr[1000];
int main()
{
	long long int n,m,x;
	cin >> n >>m;
	arr[0]=0,arr[1]=1;;
	long long int sum=1;
	for(int i=2;i<=n;i++)
	{
		arr[i] = arr[i-1] + arr[i-2];
		sum+=arr[i];
	}
	for(int i=1;i<=m;i++)
	{
		cin >> x;
		sum-=x;
	}
	cout << sum <<endl;
	
    
    return 0;
}
