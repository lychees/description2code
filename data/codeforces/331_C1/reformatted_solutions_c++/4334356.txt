#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>

using namespace std;
int ans;
long long x;

int main() {
  cin >> x;
  while (x > 0) {
    long long num, tp;
    num = 0;
    tp = 1;
    while (tp <= x) {
      num = max(num, (x / tp) % 10);
      tp = tp * 10;
    }
    x -= num;
    ans++;
  }
  cout << ans << endl;
  return 0;
}
