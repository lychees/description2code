#include<iostream>
using namespace std;
int main()
{
    int t,h,m,ha,ma,angle;
    cin>>t;
    while(t--)
    {
        cin>>h>>m;
        ha=30*h+m/2;
        if(h==12)
            ha=m/2;
        ma=6*m;
        if(ma>ha)
            angle=ma-ha;
        else
            angle=ha-ma;
        if(angle>180)
            angle=360-angle;
        cout<<angle<<endl;
    }
}
