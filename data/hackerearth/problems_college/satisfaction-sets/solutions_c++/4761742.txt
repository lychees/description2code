#include <iostream>
using namespace std;

int main()
{
	int n, a[501] = {0}, b[501] = {0};
	cin >> n;
	int s = n;
	int sum=0, sum2 = 0;
	while(n--){
		int m;
		cin >> m;
		int g[m], k[501] = {0};
		for(int i=0; i<m; i++){
			cin >> g[i];
			if(k[g[i]] == 0){
				a[g[i]] = 1;
				b[g[i]] += 1;
				k[g[i]] = 1;
			}
		}
	}
	for(int i=0; i<=500; i++){
		if(a[i]==1){
			sum+=i;
		}
	}
	for(int i=0; i<=500; i++){
		if(b[i] == s){
			a[i] = 0;
		}
	}
	for(int i=0; i<=500; i++){
		if(a[i] == 1){
			sum2 = sum2+i;
		}
	}
	cout << sum2-sum<< endl;
    return 0;
}
