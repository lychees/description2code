
#include<bits/stdc++.h>
using namespace std;
#define MOD	1000000007
#define ll	long long int
ll num[400005],den[400005];
ll pow_mod(ll a,ll b)
{
ll res=1;
while(b)
{
if(b&1)
res=(res*a)%MOD;
a=(a*a)%MOD;
b>>=1;
}
return res;
}
ll ncr(ll n, ll r)
{
return (((num[n]*den[n-r])%MOD)*den[r])%MOD;
}
ll inverse(ll x)
{
return pow_mod(x, MOD-2);;
}
void pre_cal()
{
num[0]=1;
for(int i=1;i<=400000;++i)
num[i]=(i*num[i-1])%MOD;
den[400000]=inverse(num[400000]);
for (int i=400000;i>0;--i)
den[i-1]=(i*den[i])%MOD;
}
int main()
{
    int a[100005];
pre_cal();
int t,var,i;
scanf("%d",&t);
while(t--)
{
ll n,k;
ll sum=0;
scanf("%lld %lld",&n,&k);
for(i=0;i<n;i++)
{
    scanf("%d",&var);
    sum=sum+var;
}
ll ans=(ncr(k-sum+n-1,k-sum))%MOD;

printf("%lld\n",ans);
}
return 0;
}
