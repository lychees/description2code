#include<iostream>
using namespace std;
int main()
{
    int i,j,n,temp,k=0;
    cin>>n;
    int arr[n],brr[n];
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }

    for(i=0;i<n;i++)
    {
        cin>>brr[i];
    }

    for(i=0;i<n;i++)
    {
    label:if(arr[i]!=brr[i])
        {
            temp=arr[i];
            for(j=i+1;j<n;j++)
            {
                arr[j-1]=arr[j];
            }
            arr[j-1]=temp;
            k++;
            if(arr[i]!=brr[i])
            {
                goto label;
            }
        }
     k++;
    }
    cout<<k;


    return 0;
}
