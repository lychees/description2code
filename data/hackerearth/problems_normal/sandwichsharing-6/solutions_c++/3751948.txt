#include <iostream>
using namespace std;

int main()
{
    long int p,i,newn,count,j;
    cin>>count;
    if(count==0) return 0;
    for(j=0;j<count;j++)
    {
		cin>>newn;
		if(newn<3) return 0;
		cin>>p;
		
		while(p)
		{
			if(newn==3) break;
			else
			{
				newn = (++newn/2)+1;
				p--;
			}
		}
		cout<<newn<<endl;
    }
    return 0;
}
