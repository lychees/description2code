#include<iostream>
#include<cstdio>
#include<map>
#include<cstring>
using namespace std;
int main()
{
    int test_cases,i,k;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        // your code here
        char scopy[101]={' '};
        k=0;
        for(i=0;i<strlen(str);i++){
        	if(isalpha(str[i]) || str[i]=='*'){
        		scopy[k++]=str[i];
        	}
        }
        
        for(i=0;i<strlen(scopy);i++){
        	if(scopy[i]=='*'){
        		break;
        	}
        }
        cout<<scopy[strlen(scopy)-i-1]<<endl;
       
        test_cases--;
    }
    return 0;
}
