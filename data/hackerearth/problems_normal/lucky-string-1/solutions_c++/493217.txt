#include<iostream>
#include<math.h>
using namespace std;
string soln[]={"Hacker","Earth"};
int main()
{
	
	int t;
	cin>>t;
	long long int chec[100];
	chec[0]=0;
	long long x=1;
	//cout<<(x<<50)<<endl;
	for(long long int i=1;i<=50;i++)
	{
		chec[i]=chec[i-1]+i*(x<<i);
	//	cout<<"i="<<i<<" check[i]="<<chec[i]<<endl;
	}
	long long int n;
	while(t--)
	{
		cin>>n;
		
		int i=1;
		for(;i<=50;i++)
		{
			if(n<=chec[i])
			{
				break;
			}
		}
		n=n-chec[i-1];
		long long int dig=ceil(n/(double)i)-1;//cout<<dig<<endl;
		int pos=i-(n%i);
		if(pos==i)
		{
			pos=0;
		}
		int co=-1,ans;
		while(co!=pos)
		{
			ans=dig%2;
			co++;
			dig/=2;
		}
		cout<<soln[ans]<<endl;
	}
	
}
