#include <iostream>
using namespace std;

int main()
{
 int t;
 cin>>t;
 while(t--)
 {
 	int n,i,j;
 	cin>>n;
 long long	int a[n];
 	for(i=0;i<n;i++)
 	cin>>a[i];
 	int count=0;
 	for(i=0;i<n;i++)
 	{
 		for(j=i+1;j<n;)
 		{
 			if(a[i]>a[j])
 			j++;
 			else
 			break;
 		}
 		if(j==n)
 		count++;
 	}
 	cout<<count<<endl;
 }
    return 0;
}
