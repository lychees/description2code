#include<iostream>
#include<cstdio>

#define ld long long int
#define MOD 1000000007

using namespace std;

ld sum[1000010];
ld arr[1000010];

int main()
{
    ld N,K,i;
    cin>>N>>K;
    for(i=1;i<=N;i++){cin>>arr[i];sum[i]=sum[i-1]+arr[i];}	
 	ld sum1=0,sum2=0;
 	for(i=1;i<=K;i++){
 		sum1=(sum1+(((i*i)%MOD)*arr[i])%MOD)%MOD;
 	 	sum2=(sum2+((2*i-1)*arr[i])%MOD)%MOD;
	}
	cout<<sum1<<" ";
	for(i=2;i<=(N-K+1);i++){
		sum1=(((sum1-sum2+MOD)%MOD)+(((K*K)%MOD)*arr[i+K-1])%MOD)%MOD;
		sum2=(((sum2-((arr[i-1]+2*sum[i+K-2]-2*sum[i-1])%MOD)+MOD)%MOD)+((2*K-1)*arr[i+K-1])%MOD)%MOD;
		cout<<sum1<<" ";
	}
}
