from math import *


def palindrome(s):
    return s[::-1] == s


def f(N):
    seen = {}
    sqrtN = int(sqrt(N))
    for i in xrange(1, sqrtN):
        s = i * i
        for j in xrange(i + 1, sqrtN):
            s += j * j
            if s > N:
                break
            if palindrome(str(s)):
                seen[s] = True
    return sum(iter(seen))


a, b = map(int, raw_input().split())
print f(b)-f(a-1)
