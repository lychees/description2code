#include<bits/stdc++.h>

#define MOD 1000000007
#define vi vector<int>
#define pb push_back
#define X first
#define Y second
typedef long long ll;
typedef pair<int,int> p;
int main()
{
    ios_base::sync_with_stdio(0);
    int t,i,j;
    cin>>t;
    int ans=0;
    while(t--)
    {
        string in;
        cin>>in;
        stack<char> mystack;
        int len=in.size();
        int cur=0;
        while(cur<len)
        {
            if(!mystack.empty() && in[cur]==mystack.top())
            {
                mystack.pop();
                //cout<<"Popping element from stack\n";
            }
            else
            {
                //cout<<"Pushing "<<in[cur]<<" in stack\n";
                mystack.push(in[cur]);
            }
            cur++;
        }
        if(mystack.empty())
            ans++;
    }
    cout<<ans;
}
