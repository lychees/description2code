#include <iostream>
#include<algorithm>
using namespace std;
bool isAP(int a,int b,int c)
{
	if(b==(a+c)/2&&(a+c)%2==0)
	return true;
	else
	return false;
}
int main()
{
    int n,count;
    cin>>n;
    count=0;
    int* a=new int[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
   
    for(int i=0;i<n-2;i++)
    for(int j=i+1;j<n-1;j++)
    for(int k=j+1;k<n;k++)
    {
    	if(isAP(a[i],a[j],a[k]))
    	count++;
    }
   cout<<count<<endl;
}
