#include <iostream>
#include <string.h>
using namespace std;

int a[1000000],MAX=1000000;
unsigned long long sum[1000000];
void mark(int n){
    int i=n*2,t;
    while(i <= MAX){
        a[i] = 1;
        i += n;
    }
}

void sieveOfEratosthenes(){
    memset(a,0,sizeof(a));
    a[0]=1;
    a[1]=1;
    for(int i=2;i<=MAX;++i){
        if(a[i] == 0){ // prime number
            mark(i);
        }
    }
    memset(sum,0,sizeof(sum));
    sum[0]=0;
    sum[1]=0;
    for(int i=2;i<=MAX;++i){
    	if(a[i] == 0){
    		sum[i] = sum[i-1] + i;
    	}else{
    		sum[i] = sum[i-1];
    	}
    }
}

int main(){
	int t,x,y;
	sieveOfEratosthenes();
	cin>>t;
	while(t--){
		cin>>x>>y;
		cout<<sum[y]-sum[x-1]<< endl;
	}
    return 0;
}
