#include <iostream>
using namespace std;
#include<cstring>
int main()
{
    int n,i;
    string s;
    cin >> n;
    while(n--)
    {
    	int ctr=0,i;
    	cin >> s;
    	for(i=0;s[i]!='\0';i++)
    	{
    		if(s[i]==s[i+1] && s[i+1]!='\0')
    			ctr++;
    	}
    	cout << ctr << "\n";
    }
    return 0;
}
