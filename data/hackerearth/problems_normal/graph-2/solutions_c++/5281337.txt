#include<bits/stdc++.h>
using namespace std;


int main()
{
    long t;
    cin>>t;
    while(t--)
    {
        long mat[20][20];
        long n, w;
        cin>>n>>w;
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++)
                cin>>mat[i][j];
        long mncost[70000], ans=1;
        for(int i=1; i<(1<<n); i++)
        {
            if(!(i&(i-1)))
            {
                mncost[i]=0;
                continue;
            }
            mncost[i]=INT_MAX;
            long cnt=0;
            for(int j=0; j<n; j++)
            {
                if((1<<j)&i)
                {
                    cnt++;
                    long mcost=INT_MAX;
                    for(int k=0; k<n; k++)
                        if((j!=k)&&((1<<k)&i))
                            mcost=min(mcost, mat[j][k]);
                    mncost[i]=min(mncost[i], mcost+mncost[i^(1<<j)]);
                }
            }
            if(mncost[i] <= w)
                ans=max(ans,cnt);
        }
        cout<<ans<<endl;
    }
    return 0;
}
