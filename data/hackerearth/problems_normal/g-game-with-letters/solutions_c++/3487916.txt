#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t!=0)
   	{
   		int n;
   		string s;
   		cin>>n>>s;
   		vector<long> ch(26);
   		for(long l=0;l<s.size();l++)
   				ch[s[l]-97]++;
   		vector<string> str(n);
   		long long score = -1;
   		int ind = -1;
   		int size = -1;
   		for(int i=0;i<n;i++)
   			cin>>str[i];
   		for(int i=0;i<n;i++)
   		{
   			long long in_sc = 0;
   			for(int j=0;j<str[i].size();j++)
   			{
   					in_sc += ch[(str[i])[j]-97];  
   			}
   			if(in_sc >= score)
   			{
   				if(in_sc == score)
				{
					if(str[i].size() <= size)
					{
							if(str[i].size()==size)
							{
									if(i < ind)
									{
										score = in_sc;
										ind = i;
										size = str[i].size();
									}			
							}
							else
								{
									score = in_sc;
									ind = i;
									size = str[i].size();
								}	
					}
					
				}
				else
				{
					score = in_sc;
					ind = i;
					size = str[i].size();
				}
   			}
   		}
   		cout<<ind+1<<endl;
   		t--;
   	}
   
    return 0;
}
