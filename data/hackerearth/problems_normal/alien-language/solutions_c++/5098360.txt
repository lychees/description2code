#include <bits/stdc++.h>
using namespace std;

//int has[10000];
int main()
{
    std::ios::sync_with_stdio(0);
    int t;
    cin>>t;
    string q1,q2;
    while(t--)
    {
    	cin>>q1>>q2;
    	//int n1=q1.length();
    	//int n2=q2.length();
    	set<char>s1;
    	bool ok=false;
    	for(auto i=0;i<q1.length();i++)
    	{
    		s1.insert(q1[i]);
    	}
    	for(auto i=0;i<q2.length();i++)
    	{
    		if(s1.find(q2[i])!=s1.end())
    		{
    			ok=true;
    			break;
    		}
    	}
    	if(ok)
    	cout<< "YES"<<endl;
    	else
    	cout<< "NO"<<endl;
    }
    return 0;
}
