#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int N;
    int arr[1001];
    cin>>N;
    for(int i=0;i<N;i++)
        cin>>arr[i];
    sort(arr,arr+N);
    int odd=0,sOdd=0,sEven=0;
    for(int i=0;i<N;i++)
    {
        if(arr[i]&1)
        {
            arr[odd++]=arr[i];
            sOdd+=arr[i];
        }
        else
        {
            cout<<arr[i]<<" ";
            sEven+=arr[i];
        }
    }
    cout<<sEven<<" ";
    for(int i=0;i<odd;i++)
        cout<<arr[i]<<" ";
    cout<<sOdd;
    return 0;
}
