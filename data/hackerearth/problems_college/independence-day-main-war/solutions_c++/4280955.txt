#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	int t, n, i;
	float a, b, x, y;
	scanf("%d", &t);
	
	while(t--)
	{
		x=0;
		y=0;
		
		scanf("%d", &n);
		
		for(i=0; i<n; i++)
		{
			cin >> a >> b;
			x+=a;
			y+=b;
		}
		
		x/=n;
		y/=n;
		
		printf("(%.2f , %.2f)\n", x, y);
	}
	
	return 0;
}
