using namespace std;
ll arr[100005];
ll l[100005];
ll r[100005];
ll c[100005];
ll temp[100005];
ll z[100005];

int main()
{
 //   freopen("input.txt","r",stdin);
//    freopen("output.txt","w",stdout);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,m;
		cin>>n>>m;
		
		for(int i=1;i<=n+1;i++)
		arr[i]=0;
		
		for(int i=1;i<=m;i++)
		{
			cin>>l[i]>>r[i]>>c[i];
			temp[i]=0;
		}
		
		
		ll q;
		cin>>q;
		while(q--)
		{
			ll x,y;
			cin>>x>>y;
			temp[x]++;
			temp[y+1]--;
		}
		
		
		ll count=0;
		for(int i=1;i<=m;i++)
		{
			count=count+temp[i];
			z[i]=count;
		//	temp[i]=0;
		}
		
	/*	for(int i=1;i<=m;i++)
		cout<<z[i]<<endl;*/
		
		for(int i=1;i<=n;i++)
		temp[i]=0;
		
		for(int i=1;i<=m;i++)
		{
			temp[l[i]]+=z[i]*c[i];
			temp[r[i]+1]-=z[i]*c[i];
		}
		
	/*	for(int i=1;i<=m;i++)
		cout<<temp[i]<<endl;
		*/
		count=0;
		for(int i=1;i<=n;i++)
		{
			count+=temp[i];
			arr[i]=count;
			cout<<arr[i]<<" ";
		}
		cout<<endl;
	}
    
    return 0;
}
