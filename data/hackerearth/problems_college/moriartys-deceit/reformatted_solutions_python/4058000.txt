from functools import partial


def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


def coprime(k, b):
    if gcd(k, b) == 1:
        return True
    return False


for t in range(int(raw_input())):
    n, x = map(int, raw_input().strip().split(' '))
    a = filter(partial(coprime, x), map(int, raw_input().strip().split(' ')))

    if len(a) == 0:
        print -1
    else:
        print ' '.join(map(str, a))
