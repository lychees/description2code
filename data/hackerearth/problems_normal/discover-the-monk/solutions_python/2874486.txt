def isHere(arr,data):
	l = 0
	r = len(arr)-1
 	while l<=r:
 		mid = (l+r)//2
 		if(arr[mid]==data):
 			return True
 		elif data>arr[mid]:
 			l = mid+1
 		else:
 			r = mid-1
 	return False

n,q = [int(i) for i in raw_input().split()]
arr = [int(i) for i in raw_input().split()]
arr.sort()
while(q>0):
	data = int(raw_input())
	if isHere(arr,data):
		print 'YES'
	else:
		print'NO'
	q-=1
