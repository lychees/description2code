def memoize(f):
    memo = dict()

    def wrapper(n):
        if memo.get(n) is None:
            memo[n] = f(n)
        return memo[n]
    return wrapper


@memoize
def factorial(n):
    if n == 0:
        return 1
    else:
        return n*factorial(n-1)


def combination(n, r):
    return factorial(n)/(factorial(r) * factorial(n-r))


testcases = int(raw_input())
for testcase in range(testcases):
    (girls, boys) = map(int, raw_input().split())
    if girls+1 < boys:
        print "-1"
    else:
        print(factorial(girls) * combination(girls+1, boys)
              * factorial(boys)) % 1000000007
