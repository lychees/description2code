#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n1,n2,k;
		cin>>n1>>n2>>k;
		int i;
		int j;
		int maxi=0;
		for(i=n1-k;i<=n1+k;i++)
		{
			for(j=n2-k;j<=n2+k;j++)
			{
				if(abs(n1-i) + abs(n2-j)<=k)
				{
					int p = __gcd(i,j);
					maxi = max(maxi,p);
				}
			}
			
		}
		cout<<maxi<<"\n";
	}
}
