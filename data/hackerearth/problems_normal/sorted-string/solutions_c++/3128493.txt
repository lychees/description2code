#include<iostream>

using namespace std;

struct str
{
	int a;
	char c;
};

void merge(struct str a[],int l, int p,int h)
{
	int j,i,t,x,y,k;	
	x = l;
	y = p+1;
	k = 0;
	struct str b[h-l+1];
	for(i=l;i<=h;i++)
	{
		if(x>p)
		{
			b[k].a = a[y].a;
			b[k].c = a[y].c;
			k++;
			y++;
		}
		else if(y>h)
		{
			b[k].a = a[x].a;
			b[k].c = a[x].c;
			k++;
			x++;
		}
		else if(a[x].a>a[y].a)
		{
			b[k].a = a[x].a;
			b[k].c = a[x].c;
			x++;
			k++;
		}
		else
		{
			b[k].a = a[y].a;
			b[k].c = a[y].c;
			
			y++;
			k++;
		}
	}
	for(i=0;i<k;i++)
	{
		a[l].a = b[i].a;
		a[l].c = b[i].c;
		l++;
	}
	return;
}

void sort(struct str a[],int l,int h)
{
	if(l<h)
	{
		int p = (l+h)/2;
		sort(a,l,p);
		sort(a,p+1,h);
		merge(a,l,p,h);
	}
	return;
}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		struct str x[26];
		char c[26];
		cin>>s;
		int i,l = s.size();
		for(i=0;i<26;i++)
		{
			x[i].a = 0;
			x[i].c = i + 'a';
		}
		for(i=0;i<l;i++)
		{
			x[s[i]-'a'].a++;
		}
		sort(x,0,25);
		
		for(i=25;i>=0;i--)
		{
			if(x[i].a!=0)
			{
				for(int j=0;j<x[i].a;j++)
				{
					cout<<x[i].c;		
				}
			}
		}
		cout<<endl;
	}
	return 0;
}
