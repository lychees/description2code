#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    int i;
    for(i=0;i<n;i++)
    {
    	scanf("%d",&a[i]);
    }
    int l,b,area=0;
    cin>>l>>b;
    area=l*b;
    long int amount=0;
    for(i=0;i<n;i++)
    {
    	if(a[i]>=1&&a[i]<=10)
    	{
    		amount+=floor(area*a[i]*a[i]*0.75);
    	}
    	else if(a[i]>=11&&a[i]<=20)
    	{
    		int j=a[i],sum=0;
    		sum=(a[i]%10)+(a[i]/10);
    		amount+=floor(area*a[i]*sum*0.7);
    	}
    	else
    	{
    		int k;
    		k=(a[i]%10)+(a[i]/10);
    		amount+=floor(area*a[i]*k*0.8);
    	}
    }
    printf("%ld",amount);
    return 0;
}
