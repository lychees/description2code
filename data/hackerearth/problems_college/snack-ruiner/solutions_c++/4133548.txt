#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   	int n;
   	cin>>n;
   float a[n+1],b[n+1];
   for(int i = 1;i<=n;i++){
  
   	cin>>a[i];
   	
   }
   for(int j=1 ;j<=n;j++)cin>>b[j];
   float t = 0;
   float m = 0;
   int  k = -2;
   for(int i = 1;i<=n;i++){
   	t = t + a[i]*b[i];
   if(a[i]*b[i] > m){
   	m = a[i]*b[i];
   	k = i;
   }
   }
   printf("%d %0.3f\n",k-1,m/t);
   
   }
    return 0;
}
