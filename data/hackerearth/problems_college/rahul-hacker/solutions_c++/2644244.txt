long int rev(long int c)
{   
	long int i=c%10,j,tot=0;
	while(i==0)	
	{
		c=c/10;	
		i=c%10;
	}
	while(c)
	{
		j=c%10;
		c=c/10;
		tot=((tot*10)+j); 
	}
	return tot;
}
