#include <bits/stdc++.h>

#define INF 0x3f3f3f3f
#define NINF -0x3f3f3f3f

using namespace std;

typedef pair<int,int> pii;

int arr[105];

int main (int argc, char const* argv[])
{
	int n,k;
	scanf("%d %d",&n,&k);
	
	for (int i = 0; i < n; i += 1)
	{
		scanf("%d",&arr[i]);
	}
	
	sort(arr,arr+n,greater<int>());
	
	int result = 0;
	for (int i = 0; i < n; i += 1)
	{
		int val = (i/k)+1;
		result += arr[i]*val;
	}
	
	printf("%d\n",result);
	
	return 0;
}
