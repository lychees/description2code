#include <bits/stdc++.h>
using namespace std;
#define ll long long
int n;
int A[5005],dp[5005];

int main()
{
	cin>>n;
    for(int i=0;i<n;i++){
    	cin>>A[i];
    	dp[i]=1;
    }
    
    for(int i=1;i<n;i++){
    	for(int j=0;j<i;j++){
    		if( (abs(A[i]) > abs(A[j])) ){
    			if( A[i]>0 and A[j]<0){
    				dp[i]=max(dp[i],dp[j]+1);
    			}
    			if( A[i]<0 and A[j]>0){
    				dp[i]=max(dp[i],dp[j]+1);
    			}
    		}
    	}
    }
    
    int ans=0;
    for(int i=0;i<n;i++){
    	ans=max(ans,dp[i]);
    }
    
    cout<<ans;
    
    return 0;
}
