#include <iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    vector<int> a(n);
    for(int i=0;i<n;i++)
    	cin>>a[i];
    sort(a.begin(),a.end());
    for(int i=0;i<n;i++)
    	cout<<a[i]<<endl;
    return 0;
}
