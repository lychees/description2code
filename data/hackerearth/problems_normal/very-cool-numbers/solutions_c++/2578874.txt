#include <bits/stdc++.h>

using namespace std;

int get_coolness(int n)
{
	int b = ceil(log2(n));
	int a[b];
	int i = 0;
	while (n != 0)
	{
		a[i++] = n % 2;
		n = n / 2;
	}
	int c = 0;
	for (int i = 0; i < b; i++)
	{
		if (i >= 2)
		{
			if (a[i - 2] == 1 && a[i - 1] == 0 && a[i] == 1)
				c++;
		}
	}
	return c;
}

int main()
{
	int t, r, k;
	cin >> t;
	while (t--)
	{
		cin >> r >> k;
		int c = 0;
		for (int i = 1; i <= r; ++i)
		{
			int cool = get_coolness(i);
//			cout << "coolness of i is :- " <<cool << endl;
			if (cool >= k)
				c++;
		}
		cout << c << endl;
	}
	return 0;
}
