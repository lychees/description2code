#include <bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin>>T;
    for(int t = 1;t<=T;t++){
    	int n;
    	cin>>n;
    	vector<long long> A;
    	for(int i=0;i<n;i++){
    		long long s;
    		cin>>s;
    		A.push_back(s);
    	}
    	
    	for(int i=2;i<n;i++){
    		long long p = 0;
    		for(int j =0;j<i-1;j++)
    		p = max(p , A[j]);
    		
    		A[i]+=p;
    	}
    	long long ans = 0;
    	for(int i=0;i<n;i++)
    	ans = max(ans, A[i]);
    	cout<<"Case "<<t<<": "<<ans<<endl;
    }
    return 0;
}
