#include <iostream>
using namespace std;

int main()
{
    long long int n,k;
    cin>>n>>k;
    long long int a[n],i=0;
    for(i=0;i<n;++i)
    {
    	cin>>a[i];
    }
    long long int x=0;
   	for(i=0;i<n;++i)
   	{
   		long long int large=a[i];
   		long long int pos=i;
   		for(int j=i+1;j<n;++j)
   		{
   			if(a[j]>large)
   			{
   				large=a[j];
   				pos=j;
   			}
   		}
   		if(pos==i)
   		continue;
   		else {++x;
   		long long int temp =a[pos];
   		a[pos]=a[i];
   		a[i]=temp;}
   		if(x==k)
   		break;
   	}
    for(i=0;i<n;++i)
    cout<<a[i]<<" ";
    return 0;
}
