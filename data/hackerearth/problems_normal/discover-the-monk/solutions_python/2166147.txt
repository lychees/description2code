def bSearch(listi,low,high,x):
	mid=(low+high+1)/2
	while(low<high):
		if listi[mid]==x:
			return True
		elif listi[mid]<x:
			return bSearch(listi, mid+1, high, x)
		else:
			return bSearch(listi, low, mid-1, x)
	if low==high:
		return listi[low]==x
		
inp=[int(i) for i in raw_input().split()]

q=inp[1]
listi=[int(i) for i in raw_input().split()]
listi.sort()
for i in range(q):
	t=input()
	if bSearch(listi,0,len(listi)-1,t):
		print "YES"
	else:
