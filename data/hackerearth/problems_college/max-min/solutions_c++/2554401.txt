#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n,x;
	cin>>n;
	int k;
	cin>>k;
	std::vector<int> v;
	for(int i=0;i<n;i++)
	{
		cin>>x;
		v.push_back(x);
	}
	std:: sort(v.begin(),v.begin()+n);
	int dif,min;
	for(int i=0;i<n && i+k<n ; i++)
	{
		dif=v[i+k-1]-v[i];
		if(i==0)
		{
			min=dif;
		}
		if(min>dif)
		{
			min=dif;
		}
	}
	cout<<min;
    return 0;
}
