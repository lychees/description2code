#include <iostream>
using namespace std;

int main()
{
    string s;
    int t;
    cin>>t;
    while(t--)
    {
    	
    	cin>>s;
    	for(int i=0;i<=s.length();i++)
    	{
    		
    		if(s[i]=='2')cout<<"cde";
    		if(s[i]=='6')cout<<"b3";
    		if(s[i]=='7')cout<<"acf";
    		if(s[i]=='9')cout<<"c6a";
    	}
    	cout<<"\n";
    }
    return 0;
}
