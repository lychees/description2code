#include<bits/stdc++.h>
using namespace std;
int power(int a,int b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	int temp=power(a,b/2);
	temp*=temp;
	if(b%2)
		temp*=a;
	return temp;
}
int main()
{
	int n,t,ans,temp;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		scanf("%d",&n);
		temp=n;
		if((n)&(n-1))
		{
			while(n!=1)
			{
				ans++;
				n/=2;
			}
			ans=power(2,ans);
			ans=2*(temp-ans)+1;
		}
		else
			ans=1;
		printf("%d\n",ans);
	}
	return 0;
}
