#include <iostream>
#include <map>
#include <cstring>
#include <cstdio>

using namespace std;

int main()
{
    int t,i;
    string s;
    scanf("%d",&t);
    map<string,int> a;
    
    map<string,int>::iterator it;
    for(i=0;i<t;i++)
   	{
   		cin>>s;
   		a[s]++;
   	}
   	
   	for(it=a.begin();it!=a.end();it++)
   	cout<<it->first<<" "<<it->second<<endl;
    return 0;
}
