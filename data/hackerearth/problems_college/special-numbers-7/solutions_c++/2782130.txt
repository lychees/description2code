#include <bits/stdc++.h>

using namespace std;

int main()
{
    int seive[111111],i,j;
    vector <int> list;
    memset(seive,0,sizeof(int)*111111);
    for(i=2;i<111111;i++)
    {
    	if(seive[i]!=0)
    		continue;
    	for(j=2*i;j<111111;j+=i)
    		seive[j]++;
    }
    for(i=2;i<111111;i++)
    	if(seive[i]>=3)
    		list.push_back(i);
    sort(list.begin(),list.end());
    //printf("%lld",list.size());
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	printf("%d\n",list[n-1]);
    }
    return 0;
}
