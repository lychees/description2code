#include <iostream>
#include<math.h>
using namespace std;
long int minimum(long int a,long int b)
{
	if(a<b)
		return a;
	return b;
}
long int maximum(long int a,long int b)
{
	if(a>b)
		return a;
	return b;
}
int compute_count(long int z,long int x,long int y)
{
	long int a=minimum(x,y),cnt=0;
	for(long int i=1;(i<=sqrt(z)&&i<=a);i++)
	{
		if(z%i==0 )
		{
			if(i<=x && (z/i)<=y)
				cnt++;
			if(i<=y && (z/i)<=x && (z/i)!=i)
				cnt++;
				
		}
	}
	return cnt;
	
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	long int z,x,y;
    	cin>>z>>x>>y;
    	cout<<compute_count(z,x,y)<<endl;
    }
    return 0;
}
