#include <cstdio>
#include <algorithm>

using namespace std;

long long find_max(int arr[], int n)
{
  long long incl = arr[0];
  long long excl = 0;
  long long excl_new;

  for (int i = 1; i < n; i++)
  {
     excl_new = (incl > excl)? incl: excl;

     incl = excl + arr[i];
     excl = excl_new;
  }

   return ((incl > excl)? incl : excl);
}

int main()
{
	int t;
	scanf("%d", &t);

	int arr[10001];

	while(t--)
	{
		int n;
		scanf("%d", &n);
		for(int i=0; i<n; i++)
		{
			scanf("%d", &arr[i]);
		}
		long long ans = find_max(arr, n);
		printf("%lld\n", ans);
	}
	return 0;
}
