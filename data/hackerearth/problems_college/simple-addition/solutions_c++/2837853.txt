#include<cstdio>
using namespace std;
int check(int a,int b)
{
	int ns=0;
	while(!(a==1 && b==1))
	{
		if(a==b) 
		{
			ns=10000000; break;
		}
		if(a<b) b=b-a;
		else a=a-b;
		ns++;
	}
	
	return ns;
}
int main()
{
	int ntc; scanf("%d",&ntc);
	while(ntc--)
	{
		int L; scanf("%d",&L);
		int ans=100000000;
		int ta;
		if(L==1)
		{
			printf("0\n"); continue;
		}
		for(int i=1;i<L;i++)
		{
			int ta=check(i,L-i);
			if(ta<ans) ans=ta;
		}
		
		printf("%d\n",ans+1);
	}
}
