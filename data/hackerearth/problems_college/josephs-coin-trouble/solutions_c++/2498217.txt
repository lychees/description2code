#include <iostream>
#include<cstring>
using namespace std;
int main()
{
	int N,x,y;
	cin>>N;
    int arr[4];
    int c=0;
    for(int i=0;i<4;i++)
    cin>>arr[i];
    for(int i=0;i<4&&arr[i]!=0;i++)
       c++;
       int ar[c];
       for(int i=0;i<c;i++)
       ar[i]=arr[i];
    
 
   int table[N+1][c];
    
    // Fill the enteries for 0 value case (n = 0)
    for (int i=0; i<c; i++)
        table[0][i] = 1;
 
    // Fill rest of the table enteries in bottom up manner  
    for (int i = 1; i < N+1; i++)
    {
        for (int j = 0; j < c; j++)
        {
            // Count of solutions including S[j]
            x = (i-ar[j] >= 0)? table[i - ar[j]][j]: 0;
 
            // Count of solutions excluding S[j]
            y = (j >= 1)? table[i][j-1]: 0;
 
            // total count
            table[i][j] = x + y;
        }
    }
    cout<<table[N][c-1];
}
