#include <iostream>
#include <queue>
using namespace std;

int main()
{
    long int i,n,k,index =0,min = 1000000;
    long int a[1000000];
    long int max = 1,nextMax = 1;
    cin>>n>>k;
    for(i = 0; i < n ; i++)
    {
      cin>>a[i];
      if(a[i] < min)
      min = a[i];
      else if(a[i] >= max)
      {
        int temp = max;
        max = a[i];
        nextMax = temp;
      }
    }
    if  (min > k)
    cout<<"NO";

    else if( (max+nextMax) < k )
    cout<<"NO";

    else
    {
    	cout<<"YES";
    }
}
