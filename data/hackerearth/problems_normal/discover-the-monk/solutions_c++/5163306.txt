#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

#define SORT(A) sort(A.begin(), A.end())

int main() {
	int n, q, temp;
	vector<int> inp;
	cin >> n >> q;
	while(n--) {
		cin >> temp;
		inp.push_back(temp);
	}
	SORT(inp);
	while(q--) {
		cin >> temp;
		(binary_search(inp.begin(), inp.end(), temp)) ? cout << "YES\n" : cout << "NO\n";
	}
	return 0;
}
