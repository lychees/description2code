#include<bits/stdc++.h>

using namespace std;

#define MOD 1000000007

int main() {
    int N, K;
    cin >> N >> K;
    
    long long combo[1001][1001];
	long long factorial[1001];
	long long dp[1001];

    factorial[0] = 1;
    for (int i = 1; i <= N; i++) {
        factorial[i] = (i*factorial[i-1]) % MOD;
    }
    for (int i = 0; i <= N; i++) {
        for (int j = 0; j <= i; j++) {
            if (j == 0 || j == i)
                combo[i][j] = 1;
            else
                combo[i][j] = (combo[i-1][j] + combo[i-1][j-1]) % MOD;
        }
    }
    dp[0] = 1;
    for (int i = 1; i <= N; i++) {
        for (int j = 0; j < i; j++) {
            long long ways = (combo[i-1][j] * factorial[j+1]) % MOD;
            ways = (ways * dp[i-j-1]) % MOD;
            dp[i] += ways;
            dp[i] %= MOD;
        }
    }
    
    long long ans = (dp[N] * combo[N][K]) % MOD;
    
    cout << ans << endl;
    
    return 0;
}
