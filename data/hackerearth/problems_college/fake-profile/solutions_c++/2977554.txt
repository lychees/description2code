inline void _IN(int&p)
{
    register int n=0,sign=1,a=GC();
    while((a<'0')or(a>'9'))
    {
        if(a=='-')
        {
            sign=-1;
            a=GC();
            break;
        }
        a=GC();
    }
    while((a>='0')and(a<='9'))
    {
        n=(n<<1)+(n<<3)+a-'0';
        a=GC();
    }
    p=n*sign;
}

}
