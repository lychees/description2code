#include <iostream>
#include<cstdio>
#define mod 1000000007
using namespace std;

int main()
{   ios::sync_with_stdio(0);
    long long int dp[100010];
    dp[1]=1; dp[2]=2;dp[3]=4;
    for(int i=4;i<100010;++i)
       dp[i]=(dp[i-1]+dp[i-2]+dp[i-3])%mod;
     int t;
     cin>>t;
     while(t--){
     	int n;
     	cin>>n;
     	printf("%d\n",dp[n]);
     }
    return 0;
}
