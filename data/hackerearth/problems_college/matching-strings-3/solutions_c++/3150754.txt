#include <iostream>
#include<string>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string m1,m2;
    	long int i=0,count=0;
    	cin>>m1>>m2;
    	while(m1[i] && m2[i])
    	{
    	if(m1[i]==m2[i] )
    	count++;
    	i++;
    	}
    	
    	
    	cout<<count<<endl;
    }
    return 0;
}
