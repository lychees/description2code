t=input()
for _ in range(t):
	l=map(int,raw_input().split())
	a=l[0]
	b=0
	if len(l)>1:
		b=l[1]
	carry=0
	cnt=0
	if not a or not b:
		print "No carry operation"
		continue
	while a or b:
		s=a%10+b%10+carry
		if s>=10: 
			carry=1
			cnt+=1
		else:
			carry=0
		a/=10
		b/=10
	if not cnt:
