#include <iostream>
using namespace std;

int ans[1000005]={0};

void init()
{
	for(int i=2; i<=1000000; i++){
		if(i==2 || i==3 || i==5 || i==7){
			ans[i]=1;
			continue;
		}
		if(i>=7 && ans[i-7]!=0){
			ans[i]=1+ans[i-7];
		}
		else if(i>=5 && ans[i-5]!=0){
			ans[i]=1+ans[i-5];
		}
		else if(i>=3 && ans[i-3]!=0){
			ans[i]=1+ans[i-3];
		}
		else if(i>=2){
			ans[i]=1+ans[i-2];
		}
	}
}

int main()
{
	init();
    int t;
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	if(ans[n]==0)
    		cout << "-1" << endl;
		else
    		cout << ans[n] << endl;
    }
    return 0;
}
