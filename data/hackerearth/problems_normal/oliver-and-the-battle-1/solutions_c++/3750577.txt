#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int a[1005][1005];
int vis[1005][1005];

int ct,n,m;
void dfs(int i, int j ){
	
	vis[i][j]=1;
	ct++;
	
	if(a[i-1][j]==1 && vis[i-1][j]==0 && (i-1)>=1 && i<=n && j>=1 && j<=m)
    	dfs(i-1,j);
	if(a[i+1][j]==1 && vis[i+1][j]==0 && i>=1 && (i+1)<=n && j>=1 && j<=m)
     	dfs(i+1,j);
	if(a[i][j-1]==1 && vis[i][j-1]==0 && (j-1)>=1 && j<=m && i>=1 && i<=n)
	 dfs(i,j-1);
	if(a[i][j+1]==1 && vis[i][j+1]==0 && j>=1 && (j+1)<=m && i>=1 && i<=n)
	 dfs(i,j+1);
	if(a[i+1][j+1]==1 && vis[i+1][j+1]==0 && j>=1 && (j+1)<=m && i>=1 && (i+1)<=n)
	 dfs(i+1,j+1);
	if(a[i-1][j-1]==1 && vis[i-1][j-1]==0 && (j-1)>=1 && (j)<=m && (i-1)>=1 && (i)<=n)
	 dfs(i-1,j-1);
	if(a[i+1][j-1]==1 && vis[i+1][j-1]==0 && (j-1)>=1 && (j)<=m && (i)>=1 && (i+1)<=n)
	 dfs(i+1,j-1);
	if(a[i-1][j+1]==1 && vis[i-1][j+1]==0 && (j)>=1 && (j+1)<=m && (i-1)>=1 && (i)<=n)
	 dfs(i-1,j+1);
	
}





int main()
{ 	int i,j;
    int t;
    cin>>t;
    while(t--)
    {  
    
    	cin>>n>>m;
    	
    	memset(vis,0,sizeof vis);
    		memset(a,0,sizeof a);
    	
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=m;j++)
    		cin>>a[i][j];
    	}
    	
    	
    	int conn=0,ma=0;
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=m;j++)
    		{
    			if(a[i][j]==1 && vis[i][j]==0)
    			{   ct=0;
    				dfs(i,j);
    				ma=max(ma,ct);
    			conn++;
    			
    			}
    		}
    	
    	}
  
    cout<< conn <<" "<<ma<<endl;
 
    	
    }
    return 0;
}
