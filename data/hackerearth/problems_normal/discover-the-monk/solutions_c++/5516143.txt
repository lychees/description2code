#include<iostream>
#include<algorithm>
using namespace std; 
  bool bsa(int arr[],int n,int x)
   {
    int l,h,mid;
    l=0;h=n-1;
     while(l<=h)
     {
     mid=(l+h)/2;
     if(arr[mid]==x)
      return true;
     else if(arr[mid]<x)
      l=mid+1;
     else
      h=mid-1;
     }  
   return false;
   }
   
int main()
{
 int n,q;
 cin>>n >>q;
 int *arr=new int[n];
 for(int i=0;i<n;i++)
   {
    cin>>arr[i];
   }
 sort(arr,arr+n);
 for(int i=0;i<q;i++)
 {
  int x;
  cin>>x;
  bsa(arr,n,x)?cout<<"YES"<<endl:cout<<"NO"<<endl;
 }
return 0;
}
