
def solution(string):
    n = len(string)
    comb = n*(n+1)/2
    if ('a' not in string) and ('z' not in string):
        return 0
    else:
        substrings = string.replace('z', 'a').split('a')
        for i in substrings:
            comb -= (len(i) * (len(i)+1)) / 2
        return comb

t = int( raw_input() )
for i in range( t ):
    print solution(raw_input())
