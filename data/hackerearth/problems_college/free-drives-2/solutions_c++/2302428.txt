#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
     long long int n,sum=0;
     cin>>n;
     if(n==1)
     cout<<"0"<<endl;
     else
     {
     for(int i=n-1;i>0;i--)
     {
      sum=sum+i;	
     }
     cout<<sum<<endl;
     }
    }
    return 0;
}
