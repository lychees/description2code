'''

name = raw_input('What is your name?\n')

for tc in range(input()):
    st = [0]*26
    s = raw_input()
    for i in s:
        st[ord(i)-ord('a')] += 1
    t = st[0]
    for i in st:
        if i != t:
            print 'No'
            break
    else:
        print 'Yes'
