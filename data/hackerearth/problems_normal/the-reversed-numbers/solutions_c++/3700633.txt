#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	
    	int aa=0,bb=0;
    	
    	for(int i=0,p=1;i<a.length();i++,p*=10)
    		aa+=p*(a[i]-'0');
    		
    	for(int i=0,p=1;i<b.length();i++,p*=10)
    		aa+=p*(b[i]-'0');
    		
    	aa+=bb;
    	
    	while(aa%10==0)
    		aa/=10;
    		
    	a="";
    	while(aa)
    	{
    		a+=(char)(aa%10+'0');
    		aa/=10;
    	}
    	cout<<a<<endl;
    }
    return 0;
}
