#include <bits/stdc++.h>
using namespace std;
 
long long int fun(vector<int> v, vector<int> indices, long long int max)
{
	long long int div=1;
	for( int i = 0; i<indices.size(); ++i)
	{
		div*=v[indices[i]];
	}
	long long int ans= max/div;
	return ans;
}
 
int main()
{
    int n,t ;
    cin>>n>>t;
	long long int o;
	vector<int> v;
	for(int i=0;i<n;i++){
		cin>>o;
		v.push_back(o);
	}
	while(t--){
		long long int A,B,resA=0,resB=0;
		cin>>A>>B;
		vector<int> indices;
		for(int b = 1; b < (1 << n); ++b)
		{
		     long long int x=1;
			 indices.clear();
		     for(int k = 0; k < n; ++k)
		     {
		          if(b & x)
		          {
		               indices.push_back(k);
		          }
		          x=x<<1;
		     }
		     long long int cardinality = fun(v,indices,A-1);
		     if(indices.size() % 2 == 1) resA += cardinality;
		     else resA -= cardinality;
		     cardinality = fun(v,indices,B);
		     if(indices.size() % 2 == 1) resB += cardinality;
		     else resB -= cardinality;
		}
		long long int g = resB - resA;
		cout <<g<< endl;
	}
    return 0;
}
