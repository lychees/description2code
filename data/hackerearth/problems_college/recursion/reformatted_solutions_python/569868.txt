import itertools


def main():
    t = int(raw_input())
    al = '123456789'
    for test in xrange(t):
        n = int(raw_input())
        pr = []
        perms = list(itertools.permutations(al[:n]))
        for tup in perms:
            tsum = ''
            for d in tup:
                tsum += d
            pr.append(int(tsum))
        pr.sort()
        print ' '.join(map(str, pr))


if __name__ == '__main__':
    main()
