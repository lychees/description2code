#include <iostream>
using namespace std;

int main()
{
   int t,N;
   long long int k,sum;
   cin>>t;
   while(t>0)
   {
   	sum=0;
   	cin>>N;
   	long long int a[N];
   	for(int i=0;i<N;i++)
   	{
   		cin>>a[i];
   	}
   	for(int i=0;i<N;i++)
   	{
   	sum=sum+a[i];
   	}
     k=sum/N;
   cout<<k<<"\n";
   t=t-1;
   }
   return 0;

}
