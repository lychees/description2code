#include<iostream>

using namespace std;

#define mod 1000000007

int arr[20];



int count(int x){
    int cc = 0;
    while(x != 0){
        x = x&(x-1);
        cc++;
    }
    return cc;
}

long long cal(long long n){
    long long ans = 1;
    int m = mod-2;
    while(m != 0){
        if(m%2 == 1){
            ans *= n;
            ans %= mod;
        }
        n = n*n;
        n %= mod;

        m /= 2;
    }
    return ans;
}

int main(){
    int t;
    cin >> t;
    for(int i=1;i<=t;i++){
        int n;
        cin >> n;
        for(int j=0;j<n;j++){
            cin >> arr[j];
        }
        long long ans = 0;
        for(int j=0;j<(1<<n);j++){
            long long ans1 = 1;
            long long ans2 = 0;
            if(count(j) >= 2){
                int kk = j;
                int cc = 0;
                while(kk != 0){
                    if(kk%2 == 1){
                        ans1 *= arr[cc];
                        ans1 %= mod;

                        ans2 += arr[cc];
                        ans2 %= mod;
                    }
                    kk /= 2;
                    cc++;
                }
            }
            long long temp = cal(ans2);
            temp = temp*ans1;
            ans = max(ans, temp%mod);
        }
        cout << "Case #" << i << ": " << ans << endl;
    }
}
