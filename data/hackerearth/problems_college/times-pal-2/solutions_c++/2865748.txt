#include <iostream>
using namespace std;
int palindrome(string a)
{
	int l=a.length(), i, f=0;
	for(i=0;i<l/2;i++)
	{
		if(a[i]!=a[l-i-1])
		{
			f=1;break;
		}
	}
	if(f==0)
		return 0;
	else
		return 1;
}
int main()
{
	string s, a, b;
	int t, l, i, c;
    cin>>t;
    while(t--)
    {
    	c=0;s="";
    	cin>>a>>b;
    	l=a.length()+1;
    	for(i=0;i<l;i++)
    	{
    		s=a.substr(0,i)+b+a.substr(i,a.length()-i);
    		if(palindrome(s)==0)
    			c++;
    		s="";
    	}
    	cout<<c<<endl;
    }
    return 0;
}
