#include <bits/stdc++.h>
using namespace std;

int gcd(int a, int b)
{
    while (b) {
        int r = a % b;
        a = b;
        b = r;
    }

    return a;
}

int lcm(int a, int b)
{
    return a * (b / gcd(a, b));
}

int main()
{
    int t;
    scanf("%d", &t);

    while (t--) {
        int x, y, a, b, lm, ans;
        scanf("%d %d %d %d", &x, &y, &a, &b);
        lm = lcm(x, y);
        ans = (b/lm) - (a/lm);
        if (a % lm == 0) ans++;
        printf("%d\n", ans);
    }

    return 0;
}
