#include <bits/stdc++.h>
using namespace std;
int solve(int val[],int wt[],int W,int n){
	int k[1001][1001] = {0};
	for(int i = 1;i<=n;i++){
		for(int w = 1;w<=W;w++){
			if(wt[i-1]<=w){
				k[i][w] = max(k[i-1][w-wt[i-1]]+val[i-1],k[i-1][w]);
			}
			else k[i][w] = k[i-1][w];
		}
	}
	return k[n][W];
}
int main()
{
    int test;
    scanf("%d",&test);
    while(test--){
    	int n,w;
    	scanf("%d %d",&n,&w);
    	int wt[n],val[n],i,j;
    	for(i=0;i<n;i++){
    		scanf("%d",&wt[i]);
    	}
    	for(i=0;i<n;i++){
    		scanf("%d",&val[i]);
    	}
    	int ans;
    	ans = solve(val,wt,w,n);
    	printf("%d\n",ans);
    }
    return 0;
}
