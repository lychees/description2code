#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>
using namespace std;

typedef long long ll;

ll t, n, st, et, wp;

class task {
	
	public:
	ll st, et;
	double wp;
	
};

task tasks[60];
double val[60];

int main()
{
	cin >> t;
	while( t-- ) {
		cin >> n;

		for( int i = 0; i < n; i++ )
			cin >> tasks[i].st >> tasks[i].et >> tasks[i].wp;
		
		sort( tasks, tasks+n, []( task i, task j) { return i.et < j.et; } );
		
		double ans = 0.0;
		
		for( int i = 0; i < n; i++ ) {
			val[i] = tasks[i].wp;
			for( int j = 0; j < i; j++ ) {
				if( tasks[j].et <= tasks[i].st )
					if( val[i] < val[j] + tasks[i].wp )
						val[i] = val[j] + tasks[i].wp;
			}
			if( ans < val[i] )
				ans = val[i];
		}
		
		cout << fixed << setprecision(2) << ans/100.0 << endl;
		
	}
	
	return 0;
}
