const li MOD = li(INF + 7);

li f[N];
int cnt[N];

inline void solve()
{
	f[0] = 1;
	for(int i = 1; i < N; i++)
	{
		f[i] = (f[i - 1] * i) % MOD;
	}

	forn1(i, n)
	{
		forn1(j, m)
		{
			cnt[(i ^ j)]++;
		}
	}

	li ans = 1LL;
	forn(i, N)
	{
		ans = (ans * f[cnt[i]]) % MOD;
	}

	cout << ans << endl;
    return;
}
