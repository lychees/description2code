#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
char keypad[10][6] = {
				{".,?!1"},
				{"abc2 "},
				{"def3 "},
				{"ghi4 "},
				{"jkl5 "},
				{"mno6 "},
				{"pqrs7"},
				{"tuv8 "},
				{"wxyz9"},
				{"_0   "}};
int main()
{
int T;
cin>>T;
do{
	string type;
	int movement = 0;
	int totsec = movement;
	int curpos = 0;
	cin>>type;
	for(int i = 0; i < type.length(); i++)
	{
		for (int j = 0; j < 10; j++)
			for (int k = 0; k < 5; k++)
				if (type[i] == keypad[j][k])
				{
					if( curpos == j)
					{
						movement = 0;
						curpos = j;
					}
					else
					{
						movement = 1;
						curpos = j;
					}
					totsec = totsec + movement + k + 1;
					movement = 0;
				}
	}
	cout<<totsec<<endl;
	T--;
}while(T>0);
    return 0;
}
