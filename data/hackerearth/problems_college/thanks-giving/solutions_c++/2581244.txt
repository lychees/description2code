#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    string data;
    cin >> data;
	int cnt[256] = {0};
	int len = data.length();
	for (int i = 0; i < len; i++) {
		cnt[int(data.at(i))]++;
	}
	int max = 0;
	char c = 'a';
    for (int i = 0; i < 256; i++) {
		if (cnt[i] > max) {
			max = cnt[i];
			c = char(i);
		} else if (cnt[i] == max) {
			c = char(min(i, (int)c));
		}
	}
	string p = "head rush", q = "cubing club";
	bool pb = false, qb = false;
	if (p.find(c) != -1) {
		pb = true;
	}
	if (q.find(c) != -1) {
		qb = true;
	}
	if (pb && qb) {
		cout << "Thanks both" << endl;
	} else if (pb) {
		cout << "Thanks " << p << endl;
	} else if (qb) {
		cout << "Thanks " << q << endl; 
	} else {
		cout << "none" << endl;
	}
    return 0;
}
