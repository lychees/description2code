#include <iostream>
#include <string>

using namespace std;

string getAns(string x, long long y, long long z)
{
	if (x[(y - 1) % x.length()] == x[(z - 1) % x.length()])
		return "Yes";
	return "No";
}

int main()
{
	string x;
	//int n;
	long long n, a, b;

	cin >> x;
	cin >> n;
	for (int i = 0; i < n; ++i)
	{
		cin >> a >> b;
		cout << getAns(x, a, b) << "\n";
	}
	return 0;
}
