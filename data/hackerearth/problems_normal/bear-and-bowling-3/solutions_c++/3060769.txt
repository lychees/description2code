#include <bits/stdc++.h>
using namespace std;

int n;

long long tab[1000007];

long long mod=1000000007;

long long mno[1000007];

long long wyn;

long long pot[1000007];

int main()
{
    scanf("%d", &n);
    pot[0]=1;
    for (int i=1; i<=n; i++)
    {
        scanf("%lld", &tab[i]);
        pot[i]=(pot[i-1]<<1)%mod;
    }

    mno[1]=1;
    mno[2]=4;
    wyn=(tab[1]*pot[n-1]+3*pot[n-2]*tab[2])%mod;

    for (int i=3; i<=n; i++)
    {
        mno[i]=((mno[i-1]-mno[i-3]-mno[i-3]+mod+mod)*2)%mod;
        wyn+=((mno[i]*tab[i])%mod)*pot[n-i];
        wyn%=mod;
        mno[i]+=mno[i-1];
        mno[i]%=mod;
    }
    printf("%lld\n", wyn%mod);
    return 0;
}
