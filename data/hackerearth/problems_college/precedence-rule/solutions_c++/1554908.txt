#include <iostream>
#include <string>
using namespace std;

int main() {
    
    string s; 
    long sum,i,t;
    
	// your code goes here
	cin >> t;
	while(t--)
	{
	    cin >> s;
	    sum = s[s.length()-1]-48;
	    for(i=s.length()-2;i>=0;i-=2)
	    {
	        if(s[i] =='+')
	        sum +=  s[i-1]-48;
	    else if(s[i] =='*')
	       sum *=  s[i-1]-48;
	    else 
	        sum -=  s[i-1]-48;
	   
	    
	}
	cout << sum << "\n";
	}
	
	return 0;
}
