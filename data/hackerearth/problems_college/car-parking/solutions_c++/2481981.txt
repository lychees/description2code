#include <iostream>
using namespace std;

int main()
{
    int n,k=0;
    cin>>n;
    string s;
    for(int i=0;i<n;i++)
    {
    	k=0;
    	cin>>s;
    	for(int j=0;j<s.length();j++)
    	{
    		if(s[j]!='D'&&s[j]!='B'&&s[j+1]!='B'&&s[j+2]!='B'&&s[j]!='S'&&s[j+1]!='S'&&s[j-1]!='S')
    		{
    		
    		//	cout<<j<<" ";
    			k++;
    			}
    		
    	}
    	cout<<k<<endl;
    }
    return 0;
}
