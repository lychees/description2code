#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    long long l=s.size(),q;
    cin>>q;
    for(int i=1;i<=q;i++)
    {
    	long long a,b;
    	cin>>a>>b;
    	a=a%l;
    	b=b%l;
    	
    	if(a==0)
    	a=l;
    	if(b==0)
    	b=l;
    	
    	if(s[a-1]==s[b-1])
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    	
    }
    return 0;
}
