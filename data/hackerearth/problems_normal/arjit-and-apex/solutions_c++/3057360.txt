#include <bits/stdc++.h>

using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
        int i,j,m,n,u,v;
        cin>>m>>n;
        vector< set<int> > ne(10002),old(10002);
        for(i=0;i<m;i++)
        {
            cin>>u>>v;
            ne[u].insert(v);
        }
        for(i=0;i<n;i++)
        {
            cin>>u>>v;
            old[u].insert(v);
        }
        int g,h,good=0,great=0;
        cin>>g>>h;
        set<int>::iterator it1,it2,it3;
        for(i=0;i<10001;i++)
        {
            it1=ne[i].begin();
            //it2=old[i].begin();
            good+=min(old[i].size(),ne[i].size());
            while(!old[i].empty() && it1!=ne[i].end())
            {
                it3=old[i].find(*it1);
                if(it3!=old[i].end())
                {
                    great++;
                    old[i].erase(it3);
                }
                it1++;
            }
        }
        if(good>=g && great>=h)
            cout<<"Great\n";
        else if(good>=g)
            cout<<"Good\n";
        else
            cout<<":(\n";
        //cout<<good<<" "<<great<<"\n";
    }
}
