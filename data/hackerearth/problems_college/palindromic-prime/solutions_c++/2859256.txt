#include <bits/stdc++.h>
using namespace std;

bool primes[123456];

int reverse(int n)
{
		int sum=0;
		while(n)
		{
			sum=sum*10+n%10;
			n/=10;
		}
		return sum;
}

int arr[124];

int sumofdivisors(int x)
{
   int sum=0;
   for(int i=1; i*i<=x; i++)
   {
		if(x%i==0)
		{
		   sum++;
		   if(i!=(x/i)) sum++;
		  // cout<<i<<" ";
		}
   }
   return sum;
}

int solve(int n)
{
	int ans=1;
	while(n)
	{
		if(n%10) ans*=(n%10);
		n/=10;
	}
	return ans;
}

int main()
{
    for(int i=2; i*i<=100000; i++) if(primes[i]==0) for(int j=i*i; j<=100000; j+=i) primes[j]=1;
    int counts=1;
    for(int i=2; i<=100000; i++) if(primes[i]==0 && reverse(i)==i) arr[counts++]=solve(i);
    
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n,m;
    	cin>>n>>m;
    	cout<<sumofdivisors(arr[n]*m)<<'\n';
    }
    return 0;
}
