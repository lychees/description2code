#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}

vector<long long int>v;
int main()
{
	long long int a,b;
	scanf("%lld %lld",&a,&b);
	int q;
	long long int c=gcd(a,b);
	long long int i,j;
	for(i=1;i<=sqrt(c);i++)
	{
		if(c%i==0)
		{
			if(i==(c/i))
			v.push_back(i);
			else
			{
				v.push_back(i);
				v.push_back(c/i);
			}
		}
	}
	sort(v.begin(),v.end());
	scanf("%d",&q);
	while(q--)
	{
		long long int l,r;
		scanf("%lld %lld",&l,&r);
		long long int x,y;
		x=lower_bound(v.begin(),v.end(),l)-v.begin();
		y=upper_bound(v.begin(),v.end(),r)-v.begin();  
		y--;
		if(y>=0&&y>=x)
		{
			printf("%lld\n",v[y]);
		}
		else
		printf("-1\n");
	}
	return 0;
}
