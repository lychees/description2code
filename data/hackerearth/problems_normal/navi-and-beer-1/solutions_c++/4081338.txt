#include <iostream>
using namespace std;

int main(){ 
	long long T;
   //taking input for number of test cases, T.
     cin>>T;
     for(int p=0;p<T;p++){
       long long N=0;
       long long K =0;
       long long M =0;
       long long amnt=0;
        //For every test case, taking input for N , K and M.
        cin>>N>>K>>M;
         for(long long i=1;i<=N;i++){
            amnt  += K*(i*i*i);
         }
         
         long long Ans =0;
        // cout<<M;
         //cout<<amnt;
         if(amnt>M){
         Ans = amnt-M;
         }else{
         	Ans =0;
         	
         }
        //let the answer be in variable **Ans**
        cout << Ans << endl; //print your answer for every test case.
     }
        return 0;
}
