#include <iostream>
#define abs(x) (x<0?-x:x)
inline void Winner(int D,int B){std::cout<<(D==B?"Tie\n":(D>B?"Dom\n":"Brian\n"));}
inline void High(int D,int B){std::cout<<(D>B?D:B);}
int main()
{
	int checkpoints,Dom=0,Brian=0,Dom_Previous,Brian_Previous,X;
	std::cin>>checkpoints>>Dom_Previous;
	for(int i=1;i<checkpoints;i++)
	{
		std::cin>>X;
		Dom=(Dom>abs((Dom_Previous-X))?Dom:abs((Dom_Previous-X)));
		Dom_Previous=X;
	}
	std::cin>>Brian_Previous;
	for(int i=1;i<checkpoints;i++)
	{
		std::cin>>X;
		Brian=(Brian>abs((Brian_Previous-X))?Brian:abs((Brian_Previous-X)));
		Brian_Previous=X;
	}
	Winner(Dom,Brian);
	High(Dom,Brian);
}
