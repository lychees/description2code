#include<bits/stdc++.h>

using namespace std;
const int inf=1e9+9;

int main() {
    int tt;
    cin>>tt;
    while(tt--) {
        int n1,n2,k;
        cin>>n1>>n2>>k;
        int maxVal=-inf;
        for(int i=max(1,n1-k);i<=n1+k;i++) {
            for(int j=max(1,n2-k);j<=n2+k;j++) {
                if(abs(n1-i)+abs(n2-j)<=k) {
                    maxVal=max(maxVal,__gcd(i,j));
                }
            }
        }
        cout<<maxVal<<endl;
    }
    return 0;
}
