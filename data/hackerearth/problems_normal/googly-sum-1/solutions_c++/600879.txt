#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define S(x) scanf("%lld", &x)


ll d[3000000] = {0};

int main()
{
	ll n;
	S(n);
	ll i;
	ll a[n + 5];
	for(i = 0;i < n;i++) {
		S(a[i]);
	}
	ll co = 0;
	for(i = 0;i < n;i++) {
		ll j;
		ll flag = 0;
		for(j = 0;j < i;j++) {
			if(d[300000 + a[i] - a[j]] == 1) {
				
				flag = 1;
				break;
			}
		}
		if(flag == 1) {
			co++;
		}
		for(j = 0;j <= i;j++) {
			d[300000 + a[i] + a[j] ] = 1;
		}
	}
	cout << co << "\n";
}
