#include <bits/stdc++.h>
#define fast ios_base::sync_with_stdio(false);cin.tie(0)
#define f(i,a,b) for(i=a;i<=b;i++)
#define rf(i,a,b) for(i=a;i>=b;i--)
#define ll long long 
using namespace std;

int n,i,ar[100001],a,b,c,t;
int main()
{   fast;
    cin>>t;
    while(t--)
	{cin>>n;
	f(i,0,n-1) cin>>ar[i];
	sort(ar,ar+n);
	int flag=0;
	a=ar[n-1];
	b=ar[n-2];
	for(i=n-3;i>=0;i--)
	{	c=ar[i];
		if(a+b>c&&b+c>a&&a+c>b)
		{   flag=1;
			break;
		}
	}
	if(flag==1)
	cout<<c<<" "<<b<<" "<<a<<endl;
	else
	cout<<-1<<endl;
	}
    
    
    return 0;
}
