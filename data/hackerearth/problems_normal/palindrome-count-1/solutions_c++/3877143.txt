#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

auto main()->int
{
    string s;
    cin>>s;
    long long counter = 0;

    for(auto i=0;i<s.size();i++)
    {
        for(auto j=i+1;j<=s.size();j++)
        {
            string t = s.substr(i,j-i);
            string rt = t;
            reverse(rt.begin(),rt.end());
            if(t==rt)
            {
                counter ++;
            }
        }
    }
    cout<<counter<<endl;
}
