#include<bits/stdc++.h>
using namespace std;

int main() {
	int n;
	cin >> n;
	int lis[n], a[n];
	for(int i = 0; i < n; i++)
		lis[i] = 1;
	for(int i = 0 ; i < n; i++)
		cin >> a[i];
	int ans = INT_MIN;
	for(int i = 1; i < n; i++){
      for(int j = 0; j < i; j++){
         if ( a[i] > a[j] && lis[i] < lis[j] + 1)
            lis[i] = lis[j] + 1;
        }
    }
    for(int i = 0;i < n; i++){
    	if(ans < lis[i])
    	ans = lis[i];
    }
	//cout << ans <<endl;
	string s;
	while(ans){
		s += ans%2 + '0';
		ans /= 2;
	}
	reverse(s.begin(), s.end());
	cout << s << endl;
}
