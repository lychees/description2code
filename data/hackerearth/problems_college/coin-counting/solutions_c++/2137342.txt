#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
int cs,i,j,n,s;
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&n);
    for(i=s=0;3*i<=n;i++)
      for(j=0;3*i+5*j<=n;j++)
        s++;
    printf("%d\n",s);
  }
	return 0;
}
