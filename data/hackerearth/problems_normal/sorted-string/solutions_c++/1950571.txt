#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
int main()
{
	int t,i,j,c,d;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		vector<pair<int,int> >v(26);
		for(i=0;i<26;i++)
		{
			v[i].first=0;
			v[i].second=i;
		}
		for(i=0;i<s.size();i++)
		{
			v[s[i]-'a'].first++;
		}
		sort(v.begin(),v.end());
		for(i=0;i<26;i++)
		{
			int c=v[i].first;
			char d=v[i].second+'a';
			for(j=0;j<c;j++)
			{
				cout<<d;
			}
		}
		cout<<endl;
	}
}
