#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll n,s,k,i,j,flag=0;
    cin>>n;
    ll a[n];
    for(ll i=0; i<n; ++i){
        cin>>a[i];
    }
    cin>>k;
    sort(a,a+n);
    for(i=0; i<n-1; ++i){
        for(j=i+1; j<n; ++j){
            //if(a[i]==a[j]) continue;
            if(k-a[i]-a[j]==a[i] || k - a[i] - a[j] == a[j]) continue;
            if(binary_search(a,a+n,k-a[i]-a[j])){
                cout<<"YES";
                flag=1;
                break;
            }
        }
        if(flag==1) break;
    }
    if(flag==0) cout<<"NO";

    return 0;
}
