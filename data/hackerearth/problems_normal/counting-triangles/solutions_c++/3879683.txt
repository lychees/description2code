#include<bits/stdc++.h>

#define ll long long

using namespace std;

int main()
{
	int n, i, sides[3];
	scanf("%d",&n);
	map<pair<ll, pair<ll,ll> >, int> m;
	map<pair<ll, pair<ll,ll> >, int>::iterator it;
	for( i = 0 ; i < n ; i++ )
	{
		scanf("%d%d%d",&sides[0], &sides[1], &sides[2]);
		sort(sides, sides + 3);
		pair<ll, pair<ll,ll> > tmp = make_pair(sides[0], make_pair(sides[1], sides[2]) );
		m[tmp]++;
	}
	int ans = 0;
	for( it = m.begin() ; it != m.end() ; it++ )
	{
		if(it->second == 1)
			ans++;
	}
	printf("%d\n", ans);
	return 0;
}
