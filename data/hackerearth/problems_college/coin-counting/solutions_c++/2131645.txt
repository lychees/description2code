#include<iostream>
#include<bits/stdc++.h>

using namespace std;
#define loop(p) for(int w=0;w<p;w++)

int main ()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,k=0,c=0,a;
		cin>>a;
		for(i=0;i<=a/5;i++)
		{
			for(j=0;j<=a/3;j++)	
			{
					for(k=0;k<=a;k++)
					{
						if((i*5+j*3+k)==a)
						{
							c++;
							
							break;
						}
						else if((i*5+j*3+k)>a)
						{
							break;
						}
						
					}
					
			} 
		}
		cout<<c<<endl;
	}



	
	return 0;
}
