#include <iostream>
#include<bits/stdc++.h>
using namespace std;

bool getAngle(int x,int y,int u,int v){
   int X = (u-x);
   int Y = (v-y);
   
   if(X == 0 || Y == 0) return false;

   if(abs(X) == abs(Y)){
   	 return true;
   }
   
   return false;
}

int getP(int x,int y,int u,int v){
	if(x==u && y==v) return 0;
	if(getAngle(x,y,u,v)) return 1;
	
	int ux = abs(x-u);
	int uy = abs(y-v);
	if((ux+uy)%2 == 0) return 2;
	
	return INT_MAX;
}

int main()
{

  int t;
  scanf("%d",&t);
  while(t--){
  	int p,x,y,u,v;
  	scanf("%d%d%d%d%d",&p,&x,&y,&u,&v);
  	x+=10000;
  	y+=10000;
  	u+=10000;
  	v+=10000;
  	
  	if(getP(x,y,u,v) < p){
  		cout<<"Counter-Terrorists Win !\n";
  	}
  	else{
  		cout<<"Terrorists Win !\n";
  	}
  	
  }
  return 0;
}
