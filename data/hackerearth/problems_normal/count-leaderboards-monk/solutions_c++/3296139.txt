#include<cstdio>
using namespace std;
typedef long long int ll;
const int mod = 1000000007;
ll pow(int n,int r)
{
    if (r==0) return 1;
    ll out=pow(n,r/2);
    out=(out*out)%mod;
    if (r%2)
	out=(out*n)%mod;
    return out;
}
int npr(int n,int r)
{
    ll out=1;
    for (int i=0;i<r;i++)
	out=(out*(n-i))%mod;
    return (int)out;
}
int ncr(int n,int r)
{
    ll rfact=1;
    for (int i=1;i<=r;i++)
	rfact=(rfact*i)%mod;
    ll inv;
    for (int i=1;i<mod;i++)
	if (rfact*i % mod==1)
	{
	    inv = i;
	    break;
	}
    inv = pow((int)rfact,mod-2);
    ll nr=1;
    for (int i=0;i<r;i++)
	nr=(nr*(n-i))%mod;
    nr=(nr*inv)%mod;
    return (int)nr;
}
int dp[1001];
int main()
{
    int N,K;
    scanf("%d %d",&N,&K);
    dp[0]=1;
    for (int i=1;i<=N;i++)
    {
	ll out=0,temp;
	for (int j=0;j<i;j++)
	{
	    temp=npr(i-1,j);
	    temp=(temp*(j+1))%mod;
	    out=(out+dp[i-j-1]*temp)%mod;
	  //  prllf("deb %d %d %lld\n",i,j,out);
	}
	dp[i]=out;
    }
    ll out=dp[N];
    out=(out*ncr(N,K))%mod;
    printf("%lld",out);
    return 0;
}
