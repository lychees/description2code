bool powsum(int no)
{
    int flag=0,dig;
    long long sum=0;
    while(sum<101)
    {
        sum=0;
        while(no)
        {
            dig=no%10;
            sum+=pow(dig,2);
            no=no/10;
        }
        if(sum==1)
        {
            return 1;
        }

        else
        no=sum;
    }
}
int main()
{
