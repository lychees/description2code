#include <iostream>
using namespace std;

int main()
{
	int t,n,i,m,cnt;
	long int pos[1000][3],targ[1000][2],d,x,y;
    cin>>t;
    while(t-- >0){
    
    	cin>>n>>m>>d;
    	i=0;
    	for(i=0;i<n;i++){
    		cin>>pos[i][0]>>pos[i][1];
    	
    	
    	}
    	i=cnt=0;
    
    
    	for(i=0;i<m;i++){

    		cin>>targ[i][0]>>targ[i][1];
    		for(int j=0;j<n;j++){
    			pos[j][2]=abs(targ[i][0]-pos[j][0])+abs(targ[i][1]-pos[j][1]);
    			if(pos[j][2]<=d){
    				cnt++;
    				break;
    			}	
    		}	
   
    	}
    
    //	cout<<cnt<<endl;
    	if(cnt>(m/2))
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    	
    }
    return 0;
}
