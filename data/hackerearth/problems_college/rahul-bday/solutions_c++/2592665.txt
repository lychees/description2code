#include <bits/stdc++.h>
using namespace std;
 
int main()
{
   long long int t;
   cin>>t;
   while(t--)
   {
   	long long int b,w;
   	cin>>b>>w;
   	long long int x,y,z;
   	cin>>x>>y>>z;
   	long long int t[5];
   	t[0]=b*x+w*y;
   	t[1]=b*x+w*(y+z);
   	t[2]=b*(x+z)+w*y;
   	t[3]=b*(y+z)+w*y;
   	t[4]=b*x+w*(x+z);
   	x=*min_element(t,t+5);
   	cout<<x<<endl;
   	
   	
   }
    return 0;
}
