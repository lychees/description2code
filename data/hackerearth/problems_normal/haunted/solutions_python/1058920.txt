
n, m = map(int, raw_input().split(' '))
of_the_day = map(int, raw_input().split(' '))

consistency = dict()
high = 0
high_ghost = -1
for i in of_the_day:
	ghost = i - 1
	try:
		consistency[ghost] += 1
	except:
		consistency[ghost] = 1
	if (consistency[ghost] > high):
		high = consistency[ghost]
		high_ghost = i
	elif (consistency[ghost] == high):
		if (i > high_ghost):
			high_ghost = i
	print high_ghost, high
