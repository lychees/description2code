#include <iostream>
using namespace std;
#include<cmath>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int t=min(k,n-k);
    	int a[n];
    	for(int i=0;i<n;i++)
    	a[i]=0;
    	a[0]=1;
    	for(int i=n;i>=t;i--)
    	a[i]=9;
    	a[n-t]--;
    	for(int i=1;i<=n;i++)
    	cout<<a[n-i];
    	cout<<endl;
    }
}
