#include<bits/stdc++.h>
using namespace std;

typedef unsigned u_t;
typedef long l_t;
typedef unsigned long ul_t;
typedef long long ll_t;
typedef unsigned long long ull_t;
#define PRE_CPP do { ios_base::sync_with_stdio(0); cin.tie(0); } while(0)

int main()
{
	PRE_CPP;
	ostringstream oss;
	ll_t a, b, c, ans, low, high, temp, div;
	int testcase;
	for (cin >> testcase; testcase--;) {
		cin >> a >> b >> c;
        temp = min(b, c);
        low = min(ll_t(sqrt(a)), temp);
        high = max(b, c);

        for (ans = 0; low > 0 && (a / low) <= high; --low ) {
            if ( a % low == 0) {
                ++ans;
                div = a / low;
                ans += (div != low && div <= temp);
            }
        }
        oss << ans << "\n";
	}
	cout << oss.str();
 	return 0;
}
