for _ in range(input()):
	mdict = {}
	sortlist = [char for char in raw_input()]
	inlist = [char for char in raw_input()]
	count = 0
	for val in sortlist:
		mdict[val] = count
		count += 1
	newlist = []
	for char in inlist:
		newlist.append(mdict[char])
	newlist.sort()
	finalstr = ''
	for val in newlist:
		finalstr += sortlist[val]
	print finalstr
