#define for_up(v, s, e) for (int v=s; v<e; v++)
#define for_down(v, s, e) for (int v=s; v>=e; v--)

#define vi vector<int>
#define pii pair<int, int>
#define vpii vector< pair<int, int> >

#define pb push_back

#define fs first
#define sc second

int main()
{
    int s, q;
    scan(s);
    scan(q);
    
    set<int> st;
    int p = 0;
    while (q--) {
    	int a;
    	scan(a);
    	
    	int gcd = __gcd(s, a);
    	
    	st.insert(gcd);
    	
    	if (st.size() > p) {
    		print(gcd);
    		p++;
    	}
    	else {
    		print(-1);
    	}
    }

	return 0;
}
