#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int n,k;
		cin>>n>>k;
		long long int total = pow(2,n)-1;
	    long long int i,j,p,q;
	    char ch='G';
	    p = 1;
	    q=total;
	    if(total>=k)
	    {
	    for(i=0;;i++)
	    {
	    	if(p==k)
	    	{
	    		cout<<"G"<<"\n";
	    		break;
			}
			if(q == k)
			{
				cout<<"C"<<"\n";
				break;
			}
			long long int mid =(p+q)/2;
			if(mid==k)
			{
				cout<<ch<<"\n";
				break;
			}
			
			if(mid>k)
			{
				q=mid -1;
				ch='G';
			}
			else
			if(mid<k)
			{
				p = mid+1;
				ch='C';
			}
		}
	    }
	    else
	    cout<<"-1"<<"\n";
	    }
}
