t = input()
while t > 0:
    t -= 1
    n = input()
    esum = 0
    osum = 0
    s = 0
    while n != 1:
        if n % 2 == 0:
            esum += n
            n /= 2
        else:
            osum += n
            n = 3*n+1
        s += 1
    osum += 1
    if (osum % s) > (esum % s):
        print "Odd Rules"
    elif (osum % s) < (esum % s):
        print "Even Rules"
    else:
        print "Tie"
