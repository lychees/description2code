#include<iostream>
using namespace std;
int main()
{
	int t,n;
	long int a[100001];
	a[1]=1;
	a[2]=2;
	a[3]=4;
	for(int i=4;i<100001;i++)
	{
		a[i]=((a[i-1]%1000000007)+(a[i-2]%1000000007)+(a[i-3]%1000000007))%1000000007;
	}
	cin>>t;
	while(t>0)
	{
		cin>>n;
		cout<<a[n]<<"\n";
		t--;
	}
	return 0;
}
