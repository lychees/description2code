#include <iostream>
#include<stdio.h>
#include<string>
#include<algorithm>
using namespace std;

pair<int, pair<string,int> > p[1005];
int k;
int main()
{
	int n,a,i,j,k=0,max,maxi;
	string s;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>s>>a;
		for(j=0;j<k;j++)
		{
			if(s.compare(p[j].second.first)==0)
				break;
		}
		if(j==k)
		{
			p[k++]=make_pair(i,make_pair(s,a));
		}
		else
		{
			p[j].first=i;
			p[j].second.second+=a;
		}
	}
	max=-1;
	maxi=-1;
	string ans;
	for(i=0;i<k;i++)
	{
		if(p[i].second.second>max || (p[i].second.second==max && p[i].first<maxi))
		{
			maxi=p[i].first;
			max=p[i].second.second;
			ans = p[i].second.first;
		}
	}
	cout<<ans;
    return 0;
}
