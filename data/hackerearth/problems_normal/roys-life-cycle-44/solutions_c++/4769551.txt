#include<iostream>
using namespace std;
int main()
{
    char activity[1440];
    int n,x=0,tx=0,ty,y=0;
    int i,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>activity;
        tx=0;
        for(j=0;j<1440;j++)
        {
            if(activity[j]=='C')
            {
                tx++;ty++;
                if(x<tx)
                {
                    x=tx;
                }
                 if(y<ty)
        		{
        			y=ty;
				}
            }
            else
            {
                tx=0;
                ty=0;
            }
        }
    }
        cout<<x<<" "<<y;
	return 0;

}
