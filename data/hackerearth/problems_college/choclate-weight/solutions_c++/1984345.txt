#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long long int n=0;
	scanf("%lld",&n);
	bool arr[100005];
	for(long long int i=0;i<=100000;i++)
	arr[i]=false;
	arr[0]=true;
	for(long long int i=1;i<=n;i++)
	{
		long long int e;
	scanf("%lld",&e);
	arr[e]=true;
}
	long long int t=0;
	scanf("%lld",&t);
	while(t>0)
	{
		long long int x=0;
		scanf("%lld",&x);
		bool status=false;
		for(long long int i=1;i<=x;i++)
		{
			if(arr[i]==true && arr[x-i]==true && i!=x-i)
			{
				status=true;
				break;
				}
			
			}
		if(status)printf("I Like You\n");
		else

		t--;
		}//end of the while loop
