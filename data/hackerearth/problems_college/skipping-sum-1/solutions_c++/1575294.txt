#include<cstdio>
#include<iostream>
using namespace std;
long long a[12][100005];
int arr[100005];
int main()
{
    int n,q;
 	int l,r,k;
    scanf("%d%d",&n,&q);
    k=10;
    for(int i=0;i<n;i++)
    scanf("%d",&arr[i]);;
    for(int i=1;i<=k;i++)
    {
        for(int j=1;j<=i;j++)
        {
            for(int m=j;m<=n;m=m+i)
            {
                if(m==j) a[i][m]=arr[m-1];
                else a[i][m]=a[i][m-i]+arr[m-1];
            }
        }
    }
    while(q--){
        scanf("%d%d%d",&l,&r,&k);
        if(l-k<1) cout << a[k][l+((r-l)/k)*k] << endl;
        else cout << a[k][l+((r-l)/k)*k]-a[k][l-k] << endl;
    }
    return 0;
}
