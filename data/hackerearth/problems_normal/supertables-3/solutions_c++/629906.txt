#include<iostream>
using namespace std;
int max(int a,int b)
{
    if(a>b)
    return a;
    return a;
}
int min(int a,int b)
{
    if(a<b)
    return a;
    return b;
}
int gcd(int a,int b)
{
   
        if(!(a%b))
        return b;
        gcd(b,a%b);
 
}
int lcm(int a,int b)
{
    return(a*b/gcd(a,b));
}
int main()
{
    long  a,b,noFactors;
    long long n,g,mid;
        long  l,r;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>n;
    l=min(a,b);
    r=max(a,b)*n;
    while(l<=r)
    {
          mid=(l+r)/2;
          g=lcm(a,b);
          //cout<<"g="<<g<<endl;
          noFactors=mid/a+mid/b-mid/g;
          if(noFactors==n)
          {
              cout<<max((mid/a)*a,(mid/b)*b);
              cout<<endl;
              break;
          }
          else if(noFactors>n)
          r=mid-1;
          else
          l=mid+1;
    }
    }
    return 0;
}
