#include <iostream>
using namespace std;
int c[1000][3]={0};
int n;

int price(int i,int j)
{
	if(i==n-1)
	return c[n-1][j];
	if(j==0)
	return c[i][j]+min(price(i+1,1),price(i+1,2));
	else if(j==1)
	return c[i][j]+min(price(i+1,0),price(i+1,2));
	else 
	return c[i][j]+min(price(i+1,0),price(i+1,1));
	
}

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    
    	cin >> n;
    	for(int i=0;i<n;i++)
    	for(int j=0;j<3;j++)
    	cin >> c[i][j];
    	int p=min(price(0,0),min(price(0,1),price(0,2)));
    	cout << p << endl;
    	
    }
    return 0;
}
