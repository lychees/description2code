int main()
{
    int tc,i,c;
    scanf("%d\n",&tc);
    while(tc--)
    {
        c=0;
        stack<ll>s;
        string str;
        getline(cin,str);
        s.push(str[0]);
        for(i=1;i<str.size();i++)
        {
            if(str[i]==s.top())
                c++;
            else
                s.push(str[i]);
        }
        pii(c);
    }
    return 0;
}
