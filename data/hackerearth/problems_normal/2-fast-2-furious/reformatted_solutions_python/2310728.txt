N = int(raw_input())
dom = map(int, raw_input().split())
brain = map(int, raw_input().split())
max_dom = 0
max_brain = 0

for i in range(N-1):
    val = abs(dom[i] - dom[i + 1])
    if (val > max_dom):
        max_dom = val

    val = abs(brain[i] - brain[i+1])
    if (val > max_brain):
        max_brain = val

if max_dom > max_brain:
    print "Dom"
    print max_dom
else:
    print "Brian"
    print max_brain
