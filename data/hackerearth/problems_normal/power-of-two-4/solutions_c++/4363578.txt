#include <bits/stdc++.h>
using namespace std;

bool power (int a)
{

	int temp;
	temp = 1ll;
	
	while(temp<=a) {
		if (a == temp) {
			return true;
		}
		temp = temp <<1;
	}
	return false;
}

int main (void)
{
	int test;
	scanf ("%d", &test);
	
	while (test--) {
		int n;
		scanf ("%d", &n);
		
		int arr[n];
		int i, j, ctr = 0;
		
		for (i = 0; i < n; i++) {
			scanf ("%d", &arr[i]);
		}
		if(n == 1) {
			if(power(arr[0])) {
				printf("YES\n");
			}else {
				printf("NO\n");
			}
			continue;
		}
		bool flag = false;
		int sum = 0;
		for (i = 0; i < 32; i++) {
			ctr = 0;
			for (j = 0; j < n; j++) {
				if (arr[j]&1<<i) {
					if(ctr == 0) {
						sum = arr[j];
						ctr = 1;
					}else {
						sum = sum&arr[j];
					}
				}
			}
			if(power(sum)) {
				flag = true;
				break;
			}
		}
		if(flag) {
			printf ("YES\n");
		}else {
			printf ("NO\n");
		}
	}
	return 0;
}
