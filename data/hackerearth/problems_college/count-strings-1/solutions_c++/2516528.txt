#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
char str1[1000001],str2[1000001];
int main()
{
    int test,l1,l2,i,flag;
    cin>>test;
    while(test--)
    {

    	int a1[26]={0},b1[26]={0};
    	cin>>str1>>str2;
    	l1=strlen(str1),l2=strlen(str2);
    	for(i=0;i<l1;i++)
    	a1[str1[i]-'a']++;
    	for(i=0;i<l2;i++)
    	b1[str2[i]-'a']++;
    	long long int ans=1;
    	flag=0;
    	for(i=0;i<26;i++)
    	{
    		if(a1[i]<b1[i])
    		{
    			flag=1;
    			break;
    		}
    		else
    		{
    			int ct=0,val=b1[i];
    			while(ct!=val)
    			{
    				ans=(ans*a1[i])%mod;
    				a1[i]--;
    				ct++;
    			}
    		}
    	}
    	if(flag==1)
        ans=0;
    	printf("%lld\n",ans);
    }
    return 0;
}
