#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	 int t;
	 ll n;
	 ll q;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>q;
    	n=((q*q*q)-((q-2)*(q-2)*(q-2)));
    	if(n==2)
    		cout<<1<<endl;
    	else
    		cout<<n<<endl;
	}
	return 0;
}
