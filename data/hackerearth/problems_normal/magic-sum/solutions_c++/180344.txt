ll a[520];
int n;
ll dp[520];
ll calc(int p)
{
   // cout << p << endl;
    if(p > n)
        return 0;
    ll &ret = dp[p];
    if(ret != -100000000000LL)
        return ret;
    //cout << p << endl;
    ret = -100000000000LL;
    ret = max(ret, a[p] + max(calc(p*2),calc(p*2+1)));
   // cout << p << " " << ret << endl;
    return ret;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {

      //  int n;
        scanf("%d",&n);
        for(int i=0;i<=n;i++)
            dp[i] = -100000000000LL;
        ll ans = -100000000000LL;
        for(int i=1;i<=n;i++)
        {
            scanf("%lld",&a[i]);
           // ans = max(ans,a[i]);
        }
        for(int i=1;i<=n;i++)
        {
            ans = max(ans, a[i] + calc(2*i) + calc(2*i+1));
        }
        printf("%lld\n",ans);
    }
return 0;
}
