#include<bits/stdc++.h>
#include<math.h>
using namespace std;
#define LL long long int
#define scanInt(x) scanf("%d",&x)
#define scan2Int(x,y) scanf("%d%d",&x,&y)
#define scanll(x) scanf("%lld",&x)
#define scan2ll(x,y) scanf("%lld%lld",&x,&y)
#define fors(i, a, b) for(int i=a; i<=b; i++)
#define forb(i, a, b) for(int i=a; i>=b; i--)
//Testing Harish Garg 
LL accSum[1000][1000];
LL n,k;
LL funAccSum(int i,int j){
	if(i<0||j<0||i>n-1||j>n-1)
	 return 0;
	else
	 return accSum[i][j];
}
int main(){
	
	scanll(n);
	scanll(k);
	int a[n][n];
	fors(i,0,n-1){
		fors(j,0,n-1){
			scanInt(a[i][j]);
		}
	}
	
	fors(i,0,n-1){
		fors(j,0,n-1){
			accSum[i][j]=a[i][j]+funAccSum(i-1,j)+funAccSum(i,j-1)-funAccSum(i-1,j-1);
		}
	}
	int ans=0;
	fors(i,0+k-1,n-1){
		fors(j,0+k-1,n-1){
			LL temp=funAccSum(i,j)- funAccSum(i-k,j)-funAccSum(i,j-k)+funAccSum(i-k,j-k);		
			LL no = ceil(pow(temp, (double)1 / 3));
			if(no*no*no==temp)
			ans++;
			//cout<<temp<<" "<<no;
		}
		//cout<<endl;
	}
	
	cout<<ans;
	
}
