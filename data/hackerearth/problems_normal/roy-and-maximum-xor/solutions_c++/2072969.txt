#include <iostream>
using namespace std;
long long int getBits(long long int x){
        long long int count = 0;
        while(x > 0){
            x = x>>1;
            count++;
        }
        return count;
    }
int main()
{
    long long int t,n,bitN,max,count;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	bitN    = 1ll << getBits(n-1);
        max     = (bitN - 1);
        count   = (bitN == n) ? bitN :(n -(bitN>>1))<<1 ;
    	cout<<max<<" "<<count<<endl;
    }
    return 0;
}
