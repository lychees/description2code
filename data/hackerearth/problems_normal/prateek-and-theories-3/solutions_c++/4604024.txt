#include <iostream> #include <sstream> #include <cstdio>
#include <cstdlib> #include <cmath>
#include <algorithm> #include <numeric>
#include <string> #include <cstring>
#include <map> #include <set>
#include <vector> #include <bitset>
#include <queue> #include <stack>
#include <unordered_map> #include <unordered_set>
#include <limits> #include <climits> #include <ctime>
using namespace std;

#define MOD 1000000009
#define debug(x) cout << #x << " : " << x << endl;


int main() {
	int t, n, st[10005], en[10005];
	
	cin >> t;
	
	while(t--) {
		cin >> n;
		
		for(int i=0; i<n; i++) {
			cin >> st[i] >> en[i];
			en[i]--;
		}
		
		sort(st, st+n);
		sort(en, en+n);
			
		int l = 0, r = 0, maxnum = 0, cnt = 0;
		
		while(l < n && r < n) {
			if(st[l] <= en[r]) {
				l++;
				cnt++;
			} else {
				cnt--;
				r++;
			}
			maxnum = max(cnt, maxnum);
		}
		
		maxnum = max(cnt, maxnum);
		
		cout << maxnum << endl;
	}
	
    return 0;
}
