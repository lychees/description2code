#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
    long int numPrimes[1000001] = {0};
    numPrimes[0] = numPrimes[1] = 100000;
    numPrimes[2] = numPrimes[3] = numPrimes[5] = numPrimes[7] = 1;
    numPrimes[4] = numPrimes[6] = 2;
    
    for(int i = 8; i< 1000001; ++i){
    	numPrimes[i] = 1 + min({numPrimes[i-2], numPrimes[i-3], numPrimes[i-5], numPrimes[i-7]});
    }
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	if(n==0 || n==1) cout<<-1<<"\n";
    	else
    	cout<<numPrimes[n]<<"\n";
    }
    return 0;
}
