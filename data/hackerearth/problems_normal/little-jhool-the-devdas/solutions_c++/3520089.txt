#include <iostream>
using namespace std;

int main()
{
    int i,n,minOdd,sum;
    cin>>n;
    int * arr = new int[n];
    sum=0;
    minOdd=999999;
    for(i=0;i<n;i++)
	{
		cin>>arr[i];
		if(arr[i]%2)
		{
			if(arr[i]<minOdd)
				minOdd = arr[i];
		}
		sum+= arr[i];
	}
	
	if(sum%2==0)
	{
		if(minOdd==999999)
		{
			cout<<":(";
		}
		else
		{
			cout<<sum-minOdd;
		}
	}
	else
	{
		cout<<sum;
	}
	cout<<endl;
    return 0;
}
