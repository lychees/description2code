'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
from string import ascii_lowercase as al
t = int(raw_input())
dic = {c: 0 for c in al}
for _ in xrange(t):
    s = raw_input().lower()
    d = dict(dic)
    for c in s:
        d[c] += 1
    li = [v[0] for v in sorted(d.iteritems(), key=lambda (k, v): (-v, k), reverse=True)]
    print ' '.join(li)
