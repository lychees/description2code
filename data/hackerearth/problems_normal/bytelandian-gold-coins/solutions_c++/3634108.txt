#include<bits/stdc++.h>
#define MAXN 100;
using namespace std;
map<long long int,long long int> mp;

long long int sol(long long int n)
{
	if(n==0) return 0;
	if(mp[n]!=0) return mp[n];
	long long int j=sol(n/2)+sol(n/3)+sol(n/4);
	return mp[n]=max(n,j);
}
int main()
{
     //printf("Hello World!\n");
     long long int N;
     while(scanf("%lld",&N)!=EOF)
     {
     	cout<<sol(N)<<endl;
     }
    return 0;
}
