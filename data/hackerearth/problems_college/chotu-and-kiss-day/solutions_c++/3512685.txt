#include <stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,i,flag,st,en;
    long long int a[3000],k;
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%lld",&a[i]);
      scanf("%lld",&k);
      sort(a,a+n);
      i=flag=0;
      while(i<n-2 && flag==0)
      {
      	st=i+1;
      	en=st+1;
      	while(a[i]+a[st]+a[en]<k && en<n)
      	  en++;
      	  if(a[i]+a[st]+a[en]==k)
      	  {
      	  	flag=1;
      	    break;
      	  }
      	  else
      	  {
      	  	while(st<en && flag==0)
      	    {
      	    	st++;
      	      while(a[i]+a[st]+a[en]>k && st<en)
      	        en--;
      	        if(a[i]+a[st]+a[en]==k)
      	        {
      	        	flag=1;
      	        	break;
      	        }
      	    }
      	  }
        i++;
      }
      if(flag==1)
       printf("YES\n");
       else
       printf("NO\n");
    return 0;
}
