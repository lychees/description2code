#include<iostream>
#include<string>

using namespace std;


int main()
{
        int t,sum=0,count[26],y,cot=0;
        string s;

        cin>>t;
        cin>>s;

        y=s.length();

        for(int i=0;i<=y-t;i++)
        {
            for(int k=0;k<26;k++)
            {
                count[k]=1;
            }

            sum=0;

        	for(int j=i;j<y;j++)
            {
                sum+=count[s[j]-'a'];

                count[s[j]-'a']=0;

                if(sum==t)
                {
                     cot++;
                }

                if(sum>t)
                {
                   break;
                }
            }
        }

        cout<<cot;

        return 0;
}
