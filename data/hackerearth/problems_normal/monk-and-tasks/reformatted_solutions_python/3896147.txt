from operator import itemgetter

t = int(raw_input())

for i in range(t):
    n = int(raw_input())
    arr = list(map(int, raw_input().split()))
    binarr = []
    for elem in arr:
        temp = str(bin(elem))
        binarr.append((elem, temp.count('1')))
    binarr.sort(key=itemgetter(1))
    ans = ''
    for elem in binarr:
        ans += str(elem[0]) + ' '
    print ans
