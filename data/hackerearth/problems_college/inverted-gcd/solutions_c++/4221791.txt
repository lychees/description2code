#include <bits/stdc++.h>
using namespace std;
#define ll long long

int gcd(int a,int b)
{
	return b == 0 ? a : gcd(b, a % b);
}

int main()
{
    int n;
    cin >> n;
    
    int *A=new int[n];
    for(int i=0;i<n;i++)
    	cin >> A[i];
    	
    ll c=0;
    for(int i=0;i<n-1;i++)
    {
    	for(int j=i+1;j<n;j++)
    	{
    		if(A[i]>A[j] && gcd(A[i],A[j])==1)
    			c++;
    	}
    }
    cout << c << endl;
}
