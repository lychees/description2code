#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    // first generate the fibonacci number till 100000
    int fib[31];
    fib[0]=0;fib[1]=1;
    for(int i=2;i<=30;i++)fib[i]=fib[i-1]+fib[i-2];
    //cout<<fib[30];
    int T;
    cin>>T;
    for(int i=0;i<T;i++){
    	int x,y;
    	cin>>x>>y;
    	int count=0;
    	for(int j=0;j<30;j++){
    		if(fib[j]>=x&&fib[j]<=y)count++;
    	}
    	if(count%2==0){cout<<"VALID"<<endl;}
    	else{cout<<"INVALID"<<endl;}
    }
    return 0;
}
