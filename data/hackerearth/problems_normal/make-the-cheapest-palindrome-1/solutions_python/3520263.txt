def test(t,a,b):
	total = 0
	for x in range(len(t)):
		if t[x] == '/' and t[len(t)-x-1] == 'a':
			total += a
		if t[x] == '/' and t[len(t)-x-1] == 'b':
			total +=b
		if t[x] == '/' and  t[len(t)-x-1] == '/':
			if a<b:
				total += a
			else:
				total +=b
		if t[x] != t[len(t)-x-1] and t[x] in 'ab' and t[len(t)-x-1] in 'ab':
			return -1
	return total
