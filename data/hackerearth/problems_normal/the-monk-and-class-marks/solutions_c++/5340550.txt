#include <iostream>
#include<bits/stdc++.h>
using namespace std;
bool cmp(const pair<int,string>& firstElem, const pair<int, string>& secondElem) {
if(firstElem.first != secondElem.first)
return firstElem.first > secondElem.first;
else
return firstElem.second < secondElem.second;

}
int main()
{
long long int n,k,j,i;
cin>>n;
string s;
vector<pair<int,string>> m;
for(i=0;i<n;i++)
{

cin>>s>>k;
m.push_back(make_pair(k,s));

}
sort(m.begin(),m.end(),cmp);
vector<pair<int,string>>::iterator ii;
for(ii=m.begin();ii!=m.end();ii++)
cout<<ii->second<<" "<<ii->first<<"\n";
//cout << "Hello World!" << endl;
return 0;
}
