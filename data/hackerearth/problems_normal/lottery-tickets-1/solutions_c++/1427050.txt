#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
#include <cstdio>
using namespace std;
int dp[55],n;
vector<pair<pair<int,int>,int>  > ar;
int fun(int idx)
{
	if(idx==n)
		return 0;
	if(dp[idx]!=-1)
		return dp[idx];
	int ans=ar[idx].second;
	ans=max(ans,fun(idx+1));
	for(int i=idx+1;i<n;i++)
		if(idx!=i && ar[i].first.first>=ar[idx].first.second)
			ans=max(ans,ar[idx].second + fun(i));
	return dp[idx]=ans;
}
int main()
{
	int t,x,y,c;
	scanf("%d",&t);
	while(t--)
	{
		ar.clear();
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			dp[i]=-1;
			scanf("%d%d%d",&x,&y,&c);
			ar.push_back(make_pair(make_pair(x,y),c));
		}
		sort(ar.begin(),ar.end());
		printf("%.2lf\n",fun(0)/100.0);
	}
    return 0;
}
