#include <iostream>
#include <set>
#include <string>

using namespace std;
long n;

int main()
{
 
    cin>>n;
    long rn = 1;
    rn = rn<<n;
    
    set<string> drivers;
     string d1;
     string d2;
    string results [2][32768]; 
    long i;
    for(i=0;i<rn-1;i++)
    {
    	cin>>d1>>d2;
    	drivers.insert(d1);
    	drivers.insert(d2);
		results[0][i] =d1;
		results[1][i] =d2;
	}
    for(i=0;i<rn-1;i++)
    {
    	 drivers.erase(results[1][i]); 
    	 
    }
    
    cout<<*drivers.begin();
    
    return 0;
}
