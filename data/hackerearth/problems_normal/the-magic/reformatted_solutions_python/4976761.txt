#!/usr/local/bin/python

import sys

total_test_cases = int(sys.stdin.readline())
for _ in range(total_test_cases):
    stones_needed = int(input())
    days_needed = 1
    current_no_of_stones = 0
    while True:
        if stones_needed <= 0:
            break
        if stones_needed % 2 == 0:
            stones_needed = stones_needed / 2
        else:
            days_needed = days_needed + 1
            stones_needed = stones_needed - 1
    print(days_needed-1)
