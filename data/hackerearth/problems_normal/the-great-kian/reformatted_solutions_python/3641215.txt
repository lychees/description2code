n = int(raw_input())
no = map(int, raw_input().split())
a1, a2, a3 = 0, 0, 0
for i in range(0, n, 3):
    a1 = a1+no[i]
    if i+1 <= n-1:
        a2 = a2+no[i+1]
    if i+2 <= n-1:
        a3 = a3+no[i+2]
print a1, a2, a3
