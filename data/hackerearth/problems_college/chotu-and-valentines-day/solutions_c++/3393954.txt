#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,q;
	cin>>n>>q;
	int i,a[n+1];
	
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	long long sum[n+1];
	sum[1]=a[1];
	for(i=2;i<=n;i++)
	{
		sum[i] = sum[i-1] + a[i];
	}
	while(q--)
	{
		int a,b;
		cin>>a>>b;
	    
		if(a==1)
		{
			cout<<sum[b]<<"\n";
		}
		else
		{
			cout<<sum[b]-sum[a-1]<<"\n";
		}
	}
	cout<<"\n";
}
