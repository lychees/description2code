#include <bits/stdc++.h>
using namespace std;
bool cmp1(pair<int, int > p1, pair<int, int > p2){
		return p1.first<p2.first;
}
int main()
{
    int n, x, y;
    cin>>n;
    vector<pair<int, int > > v;
    for(int i=0; i<n; i++){
    	cin>>x>>y;
    	v.push_back(make_pair(x, y));
    }
    sort(v.begin(), v.end(), cmp1);
    vector<int >::iterator j;

    int max_[v.size()];
    vector<int > a;
    for(int i=0; i<v.size(); i++){
		a.push_back(v[i].first);
		max_[i]=v[i].second;
    }
    for(int i=v.size()-2; i>=0; i--){
    	max_[i]=max(max_[i+1], max_[i]);
    }
    int q;
    long long total=0;
    cin>>q;
    for(int i=0; i<q; i++){
    	cin>>x;
    	j = lower_bound(a.begin(), a.end(), x);
    	if(j!=a.end()){
    		total += 100*(max_[distance(a.begin(), j)]/x);
    	}
    }
    cout<<total<<endl;
    return 0;
}
