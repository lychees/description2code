#include <iostream>
#include <algorithm>
#include <map>
 
using namespace std;
 
int main()
{
    ios::sync_with_stdio(0);
    int t;
    map<string, int> m;
    string s;
    cin >> t;
    while (t--) {
        cin >> s;
        m[s]++;
    }
    for (map<string, int>::iterator i = m.begin(); i != m.end(); i++) {
        cout << i->first << " " << m[i->first] << endl;
    }
 
    return 0;
}
