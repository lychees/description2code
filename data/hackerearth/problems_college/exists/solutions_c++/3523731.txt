#include <iostream>
#include<algorithm>
using namespace std;

int main()
{long long int t,n,q,k,y=0,a[100000],b=0,la,m;
    cin>>t;
    while(t--){
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>a[i];
    sort(a+0,a+n);
    cin>>k;
    for(int j=0;j<k;j++)
    {
    	y=0;
    	
    	cin>>q;
    	b=0;la=n-1;
    	while(b<=la)
    	{
    		m=(b+la)/2;
    		if(q==a[m])
    		{
    			y=1;
    			break;
    		}
    		else if(q>a[m])b=m+1;
    		else la=m-1;
    	}
    	
    	if(y)
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }}
    return 0;
}
