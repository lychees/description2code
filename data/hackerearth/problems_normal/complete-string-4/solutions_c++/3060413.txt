#include <iostream>
#include <string>

#define _ ios_base::sync_with_stdio(0);cin.tie(0);
#define FOR(i,a,b) for(i=a;i<b;i++)

using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--) {
		string s;
		bool a[26] = {0};
		cin >> s;
		int l = s.length(), i = 26;
		l--;
		while(l >= 0){
			if(a[s.at(l) - 'a'] == false) {
				a[s.at(l) - 'a'] = true;
				i--;
			}
			l--;
		}
		if(i == 0)
			cout << "YES";
		else
			cout << "NO";
		cout << endl;
	}
    return 0;
}
