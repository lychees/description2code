#include <iostream>
using namespace std;
long long l[100000],d[100000],sum,n,x,count;
int f(long long i)
{long long sum=0;
	while(i>9)
         {while(i)
         { sum+=i%10;
           i/=10;
         }
         i=sum;
         sum=0;
         }
         return i;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {cin>>n;
    count=0;
     for(int i=0;i<n;i++)
     cin>>l[i]>>d[i];
     for(int i=0;i<n;i++)
      { x=f(l[i]);
        x=f(x*d[i]);
        count+=x;
      }
      cout<<f(count)<<endl;
      }
    return 0;
}
