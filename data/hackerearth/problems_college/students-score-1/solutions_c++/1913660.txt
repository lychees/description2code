#include <iostream>
#include<stdlib.h>
using namespace std;

int main()
{
	long long int n;
	cin>>n;
	long long int *ar=(long long int* )malloc(sizeof(long long int)*n);
		long long int i=0;
			while(n--)
			cin>>ar[i++];
			n=i;
			i=0;
			long long int sum=0,prevsum=0;
			while(n--)
			{
				prevsum+=ar[i];
				if(ar[i]>prevsum)
				prevsum=ar[i];
				if(prevsum>sum)
				sum=prevsum;
				i++;
				
				
			}
			cout<<sum<<"\n";
    return 0;
}
