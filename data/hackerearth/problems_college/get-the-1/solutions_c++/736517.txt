#include<iostream>
using namespace std;
int newn(long long int k)
{
    long long int r,s=0;
    while(k>0)
    {
        r=k%10;
        s=s+(r*r);
        k=k/10;
    }
    return s;
}
int main()
{
    int n;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
        while(a[i]!=1&&a[i]!=89)
        {
            long long int h;
            h=newn(a[i]);
            a[i]=h;
        }
        if(a[i]==1)
            cout<<"Case #"<<i+1<<":YES"<<"\n";
        else
            cout<<"Case #"<<i+1<<":NO"<<"\n";
    }
    return 0;
}
