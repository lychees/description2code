#include <iostream>
using namespace std;
void quicksort(int *a,int left,int right){
	int i=left;
	int j=right,pivote=a[left];
	if(i<j){
		do{
			do{
				i++;
			}while(pivote<a[i] && i<right);
			do{
				j--;
			}while(pivote>a[j] && j>=left);
			if(i<j){
				int temp=a[i];
				a[i]=a[j];
				a[j]=temp;
			}
		}while(i<j);
		a[left]=a[j];
		a[j]=pivote;
		quicksort(a,left,j);
		quicksort(a,j+1,right);
		}
	}
int main()
{
    int T,nr,ng;
    cin>>T;
    while(T--){
    	cin>>nr>>ng;
    	int players[nr],i,j,flag=1;
    	float c= float(nr)/ng;
    	int count = int(c+1);
    	//int group[m][count];
    	for(i=0;i<nr;i++)
    		cin>>players[i];
    	int me=players[0];
    	quicksort(players,0,nr);
    	int k=0;
    	for(i=0;i<count && flag;i++){
    		for(j=0;j<ng;j++){
    			if(players[k]==me){
    				cout<<i<<endl;
    				flag=0;
    				break;
    			}
    			k++;
    		}
    	}
    }
}
