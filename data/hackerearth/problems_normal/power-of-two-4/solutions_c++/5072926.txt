#include <iostream>
using namespace std;

int main()
{
    int t,n,i,j,f;
    cin>>t;
    p:
    while(t--){
    	cin>>n;
    	f=0;
    	int ans;
    	long long a[n],x;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	int tot=0;
    	for(i=0;i<32;i++){
    		tot+=(1<<i);
    	}
    	for(i=0;i<32;i++){
    		ans=tot;
    		for(j=0;j<n;j++){
    			if(a[j]&(1<<i)){
    				ans=ans&a[j];
    			}
    		}
    		if(!(ans&(ans-1))){
    			f=1;
    			break;
    		}
    	}
    	if(f){
    		cout<<"YES"<<endl;
    	}
    	else{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
