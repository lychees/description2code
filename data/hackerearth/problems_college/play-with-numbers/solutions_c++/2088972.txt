#include <iostream>
#include <cmath>
#include <bits/stdc++.h>

using namespace std;

int main() {
    long long Z, M, N, T, low, high, temp, d1;
    int ans;

    for (cin >> T; T--; ) {

        cin >> Z >> M >> N;
        temp = min(M, N);
        long long int h=sqrt(Z);
        low = min(h, temp);
        high = max(M, N);

        for( ans = 0; low > 0 && (Z / low) <= high; --low ) {
            if ( Z % low == 0) {
                ++ans;
                d1 = Z/low;
                ans += (d1!=low && d1<=temp);
            }
           
        }
        cout << ans << "\n";
    }

    return 0;
}
