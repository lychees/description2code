#include<bits/stdc++.h>
#define ll long long
using namespace std;
const ll ma=1e6+7;
ll toi[ma], tp[ma];
bool pri[ma];
void seive()
{
    for(int i=2;i*i<=ma;i++)
    {
        if(!pri[i])
        {
            for(int j=2*i;j<=ma;j+=i)
            {
                pri[j]=true;
            }
        }
    }
}
void segmented_seive(ll l,ll r)
{
    for(ll i=l;i<=r;i++)
    {
        toi[i-l]=i;
        tp[i-l]=i;
    }
    for(ll i=2;i*i<=r;i++)
    {
        if(!pri[i])
        {
        for(ll j=max(i*i, (l+(i-1))/i*i);j<=r;j+=i)
        {
            toi[j-l]-=toi[j-l]/i;
            while(tp[j-l]%i==0)
            tp[j-l]/=i;
        }
        }
    }
    ll p=2;

    for(ll i=max(l,p);i<=r;i++)
    {
        if(tp[i-l]>1)
        {
            toi[i-l]-=toi[i-l]/tp[i-l];
        }
    }
}
int main()
{

    //freopen("i.txt","r",stdin);
    int t;
    seive();
    for(scanf("%lld",&t);t;--t)
    {
        ll l,r,k;
        scanf("%lld%lld%lld",&l,&r,&k);
        ll c=0;
        segmented_seive(l,r);
        /*for(ll i=l;i<=r;i++)
        {
            cout<<toi[i-l]<<endl;
        }*/
        for(ll i=l;i<=r;i++)
        {
            if(toi[i-l]%k==0)
            c++;
        }
        float ans=(c*1.0)/((r-l+1)*1.0);
        printf("%.6f\n",ans);
    }
       return 0;
}
