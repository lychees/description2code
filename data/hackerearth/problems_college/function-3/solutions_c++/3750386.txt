long long int func(int x,long long int f0)
{
	if(x==0 || x==1 || x==2) return f0;
	long long int ans;
	ans=(2*x*func(x-1,f0) - (x-1)*func(x-2,f0) - (x-2)*func(x-3,f0))/3;
	return ans;
}
