#include <iostream>
using namespace std;

int main()
{
	
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int i;
    	int arr[1000009];
    	long long int final=n;
    	for(i=2;i*i<=n;i++)
    	arr[i]=1;
    	for(i=2;i*i<=n;i++)
    	{
    		if(arr[i]==1)
    		{
    			if(n%i==0)
    			final=final-final/i;
    			while(n%i==0)
    			n=n/i;
    			int j;
    			for(j=2*i;j*j<=n;j+=i)
    			arr[j]=0;
    		}
    	}
    	if(n>1)
    	final=final-final/n;
    	cout << final << endl;
    }
    return 0;
}
