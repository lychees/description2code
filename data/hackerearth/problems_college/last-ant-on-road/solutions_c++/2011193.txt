#include <bits/stdc++.h>

using namespace std;

int main() {
  int t; scanf("%d", &t);
  while(t--) {
    int n; scanf("%d", &n);
    int l = 0, r = 0;
    for(int i = 0; i < n; ++i) {
      int foo; scanf("%d", &foo);
      if(foo == -1) ++l;
      else ++r;
    }
    if(l < r) ++l;
    printf("%d\n", l);
  }
  return 0;
}
