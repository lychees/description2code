#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	vector<int>arr(n);
	for(int i=0;i < n;i++){
		cin >> arr[i];
	}
	int maxi = 0,cur=0;
	for(int i=0;i <n;i++){
		cur = max(arr[i] , arr[i] + cur);
		maxi = max(maxi, cur);
	}
	if(maxi >= 0)cout << maxi << endl;
	else cout << "0\n";
	return 0;
}
