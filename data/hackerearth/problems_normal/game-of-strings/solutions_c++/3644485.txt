#include <bits/stdc++.h>
using namespace std;

bool is_vowel(char s)
{
	if(s=='a'|| s=='e'|| s=='i'|| s=='o'|| s=='u' )
		return true;
	return false;
}

int main() 
{
	char a[5005], b[5005];
	cin>>a;
	cin>>b;
	int n=strlen(a), m=strlen(b), c[2][m+1], i, j, k=1;
	for(i=0;i<=1;i++)
		c[i][0]=0;
	for(j=0;j<=m;j++)
		c[0][j]=0;
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			if(a[i-1]==b[j-1] && is_vowel(a[i-1])==false)
				c[k][j]=c[ (k+1)%2 ][j-1]+1;
			else
				c[k][j]=max(c[(k+1)%2 ][j], c[k][j-1]);
		}
		
		k=(k+1)%2;
	}
	cout<<c[(k+1)%2][m];
	return 0;
}
