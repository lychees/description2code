n=int(raw_input())
a=map(int,raw_input().split())
fn=sn=tn=0
c1=1
c2=2
c3=3
for i in range(n):
    if c1 == i+1:
        fn=fn+a[i]
        c1=c1+3
    if c2 == i+1:
        sn=sn+a[i]
        c2=c2+3
    if c3 == i+1:
        tn=tn+a[i]
        c3=c3+3
print fn,sn,tn
