#include <iostream>
using namespace std;

int main()
{
    int t,n,k,count=0,a;
    cin>>t;
    int res[t];
    for(int i=0;i<t;i++)
    {  count=0;
    	cin>>n>>k;
    	for(int j=0;j<n;j++)
    	{
    		cin>>a;
    		if(a<=0)
    		count++;
    	}
    	if(count<k)
    	res[i]=0;
    	else
    	res[i]=1;
    }
    for(int i=0;i<t;i++)
    {
    	if(res[i]==1)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;
    }
    return 0;
}
