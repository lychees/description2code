#include <iostream>
#include<cstdio>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(a%b==0)
	return b;
	else
	return gcd(b,a%b);
}
int main() 
{
	std::ios_base::sync_with_stdio(false);

	long long int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long int arr[n],ans;
		for( i=0;i<n;i++)
		cin>>arr[i];
		ans=arr[0];
		for(i=1;i<n;i++)
		ans=gcd(ans,arr[i]);
		cout<<ans*n<<endl;
	}
		return 0;
}
