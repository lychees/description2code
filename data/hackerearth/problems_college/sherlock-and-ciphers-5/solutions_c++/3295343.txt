#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    char ch;
    cin>>ch;
    int flag=0;
    while(n--)
    {
    	
    	//cout<<n<<endl;
    	string s;
    	getline(cin,s);
    	if(ch!='\n' && flag==0)
    	{
    		s=ch+s;
    		flag=1;
    	}
    	//cout<<s<<endl;
    	int x = sqrt(s.length());
    	for(int i=0;i<x;i++)
    	{
    		for(int j=0;j<s.length();j+=x)
    			cout<<s[i+j];
    	}
    	
    	cout<<endl;
    	
    }
    return 0;
}
