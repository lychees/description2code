#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int i,temp;
    vector<int>vec;
    vector<int>::iterator it,it1;
    for(i=0;i<n;i++)
    {
    	cin>>temp;
    	vec.push_back(temp);
    }
    int res = 0,maxi = 0;
    int size = n;
    while(size != 2)
    {
    	maxi=0;
    	for(it=vec.begin()+1;it!=vec.end()-1;it++)
    	{
    		if(*(it-1)*(*(it+1))>maxi)
    		{
    			maxi=*(it-1)*(*(it+1));
    			it1=it;
    		}
    	}
    	vec.erase(it1);
    	res += maxi;
    	size--;	
    }
    cout<<res<<endl;
}
