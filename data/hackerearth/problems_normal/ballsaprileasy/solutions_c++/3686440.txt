#include <iostream>
#include <algorithm> //std::sort
using namespace std;

int main()
{
    int n;
    cin>>n;
    long int *num;
    long int sum=0,ball_need; //number of required balls
    long int k = 1; //layor number
    
    num = new long[n];
    
    if(num==NULL)
    	cout<<"Memory error";
    else
    {
    	for(int i=0; i<n; i++)
    	{
    		cin>>num[i];
    	}
    	sort(num, num+n); //sort the inputs in increasing order
    	for(int i=0; i<n; i++)
    	{
    		ball_need = k*k;
    		if(ball_need-num[i]<0)
    		{
    			sum = sum + ball_need;
    			i--;
    		}
    		else
    		{
    			sum = sum + (ball_need-num[i]);
    		}
    		k++;
    	}
    }
    cout<<sum;
    delete num;
    return 0;
}
