#include<stdio.h>
using namespace std;


int main(){
	int T;
	scanf("%d",&T);
	
	while(T--){

		int N;
		scanf("%d",&N);

		long long distance[N];
		long long total_dist = 0;
		for(int i = 0; i < N; i++){
			scanf("%lld",&distance[i]);
			total_dist += distance[i];
		}		
		
		long long M;
		scanf("%lld",&M);
		long long ans = 0;		

		if(M > total_dist){
			M = M % total_dist;
		}
		if(M == 0){
			M = total_dist;
		}
			
		for(int i = 0; i < N; i++){
			if(M > distance[i]){
				M = M - distance[i];
			}else{
				ans = i+1;
				break;	
			}
		}	
		if(M == 0){
			printf("0\n");
		}else{
			printf("%lld\n",ans);	
		}

	}
}
