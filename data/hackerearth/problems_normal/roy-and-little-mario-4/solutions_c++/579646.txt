#include <iostream>
using namespace std;
#define mod 1000000007
long long trib[100005];
void dp()
{
	trib[0]=1,trib[1]=1,trib[2]=2;
	for(int i=3;i<100005;i++)
	{
		trib[i]=((trib[i-1]%mod)+(trib[i-2]%mod)+(trib[i-3]%mod))%mod;
	}
}
int main()
{
    int t,n,i;
    cin>>t;
    dp();
    while(t--)
    {
    	cin>>n;
    	cout<<trib[n]<<endl;
    }
    return 0;
}
