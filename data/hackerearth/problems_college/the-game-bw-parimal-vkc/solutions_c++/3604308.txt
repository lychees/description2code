#include<bits/stdc++.h>

using namespace std;
typedef  long long LL;

#define mod 1000000007


char a[2000000];

int main()
{
	int i;
   	scanf("%s",a);
    int l =strlen(a);
    LL out = a[0] - '0'; 
    LL temp = a[0] - '0'; 
    for( i=1 ; i< l ; i++)
    {
        temp = temp*10 + (i+1)*(a[i] - '0');
        temp = temp%mod;
        out = (out +temp)%mod;
    }
        printf("%lld\n",out);
    
    return 0;
}
