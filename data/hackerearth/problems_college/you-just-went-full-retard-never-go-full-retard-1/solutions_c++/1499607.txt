#include <bits/stdc++.h>
using namespace std;

bool isPrime(const unsigned long int & number)
{
    if (number < 2){
            return false;
    }
    for (int i = 2; i < sqrt(number+1); i++){
        if (number%i == 0){
            return false;
        }
    }
    return true;
}

int main()
{
    int x=0;
    unsigned long int n=0;
    cin >> x;
    while (x>0){
    	cin >> n;
    	if (isPrime(n))
    		cout << "YES\n";
    	else
    		cout << "NO\n";
    	x--;
    }
    return 0;
}
