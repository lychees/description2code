#include <iostream>
using namespace std;

int main()
{
    long largest=0, larger=0, smaller=0, smallest=0;
    int n;
    cin>>n;
    int k;
    cin>>k;
    n--;
    if(k>0)
    	largest=k;
    else
    	smallest=k;
    cin>>k;
    n--;
    if(k>0)
    {
    	if(k>largest)
    	{
    		larger = largest;
    		largest = k;
    	}
    	else
    		larger = k;
    }
    else
    {
    	if(k<smallest)
    	{
    		smaller = smallest;
    		smallest = k;
    	}
    	else
    		smaller = k;
    }
    if(n==0)
    {
    	if(smallest<0 && largest>0)
    		cout<<largest*smallest;
    	else
    	{
    		if(largest>0)
    			cout<<largest*larger;
    		else
    			cout<<smallest*smaller;
    	}
    }
    else
    {
	    while(n--)
	    {
	    	cin>>k;
	    	if(k>0)
	    	{
	    		if(k>largest)
		    	{
		    		larger = largest;
		    		largest = k;
		    	}
		    	else if(k>larger)
		    		larger = k;
	    	}
	    	else
	    	{
	    		if(k<smallest)
		    	{
		    		smaller = smallest;
		    		smallest = k;
		    	}
		    	else if(k<smaller)
		    		smaller = k;
	    	}
	    }
	    long p = largest*larger;
	    long q = smallest*smaller;
	    if(p>q)
	    	cout<<p;
	    else
	    	cout<<q;
    }
    return 0;
}
