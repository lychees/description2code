#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long long int n,sum=1,pre=1;
    cin>>n;
    long long int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    for(int i=0;i<n-1;i++)
    {
    	if(abs(a[i+1]-a[i])==1)
    	{
    		sum++;
    		if(sum>pre)
    		pre=sum;
    	}
    	else
    	sum=1;
    }
    cout<<pre;
    return 0;
}
