import itertools
cn = int(raw_input())
ca = map(int, raw_input().split())
ca.append(int(0))
case = int(raw_input())
while case > 0:
    n = int(raw_input())
    if n in ca:
        print("I Like You")

    else:

        a = [i for i in ca if i < n]
        k = list(itertools.combinations(a, 2))
        n = [(x, y) for (x, y) in k if x+y == n]
        if len(n):
            print("I Like You")
        else:
            print("I Hate You")
    case -= 1
