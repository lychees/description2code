#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
char s[100016];
int n;
cin>>n;
int p=0;
memset(s,0,sizeof(s));
if(n%3!=0)
{
for(int i=0;i<n;i++)
{
s[i]='5';
}
for(int j=n-1;j>=0;j--)
{
int x=0,y=0;
s[j]='3';
for(int i=0;i<n;i++)
{
if(s[i]=='5')
x++;
if(s[i]=='3')
y++;
}
if(x%3==0&&y%5==0)
{
p=1;
break;
}
}
if(p==1)
cout<<s<<endl;
else
cout<<"-1"<<endl;
}
else
{
for(int i=0;i<n;i++)
                cout<<5;
cout<<endl;
}
}
return 0;
}
