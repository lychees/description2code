#include <iostream>
#include<ctype.h>
using namespace std;
int main()
{
    int n,i,j;
    char ch;
    while(cin>>ch){
    	cin>>n;
    	if(n==1 || (toupper(ch)+n-1)>'Z'){
    		cout<<"INVALID"<<endl;
    		cout<<"\n";
    		continue;
    	}
    	for(i=0;i<n;i++){
    		for(j=0;j<=i;j++){
    			cout<<ch<<" ";
    		}
    		ch++;
    		cout<<endl;
    	}
    		cout<<"\n";
    }
    return 0;
}
