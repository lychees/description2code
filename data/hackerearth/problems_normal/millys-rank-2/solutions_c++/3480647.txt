#include <iostream>
using namespace std;

int main()
{
   long long int t,n,x,y,leftx,rightx,lefty,righty,ans;
   cin>>t;
   while(t--){
   	cin>>n>>x>>y;
   	leftx=x+1;
   	rightx=n;
    lefty=n-y;
    righty=n;
    if(leftx>lefty){
    ans=n-leftx+1;	
    } 
   else{
   	ans=n-lefty+1;
   }	
   cout<<ans<<endl;
   	
   }
    return 0;
}
