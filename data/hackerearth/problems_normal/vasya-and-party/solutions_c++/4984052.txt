#include <bits/stdc++.h>
using namespace std;
#define MAX 100001
bool visited[MAX];
int iq[MAX];
int res;
int maxiq;
vector<vector<int> >graph(MAX);
int dfs(int u)
{
	
	visited[u]=1;
	if(maxiq<iq[u])
	{
		maxiq=iq[u];
		res=1;
	}
	
	else if(maxiq==iq[u])
	{
		res++;
	}
	//cout<<u<<" "<<res<<" "<<maxiq<<endl;
	for(int v=0;v<graph[u].size();v++)
	{
		if(!visited[graph[u][v]])
		{
			dfs(graph[u][v]);
		}
	}
	return 0;
}
int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n;i++)
	cin>>iq[i];
	for(int i=0;i<m;i++)
	{
		int a,b;
		cin>>a>>b;
		graph[a].push_back(b);
		graph[b].push_back(a);
	}
	long long int temp=1;
	for(int i=1;i<=n;i++)
	{
		res=0;maxiq=0;
		if(!visited[i])
		{
			dfs(i);
			temp=((temp)*(res))%1000000007;
			//cout<<"Next"<<endl;
		}
	}
	cout<<temp<<endl;
    return 0;
}
