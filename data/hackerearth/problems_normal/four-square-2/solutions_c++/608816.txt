#include <iostream>
#include <cstdio>
#include <cmath>
#define lli long long int
#define s(x) scanf("%lld", &x)

using namespace std;

int main()
{
    lli n,m,tcase,sum,i,j,k,temp,pos;
    
    s(tcase);
    
    while (tcase--) {
    	s(n);
    	
    	k = (lli)sqrtl((long double)n);
    	
    	sum = 0;
    	
    	for(i = 1; i <= k ; ++i) {
    		if (n%i == 0) {
    			if (i&1)
    			sum = sum +i;
    			if ((n/i)&1)
    			sum = sum + n/i;
    		}
    	}
    	
    	if (k*k == n) {
    		if (k&1)
    			sum = sum-k;
    	}
    	
    	if (n&1)
    		sum = sum*8;
    	else
    		sum = sum*24;
    	
    	printf("%lld\n", sum);
    }
    return 0;
}
