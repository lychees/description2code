#include <bits/stdc++.h>
#include <iostream>
#define ll long long
using namespace std;
int main()
{
    ll t,n,t1,t2;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll d1[n+1],d2[n+1],t1,t2;
        for(ll i=1;i<=n;i++)
        {
            cin>>t1>>t2;
            d1[t1]=i;
            d2[t2]=i;
        }
        ll dp[n+1][n+1],ans=1;
       
        for(ll i=0;i<=n;i++)
        {
            for(ll j=0;j<=n;j++)
            {
                if(i==0||j==0)
                 dp[i][j]=0;
                else if(d1[i]==d2[j])
                 dp[i][j]=dp[i-1][j-1]+1;
                 else
                 dp[i][j]=max(dp[i-1][j],dp[i][j-1]);
            }
        }
     cout<<max(ans,dp[n][n])<<endl;   
    }
    return 0;
}
