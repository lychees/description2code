#include<iostream>
using namespace std;
int main()
{

    int t,n,i;
    long long int mi,s;
    cin>>t;
    while(t--)
    {s=0;
    mi=10000000000;
        cin>>n;

        long long r[n];
        for(i=0;i<n;i++)
        {cin>>r[i];
        if(r[i]<mi)
            mi=r[i];

        s+=r[i];
        }

        s+=mi*(n-2);
        cout<<s<<'\n';
    }
    return 0;
}
