#include<iostream>

using namespace std;

int main()
{
    long long int t; 
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        long long int a[26]={0};
        for(int i=0;i<26;i++)
            cin>>a[i];
                            
        long long int q = (12*n)/26;
        long long int r = (12*n)%26;
        long long int f[26] ={0};
        
        for(int i=0;i<26;i++)
            f[i] = q;
        for(int i=0; i<r; i++)
            f[i] += 1;
        
              
        long long int result = 0;
        
        for(int i=0;i<26;i++)
        {
            result += f[i]*a[i] + (f[i]*(f[i]-1))/2;
        }        
        cout<<result<<endl;      
        
        
    }
    return 0;
}
