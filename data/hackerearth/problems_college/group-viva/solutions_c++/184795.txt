#include<iostream>
using namespace std;
#define ll long long
int mark[200000];
ll p[200000];
ll gcd(ll a,ll b)
{
    if(a<b)
    {
        ll t=a;
        a=b;
        b=t;
    }
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    ll i,j,t,n,r,m;
    for(i=0;i<200000;++i)
        mark[i]=0;
    i=2;
    int np=0;
    while(i<200000)
    {
        if(mark[i]==0)
        {
            p[np++]=i;
            j=i;
            while(j<200000)
            {
                mark[j]=1;
                j+=i;
            }
        }
        i++;
    }

    cin>>t;
    while(t--)
    {
        int num=1;
        cin>>n>>r;
        r--;
        for(i=0;i<np;++i)
            if(r%p[i]==0)
        {
            int d=0;
            while(r%p[i]==0)
            {
                d++;
                r/=p[i];
            }
            num*=d+1;

        }
        if(r>1)
            num*=2;

        num=n-num;
        ll tm=gcd(num,n);
        cout<<num/tm<<"/"<<n/tm<<"\n";
    }

}
