#include <iostream>
using namespace std;
int check(string s)
{
	int n=s.length();
	int count[27]={0};
	for(int i=0;i<n;i++)
	{
		count[s[i]-'a']++;
	}
	int odd=0;
	for(int i=0;i<=25;i++)
	{
		if(odd>1)
		return 0;
		else if(count[i]%2==1)
		odd++;
	}
	return 1;
}
int main()
{
    string s;
    cin>>s;
    if(check(s))
    cout<<"YES";
    else
    cout<<"NO";
    return 0;
}
