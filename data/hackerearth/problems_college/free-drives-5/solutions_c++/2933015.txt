#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int tc;
   scanf("%d",&tc);
   while(tc--)
   {
   	int a;
   	long long ans;
   	scanf("%d",&a);
   	if(a%2==0)
   	{
   		ans=a/2;
   		ans=ans*(a-1);
   	}
   	else
   	{
   		ans=(a-1)/2;
   		ans=ans*a;
   	}
   	printf("%lld\n",ans);
   }
    return 0;
}
