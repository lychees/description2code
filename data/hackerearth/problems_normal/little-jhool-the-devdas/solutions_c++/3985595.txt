//Header files used...
#include<bits/stdc++.h>

using namespace std;
int main()
{
	int n,i,sum =0,t=0,m=99;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		sum+= a[i];
		if(a[i]%2==1 && a[i]<m)
		{
			m=a[i];
			
		}
		if(a[i]%2==1)
			t=1;
	}
	if(sum%2==1)
		cout<<sum;
	else
	{
		if(t==0)
			cout<<":(";
		else
			cout<<sum-m;	
	}
}
