maxscore = None
best_playername = None
player_dict = {}
no_players, no_rounds = map(int, raw_input().split())
for _ in range(no_rounds):
	player_name, score = raw_input().split()
	score = int(score)
	if player_dict.get(player_name, "not") == "not":
		player_dict[player_name] = score
	else:
		player_dict[player_name] += score
		score = player_dict[player_name]
	if maxscore is None or maxscore < score:
		best_playername = player_name
		maxscore = score
print best_playername
