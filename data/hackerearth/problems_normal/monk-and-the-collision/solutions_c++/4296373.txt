#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll hashfunc(ll no)
{
    ll index;
    index= no%10;
    return index;
}
  int main()
{ ll t,i;
cin>>t;
while(t--)
{ll n,x;
    cin>>n;
    ll arr[10];
    memset(arr,0,sizeof(arr));
    for(i=0;i<n;i++)
    {
        cin>>x;
        ll in=hashfunc(x);
        arr[in]++;
    }
    ll c=0;
    for(i=0;i<=9;i++)
{
    if(arr[i]>1)
        c+=arr[i]-1;
}
    cout<<c<<endl;c=0;
}

}
