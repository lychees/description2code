#include <iostream>
using namespace std;

int main()
{
	int T,L,max;
	cin>>T;
	max=0;
	cin>>L;
	int array[L-T+1][L-T+1];
	for(int i=T;i<=L;i++)
	{
		for(int j=T;j<=L;j++)
		{
			array[i-T][j-T]=((~i)&j)|(i&(~j));
		}
	}
	for(int i=0;i<=L-T;i++)
	{
		for(int j=0;j<=L-T;j++)
		{
			max=(max<array[i][j])?array[i][j]:max;
		}
	}
	cout<<max<<"\n";
	
    return 0;
}
