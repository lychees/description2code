#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int T;
    cin>>T;
    while(T--)
    {
    	map<char,int,greater<int>> m;
    	int n;
    	string s;
    	cin>>n>>s;
    	for(auto i : s)
    	{
    		if(m.find(i)!=m.end())
    		{
    			m[i]+=1;
    		}
    		else
    		{
    			m.insert(make_pair(i,1));
    		}
    	}
    	int max1 = 0;
    	
    	for(auto i :m)
    	{
    		max1=max(i.second,max1);
    		
    	}
    	if(max1%2==0)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    
    return 0;
}
