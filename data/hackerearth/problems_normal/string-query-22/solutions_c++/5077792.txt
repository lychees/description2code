#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin >> s;
    vector<int> V(s.length());
    unordered_map<char,vector<int>> my_map;
    for(int i=0; i<s.length(); i++){
    	my_map[s[i]-'a'].push_back(i);
    	V[i]=s[i]-'a';
    };
    
    int m;
    cin >> m;
    for(int i=0; i<m; i++){
    	int pos;
    	char c;
    	cin >> pos >> c;
    	if(my_map.size()==1){
    		V[pos-1]=-1;
    		continue;
    	};
    	vector<int> &curr= my_map[c-'a'];
    	int count=0;
    	for(int j=0; j<curr.size(); j++){
    		
    		if(curr[j]!=-1) count++;
    		//cout << "asjn" << endl;
    		if(count==pos) {
    			V[curr[j]]=-1;
    			curr[j]=-1;
    			break;
    		};
    	};
    	
    };
    string ans="";
    for(int i=0; i<V.size(); i++){
    	if(V[i]!=-1) ans.push_back(V[i]+'a');
    };
    cout << ans ;
    return 0;
}
