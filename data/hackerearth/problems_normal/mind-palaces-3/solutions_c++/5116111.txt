#include <iostream>
using namespace std;

int main()
{
    int n,m,q,w,s,e;
    cin>>n;
    cin>>m;
    int a[n][m];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		cin>>a[i][j];
    		
    	}
    }
    cin>>q;
    while(q--)
    {
    	cin>>w;
    	s=0;
    	e=m-1;
    	while(s<n&&e>=0&&a[s][e]!=w)
    	{
    		if(a[s][e]>w)
    		{
    			e--;
    		}
    		else
    		{
    			s++;
    		}
    	}
    	if(s>=n||e<0)
    	{
    		cout<<"-1"<<" "<<"-1"<<endl;
    	}
    	else
    	{
    		cout<<s<<" "<<e<<endl;
    	}
    }
    return 0;
}
