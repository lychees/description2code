#include <iostream>
using namespace std;

#define N 1000000

char in[N];
string s;

int gcd(int a, int b){
	if(a < b){
		return gcd(b, a);
	}
	if(b == 0){
		return a;
	}	
	return gcd(b, a % b);
}

inline void solve(){
	int i, l, j, k;
	scanf("%s", in);	
	s = in;
	l = s.length();
	k = s[0] - '0';
	for(i = 1; i < l; i++){
		if(s[i] != 0){
			j = s[i] - '0';
			k = gcd(k, j);
		}
	}
	for(i = 0; i < l; i++){
		printf("%d", (s[i] - '0') / k);
	}
	printf("\n");
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	solve();
    }
    return 0;
}
