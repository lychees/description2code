#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
#define mod 1000000007
LL DP[501][501];
int N,H,n,h;
int main()
{
    cin>>N>>H;
    assert(N>0 and H>0);
    assert(N<=500 and H<=500);
    for(int h = 0; h<H; h++)
    {
        DP[0][h] = 1;
        DP[1][h] = 1;
    }
    DP[1][0] = 0;
    for(int n = 2; n<N; n++)
    {
        for(int h = 2; h<H; h++)
        {
            for(int l = 0; l < n; l++)
            {
                int r = n-(1+l);
                DP[n][h] = (DP[n][h] + DP[l][h-1]*DP[r][h-1]) % mod;
            }
        }
    }
    LL ans =0;
    for(int rt = 1; rt<=N; rt++)
    {
        ans = (ans + ((rt*DP[rt-1][H-1])%mod)*DP[N-rt][H-1])% mod;
    }
    cout<<ans<<endl;
    return 0;
}
