#include <iostream>
using namespace std;

int main()
{
	int t,n,p;
    cin >> t; 
    for(int i=0;i<t;i++)
    {
    	cin >> n;
    	cin >> p;
    	
    	for(int j=0;j<p&&n>3;j++)
    	{
    		
    		if(n%2==0)
    			n=n/2+1;
    		else
    			n=(((int)n/2)+2);
    	}
    	cout << n <<endl;
    }
    return 0;
}
