#include<bits/stdc++.h>

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

const int lmt = 100005;
int in[lmt];

bool ok(int n,int m,int c){
    int prev = in[0];
    int cnt = 1;
    if(cnt>=c) return true;
    for(int i=1;i<n;i++){
        if(in[i]-prev >= m){
            cnt++;
            prev = in[i];
            if(cnt>=c) return true;
        }
    }
    return false;
}

int main(){
    int t,n,c;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&n,&c);
        for(int i=0;i<n;i++)
            scanf("%d",&in[i]);
        sort(in,in+n);
        int l=0,r=in[n-1]-in[0],m;
//        DEBUG(l);
//        DEBUG(r);
        while(l<r){
            m = l + (r-l+1)/2;
//            DEBUG(m);
            if(ok(n,m,c))
                l = m;
            else
                r = m-1;
//            DEBUG(l);
//            DEBUG(r);
        }
        printf("%d\n",l);
    }
    return 0;
}
