import collections

def how_many_words(n):
    i = 1
    while True:
        if (i*(i+1))/2 > n:
            return i - 1
        i += 1
        
for _ in range(input()):

    r = map(str,raw_input())
    a = collections.Counter(r).values()
    b = collections.Counter(r).keys()
    ans = 0
    for i in a:
        ans += how_many_words(i)
    if ans >= 4:
        print "YES"
    else:
        print "NO"
