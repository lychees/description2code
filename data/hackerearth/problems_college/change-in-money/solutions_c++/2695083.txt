    #include <stdio.h>
    long long int dp[500][500];
     
    long long int count(int s[], int m , int n )
    {
    	if(dp[m][n]!=0)
    	return dp[m][n];
    	if(n==0)
    	return 1;
    	if(n<0)
    	return 0;
    	if(m==0 && n>=1)
    	return 0;
    	dp[m][n] = count(s,m-1,n)+count(s,m,n-s[m-1]);
    	return dp[m][n];
    }
     
    int main()
    {
    //	memset(dp,0,sizeof dp);
        int m,n;
        scanf("%d %d",&n,&m);
        int s[m],i;
        for(i=0;i<m;i++)
        {
        	scanf("%d",&s[i]);
        }
        
        printf("%lld",count(s,m,n));
        return 0;
    }
