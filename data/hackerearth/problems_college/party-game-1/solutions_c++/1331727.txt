#include<bits/stdc++.h>
#include<math.h>
using namespace std;
typedef long long int ll;
#define s(x) scanf("%lld",&x)
#define p(x) printf("%lld\n",x)
#define f1(x,r) for(x=0;x<r;x++)
#define f2(x,r) for(x=1;x<=r;x++)
#define f3(x) for(x=0;;x++)
#define f4(x,r) for(x=r;x>0;x--)
int main()
{
    ll t,i,j,k,x,y,n,sum1,sum2;
    s(t);
    f1(i,t)
    {
        ll count1=0,count2=0;
        sum1=0;sum2=0;
        s(x);s(y);s(n);
        ll a[n];
        f1(j,n)
        s(a[j]);
        sort(a,a+n);
        f1(j,n)
        {
            sum1=sum1+a[j];
            sum2=sum2+a[j];
            if(sum1<=x)
                count1++;
            if(sum2<=y)
                count2++;
        }
        if(count1>count2)
            cout<<"Chef Won"<<endl;
        if(count1<count2)
            cout<<"Jhool Won"<<endl;
        if(count1==count2)
            cout<<"Tie"<<endl;
    }
}
