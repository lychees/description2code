ll sq(ll x)
{

ll bmod(ll num,ll pw,ll md)
{
    if(pw==0)
    return 1;
    else if(pw%2==0)
    return sq(bmod(num,pw/2,md))%md;
    else
    return (num*bmod(num,pw-1,md))%md;
}
