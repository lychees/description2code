#include<bits/stdc++.h>
#define MAX 10000001
using namespace std;
long long sp[MAX];
long long phi[MAX];
int main(){
	sp[1]=1;
	for (long i=2;i<MAX;i++)
		if (sp[i]==0)
			for (long j=i*2;j<MAX;j+=i)
				sp[j]=i;
	int t;
	scanf("%d",&t);
	long num,p,tmp,ans,q;
	phi[1]=1;
	for(num=2;num<MAX;num++)
    {
        p=sp[num];
        q=num;
        if(p==0)
        {
            phi[num]=num-1;
        }
        else
        {
            while(q%p==0)
            {
                q/=p;
            }
            phi[num]=phi[q]*(num/q)*(p-1)/p;
        }
    }
    for(long i=2;i<MAX;i++)
        phi[i]=phi[i-1]+phi[i];
    while(t--)
    {
        long n;
        scanf("%ld",&n);
        printf("%lld\n",phi[n]);
    }
    return 0;
}
