typedef long long ll;

#define mod 1000000007
#define pb push_back

int prime[1010001];

void markMultiples(int a){
    int i=2, num;
    while((num = i*a) <= 1010000){
        prime[num] = 1;
        i++;
    }
}

void precalc(){
    prime[0]=1;
    prime[1]=1;
    for(int i=2;i<=101000;i++){
        if(prime[i]==0){
            markMultiples(i);
        }
    }
}

/* Iterative function to reverse digits of num*/
int reversDigits(int num)
{
    int rev_num = 0;
    while(num > 0)
    {
        rev_num = rev_num*10 + num%10;
        num = num/10;
    }
    return rev_num;
}

int main(){
    ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    memset(prime, 0, sizeof(prime));
    precalc();
    int T;
    cin>>T;
    ll N;
    while(T--){
        cin>>N;
        while(N<1010000){
            if(prime[N]==0){
                if(N<9 || (N == reversDigits(N))){
                    cout<<N<<endl;
                    break;
                }
            }
            N++;
        }
        if(N>=1010000)
            cout<<0<<endl;
    }


    return 0;
}
