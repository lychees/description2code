#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define sc(x) scanf("%d",&x)
int arr[500005];
int main()
{
    int t,n,ans,i,mn,mm,c,pn,pm;
    sc(t);
    while(t--)
    {
    	sc(n);
    	for(i=0;i<n;i++)
    	{
    		sc(arr[i]);
    	}
    	ans=0;
    	mn=arr[0];
    	mm=arr[0];
    	pn=0;
    	pm=0;
 		c=1;
    	for(i=1;i<n;i++)
    	{
    		if(mn>=arr[i])
    		{
    			mn=arr[i];
    			pn=i;
    		}
    		else if(mm<=arr[i])
    		{
    			pm=i;
    			mm=arr[i];
    		}
    		if(abs(mn-mm)<=1)
    		{
    			c++;
    		}
    		else
    		{
    			ans=max(ans,c);
    			if(arr[i]==mn)
    			{
    				i=pm+1;
    			}
    			else
    			{
    				i=pn+1;
    			}
    			c=1;
    			mn=arr[i];
    			mm=arr[i];
    			pm=i;
    			pn=i;
    		}
    	}
    	if(ans<c)
    	{
    		ans=c;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
