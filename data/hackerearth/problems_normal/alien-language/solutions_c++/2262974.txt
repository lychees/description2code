#include <iostream>
#include<string>
#include<cstring>
using namespace std;

int main()
{int t;
string s1,s2;
    cin>>t;
    while(t--)
    {bool b=false;
    	cin>>s1;
    	cin>>s2;
    bool mp1[26];
    bool mp2[26];
    memset(mp1,0,26);
    memset(mp2,0,26);
   	for(int i=0;i<s1.length();i++)
    	mp1[s1[i]-'a']=1;
    for(int i=0;i<s2.length();i++)
    	mp2[s2[i]-'a']=1;
   for(int i=0;i<26;i++)
   {
   		if(mp1[i]&&mp2[i])
   			b=true;
   }
   if(b)
   	cout<<"YES"<<endl;
   else cout<<"NO"<<endl;
    }return 0;
}
