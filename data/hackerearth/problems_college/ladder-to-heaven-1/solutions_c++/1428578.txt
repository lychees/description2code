#include <iostream>
using namespace std;

int main()
{
    long long t;
    long long x,y,sum,limit,evensum;
    
    
    
    sum=0;
    cin>>t;
    while(t--)
    {
    	evensum=0;
    	sum=0;
    
    x=0;
    y=1;
    	
    	cin>>limit;
     while( sum<=limit ) {
     	if (sum%2==0)
           evensum+=sum;
        sum=x+y;
        x=y;
        y=sum;
        
    }
    	cout<<evensum<<endl;
    }
    return 0;
}
