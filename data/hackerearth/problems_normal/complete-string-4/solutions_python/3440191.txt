'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
N = input()
constants = 'abcdefghijklmnopqrstuvwxyz'
for n in range(N):
	s = raw_input()
	if len(s) >= 26 and len([i for i in constants if i in s]) == 26:
		print 'YES'
	else:
		print 'NO'
