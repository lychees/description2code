#include <iostream>
using namespace std;

int main()
{
    int t, n, k, i, j, count;
    cin>>t;
    for ( ; t; t--) {
    	cin>>n>>k;
    	int arr[n][n]; count = 1;
    	for (i = 0; i < n; i++) {
    		for (j = 0; j < n; j++) {
    			arr[i][j] = count++;
    		}
    	}
    	count = k - 1;
    	for (i = (n - 2); i >= 0; i--) {
    		arr[i][k-1] = arr[i][k - 1] + count;
    		count = count + (k - 1);
    	}
    	count = 0;
    	for (i = 0; i < n; i++) {
    		count = count + arr[i][k-1];
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
