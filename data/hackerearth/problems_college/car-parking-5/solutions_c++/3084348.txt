#include <iostream>
#include<string.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int n =  str.length();
    	int arr[100001]={};
    	
    	for(int i=0; i<n; i++)
    	{
    		if(str[i]=='B')
    		{
    			arr[i]=1;
    			arr[i-1]=1;
    			arr[i-2]=1;
    		}
    		if(str[i]=='D')
    		{
    			arr[i]=1;
    		}
    		if(str[i]=='S')
    		{
    			arr[i-1]=1;
    			arr[i]=1;
    			arr[i+1]=1;
    		}
    	}
    	int count = 0;
    	for(int i=0; i<n; i++)
    	{
    		if(arr[i]==0)
    		count++;
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
