#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, n, a[26];
    string s;

    scanf("%d", &t);

    while (t--) {
        bool flag = false;
        memset(a, 0, sizeof(a));
        scanf("%d", &n);
        for (int i=0; i<n; i++) {
            cin >> s;
            int x = s[0]-'a';
            if (a[x]) {
                flag = true;
            } else {
                a[x] = 1;
            }
        }
        if (flag) {
            printf("NO\n");
        } else {
            printf("YES\n");
        }
    }

    return 0;
}
