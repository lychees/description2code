#include<bits/stdc++.h>
using namespace std;
bool prime(int n)
{
    if(n==0 || n==1)
        return 0;
    if(n==2 || n==3)
        return 1;
    int i;
    for(i=2;i*i<=n;i++)
        if(n%i==0)
            return 0;
    return 1;
}
int main()
{
    int t;
    scanf(" %d",&t);
    while(t--)
    {
        int n;
        scanf(" %d",&n);
        int p[10];
        int c[10];
        int i;
        int Vc=0,Vp=0;
        for(i=0;i<n;i++)
        {
            scanf(" %d",&c[i]);
        }
        for(i=0;i<n;i++)
        {
            scanf(" %d",&p[i]);
        }
        int vic=0;
        for(i=0;i<n;i++)
        {
            if(!prime(c[i]) && prime(p[i]))
            {
                if(c[i]>p[i])
                    Vc+=c[i]-p[i];
                else
                    Vp+=p[i]-c[i];
            }
            else if(prime(p[i]))
            {
                Vp+=p[i];
            }
            else if(!prime(c[i]))
            {
                Vc+=c[i];
            }
            else
            {
                if(c[i]>p[i])
                    Vc+=c[i]-p[i];
                else if(c[i]<p[i])
                    Vp+=p[i]-c[i];
            }
        }
        if(Vc>Vp)
        {
            printf("The Composites\n");
        }
        else if(Vc==Vp)
        {
            printf("Tie\n");
        }
        else
        {
            printf("The Primes\n");
        }
    }
    return 0;
}
