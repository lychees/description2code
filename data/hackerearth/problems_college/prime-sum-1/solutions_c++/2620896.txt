#include <iostream>
using namespace std;
#define ll long long
ll pr(ll a)
{	if(a==4||a==6||a==8||a==9||a==10)
    	return 2;
    else if(a<10)
    	return 1;
}

int main()
{	ll t;
    cin>>t;
    while(t--)
    {	ll a,s;
    	cin>>a;
    	if(a<=10)
    		cout<<pr(a)<<"\n";
    	else
    	{	if(a%7==0)
    			s=0;
    		else if(a%7==1)
    			s=1;
    		else if(a%7==2)
    			s=1;
    		else if(a%7==3)
    			s=1;
    		else if(a%7==4)
    			s=2;
    		else if(a%7==5)
    			s=1;
    		else if(a%7==6)
    			s=2;
    		cout<<s+a/7<<"\n";
    	}
    }
    return 0;
}
