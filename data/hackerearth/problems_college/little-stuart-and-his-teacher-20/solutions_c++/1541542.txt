#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}

int euler(int val){
    int ans = val; 
    for(int i=2;i*i<=val;i++) { 
        if(val%i==0)ans-=ans/i; 
        while(val%i==0)val/=i; 
    } 
    if(val>1)ans-=ans/val; 
    return ans; 
}
long long int arr[1000001];
int main(){
	int t;
	t=scan();
	arr[0]=0;
    arr[1]=0;
    arr[2]=0;
	for(int i=3; i<1000000; i++){
		arr[i]=arr[i-1];
		long long int val= i-1-euler(i);
		arr[i]= arr[i] + val;
	//	cout<<arr[i]<<endl;
	}
  //  cout<<euler(3)<<" "<<euler(4)<<" "<<euler(5)<<endl;
	while(t--){
		int n;

		//ut<<arr[n]<<endl;
	}
}
