#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		vector< pair<char,int> > res1;
		vector< pair<char,int> > res2;
		for(i=0;i<s1.length();i++)
			if(s1[i]!='#')
				res1.push_back(make_pair(s1[i],i));
		for(i=0;i<s2.length();i++)
		  if(s2[i]!='#')
				res2.push_back(make_pair(s2[i],i));
		int flag=1;
		for(i=0;i<res1.size();i++)
		{
			if(res1[i].first!=res2[i].first)
			{
				flag=0;
				break;
			}
			else
			{
				if(res1[i].first=='A'&& res1[i].second<res2[i].second)
				{
					flag=0;
					break;
				}
				else if(res1[i].first=='B'&&res1[i].second>res2[i].second)
				{
					flag=0;
					break;
				}
				
			}
		
		}
		if(flag)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
		
	}
	return(0);
}
