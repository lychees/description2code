#include <iostream>
using namespace std;
int a[1000000],n,c,flag=0,sum,m=0;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {cin>>a[i];
     if(!flag)
      if(a[i]>0)
       {c=i;
       flag=1;
       }
    }
    
    for(int i=c;i<n;i++)
    {sum=0;
    	for(int j=i;j<n;j++)
        { sum+=a[j];
          
         m=max(sum,m);
         if(sum<0)
         break;
        }
    }
    cout<<m;
    return 0;
}
