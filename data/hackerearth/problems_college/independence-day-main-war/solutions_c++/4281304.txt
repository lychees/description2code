#include<bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
   double x[20000],y[20000];
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	double sumx=0,sumy=0;
   	cin>>n;
   	for(int i=0;i<n;i++)
   	cin>>x[i]>>y[i],sumx+=x[i],sumy+=y[i];
   	printf("(%.2lf , %.2lf)\n",sumx/n,sumy/n);
   }
    return 0;
}
