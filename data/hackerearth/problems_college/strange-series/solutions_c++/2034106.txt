#include <iostream>
using namespace std;
long long C(int n, int r) {
    if(r > n / 2) r = n - r; // because C(n, r) == C(n, n - r)
    long long ans = 1;
    int i;

    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }

    return ans;
}

int main()
{  int t,n;
cin>>t;
while(t--){
	cin>>n;
	long long ans=0;
	for(int i=0;i<n;++i)
	ans+=C(n,i);
	cout<<ans<<endl;
	
	}
    return 0;
}
