#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    int N,i,j,status,c=0,a,pr[100][100];
    char word[100];
    cin >> N;
    for (i=0;i<N;i++)
    {
    	cin >> word;
    	for (j=0;j<strlen(word);j++)
    	{
    	a=(int)word[j];
    	pr[a-97][i]=1234;
    	}
    }
    for (i=0;i<26;i++)
    {
    	status=1;
    	for (j=0;j<N;j++)
    	{
    	if (pr[i][j]!=1234)
    	{
    		status=0;
    		break;
    	}
    	}
    	if (status) c++;
    }
    cout << c;
    return 0;
}
