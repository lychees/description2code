#include<bits/stdc++.h>
using namespace std;
long long int ar[1000005];
long long int counter[24];

int main() {
	long long int n,q,k,i,j,c;
	scanf("%lld%lld",&n,&q);
	memset(ar,0,sizeof(ar));
	memset(counter,0,sizeof(counter));

	for(i=2;i<=n;i++) {
		if(!ar[i]) {
			for(j=i;j<=n;j+=i)
				ar[j]++;
		}
		counter[ar[i]]++;
	}
	while(q--) {
		c=0;
		scanf("%lld",&k);
		if(k>23)
			printf("0\n");
		else
			printf("%lld\n",counter[k]);
	}
	return 0;
}
