#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
   long long int i,j,sum1=0,sum2=0,ans=0;
    int arr[100][100];
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n;j++)
    	{cin>>arr[i][j];
    	 if(i==j)
    	 sum1+=arr[i][j];
    	 if(i+j==n-1) sum2+=arr[i][j];
    	}
    }
    
    if(sum1>sum2)
    {ans=sum1-sum2;
    cout<<ans<<endl;}
    else
    {ans=sum2-sum1;  
     cout<<ans<<endl;}
    return 0;
}
