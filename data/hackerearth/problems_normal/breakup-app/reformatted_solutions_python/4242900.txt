from re import findall
counts = [0]*31
for N in range(input()):
    s = raw_input().strip()
    x = map(int, findall('[0-9]+', s))
    m = 2 if s[0] == 'G' else 1
    for i in x:
        counts[i] += m
m = counts.index(max(counts))
c = counts.count(counts[m])
print 'Date' if (m == 19 or m == 20) and c == 1 else 'No Date'
