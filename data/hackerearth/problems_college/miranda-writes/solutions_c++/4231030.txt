#include <bits/stdc++.h>

#define INF 1000000007

typedef long long ll;
typedef unsigned long long ull;

#define all(x) x.begin(), x.end()
#define tr(x,it) for(auto it = x.begin();it!=x.end();++it)
#define sz(a) int((a).size()) 
#define pb push_back 
#define mp make_pair
#define fpresent(c,x) ((c).find(x) != (c).end())  // set,map
#define present(c,x) (find(all(c),x) != (c).end())  //vector
#define X first
#define Y second
#define FOR(i,a,b) for(int i=a;i<=b;++i)
#define NFOR(i,a,b) for(int i=a;i>=b;--i)
#define fill(a,b) memset(a,b,sizeof(a))


int n,m,z;
int dist[400][400];
int x,y,k;
int u[15];
int memo[1<<15][15];
int f(int mask, int now)
{

	if(memo[mask][now]!=-1)
	return memo[mask][now];
	
	if(mask==((1<<k)-1))
	return memo[mask][now]=dist[u[now]][n-1];
	
	int temp=INF;
	FOR(i,0,k-1)
	{
		if(!((mask>>i)&1))
		temp=min(temp,dist[u[now]][u[i]]+f(mask | (1<<i) ,i));
	}
	return memo[mask][now]=temp;
}


int main() 
{
    ios_base::sync_with_stdio(false);cin.tie(0);
    
    	fill(memo,-1);
    	cin>>n>>m>>k;
    	FOR(i,0,n-1)FOR(j,0,n-1)dist[i][j]=INF;
    	FOR(i,1,m)
    	{
    		cin>>x>>y>>z;
    		dist[x-1][y-1]=min(dist[x-1][y-1],z);
    		dist[y-1][x-1]=min(dist[y-1][x-1],z);
    	}
    	FOR(i,0,n)dist[i][i]=0;
    	
    	FOR(k,0,n-1)FOR(i,0,n-1)FOR(j,0,n-1)dist[i][j]=min(dist[i][j],dist[i][k]+dist[k][j]);
    	
    	FOR(i,0,k-1)cin>>u[i],--u[i];
    	int ans = INF;
    	
    	if(k==0){
    		cout<<dist[0][n-1]<<"\n";
    		return 0;
    	}

    	FOR(i,0,k-1){
    		ans = min(ans, f(1<<i,i)+dist[0][u[i]]);
    	}

    	cout<<ans<<"\n";
	return 0;
}
