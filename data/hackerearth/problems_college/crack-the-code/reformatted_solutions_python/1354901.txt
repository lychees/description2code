fib = [1, 1]
for i in range(50):
    fib.append(fib[-1]+fib[-2])
for _ in range(int(raw_input())):
    n, m = map(int, raw_input().split())
    count = 0
    i = 0
    while fib[i] <= m:
        if fib[i] >= n and fib[i] <= m:
            count += 1
        i += 1
    if count % 2 == 0:
        print "VALID"
    else:
        print "INVALID"
