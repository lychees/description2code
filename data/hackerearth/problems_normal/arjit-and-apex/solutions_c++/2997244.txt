#include<iostream>
#include<stdlib.h>
#include<vector>
#include<limits.h>
#include<queue>
#include<stdio.h>
#include<map>
using namespace std;
typedef long long ll;
ll tc,i,j,x,y,n,count,a[1000000],m,u,v,w,good,great,g,h,f;
int main()
{
    cin>>tc;
    multimap<int,int> nw,old;
    while(tc--)
    {
        good=great=0;
        scanf("%lld %lld",&m,&n);
        nw.clear();
        old.clear();
        for(i=0;i<m;i++)
        {
            scanf("%d %d",&u,&v);
            nw.insert(make_pair(u-1,v-1));
        }
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&u,&v);
            old.insert(make_pair(u-1,v-1));
        }
        scanf("%lld %lld",&g,&h);
        multimap<int,int>::iterator it,it1,it2=nw.end(),it3;
        pair<multimap<int,int>::iterator,multimap<int,int>::iterator> p;
        for(it=nw.begin();it!=nw.end();it++)
        {
         //   cout<<"c";
         //   if(old.size()==141)
         //   {
         //       for(it3=old.begin();it3!=old.end();it3++)
         //           cout<<it3->first<<" ";
         //   }
            if(old.find(it->first)==old.end())
                continue;
        //    cout<<".";
            p=old.equal_range(it->first);
         //   cout<<"q";
            f=1;
         //   cout<<"d";
            for(it1=p.first;it1!=p.second;it1++)
            {
           //     cout<<"e";
                if(it1->second==it->second)
                {
                    great++;
                    if(it1==p.first)
                        good++;
                    else
                        it2=nw.end();
                    old.erase(it1);
                    break;
                }
                else if(f==1)
                {
                    good++;
                    it2=it1;
                    f=0;
                }
            }
      //      cout<<"f";
            if(it2!=nw.end() && f==0)
                old.erase(it2);
        }
        if(good>=g && great>=h)
            printf("Great\n");
        else if(good>=g)
            printf("Good\n");
        else
            printf(":(\n");
    }
}
