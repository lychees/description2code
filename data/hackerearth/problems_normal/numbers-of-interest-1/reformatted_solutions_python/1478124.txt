tc = int(raw_input())


def sumOfLastX(output, X):
    total = 0
    for i in range(len(output)-1, len(output)-X-1, -1):
        total += output[i]
    output.append(total)


while (tc):
    tc -= 1
    inputs = list(map(int, raw_input().split()))
    X = inputs[0]
    Y = inputs[1]
    N = inputs[2]

    output = []

    for i in range(X):
        output.append(Y)

    for i in range(N-X):
        sumOfLastX(output, X)

    print(output[N-1])
