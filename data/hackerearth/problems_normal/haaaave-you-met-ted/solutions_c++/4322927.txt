#include <bits/stdc++.h>
using namespace std;
const int N=1e5+5;
int arr[N];
int Count(int x){
	int c=0;
	while(x){
		x=x&(x-1);
		c++;
	}
	return c;
}
int main(){
   int t,n;
   scanf("%d",&t);
   while(t--){
   		scanf("%d",&n);
   		int mn=N;
   		for(int i=0;i<n;i++){
   			scanf("%d",&arr[i]);
			int c=Count(arr[i]);
  			mn=min(c,mn);
   		}	
		printf("%d\n",mn);
   }
   return 0;
}
