#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
#include <string>
#include <set>
#include <sstream>
#include <fstream>

#include <cstdio>
#include <cmath>

struct Student
{
    int age;
    std::string name;
};

bool operator<(const Student& lhs,const Student& rhs)
{
    return lhs.age<rhs.age;
}

int main(void)
{
    short T;
    short age,R;
    std::string name;
    short N;
    std::vector<Student> students;
    
    std::cin>>T;
    while (T--)
    {
        std::cin>>N;
        students.clear();
        while (N--)
        {
            std::cin>>age>>name;
            Student s;
            s.age = age;
            s.name = name;
            students.push_back(s);
        }
        
        std::sort(students.begin(), students.end());
        std::cin>>R;
        
        std::cout<<students[R-1].name<<std::endl;
    }
    return 0;
}
