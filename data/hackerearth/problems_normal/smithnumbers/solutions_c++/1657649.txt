#include<bits/stdc++.h>
using namespace std;

const int N=1e6+10;

bool isPrime[N];
int dp[N];
int DP[N];

int main() {
  for(int i=2;i<N;i++) {
    if(isPrime[i]==false) {
      for(int j=i;j<N;j+=i) {
        int tmp=i,sum=0;
        while(tmp) {
          sum+=tmp%10;
          tmp/=10;
        }
        int res=j;
        while(res%i==0) {
          dp[j]+=sum;
          res/=i;
        }
        isPrime[j]=true;
      }
    }
  }
  for(int i=2;i<N;i++) {
    int cnt=0;
    int tmp=i;
    while(tmp) {
      cnt+=tmp%10;
      tmp/=10;
    }
    if(cnt==dp[i]) {
      DP[i]=1;
    }
  }
  DP[1]=1;
  for(int i=2;i<N;i++) {
    DP[i]+=DP[i-1];
  }
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    int l,r;
    scanf("%d%d",&l,&r);
    printf("%d\012",DP[r]-DP[l-1]);
  }
  return 0;
}
