#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    string s;
    cin>>t;
    getline(cin,s);
    while(t--){
    	
    	getline(cin,s);
    	//cout<<s<<endl;
    	int l=s.length();
    	int sum=0;
    	for(int i=0;i<l;i++){
    		sum+=(int)s[i];
    	}
    	int f=sum/l;
    	if(f%2!=0){
    		for(int i=0;i<l/2;i++){
    			swap(s[i],s[l-i-1]);
    		}
    		cout<<s<<endl;
    	}
    	else{
    		cout<<s<<endl;
    	}
    }
    
    
    return 0;
}
