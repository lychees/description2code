#include<bits/stdc++.h>
using namespace std;
int main()
{
    long t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        long n,k;
        cin>>n>>k;
        long arr[n+1];
        long i;
        for(i=1;i<=n;i++)
            cin>>arr[i];
        long long dp[n+1];
        if(arr[1]<=k)
            dp[1]=arr[1];
        else
            dp[1]=0;
        for(i=2;i<=n;i++)
        {
            if(arr[i]<=k)
                dp[i]=dp[i-1]+arr[i];
            else
                dp[i]=0;
        }
        long long ans=0;
        for(i=1;i<=n;i++)
        {
            if(ans<dp[i])
                ans=dp[i];
        }
        cout<<ans<<"\n";
    }
    return 0;
}
