#include<iostream>
#include<stdio.h>

using namespace std;

long long int cnt(int n)
{
    long long int k=1;
    n--;
    while(n--)
        k=k*2+1;
    return k;
}

bool check(long long int k,long long int n)
{
    long long int i,j;
    i=1;
    j=1;
    if(k==1)
        return false;
    while(j<n)
    {
        i=i*2+1;
        if(i>=k)
            return false;
        j++;
    }
    return true;
}

char find(long long int n,long long int k,bool stat)
{
    char c;
    if(n==1)
    {
        if(stat)
            return 'G';
        else
            return 'C';
    }
    if(n==2)
    {
        if(k==3)
        {
            if(stat)
                return 'C';
            else
                return 'G';
        }
        else
        {
            if(stat)
                return 'G';
            else
                return 'C';
        }
    }
    if(k==cnt(n-1)+1)
    {
        if(stat)
            return 'G';
        else
            return 'C';
    }
    else
    {
        if(!check(k,n-1))//if(k<=cnt[n]/2)
        {
            return find(n-1,k,stat);
        }
        else
            return find(n-1,2*cnt(n-1)+2-k,!stat);
    }
}

int main()
{
    long long int t;
    scanf("%lld",&t);
    while(t--)
    {
        long long int n,k;
        scanf("%lld %lld",&n,&k);
        if(check(k,n))//k>cnt
        {
            //cout<<k<<" "<<cnt[n];
            printf("-1\n");
            continue;
        }
        char c=find(n,k,true);
        printf("%c\n",c);
    }
    //cout<<check(5,2);
    //cout<<check(12,4)<<check(16,4);
    return 0;
}
