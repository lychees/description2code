#include <iostream>
#include<stdio.h>
using namespace std;
typedef long long int ll;
ll bin(ll n) 
{
	ll s = 0;
	while (n>0) {
		s = s+ n%2;
		n = n/2;
		
	}	
	return s;
}
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--) {
		ll n;
		scanf("%lld",&n);
		ll a;
		a = bin(n);
		printf("%lld\n",a);
	}
    
    return 0;
}
