#include <bits/stdc++.h>
using namespace std;

int main()
{
	int nx[1000],ny[1000],mx[1000],my[1000];
	int tc,i,j,count,res,n,m,d;
	cin>>tc;
	while(tc--)
	{
		count=0;
		cin>>n>>m>>d;
		for(i=0;i<n;i++)
		{
			cin>>nx[i];
			cin>>ny[i];
		}
		for(i=0;i<m;i++)
		{
			cin>>mx[i];
			cin>>my[i];
		}
		for(i=0;i<m;i++)
		{
			for(j=0;j<n;j++)
			{
			    res=abs(nx[j]-mx[i])+abs(ny[j]-my[i]);
			    if(res<=d)
			    {
			    	count++;
			    	break;
			    }
			}
		}
		if(count>m/2)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
	}
    return 0;
}
