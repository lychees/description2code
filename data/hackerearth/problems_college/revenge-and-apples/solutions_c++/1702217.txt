#include <bits/stdc++.h>
#include <iostream>
#include <algorithm>
#include <vector>

#include <stack>

#include <stdlib.h>
#include <map>
using namespace std;
#define MAX 2005
 
char str[MAX][MAX];
int dp[MAX][MAX][2];
int rec(int i, int j, int orie, int n, int m)
{
	if (i < 0 || j < 0 || i >= n || j >= m) {
		return 0;
	}
	if (str[i][j] == '#') {
		return 0;
	}
	if (dp[i][j][orie] != -1) {
		return dp[i][j][orie];
	}
	if (orie) {
		if (str[i][j] >= '0' && str[i][j] <= '9') {
			return dp[i][j][orie] = (str[i][j] - '0') + max(rec(i, j + 1, orie, n, m), rec(i + 1, j, orie ^ 1, n, m));
		}
		else {
			return dp[i][j][orie] = max(rec(i, j + 1, orie, n, m), rec(i + 1, j, orie ^ 1, n, m));
		}
	}
	else {
		if (str[i][j] >= '0' && str[i][j] <= '9') {
			return dp[i][j][orie] = ((str[i][j] - '0')) + max(rec(i, j - 1, orie, n, m), rec(i + 1, j, orie ^ 1, n, m));
		}
		else {
			return dp[i][j][orie] = max(rec(i, j - 1, orie, n, m), rec(i + 1, j, orie ^ 1, n, m));
		}
	}
}
 
int main()
{
	int t;
	cin >> t;
	while (t--) {
		memset(dp, -1, sizeof(dp));
		int n, m;
		cin >> n >> m;
		for (int i = 0; i < n; i++) {
			scanf("%s", str[i]);
		}
		int ans = rec(0, 0, 1, n, m);
		cout << ans << endl;
	}
	return 0;
}
