#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        long long n,i,j,y=0;
        cin>>n;
        for(i=5;i<=n;i*=5)
        {
            y+=(n/i);
        }
        cout<<y<<endl;
    }
    return 0;
}
