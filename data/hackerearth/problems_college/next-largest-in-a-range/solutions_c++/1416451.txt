#include <iostream>
#include<climits>
using namespace std;

int main()
{
 //   cout << "Hello World!" << endl;
 	int n,l,r,q;
 	cin>>n>>l>>r>>q;
 	unsigned long long a[n];
 	for(unsigned long long i=0;i<n;i++)
 	cin>>a[i];
 	while(q--)
 	{
 		unsigned long long query;
 		cin>>query;
 		unsigned long long diff=ULLONG_MAX;
 		int flag=0;
 		for(unsigned long long i=l;i<=r;i++)
 		{
 			
 			if(a[i]>query)
 			{
 				flag=1;
 				diff=(diff<(a[i]-query)?diff:(a[i]-query));
 			}
 			
 		}
 		if(flag==0)
 		cout<<"-1"<<endl;
 		else
 		cout<<(query+diff)<<endl;
 		
 	}
 	
    return 0;
}
