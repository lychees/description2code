#include <iostream>
using namespace std;

int main()
{
    long long int t,a,b;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b;
    	int pow=1;
    	int z=0,cnt=0;
    	while(a && b)
    	{
    		int x=a%10;
    		a/=10;
    		int y=b%10;
    		b/=10;
    		if(x+y+z>=10)
 			{   if(x+y+z==10)
 				z=1;
 				else
 				z=(z+x+y)%10;
 				cnt++;
 			}
 			else
 			z=0;
 			
    	}
    	if(cnt==0)
    	cout<<"No carry operation"<<endl;
    	else if(cnt==1)
    	cout<<cnt<<" carry operation"<<endl;
    	else
    	cout<<cnt<<" carry operations"<<endl;
    }
    return 0;
}
