#include<bits/stdc++.h>
using namespace std;

int main()
{

    int t,n;
    cin>>t;
    for (int i=0; i<t; ++i)
    {
        double dist,res,xx,yy,zz,vxx,vyy,vzz,den,time,x,y,z,vx,vy,vz;
        xx=0;
        yy=0;
        zz=0;
        vxx=0;
        vyy=0;
        vzz=0;
        {
            scanf("%d",&n);
            for (int j=0; j<n; ++j)
            {
                cin>>x>>y>>z>>vx>>vy>>vz;
                xx+=x;
                yy+=y;
                zz+=z;
                vxx+=vx;
                vyy+=vy;
                vzz+=vz;
            }
            xx=xx/n;
            yy=yy/n;
            zz=zz/n;
            vxx=vxx/n;
            vyy=vyy/n;
            vzz=vzz/n;
            time=((xx*vxx)+(yy*vyy)+(zz*vzz));
            den=(vxx*vxx)+(vyy*vyy)+(vzz*vzz);
            if(den< 1e-09)
            {
                res=0;
                dist=sqrt(xx*xx + yy*yy + zz*zz);;
            }
            else
            {
                res=time/den;
                res=res*(-1);
                if(res<0)
                {
                    res=0;
                    dist=sqrt(pow(xx+(vxx*res),2) +pow((yy+(vyy*res)),2)+pow((zz+(vzz*res)),2));;
                }
                else
                {
                    dist=sqrt(pow(xx+(vxx*res),2) +pow((yy+(vyy*res)),2)+pow((zz+(vzz*res)),2));;
                }
            }
            cout << "Case #" << (i+1) << ": ";
            printf("%0.8lf %0.8lf\n",dist,res);
        }
    }
}
