#include <cstdio>
#include <iostream>
#include <map>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t,n,i,j,q;
    char st[15];
    string tmp;
    map< string,vector<int> > m;
    map< string,vector<int> > ::iterator mi;
    scanf("%d",&n);
    for(t=1;t<=n;t++)
    {
        scanf("\n%s",st);
        for(i=0;st[i]!='\0';i++)
        {
            tmp="";
            for(j=i;j<st[j]!='\0';j++)
            {
                tmp+=st[j];
                mi=m.find(tmp);
                if(mi==m.end()) m[tmp].push_back(t);
                else if(mi->second.back()!=t) mi->second.push_back(t);
            }
        }
    }
    scanf("\n%d",&q);
    while(q--)
    {
        scanf("\n%d %d ",&i,&j);cin>>tmp;
        mi=m.find(tmp);
        printf("%d\n",upper_bound(mi->second.begin(),mi->second.end(),j)-lower_bound(mi->second.begin(),mi->second.end(),i));
    }
    return 0;
}
