#include <iostream>
#include <algorithm>
using namespace std;
#define ll long long

void sorting(ll &a,ll &b,ll &c){
	ll arr[3];
	arr[0]=a,arr[1]=b,arr[2]=c;
	sort(arr,arr+3);
	a=arr[0],b=arr[1],c=arr[2];
}

int main()
{
    int t;
    ll a,b,c;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c;
    	sorting(a,b,c);
    	if(a>1000000007)
    		a%=1000000007;
    	if(b>1000000007)
    		b%=1000000007;
    	if(c>1000000007)
    		c%=1000000007;
    	ll ans=1;
    	if(b-1<=0){
    		cout<<0<<"\n";
    		return 0;
    	}
    	ans=a*(b-1);
    	if(ans>1000000007)
    		ans%=1000000007;
    	if(c-2<=0){
    		cout<<0<<"\n";
    		return 0;
    	}
    	ans=ans*(c-2);
    	if(ans>1000000007)
    		ans%=1000000007;
    	cout<<ans<<"\n";
    }
    return 0;
}
