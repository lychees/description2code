#include<bits/stdc++.h>

using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a[1001],sum,n,count=0,res[10001],k=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    		cin>>a[i];
   		for(int i=0;i<n;i++)
   		{	
   			for(int j=i+1;j<n;j++)
   			{
   				sum=a[i];
    			if(sum!=a[j])
    			{
    				sum=sum+a[j];
    				if(sum%2==0)
    				{
    					res[k++]=2*sum;	
    				}
    			}
    		}
    	}
    	cout<<k<<endl;
    }
    return 0;
}
