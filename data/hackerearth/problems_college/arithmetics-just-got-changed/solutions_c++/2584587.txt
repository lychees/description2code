#include <iostream>
using namespace std;

int reverse(int x){
	int r,temp = x,rev=0;
	while(temp!=0){
        r = temp%10;
        temp = temp/10;
        rev = rev*10+r;
    }
	return rev;	
}

int main()
{
    int t,a,b,s;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&a,&b);
    	s = reverse(reverse(a)+reverse(b));
    	printf("%d\n",s);
    }
    return 0;
}
