#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long int cont=0;
void cal(long long int i,long long int j)
{
	if(i>j)
	{
		long long int c=i;
		i=j;
		j=c;
	}
	if((j%i==0 && i!=1)||(i==j))cont++;
	else if(i==1)cont++;
	else
	{
		if(2*i<j)
		cont=cont+2;
		else
		cont=cont+1;
		
	//	printf("akshay\n");
		int t=j/i;
		cal(i,j-i*t);
		
	}	
}
int main()
{
long long int  t,i,j,a1,a2,b1,b2,p;
scanf("%lld",&t);
for( p=1;p<=t;p++)
{
	scanf("%lld %lld %lld %lld",&a1,&a2,&b1,&b2);
	long long int  ans=0;
	if(a1<=0)a1=1;
	if(b1<=0)b1=1;
	for(i=a1;i<=a2;i++)
	{
		
		for(j=b1;j<=b2;j++)
		{
			 cont=0;
			 if(i!=j)
			 {
			 	cal(i,j);
			if(cont%2!=0)
			ans++;
			 	
			 }
			
			
			
		}
		
	}
	
	printf("Case #%lld: %lld\n",p,ans);
	
}

}
