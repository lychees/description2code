#include<iostream>
#include<math.h>
using namespace std;
#define mod 1000000007
long long int dp[65][65];
int n,k;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	cin>>n>>k;
	dp[0][0]=1;
	dp[0][1]=1;
	long long int s=0;
	for(int i=1;i<n;i++)
	{
		s=0;
		for(int j=0;j<k;j++)
		{
			s+=dp[i-1][j];
			dp[i][j]+=dp[i-1][j-1];
		}
		dp[i][0]=s;
	}
	/*for(int i=0;i<=n;i++)
	{
		for(int j=0;j<=n;j++)
		{
			cout<<dp[i][j]<<"  ";
		}
		cout<<endl;
	}*/
	long long int x=0;
	long long int y=pow(2,n);
	for(int i=0;i<k;i++)
	{
		x=x+dp[n-1][i];
	}
	x=y-x;
	long long int a=gcd(x,y);
	x=x/a;
	y=y/a;
	cout<<x<<"/"<<y;
}
