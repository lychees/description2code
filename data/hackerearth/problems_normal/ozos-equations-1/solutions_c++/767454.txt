#include <iostream>
#include<math.h>
using namespace std;
typedef unsigned long long ll;
ll su(ll n){
	ll total = 0;
	while(n){
		total+=n%10;
		n/=10;
	}
	return total;
}
void sol(ll n){
	ll x=(ll)sqrt(n);
	x++;

ll b;
ll st ;
if(x>70)st = x-70;
else st = 1;
for(;st<=x;st++)
{if(n%st==0){
	b = st+su(st);
	if(b*st==n){
		printf("%llu\n",st);
		return;
	}
	
	}
}
printf("-1\n");

}
int main()
{
int t;
ll n;
scanf("%d",&t);
while(t--){
scanf("%llu",&n);
sol(n);
}

    return 0;
}
