#include <iostream>
using namespace std;

int main()
{
    long t,b,w,x,y,z;
    cin>>t;
    while(t--)
    {
    	long sum=0,sum1=0;
    	cin>>b>>w>>x>>y>>z;
    	sum+=b*x+w*y;
    	if(x<y) sum1+=(b+w)*x+w*z;
    	else if(x>y) sum1+=(b+w)*y+b*z;
    	else sum1+=sum;
    	if(sum<=sum1) cout<<sum<<endl;
    	else cout<<sum1<<endl;
    }
    return 0;
}
