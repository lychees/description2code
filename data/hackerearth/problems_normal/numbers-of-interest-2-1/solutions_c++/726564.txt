#include <iostream>
#include <cstring>
#include <cstdio>

using namespace std;

#define N 1000001
int x[N];
int y[N];

void init()
{
	memset(x, 0, sizeof(x));
	memset(y, 0, sizeof(y));

	for (int i = 2; i <= N; i++) {
		if (!x[i]) {
			for (int j = i; j <= N; j += i) {
				if (!x[j]) x[j] = i;
			}
		}
	}

	for (int i = 2; i < N; i++) {
        //if (!y[i]) {
            for (int j = i + i; j <= N; j += i) {
                y[j] += i;
            }
        //}
        y[i]++;
	}

    /*for(int i = 0; i < N; i++) {
        cout << x[i] << " ";
    }
    cout << endl;
    */

	for (int i = 1; i < N; i++) {
        x[i] += x[i-1];
        y[i] += y[i-1];
	}

	/*for(int i = 0; i < N; i++) {
        cout << x[i] << " ";
	}
	cout << endl;
	*/
}

int main()
{
    init();
    int t, n;
    cin >> t;
    while(t--) {
        cin >> n;
        cout << (x[n] + y[n]) % n << endl;
    }
    return 0;
}
