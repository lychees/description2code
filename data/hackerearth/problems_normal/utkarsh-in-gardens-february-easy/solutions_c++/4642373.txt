#include<iostream>
#include<bits/stdc++.h>
using namespace std;
const int maxi=2010;
bitset<maxi> matrix[maxi];
int main()
{
	long long int n,ans,t;
	cin >> n;
	for(int i=1;i<=n;i++)
	{
	for(int j=1;j<=n;j++)
	{
	int x;cin >> x;
	matrix[i][j]=x;
	}
	}
	for(int i=1;i<=n;i++)
	{
	for(int j=i+1;j<=n;j++)
	{
	t=(matrix[i] & matrix[j]).count();
	ans+=(t*(t-1))/2;
	}
	}
	cout << ans/2 << '\n';
}
