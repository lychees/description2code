#include <bits/stdc++.h>
using namespace std;
const int MAX = 1000000 + 7;

int main()
{
    
    int T , L , N ;
    int  c[MAX];
    int  a[MAX];
    cin >> T;
    while (T--)
    {
    	 int index = 0;
    	cin >> N;
    	for(int i = 0; i < N; i++)
    	c[i]  = 0;
    	 for(int i = 0; i < N ; i++)
    	 cin >>a[i];
    	 
    	 c[0] = a[0];
    	 
    	 for(int i = 1; i < N ; i++)
    	 {
    	     if(c[index] < a[i])
    	      c[++index] = a[i];
    	     
    	     else 
    	     {
    	     	int low = 0;
    	     	int high = index;
    	     	
    	     	while (low <= high)
    	     	{
    	     		
    	     		int mid = (low + high)/2;
    	     		
    	     		if(c[mid] < a[i])
    	     		low = mid+1;
    	     		else
    	     	   high = mid-1;
    	     		
    	     		
    	     	}
    	     	
    	     c[low] = a[i];	
    	     	
    	     	
    	     	
    	     	
    	     	
    	     	
    	     }
    	     
    	 	
    	 }
    	
    	cin >> L;
    	L--;
    	if(L > index)
    	 cout <<"-1"<<endl;
    	 else
    	 cout <<c[L] <<endl;
    	
    	
    }
    
    return 0;
}
