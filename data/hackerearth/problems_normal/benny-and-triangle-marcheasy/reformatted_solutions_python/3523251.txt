# Import libraries
import sys
import math
import pdb


def getOneLine(inFile):
    if inFile != "":
        return (inFile.readline().rstrip())
    return (sys.stdin.readline().rstrip())


def main():

    # Toggle mode of input
    # inFile = open("input5.txt", 'r')
    inFile = ""

    # number of cases
    input = getOneLine(inFile)
    input = input.split(" ")
    a = int(input[0])
    b = int(input[1])
    p = int(input[2]) / 100.0

    requiredArea = 0.5 * a * b * p
    result = b - math.sqrt(2*requiredArea * b / a)
    result = round(result, 2)
    print("%.2f" % result)


main()
