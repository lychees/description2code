#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    int n, i, count=0, max=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(max<a[i])
    	max=a[i];
    }
    int index[max+1];
    memset(index,0,sizeof(index));
    for(i=0;i<n;i++)
    index[a[i]]++;

    while(1)
    {
    int s=0, pos=max;
    for(i=0;i<=max;i++)
    {
    	if(index[i]!=0)
    	{
    		s+=index[i];
    		if(pos>=i)
    		pos=i;
    	}
    }
    index[pos]=0;
   	if(s==0)
   		break;
    else
        cout<<s<<endl;
    }
return 0;
}
