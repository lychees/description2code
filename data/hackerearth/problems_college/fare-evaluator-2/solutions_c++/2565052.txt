#include <iostream>
using namespace std;

int main()
{
    int sum,d,t,wt;
    cin>>t;
    while(t>0)
    {
    	sum=0;
    	cin>>d>>wt;
    	if(d>60)
    	  cout<<"NA"<<endl;
    	else
    	{
    		if(d==1)
    		cout<<(50+(2*wt))<<endl;
    		else{
    		d-=2;
    		sum=50 + (d*12) +(2*wt);
    		cout<<sum<<endl;}
    	
    	}
    	
    	t--;
    }
    return 0;
}
