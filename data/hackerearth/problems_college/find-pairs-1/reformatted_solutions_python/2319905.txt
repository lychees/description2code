if __name__ == '__main__':
    T = int(raw_input())
    for i in range(0, T):
        t = int(raw_input())
        numlist = list(map(int, raw_input().split()))
        count = 0
        for k in range(0, t):
            for j in range(k+1, t):
                if numlist[k] == numlist[j]:
                    count += 1
        print count+t
