#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


bool check(int A[],int B[],int r,int x,int y)
{int i;
for(i=0;i<=3;i++){
	 if (((x - A[i])*(x-A[i]) + (y - B[i])*(y-B[i])) > r*r) {
            return false;
        }
    
   
}
  return true;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int r,c_x,c_y;
   	cin>>r>>c_x>>c_y;
   	int A[3],B[3],i;
   	for(i=0;i<=3;i++)
   	{
   		cin>>A[i]>>B[i];
   	}
   
   if(check(A,B,r,c_x,c_y))
   cout<<"Yes\n";
   else
   cout<<"No\n";
   }
    return 0;
}
