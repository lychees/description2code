#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		map<long long int,bool> m;
		m[1]=true;
		if(n==2)
	{	cout<<"Yes"<<endl;
		continue;
	}
	int f=0;
 for (int x = 2; x <= sqrt(n); x++) {
 
                int lgx = (int)ceil((log(n) / log(x)));
          
                if (f == 1) {
                    break;
                }
                for (int y = 2; y <= lgx; y++) {
 
                    long num = (long)pow(x, y);
                    if (m.find(n - num)!=m.end() || num+num==n) {
                        f = 1;
                        break;
                    }
                    m[num]=true;
 
                }
 }
		
		
	
	 if (f == 1) {
               cout<<"Yes"<<endl;
            } else {
            cout<<"No"<<endl;
            }
	
	
 
	}
           
            
           
	
	
	
    //cout << "Hello World!" << endl;
    return 0;
}
