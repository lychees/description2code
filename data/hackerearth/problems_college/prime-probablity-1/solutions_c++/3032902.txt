#include <cstdio>
#include <vector>
#include <cmath>
#include <iostream>
using namespace std;

vector<int> A;

 bool check_prime(int k){
    if(k==2)
        return true;
    for(int i=2;i<sqrt(k)+1;i++)
        if(k%i==0)
            return false;
    return true;
 }
void set_primes(){
    for(int i=2;i<1000000;i++)
        if(check_prime(i))
            A.push_back(i);
}

int bin_search(int l,int r,int key){
    int mid=(l+r)/2;
    if(r>l){
        if(A[mid]<key)
            return bin_search(mid+1,r,key);
        else if(A[mid] > key)
            return bin_search(l,mid-1,key);
        else return mid;
    }
    else return mid;
}

int main(){
    int test;
    cin>>test;
    float L,R;
    float l_i,r_i;
    set_primes();

    int size=A.size();
    for(int i=0;i<test;i++){
        cin>>L>>R;
        l_i=(float) bin_search(0,size-1,L);
        r_i=(float) bin_search(0,size-1,R);
        //cout<<l_i<<","<<r_i<<endl;
        if(A[l_i]<L)
            l_i++;
        if(A[r_i] > R)
            r_i--;
        //cout<<l_i<<","<<r_i<<endl;
        float answer= (r_i-l_i+1)/(R-L+1);
        printf("%.6f\n",answer);

    }
}
