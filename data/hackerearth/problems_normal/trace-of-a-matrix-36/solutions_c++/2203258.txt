#include <bits/stdc++.h>

using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	int t;
	cin >> t;
	while (t--) {
		long long int r, c;
		cin >> r >> c;
		long long int n = min(r, c);
		long long trace1 = (n*(2+(n-1)*(r+1))) / 2;
		long long trace2 = (n*(2+(n-1)*(c+1))) / 2;
		cout << trace1 + trace2 << endl;
	}	
	return 0;
}
