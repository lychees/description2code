#include <iostream>
#include <map>
#include <vector>
#include <string>
#include <utility>
using namespace std;

int main()
{
	map<string,int> my_map;
	vector<char> caps;
	int k,n;
	int last = 0;
	cin>>k;
	string temp;
	char res;
	while(k--)
	{
		cin>>temp; 
		my_map.insert(make_pair(temp,1));
	}
	cin>>n;
	for(int i = 0;i<n;i++)
	{	cin>>temp;
		if(my_map.find(temp)==my_map.end())
		{
			res = (temp[0]-97+65);
			caps.push_back(res);
		}
	}
	
	
	for(auto it = caps.begin();it!=caps.end();it++)
	{
		cout<<*it;
		if(it!=caps.end()-1)
		cout<<".";
	}
    return 0;
}
