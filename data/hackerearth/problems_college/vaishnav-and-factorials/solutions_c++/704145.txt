int main()
{
    int t,n,i,j,k,l,carry,fact[999999],cnt;
    cin>>t;
    while(t--)
    {
        cin>>n;
        fact[0]=1;
        l=1;
        carry=0;
        for(i=1; i<=n; i++)
        {
            for(j=0; j<l; j++)
            {
                k=fact[j]*i+carry;
                fact[j]=k%10;
                carry=k/10;
            }
            while(carry)
            {
                fact[l]=carry%10;
                carry/=10;
                l++;
            }
        }
        cnt=0;
        for(i=l-1; i>=0; i--) if(fact[i]==4 ||fact[i]==7) cnt++;
        cout<<cnt<<endl;
    }
    return 0;
}
