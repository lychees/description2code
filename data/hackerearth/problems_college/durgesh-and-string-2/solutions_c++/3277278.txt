#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main() 
{
    int n,i,j;
    cin>>n;
    string str[n];
    int arr[n][2];
    for(i = 0; i < n; i++) {
    	cin>>str[i];
    	arr[i][0] = arr[i][1] = 0;
    	for(j = 0; str[i][j]; j++) {
    		arr[i][str[i][j]=='0'?0:1]++;
    	}
    }
    int p,q;
    cin>>p>>q;
    int x = 1<<n;
    int ans = 0;
    for(i = 0; i < x; i++) {
    	int c,d,y;
    	c = d = y = 0;
    	for(j = 0; j < n; j++) {
    		if(i&(1<<j)) {
    			c += arr[j][0];
    			d += arr[j][1];
    			y++;
    		}
    	}
    	if(c <= q && d <= p) {
    		ans = max(ans,y);
    	}
    }
    cout<<ans<<endl;
	return 0;
}
