#include <cstdlib>
#include <iostream>
#include <deque>
#include <algorithm>
using namespace std;
int mod(int a)
{
    if( a > 0) return a;
    return -1*a;
}
int main()
{
    int ceo, coo , cto,n,temp;
    scanf("%d%d%d",&ceo,&coo,&cto);
    scanf("%d",&n);
    deque <int > A;
    for(int i = 0;i<n;++i){
        scanf("%d",&temp);
        A.push_back(temp);
    }
    sort(A.begin(),A.end());
    if(coo > cto){
       A.push_back(coo);
       A.push_front(cto);
    }
    else{
       A.push_back(cto);
       A.push_front(coo);
    }             
    deque <int >::iterator it;
    it = A.begin();
    int size = A.size();
    if(size%2==0){  
          advance(it,size/2)  ;
    }
    else{
         if(ceo > A[size/2]) advance(it,size/2+1);
         else                  advance(it,size/2);
    }
    A.insert(it,ceo);    
    long long int sum =  0;
    for(int i = 1;i<A.size();++i){
        sum+=(mod(A[i] - A[i-1]));
    }
    printf("%lld",sum);      
    
    return 0;
}
