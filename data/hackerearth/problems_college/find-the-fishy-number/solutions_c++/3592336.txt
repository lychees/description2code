#include<bits/stdc++.h>

typedef long long ll;
using namespace std;

double explog[10]={0,0.3010300,0.1760913,0.1249387366,0.096910013,0.07918124604,0.06694678963,0.05799194697,0.05115252244,0.04575749056};
int main(){
    std::ios::sync_with_stdio(false);
    int t,n,threshold;
    string s;
    cin>>t;
    while(t-->0){
        cin>>n;
        string a[n];
        int frequency[10];
        double divide[10];
        memset(frequency,0,sizeof(frequency));
        for(int i=0; i<n; i++){
            cin>>a[i];
            s=a[i];
            frequency[s[0]-48]++;
            //cout<<s<<endl;
        }
        cin>>threshold;
        bool flag=false;
        for(int i=1; i<10; i++){
           if(frequency[i]>=(threshold*n*explog[i]) || frequency[i]<=(n*explog[i]/threshold)){
                cout<<i<<endl;
                flag=true;
                break;
           }
        }
        if(flag==false) cout<<"-1"<<endl;
    }
    return 0;
}
