#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int a1,a2,x,y,b1,z,b2,h,l=0,i,j,k;
    while(t--)
    {
    	h=0;
    	cin>>a1>>a2>>b1>>b2;
    	for(i=a1;i<=a2;i++)
    	{
    		for(j=b1;j<=b2;j++)
    		{
    		 x=i;
    		 y=j;
    		 k=0;
    		 while(x>0 && y>0)
    		 {
    		 	if(x<y)
    		 	{
    		 		z=y/x;
    		 		if(z!=1)
    		 		y=y-((z-1)*x);
    		 		else
    		 		y=y-x;
    		 	}
    		 	else if(y<x)
    		 	{
    		 		z=x/y;
    		 		if(z!=1)
    		 		x=x-((z-1)*y);
    		 		else
    		 		x=x-y;
    		 	}
    		 	else
    		 	x=x-y;
    		 	k=k+1;
    		 }
    		 if(k%2==0)
    		 h=h+1;
    		}
    	}
    	    		 cout<<"Case #"<<l+1<<": "<<h<<"\n";

    	l=l+1;
    }
    return 0;
}
