k = int(raw_input())
alph = list(raw_input())
totalCount = 0
for index in range(len(alph)):
    count = 0
    countAlp = [0]*26
    for indexJ in range(index, len(alph)):
        if (countAlp[ord(alph[indexJ])-97] == 0):
            count += 1
        countAlp[ord(alph[indexJ])-97] = 1
        if (count == k):
            totalCount += 1
        if (count > k):
            break

print totalCount
