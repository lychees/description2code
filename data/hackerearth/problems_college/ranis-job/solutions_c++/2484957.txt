#include<bits/stdc++.h>
using namespace std;
bool fun(int i,int j)
{
	return (i>j);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int c[n];
    	long long int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin>>c[i];
    	}
    	//cout<<sum<<endl;
    	sort(c,c+n,fun);
    	int count=1;
    	for(int i=0;i<n;i++)
    	{
    		sum-=c[i];
    		//cout<<sum<<endl;
    		if(sum>0)
    		{
    			count++;
    		}
    		else
    		{
    			break;
    		}
    		
    	}
    	cout<<count<<endl;
    }
}
