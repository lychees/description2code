#include <bits/stdc++.h>
using namespace std;

struct entry{
	char name[100];
	int count;
	entry(){
        count=0;
	}
};
bool myfn(pair<string,int> i,pair<string,int> j){
    return (i.second>j.second);
}
int main()
{
    int n,i;
    cin>>n;
    entry a[n];
    map <string,int> mp;
    for(i=0;i<n;i++){
    	cin>>a[i].name;
    	mp[a[i].name]=0;
    	a[i].count++;
    }
    for(i=0;i<n;i++){
        mp[a[i].name]+=a[i].count;
    }

    map <string,int>::reverse_iterator it;
    for(it=mp.rbegin();it!=mp.rend();it++){
        cout<<it->first<<" "<<it->second<<endl;
    }
    return 0;
}
