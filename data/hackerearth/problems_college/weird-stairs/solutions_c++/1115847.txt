#include<iostream>
using namespace std;

 int main()
 {
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
 {
    int n;
    long long w, a, res=0;
    cin>>w>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a;
        if(a==1)
            res = 2*i;
        else if(a==2)
            res = res+1;
    }
    cout<<res+n*w<<endl;
 }
  return 0;
}
