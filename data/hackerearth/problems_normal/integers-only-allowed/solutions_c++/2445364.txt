#include <bits/stdc++.h>

using namespace std;

#define ll long long

ll x[1010], y[1010];

int gcd(int a, int b)
{
	if(b==0)
		return a;
	return gcd(b, a%b);
}

ll find_points(int i, int j)
{
	int p, q;
	p = abs(x[j]-x[i]), q = abs(y[j]-y[i]);
	if(p==0)
		return q-1;
	else if(q==0)
		return p-1;
	int ret = gcd(p, q)-1;
	return ret;
}

int main()
{
	int n, i, j, k;
	scanf("%d", &n);
	for(i=1; i<=n; i++)
		scanf("%lld%lld", &x[i], &y[i]);
	ll area = 0, temp, b = 0;
	for(i=1; i<=n; i++){
		if(i==1)
			j = n;
		else
			j = (i-1);
		if(i==n)
			k = 1;
		else
			k = i+1;

		temp = x[i]*(y[k]-y[j]);
		b+=find_points(i, j);
		b+=find_points(i, k);
		area+=temp;
	}
	b/=2;
	b+=n;
	//cout << b << "\n";
	if(area<0)
		area*=-1;
	double act = (double)area/2.0;
	//cout << act << " ";
	ll ans = act+1-(double)b/2.0;
	printf("%lld\n", ans);
	return 0;
}
