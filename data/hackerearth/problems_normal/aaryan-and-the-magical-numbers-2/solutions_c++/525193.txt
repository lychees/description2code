#include <stdio.h>
int main()
{
  int t;
 scanf("%d",&t);
 long long c=0,mn[t];
 while(c<t)
 {
  scanf("%llu",&mn[c]);
  c++;
 }
 c=0;

   long long pre,aux,count,power,last,n;
  int new1,crem,cprime;
  
   while(c<t)
  {
  count=0;
  pre=0;
  power=1;
  last=0;
  n=mn[c++];
   
  while(n!=0)
  {
  
  
  crem=(n%10);
  


  
  switch(crem)
  {case 0:cprime=0;new1=0;break;
  case 1:cprime=0;new1=0;break;
  case 2:cprime=0;new1=1;break;
  case 3:cprime=1;new1=1;break;
  case 4:cprime=2;new1=0;break;
  case 5:cprime=2;new1=1;break;
  case 6:cprime=3;new1=0;break;
  case 7:cprime=3;new1=1;break;
  case 8:cprime=4;new1=0;break;
  default:cprime=4;new1=0;break;
  }
  if(crem==2||crem==3||crem==5||crem==7)
  {count=last;}
  aux=(crem-cprime)*pre +(cprime*power);
  count=count+aux+new1;
 last=(crem*power)+(last);
  pre=(power=power<<2) + (6*pre);
  power=(power<<1)+(power>>1);
  n=n/10; 
  }
 printf("%llu\n",count);
}
return 0;}
