int main()
{
    int t,n,a;
    char city[100];

    scanf("%d",&t);
    while(t--){
        int cities[30];
        string ans = "YES";
        memset(cities,0,sizeof cities);
        scanf("%d",&n);
        while(n--){
            scanf("%s",&city[0]);

                ans = "NO";
            }

        }
        cout<<ans<<endl;
    }

    return 0;
