#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    while(cin >> n){;
    int a[n];
    for (int i = 0;i <n;i++)
    {
        cin >> a[i];
    }
    int k = 0;
    for (int i = n-1;i >= 0;i--)
    {
        if (a[i] % 2 == 0)
        {
            cout << a[i] << " ";
            k = 1;
        }
    }

    if (k == 0)
        cout << "None to display";
        
        cout << endl;
    }

    return 0;
}
