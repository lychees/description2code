    #include <iostream>
    using namespace std;
    int main()
    {
    	int _,t,a1,a2,b1,b2,i,j,win;
        cin>>t;
        for(_=1;_<=t;++_)
        {
        	cin>>a1>>a2>>b1>>b2;
        	win=0;
        	for(i=a1;i<=a2;++i)
        	{
        		
        		for(j=b1;j<=b2;++j)
        		{
        			
        			if((float)i/j>1.618 || (float)j/i>1.618){win++;}
        			//Just trial and error
        		}
        	}
        	cout<<"Case #"<<_<<": "<<win<<endl;
        }
        
        return 0;
    }
