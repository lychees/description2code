int arr1[100005]={0};

{
    int n;

    int arr[n];
    for(int i=0;i<n;i++)
        cin>>arr[i];
        int max_freq1=0,c2=0,max_freq2=0;
    for(int i=0;i<n;i++)
    {
        arr2[arr[i]]++;
        max_freq1=max(max_freq1,arr2[arr[i]]);
        if(i!=0)
        {
            if(arr[i]==arr[i-1])
            {

                arr1[arr[i]]++;
                max_freq2=max(max_freq2,arr1[arr[i]]);
                c2++;
            }
        }

    }
    if(max_freq1>(n+1)/2)
    {
        cout<<-1<<endl;
        return 0;
    }
    else
    {
        cout<<max(((c2+1)/2),max_freq2)<<endl;
    }
