#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int flag=0;
   	int n;
   	cin>>n;
   	int w[n];
   	int h[n];
   	for(int i=0;i<n;i++)
   	{
   		cin>>w[i];
   	}
   	for(int i=0;i<n;i++)
   	{
   		cin>>h[i];
   	}
   	sort(w+1,w+n);
   	sort(h,h+n);
   	reverse(h,h+n);
   	for(int i=1;i<n;i++)
   	{
   	 if(w[i]*h[i]>=w[0]*h[0])
   	 
   	  flag=1;
    	
   	 }
   	if(flag==1)
   	cout<<"NO"<<"\n";
   	if(flag==0)
   	cout<<"YES"<<"\n";
   	
   
   }
}
