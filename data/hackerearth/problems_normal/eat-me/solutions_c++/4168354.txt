#include <iostream>
#include<cstring>
#define MOD 1000000007
using namespace std;
typedef unsigned long long ll;
int main()
{
    int n,k;
    cin>>n>>k;
    ll dp[n+1][k+1],sum=0;
    memset(dp,0,sizeof(dp));
    int i,j;
    for(i=1;i<=n;i++)
    {
    	dp[i][1]=1;
    }
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=k;j++)
    	{
    		dp[i][j]+=dp[i][j-1]+dp[i-1][j];
    		dp[i][j]%=MOD;
    	}
    }
    for(i=1;i<=k;i++)
    {
    	sum+=dp[n][i];
    }
    cout<<sum%MOD<<"\n";
    return 0;
}
