#include <bits/stdc++.h>
using namespace std;

int main()
{
	stack<int>s;
    int n,q,c;
    cin>>n;
    while(n--){
    	cin>>q;
    	if(q==1){
    		if(!s.empty()){
    			cout<<s.top()<<"\n";s.pop();
    		}
    		else cout<<"No Food\n";
    		
    	}
    	else{
    		cin>>c;
    		s.push(c);
    	}
    }
    return 0;
}
