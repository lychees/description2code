#include <iostream>
#include <cstdio>
#include <vector>
#include <string>

#define MAX(x,y) x>y?x:y
#define MIN(x,y) x<y?x:y
#define pb push_back
#define mp make_pair
#define rep(i,n) for(i=0;i<n;i++)
#define repr(i,j,n) for(i=j;i<=n;i++)
#define endl '\n'
using namespace std;
typedef long long ll;
typedef vector<ll> vll;
const ll maxn = 1009;
int wt[maxn];
int val[maxn];
ll dp[1024][1024];

int main()
{
	std::ios::sync_with_stdio(0);
	ll i,j,k,t,q,m,l,r,n,w;
	//freopen("input.in","r",stdin);
	cin>>t;
	while(t--)
	{
		cin>>n>>w;
		rep(i,n)cin>>wt[i];
		rep(i,n)cin>>val[i];

		dp[0][0]=0;
		for(j=0;j<=w;j++)
		{
			if(j<wt[0])dp[0][j]=0;
			else dp[0][j]=val[0];
		}
		for(i=1;i<n;i++)
		{
			for(j=0;j<=w;j++)
			{
				if(j==0)dp[i][j]=0;
				else
				{
					if(wt[i]>j)dp[i][j]=dp[i-1][j];
					else
						dp[i][j]=MAX(dp[i-1][j-wt[i]]+val[i],dp[i-1][j]);
				}
			}
		}
		if(n)cout<<dp[n-1][w]<<endl;
		else cout<<0<<endl;
	}
}
