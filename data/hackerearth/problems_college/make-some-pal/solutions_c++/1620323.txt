#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i,l,j,ans,k;
    char s[60];

    cin>>t;

    while(t--)
    {
        cin>>s;

        l = strlen(s);
        l--;
        for(i=0;i<=l;i++)
        {
            if(s[i]==s[l])
            {
                for(j=i,k=l;j<k;j++,k--)
                    if(s[j]!=s[k])
                    break;
                if(j>=k)
                {
                    ans = l - i + 1 + 2 * i;
                    break;
                }
            }
        }

        cout<<ans<<'\n';
    }
    return 0;
}
