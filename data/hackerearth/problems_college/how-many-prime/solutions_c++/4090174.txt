#include <bits/stdc++.h>

using namespace std;

int main(){

  //  FILE *in  = freopen("Input5.txt", "r", stdin);
   // FILE *out = freopen("Output5.txt", "w", stdout);
    //assert(in != NULL);

    long T;
    cin>>T;
    assert(T>=1 && T<=50);
    while(T--){

        long N;
        cin>>N;
        assert(N>=2 && N<=1000000);
        vector < long > numbers(N + 1);
        for(long i=2; i<=N; i++)
            numbers[i] = i;
        vector< long > prime;
        for(long i=2; i<=N; i++){
            if(numbers[i] != 0)
                for(long j=2*i; j<=N; j=j+i)
                    numbers[j] = 0;
        }

        for(long i=2; i<=N; i++)
            if(numbers[i] != 0)
                prime.push_back(i);

        int a = -1, b = -1;
        for(long i=0; i<prime.size() - 1; i++)
            for(long k=i+1; k<prime.size(); k++)
                if(N == (prime[i] + prime[k])){
                    a = prime[i];
                    b = prime[k];
                }

        cout<<a<<" "<<b<<endl;
    }

    return 0;
}
