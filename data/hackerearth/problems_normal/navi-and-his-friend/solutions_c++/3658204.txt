#include<bits/stdc++.h>
using namespace std;
const int MAXN = 20;
long long pr[MAXN],wt[MAXN],POWER[MAXN];
int main() {
	POWER[0]=1;
	for(int i=1;i<20;i++) POWER[i]=POWER[i-1]<<1;
	int t;
	cin>>t;
	int counter=0;
	while(t--) {
		++counter;
		int n;
		cin>>n;
		long long ans=-1;
		for(int i=0;i<n;i++) scanf("%lld%lld",&pr[i],&wt[i]);
		long long WT,C;
		scanf("%lld%lld",&WT,&C);
		for(int i=1;i<POWER[n];i++) {
			int cc=i;
			int c=0;
			long long cur=0,val=0;
			for(int j=0;j<n;j++) if(POWER[j]&cc) c++,cur+=wt[j],val+=pr[j];
			if(c<=C and cur<=WT) ans=max(ans,val);
		}
		printf("For Day #%d:\n%lld\n",counter,ans);
	}
	return 0;
}
