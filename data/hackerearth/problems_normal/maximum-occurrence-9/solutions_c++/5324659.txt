#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
int main()
{ string s;
getline(cin,s);
	map<int,int> m;
	map<int,int> :: iterator itr;
	for(int i=0;s[i];i++)
	{
		m[(int)s[i]]+=1;
	}  
	int ans=(m.begin())->first;
	for(itr=m.begin();itr!=m.end();itr++)
	{
		if(itr->second>m[ans])
		ans=itr->first;
	}
	cout<<(char)ans<<" "<<m[ans]<<"\n";
	

}
