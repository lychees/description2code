#include <bits/stdc++.h>
using namespace std;
char s1[52];
int pall(int i,int j) {
	int k = j;
	int flag = 1;
	for(k=0;k<j;k++,j--) {
		if(s1[k]!=s1[j]) {
			flag = 0;
		}
	}
	return flag;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
    	char s[52];
    	scanf("%s",s);
    	int l = strlen(s);
    	l--;
    	int i,j,k;
    	for(i=l;i>=0;i--) {
    		s1[l-i] = s[i];
    	}
    	s1[l+1]='\0';
    //	printf("%s",s1);
    	k = 0;
    	for(i=l;i>=0;i--) {
    		int ans = pall(0,i);
    	//	printf("%d ",ans);
    		if(ans) {
    			k = l - i;
    			break;
    		}
    	}
    	printf("%d\n",k+l+1);
    }
    return 0;
}
