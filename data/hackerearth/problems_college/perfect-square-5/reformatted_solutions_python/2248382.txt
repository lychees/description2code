import math
for i in range(input()):
    n = input()
    if math.sqrt(n) == math.ceil(math.sqrt(n)):
        print "YES"
    else:
        x = (math.ceil(math.sqrt(n))-1)**2
        y = (math.ceil(math.sqrt(n)))**2
        if math.fabs(n-x) < math.fabs(n-y):
            print int(x-n)
        else:
            print "+"+str(int(y-n))
