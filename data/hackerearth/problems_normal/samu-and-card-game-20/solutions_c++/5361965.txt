#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin>>T;
    while(T--)
    {
    	int n,m;
    	cin>>n>>m;
    	int k;
    	cin>>k;
    	int count = 0,x=1,y=1;
    	for(int i=0;i<k;i++)
    	{
    		int a,b;
    		cin>>a>>b;
    		while((x+a)<=n && (y+b)<=m && (x+a)>=1 && (y+b)>=1)
    		{
    			count++;
    			x+=a;
    			y+=b;
    		}
    	}
    	cout<<count<<"\n";
    }
    return 0;
}
