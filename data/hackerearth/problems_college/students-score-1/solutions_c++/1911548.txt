#include <iostream>
#include <algorithm>
using namespace std;
int N, inp, maxn, cur;
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cin >> N;
    for (int a = 0; a < N; ++a) {
        cin >> inp;
        cur += inp;
        if (cur < 0)
            cur = 0;
        maxn = max(maxn, cur);
    }
    cout << maxn << "\n";
    return 0;
}
