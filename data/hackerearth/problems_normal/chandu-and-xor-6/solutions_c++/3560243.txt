# include <bits/stdc++.h>

using namespace std;


int main()
{
	int n,m;

	cin>>n>>m;
	int arr[n][m];
	int row[n];
	int col[m];

	for(int i=0;i<n;i++)
	{
		row[i]=0;
	}


	for(int i=0;i<m;i++)
	{
		col[i]=0;
	}

	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>arr[i][j];
			row[i] = row[i] + arr[i][j];
			col[j] = col[j] + arr[i][j];
		}

	}
	int ans=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			ans = max(ans, row[i]^col[j]);
		}

	}	

		
	cout<<ans<<endl;

}
