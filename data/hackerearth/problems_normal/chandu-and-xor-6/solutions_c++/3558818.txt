#include <iostream>
using namespace std;

int main()
{
	int N, M, arr[1000][1000], rows[1000]={0}, cols[1000]={0};
    cin>>N>>M;
    
    for(int i=0; i<N;i++) {
    	for(int j=0;j<M; j++) {
    		cin>>arr[i][j];
    	}
    }
    for(int i=0; i<N;i++) {
    	
    	for(int j=0;j<M; j++) {
    	
    		rows[i] += arr[i][j];
    		
    		
    	}
    }
    for(int i=0; i<M;i++) {
    	
    	for(int j=0;j<N; j++) {
    	
    		cols[i] += arr[j][i];
   
    	}
    }
	int max=0, xor_val;
    for(int i=0; i<N;i++) {
    	
    	for(int j=0;j<M; j++) {
    	
    		xor_val = rows[i]^cols[j];
    		if (xor_val > max)
   				max = xor_val;
    	}
    }
	cout<<max;
    return 0;
}
