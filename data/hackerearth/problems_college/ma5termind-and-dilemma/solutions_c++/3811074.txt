#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int n,i,a,b,c,d,e,f;
	char s[100002];
	while(t--)
	{
		cin>>n;
		cin>>s;
		a=0;
		b=0;
		c=0;
		d=0;
		e=0;
		f=0;
		for(i=0;i<n;i++)
		{
			if(s[i]=='0')
			{
				a++;
			}
			else
				b++;
		}
		for(i=0;i<a;i++)
		{
			if(s[i]=='0')
			c++;
			else
			d++;
		}
		for(i=0;i<b;i++)
		{
			if(s[i]=='0')
			e++;
			else
			f++;
		}
		if(a!=b)
		{
			if(d<e)
			cout<<d<<endl;
			else
			cout<<e<<endl;
			
		}
		else
		{
			if(c<d)
			cout<<c<<endl;
			else
			cout<<d<<endl;
		}
	}
	return 0;
}
