#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    int q;
    cin>>q;
  for(int j=1;j<=q;j++)
    {
    	int n;
    	cin>>n;
    	double s1=0,s2=0,s3=0,e=0,f=0,t=0;
    	for(int i=0;i<n;i++)
    	{
    		double x,y,z,a,b,c;
    		cin>>x>>y>>z>>a>>b>>c;
    	//	cout<<x<<y<<z<<a<<b<<c;
    		s1+=x;
    		s2+=y;
    		s3+=z;
    		e+=a;
    		f+=b;
    		t+=c;
    		
    	}
    	s1=s1/n;
    	s2=s2/n;
    	s3=s3/n;
    	e=e/n;
    	f=f/n;
    	t=t/n;
    	long double ss;
  if(e==0&&t==0&&f==0)
  ss=0;
  else
    	 ss=(s1*e+s2*f+s3*t)/((e*e)+(f*f)+(t*t));
    	 
    	 
    	ss=ss*-1.00000000;
    long double ff=sqrt((((s1+e*ss)*(s1+e*ss))+((s2+f*ss)*(s2+f*ss))+((s3+t*ss)*(s3+t*ss))));
    	//cout<<"case #"<<j<<":"<<ff<<" "<<ss<<endl;
    	printf("Case #%d: %.8llf% .8llf\n",j,ff,ss);
    }
    return 0;
}
