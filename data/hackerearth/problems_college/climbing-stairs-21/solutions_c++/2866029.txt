#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define scl(a) scanf("%lld",&a)
#define ll long long 

ll f[1001];

void solve()
{
	int i;
	f[1]=1;
	f[2]=2;
	for(i=3;i<=1001;i++)
		f[i]=f[i-1]+f[i-2];
}

int main()
{
	solve();
    long long t,n;
    scl(t);
    while(t--)
    {
    	scl(n);
    	printf("%lld\n",f[n]);
    }
    return 0;
}
