#include<stdio.h>
#include <string.h>
#include <map>
using namespace std; 
map<string, int> ma;
char m[101], n[101], x, y, sum=0;
void issub(int I, int j){
	int k, l, p=0, res=0;
	string a="";
	for (k=I;k+y-1<=j;k++){
		p=0;
		for (l=0;l<y;l++){
	//		printf("%c %c\n",m[k+l], n[l]);
			if (m[k+l]!=n[l]){
				p=1; break;}
		}
		if (p==0)
		  res=1;
	}
	if (res==1){
		for (k=I;k<=j;k++)
		a+=m[k];
		if (ma[a]==0)
		 sum++;
		 ma[a]=1;
	}
}
int main(){
	int I, j;
	scanf("%s %s", m, n);
	x=strlen(m),
	y=strlen(n);
//	printf("%d %d", x, y);
	for (I=0; I<x;I++){
	for (j=I;j<x; j++){
//		printf("%d %d\n", I, j);
		   issub(I,j);
	}
//	printf("%d %d\n", I, j);
	}
	printf("%d\n", sum);
	return 0; 
}
