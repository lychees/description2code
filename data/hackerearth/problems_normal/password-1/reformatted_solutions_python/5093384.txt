n = int(raw_input())
l = []
d = []
for i in range(n):
    l.append(raw_input())
for i in range(len(l)):
    for j in range(len(l)):
        if i == j:
            continue
        else:
            d.append([l[i], l[j]])
for i in d:
    if i[0][::-1] == i[1]:
        print len(i[0]), i[0][(0+len(i[0])-1)/2]
        break
