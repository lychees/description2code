#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll a[100005];
int main()
{
    //cout << "Hello World!" << endl;
    ll n,i,j,k,ans=0;
    cin>>n;
    for(i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    //if(a[n-1]*a[n-2]>0) cout<<a[n-1]*a[n-2]<<endl;
    ll c1=a[n-1]*a[n-2];
    ll c2=a[0]*a[1];
    ll c3=0;
    for(i=0;i<n-1;i++) c3=max(c3,a[i]*a[i+1]);
    cout<<max(c1,max(c2,c3));
    
    
    return 0;
}
