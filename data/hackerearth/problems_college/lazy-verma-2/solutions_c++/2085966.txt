#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

int main()
{
	ll n,m;
	scanf("%lld%lld",&n,&m);
	ll ar[n],tmp,sum = 0;
	for(ll i=0;i<n;i++)
	scanf("%lld",&ar[i]);
	for(ll i=0;i<m;i++)
	{
		scanf("%lld",&tmp);
		sum+=tmp;
	}
	ll ans = 0;
	for(ll i=0;i<n;i++)
	{
		ans+=ar[i]*sum;
	}
	printf("%lld\n",ans);
	return 0;
}
