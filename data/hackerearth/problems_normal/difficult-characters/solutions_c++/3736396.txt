#include <iostream>
#include<map>
#include<vector>
#include<algorithm>

using namespace std;

bool comp(const pair<int, int> &a, const pair<int, int> &b)
{
	if(a.second==b.second)
		return a.first>b.first;
	return a.second<b.second;
	
}

int main()
{
    int test_cases;
    cin>>test_cases;
    for(int b=0;b<test_cases;b++)
    {
    	string s;
    	cin>>s;
    	
    	int n = s.size();
    	map<char, int > mymap;
    	for(int i=0;i<26;i++)
	    {
	    	char c = i+'a';
	    	mymap[c] = 0;
	    	
	    }
    	for(int i=0;i<n;i++)
    	{
    		char cur = s[i];
    		mymap[cur] = mymap[cur]+1;
    	}
    	
    	//Iterate the map
    	vector<pair<int, int> > data;
    	
    	for(int i=0;i<26;i++)
    	{
    		char c = i+'a';
    		//cout<<i<<":"<<mymap[c];
    		data.push_back(make_pair(i, mymap[c]));
    	}
    	
    	//Sort
    	std::sort(data.begin(), data.end(), comp);
    	
    	//print
    	for(int i=0;i<26;i++)
    	{
    		char cur = data[i].first + 'a';
    		cout<<cur<<" ";
    	}
    	cout<<endl;
    }
    
    return 0;
}
