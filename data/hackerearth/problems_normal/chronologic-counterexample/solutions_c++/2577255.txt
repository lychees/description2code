#include <bits/stdc++.h>

#define pb push_back
#define pp pop_back
#define f first
#define s second
#define mp make_pair
#define sz(a) int((a).size())
#define fname "."

typedef long long ll;
typedef unsigned long long ull;

const int MAX_N = (int)1e5 + 123;
const double eps = 1e-6;
const int inf = (int)1e9 + 123;

using namespace std;

int n, m;
int l[60], r[60];
int p[60];
int p2[60];
bool u[60];
vector < int > q;

bool check() {
	for (int i = 1; i <= n; i++)
		p2[i] = p[i];
	for (int i = 1; i <= m; i++)
		sort(p2 + l[i], p2 + r[i] + 1);
	for (int i = 1; i <= n; i++)
		if (p2[i] != i)
			return 1;
	return 0;
}

int main() {
	#ifdef Nick
	freopen(fname"in","r",stdin);
	freopen(fname"out","w",stdout);
	#endif
	cin >> n >> m;
	for (int i = 1; i <= m; i++)
		cin >> l[i] >> r[i];
	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= n; j++) {
			if (!u[j]) {
				p[i] = j;
				for (int k = n, id = i + 1; k > 0; k--)
					if (!u[k] && k != j)	
						p[id++] = k;
				if (check()) {
					u[j] = 1;
					p[i] = j;
					q.pb(j);
					break;
				}
			}
		}
		if (sz(q) != i) {
			cout << "All ok!";
			return 0;
		}
	}
	for (auto i : q)
		cout << i << ' ';
	return 0;
}
