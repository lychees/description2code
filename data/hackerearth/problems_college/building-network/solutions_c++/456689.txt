#include <bits/stdc++.h>
using namespace std;
int main()
{
       int t,i,j,k;
       long long ans=0;
       scanf("%d",&t);
       for(i=0;i<t;i++)
       {
               scanf("%d",&j);
               ans=0;
               long long A[j];
               for(k=0;k<j;k++)
               scanf("%lld",&A[k]);
               sort(A,A+j);
               for(k=1;k<j;k++)
               ans+=(A[k]+A[0]);
               printf("%lld\n",ans);
       }
       return 0;
}
