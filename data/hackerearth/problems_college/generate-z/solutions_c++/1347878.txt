#include <iostream>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    for(i=1;i<=n;i++){
    	cout<<"*";
    }
    cout<<"\n";
    for(i=2;i<n;i++){
    	for(int j=1;j<=(n-i);j++){
    		cout<<" ";
    	}
    	cout<<"*"<<"\n";
    }
    for(i=1;i<=n;i++){
    	cout<<"*";
    }
    return 0;
}
