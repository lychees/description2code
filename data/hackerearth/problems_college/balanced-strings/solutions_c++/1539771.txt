#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int a[26];
	ll t;
	cin>>t;
	while(t--)
	{
		memset(a,0,sizeof(a));
		string s;
		cin>>s;
		for(ll i=0;i<s.size();i++)
		a[s[i]-'a']++;
		
		ll t=a[0];
		ll flag=0;
		for(ll i=1;i<26;i++)
		{
			if(a[i]!=t)
			{
				flag=1;
				break;
			}
		}
		if(flag)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
	}
	
}
