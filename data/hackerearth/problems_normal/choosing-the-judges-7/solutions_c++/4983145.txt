#include<bits/stdc++.h>
using namespace std;
#define LL long long 
LL fun(LL a[],int n)
{
	LL incl=a[0];
	LL excl=0;
	LL excl_new;
	for(int i=1;i<n;i++)
	{
		excl_new=(incl>excl)?incl:excl;
		incl=excl+a[i];
		excl=excl_new;
	}
	return (incl>excl?incl:excl);
}
int main()
{
	int t; cin>>t;
	for(int i=1;i<=t;i++)
	{
		int n; cin>>n; LL a[n];
		for(int j=0;j<n;j++)
			cin>>a[j];
		cout<<"Case "<<i<<": "<<fun(a,n)<<endl;
	}
}
