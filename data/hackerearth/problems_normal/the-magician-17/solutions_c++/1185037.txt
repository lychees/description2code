int go(int n,int k){
    if(n==1)
        return 1;
    if(n&1){
        if(k<=(n+1)/2)  return 2*k-1;
        else{
            k-=(n+1)/2;
            int t=go(n/2,k);
            t=t==n/2?1:t+1;
            return 2*t;
        }
    }else{
        if(k<=n/2)  return 2*k-1;
        else        return 2*go(n/2,k-n/2);
    }
}
