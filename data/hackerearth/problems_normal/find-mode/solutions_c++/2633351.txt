#include<bits/stdc++.h>
using namespace std;
int a[1000001];
int main()
{
    int t,i,j,n,m,k;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);	
    	memset(a, 0, sizeof(a));
    	
    	for(i=0;i<n;i++)
    	{
    	scanf("%d",&k);
    	a[k]++;
    	}
    	m=a[0];
    	for(i=0;i<1000001;i++)
    	{
    		if(a[i]>m)
    		m=a[i];
    	}
    	
    	for(i=1000001;i>0;i--)
    	{
    		if(a[i]==m)
    		cout<<i<<" ";
    	}
    	cout<<"\n";
    	
	     }
    return 0;
}
