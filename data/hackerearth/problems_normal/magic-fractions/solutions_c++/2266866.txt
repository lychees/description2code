#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
int p[505];
void pre()
{
	for(int i=2;i<=501;i++)
	{
		for( int j=i;j<=501;j++)
		{
			if(i*j <= 501)
			{
				p[i*j] =1;
			}
		}	
	}
}
int main()
{   
	int n,c=0;
	long long int s=0;
	cin>>n;
	memset(p,0,sizeof(p));
	pre();
    for(int i=2;i<=n;i++)
	{
		if(p[i] ==0) c++;
		s += pow(2,c-1);
	}	
	cout<<s<<endl;
}
