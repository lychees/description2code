#include <iostream>
#include <algorithm>
using namespace std;

long long int FEM(long long int a, long long int b, long long int m)
{
    long long int c=a%m,p=1;
    while(b>0)
        {
            if(b%2==1)
            {
                p=p*c;
                p=p%m;
            }
            b=b/2;
            c=(c*c)%m;
        }
    return p;
}

int main()
{
	ios_base::sync_with_stdio(false);
	long long int T,n,i,a,m=1000000007;
	cin>>T;
	while(T--)
	{
		cin>>n;
		if(n==1)
			cout<<"2\n";
		else
		{
			n--;
			a=FEM(2,n,m);
			a=(a*3)%m;
			if(n%2!=0)
				cout<<(a+1+m)%m<<"\n";
			else
				cout<<(a-1+m)%m<<"\n";
		}
		
		//cout<<c<<"\n";
	}
	return 0;
}
