#include<bits/stdc++.h>
using namespace std;

bool check(char*a,char*b){
	
	while(a<b){
		if(*a==*b){
			a++;
			b--;
		}
		else
			return 0;
	}

	return 1;
}

int main(){
	int n,q,l,r,type;
	scanf("%d%d",&n,&q);
	char arr[n+2];
	scanf("%s",arr);
	//printf("query is %d\n",q);
	while(q--){
		scanf("%d",&type);
		//printf("type is %d for query %d\n",type,q+1);
		//cout<<p<<endl<<endl;
		if(type==1){
			scanf("%d%d",&l,&r);
			if(check(arr+l-1,arr+r-1))
				cout<<"Yes"<<endl;
			else
				cout<<"No"<<endl;
		}
		else{
			int i;
			char x;
			scanf("%d",&i);
			getchar();
			scanf("%c",&x);
			arr[i-1]=x;
		}
	}
	return 0;

}
