#include<iostream>
#include<bits/stdc++.h>
using namespace std;
char s[8][16];
int n;
int minlen=-1,curlen=0;
char temps[1000],siz=0;
int thesetwo(char *s1,char *s2)
{
    int i=0,j=0;
    while(s1[i]!='\0')
    {
        j=0;
        while(s2[j]!='\0' && s1[j+i]!='\0')
        {
            if(s1[i+j]!=s2[j])
            {
                break;
            }
            j++;
        }
        if(s1[j+i]=='\0')
        {
            return i;
        }
        i++;
    }
    return i;
}
void make(int *order)
{
        int i=1;
        strcpy(temps,s[order[0]]);
        while(i<n)
        {
            int joinat=thesetwo(temps,s[order[i]]);
         //   cout<<"join at"<<joinat;
            strcpy(&temps[joinat],s[order[i]]);
            i++;
        }
   //     cout<<temps<<endl;
        curlen=strlen(temps);
        if(minlen==-1 || minlen>curlen)
        {
            minlen=curlen;
        }
}
void go(int at,int *order,int *done)
{
    if(at==n)
    {
        make(order);
        return;
    }
    int i;
    i=0;
    while(i<n)
    {
        if(done[i]!=1)
        {
            done[i]=1;
            order[at]=i;
            go(at+1,order,done);
            done[i]=0;
        }
        i++;
    }
    return;
}
int main()
{
	ios::sync_with_stdio(false);
    int i;
    cin>>n;
    i=0;
    while(i<n)
    {
        cin>>s[i];
        i++;
    }
    int order[20]={0};
    int done[20]={0};
    go(0,order,done);
    cout<<minlen;
	return 0;
}
