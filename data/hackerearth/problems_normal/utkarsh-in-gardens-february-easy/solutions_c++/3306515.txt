#include <bits/stdc++.h>
using namespace std;

#define icin(x) scanf("%d",&x)
#define lcin(x) scanf("%lld",&x)
#define pb push_back
#define mp make_pair
typedef long long LL;
typedef vector<int> vi;
typedef vector<vi> vvi;
typedef pair<int,int> pii;

int main(){
	int n,a;
	icin(n);
	vector<bitset<2000> > v(n);
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			icin(a);
			v[i][j]=a;
		}
	}
	LL ans=0;
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++){
			a = (v[i]&v[j]).count();
			ans += ((a*(a-1))/2);
		}
	}
	cout<<(ans)/2<<endl;
  return 0;
}
