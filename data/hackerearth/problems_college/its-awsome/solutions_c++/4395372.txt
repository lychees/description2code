#include <iostream>
using namespace std;

int main()
{
    long long int t,n,s,flag;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	s=0;
    	while(n>0)
    	{
    		if(n%10==4||n%10==7)
    		{
    			s++;
    		}
    		n/=10;
    	}
    	flag=1;
    	if(s==0)
    	{
    		flag=0;
    	}
    	while(s>0)
    	{
    		if(s%10!=4&&s%10!=7)
    		{
    			flag=0;
    			break;
    		}
    		s/=10;
    	}
    	if(flag)
    	{
    		cout<<"YES"<<endl;
    	}
    	else
    	{
    		cout<<"NO"<<endl;
    	}
    }
    return 0;
}
