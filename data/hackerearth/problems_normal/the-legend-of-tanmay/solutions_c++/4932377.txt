#include <iostream>
#include<set>
using namespace std;
multiset<long long> s;

int main()
{long long t,na,x=1,y=1,m=0,n=0,i,a,z=0;
cin>>t;
 	while(t--)
 	{m=0;
 	n=0;
 	x=1;
 	y=1;
 	z=0;
 	s.clear();
 	cin>>na;
 	  for(i=0;i<na;i++)
 	  {cin>>a;
 	   s.insert(a);
 	  }
 	  multiset<long long>::iterator it;
 	  for(it=s.begin();it!=s.end();it++)
 	  {if(*it!=0)
 	    {  x*=*it;
 	      
 	    }
 	    z++;
 	  
 	   if(*it<0)
 	       y=*it;
 	}
 	   if(x<0)
 	    	{m=x/(y);
 	    	 n=x;
 	    	}
 	   if(x>0)
 	   		{m=x;
 	   		n=m/(y);
 	   		}
 	    if(*s.begin()>=0)
 	    n=*s.begin();
 	    if(*(--s.end())==0)
 	    m=0;
 	    if(z==1)
 	    {m=*s.begin();
 	    n=*s.begin();
 	     }
 	  
 	  cout<<m<<" "<<n<<endl;
 	}
 	  return 0;
}
