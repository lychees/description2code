#include <iostream>
#include<algorithm>
#include<cstring>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    string str[n];
    int size=sizeof(str)/sizeof(str[0]);
    for(i=0;i<n;i++){
        cin>>str[i];
        sort(str[i].begin(),str[i].end());
    }
    sort(str,str+size);
    int cnt=1,max=0;
    for(i=0;i<n-1;i++){
        if(!(str[i].compare(str[i+1])))
             cnt++;
        else{
            if(cnt>max)
                max=cnt;
            cnt=1;
        }
    }
    if(cnt>max)
        max=cnt;
    cout<<max;
            
    return 0;
}
