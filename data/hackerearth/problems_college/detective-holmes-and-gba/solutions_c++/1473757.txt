#include <iostream>
using namespace std;

int main()
{
    long long t,n,m,s,x;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m>>s;
    	x=m%n;
    	if(x==0)
    	{
    		if(s==1)
    		cout<<n<<endl;
    		else
    		cout<<s-1<<endl;
    	}
    	else
    	{
    		s=s+x-1;
    		if(s>n)
    		s=s-n;
    		cout<<s<<endl;
    	}
    }
}
