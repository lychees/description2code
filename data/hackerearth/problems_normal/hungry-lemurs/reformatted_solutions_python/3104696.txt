n, k = map(int, raw_input().split())

if k >= n:
    print min(n, k-n)
else:
    ans = 1 << 30
    for v in range(-(n-k), n-k+1):
        d = k+v
        if d == 0:
            continue
        x = abs(v) + n - (n/d)*d
        y = abs(v) + ((n+d-1)/d) * d - n
        ans = min(ans, x, y)
    print ans
