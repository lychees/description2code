#include <iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string s;
	
	for(int i=0;i<t;i++)
	{
		int a[26]={0};
		cin>>s;
		int len=s.size();
		for(int j=0;j<len;j++)
		{
			int temp=(int)s[j];
			a[temp%65]+=1;
		}
		for(int k=0;k<26;k++)
		{
			if(a[k]!=0)
			{
				for(int h=0;h<a[k];h++)
				{
					int x=k+65;
					char c=(char)x;
					cout<<c;
				}
			}
		}
		cout<<endl;
		
	}
	return 0;
}
