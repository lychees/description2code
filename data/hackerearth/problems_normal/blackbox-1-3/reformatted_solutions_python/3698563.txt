from fractions import gcd
l = []


for x in range(int(raw_input())):
    l.append(sorted(list(map(int, raw_input().split()))))

for tri in l:
    a = tri[0]
    b = tri[1]
    c = tri[2]
    if a**2 + b**2 == c**2 and gcd(a,
                                   b) == 1 and gcd(b,
                                                   c) == 1 and gcd(a,
                                                                   c) == 1:
        print('YES')
    else:
        print('NO')
