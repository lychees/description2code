#include <math.h>
    #include <stdio.h>
    int prime(int num)
    {
    int i;
    if(num%2==0)
    return 0;
    for(i=3;i<=sqrt(num);i += 2)
    {
    if(num%i==0)
    return 0;
    }
    return 1;
    }
    int main()
    {
    int t,n,x,y,i,count,m,total;
    double ans[10000];
    scanf("%d",&t);
    ans[1]=0;x=0;
    for(n=3;n<10000;n+=2)
    {
    count = 0;
   
    x = x+2;
    if(prime((x*x)+1))
    ++count;
    if(prime((x*x)+1+x))
    ++count;
    if(prime((x*x)+1-x))
    ++count;
    ans[n]=ans[n-2]+count;
    }
    while(t--)
    {
    scanf("%d",&n);
    total = (n*2) - 1;
    printf("%lf\n",(ans[n]/total)*100);
    }
    return 0;
    }
