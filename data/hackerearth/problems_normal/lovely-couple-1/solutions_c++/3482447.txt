unsigned GCD(unsigned u, unsigned v) {
    while ( v != 0) {
        unsigned r = u % v;
        u = v;
        v = r;
    }
    return u;
}

ULL checkfact(ULL a)
{
	LLI i =2,count = 0;
	while(a > 1)
	{
		if(a % i == 0)
		{
			++count;
			while(a % i == 0)
			{
				a = a/i;
			}
		}
		++i;
	}
return count;
}

bool isPrime(int number){

    if(number < 2) return false;
    if(number == 2) return true;
    if(number % 2 == 0) return false;
    for(int i=3; (i*i)<=number; i+=2){
        if(number % i == 0 ) return false;
    }
    return true;

}
