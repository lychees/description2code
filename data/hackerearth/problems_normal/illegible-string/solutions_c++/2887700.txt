#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,n;
	int x=0,w=0;
	string s;
	cin>>n;
	cin>>s;
	for(i=0;i<n-1;i++)
	{
		if(s[i]=='w')
			w++;
		else if(s[i]=='v'&&s[i+1]=='v')
		{
			w++;
			i++;
		}
		else if(s[i]=='v'&&s[i+1]=='w')
		{
			w++;
			s[i+1]='v';
		}
		else
			x++;
	}
	if(i==n-1)
	{
		if(s[i]=='w')
			w++;
		else
			x++;
	}
	cout<<x+w<<' '<<x+w+w<<endl;
	return 0;
}
