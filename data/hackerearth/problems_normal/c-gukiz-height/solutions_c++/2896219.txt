#include <iostream>
#include<cmath>
#define ll long long
using namespace std;

int main()
{
    int n;
    ll h;
    cin>>n>>h;
    double a[n];
    int i;
    for(i=0;i<n;++i)
    	cin>>a[i];
    for(i=1;i<n;++i)
    	a[i]+=a[i-1];

     double an = 4e17;
     for(i=0;i<n;++i){
        double tot = i+1;
     		double go = h;
     		go-=(tot*(tot+1)/2.0);
     		double su = a[i];

     		if(su>=go){
     			an = tot;
     			break;
     		}
     }

     ll lo = 0;
     if(a[n-1]<0){
        ll low = 0;
        ll hig = 1e8;
        while(low<=hig){
           ll mii = (low+hig)/2;
           ll nn = (mii+1)*n;
           ll su = nn*(nn+1)/2;
           nn = mii*n;
           su-=nn*(nn+1)/2;
           ll cum = abs((ll)a[n-1]);
           if(su>=cum){
              hig = mii-1;
           }else{
              lo = mii;
              low = mii+1;
           }
        }
     }
     ll hi = 4e17;

     while(lo<=hi){

     	double mid = (lo+hi)/2;
     	double da = mid*n;
     	int ff = 0;
     	for(i=0;i<n;++i){
     		double tot = da+i+1;
     		double go = h;
     		go-=(tot*(tot+1)/2.0);
     		double su = mid*a[n-1]+a[i];

     		if(su>=go){
     			an = min(an,tot);
     			ff = 1;
     			break;
     		}
     	}

     	if(ff){
     		hi = (ll)mid-1;
     	}else
     		lo = (ll)mid+1;
     }
     cout<<(ll)an<<endl;
    return 0;
}
