#include <iostream>
#include<queue>
#include<utility>
#include<map>
#include<vector>
#define ll long long
using namespace std;

class mycompare {
	
	public:
	bool operator() (pair<ll, pair<ll,ll>> p1, pair<ll, pair<ll,ll>> p2) {
		if(p1.first != p2.first) {
			return p1.first < p2.first;
		} else {
			return p1.second.first < p2.second.first;
		}
	}
};


int main()
{
	int n;
	cin >> n;
	priority_queue<pair<ll, pair<ll,ll>>, vector<pair<ll, pair<ll,ll>>>, mycompare> q;
	ll id, oldScore, p, l, c, s, newScore;
	while(n--) {
		cin >> id >> oldScore >> p >> l >> c >> s; 
		newScore = p*50 + 5*l + 10*c + 20*s;
		q.push(make_pair(newScore-oldScore, make_pair(id,newScore)));
	}
	
	for(int i = 1; i <= 5; i++) {
		ll tid = q.top().second.first;
		ll score = q.top().second.second;
		cout << tid << " " << score << endl;
		q.pop();
	}
    return 0;
}
