#include<bits/stdc++.h>
using namespace std;
int main ()
{	long long int T,count,a,b,ar[10000],in,k,i,l,j,flag;
	cin>>T;
	while(T--)
	{count=0;
	cin>>a>>b;
	for(i=a;i<=b;i++)
	{k=0;
	in=i;
	memset(ar,0,sizeof(ar));
	while(in!=0)
	{
	ar[k]=in%10;
	in/=10;
	k++;
	}
	flag=0;
	for(j=0,l=k-1;i!=0&&j<=k/2;j++,l--)
	{
	if(ar[j]!=ar[l])
	{flag=1;break;}
	}
	if(flag==0)
	count++;
	}
	cout<<count<<endl;
}
return 0;
}
