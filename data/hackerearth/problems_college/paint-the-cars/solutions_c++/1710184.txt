using namespace std;

typedef pair<int, int> pii;

const double eps = 1e-9;
const double pi = acos(-1.0);

const int inf = (int)1e9;

int cost[22][3];
int dp[22][3];

int main() {
	#ifdef LOCAL
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif

	int T; cin >> T;

	while(T--) {
		int n; scanf("%d",&n);
		for (int i = 1; i <= n; i++) {
			for (int j = 0; j < 3; j++) {
				scanf("%d",&cost[i][j]);
				dp[i][j] = inf;
				for (int k = 0; k < 3; k++) if (k != j) {
					dp[i][j] = min(dp[i][j],dp[i - 1][k] + cost[i][j]);
				}
			}
		}
		int res = inf;
		for (int i = 0; i < 3; i++) {
			res = min(res,dp[n][i]);
		}
		printf("%d\n",res);
	}
	
	return 0;
}
