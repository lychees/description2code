n, m = map(int, raw_input().split())
a = []
mid = (m+1)/2
a.append(mid)
if m % 2 == 0 :
      for i in range(1, mid) :
            a.append(mid + i)
            a.append(mid - i)
      a.append(m)
else :
      for i in range(1, mid) :
            a.append(mid - i)
            a.append(mid + i)
for i in range(n) :
      print a[i%m]
