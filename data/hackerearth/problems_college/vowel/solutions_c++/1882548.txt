#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    		int flag=0;
    	char s[100000];
    	cin>>s;
    	for(int i=0;s[i]!='\0';i++)
    	{
    		if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='E'||s[i]=='I'||s[i]=='O'||s[i]=='U')
    		{
    			flag=1;
    			cout<<s[i];
    		}
    	}
    	if(flag==0)
    		cout<<"No";
    	cout<<endl;
    }
    return 0;
}
