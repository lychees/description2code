#include <iostream>
using namespace std;
int f(int x)
{  int count=0;
	for(int i=0;i<32;i++)
     if(x&(1<<i))count++;
     return count;
}
     
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(f(n)%2==0)cout<<"even"<<endl;
    	else
    	cout<<"odd"<<endl;
    }
    return 0;
}
