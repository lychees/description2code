#include <iostream>
using namespace std;
#include<map>
#include<algorithm>
#include<cstring>
bool comp( pair<int, string> a,pair<int, string> b)
{
		if(a.first<b.first)
		return true;
		else
		return false;
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
 		int n;
 		cin>>n;
 		pair<int ,string> p[n];
 		for(int i=0;i<n;i++)
 		{
 			int a;
 			string x;
 			cin>>a>>x;
 			p[i].first=a;
 			p[i].second=x;
 		}
 		sort(p,p+n,comp);
 		int y;
 		cin>>y;
 		cout<<p[y-1].second<<endl;
   }
}
