#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll Mod = 1e9 + 7;
ll pw(ll a,ll b){ return b?(pw((a * a) % Mod,b >> 1) * (b & 1?a:1))%Mod:1; }
ll inv(ll x){
	return pw(x,Mod-2);
}
ll til(ll x){
	return (((((x * (x + 1)) % Mod) * (2 * x - 2)) % Mod) * inv(6))% Mod;
}
int main(){
	ll n,m,p,q;
	cin >> n >> m >> p >> q;
	ll ans = 0;
	if(m > q && n > p) ans += max((m - q) * (n - p) * 4,(ll)0);
	ans %= Mod;
	if(m > p && n > q) ans += max((n - q) * (m - p) * 4,(ll)0);
	ans %= Mod;
	ans %= Mod;
	ll num = abs(n - m) + 1;
	ll mn = min(n,m) - 1;
	ans += til(mn) * 4;
	ans %= Mod;
	mn++;
	ans += ((((mn * (mn - 1)) % Mod) * num) % Mod) * 2;
	ans %= Mod;
	cout << ((((((((n * m) % Mod) * ((n * m - 1) % Mod))) % Mod) - ans + Mod) % Mod) + Mod) % Mod<< endl;
	return 0;
}
