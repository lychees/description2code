#include <iostream>
using namespace std;

int main()
{
	long long t,a,b,c,sum,i;
    cin >> t;
    while(t--){
    	cin >> a >> b >> c;
    	if (c==0) cout << a << endl;
		if (c==1) cout <<  b << endl;
		else{
		for(int i=0; i<c-1; i++)
		{
			sum = a + b;
			a = b;
			b = sum;
		}
			cout << sum << endl;
		}
    }
    return 0;
}
