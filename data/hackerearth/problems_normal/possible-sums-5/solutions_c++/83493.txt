#include<iostream>
#include<stdio.h>
#include<string>

using namespace std;

int isSubsetSum(int set[], int n, int sum)
{
    bool subset[sum+1][n+1];
 	int ans=1;
	  
    for (int i = 0; i <= n; i++)
      subset[0][i] = true;
  
    for (int i = 1; i <= sum; i++)
      subset[i][0] = false;
  
     for (int i = 1; i <= sum; i++)
     { 
       for (int j = 1; j <= n; j++)
       {
         subset[i][j] = subset[i][j-1];
         if (i >= set[j-1])
           subset[i][j] = subset[i][j] || subset[i - set[j-1]][j-1];      
       }
//       cout<<subset[i][n]<<endl;
       if(subset[i][n]==true){
	   	ans++;	  
       }
     }
 	/*
     // uncomment this code to print table
     for (int i = 0; i <= sum; i++)
     {
       for (int j = 0; j <= n; j++)
          printf ("%d", subset[i][j]);
       printf("\n");
     } 
 	*/
	return ans;     
}



int main(){
  
  int t,n,set[200],i,sum;
  
  cin>>t;
  
  while(t--){
  	
  	cin>>n;
  	for(i=sum=0;i<n;i++)	cin>>set[i],sum+=set[i];
  	cout<<isSubsetSum(set,n,sum)<<endl;
  }
  
  return 0;
}
