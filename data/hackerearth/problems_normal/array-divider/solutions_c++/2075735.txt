#include<iostream>
using namespace std;
const int MAXN = 500 + 5;
const long long INF = 1e16;
long long a[MAXN];
long long dp[MAXN][MAXN];
int main()
{
    int n , k;
    cin >> n >> k;
    for(int i = 1;i <= n;i ++)
    {
        cin >> a[i];
        a[i] += a[i - 1];
    }
    for(int i = 0;i <= n;i ++)
        for(int j = 0;j <= k;j ++)
            dp[i][j] = INF;
    dp[0][0] = 0;
    for(int i = 1;i <= n;i ++)
        for(int j = 1;j <= k;j ++)
            for(int p = 1;p <= i;p ++)
            {
                long long t = a[i] - a[p - 1];
                dp[i][j] = min(dp[i][j] , dp[p - 1][j - 1] + t * t * (k - 1) - 2ll * t * a[p - 1]);
            }
    cout << dp[n][k] << endl;
    return 0;
}
