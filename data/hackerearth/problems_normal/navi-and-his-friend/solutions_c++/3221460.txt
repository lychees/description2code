#include <bits/stdc++.h>
using namespace std;
int price[20];
int weight[20];
int main()
{
	int t,n,x,y,w,c;
	cin>>t;
	for(int test=1;test<=t;test++)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&price[i],&weight[i]);
		}
		scanf("%d%d",&w,&c);
		long long int ans=-1;
		for(int i=0;i<pow(2,n);i++)
		{
			vector<int>v;
			for(int j=0;j<n;j++)
			{
				if(i&(1<<j))
				v.push_back(j);
			}
			if(v.size()>0&&v.size()<=c)
			{
				long long int temp1=0;
				long long int temp2=0;
				for(int k=0;k<v.size();k++)
				{
					temp1=temp1+price[v[k]];
					temp2=temp2+weight[v[k]];
				}
				if(temp2<=w)
				ans=max(ans,temp1);
			}
		}
		cout<<"For Day #"<<test<<":\n"<<ans<<"\n";
	}
	return 0;
}
