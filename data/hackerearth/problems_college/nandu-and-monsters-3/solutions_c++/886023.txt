#include<stdio.h>
#include<math.h>
#include<string.h>
char arr[100001];
void work(int a,int b)
{
    int l1,l2,t;
    l1=log(a)/log(2);
    l2=log(b)/log(2);
    //printf("%d %d\n",l1,l2);
    if(l2<l1)
    {
        t=a;
        a=b;
        b=t;
        t=l1;
        l1=l2;
        l2=t;
    }
    if(arr[a-1]=='1'||arr[b-1]=='1')
    {
        printf("No\n");

        return;
    }
    //printf("a=%d b=%d \n",a,b);
    for(;l2>l1;l2--)
    {
        b=b/2;
        if(arr[b-1]=='1')
        {
            printf("No\n");
            return;
        }
    }
    while(a!=b)
    {
        a=a/2;
        b=b/2;
        if(arr[a-1]=='1'||arr[b-1]=='1')
        {
        printf("No\n");
        return;
        }
    }
    printf("Yes\n");
}


int main()
{
    int n,q,i,a,b;
    scanf("%d%d",&n,&q);


    scanf("%s",&arr);



    while(q--)
    {
        scanf("%d%d",&a,&b);
        work(a,b);
    }
}
