#include <iostream>
using namespace std;

int main()
{
	int t, aa, bb, ab, ba, i, j;
	char str1[2004], str2[2004];
	cin>>t;
	for ( ; t; t--) {
		cin>>aa>>bb>>ab>>ba;
		i = 0; j = 0;
		while (aa >= 2) {
			str1[i++] = 'a'; str1[i++] = 'a';
			str2[j++] = 'a'; str2[j++] = 'a';
			aa = aa - 2;
		}
		while ((ab > 0) && (ba > 0)) {
			str1[i++] = 'a'; str1[i++] = 'b';
			str2[j++] = 'a'; str2[j++] = 'b';
			ab--;
			ba--;
		}
		while (bb >= 2) {
			str1[i++] = 'b'; str1[i++] = 'b';
			str2[j++] = 'b'; str2[j++] = 'b';
			bb = bb - 2;
		}
		if (aa == 1) {
			str1[i++] = 'a'; str1[i++] = 'a';
		} else if (bb == 1) {
			str1[i++] = 'b'; str1[i++] = 'b';
		}
		str1[i] = '\0';
		cout<<str1;
		for (i = (j-1); i>= 0; i--) {
			cout<<str2[i];
		}
		cout<<endl;
	}
    return 0;
}
