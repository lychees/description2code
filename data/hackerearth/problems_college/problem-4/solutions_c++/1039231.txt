#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char str[1001];
    cin>>str;
    int n=strlen(str);
    for(int i=0;i<n;i++)
	{
    	for(int j=i+1;j<n-1;j++)
    	{
    		if((int)str[i]>(int)str[j])
    		{
    			int temp=str[i];
    			str[i]=str[j];
    			str[j]=temp;
    		}
    	}
    }
    cout<<str;
    return 0;
}
