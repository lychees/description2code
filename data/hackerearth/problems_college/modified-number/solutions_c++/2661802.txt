#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[]={1,9,45 ,55, 99 ,297 ,703 ,999 ,2223 ,2728 ,7272 ,7777 ,9999 ,17344 ,22222 ,77778 ,82656 ,95121 ,99630 ,99999 };

int main()
{
    int p,q,c=0;
    cin>>p>>q;
    for(int i=0;i<=19;i++)
        if(a[i]>=p && a[i]<=q)
        {
            cout<<a[i]<<" ";
            c=1;
        }
        if(c==0)
            cout<<"INVALID RANGE";
        cout<<endl;
}
