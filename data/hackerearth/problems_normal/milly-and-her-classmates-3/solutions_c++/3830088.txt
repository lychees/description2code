#include <iostream>
using namespace std;
#include<vector>

int main() 
{
	int t;
	int n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		vector<long long> arr(n);
		for(int i=0;i<n;i++)cin>>arr[i];
		int beg=-1,end=-1;
		int flag=0;
		int i;
		for(i=0;i<n-1;)
		{
			if(arr[i+1]<arr[i])
			{
				flag=1;
				if(beg!=-1)break;
				//may  be beginning found
				beg = i;
				end=i+1;
				i++;
				while(i<n-1 && arr[i+1]<arr[i])
				{
					end++;i++;
				}
				i++;
				//beg to end ends.So this must be ans
			}
			else
			{
				i++;
			}
		}
		if(i<n-1 || flag==0)cout<<"-1"<<" "<<"-1"<<endl;
		else
		{if((beg==0 || arr[beg-1]<arr[end]) && (end==n-1 || arr[end+1]>arr[beg]))
		cout<<beg+1<<" "<<end+1<<endl;
		else
		cout<<"-1"<<" "<<"-1"<<endl;}
	}
	// your code goes here
	return 0;
}
