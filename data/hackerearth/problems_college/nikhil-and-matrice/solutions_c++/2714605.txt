#include <iostream>
using namespace std;
void output(int s)
{
    int i, k = 0, l = 0;
 	int m = s, n = s;
 	int a[s][s];
    /*  k - starting row index
        m - ending row index
        l - starting column index
        n - ending column index
        i - iterator
    */
 	int num = 1;
    while (k < m && l < n)
    {
        /* Print the first row from the remaining rows */
        for (i = l; i < n; ++i)
        {
            a[k][i] = num++;
        }
        k++;
 
        /* Print the last column from the remaining columns */
        for (i = k; i < m; ++i)
        {
            a[i][n-1] = num++;
        }
        n--;
 
        /* Print the last row from the remaining rows */
        if ( k < m)
        {
            for (i = n-1; i >= l; --i)
            {
                a[m-1][i] = num++;
            }
            m--;
        }
 
        /* Print the first column from the remaining columns */
        if (l < n)
        {
            for (i = m-1; i >= k; --i)
            {
                a[i][l] = num++;
            }
            l++;    
        }        
    }
    for(int p=0;p<s;p++) {
    	for(int q=0;q<s;q++) {
    		cout<<a[p][q];
    	}
    	cout<<endl;
    }
    cout<<endl;
}
int main()
{
	int n, s;
	cin>>n;
	while(n--) {
		cin>>s;
		output(s);
	}
    return 0;
}
