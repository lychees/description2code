#include<iostream>
#include<algorithm>

using namespace std;
int arr[10001];
int main()
{
	int finalSum[10001] = { 0 };
	int n,sum;
	sum = 0;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> arr[i];
	}
	if (arr[0] > 0)
		sum = finalSum[0] = arr[0];
	for (int i = 1; i < n; i++)
	{
		finalSum[i] = max(finalSum[i - 1] + arr[i], arr[i]);
		if (finalSum[i] > sum)
			sum = finalSum[i];
	}
	cout << sum;
	return 0;
}
