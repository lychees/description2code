#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main(){
	int t,a[26],i,count,c,l,j;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		memset(a,0,sizeof(a));
		for(i=0,count=0,c=0;i<s.length();i++){
			++a[(int)s[i]-97];
			if(a[(int)s[i]-97]==1)
			++c;
			if(c>3||(s.length())%3!=0){
				count=10;
				break;
			}
		}
		if(count==10)
		cout<<"Not OK";
		else{
			sort(a,a+26);
			if(a[25]==a[24]&&a[24]==a[23]){
				l=s.length()/3;
			for(i=0,count=0;i<s.length();i=i+l){
				for(j=i;j+1<l;j++){
					if(s[j]!=s[j+1]){
						count=10;
						break;
					}
				}
				if(count==10)
				break;
			}
			if(count!=10)
			cout<<"OK";
			else
			cout<<"Not OK";
			}
			else
			cout<<"Not OK";
		}
		cout<<endl;
	}
	return 0;
}
