#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n, i;
		cin>>n;
		int a[n], b[n], xr=0;
		for(i=0;i<n;++i) cin>>a[i];
		for(i=0;i<n;++i){
			cin>>b[i];
			xr ^=b[i]-1;
			xr^=a[i]-b[i];
		}
		if(xr==0) cout<<"Second"<<'\n';
		else cout<<"First"<<'\n';
	}
    return 0;
}
