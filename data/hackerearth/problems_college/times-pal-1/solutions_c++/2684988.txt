#include <bits/stdc++.h>

using namespace std;

int main()
{
        int t;
        string a,b,c,d;

        cin >> t;

        while(t--) {
                cin >> a >> b;
                c = b+a;
                d = c;
                reverse(c.rbegin(), c.rend());
                int count = 0;
                if(c == d) {
                        count++;
                }
//                cout << c << " " << d << endl;
                for(int i = 0; i < a.size(); i++) {
                        c = a.substr(0,i+1);
                        c = c+b;
                        c = c+a.substr(i+1,a.size()-i);
                        d = c;
                        reverse(c.rbegin(), c.rend());
                        if(c == d) {
                                count++;
                        }
  //                      cout << c << " " << d << endl;
                }
                printf("%d\n",count);
        }
        return 0;
}
