t = input()

for i in range(t):
    s = list(raw_input())
    t = list(raw_input())
    for each in s[:]:
        if each in t:
            s.remove(each)
            t.remove(each)

    s = set(s)
    t = set(t)
    inter = s.intersection(t)
    s -= inter
    t -= inter
    if len(s - t) != 0:
        print -1
    else:
        print len(t - s)
