#include <iostream>
using namespace std;

int main()
{
  string a;
  cin>>a;
  int q;
  cin>>q;
  while(q--)
  {
  	int l1,l2,l3,l4;
  	cin>>l1>>l2>>l3>>l4;
  	l1--;
  	l2=l2-l1;
  	l3--;
  	l4=l4-l3;
  
  	
  //	cout<<(a.substr(l1,l2))<<endl<<(a.substr(l3,l4))<<endl;
  	if(a.compare(l1,l2,a,l3,l4) == 0)
  	
  	cout<<"Yes\n";
  	else
  	cout<<"No\n";
  	
  	
  	
  }
  return 0;
 
}
