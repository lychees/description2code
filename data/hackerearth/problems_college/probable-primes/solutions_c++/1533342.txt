#include <iostream>
#include <stdio.h>
#define ll long int
#define MAXSIZE 1000000
using namespace std;
ll notprime[1000001]={0};

int main()
{
	ll i,k,cnt;
	notprime[1]=notprime[0]=-1;
	notprime[2]=1;
	cnt=2;
	for(i=3;i<=MAXSIZE;i=i+2)
	{
		if(notprime[i]==0)
		{
			notprime[i]=cnt;
			cnt++;
			k=2*i;
			while(k<=MAXSIZE)
			{
				notprime[k]=-1;
				k=k+i;
			}
		}
	}
	int t;
	ll l,r,j;
	float prob;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld %ld",&l,&r);
		j=r;
		while(j>=l && notprime[j]<=0)
			j--;

		if(j<l)
		{
			cnt=0;
		}
		else
		{
			i=l;
			while(i>0 && notprime[i]<=0)
				i--;

			if(i==0)
			{
				cnt=notprime[j];
			}
			else
			{	
				if(i==l)
					cnt=notprime[j]-notprime[i]+1;
				else
					cnt=notprime[j]-notprime[i];
			}
		}

		prob=(float)cnt/(r-l+1);
		printf("%.6f\n",prob);
	}
	return 0;
}
