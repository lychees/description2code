#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		long int n,m;
		cin >> n >> m;
		
		long int arr[n];
		for(long int i=0;i<n;i++)
			cin >> arr[i];
		long int brr[m];
		for(long int j=0;j<m;j++)
			cin >> brr[j];
		
		long int i=0,j=0;
		while(i<n && j<m)
		{
			if(arr[i] > brr[j])
			{
				cout << arr[i] << " ";
				i++;
			}
			else
			{
				cout << brr[j] << " ";
				j++;
			}
		}
		while(i<n)
		{
			cout << arr[i] << " ";
			i++;
		}
		while(j<m)
		{
			cout << brr[j] << " ";
			j++; 
		}
		cout << endl;
	}
	return 0;
}
