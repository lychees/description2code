#include <bits/stdc++.h>
using namespace std;
int binarySearch(int a[],int l,int h,int key)
{
	if(l<=h)
	{
		int mid=h-(h-l)/2;
		
		if(a[mid]==key)return 1;
		else if(a[mid]<key) return binarySearch(a,mid+1,h,key);
		else return binarySearch(a,l,mid-1,key);
	}
	return -1;
}
int main()
{
    int n,q,x;
    cin>>n>>q;
    int a[n];
    
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    
    for(int i=0;i<q;i++)
    {
    	cin>>x;
    	
    	if(binarySearch(a,0,n-1,x)!=-1) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
