int v [ MAX ] ;

int main (   )
{
    int n ;
    long long k ;
    cin >> n >> k ;
    FORN ( i , 1 , n )
    {
        cin >> v [ i ] ;
    }
    sort ( v + 1 , v + n + 1 ) ;
    long long CET = 0 ;
    FORN ( i , 1 , n - 2 )
    {
        FORN ( j , i + 1 , n - 1 )
        {
            int st = j + 1 ;
            int dr = n ;
            int soll =-1 ;
            while ( st <= dr )
            {
                int mij = ( st + dr ) >> 1 ;
                if ( 1LL * v [ i ] * v [ j ] * v [ mij ] <= k )
                {
                    soll = mij ;
                    st = mij + 1 ;
                }
                else {
                    dr = mij - 1 ;
                }
            }
            if ( soll != -1 )
            {
                CET = CET + soll - ( j + 1 ) + 1 ;
            }
        }
    }
    cout << CET << '\n' ;
    return 0;
}
