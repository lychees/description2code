#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int arr[100005];
    	for(int i=0;i<n;i++) cin>>arr[i];
    	map< int , pair< int,int > > m;
    	int l=0;
    	int a=0,b=0;
    	for(int i=0;i<n;i++)
    	{
    		if(m[arr[i]].first)
    		{
    			int r=m[arr[i]].second;
    			for(int j=l;j<r;j++) m[arr[j]].first=0;
    			if(b-a+1<r-l)
    			{
    				a=l;
    				b=r-1;
    			}
    			l=r+1;
    			m[arr[i]].second=i;
    			//cout<<a<<" if "<<b<<endl;
    		}
    		else 
    		{
    			m[arr[i]]=make_pair(1,i);
    			if(b-a+1<i-l+1)
    			{
    				a=l;
    				b=i;
    			}
    		//	cout<<a<<" else "<<b<<endl;
    		}
    	}
    	cout<<a+1<<" "<<b+1<<endl;
    }
    return 0;
}
