int main()
{
     std::ios_base::sync_with_stdio (false);
     cin.tie(NULL);
     cout.tie(NULL);
     int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<string>vv;
    	for(int i=0;i<n;i++)
    	{
    		string s;
    		cin>>s;
    		vv.pb(s);
    	}
    	int fl=1;
    	for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i==j)continue;
                string s1=vv[i],s2=vv[j];
                fl=0;
                for(int k=0;k<min(s1.size(),s2.size());k++)
                {
                    if(s1[k]!=s2[k])
                    {
                        fl=1;
                        break;
                    }
                    fl=0;
                }
                if(!fl)break;
            }
            if(!fl)break;
        }
    	if(!fl)cout<<"NO\n";
    	else cout<<"YES\n";
    }
    return 0;
}
