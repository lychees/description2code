#include <iostream>
using namespace std;

int main()
{
	int T;
	long N, M, A, B, C;
	long max;
    cin >> T;
    for(int i=0; i<T; i++){
    	cin >> N >> M >> A >> B >> C;
    	if((A+B) >= (2*C)){
    		max = (A*N)+(B*M);
    	}
    	else{
    		if(N<M){
    			max = (2*N*C)+((M-N)*B);
    		}
    		else{
    			max = (2*M*C)+((N-M)*A);
    		}
    	}
    
    	cout << max << endl;
    }
    
    return 0;
}
