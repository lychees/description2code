#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	long long n;
    	cin>>n;
    	long long arr[n];
    	for(int j=0;j<n;j++)
    	cin>>arr[j];
    	int count = 0;
    	for(int j=0;j<n;j++)
    	{
    		if(arr[j]%2!=0)
    		count++;
    	}
    	cout<<(n-count)*count<<endl;
    }
    return 0;
}
