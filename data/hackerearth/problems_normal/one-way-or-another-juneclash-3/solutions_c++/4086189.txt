#include<bits/stdc++.h>
using namespace std;

int T, N;
long long r, c, v, t;

int main() {
	scanf("%d", &T);
	while (T--) {
		scanf("%d", &N);
		scanf("%lld %lld %lld", &r, &c, &v);
		bool ok = 1;
		if ((r+c)&1) {
			if (v&1) t = 1;
			else t = 0;
		} else {
			if (v&1) t = 0;
			else t = 1;
		}
		for (int i = 0; i < N-1; i++) {
			scanf("%lld %lld %lld", &r, &c, &v);
			if ((r+c)&1) {
				if (t != (v&1)) ok = 0;
			} else {
				if (t != !(v&1)) ok = 0;
			}
		}
		puts (ok ? "Yes" : "No");
		
	}
    return 0;
}
