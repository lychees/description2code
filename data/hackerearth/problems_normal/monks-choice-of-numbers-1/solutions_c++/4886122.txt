#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int a[n],cnt[n];
    	memset(cnt,0,sizeof(cnt));
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
			while(a[i])
			{
				a[i]=a[i]&(a[i]-1);
				cnt[i]++;
			}
    	}
    	sort(cnt,cnt+n);
    	int sum=0;
    	int i=1;
    	while(k!=0)
    	{
    		sum+=cnt[n-i];
    		i++;
    		k--;
    	}
    	cout<<sum<<endl;
    }
}
