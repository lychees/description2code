#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,p,ans2,ans=INT_MIN;
    	cin>>n>>p;
    	int price[n+2],cost[n+2],dp[10000+2]={0};
    	for(int i=0;i<n;i++){
    		cin>>price[i]>>cost[i];
    	}
    	for(int i=0;i<n;i++){
    		for(int j=p;j>=cost[i];j--){
    			if(dp[j]<dp[j-cost[i]]+price[i]){
    				dp[j]=dp[j-cost[i]]+price[i];
    			}
    		}
    	}
    	for(int j=0;j<=p;j++){
    		if(dp[j]-j>ans){
    			ans=dp[j]-j;
    			ans2=j;
    		}
    	}
    	cout<<ans2<<" "<<ans+p<<endl;
    }
    return 0;
}
