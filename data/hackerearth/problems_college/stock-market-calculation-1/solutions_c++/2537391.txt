#include <iostream>
using namespace std;

int main()
{
    long long int test,n,i,j,k,ans,l,prev,sum;
    cin>>test;
    while(test--){
    	cin>>n;
    	j=0;
    	long long int arr[n+2];
    	for(i=0;i<n;i++){
    		cin>>arr[i];
    		if(arr[i]>=j){
    			j=arr[i];
    			k=i;
    		}
    	}
         i=0;
         prev=0;
         ans=0;
         while(prev<n){
         	sum=0;
         	for(l=prev;l<k;l++){
         		sum+=arr[l];
         	}
         	ans+=(k-prev)*j-sum;
         	prev=k+1;
         	j=0;
         	for(l=prev;l<n;l++){
         		if(arr[l]>=j){
         			j=arr[l];
         			k=l;
         		}
         	}
         }
         cout<<ans<<endl;
    }
    return 0;
}
