#include <bits/stdc++.h>

using namespace std;
char s[25], ans[1000005];

int main()
{
    int t, la, l, k;
    for(scanf("%d", &t);t--;)
    {
        scanf("%s", s);
        l = strlen(s);
        k = 0;
        for(int i= 0;i < l;++i)
        {
            if(s[i] >= 'a' and s[i] <= 'z')
                ans[k++] = s[i];
            else if(s[i] >= 'A' and s[i] <= 'Z')
                ans[k++] = s[i];
            else
            {
                la = k;
                for(int j = 1;j < (s[i] - '0');++j)
                {
                    for(int p = 0;p < la;++p)
                    {
                        ans[k++] = ans[p];
                    }
                }
            }
        }
        ans[k] = '\0';
        printf("%s\n", ans);
    }
    return 0;
}
