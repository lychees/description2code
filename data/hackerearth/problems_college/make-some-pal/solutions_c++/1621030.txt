#include <iostream>
#include <string>
#include <vector>

using std::cout;
using std::cin;
using std::vector;
using std::string;

int main()
{
    int T;
    cin>>T;
    string s;
    while(T)
    {
    	  int min;
    	  bool diff;
    	  int y;
    	  cin>>s;
    	  int n=s.length();
    	  vector <string> w;
    	  for(int i=0;i<n;i++)
    	  {
    	  	    string y=s.substr(i,n-i);
    	  	    w.push_back(y);
    	  }
    	  for(int j=0;j<w.size();j++)
    	  {
    	  	  bool palin;
    	  	  int m=w[j].length();
              string z=w[j];
    	  	  int y=0;
    	  	  for(int x=m-1;x>y;x--)
    	  	  {
    	  	  	    if(z[x]==z[y])
    	  	  	    {
    	  	  	    	 palin=true;
    	  	  	    	 y++;
    	  	  	    }
    	  	  	    else
    	  	  	    {
    	  	  	    	palin=false;
    	  	  	    	diff=true;
    	  	  	    	break;
    	  	  	    }
    	  	  }
    	  	  if(palin==true)
    	  	  {
    	  	  	    min=n-m;
    	  	  	    diff=false;
    	  	  	    cout<<min+n<<"\n";
    	  	  	    break;
    	  	  }
    	  }
    	  if(diff==true)
    	  {
    	  	       cout<<2*n-1<<"\n";
    	  }
    	  T--;
    }
}
