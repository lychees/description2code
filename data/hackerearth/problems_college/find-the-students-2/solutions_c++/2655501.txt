#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long test;
    cin>>test;
    long long arr[test];
    for(int i=0;i<test;i++)
    	cin>>arr[i];
    	
    sort(arr,arr+test);
    
    int c=1,r=test;
    cout<<test<<endl;
    for(int i=1;i<test;i++){
    	if(arr[i]==arr[i-1]){
    		c++;
    	}else{
    		r-=c;
    		c=1;
    		cout<<r<<endl;
    	}
    }
    return 0;
}
