#define CLR(a,b) memset(a,b,sizeof(a))
#define REPN(x,a,b) for (int x=a; x<b;++x)
#define REP(x,b) REPN(x, 0, b)

char A[1020], B[1020];

int main() {
    int TC;
    scanf("%d", &TC);
    while (TC--) {
        scanf("%s", A);
        scanf("%s", B);
        int na = strlen(A), nb = strlen(B);
        int res = na;
        REP(i, na-nb+1) {
            int cnt = 0;
            REP(j, nb) {
                if (A[j+i] != B[j]) cnt++;
            }
            res = min(cnt, res);
        }
        printf("%d\n", res);
    }

    return 0;
}
