#include<iostream>
#define _USE_MATH_DEFINES
#include<cmath>
#define e 2.71828182845904523536028747135266250
using namespace std;

int main()
{long long int t,n,ans,div,k;

    cin>>t;

    while(t)
    {
        cin>>n;         //(log(2?n)/2+n(logn?loge))/log10]+1

        ans=(((log(2*n*M_PI))/2+n*(log(n/e)))/log(10));
        ans=ans+1;
        div=1;k=5;
        while(div!=0)
        {
         div=n/k;
         ans-=div;
         k=k*5;
        }
        cout<<ans<<endl;
        t--;
    }
    return 0;
}
