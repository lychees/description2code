#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,count=0;
    	cin>>n;
    	map<int,int>pushpa;
    //	map<ll>::iterator it;
    	for(ll i=0;i<n;i++)
    	{
    		ll temp;
    		cin>>temp;
    		pushpa[temp%10]++;
    		if(pushpa[temp%10]>=2)
    		{
    			count++;
    			pushpa[temp%10]=1;
    		}
    	}
        cout<<count<<endl;
    }
}
