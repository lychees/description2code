n=int(raw_input())
for j in range(n):
	n,k=raw_input().split()
	n=list(n)
	n=map(int,n)
	k=list(k)
	k=map(int,k)
	s=min(len(n),len(k))
	i=s-1
	carry=0
	while i>=0:
		m=n[i]+k[i]
		if m>=10 and i!=0:
			carry+=1
			n[i-1]+=1
		elif m>=10:
			carry+=1
		m=0
		i-=1
	if carry==0:
		print "No carry operation"
	elif carry==1:
		print "1 carry operation"
	else:
		print "%s carry operations" % carry
