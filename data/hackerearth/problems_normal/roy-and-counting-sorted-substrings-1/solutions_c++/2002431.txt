#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int t,n,i,l,ans,p;
    string s;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>s;
    	l=s.size();
    	ans=l;
    	p=1;
    	for(i=0;i<l;i++)
    	{
    		if(s[i+1]>=s[i])
    		{
    			ans+=p;
    			p++;
    		}
    		else
    			p=1;
    	}
    	cout<<ans<<"\n";
    }
    return 0;
}
