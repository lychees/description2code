using namespace std;
typedef long long ll;
int prime[1000105],prime_num[78600],k=0;
int phival[1000002];
void mm(int m)
{
	int i=2;
	while(i*m<=1000103)
	{
		prime[i*m] = 1;
		i++;
	}
}
void sieve()
{
	prime[1]=1;
	for(int i=2;i<=1000100;i++)
		if(!prime[i])
		{
			mm(i); prime_num[k++]=i;
		}
}
int main()
{
	sieve();k=1;phival[1]=1;
	for(int i=2;i<=1000000;i++)
	{
		if(i>=prime_num[k])
			k++;
		phival[i] = prime_num[k];
	}
	//cout << "k : " << k << endl;
	int t; get(t);
    while(t--)
    {
    	int n; get(n);
    	int x; ll sum=0;
    	for(int i=0;i<n;i++)
    	{
    		get(x); sum += phival[x];
    	}
    	printl(sum);
    }
    return 0;
}
