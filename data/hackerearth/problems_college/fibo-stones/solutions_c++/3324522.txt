ll modpow(ll base, ll exponent, ll modulus)
{
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}

ll gcd(ll u, ll v)
{
    return (v != 0) ? gcd(v, u % v) : u;
}

ll mulmod(ll a, ll b, ll c){
    ll x = 0, y = a%c;
    while(b > 0){
        if(b%2 == 1){
            x = (x+y) % c;
        }
        y = (y*2) % c;
        b /= 2;
    }
    return x%c;
}

int n, m; bool taken[101];
ll fib[101];

int main() {
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    fib[0] = fib[1] = 1;
    for(int i = 2; i < 80; ++i)
        fib[i] = fib[i-1]+fib[i-2];
    cin >> n >> m;
    rep(i, 0, m) {
        ll x;
        cin >> x;
        rep(j, 0, n) {
            if(fib[j] == x && !taken[j]) {
                taken[j] = 1;
                break;
            }
        }
    }
    ll ans = 0;
    for(int i = 0; i < n; ++i)
        if(!taken[i])
            ans += fib[i];
    cout << ans;    
    return 0;
}
