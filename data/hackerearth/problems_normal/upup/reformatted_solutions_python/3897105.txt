inp = raw_input()
# print(len(inp))
listStr = list(inp)
start = 0
end = start
while (end < len(listStr)):
    while (start < len(listStr) and listStr[start] == ' '):
        start = start + 1

    end = start + 1
    while (end < len(listStr) and listStr[end] != ' '):
        end = end + 1

    if (start < len(listStr)):
        # print str(start) + " has character " + listStr[start]
        listStr[start] = listStr[start].upper()
        # print str(start) + " has character " + listStr[start]

    start = end

inp = "".join(listStr)
# print(len(inp))
print(inp)
