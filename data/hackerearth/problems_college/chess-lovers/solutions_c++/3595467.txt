#include <iostream>
using namespace std;

int main()
{
    int x1,x2,y1,y2,t;
    cin>>t; 
    while(t--)
    {
    cin>>x1>>y1>>x2>>y2;
    if((x1+y1)%2!=(x2+y2)%2) 
    {
    	cout<<"-1\n";
    }
    else if( abs(x1-x2)==abs(y1-y2) )
    {
    	cout<<"0\n";
    }
    else cout<<"1\n";
    }
    return 0;
}
