#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MAX 20000

ll dp[200][MAX];
int main()
{
  int test;
  cin>>test;int t;int i,j;
  for(t=0;t<test;t++)
  {
    memset(dp,0,sizeof(dp));
    for(i=0;i<200;i++)
    {
      dp[i][0]=1;
    }
    int n;
    cin>>n;
    ll arr[n+1];
    for(i=1;i<=n;i++)
    {
      cin>>arr[i];
    }

    for(i=1;i<=n;i++)
    {
      for(j=1;j<MAX;j++)
      {
        if (j>=arr[i])
        dp[i][j]=max(dp[i-1][j],dp[i-1][j-arr[i]]);
        else
        {
          dp[i][j]=dp[i-1][j];
        }
      }
    }
    ll count1=0;
    for(i=0;i<MAX;i++)
    {
    if (dp[n][i]==1)
      count1++;
    }
    cout<<count1<<"\n";
  }






}
