T = input()

for ii in range(T):
	N = input()
	days = map(int, raw_input().split())
	
	color_to_index = {}
	first_index = 0
	max_so_far = 0
	so_far = 0
	max_first_index = first_index
	max_last_index = first_index

	for index in range(N):
		if not color_to_index.has_key(days[index]):
			so_far += 1
			color_to_index[days[index]] = index
			if max_so_far < so_far:
				max_first_index = first_index
				max_last_index = index
				max_so_far = so_far
		else:
			matched_index = color_to_index.get(days[index])
			color_to_index[days[index]] = index
			if matched_index < first_index:
				so_far += 1
				if max_so_far < so_far:
					max_first_index = first_index
					max_last_index = index
					max_so_far = so_far
			else:
				first_index = matched_index + 1
				so_far = index - first_index + 1

	print str(max_first_index + 1) + " " + str(max_last_index + 1)
