#include <iostream>
using namespace std;
#define mod 1000000007

int main()
{
	int test;
	cin>>test;
	while(test--) {
		string first , second;
		cin>>first>>second;
		int hash1[26] ={0};
	 	int hash2[26] = {0};
	 	int i = 0;
		while(first[i]) {
			hash1[first[i]- 'a']++;
			i++;
		}
		i = 0;
		while(second[i]) {
		  	hash2[second[i]-'a']++;
		  	i++;
		}
		long long int answer = 1;
		for(i =0; second[i]; i++) {
			answer *= hash1[second[i]- 'a'];
			hash1[second[i]- 'a']--;
			answer%=mod;
		}
		cout<<answer<<endl;
 	}
 	return 0;
}
