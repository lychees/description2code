#include <iostream>
#define MODVAL 1000000007
using namespace std;

int main()
{
	int T;
	string A,B;
	int Achar[26],Bchar[26];
	long long count = 1;
	cin>>T;
	
	for(int i=0;i<T;i++){
		cin>>A>>B;
		for(int i=0;i<26;i++){ Achar[i]=0;  Bchar[i]=0; }
				//for(int i=0;i<26;i++){ cout<<i<<" "<<Achar[i]<<" "<<Bchar[i]<<endl; }

		for(auto &ele : A){	Achar[ele-97]++; }
		for(auto &ele : B){	Bchar[ele-97]++; }
		
				//for(int i=0;i<26;i++){ cout<<i<<" "<<Achar[i]<<" "<<Bchar[i]<<endl; }
		count =1;
		int flag=0;
		for(int i=0;i<26;i++){ 
			while(Bchar[i]){
				flag=1;
				if(Achar[i] == 0)
					count = 0;
				count = (count * Achar[i]) % MODVAL;
				Achar[i]--;
				Bchar[i]--;
			}	
		}
		if(flag==0)
			count=0;
		cout<<count<<endl;
	}
return 0;
}
