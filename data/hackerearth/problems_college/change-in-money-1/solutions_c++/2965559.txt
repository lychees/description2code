#include<iostream>
#include<algorithm>

using namespace std;
int coins[50];
int main() {

	int c, n, m;
	cin >> c >> m;
	for (int i = 0; i < m; i++)
	{
		cin >> coins[i];
	}
	//sort(coins, coins + m);
	long long total[251] = { 0 };
	total[0] = 1;
	for (int i = 0; i < m; i++)
	{
		for (int j = coins[i]; j <= c; j++) {
			total[j]  += total[j - coins[i]];
		}
	}
	if(c!=0)
	cout << total[c];
  else
  cout<<"0";
	return 0;
}
