#include <iostream>
using namespace std;

int main()
{
    int hr,min;
    scanf("%d:%d",&hr,&min);
    double minangle = min*6.0;
    double hrangle = hr*30.0+min*(30.0/60);
    double result ;
    if(hrangle>minangle){
    	result = hrangle-minangle;
    }else{
    	result = minangle-hrangle;
    }
    if(result>180.0){
    	result = 360.0-result;
    }
    printf("%6lf",result);
}
