#include <iostream>
#include <vector>
using namespace std;

int main()
{
    long t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	vector<int>v;
    	while(n>0)
    	{
    		v.push_back(n%10);
    		n/=10;
    	}
    	long sum=0,num=0;
    	for(int i=0;i<v.size();i++) sum+=v[i];
    	if(sum==1) cout<<"NO\n";
    	else
    	{
    		for(int j=2;j<sum;j++)
    		{
    			if(sum%j==0)
    			{
    				num=1;
    				cout<<"NO\n";
    				break;
    			}
    		}
    		if(num==0) cout<<"YES\n";
    	}
    }
    return 0;
}
