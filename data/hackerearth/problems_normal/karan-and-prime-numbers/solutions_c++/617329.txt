#include<iostream>
#include<cmath>
using namespace std;

struct node{
	long data;
	node* right;
};

int main(){
	int nooftest;
	cin>>nooftest;
	for(int z=1;z<=nooftest;z++){
		node *head,*temp,*rightmost;
		head=new node;
		head->data=2;
		head->right=NULL;
		temp=head;
		rightmost=head;
		int lowlimit,uprlimit;
		//int count=1;
		cin>>lowlimit;
		cin>>uprlimit;
		//int maxnoofprime;
		//cin>>maxnoofprime;
		for(int i=3;i<=uprlimit;i=i+2){
			for(temp=head;temp!=NULL && temp->data<=sqrt(i);temp=temp->right){
				if(i%(temp->data)==0){
					break;
				}
			}
			if(temp==NULL || temp->data>sqrt(i)){
				temp=new node;
				temp->data=i;
				temp->right=NULL;
				rightmost->right=temp;
				rightmost=temp;
			}
		}
		long sum=0;
		for(temp=head;temp!=NULL;temp=temp->right){
			if(temp->data>=lowlimit && temp->data<=uprlimit){
				sum=sum+temp->data;
			}
		}
		cout<<sum<<'\n';
	}
	
}
