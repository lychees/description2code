int main()
{
	ios_base::sync_with_stdio(false);
	int fin[105],mark[105],n,ans=0;
	cin>>n;
	FOR(i,0,n) cin>>fin[i]; FOR(i,0,n) cin>>mark[i];
	sort(mark,mark+n); sort(fin,fin+n); //reverse(fin,fin+n);
	FOR(i,0,n)
	{
		int val=fin[i],index=-1;
		FOR(j,0,n)
		{
			if(mark[j]<val)
				index=j;
		}
		if(index!=-1)
		{
			//if(mark[index]==val) ans+=50;
			ans+=100;
			mark[index]=INT_MAX;
			fin[i]=-1;
		}
	}
	FOR(i,0,n)
	{
		int index=-1;
		if(fin[i]!=-1)
		{
			int val=fin[i];
			FOR(k,0,n)
			{
				if(mark[k]==val)
					index=k;
			}
			if(index!=-1)
			{
				mark[index]=INT_MAX;
				ans+=50;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
