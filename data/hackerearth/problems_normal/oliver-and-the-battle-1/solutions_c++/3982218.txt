#include <bits/stdc++.h>
using namespace std;
int adj[1001][1001];
int vis[1001][1001];
int n,m;

int bfs(int a,int b){
	int i,j,cnt=1;
	queue<pair<int,int> > q;
	q.push(make_pair(a,b));
	vis[a][b]=1;
	while(!q.empty()){
		int x= q.front().first;
		int y= q.front().second;
		q.pop();
		for(i=x-1;i<x+2;i++){
			for(j=y-1;j<y+2;j++){
				if(adj[i][j] && !vis[i][j]){
					q.push(make_pair(i,j));
					vis[i][j]=1;
					cnt++;
				}
			}
		}
	}
	return cnt;
}
int main()
{
	int t,i,j,cnt,ans;
	cin >> t;
	while(t--){
		cin >> n >>m;
		memset(vis,0,sizeof(vis));
		memset(adj,0,sizeof(adj));
		for(i=1;i<=n;i++){
			for(j=1;j<=m;j++){
				cin >> adj[i][j];
			}
		}
	    cnt=0;ans=0;
		for(i=1;i<=n;i++){
			for(j=1;j<=m;j++){
				if( adj[i][j] && !vis[i][j]){
					ans=max(ans,bfs(i,j));
				//	cout << ans << endl;
					cnt++;
				}
			}
		}
		cout << cnt << " " << ans << endl;
	}
    return 0;
}
