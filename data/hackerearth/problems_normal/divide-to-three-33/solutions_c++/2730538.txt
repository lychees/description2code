#define inf 2000000000

int maxx = inf ,N = 0 ;
vi a ;
void solve(int i,int j,int k,int pos){
	if(pos==N){
		int res = max(i,max(j,k)) ;
		if(res<maxx) maxx=res ;
	}else{
		solve(i+a[pos],j,k,pos+1) ;
		solve(i,j+a[pos],k,pos+1) ;
		solve(i,j,k+a[pos],pos+1) ;
	}
	return ;
}

	rep(i,0,n) cin>>a[i] ;
