#include <iostream>
#include<math.h>
using namespace std;


int main()
{
    long long int t,n,ctr=0,i;
    cin>>t;
    while(t--)
    {
        ctr=0;
        cin>>n;
        long long int current=n;
        for(i=2;i<=sqrt(n);++i)
        {
            if(n%i==0)
            {
            	while(n%i==0)
            	{
            		n=n/i;
            	}
            	current=current-current/i;
            }
        }
        if(n!=1)
        current=current-current/n;
        cout<<current<<"\n";
    }
    return 0;
}
