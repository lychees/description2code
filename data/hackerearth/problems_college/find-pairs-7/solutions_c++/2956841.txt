#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t!=0)
    {
    	int n,s,cnt=0;
    	cin>>n;
    	vector<int >a;
    	for(int i=0;i<n;i++)
    	{
    		cin>>s;
    		a.push_back(s);
    		for(int j=0;j<=i;j++)
    		if(a[i]==a[j])
    		cnt++;
    	}
    	cout<<cnt<<endl;
    	t--;
    }
    return 0;
}
