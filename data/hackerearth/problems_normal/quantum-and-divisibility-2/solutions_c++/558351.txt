int arr[100010];

map<long, long> calc(int x) {
    map<long, long> res;
    for (int p = 2; p*p <= x; p++) {
        if (x % p == 0) {
            long r = 0;
            while (x % p == 0) {
                x /= p;
                r++;
            }
            res[p] = r;
        }
    }
    if (x != 1) {
        res[x] = 1;
    }
    return res;
}

int main(void) {
	//freopen("input.txt", "r", stdin);

	int t;
	int a, b, c, d;
	cin >> t;
	while(t--) {
		cin >> a >> b >> c >> d;	
		map<long, long> x = calc(a);
		map<long, long> y = calc(c);

		bool flag = true;
		map<long, long> :: iterator it;
		for(it = y.begin(); it != y.end(); ++it) {
			if(it->second * d > x[it->first] * b) {
				flag = false;
				break;
			}
		}

		if(!flag) {
			cout << "Not divisible" << endl;
		}
		else
			cout << "Divisible" << endl;
	}
	return 0;
}
