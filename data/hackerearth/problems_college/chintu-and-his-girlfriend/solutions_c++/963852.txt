#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int i,j,k,p=0,n;
    char s1[100000];
    int a[30]={0};
    cin>>n>>k;
    for(j=0;j<n;j++)
    {
    scanf("%s",&s1);

    for(i=0;i<strlen(s1);i++)
    {
        if(a[s1[i]-97]==j)
            a[s1[i]-97]++;
    }
    }


    for(i=0;i<26;i++)
    {
        if(a[i]==n)
            p++;
    }
    if(p==k)
        cout<<":-)";
    else
        cout<<":-(";

}
