#include<cstdio>
#include<iostream>
#include<map>
using namespace std;
void givevalue(map<string, int> &a){
	int x=1;
	for(map<string, int>::iterator it=a.begin();it!=a.end();it++){
	it->second=x++;
		}
}

int main(){
ios_base::sync_with_stdio(false);
cin.tie(NULL);
	int n;
	cin>>n;
		map<string,int> mymap;
		
		string input;
		
		for(int i=0;i<n;i++){
			cin>>input;
			mymap[input]=0;
		}		
		int q;
		cin>>q;
		givevalue(mymap);
		while(q--){
		string query;
		cin>>query;
		int ans=mymap[query];
		if(ans==0)	
		ans=-1;
	cout<<ans<<"\n";
		}
}
