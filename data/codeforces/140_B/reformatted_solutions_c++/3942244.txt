#include <algorithm>

#include <vector>
#define PI acos(-1.0)
using namespace std;
int n, a[305][305], p[305];
int main() {
  cin >> n;
  for (int i = 1; i <= n; ++i)
    for (int j = 1; j <= n; ++j)
      cin >> a[i][j];
  for (int i = 1; i <= n; ++i)
    cin >> p[i];
  for (int i = 1; i <= n; ++i) {
    for (int j = 1; j <= n; ++j) {
      if (a[i][j] == i)
        continue;
      int k;
      for (k = 1; k <= n; ++k)
        if (p[k] != i && p[k] <= a[i][j])
          break;
      if (p[k] == a[i][j]) {
        cout << a[i][j];
        break;
      }
    }
    if (i < n)
      cout << " ";
    else
      cout << endl;
  }
  return 0;
}
