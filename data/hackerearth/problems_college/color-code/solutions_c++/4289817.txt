#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,count1=1;
	cin >> n;
	while(n--)
	{
		int a[3];
		cin >>a[0]>>a[1]>>a[2];
		cout << "Case" << " " <<count1<< ":" << " ";
 		for(i=0;i<3;i++)
		{
			int rem;
			rem = a[i]/16;
			if(rem < 10)
			{
				cout << rem ;
			}
			else
			{
				if(rem==10) cout << "U";
				if(rem==11) cout << "V";
				if(rem==12)	cout << "W";
				if(rem==13) cout << "X";
				if(rem==14) cout << "Y";
				if(rem==15)	cout << "Z";
			}
			rem = a[i]%16;
			
			if(rem < 10)
			{
				cout << rem;
			}
			else
			{
				if(rem==10)	cout << "U";
				if(rem==11) cout << "V";
				if(rem==12) cout << "W";
				if(rem==13) cout << "X";
				if(rem==14) cout << "Y";
				if(rem==15) cout << "Z"; 
			}
			cout << " ";
		}
		count1++;
		cout << endl;
	}
}
