#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int x;
    cin>>t;
    while(t--){
    	cin>>x;
    	int c=0;
    	if(x>=1000){
    		c+= (x/1000);
    		x%=1000;
    	}
    	
    	if(x>=500){
    		c+= (x/500);
    		x%=500;
    	}
    	
    	if(x>=100){
    		c+= (x/100);
    		x%=100;
    	}
    	
    	if(x>=50){
    		c+= (x/50);
    		x%=50;
    	}
    	
    	if(x>=20){
    		c+= (x/20);
    		x%=20;
    	}
    	
    	if(x>=10){
    		c+= (x/10);
    		x%=10;
    	}
    	
    	if(x>=5){
    		c+= (x/5);
    		x%=5;
    	}
    	
    	if(x>=2){
    		c+= (x/2);
    		x%=2;
    	}
    	
    	if(x>=1){
    		c+= (x/1);
    		x%=1;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
