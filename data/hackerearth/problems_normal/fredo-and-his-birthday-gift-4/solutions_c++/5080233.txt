#include<iostream>

using namespace std;

#define MAXN 15
#define MAX 100000000

bool adj[MAXN][MAXN];
bool mic[MAXN][1<<MAXN];


int countbit(int n){
    int co = 0;
    while(n){
        n &= n-1;
        co++;
    }
    return co;
}

void solve(int n, int m){
    for(int i=0;i<n;i++){
        for(int j=0;j<(1<<n);j++) mic[i][j] = 0;
    }
    for(int i=0;i<n;i++){
        mic[i][1<<i] = 1;
    }
    for(int i=0;i<(1<<n);i++){
        for(int j=0;j<n;j++){
            if(i&(1<<j)){
                for(int k=0;k<n;k++){
                    if(adj[j][k] && mic[k][i^(1<<j)] && (i&(1<<k))){
                        mic[j][i] = 1;
                        break;
                    }
                }
            }
        }
    }
    for(int i=0;i<n;i++){
        int ans = 0;
        for(int j=0;j<(1<<n);j++){
            ans = max(ans, (countbit(j)-1)*mic[i][j]);
        }
        cout << ans << " ";
    }
    cout << endl;
}

void init(){
    for(int i=0;i<MAXN;i++) for(int j=0;j<MAXN;j++) adj[i][j] = 0;
}

int main(){
    int t;
    cin >> t;
    while(t--){
        init();
        int n, m;
        cin >> n >> m;
        for(int i=0;i<m;i++){
            int x, y;
            cin >> x >> y;
            x--; y--;
            adj[x][y] = 1;
            adj[y][x] = 1;
        }
        solve(n, m);
    }
}
