#include <iostream>
using namespace std;

int main()
{
    long long int n,pos=0,max=0;
    cin>>n;
    while(n--)
     {
     	long long int t,p;
     	cin>>t>>p;
     	if(t>=max)
     	 {
     	 	max=t;
     	 	pos=p;
     	 }
     }
     if(pos>0)
      cout<<"Left Corner"<<endl;
     else if(pos<0)
      cout<<"Right Corner"<<endl;
     else
      cout<<"Either Corner"<<endl;
    return 0;
}
