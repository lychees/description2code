#include <iostream>
using namespace std;

int reverse(int n)
{
	int a=0;
	int x=n;
	while(x)
	{
		a=a*10+x%10;
		x=x/10;
	}
	if(a==n)
	return 1;
	else
	return 0;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int a,b;
    	cin>>a>>b;
    	int ans=0;
    	for(int i=a;i<=b;i++)
    	{
    		ans+=reverse(i); 
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
