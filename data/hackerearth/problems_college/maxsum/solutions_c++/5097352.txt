#include<bits/stdc++.h>
using namespace std;
int arr[10004],n,dp[10004];
int maxsum(int idx)
{
    if(idx>=n)
        return 0;
    if(dp[idx]!=-1)
        return dp[idx];
    dp[idx]=max(arr[idx]+maxsum(idx+2),maxsum(idx+1));
        return dp[idx];
}
int main()
{
    ios::sync_with_stdio(false);
    int t,i;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>arr[i];
        memset(dp,-1,sizeof(dp));
        cout<<maxsum(0)<<"\n";
    }
    return 0;
}
