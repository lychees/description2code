#include <bits/stdc++.h>
 
using namespace std;

#define reads() freopen("input.txt", "r", stdin)
#define write() freopen("output.txt", "w", stdout)

# define MOD 1000000007
# define token 1.414213562

template<class T> inline void read(T& x) {
    char ch;
    while(!isdigit(ch = getchar_unlocked()));
    x = ch-'0';
    while( isdigit(ch = getchar_unlocked())) x *= 10, x += ch-'0';
}

long n;
vector <double> a,b;

int main(){
	//cout << token << "\n";
	cin >> n;
	double first,second;
	while(n--){
		cin >> first >> second;
		if(first == 2){
			a.push_back(second);
		}
		else{
			b.push_back(second);
		}
	}
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	long answer = 0;
	long i = 0;
	long j = 0;
	while(b[i] < (a[j]/token)){
		if(i >= b.size()){
			break;
		}
		i++;
	}
	while(1){
		if(i >= b.size() || j >= a.size()){
			break;
		}
		if(b[i] > (a[j]/token) && b[i] < (a[j]*token) ){
			i++;
			j++;
			answer++;
		}
		else if(b[i] < (a[j]/token)){
			i++;
		}
		else if(b[i] > (a[j]*token)){
			j++;
		}
	}
	cout << answer << "\n";
	return 0;
}
