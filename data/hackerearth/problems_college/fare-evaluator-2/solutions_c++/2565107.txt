#include <iostream>
using namespace std;

int main()
{
    int test;
    cin>>test;
    
    while(test--) {
    	int d,wt;
    	cin>>d>>wt;
    	if(d>60) {
    		cout<<"NA"<<endl;
    		continue;
    	}
    	
    	int ans=0;
    	if(d<=2) {
    		ans = 50;
    	}
    	else {
    		ans = 50 + (d-2)*12;
    	}
    	
    	ans += wt*2;
    	cout<<ans<<endl;
    }
    return 0;
}
