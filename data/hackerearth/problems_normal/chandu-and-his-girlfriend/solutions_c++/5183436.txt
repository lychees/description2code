#include<stdio.h>

void merge(long long int a[],int l, int p,int h)
{
       int j,i,t,x,y,k;    
       x = l;
       y = p+1;
       k = 0;
       long long int b[h-l+1];
       for(i=l;i<=h;i++)
       {
              if(x>p)
              {
                     b[k] = a[y];
                     k++;
                     y++;
              }
              else if(y>h)
              {
                     b[k] = a[x];
                     k++;
                     x++;
              }
              else if(a[x]>a[y])
              {
                     b[k] = a[x];
                     x++;
                     k++;
              }
              else
              {
                     b[k] = a[y];
                     y++;
                     k++;
              }
       }
       for(i=0;i<k;i++)
       {
              a[l] = b[i];
              l++;
       }
       return;
}

void sort(long long int a[],int l,int h)
{
       if(l<h)
       {
              int p = (l+h)/2;
              //printf("%d ",p);
              sort(a,l,p);
              sort(a,p+1,h);
             
              merge(a,l,p,h);
       }
       return;
}

{
       int t;
       scanf("%d",&t);
       while(t--)
       {
              int n,i;

              long long int a[n];
              for(i=0;i<n;i++)
              {
                     scanf("%lld",&a[i]);
              }
              sort(a,0,n-1);
              for(i=0;i<n;i++)
              {
                     printf("%lld ",a[i]);
              }

       }

}
