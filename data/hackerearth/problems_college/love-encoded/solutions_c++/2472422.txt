#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        char a[50];
        int c,b;
        cin>>c;
        cin>>a;
        b='Z'-'A'+1;
        for(int j=0;j<strlen(a);j++)
        {
            a[j]=a[j]-c;
            if(a[j]<'A')
                a[j]+=b;
        }
        cout<<a<<endl;
    }
    return 0;
}
