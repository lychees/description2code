#include <iostream>
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int dp[100001];
    dp[0] = 1, dp[1] = 1, dp[2] = 2;
    for(int i=3; i<=100000; i++)
    dp[i] = ((dp[i-1])%mod + (dp[i-2])%mod + (dp[i-3])%mod)%mod;
    while(t--)
    {
    	long long int N;
    	cin>>N;
    	cout<<dp[N];
    	cout<<"\n";
    }
    return 0;
}
