#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll a[100005];
ll lsum[100005],rsum[100005];
int main()
{
   // printf("Hello World!\n");
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	int i;
   	for(i=1;i<=n;i++)scanf("%lld",&a[i]);
   lsum[n+1] = rsum[n+1] = 	a[n+1] = 0;
   	for(i=1;i<=n;i++)lsum[i] = lsum[i-1] + a[i];
   	for(i=n;i>=1;i--)rsum[i] = rsum[i+1] + a[i];
   	ll ma = (ll)1<<60;
   	for(i=1;i<n;i++)
   	{
   		ll dif = abs(lsum[i] - rsum[i+1]);
   		ma = min(ma,dif);
   	}
   	printf("%lld\n",ma);
   }
    return 0;
}
