#include<iostream>
using namespace std;
#include<vector>
#include<algorithm>
#include<cmath>

typedef long long ll;
int main(){
    ll n;
    cin>>n;
    ll x,i;
    vector<ll> v;
    for(i=0;i<n;i++){
        cin>>x;
        v.push_back(x);
    }
    sort(v.begin(),v.end());
    ll res=0;
    for(i=0;i<=(n-1)/2;i++)
        res+=(abs(v[n-1-i]-v[i])+abs(v[n-1-i]-v[i+1]));
    if(n%2==0)
        res+=abs(v[(n)/2]-v[0]);
    cout<<res;
return 0;
}
