#include <bits/stdc++.h>

using namespace std;
int h[100];
int main()
{

    int n,q;
    cin>>n;
    cin>>q;
    while( q--)
    {
        int mak[100005];
        memset(mak, 0, sizeof( mak));
        memset(h, 0, sizeof(h));
        int ans = 0;
        int x;
        cin>>x;
        while(x--)
        {
            int p;
            cin>>p;
            h[p] = 1;
        }

        for(int i = 1; i <= n; i++)
        {
            int k = i;
            while( k != 0)
            {
                if( h[k%10] == 1 && mak[i] == 0)
                {
                    ans++;
                    mak[i] = 1;
                    //cout<<i<<" ";
                    goto lab;
                }
                k = k/10;
            }
            lab:;
        }
        //cout<<endl;
        cout<<ans<<endl;
    }
}
