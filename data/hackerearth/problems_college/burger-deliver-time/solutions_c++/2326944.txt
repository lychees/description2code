#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	vector<int> v1;
	vector<int> v2;
	for(int i=1;i<=t;i++){
		int a,b;
		cin>>a>>b;
		v1.push_back(a+b);
	}
	for(int i=0;i<t;i++){
		v2.push_back(v1[i]);
	}
	sort(v1.begin(),v1.end());
	int visited[t+1];
	memset(visited,0,sizeof(visited));
	for(auto x: v1){
	    for(int i=0;i<t;i++){
			if(x==v2[i]&&visited[i]==0){
				cout<<(i+1)<<" ";
				visited[i]=1;
			}
		}
	}
}
