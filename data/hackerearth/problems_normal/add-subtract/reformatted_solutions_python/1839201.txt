for _ in range(input()):
    no_elems, elem_count = map(int, raw_input().split())
    mylist = map(int, raw_input().split())
    costfinal = None
    for num in range(1, 101):
        cost = []
        for val in mylist:
            if num >= val:
                cost.append((num - val) * 3)
            else:
                cost.append((val - num) * 5)
        cost.sort()
        value = sum(cost[:elem_count])
        if costfinal is None:
            costfinal = value
        else:
            costfinal = min(costfinal, value)
    print costfinal
