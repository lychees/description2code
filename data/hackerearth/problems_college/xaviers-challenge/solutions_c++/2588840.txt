#include<bits/stdc++.h>
using namespace std;

int main()
 {
    int t;
    scanf("%d",&t);
    char ch;
    scanf("%c",&ch);
    while(t--)
     {
       char str[200];

       scanf("%[^\n]",str);

       int l = strlen(str);
       int i;
       int sum = 0;

       for(i=0; i<l ;i++)
         sum+= (int) str[i];

        int x = sum/l;

        if(x & 1)
          {
            for(i=l-1; i>=0; i--)
              cout<<str[i];

             cout<<endl;
          }

        else
           cout<<str<<endl;
     }
return 0;
}
