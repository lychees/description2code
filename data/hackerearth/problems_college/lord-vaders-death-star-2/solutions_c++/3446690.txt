#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n;
    cin >> n;
    map<long int,long int> pr;
    map<long int,long int> :: iterator it;

    for (int i = 0;i < n;i++)
    {
        long int num;
        cin >> num;
        if (pr[num] == 0)
            pr[num] = 1;
        else if (pr[num] == 1)
            pr[num] = 2;
        else if (pr[num] == 2)
            pr[num] = 2;
    }
    long int ans = -1;

    for (it = pr.begin();it != pr.end();it++)
    {
        if (it -> second == 1){
            ans = max(ans,it -> first);
        }
            //cout << it -> first << endl;
    }

    cout << ans << endl;

    return 0;
}
