#include <iostream>
using namespace std;


int ip[100005] = {0};

int N=0;


void zero(int x, int y)
{
    ip[x]=y;
}

int one(int v)
{
    int i=0;
    
    for(i=1; i<=N; i++)
    {
        if(ip[i]>=v)
        return i;
        
        
    }
    return -1;
}


// Driver program
int main()
{
	int tc=0;
	int T=0;
	cin>> N>>T;
//	cout<<N<<" "<<tc<<endl;
	
	
	int i= 0;
	
	for(i=1; i<=N; i++)
	{
	    cin>> ip[i];  
	    //cout<<ip[i]<<" ";
	}
	
//	cout<<endl;
	for(tc=0; tc<T; tc++)
	{
	    //init();
	    
	    int option=-1;
	    cin>>option;
	    
	    if(option ==0)
	    {
	        int x=-1; 
	        int y=-1;
	        cin>> x>>y;
	     //   cout<<option<<" "<<x<<" "<<y;
	        zero(x,y);
	        //cout<<endl;
	    }
	    
	    else if(option ==1)
	    {
	       int v=-1;
	       cin>>v;
	       //cout<<option<<" "<<v;
	       cout<<one(v)<<endl;
	       // cout<<endl;
	        one(v);
	        
	        
	    }
	}


	return 0;
}
