#include <bits/stdc++.h>
using namespace std;
int a[500000];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		int temp,co=0;
		scanf("%d",&temp);
		while(temp%2==0)
		{
			temp=temp/2;
			co++;
		}
		a[2]=a[2]+co;
		for(int i=3;i<=sqrt(temp);i++)
		{
			co=0;
			while(temp%i==0)
			{
				temp=temp/i;
				co++;
			}
			a[i]=a[i]+co;
		}
		if(temp>1)
		{
			a[temp]++;
		}
	}
	int q;
	scanf("%d",&q);
	while(q--)
	{
		int temp,mi=INT_MAX;
		scanf("%d",&temp);
		int flag=0,co=0;
		while(temp%2==0)
		{
			flag=1;
			temp=temp/2;
			co++;
		}
		if(flag)
		if(mi>a[2]/co)mi=a[2]/co;
		for(int i=3;i<=sqrt(temp);i++)
		{
			flag=0;
			co=0;
			while(temp%i==0)
			{
				co++;
				flag=1;
				temp=temp/i;
			}
			if(flag)
			if(mi>a[i]/co)mi=a[i]/co;
		}
		if(temp>1)
		{
			if(mi>a[temp])mi=a[temp];
		}
		printf("%d\n",mi);
	}
    return 0;
}
