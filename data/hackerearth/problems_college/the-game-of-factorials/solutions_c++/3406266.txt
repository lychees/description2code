
#include<iostream>
using namespace std;
int prod(int x, int r[], int r_size)
{
    int c = 0;


    for (int i=0; i<r_size; i++)
    {
        int prod = r[i] * x + c;
        r[i] = prod % 10;
        c  = prod/10;
    }


    while (c)
    {
        r[r_size] = c%10;
        c = c/10;
        r_size++;
    }
    return r_size;
}


void fact(int n)
{
    int r[10000];
    r[0] = 1;
    int sum=0;
    int r_size = 1;
    for (int x=2; x<=n; x++)
        r_size = prod(x, r, r_size);

    for (int i=r_size-1; i>=0; i--)
        sum+=r[i];
        cout<<sum<<endl;
}

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
     cin>>n;
      fact(n);}
    return 0;
}
