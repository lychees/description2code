#include <iostream>
#include <bits/stdc++.h>
#define upperlimit 11111
using namespace std;
int arr[upperlimit+1];
int main()
{
    int n,i,j,answer,sum=0;
    cin>>n;
    for(i=0;i<n;i++)cin>>arr[i],sum+=arr[i];
    answer=sum;
    int pw=1;
    int s1,s2,s3;
    for(i=0;i<n;i++)pw*=3;
    for(i=0;i<pw;i++){
    	int temp=i;
    	s1=0;
    	s2=0;
    	s3=0;
    	for(j=0;j<n;j++){
    		if(temp%3==0)s1+=arr[j];
    		if(temp%3==1)s2+=arr[j];
    		if(temp%3==2)s3+=arr[j];
    		temp/=3;
    	}
    	answer=min(answer,max(s1,max(s2,s3)));
    }
    cout<<answer<<endl;
    return 0;
}
