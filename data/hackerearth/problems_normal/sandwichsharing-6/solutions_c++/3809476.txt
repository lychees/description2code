#include<iostream>

int main()
{
  unsigned int Testcases;
  std::cin>>Testcases;
  while(Testcases--){
    unsigned int sides;
    std::cin>>sides;
    unsigned int person ;
    std::cin>>person;
    for(int i=1;i<=person;i++){

      if(sides%2 == 0){
	sides = sides/2 +1;
      }else{

	sides = sides/2 +2;
      }
      if(sides ==3 ){
	break;
      }
    }
    std::cout<<sides<<std::endl;  
  }
  
  return 0;
}
