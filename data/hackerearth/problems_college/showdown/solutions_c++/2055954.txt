#include<iostream>
#include<algorithm>
using namespace std;

int N;
int A[100],B[100];

bool comp(int a,int b)
{
    if(a<b) return false;
    else return true;
}

int calc(int i,int j)
{
    if(i>=N||j>=N) return 0;
    if(A[i]<B[j]) return calc(i,j+1);
    else if(A[i]>B[j]) return 100+calc(i+1,j+1);
    else return max(50+calc(i+1,j+1),calc(i,j+1));
}
int main()
{
    cin>>N;
    for(int i=0;i<N;i++)
        cin>>A[i];
    for(int i=0;i<N;i++)
        cin>>B[i];
    sort(A,A+N,comp);
    sort(B,B+N,comp);

    cout<<calc(0,0)<<"\n";
}
