#include <stdio.h>
#include <math.h>
#define ll unsigned long long
long long joseph (long long n,long long k) {
    if (n==1LL) return 0LL;
    if (k==1LL) return n-1LL;
    if (k>n) return (joseph(n-1LL,k)+k)%n;
    long long cnt=n/k;
    long long res=joseph(n-cnt,k);
    res-=n%k;
    if (res<0LL) res+=n;
    else res+=res/(k-1LL);
    return res;
}
int main() {
	ll t, n;
	scanf("%lld\n",&t);
	while(t--) {
		scanf("%lld",&n);
		printf("%lld\n",joseph(n,2)+1);
	}
	return 0;
}
