#include <iostream>
#include <vector>
using namespace std;

const int MAX = 2000;

int main()
{
    int T, M, L;
    scanf("%d", &T);
    while (T-- > 0) {
    	scanf("%d", &M);
    	scanf("%d", &L);
    	int W[M];
    	int V[M];
    	for (int i = 0; i < M; i++) {
    		scanf("%d", &W[i]);
    	}
    	for (int i = 0; i < M; i++) {
    		scanf("%d", &V[i]);
    	}
    	int dp[MAX + 1] = { 0 };
    	for (int i = 0; i < M; i++) {
    		for (int j = MAX; j >= 0; j--) {
    			int wNew = j + W[i];
    			int vNew = dp[j] + V[i];
    			if (wNew <= MAX) {
    				dp[wNew] = max(dp[wNew], vNew);
    			}
    		}
    	}
    	int ans = 0;
    	for (int i = 0; i <= L; i++) {
    		ans = max(ans, dp[i]);
    	}
    	printf("%d\n", ans);
    }
}
