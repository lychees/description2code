#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n];
    int i;
    for(i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    int fs=0,count=n,x;
    for(i=0;i<n;i++){
    	out:
    	if(fs==0){
    	x=arr[i]+4;
    	fs=1;
    	}
    	else{
    		if(arr[i]<=x){
    		count--;
    		}
    		else
    		{
    			fs=0;
    			goto out;
    		}
    	}
    }
    
    cout<<count<<endl;
    return 0;
}
