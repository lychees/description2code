#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define inf 10e12

using namespace std;

ll pos[150001];
ll dish[150001];
ll n,m;

ll valid(ll mid)
{
    ll i=0,j=0;
    while(i<n && j<m) {
        if(abs(pos[i]-dish[j])<=mid) {
            i++;j++;
        }
        else j++;
    }
    if(i==n) return 1;
    return 0;
}

int main()
{
    ll t,i,j,high,low,mid;
    //freopen("input.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    s(t);
    while(t--) {
        s(n);s(m);
        for(i=0;i<n;i++) {
            s(pos[i]);
        }
        for(i=0;i<m;i++) {
            s(dish[i]);
        }
        sort(pos,pos+n);
        sort(dish,dish+m);
        low=0;
        high=inf;
        while(low<high) {
            mid=(low+high)/2;
            if(valid(mid)==1) {
                high=mid;
            }
            else {
                low=mid+1;
            }
        }
        cout<<low<<endl;
    }
    return 0;
}
