#include<bits/stdc++.h>
using namespace std;
double dp[1000006];
int main()
{
    int n,p,i;
    scanf("%d %d",&n,&p);
    dp[0]=1;
    dp[1]=0;
    dp[2]=p/100.0;
    dp[3]=1-p/100.0;
    for(i=4;i<=n;i++)
    {
        dp[i]=(dp[i-2]*p/100.0)+(dp[i-3]*(1-p/100.0));
    }
    printf("%0.6lf ",dp[n]);
}
