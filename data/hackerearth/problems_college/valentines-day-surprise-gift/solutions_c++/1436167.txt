using namespace std;

#define ll long long
#define vt vector
#define mod 1000000007

long long powmod(ll base, ll expo)
{
	if(expo==0)
		return 1;
	else if(expo&1)
		return (long long)base*powmod(base, expo-1)%mod;
	else{
		ll root=powmod(base, expo>>1);
		return (long long)(root*root)%mod;
	}
}

long long inverse(ll x)
{
	return powmod(x, mod-2);
}

long long factmod(long long val)
{
	long long answer = 1;
	for(ll i = 1; i <= val; i++)
		answer = (answer * i)%mod;
	return answer % mod;
}

ll gcd(ll a,ll b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}


ll R(ll n)
{
	if(n==0 ||n==1 ||n==2 )
		return 1;
	else if(n<=4)
		return 3;
	else if(n<=6)
		return 15;
	else if(n<=10)
		return 105;
	else
		return ((R((n+4)/5)*R((n+9)/10))%mod*powmod(2,(n+3)/4))%mod;
}

int main()
{
	ll t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll rvalue = R(n);
		ll maxans=-1;
		for(int i=1;i<=5;i++)
		{
			for(int j=1;j<=5;j++)
			{
				ll x = i*rvalue;
				ll y = j*rvalue;
				if(gcd(i,j)==1)
					maxans = max(maxans,powmod(x,y));
			}
		}
		cout << maxans << "\n";
	}
	return 0;
}
