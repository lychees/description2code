#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#include<vector>

int main()
{
	ll t, m, n, i;
	cin>>t;
	while(t--)
	{cin>>n>>m;
vector<ll> mat;
i=0;
while(m)
{
if((m & 1)==1)
{
mat.push_back(i);}
m=m>>1;
i++;
}
ll j=mat.size()-1;
cout<<"("<<n<<"<<"<<mat[j]<<")";
j--;
while(j>=0)
{
	cout<<" + "<<"("<<n<<"<<"<<mat[j]<<")";
	
	j--;
}
		
	cout<<endl;	
		
		
		
		
		
	}
	
	
}
