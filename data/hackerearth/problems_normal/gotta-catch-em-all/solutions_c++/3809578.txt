#include <iostream>
#include <algorithm>
using namespace std;

typedef unsigned long long ull;
int main()
{
    int n;cin>>n;
    int a[n],i;
    for(i=0;i<n;i++)	cin>>a[i];
    sort(a,a+n);
    ull prev=a[n-1],cnt=1;
    for(i=n-2;i>=0;i--){
    	if(a[i]<prev){
    		cnt++;
    		prev--;
    	}
    	else{
    		cnt++;
    		prev=a[i];
    		
    	}
    }
    cnt+=prev;
    cout<<cnt+1<<endl;
    return 0;
}
