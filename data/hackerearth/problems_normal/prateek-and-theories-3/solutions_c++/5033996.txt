#include<iostream>
#include<vector>
#include<utility>
#include<algorithm>
using namespace std;
int main()
{
    int test,n,i,j,st,ed,maximum,rel;
    cin>>test;
    for(i=0;i<test;i++)
    {
        cin>>n;
        vector<pair<int,int> >times(2*n);
        maximum=0,rel=0;
        for(j=0;j<n;j++)
        {
            cin>>st>>ed;
            times[j]=make_pair(st,1);
            times[n+j]=make_pair(ed,-1);
        }
        sort(times.begin(),times.end());
        for(j=0;j<2*n;j++)
        {
            rel+=times[j].second;
            maximum=max(maximum,rel);
        }
        cout<<maximum<<endl;
    }
    return 0;
}
