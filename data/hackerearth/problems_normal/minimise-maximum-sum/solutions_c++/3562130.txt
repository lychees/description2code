#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j,m,temp;
    int a[100010];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<(2*n);i++)
    	scanf("%d",&a[i]);
    	
    	sort(a,a+(2*n));
    	m=0;temp=0;
    	for(i=0,j=((2*n)-1);i<=j;i++,j--)
    	{
    		temp=a[i]+a[j];
    		if(temp>m)
    		m=temp;
    	}
    	cout<<m<<endl;
    }
    return 0;
}
