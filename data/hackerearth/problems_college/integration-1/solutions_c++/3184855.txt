#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    for(int te=1;te<=t;te++){
    	int n,p,u,r,s;
    	scanf("%d %d %d %d %d",&n,&p,&u,&r,&s);
    	for(int i=0;i<n;i++){
    		if(p+r>u){
    			if(p-s<0)
    			break;
    			else p-=s;
    		}
    		else
    		p+=r;
    	}
    	printf("Case #%d: %d\n",te,p);
    }
	return 0;
}
