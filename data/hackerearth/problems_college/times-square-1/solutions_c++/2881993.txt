#include <iostream>
using namespace std;

int main()
{
    long long n,m,a;
    cin>>n>>m>>a;
    long long ax=0,bx=0;
    if(n%a==0)
    {
    	ax =n/a;
    	
    }
    else
    ax = (n/a)+1;
    
    if(m%a==0)
    {
    	bx =m/a;
    	
    }
    else
    bx = (m/a)+1;
    
    cout<<ax*bx<<endl;
    
    
    
    return 0;
}
