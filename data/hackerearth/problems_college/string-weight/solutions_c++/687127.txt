#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int t;cin>>t;
	while(t--){
		char s[10001];
		int a[27]={0},index;
		cin>>s;
		int size=strlen(s);
		for(int i=0;i<size;i++){
			index= s[i]-96;
			a[index]++;
		}
		int skip;char c;
		cin>>skip;
		while(skip--){
			cin>>c;
			a[c-96]=0;
		}
		long long wt=0;
		for(int i=1;i<=26;i++)
		wt= wt+i*a[i];
		
		cout<<wt<<endl;
	}
    return 0;
}
