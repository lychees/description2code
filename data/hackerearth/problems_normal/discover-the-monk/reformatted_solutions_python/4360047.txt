x, y = map(int, raw_input().strip().split())
arr = sorted(map(int, raw_input().strip().split()))
for i in range(y):
    n = int(raw_input())
    low = 0
    high = x-1
    flag = 0
    while (low <= high and low >= 0 and high <= x-1):
        mid = (low+high)/2
        if arr[mid] == n:
            print "YES"
            flag = 1
            break
        elif arr[mid] > n:
            high = mid-1
        else:
            low = mid+1
    if flag != 1:
        print "NO"
