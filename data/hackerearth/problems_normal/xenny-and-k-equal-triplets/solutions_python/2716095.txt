def gcd(a,b):
    if a == 0:
        return b
    if b == 0:
        return a;
    return gcd(b%a, a)

test = int(raw_input())
for i in range(test):
    n,k = map(long, raw_input().split())
    arr = map(long, raw_input().split())
    cnt = 0
    for j in range(0, n):
        x = arr[j]
        if x==k:
            cnt = cnt+1
    ans1 = n*(n-1)*(n-2)
    ans2 = cnt*(cnt-1)*(cnt-2)
    ans = gcd(ans1,ans2)
    print str(ans2/ans)+'/'+str(ans1/ans)
