typedef long long LL;

using namespace std;



LL a[100000];
LL mini(LL n)
{
	LL minimum=10000;
	LL ct=0;
	for(LL i=0;i<n;i++)
	{
		if(a[i]<minimum && a[i]!=-10000)
		minimum=a[i];
	}
	
	for(LL i=0;i<n;i++)
	{
		if(a[i]==minimum)
		ct++;
	}
	return ct;
}

LL findmin(LL n)
{
	LL minimum=10000;
	for(LL i=0;i<n;i++)
	{
		if(a[i]<minimum && a[i]!=-10000)
		minimum=a[i];
	}
	return minimum;
}

LL cont(LL n)
{
	for(LL i=0;i<n;i++)
	{
		if(a[i]>0)
		return 1;
	}
	return 0;
}

main()
{
	LL n,ct;
	cin>>n;
	
	ct=n;
	for(LL i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	cout<<ct<<endl;
	
	while(cont(n))
	{
		LL temp=findmin(n);
		LL ans=mini(n);
		for(LL i=0;i<n;i++)
		{
			a[i]=a[i]-temp;
			if(a[i]<=0)
			a[i]=-10000;
		}		
		ct=ct-ans;
		if(ct!=0)
		cout<<ct<<endl;
	}

}
