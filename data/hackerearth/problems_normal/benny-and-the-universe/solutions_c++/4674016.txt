#include<bits/stdc++.h>
using namespace std;
int n,q;
int a[1100];
long long f[101000];
int main()
{
    #define INF (long long)2e18
    scanf("%d%d",&n,&q);
    for(int i=1;i<=n;++i)
        scanf("%d",&a[i]);
    int min_ai=a[1];
    for(int i=2;i<=n;++i)
        min_ai=min(min_ai,a[i]);
    for(int i=0;i<=min_ai;++i) f[i]=INF;
    priority_queue<pair<long long,int> > pb;
    f[0]=0ll;
    pb.push(make_pair(0,0));
    while(!pb.empty())
    {
        long long dis=-pb.top().first;
        int v=pb.top().second;
        pb.pop();
        if(f[v]<dis) continue;
        for(int i=1;i<=n;++i)
        {
            int nxt=(v+a[i])%min_ai;
            if(f[nxt]>f[v]+1ll*a[i])
            {
                f[nxt]=f[v]+1ll*a[i];
                pb.push(make_pair(-f[nxt],nxt));
            }
        }
    }
    bool pp,mm;
    pp=mm=false;
    while(q--)
    {
        int v;
        scanf("%d",&v);
        int real=v;
        v%=min_ai;
        if(f[v]<=1ll*real)
        {
            printf("YES\n");
            pp=true;
        }
        else
        {
            printf("NO\n");
            mm==true;
        }
    }
    return 0;
}
