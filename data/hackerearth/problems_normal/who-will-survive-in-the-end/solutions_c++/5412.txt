#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        i=0;
        while(pow(2,i)<n)
        i++;
        i-=1;
        n-=pow(2,i);
        n+=1;
        printf("%d\n",2*n-1);
    }
    return 0;
}
