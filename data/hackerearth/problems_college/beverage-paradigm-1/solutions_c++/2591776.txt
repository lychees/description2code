#include <iostream>
#include <string.h>
#include <algorithm>
using namespace std;

int main()
{
    int Tc;
    cin>>Tc;
    int *Arr;
    int SearchKey;
    Arr = (int *)malloc(sizeof(int )*Tc);
    for(int i=0;i<Tc;i++)
    cin>>Arr[i];
    sort(Arr,Arr+Tc);
    cin>>SearchKey;
  //  for(int i=0;i<Tc;i++)
  //  cout<<Arr[i]<<"   ";
    int p;
    int h=0;
    for(int i=Tc-1;i>=0;i--)
    {
    	int t=SearchKey-Arr[i];
    //	cout<<"\n OI  ::           "<<t<<endl;
    	if(t>=0)
        {
        	
        	int op = t;
        		for(int j=i-1;j>=0;j--)
    			{
    				int ak = op-Arr[j];
    			//	cout<<"##### T %%%          "<<ak<<endl;
					if(ak >= 0)
					{
						
    						for(int k=j-1;k>=0;k--)
    						{
    							//	cout<<"&&&  Arr[k]       "<<Arr[k]<<endl;
    								if(ak==Arr[k])
    								{
    									h=1;
    									goto label;
    				
    								}
    						}
					}	
    			}
        }
    
    }
    label:
    if(h)
    cout<<"True"<<endl;
    else
    cout<<"False"<<endl;
    return 0;
}
