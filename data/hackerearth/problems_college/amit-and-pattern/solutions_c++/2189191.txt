
#include<bits/stdc++.h>

typedef long long ll;
#define deb(x) cerr << #x << " = " << x << '\n';

using namespace std;

int main()
{
	std::ios_base::sync_with_stdio(false);
	//freopen("input1.txt","r",stdin);
    int n,m;

    cin>>n>>m;

    int arr[n+2][m+2];

    for(int i = 0; i < n; i++)
    {
        for(int j = 0; j < m; j++)
            cin>>arr[i][j];
    }

    for(int col = 0; col < m; col++)
    {
        int row = 0;
        int cnt = 1;
        int temp_col = col;
        while(row < n && temp_col >= 0)
        {
            cout<<arr[row][temp_col]<<" ";
            row++;
            temp_col--;
            cnt++;
        }

        cout<<endl;

    }
    for(int row = 1; row < n; row++)
    {
        int temp_row = row;
        int col = m - 1;
        while(temp_row < n && col >=0)
        {
            cout<<arr[temp_row][col]<<" ";
            col--;
            temp_row++;
        }
        cout<<endl;
    }

	return 0;

}
