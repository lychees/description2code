#include <iostream>
#include<vector>
using namespace std;
int temp=0;
int main()
{
    vector<int>v1;
    vector<int>v;
    int n,a;
    cin>>n;
    for(int i=0;i<n;i++)
     {
     	  cin>>a;
     	  v1.push_back(a);
          v.push_back(a);
     }
      /*for(int j=0;j<n;j++)
       	    {
                
       	    	cout<<v[j]<<" ";
       	    }
       cout<<endl;*/
     
     for(int k=1;k<n;k=k+1)
       {
       	 if(v1[k]<v1[k-1])
       	    {
       	    	 v[k]=v[k]-v[k];
       	    	  v[k-1]=v[k-1]-v1[k];
       	    }
       	   else
       	   {
       	   	  v[k]=v[k]-v[k-1];
       	   	   v[k-1]=v[k-1]-v[k-1];
       	   }
            v1[k]=v[k];
             v1[k-1]=v[k-1];
       }
       	   for(int j=0;j<n;j++)
       	    {
                 if(v[j]!=0)
                  {
                         temp=1;
                        break;
                   }
             }

      if(temp==0)
         cout<<"YES"<<endl;
       else
          cout<<"NO"<<endl;


          
  return 0;
}
