#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		bool f=0;
		int n;
		cin>>n;
		int a[n];
		cin>>a[n-1]>>a[n-2];
		for(int i=n-3;i>=0;i--)
			{
				cin>>a[i];
			}
		sort(a,a+n);
			for(int i=n-3;i>=0;i--)
			{
				//cin>>a[i];
				if(a[i]+a[i+1]>a[i+2])
				{cout<<a[i]<<" "<<a[i+1]<<" "<<a[i+2]; f=1; break;}
			}
		if(!f)
		cout<<"-1";
		cout<<"\n";
	}
	return 0;
}
