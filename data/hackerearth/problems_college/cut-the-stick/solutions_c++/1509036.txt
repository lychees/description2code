#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int S,n,L;
        scanf("%d%d%d",&S,&n,&L);
        int arr[n];
        for(int i=0;i<n;i++)
            scanf("%d",&arr[i]);
        sort(arr,arr+n);
        int cumulative[S+9];
        cumulative[0]=0;
        int cnt=0;
        for(int i=1;i<=S;i++)
        {
            cumulative[i]=0;
            if(cnt<n&&arr[cnt]==i)
            {
                cumulative[i]=1;
                cnt++;
            }
            cumulative[i]+=cumulative[i-1];
        }
        ll ans=0;
        for(int i=0;i<n;i++)
        {
            int fb=arr[i];
            if(fb<=L)
            {
                int num1=0,num2=0,num3=0;
                if(fb+L<=S&&S-L-1>=0&&fb+L<S-L-1)
                    num3=cumulative[S-L-1]-cumulative[fb+L];
                if(fb+L<=S)
                    num1=n-cumulative[fb+L];
                if(S-L-1>fb)
                    num2=cumulative[S-L-1]-cumulative[fb];
                ans+=num1+num2-num3;
            }
            else
            {
                ans+=n-cumulative[fb];
            }
        }
        printf("%lld\n",ans);
    }
}
