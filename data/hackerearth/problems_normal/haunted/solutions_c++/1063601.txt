#include<stdio.h>
#include<iostream>
#include<map>
#include<algorithm>
using namespace std;

struct tempt
{
	int num;
	int ind;
};

int comp(struct tempt a,struct tempt b)
{
	return a.num<b.num;
}

int final_ind[100005];

int hash3[100005];

int hash1[100005];

int hash2[100005];

int main()
{
	int N,M;
	cin>>M>>N;
	int arr[M];
	struct tempt arr2[M];
	for(int i=0;i<M;i++)
	{
		scanf("%d",&arr[i]);
		arr2[i].num=arr[i];
		arr2[i].ind=i;
	}
	sort(arr2,arr2+M,comp);
	for(int i=0;i<M;i++)
	final_ind[arr2[i].ind]=i;
	hash3[0]=1;
	int c=2;
	for(int i=1;i<M;i++)
	{
		if(arr2[i].num!=arr2[i-1].num)
		{
			hash3[i]=c;
			c++;
		}
		else
		hash3[i]=hash3[i-1];
	}
	
	for(int i=0;i<M;i++)
	hash1[hash3[i]]=arr2[i].num;
	
	int max_val=0;
	int max_age=-1;
	
	for(int i=0;i<M;i++)
	{
		int ind=hash3[final_ind[i]];
		hash2[ind]++;
		if(hash2[ind]>max_val)
		{
			max_val=hash2[ind];
			max_age=hash1[ind];
		}
		else if(hash2[ind]==max_val)
		{
			if(arr[i]>max_age)
			max_age=arr[i];
		}
		printf("%d %d\n",max_age,max_val);
	}
}
