#include <iostream>
#include <math.h>
using namespace std;


int main()
{
	int powers[]={4,9,16,25,36,49,100};
	int pairs[]={2,3,4,5,6,7,10};
	
	int iter,a,b;
	float x;
	cin>>iter;
	for(int i=0;i<iter;i++)
	{
		cin>>x;
		b=1;
		a=x-b;
		float c=x-(int)x;
		if(c==0)
		{
		if(x==1)
			cout<<"\n"<<1<<" "<<0;
		else
		cout<<"\n"<<a<<" "<<b;
		} else {
			int x_mod=(int)x;
			float frac=x-x_mod;
			for(int j=0;j<7;j++)
			{
				if(x_mod==powers[j]) {
					if(frac==(1.0/(pow(2,pairs[j]))))
						{
							b=-2;
							a=pairs[j];
						}
						break;
	
				} else if(x_mod==64) {
					if(frac==(1.0/(pow(2,8)))) {
						b=-2;
						a=8;
					} else {
						a=4;
						b=-4;
					}
					break;
				}
				
				else if(x_mod==81) {
					if(frac==((1.0/pow(4,3)))) {
						b=-4;
						a=3;
					} else {
						a=9;
						b=-2;
					}
					break;
				}
				
				
				
			}
			cout<<"\n"<<a<<" "<<b;
		}
		
	}
  
    return 0;
}
