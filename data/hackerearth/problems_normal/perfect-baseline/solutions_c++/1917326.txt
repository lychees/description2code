#include <bits/stdc++.h>
#define ll long long 
using namespace std;
 
int main()
{
	ll n,t,k,i,j;
	
	char s[10002][22];
	
	cin >> t;
	
	while(t--)
	{
		cin >> n >> k;
	     for(i=0;i<n;i++)
	     {
			cin >> s[i]; 
		 }
		 char c;
		 for(j=0;j<k;j++)
		 {
		 	ll dis=10000000;
		 	for(int l='a';l<='z';l++)
		 	{
		 	   ll dis1=0;
		 	   for(i=0;i<n;i++)
				{
				   	dis1+=abs(s[i][j]-l);
				}
				if(dis>dis1)
				{
				  dis=dis1;
				  c=l;  	
				}	
			}
			cout << (char) c ;
		 }
		 cout << endl;
	}
     return 0;
}
