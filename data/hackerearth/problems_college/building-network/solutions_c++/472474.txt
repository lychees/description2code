#include <cstdio>
#include <iostream>
#include <climits>
using namespace std;
typedef long long int ll;
ll a[1005];
int main(){

    int t,n;
    cin>>t;
    while(t--){
        ll sum=0,min=INT_MAX;
        cin>>n;
        for(int i=1;i<=n;i++){
            cin>>a[i];
            if(min>a[i])
                min=a[i];

            sum+=a[i];
        }
        ll val=sum+(n-2)*min;
        cout<<val<<"\n";
    }
    return 0;
}
