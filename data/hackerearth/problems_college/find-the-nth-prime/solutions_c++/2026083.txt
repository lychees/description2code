#include <iostream>

using namespace std;
int prime(int n)
{
	int flag=0;
	for(int i=2;i<=n/2;++i)
  	{

	    if(n%i==0)
      	{
       	   flag=1;
       	   break;
      	}
  }
  if (flag==0)
      return 1;
  else
      return 0;
}

int main()
{
	int i,p=0,n;
	cin>>n;
	for(i=2;;i++)
	{
		if(prime(i))
			p++;
		if(p==n)
			break;
	}
	cout<<i<<endl;
	return 0;
	
}
