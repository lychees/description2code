#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,j,l;
		scanf("%d",&n);
		int a[n];
		set<int>s;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			s.insert(a[i]);
		}
		l=s.size();
		set<int>::iterator it1;
		i=0;
		int b[l];
		for(it1 = s.begin();it1!=s.end();it1++)
		{
			b[i] = *it1;
			i++;
		}
		
		for(i=0;i<l/2;i++)
		{
			printf("%d ",b[l-1-i]);
			if(b[l-1-i]==b[i])
			 continue;
			printf("%d ",b[i]);
		}
		if((l%2==1)&&(l>1))
		{
			printf("%d ",b[l/2]);
		}
		if(l==1)
		{
			printf("%d",b[0]);
		}
		printf("\n");
	}
}
