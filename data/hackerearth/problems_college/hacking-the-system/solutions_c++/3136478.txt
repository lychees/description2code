#include<iostream>
using namespace std;
int main()
{
  long int a,b,count=0;
  int p,q;
  cin>>a;
  cin>>b;
  while(a>0&&b>0)
  {
  p=a%10;
  q=b%10;
  if(p!=q)
  count++;
  a=a/10;
  b=b/10;
  }
  if(count>1&&count!=0)
  {
   cout<<"loses";
  }
  else
  {
   cout<<"wins";
  }
}
