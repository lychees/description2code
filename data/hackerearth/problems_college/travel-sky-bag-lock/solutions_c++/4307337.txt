#include <iostream>
using namespace std;

int main()
{
    int t,n,a,b,sum;
    
    cin>>t;
    while(t--)
    {	sum=0;
    	cin>>n;
    		cin>>a;
    		while(n--)
    		{
    			b=a%10;
    			a=a/10;
    			if((b==1)||(b==4)||(b==6)||(b==8))
    				sum+=1;
    			else if(b==9)
    				sum+=2;
    		}
    		
    	cout<<sum<<endl;
    }
    return 0;
}
