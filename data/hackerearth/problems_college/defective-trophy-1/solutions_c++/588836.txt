#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int T,fla=0;
    cin>>T;
    if(cin.fail()||T<1||T>30){fla=1;}
    else
    {
    	int N[T],i,j;
    	for(i=0;i<T;i++)
    	{
    		cin>>N[i];
    		if(cin.fail()||N[i]>10000||N[i]<0){fla=1;break;}
    		for(j=0;pow(3,j)<N[i];j++);
    		cout<<j<<endl;
    	}
    }
    return 0;
}
