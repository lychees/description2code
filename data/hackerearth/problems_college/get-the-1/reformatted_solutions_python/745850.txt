tc = int(raw_input())
inp = []
i = 0
while i < tc:
    inp.append(raw_input())
    i = i + 1
result = []
for inp1 in inp:
    while len(inp1) != 1:
        s = 0
        for j in inp1:
            s = s + (int(j))**2
        inp1 = str(s)
    if (int(inp1) == 1):
        result.append("YES")
    else:
        result.append("NO")
for z in range(len(result)):
    print 'Case #'+str(z+1)+':'+str(result[z])
