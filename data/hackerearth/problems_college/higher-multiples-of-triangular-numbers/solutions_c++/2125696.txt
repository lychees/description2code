#include <iostream>
#include <math.h>
using namespace std;

int main()
{   long long z,x,s,i,r,j,p;
    cin>>z;
    while(z)
    {   cin>>x;
        r=2;j=1;
        while(1)
        {   s=0;
            p=sqrt(j);
            if(p*p==j)
                s-=1;
            for(i=1;i<=p;++i)
            {   if(j%i==0)
                {   s+=2;
                }
            }
            if(s>x)
            {   cout<<j<<"\n";
                break;
            }
            j+=r;
            ++r;
        }
        --z;
    }
    return 0;
}
