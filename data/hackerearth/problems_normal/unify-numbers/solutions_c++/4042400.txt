#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>


int main()
{
    using namespace std;
    
    int T;
    cin >> T;
    for (int count = 0; count < T; ++count)
    {
    	int N;
    	cin >> N;
    	std::vector<int> Ai;
    	for (int ii = 0; ii < N; ++ii)
    	{
    		int num;
    		cin >> num;
    		Ai.push_back(num);
    	}
    	
    	sort(Ai.begin(), Ai.end());
    	
    	// Avg:
    	
    	// double sumAi;
    	// for (int ii = 0; ii < N; ++ii)
    	// {
    	// 	sumAi += Ai[ii];
    	// }
    	// double dAvgAi = sumAi / N;
    	// int nAvgAi = floor(dAvgAi);
    	
    	// Median:
    	
    	int med;
    	if (N % 2 == 0)
    	{
    		med = (Ai[(N-1) / 2] + Ai[N/2]) / 2;
    	}
    	else
    	{
    		med = Ai[N/2];
    	}
    	
    	long long M = 0;
    	for (int ii = 0; ii < N; ++ii)
    	{
    		M += abs(Ai[ii] - med);
    	}
    	cout << M << "\n";
    }
    
    return 0;
}
