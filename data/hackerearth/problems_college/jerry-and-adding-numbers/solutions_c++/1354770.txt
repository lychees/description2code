#include <bits/stdc++.h>
using namespace std;
#define in(x) scanf("%d",&x)
#define o(x,y) printf("%d %d\n",x,y)

int main()
{
    int t,n,k;
    in(t);
    while(t--)
    {
    	in(n);
    	in(k);
    	int i;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		in(a[i]);
    	}
    	sort(a,a+n);
    	int h,l,m,v=0,b=-1;
    	long long int s[n+1];
    	s[1]=a[0];
    	s[0]=0;
    	for(i=1;i<n;i++)
    	s[i+1]=s[i]+a[i];
    	for(i=0;i<n;i++)
    	{
    		l=0;h=i;
    		while(l<h)
    		{
    			m=(l+h)/2;
    			if((((long long)(i-m+1)*a[i])-(s[i+1]-s[m]))<=k)
    			h=m;
    			else
    			l=m+1;
    		}
    		if((i-l+1)>b)
    		{
    			b=i-l+1;
    			v=a[i];
    		}
    	}
    	o(b,v);
    }
    return 0;
}
