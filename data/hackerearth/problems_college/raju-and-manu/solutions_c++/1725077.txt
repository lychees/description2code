int power(int a, int b)
{
	if(!b)return 1;
	if(b==1)return a;
	int tmp=power(a,b/2);
	if(b%2)return a*tmp*tmp;
	return tmp*tmp;
}

inline void ifast(int &x) 
{
    register int c = getchar_unlocked();
    x = 0;
    int neg = 0;
    for(; ((c<48 || c>57) && c != '-'); c = getchar_unlocked());
    if(c=='-') {
        neg = 1;
        c = getchar_unlocked();
    }   
    for(; c>47 && c<58 ; c = getchar_unlocked()) {
        x = (x<<1) + (x<<3) + c - 48; 
    }   
    if(neg)
        x = -x; 
}

#define LEN 100005

int cnt(int n)
{
	int c=0,m=1;
	while(n) {
		if(n&m)c++;
		n>>=1;
	}
	return c;
}

int main()
{
	int arr[LEN];
	int brr[LEN];
	arr[0]=0;
	arr[1]=1;
	brr[0]=0;
	brr[1]=1;
	for(int i=2;i<LEN;i++) {
		int temp=cnt(i);
		if(arr[i-1]>temp) {
			arr[i]=arr[i-1];
			brr[i]=brr[i-1];
		} else {
			arr[i]=temp;
			brr[i]=i;
		}
//		p(arr[i]);sp;p(brr[i]);nl;
	}
	int T;
	ifast(T);
	while(T--) {
		int n;
		ifast(n);
		p(brr[n]);nl;
	}
	return 0;
}
