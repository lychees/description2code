#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--){
    	long long n,q; cin>>n>>q;
    	long long  maxi=0,mini=INT_MAX,ans;
    	while(n--){
    		long long  temp; cin>>temp;
    		maxi=max(temp,maxi);
    		mini=min(temp,mini);
    	}
    	ans=q*(maxi-mini);
    	cout<<ans<<endl;
    }
    return 0;
}
