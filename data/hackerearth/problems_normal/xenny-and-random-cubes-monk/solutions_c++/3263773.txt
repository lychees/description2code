#include <bits/stdc++.h>
#define MOD 1000000007

using namespace std;

typedef long long int ll;

ll letters [11][26];
ll aux[11];
int n,k;
ll ans = 0;
string s;

inline void solve(int idx, int curr){
	if(curr >= k){
        sort(aux,aux+k);
        do{
            ll temp = 1LL;
            for(int i=0; i<k; i++)
                temp = (temp * letters[ aux[i] ][ s[i] - 'a' ]) % MOD;

            ans = (ans + temp) % MOD;

        }while( next_permutation(aux,aux+k) );

		return;
	}


    for(int j=idx; j<n; j++){
        aux[curr] = j;
        solve(j+1,curr+1);
    }

}

int main(){

    cin >> n >> k;

    char c;

    memset(letters,0,sizeof letters);
    for(int i=0; i<n; ++i){
    	for(int j=0; j<6; ++j){
    		cin >> c;
    		letters[i][c-'a']++;
    	}
    }


    cin >> s;
    solve(0,0);
    cout << ans;

    return 0;
}
