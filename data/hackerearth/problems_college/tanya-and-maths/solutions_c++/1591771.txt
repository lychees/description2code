#include <iostream>
using namespace std;

inline int insert(){
	int x=0;
	char c=getchar();
	for(;c<48 || c>57;c=getchar());
	for(;c>47 && c<58;c=getchar()) x = (x<<1) + (x<<3) + c -48;
	
	return x;
}
int gcd(int a,int b){
	if( a == b)
		return a;
	if(a>b)
		return gcd(a-b,b);
	return gcd(a,b-a);
}
int main()
{
 	int t=insert();
 	while(t--){
 		int a=insert();
 		int b=insert();
 		cout<<gcd(a,b)<<endl;
 	}
    return 0;
}
