int main()
{
	int test;
	int i;
	cin>>test;
	while(test--) {
		string st;
		cin>>st;
		int l = st.length();
		int pos = 0;
		while(st[pos] != 'a' && st[pos] != 'z' && pos < l) pos++;
		if(pos == l) puts("0");
		else {
			int last = -1;
			ll sum = 0;
			for(i = 0; i < l; i++) {
				if(st[i] == 'a' || st[i] == 'z') {
					sum+=((i+1)-(last+1))*(l-(i));
					last = i;
				}
			}
			if(l == 1) {
				if(st[0] == 'a' || st[0] == 'z') puts("1");
			} else
				cout<<sum<<"\n";
		}
	}
	return 0;
}
