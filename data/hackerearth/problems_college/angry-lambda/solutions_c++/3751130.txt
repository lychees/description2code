#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--){
   	int t,k;
   	cin>>t>>k;
   	int a[t+1];
   	for(int i=0;i<t;i++){
   		cin>>a[i];
   	}
   	int count=0;
   	for(int j=0;j<t;j++){
   		if(a[j]<=0){
   			count++;
   		}
   	}
   //	cout<<count<<"gg"<<k<<endl;
   	if(count>=k){
   		cout<<"NO"<<endl;
   	}
   	else{
   		cout<<"YES"<<endl;
   	}
   }
    return 0;
}
