#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		long long int z,k=1000000007,sum1=0,sum2=0;
		for(int i=0;i<n;i++)
		{
			if(n%2==1)
			{
			while(i<=n/2)
			{
		    	sum1=sum1+(-1)*a[i];
		    	i++;
	    	}     
			sum1=sum1+a[i];
	    	}
	    	else
	    	{
	    		while(i<=n/2-1)
			{
		    	sum1=sum1+(-1)*a[i];
		    	i++;
	    	}     
			sum1=sum1+a[i];
	    	}
		}
		if(sum1>0)
		{
			for(int i=0;i<n;i++)
	    	{
			    if(a[i]<0)
			    sum2=sum2+(-1)*a[i];
		    	else
			    sum2=sum2+a[i];
	    	}
		}
		else
		{
			for(int i=0;i<n;i++)
	    	{
			    if(a[i]>0)
			    sum2=sum2+(-1)*a[i];
		    	else
			    sum2=sum2+a[i];
	    	}
		}
		z=((sum1%k)*(sum2%k))%k;
		cout<<z<<endl;
	}
	return 0;
}
