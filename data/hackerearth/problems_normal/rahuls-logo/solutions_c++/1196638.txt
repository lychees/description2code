#include<cstdio>

void space()
{
	putchar(' ');
}

int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n-i-1;j++)
			space();
		putchar('/');
		for(int j=0;j<i/2;j++)
			printf(" /");
		if(i&1)
			space(),space();
		for(int j=0;j<i/2;j++)
			printf("\\ ");
		putchar('\\');
		putchar('\n');
	}
	for(int i=n-1;i>=0;i--)
	{
		for(int j=0;j<n-i+1;j++)
			space();
		putchar('\\');
		for(int j=0;j<i/2;j++)
			printf(" \\");
		if(i&1)
			space(),space();
		for(int j=0;j<i/2;j++)
			printf("/ ");
		putchar('/');
		putchar('\n');
	}
	n++;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n-i-1;j++)
			space();
		putchar('/');
		for(int j=0;j<i/2;j++)
			printf(" /");
		if(i&1)
			space(),space();
		for(int j=0;j<i/2;j++)
			printf("\\ ");
		putchar('\\');
		putchar('\n');
	}
	for(int i=n-1;i>=0;i--)
	{
		for(int j=0;j<n-i+1;j++)
			space();
		putchar('\\');
		for(int j=0;j<i/2;j++)
			printf(" \\");
		if(i&1)
			space(),space();
		for(int j=0;j<i/2;j++)
			printf("/ ");
		putchar('/');
		putchar('\n');
	}
	return 0;
}
