#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    { 
    	  int fd=0,rd=0;
    	 int n,d;
    	 cin>>n>>d;
    	 while(n>0&&fd!=d)
    	 {
    	 	  n--;
    	 	  fd=fd+2;
    	 	  rd=rd+2;
    	 	  if(rd>=6)
    	 	  {
    	 	  	 rd=0;
    	 	  	 n=n+2;
    	 	  }
    	 }
    	 	  if(fd<d)
    	 	  cout<<"No"<<"\n";
    	 	  else if(fd>=d)
    	 	  cout<<"Yes"<<"\n";
    	 }
    	 
    
    return 0;
    
}
