#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,m;
		cin>>n>>m;
		ll x1,y1,x2,y2,p;
		cin>>x1>>y1;
		cin>>x2>>y2;
		cin>>p;
		ll ans=p*(min(abs(x1-x2),n-abs(x1-x2))+min(abs(y1-y2),m-abs(y1-y2)));
		cout<<ans<<endl;
		if(ans<=1000)
			cout<<"You saved the group.\n";
		else
			cout<<"Let go of the group.\n";
	}
	return 0;
}
