#include<iostream>
#include<string>
#include<stack>
using namespace std;
int main()
{
    int t,i,j,k , n , m , cnt =0 , score =0;
    string s ;
    stack<char> stk , temp;
    cin>>m;
    for(k=0;k<m;k++)
    {
          cin>>s;
          cnt =0;
          n = s.length();
          while(!stk.empty())
            stk.pop();
          while(!temp.empty())
            temp.pop();
          stk.push(s[0]);cnt++;
          for(i=1;i<n;i++)
          {
              if((!stk.empty())&&(stk.top() == s[i]))
              {
                  stk.pop();cnt--;
              }
              else
              {
                  stk.push(s[i]);cnt++;
              }
          }
          if(!stk.empty())
          {
              if(cnt%2!=0)
              {
                  continue;
              }
              else
              {
                  for(i=0;i<cnt/2;i++)
                  {
                      temp.push(stk.top()); stk.pop();
                  }
                  while(!temp.empty()&&(!stk.empty()))
                  {
                      if(stk.top()==temp.top())
                      {
                          stk.pop();
                          temp.pop();
                      }
                      else
                      {
                          break;
                      }
                  }
                  if(!temp.empty()&&(!stk.empty()))
                    continue;
              }
          }
          score++;
    }
    cout<<score<<endl;
}
