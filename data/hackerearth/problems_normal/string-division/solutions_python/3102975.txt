def sol(s):
	for i in range(1,len(s)):
		for j in range(i+1,len(s)):
			for k in range(j+1,len(s)):
				myset = set([])
				myset.add(s[0:i])
				myset.add(s[i:j])
				myset.add(s[j:k])
				myset.add(s[k:])
				if len(myset) == 4 :
					return 1
	return 0

for t in range(input()):
	s = raw_input()
	if len(s) >= 10 or sol(s):
		print 'YES'
	else:
		print 'NO'
