#include <iostream>
#define ll long long int
using namespace std;

int main()
{
	ll n;
	scanf("%lld", &n);
	ll a[n];
	ll ans = 0, max = -1;
	for(ll i = 0; i < n; i++)
	{
		scanf("%lld", &a[i]);
		if(a[i] > max)
			max = a[i];
	}
	for(ll i = 1; i <= max; i*=2)
	{
		for(ll j = 0; j < n; j++)
		{
			if((a[j]&i) == i)
			{
				ans = ans | i;
				break;
			}
		}
	}
	printf("%lld\n", ans);
    return 0;
}
