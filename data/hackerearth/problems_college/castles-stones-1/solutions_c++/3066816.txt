#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <vector>
#include <algorithm>
using namespace std;
#define SIZE 1234
 
int main(void) {
	
	int T = 0;
	scanf("%d",&T);
	while(T--) {
		int M = 0,N = 0,res = 0;
		scanf("%d",&M);
		scanf("%d",&N);
		vector<int> v1(N);
		int it = 0;
		for(;it != N; ++it) {
			scanf("%d",&v1[it]);
		}
		int i = 0, j = N - 1;
		sort(v1.begin(),v1.end());
		for(i = 0; i <= j;) {
			if(v1[j] > M)
				--j;
			else if (v1[i] + v1[j] > M) {
				--j;
				++res;
			}
			else {
				--j;
				++i;
				++res;
			}
		}
		printf("%d\n",res);
	}
	
	
	return 0;
 
}
