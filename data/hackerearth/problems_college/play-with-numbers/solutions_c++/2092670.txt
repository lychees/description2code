#include <iostream>
#include <cmath>


using namespace std;

long long int max(long long x,long long int y)
{
	if(x>y)
	return x;
	else
	return y;
}
long long int min(long long x,long long int y)
{
	if(x>y)
	return y;
	else
	return x;
}

int main() {
    long long int z, m, n, t, low, high, temp, div;
    int ans;
    cin>>t;
    while(t--)
    {
        cin >> z >> m >> n;
        temp =min(m,n);
        low = min(sqrt(z), temp);
        high = max(m, n); //Maximum value

        for( ans = 0; low > 0 && (z / low) <= high; --low ) {
            if ( z % low == 0) {
                ++ans;
                div = z / low;
                ans += (div != low && div <= temp);
            }
                
            }
             cout << ans << "\n";
        }
       
    
    return 0;
}
