int wrk(int x, int y, int l, int r, int k) {
  int m = (1 << k);
  if (l > r || x > y)
    return 0;
  if (l < x || l == x && r > y)
    return wrk(l, r, x, y, k);
  if (y >= r)
    return r - l + 1;
  if (l > m)
    return wrk(x, y, l - m, r - m, k);
  if (y < m && r < m)
    return wrk(x, y, l, r, k - 1);
  return max(max(wrk(x, y, l, m - 1, k), wrk(1, r - m, x, y, k)), y - l + 1);
}
