    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int i,j,k,n;
    	cin>>n>>k;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
     
    	sort(a,a+n);
    	int min=INT_MAX;
    	int diff;
    	for(i=0;i<(n-k+1);i++)
    	{
    		diff=a[i+k-1]-a[i];
    		if(diff<min)
    		min=diff;
    	}
    	cout<<min;
    	return 0;
    }
