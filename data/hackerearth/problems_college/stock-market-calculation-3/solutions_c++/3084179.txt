#include <iostream>
using namespace std;
typedef long long int64;

int main()
{
	int T,N;
	cin>>T;
	while(T) {
		cin>>N;
		int arr[N];
		int max[N];
		for(int i=0;i<N;i++) {
			cin>>arr[i];
		}
		int64 res=0;
		max[N-1] = N-1;
		for(int j=N-2;j>=0;j--) {
			max[j] = max[j+1];
			if(arr[max[j]]<arr[j]) {
				max[j] = j;
			}
		}
		int i=0;
		while(i<N-1) {
			if(arr[i]<arr[max[i+1]]) {
				int j = max[i+1];
				while(i<j) {
					res+=arr[j]-arr[i];
					i++;
				}	 
			}
			i++;
		}
		cout<<res<<endl;
		T--;
	}
    return 0;
}
