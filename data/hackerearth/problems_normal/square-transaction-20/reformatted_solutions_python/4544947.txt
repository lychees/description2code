T = int(raw_input())
w = map(int, raw_input().strip().split())
q = int(raw_input())


def binarySearch(alist, item):
    first = 0
    last = len(alist)-1
    found = False

    while first <= last and not found:
        midpoint = (first + last)//2
        if alist[midpoint] > item and alist[midpoint - 1] < item:
            found = True
        elif alist[midpoint] == item:
            found = True
        else:
            if item < alist[midpoint]:
                last = midpoint-1
            else:
                first = midpoint+1

    return midpoint


list_sum = 0
w1 = []
for x in w:
    list_sum += x
    w1.append(list_sum)

for query in range(q):
    query = int(raw_input())
    list1 = w1
    if query > list1[-1]:
        print -1
    elif query < list1[0]:
        print 1
    else:
        found = binarySearch(list1, query)
        print int(found)+1
