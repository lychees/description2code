#include <iostream>
#define mod 1000000007
using namespace std;

long long pow(long long n)
{
	if(n==0)
		return 1;
	if(n==1)
		return 2;
	if(n%2==0)
	{
		return (((pow(n/2)%mod)*(pow(n/2)%mod))%mod);
	}
	else
	{
		return (((pow(n/2)%mod)*(pow(n/2)%mod)*2)%mod);
	}
}
int main()
{
    long long n,t;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	
    	cout<<(pow(n)-1)%mod<<endl;
    }
    return 0;
}
