#include <iostream>
using namespace std;
#define MAX 10000
int main()
{
    int N,arr[MAX],q,l,r,sum;
    cin >> N;
    for(int i=1;i<=N;i++)
    {
    	cin >> arr[i];
    }
    cin >> q;
    while(q--)
    {
    	 sum=0;
    	cin >> l >> r;
    	for(int i=l;i<=r;i++)
    		sum=sum+arr[i];
    	cout << sum <<endl;
    }
    
    return 0;
}
