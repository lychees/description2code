#include<stdio.h>

int main()
{
int a,b;
long int t;
long long int c;
scanf("%ld",&t);
while(t--)
{
scanf("%d %d %lld",&a,&b,&c);
//printf("%lld %lld %lld",a,b,c);
long long int d=0,an=a,bn=b,L,R,M,val;
long long int ans=0,p,n1,n2;
n1=a;
n2=b;
while(n1!=n2)
{
if(n1>n2-1)
n1=n1-n2;
else n2=n2-n1;
}

p = (a*b)/n1;
L = 1;
R = 100000000000000000;
while ( L <= R )
{
M = (L+R)/2;
val = M/a + M/b - M/p;
if ( val == c ) 
{
if((M/a)*a>(M/b)*b)
ans=((M/a)*a);
else
ans=((M/b)*b);
printf("%lld\n",ans);
break;
}
else if ( val < c ) 
{
L = M+1;
}
else 
{
R = M-1;
}
}
}
return 0;
}
