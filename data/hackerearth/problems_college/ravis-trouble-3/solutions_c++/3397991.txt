#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	int t;
	cin >> t;
	while(t--){
		string a,b;
		cin >> a >> b;
		int m = a.size();
		int n = b.size(); 
		int i=0,j=0,count=0;
		while(i < m && j < n){
			if(a[i]==b[j]){
				i++;
				j++;
				count++;
				if(b.size()==count)
				 break;
			}
			else
			 i++;
		}
		if(n == count)
		 printf("YES\n");
		else
		 printf("NO\n"); 
	}
	return 0;
}
