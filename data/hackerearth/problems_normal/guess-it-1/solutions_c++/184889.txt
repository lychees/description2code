#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;

int main()
{
    int t;
    char s[200];
    int sum,rem;

    cin>>t;
    getchar();

    while(t--)
    {
        sum=0;
        //cin.ignore(1e50,'\n');
        cin.getline(s,200);
        for(int i=0;s[i]!='\0';i++)
        {
            //cout<<"s="<<s<<endl;
            if(s[i]!='_')
            {
                sum+=s[i]-'0';
            }


        }
        //cout<<"sum="<<sum<<endl;
        rem=sum%9;
        //cout<<rem<<"rem"<<endl;
        if(rem==0)
        {
            cout<<"0 or 9"<<endl;
        }
        //cout<<1e5;
        else
            {
            cout<<9-rem<<endl;
        }
       // cout<<1<<endl;
    }
}
