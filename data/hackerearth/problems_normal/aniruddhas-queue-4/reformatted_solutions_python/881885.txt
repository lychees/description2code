#! /usr/bin/python

t = int(raw_input())
for x in xrange(t):
    n = long(raw_input())
    s = raw_input()
    d = s.split()
    m = long(raw_input())
    yd = long(0)
    # total distance in 1 year
    for i in d:
        yd = yd + long(i)

    ed = long(m % yd)
    ans = 0
    if (ed != 0):
        tmp = long(0)
        count = long(0)
        for j in d:
            tmp += long(j)
            count += 1
            if (tmp >= ed):
                ans = count
                break
    elif (m == 0):
        ans = 1
    else:
        for k in xrange(n):
            if (long(d[k]) > 0):
                ans = k + 1
    print ans
