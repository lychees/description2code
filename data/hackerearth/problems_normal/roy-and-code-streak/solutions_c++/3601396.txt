#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t = 0,n = 0,s = 0,r = 0,ans = INT_MIN;
	cin>>t;
	while(t--){
		cin>>n;
		map<int,bool> ma;
		ans = INT_MIN;
		int temp = 0;
		while(n--){
			cin>>s>>r;
			if(r == 0){
				ans = max(temp,ans);
				temp = 0;
			}
			else{
				if(ma.find(s) == ma.end()){
					temp++;
					ma[s] = true;
				}
			}
		}
		ans = max(temp,ans);
		cout<<ans<<endl;
	}
    return 0;
}
