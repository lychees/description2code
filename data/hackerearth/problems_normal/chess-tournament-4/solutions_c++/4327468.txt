#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
    int n;ios::sync_with_stdio(false);
    cin>>n;
    int size = (int)pow(2.0,(double)n)-1;
    int arr[size+1][size+1];
    //cout<<size<<endl;
    for(int i= 0;i<size;i++)
    {
        for(int j=0;j<=i;j++)
        {
            cin>>arr[i+1][j];
            if(arr[i+1][j]==0)
                arr[j][i+1]=1;
            else
                arr[j][i+1]=0;
        }
    }
    int newarr[size+1][size+1];
    for(int i=0;i<size+1;i++)
        newarr[0][i] = i;
    int x = (size+1)/2;
    //cout<<x<<endl;
    for(int j = 1;j<size;j++)
    {
        for(int k = 0;k<x;k++)
        {
            //cout<<k<<endl;
            if(arr[newarr[j-1][k*2]][newarr[j-1][k*2+1]]==1)
            {
                newarr[j][k]=newarr[j-1][k*2];
            }
            else
                newarr[j][k] = newarr[j-1][k*2+1];
        }
        x/=2;
    }
    /*for(int i = 0;i<size;i++)
    {
        for(int j=0;j<size+1;j++)
            cout<<newarr[i][j]<<' ';
        cout<<endl;
    }*/
    cout<<newarr[n][0]+1<<endl;

}
