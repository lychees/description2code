#include <iostream>
#include <bits/stdc++.h>
#define ll long long int
#define big 1000000000
using namespace std;

ll solve(ll n){
	
		ll dig[10];
		ll ans=0,temp=0;
		int i=0,c=0;
		
		while(n>0){
			dig[i++] = n%10;
			c++;
			temp = temp*10+dig[i-1];
			n = n/10;
		}
			
		
		for(int j=c-1;j>=0;j--){
			for(int k=0;k<dig[j];k++){
				
				if(k==3){
					continue;
				}
				if(j!=0){
					ans += (ll)pow(9.0,(double)j);
				}
				else{
					if(dig[0]>3){
						ans += dig[0]-1;
						break;	
					}
					else{
						ans += dig[0];
						break;
					}
				}
			}
		}
		return ans;
}

int main()
{	
	int t;
	cin >>t;
	while(t--){
		ll n;
		ll ans;
		
		cin >>n;
		
		ans = solve(n);
		cout <<ans<<endl;
	}
    return 0;
}
