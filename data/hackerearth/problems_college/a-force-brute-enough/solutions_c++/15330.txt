#include <iostream>

using namespace std;
void db(int t,int p,int n,int a[30]){
    int j;
    if(t>n){
        if(n%p==0){
            for(j=1;j<p+1;j++){
                cout<<a[j];
            }
        }
    }
    else{
        a[t]=a[t-p];
        db(t+1,p,n,a);
        for(j=a[t-p]+1;j<10;j++){
            a[t]=j;
            db(t+1,t,n,a);
        }
    }
}

int main() {
    int n,i;
    cin>>n;
    int a[100];
    while(n){
        for(i=0;i<n*10;i++){
            a[i]=0;
        }
        db(1,1,n,a);
        for(i=1;i<n;i++)
            cout<<0;
        cout<<endl;
        cin>>n;
    }
	return 0;
}
