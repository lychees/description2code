#include<iostream>
#include<map>
using namespace std;
int main(){
	long int T,i,max=0,count=0;
	cin>>T;
	map<string,int> list;
	map<string,int>:: iterator it;
	for(i=0;i<T;i++){
		string s;
		cin>>s;
		it=list.find(s);
		if(it==list.end())
			list.insert(make_pair(s,1));
		else{
			it->second++;
		}
	}
	for(it=list.begin(); it!=list.end(); ++it){
		if(it->second > max){
			max=it->second;
			count=it->second;
		}
		else if(it->second ==max)
			count+=max;
	}
    	cout<<count;
	return 0;
}
