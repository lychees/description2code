#include<bits/stdc++.h>
using namespace std;
map<string,bool> ma; 
string artostr(char a[]){
	string   temp="";
	for(int i=0;a[i]!='\0';i++){
		temp+=a[i];
	}
	return temp;
}
int main(){
	int t;
	scanf("%d\n",&t);
	while(t--){
		char m[100001];
		gets(m);
		char n[100011];
		int c=0;
		for(int i=0;m[i]!='\0';i++){
			if(m[i]==' '){
				n[c]=0;
				if(ma[artostr(n)]==false){
					printf("%s ",n);
				}
				ma[artostr(n)]=true;
				c=0;
			} else
			n[c++]=m[i];
		}
		n[c]=0;
		if(ma[artostr(n)]==false){
			printf("%s ",n);
		}
		printf("\n");
		ma.clear();
	}
	return 0;
}
