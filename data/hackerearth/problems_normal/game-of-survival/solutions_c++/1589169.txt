#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	int swords[n], animals[n];    	
    	
    	for (int i = 0; i < n; ++i)
    		cin >> swords[i];
    	for (int i = 0; i < n; ++i)
    		cin >> animals[i];
    	sort(swords, swords+n);
    	sort(animals, animals+n);
    	
    	bool flag = true;
    	for (int i = 0; i < n && flag; ++i)
    		if (animals[i] > swords[i])
    			flag = false;
    	
    	if (flag)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
