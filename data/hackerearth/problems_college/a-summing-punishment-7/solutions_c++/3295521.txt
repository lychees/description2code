#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 100010

ll arr[MAXN];
ll sum[MAXN];

int main()
{
	ll i,n,l,r,q;

	memset(sum,0,sizeof(sum));

	scanf("%lld", &n);
	
	for(i=1;i<=n;i++) {
		scanf("%lld", &arr[i]);
		sum[i] = sum[i-1] + arr[i];
	}
	
	scanf("%lld", &q);

	while(q--) {
		scanf("%lld%lld", &l,&r);
		
		printf("%lld\n", sum[r]-sum[l-1]);
	}

	return 0;
}
