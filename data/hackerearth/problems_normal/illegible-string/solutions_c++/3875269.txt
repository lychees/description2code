#include <iostream>
using namespace std;
 
int main()
{
    int n; cin >> n;
    string s; cin >> s;
    int v, w; v = w = 0;
    for (char ch : s) {
    	if (ch == 'w') ++w;
    	else if (ch == 'v') ++v;
    }
    
    int cnt = 0;
    for (int i = 0; i < s.length() - 1; ++i) {
    	if (s[i] == 'v' && s[i+1] == 'w') {
    	//	s[i] = 'w';
    		s[i + 1] = 'v';
    	}
    	else if (s[i] == 'v' && s[i + 1] == 'v') {
    		++cnt;
    		++i;
    	}
    }
    cout << s.length() - cnt << " " << s.length() + w << endl;
    return 0;
}
