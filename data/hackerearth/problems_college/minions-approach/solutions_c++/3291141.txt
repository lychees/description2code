#include <iostream>
#include<bits/stdc++.h>
using namespace std;

bool p[100001];
void calculate()
{
	memset(p,true,sizeof(p));
	long long int i,j;
	p[1]=false;
	for(i=2;i*i<=sqrt(100000);++i)
	{
		if(p[i]==true)
		{
			for(j=2*i;j<=100000;j=j+i)
			{
				p[j]=false;
			}
		}
	}
}

long long int gcd(long long int a,long long int b)
{
	if(a%b==0)
	return b;
	else return gcd(b,a%b);
}
int main()
{
    calculate();
    //calc();
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long int ctr=1,i;
    	for(i=2;i<n;++i)
    	{
    		if(gcd(n,i)==1)
    		{
    			++ctr;
    		}
    	}
    	if(p[ctr])
    	cout<<"Correct\n";
    	else cout<<"Incorrect\n";
    }
    return 0;
}
