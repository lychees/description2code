l = [i for i in range(26, 0, -1)]
c = "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
# print l
for _ in xrange(input()):
    chk = True
    s = raw_input()
    for ch in c:
        if ch in s:
            a = s.count(ch)
            b = ord(ch)-ord('A')
            # print a,b,l[b]
            if a != l[b]:
                chk = False
                break
    if chk:
        print "Yes"
    else:
        print "No"
