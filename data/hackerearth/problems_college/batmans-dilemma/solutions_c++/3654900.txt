#include <iostream>

using namespace std;
#define MOD 1000000007
#define SIZE 10000000

long long int fact[SIZE];

int main()
{
	long long int t, n, i, j, ans, m;
	long long int hash[6000];
	fact[0] = 1;
	fact[1] = 1;
	for (i = 2; i < SIZE; ++i) {
		fact[i] = (fact[i-1] * i) % MOD;
	}

	cin >> t;
	for (; t; --t) {
		cin >> n >> m;
		for (i = 0; i < 6000; ++i) {
			hash[i] = 0;
		}

		for (i = 1; i <= n; ++i) {
			for (j = 1; j <= m; ++j) {
				hash[i | j]++;
			}
		}

		ans = 1;
		for (i = 0; i < 6000; ++i) {
			if (hash[i]) {
				ans = (ans * fact[hash[i]]) % MOD;
			}
		}

		cout << ans << endl;
	}

	return 0;
}
