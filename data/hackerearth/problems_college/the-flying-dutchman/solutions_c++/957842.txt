#include<bits/stdc++.h>
#define assn(n,a,b) assert(n>=a && n<=b)
using namespace std;
typedef long long LL; 
LL dp[100009]={},val[100009],pre[100009];

int main() 
{ 
    int t; 
    scanf("%d",&t);
    while(t--) 
    {
        pre[0]=0;
        LL n,i,j,sum=0,p2; 
        scanf("%lld",&n); 
        assn(n,1,100000);
        for(i=n; i>0; i--)
        {
            scanf("%lld",&val[i]);
            sum=sum+val[i];
            assn(val[i],0ll,1000000000ll);
        } 
        for(i=1; i<=n; i++) 
            pre[i]=pre[i-1]+val[i];
        dp[0]=0;
        dp[1]=val[1];
        dp[2]=dp[1]+val[2];
        for(i=3; i<=n; i++)
        {
            dp[i]=max(-1ll,pre[i-1]-dp[i-1]+val[i]);
            dp[i]=max(dp[i],pre[i-2]-dp[i-2]+val[i]+val[i-1]); 
        }
        p2=sum-dp[n];
        if(dp[n]>p2) printf("LOSE\n");
        else if(dp[n]<p2) printf("WIN\n");
        else printf("DRAW\n");
        
    }
    return 0; 
}
