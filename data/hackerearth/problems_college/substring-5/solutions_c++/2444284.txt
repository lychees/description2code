#include<bits/stdc++.h>
#include<string.h>
using namespace std;
int main(){
    char x[102];
    int i,l,s=1,n=0;
    cin>>x;
    l=strlen(x);
    for(i=1;i<l;i++){
        if(x[i]==x[i-1])
            s++;
        else{
            n+=(s*(s+1))/2;
            s=1;
            }
        }
    n+=(s*(s+1))/2;
    cout<<n;
return 0;
}
