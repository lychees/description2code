#include <iostream>
using namespace std;
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	int p[]={1,1,2,6,4};
	int t;
	cin>>t;
	while(t--) {
		int a, b, ans=0;
		cin>>a>>b;
		for(int i=a; i<=b; i++) {
			if(i>=5)
				break;
			ans+=p[i];
		}
		cout<<ans%10<<"\n";
	}
	return 0;
}
