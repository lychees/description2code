#include <iostream>
#include<algorithm>
using namespace std;

int hcf(int a,int b)


         {
         	if(b==0)
			return a;


              if(a%b==0)


               {


                      return b;


               }


              else


              {


                      return(hcf(b,a%b));


              }


         } 

            


int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int a[n];
    	for(int i=0;i<n;i++)
    	scanf("%d",&a[i]);
    	sort(a,a+n);
    	int h;
    	for(int i=0;i<n-1;i++)
        {
            h=hcf(a[i],a[i+1]);
            a[i+1]=h;
        }
    	printf("%d\n",h);
    }
    
}
