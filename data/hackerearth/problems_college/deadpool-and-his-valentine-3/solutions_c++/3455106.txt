#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,m,l,n,x,f=0;
	cin>>i>>j;
	int a[j];
	for(m=0;m<j;m++)
	a[m]=0;
	
	for(m=0; m<i;m++){
		cin>>l;
		for(x=0;x<l;x++){
			cin>>n;
			a[n-1]=1;
		}
		
	}
	for(m=0; m<j;m++){
	    
		if(a[m]==0){
			f=1;break;
		}
	}
    if(f)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    return 0;
}
