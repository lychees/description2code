#include <iostream>
using namespace std;

int main()
{
    
    int test;
    
    long int side[100];
    long int vol;
    cin>>test;
    for(int i=0;i<test;i++){
    	cin>>side[i];
    }
    for(int i=0;i<test;i++){
    	
    	if(side[i] == 1){
    		vol = 1;
    	}
    		
    	else{
    		vol = (6*(side[i]-2)*(side[i]-2))+(12*(side[i]-2)) + 8;

    	}
    	//cout<<fixed;
    	cout<<vol<<"\n";
    }
    return 0;
}
