#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll MOD = 1e9 + 7;

ll binpow(ll a, ll k)
{
	ll res = 1;
	while (k) {
		if (k & 1)
			res = res * a % MOD;
		k >>= 1;
		a = a * a % MOD;
	}
	return res;
}

ll inv(ll a)
{
	return binpow(a, MOD - 2);
}

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
    ll n;
    cin >> n;
    
    vector<ll> p(n), q(n);
    for (auto& t : p)
    	cin >> t;
    for (auto& t : q)
    	cin >> t;
    	
    ll ans = 1;
    for (int i = 0; i < n; i++) {
    	ll a = p[i], b = q[i];
    	ll cur = binpow(a, b + 1) - 1;
    	if (cur < 0) cur += MOD;
    	cur *= inv(a - 1);
    	cur--;
    	if (cur < 0) cur += MOD;
    	cur %= MOD;
    	ans *= cur;
    	ans %= MOD;
    	//cout << cur << " ";
    }
    cout << ans << endl;
	}
}
