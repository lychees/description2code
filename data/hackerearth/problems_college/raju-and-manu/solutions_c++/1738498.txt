#include <iostream>
using namespace std;

int main()
{
	int t = 0;
	cin >> t;
	for (int i = 0; i < t; i++)
	{
		int n;
		cin >> n;
		int sum = 1;
		int max = 0;
		int ind = 0;
		for (int j = n; j >= 0; j--)
		{
			int m = j;
			if (j == 0)
			{
				sum = 0;
			}
			while (m > 1)
			{
				sum += m % 2;
				m /= 2;
			}

			if (sum > max)
			{
				max = sum;
				ind = j;
			}
			sum = 1;
		}
		cout << ind << endl;
	}
	return 0;
}
