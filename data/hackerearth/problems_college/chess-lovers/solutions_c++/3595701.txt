#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,x1,x2,y1,y2,d1,d2;
	cin>>t;
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2;
		d1=abs(x1-y1);
		d2=abs(x2-y2);
		if(!((d1%2&&d2%2)||(d1%2==0&&d2%2==0)))
		cout<<"-1\n";
		else
		{
			if(x1-y1==x2-y2)
			cout<<"0\n";
			else if(x1+y1==x2+y2)
			cout<<"0\n";
			else
			cout<<"1\n";
		}
	}
}
