#include <iostream>
#include<string>
using namespace std;

int main()
{
   int n,i,j,count=0;
   string s;
   cin>>n;
   int a[26]={0};
   for(i=1;i<=n;i++){
   		j=0;
   		cin>>s;
   	    while(s[j]!='\0'){
   			     if(a[(int)s[j]-97]==(i-1))
   			      a[(int)s[j]-97]=i;
   			      j++;
   			    }
   }
   
   for(i=0;i<26;i++){
   	if(a[i]==n)
   	  count++;
    }
   cout<<count;
    return 0;
}
