def hcf(a,b):
	if(b==0):
		return a;
	else:
		return hcf(b,a%b)

a = int(raw_input())
n = int(raw_input())
numbers = raw_input().split(' ')
friends = 0
for i in numbers:
	if a>int(i):
	    gcd = hcf(a,int(i))
	else:
		gcd = hcf(int(i),a)
	if gcd>1:
		friends += 1
print friends
