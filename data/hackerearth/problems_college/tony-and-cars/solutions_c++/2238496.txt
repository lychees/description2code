#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<math.h>
using namespace std;
int main()
{
    long long int t,i,x,n,l,r,count;
    cin>>t;
    while(t--){
        cin>>x;
        n=0;
        l=1;r=9;
        count=1;
        while(1){
            //cout<<l<<" "<<r<<" "<<n<<" "<<count<<"\n";

            if(x<=r){
                n+=count*(x-l+1);
                break;
            }else{
                n+=count*(r-l+1);
            }
            l=r+1;
            r=(r+1)*10-1;
            count++;
        }
        cout<<n<<"\n";
    }
    return 0;
}
