
#include <cstdio>
#include <cstring>
#include <cstdlib>
 #include <cmath>
 #include <iostream>
 #include <algorithm>
 using namespace std;
 #define ll long long
 
 int main(){
 	int t;
 	cin>>t;
 	
 	while(t--){
     int sum;
     int v[21];
     ll dp[10005];
     for(int i = 1; i <= 21; ++i)
         v[i - 1] = i * i * i;
     memset(dp,0,sizeof(dp));
     dp[0] = 1;
     for(int i = 0; i < 21; ++i){
         for(int j = v[i]; j <= 10000; ++j){
             dp[j] += dp[j - v[i]];
         }
     }
     while(scanf("%d",&sum) == 1)
         printf("%lld\n",dp[sum]);
 	}
     return 0;
 }
