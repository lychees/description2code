#include<iostream>
using namespace std;
long long a[1000000];
int main()
{
 	int n;
 	cin>>n;
 	
 	for(int i=0;i<n;i++)
 	{
	 		cin>>a[i];
			}
			long long maxsum=0,cur=0;
			for(int i=0;i<n;i++)
			{
			 		cur=cur+a[i];
			 		if(cur<0)
			 		{
					 		 cur=0;
							 }
							 if(cur>maxsum)
							 maxsum=cur;
			}
			cout<<maxsum<<"\n";
			}
