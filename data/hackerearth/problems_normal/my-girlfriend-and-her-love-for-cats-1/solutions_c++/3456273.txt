#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
    long long int n,i;
    scanf("%lld",&n);
    unsigned long long int sum=0;
    vector<long long int>c;
    vector<long long int>s;
    s.resize(n);
    c.resize(n);
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&s[i]);
    }
    for(i=0;i<n;i++)
    {
    	scanf("%lld",&c[i]);
    }
    sort(c.begin(),c.end());
    sort(s.begin(),s.end());
    for(i=0;i<n;i++)
    {
    	sum+=s[i]*c[i];
    }
    printf("%lld",sum);
    return 0;
}
