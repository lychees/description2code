#include <iostream>
using namespace std;
void swap(int* x,int* y) {
	int t = *x;
	*x = *y;
	*y = t;
	return ;
}
int main() {
	int i,j,k,l,n,x,v,a[4][105],y;
	cin>>x>>n;
	y = 1000000;
	for(i=0;i<n;i++) {
		cin>>j>>k>>l>>v;
		if(v>k) {
			a[0][i] = j;
			a[1][i] = k;
			a[2][i] = l;
			a[3][i] = v;
		}
		else {
			a[0][i] = l;
			a[1][i] = v;
			a[2][i] = j;
			a[3][i] = k;
		}
	}
	for(i=n-1;i>0;i--)
		for(j=0;j<i;j++)
			if(a[3][j]>a[3][j+1])
				for(k=0;k<4;k++)
					swap(&a[k][j],&a[k][j+1]);
//	for(i=0;i<n;i++)
//		cout<<a[0][i]<<" "<<a[1][i]<<" "<<a[2][i]<<" "<<a[3][i]<<endl;
	while(1) {
		for(i=n-1;i>=0;i--)
			if(a[3][i]<=y && ((a[2][i]>=x && a[0][i]<=x) || (a[2][i]<=x && a[0][i]>=x))) {
				y = a[1][i];
				x = a[0][i];
				break;
			}
//		cout<<i<<" "<<x<<" "<<y<<endl;
		if(i<=0)
			break;
	}
	cout<<x<<endl;
	return 0;
}
