#include<bits/stdc++.h>
#define ll long long int
#define size 100005
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	ll test;
	cin>>test;
	while(test--)
	{
		ll cntt[27],cnts[27],cnt=0;
	char s[size],t[size];
	cin>>s;
	cin>>t;
	memset(cntt,0,sizeof(cntt));
	memset(cnts,0,sizeof(cnts));
	for(int i=0;i<strlen(t);i++)
	{
		cntt[t[i]-96]++;
	}
	for(int i=0;i<strlen(s);i++)
	{
			cnts[s[i]-96]++;
	}
	int flag=0;
	for(int i=1;i<27;i++)
	{
		if(cnts[i]>cntt[i])
		{
			flag=1;
			break;
		}
		else if(cntt[i]>0 && cntt[i]>cnts[i])
		{
			cnt++;
		}
	}
	if(flag==1)
	{
		cout<<"-1"<<endl;
	}
	else
	{
		cout<<cnt<<endl;
	}
	}
	
	
	return 0;
}
