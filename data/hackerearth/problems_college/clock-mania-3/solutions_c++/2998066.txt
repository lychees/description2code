#include <iostream>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	
   	while(t--){
   		int h,m,H,M;
   		cin>>h>>m;
   		
   		H=0.5*m+ 30*(h%12);
   		M=6*m;
   		
   		int s;
   		s=M-H;
   		if(s<0) s=-s;
   		
   		if(360-s>s)cout<<s<<endl;
   		else cout<<360-s<<endl;
   		
   	}
    return 0;
}
