#include <bits/stdc++.h>
using namespace std;
bool binarysearch(int arr[],int key,int n){
	int low=0,high=n-1,mid;
	while(low<=high){
		mid=low+(high-low)/2;
		if(mid<0||mid>=n) return 0;
		if(arr[mid]>key) high=mid-1;
		else if(arr[mid]<key) low=mid+1;
		else if(arr[mid]==key) return 1;
	}
	return 0;
}
int main()
{
    int n,q;
    cin>>n>>q;
    int arr[n];
    for(int i=0;i<n;i++) cin>>arr[i];
    sort(arr,arr+n);
    while(q--){
    	bool flag=1;
    	int key; cin>>key;
    	flag=binarysearch(arr,key,n);
    	if(flag==1) cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
