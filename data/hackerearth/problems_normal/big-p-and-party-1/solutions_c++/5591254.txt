#include <bits/stdc++.h>
using namespace std;
vector<int>a[1005];
int luck[1005];
bool visited[1005];
void sol(int n){
	memset(luck,-1,sizeof(luck));
	memset(visited,false,sizeof(visited));
	queue<int>q;
	for(int i=0;i<a[0].size();i++){
		int v=a[0][i];
		q.push(v);
		visited[v]=true;
		luck[v]=1;
	}
	while(!q.empty()){
		int u=q.front();
		q.pop();
		for(int i=0;i<a[u].size();i++){
			int v=a[u][i];
			if(!visited[v]){
				q.push(v);
				visited[v]=true;
				luck[v]=1+luck[u];
			}
		}
	}
	for(int i=1;i<n;i++){
		cout<<luck[i]<<endl;
	}
}
int main() {
	int n,m;
	cin>>n>>m;
	while(m--){
		int u,v;
		cin>>u>>v;
		a[u].push_back(v);
		a[v].push_back(u);
	}
	sol(n);
	return 0;
}
