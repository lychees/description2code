#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int GCD(long long, long long);

int main()
{
    long long T, temp;
    vector<long long> P;
    cin >> T;
    
    while (T--)
    {
    	P.clear();
        int i = 0;
        while (i++ < 3) { cin >> temp; P.push_back(temp); }
        sort(P.begin(), P.end());
        if (GCD(GCD(P[0], P[1]), P[2]) > 1) cout << "NO" << endl;
        else
        {
        	long long t = P[0]*P[0] + P[1]*P[1];
        	if (t == P[2]*P[2]) cout << "YES" << endl;
        	else cout << "NO" << endl;
        }
    }
    return 0;
}

int GCD(long long a, long long b)
{
	if (!(a%b)) return b;
	else GCD(b, a%b);
}
