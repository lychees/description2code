#include <iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
int verify(string s,string s2)
{
	int a,b;
	a=s.size();
	b=s2.size();
	int flag=1;
	if(a==b)
	{
		if(s[0]==s2[0] && s[a-1]==s2[b-1] )
		{
			sort(s.begin(),s.end());
			sort(s2.begin(),s2.end());
			if(s!=s2)
			{
			flag=0;
			}
		}
		else
		flag=0;
	}
	else
	flag=0;
	return flag;
}
int main()
{
    int n;
    cin>>n;
    vector<string> v;
    for(int i=0;i<n;i++)
    {
    	string s;
    	cin>>s;
    	v.push_back(s);
    }
    vector<int> chk(n,1);
    for(int i=0;i<n;i++)
    {
    	if(chk[i]==1)
    	{
    		for(int j=i+1;j<n;j++)
    		{
    			if(verify(v[i],v[j]))
    			{
    				chk[j]=0;
    			}
    		}
    	}
    }
    int c=0;
    for(int i=0;i<n;i++)
    {
    	if(chk[i]==1)
    	c++;
    }
    cout<<c<<endl;
}
