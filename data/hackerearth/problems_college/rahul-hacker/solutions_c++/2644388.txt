#include <iostream>
using namespace std;

long int reverse(long int n)
{
	int final=0;
	while(n)
	{
		int a=n%10;
		n=n/10;
		final=final*10+a;
	}
	return final;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n1,n2,ans;
    	cin>>n1>>n2;
    	n1=reverse(n1);
    	n2=reverse(n2);
    	ans=n1+n2;
    	cout<<reverse(ans)<<endl;
    }
    return 0;
}
