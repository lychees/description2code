#include <iostream>
using namespace std;

int main()
{
	int t,n,i,j,count,a[100000];
	cin>>t;
   while(t--)
   {
   	
   	cin>>n;
   	for(i=0;i<n;i++)
   	cin>>a[i];
   	count=0;
   	
   	for(i=0;i<n;i++)
   	{
   		for(j=0;j<n;j++)
   		{
   			if(j>=i&&(a[i]==a[j]))
   			{
   			count++;
   			
   			}
   		}
   	}
   	cout<<count<<endl;
   }
    return 0;
}
