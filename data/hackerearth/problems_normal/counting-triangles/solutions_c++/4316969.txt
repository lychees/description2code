#include <bits/stdc++.h>

using namespace std;

int n, res=0;
map<vector<long long>,int > mapa;

int main()
{
    cin >> n ;
    for (int i=0; i<n; i++){
        vector<long long> arr(3);
        cin >> arr[0] >> arr[1] >> arr[2];
        sort(arr.begin(), arr.end());
        mapa[arr] ++;
    }
    for (auto it : mapa){
        if (it.second == 1) {
            res ++;
        }
    }
    cout << res << "\n";
}
