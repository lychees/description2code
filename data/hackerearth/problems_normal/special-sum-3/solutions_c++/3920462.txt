int gcd(int x,int y)
{
    int m,i;
    if(x>y)
         m=y;
    else
         m=x;

    for(i=m;i>=1;i--){
         if(x%i==0&&y%i==0){

             return i;
         }
    }
}
int foo (int n)
{
    int ret=0;
    int i;
    for(i=1;i<=n;i++)
    {
        if (gcd(n,i)==1)
        {
            ret++;
        }
    }
    return ret;
}
int specialsum(int n)
{
    int ret,i;
    ret=0;
    for (i=1;i<=n;i++)
    {
        if (n%i==0)
        {
            ret=ret+foo(i);
        }
    }
    return ret;

}
