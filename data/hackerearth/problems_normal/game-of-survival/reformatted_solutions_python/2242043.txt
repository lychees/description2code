t = int(raw_input())
while t:
    t -= 1
    a = []
    b = []
    n = int(raw_input())
    for i in range(n):
        temp = int(raw_input())
        a.append(temp)
    for i in range(n):
        temp = int(raw_input())
        b.append(temp)

    a.sort()
    b.sort()
    flag = True
    for i in range(n):
        if (a[i] <= b[i]):
            flag = False
    if (flag):
        print "YES"
    else:
        print "NO"
