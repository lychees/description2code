#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        long long n,i,x=0,temp,y=0,ans=0;
        cin>>n;
        temp=n;
        while(temp!=0){y+=temp%10;temp/=10;}
        if(y==1){cout<<"NO\n";continue;}
        for(i=1;i*i<=y;i++)
        {
            if(y%i==0)
            {
                x++;
                if(y/i!=i)x++;
            }
        }
        if(x>2)cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
