#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
bool B[30];
int main(int argc, char* argv[])
{
    if(argc == 2 or argc == 3)
        freopen(argv[1], "r", stdin);
    if(argc == 3)
        freopen(argv[2], "w", stdout);
    string s;
    int ans, count, k;
    cin >> k >> s;
    ans = 0;
    for(int i = 0;i < s.length();++i)
    {
        count = 0;
        memset(B, false, sizeof(B));
        for(int j = i;j < s.length();++j)
        {
            if(B[s[j] - 'a'] == false) count++;
            B[s[j] - 'a'] = true;
            if(count == k)
                ans++;
        }
    }
    cout << ans << endl;
    return 0;
}
