#include<iostream>
using namespace std;
int main()
{
    int t, a, b, c, d, counter;
    cin >> t;
    while(t--)
    {
        cin >> a >> b;
        c = a^b;
        counter = 0;
        while(c > 0)
        {
            d = c%2;
            if(d == 1)
            {
                counter++;
            }
            c /= 2;
        }
        if(counter%2 == 0)
        {
            cout << "YES" << endl;
        }
        else
        {
            cout << "NO" << endl;
        }
    }
    return 0;
}
