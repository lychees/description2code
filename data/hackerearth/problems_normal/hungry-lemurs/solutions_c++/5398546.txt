#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int ans=1000000,tempf,temps,tempm;
    for(int i=1;i<100009;i++){
    	tempf=n%i;
    	temps=i-tempf;
    	tempm=min(tempf,temps);
    	tempm+=abs(k-i);
    	if(tempm<ans)
    	ans=tempm;
    }
    
    cout<<ans<<endl;
    return 0;
}
