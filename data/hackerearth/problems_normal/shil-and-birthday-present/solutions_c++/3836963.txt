#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,x;
    cin>>n;
    int a[1000001]={0};
   long long int dist=0;
    for(int i=0;i<n;i++){
    	
    	cin>>x;
    	if(a[x]==0){
    	a[x]=1;
    	dist++;
    		
    	}
    }
    cout<<dist*(dist-1)/2;
    return 0;
}
