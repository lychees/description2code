#include <vector>
#include <iostream>
#include <algorithm>
#include<stdlib.h>
using namespace std;

struct node
{
	int a;
    int index;
};

bool compare(node lhs,node rhs)
{
	 return (lhs.a<rhs.a);
}
int main()
{
    int t,a,b;
    cin>>t;
    struct node temp[t];
    
       for(int i=0;i<t;i++)
      {
        cin>>a>>b;
        temp[i].a=a+b;
        temp[i].index=i;
      }
      
      sort(temp,temp+t,compare);
      for(int i=0;i<t;i++)
      cout<<(temp[i].index)+1<<" ";
      
   return 0;
}
