#include <bits/stdc++.h>
 
using namespace std;
 
void primeFactors(int n) {
    int c=0;
    while (n%2 == 0) {
        c++;
        n = n/2;
    }
    cout<<"2^"<<c;
    int t,f=0, sq = sqrt(n);
    for (int i = 3; i <= sq; i+=2){
        t=0;
        while (n%i == 0){
            t++;
            n = n/i;
        }
        if(t) {
        	cout<<"*"<<i<<"^"<<t;
        	f = 1;
        }
    }
    int p=0;
    if (n > 2) {
        p++;
    }
    if((c || f) && p) cout<<"*"<<n<<"^"<<p;
}
 
int main() {
	int t,n;
	cin>>t;
	while(t--) {
		cin>>n;
		primeFactors(n);
		cout<<endl;
	}
    return 0;
}
