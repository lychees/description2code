#include<iostream>
#include<stdio.h>
using namespace std;
int main ()
{
	int a[1000],k,g,j=999,car, t, n, count=0;
	scanf("%d", &t);
	while(t--)
	{
		for(int i=0; i<1000; i++)
		a[i]=0;
		a[0]=1;
		scanf("%d",&n);
		for(int i=1;i<=n;i++)
		{
			k=0;
			car=0;
			while(a[k]==0)
			k++;
			while(k<1000)
			{
				g=a[k]*i+car;
				//cout<<g<<"&&";
				a[k]=g%10;
				car=g/10;
				k++;
			}
		}
		for(int i=0; i<1000; i++)
		{
			if((a[i]==4)||(a[i]==7))
			count++;
		}
		printf("%d\n", count);
		count=0;
		j=999;
	}
	return 0;
}
