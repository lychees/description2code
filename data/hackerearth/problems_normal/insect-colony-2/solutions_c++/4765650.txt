#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main() {
	int T;
	cin >> T;
	
	int a[105];
	while(T--) {
		int N;
		cin >> N;
		for(int i = 0;i<N; i++) cin >> a[i];
		
		int o = 0;
		for(int i = 0;i<N; i++) o = o^a[i];
		if(o%2 == 0) cout << "Yes\n";
		else cout <<"No\n";
	}
	return 0;
}
