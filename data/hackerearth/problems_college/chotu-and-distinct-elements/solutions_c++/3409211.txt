#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,t,n,p,i,x,y;
	scanf("%d",&t);
	while(t--) {
		x=0;
		y=0;
		set <int> s;
		set <int> :: iterator it,ti;
		scanf("%d",&n);
		while(n--) {
			scanf("%d",&p);
			s.insert(p);
		}
		ti=s.end();
		ti--;
		k=s.size();
		for(it=s.begin(),i=0;;i++) {
			if(i==k)
			break;
			if(i%2==0) {
				x=*(ti);
				printf("%d ",*(ti));
				ti--;
			}
			else {
				y=*(it);
				printf("%d ",*(it));
				it++;
			}
		}
		printf("\n");
	}
	return 0;
}
