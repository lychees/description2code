#include <iostream>
using namespace std;

int main()
{
    int t,i,f,j;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	f = 1;
    	i = 0;
    	j = s.size()-1;
    	while(i<j){
    		if(s[i]!=s[j]){
    			f=0;
    			break;
    		}
    		i++;
    		j--;
    	}
    	if(f==0){
    		cout<<-1<<endl;
    	}else{
    		cout<<int(s[s.size()/2])<<endl;
    	}
    }
    return 0;
}
