int GCD (int a,int b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : GCD(b,a % b));
}

long long int POW (long long int base,long long int exp) {
	long long int val;
	val = 1;
	while (exp > 0) {
		if (exp % 2 == 1) {
			val = (val * base) % MOD;
		}
		base = (base * base) % MOD;
		exp = exp / 2;
	}
	return val;
}

int a[1500];
vector <int> v(1500);
int ans[1500][1500];

int main()
{
	int n,m,i,j,sz,idx,q,l,r;
	scanf("%d",&n);
	for (i = 0; i < n; i++) {
		scanf("%d",&a[i]);
	}
	for (i = 0; i < n; i++) {
		v.clear();
		sz = 0;
		for (j = i; j < n; j++) {
			if (j - i == 0) {
				v[sz] = a[j];
				sz++;
			} else {
				idx = sz - 1;
				while (idx >= 0 && v[idx] > a[j]) {
					v[idx + 1] = v[idx];
					idx--;
				}
				v[idx + 1] = a[j];
				sz++;
			}
			ans[i][j] = v[(j - i) / 2];
		}
	}
	scanf("%d",&q);
	while (q --) {
		scanf("%d%d",&l,&r);
		l--;
		r--;
		printf("%d\n",ans[l][r]);
	}
	return 0;
}
