#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int no;
    	cin>>no;
    	long int a[no];
    	for(int i=0;i<no;i++)
    			cin>>a[i];
    	
    	sort(a,a+no);
    	
    	long int big=a[no-1];
    	long int sml=a[0];
    	long int diff=big-sml;
    	if(diff%2==1)
    			cout<<"Yes\n";
    	else
    			cout<<"No\n";
    }
    return 0;
}
