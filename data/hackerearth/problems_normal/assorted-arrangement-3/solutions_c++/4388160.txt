#include<iostream>
#include<stdio.h>
using namespace std;
inline void scan_f(int *a)
{
	char c = 0;
	while(c<33)
		c = getc(stdin);
	(*a) = 0;
	while(c>33)
	{
		(*a) = (*a)*10 + c - '0';
		c = getc(stdin);
	}
}
int m;
int solve(int c[55],int index,int ans)
{
	if(ans%c[index])
		return 1;
	for(int i=index+1;i<=m;i++)
	{
		if(ans%c[i]==0)
			return 1;
	}
	return 0;
}
int main()
{
	int n,c[55],x,ans=0;
	cin >> n >> m;
	for(int i=1;i<=m;i++)
		scan_f(&c[i]);
	for(int i=1;i<=n;i++)
	{
		ans++;
		cin >> x;
		while(solve(c,x,ans))
			ans++;

	}
	cout << ans << '\n';
	return 0;
}
