#include <iostream>
#include<math.h>
using namespace std;

int main()
{   int T,i,N;
    int64_t sum;
    cin>>T;
    
    while(T--)
    {   
    	cin>>N;
    	int s=sqrt(N);i=1;sum=0;
       while(i<=s)
          {
          sum+=N/i;
          i++;
          }
         sum=sum*2-N-s*s;
         
    	cout<<sum<<"\n";
    }
    return 0;
}
