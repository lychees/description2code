#include <bits/stdc++.h>
using namespace std;
#define maxn 200002
#define maxc 26

char s[maxn];
int BIT[maxn][maxc];
void update(int n, int i, int j, int val)
{
	while(i <= n)
	{
		BIT[i][j] += val;
		i += i&-i;
	}
}
int get(int i, int j)
{
	int ret = 0;
	while(i)
	{
		ret += BIT[i][j];
		i -= i&-i;
	}
	return ret;
}

int main()
{
	int n, i, q, k, l, r, m, x, cnt = 0;
	char c;
	scanf("%s", s + 1);
	n = strlen(s + 1);
	for(i = 1; i <= n; i++)
		update(n, i, s[i] - 'a', 1);
	scanf("%d", &q);
	while(q--)
	{
		scanf("%d %c", &k, &c);
		l = 1;
		r = n;
		while(l <= r)
		{
			m = (l + r) >> 1;
			if(get(m, c - 'a') >= k)
			{
				x = m;
				r = m - 1;
			}
			else
				l = m + 1;
		}
		update(n, x, c - 'a', -1);
	}
	for(i = 1; i <= n; i++)
	{
		for(k = 0; k < maxc; k++)
			if(get(i, k) - get(i - 1, k))
				break;
		if(k !=  maxc)
			s[cnt++] = 'a' + k;
	}
	s[cnt++] = '\0';
	printf("%s", s);
	return 0;
}
