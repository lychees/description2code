#include <iostream>
using namespace std;

int main()
{
	int k,i,j,u,v,n;
	string a,b,s[10000],c;
	cin>>k;
	cin>>a;
	j=0;b="";
	c=a;
	n=a.size();
	while (j<=k&&b!=c)
	{
		b="";
		if (n%2==1)
		{
		for (i=0;i<n;i+=2)
		b=b+a[i];
		i=n-2; 
		for (i;i>=1;i-=2)
		b=b+a[i];
		}
		else 
		{
		for (i=0;i<n-2;i+=2)
		b=b+a[i];
		b=b+a[n-2]+a[n-1];
		i=n-3;
		for (i;i>=1;i-=2)
		b=b+a[i];
		}
		j+=1;
		s[j]=b;
		a=b;
	}
	if (j==k) cout<<s[j];
	else cout<<s[k%j];
}
