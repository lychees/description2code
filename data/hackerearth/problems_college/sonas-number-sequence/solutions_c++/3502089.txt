#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i;
    cin>>n;
    long b[n];
    long a[n];
    for(i=0;i<n;i++)
    {
        cin>>b[i];
    }
    a[0] = b[0];
    for(i=1;i<n;i++)
    {
        a[i] = (i+1)*b[i] - i*b[i-1];
    }
    for(i=0;i<n;i++)
    {
        cout<<a[i]<<" ";
    }
    cout<<endl;
    return 0;
}
