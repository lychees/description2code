#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6+3;
int ar[MAXN];
int main() {
	int t,i;
	scanf("%d",&t);
	while(t--) {
		int n;
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&ar[i]);
		unordered_map<int,int> umap;
		long long ans=0;
		for(i=n-1;i>=0;i--) {
			umap[ar[i]]++;
			ans+=umap[ar[i]];
		}
		printf("%lld\n",ans);
	}
	return 0;
}
