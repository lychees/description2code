#include <iostream>
#include <cstdio>
#include <string>
#include <cmath>
#include <algorithm>
 
using namespace std;
 
string f(long long n)
{
       string p = "45";
       string s,ans("");
       n--;
       int i;
       while ( n >= 0 ) {
             s.push_back((char)(n%2 + 48));
             n = n/2 - 1;
       }
       reverse(s.begin(),s.end());
       for ( i = 0; i < s.size(); i++ ) ans += p[s[i]-48];
       return ans;
}

int main()
{
    int t;
    long long n,pro,val,len,num;
    cin >> t;
    while ( t-- ) {
          cin >> n;
          val = 0;
          len = 1;
          pro = 1;
          num = 0;
          while ( 1 ) {
                if ( val + (len*pro*2) >= n ) {
                     break;
                }
                else {
                     val += len*pro*2;
                     len++;
                     num += pro*2;
                     pro *= 2;
                }
          }
          n -= val;
          num += n/len;
          char p1;
          if ( n%len != 0 ) num++;
          string ans = f(num);
          if ( n%len == 0 ) {
               p1 = ans[ans.size()-1];
          }
          if ( n%len != 0 ) {
               p1 = ans[(n%len)-1];
          }
          if ( p1 == '4' ) cout << "Hacker" << endl;
          else cout << "Earth" << endl;
    }
    return 0;
}
