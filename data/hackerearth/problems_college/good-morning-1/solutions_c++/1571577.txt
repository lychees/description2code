#include <iostream>
using namespace std;

int dp[101];
void init()
{
	for(int i=0;i<100;i++)
	{
		if(i==0)
		    dp[i]=0;
		else if(i==1)
		     dp[i]=2;
		else
		    dp[i]=6*(i-2)+4;
	}
}	
int main()
{
    int t,n,d;
    cin >> t;
    init();
    while(t--)
    {
    	cin >> n >> d;
    	if(dp[n]>=d)
    	    cout << "Yes" << "\n";
    	else
    	   cout << "No"<< "\n";
    }
    
    return 0;
}
