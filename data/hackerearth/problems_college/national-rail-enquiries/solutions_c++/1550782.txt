#include <iostream>
using namespace std;
long long int gcd( long long int a, long long int b)
{
	if(b==0)
	return a;
	else return gcd(b,a%b);
}
long long int factorial(long long int n)
{
	if(n==1||n==0)
	return 1;
	else 
	return factorial(n-1)*n;
}
int main()
{
	int t;
	cin>>t;
	int c=1;
	while(t--)
	{
		long long int x;
		cin>>x;
		if(x<=110)
		{
			 long long int res=factorial(x-1)%110;
			res=res%110;
			if(res>0 && gcd(res,x)==1)
			{
				cout<<"Case"<<" "<<c<<":"<<"Yes"<<endl;
			}
			else 
			cout<<"Case"<<" "<<c<<":"<<"No"<<endl;
			
		}
		else
		cout<<"Case"<<" "<<c<<":"<<"No"<<endl;
		
		c++;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
