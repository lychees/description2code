#include <iostream>
#include <cstring>
#include <algorithm>
#include <cstdio>
#include <map>
#include <vector>
#include <algorithm>

#define FOR(i,a,b) for (int i = a; i <= b; i++)
#define DFOR(i,a,b) for (int i = a; i >= b; i--)

using namespace std;

const int MAX = 111000;

int n,d[MAX+1],t;
bool mark[MAX+1];

main()
{
	memset(mark, true, sizeof(mark));
	memset(d, 0, sizeof(d));
	FOR(i,2,MAX)
	if (mark[i])
	{
		d[i] = 1;
		FOR(j,2,MAX/i)
		{
			t = i*j;
			while (t % i == 0)
			{
				d[i*j]++;
				t/=i;
			}
			mark[i*j] = false;
		}
	}
	scanf("%d",&n);
	int result = 0, u;
	FOR(i,1,n)
	{
		scanf("%d",&u);
		result ^= d[u];
	}
	if (result == 0) cout << "BOB" << endl;
	else cout << "ALICE" << endl;
}
