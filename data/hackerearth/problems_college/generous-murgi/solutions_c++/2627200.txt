
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,b,c=0;
        cin>>n>>b;
        long long k=n-b;
         if(b>0)
  {

        for(long long i=1;i<=(sqrt(k));i++)
        {

                if(n%i!=0&&k%i==0&&n%i==b)
                {
                    c++;
                }
                long long j=k/i;
                if(n%j!=0&&k%j==0&&n%j==b&&k/i!=i)
                {
                    c++;
                }
            }
  }
  else if(b==0)
  {
      for(long long i=1;i<=(sqrt(k));i++)
        {

                if(k%i==0)
                {
                    c++;
                }
                long long j=k/i;
                if(k%j==0&&k/i!=i)
                {
                    c++;
                }
            }

  }
if(n==b)
{
    cout<<"-1"<<"\n";
}
else

        cout<<c<<"\n";
    }
    return 0;
}
