#include <iostream>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		string m,n;
		cin>>m>>n;
		int count=0;
		int temp=0;
		for(int i=m.size()-1;i>=0;i--)
		{
			if(temp+m[i]+n[i]-96>=10)
			{
			temp=1;
			count++;
		    }
		}
		if(count==0)
		cout<<"No carry operation"<<endl;
	    else if(count==1)
		cout<<"1 carry operation"<<endl;
		else
		cout<<count<<" carry operations"<<endl;
	}
}
