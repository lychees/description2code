#include <iostream>
using namespace std;

int main()
{
 int t;
 cin>>t;
 int n;
 while(t--){
 	cin>>n;
 	if(n==1){
 		cout<<"-1"<<"\n";
 	}else if(n%3==0){
 		while(n--){
 			cout<<"5";
 		}
 		cout<<"\n";
 	}else if(n%5==0){
 		while(n--){
 			cout<<"3";
 		}
 		cout<<"\n";
 	}else {
 		int count=0;
 		 while(n%3!=0){
 		 	n=n-5;
 		 	count++;
 		 }
 		 count*=5;
 		 while(n--){
 		 	cout<<"5";
 		 }
 		 while(count--){
 		 	cout<<"3";
 		 }
 		 cout<<"\n";
 	}
 }
 
 return 0;
}
