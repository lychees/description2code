#include <iostream>
using namespace std;

int main()
{
	int t;
    int n,a[1005];
    long long int u,s;
    cin>>t;
    while(t--){
    cin>>n;
    u = 1002;
    s = 0;
    for(int i=0;i<n;i++){
    	cin>>a[i];
    	s += a[i]-1;
    	u = u>a[i]?a[i]:u;
    }
    cout<<u-1<<" "<<s<<endl;
    }
    return 0;
}
