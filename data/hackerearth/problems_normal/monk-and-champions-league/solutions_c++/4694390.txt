#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#include<algorithm>

int main()
{
  long long n,m,count=0,temp;
  cin>>m;
  cin>>n;
  priority_queue<long long>q;
  for(long long i=0;i<m;i++)
  {
  	long long el;
  	cin>>el;
  	q.push(el);
  }
  while(n--)
  {
  	count=count+q.top();
  	temp=q.top();
  	q.pop();
  	temp--;
  	q.push(temp);
  }
  cout<<count;
  return 0;
}
