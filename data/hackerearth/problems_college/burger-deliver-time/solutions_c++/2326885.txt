#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n;
	cin>>n;
	int i,j;
	int a[n],b[n],c[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		cin>>b[i];
		c[i]=a[i]+b[i];
	}
	sort(c,c+n);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(a[j]+b[j]==c[i] && c[i]!=0)
			{
				a[j]=0;
				b[j]=0;
				cout<<j+1<<" ";
				break;
			}
		}
	}
	return 0;
}
