import math 
for index in range(input()):
    count = 0
    a = map(int,raw_input().split(' '))
    for sindex in range(1,int(math.sqrt(a[0])+1)):
        n2 = a[0]/sindex
        if sindex*n2 == a[0]:
            if n2 <= a[1]:
                count+=1
            if n2 <= a[2] and n2 != sindex:
                count+=1
    print count
