#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%d\n", x)
#define FOR(i, n) for(int i=0;i<n;i++)
#define pp pair<int, int>

#define MAX 1000001
#define MOD 1000000007LL

int m;
bool isSet[51][2001][101];
LL dp[51][2001][101];

int dfs(int n, int k, int total) {
	if(k<0) return 0;

	if(isSet[n][k][total]) return dp[n][k][total];

	isSet[n][k][total] = true;
	if(n) {
		LL result = 0;
		for(int i=0;i<=9;i++) {
			result = (result + dfs(n-1, k-i*i, ((total*10) + i)%m)) % MOD;
		}
		return dp[n][k][total] = result;
	} 
	if(k || total) return dp[n][k][total] = 0;
	return dp[n][k][total] = 1;
}


int func(int n, int k) {
	LL result = 0;
	for(int i=1;i<=9;i++) {
		result = (result + dfs(n-1, k-i*i, (i)%m)) % MOD;
	}
	return result;
}

int main() {

	int n, k;

	sc3(n, m, k);

	pr1(func(n, k));

	return 0;
}
