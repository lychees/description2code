#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll    long long int
#define all(c) c.begin(),c.end()
#define tr(c,i) for(typeof((c).begin()) i = (c).begin(); i != (c).end(); i++)
#define scan(x) scanf("%d", &x)
#define scanl(x) scanf("%lld", &x)
#define print(a) printf("%d\n",a)
#define printl(a) printf("%lld\n",a)
#define pb push_back
#define maxn 10000005
ll a[maxn],cum[maxn],ans[maxn];
// 1.37am
ll sum1=0,sum2=0,k;
ll getsum2(ll x)
{
    sum2=(sum2+(((k*2-1)%mod)*(ll)a[x+k-2])%mod)%mod;
    sum2=(sum2-2*(cum[k+x-3]-cum[x-2])-a[x-2])%mod;
    return 0;
}
int main()
{
    ll i,j,n,t;
    cin>>n>>k;
    for(i=1;i<=n;i++){cin>>a[i];cum[i]=(cum[i-1]+a[i])%mod;}
    for(i=1;i<=k;i++)sum1=(sum1+((((ll)i*i)%mod)*a[i])%mod)%mod;
    ans[1]=sum1;
    for(i=1;i<=k;i++)sum2=(sum2+(ll)a[i]*(2*i-1))%mod;
    ans[2]=(sum1-sum2+(((ll)k*k)%mod*a[k+1])%mod)%mod;
    while(ans[2]<0)ans[2]=ans[2]+mod;
    sum1=ans[2];
    for(i=3;i<=n-k+1;i++)
    {
        getsum2(i);
        ans[i]=(sum1-sum2+((((ll)k*k)%mod)*a[k+i-1])%mod)%mod;
        while(ans[i]<0)ans[i]=ans[i]+mod;
        sum1=ans[i];
    }
    for(i=1;i<=n-k+1;i++)cout<<ans[i]<<" ";
    return 0;
}
