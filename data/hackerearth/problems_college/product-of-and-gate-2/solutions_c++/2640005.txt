#include <cmath>
#include <iostream>
#include <algorithm>
using namespace std;

typedef long long ll;

ll t, a, b;

string bits( ll i ) {
	
	string s;
	
	while( i > 0 ) {
		s.push_back( i%2+'0' );
		i /= 2;
	}
	while( s.size() < 32 )
		s.push_back( '0' );
	
	
	reverse( s.begin(), s.end());
	return s;
}

int main()
{
    cin >> t;
    
    while( t-- ) {
    	cin >> a >> b;
    	
    	string as = bits(a);
    	string bs = bits(b);
    	
    	ll i = 0, c = 0;
    	for( i = 0; i < 32; i++ ) {
    		if( as[i] == '1' && bs[i] == '1' )
    			c += pow( 2, 31-i);
    		else if( bs[i] == '1' )
    			break;
    	}

		cout << c << endl;
    }
    
    return 0;
}
