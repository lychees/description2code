#include <iostream>
using namespace std;

int A[2][101];
int profit[101];


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       int n,amt;
       cin>>n>>amt;
       for(int i=0;i<n;i++)
       {
           cin>>A[0][i]>>A[1][i];
           profit[i] = A[0][i] - A[1][i];
       }
       int dp[101][10002]={0};
       for(int i=0;i<=n;i++)
       {
           for(int j=0;j<=amt;j++)
           {
               if(i==0 || j==0)
                    dp[i][j] = 0;
                else if((A[1][i-1])<=j)
                    dp[i][j] = max(dp[i-1][j],dp[i-1][j-A[1][i-1]]+profit[i-1]);
                else
                    dp[i][j] = dp[i-1][j];
                //cout<<dp[i][j]<<" ";
           }
           //cout<<endl;
       }
       int i,j,maxProfit;
        i = n;
    	j = amt;
    	maxProfit = dp[n][amt];
    	while (j >= 0 && maxProfit == dp[i][j])
    		j -= 1;
    	j += 1;
       cout<<j<<" "<<maxProfit+amt<<endl;
       
    }
    return 0;
}
