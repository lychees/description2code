import math


def nCr(n, r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)


if __name__ == '__main__':
    line = raw_input()
    line = line.split(' ')
    n = int(line[0])
    r = int(line[1])
    if n < r:
        print -1
    else:
        print nCr(n-1, r-1) % 10000007
