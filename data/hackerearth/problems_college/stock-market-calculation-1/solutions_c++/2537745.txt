#include <iostream>
using namespace std;

long long calcProfit(int *arr,int n)
{
	int max = arr[n-1];
	//arr2[n-1] = 0;
	long long sum = 0;
	for(int i=n-2;i>=0;i--)
	{
		if(arr[i] < max)	sum+= (max - arr[i]);
		else 
		{
			max = arr[i];
			//arr2[i] = 0;
		}
		
	}
	return sum;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,arr[50000];
    long long profit;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	cin>>n;
    	for(int j=0;j<n;j++)
    	{
    		cin>>arr[j];
    	}
    	profit = calcProfit(arr,n);
    	cout<<profit<<endl;
    }
    return 0;
}
