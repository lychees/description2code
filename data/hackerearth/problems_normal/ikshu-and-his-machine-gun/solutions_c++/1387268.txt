#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	vector<int> val(t);
	for(int i=0;i<t;i++)
	  cin>>val[i];
	  
	  sort(val.begin(),val.end());
	  int s=0;
	  for(int i=0;i<t;i++)
	    { val[i]-=s;
	      cout<<val[i]<<" ";
	      s++;
	    }
	    cout<<endl;
	return 0;
}
