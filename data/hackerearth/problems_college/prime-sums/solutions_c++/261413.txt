#include <iostream>
#include <vector>
using namespace std;

int main()
{
    long n;
    cin>>n;
    vector<bool>primes(n,true);
    primes[1] = false;
    for(int a = 2; a*a < primes.size(); ++a){
        if(primes[a]==true){
            for(int b = 0; a*a+b*a < primes.size(); ++b){
                    primes[a*a+b*a] = false;
            }
        }
    }
    long long total = 0;
    for(int c = 1; c < primes.size(); c++){
        if(primes[c] == true)
            total +=c;
    }
    cout << total;

return 0;
}
