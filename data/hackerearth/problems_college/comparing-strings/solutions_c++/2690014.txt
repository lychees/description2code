#include<bits/stdc++.h>
using namespace std;
int main()
{
	unsigned long int t,n,i,j,len1,len2,k;
	string a,b;
	cin>>t;
	cin.ignore();
	for(i=0;i<t;i++)
	{
		int flag=0;
		getline(cin,a);
		getline(cin,b);
		len1=a.size();
		len2=b.size();
		for(j=0;j<len1;j++)
		{
			for(k=0;k<len2;k++)
			{
				if(a.at(j)==b.at(k))
				{
					flag=1;
					break;
				}
			}
				if(flag==1)
					break;
		}
		if(flag==1)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
