#include<iostream>
#include<algorithm>
#include<vector>
#include<climits>
#include<string.h>
#include<queue>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int x;
        cin>>x;
        int arr[500] = {0};
        arr[0] = 1;
        for(int i=2; i<=x; i++)
        {
            int carry = 0;
            for(int j=0; j<500; j++)
            {
                arr[j] = arr[j] * i + carry;
                carry = arr[j]/10;
                arr[j] = arr[j]%10;
            }
        }
        int done = 0;
        for(int i=499; i>=0; i--)
        {
            if(arr[i] == 4 || arr[i] == 7) done++;
        }
        cout<<done<<endl;
    }
    return 0;
}
