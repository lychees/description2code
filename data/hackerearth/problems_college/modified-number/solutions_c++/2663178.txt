    #include <cmath>
    #include <cstdio>
    #include <vector>
    #include <iostream>
    #include <algorithm>
    using namespace std;
    bool isKap(int n) {
        int d = 0;
        int temp = n;
        
        while (temp > 0) {
            temp /= 10;
            d++;
        }
        
        int lhs = (long long) n*n / (long long) pow(10, d);
        int rhs = (long long) n*n % (long long) pow(10, d);
        
        return lhs + rhs == n;
    }
    int main() {
        int p, q;
        bool validRange = false;
        cin >> p >> q;
        
        for (int n=p; n<=q; n++) {
            if (isKap(n)) {
                cout << n << " ";
                validRange = true;
            }
        }
        
        if (!validRange)
            cout << "INVALID RANGE" << endl;
        return 0;
    }
