#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--!=0){
    	int n;
    	cin>>n;
    	int *a=new int[n+1];
    	a[0]=0;
    	a[1]=1;
    	for(int i=2;i<=n;i++){
    		if(i<4){
    			if(a[i-1]==1)
    			a[i]=0;
    			else{
    				a[i]=1;
    			}
    		}else{
    			if(a[i-1]==1 && a[i-4]==1)
    			a[i]=0;
    			else
    			a[i]=1;
    		}
    	}
    	if(a[n]==1)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
