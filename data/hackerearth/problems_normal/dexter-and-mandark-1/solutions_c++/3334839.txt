//GAUSS
#include<bits/stdc++.h>
using namespace std;

#define mod	1000000007
int dp[1004][1004]={0};
int main() {
	int t; cin>>t;
	while(t--) {
		int n, i, j, k;
		cin>>n>>k;
		memset(dp, 0, sizeof dp);
		dp[0][0] = 1;
		
		int f = 0, s = 0;
		for(i = 1; i <= n; i++) {
			for(j = 0; j <= n; j++) {
				f = i-1-j;
				if(f < 0) continue;
				f++;
				if(f >= k*j) dp[i][j] += (dp[i-1][j])%mod;
				f--;
				if(f >= k*(j+1)) dp[i][j+1] += (dp[i-1][j])%mod;
			}
		}
		long long int ans = 0;
		for(i = 0; i <= n; i++) ans = (ans+ dp[n][i])%mod;
		if(!(n&1) && k==1)
			ans=(ans-dp[n][n/2]+mod)%mod;
		cout<<ans<<endl;
	}

	return 0;
}
