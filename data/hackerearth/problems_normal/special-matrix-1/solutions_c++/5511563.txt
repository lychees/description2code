#include<iostream>
using namespace std;
int main(){
	int t,n,m,i,j,k,l;
	cin>>t;
	string s;
	while(t--){
		cin>>n;
		for(i=0,j=-1,k=-1;i<n;i++){
			cin>>s;
			if(j==-1&&k==-1){
				for(l=0;l<s.length();l++){
					if(s[l]=='*')
					j=i+1,k=l+1;
				}
			}
		}
		m=n/2+1;
		if(j>m&&k>m)
		cout<<(j-m+k-m);
		else if(j>m&&k<=m)
		cout<<(j-m+m-k);
		else if(j<=m&&k>m)
		cout<<(m-j+k-m);
		else
		cout<<(2*m-j-k);
		cout<<endl;
	}
	return 0;
}
