#include <iostream>
#include <cmath>
#include<iomanip>
#include<stdio.h>
using namespace std;
	double pi=3.1415;

int main()
{
	int t,i;
	double a[1001],b[1001],c[1001];
	double s[1001],areat[1001], base[1001], height[1001],areac[1001];
	cin>>t;
	for(i=1;i<=t;i++)
	{
		cin>>a[i]>>b[i]>>c[i];

		base[i]=a[i];				//find greatest of the 3
		if(base[i]<b[i])
		{
			base[i]=b[i];
		}
		if(base[i]<c[i])
		{
			base[i]=c[i];
		}

	}

	for(i=1;i<=t;i++)
	{
		s[i]=(a[i]+b[i]+c[i])/2;
		areat[i]=sqrt(s[i]*(s[i]-a[i])*(s[i]-b[i])*(s[i]-c[i]));
		height[i]=2*(areat[i]/base[i]);
	}

	for(i=1;i<=t;i++)
	{
		double diam;
	
		{
			diam=(a[i]*b[i]*c[i])/(2*areat[i]);
			areac[i]=pi*(diam/2)*(diam/2);
	
		}
			
			printf("%.4f \n",areac[i]);
		}

	return 0;
}
