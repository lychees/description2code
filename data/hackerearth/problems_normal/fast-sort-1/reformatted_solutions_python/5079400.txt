t = int(raw_input())

for i in range(0, t):
    alphabet = list(raw_input())
    word = list(raw_input())
    translation = []
    translation_table = [0 for x in range(0, 26)]
    for j in range(0, 26):
        translation_table[ord(alphabet[j]) - ord('a')] = j

    for letter in word:
        translation.append(translation_table[ord(letter) - ord('a')])

    translation = sorted([chr(x + ord('a')) for x in translation])

    translation = [alphabet[ord(letter) - ord('a')] for letter in translation]

    print ''.join(translation)
