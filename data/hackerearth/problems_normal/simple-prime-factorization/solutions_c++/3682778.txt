#include <iostream>
#include <cmath>

using namespace std;

void factor(int n)
{
	int cnt=0,i,a=n;
	while(n%2==0)
	{
		n=n/2;
		cnt++;
	}
	cout<<"2^"<<cnt;
	for(i=3;i<=a/2;i=i+2)
    {
    	cnt=0;
        while (n%i == 0)
        {
            
            n = n/i;
            cnt++;
        }
        if(cnt!=0)
        cout<<"*"<<i<<"^"<<cnt;
	}
}
int main()
{
   int t,i,n;

   {

}
