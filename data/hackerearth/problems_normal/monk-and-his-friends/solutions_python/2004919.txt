'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
T = int(raw_input())
for i in range(T):
	first = map(int,raw_input().split())
	N = first[0]
	M = first[1]
	second = map(int,raw_input().split())
	classroom = {}
	for x in range(N):
		if second[x] not in classroom:
			classroom[second[x]] = 1
		else:
			classroom[second[x]] += 1
	for y in range(M):
		if second[y+N] in classroom:
			classroom[second[y+N]] += 1
			print "YES"
		else:
			classroom[second[y+N]] = 1
			print "NO"
