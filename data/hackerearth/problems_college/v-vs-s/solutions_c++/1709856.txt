#include<bits/stdc++.h>
using namespace std;

int main() {
  int tt;
  scanf("%d",&tt);
  while(tt--) {
    string s;
    cin>>s;
    int len=s.length();
    int ans;
    bool flag=false;
    for(int i=0;i<len;i++) {
      if(s[i]=='1'||s[i]=='2') {
        flag=true;
        break;
      }
    }
    if(flag==false) {
      puts("0");
      continue;
    }
    int res1=0,res2=0;
    if(len>=1) {
      res1=s[len-1]-'0';
    }
    if(len>=2) {
      res2=s[len-2]-'0';
    }
    ans=(res1*1+res2*3)%9;
    if(ans<=0) {
      ans+=9;
    }
    printf("%d\012",ans);
  }
  return 0;
}
