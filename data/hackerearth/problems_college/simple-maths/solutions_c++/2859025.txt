#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t, n, a, b, ans, da, db, dab, lcm;
    scanf("%lld", &t);
    while(t--)
    {
    	scanf("%lld%lld%lld", &n, &a, &b);
    	if(a == 0 || b == 0)
    	{
    		printf("-1\n");
    		continue;
    	}
    	lcm = a * b / __gcd(a, b);
    	dab = n / lcm;
    	da = n / a;
    	db = n / b;
    	ans = da + db - dab;
    	printf("%lld\n", ans);
    }
    return 0;
}
