#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		scanf("%d",&n);
		int c=0,ans=0;
		bool arr[1000002]={0};
		for(i=0;i<n;i++)
		{
			int s,r;
			scanf("%d%d",&s,&r);
			if(r==1)
			{
				if(arr[s]==0)
				{
				c++;
				ans=max(c,ans);
				arr[s]=1;
				}
			}
			else
			{
				c=0;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
