#include <iostream>
#include<string>
using namespace std;
int main()
{
	int n,i,j;
   string str;
   int hash[109][26]={0};
   cin>>n;
   for(i=0;i<n;i++)
   {
   cin>>str;
   int j=0;
   for(j=0;j<str.size();j++)
   	hash[i][str[j]-'a']++;
  }
  int count1=0;
  for(i=0;i<26;i++)
  {
  	int count=0;
  	for(j=0;j<n;j++)
  	{
  		if(hash[j][i])
  		count++;
  	}
  	if(count==n)
  	count1++;
  }
 cout<<count1;
    return 0;
}
