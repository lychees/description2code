// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

int isP[100005];
vector <int> p;

void sieve()
{
    int i, j;
    for (i = 2; i <= 100000; i++) {
        if (!isP[i]) {
            p.push_back(i);
            for (j = 2; i*j <= 100000; j++)
                isP[i*j] = 1;
        }
    }
}

void calculate(vector <int> &tmp, map <int, int> &m, int ans, int i, vector <int> &v)
{
    if (i == v.size()) {
        map <int, int> :: iterator it;
        it = m.find(ans);
        if (it == m.end()) {
            m[ans] = 1;
            tmp.push_back(ans);
        }
        return;
    }
    calculate(tmp, m, ans, i+1, v);
    calculate(tmp, m, ans*v[i], i+1, v);
}

int main()
{
    //freopen("in10.txt", "r", stdin);
    //freopen("out10.txt", "w", stdout);
    sieve();
    int x, y, z, i, n;
    scanf("%d%d", &x, &y);
    z = __gcd(x,y);
    vector <int> v;
    for (i = 0; i < p.size(); i++) {
        while (z % p[i] == 0) {
            v.push_back(p[i]);
            z /= p[i];
        }
    }
    if (z > 1)
        v.push_back(z);
    vector <int> tmp;
    map <int, int> m;
    calculate(tmp, m, 1, 0, v);
    sort(tmp.begin(), tmp.end());
    scanf("%d", &n);
    while (n--) {
        scanf("%d%d", &x, &y);
        z = upper_bound(tmp.begin(), tmp.end(), y) - tmp.begin() - 1;
        if (0 <= z && z < tmp.size() && tmp[z] >= x)
            printf("%d\n", tmp[z]);
        else
            printf("-1\n");
    }
    return 0;
}
