#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;



int main()
{
	int t;
	int n,a,b;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int sum=0;
		vector<int> vec;
		vec.resize(0);
		for(int j=0;j<7;j++)
		{
			cin>>n;
			for(int k=0;k<n;k++)
			{
				cin>>a>>b;
				vec.push_back(b-a);
			}
			
		}
		sort(vec.begin(),vec.end());
		n=vec.size();
		int qw=(n*3)/4;
		while(qw*4<n*3){qw++;}
		//cout<<qw<<endl;
		for(int k=0;k<qw;k++)sum+=vec[k];
		cout<<86400*7-sum<<endl;
	}
}
