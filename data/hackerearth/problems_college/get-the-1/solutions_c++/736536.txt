#include<iostream>
using namespace std;
int main()
{
 unsigned long long n,s=0,temp;
 int rem;
 int t;
 cin>>t;
 for(int m=0;m<t;m++)
 {
 s=0;
 cin>>n;
 temp=n;
 while(s!=1 && s!=2 && s!=3 && s!=4 && s!=5 && s!= 6 && s!=8 && s!=9)
 {
  s=0;
  while(temp!=0)
  {
   rem=temp%10;
   s+=(rem*rem);
  
   temp=temp/10;
  }
  temp=s;
  
 }
 if(s==1)
  cout<<"Case #"<<m+1<<":"<<"YES"<<endl;
 else
  cout<<"Case #"<<m+1<<":"<<"NO"<<endl;
}
 return 0;
}
