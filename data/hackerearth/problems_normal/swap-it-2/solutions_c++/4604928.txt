#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
bool comp(pair<l1,l1> p1,pair<l1,l1> p2)
{
	return p1.first<p2.first;
}
int main()
{
	ios::sync_with_stdio(false);
	l1 i,j,n,t,k,f,s,ind,r,temp;
	cin>>t;
	while(t--)
	{
		cin>>n>>s;
		f=0;
		l1 num[n+9];
		for(i=0;i<n;i++)
		cin>>num[i];
			vector<pair<l1,l1> > vec;
		for(i=0;i<n;i++)
		{
			f=0;
			ind=-1;
			if(s==0)
			break;
			vec.clear();
			for(j=i+1;j<n;j++)
			{
				if(num[j]<num[i])
				{
					f=1;
					vec.push_back(make_pair(num[j],j-i));
				}
			}
			if(f==1)
			stable_sort(vec.begin(),vec.end(),comp);
		//	for(int h=0;h<vec.size();h++)
		//	cout<<vec[h].first<<" "<<vec[h].second<<endl;
			
			if(f==1)
			{
			for(k=0;k<vec.size();k++)
			{
				if(vec[k].second<=s)
				{
					ind=vec[k].second;
					break;
				}
			}
			//cout<<ind<<endl;
			if(ind!=-1)
			{
			s=s-(ind);
			for(r=ind+i;r>i;r--)
			{
				temp=num[r];
				num[r]=num[r-1];
				num[r-1]=temp;
			}
			}
			}
			
		}
	
for(i=0;i<n;i++)
cout<<num[i]<<" ";
	cout<<endl;
}
	return 0;
	
}
