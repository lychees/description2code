#include<bits/stdc++.h>

using namespace std;

#define MOD 1000000007

long long mod(long long a, long long b, long long m) {
    long long r = 1;
    while (b != 0) {
        if (b & 1 != 0) {
            r *= a;
            r %= m;
        }
        b /= 2;;
        a *= a;
        a %= m;
    }
    return r;
}
int main() {
	long long derange[101];
	long long combo[101][101];
	derange[0] = 1;
    derange[1] = 0;
    for (int i = 2; i <= 100; i++) {
        derange[i] = ((i-1) * (derange[i-1]+derange[i-2])) % MOD;
    }
    for (int i = 0; i <= 100; i++) {
        for (int j = 0; j <= i; j++) {
            if (i == j || j == 0)
                combo[i][j] = 1;
            else
                combo[i][j] = (combo[i-1][j] + combo[i-1][j-1]) % MOD;
        }
    }
	
    int N, L, R;
    cin >> N >> L >> R;
    long long num = 0;
    long long det = 0;
    for (int i = 0; i <= N; i++) {
        long long ways = (derange[i] * combo[N][i]) % MOD;
        if (L <= i && i <= R) {
            num += ways;
            num %= MOD;
        }
        det += ways;
        det %= MOD;
    }
    long long ans = (num * mod(det, MOD-2, MOD)) % MOD;
    
    cout << ans << endl;
    
    return 0;
}
