#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;

typedef long long LL;
typedef vector<int> VI;

#define REP(i,n) for(int i=0, i##_len=(n); i<i##_len; ++i)
#define EACH(i,c) for(__typeof((c).begin()) i=(c).begin(),i##_end=(c).end();i!=i##_end;++i)

int T, N;
LL A[55], L[55];
int main() {
    scanf("%d", &T);
    REP ($, T) {
	scanf("%d", &N);
	REP (i, N) scanf("%lld", A+i);
	REP (i, N) scanf("%lld", L+i);
	REP (i, N) REP (j, N-1) {
	    if (A[j] > A[j+1]) {
		swap(A[j], A[j+1]);
		swap(L[j], L[j+1]);
	    }
	}
	LL ans = 0;
	for (int i=N/2-1; i>=0; i--) {
	    LL d = A[i+1] - (A[i]+L[i]);
	    ans += d;
	    A[i] += d;
	}
	for (int i=N/2+1; i<N; i++) {
	    LL d = A[i] - (A[i-1]+L[i-1]);
	    ans += d;
	    A[i] -= d;
	}
	printf("%lld\n", ans);
    }
    return 0;
}
