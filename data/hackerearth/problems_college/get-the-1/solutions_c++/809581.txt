long long int sqsum(long long int x)
{
	long long int y=x;int count=1;long long int ans=0;int i;
	while(y/=10)
	count++;
	
	for(i=0;i<count;i++)
	{
		ans+=pow((x%10),2);
		x/=10;
	
	}
	return ans;
	
}
