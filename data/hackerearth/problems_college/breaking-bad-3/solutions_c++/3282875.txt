#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i;
    long long ans;
    long arr[200005];
    cin>>t;
    while(t--)
    {	ans=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>arr[i];
		sort(arr,arr+n);
    	for(i=0;i<n;i+=2)
    	{
    		ans+=(arr[i]*arr[i+1]);
    	}
		cout<<ans<<endl;
    }
    return 0;
}
