t=input()
while t:
    a=map(str, raw_input("").split())
    base=int(a[0])
    s=a[1]
    k=1
    ans=0
    while len(s)>0:
        l=s[-1]
        if ord(l)<59:
            l=ord(l)-48
        else:
            l=ord(l)-87
        ans+=k*l
        k*=base
        s=s[:-1]
    x=str(ans)
    print sum(int(i) for i in x)
    t-=1
