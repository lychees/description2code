#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <map>
#define ll long long int

using namespace std;

int main()
{
    ll n;
    ll i, j;
    cin >> n;

    for (i = 0; i < n; i++) {
        cout << "*";
    }
    cout << endl;

    for (j = n - 2; j > 0; j--) {

        for (i = 0; i < j; i++) {
            cout << " ";
        }
        cout << "*\n";
    }




    for (i = 0; i < n; i++) {
        cout << "*";
    }
    cout << endl;
}
