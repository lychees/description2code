bool prime(long long int x){
    
    if(x==2)
    return true;
    
    else if (x==3)
    return true;
    
    else{
        for(long long int i=3;i<=x/2;i+=2)
        {
            if(x%i==0)
            return false;
        }
    }
    
    return true;
    
}

}
