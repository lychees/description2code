T = int(raw_input())
for t in range(T):
    s1 = raw_input()
    s2 = raw_input()
    s1 = list(set(s1))
    s2 = list(set(s2))
    s =[]
    p=[]
    res= False
    if len(s1) < len(s2):
        s = s1
        p=s2
    else:
        s= s2
        p = s1
    for x in s:
        if x in p:
            res=True
            break
    if res:
        print('Yes')
    else:
        print('No')
