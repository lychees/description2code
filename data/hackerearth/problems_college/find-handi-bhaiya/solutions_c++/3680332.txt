#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long int
#define loop(i,n) for(i=0;i<n;i++)
int main(){
    ll t,a,b,c,i,j,k,l1,l2,l3;
    cin>>t;
    while(t--){
        scanf("%llu %llu %llu",&a,&b,&c);
        if(a==0 || b==0 || c==0){
            cout<<"NO"<<endl;
        }
        else{
        	l1=a+b;
        	l2=b+c;
        	l3=c+a;
        if(l1>c && l2>a && l3>b){
            cout<<"YES. ";
            if(a==b  && b==c ){
                cout<<"It is a equilateral triangle."<<endl;
            }
            else if((a==b && c!=a) || (b==c && a!=b) || (a==c && c!=b)){
                cout<<"It is a isosceles triangle."<<endl;
            }
            else{
                cout<<"It is a scalene triangle."<<endl;
            }
        }
        else{
            cout<<"NO"<<endl;
        }
        }
    }
    return 0;
}
