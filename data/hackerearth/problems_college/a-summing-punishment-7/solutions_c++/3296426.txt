#include <iostream>

using namespace std;
int main()
{
	int n;
	cin >> n;
	
	int a[n];
	
	for(int i = 0; i < n; i++) {
		cin >> a[i];
	}
	
	int t;
	cin >> t;
	while(t--) {
		int aa, b;
		cin >> aa >> b;
		int sum = 0;
		for(int i = aa-1; i <= b-1; i++) {
			sum = sum + a[i];
		}
		
		cout << sum << endl;
	}
    //printf("Hello World!\n");
    return 0;
}
