#define N 1000005
using namespace std;
int t,n;
ll a[N];
int check(ll val) {
    ll last=0;
    FOR(i,1,n) {
        if(a[i]>last) {
            last=max(last+1,a[i]-val);
        }
        else {
            ll lastt=min(last+1,a[i]+val);
            if(lastt<=last) {
                return 0;
            }
            last=lastt;
        }
    }
    return 1;
}
int main() {
    scanf("%d",&t);
    while(t--) {
        scanf("%d",&n);
        FOR(i,1,n) {
            scanf("%lld",&a[i]);
        }
        ll bas=0,son=inf;
        while(bas<son) {
            if(check(orta)) {
                son=orta;
            }
            else {
                bas=orta+1;
            }
        }
        printf("%lld\n",bas);
    }
}
