#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	scanf("%lld",&t);
	
	for(;t>0;t--)
	{
		long long int n;
		scanf("%lld",&n);
		long long int a[n];
		
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		
		sort(a,a+n);
		long long int sum=n;
		
		int i=0,c=1,f=0;
		while(i<n-1)
		{
			if(a[i]==a[i+1])
			{
				c++;
				f=0;
			}
			else
			{
				sum+=(c*(c-1))/2;
				c=1;
				f=1;
			}
			
			i++;
		}	
		
		if(f==0)
		{
			sum+=(c*(c-1))/2;
		}
		printf("%lld\n",sum);
	}
	
	return 0;
}
