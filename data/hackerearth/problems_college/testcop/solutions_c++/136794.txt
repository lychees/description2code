#include <iostream>
#include<stdio.h>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        gcd(b,a%b);
}
int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int N,i;
		scanf("%d",&N);
		int a[N];
		for(i=0;i<N;i++)
			scanf("%d",&a[i]);
		if(N==1)
		{
			if(a[0]==1)
				printf("YES\n");
			else
				printf("NO\n");
		}
		else
		{
			int G=gcd(a[0],a[1]);
			for(i=2;i<N;i++)
				G=gcd(G,a[i]);
			if(G==1)
				printf("YES\n");
			else
				printf("NO\n");
		}
	}
	return 0;
}
