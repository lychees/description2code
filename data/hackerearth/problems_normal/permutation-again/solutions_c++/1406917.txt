#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    while(T--)
    {
    	unsigned long long int N;
    	cin>>N;
    	
    	if (N == 1){
    		cout<<"1"<<endl;
    		continue;
    	}
    	
    	
    	unsigned long long int answer = (N * N)/2 - 1;
    	
    	cout<<answer<<endl;
    	
    }
}
