#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int num_nodes;
struct node
{
	struct node *child[26];
};


struct node *insert(struct node *root,char *str);
int main() 
{
	int n,i;
	char str[31];
	
	struct node *root;
	root=new node;
	for(i=0;i<26;i++)
		root->child[i]=NULL;
	
	num_nodes=1;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>str;
		root=insert(root,str);
	}
	
	cout<<num_nodes<<"\n";
	return 0;
}

struct node *insert(struct node *root,char *str)
{
	int i;
	if(str[0]=='\0')
		return root;
	
	struct node *child;
	child=root->child[str[0]-'a'];
	
	if(child==NULL)
	{
//printf("Create for %c\n",str[0]);	
		child=new node;
		num_nodes++;
		for(i=0;i<26;i++)
			child->child[i]=NULL;
		
		child=insert(child,str+1);
	
	}
	else
	{
//printf("not Create for %c\n",str[0]);	

		child=insert(child,str+1);
		
	}
	root->child[str[0]-'a']=child;
	return root;
}
