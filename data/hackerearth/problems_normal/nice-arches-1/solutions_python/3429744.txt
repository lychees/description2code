#print 'Hello World!'
M = int(raw_input())
ans = 0
for i in xrange(M):
	word = list()
	word_in = raw_input()
	for j in xrange(len(word_in)):
		if len(word)>0 and word_in[j] == word[-1]:
			word.pop()
		
		else:
			word.append(word_in[j])
	if len(word)==0:
		ans+=1
print ans
