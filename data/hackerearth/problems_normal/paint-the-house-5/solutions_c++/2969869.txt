#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define mod 1000000007

vector < pair <ll, pair <ll, ll> > > v;

int main()
{
    ll n, d, i, x, y, t, mn, mx, j;
    sl(n);
    sl(d);
    for (i = 0; i < n; ++i) {
        sl(t);
        sl(x);
        sl(y);
        v.pb(mp(t, mp(x,y)));
    }
    sort(v.begin(), v.end());
    ll cost = 0, time = 0, speed = 0, done = 0;
    for (i = 0; i < n; ) {
        done += (v[i].first - time) * speed;
        if (done >= d)
            break;
        j = i;
        mx = 0;
        time = v[j].first;
        while (j < n && v[i].first == v[j].first) {
            if (v[j].second.second > mx) {
                mx = v[j].second.second;
                mn = v[j].second.first;
            }
            else if (v[j].second.second == mx)
                mn = min(mn, v[j].second.first);
            ++j;
        }
        i = j;
        if (speed < mx) {
            cost += mn;
            speed = mx;
        }
    }
    printf("%lld\n", cost);
    return 0;
}
