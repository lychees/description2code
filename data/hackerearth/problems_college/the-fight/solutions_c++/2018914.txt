#include <bits/stdc++.h>
using namespace std;
long long int a[1000003],b[1000003]={0};
int main()
{
   	long long int l,r,k,m,n;
   	scanf("%lld",&n);
   	//long long int a[1000000],b[100000]={0};
   	for(int i=1;i<=n;i++)
   	{
   		scanf("%lld",&a[i]);
   	}
   	scanf("%lld",&m);
   	while(m--)
   	{
   		scanf("%lld %lld %lld",&l,&r,&k);
   		b[l]+=k;
   		b[r+1]-=k;
   	}
   	long long int value=0;
   	for(int i=1;i<=n;i++)
   	{
   		value+=b[i];
   		printf("%lld\n",a[i]+value);
   	}
}
