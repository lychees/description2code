#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int food,gold,s,p,c,power_lal,power_faj=-1;
		scanf("%d%d%d%d%d%d",&food,&gold,&s,&p,&c,&power_lal);
		int sams,pals,champs;
		/*Samurai is: 100 food, 0 gold.
The cost of a Paladin is: 125 food, 50 gold.
The cost of a Champion is: 50 food, 100 gold.*/
		for(sams=0;sams*100<=food;sams++){
			for(pals=0;pals*125+sams*100<=food&&pals*50<=gold;pals++){
				int foodleft=food-pals*125-sams*100;
				int goldleft=gold-pals*50;
				champs=min(foodleft/50,goldleft/100);
				power_faj=max(power_faj,sams*s+pals*p+champs*c);
			}
		}
		if(power_faj>power_lal){
			cout<<power_faj-power_lal<<endl;
		}
		else cout<<-1<<endl;
	}
}
