#include <bits/stdc++.h>
using namespace std;
int good=0,G,H,great=0;
struct node
{
	int s;
	int p;
};
struct node news[100005],v[100005];
bool compare(const node &a,const node &b)
{
	if(a.s==b.s)
	{
		return a.p<b.p;
	}
	else
	{
		return a.s<b.s;
	}
}
int main()
{
    ios::sync_with_stdio(false);
    int t,m,n;
    cin>>t;
    while(t--)
    {
    	good=0;great=0;
    	cin>>m>>n;
    	for(int i=0;i<m;i++)
    	{
    		cin>>news[i].s>>news[i].p;
    	}
    	
    	for(int i=0;i<n;i++)
    	{
    		cin>>v[i].s>>v[i].p;
    	}
    	
    	cin>>G>>H;
    	sort(news,news+m,compare);
    	sort(v,v+n,compare);
    	int i=0,j=0;
    	while(i<m && j<n)
    	{
    		if(news[i].s==v[j].s)
    		{
    			good++;
    			if(news[i].p==v[j].p)
    			{
    				great++;
    			}
    			j++;
    			i++;
    		}
    		else if(news[i].s>v[j].s)
    		{
    			while(news[i].s>v[j].s)
    			{
    				j++;
    			}
    			i++;
    		}
    		else
    		{
    			i++;
    		}
    	}
    	
    	if(good>=G && great>=H)
    	{
    		cout<<"Great"<<endl;
    	}
    	else if(good>=G)
    	{
    		cout<<"Good"<<endl;
    	}
    	else
    	{
    		cout<<":("<<endl;
    	}
    }
    
    return 0;
}
