#include <iostream>
#include<limits.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    int c=n;
    while(c)
    {
    	int min=INT_MAX;
    	c=n;
    	for(int i=0;i<n;i++)
    	{
    		if(min>a[i]&&a[i]!=0)
    		min=a[i];
    	}
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]!=0)
    		a[i]=a[i]-min;
    		else
    		c--;
    	}
    	if(c!=0)
    	cout<<c<<endl;
    }
    return 0;
}
