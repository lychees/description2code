#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007LL
#define MAXN 100010

ll A[MAXN];
map<string,int> mp;

int main()
{
	mp["000"] = 0;
	mp["001"] = 1;
	mp["010"] = 2;
	mp["011"] = 3;
	mp["100"] = 4;
	mp["101"] = 5;
	mp["110"] = 6;
	mp["111"] = 7;
	string s;
	cin >> s;
	bool dot = false;
	for(int i=0;i<s.size();++i)
		if(s[i]=='.')
			dot = true;
	if(!dot)
	{
		while((s.size()%3)!=0)
			s = "0" + s;
		int j = s.size()/3;
		for(int k=0;k<j;++k)
			cout << mp[s.substr(3*k,3)];
		cout << " ";
		for(int k=0;k<j;++k)
			if(mp[s.substr(3*k,3)]==0)
				cout << " ";
			else
				cout << char(mp[s.substr(3*k,3)] - 1 + 'A');
	}
	else
	{
		int idx;
		for(int i=0;i<s.size();++i)
		if(s[i]=='.')
			idx = i;
		string A = s.substr(0,idx), B = s.substr(idx+1);
		while((A.size()%3)!=0)
			A = "0" + A;
		while((B.size()%3)!=0)
			B = B + "0";
		s = A;
		int j = s.size()/3;
		for(int k=0;k<j;++k)
			cout << mp[s.substr(3*k,3)];
		cout << ".";
		s = B;
		j = s.size()/3;
		for(int k=0;k<j;++k)
			cout << mp[s.substr(3*k,3)];
		cout << " ";
		s = A;
		j = s.size()/3;
		for(int k=0;k<j;++k)
			if(mp[s.substr(3*k,3)]==0)
				cout << " ";
			else
				cout << char(mp[s.substr(3*k,3)] - 1 + 'A');
		cout << " ";
		s = B;
		j = s.size()/3;
		for(int k=0;k<j;++k)
			if(mp[s.substr(3*k,3)]==0)
				cout << " ";
			else
				cout << char(mp[s.substr(3*k,3)] - 1 + 'A');
	}
	return 0;
}
