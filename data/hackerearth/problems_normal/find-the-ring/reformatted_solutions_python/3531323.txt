T = int(raw_input())

for i in range(T):
    pos, n = map(int, raw_input().split())
    if n % 2 == 0:
        if n == 0:
            ans = pos
        elif pos == 1:
            ans = 1
        else:
            ans = 0
    else:
        if pos == 1:
            ans = 0
        else:
            ans = 1
    print ans
