#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int ma = 0;
    string sa;
    map<string,int> m;
    for(int i=0;i<n;i++) {
    	string temp;
    	int g;
    	cin >> temp >> g;
    	m[temp] += g;
    	if(m[temp] > ma) {
    		ma = m[temp];
    		sa = temp;
    	}
    }
    cout << sa << endl;
    return 0;
}
