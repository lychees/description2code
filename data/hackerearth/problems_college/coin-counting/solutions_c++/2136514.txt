#include <iostream>
using namespace std;
int main()
{
    int t,i=0;
    cin>>t;
    while(t--)
    {  int amt;
    cin>>amt;
    int a,b,c,ans=0;
       for(a=0;a<=amt;a++)
        {
        	for(b=0;b<=amt;b++)
         {
         	for(c=0;c<=amt;c++)
          {
          	if((a+3*b+5*c)==amt)
          	 ans++;
          }
         }
        }
    cout<<ans<<endl;	
    	i++;
    }
}
