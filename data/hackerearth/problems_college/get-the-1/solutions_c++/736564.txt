#include <iostream>

using namespace std;

int generatesum(int n)
{
    int sum = 0;
    int digit;
    while(n)
    {
        digit = n%10;
        sum  += digit*digit;
        n/=10;
    }
    return sum;
}

int main()
{
    int a[]={4, 16, 37, 58, 89, 145, 42, 20};
    int T,n;
    int idx=0;
    cin>>T;
    while(T--)
    {
        idx++;
        cin>>n;
        int temp = n;
        int flag=0;
        while(1)
        {
            temp = generatesum(temp);
            for(int i=0;i<8;i++)
                if(temp==a[i])
                    {cout<<"Case #"<<idx<<":"<<"NO\n";flag=1;break;}
            if(flag==1)
                break;
            if(temp==1)
                {cout<<"Case #"<<idx<<":"<<"YES\n";break;}
        }
    }
    return 0;
}
