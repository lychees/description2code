#include <iostream>
#include<string.h>
#include<stdio.h>>
using namespace std;

int main()
{
    int t,i,n,j;
    char a[100];
    
    cin>>t;
    for(j=0;j<t;j++){
    	
    	scanf("%s",&a);
    	for(i=0;a[i]!=NULL;i++){
    		n=(int)a[i];
    		
    		if(n>=97&&n<=99)cout<<2;
    		else if(n>=100&&n<=102)cout<<3;
    		else if(n>=103&&n<=105)cout<<4;
    		else if(n>=106&&n<=108)cout<<5;
    		else if(n>=109&&n<=111)cout<<6;
    		else if(n>=112&&n<=115)cout<<7;
    		else if(n>=116&&n<=118)cout<<8;
    		else cout<<9;
    
    	}
    	cout<<endl;
    }
    return 0;
}
