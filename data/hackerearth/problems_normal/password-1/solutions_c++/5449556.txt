#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n; cin >> n;
	vector<string> arr(n);
	
	for(int i=0;i<n;i++)
	{
		cin >> arr[i];
	}
	
	bool flag = false;
	for(int i=0;i<n;i++)
	{
		string s;
		int l = arr[i].length();
		for(int k=l-1;k>=0;k--) s += arr[i][k];

		for(int j = i+1;j<n;j++)
		{
			if(arr[j] == s)
			{
				cout << l << " " << s[(l/2)];
				flag  = true;
				break;
			}
		}
		if(flag) break;
	}
}
