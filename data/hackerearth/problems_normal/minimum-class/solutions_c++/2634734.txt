int solve(int n)
 {
 	int i=2,p=0,l=0;
 	while(n>0)
 	{
 			l=l+1;
 			p=p+1;
 			n=n-i;
 			if(l==2)
 			{
 			  l=0;
 			  i=i+1;
 			}	
    }
    return p;
 }
int main()
{
    int t;
