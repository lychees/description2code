def gcd(a,b):
	t = b
	b = a % b
	if b == 0:
		return t
	else:
		return gcd(t,b)

def isCoprime(a,b,c):
	return gcd(a,b)==1 and gcd(b,c)==1 and gcd(a,c)==1

t=input()
while t>0:
	t-=1
	triples=map(int, raw_input().split())[:3]
	triples.sort()
	a=triples[0]
	b=triples[1]
	c=triples[2]
	if isCoprime(a,b,c) and c*c==(b*b+a*a):
		print "YES"
	else:
		print "NO"
