t = input()
gSize = dict()


def gs(n, m):
    if m < 2:
        return 0
    if (m == n):
        return 1
    if n < m:
        return 0
    if (n, m) in gSize:
        return gSize[(n, m)]
    res = 1
    for i in range(m, n-1):
        res += gs(n-i, i)
    gSize[(n, m)] = res
    return res


for _ in range(t):
    n = input()
    print gs(n, 2)
