#include <iostream>
using namespace std;

int main()
{
    long int t,n,m,x,avg=0,sum=0,ans=0,a;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n>>m>>x;
    	for(int i=0;i<n;++i)
    	{
    		cin>>a;
    		sum=sum+a;
    	}
    	ans=x*(n+1)-sum;
    	if(ans<=0)
    	cout<<"1"<<endl;
    	else
    	if(ans>m)
    	cout<<"Impossible"<<endl;
    	else
    	cout<<ans<<endl;
    }
    return 0;
}
