#include<cstdio>
#include<vector>
#include<map>
#include<algorithm>
 
using namespace std;
 
int main(){
    int N; scanf("%d", &N);
    vector<pair<int, int> > A;
    int s, f;
    for(int n = 0; n<N; n++){
        scanf("%d %d", &s, &f);
        A.push_back(make_pair(s+f, n));
    }
    sort(A.begin(), A.end());
    for(int n = 0; n<N; n++){
        printf("%d ", A[n].second+1);
    }
    return 0;
}
