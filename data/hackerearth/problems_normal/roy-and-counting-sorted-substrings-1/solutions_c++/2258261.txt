#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
string str;
//set<char> s;
long long int number(long long int a,long long int b)
{
		return a-b;
}
int main()
{
    int t;
    long long int n,i,count,k;
    cin>>t;
    while(t--)
    {
    //	s.clear();
    	count=1;
    	cin>>n>>str;
    	k=0;
    	for(i=1;i<n;i++)
    	{
    	//	s.insert(str[i]);
    		if(str[i]>=str[i-1])
    		count=count+1+number(i,k);
    		else
    		{
    			 k=i;
    		//	 cout<<k;
    			count=count+1+number(i,k);
    		}
    	}
    	cout<<count<<"\n";	
    }
    return 0;
}
