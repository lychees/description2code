#include<bits/stdc++.h>
using namespace std;
int main()
{unsigned long long prod;
    priority_queue<long long int> pq;
    int n,z=1;long long a,b,c,d,i;
    cin>>n;
    for(i=0;i<n;i++)
    {
    cin>>a;
    pq.push(a);
    if(z<=2)
    {
        cout<<"-1"<<"\n";
    }
    else if(z>2)
    {b=pq.top();pq.pop();
    c=pq.top();pq.pop();
        d=pq.top();pq.pop();
        pq.push(b);pq.push(c);pq.push(d);
        prod=b*c*d;
        cout<<prod<<"\n";
    }
    ++z;
    }

}
