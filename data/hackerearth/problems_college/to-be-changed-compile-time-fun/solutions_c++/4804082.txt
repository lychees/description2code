#include <iostream>
using namespace std;
void delandinsert(int *ar,int index,int len)
{
	int k;
	for(k=index;k<len;k++)
	ar[k]=ar[k+1];
}
int main()
{
    int n;
    cin>>n;
    int a[100];
    
    int i,j;
    for(i=0;i<n;i++)
    cin>>a[i];
    int flag=n-1;int curr,sum=0,multi=1,prev=1;
    for(j=1;j<flag;j++)
    {
    	prev=1;
    	for(i=1;i<(n-1);i++)
    	{
    		multi=(a[i-1]*a[i+1]);
    		if(multi>prev)
    		{
    			//cout<<"Multi"<<multi<<"\n";
    			curr=i;
    			prev=multi;
    		}
    	}
    	delandinsert(a,curr,n);
    	n--;
    	//cout<<prev<<"\n";
    	sum=sum+prev;
    }
    cout<<sum;
    return 0;
}
