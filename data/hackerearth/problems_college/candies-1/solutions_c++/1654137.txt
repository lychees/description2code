#include <iostream>
using namespace std;

int main() {
	int T;
	scanf("%d",&T);
	while(T--){
		int N; 
		scanf("%d",&N);
		long long a[N+1];
		long long sum = 0;
		for(int i = 0; i<N; i++){
			scanf("%lld",&a[i]);
			sum+=a[i];
		}
		long long ans = sum%N;
		if(ans == 0){
			cout<<"YES\n";
		}
		else{
			cout<<"NO\n";
		}
		cout<<sum/N<<" "<<ans<<endl;
		
	}
	return 0;
}
