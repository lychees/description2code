t = int(raw_input())
while t:
    r = int(raw_input())
    arr = sorted(map(int, raw_input().split()))
    s = 0
    i = r - 1
    while i >= 0:
        s += arr[i] * i - arr[i] * (r - i - 1)
        i -= 1
    s %= 1000000007
    m = max(arr) % 1000000007
    print((s * m) % 1000000007)
    t -= 1
