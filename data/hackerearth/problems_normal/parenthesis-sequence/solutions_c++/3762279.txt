#include <iostream>
#include <list>
#include <string.h>
using namespace std;

void readInp(char inp[])
{
	cin>>inp;
}

void calc(char inp[])
{
	std::list<char> ans;
	char op[1000000];
	int n = strlen(inp);
	int cnt = 0;
	for(int i=0; i<n; i++){
		if(inp[i]=='('){
			ans.push_back('(');
			cnt++;
		}else{
			ans.push_back(')');
			cnt--;
		}
		if(cnt<0){
			ans.push_front('(');
			cnt++;
		}
	}
	while(cnt>0){
		ans.push_back(')');
		cnt--;
	}
	int k = 0;
	for(list<char>::iterator it=ans.begin(); it!=ans.end(); it++){
		op[k++] = *it;
	}
	op[k++] = '\0';
	cout<<op<<endl;
}

int main(int argc, char const *argv[])
{
	char inp[1000000];
	int t;
	cin>>t;
	while(t){
		readInp(inp);
		calc(inp);
	t--;
	}
	return 0;
}
