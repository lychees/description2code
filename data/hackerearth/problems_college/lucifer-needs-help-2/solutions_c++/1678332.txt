#include <iostream>
#include <cstdio>
#include <string.h>
#define ll long long int
#define getcx getchar

using namespace std;

inline void inp( ll &n )//fast input function
{
	n=0;
	ll ch=getcx();
	ll sign=1;
	while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}
	while( ch >= '0' && ch <= '9' )
	n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
	n=n*sign;
} 

ll arr[1000005];
ll arr1[1000005];
ll arr2[1000005];
ll temp[1000005];


int main()
{

	

	ll n;
	
	memset(arr, 0, sizeof(arr));
	memset(arr1, 0, sizeof(arr1));
	memset(arr2, 0 ,sizeof(arr2));
	ll u, i, a, b, c;
	inp(n);
	
	//cin >> n;
	//cin >> u;
	
	for (i = 0; i < n; i++) {
		inp(temp[i]);
	}
	
	inp(u);
	
	
	for (i = 0; i < u; i++) {
		//cin >> a;
		//cin >> b;
		//cin >> c;
		inp(a);
		inp(b);
		inp(c);
		a--;
		b--;
		
		arr1[a] = arr1[a] + c;
		arr2[b] = arr2[b] - c;
	}
	
	ll sum = 0;
	
	for (i = 0; i < n; i++) {
		sum = sum + arr1[i];
		arr[i] = sum;
		sum = sum + arr2[i];
		//cout << sum << " ";
	}
	
	for (i = 0; i < n; i++) {
		printf("%lld\n", arr[i] + temp[i]);
	}
		 
	
	
	return 0;
}
