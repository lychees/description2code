#include<bits/stdc++.h>
using namespace std;
static bool isfib[1000001];
bool isprime(long long x){
	for(long long i=2;i<=sqrt(x);i++){
		if(x%i==0){
			return false;
		}
	}
	return true;
}
int main(){
	memset(isfib,false,sizeof(isfib));
	int f=1,s=1;
	isfib[1]=true;
	while(s<=1e6){
		int z=f+s;
		f=s;
		s=z;
		if(s<=1e6)
		isfib[s]=true;
	}
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		long long sum=0;
		for(int i=1;i<n;i++){
			if(!isfib[i]){
				sum+=i;
			}
		}
	//	printf("%lld\n",sum);
		if(isprime(sum)){
			printf("Tic\n");
		} else
		if(sum%2==0){
			printf("Tac\n");
		}
		else
		printf("Toe\n");
	}
	return 0;
}
