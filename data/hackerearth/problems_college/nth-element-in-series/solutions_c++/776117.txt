matrix multiply(matrix a,matrix b)
{
    matrix c;
    int row=a.size();
    int col=b[0].size();
    c.resize(row);
    for(int i=0;i<row;i++)
        c[i].resize(col,0);

    int i,j,k;
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            c[i][j]=0;
            for(k=0;k<a[0].size();k++)
            {
                c[i][j]=(c[i][j]+a[i][k]*1LL*b[k][j]%mod)%mod;
            }
        }
    }
    return c;
}

matrix matpow(matrix a,ill b)
{
    matrix c;
    c.resize(a.size());
    int i;
    for(i=0;i<a.size();i++)
        c[i].resize(a[0].size(),0);
    for(i=0;i<a.size();i++)
        c[i][i]=1;

    while(b)
    {
        if(b&1)
            c=multiply(a,c);
        a=multiply(a,a);
        b/=2;
    }
    return c;
}
