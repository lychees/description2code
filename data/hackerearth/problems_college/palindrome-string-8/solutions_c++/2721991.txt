#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

void printArr(int* f)
{
	for(int i = 0; i < 26; i++)
		cout << f[i] << " ";
	return;
}
bool isPalind(string s)
{
	int siz = s.size();
	int* freq = new int[26];
	for(int i = 0; i < siz; i++)
		freq[(int)s[i] - 97]++;
	//printArr(freq);
	if(siz % 2 == 0)
	{
		for(int i = 0; i < 26; i++)
		{
			if(freq[i] % 2 == 1)
			return false;
		}
	}
	else
	{
		bool f = false;
		for(int i = 0; i < 26; i++)
		{
			if(freq[i] % 2 == 1)
			{
				if(f) return false;
				else f = true;
			}
		}
	}
	return true;
}

int main()
{
    int T;
    string ss;
    cin >> T;
    while(T--)
    {
    	cin >> ss;
    	transform(ss.begin(), ss.end(), ss.begin(), ::tolower);
    	if(isPalind(ss))
    		cout << "Yes" << endl;
    	else
    		cout << "No" << endl;
    }
    return 0;
}
