#include<iostream>
using namespace std;
int main()
{
	int t,t1,n,even,odd,step;
	cin>>t;
	for(t1=0;t1<t;t1++)
	{
		even=0;
		odd=0;
		step=0;
		cin>>n;
		while(n!=1)
		{
			if(n%2==0)
			{
				even+=n;
				n=n/2;
			}
			else
			{
				odd+=n;
				n=3*n+1;
			}
			step++;
		}
		odd++;
		if(even%step<odd%step)
			cout<<"Odd Rules"<<endl;
		else if(even%step>odd%step)
			cout<<"Even Rules"<<endl;
		else
			cout<<"Tie"<<endl;
	}
	return 0;
}
