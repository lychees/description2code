#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define print(a) cout<<a
#define T ll t; cin>>t ; while(t--)
#define fin freopen("input.txt","r",stdin)


 ll a[100005];

int main(){
    ll n,k;

    scanf("%lld %lld",&n,&k);


    for(ll i=0;i<n;i++)
        scanf("%lld",&a[i]);
	
	if(k>n) {
		for(ll i=0;i<n;i++) {
			printf("%lld ",n-i);
		}
		return 0;
	}
	
    for(ll i=0;i<n && k>0;i++) {

        ll kk=i;
        for(ll j=i+1;j<n;j++) {
            if(a[kk] <= a[j]) {
                kk=j;
            }
        }

        if(a[i]!=a[kk]) {
    		k--;
            ll temp=a[i];
            a[i]=a[kk];
            a[kk]=temp;
        }      


    }

    for(ll i=0;i<n;i++) {
        printf("%lld ",a[i]);
    }

    return 0;
}
