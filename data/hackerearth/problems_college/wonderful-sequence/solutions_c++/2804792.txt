#include <cstdio>
using namespace std;
int main()
{
  int test,b,i;
  scanf("%d",&test);
  while(test--){
  	int count=0;
  	scanf("%d",&b);
  	int a[b];
  	for(i=0;i<b;i++){
  		scanf("%d",&a[i]);
  	}
  	for(i=0;i<b-1;i++){
  		if(a[i]>=a[i+1]||a[i]<0){
  			count++;
  		}
  	}
  	printf("%d\n",count);
  	
  }
}
