#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
typedef long long LL;
int main()
{
    LL arr[100],i;
    arr[0]=1;
    arr[1]=0;
    for(i=2; i<100; i++)
	arr[i]=((i-1)*((arr[i-1]+arr[i-2])%MOD))%MOD;
    int l;
    string s;
    cin >> l >> s;
    if(l==1)cout << 1 << endl;
    else
    cout << arr[l] << endl;
    return 0;
}
