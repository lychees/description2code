for i in range(int(raw_input())):
    n = raw_input()
    a = map(int, raw_input().split())
    already_reach = []

    loc = 0
    is_win = False
    while True:
        if a[loc] == -1:
            is_win = True

        if loc in already_reach:
            break

        already_reach.append(loc)
        loc = a[loc]

    if is_win:
        print("Win")
    else:
        print("Lose")
