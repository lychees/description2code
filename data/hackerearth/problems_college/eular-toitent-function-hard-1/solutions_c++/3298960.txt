#include<iostream>
#include<vector>
#include<cmath>
#include<cstdio>
using namespace std;

#define N 10000000
#define MAXN 10000000
#define LL long long int
LL phi[MAXN + 1], prime[MAXN/10], sz=0;
vector<bool> mark(MAXN + 1);

int main()
{
    
   phi[1] = 1; 
   for (LL i = 2; i <= MAXN; i++ ){
    if(!mark[i]){
        phi[i] = i-1;
        prime[sz++]= i;
    }
    for (LL j=0; j<sz && prime[j]*i <= MAXN; j++ ){
        mark[prime[j]*i]=1;
        if(i%prime[j]==0){
            LL ll = 0;LL xx = i;
            while(xx%prime[j]==0)
            {
                           xx/=prime[j];
                           ll++;         
                       }
            LL mm = 1;
            for(LL k=0;k<ll;k++)mm*=prime[j];
            phi[i*prime[j]] = phi[xx]*mm*(prime[j]-1);
            break;
    }
        else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
    }
}
   long int t;
   scanf("%ld",&t);
   while(t--)
   {
    long int n;
    scanf("%ld",&n);
    printf("%lld\n",phi[n]);             
   } 
   return 0;    
}
