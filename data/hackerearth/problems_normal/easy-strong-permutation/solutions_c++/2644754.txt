#include <iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;

int main()
{
    long long int n;
    cin>>n;
    long int x;
    vector<long int> a,d;
    
    
    for(long int i=0;i<n;i++)
    {cin>>x;
    a.push_back(x);
    }
    sort(a.begin(),a.end());
    long int l=a.size();
    long int  i=0;
    while(i<l/2)
    {d.push_back(a[i]);
    d.push_back(a[l-1-i]);
    i++;
    }
    if(l%2!=0)
    {d.push_back(a[l/2]);
    }
    long int strength=0;
    for(long int j=0;j<n-1;j++)
    {strength=strength+abs(d[j]-d[j+1]);
    }
    strength=strength+abs(d[n-1]-d[0]);
    cout<<strength<<endl;
    return 0;
}
