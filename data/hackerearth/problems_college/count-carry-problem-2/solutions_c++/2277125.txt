# include<bits/stdc++.h>
using namespace std;
 
int main(){
	char A[10],B[10];
	int l,i,s,c,count,t;
	cin>>t;
	
	while(t--){
		scanf("%s %s",A,B);
		c=0,count=0;
		if(strcmp(A,"0")==0 && strcmp(B,"0")==0)
			return 0;
		else{
			if(strlen(A)>strlen(B))
				l = strlen(B);
			else
				l = strlen(A);
			for(i=0;i<l;i++){
				s = A[strlen(A)-1]-'0' + B[strlen(B)-1]-'0' + c;
				A[strlen(A)-1] = '\0';
				B[strlen(B)-1] = '\0';
				c = 0;
				if(s>9){
					c = s/10;
					count++;
				}
			}
			if(count == 1)
			printf("%d carry operation\n",count);
			else if(count > 1)
			printf("%d carry operations\n",count);
			else
			printf("No carry operation\n");
		}
	}
}
