#include <bits/stdc++.h>
using namespace std;

long long int max(long long int a,long long int b)
{
	if(a>b)
	return a;
	return b;
}

int main()
{
	int T;
	scanf("%d",&T);
	int j = 1;
	while(j <= T)
	{
		int n;
		vector<long long int> x;
        scanf("%d",&n);
       
  		for(int i =0; i < n;i++)
		{ 
			long long int q;
			scanf("%lld",&q);
			x.push_back(q);
		}
		vector<long long int>dp;
		dp.push_back(x[0]);
		int i = 1;
		while(i<n)
		{
			if(i >= 2)
			dp.push_back(max(dp[i-1], dp[i-2]+x[i]));
			else
			dp.push_back(max(dp[i-1], x[i]));
			
			i++;
			
		}
	   //cout << "Case " << j << ": " << dp[n-1] << endl;
	   printf("Case %d: %lld\n",j,dp[n-1]);
		j++;
	}
}
