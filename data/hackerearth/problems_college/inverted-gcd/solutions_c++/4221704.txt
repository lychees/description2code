#include<iostream>
#define size 100000
using namespace std;

unsigned long long gcd(unsigned long long a, unsigned long long b) { return b ? gcd(b, a%b) : a; }

int main()
{
	long long a[size],n,i,count=0,j;
	cin>>n;
	for(i=0; i<n; i++)
	cin>>a[i];
	
	for(i=0; i<n-1; i++)
	{
		for(j=i+1; j<n; j++)
		{
			if(a[i]>a[j] && gcd(a[i],a[j])==1)
			count++;
		}
	}
	
	cout<<count<<endl;
}
