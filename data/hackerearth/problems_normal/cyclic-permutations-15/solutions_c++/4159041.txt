#include<bits/stdc++.h>
using namespace std;
string a,b;
string con;
int z[300005];
int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				cin>>a>>b;
				a = a;
				char zi;
				zi = '$';
				con = b + zi + a + a;
				int i;
				int l;
				int r;
				l = r = 0;
				int n;
				n = con.size();
				int cnt = 0;
				int flag=0;
				for(i=1;i<b.size();++i)
				{
						if(b[i]==b[i-1]) cnt++;
				}
				if(cnt == b.size()-1) flag=1;
				cnt=0;
				for(i=1;i<con.size();++i)
				{
						if(i > r)
						{
								l = r = i;
								while(r<n && con[r-l]==con[r])
										r++;
								z[i]=r-l;
								r--;
						}
						else
						{
								int k = i - l;
								if(z[k]<r-i+1)
								{
										z[i] = z[k];
								}
								else
								{
										l = i;
										while(r<n && con[r-l]==con[r])
												r++;
										z[i]=r-l;
										r--;
								}
						}
						if(i >= n/3 && z[i] == b.size()) cnt++;
				}
				/*cout<<con<<endl;
				  for(i=0;i<n;++i)
				  cout<<z[i]<<" ";
				  cout<<endl;*/
				if(a == b)
				{
						cout<<cnt-1<<endl;
				}
				else
				{
						if(cnt > 0) cout<<cnt<<endl;
						else cout<<0<<endl;
				}

		}
}
