def modulo(a, b, mod):
    x, y = 1, a % mod
    while b:
        if b & 1:
            x = (x*y) % mod
        b >>= 1
        y = (y*y) % mod
    return x % mod


mod = 10**9 + 7
for __ in range(input()):
    vertices = input()
    Q = modulo(modulo(2, vertices-1, mod), mod-2, mod)
    ans = (1*Q) % mod
    print ans
