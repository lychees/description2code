#include<bits/stdc++.h>
using namespace std;
bool pandatic[1000001];

long int fpow(long int x,long int n)
{
    if(n==0)
        return 1;
    long int ret = fpow(x,n/2);
    if(n%2)
        ret = x*ret*ret;
    else
        ret = ret*ret;
    return ret;
}

bool dp[1000005];
int main()
{
    for(int i=1;;i++)
    {
        long long int k = fpow(i,i);
        if(k>1e6)
            break;
        pandatic[k] = true;
    }

    for(long int i=1;i<=1e6;i++)
    {
        dp[i] = false;
        if(pandatic[i])
        {
            dp[i] = true;
            continue;
        }
        long long int n = i;
        if(i>9)
        while(n)
        {
           if(dp[i-(n%10)*(n%10)])
           {
               dp[i] = true;
               break;
           }
           n/=10;
        }
    }

    int t;
    cin>>t;
    while(t--)
    {
        long int n;
        cin>>n;
        if(dp[n])
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
