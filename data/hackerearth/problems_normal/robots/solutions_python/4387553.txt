mod = 1000000007
for x in xrange(int(raw_input())):
	robots = int(raw_input())
	names = map(int, raw_input().split())
	possi = True
	res = 1
	names.sort()
	for i in xrange(0, robots):
		if (i >= names[i]):
			possi = False
			break
		res = ((res % mod) * (names[i] - i) % mod) % mod
		#res = res * (names[i] - i)
	if possi:
		print res
	else:
		print 0
