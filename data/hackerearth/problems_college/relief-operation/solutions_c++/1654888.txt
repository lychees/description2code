int dp[2001][2001];

int w[2001];
int n;

int solve(int ind, int rest)
{
    if(ind == n)
    {
        return 0;
    }
    int &ret = dp[ind][rest];
    if(ret != -1) return ret;
    ret = solve(ind + 1, rest);
    if(arr[ind] <= rest)
        ret = max(ret, w[ind] + solve(ind + 1, rest - arr[ind]));

return ret;
}
