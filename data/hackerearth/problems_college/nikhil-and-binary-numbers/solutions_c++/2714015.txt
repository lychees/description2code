#include<bits/stdc++.h>
using namespace std;

int main() {
  int tt;
  cin>>tt;
  while(tt--) {
    int n;
    cin>>n;
    int one=0,zero=0;
    int val;
    for(int i=0;i<n;i++) {
      cin>>val;
      if(val&1) {
        one++;
      } else {
        zero++;
      }
    }
    if(one&1) {
      cout<<one<<endl;
    } else {
      cout<<zero<<endl;
    }
  }
  return 0;
}
