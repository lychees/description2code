#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n,k;
    cin >> n >> k;
    vector<long long> matrix[n];
    for(int i = 0;i < n; ++i){
    	for(int j = 0;j < n;++j){
    		int m;
    		cin >> m;
    		matrix[i].push_back(m);
    		if(i == 0 && j > 0)matrix[i][j] += matrix[i][j - 1];
    	}
    	if(i > 0)matrix[i][0] += matrix[i - 1][0];
    }
    for(int i = 1;i < n;++i){
    	for(int j = 1;j < n;++j){
    		matrix[i][j] += matrix[i - 1][j] + matrix[i][j - 1] - matrix[i - 1][j - 1];
    	}
    }
    int res = 0;
    for(int i = k - 1;i < n;++i){
    	for(int j = k - 1;j < n;++j){
    		long long ans = matrix[i][j];
    		int r = i - k, c = j - k;
    		if(r > -1)ans -= matrix[r][j];
    		if(c > -1)ans -= matrix[i][c];
    		if(r > -1 && c > -1)ans += matrix[r][c];
    		long long left = -1, right = 1000000;
	    	while(right - left > 1){
	    		long long mid = (left + right) / 2;
	    		if(mid * mid * mid >= ans)right = mid;
	    		else left = mid;
	    	}
	    	if(ans == right*right*right)res++;
    	}
    	
    }
    cout << res;
    
}
