#include <iostream>
using namespace std;
long long int n,k,l=0;
int main()
{
	cin>>n>>k;
	long long int a[n],t,i,j,y,max,m;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}

	 for(j=0;j<n;j++)
		{
			max=0;
			
			for(m=j;m<n;m++)
			{
				if(a[m]>max)
				{
					max=a[m];
					y=m;
				}
			}
				if(l<k)
				{
					t=a[j];
					a[j]=a[y];
					a[y]=t;
					if(a[j]!=a[y])
					{
						l++;
					}
				}
		}
	
	for(long int l=0;l<n;l++)
	{
		cout<<a[l]<<" ";
	}
    return 0;
}
