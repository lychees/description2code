import re
s = raw_input()

username = re.findall('username=(\\S.*)&pwd', s)
print "username:", username[0]

pwd = re.findall('pwd=(\\S.*)&profile', s)
print "pwd:", pwd[0]

profile = re.findall('profile=(\\S.*)&role', s)
print "profile:", profile[0]

role = re.findall('role=(\\S.*)&key', s)
print "role:", role[0]

key = re.findall('key=(\\S.*)', s)
print "key:", key[0]
