#include <iostream>
#include<string>
using namespace std;

int main()
{
    int T,i;
    //char a[100]={0}, b[100]={0};
    string a,b;
    cin>>T;
    while(T!=0){
    	cin>>a;
    	cin>>b;
    	for(i=0;a[i]!='\0';i++){
    		if(a[i]!=b[i])
    			cout<<"1";
    		else
    			cout<<"0";
    	}
    	T--;
    	cout<<endl;
    }
    return 0;
}
