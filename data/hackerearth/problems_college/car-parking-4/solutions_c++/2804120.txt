#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,i,j,ctr;
    scanf("%d",&t);
    while(t--)
    {
    	char s[100001];
    	scanf("%s",s);
    	ctr=0;
    	int c=strlen(s);
    	int a[c];
    	for(i=0;i<c;i++)
    	a[i]=0;
    	for(i=0;s[i]!='\0';i++)
    	{
    		
    		if(s[i]=='-')
    		{
    			ctr++;
    			//continue;
    		}
    		else if(s[i]=='D')
    		continue;
    		else if(s[i]=='B')
    		{
    			if(s[i-1]=='-'&&a[i-1]==0)
    			{
    				ctr--;
    				a[i-1]=1;
    			}
    			if(s[i-2]=='-'&&a[i-2]==0)
    			{
    				ctr--;
    				a[i-1]=1;
    			}
    			//continue;
    		}
    		else if(s[i]=='S')
    		{
    			if(s[i-1]=='-'&&a[i-1]==0)
    			{
    				ctr--;
    				a[i-1]=1;
    			}
    			if(s[i+1]=='-'&&a[i+1]==0)
    			{
    				ctr--;
    				a[i+1]=1;
    			}
    			//continue;
    		}
    //	printf("%d-\n",ctr);	
    	}
    	printf("%d\n",ctr);
    }
    return 0;
}
