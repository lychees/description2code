#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
double n;
double pi = 2*acos(0);
long long int len,l;

while(t-->0)
{
    int i,count=0;
        cin>>n;
        if(n==0.00||n==1.00)l=1;

        else{
        l=(((log(2.00*pi*n)/2.00)+(n*(log(n)-1.00)))/log(10.0))+1;
        }
        for(i=5;n/i>=1;i*=5)
            count+=n/i;

        cout<<l-count<<endl;
}
return 0;
}
