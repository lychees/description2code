#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t,n,a[500][3],v[500][3],i;double x,y,z;
   double md=0;long int tm=0;
   cin>>t;
   for(int j=1;j<=t;j++)
   {
   	cin>>n;
   	for(i=0;i<n;i++)
   	cin>>a[i][0]>>a[i][1]>>a[i][2]>>v[i][0]>>v[i][1]>>v[i][2];
   	int r=0,r1;
   	int st=0;
   	double ch,d;
   	while(1)
   	{
   		x=0;y=0;z=0;
   		for(i=0;i<n;i++)
   		{
   			x+=a[i][0]+v[i][0]*r;
   			y+=a[i][1]+v[i][1]*r;
   			z+=a[i][2]+v[i][2]*r;
   		}
   		x=x/(double)n;
   		y=y/(double)n;
   		z=z/(double)n;
   	//cout<<x<<y<<z<<endl;
   		d=sqrt(x*x+y*y+z*z);
   	//cout<<d<<" "<<r<<endl;
   		if(st==0)
   		{
   			st=1;
   			md=d;
   			r1=0;
   			ch=d;
   		
   		}
   		else
   		{
   			if(d>ch)
   			break;
   			if(d<md)
   			{md=d;
   			r1=r;
   			}
   			ch=d;
   		}
   		r++;
   		
   	}
   	printf("Case #%d: %.8lf %d.00000000\n",j,md,r1);
   }
    return 0;
}
