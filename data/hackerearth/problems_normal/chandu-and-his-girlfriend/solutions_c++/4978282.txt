#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

void getNumbers(vector<int>& vec);
void printVector(const vector<int>& vec);

int main()
{
    int tests;
    cin >> tests;
    for(int test(0) ; test<tests ; ++test){
    	int size;
    	cin >> size;
    	vector<int> vec(size);
    	getNumbers(vec);
    	sort(vec.rbegin(),vec.rend());
    	printVector(vec);
    }
    return 0;
}

void getNumbers(vector<int>& vec){
	for(int i(0) ; i<vec.size() ; ++i){
		cin >> vec[i];
	}
}

void printVector(const vector<int>& vec){
	for(const auto& elem : vec){
		cout << elem << ' ';
	}
	cout << endl;
}
