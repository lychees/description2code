#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
typedef struct node
{
	int tim;
	int score;
};
bool mycompare (struct node node1,struct node node2) { return(node1.score>node2.score); }
int main()
{
   int t,i,j,k,l,m,n,q;
   struct node arr[100000];
   scanf("%d%d",&n,&q);
   for(i=0;i<n;i++)
   scanf("%d",&arr[i].tim);
   for(i=0;i<n;i++)
   scanf("%d",&arr[i].score);
   sort(arr,arr+n,mycompare);
   int timesum[100000];
   timesum[0]=arr[0].tim;
   for(i=1;i<n;i++)
   timesum[i]=timesum[i-1]+arr[i].tim;
   for(i=0;i<q;i++)
   {
   	scanf("%d",&m);
   	printf("%d\n",timesum[m-1]);
   }
   return 0;
}
