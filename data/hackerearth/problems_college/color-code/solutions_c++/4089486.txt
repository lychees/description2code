string convert(int a)
{
    string s;
    while(a>0)
    {
        int q=a%16;
        if(q>=10 && q<=15)
        {
            char h=m[q];
            s=h+s;
        }
        else
            {
                char h=(char)(q+48);
                s=h+s;
            }
        a/=16;
    }
    if(s.length()==1)
        s="0"+s;
     if(s.empty())
     s="00";
    return s;
}
