#include<bits/stdc++.h>
using namespace std;
int dp[20000],ar[20000];
int main() {
	int n,i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&ar[i]);
	dp[0]=0;
	int ans=-123456789;
	for(i=1;i<=n;i++) {
		dp[i]=max(dp[i-1]+ar[i],ar[i]);
		ans=max(dp[i],ans);
	}
	printf("%d\n",ans);
	return 0;
}
