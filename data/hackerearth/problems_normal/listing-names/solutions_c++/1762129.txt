#include <bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false); cin.tie(0);
    int t,i;
    cin>>t;
    string s;
    map<string,int> m;
    map<string,int>::iterator it;
    for(i=0;i<t;i++)
    {
        cin>>s;
        m[s]++;
    }
    for(it=m.begin();it!=m.end();it++)
    {
        cout<<(it->first)<<" "<<(it->second)<<endl;
    }
    return 0;
}
