#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int dom[n],brian[n];
	for(int i=0;i<n;i++)
		cin >> dom[i];
	for(int i=0;i<n;i++)
		cin >> brian[i];
	int max1=INT_MIN,max2=INT_MIN;
	for(int i=0;i<n-1;i++)
	{
		int diff=abs(dom[i]-dom[i+1]);
		if(diff>max1)
			max1=diff;
		diff=abs(brian[i]-brian[i+1]);
		if(diff>max2)
			max2=diff;
	}
	if(max1>max2)
	{
		cout << "Dom" << endl;
		cout << max1 << endl;
	}
	else
	{
		cout << "Brian" << endl;
		cout << max2 << endl;
	}
	return 0;
}
