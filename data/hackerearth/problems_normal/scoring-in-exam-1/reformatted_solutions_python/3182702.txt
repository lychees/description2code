import sys


def readoneline():
    linenostrip = sys.stdin.readline()
    linestrip = linenostrip.strip()
    lst = linestrip.split(" ")
    return lst


nqlist = readoneline()
n = int(nqlist[0])
q = int(nqlist[1])


tlist = readoneline()
slist = readoneline()


unsortedlist = []
for i in range(0, n):
    ti = int(tlist[i])
    si = int(slist[i])
    unsortedlist.append((si, ti))

sortedlist = sorted(unsortedlist, key=lambda x: x[0])
srevlist = list(reversed(sortedlist))

prefixsumlist = []

psum = 0
for tup in srevlist:
    psum = psum + tup[1]
    prefixsumlist.append(psum)


for i in range(0, q):
    klist = readoneline()
    k = int(klist[0])
    print prefixsumlist[k-1]
