#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	long long th,m,i,j,k,d;
	int f,t;
	scanf("%lld",&th);
	
	while(th--)
	{
		f=0;t=0;
		scanf("%lld",&m);
		long long a[m],s[m];
		for(i=0;i<m;i++)
		scanf("%lld",&a[i]);
		i=0;
		k=0;
		for(j=a[i];i<m;j++)
		{
			if(j==0)
			break;
			
			for(d=0;d<k;d++)
			{
				if(j==s[d])
				{
				t=1;
				break;}
			}
			if(t==1)
			break;
			
			if(j==-1)
			{
				printf("Win\n");
				f=1;
				break;
			}
			i=j;
			s[k]=j;
			k++;
			j=a[i]-1;
		}
		if(f==0)
		printf("Lose\n");
	}
	return 0;
}
