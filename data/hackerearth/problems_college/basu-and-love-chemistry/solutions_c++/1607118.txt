#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int main() {
    char arr[200007];
	int len;
	ll ctr=0, seq=0;
	cin>>arr;	
	len = strlen(arr);
	for(int i=0;i<len;i++) {
		ctr = ctr*10;
		seq += (i+1)*(arr[i]-'0');	
		ctr += seq;
		ctr = ctr%mod;
	}
	cout<<ctr<<endl;
}
