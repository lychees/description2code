#include <iostream>
using namespace std;

int main()
{
   int n;
   cin>>n;
   for(int i=1;i<=n;i++)
   cout<<"*";
   cout<<"\n";
   for(int j=n-2;j>=1;j--)
   {
   
   	for(int k=j;k>=1;k--)
   	cout<<" ";
   	
   	cout<<"*"<<"\n";
   }
   
   for(int l=1;l<=n;l++)
   cout<<"*";
   
   
   return 0;
}
