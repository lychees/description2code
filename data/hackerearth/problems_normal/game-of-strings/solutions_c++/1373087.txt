#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define MAXN 5001

int DP[MAXN][MAXN] ;
string str1,str2 ;
int N,M ;
int solve(int N,int M){
	
	if(N<0||M<0){
		return 0 ;
	}
	int &ret = DP[N][M] ;
	if(ret != -1){
		return ret ;
	}
	if(str1[N] == str2[M]){
		ret = 1 + solve(N-1,M-1) ;
	}else{
		ret = max(solve(N-1,M),solve(N,M-1)) ;
	}
	return ret ;
}

string remove_vowel(string &str){
	string x = "" ;
	for(int i=0;i<str.length();i++){
		if(!(str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u'))
			x += str[i] ;
	}
	return x ;
}
int main()
{
    cin >> str1 ;
    cin >> str2 ;
    str1 = remove_vowel(str1) ;
    str2 = remove_vowel(str2) ;
    N = str1.length() ;
    M = str2.length() ;
    N -- ;
    M -- ;
    memset(DP,-1,sizeof DP) ;
    cout << solve(N,M) << endl ;
    return 0;
}
