lld modpow(lld a,lld n,lld temp){lld res=1;while(n>0){if(n&1)res=(res*a)%temp;a=(a*a)%temp;n>>=1;}return res%temp;} 
 
int main()
{
  int t;
  cin>>t;
  while (t--) {
    int n, i, j;
    cin>>n;
    vi a;
    string s;
    rep (i, n) {
      cin>>s;
      if (a.sz == 0)
      	a = vi (s.sz);
      rep (j, s.sz) {
        if (s[j] == 'Y')
          a[j] += 1<<i;
      }
    }
    map<int, int> dp;
    rep (i, a.sz) {
      map<int,int>::iterator ed = dp.end();
      for (map<int, int>::iterator it = dp.begin(); it!= ed; it++) {
          if (dp[it->first|a[i]] == 0)
            dp[it->first|a[i]] = it->second + 1;
          dp[it->first|a[i]] = min(dp[it->first|a[i]], it->second + 1);
      }
      dp[a[i]] = 1;
    }
   cout<<dp[(1<<n)-1]<<"\n";
  }
 
	return 0;
}
