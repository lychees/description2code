#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<long long int> arr;
    long long int ans=0,x;
    for(int i=0;i<n;i++)
    {
    	cin>>x;
    	arr.push_back(x);
    }
    sort(arr.begin(),arr.end());
    if(n%2==0)
    {
    	for(int i=0,j=n-1;i<=j;i++,j--)
    	{
    		ans=ans+abs(arr[i]-arr[j]);
    		if(j-1>=i)ans=ans+abs(arr[i]-arr[j-1]);
    	}
    }
    else
    {
    	for(int i=0,j=n-1;i<=j;i++,j--)
    	{
    		ans=ans+abs(arr[i]-arr[j]);
    		if(j+1<n)ans=ans+abs(arr[i]-arr[j+1]);
    	}
    }
    cout<<ans;
    return 0;
}
