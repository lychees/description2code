'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
lines = int(raw_input())
line_set = []

for i in xrange(lines):
    inp = raw_input()
    line_set.append(list(int(i) for i in iter(inp.split())))

line_set.sort(key = lambda x: x[0])

result = 0
sp = 1
for x, _, d in line_set:
    sp = max(sp, x)
    if x+d >= sp:
        result += x+d-sp+1
        sp = x+d+1

print result
