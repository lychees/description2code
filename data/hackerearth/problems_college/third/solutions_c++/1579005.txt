int max(int a,int b){
return a>b?a:b;
}
int maximiumsubcontigous(int *a,int n){
	m[0]=a[0];
	m[1]=a[1];
	for(int i=2;i<n;i++){
		m[i]=max(m[i-1],a[i]+m[i-2]);
	}
	return m[n-1];
	}
