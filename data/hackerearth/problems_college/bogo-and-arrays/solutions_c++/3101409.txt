#include<stdio.h>
long long int res;
int main()
{
long long int res,tmp;
int t,n,k,i,val,nl;
scanf("%d",&t);
while(t--)
{
res=0;
nl=0;
scanf("%d %d",&n,&k);
for(i=0;i<n;i++)
 {
   scanf("%d",&val);
   if(val<=k)
   {
     nl++;
   }
   else
   {
    tmp=n-i;
    tmp=tmp*(nl+1);
    res=res+tmp;   
    nl=0;
   }
 }
 printf("%lld\n",res);
}
return 0;
}
