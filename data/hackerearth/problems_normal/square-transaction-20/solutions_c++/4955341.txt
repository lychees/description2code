#include <iostream>
using namespace std;
typedef long long int lli;
int b_search (lli arr[], int i, int j, lli s);
int main()
{
	int i, t, q;
	cin >> t;
	lli arr[t], x;
	cin >> arr[0];
	for (i = 1; i < t; ++i) {
		cin >> x;
		arr[i] = arr[i - 1] + x;
	}
	for (cin >> q; q; --q) {
		cin >> x;
		if (x > arr[t - 1]) {
			cout << "-1" << endl;
			continue;
		}
		i = b_search (arr, 0, t - 1, x);
		cout << i + 1 << endl;
	}
	return 0;
}
int b_search (lli arr[], int i, int j, lli s)
{
	int m;
	for (m = (i + j) / 2; i <= j; m = (i + j) / 2) {
		if (arr[m] == s) {
			return m;
		} else {
			if (arr[m] < s) {
				i = m + 1;
			} else {
				j = m - 1;
			}
		}
	}
	return i;
}
