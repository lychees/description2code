#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k;
    cin>>n;
    
    while(n>0){
    	string s1;
    	cin>>s1;
    	vector<char> vec(s1.begin(),s1.end()) ;
    	reverse(vec.begin(),vec.end());
    	k=0;
    	for(int i=0;i<s1.length()-1;i++){
    		if( abs(s1[i]-s1[i+1]) != abs(vec[i]-vec[i+1]) ){
    			k=1;break;
    		}
    	
    	}
    	if(k==0){
    		cout<<"Funny"<<endl;
    	}else
    		cout<<"Not Funny"<<endl;
    	n--;
    }
    return 0;
}
