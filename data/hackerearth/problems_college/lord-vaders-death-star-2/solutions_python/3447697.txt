count = int(raw_input())

nos = raw_input().split(" ")

ma = {}

for n in nos:
	n = int(n)
	if(ma.has_key(n) == False):
		ma[n] = 1
	else:
		ma[n] += 1

uniqueNos = []

for key in ma.keys():
	if(ma[key] == 1):
		uniqueNos.append(key)

if(len(uniqueNos) == 0):
	print -1
else:
	print max(uniqueNos)
