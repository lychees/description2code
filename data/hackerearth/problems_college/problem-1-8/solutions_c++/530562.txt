#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
#include<string>

#define getcx getchar_unlocked

inline void input(int &n)
{
n=0;
int ch=getcx();
while( ch < '0' || ch > '9' ){ ch=getcx();}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
}


using namespace std;

int main(){

	string word;
	cin>>word;
	
	string str;
	vector<string> v;
	
	int n;
	cin>>n;
	
	string subst;
	while(n--){
		cin>>str;		
		subst = str.substr(0,word.length());
		if(word.compare(subst)==0)
			v.push_back(str);
	}
	
	sort(v.begin(),v.end());
	
	if(v.empty())
		cout<<word<<"\n";
	else
		cout<<v[0]<<"\n";	

}
