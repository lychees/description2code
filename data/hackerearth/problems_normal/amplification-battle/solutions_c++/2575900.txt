#include <iostream>
using namespace std;

int a,b;

int winner(long long num,int player)
{
	if(num==1)
		return player;
		
	if(!player)
		winner((num-1)/b+1,!player);
	else
		winner((num-1)/a+1,!player);
		

}

int main()
{
    int T,w;
    long long c;
    int flag;
    
    cin>>T;
    
    while(T--)
    {
    	cin>>a>>b>>c;
    	flag=0;
    	
    	w=winner(c,flag); //0 is adam 1 is bob
    	
    	if(w)
    		cout<<"Adam"<<endl;
    	else
    		cout<<"Bob"<<endl;
    }
    return 0;
}
