#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int dp[1000010][5];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int i,j,a[]={2,3,5,7};
	for(i=1;i<1000010;i++)
	{
		for(j=0;j<=4;j++)
		dp[i][j]=INT_MAX;//Not possible
	}
	for(i=0;i<=4;i++)
	{
		dp[0][i]=0;//Possible
	}
	for(i=1;i<1000010;i++)
	{
		for(j=1;j<=4;j++)
		{
			if((a[j-1]<=i))
			{
				dp[i][j]=min(dp[i][j],1+dp[i-a[j-1]][j]);
			}
			dp[i][j]=min(dp[i][j],dp[i][j-1]);
		}
	}
	//~ for(i=0;i<10;i++)
	//~ {
		//~ for(j=0;j<=4;j++)
		//~ cout<<dp[i][j]<<" ";
		//~ cout<<endl;
	//~ }
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n;
		cin>>n;
		if(dp[n][4]>=INT_MAX)
		cout<<-1<<endl;
		else
		cout<<dp[n][4]<<endl;
	}
	return 0;
}
