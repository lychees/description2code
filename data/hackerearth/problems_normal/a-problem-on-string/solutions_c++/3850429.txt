#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <map>
#include <utility>
using namespace std;

map<pair<char,char> ,long long int > mp;
map<pair<char,char> ,long long int > :: iterator it;
char str[1000003];
    
int main()
{
    //cout << "Hello World!" << endl;
    int len,q;
    long long int ans,cnt;
    char st,end;
    scanf(" %d %s %d ",&len,&str,&q);
    
    for(char i='a';i<='z';i++)
    {
    	for(char j=i;j<='z';j++)
    	{
    		ans = 0;
    		cnt = 0;
    		if(i==j)
    		{
    			int single = 0;
    			for(int k=len-1;k>=0;k--)
    			{
    				if(str[k]==j)cnt++,single++;
    				if(str[k]==i)ans += cnt;
    			}
    			ans = ans - single;
    		}
    		else
    		{
    			for(int k=len-1;k>=0;k--)
    			{
    				if(str[k]==j)cnt++;	
    				if(str[k]==i)ans += cnt;
    			}
    			cnt = 0;
    			for(int k=len-1;k>=0;k--)
    			{
    				if(str[k]==i)cnt++;
    				if(str[k]==j)ans += cnt;
    			}
    		}
    		mp.insert(make_pair(make_pair(i,j) , ans));	
    		
    	}
    }
    while(q--)
    {
    	scanf(" %c %c",&st,&end);
    	if(st<end)
    	{
    		it = mp.find(make_pair(st,end));
    	}
    	else
    	{
    		it = mp.find(make_pair(end,st));
    	}
    	printf("%lld\n",it->second);
    	
    }
    mp.clear();
    return 0;
}
