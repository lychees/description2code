#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   int t;
   cin>>t;
   string s;
   for (int i=0;i<t;i++)
  {
  	cin>>s;
  	int j;
  int m=0;
  	for ( j=s.size()-1;j>0;j--)
  	{ if (s[j]>s[j-1]) {m=1; break;}
  	}
  	char c=s[j-1];
  	char min=s[j];
  	int k=j;
  	for (int m=j+1;m<s.size()-1;m++)
  	{
  		if (c<s[m] && min>s[m] ) {min=s[m];k=m;}
  	}
  //	cout<<j<<" "<<k<<endl;
    swap(s[j-1],s[k]);
    sort(s.begin()+j,s.end());
    if (m==0 ) cout<<"no answer"<<endl;
    else
  	cout<<s<<endl;
  //	if (c==1) cout<<s<<endl;
  //	else cout<<"no answer"<<endl;
  	
  }
    return 0;
}
