#include <iostream>
#include <string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int l = s.length();
    	int pos =0;
		for(int i=0;i<l;i++){
			if(s[i]!=s[l-1]){
				continue;
			}
			else{
				int j=i+1;
				int last = l-2;
				while(j<=last && s[j]==s[last]){
					j++;
					last--;
				}
				if(j >= last){
					pos = i;
					break;
				}
			}
			
		}
		cout<<(l+pos)<<endl;
    }
    return 0;
}
