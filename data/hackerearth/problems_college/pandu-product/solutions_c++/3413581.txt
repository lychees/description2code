#include <iostream>
using namespace std;
 
int main()
{
	long long int mul_max=1,mul_min=1,max1=-1000000000,max2=-1000000000,min1=1000000000,min2=1000000000;
	int n;
	cin>>n;
	long long int *arr=new long long int[n]; 
	for(int i=0;i<n;i++)
		cin>>arr[i];
		for(int i=0;i<n;i++)
		{
			if(arr[i]>max1)
			{
				max2=max1;
				max1=arr[i];
			}
			else if(arr[i]>max2)
			{
				max2=arr[i];
			}
			if(arr[i]<min1)
			{
				min2=min1;
				min1=arr[i];
			}
			else if(arr[i]<min2)
			{
				min2=arr[i];
			}
		}
		mul_max=max1*max2;
		mul_min=min1*min2;
		if(mul_max>=mul_min)
			cout<<mul_max<<endl;
		else
			cout<<mul_min<<endl;
		
}
