def CompilerVersion():
    while True:
        try:
            string = raw_input()
        except BaseException:
            break

        if '//' in string:
            pos = string.find("//")
            newstring = string[:pos].replace('->', '.')+string[pos:]
            print newstring
        else:
            newstring = string.replace('->', '.')
            print newstring


if __name__ == "__main__":
    CompilerVersion()
