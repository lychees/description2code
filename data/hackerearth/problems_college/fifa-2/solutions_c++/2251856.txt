using namespace std;
int main()
{
    int n,i;
    scanf("%d",&n);
    int x[100001],y[100001],h[100001]={0},a[100001]={0};
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&x[i],&y[i]);
        h[x[i]]++;
        a[y[i]]++;
    }
    for(i=0;i<n;i++)
    {
        printf("%d %d\n",n-1+h[y[i]],n-1-h[y[i]]);
    }
    return 0;
}
