#include<bits/stdc++.h>
using namespace std;
int arr[100];
int check(int start){
    for(int i=start;i>0;i--){
        if(arr[i] && arr[i-1])
                return 0;
    }
    return 1;
}
int NxtBin(int num){
    int i=0;
    while(pow(2,i)<=num)
        i++;
    int index=i-1;
    for(i=index;i>=0;i--){
        if(num>=pow(2,i)){
            num-=pow(2,i);
            arr[i]=1;
        }
        else
            arr[i]=0;

    }
    return check(index);
}

int main(){
    int count,num,i;
    scanf("%d",&count);
    while(count--){
        scanf("%d",&num);
        for(i=num;;i++)
            if(NxtBin(i))
                break;
        printf("%d\n",i);



    }


}
