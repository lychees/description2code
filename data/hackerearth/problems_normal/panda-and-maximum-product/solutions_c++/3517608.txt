#include <iostream>
using namespace std;

int main()
{
    long n;
    cin>>n;
    long long num;
    long long a=0,b=0,c=0,d=0;
    for(long i=0;i<n;i++) {
    	cin>>num;
    	if(num>=0) {
    		if(num>=a) {
    			b=a;
    			a=num;
    		} else if(num>b) {
    			b=num;
    		}
    	} else {
    		if(num<=c) {
    			d=c;
    			c=num;
    		} else if(num<d) {
    			d=num;
    		}
    	}
    }
    unsigned long long max = a*b;
    unsigned long long temp = c*d;
    if(temp > max)
    	max = temp;
    cout<<max;
    return 0;
}
