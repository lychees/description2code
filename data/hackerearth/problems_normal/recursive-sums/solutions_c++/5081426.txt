#include <iostream>
using namespace std;

int recur(long long int num){
	int s = 0;
	while(num){
		s+=num%10;
		num/=10;
	}
	return s;
}

int main()
{
    int T,N,r;
    long long int d,sum;
    
    scanf("%d",&T);
    while(T--){
    	scanf("%d",&N);
    	sum = 0;
    	
    	while(N--){
    		scanf("%lld%d",&d,&r);
    		if(d*r >= 10){
    			sum += recur(d*r);	
    		}else{
    			sum += d*r;
    		}
    		
    	}
    	
    	while(sum>=10){
    		sum = recur(sum);
    	}
    	printf("%lld\n",sum);
    }
    
    return 0;
}
