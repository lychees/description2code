#include <iostream>
#include<vector>
#include<string>
#include<algorithm>

using namespace std;

int main()
{
   int n;
   cin>>n;
   vector<string> v(n);
   for(int i=0;i<n;i++)
   {
   	cin>>v[i];
   }
   vector<string>::iterator it;
   for(int i=0;i<n;i++)
   {
   		string temp=v[i];
   		reverse(temp.begin(),temp.end());
   		it=find(v.begin(),v.end(),temp);
   		if(it!=v.end())
   		{
   			cout<<temp.size()<<" ";
   			cout<<v[i][temp.size()/2]<<endl;
   			break;
   		}
   }
  
}
