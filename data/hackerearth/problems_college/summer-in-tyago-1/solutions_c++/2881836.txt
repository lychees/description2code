#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
// codeforces 549G Happy Line
// Problem F
// greed

int main()
{
	int n;
	int man[200005];
	while(~scanf("%d", &n)){
		for(int i=0; i<n; i++){
			scanf("%d", &man[i]);
			man[i] += i;
		}
		sort(man,man+n);
		bool findans = 0;
		for(int i=0; i<n; i++){
			man[i] -= i;
			if(man[i] < man[i-1] && (i!=0)){
				printf(":(\n");
				findans = true;
				break;
			}
		}
		if(!findans)
			for(int i=0; i<n; i++){
				if(i == n-1)
					printf("%d\n", man[i]);
				else
					printf("%d ", man[i]);
			}
	
	}
	return 0;
}
