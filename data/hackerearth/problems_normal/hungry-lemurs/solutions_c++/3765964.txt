#include <iostream>
#include <cmath>
using namespace std;
double min(double,double);
int main()
{
    int n, k;
    cin>>n>>k;
    int minimum_minutes=1e9,i;
    
    for(int j=2*k;j>=1;j--)
    {
    	i=n-(n%j);
    	minimum_minutes=min(minimum_minutes,abs(n-i)+abs(k-j));
    	i+=j;
    	minimum_minutes=min(minimum_minutes,abs(n-i)+abs(k-j));
    }
    
    cout<<minimum_minutes<<"\n";
    return 0;
}
double min(double a,double b){
	return (a>b?b:a);
}
