#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   while(t--)
   {
   	int a[100005],b[100005];
   	int n;
   	cin>>n;
   	for(int i=1;i<=n;i++)
   	{
   		cin>>a[i];
   		b[a[i]]=i;
   	}
   	int flag=0;
   	for(int i=1;i<=n;i++)
   	{
   		if(a[i]!=b[i])
   		{
   			
   		flag=1;
   		break;
   		}
   	}
   	if(flag)
   	cout<<"not inverse"<<endl;
   	else
   	cout<<"inverse"<<endl;
   	
   }
    return 0;
}
