#include <bits/stdc++.h>
using namespace std;
#define NO_OF_CHARS 256
int main()
{
    int t=0;
    scanf("%d",&t);
    while(t--)
    {
    string s;
    cin>>s;
    int c=1,m=1,p,i=0,n=s.size();
    int *v = (int *)malloc(sizeof(int)*NO_OF_CHARS);
    for (i = 0; i < NO_OF_CHARS;  i++)
        v[i] = -1;
        v[s[0]] = 0;
        
        for(i=1;i<n;i++)
        {
        p=v[s[i]];
       if (p==-1 || i-c>p)
            c++;
        else
        {
            if (c>m)
                m=c;
             c=i-p;
        }
 
        v[s[i]] = i; 
       }
 
   
    if (c>m)
        m=c;
    cout<<m<<endl;
    }
    return 0;
}
