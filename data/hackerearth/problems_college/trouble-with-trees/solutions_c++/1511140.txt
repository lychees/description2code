#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
unsigned long long int catalanDP(int n)
{
    unsigned long long int catalan[n+1];
   catalan[0] = catalan[1] = 1;
   for (int i=2; i<=n; i++)
    {
        catalan[i] = 0;
        for (int j=0; j<i; j++)
            catalan[i] = (catalan[i] + ( catalan[j] * catalan[i-j-1])%mod)%mod;
    }
   return catalan[n];
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        cout<<catalanDP(n)<<endl;
    }
    return 0;
}
