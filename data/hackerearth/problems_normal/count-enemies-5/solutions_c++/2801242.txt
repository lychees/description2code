#include <iostream>
#include <string>

using namespace std;

int util(string x)
{
	int count = 0 , length = x.length() ; 
	
	for(int i = 0 ; i < length ; i++)
	{
		if(x[i] == 'X')
		{
			return 0 ;
		}
		count ++ ;
	}
	
	return count ;
}

int main()
{
    int test ;
    scanf("%d",&test) ;
    
    
    while(test--)
    {
    	string str ; 
    	cin >> str ;
    	
    	int count = 0 ;
    	
    	while(str.length() != 0)
    	{
    		int index = str.find('*') ;
    		
    		//cout << " index >> " << index << endl ;
    		
    		string temp ;
    		int temp_count = 0 ; 
    		
    		if(index != str.length() && index != EOF)
    		{
    			temp = str.substr(0,index) ;
    			count += util(temp) ;
    			str = str.substr(index+1) ;
    			
    			//cout << " temp >> " << temp << "  str >> " << str << endl ; 
    		}
    		else
    		{
    			//cout << " str >> " << str << endl ; 
    			count += util(str) ; 
    			str = "" ;
    		}
    	}
    	printf("%d\n",count) ;
    	
    }
    
    
    return 0;
}
