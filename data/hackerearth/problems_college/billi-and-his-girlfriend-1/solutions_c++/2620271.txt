#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll fastmul(ll a,ll b,int k)
{
	ll num=1, mod=pow(10,k);
	while(b)
	{
		if(b%2==1)
			{num*=a;
			num%=mod;
			}
		a*=a;
		a%=mod;
		b/=2;

	}
return num%mod;
}
int main()
{
ll t,a,b,k,ans,res;
cin>>t;
while(t--){
	res=0;
cin>>a>>b>>k;
ans=fastmul(a,b,k);
while(ans)
{
	res+=(ans%10);
	ans/=10;
}
cout<<res<<endl;
}
}
