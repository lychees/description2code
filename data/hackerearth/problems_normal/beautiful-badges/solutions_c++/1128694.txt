#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    int n=scan();
    int aa[100001];
    int bb[100001];
    int x=0;
    int y=0;
    for(int i=0;i<n;i++){
        int c=scan();
        int d=scan();
        if(c==1){
            aa[x]=d;
            x++;
        }
        else{
            bb[y]=d;
            y++;
        }
    }
    sort(aa,aa+x);
    sort(bb,bb+y);
    int ans=0;
    while(x>=1&&y>=1){
        int a=aa[x-1];
        int b=bb[y-1];
        if(b>=a){
            if(b*b<2*a*a){
                ans++;
                x--;
                y--;
            }
            else{
                y--;
            }
        }
        else{
            if(a*a<2*b*b){
                ans++;
                x--;
                y--;
            }
            else{
                x--;
            }
        }
    }
    printf("%d",ans);
}
