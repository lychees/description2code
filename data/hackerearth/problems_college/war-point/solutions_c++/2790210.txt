#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
     ll t;
	 scanf("%lld",&t);
	 while(t--)
	 {
     ll n;
     ll sum=0;
	 scanf("%lld",&n);
	 ll arr[n];
	 for(ll i=0;i<n;i++)
	 {
	 scanf("%lld",&arr[i]);
	 sum+=arr[i];
	 }
	 ll sum2=sum-arr[0];
	 ll diff=abs(sum2-arr[0]);
	 ll sum1=arr[0];
	 for(ll i=1;i<=n-2;i++)
	 {
	     sum1+=arr[i];
		 sum2-=arr[i];
		 //cout<<sum1<<" "<<sum2<<endl;
		 if(abs(sum1-sum2)<diff)
		 diff=abs(sum1-sum2);
	 }
	 printf("%lld",diff);
	 printf("\n");
	 }
	 
	 return 0;
}
