def get_limit_and_difficulty(input):
	limit, difficulty = input.split(' ')
	return int(limit), int(difficulty)

def get_score(questions, difficulty):
	skipped = False
	marks = 0
	for i in questions:
		if i <= difficulty:
			marks += 1
		else:
			if not skipped:
				skipped = True
			else:
				break
	return marks

def main():
	input_str = raw_input()
	limit, difficulty = get_limit_and_difficulty(input_str)
	questions = raw_input()
	questions = questions.split(' ')
	questions = [int(x) for x in questions]
	print get_score(questions, difficulty)

main()
