#include<bits/stdc++.h>
using namespace std;
#define gc getchar
inline long long int read()
{
	register char c = gc();
	while(c<'0' || c>'9') c = gc();
	long long int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - 48;
		c = gc();
	}
	return ret;
}
int main()
{
    int t;
    t=read();
    while(t--)
    {
    	int n,i,j,count=0;
    	int a[10003]={0};
    	n=read();
    	for(i=0;i<n;i++)
    	{
    		j=read();
    		a[j]++;
    	}
    	for(i=0;i<10002;i++)
    	{
    		if(count<a[i])
    		count=a[i];
    	}
    	printf("%d\n",count);
    }
	return 0;
}
