#include <bits/stdc++.h>
using namespace std;
#define MAXN 500005
#define ll long long

int m,n;
//int cnt[MAXN];
int col[100005];
int a[55];
int main()
{	
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>m;
	for(int i=1;i<=m;i++)
		cin>>a[i];
	for(int i=1;i<=n;i++)
		cin>>col[i];

//	memset(cnt,0,sizeof(cnt));
	ll ans=1;
	for(int i=1;i<=n;i++)
	{
		ll int c=a[col[i]];

		ll low=ans/c;
		low++;
		low*=c;
		//cout<<i<<": ";
		while(true)
		{
			bool posi=true;
			//cout<<low<<"__";
			//if(low>50)
			//	return 0;
			for(int j=col[i]+1;j<=m;j++)
			{	
				//if(j==col[i])continue;
				if(low%a[j]==0)
					posi=false;

			}
			if(posi)
			{
				ans=low;
				break;
			}
			low+=c;
		}
		//cout<<endl;
	}

	cout<<ans<<endl;
	return 0;
}
