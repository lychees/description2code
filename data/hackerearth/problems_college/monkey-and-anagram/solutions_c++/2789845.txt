#include <bits/stdc++.h>
#define ll long long
using namespace std;
int ht[26],hs[26];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		memset(hs,0,sizeof(hs));
		memset(ht,0,sizeof(ht));
		string s,t;
		cin>>s>>t;
		int ls,lt,i;
		ls=s.length();
		lt=t.length();
		for(i=0;i<ls;i++)
		{
			hs[s.at(i)-'a']++;
		}
		for(i=0;i<lt;i++)
		{
			ht[t.at(i)-'a']++;
		}
		ll int ans=0;
		for(i=0;i<26;i++)
		{
			if(hs[i]>ht[i])
			{
				ans=-1;
				break;
			}
			else if(ht[i] > hs[i])
			{
				ans++;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
