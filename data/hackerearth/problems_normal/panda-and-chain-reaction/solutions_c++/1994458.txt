#include <iostream>
#include <cstdio>
using namespace std;
#define m 1000003
int main()
{
    long long int fact[1000004], n, x, i;
    fact[0]=1;
    for(i=1;i<=1000003;i++)
    {
    	fact[i]=(fact[i-1]*i)%m;
   	}
    int t;
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%lld%lld", &n, &x);
    	if(n<m)
    	printf("%lld\n", ((x%m)*fact[n])%m);
    	else printf("0\n");
    }
    return 0;
}
