#include<bits/stdc++.h>
#define modulo 1000000007
using namespace std;
vector<long long int> first(1002);
vector<long long int> multiple(1002);
vector<long long int> totalways(1002);
void single(int a)
{
 
    for(int i=5;i<=a;i++)
        {
            first[i]=(first[i-1]+first[i-2]+first[i-3]+first[i-4])%modulo;
        }
 
}
 
long long int power(long long int a,int n)
{   long long int Final=a;
    if(a==0 || a==1)
        return 1;
    else
    {
       for(int i=2;i<=n;i++)
       {
            Final=Final*a;
            Final=Final%modulo;
       }
       return Final;
    }
}
 
 
void multiplearr(int m,int n)
{
    for(int i=0;i<=m;i++){
        multiple[i]=power(first[i],n);
    }
 
 
}
int main(){
 
    int t,n,m;
    long long int Count=0;
    first[0]=1,first[1]=1,first[2]=2,first[3]=4,first[4]=8;
 
 
    single(1000);
 
    cin>>t;
    while(t--){
       cin>>n>>m;
       multiplearr(m,n);
       totalways[0]=1,totalways[1]=1;
       for(int i=2;i<=m;i++){
            Count=0;
            for(int j=1;j<i;j++){
                Count+=(totalways[j]*multiple[i-j])%modulo;
            }
            totalways[i]=(multiple[i]-Count)%modulo;
       }
       while(totalways[m]<0)
       {
            totalways[m]+=modulo;
       }
      /* for(int i=0;i<=m;i++)
            cout<<first[i]<<" ";
            cout<<endl;*/
            
       cout<<totalways[m]<<endl;
        multiple.clear();
        totalways.clear();
    }
 
    return 0;
}
