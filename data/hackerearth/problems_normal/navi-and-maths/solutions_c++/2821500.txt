#include <bits/stdc++.h>

using namespace std;

#define ll long long

const ll mod = 1000000007;

ll modpow(ll a, ll b)
{
	ll x=1ll, y=a;
	while(b){
		if(b%2)
			x=(x*y)%mod;
		y=(y*y)%mod;
		b/=2;
	}
	return x;
}

ll arr[20];

int main()
{
	int t, n, i, j, c=1;
	ll pro, sum, res, ans;
	scanf("%d", &t);
	while(t--){
		scanf("%d", &n);
		for(i=0; i<n; i++)
			scanf("%lld", &arr[i]);
		ans = 0;
		for(i=0; i<(1<<n); i++){
			pro = 1;
			sum = 0;
			for(j=0; j<n; j++){
				if((i&(1<<j))){
					pro = (pro*arr[j])%mod;
					sum = (sum+arr[j])%mod;
				}
			}
			sum = modpow(sum, mod-2);
			res = (pro*sum)%mod;
			ans = max(ans, res);
		}
		printf("Case #%d: %lld\n", c++, ans);
	}
	return 0;
}
