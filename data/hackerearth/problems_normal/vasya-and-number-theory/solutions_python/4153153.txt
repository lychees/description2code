N = int(raw_input())
ans = -100

array = map(int,raw_input().strip().split())
array.sort()

div = [0] * N
for i in range(0,N):
        div[i] =1
        for j in range(0,i):
                if array[i] % array[j] == 0:
                        div[i] = max(div[i],div[j]+1)
#print div
ans = max(div)
print ans if ans >=2 else -1
