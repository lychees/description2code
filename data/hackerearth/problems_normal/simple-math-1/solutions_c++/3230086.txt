#include<bits/stdc++.h>
using namespace std;
int dp[1005][1005];
#define mod 10000007
void compute()
{
    for(int i=0;i<1005;i++)
    {
        for(int j=0;j<1005;j++)
        {
            if(i==j||j==0)
                dp[i][j]=1;
                else
        dp[i][j]=(dp[i-1][j-1]%mod+dp[i-1][j]%mod)%mod;
        }
    }
}
int main()
{
    compute();

        int n,p;
        cin>>n>>p;
        int ans=0;
        if(p>n)
        {
        	cout<<-1<<endl;
			return 0;     	
        }
        ans=(ans%mod+dp[n-1][p-1]%mod)%mod;

        cout<<ans<<endl;
    }
