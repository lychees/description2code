#include <iostream>
using namespace std;
int main()
{
	
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	cin>>s;
    	int i=0,j;
    	while(s[i]!='\0')
    	{
    		j=s[i];
    	   	if(j>=97&&j<=99)
    	   	cout<<2;
    		else if(j>=100&&j<=102)
    		cout<<3;
    		else if(j>=103&&j<=105)
    		cout<<4;
    		else if(j>=106&&j<=108)
    		cout<<5;
    		else if(j>=109&&j<=111)
    		cout<<6;
    		else if(j>=112&&j<=115)
    		cout<<7;
    		else if(j>=116&&j<=118)
    		cout<<8;
    		else 
    		cout<<9;
    	
    		i++;
    	}
    	cout<<endl;
    }
    return 0;
}
