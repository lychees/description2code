#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
		int n;
		scanf("%d",&n);
		vector<int> a;
		int k;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&k);
			a.push_back(k);
		}
		sort(a.begin(),a.end());
		int ans=1;
		int prev=-1;
		int counter=0;
		for(int i=0;i<n;i++)
		{
			if(a[i]==prev)
			counter++;
			else
			counter=1;
			prev=a[i];
			if(ans<counter)
			ans=counter;
			
		}
		printf("%d\n",ans);
	}
}
