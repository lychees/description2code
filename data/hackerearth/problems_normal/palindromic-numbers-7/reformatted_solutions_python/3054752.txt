for i in xrange(int(raw_input())):
    count = 0
    startRng, endRng = map(int, raw_input().split())
    for j in xrange(startRng, endRng + 1):
        string = str(j)
        revStr = reversed(string)
        if list(string) == list(revStr):
            count += 1
    print count
