#include <bits/stdc++.h>
#include<string.h>
using namespace std;
int main()
{

    int a,b;
    int i,j,k,l;
  long long int arr[1000][1000];
    cin>>a>>b;
    for(i=0;i<a;i++){
        for(j=0;j<b;j++){
            cin>>arr[i][j];
        }

    }
    int query;
    cin>>query;
    while(query--){
        int aa;
        bool flag=false;
        cin>>aa;
        j=b-1;
        for(i=0;i<a;i++){
            for(;j>=0;j--){
                if(arr[i][j] > aa ){
                    continue;
                }
                else if(arr[i][j]<aa){
                    break;
                }
                else{
                    flag=true;
                    cout<<i<<" "<<j<<endl;
                    break;
                }
            }
            if(flag)
                break;
        }
        if(!flag)
            cout<<"-1"<<" "<<"-1"<<endl;
    }


    return 0;
}
