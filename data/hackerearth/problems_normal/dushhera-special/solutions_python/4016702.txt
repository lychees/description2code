from __future__ import print_function
t=int(raw_input())
while t:
	t-=1
	s=raw_input()
	p,k=s.split(" ")
	p=int(p)
	k=int(k)
	for i in range(1,p+1):
		print (i*k+i, end=' ')
	print('')
