int gcd(int a,int b)
{
	if(a == 0) return b;
	if(b == 0) return a;
	
	if(a>b) return gcd(b,a%b);
	return gcd(a,b%a);
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
   	int t;
   	
   	cin>>t;
   	
   	while(t--)
   	{
   		int a,b,c;
   		cin>>a>>b>>c;
   		
   		if ((a*a+b*b == c*c) || (a*a+c*c == b*b) || (b*b+c*c == a*a))
   		{
   			if(gcd(a,gcd(b,c)) == 1) cout<<"YES\n";
   			else cout<<"NO\n";
   		}
   		
   		else cout<<"NO\n";
   		
   	}
   	
   	return 0;
}
