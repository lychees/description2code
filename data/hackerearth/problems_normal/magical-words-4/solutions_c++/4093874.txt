#include <bits/stdc++.h>

using namespace std;

int ans[1005][1005];

bool is_palindrome(char a[],int i,int j)
{
    while(i < j)
    {
        if (a[i] != a[j])
            return false;
        i++;
        j--;
    }

    return true;
}

int main()
{
	int tc;
	cin >> tc;

	while(tc--){
    char a[1005];
    scanf("%s",a);

    int n = strlen(a);

    for (int i = 0;i < n;i++)
        for (int j = 0;j <n;j++)
            ans[i][j] = -1;

    for (int i = 0;i < n;i++)
        ans[i][i] = 1;

    long long int sum = n;

    for (int i = 0;i < n-1;i++)
    {
        if (a[i] == a[i+1]) {
            ans[i][i+1] = 1;
            sum += 4;
        }
        else
            ans[i][i+1] = 0;
    }

    int j;

    for (int l = 3;l <= n;l++)
    {
        for (int i = 0;i < n;i++)
        {
            j = min(i + l -1,n);

            if (a[i] == a[j] && ans[i+1][j-1] == 1)
            {
                ans[i][j] = 1;
                sum += pow(l,2);
            } else
                ans[i][j] = 0;

            /*if (!is_palindrome(a,i,j))
                ans[i][j] = 0;

                ans[i][j] = 1;
                sum += pow(l,2);*/
        }
    }


    printf("%ld\n",sum);// << endl;
	}

    return 0;
}
