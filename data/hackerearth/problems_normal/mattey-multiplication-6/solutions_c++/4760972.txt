#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--) {
		vector <int> v;
		long long a,n,m,p,q=1;
		scanf("%lld%lld",&n,&m);
		int i=0,c,j;
		p=m;
		while(p!=1) {
			p=p/2;
			i++;
		}
		j=i;
		while(i--) {
			q=q*2;
		}
		while(q!=0) {
			a=q&m;
			if(a) {
				v.push_back(j);
			}
			j--;
			q=q/2;
		}
		vector <int> :: const_iterator it;
		for(it=v.begin();it!=v.end();it++) {
			if((++it)==v.end()) {
				it--;
				cout<<"("<<n<<"<<"<<*it<<")\n";
				break;
			}
			it--;
			cout<<"("<<n<<"<<"<<*it<<") + ";
		}
	}
	return 0;
}
