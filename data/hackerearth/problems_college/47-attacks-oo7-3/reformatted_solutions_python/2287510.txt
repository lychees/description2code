tt = input()
while (tt > 0):
    tt -= 1
    n = input()
    ans = 0
    if (n == 0):
        print(0)
    elif (n > 120):
        print("UNABLE TO FIRE")
    else:
        ans = 1
        for i in range(1, n+1):
            if (i % 2 == 0):
                ans = ans+2
            else:
                ans = ans*3
        print(ans)
