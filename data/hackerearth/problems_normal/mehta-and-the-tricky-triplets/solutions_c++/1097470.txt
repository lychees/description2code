#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000007
#define total 5000005
#define M 1000000000001
#define NIL 0
#define MAXN 200001
#define EPS 1e-5
#define INF (1<<28)
typedef unsigned long long int uint64;
typedef long long int int64;
int cnt[20];
int main(){
	int n,i;
	int64 x;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>x;
		int cnt2=0;
		int cnt3=0;
		int cnt5=0;
		int cnt7=0;
		while(x){
			int rem=x%10;
			if(rem==2)
			cnt2=1;
			if(rem==3)
			cnt3=1;
			if(rem==5)
			cnt5=1;
			if(rem==7)
			cnt7=1;
			x=x/10;
		}
		for(int j=1;j<16;j++){
		int ret=1;
		if(j&(1<<0))
		ret&=cnt2;
		if(j&(1<<1))
		ret&=cnt3;
		if(j&(1<<2))
		ret&=cnt5;
		if(j&(1<<3))
		ret&=cnt7;
		cnt[j]+=ret;
		}
	}
	int64 ans=0;
	for(int j=1;j<16;j++){
		int64 tmp=cnt[j];
		tmp=tmp*(tmp-1)*(tmp-2);
		tmp/=6;
		i=bitcnt(j);
		if(i&1)
		ans+=tmp;
		else
		ans-=tmp;
	}
	cout<<ans;
	return 0;
}
