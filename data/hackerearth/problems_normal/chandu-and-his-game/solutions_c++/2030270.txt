#include <bits/stdc++.h>

using namespace std;

long long int x[100001],y[100001];
int main()
{
    long long int t,n,a,b,k,i,j,sum ,sum1 ,sum2 ,sum3,sum4,sums,ans,mov = 0;

    scanf("%lld %lld", &n, &k);

    for(i = 0; i < k; i++){
        scanf("%lld", &x[i]);

    }
    for(i = 0; i < k; i++){
        scanf("%lld", &y[i]);

    }

    for(i = 0; i < k; i++){

            sum1 =  x[i] + y[i] - 2;
             sum2 =  2*n -x[i] - y[i];
              sum3 =  x[i] -1 + n - y[i] ;
               sum4 =  n -x[i] -1+ y[i];
               sum = min(sum1,sum2);
    sums = min(sum4,sum3);
    ans = min(sum,sums);

    mov = mov + ans;

    }


    cout << mov << endl;
    return 0;
}
