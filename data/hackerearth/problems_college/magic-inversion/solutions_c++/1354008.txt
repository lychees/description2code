#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int n,a[1000001],b[1000001],cnt,flg;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
    	cin>>b[i];
    }
    cnt=0;
    for(int i=0;i<n;i++)
    {
    	if(a[i]>b[i]) cnt++;
    }
    flg=0;
    if(cnt%2==0||cnt==0) flg=1;
    else{
    	for(int i=3;i<=sqrt(cnt);i++)
    	{
    		if(cnt%i==0) 
    		{
    			flg=1;
    			break;
    		}
    	}
    }
    if(flg) cout<<"SIMPLE INVERSION"<<endl;
    else cout<<"MAGIC INVERSION\n";
    return 0;
}
