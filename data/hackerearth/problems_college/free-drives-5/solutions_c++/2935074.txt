#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i;
    long long int r;
    cin>>t;
    while(t--)
    {
    	r=0;
    	cin>>n;
    	int p=n-1;
    	for(i=1;i<=n;i++)
    	{
    	r=r+p;
    	p=p-1;
    	}
    	cout<<r<<endl;
    }
    return 0;
}
