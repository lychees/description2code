#include <iostream>
#include <limits.h>
#include <algorithm>
using namespace std;
int main()
{
        int t;
        cin >> t;
        while(t--)
        {
                int n, arr[20], i, count=0;
                cin >> n;
                for(i=0; i<n; i++)
                        cin >> arr[i];
                long long mini = 1, maxi = 1;
                sort(arr, arr+n);
                int flag=1;
                for(i=0; i<n; i++)
                {
                        if(arr[i]<0)
                                count++;
                        else if(arr[i]>0 || count>=2)
                        {
                                flag=0;
                                break;
                        }
                }
                if(count>=2)
                        flag=0;
                if(flag==1)
                        maxi=arr[n-1];
                else
                {
                        for(i=0; i<n; i++)
                        {
                                if(arr[i]>0)
                                        maxi*=arr[i];
                                else if(arr[i]<0)
                                {
                                        if(maxi<0 && i<=count-1)
                                                maxi*=arr[i];
                                        else if(maxi>0 && i<=count-2)
                                                maxi*=arr[i];
                                }
                        }
                }
                if(count%2==0)
                        count--;
                for(i=0; i<count; i++)
                        mini*=arr[i];
                if(mini==1)
                        mini=arr[i];
                else
                        for(i=0; i<n; i++)
                                if(arr[i]>0)
                                        mini*=arr[i];
                cout << maxi << ' ' << mini << endl;
        }
        return 0;
}
