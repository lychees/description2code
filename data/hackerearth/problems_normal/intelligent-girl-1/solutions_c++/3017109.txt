#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    string S;
    cin>>S;
    int n = S.length();
    int mapping[n];
    memset(mapping, 0, sizeof(mapping));
    
    for (int i=0;i<n;i++) {
    	if (!((S[i]-int('0'))&1))
    		mapping[i]=1;
    }
    
    for (int i=n-1;i>0;i--) {
    	mapping[i-1]+=mapping[i];
    }
    
    for (int i=0;i<n;i++)
    	cout<<mapping[i]<<" ";
    
    
    return 0;
}
