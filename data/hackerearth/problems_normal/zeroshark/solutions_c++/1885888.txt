#include <bits/stdc++.h>

using namespace std;

long long int MOD=1000000007;
long long int nomorethantwozeroestogether(int prev, int index, int done);

long long int cache[1000005][2][2];
int n;

int main()
{
        int T;
        cin>>T;
        int i, j, k;
        while(T--)
        {
                cin>>n;
                for(i=0; i<100005; i++)
                {
                        for(j=0; j<2; j++)
                        {
                                for(k=0; k<2; k++)
                                {
                                        cache[i][j][k]=-1;
                                }
                        }
                }
                cout<<nomorethantwozeroestogether(1, 0, false)<<endl;
        }
    return 0;
}

long long int nomorethantwozeroestogether(int prev, int index, int done)
{
        if(index==n && done)
                return 1;
        if(index==n && !done)
                return 0;

        long long int ans;

        if(cache[index][done][prev]!=-1)
        {
                        return cache[index][done][prev];
        }

        if(prev==0 && !done)
        {
                ans=(nomorethantwozeroestogether(0, index+1, !done)%MOD + nomorethantwozeroestogether(1, index+1, done)%MOD)%MOD;
        }
        if(prev==0 && done)
        {
                ans=(nomorethantwozeroestogether(1, index+1, done))%MOD;
        }
        if(prev==1 && !done)
        {
                ans=(nomorethantwozeroestogether(0, index+1, done)%MOD + nomorethantwozeroestogether(1, index+1, done)%MOD)%MOD;
        }
        if(prev==1 && done)
        {
                ans=(nomorethantwozeroestogether(0, index+1, done)%MOD + nomorethantwozeroestogether(1, index+1, done)%MOD)%MOD;
        }

        cache[index][done][prev]=ans;

        return ans;
}
