#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <queue>
#include <algorithm>
#define lli long long int
#define pb push_back
#define mod 1000000007
#define pii pair<int,int>

using namespace std;
lli dp[1000009][3];
int main()
{
    int t,i,j,k,a,b,c,n;
    scanf("%d",&t);
    while(t--)
    {
        dp[0][0]=0;
        dp[0][1]=0;
        dp[0][2]=0;
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
            scanf("%d%d%d",&a,&b,&c);
            dp[i][0]=a+ min(dp[i-1][1] ,dp[i-1][2]);
            dp[i][1]=b+ min(dp[i-1][0] ,dp[i-1][2]);
            dp[i][2]=c+ min(dp[i-1][0], dp[i-1][1]);
        }
        lli ans=min(dp[n][0],min(dp[n][1],dp[n][2]));
        printf("%lld\n",ans);
    }

    return 0;
}
