#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
    	int n,arr[10];
    	cin>>n;
    	if(n%21==0)
    		cout<<"The streak is broken!"<<endl;
    	else
    	{
    		int tmp,k=0,flag=0;
    		while(n>0)
    		{
    			arr[k]=n%10;
    			n=n/10;
    			k++;
    		}	
    		
    		for(int i=k-1;i>0;i--)
    		{
    			if(arr[i]==2 && arr[i-1]==1)
    			{
    			cout<<"The streak is broken!"<<endl;
    			flag=1;
    			}		
    		}
    		if(flag==0)
    		cout<<"The streak lives still in our heart!"<<endl;
    	}
    	
    }
    return 0;
}
