#define MOD 1000000007LL

long long int ppow(long long int i, long long int j){
	long long int res = 1LL;
	while (j){
		if ((j & 1LL)){
			res *= i;
			res %= MOD;
		}
		i *= i;
		i %= MOD;
	}
	return res;
}
