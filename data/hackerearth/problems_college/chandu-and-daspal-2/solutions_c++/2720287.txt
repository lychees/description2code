#include <iostream>
using namespace std;

int gcd(int x,int y){
	if(y==0)	
		return x;
		
	return gcd(y,x%y);
}

int main()
{
    int t,m,n,g,i;
    string s1,s2;
    
    cin>>t;
    while(t--){
    	cin>>m>>n;
    	cin>>s1;
    	
    	if(s1=="Chandu"){
    		s2="Daspal";
    	}else{
    		s2="Chandu";
    	}
    	
    	i=0;
    	while(m!=0 && n!=0){
    		g=gcd(m,n);
    		m-=g;
    		n-=g;
    		i++;
    	}
    	
    	if(i%2){
    		cout<<s1<<endl;
    	}else{
    		cout<<s2<<endl;
    	}
    	
    }
    
    return 0;
}
