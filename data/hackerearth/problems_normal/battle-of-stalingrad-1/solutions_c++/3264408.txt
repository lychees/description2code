#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

long double Area(int x1,int y1,int x2,int y2,int x3,int y3){
	long double result =  ( ( x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2) )/2.0 );
	if(result<0) 
		result = result*(-1);
	return result;	
}

bool check(int x1,int y1,int x2,int y2,int x3,int y3,int x4,int y4){
	long double area = Area(x1,y1,x2,y2,x3,y3);
	long double area1 = Area(x4,y4,x2,y2,x3,y3);
	long double area2 = Area(x1,y1,x4,y4,x3,y3);
	long double area3 = Area(x1,y1,x2,y2,x4,y4);

	return (area == area1 + area2 + area3);
}

int main()
{
	int t;
	cin >> t;
	while(t--){
		int x1,y1,x2,y2,x3,y3,x4,y4;
		cin >> x1 >> y1 >> x2 >> y2 >> x3 >> y3 >> x4 >> y4;
		
		if( check(x1,y1,x2,y2,x3,y3,x4,y4) ){
			cout << "INSIDE";
		}
		else{
			cout << "OUTSIDE";
		}
		cout << endl;
	}
	return 0;
}
