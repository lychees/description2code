#include <iostream>
#include <set>
using namespace std;

int main()
{
    int t;
    string s;
    set <string> se;
    
    cin>>t;
    while(t--){
    	cin>>s;
    	se.insert(s);
    }
    
    set <string> :: iterator it;
    for(it=se.begin();it!=se.end();it++){
    	cout<<*it<<endl;
    }
    
    return 0;
}
