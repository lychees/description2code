#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int a[n];
    for(int i=0;i<n+1;i++){
    	cin>>a[i];
    }
  sort(a,a+n);
 int i;
  for( i=0;i<n;i++){
  if(a[i+1]-a[i]>k){
  	cout<<a[i]<<"\n";
  	break;
  }
  if(i==n-1){
  	cout<<a[n-1]<<"\n";
  }
 
 }
    
    return 0;
}
