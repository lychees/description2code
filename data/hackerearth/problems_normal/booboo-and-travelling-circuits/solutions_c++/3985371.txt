//typedef unsigned long long int ull;
typedef unsigned long long int ll;
typedef vector<vector<ll> > mat;

int mod=1000000003;

int mul(ll a,ll b)
{
	ll r=a;
	r*=b;
	if(r>=mod)r%=mod;
	return r;
}

int add(int a,int b)
{
	a+=b;
	if(a>=mod)
	a%=mod;
	return a;
}

ll dp[2005][10005]={0};

int pp[10005],sm[10005];
vii pr[10005];

void seive()
{
	pp[0]=1,pp[1]=1;
	rep(i,1,10004)sm[i]=1e9+7;
	sm[1]=1;
	rep(i,2,10004)
	{
		if(pp[i])continue;
		sm[i]=i;
		for(int j=2*i;j<=10004;j+=i)
		pp[j]=1,sm[j]=min(sm[j],i);
	}
	rep(i,2,10000)
	pr[sm[i]].pb(i);
	
}

int main()

	rep(j,1,m)
	{
		dp[i][j]=min(dp[i][j],dp[i-1][j]+a[j]);
		if(j!=1)
		mu=min(mu,dp[i-1][j]+b[j]);
		
	}
	dp[i][1]=min(dp[i][1],mu);
	rep(j,2,m)
	{
		ll mn1=1e18,mn2=1e18,idx1=-1,idx2=-1;
