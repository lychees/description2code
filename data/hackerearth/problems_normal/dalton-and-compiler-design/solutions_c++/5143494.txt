#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int ar[t];
    for(int i=0;i<t;i++)
    {
        cin>>ar[i];
        if(ar[i]%2==0)
        {
            int a;
            a=ar[i]/2;
            cout<<a<<" "<<a<<endl;
        }
        else if((ar[i]%2)!=0)
        {
            float b;
        b=ar[i]/2;
        cout<<b<<" "<<b+1<<endl;
        }
    }
    return 0;
}
