#include <bits/stdc++.h>

using namespace std;

int main()
{
     long long  int t,n,i,j,l,temp;
  		cin>>t;
  while(t--)
  {
      cin>>n;
     long long int a[n],b[n];
     long long sum=0,c=0;
      for(i=0;i<n;i++)
      {
          cin>>a[i];
          sum=sum+a[i];
      }
      for(i=0;i<n;i++)
      cin>>b[i];
      for(i=0;i<n;i++)
      for(j=0;j<n;j++)
          if(b[i]>b[j])
              {
                  temp = b[i];
                  b[i] = b[j];
                  b[j] = temp;
              }
      while(sum>0)
      {
          sum=sum-b[c];
          c++;
      }
      cout<<c<<endl;
  }
  return 0;
}
