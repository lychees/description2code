#include<iostream>
using namespace std;
int main()
{
    long long int t,n,k;
    cin>>t;
    while(t--)
    {
    k=0;

        cin>>n;
       long long int num[n];
        for( long long int i=0;i<n;i++)
            cin>>num[i];
        for( long long int i=0;i<n;i++)
        {
            if(num[i]>num[i+1])
            {
                k=1;


            }
        }
        if(k==1)
            cout<<"NO"<<endl;
        else
            cout<<"YES"<<endl;
    }
    return 0;


}
