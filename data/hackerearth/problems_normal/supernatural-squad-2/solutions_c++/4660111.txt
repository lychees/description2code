void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}


int k;
ll dp[201][201];

ll cal(int n, int h)
{
    if(n==0) return 1;
    if(n<k || h<k) return 0;
    if(dp[n][h]!=-1) return dp[n][h];
    ll ans=cal(n-h,h)+cal(n,h-1);
    return dp[n][h]=ans;
}

    }

}
