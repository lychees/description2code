#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,boys,girls,total;
    char jhool[]="Little Jhool wins!";
    char teacher[]="The teacher wins!";
    cin>>t;
    while(t--)
    {
        cin>>total>>boys>>girls;
        if(abs(girls-boys)>1)
            cout<<jhool<<endl;
        else
            cout<<teacher<<endl;
    }
}
