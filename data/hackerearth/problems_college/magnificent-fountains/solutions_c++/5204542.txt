#include<bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>
using namespace std;
namespace mp = boost::multiprecision;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
cin>>n;
mp::cpp_int a[n];
mp::cpp_int lcm = 1;
for(int i=0;i<n;i++)
cin>>a[i];
for(int i=0;i<n;i++)
{
lcm = (((lcm*a[i]))/__gcd(lcm,a[i]));
}
cout<<lcm%1000000007<<"\n";
}
return 0;
}
