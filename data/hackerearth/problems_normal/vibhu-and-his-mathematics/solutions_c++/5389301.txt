#include <bits/stdc++.h>
#define ll long long int
#define MAX 1000006
#define MOD 1000000007

using namespace std ;

ll f[MAX] ;

void gen()
{
	f[0] = 0 ;
	f[1] = 1 ;
	f[2] = 2 ;
	
	for(int i = 3 ; i < MAX ; i++)
	{
		f[i] = f[i - 2] * (i - 1) + f[i - 1]  ;
		
		if(f[i] >= MOD)
			f[i] %= MOD ;
	}
}

int main()
{
	gen() ;
	
	int T ;
	
	cin >> T ;
	
	while(T--)
	{
		int N ;
		
		cin >> N ;
		
		cout << f[N] << '\n' ;
	}
	
	return 0 ;
}
