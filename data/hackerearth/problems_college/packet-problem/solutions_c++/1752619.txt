#include <iostream>
#include <vector>

using namespace std;

typedef long long int Lint;

int main()
{
	ios_base::sync_with_stdio( false );
	Lint t, n, i;
	
	vector<Lint> A( 92 );
	A[ 0 ] = 0;
	A[ 1 ] = 1;
	
	for( i = 2;i <= 91;i++ )
		A[ i ] = A[ i - 1 ] + A[ i - 2 ];
	
	cin >> t;
	
	while( t-- ) {

		cin >> n;
		
		if( n == 1 )
			cout << "0 1\n";
		else {
			cout << A[ n - 1 ] << " " << A[ n ] << endl;
		}
	}
	
	return 0;
}
