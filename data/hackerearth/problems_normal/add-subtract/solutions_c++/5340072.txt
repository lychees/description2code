 #include<bits/stdc++.h>
  using namespace std;

int arr[101];
int main(){
	int t,n,i,k,j,cost;
	
	cin>>t;
	while(t--){

                cin>>n>>k;
		for(i=0;i<n;i++){
			cin>>arr[i];
		}
		int ans=1e9;
		for(i=1;i<=100;i++){
			vector<int>v;
			for(j=0;j<n;j++){
				if(i>=arr[j]){
				cost=(i-arr[j])*3;
				}
				else{
					cost=(arr[j]-i)*5;
				}
				v.push_back(cost);
			}
			sort(v.begin(),v.end());
			cost=0;
			for(j=0;j<k;j++){
				cost+=v[j];
			}
			ans=min(ans,cost);
		}
		cout<<ans<<endl;
	}
	
	return 0;
}
