def gcd(i,j):
	if(j==0):
		return i
	else:
		return gcd(j,i%j)



def isCoprime(a,b,c):
	if gcd(a,b)==1 and gcd(b,c)==1 and gcd(a,c)==1:
		return True
	return False


t=input()
while t>0:
	arr=[int(x) for x in raw_input().split()]
	h=max(arr)
	arr.remove(h)
	if((h*h==(arr[0]*arr[0]+arr[1]*arr[1]))and isCoprime(h,arr[0],arr[1])):
		print 'YES'
	else:
		print 'NO'
	t=t-1
