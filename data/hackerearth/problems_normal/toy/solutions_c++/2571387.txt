#include <bits/stdc++.h>

using namespace std;

int arr[100000+10];
int mark[10000+10];

int main()
{
    int n;
    cin >> n;
   for (int i = 0 ; i < n ; i++)cin >> arr[i];
   memset(mark , 0 , sizeof(mark));
   sort(arr , arr + n);
   int no = 0;
   for (int i = 0 ; i < n ; i++){
    if (!mark[arr[i]]){
            mark[arr[i]] = 1;
            no++;
        for (int j = arr[i] ; j <= arr[i]+4 ; j++){
            mark[j] = 1;
        }
    }
   }
   cout << no << "\n";
   return 0;

}
