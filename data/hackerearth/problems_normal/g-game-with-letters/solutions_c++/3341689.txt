int main(){

	int T, N;
	cin>>T;
	
	while(T--){
		string S;
		map<char,int>mp;
		vector<pair<int,pii> >V;
		cin>>N>>S;
		for(int i=0;i<S.size();++i){
			mp[S[i]]++;
		}
		rep(i,1,N){
			cin>>S;
			int sc = 0;
			for(int j=0;j<S.size();++j){
				sc += mp[S[j]];
			}
			V.push_back({sc,{-S.size(),-i}});
		}
		sort(V.begin(),V.end());
		cout<<V[N-1].second.second*-1<<endl;
	}
