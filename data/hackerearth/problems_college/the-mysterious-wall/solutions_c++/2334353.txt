#include <iostream>
#include<cstdio>
#include<string>
using namespace std;
int a1[26],a2[26];
int main()
{
    int t,going,coming;
    long i,j,k,count,len1,len2;
    string tar,pat;
    int flag,ans;
    scanf("%d",&t);
    while(t--){
        flag = 0;
        for(i=0;i<26;i++){
            a1[i] = 0;
            a2[i] = 0;
        }
        count = 0;
        cin>>tar;
        cin>>pat;
        ans=0;
        len2 = pat.length();
        len1 = tar.length();
        if(len2 > len1)
            printf("NO\n");
        else{
        for(i=0;i<len2;i++){
            a2[pat[i]-97]++;
        }
        for(i=0;i<26;i++){
            if(a2[i])
                ans++;
        }
       // printf("Ans: %d\n",ans);
        for(i=0;i<len2;i++){
            k = tar[i] - 97;
            a1[k]++;
            if(a1[k] == a2[k])
                count++;

        }
        //printf("Out count: %d\n",count);
        if(count == ans)
            printf("YES\n");
        else{
            for(i=len2;i<len1;i++){
                going = tar[i-len2] - 97;
                a1[going]--;
                if(a1[going] == a2[going] - 1)
                    count--;
                coming = tar[i] - 97;
                a1[coming]++;
                if(a1[coming] == a2[coming])
                    count++;
                if(count == ans)
                {
                    flag = 1;
                    break;
                }
            }
            if(flag == 1)
                printf("YES\n");
            else
                printf("NO\n");
            flag = 0;
        }
        }
    }
    return 0;
}
