int gcd(int n1,int n2)
{
	int a,b,c;
	if(n1>=n2){
		a=n1;
		b=n2;
	}
	else{
		b=n1;
		a=n2;
	}
	while(1){	
		c=a%b;
		if(c==0){
			break;
		}
		a=b;
		b=c;
	}
	return b;
}
