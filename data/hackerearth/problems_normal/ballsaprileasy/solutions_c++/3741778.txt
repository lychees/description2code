#include <iostream>
#include <algorithm>

using namespace std;

int main(void){
    
    long long n, sum=0, i=0, b=1, j=0;
    cin>>n;
    
    int a[n];
    
    for(i = 0; i<n; i++)
        cin>>a[i];
        
    sort(a, a+n);
    
    for(i=0; i<n; i++){
        
        for(j=b; ; j++){
            
            if(a[i]<=j*j){
                sum = sum + ( j*j - a[i] );
                b=j+1;
                break;
            }
            else{
                sum = sum + j*j;
                continue;
            }
        }
        
    }
    
    cout<<sum;

    return 0;
}
