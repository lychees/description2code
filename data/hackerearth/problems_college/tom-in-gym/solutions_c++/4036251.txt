#include <iostream>
#include <boost/multiprecision/cpp_int.hpp>
using namespace std;
using namespace boost::multiprecision;

int main()
{
    int128_t t,n,i,j,k;
    cin>>t;
    while(t--){
    	cin>>n;
    	k = 1;
    	for(i=1;i<n;i++){
    		k *= (2*(i+1)-1);
    	}
    	//k++;
    	cout<<k<<endl;
    }
    return 0;
}
