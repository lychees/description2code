#include <iostream>
#include <string>
void getLargest(int &grow, int &max)
{
if (grow > max)
max = grow;
grow = 0;
}
int main()
{
int days;
std::cin >> days;

int maxStreak(0);
int dayStreak(0);
int maxContinuous(0);
int continuous(0);
while(days--)
{
std::string log;
std::cin >> log;
for (char action: log)
if (action == 'C')
{
++dayStreak;
++continuous;
}
else
{
getLargest(dayStreak, maxStreak);
getLargest(continuous, maxContinuous);
}
getLargest(dayStreak, maxStreak);
}
getLargest(continuous, maxContinuous);

std::cout << maxStreak << ' ' << maxContinuous;
return 0;
}
