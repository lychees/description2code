#include<bits/stdc++.h>
using namespace std;
#define M 100005
char str[M];
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    cin>>str;
    int n=strlen(str);
    int ans=-1;
    for(int i=0;i<n/2;i++)
    {
      if(str[i]!=str[n-i-1])
      {
         if(str[i+1]==str[n-i-1])
          ans=i;
         else if(str[i]==str[n-i-2])
          ans=n-i-1;
         break;
      }
    }
    cout<<ans<<endl;

  }
  return 0;
}
