#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int N,Q; ll L,R;
    scanf("%d",&N);
    vector<ll> P(N);
    for(int i = 0; i < N; i++){
    	scanf("%lld",&P[i]);
    }
    sort(P.begin(),P.end());
    scanf("%d",&Q);
    while(Q--){
    	scanf("%lld%lld",&L,&R);
    	printf("%d\n",upper_bound(P.begin(),P.end(),R) - lower_bound(P.begin(),P.end(),L));
    }
    return 0;
}
