#include <cstdio>
#include <algorithm>
#include <vector>
#include <cstdlib>
#include <ctime>
#include <iostream>
#include <string>
#include <math.h>

using namespace std;

int main() {
	long long size;
	cin>>size;
	
	long long arr[size];
	for(long long i=0;i<size;i++)
		cin>>arr[i];

	sort(arr,arr+size);

	long long query;
	cin>>query;
	int temp[1];
	while(query--){
		cin>>temp[0];
		if(binary_search(arr,arr+size,temp[0])){
			cout<<"I Like You"<<endl;
			continue;
		}

		for(long long i=0;i<size;i++){
			for(long long j=i+1;j<size;j++){
				if(arr[i]+arr[j] == temp[0]){
					cout<<"I Like You"<<endl;
								goto b;
				}
			}
		}

		cout<<"I Hate You"<<endl;
		b:;

	}
	return 0;
}
