for i in xrange(int(raw_input())):
    Z, M, N = str(raw_input()).split()
    Z = int(Z)
    M = int(M)
    N = int(N)
    cnt = 0
    for j in xrange(1, int(Z**(0.5))+1):
        if Z % j == 0:
            if j <= M and Z/j <= N:
                cnt += 1
            if j <= N and Z/j <= M and j != Z/j:
                cnt += 1
    print cnt
