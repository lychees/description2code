#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    long test;
    cin>>test;

    while(test)
    {
        test--;
        string s;
        cin>>s;
        int len1 = s.length();

        char a1[len1];
        strcpy(a1, s.c_str());

        cin>>s;
        int len2 = s.length();

        char a2[len2];
        strcpy(a2, s.c_str());

        int
            netLength = len1 + len2,
            firstTraversal = 0,
            counter = 0;

        bool flag = 0;

        while(firstTraversal <= len1)
        {
            char *temp = new char[netLength];

            int ftemp = 0,
                stemp = 0,
                actual = 0;

            while(ftemp < firstTraversal)
            {
                temp[actual++] = a1[ftemp++];
            }

            while(stemp < len2)
            {
                temp[actual++] = a2[stemp++];
            }

            while(ftemp < len1)
            {
                temp[actual++] = a1[ftemp++];
            }

            for(int i=0;i < netLength ;i++)
                {
                    if(temp[i] != temp[netLength-i-1])
                    {
                      flag = 1;
                      break;
                    }
                }

            if(flag)
            {
                flag = 0;
            }
            else
            {
                counter++;;

            }

            firstTraversal++;
        }
        cout<<counter<<"\n";
    }
    return 0;
}
