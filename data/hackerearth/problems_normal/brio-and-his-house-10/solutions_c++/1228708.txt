#include <iostream>
#include <cmath>
#include <cstdio>
#define pi 3.1415

using namespace std;

int main()
{
    int t;
    double a,b,c,prod,root,radius;
    cin>>t;
    while(t--)
    {
            cin>>a>>b>>c;
            root=(a+b+c)*(b+c-a)*(c+a-b)*(a+b-c);
            root=sqrt(root);
            prod=a*b*c;
            radius=prod/root;
            printf("%0.4f\n",pi*radius*radius);


    }

    return 0;
}
