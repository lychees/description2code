#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,max,k;
	cin>>t;
	while(t--)
	{
		cin>>n;
		max=0;
		int bl[n+9],gl[n+9];
		for(i=1;i<=n;i++)
		{
			cin>>k;
			bl[i]=k;
		}
		for(i=1;i<=n;i++)
		{
			cin>>k;
			gl[i]=k;
		}
		int bb[100000]={0},gb[100000]={0};
		int b[100000]={0},g[100000]={0};
		for(i=1;i<=n;i++)
		{
			if(bl[i]!=gl[bl[i]])
			{
				
				bb[i]=gl[bl[i]];
				
				//cout<<"boys "<<i<<"beat"<<bb[i]<<endl;
				b[gl[bl[i]]]++;
				j=b[gl[bl[i]]];
				if(j>max)
				max=j;
				
			}
			
		}
		for(i=1;i<=n;i++)
		{
			if(gl[i]!=bl[gl[i]])
			{
				gb[i]=bl[gl[i]];
				
				g[bl[gl[i]]]++;
				j=g[bl[gl[i]]];
				//cout<<"girls "<<i<<"beat"<<gb[i]<<endl;
				if(j>max)
				max=j;
				
			}
		}
		int s=0;
		for(i=1;i<=100000;i++)
		{
			k=bb[i];
			if(bb[k]==i&&bb[i]!=0&&bb[i]!=i)
			s++;
			}
		for(i=1;i<=100000;i++)
		{
			k=gb[i];
			if(gb[i]==i&&gb[i]!=0&&gb[i]!=i)
			s++;
		}
		
		cout<<max<<" "<<s<<endl;
	
		
	}
	return 0;
}
