#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		ll  int es=0,os=0,mc=0; //even sum,odd sum & steps-count
		while(n!=1)
		{
			if((n%2)==0)
			{
				es+=n;
				n/=2;
			}
			else
			{
				os+=n;
				n=3*n+1;
			}
			mc++;
		}
		os++; //Add for 1
		if((os%mc) == (es%mc))
		{
			printf("Tie\n");
		}
		else if((os%mc) > (es%mc))
		{
			printf("Odd Rules\n");
		}
		else
		{
			printf("Even Rules\n");
		}
	}
	return 0;
}
