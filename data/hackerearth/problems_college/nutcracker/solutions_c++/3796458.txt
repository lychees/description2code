#include <bits/stdc++.h>

using namespace std;


bool a[100001];


void fig()
{
	int i,j;
	
	a[1] = 1;
	a[2] = 0;
	a[3] = 1;
	a[4] = 1;
	
	for(i=5;i<100001;i++)
	{
		if(a[i-1] == 0 || a[i-4] == 0)
		{
			a[i] = 1;
		}
		else
			a[i] = 0;
	}
}
int main()
{
	long long int i,j,k,l,m,n,t;
	
	cin >> t;
	
	fig();
	
	
	while(t--)
	{
		cin >> n;
		
		if(a[n] == 1 )
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
}
