using namespace std;

int n;
int a[105];		//maximum element value is 1000
int dp[105][100005];	//stores the left most index
						//greater than i 
						//such that sum is s
int csum[105]={0};	//cumulative sum
vector<int> ans;

int main()
{
    int i,j,k,m;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
    	scanf("%d",&a[i]);
    	csum[i]=csum[i-1]+a[i];
    }
    for(j=0;j<=csum[n];j++)
    	dp[n+1][j]=n+1;
    for(j=1;j<=csum[n];j++)
    {
    	k=n;
    	for(i=n;i>=1;i--)
    	{
    		dp[i][j]=dp[i+1][j];
    		while((csum[k]-csum[i-1])>j)
    			k--;
    		if((csum[k]-csum[i-1])==j)
    			dp[i][j]=k;
    	}
    }
    m=0;
    for(j=1;j<=csum[n];j++)
    {
    	i=1;
    	k=0;
    	while(i<=n)
    	{
    		if(dp[i][j]!=(n+1))
    			k++;
    		i=dp[i][j]+1;
    	}
    	if(k>m)
    	{
    		ans.clear();
    		m=k;
    	}
    	if(m==k)
    		ans.push_back(j);
    }
    printf("%d\n",m);
    for(i=0;i<(int)ans.size();i++)
    	printf("%d ",ans[i]);
    return 0;
}
