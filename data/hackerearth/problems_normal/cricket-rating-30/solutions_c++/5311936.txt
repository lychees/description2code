#include<iostream>
using namespace std;

int max(int a,int b)
{
    if(a>b)
    return a;
    else
    return b;
}
int sprayarray(int arr[],int n)
{
    int max_ending=0,max_sofar=0;
    for(int i=0;i<n;i++)
    {
        max_ending+=arr[i];
        if(max_ending<0)
        {
            max_ending=0;
        }
        if(max_sofar<max_ending)
        {
            max_sofar=max_ending;
        }
    }
    return max_sofar;
}
int arraymax(int arr[],int n)
{
    int currmax=arr[0],finalmax=arr[0];
    for(int i=1;i<n;i++)
    {
        currmax=max(arr[i],currmax+arr[i]);
        finalmax=max(finalmax,currmax);
    }
    return finalmax;
}
