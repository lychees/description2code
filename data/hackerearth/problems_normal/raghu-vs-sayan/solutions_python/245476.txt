t=input()
while t>0:
    ctr=0
    ctr1=0
    a,b,n=map(int,raw_input().split())
    x=map(int,raw_input().split())
    x.sort()
    for i in range(0,n):
        if x[i]<=a:
            a=a-x[i]
            ctr=ctr+1
        if x[i]<=b:
            b=b-x[i]
            ctr1=ctr1+1
    if ctr>ctr1:
        print "Raghu Won"
    elif ctr<ctr1:
        print "Sayan Won"
    else:
        print "Tie"
    t=t-1
