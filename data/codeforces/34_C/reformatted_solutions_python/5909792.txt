import sys
import itertools
from itertools import groupby
from operator import itemgetter
res = ""
data = sorted(set([int(i) for i in sys.stdin.readline().split(',')]))
for k, g in groupby(enumerate(data), lambda (i, x): i-x):
    a = map(itemgetter(1), g)
    if len(a) >= 2:
        res = res+str(a[0])+"-"+str(a[len(a)-1])+","
    else:
        res = res+str(a[0])+","
print res[:-1]
