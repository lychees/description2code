#include <iostream>
#include <set>
using namespace std;

int main()
{
    int t;
    long int n,i;
    long long int num;
    set<long long int> s;
    
    cin>>t;
    while(t--){
    	cin>>n;
    	s.clear();
    	for(i=0;i<n;i++){
    		cin>>num;
    		s.insert(num);
    	}
    	cout<<s.size()<<endl;
    }
    return 0;
}
