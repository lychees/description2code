#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
#include<map>

using namespace std;

#define bs 1000000007

int fact[1<<20],invf[1<<20];

int tests,n,m,ans,ways;
int C;
long long pw(long long a,long long b)
{
 if (b==0)
  return 1;
 if (b%2) return a*pw(a,b-1)%bs;
 return pw(a*a%bs,b/2);
}


long long inv(long long x)
{
 return pw(x,bs-2);
}


long long c(long n,long m)
{
     if (n<m)return 0;
     if (m<0)return 0;
     
     long long res=fact[n];
     res*=invf[m];
     res%=bs;
     res*=invf[n-m];
     return res%bs;
}
int main(){
ios_base::sync_with_stdio(0);
//cin.tie(0);
fact[0]=1;

for (int i=1;i<=100000;i++)
 fact[i]=1ll*fact[i-1]*i%bs;
 
 
 
for (int i=0;i<=100000;i++)
 invf[i]=inv(fact[i]);
 
 
cin>>tests;

for (;tests;--tests)
{
 cin>>n>>m>>C;
 if (C==0)
 {
  if (m==n)cout<<1<<endl;
  else cout<<0<<endl;
  continue;
 }
 cout<<c(m+1,m-(C-1))*c(n-m-1,C-1)%bs<<"\n";
}
return 0;}
