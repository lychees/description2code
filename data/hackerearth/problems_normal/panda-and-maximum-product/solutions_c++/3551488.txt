#include<bits/stdc++.h>
using namespace std;
main()
{
     long long int n,i,a[100000],b[100010],p,ans1,ans2;

    cin>>n;
    int k=0;
    int r=0;


    for(i=0;i<n;i++)
    {
        cin>>p;
        if(p>=0)
        {
            a[k]=p;
            k=k+1;

        }
        else
        {
            b[r]=p;
            r=r+1;
        }
    }
    sort(a,a+k);
    sort(b,b+r);
    ans1=b[0]*b[1];
    ans2=a[k-2]*a[k-1];
    ans1=max(ans1,ans2);
    cout<<ans1;
}
