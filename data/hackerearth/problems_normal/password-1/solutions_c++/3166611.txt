#include <iostream>
#include <cstring>
using namespace std;
 
int main()
{	
	int n,i,j,c,r,d,k,l,m=0,p=0;
	char a[100][14];int g[100];
	string s;
	cin>>n;
	for(r=0;r<n;r++)
	{
	cin>>s;
	int f=s.length();
	for(i=0;i<f;i++)
	{
		a[p][i]=s[i];
	}
	g[r]=f;
	p++;
	}
	
	for(i=0;i<n;i++)
	{
		for(j=1;j<n;j++)
		{
			if(g[i]==g[j])
			{
				
			A:	for(k=0;k<g[i];k++)
				{
					if(a[i][k]==a[j][g[i]-k-1])
					{
						m++;
					}
					else 
					{
						break;
					}
			
				}
				if(m==g[i])
				{
					cout<<m<<" "<<a[i][m/2]<<endl;goto B;
				}
				m=0;
			}
			else {j++;goto A;
			};
		}
	} B:;

    return 0;
}
