#include <iostream>
using namespace std;
int readInt () {
	bool minus = false;
	int result = 0;
	char ch;
	ch = getchar();
	while (true) {
		if (ch == '-') break;
		if (ch >= '0' && ch <= '9') break;
		ch = getchar();
	}
	if (ch == '-') minus = true; else result = ch-'0';
	while (true) {
		ch = getchar();
		if (ch < '0' || ch > '9') break;
		result = result*10 + (ch - '0');
	}
	if (minus)
		return -result;
	else
		return result;
}
typedef unsigned long long ll;
int main()
{
    //cout << "Hello World!" << endl;
    int n,m;
    n=readInt();
    m=readInt();
    int a[n][m];
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<m;j++)
    	{
    		a[i][j]=readInt();
    	}
    }
    for(int j=0,i=0;j<m,i<n;j++)
    {
    	int k=j,p=i;
    	while(k>=0&&p<n)
    	{
    		cout<<a[p][k]<<" ";
    		p++;k--;
    	}
    	if(j==m-1){i++;j=m-2;}
    cout<<"\n";
    }
    return 0;
}
