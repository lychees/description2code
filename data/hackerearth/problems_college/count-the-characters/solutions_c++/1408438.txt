#include <iostream>
#include<string>
using namespace std;

int main()
{
	string str;
	int test,j,i;
	cin>>test;
    int c[28]={};
    cin>>str;
    for(i=0;i<str.length();i++)
    {
    	if(c[str[i]-'a']==0)
    	{
    		c[str[i]-'a']++;
    	}
    }
    for(i=1;i<test;i++)
    {
    	
    	cin>>str;
    	for(j=0;j<str.length();j++)
    	{
    		if(c[str[j]-'a']==i)
    		{
    			c[str[j]-'a']++;
    		}
    	}
    }
    int cnt=0;
    for(i=0;i<26;i++)
    {
    	if(c[i]==test)
    	cnt++;
    }
    cout<<cnt;
    return 0;
}
