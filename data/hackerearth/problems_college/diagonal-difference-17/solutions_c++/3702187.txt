#include <iostream>
using namespace std;

int main()
{
	int n,d,i,j,tot=0,tot1=0;
    cin >>n;
    int a[n][n];
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<=n;j++)
    	{
    		cin>>a[i][j];
    		if(i==j)
    		{
    			tot=tot+a[i][j];
    		}
    		 if(i+j==n+1)
    		{
    			tot1=tot1+a[i][j];
    		//	cout<<tot1<<endl;
    		}
    	}
    	
    }
    d=tot-tot1;
    if(d>0)
    {
    	cout<<d;
    }
    else
    {
    	cout<<-1*d;
    }
    return 0;
}
