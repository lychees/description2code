#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    int max1=0;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        if(max1<a[i])
            max1=a[i];
    }
    if(max1+1 <= n)
        cout<<"Yes\n";
    else
        cout<<"No\n";
    return 0;
}
