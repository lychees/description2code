#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,n,a;
    	map<int,int> mp;
    	cin>>n;
    	for(i=1;i<=n;i++)
    	{
    		cin>>a;
    		mp[a]++;
    	}
    	for(map<int,int>::iterator it=mp.begin();it!=mp.end();it++)
    	{
    		if(it->second==1)
    		{
    			cout<<it->first<<"\n";
    			break;
    		}
    	}
    }
    return 0;
}
