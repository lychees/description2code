#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    int t,n;
    cin>>t;
    long long int inp,sum;
    vector<long long int>v;
    while(t--)
    {
    	cin>>n;
    	sum=0;
    	int maxy=0;
    	v.resize(n);
    	for(int i=0;i<n;i++)
    	{
    		cin>>v[i];
    		if(maxy<v[i])
    		{
    			maxy=v[i];
    		}
    	}
    	sort(v.begin(),v.end());
    	int p=n-1;
    	for(int i=n-1;i>=0;i--)
    	{
    		sum=sum+(v[i]*p);
    		
    		p=p-2;
    	}
    	sum=sum%mod;
    	cout<<(sum*maxy)%mod<<endl;
    	v.clear();
    }
    
}
