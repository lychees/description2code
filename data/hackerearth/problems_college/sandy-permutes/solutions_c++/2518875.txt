#include <iostream>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int a[n+1];
    for(int i=1;i<=n;i++)
    cin>>a[i];
    
    int pos[n+1];
    for(int i=1;i<=n;i++)
    {
    	pos[a[i]]=i;
    }
    int i=1,j=1;
    while(k>0 && i<=n)
    {
     if(a[i]==n-i+1)
     i++;
     else
     { 
     swap(a[i],a[pos[n-i+1]]);
     //swap(pos[a[i]],pos[n-i+1]);
  /// cout<<n-i+1<<" "<<pos[n-i+1]<<endl;
    pos[a[pos[n-i+1]]]=pos[n-i+1];
    //cout<<a[i]<<" "<<pos[a[i]]<<endl;
     i++;
    // cout<<n-i+1<<" "<<pos[n-i+1]<<endl;
     k--;

     }
    
    }
    for(int i=1;i<=n;i++)
    cout<<a[i]<<" ";
    cout<<endl;
    return 0;
}
