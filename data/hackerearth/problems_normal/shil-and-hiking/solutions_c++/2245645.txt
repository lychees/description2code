typedef unsigned long long int ull;

int main()
{ios_base::sync_with_stdio(false);
int n,ans=0;
cin>>n;
double x[1005],y[1005];
rep(i,0,n-1)cin>>x[i]>>y[i];
rep(i,0,n-2)
{
	double mn=(y[i]-y[i+1])/(x[i+1]-x[i]);ans++;
	rep(j,i+2,n-1)
	{
		if(y[j]>=y[i]-mn*(x[j]-x[i]))
		ans++;
		mn=min(mn,(y[i]-y[j])/(x[j]-x[i]));
	}
}
ol(ans);
return 0;
}
