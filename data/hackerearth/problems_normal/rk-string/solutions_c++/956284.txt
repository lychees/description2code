#include <iostream>
#include <vector>
#include <cstdio>
#include <string>

#define ll long long

using namespace std;

int main()
{
    ll t,m,n;
    scanf("%lld",&t);
    string temp;
    while(t--)
    {
    	scanf("%lld %lld",&m,&n);
    	cin>>temp;
    	vector<ll> dp1(1e6,0),dp2(1e6,0);
    	dp1[0]=1,dp2[0]=1;
    	ll r=0,k=0,ans1=0,ans2=0;
    	for(ll i=0;i<temp.length();i++)
    	{
    		if(temp[i]=='R')
    			r++;
    		if(r>=m)
    			ans1 += dp1[r-m];
    		dp1[r]++;
    		if(temp[i]=='K')
    			k++;
    		if(k>=n)
    			ans2 += dp2[k-n];
    		dp2[k]++;
    	}
    	printf("%lld %lld\n",ans1,ans2);
    }
    return 0;
}
