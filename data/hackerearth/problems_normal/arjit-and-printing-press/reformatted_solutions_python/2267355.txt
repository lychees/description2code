T = int(raw_input())
for _ in xrange(T):
    W = raw_input()
    R = raw_input()
    R = sorted(R)
    W = list(W)
    i = 0
    j = 0
    for i in xrange(len(W)):
        if W[i] > R[j]:
            W[i] = R[j]
            j += 1
            if j == len(R):
                break
    print "".join(W)
