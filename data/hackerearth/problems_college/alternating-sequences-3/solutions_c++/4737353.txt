#include <iostream>
using namespace std;
long long n,i,ans,dp[10000],a[10000],j;
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i],dp[i]=1;
    ans=1;
    for(i=1;i<n;i++)
    {
    	for(j=0;j<i;j++)
    	{
    		if(a[i]>0)
    		{
    			if(a[j]<0&&abs(a[j])<abs(a[i]))
    			dp[i]=max(dp[i],dp[j]+1);
    		}
    		else
    		{
    			if(a[j]>0&&abs(a[j])<abs(a[i]))
    			dp[i]=max(dp[i],dp[j]+1);
    		}
    		ans=max(ans,dp[i]);
    	}
    }
    cout<<ans<<endl;
    return 0;
}
