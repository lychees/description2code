n = raw_input()
noQues = int(n.split()[0])
maxDifficulty = int(n.split()[1])

arrStr = raw_input()
arr = arrStr.split()
arr = [int(x) for x in arr]
count = 0
skip = 0
for i in arr:
    if i > maxDifficulty:
        skip = skip + 1
    else:
        count = count + 1
    if skip >= 2:
        break
