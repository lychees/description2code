#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;


int main()
{
    int t,n;
    lli g,tmp;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %d",&g,&n);
        tmp=2*g;
        n--;
        while(n--)
        {
            printf("%lld ",tmp);
           // tmp+=g;
        }
        printf("%lld",tmp+g);
        printf("\n");
    }
    return 0;
}
