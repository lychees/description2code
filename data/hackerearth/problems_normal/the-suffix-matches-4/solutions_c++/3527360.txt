
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pb push_back
#define mp make_pair

int Z[200005];

void z(char a[]) {
    int n = strlen(a);
    Z[0]=-1;
    int l,r;
    l = r = 0;
    for(int i=1;i<n;++i) {
	if(i>r) {
	    l = r = i;
	    while(r<n && a[r]==a[r-l]) {
		r++;
	    }
	    r--;
	    Z[i] = r-l+1;
	}
	else {
	    int k = i-l;
	    if(Z[k] < r-i+1) {
		Z[i] = Z[k];
	    }
	    else {
		l = i;
		while(r<n && a[r]==a[r-l]) {
		    r++;
		}
		r--;
		Z[i] = r-l+1;
	    }
	}
    }
}

int main()
{
    char s[200005];
    sstr(s);
    char b[200005];
    int n = strlen(s);
    for(int i=0,j=n-1;j>=0;--j,++i) {
	b[i] = s[j];
    }
    z(b);
    int q;
    si(q);
    while(q--) {
	int i;
	si(i);
	i--;
	if(Z[n-1-i]==-1) {
	    pri(n); nl;
	}
	else {
	    pri(Z[n-1-i]); nl;
	}
    }
    return 0;
}
