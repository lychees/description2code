#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
  	int n,k,a[1000001],c,i;
  	scanf("%d%d",&n,&k);
  	for(i=0;i<n;i++)
  		scanf("%d",&a[i]);
  	sort(a+0,a+n);
  	for(i=0;i<n-1;i++)
  		{
  			if(a[i+1]-a[i]>k)
  				{
  					c=a[i];
  					break;
  				}
  		}
  		if(c==0)
  			printf("32767\n");
  		else
  		cout<<c<<endl;
    return 0;
}
