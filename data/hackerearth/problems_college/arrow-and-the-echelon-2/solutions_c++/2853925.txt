#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long int num;
        scanf("%ld",&num);

        if(num%3 == 0)
        {
            for(long int j = 1; j <= num;j++)
                printf("%d",5);
            printf("\n");
        }
        else if(num%5 == 0)
        {
            for(long int j = 1; j <= num;j++)
                printf("%d",3);
            printf("\n");
        }

        else
        {


            int flag = 0;
            long int k = 0;
            for(long int i=5;num > 0;)
            {
                num = num - 5;
                k++;
                if(num < 0)
                    break;
                if(num%3 == 0)
                {

                    for(long int j = 1; j <= num;j++)
                        printf("%d",5);
                    for(long int j = 1; j <= k*5;j++)
                        printf("%d",3);
                    printf("\n");
                    flag = 1;
                    break;
                }
            }
            if(flag == 0)
                printf("%d\n",-1);
        }
    }

    return 0;
}
