def sortstr(s):
    s = list(s)
    dict1 = {}
    for i in s:
        if s.count(i) in dict1.keys():
            dict1[s.count(i)].append(i)
        else:
            dict1[s.count(i)] = [i]
    st = ''
    temp = dict1.keys()
    temp.sort()
    for i in temp:
        dict1[i].sort()
        st += ''.join(dict1[i])
    return st
