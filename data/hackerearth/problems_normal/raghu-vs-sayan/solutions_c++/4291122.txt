#include<bits/stdc++.h>
using namespace std;
priority_queue<int, std::vector<int>, greater<int>> minHeap;

int main()
	{
		int T;
		cin>>T;
		while(T--)
			{
				int R,S,N;
				cin>>R>>S>>N;
				int x;
				for(int i=0;i<N;i++)
					{
						cin>>x;
						minHeap.push(x);
					}
				int R_cnt=0;
				int S_cnt=0;
				int R_sum=R;
				int S_sum=S;
				while(!(minHeap.empty()))
					{
						int ele=minHeap.top();
						if(ele<=R_sum)
							{
								R_sum=R_sum-ele;
								R_cnt++;
							}
						if(ele<=S_sum)
							{
								S_sum=S_sum-ele;
								S_cnt++;
							}
						minHeap.pop();
					}
				if(R_cnt==S_cnt)
					cout<<"Tie"<<endl;
				else if(R_cnt>S_cnt)
					cout<<"Raghu Won"<<endl;
				else
					cout<<"Sayan Won"<<endl;
			}
		return 0;
	}
