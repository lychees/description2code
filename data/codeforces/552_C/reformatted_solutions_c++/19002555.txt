#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
int main() {
  int w, m;
  cin >> w >> m;
  while (m) {
    if ((m - 1) % w == 0)
      m--;
    else if ((m + 1) % w == 0)
      m++;
    else if (m % w != 0) {
      cout << "NO";
      return 0;
    }
    m /= w;
  }
  cout << "YES";
  return 0;
}
