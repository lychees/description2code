#include <iostream>
using namespace std;

int main()
{
    long n;
    cin>>n;
    long a,b,c[50000],l=0,sum=0;
    cin>>a>>b;
    for(int m=0;m<50000;m++)
    {
    	c[m]=0;
    }
  	for(int j=1;j<=n;j++)
    	{
    		if(j%b==0 || j%a==0)
    		{
    			c[l]=j;
    			l++;
    		}
    	}
    	for(int k=0;k<l-1;k++)
    	{
    		sum=sum+c[k];
    	}
    	cout<<sum<<'\n';
    return 0;
}
