/*Carry*/

#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
char num1[100],num2[100];

int main()
{
	int carry,i,j,length1,length2,ncarries,total;
	int t;
	cin>>t;
	while(t--)
	{
	scanf("%s %s",num1,num2);

		length1=strlen(num1);
		length2=strlen(num2);
		i=length1-1;
		j=length2-1;
		carry=ncarries=0;
		while((i>=0) && (j>=0))
		{
			total=carry+num1[i]+num2[j]-96;
			carry=total/10;
			if(carry>0)
				ncarries++;
			i--;
			j--;
		}
		while(i>=0)
		{
			total=carry+num1[i]-48;
			carry=total/10;
			if(carry>0)
				ncarries++;
			i--;
		}
		while(j>=0)
		{
			total=carry+num2[j]-48;
			carry=total/10;
			if(carry>0)
				ncarries++;
			j--;
		}
		if(ncarries==0)
			printf("No carry operation\n");
		else if(ncarries==1)
			printf("1 carry operation\n");
		else
			printf("%d carry operations\n",ncarries);
	
	}
	return 0;
}
