#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    long long t;
    cin>>t;
    while(t--)
    {
       int  i,sum=0,x;
    	vector<int >a;
    	for(i=0;i<3;i++){
    		
    	cin>>x;
    	a.push_back(x);
    	}
    	
    	for(i=0;i<3;i++)
    	{
    		 if(a[i]==9)
    		{
    			sum+=10;
    			break;
    		}	 
    		if(a[i]==0)
    		sum+=0;
    		
    		 if(a[i]>6&&a[i]<9)
    		sum+=2;
    		
    		 if(a[i]>0&&a[i]<7)
    		sum+=1;
    		
    	}
    		if(i==1)
    		sum-=3;
    		if(i==2)
    		sum-=6;
    if(i==3)
    cout<<"0"<<endl;
    else
    	cout<<sum<<endl;
    	a.clear();
    }
    return 0;
}
