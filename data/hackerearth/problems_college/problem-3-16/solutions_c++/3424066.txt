#include <iostream>
using namespace std;

int main()
{
    int t,k,n,a[11],i,j,p,max;
    int q;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	for(j=0;j<11;j++)
    	a[j]=0;
    	cin>>n>>k;
    	for(j=0;j<n;j++)
    	{
    		cin>>p;
    		a[p]++;
    	}
    	max=a[1];
    	for(j=1;j<=k;j++)
    	{
    		if(a[j]>max)
    		{max=a[j];}
    	}
    	//cout<<max<<endl;
    	q=n-max;
    	cout<<q<<endl;
    }
    return 0;
}
