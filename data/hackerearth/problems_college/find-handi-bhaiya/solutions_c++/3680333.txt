#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n;
    cin>>n;
    for(long long int i=0;i<n;i++){
    float a,b,c;
    scanf("%f",&a);
    scanf("%f",&b);
    scanf("%f",&c);
 
    if(a<(b+c) &&b<(a+c)&&c<(a+b))
    {
        if(a==b&&a==c&&b==c)
        printf("YES. It is a equilateral triangle.\n");
        else if(a==b||a==c||b==c)
        printf("YES. It is a isosceles triangle.\n");
        else if(a!=b&&a!=c&&b!=c)
        printf("YES. It is a scalene triangle.\n");
    }
    else
        printf("NO\n");}
    
    return 0;
}
