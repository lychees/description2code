#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll n,m,first,last,mid,temp,maxm;
vector<ll> v;

bool check(ll val)      // checking for a given T
{
  ll len = v.size();
  ll sum=0;
  ll ctr=0;

  for(ll i=0; i<len; i++)
  {
     if(v[i]>val)
     return false;

     sum+=v[i];
     if(sum+v[i+1]>val)
      {
        ctr++;
        sum=0;
      }
  }
    if(ctr+1>m)
     return false;
    else
    return true;

}

int main()
{
  cin>>n>>m;
  maxm=0;

  while(n--)
  {
    cin>>temp;
    maxm = max(maxm,temp);
    v.push_back(temp);
  }

  first = maxm;
  last = 1e18;

  while(first <= last)
  {
    mid = (first+last)/2;

    if(check(mid) &&  !check(mid-1))
    {
      cout<<mid<<endl;
      break;
    }

    else if(check(mid))
     last = mid-1;

    else if(!check(mid))
     first = mid+1;

  }
return 0;
}
