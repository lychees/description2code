#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
#define PII       pair<int,int>
#define all(c)    c.begin(),c.end()
#define sz(c)     (int)c.size()
#define clr(c)    c.clear()
#define pb        push_back
#define mp        make_pair
#define cin(x)    scanf("%d",&x)
#define MOD       1000000007

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		int n, k;
		cin(n);
		cin(k);
		int one = 0 , zero = 0;
		for(int i = 0; i < n; i++)
		{
			int x;
			cin(x);
			if(x == 0) zero += 1;
			else one += 1;
		}
		int p = min(zero , k);
		one += p;
		zero -= p;
		k -= p;
		k %= 2;
		one -= k;
		zero += k;
		printf("%d\n", one);
	}
	return 0;
}
