#include <bits/stdc++.h>

using namespace std;

int main()
{
	int ts;
	cin>>ts;
	while(ts--){
		int n,count=0;
		cin>>n;
		for(int i=1;i<=n;i++){
			for(int j=i+1;j<=n;j++){
				if(((i*j)%(i+j))==0){
					count++;
				}
				else{
				count=count+0;
					}
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
