#include <bits/stdc++.h>
using namespace std;
unsigned long long arr[90];
unsigned long long fibo(int i)
{
	if(arr[i])
		return arr[i];
	else if(i==1||i==0)
		return 1;
	arr[i]=fibo(i-1)+fibo(i-2);
	return arr[i];
}
int main()
{
	ios_base::sync_with_stdio(false);
	int n,m;
	memset(arr,0,sizeof(arr));
	arr[0]=1;
	arr[1]=1;
	cin>>n;
	arr[n]=fibo(n-1);
	unsigned long long total=0;
	for(int i=0;i<n;i++)
		total+=arr[i];
	cin>>m;
	unsigned long long x;
	while(m--)
	{
		cin>>x;
		total-=x;
	}
	cout<<total<<endl;
	return 0;
}
