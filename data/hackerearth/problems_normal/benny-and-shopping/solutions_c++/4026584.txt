#include<iostream>
#include<string.h>
#include<stdio.h>

using namespace std;

int main(){
	int t,len,amountReached,amountStarted,j;
	char string[10000],result[10000];
	cin>>t;
	getchar();
	while(t--){

		scanf ("%[^\n]%*c", string);
		len=strlen(string);
		amountReached=0;
		amountStarted=0;
		j=0;
		for(int i=0;i<len;i++){
			if(amountReached==1){
				if(amountStarted==1){
					if(string[i]<=57 && string[i]>=48){
						result[j++]=string[i];
					}else if(string[i]==' '){
						continue;
					}else
						break;
				}else{
					if(string[i]<=57 && string[i]>=49){
						result[j++]=string[i];
						amountStarted=1;
					}else if(string[i]==' ' || string[i]=='0'){
						continue;
					}else{
						break;
					}	
				}
			}else if(string[i]=='$'){
				amountReached=1;
			}
		}
		if(j==0){
			cout<<"$0";
		}else{
		cout<<"$";
		for(int i=0;i<j;i++)
		cout<<result[i];
		}
		
		cout<<endl;
	}
return 0;

}
