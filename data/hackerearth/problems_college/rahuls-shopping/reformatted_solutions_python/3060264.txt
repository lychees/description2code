n = input()
a = raw_input()
b = map(int, a.split(" "))
c = raw_input()
l, h = map(int, c.split(" "))
ar = l*h
su = 0
for j in b:
    t = j
    s = 0
    while (t != 0):
        y = t % 10
        t = t/10
        s = s + y
    x = (s * j * ar)
    m = 0
    if j in range(1, 11):
        m = int(x*0.75)
        su = su + m
    elif j in range(11, 21):
        m = int(x*0.7)
        su = su + m
    elif j in range(21, 51):
        m = int(x*0.8)
        su = su + m
print su
