#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   int n;
   long long int sum=0;
   cin>>t;
   while(t--)
   {
   		
   		cin>>n;
   		sum=0;
   		long long int d;
   		int i;
   		for(i=0;i<n;i++)
   		{
   			cin>>d;
   			sum=sum+d;
   		}
   		cout<<sum/n;
   	cout<<"\n";
   	
   }
    return 0;
}
