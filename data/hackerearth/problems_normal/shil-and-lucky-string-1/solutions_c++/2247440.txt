#include<bits/stdc++.h>
#define sci(a) scanf("%d",&a)
#define scll(a) scanf("%lld",&a)
#define scs(a) scanf("%s",a)
#define INCR(x,a,b) for(int x=a;x<=b;x++)
#define DCR(x,a,b) for(int x=a;x>=b;x--)
#define ff first
#define ss second
#define pb(a) push_back(a)
#define all(x) (x).begin(),(x).end()
#define sz(x) ((int)(x).size())
#define mp(x,y) make_pair(x,y)
#define pf(a) push_front(a)

using namespace std;
typedef long long ll;


int main()
{
	int n;
	sci(n);
	char str[100009];
	scs(str);
	int mid=n/2;
	int forw[26];
	int back[26];
	memset(forw,0,sizeof(forw));
	memset(back,0,sizeof(back));
	INCR(i,0,mid-1)
	{
		forw[str[i]-'a']++;
		back[str[mid+i]-'a']++;
	}
	int gret=0,small=0,eq=0;
	int tf[26],tb[26];
	INCR(i,0,25)
	{
		tf[i]=forw[i];
		tb[i]=back[i];
	}
	INCR(i,0,25)
	{
		if(forw[i]>back[i])
		{
			forw[i]-=back[i];
			back[i]=0;
		}
		else if(back[i]>forw[i])
		{
			back[i]-=forw[i];
			forw[i]=0;
		}
		else
			forw[i]=back[i]=0;
	}
	INCR(i,0,25)
		eq+=forw[i];
	INCR(i,0,25)
	{
		forw[i]=tf[i];
		back[i]=tb[i];
	}
	gret=back[0]+forw[25];
	forw[0]+=forw[25];
	back[25]+=back[0];
	INCR(i,1,25)
	{
		if(forw[i-1]<back[i])
		{
			back[25]+=back[i]-forw[i-1];
			gret+=back[i]-forw[i-1];
		}
		else
		{
			forw[i]=forw[i]+(forw[i-1]-back[i]);
		}
	}
	//gret+=forw[25];
	INCR(i,0,25)
	{
		forw[i]=tf[i];
		back[i]=tb[i];
	}
	small=forw[0]+back[25];
	back[0]+=back[25];
	forw[25]+=forw[0];
	INCR(i,1,25)
	{
		if(back[i-1]<forw[i])
		{
			forw[25]+=forw[i]-back[i-1];
			small+=forw[i]-back[i-1];
		}
		else
		{
			back[i]=back[i]+(back[i-1]-forw[i]);
		}
	}
	//small+=back[25];
	//cout<<eq<<" "<<small<<" "<<gret<<endl;
	int ans=min(eq,min(small,gret));
	cout<<ans;
	
    return 0;
}
