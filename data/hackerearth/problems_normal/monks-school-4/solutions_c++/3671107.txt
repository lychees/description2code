#include <bits/stdc++.h>
#define inf 1000000000
using namespace std;

int abs(int n) {
    if (n < 0) return -n;
    return n;
}

struct table {
    string teacher, student;
    int age;
};

bool cmp(const struct table &p1, const struct table &p2) {
    return (p1.teacher < p2.teacher);
}

int main()
{
    int n, m;
    string s;
    table a[100005];

    cin >> n >> m;

    for (int i = 0; i < n; i++) {
        cin >> s;
    }

    for (int i = 0; i < m; i++) {
        cin >> a[i].teacher >> a[i].student >> a[i].age;
    }

    sort(a, a + m, cmp);

    int i = 1;
    vector<pair<int, string> > v;
    int ag = a[0].age;
    string st = a[0].student;
    v.push_back(make_pair(ag, st));

    while (i < m) {
        if (a[i].teacher == a[i - 1].teacher) {
            ag = a[i].age;
            st = a[i].student;
            v.push_back(make_pair(ag, st));
            i++;
        } else {
            sort(v.begin(), v.end());
            cout << a[i - 1].teacher << endl;
            for (int j = 0; j < v.size(); j++) {
                cout << v[j].second << " " << v[j].first << endl;
            }
            v.clear();
            ag = a[i].age;
            st = a[i].student;
            v.push_back(make_pair(ag, st));
            i++;
        }
    }
    sort(v.begin(), v.end());
    cout << a[i - 1].teacher << endl;
    for (int j = 0; j < v.size(); j++) {
        cout << v[j].second << " " << v[j].first << endl;
    }


    return 0;
}
