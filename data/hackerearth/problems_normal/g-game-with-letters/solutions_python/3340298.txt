from collections import defaultdict
for t in xrange( input() ) :
	n, s = raw_input().split()
	score = defaultdict(int)
	for c in s :
		score[c] += 1
	have = []
	for i in xrange( int(n) ) :
		t = raw_input()
		have.append( (sum(score[x] for x in t),-len(t),-(i+1) ) )
	have.sort( reverse = True )
	print -have[0][2]
