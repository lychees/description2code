#include<cstdio>

int p[1000001] = {0};

void init(int n){
    p[1] = 0;
    for(int i=2;i<=n;i++){
        if(!p[i]){
            for(int j=i;j<=n;j+=i){
                p[j] += i;
            }
        }
    }
}

int main(){
    int n, k;
    scanf("%d %d", &n, &k);
    init(n);
    int cc = 0;
    for(int i=0;i<=n;i++){
        if(p[i] == k) cc++;
    }
    printf("%d\n", cc);
}
