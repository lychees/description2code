import sys
import os
def bin(n):
    Num=0
    j=1
    while(n!=0):
        rem=n%10
        Num=Num+rem*j
        j=j*2
        n=n/10
    return Num

t = int(input())
while(t>0):
    a , b = map(int,raw_input().split())
    n1 = bin(a)
    n2 = bin(b)
    if(n1**n2<=n2**n1):
        print("The Hound")
    elif(n1**n2>n2**n1):
        print("The Mountain")
    t-=1
