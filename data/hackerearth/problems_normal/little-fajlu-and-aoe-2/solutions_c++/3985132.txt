#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
long long max1(long long &a,long long &b,long long& c)
{
    long long temp;
    if(a>b) temp=a;
    else temp=b;
    if(c>temp) temp=c;
    return temp;
 
 
}
 
int main()
{
    static long long arr[1601][801]={0};
    int t;
    scanf("%d",&t);
    int f,g,s,p,c,power;
    while(t--)
    {
scanf("%d%d%d%d%d%d",&f,&g,&s,&p,&c,&power);
f=f/25;
g=g/50;
 
for(int i=2;i<=f;i++)
{
    for(int j=0;j<=g;j++)
    {
        long long k1=0,k2=0,k3=0;
        if(i>=4) k1=arr[i-4][j]+s;
        if(i>=5&&j>=1) k2=arr[i-5][j-1]+p;
        if(i>=2&&j>=2) k3=arr[i-2][j-2]+c;
        arr[i][j]=max1(k1,k2,k3);
 
 
    }
 
 
 
}
 
if(arr[f][g]>power) printf("%lld\n",arr[f][g]-power);
else printf("-1\n");
 
for(int i=2;i<=f;i++)
{
    for(int j=0;j<=g;j++)
    {
 arr[i][j]=0;
 
    }
 
 
}
    }
}
