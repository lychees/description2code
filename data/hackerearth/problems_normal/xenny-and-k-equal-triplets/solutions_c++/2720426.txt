#include<iostream>
#include<vector>
using namespace std;

long long ncr(int n, int r){
	long long result = n;
	for(int i=2; i<=r; ++i){
		result *= (n-i+1);
		result /= i;
	}
	return result;
}

long long GCD(long long a, long long b){
	if(a==0){
		return b;
	} 
	return GCD(b%a, a);
}

int main(){
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--){
		int n, k;
		cin>>n>>k;
		int count=0, temp;
		for(int i=0; i<n; i++){
			cin>>temp;
			if(temp == k){
				count++;
			}
		}
		long long deno, numer;
		deno = ncr(n, 3);
		numer = ncr(count, 3);
		int gcd = GCD(deno, numer);
		deno /= gcd;
		numer /= gcd;
		cout<<numer<<"/"<<deno<<endl;
		
	}
	return 0;
}
