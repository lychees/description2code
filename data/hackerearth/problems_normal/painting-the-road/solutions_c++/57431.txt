#include <cstdio>
#include <vector>
#include <queue>
#include <iostream>
#include <algorithm>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <map>
#include <utility>
#define MOD 1000000007
#define INF 100000000
#define PB push_back
#define pii pair<int, int>
typedef long long LLD;
using namespace std;
struct proposal
{
    int l, r, c;
};
proposal ar[1000];
int dp[1000];
bool comp(proposal a, proposal b)
{
    return a.l < b.l;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
	int L, n;
	cin >> L >> n;
	for(int i = 0; i < n; i ++)
	    cin >> ar[i].l >> ar[i].r >> ar[i].c;
	sort(ar, ar + n, comp);
	for(int i = 0; i <= L; i ++)
	    dp[i] = INF;
	dp[0] = 0;
	for(int i = 0; i < n; i ++)
	{
	    for(int j = ar[i].l; j <= ar[i].r; j ++)
		dp[ar[i].r] =  min(dp[ar[i].r], dp[j] + ar[i].c);
	}
	if(dp[L] >= INF)
	    dp[L] = -1;
	cout << dp[L] << endl;
    }
    return 0;
}
