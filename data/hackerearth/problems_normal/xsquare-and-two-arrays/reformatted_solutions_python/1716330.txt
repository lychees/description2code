a, b = map(int, raw_input().split())
seriesA = map(int, raw_input().split())
seriesB = map(int, raw_input().split())
while b > 0:
    Q, L, R = map(int, raw_input().split())
    if Q == 1:
        AA = seriesA[L-1:R:2]
        AB = seriesB[L:R:2]
    else:
        AA = seriesA[L:R:2]
        AB = seriesB[L-1:R:2]
    b -= 1
    print sum(AA) + sum(AB)
