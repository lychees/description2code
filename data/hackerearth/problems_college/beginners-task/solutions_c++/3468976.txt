#include <iostream>
using namespace std;
int main()
{
   int t,c,a[10000],n;
   cin>>t;
   for(int i=1;i<=t;i++ )
   {c=0;
   	cin>>n;
   	for(int j=0;j<n;j++)
   	{
   		cin>>a[j];
   	}
   	for(int j=0;j<n;j++)
   	{
   		if(a[j]%2==0)
   		{
   			c++;
   		}}
   		cout<<c<<"\n";
   }
    return 0;
}
