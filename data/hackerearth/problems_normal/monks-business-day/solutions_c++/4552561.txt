#include <bits/stdc++.h>
using namespace std;

#define MAX 99999999

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n;
        cin>>m;
        vector<vector<pair<int,int> > > graph(n+1);
        int dist[n+1];
        for(int i=1;i<=n;i++)
        {
            dist[i]=MAX;
        }

        dist[1]=0;
        for(int i=1;i<=m;i++)
        {
            int x,y,c;
            cin>>x>>y>>c;
            graph[x].push_back(make_pair(y,-c));
        
        }
        for(int k=1;k<n;k++)
        {
            for(int i=1;i<=n;i++)
            {
                if(dist[i]==MAX)
                    continue;

                for(vector<pair<int,int> >::iterator it=graph[i].begin();it!=graph[i].end();it++)
                {
                    int point=(*it).first;
                    int weight=(*it).second;
                    dist[point]=min(dist[point],dist[i]+weight);
                }
            }
        }
        int flag=0;
        for(int i=1;i<=n;i++)
            {
                if(dist[i]==MAX)
                    continue;

                for(vector<pair<int,int> >::iterator it=graph[i].begin();it!=graph[i].end();it++)
                {
                    int point=(*it).first;
                    int weight=(*it).second;
                    if(dist[point]>dist[i]+weight){
                        flag=1;
                        break;
                    }
                }
            }

        if(flag==1)
            printf("Yes\n");
        else
            printf("No\n");
        
    }
    return 0;
}
