#include<iostream>
#include<cmath>
using namespace std;
long double area(int x1, int y1, int x2, int y2, int x3, int y3)
{
	return abs((x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2.0);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int x1, y1, x2, y2, x3, y3, x4, y4;
    	long double a, a1, a2, a3;
    	cin>>x1>>y1>>x2>>y2>>x3>>y3>>x4>>y4;
    	a=area(x1,y1,x2,y2,x3,y3);
    	//cout<<a<<endl;
    	a1=area(x4,y4,x2,y2,x3,y3);
    	//cout<<a1<<endl;
    	a2=area(x1,y1,x4,y4,x3,y3);
    	//cout<<a2<<endl;
    	a3=area(x1,y1,x2,y2,x4,y4);
    	//cout<<a3<<endl;
    	if((a1+a2+a3)==a)
    	cout<<"INSIDE"<<endl;
    	else
    	cout<<"OUTSIDE"<<endl;
    }
return 0;
}
