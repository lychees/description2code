    using namespace std;
    const int MAXN = 1000005;
    int S[MAXN];
    int size;
    int n, x;
     
     
    void wstaw(int w)
    {
      int poc = 1;
      int kon = size;
      int ret = 1;
      while(poc <= kon)
      {
        int sr = (poc + kon) / 2;
        if(w > S[sr] - 1)
        {
          ret = sr + 1;
          poc = sr + 1;      
        }
        else
        {
          kon = sr - 1;
        }
      }
      if(ret > size)
      {
        S[++size] = w;
      }
      else
        S[ret] = w;
    }
     
    int main()
    {
      scanf("%d", &n);
      for(int i = 1; i <= n; i++)
      {
        scanf("%d", &x);
        if(x >= i)
        wstaw(x - i);
    /*

        cout<<S[j]<<" ";
        cout<<endl;*/
      }
      printf("%d", n - size);
      return 0;
    }
