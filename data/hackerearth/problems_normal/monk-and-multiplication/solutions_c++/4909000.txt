#include <iostream>
using namespace std;

int main()
{
    long long n;
    cin>>n;
    long long a[n];
    for(long long i=0;i<n;i++)
    cin>>a[i];
    
    long long first = -1;
    long long second = -1;
    long long third = -1;
    for(long long i=0;i<n;i++)
    {
    	if(a[i]>=first)
    	{
    		third = second;
    		second = first;
    		first = a[i];
    	}
    	else if (a[i]>=second)
    	{
    		third = second;
    		second = a[i];
    	}
    	else if(a[i]>=third)
    	{
    		third = a[i];
    	}
    	
    	
    	if(i>1) cout<<first*second*third<<endl;
    	else cout<<-1<<endl;
    }
    return 0;
}
