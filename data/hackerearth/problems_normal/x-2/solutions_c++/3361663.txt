#include<bits/stdc++.h>
#define nax 1000005
#define ll long long
using namespace std;
ll int arr[nax],pref[nax],cl[nax];
ll int binarySearch(ll int start,ll int n,ll int pref[],ll int data)
{
	ll int high=n;
	ll int low=start;
	ll int mid;
	while(high>=low)
	{
		mid=(high+low)/2;
		if(pref[mid]==data)
			return mid;
		else if(pref[mid]>data)
		{
			high=mid-1;
		}
		else
			low=mid+1;
	}
	return high;
}
int main()
{
	ll int t,n,k,x;
	cin >> t;
	while(t--)
	{
		cin>>n>>k;
		for(ll int i=1;i<=n;i++)
		{
			cin>>arr[i];
		}
		arr[0]=0;
		pref[0]=0;
		for(ll int i=1;i<=n;i++)
		{
			pref[i]=pref[i-1]+arr[i];
		}
		for(ll int i=0;i<nax;i++)
			cl[i]=0;
		for(ll int i=1;i<=n;i++)
		{
			if(arr[i]>k)
				continue;
			x=binarySearch(i,n,pref,pref[i-1]+k);
			cl[x-i+1]++;
		}
//		for(int i=1;i<=n;i++)
//			cout <<cl[i]<<" "<<endl;
		for(ll int i=n-1;i>=1;i--)
			cl[i] +=cl[i+1];
		for(ll int i=1;i<=n;i++)
			cout<<cl[i]<<" ";
		cout<<"\n";
	}
	return 0;
}
