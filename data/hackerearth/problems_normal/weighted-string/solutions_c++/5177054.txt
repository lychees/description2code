#include <bits/stdc++.h>
using namespace std;

set<int> se;
string s;
int k;

int main() {
    int t; scanf("%d", &t);
    while (t--) {
    	cin >> k >> s;
    	se.clear();
    	se.insert(0);
    	int sum = 0;
    	for (int i = 0; i < (int) s.size(); ++i) {
    		sum += s[i] - 'a' + 1;
    		se.insert(sum);
    	}
    	int ans = 0;
    	for (auto& t: se) {
    		ans += se.find(t + k) != se.end();
    	}
    	cout << ans << endl;
    }
    return 0;
}
