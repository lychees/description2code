#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
bool power(ll n)
{
	if((n&(n-1))==0)
	return(true);
	return(false);
}
int main()
{
	ll t,n;
	cin>>t;
	ll total=0;
	for(ll i=0;i<32;i++)
	total+=1<<i;
	while(t--)
	{
		cin>>n;
		ll arr[n];
		for(ll i=0;i<n;i++)
		cin>>arr[i];
		int flag=0;
		if(n==1)
		{
		if(power(arr[0]))
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		}
		else
		{
		 for(ll i=0;i<32;i++)
		 {
			ll ans=total;
			for(ll j=0;j<n;j++)
			{
				if(arr[j]&(1<<i))
				ans=ans&arr[j];
			}
			if(power(ans))
			{
				flag=1;
				break;
		 	}
		 }
		if(flag)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		}
	}
	return(0);
}
