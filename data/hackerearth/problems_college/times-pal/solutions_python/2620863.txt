from sys import stdin
def pal(a):
	b = list(a)
	b.reverse()
	x = ''.join(b)
	return x==a
	
t=int(stdin.readline())
for i in xrange(t):
		
	a = stdin.readline().strip()
	b = stdin.readline().strip()
	ans  = 0
	l = len(a)
	for i in xrange(l):
		y = a[:i] + b + a[i:]
		#print "check", y
		if pal(y):
			ans+=1
	if pal(a+b):
		ans+=1
	print ans
