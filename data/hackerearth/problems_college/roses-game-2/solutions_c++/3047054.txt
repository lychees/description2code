#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[1000010];
int main()
{
    ll t,n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        for(int i=0;i<n;i++)
            scanf("%lld",&a[i]);
        ll x=a[0],c=0;
        while(x!=0&&x!=-1&&c<n)
        {
            x=a[x];c++;
        }
        if(x==-1)
            printf("Win\n");
        else
            printf("Lose\n");
    }
}
