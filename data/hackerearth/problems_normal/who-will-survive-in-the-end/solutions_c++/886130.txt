//Special case where the alternate person is killed.
//f(n) = 2(n-2^{\lfloor \log_2(n) \rfloor})+1 

#include<iostream>
#include<stdio.h>

using namespace std;

long long int nearestpow(long long int n){
	long long  int nearest = 1;
	while(nearest<=n){
		nearest <<=1;
	}
	return nearest>>1;	
}

int main(void){
	int T=0;
	scanf("%d",&T);
	while(T--){
		long long int N = 0;
		scanf("%lld",&N);
		long long int nearest = nearestpow(N);		
		long long int result = 2*(N-nearest)+1;
		printf("%lld\n",result);
	}
	return 0;
}
