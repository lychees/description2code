#include <iostream>
using namespace std;

int main() {
    int  t;
    cin >> t;
    while(t--){
        char ch[100];
        cin >> ch;
        int S = 0;
        while(ch[S] != '\0') S++;
        S--;
        for(int i = 0; ; i++) {
            if(i >= S - i) {
                cout << ch[S/2] + 0<< endl;  
                break;
            } 
            if(ch[i] != ch[S - i]) {
                cout << -1 << endl;
                break;
            }
        }
    }
}
