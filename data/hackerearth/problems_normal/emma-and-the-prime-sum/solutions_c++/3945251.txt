using namespace std;
ll sum[1000001];bool prime[1000001];
void seivec()
{
	prime[0]=prime[1]=true;
	sum[0]=0;
	for(int i=2;i*i<1000001;i++)
		if(prime[i]==false)
			for(int j=2*i;j<1000001;j+=i)
				prime[j]=true;
	for(int i=1;i<1000001;i++)
	{
		if(prime[i]==false)
			sum[i]=sum[i-1]+i;
		else
			sum[i]=sum[i-1];
	}
}
int main()
{
	seivec();
	int t;
	si(t);
	while(t--)
	{
		int l,r;
		si(l);si(r);
		ll an=sum[r]-sum[l-1];
		pll(an);
	}
	return 0;
}
