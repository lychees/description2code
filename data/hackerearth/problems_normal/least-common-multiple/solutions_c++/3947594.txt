#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;

bool isprime(ll n)
{
	for(ll i=2;i<=sqrt(n);i++)
	{
	if(n%i==0)
	return false;
	}
	return true;
}

ll power(ll m,ll n)
{
    ll res = 1;     // Initialize result
 
    while (n > 0)
    {
        // If y is odd, multiply x with result
        if (n & 1)
            res = res*m;
 
        // n must be even now
        n = n>>1; // y = y/2
        m = m*m;  // Change x to x^2
    }
    return res;
}


int main()
{
 ll n;
 cin>>n;
 ll min1=n,min2=n;
 for(ll i=n/2+1;i<=n;i++)
 {
 	if(isprime(i))
 	{
 		min1=i;
 		break;
 	}
 }

 for(ll i=2;i<=log2(min1);i++)
 	{
 		ll k=1;
 		while(power(k,i)<=n/2)
 		k++;
 		while(power(k,i)<=min1)
 		{
 			if(power(k,i)<min2)
 			min2=power(k,i);
 			k++;
 		}
 		
 	}
 	//cout<<min2<<" ";
 	cout<<min(min1,min2)<<"\n";
    return 0;
}
