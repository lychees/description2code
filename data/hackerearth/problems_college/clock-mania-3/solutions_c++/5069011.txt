#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;
   cin>>t;
   for(;t--;)
   {
   	int h,m;
   	cin>>h>>m;
   	h*=30;
   	m*=6;
   	h+=(m/12);
   	if(h>m)
   	h-=m;
   	else
   	h=m-h;
   	if(h>180)
   	cout<<(360-h)<<endl;
   	else
   	cout<<h<<endl;
   }
    return 0;
}
