from math import *
n = int(raw_input())
a = map(int,raw_input().split())
ans = {}
mod = 10**9+7
for v in a:
	for i in xrange(2,int(sqrt(v))+1):
		while (v%i==0):
			if not i in ans:
				ans[i]=0
			ans[i]+=1
			v=v/i
	if v>1:
		if not v in ans:
			ans[v]=0;
		ans[v]+=1
res = 1
for key in ans:
	res*=(ans[key]+1)
print res%mod
