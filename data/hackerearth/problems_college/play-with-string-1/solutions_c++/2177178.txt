#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int main()
{
	int t,i;
	char a[100009];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",a);
		int n=strlen(a);
		i=1;		
		while(a[i]!='\0')
		{
			if(abs(a[i]-a[i-1])!=abs(a[n-i]-a[n-i-1]))
			{
				i=-1;
				break;
			}
			i++;
		}
		if(i!=-1)
		{
			printf("Funny\n");
		}
		else
		{
			printf("Not Funny\n");
		}
	}
	return 0;
}
