#include <iostream>
#include <stdio.h>
using namespace std;

typedef long long int Lint;

Lint Raise( Lint a, Lint b ) {
	
	Lint v, v2, v3;
	
	if( b == 0 )
		return 1;
	else {
		if( b & 0x1 ) {
			v = Raise( a, ( b - 1 ) / 2 );
			v3 = v * v;
			v2 = a * v3;
		} else {
			v = Raise( a, b / 2 );
			v2 = v * v;
		}
		return v2;
	}
}

int main()
{

	ios_base::sync_with_stdio( false );
	Lint t, n, val, v;
	
	cin >> t;
	
	while( t-- ) {
		cin >> n;
		v = Raise( 2, n - 1 );
		val = n * ( n + 1 ) / 2;
		cout << ( v * val ) << "\n";
	}
	
	return 0;
}
