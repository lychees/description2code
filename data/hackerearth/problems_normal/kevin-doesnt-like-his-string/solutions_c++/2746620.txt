int main() {
	srand (time(NULL));
	ios_base::sync_with_stdio(false); cin.tie(NULL);
	
	next(int, n);
	next(int, k);

	next(string, s);
	
	vector<int> to_insert(200);

	vector<int> rightmost(200, n - 1);
	FOR (c, 'a', 'z' + 1) {
		while (rightmost[c] >= 0 && s[rightmost[c]] != c) rightmost[c]--;		
	}

	FOR (i, 0, n) {
		for (char c = 'a' ; c < s[i] || s[i] == '_'; c++) {
			if (s[i] == '_' && to_insert[c]) {
				s[i] = c;
				to_insert[c]--;
				break;
			}
			if (s[i] != '_' && k && to_insert[c]) {
				to_insert[c]--;
				to_insert[s[i]]++;
				s[i] = c;
				k--;
				break;
			}
			if (s[i] == '_' && k && rightmost[c] > i) {
				s[i] = c;
				s[rightmost[c]] = '_';
				k--;
				while (rightmost[c] >= 0 && s[rightmost[c]] != c) rightmost[c]--;
				break;
			}

			if (s[i] != '_' && rightmost[c] > i && k > 1) {
				to_insert[s[i]]++;
				s[i] = c;
				s[rightmost[c]] = '_';
				k -= 2;

				while (rightmost[c] >= 0 && s[rightmost[c]] != c) rightmost[c]--;
				break;
			}

		}
	}

	cout << s;
}
