def ans(n):

    if (n == 0):
        return True
    elif (n < 0):
        return False
    else:
        return ans(n-5) or ans(n-3)


for _ in xrange(input()):
    n = input()

    if ans(n):
        if (n % 5 == 0):
            print str(3)*n
        else:
            c = 5*(2*n % 3)
            print str(5)*(n-c)+str(3)*c
    else:
        print -1
