#include<iostream>
#define MOD 1000000007
using namespace std;
long long int nMat[501][501];
long long int nAr[501];

void fillWays(long long int *nAr)
{
    nAr[0] = 1;
    nAr[1] = 1;
    nAr[2] = 2;
    for(int nNum = 3; nNum < 501; nNum++)
    {
        long long int nSum = 0;
        for(int i = 0; i <= nNum - 1; i++)
        {
            nSum += (nMat[nNum - 1][i] * nAr[i])%MOD;
        }
        nAr[nNum] = nSum%MOD;
    }
}

void fillAr(long long int nMat[501][501])
{
    nMat[0][0] = 1;
    nMat[1][1] = 1;
    nMat[1][0] = 1;
    for(int i = 2; i < 501; i++)
    {
        nMat[i][0] = 1;
        nMat[i][1] = i;
        nMat[i][i] = 1;
        for(int j = 2; j < i; j++)
        {
                nMat[i][j] = (nMat[i - 1][j] + nMat[i-1][j-1])%MOD;
        }
    }
}


int main()
{
    int nCases, nNum;
    fillAr(nMat);
    fillWays(nAr);
    cin>>nCases;
    while(nCases--)
    {
        cin>>nNum;
        cout<<nAr[nNum]<<endl;
    }
    return 0;
}
