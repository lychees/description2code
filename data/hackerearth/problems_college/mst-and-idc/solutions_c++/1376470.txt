    #define getcx getchar

    inline long long getint (void) {
    long x = 0; char ch = getcx();
    while (ch<'0' || ch>'9')
    ch = getcx();
    while ('0'<=ch && ch<='9') {
    x = x*10+ch-'0';
    ch = getcx();
    }
    return x;
    }


int main()
{
    lli t,n,i,min,max;
    t=getint();
    while(t--)
    {
              n=getint();
              
              lli a[n];
              
              for(i=0;i<n;i++)
              a[i]=getint();
              
              min=a[0];max=a[0];
              for(i=0;i<n;i++)
              {
                              if(a[i]>max)
                              max=a[i];
                              else if(a[i]<min)
                              min=a[i];
              }
              
              printf("%lld\n",(max-min)*2);
    }
    
    return 0;
}
