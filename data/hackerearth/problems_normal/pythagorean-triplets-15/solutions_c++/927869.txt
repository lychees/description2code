#include<iostream>
#include<cmath>
using namespace std;

int gcd(int a,int b)
{
    return b==0 ? a : gcd(b,a%b);
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,ct=0;
        cin>>n;
        long long int c;
        int i,j,k;
        for(i=2;i<=sqrt(n-1)+1;i++)
        {
            for(j=1;j<i;j++)
            {
                if(gcd(i,j)==1 && (i-j)%2 && (i*i + j*j <=n))
                {
                    k=0;
                    c=i*i + j*j;
                    while((++k)*c<=n)
                    {
                        ct++;
                    }
                }
            }
        }
        cout<<ct<<endl;
    }
    return 0;
}
