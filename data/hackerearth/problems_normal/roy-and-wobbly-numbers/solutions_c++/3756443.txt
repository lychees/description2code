#include <iostream>
using namespace std;

int main()
{
    int res[81];
    int k=0;
    for(int i=1; i<10; i++)
    {
    	for(int j=0; j<10; j++)
    	{
    		if(i != j)
    		{
    			res[k] = 10*i + j;
    			k++;
    		}
    	}
    }
    
    int t;
    cin >>t;
    for(int i=0; i<t; i++)
    {
    	int n,k;
    	cin >>n>>k;
    	if(k > 81)
    	{
    		cout << "-1" << endl;
    		continue;
    	}
    	int a = res[k-1]/10;
    	int b = res[k-1]%10;
    	for(int j=0; j<n; j++)
    	{
    		if(j%2 == 0) cout << a;
    		else cout << b;
    	}
    	cout << endl;
    }
    
    return 0;
}
