#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
 long long  int n;
 cin>>n;
  
long long int ar[100005];
  
  for(long long int i=0;i<n;i++){
  cin>>ar[i];
  }
  
  sort(ar,ar+n);
  
  long long int result1=ar[n-1]*ar[n-2];
  long long int result2=ar[0]*ar[1];
  
  if (result1>result2)
  cout<<result1;
  
  else cout<<result2;
    return 0;
}
