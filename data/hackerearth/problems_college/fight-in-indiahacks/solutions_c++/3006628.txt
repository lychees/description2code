#include <bits/stdc++.h>
#define ll long long
using namespace std;
bool dp[1000010];
ll int n,m1,m2;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		cin>>n>>m1>>m2;
		ll int i;dp[0]=0;//I have noting to select
		for(i=1;i<=n;i++)
		{
			if(m1>i && m2>i) //I can't make a move
			dp[i]=0;
			else if(m1>i && m2<=i)
			{
				dp[i]=!dp[i-m2];
			}
			else if(m1<=i && m2>i)
			{
				dp[i]=!dp[i-m1];
			}
			else
			{
				dp[i]=!(dp[i-m1] & dp[i-m2]);
			}
		}
		cout<<dp[n]<<endl;
	}
	
	return 0;
}
