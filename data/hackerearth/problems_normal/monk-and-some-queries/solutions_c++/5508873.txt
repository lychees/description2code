#include <bits/stdc++.h>

using namespace std;

int main(int argc, char const *argv[])
{
	map<int,int> m;
	int q;
	cin>>q;
	int x;
	while(q--)
	{
		cin>>x;
		if(x==1)
		{
			cin>>x;
			if(m.count(x)>0)
			{
				m[x]++;
			}
			else
			{
				m.insert(pair<int,int>(x,1));
			}
		}
		else if(x==2)
		{
			cin>>x;
			if(m.count(x)>0)
			{
				m[x]--;
				if(m[x]==0)
					m.erase(x);
			}
			else
				cout<<-1<<endl;
		}
		else if(x==3)
		{	
			if(m.size()==0)
				cout<<-1<<endl;
			else
			{
			map<int,int>::iterator it;
			it = m.end();
			it--;
			cout<<it->first<<endl;
			}
		}
		else if(x==4)
		{
			if(m.size()==0)
				 cout<<-1<<endl;
			else
			{
				map<int,int>::iterator it;
				it = m.begin();
				cout<<it->first<<endl;
			}
		}
	}
	return 0;
}
