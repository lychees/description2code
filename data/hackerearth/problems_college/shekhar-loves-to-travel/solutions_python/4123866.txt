def distance(dis,alist):
    i=0
    next=0
    mx  =0
    t=0
    while i<dis-1 and next<dis-1:
        mx=max(mx,i+int(alist[i]))
        if i==next:
            if mx == next:
                return -1
            next = mx
            t+=1
        i+=1
    return t

test=int(raw_input())

while test>0:
    dis=int(raw_input())
    str=raw_input()
    alist=str.split(' ')
    result=distance(dis,alist)
    print(result-1)
    test-=1
