#include<bits/stdc++.h>
using namespace std;
int main(){
int t;
cin>>t;
   while(t--)
  {
      int l=0,i=0,j=0,y=0;
    string s;
    cin>>s;
    l=s.length();
    j=l-1;
    for(i=0;i<l/2;i++)
    {
     if(s.at(i) > s.at(j))
    {

         y+=int(s.at(i))-int(s.at(j));
    }
    else if(s.at(i) < s.at(j))
    {
       y+=int(s.at(j))-int(s.at(i));
    }
    j--;
    }
    cout<<y<<endl;
  }
return 0;
}
