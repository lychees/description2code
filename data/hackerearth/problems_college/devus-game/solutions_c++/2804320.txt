    #include <iostream>
    #include <stdio.h>
    using namespace std;
    int main()
    {
        int t;
        cin>>t;
        while(t--)
        {
        	int n;
        	cin>>n;
        	int a[n+1];
        	for(int i=0;i<n;i++)
        	{
        		cin>>a[i];
    		}
    		int i=0;
    		int count=n;
    		while(i!=-1   && count!=0)
    		{
    			i=a[i];
    			count--;
    		}
    		if(i==-1)
    			cout<<"Win"<<endl;
    		else
    			cout<<"Lose"<<endl;
        }
        return 0;
    }
