#include <iostream>

int main()
{
	int tests;
	scanf("%d", &tests);
	//std::cin >> tests;
	
	while (tests-- > 0) {
		int chars;
		scanf("%d", &chars);
		//std::cin >> chars;
		
		while (chars-- > 0) {
			int cValue;
			scanf("%d", &cValue);
			//std::cin >> cValue;
			
			cValue = (255 - cValue);
			
			std::cout << static_cast<char>(cValue);
		}
		
		std::cout << std::endl;
	}
	
	return 0;
}
