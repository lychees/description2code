#include <iostream>
using namespace std;

int main()
{

		int N;
		int C[100000];
		
		cin >> N;
		
		int intersections = 0;
		
		for (int j=0;j<N;j++)
			cin >> C[j];
		
		for ( int j = 0; j < N-1; j++)
		{
			for(int k = j+1; k < N ; k++)
			{
				if(C[j] == C[k])
				{
					
						for(int q = k+1; q < N ; q++)
						{
							for ( int p = j+1 ; p < k ; p++)
							{
							if(C[p] == C[q] && C[p] != C[j])
								intersections++;
						}
					}
				}
			}
		}
		
		cout << intersections << endl;

    return 0;
}
