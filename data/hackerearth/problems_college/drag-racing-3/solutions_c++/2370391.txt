#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	ll n;
	cin>>n;
	int max=0;
	map<string,int> mp;
	ll t=pow(2,n)-1;
	string res_string="";
	while(t--){
		string str1,str2;
		cin>>str1>>str2;
		if(mp.find(str1)!=mp.end()){
			mp[str1]++;
			if(max<mp[str1]){
				max=mp[str1];
				res_string=str1;
			}
		}
		else{
			mp[str1]=1;
			if(max<mp[str1]){
			  max=mp[str1];
				res_string=str1;	
			}
		}
	}
	cout<<res_string;
}
