#include<iostream>
#include<stdlib.h>
#include<string.h>
#include<stdio.h>
using namespace std;
int pal(char str[],int a,int b)
{
	for(int i=a,j=b;i<j;i++,j--)
	{
		if(str[i]!=str[j])
		return 0;
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[30];
		int max=0;
		cin>>str;
		int len=strlen(str);
		for(int i=0;i<len;i++)
		{
			for(int j=i+1;j<len;j++)
			{
				if(pal(str,i,j))
				{
					int check=abs(j-i)+1;
					if(check>max)
					max=check;
				}
			}
		}
		cout<<max<<endl;
	}
}
