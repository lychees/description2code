T = int(raw_input())
for t in range(T):
    x, y, a, b, c = [int(i) for i in raw_input().split()]
    if (a >= b and a >= c) or (b >= a and b >= c):
        ans = x*a + y*b
        if x > y:
            ans1 = 2*y*c + (x-y)*a
        else:
            ans1 = 2*x*c + (y-x)*b
        if ans < ans1:
            ans = ans1
    elif c >= a and c >= b:
        if x > y:
            ans = 2*y*c + (x-y)*a
        else:
            ans = 2*x*c + (y-x)*b
    print ans
