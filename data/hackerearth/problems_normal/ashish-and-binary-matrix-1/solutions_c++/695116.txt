#include <iostream>
#include <set>
using namespace std;

int t,n,m;
int main()
{
    ///cout << "Hello World!" << endl;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	set<string> s;
    	string s1;
    	for(int i=0; i<n; i++)
		{
			cin>>s1;
			s.insert(s1);			
		}
		if(s.size() == n)
			cout<<"Yes\n";
		else 
			cout<<"No\n";
		s.clear();
    }
    return 0;
}
