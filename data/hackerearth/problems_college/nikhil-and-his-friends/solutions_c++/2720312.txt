    #include<iostream>
    #include<cstdio>
    #include<cstring>
    using namespace std;
    int main()
    {
              int t;
              cin>>t;
              while(t--)
              {
              long long int i,j,n,sum,k,flag=0;
              cin>>n>>k;
              long long int a[n];
              for(i=0;i<n;i++)
                        cin>>a[i];
              for(i=0;i<n;i++)
              {
                        sum=0;
                        for(j=i;j<n;j++)
                        {
                        sum+=a[j];
                        if(sum==k)
                        {
                                  flag=1;
                                  break;
                        }
                        if(sum>k)
                                  break;
                        }
                        if(flag==1)
                                  break;
              }
              if(flag==1)
                        printf("YES\n");
              else
                        printf("NO\n");
              }
    }
