'''
# Read input from stdin and provide input before running code

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
st=raw_input()
n=input()
for i in range(n):
	l=raw_input()
	m=l.split()
	a=int(m[0])-1
	b=int(m[1])-1
	l1=len(st)
	if a>l1:
		a=a%l1
	if b>l1:
		b=b%l1
	if st[a]==st[b]:
		print "Yes"
	else:
		print "No"
