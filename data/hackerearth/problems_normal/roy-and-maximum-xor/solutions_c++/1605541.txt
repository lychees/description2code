#include<iostream>

using namespace std;

long mXOR(unsigned long long x, unsigned long long y) {
    unsigned long long z = x ^ y, a = 1;
    while(z){
        z =z/2;
        a = a<<1;
    }
	a=a-1;
    return a;
}

int main()
{
	long tcase;
	
	cin>>tcase;
	
	for(long i = 0;i < tcase; i++)
	{
		unsigned long long n;

		cin>>n;
		if(n>1)
		{
			unsigned long long ans = mXOR(0, n-1);
			cout<<ans<<" "<< (n-(ans+1)/2)*2<<endl;
		}
		else{
			cout<<"0 1"<<endl;
		}
	}	
	return 0;
}
