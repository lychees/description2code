const int MOD = 1000000007;
LL modexp(LL a,LL b)
{
      if(b==0)return 1;
      if(b==1)return a%MOD;
      LL temp = modexp(a,b/2);
      temp = (temp*1LL*temp)%MOD;
      if(b&1)temp = (temp*1LL*a)%MOD;
      return temp;
}
int arr[11]={1,1,1,3,3,15,15,105,105,105,105};
int findr(int n)
{
      if(n<=10)return arr[n];
      if(mp.find(n)!=mp.end())
            return mp[n];
      int a = findr(ceil(n/5.0));
      int b = findr(ceil(n/10.0));
      int c = modexp(2,ceil(n/4.0));
      int ret = (a*1LL*b)%MOD;
      ret = (ret*1LL*c)%MOD;
      return ret;
}
