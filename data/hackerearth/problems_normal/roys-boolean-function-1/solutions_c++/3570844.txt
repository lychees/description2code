#include <bits/stdc++.h>
using namespace std;
bool f;
long int fi(long int n) 
{          
		long int result = n;          
		for(long int i=2;i*i <= n;i++) {            
			if (n % i == 0) result -= result / i;            
			while (n % i == 0) n /= i;          
		}          
		if (n > 1) result -= result / n;          
		return result;        
}   
bool prime(long int a)
{
	if(a==1)
	return false;
	for(long int i=2;i*i<=a;i++)
	{
		if(a%i==0)
		return false;
	}
	return true;
}

int main()
{
    long int t;
    scanf("%ld",&t);
    while(t--)
    {
    	long int z;
    	scanf("%ld",&z);
    	long int k = fi(z);
    	f = prime(k);
    	if(f)
    	printf("TRUE\n");
    	else
    	printf("FALSE\n");
    }
}
