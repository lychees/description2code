#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int main()
{
	ll t,k,i;
	cin>>t;
	while(t--) {
		cin>>k;
		ll x,y,z;
		cin>>x>>y>>z;
		x = x+(y+z);
		ll ans = 1;
		for(i = 0; i < k; i++) {
			ans *= x;
		}
		cout<<ans<<endl;
	}
	return 0;
}
