#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int p=0;
		int n,j,i;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		for( i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(a[i]<=a[j])
				break;
			}
			if(j==n)
		     p++;
		}
		cout<<p<<endl;
			}
    return 0;
}
