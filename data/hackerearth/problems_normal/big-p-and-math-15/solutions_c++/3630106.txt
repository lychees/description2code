#include<iostream>
#include<vector>
#define lli long long int
using namespace std;
int main()
{
    lli a,b;
    cin>>a>>b;
    lli x,y;
    x=a;
    y=b;
    vector<lli> v;
    vector<lli> w;
    while(x!=0)
    {
        int rem=x%10;
        v.push_back(rem);
        x=x/10;
    }
    while(y!=0)
    {
        int rem=y%10;
        w.push_back(rem);
        y=y/10;
    }
    for(lli i=0;i<v.size();i++)
    {
        if(v[i]==5)
            v[i]=6;
    }
    for(lli i=0;i<w.size();i++)
    {
        if(w[i]==5)
            w[i]=6;
    }
    lli s=0;
    for(lli i=v.size()-1;i>=0;i--)
        s=s*10+v[i];
    lli t=0;
    for(lli i=w.size()-1;i>=0;i--)
        t=t*10+w[i];
    lli max_sum=0;
    max_sum=s+t;
    for(lli i=0;i<v.size();i++)
    {
        if(v[i]==6)
            v[i]=5;
    }
    for(lli i=0;i<w.size();i++)
    {
        if(w[i]==6)
            w[i]=5;
    }
    s=0;
    for(lli i=v.size()-1;i>=0;i--)
        s=s*10+v[i];
    t=0;
    for(lli i=w.size()-1;i>=0;i--)
        t=t*10+w[i];
    lli min_sum=0;
    min_sum=s+t;
    cout<<min_sum<<" "<<max_sum;
    return 0;
}
