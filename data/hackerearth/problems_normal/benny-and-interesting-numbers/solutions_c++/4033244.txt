#include<bits/stdc++.h>
using namespace std;
#define max 100000000000

long long prime[316235],cnt=0;
bool check[316235]={false};

void seive()
{
	long long i,j;
	
	i=2,j=2;
	prime[cnt++]=2;
	while(i*j<=sqrt(max))
	{
		check[i*j]=true;
		j++;
	}
	
	i=3,j=2;
	prime[cnt++]=3;
	while(i*j<=sqrt(max))
	{
		check[i*j]=true;
		j++;
	}
	
	for(i=5;i<=sqrt(max)+2;i+=6)
	{
		if(!check[i])
		{
			prime[cnt++]=i,j=2;
			while(i*j<=sqrt(max))
			{
				check[i*j]=true;
				j++;
			}
		}
		if(!check[i+2])
		{
			prime[cnt++]=i+2,j=2;
			while((i+2)*j<=sqrt(max))
			{
				check[(i+2)*j]=true;
				j++;
			}
		}
	}
}

int main()
{
	seive();
	long long t;
	cin>>t;
	while(t--)
	{
		long long n,i,j,k,c=0;
		cin>>n;
		
		for(i=0;i<cnt && prime[i]*prime[i]<=n;i++)
		{
			 if(n%prime[i]==0)
			 {
			 	  c=0,j=n;
			 	  while(j%prime[i]==0 && j>=prime[i])
			 	  {
			 	  	  c++;
			 	  	  j=j/prime[i];
				  }
				  if(c>1)
				  {
				  	cout<<"No\n";
				  	goto hell;
				  }
			 }
		}
		cout<<"Yes\n";
		hell:;
	}	
}
