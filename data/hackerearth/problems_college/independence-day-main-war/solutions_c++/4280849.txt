#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n;
    	double sumx=0,sumy=0;
    	for (int i=0;i<n;i++)
    	{
    		double x,y;
    		cin >> x >> y;
    		sumx+=x;
    		sumy+=y;
    		
    	}
    	double a=sumx/n,b=sumy/n;
    	printf("(%.2lf , %.2lf)\n",a,b);
    	
    	
    }
    return 0;
}
