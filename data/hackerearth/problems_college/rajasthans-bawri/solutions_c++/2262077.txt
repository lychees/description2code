#include <iostream>
using namespace std;

int t,bx,by,px,py;

int main()
{
	scanf("%d",&t);
	while(t--){
		scanf("%d%d%d%d",&px,&py,&bx,&by);
		if(px*py < bx*by){ printf("Not Possible\n");}
		else{
			if( (px>=bx && py>=by) || (py>=bx && px>=by) ){ printf("Possible\n");}
			else{ printf("Not Possible\n");}
		}
	}
    return 0;
}
