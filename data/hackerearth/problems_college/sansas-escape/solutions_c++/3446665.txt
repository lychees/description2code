ll n;
ll a[N], st[4*N];
void build(ll id,ll s,ll e)
{
    if(s==e)
    {
        st[id] = a[s];
        return ;
    }
    ll mid= (s+e)/2;
    build(id*2,s,mid);
    build(id*2+1,mid+1,e);
    st[id] = st[id*2] ^ st[id*2+1];
}
ll query(ll id,ll s,ll e,ll l,ll r)
{
    if(s>e||s>r||e<l)
    {
        return 0;
    }
    if(s>=l&&r>=e)
    {
        return st[id];
    }
    ll mid = (s+e)/2;
    return query(id*2,s,mid,l,r) ^ query(id*2+1,mid+1,e,l,r);
}

    ll q;

    build(1,0,n-1);
    while(q--)
    {
        ll l, r;
        cin >> l >> r;

        cout << query(1,0,n-1,l,r) << "\n";
    }
    return 0;
}
