#include<stdio.h>
#include<math.h>
long long table[100015][11];
int num[100005];
void init(int a)
{
    int i,j;
    for(i=a;i>0;i--)
    {
        for(j=1;j<11;j++)
        {
            table[i][j]=num[i];
            if(i+j<=a)
                table[i][j]+=table[i+j][j];
        }
    }
}
int main()
{
    int i,j,tp,N,Q,L,R,K;
    scanf("%d %d",&N,&Q);
    for(i=1;i<N+1;i++)
        scanf("%d",&num[i]);
    init(N);
    for(i=0;i<Q;i++)
    {
        scanf("%d %d %d",&L,&R,&K);
        tp=(ceil)((R+1-L)*1.0000/K);
        printf("%lld\n",table[L][K]-table[L+tp*K][K]);
    }
}
