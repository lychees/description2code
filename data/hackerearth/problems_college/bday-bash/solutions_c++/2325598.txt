#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long c,sp,bf,f;
    	cin>>c>>sp>>bf>>f;
    	if(c%25!=0){
    		cout<<"No\n";
    		continue;
    	}
    	sp*=100;
    	bf*=50;
    	f*=25;
    	long long spr,bfr,fr;
    	spr=bfr=fr=0;
    	if(c>=sp){
    		spr++;
    		c-=sp;
    	}
    	if(c>=bf){
    		bfr++;
    		c-=bf;
    	}
    	if(c>=f){
    		fr++;
    		c-=f;
    	}
    	while(1){
    		if(c>=sp){
	    		spr++;
	    		c-=sp;
    		}
    		if(c>=bf){
    			bfr++;
    			c-=bf;
    		}
    		if(c>=f){
    			fr++;
    			c-=f;
    		}
    		else break;    		
    	}
    	if(c!=0||f!=0&&fr==0||bf!=0&&bfr==0||sp!=0&&spr==0)
    		cout<<"No\n";
    	else cout<<spr*sp<<" "<<bfr*bf<<" "<<fr*f<<"\n";
    }
    return 0;
}
