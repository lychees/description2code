#include <iostream>
using namespace std;

unsigned long long a[1000000];

int main()
{
    unsigned long long t,n;
    cin >> t;
    a[0] = 1;
	a[1] = 1;
	a[2] = 2;
	a[3] = 4;
    while(t){
    	cin >> n;
    	if(n>3 && a[n]==0){
	    	for(int i=4;i<=n;i++){
	    		a[i] = (a[i-1]+a[i-2]+a[i-3])%1000000007 ;	
	    	}
    	}
    	if (n>0)cout << a[n]<<endl;
    	else cout << 0<<endl;
    	t--;
    }
    return 0;
}
