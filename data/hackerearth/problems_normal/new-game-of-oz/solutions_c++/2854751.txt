#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
using namespace std;
static ll a[1000009];
int main(){
	ll t,n,c;
	inp(t);
	while(t--){
		c = 0;
		inp(n);
		for(ll i=0;i<n;++i){
			inp(a[i]);
		}
		sort(a,a+n);
		a[n] = 0;
		for(ll i=0;i<n;++i){
			if(a[i]+1==a[i+1]){
				++c;++i;
			}else{
				++c;
			}
		}
		printf("%lld\n",c);
	}
	return 0;
}
