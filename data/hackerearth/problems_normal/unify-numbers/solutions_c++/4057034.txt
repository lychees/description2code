#include <iostream>
#include <vector>
#include <stdlib.h>
#include <algorithm>
using namespace std;

bool mycomp(long long a, long long b){
	return a <b;
}

int main()
{
	int t=0;
	cin >> t;
	long long n,temp,mean,res;
	vector< long long> array;
	while(t--){
		res=0;
		array.clear();
		cin >> n;
		for(long int i=0;i<n;i++){
			cin >> temp;
			array.push_back(temp);
		}
		std::sort(array.begin(),array.end(),mycomp);
		mean = array[array.size()/2];
		for(long int i=0;i<n;i++){
			res += abs(array.at(i) - mean);
		}
		cout << res <<endl;
	}
    return 0;
}
