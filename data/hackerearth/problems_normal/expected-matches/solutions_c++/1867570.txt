#include<cstdio>
#include<algorithm>
#include<vector>
#include<cstring>
#include<cmath>
#include<stack>
#include<queue>
#include<string>
#include<iostream>
#include<map>
#include<stdlib.h>
#include<time.h>  
using namespace std;
 
#define sd(a) scanf("%d",&a)
#define pd(a) printf("%d\n",(a))
#define LL long long
#define F first
#define S second
#define MP make_pair
#define PB push_back
double dp[1010][1010];
double p[1010][1010];
double ans;
int main()
{
	int t,n,m,i,j;
	sd(t);
	while(t--)
	{
		sd(n);
		sd(m);
		for(i=0;i<n;++i)
		{
			for(j=0;j<m;++j)
			{
				scanf("%lf",&p[i][j]);
			}
		}
		dp[0][0]=1;
		for(i=0;i<n;++i)
		{
			for(j=0;j<m;++j)
			{
				if(i||j)
					dp[i][j]=0;
				if(j>0)
					dp[i][j]=dp[i][j]+dp[i][j-1]*p[i][j-1];
				if(i>0)
					dp[i][j]=dp[i][j]+dp[i-1][j]*(1-p[i-1][j]);
			}
		}
		ans=0;
		for(i=0;i<n;++i)
		{
			ans=ans+dp[i][m-1]*p[i][m-1]*(i+m);
		}
		for(i=0;i<m;++i)
		{
			ans=ans+dp[n-1][i]*(1-p[n-1][i])*(n+i);
		}
		printf("%.6lf\n",ans);
	}
}
