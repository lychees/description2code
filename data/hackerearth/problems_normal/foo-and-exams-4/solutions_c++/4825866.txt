#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int t;
    long long a,b,c,d,k,low,high,mid,stress,ans;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c>>d>>k;
    low=0;
    high=(int)cbrt(k);
    ans=(low+high)/2;
    while(low<=high){
    	mid=(low+high)/2;
    	stress=a*pow(mid,3)+b*pow(mid,2)+c*mid+d;
    	if(stress<k){
    		low=mid+1;
    		ans=mid;
    	}
    	else if(stress>k){
    		high=mid-1;
    	}
    	else{
    		ans=mid;
    		break;
    	}
    }
    cout<<ans<<endl;
    }
    return 0;
}
