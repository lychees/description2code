#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		string str;
		cin >> str;
		str[0] -= '0';
		str[1] -= '0';
		int hrs = str[0]*10 + str[1];
		cin >> str;
		str[0] -= '0';
		str[1] -= '0';
		int mins = str[0]*10 + str[1];
		if (hrs == 12) {
			hrs = 0;
		}
		mins *= 6;
		
		int p = mins / 12;
		hrs = hrs*30 + p;
		
		int diff = abs(hrs - mins);
		if (diff > 180) {
			diff = 360 - diff;
		}
		
		cout << diff << endl;
	}
    return 0;
}
