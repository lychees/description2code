#include<iostream>
#include<cstring>
#include<stdio.h>
#include<algorithm>
using namespace std;
long long int a[1000005],b[1000005]={0};
int main()
{
	long long int n,k,g1,g2,p1,p2,g,p;
	scanf("%lld%lld",&n,&k);
	int i,j;
	b[0]=0;
	scanf("%lld",&a[0]);
	b[1]=a[0];
	for(i=1;i<n;i++)
	{
		scanf("%lld",&a[i]);
		
		b[i+1]=a[i]+b[i];
	}
	for(i=0;i<k;i++)
	{
		long long int flag=0,ans=0;
		scanf("%lld%lld",&g1,&p1);
		scanf("%lld%lld",&g2,&p2);
		g=min(min(g1,g2),min(p1,p2));
		p=max(max(p1,p2),max(g1,g2));
		if(max(g1,g2)<min(p1,p2))
		{
			flag=1;
		}
		if(min(g1,g2)>max(p1,p2))
		{
			flag=1;
		}
		if(max(g1,g2)<min(p1,p2))
		{
			flag=1;
		}
		if(max(p1,p2)<min(g1,g2))
		{
			flag=1;
		}
		if(flag==1)
		{
			ans=ans+b[max(g1,g2)+1]-b[min(g1,g2)];
			ans=ans+b[max(p1,p2)+1]-b[min(p1,p2)];
		}
		if(flag==0)
		{
			ans=b[p+1]-b[g];	
		}
		printf("%lld\n",ans);
	}
	return 0;
}
