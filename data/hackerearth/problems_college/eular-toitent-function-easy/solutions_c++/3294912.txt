#include <iostream>
using namespace std;
int cop(int n)
{
	int c,i,j,k,n1;
	for(i=2,c=1;i<n;i++)
	{
		j=i;
		n1=n;
		while(j!=1)
		{
			k=n1%j;
			if(k==0)
			break;
			n1=j;
			j=k;
			
		}
		if(k==1)
		c++;
		
	}
	return c;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<cop(n)<<"\n";
    }
    return 0;
}
