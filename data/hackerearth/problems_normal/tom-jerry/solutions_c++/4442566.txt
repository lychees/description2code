#include<bits/stdc++.h>
using namespace std;
#define ll long long
int vis[110][110];
ll n, ans=0, mat[110][110];
void solve(ll x, ll y){
	if(!x || !y || x > n || y > n)return;
	if(mat[x][y] == 1)return;
	if(x == n && y == n){ans++;return;}
	if(vis[x][y] == 1)return;
	vis[x][y]=1;
	solve(x-1, y);
	solve(x+1, y);
	solve(x, y-1);
	solve(x, y+1);
	vis[x][y]=0;
}

int main(){
	cin>>n;
	for(int i=1; i<=n; i++)
	for(int j=1; j<=n; j++)
		cin>>mat[i][j];
	solve(1, 1);
	cout<<ans<<endl;
	return 0;
}
