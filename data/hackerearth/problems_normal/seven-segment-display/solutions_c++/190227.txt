#include<iostream>
using namespace std;
char s[1000001];
main()
{
long long t,c,i;
cin>>t;
while(t--)
{c=0;
cin>>s;
for(i=0;s[i];i++)
{if(s[i]=='0')c+=6;
else if(s[i]=='1')c+=2;
else if(s[i]=='2')c+=5;
else if(s[i]=='3')c+=5;
else if(s[i]=='4')c+=4;
else if(s[i]=='5')c+=5;
else if(s[i]=='6')c+=6;
else if(s[i]=='7')c+=3;
else if(s[i]=='8')c+=7;
else c+=6;
}
cout<<c<<endl;
}
return 0;
}
