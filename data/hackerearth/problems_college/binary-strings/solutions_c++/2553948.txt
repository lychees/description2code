#include <stdio.h>
#include <iostream>
using namespace std;
#define mod 1000000007
long long f[1000001];
int main()
{
    //printf("Hello World!\n");
    /* Declare an array to store Fibonacci numbers. */
  //int f[n+1];
  int i;
 
  /* 0th and 1st number of the series are 0 and 1*/
  f[0] = 1;
  f[1] = 1;
  
  for (i = 2; i <= 1000000; i++)
  {
      /* Add the previous 2 numbers in the series
         and store it */
      f[i] = (f[i-1] + f[i-2])%mod;
  }
  int t;
  cin>>t;
  while(t--){
  	int n;
  	cin>>n;
  	cout<<f[n]<<endl;
  }
  
    return 0;
}
