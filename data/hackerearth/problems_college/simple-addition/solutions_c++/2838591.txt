#include<iostream>
using namespace std;

int depth(int x,int y);

int main(){
	int T;
	int d=0,dNw=0;
	int n[40];
	int x,y;
	cin>>T;
	while(T--){
		cin>>n[T];
		d = 0;
		for(int i=1;i<=n[T]/2;i++){
			x=i;
			y=n[T]-i;
			dNw=0;
			while(1){
				if(x==1 && y==1){
					dNw++;
					break;
				}
				else if(x<=0 || y<=0){
					dNw=-1;
					break;
				}
				else if(x>y){
					x=x-y;
				}
				else{
					y=y-x;
				}
				dNw++;
			}
			
			if((dNw<d && dNw>0) || d==0){
				d = dNw;
			}
		}
		cout<<d<<"\n";
	}
}
