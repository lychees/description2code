#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int  main(int argc, char const *argv[])
{
	int T;
	cin>>T;
	while(T--){
		ll n;
		cin>>n;
		ll a[n+1],i;
		for (i = 1; i <= n; ++i)
		{
			cin>>a[i];
		}
		ll res=0,j,one=0,zero=0;
		for(j=1;j<=n;j++){
			if(a[j]==1)
				one++;
			else
				zero++;
		}
		if(one%2!=0)
			res+=one;
		else
			res+=zero;
		if(zero==n)
			cout<<n<<endl;
		else
		cout<<res<<endl;
	}
	return 0;
}
