#include <iostream>
using namespace std;

int main()
{
    int T;
    long long N;
    
    cin>>T;
    
    for(int k=0;k<T;k++)
    {
        cin>>N;
        long long arr[N];
        long long e=0,o=0;
        
        for(int i=0;i<N;i++)
        {
            cin>>arr[i];
        }
        
        for(int i=0;i<N;i++)
        {
        	long long q = (arr[i])/2;
        	int check = (arr[i]) - q*2;
        	if(check==0)
        		e++;
        	else
        		o++;
        }
       	
        cout<<(e*o)<<endl;
    }
    return 0;
}
