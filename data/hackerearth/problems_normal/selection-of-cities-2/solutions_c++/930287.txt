#include <iostream>
#include <stdio.h>
using namespace std;
long long mul(long long a,long long b, long long mod)
{
	long long ans=1;
	while(b)
	{
		if(b%2)
			ans=(ans*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long n;
    	scanf("%lld",&n);
    	long long ans=mul(2,n,1000000007)-1;
    	printf("%lld\n",ans);
    }
    return 0;
}
