
#include <iostream>
#include<algorithm>
#include<math.h>
#include<vector>
using namespace std;
 int table[1300][20];
int main()
{
   int n;
   cin>>n;
   int sum=0;
   vector<int> strength;
   for(int i=0;i<n;i++)
   {
   	int s;
   	cin>>s;
   	sum+=s;
   	strength.push_back(s);
   }
   sort(strength.rbegin(),strength.rend());
   if(n==3) cout<<strength[0];
   else{
   int req=ceil(sum/3.0);

   for(int i=0;i<=n;i++) table[0][i]=1;
   for(int i=1;i<=sum;i++) table[i][0]=0;
   for(int i=1;i<=sum+1;i++){
    for(int j=1;j<=n;j++){
        table[i][j]=table[i][j-1];
        if(strength[j-1]<=i)
            table[i][j]=table[i][j]||table[i-strength[j-1]][j-1];
    }
   }int ans=0;
   for(int i=sum;i>=req;i--)
   {
       if(table[i][n]) ans=i;
   }
   /*for(int i=0;i<n;i++) cout<<strength[i]<<" ";
   cout<<endl;
   cout<<sum;
   cout<<req;
   int ans=0;
   for(int i=0;i<n;i++)
   {
   	if(ans>=req) break;
   	ans+= strength[i];
   }*/
   cout<<ans;
   }
}
