#include<iostream>
#include<stack>
#include <queue>
#include<string.h>
#include<algorithm>

using namespace std;


int main()
{
	int t,a[150]={0},d;
	char w[31];
	cin>>t;
	while(t--)
	{
		for(int i=0;i<150;i++)
		a[i]=0;
		
		cin>>w;
		for(int i=0;i<strlen(w);i++)
		{
			d=w[i];
			if(a[d]==0)
			{
				a[d]=1;
				cout<<w[i];
			}
		}
		cout<<endl;
	}
	return 0;
}
