ll power(ll a,ll b)
{
	if(!b)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp);
	if(b&1)
		temp=(temp*a);
	return temp;
}

void pre_order( int val, vector<pp >& m)
{
	if( val )
	{
		cout << val <<" ";
		pre_order( m[val].first, m);
		pre_order( m[val].second, m);
	}
}

			m[val].first = x;

		}
