#include <iostream>
using namespace std;

int main()
{
	int t,n,k;
	cin>>t;
	
	int students,tm;
	
	while(t--){
		students=0;
		cin>>n>>k;
		
		for(int i=0;i<n;i++){
			cin>>tm;
			
			if(tm<=0){
				students++;
			}
		}
		
		if(students>=k){
			cout<<"NO\n";
		}else{
			cout<<"YES\n";
		}
	}
    return 0;
}
