#include <iostream>
#include <algorithm>
#include <string.h>
using namespace std;

struct node {
	char letter;
	int weight;
};

bool acompare(node lhs, node rhs) 
{ 
	if(lhs.weight < rhs.weight)
		return true;
	else if(lhs.weight > rhs.weight)
		return false;
	else
		return lhs.letter > rhs.letter; 
}

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		node L[26];
		char c;
		for(int i=0, c = 'a'; i<26; i++, c++){
			L[i].letter = c;
			L[i].weight = 0;
		}
		
		for(int i=0; i<s.size(); i++){
			L[s[i] - 'a'].weight++;
		}
		
		sort(L, L+26, acompare);
		
		for(int i=0; i<26; i++){
			cout<<L[i].letter<<" ";
		}
		cout<<endl;
		
	}
	return 0;
}
