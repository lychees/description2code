#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main(){
int t,food,gold,sam,pal,champ,powerBadshah;
int nsam,npal,nchamp,maxPower,ans;
scanf("%d",&t);
while(t--){
        maxPower=-1;
        scanf("%d%d%d%d%d%d",&food,&gold,&sam,&pal,&champ,&powerBadshah);

        for(nsam=0;nsam*100<=food;nsam++){
            for(npal=0;npal*125+nsam*100<=food && npal*50<=gold;npal++){
                int foodLeft = food - nsam*100 - npal*125;
                int goldLeft = gold -npal*50;

                nchamp = min(foodLeft/50,goldLeft/100);
                maxPower = max(maxPower,nsam*sam+npal*pal+nchamp*champ);

            }
        }
        if(maxPower>powerBadshah)
            ans = maxPower - powerBadshah;
        else
            ans=-1;
        
        printf("%d\n",ans);
    }

return 0;
}
