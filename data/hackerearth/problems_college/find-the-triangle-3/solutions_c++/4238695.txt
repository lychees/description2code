#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main(){
	ll t,n;
	ll arr[100005];
	ll brr[3];
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++) cin>>arr[i];
		sort(arr,arr+n);
		ll flag=0;
		ll a,b,c;
		for(int i=n-1;i>=2;i--){
			for(int j=i-1;j>=1;j--){
				for(int k=j-1;k>=0;k--){
					a=arr[i];
					b=arr[j];
					c=arr[k];
					if(a+b>c && a+c>b && b+c>a){
						flag=1;
						brr[0]=a;
						brr[1]=b;
						brr[2]=c;
						sort(brr,brr+3);
						break;
					}
				}
				if(flag==1) break;
			}
			if(flag==1) break;
		}
		(flag==0)?(cout<<"-1\n"):(cout<<brr[0]<<" "<<brr[1]<<" "<<brr[2]<<endl);
	}
	return 0;
}
