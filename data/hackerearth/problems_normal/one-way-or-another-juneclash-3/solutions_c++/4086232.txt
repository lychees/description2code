#include <bits/stdc++.h>
using namespace std;
#define ull unsigned long long

vector < pair <ull, ull> > odd, even;

int main()
{
	ios::sync_with_stdio(0);
	ull t, n, r, c, x, f, i, j, dx, dy;
	cin >> t;
	while(t --)
	{
		odd.clear();
		even.clear();
		cin >> n;
		f = 1;
		while(n --)
		{
			cin >> r >> c >> x;
			if(x % 2)
				odd.push_back(make_pair(r, c));
			else
				even.push_back(make_pair(r, c));
		}
        for(i = 0; i < odd.size(); i ++)
        {
        	for(j = 0; j < odd.size() && j != i; j ++)
        	{
        		dx = llabs(odd[i].first - odd[j].first);
        		dy = llabs(odd[i].second - odd[j].second);
                if(dx % 2 && ! (dy % 2) || ! (dx % 2) && (dy % 2))
                {
                	f = 0;
                	break;
                }
        	}
        }
        for(i = 0; i < even.size(); i ++)
        {
        	for(j = 0; j < even.size() && j != i; j ++)
        	{
        		dx = llabs(even[i].first - even[j].first);
        		dy = llabs(even[i].second - even[j].second);
                if(dx % 2 && ! (dy % 2) || ! (dx % 2) && (dy % 2))
                {
                	f = 0;
                	break;
                }
        	}
        }
        if(! f)
        	cout << "No \n";
        else
        	cout << "Yes \n";
	}
	return 0;
}
