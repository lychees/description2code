#include <iostream>
using namespace std;

int main()
{
   int t;
   long int n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	int a[n],flag=0;
   	long int sum=0;
   	for(int i=0;i<n;i++)
   	{
   		cin>>a[i];
   		sum=sum + a[i];
   	}
   	long int sum1=0;
   	for(int i=0;i<n;i++)
   	{
   		
   		if(sum1==(sum-a[i]-sum1))
   		{
   			flag=1;
   			break;
   		}
   		sum1=sum1 + a[i];
   	}
   	if(flag)
   	cout<<"YES"<<endl;
   	else
   	cout<<"NO"<<endl;
   }
    return 0;
}
