#include <bits/stdc++.h>
using namespace std;
int cmp(string a, string b)
{
	return a+b < b+a;
}
string a[100005];
int main()
{
   // cout << "Hello World!" << endl;
   int n;
   scanf("%d",&n);
   int i;
   for(i=0;i<n;i++)cin>>a[i];
   sort(a,a+n,cmp);
   string ans = "";
   for(i=0;i<n;i++)ans += a[i];
   cout<<ans;
    return 0;
}
