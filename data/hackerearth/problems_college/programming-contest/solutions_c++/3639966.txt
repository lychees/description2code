lld solve(string A,string B,string C,lld a,lld b,lld c){
	// cout<<A<<B<<C;
    lld t=0,p,r=0;
    rep(i,0,a-1){
        if(a-i>b) continue;
        p=1;
        rep(j,i,a-1) if(A[j]!=B[j-i]) p=0;
        if(p==1){
            t=a-i;
            break;
        }
    }
    A=B;
    B=C;
    a=b;
    b=c;
    rep(i,0,a-1){
        if(a-i>b) continue;
        p=1;
        rep(j,i,a-1) if(A[j]!=B[j-i]) p=0;
        if(p==1){
            r=a-i;
            break;
        }
    }
    // cout<<p<<" "<<r<<endl;
    return t+r;
}
