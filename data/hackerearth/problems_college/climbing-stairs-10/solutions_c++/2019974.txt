#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
long long int arr[10000];
int main()
{
	int t;
	arr[0]=1;
	arr[1]=1;
	arr[2]=2;
	for(int i=3;i<=1000;i++)
	{
		arr[i]=arr[i-1]+arr[i-2];
	}
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<arr[n]<<endl;
	}
}
