#include<bits/stdc++.h>
using namespace std;

bool isPrime(int n) {
    if(n<2) {
        return false;
    }
    for(int i=2;i*i<=n;i++) {
        if(n%i==0) {
            return false;
        }
    }
    return true;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int n;
        scanf("%d",&n);
        if(n==3) {
            puts("3");
        } else if(isPrime(n)==true) {
            puts("2");
        } else {
            puts("1");
        }
    }
    return 0;
}
