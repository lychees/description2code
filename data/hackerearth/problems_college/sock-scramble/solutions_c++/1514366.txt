#include<iostream>
using namespace std;
int sp(long int a[],int t)
{
		int r=a[0];
		for(int i=1;i<t;i++)
		{
			r ^=a[i];
		}
		return r;
}
 int main()
 {
		int n;
		cin>>n;
		int t;
		long int a[100000];
		while(n--)
		{
			cin>>t;
			for(int i=0;i<t;i++)
			{
				cin>>a[i];
			}
			cout<<sp(a,t)<<endl;	
			}
			
		return 0;
		}
