#include <iostream>
using namespace std;

int main()
{
    string s;
    
    cin>>s;
    long long c=0,p=1;
    long long count[26]={0};
    for(long long i=s.length()-1;i>=0;i--) {
    	
    	if(s[i]>='a' && s[i]<='z') {
    		count[s[i]-'a']+= c;
    		c = 0;
    		p = 1;
    	}
    	else {
    		c+= (s[i]-'0')*p;
    		p*=10;
    	}
    }
	long long prev = count[0];

    for(long long i=1;i<26;i++) {
    	if(count[i]>0) {
    		count[i] += prev;
    		//cout<<" "<<count[i];
    		prev = count[i];
    	}
    }
    long long q,k;
    
    cin>>q;
    
    while(q) {
    	
    	cin>>k;
    	bool flag = false;
    	for(long long i=0;i<26;i++) {
    		char ch;
    		if(count[i]>=k) {
    			ch = 'a' + i;
    			cout<<ch<<endl;
    			flag = true;
    			break;
    		}
    	}
    	if(!flag)
    		cout<<-1<<endl;
    	q--;
    }
    
    return 0;
}
