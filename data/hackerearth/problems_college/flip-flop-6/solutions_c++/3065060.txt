#include <bits/stdc++.h>
#define ll long long
using namespace std;
string s;
int len;
bool works(int r)
{
	int i,ans=0;
	char p=s.at(0);
	for(i=1;i<len;i++)
	{
		if(s.at(i)!=p)
		{
			p=s.at(i);
		}
		else
		ans++;
	}
	//~ cout<<ans<<" "<<r<<endl;
	if(ans>r)
	return 0;
	return 1;
}
int myfind(int l,int h)
{
	int ans,m;
	while(l<=h)
	{
		m=((h-l)>>1)+l;
		if(works(m))
		{
			ans=m;
			h=m-1;
		}
		else
		{
			l=m+1;
		}
	}
	return ans;                                                                                                                                                                                                                                                                                                                                                                                                                     
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int test;
	cin>>test;
	while(test--)
	{
		cin>>s;
		int ans;
		len=s.length();
		ans=myfind(0,len);
		cout<<ans<<endl;
	}
	return 0;
}
