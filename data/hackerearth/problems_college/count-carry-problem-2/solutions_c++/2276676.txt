#include<bits/stdc++.h>
using namespace std;


typedef long long int LL;

#define scll(x) scanf("%lld",&x);
#define sci(x) scanf("%d",&x);
#define prll(x) printf("%lld\n",x)
#define pri(x) printf("%d\n",x)


int main()
{
	LL a,b,c,x,y,t;
	scll(t);
	while(t--)
	{
		c=0;
		scll(a);
		scll(b);
		while(a>0 && b>0){
			x=a%10; y=b%10;
			if((x+y)>9)
				c++;
			a=a/10; b=b/10;	
		}
		
		if(c==0)
		cout<<"No carry operation\n";
		else if(c==1)
		cout<<"1 carry operation\n";
		else
		cout<<c<<" carry operations\n";
	}
	return 0;
}
