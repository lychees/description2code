#!/usr/bin/env python
# encoding: utf-8

def gao(x, a, n, m):
    i = 0
    times = 0
    while i < n:
        j = i
        y = x
        times += 1
        while j < n and a[j] <= y:
            y -= a[j]
            j += 1
        if j == i:
            return False
        i = j
    return True if m >= times else False


n, m = map(int, raw_input().split())
a = map(int, raw_input().split())

lb, ub = 0, sum(a)
while ub - lb > 1:
    mid = (lb + ub) / 2
    if gao(mid, a, n, m):
        ub = mid
    else:
        lb = mid

print ub
