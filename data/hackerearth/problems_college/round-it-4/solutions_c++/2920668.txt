#include <iostream>
using namespace std;

int main()
{
    int t,i,multiple,low,high,ans,tlow,thigh;
    cin>>t;
    int n[t],b[t];
    for(i=0;i<t;i++)
    {
    	cin>>n[i]>>b[i];
    }
    for(i=0;i<t;i++)
    {
    	multiple = n[i]/b[i];
    	low=b[i]*multiple;
    	tlow=n[i]-low;
    	high = b[i]*(multiple+1);
    	thigh = high-n[i];
    	if(tlow<thigh)
    	ans = low;
    	else
    	ans = high;
    	cout<<ans<<endl;
    }
}
