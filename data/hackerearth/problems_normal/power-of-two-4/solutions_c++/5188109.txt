#include <iostream>
#include<cmath>
#include<algorithm>
using namespace std;


int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n,max=-1;
    	cin >> n;
    	int arr[n];
    	for(int i=0;i<n;i++)
    	{
    	cin >> arr[i];
    	if(max < arr[i])
    	max=arr[i];
    	}
    	bool ans=false;
    	int len=log2(max);
    	for(int i=0;i<=len;i++)
    	{
    		int mask= 1 << i;
    		int temp=-1;
    		for(int j=0;j<n;j++)
    		if(( arr[j] & mask)!=0)
    		{
    			if(temp == -1)
    			temp=arr[j];
    			else
    			temp= temp & arr[j];
    		}
    		if(temp == mask)
    		{
    		ans=true;
    		break;
    		}
    	}
    	if(ans)
    	cout << "YES" << endl;
    	else
    	cout << "NO" << endl;
    }
    return 0;
}
