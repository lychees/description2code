#include <iostream>
#include <cmath>
#include <cstdio>
using namespace std;

int gcd(int a,int b)
{
	int r;
	while(b!=0)
	{
		r=a%b;
		a=b;
		b=r;
	}
	return a;
}
int arr[101][101];
int count[101];
int main()
{
	for(int i=1;i<=100;++i)
	{
		for(int j=i;j<=100;++j)
		{
			arr[i][j]=gcd(j,i);
			arr[j][i]=arr[i][j];
		}
	}
	for(int i=1;i<=100;i++)
    	{
    		for(int j=i+1;j<=100;++j)
    		{
    			for(int k=j+1;k<=100;++k)
    			{
    				count[arr[i][arr[j][k]]]++;
    			}
    		}
    	}
	int t;
    long long x,z;
    cin>>t;
    while(t--)
    {
    	cin>>x>>z;
    	int ans=0;
    	for(int i=1;i<=100;++i)
    	{
    		if(abs(x-z)%i==0) ans+=count[i];
    	}
    	double a=1.0*ans/(100*33*49);
    	printf("%.6f\n",a);
    }
    return 0;
}
