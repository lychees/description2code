n = input()
m = map(int, raw_input().split())
l = []
for i in xrange(n):
    l.append(1)
for i in xrange(n):
    for j in xrange(i):
        if m[i] > m[j] and l[i] < l[j]+1:
            l[i] = l[j]+1

print bin(max(l))[2:]
