#include<iostream>
using namespace std;
#include<map>
#include<string>

int main() {
	long t, i, l, j, r, u, b, y, result;
	string str;
	std::map<char, long> myMap;
	
	cin>>t;
	for(i=0;i<t;i++) {
		cin>>str;
		myMap.clear();
		result = 0;
		l = str.length();
		for(j=0;j<l;j++)
			myMap[str[j]]++;
		r = myMap['r'];
		u = myMap['u'];
		b = myMap['b'];
		y = myMap['y'];
		while(r!=0 && u!=0 && b!=0 && y!=0) {
			r--;
			u--;
			b--;
			y--;
			result++;
		}
		cout<<result<<endl;
	}
	return 0;
}
