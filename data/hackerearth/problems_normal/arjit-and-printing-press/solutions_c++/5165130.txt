#include <iostream>
#include <string>
#include <algorithm>
#include <functional>
using namespace std;
#define size 10000
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string word;
    	cin>>word;
    	string reserve;
    	cin>>reserve;
    	sort(reserve.begin(),reserve.end(),less<char>());
    	for(int i=0;i<reserve.length();i++)
             for(int j=i;j<word.length();j++)
                  {
                      if(reserve[i]<word[j])
                      {
                          word[j]=reserve[i];
                          break;
                      }

                  }
        cout<<word<<endl;
   }

    return 0;
}
