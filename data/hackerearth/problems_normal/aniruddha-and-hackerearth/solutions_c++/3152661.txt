#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
const int MAX =1e6+3;
long long sum[MAX]={0};
int main()
{
	ios_base::sync_with_stdio(0);
    
    vector<long long int>v;
    v.push_back(0);
    v.push_back(2);
    v.push_back(3);
   	sum[1]=2;
    sum[2]=5;
    	for(int i=3;i<MAX;i++)
    		{
    			v.push_back((v[i-1]+v[i-2])%mod);
    			sum[i]=(sum[i-1]+v[i])%mod;
    		}
    		
    long long t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<sum[n]<<endl;
    }
    
    return 0;
}
