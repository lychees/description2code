cp = int(raw_input())

dom = map(int, raw_input().split())
brian = map(int, raw_input().split())

b_max = 0
d_max = 0

for x in range(cp-1):
    d_max = max(d_max, abs(dom[x+1] - dom[x]))
    b_max = max(b_max, abs(brian[x+1] - brian[x]))

if b_max == d_max:
    print "Tie"
elif b_max > d_max:
    print "Brian"
    print b_max
else:
    print "Dom"
    print d_max
