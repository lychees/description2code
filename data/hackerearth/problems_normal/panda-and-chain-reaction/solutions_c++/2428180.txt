#include<iomanip>
using namespace std;

long long a[1000005];
const int mod=1000003;
int main()
{
	int t;
	long long n,k,ans;
	scanf("%d",&t);
	a[0]=1;
	for(long long i=1 ; i<=1000005; i++){
		a[i]=i*a[i-1];
		a[i]%=mod;
	}
	while(t--){
		scanf("%lld%lld",&n,&k);
		k%=mod;
		if(n>mod)n=mod;
		ans=k*a[n];
		ans%=mod;
		printf("%lld\n",ans);
	}
	return 0;
}
