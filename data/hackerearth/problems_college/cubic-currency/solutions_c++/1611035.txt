#include <cstdio>  
#include <iostream>
using namespace std;
const int maxn = 10000;  
    
long long dp[maxn];  
int a[22];  
    
int main()  
{  
	
    int i, j, n, t, p;
    cin>>t;
    for(p=0; p<t; p++)
    {
    for (i = 1; i <= 21; ++i)  
        a[i] = i * i * i;  
    for (i = 0; i <= maxn; ++i)  
        dp[i] = 1;  
    for (j = 2; j <= 21; ++j)  
        for (i = a[j]; i <= maxn; ++i)  
            dp[i] += dp[i - a[j]];  
    while (~scanf("%d", &n))  
        printf("%lld\n", dp[n]);  
    }
    return 0;  
}
