#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp1[1000009];
ll dp2[1000009];
int main()
{
    ll n,m,l,r,i,x,q;
    scanf("%lld",&n);
    scanf("%lld",&m);
    memset(dp1,0,sizeof(dp1));
    memset(dp2,0,sizeof(dp2));
    for(i=0;i<m;i++)
    {
        scanf("%lld %lld",&l,&r);
        l--;
        r--;
        dp1[l]++;
        dp1[r+1]--;
    }
    dp2[dp1[0]]++;
    for(i=1;i<n;i++)
    {
        dp1[i]+=dp1[i-1];
        dp2[dp1[i]]++;
    }
    /*for(i=0;i<n;i++)
        cout<<i<<"-->"<<dp1[i]<<" ";
    cout<<endl;
    for(i=0;i<=m;i++)
        cout<<i<<"-->"<<dp2[i]<<" ";
    cout<<endl;*/
    for(i=m-1;i>=0;i--)
        dp2[i]+=dp2[i+1];
    /*for(i=0;i<=m;i++)
        cout<<i<<"-->"<<dp2[i]<<" ";
    cout<<endl;*/
    scanf("%lld",&q);
    while(q--)
    {
        scanf("%lld",&x);
        printf("%lld\n",dp2[x]);
    }
    return 0;
}
