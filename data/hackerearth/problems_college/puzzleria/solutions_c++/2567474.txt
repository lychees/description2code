long long int com(long long int n,long long int k)
{
	if(n==k||k==0)
	    return 1;
	    else return com(n-1,k-1)+com(n-1,k);
}

long long int fact(long long int n)
{
	if(n==0||n==1)
	   return 1;
	  else return n*fact(n-1);
}
