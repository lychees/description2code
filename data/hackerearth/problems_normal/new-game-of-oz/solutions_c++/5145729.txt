#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-- >0)
    {
    	int n,i=0,count=0;
    	cin>> n;
    	int a[10000];
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	sort(a,a+n);
    		
    	for(i=n-1;i>=0;){
    		//cout<<"a[i-1] "<<a[i-1]<<endl;
    		//cout<<"a[i]+1"<<a[i]+1<<endl;
    		if(a[i]==(a[i-1]+1)){
    		//	cout<<a[i]<<" ";
    		count++;
    	//	cout<<count<<endl;
    		i=i-2;
    		}
    		else{
    		//	cout<<a[i]<<" ";
    		count++;
    		i--;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
