#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	long long int TC, NUM, swapCount, boyCount;
	string S;
	char genSearch;
	cin >> TC;
	while(TC--)
	{
		swapCount=boyCount=0;
		cin >> NUM >> S;
		for(long long int i=0;i<2*NUM;i+=2)
		{
			if(S[i]=='B')
				boyCount+=1;
		}

		if(boyCount > NUM - boyCount)
			genSearch='B';
		else if(boyCount < NUM - boyCount)
			genSearch='G';
		else
			genSearch=S[0];

		for(long long int i=0;i<2*NUM;i+=2)
		{
			if(S[i]!=genSearch)
				swapCount+=1;
		}
		cout << swapCount << endl;
	}
}
