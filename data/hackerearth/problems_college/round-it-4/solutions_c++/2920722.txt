#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n, b, p;
    	cin>>n>>b;
    	p=n%b;
    	if(b%2==0)
    	{
    		if(p<b/2)
    		cout<<n-p<<endl;
    		else
    		cout<<n+b-p<<endl;
    	}
    	else
    	{
    		if(p<=b/2)
    		cout<<n-p<<endl;
    		else
    		cout<<n+b-p<<endl;
    	}
    }
    return 0;
}
