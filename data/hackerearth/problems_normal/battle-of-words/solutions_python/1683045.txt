t=input()
while t>0:
	alice = raw_input()
	bob   = raw_input()
	a = [0 for x in xrange(1,30)]
	for x in alice:
		if x == ' ': continue
		a[ord(x)-ord('a')]= (a[ord(x)-ord('a')]+1) 
	for x in bob:
		if x == ' ': continue
		a[ord(x)-ord('a')]= (a[ord(x)-ord('a')]-1) 
	alice = 0
	bob   = 0
	for x in a:
		if x < 0:
			bob = 1
		elif x > 0:
			alice = 1
	if bob == alice:
		print 'You draw some.'
	elif bob == 1:
		print 'You lose some.'
	else:
		print 'You win some.'
	t=t-1
