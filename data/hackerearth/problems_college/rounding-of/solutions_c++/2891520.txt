#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
	int t,i,b,n;
	cin>>t;
	while(t--)
	{
		
		cin>>n>>b;
		i=0;
		if(n<(float)b/2)
		cout<<"0"<<endl;
		else
		if(n%b==0)
		cout<<n<<endl;
		else
		
		if(n>=(float)b/2 && n<=b)
		cout<<b<<endl;
		else
		{
		while(1)
		{
			b*i;
			if(b*i>n)
			break;
			i++;
			
		}
		if((b*i)-n>n-(b*(i-1)))
		cout<<b*(i-1)<<endl;
		else
		if((b*i)-n<=n-(b*(i-1)))
		
		cout<<b*i<<endl;
	}
		
	}
return 0;
}
