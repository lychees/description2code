#include <iostream>
using namespace std;

int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
    	long x=0,y=0;
    	cin>>x>>y;
    	long d=x^y;
    	int count=0;
    	while(d!=0)
        {
        	if((d%2)==1)
        	count++;
        	d/=2;
        }
        cout<<count<<endl;
 }
    return 0;
}
