#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int X,Y,n,w,p1,p2;
    	cin>>X>>Y>>n>>w>>p1>>p2;
    	
    	long double a1=p1/100.0,a2=p2/100.0;
    	long double dp[n+1][w+1];
    	for(int i=0;i<=n;i++)
    	{
    		for(int j=0;j<=w;j++)
    		{
    			
    			if(j==0)
    			dp[i][j]=1;
    			else if(i==0)
    			dp[i][j]=0;
    			else
    			{
    				dp[i][j] = max(a1*dp[i-1][max(j-X,0)] + (1-a1)*dp[i-1][max(j,0)], a2*dp[i-1][max(j-Y,0)] + (1-a2)*dp[i-1][max(j,0)]); 
    			}
    		}
    	}
    	
    	long double ans=dp[n][w]*100;
    	cout<<fixed<<setprecision(6);
    	cout<<ans<<endl;
    }
    return 0;
}
