ll sl(){ 
  int c =gc();
  while((c<'0' || c>'9') && c!='-') {
    c = gc();
  }
  bool sign = 0;
  if(c=='-') {
    sign = 1;
    c = gc();
  }
  ll ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  if(sign) return -ret;
  return ret;
}

int ss(char *s) {
  int n = 0;
  int c = gc();
  while(!((c>='a' && c<='z') || (c>='A' && c<='Z'))) c = gc();
  while(((c>='a' && c<='z') || (c>='A' && c<='Z'))) {
    s[n] = (char)(c);
    n++;
    c = gc();
  }
  return n;
}

int p[100005], r[100005];

int find(int x) {
  if(p[x]==x) return x;
  return p[x] = find(p[x]);
}

void uni(int x, int y) {
  if(r[x]<=r[y]) {
    p[x] = y;
    if(r[x]==r[y]) r[y]++;
  }
  else p[y] = x;
}
