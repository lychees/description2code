T = int(raw_input())
for t in range(0, T):
    String = raw_input()
    freq = {}
    for s in String :
        if s in freq.keys() :
            freq[s] += 1
        else :
            freq[s] = 1
    insertion = 0
    one_odd = False
    for key in freq.keys() :
        if freq[key]%2 == 0 :
            continue
        elif freq[key]%2 == 1 and one_odd :
            insertion += 1
        else :
            one_odd = True

    print(insertion)
