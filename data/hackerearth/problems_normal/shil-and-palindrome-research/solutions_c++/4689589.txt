void update(ll idx ,ll val, ll i) {
	while (idx <= MAX) {
		BIT[idx][i] += val;
		idx += (idx & -idx);
	}
}
void read(int index,int* a)
{
    while(index>0)
    {
        for(int i=1;i<=26;i++)
        {
            a[i]+=BIT[index][i];
        }
        index -= (index&-index);
    }
}
void print(int n)
{
    for(int i=1;i<=n;i++)
       {for(int j=1;j<=26;j++)
            cout<<BIT[i][j]<<" ";
        cout<<endl;}
}
