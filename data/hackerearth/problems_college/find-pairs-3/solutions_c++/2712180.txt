#include<iostream>
#include<cstring>
using namespace std;
//Given an array A of n integers,find number of distinct pairs(i,j) such that A[i]==A[j] & j>=i
//-1000000<A[i]<1000000
int positive[1000006];//store frequency of positive integers of Array
int negative[1000006];//store frequency of negative integers of Array
int main(void){
	int tests;
	cin>>tests;
	while(tests--){
		memset(positive,0,sizeof(positive));
		memset(negative,0,sizeof(negative));
		int n,input;
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>input;
			if(input>=0)positive[input]++;
			else negative[-input]++;
		}
		int ans=0;
		for(int i=0;i<1000006;i++){//check for all possible values of A[i]
			if(positive[i]==0 and negative[i]==0)continue;
			else{
				if(positive[i]>0)
					ans=ans+((positive[i]*(positive[i]+1))/2);
				if(negative[i]>0)
					ans=ans+((negative[i]*(negative[i]+1))/2);
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
