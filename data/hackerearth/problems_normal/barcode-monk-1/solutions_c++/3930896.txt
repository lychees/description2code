#include <cstdio>

int n, k, w[1000], ans[1001][1010] = {{0}}, mod = 1000000007;

int main()
{
	scanf("%d %d", &n, &k);
	for (int i = 0; i < k; i++)
	{
		scanf("%d", &w[i]);
	}
	ans[0][0] = 1;
	ans[0][1001] = 1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 0; j < k; j++)
		{
			if (w[j] > i)
			{
				continue;
			}
			for (int l = 0; l <= n; l++)
			{
				if (ans[i-w[j]][l] > 0 && ans[i][l+w[j]] == 0)
				{
					ans[i][l+w[j]] = ans[i-w[j]][l];
				}
			}
		}
		ans[i][0] = ans[i-1][1001];
		for (int j = 0; j <= n; j++)
		{
			ans[i][1001] += ans[i][j];
			ans[i][1001] %= mod;
		}
	}
	printf("%d\n", ans[n][1001]);
}
