int main() {
	int T;
	scanf("%d", &T);
	rep(ii, T) {
		int P, x, y, u, v;
		scanf("%d%d%d%d%d", &P, &x, &y, &u, &v);
		int dist;
		if (((x + y) & 1) != ((u + v) & 1))
			dist = INF;
		else if (mp(x, y) == mp(u, v))
			dist = 0;
		else if (x + y == u + v || x - y == u - v)
			dist = 1;
		else
			dist = 2;
		puts(dist < P ? "Counter-Terrorists Win !" : "Terrorists Win !");
	}
	return 0;
