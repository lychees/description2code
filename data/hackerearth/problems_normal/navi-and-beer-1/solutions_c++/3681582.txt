  #include <iostream>
    using namespace std;

    int main()
    {
    	unsigned long long int n,k,m,t;
    	
        cin>>t;
		while(t--)
		{
        cin>>n>>k>>m;
        if(n==0 || k==0)
        	{
        		cout<<"0"<<endl;
        	}
        else
        {
        	unsigned long long int ans=0;
        	ans= (n*(n+1)/2);
        	ans*=ans;
        	ans*=k;
        	if(ans < m)
        		cout<<"0"<<endl;
        	else
        		cout<< (ans-m)<<endl;
        }
		}
        return 0;
    }
