#include<bits/stdc++.h>

using namespace std ;
int main()
{
    int n,temp ;
    cin>>n;
    vector< vector<int> > cum(n+1, vector<int>(105,0));
    vector<int> place ;
    for(int i = 0 ; i<n;i++)
    {
         cin>>temp;
         place.push_back(temp);
        cum[i][temp]+=1;
        for( int  j = 0 ; j <103 ; j++)
            cum[i+1][j]=cum[i][j];

    }
    int q;
    cin>>q;
    while(q--)
    {
        int s ,e ,val;
        cin>>s>>e>>val;
        if(place[s]==val)
        cout<<cum[e][val]-cum[s][val]+1<<endl;
        else
            cout<<cum[e][val]-cum[s][val]<<endl;
    }




    return 0 ;
}
