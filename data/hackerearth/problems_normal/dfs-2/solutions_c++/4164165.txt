#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll n,q,i,l,r;
    cin>>n;
    vector<ll> v(n);
    for(i=0;i<n;i++)
    	scanf("%lld",&v[i]);
    sort(v.begin(),v.end());
    cin>>q;
    while(q--)
    {
    	scanf("%lld%lld",&l,&r);
    	ll s=0,e=n-1;
    	ll ans1=-1;
    	while(s<=e)
    	{
    		ll m=(s+e)>>1;
    		if(v[m]>=l)
    		{
    			ans1=m;
    			e=m-1;
    		}
    		else
    			s=m+1;
    	}
    	e=n-1;
    	ll ans2=-1;
    	if(ans1!=-1)
    	{	s=ans1;
    	while(s<=e)
    	{
    		ll m=(s+e)>>1;
    		if(v[m]<=r)
    		{
    			ans2=m;
    			s=m+1;
    		}
    		else
    			e=m-1;
    	}
    	}
    	//cout<<ans1<<" "<<ans2<<endl;
    	if(ans1==-1 && ans2==-1)
    		cout<<"0\n";
    	else
    		cout<<ans2-ans1+1<<endl;
    }
    return 0;
}
