#include<cstdio>
#include<algorithm>
#include<vector>

using namespace std;

typedef long long ll;

const int MAXN = 2e3 + 4;

int n, cnt[MAXN][MAXN];
int e[MAXN][MAXN];
ll ans;

void f(ll c){
	ans += c * (c - 1)/ 2;
}

int main(){
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		for (int j = 0; j < n; j++){
			bool x;
			scanf("%d", &x);
			if (x)
				e[i][j / 31] |= 1<<(j % 31);
		}

	for (int i = 0; i < n; i++)
		for (int j = 0; j < i; j++){
			int c = 0;
			int w;
			for (w = 0; w + 31 < i; w += 31)
				c += __builtin_popcount(e[i][w/31] & e[j][w/31]);
			for (; w < i; w++)
				if (((e[i][w/31] >> (w % 31)) & 1) && ((e[j][w/31] >> (w % 31)) & 1))	c++;
			f(c);
		}
	printf("%lld\n", ans);
	return	0;
}
