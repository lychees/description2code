#include <iostream>
using namespace std;
#define ll long long 
int main()
{
    int t;
	scanf("%d",&t);
	while(t--)
	{
              int x;
              scanf("%d",&x);
              int a[500],m=1;
              a[0]=1;
              for(int i=1;i<=x;i++)
              {
                      int temp=0,x;
                      int indx=0;
                      while(indx<m)
                      {
                                   x=a[indx]*i+temp;
                                   temp=x/10;
                                   a[indx]=x%10;
                                   indx++;
                      }
                      if(temp!=0)
                      {
                                 while(temp!=0)
                                 {
                                               a[indx]=temp%10;
                                               temp/=10;
                                               indx++;
                                               m++;
                                 }
                      }
              }
              ll count=0;
              for(int i=0;i<m;i++)
              {
              	      if(a[m-i-1]==4|| a[m-i-1]==7)
              	      count++;
                      //cout<<a[m-i-1];
              }
              printf("%lld\n",count);
                      
	}
	//system("pause");
	return 0;
}
