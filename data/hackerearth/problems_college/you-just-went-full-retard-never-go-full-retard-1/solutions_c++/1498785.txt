#include <iostream>
#include<cmath>
#include <cstdint>
using namespace std;
 bool isPrime(long long int i) {
        if (i < 2) {
            return false;
        } else if (i % 2 == 0 && i != 2) {
            return false;
        } else {
            for (int j = 3; j <= sqrt(i); j = j + 2) {
                if (i % j == 0) {
                    return false;
                }
            }
            return true;
        }
    }

int main()
{
   
  long long int t,n;
 cin>>t;
   while(t--)
   {
  cin>>n;
  if(isPrime(n))
  cout<<"YES\n";
  else
  cout<<"NO\n";

   }
    return 0;
}
