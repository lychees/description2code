#!usr/bin/python
from __future__ import print_function

n = input()

for i in range(0, n):
    if (i == 0 or i == (n - 1)):
        for j in range(0, n):
            print('*', end='')
        print(end='\n')

    else:
        for j in range(0, n - i - 1):
            print(' ', end='')

        print('*')
