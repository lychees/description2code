#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin >> n;
	int a[n];
	for(int i = 0; i < n; ++i) cin >> a[i];
	
	int max_so_far, max_cur;
	max_so_far = max_cur = a[0];
	for(int i = 1; i < n; ++i) {
		max_cur = max(0, max_cur + a[i]);
		max_so_far = max(max_cur, max_so_far);
	}
	cout << max_so_far << endl;
}
