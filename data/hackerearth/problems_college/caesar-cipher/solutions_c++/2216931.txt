#include <iostream>
#include <string>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
   
 
    int k;
    cin>>k;
   string s;
   // cin>>n;
    cin>>s;
    int n=s.length();
    k%=26;
    for(int i=0;i<n;i++)
    {
        int c=s[i];
        if(c>='a' && c<='z') 
        {
            c+=k;
            if(c>'z')
            {
               c=96+(c%122); 
            }
        }
        else if(c>='A' && c<='Z') 
        {
            c+=k;
            if(c>'Z') 
            {
                c=64+(c%90); 
            }
        }
        cout<<(char)c;
    }
    cout<<endl;
	}
    return 0;
}
