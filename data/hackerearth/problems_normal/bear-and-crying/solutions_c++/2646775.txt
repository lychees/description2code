template <class T>T Max2(T a,T b){return a<b?b:a;}
template <class T>T Min2(T a,T b){return a<b?a:b;}
template <class T>T Max3(T a,T b,T c){return Max2(Max2(a,b),c);}
template <class T>T Min3(T a,T b,T c){return Min2(Min2(a,b),c);}
template <class T>T Max4(T a,T b,T c,T d){return Max2(Max2(a,b),Max2(c,d));}

using namespace std;

lli bit_count(lli _x){lli _ret=0;while(_x){if(_x%2==1)_ret++;_x/=2;}return _ret;}
lli bit(lli _mask,lli _i){return (_mask&(1<<_i))==0?0:1;}
lli powermod(lli _a,lli _b,lli _m){lli _r=1;while(_b){if(_b%2==1)_r=(_r*_a)%_m;_b/=2;_a=(_a*_a)%_m;}return _r;}

string s;
lli ans,n;
vector<lli> v;
lli dp[100010];
lli dp2[100010];
lli sub;

int main()
{
#ifndef ONLINE_JUDGE
    freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
#endif
    std::ios::sync_with_stdio(false);
    srand (time(NULL));

    cin>>s;
    n=s.size();
    int b=0,e=n-1;
    while(s[b]=='_' && b<n)
        b++;
    while(s[e]=='_' && e>=0)
        e--;
    if(e-b<=1)
    {
        cout<<0;
        return 0;
    }
    for(int i=b;i<=e;i++)
    {
        lli c=1;
        while(i+1<=e && s[i+1]==s[i])
            i++,c++;
        v.push_back(c);
    }
    /*for(lli x:v)
        cout<<x<<" ";*/
    for(int i=1;i<v.size();i+=2)
    {
        lli add;
        add=(v[i-1]*v[i+1])%MOD;
        add=(add*(powermod(2,v[i],MOD)-1))%MOD;
        ans=(ans+add)%MOD;

        if(i>=3)
        {
            add=(dp[i-2]*v[i+1])%MOD;
            add=(add*powermod(2,v[i],MOD))%MOD;
            ans=(ans+add)%MOD;

            dp[i]=(dp[i-2]*powermod(2,v[i],MOD))%MOD;
            add=(v[i-1]*powermod(2,v[i],MOD))%MOD;
            dp[i]=(dp[i]+add)%MOD;
        }
        else
        {
            dp[i]=(v[i-1]*powermod(2,v[i],MOD))%MOD;
        }

        if(i>=3)
        {
            sub=(sub+dp2[i-2]*v[i+1])%MOD;
            dp2[i]=(dp2[i-2]+v[i-1])%MOD;
        }
        else
        {
            dp2[i]=v[i-1];
        }
    }
    ans=(ans-sub)%MOD;
    ans=(ans+MOD)%MOD;
    cout<<ans;

    return 0;
}
