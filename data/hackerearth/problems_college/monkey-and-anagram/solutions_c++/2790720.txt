#include <bits/stdc++.h>
using namespace std;

int main()
{
	int p,i,cnt;
	bool flag;
	char s[1001],t[1001];
    cin>>p;
    while(p--){
    	int arr[26] = {0};
    	cin>>s>>t;
    	for(i=0;i<strlen(t);i++)
    		arr[t[i]-'a']++;
    	flag = false;
    	for(i=0;i<strlen(s);i++){
    		if(arr[s[i]-'a'] == 0){
    			flag = true;
    			break;
    		}
    		else
    			arr[s[i]-'a']--;
    	}
    	if(flag)
    		cout<<-1<<endl;
    	else{
    		cnt=0;
    		for(i=0;i<26;i++)
    			if(arr[i]!=0)
    				cnt++;
    		cout<<cnt<<endl;
    	}
    }
    return 0;
}
