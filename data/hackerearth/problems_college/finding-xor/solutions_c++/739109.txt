#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while (t--)
    {
    	char n1[10000];
    	char n2[10000];
    	cin>>n1>>n2;
    	int l1=strlen(n1);
    	for(int i=0;i<l1;i++)
    	{
    		if(n1[i]==n2[i])
    			cout<<0;
    		else
    			cout<<1;
    	}
    	cout<<endl;
    }
    return 0;
}
