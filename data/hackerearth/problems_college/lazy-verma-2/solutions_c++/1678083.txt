#include <bits/stdc++.h>
#define DEBUG 0

using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	long long n, m;
	cin >> n >> m;
	long long a = 0, b = 0, x;
	for (int i = 0; i < n; ++i) {
	    cin >> x;
	    a += x;
	}
	for (int i = 0; i < m; ++i) {
	    cin >> x;
	    b += x;
	}

	cout << a * b;



    return 0;
}
