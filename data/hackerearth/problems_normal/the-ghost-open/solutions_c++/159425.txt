#include<iostream>
using namespace std;
int main()
{
short t;
long n,matches=0;
cin>>t;
while(t--)
{
    cin>>n;
    while(n>1)
    {
        
        if(n%2==0)
            {
                matches =matches + n/2;
             n=n/2;
            }
            else
                {
                     matches = matches + (n)/2;
                n=(n)/2+1;
                }
    }
    cout<<matches<<endl;
    matches =0;
}
    return 0;
}
