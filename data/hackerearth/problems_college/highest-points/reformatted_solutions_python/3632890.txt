def score_team(w, t, l):
    w = int(w)
    t = int(t)
    l = int(l)
    return (3*w)+(t)-(l)


l = []
t = float(raw_input())
for i in range(0, int(t)):
    temp = []
    b_temp = '0'
    a = raw_input()
    for j in range(0, len(a)):
        if (a[j] != ' '):
            b_temp = b_temp+a[j]
        if (a[j] == ' ' or j == len(a)-1):
            temp.append(b_temp)
            b_temp = '0'
    l.append(score_team(temp[0], temp[1], temp[2]))
print(max(l))
