#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	int c=0;
	cin>>t;
	string s;
	while(t>0)
	{
	    cin>>s;
	    c=0;
	    for(int i=0;i<(s.length()-1);i++)
	    {
	        if(((s[i]=='X' &&s[i+1]=='Y'))||(s[i]=='Y' &&s[i+1]=='X'))
	        continue;
	        else
	        c+=1;
	    }
	    cout<<c<<endl;
	    t--;
	}
	return 0;
}
