#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll a[10000001];
ll mod = 1e9 + 7;
void fn()
{
	int i;
	a[0] = 1;
	for(i = 1 ; i < 1000001; i++) {
		a[i] = 3*a[i-1];
		a[i] = a[i] % mod;
	}
}
int main()
{
    ll t, n,k1,k2;
    cin>>t;
    fn();
    while(t--) {
    	cin>>n;
    	if(n <= 1000000)
    		cout<<a[n]<<endl;
    	else {
    		k1 = n/1000000;
    		k2 = n%1000000;
    	//	cout << k1 << " " << k2 << endl;
    		ll ans = 1,data = a[1000000];
    	//	cout << data << endl;
    		for(int i = 0; i < (int)k1; i++) {
    			ans = ans * data;
    			ans %= mod;
    		}
    		ans *= a[(int)k2];
    		ans %= mod;
    		cout << ans << endl;
    	}
    }
    return 0;
}
