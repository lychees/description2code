def gcd(a,b):
    rem=1
    div=a
    divd=b
    while rem!=0:   
        rem=divd%div
        if rem==0:
            return div
        else:
            divd=div
            div=rem
        

a,query=map(int,raw_input().split())
l=[]
for i in range(query):
    b=int(raw_input())
    nothing=gcd(a,b)
    if nothing in l:
        print -1
    else:
        l.append(nothing)
        print nothing
