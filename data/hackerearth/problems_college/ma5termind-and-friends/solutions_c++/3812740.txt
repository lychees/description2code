#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
#define sc(x) scanf("%d",&x)
#define sc2(x,y) scanf("%d%d",&x,&y)
#define sc3(x,y,z) scanf("%d%d%d",&x,&y,&z)
#define sc4(x,y,z,u) scanf("%d%d%d%d",&x,&y,&z,&u)
#define scll(x) scanf("%lld",&x)
#define scll2(x,y) scanf("%lld%lld",&x,&y)
#define scll3(x,y,z) scanf("%lld%lld%lld",&x,&y,&z)
#define pr(x) printf("%d\n",x)
#define prll(x) printf("%lld\n",x)
ll a[100005],b[100005];
struct ok
{
    ll l,r,c;
}ab[100005];
int main()
{
    ll i,j,k,l,m,x,y,n,t,q;
    scll(t);
    while(t--)
    {
        scll2(n,m);
        for(i=1;i<=n;i++)
            a[i]=0;
        for(i=1;i<=m;i++)
        {
            scll3(ab[i].l,ab[i].r,ab[i].c);
        }
        scll(q);
        for(i=1;i<=m;i++)
            b[i]=0;
        while(q--)
        {
            scll2(x,y);
            b[x]+=1;
            b[y+1]-=1;
        }
        for(i=2;i<=m;i++)
            b[i]+=b[i-1];
        for(i=1;i<=m;i++)
        {
            ab[i].c=ab[i].c*b[i];
        }
        for(i=1;i<=m;i++)
        {
            a[ab[i].l]+=ab[i].c;
            a[ab[i].r+1]-=ab[i].c;
        }
        for(i=2;i<=n;i++)
        {
            a[i]+=a[i-1];
        }
        for(i=1;i<=n;i++)
            printf("%lld ",a[i]);
        printf("\n");
    }
    return 0;
}
