#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    vector<int>strength(n+1),ans(n+1,0),fight(n);
    for(int i=1;i<=n;i++)
    	scanf("%d",&strength[i]);
    for(int i=0;i<n;i++)fight[i]=i+1;
    
    while(1)
    {
    	vector<int>temp;
    	int l=fight.size();
    	if(l%2==0)
    	for(int i=0;i<l;i++)
    	{
    		ans[fight[i]]++;
    		ans[fight[i+1]]++;
    		if(strength[fight[i]]>strength[fight[i+1]])
    			temp.push_back(fight[i]);
    		else temp.push_back(fight[i+1]);
    		i++;
    	}
    	else {
	    	for(int i=0;i<l-1;i++)
	    	{
	    		ans[fight[i]]++;
	    		ans[fight[i+1]]++;
	    		if(strength[fight[i]]>strength[fight[i+1]])
	    			temp.push_back(fight[i]);
	    		else temp.push_back(fight[i+1]);
	    		i++;
    		}
    		temp.push_back(fight[l-1]);
    	}
    	fight.clear();
    	fight=temp;
    	if(fight.size()<=1)
    		break;
    }
    int t;
    for(int i=0;i<q;i++)
    {
    	scanf("%d",&t);
    	printf("%d\n",ans[t]);
    }
    return 0;
}
