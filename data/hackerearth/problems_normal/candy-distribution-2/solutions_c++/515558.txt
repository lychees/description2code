#include <bits/stdc++.h>


using namespace std;




int main()
{
    ios_base::sync_with_stdio(false);

    register unsigned long long t,n,sum,i,x;
    cin>>t;
    while(t--)
    {

        cin>>n;
        sum=0;
        for(i=0;i<n;i++)
        {
            cin>>x;
            sum+=x;
        }
        sum-=(n-1);
        cout<<sum<<"\n";
    }

    return 0;
}
