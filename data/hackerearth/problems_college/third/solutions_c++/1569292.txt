#include <iostream>
using namespace std;
#include <algorithm>

long int answer(int a[], int n)
{	
	if(n==0)
		return 0;
	int ans[n];
	ans[0] = a[0];
	if(n==1)
		return ans[0];
	ans[1] = max(a[0], a[1]);
	
	for(int i=2; i<n; i++)
	{
		ans[i] = max(ans[i-1], ans[i-2]+a[i]);
	}
	return ans[n-1];
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0; i<n; i++)
    		cin>>a[i];
    	cout<<answer(a, n)<<endl;
    }
    return 0;
}
