N = int(raw_input())
i = 0
main = []
while i < N:
    data = raw_input()
    main.append(data)
    i += 1

for s in main:
    main1 = []
    main1 = list(s)
    main1.reverse()

    if "".join(main1) in main:
        main.remove("".join(main1))
        char = main1[len(main1)/2]
        print("%d %c" % (len(main1), char))
