#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
int main(int argc, char const *argv[])
{  vector<string> ansr;
	int ts;
	cin>>ts;
	while(ts--)
  {string te="";
	string s;
	int i;
	bool l=false;
	vector<char> ans;
	int  c,j=0,ic;
	vector<int>  v;
	cin>>ic>>s;
	v.push_back(0);
	if(ic==0)ans.push_back(s.at(0));
    for(i=1;i<s.size();i++)
    {
    	if(s.at(i)=='(')
    		{l=true;continue;}
    	else if(s.at(i)=='.')
    	    {l=false; continue;}
    	else if(s.at(i)==')')
    	    {v.erase(v.begin()+j);j--;continue;}
    	else
    		{c=v.at(j);
    		if(l)
    			c=c-1;
    		else 
    			c=c+1;
    		l=false;
    		if(c==ic){ans.push_back(s.at(i));}
    		v.push_back(c);
    		j++;
    		//cout<<"\n"<<s.at(i)<<c;
      		}
    }

if(ans.size()>0)
{
    sort(ans.begin(),ans.end());
    for(i=0;i<ans.size();i++)te+=ans.at(i);
    	ansr.push_back(te);
 }
else
	ansr.push_back("Common Gandhijee!");
}
for(int j=0;j<ansr.size();j++)
cout<<ansr.at(j)<<"\n";

	return 0;
}
