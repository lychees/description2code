#include <iostream>
using namespace std;

int main()
{
    int t,i=0;
    cin>>t;
    while(i<t)
    {  int a,b,c,score=0;
       cin>>a>>b>>c;
      if(a==9||b==9||c==9)
      {if(a==9)
       { score=10;goto label;}
       else{ if(a>6&&a<9)score=2;
             else if(a>0&&a<7)score=1;}
       if(b==9)
       {
       	score=score+10-3;goto label1;
       }
       else{if(b>6&&b<9)score=score+2;
            else if(b>0&&b<7)score=score+1;
           }
      if(c==9)
    	{
    		score=score+10-6;goto label2;
    	}
      else{ if(c>6&&c<9)score=score+2;
            else if(c>0&&c<7)score=score+1;}
      }
      else
      {score=0;}
        label:
        label1:
        label2:
        
    	cout<<score<<endl;
    	i++;
    }
    return 0;
}
