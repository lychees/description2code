#include <bits/stdc++.h>
using namespace std;
int n,arr[1000009];
int num1[1000009],num2[1000009];
int main(){
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>arr[i];
    }
    for(int i=1;i<=n;i++){
        num1[arr[i]]++;
        num2[arr[i]]++;
    }
    for(int i=1000000;i>=0;i--){
        num1[i] += num1[i+1];
    }
    for(int i=1;i<=1000000;i++){
        num2[i] += num2[i-1];
    }
    long double ans1 = 0 , ans2 = 0;
    for(int i=1;i<=n;i++){
        long double a = (1/(double)n) , b = (num1[arr[i]+1]/(double)(n-1));
        ans1 += a * b ;
    }
    for(int i=0;i<=n;i++){
        long double a = (1/(double)n) , b = (num2[arr[i]-1]/(double)(n-1));
        ans2 += a * b ;
    }
    cout.precision(6);
    cout<<fixed;
    cout<<ans1<<" "<<ans2<<endl;
}
