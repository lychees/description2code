#include <iostream>
#include<math.h>
#include<vector>
using namespace std;
int cnt=1;
vector<vector<int> > a(1025,vector<int> (1025));
int recur(int i,int j,int l)
{
	if(l==1)
		{a[i][j]=cnt++;return 0;}
	recur(i,j,l/2);
	recur(i,j+l/2,l/2);
	recur(i+l/2,j,l/2);
	recur(i+l/2,j+l/2,l/2);
}

int main()
{
	int n;
	cin>>n;
	int l=pow(2,n);
	cnt=1;

	recur(0,0,l);

	for(int i=0;i<l;i++)
	{
		for(int j=0;j<l;j++)
		{
			cout<<a[i][j]<<" ";
		}
		cout<<endl;
	}
    return 0;
}
