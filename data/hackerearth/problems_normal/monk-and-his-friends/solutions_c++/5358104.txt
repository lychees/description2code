#include <bits/stdc++.h>
using namespace std;
#define ll long long

int main(int argc, char const *argv[])
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,m,num;
	cin>>n>>m;
	set<ll>s;
	while(n--)
	{
		cin>>num;
		s.insert(num);
	}	
	while(m--)
	{
		cin>>num;
		if(s.find(num) != s.end())
			cout<<"YES"<<endl;
		else
		{
			cout<<"NO"<<endl;
			s.insert(num);
			
		}
	}
	
	}
	
	return 0;
}
