typedef long long ll;
#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
#include <vector>
#include <string>
#include <string.h>
#include <map>
#include <queue>
#include <stack>
#include <sstream>
#include <fstream>
#include <stdlib.h>
#include <utility>

using namespace std;
ll dp[2][26];
int main()
{
    ll  n,j,i,c;
    string x;
    scanf("%lld",&n);
    for(i=0;i<2;i++)
    {
        for(j=0;j<26;j++)
        dp[i][j]=0;
    }
    for(j=0;j<n;j++)
    {
        cin>>x;
        for(i=0;i<x.length();i++)
        {
                if(j==0)
                dp[1][x[i]-97]=1;
                else
                {
                    if(dp[0][x[i]-97]==1)
                    dp[1][x[i]-97]=1;
                }
        }
        for(i=0;i<26;i++)
        {
            dp[0][i]=dp[1][i];
            dp[1][i]=0;
        }
    }
    c=0;
    for(i=0;i<26;i++)
    {
        if(dp[0][i]==1)
        c++;
    }
    printf("%lld\n",c);
	return 0;
}
