#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long n,m,i,x,y,num;
	priority_queue <pair<long long,long long>> q;
	map<long long,long long> mp;
	pair<long long,long long> a[1000000];
scanf("%lld %lld",&n,&m);
for(i=0;i<n;i++){
	scanf("%lld",&num);
	mp[num]++;
	q.push(make_pair(mp[num],num));
	a[i]=q.top();
	x=a[i].second;
	y=a[i].first;
	printf("%lld %lld\n",x,y);
}
    return 0;
}
