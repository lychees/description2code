#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 100;
int a[N];
map<int, int> m;

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		for (int i = 0; i < n; ++i) cin >> a[i];
		int k;
		cin >> k;
		m.clear();
		long long ans = 0;
		for (int i = 0; i < n; ++i) {
			if (k - a[i] >= 0) ans += m[k-a[i]];
			++m[a[i]];
		}
		cout << ans << '\n';
	}
    return 0;
}
