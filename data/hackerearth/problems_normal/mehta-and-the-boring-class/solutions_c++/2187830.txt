#include <bits/stdc++.h>
using namespace std;

int main()
{
    int arr[1000001]={0};
    for(int i=2;i<1000001;i++)
    {
    	if(arr[i]==0)
    	{
    		for(int j=1;j<(1000001/i)+1;j++)
    		{
    			int temp=j;
    			int ct=0;
    			while(temp%i==0)
    				{
    					ct++;
    					temp=temp/i;
    				}
    			arr[i*j]+=ct+1;
    		}
    	
    	}
    }
    for(int i=1;i<1000001;i++)
    	arr[i]+=arr[i-1];
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int a,b;
    	scanf("%d%d",&a,&b);
    	printf("%d\n",(arr[b]-arr[a-1]));
    }
    
}
