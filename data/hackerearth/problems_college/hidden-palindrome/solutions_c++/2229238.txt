#include<bits/stdc++.h>
using namespace std;
int n;
string s;
int dp[30][30];
int precomp()
{
	int i,j,k;
	int mx=1;
	for(i=0;i<n;i++)
	{
		dp[i][i]=1;
	}
	for(i=0;i<n-1;i++)
	{
		if(s[i]==s[i+1])
		{
			dp[i][i+1]=2;
		}
	}
	for(i=2;i<n;i++)
	{
		for(j=0;j<n-i;j++)
		{
			k=j+i;
			if(s[j]==s[k] && dp[j+1][k-1])
			{
				dp[j][k]=2+dp[j+1][k-1];
				mx=max(mx,dp[j][k]);
			}
		}
	}
	return mx;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		s.clear();
		cin>>s;
		n=s.length();
		int k=precomp();
		cout<<k<<endl;
	}
	return 0;
}
