using namespace std;
 
typedef unsigned long long int ull;
typedef long long int ll;
typedef vector<vector<ll> > mat;

vector<pll> v[100005];ll a[100005];

int main()
{ios_base::sync_with_stdio(false);
int n,m,s;
cin>>n>>m>>s;
rep(i,1,n)cin>>a[i];
while(m--)
{
	int a,b,c;
	cin>>a>>b>>c;
	v[a].pb(mp(b,c));
	v[b].pb(mp(a,c));
}
ll d[100005];
rep(i,0,n+1)d[i]=1e18;

	d[s] = 0;
	priority_queue < pll > q;
	q.push (make_pair (0, s));
	while (!q.empty()) 
	{
		int vv = q.top().second,  cur_d = -q.top().first;
		q.pop();
		if (cur_d > d[vv])  continue;
 
		for (size_t j=0; j<v[vv].size(); ++j) 
		{
			int to = v[vv][j].first,
				len = v[vv][j].second;
			if (d[vv] + len < d[to]) 
			{
				d[to] = d[vv] + len;
				q.push (make_pair (-d[to], to));
			}
		}
	}
	ll sum=0,cc=n,s1=0;
rep(i,1,n)
{
 	if(d[i]==1e18){d[i]=0,cc--;continue;}
 	sum+=d[i]*a[i];
 	s1+=a[i];
}

rep(i,1,n)
{
	if(d[i]==0&&i!=s)
	ol("0 ");
	else
	{
		ol(sum-d[i]*a[i]+d[i]*(s1-a[i]));os;
	}
}
//ol(sum);


return 0;
}
