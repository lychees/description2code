#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int t,j,k,p=0,l=0;
    cin>>t;
    int arr[101],arr1[10000];
    for(j=1;j<=100;j++)
    {
        arr[j] = j;
    }

    for(j=1;j<=99;j++)
    {
        for(k=j+1;k<=100;k++)
        {
            arr1[p] = (arr[j]*arr[j]*arr[j] + arr[k]*arr[k]*arr[k]);
            p++;
        }
    }
    
    sort(arr1,arr1+99*50);
    while(t--)
    {
    	int ans=-1;
        int n,i;
        cin>>n;
        for(i=1;i<99*50;i++)
        {
            if(arr1[i-1]==arr1[i])
            {
                if(arr1[i]<n )
                ans=arr1[i];
                else
                    break;
                //else cout<<"-1"<<endl;
            }
        }
        cout<<ans<<endl;
    }
}
