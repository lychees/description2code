#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    long long int t,l,b,d,x,y;
    cin>>t;
    while(t--)
    {
    	cin>>l;
    	cin>>b;
    	cin>>d;
    	if(d>b || d>l)
    	cout<<0<<endl;
    	else
    	{
    		x=(l/d)%1000000007;
    		y=(b/d)%1000000007;
    		x=(x*y)%1000000007;
    		cout<<x<<endl;
    	}
    }
    return 0;
}
