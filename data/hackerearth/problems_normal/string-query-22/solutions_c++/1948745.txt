#include <bits/stdc++.h>
using namespace std;
#define MAX 40000
#define MAX1 200000
int main()
{
    string str;
    cin>>str;
    int m;
    cin>>m;
    int n;
    char c;
    vector<int>vec[150];
    int i;
    for(i=0;i<str.size();i++)
    {
    	vec[str[i]].push_back(i);
    }
    while(m--)
    {
    	cin>>n;
    	cin>>c;
    	str[vec[c][n-1]]='0';
    	vec[c].erase(vec[c].begin()+n-1);
    }
    for(i=0;i<str.size();i++)
    {
    	if(str[i]!='0')
    	{
    		cout<<str[i];
    	}
    }
    cout<<endl;
}
