#include <bits/stdc++.h>
using namespace std;

int main()
{
	int tc;
	scanf("%d",&tc);
	
	int n;
	long long b;
    while(tc--)
    {
	scanf("%d",&n);
	int a[53][53][53]={0};
	//memset(a,0,53*53*53);
	for(int i=2;i<=n+1;i++)
	{
		for(int j =2;j<=n+1;j++ )
		{
			for(int k=2;k<=n+1;k++)
			{
			   scanf("%d",&a[i][j][k]);
			}
		}
	}	
		for(int i=2;i<=n+1;i++)
	{
		for(int j =2;j<=n+1;j++ )
		{
			for(int k=2;k<=n+1;k++)
			{
	            b= a[i][j][k]-a[i-1][j][k]-a[i][j-1][k]-a[i][j][k-1]+a[i][j-1][k-1]+a[i-1][j][k-1]
                    +a[i-1][j-1][k]-a[i-1][j-1][k-1];
                    
			   printf("%d ",b);
					
             }
             printf("\n");
		}
	}
	//printf("\n");
}
     return 0;
}
