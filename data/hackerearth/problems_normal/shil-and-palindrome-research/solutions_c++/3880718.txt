#include<bits/stdc++.h>
using namespace std;
#define MAX 100005
#define ll long long int
int n;
string str;
int bit[26][MAX];
void update(int i, int p, int val)
{
		for(;i <= n; i += i&-i)
				bit[p][i] += val;
}
int query(int i, int p)
{
		int sum = 0;
		for(; i > 0; i -= i&-i)
				sum += bit[p][i];
		return sum;
}
int main()
{
		int q;
		cin>>n>>q;
		int i,j;
		for(i=0;i<26;++i)
		{
				for(j=0;j<MAX;++j)
						bit[i][j] = 0;
		}
		int tmp;
		char c = 'z';
		//str = c + str;
		//str[0] = c;
		//scanf("%s",str+1);
		cin>>str;
		str=c+str;
		int len = str.size();

		//cout<<"len is "<<len<<endl;
		for(i=1;i<len;++i)
		{
				update(i,str[i]-'a',1);
		}
		int l,r;
		int cnt=0;
		int cnt2=0;
		for(i=0;i<q;++i)
		{
				cin>>tmp;
				if(tmp == 1)
				{
						cin>>l>>c;
						update(l,str[l]-'a',-1);
						update(l,c-'a',1);
						str[l] = c;
						cnt2++;
				}
				else
				{
						cnt++;
						cin>>l>>r;
						int odd,even;
						odd=even=0;
						int res;
						for(j=0;j<26;++j)
						{
								res = query(r,j) - query(l-1,j);
								if(res&1) odd++;
								else even++;
						}
						if(odd <= 1) 
						{
								cout<<"yes"<<endl;
								//cout<<"l="<<l<<" r="<<r<<endl;
								//for(j=l;j<=r;++j)
								//		cout<<str[i];
								//cout<<endl;
						}
						else cout<<"no"<<endl;
				}
		}
		//cout<<cnt<<" "<<cnt2<<endl;

}
