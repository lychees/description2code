#include <bits/stdc++.h>
using namespace std;
int myCompare(string X, string Y)
{
    string XY = X.append(Y);
 
    string YX = Y.append(X);
 
    return XY.compare(YX) > 0 ? 1: 0;
}
 
void printLargest(vector<string> arr)
{
    sort(arr.begin(), arr.end(), myCompare);
 
    for (int i=0; i < arr.size() ; i++ )
        cout << arr[i];
}
int main() {
	// your code goes here
	int N;
	cin>>N;
	vector<string> arr;
	for(int i=0; i<N; i++)
	{
		string Num;
		cin>>Num;
		arr.push_back(Num);
	}
	printLargest(arr);
	return 0;
}
