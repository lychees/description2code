#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll int test;
	scanf("%lld",&test);
	while(test--)
	{
		ll int n;
		scanf("%lld",&n);
		ll int digit=1,ans=0,x=10;
		while(1)
		{
			if((n-(x-1))>=0)
			{
				//cout<<(x-x/10)*digit<<endl;
				ans+=(x-x/10)*digit;
			}
			else
			break;
			digit++;
			x*=10;
		}
		ans+=((n-x/10 +1)*digit);
		printf("%lld\n",ans);
	}
	return 0;
}
