ll fib[10000005];
ll arr[1000005];

void fibo()
{
    ll a,b,c;
    fib[0]=-1;
    fib[1]=-1;
    a=0;
    b=1;
    c=0;
    while(c<=2000000)
    {
        c=a+b;
        fib[c]=-1;
        a=b;
        b=c;
    }
}

ll isprime(ll n)
{
    ll i,flag=0;
    for(i=2;i<=sqrt(n);i++)
    {
        if(n%i==0)
        {
            flag++;
            break;
        }
    }
    if(flag==0)
        return 1;
    else
        return 0;
}

int main()
{
    //printf("Pro\n");
    fibo();
    //sieve();
    ll tc,sum,i,n,pr;
    sll(tc);
    while(tc--)
    {
        sum=0;
        sll(n);
        for(i=1;i<n;i++)
        {
            if(fib[i]!=-1)
                sum+=i;
        }
        pr=isprime(sum);
        //printf("sum=%lld prime=%lld\n",sum,pr);
        if(pr==1)
            printf("Tic\n");
        else if(sum%2==0)
            printf("Tac\n");
        else if(sum%2==1 && pr==0)
            printf("Toe\n");
        //pll(sum);
    }
    return 0;
}
