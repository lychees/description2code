#include <iostream>
using namespace std;

int main()
{
    int T, i, j, sum;
    cin>>T;
    string s;
    while(T--)
    {
    	cin>>s;
    	int n=s.length();
    	sum=0;
    	for(i=0; i<n/2; ++i)
    		sum+=(s[i]-'0');
    	int sum1=0;
    	if(n%2)
    	{
    		for(i=n/2+1; i<n; ++i)
    			sum1+=(s[i]-'0');
    	}
    	else
    	{
    		for(i=n/2; i<n; ++i)
    			sum1+=(s[i]-'0');
    	}
    	if(sum==sum1)
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
