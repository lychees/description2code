#include <iostream>
#include <cstring>
using namespace std;
int checkP (string st)
{
	int i, j = st.size() - 1;
	for (i = 0; i < j; i++, j--) {
		if (st[i] != st[j]) {
			return 0;
		}
	}
	return 1;
}
int main()
{
	int i, j, t;
	string s1, s2;
	cin >> t;
	for ( ; t; t--) {
		cin >> s1;
		cin >> s2;
		if (checkP (s1 + s2) == 1 || checkP (s2 + s1) == 1) {
			printf("YES\n");
		} else {
			printf("NO\n");
		}
	}
    return 0;
}
