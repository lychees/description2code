long long int power(long long int x, long long int y)
{
    long long int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return temp%1000000007*temp%1000000007;
    else
        return (((x%1000000007*temp%1000000007)%1000000007)%1000000007*temp%1000000007)%1000000007;
}
