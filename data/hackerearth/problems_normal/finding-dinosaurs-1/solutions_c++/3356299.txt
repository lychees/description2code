#include <iostream>
#include <map>
using namespace std;

unsigned hashAdd(unsigned short v, unsigned hash)
{
	return (hash << 4) | v;
}

int main()
{
	map<unsigned, string> dinosaurs;
	
    int n, k, l, q;
    cin >> n >> k >> l >> q;
    
    string name;
    unsigned short v;
    unsigned h;
    for(int i = 0; i<n; i++)
    {
    	cin >> name;
    	
    	h = 0;
    	
    	for (int j=0;j<k;j++)
    	{
    		cin >> v;
    		h = hashAdd(v, h);
    		
    	}
    	dinosaurs.insert(pair<unsigned, string>(h, name));
    }
    
    for(int i=0;i<q;i++)
    {
    	h = 0;
    	for(int j=0;j<k;j++)
    	{
    		cin >> v;
    		h = hashAdd(v, h);
    	}
    	map<unsigned, string>::iterator it = dinosaurs.find(h);
    	if(it!=dinosaurs.end())
    		cout << it->second << endl;
    	else
    		cout << "You cant fool me :P" << endl;
    }
}
