#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
#define gc getchar_unlocked

inline int readInt()
{
    register int x;
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    return x;
}
inline int power(int x, unsigned int y)
{
    register int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return temp*temp;
    else
        return x*temp*temp;
}
int main() {
	
	int t,x;
	 t = readInt();
	while(t--){
	     x = readInt();
	     int max = 0;
	     int ans1=0,ans2=0;
	    for(int i=0;i<=x;i++){
	           for(int j=0;j<=x;j++){
	               
	               if(power(i,j)+power(j,i)==x){
	                   if(abs(i-j) >= max){
	                       max = i-j;
	                       ans1 = i;
	                       ans2 = j;
	                   }
	               }
	               
	           }
	        
	    }
	    cout<<ans1<<" "<<ans2<<endl;
	}
	
	return 0;
}
