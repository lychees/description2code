no_of_lines = int(raw_input())

for i in range(no_of_lines):
    broken = False
    num_str = raw_input()
    for j in range(len(num_str)-1):
        if num_str[j] == '2' and num_str[j+1] == '1':
            broken = True
            break

    if broken:
        print "The streak is broken!"
    elif not broken and int(num_str) % 21 == 0:
        print "The streak is broken!"
    else:
        print "The streak lives still in our heart!"
