#include<bits/stdc++.h>
#define LL long long
using namespace std;
map<int,LL> factor(int n)
{
	map<int,LL>r;
	for(int i=2;i*i<=(n);i++)
	{     
		if(n%i==0)
		{
			while(n%i==0)
			{
				r[i]++;
				n/=i;             
			}          
		}  
  }
  if(n>1)r[n]=1;       
  return r;    
}
int main()
{

  int tc;
  scanf("%d",&tc);
  while(tc--)
  {
    int a,b,c,d,flag=1;
    scanf("%d %d %d %d",&a,&b,&c,&d);
    map<int,LL> aa=factor(a),bb=factor(c);
    map<int,LL>::iterator it;
    for(it=bb.begin();it!=bb.end();it++)
    {
      if((it->second)*d > aa[it->first]*b)
      {
        flag=0;                 
      }                               
    }       
    if(!flag)
    {
      puts("Not divisible");         
    }   
    else
    {
      puts("Divisible");    
    }
  }  
  return 0; 
}
