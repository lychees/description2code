#include<iostream>
#include<string>
//works
using namespace std;
int main()
{
	string str,final[2000];
	int len[2000], total_lines=0;
	while(getline(cin,str))
	{final[total_lines]=str;
	int n=str.length(), count=0;
		for(int i=1;i<n;i++)
		{
		if(str[i-1]=='-'&&str[i]=='>')
		count++;
		if(str[i-1]=='/'&&str[i]=='/')
		break;}
	int last_pos=1, y=count;
	while(count--)
	{
	bool shift=0;
	for(int i=last_pos;i<n;i++)
	{
		if(str[i-1]=='-'&&str[i]=='>')
		{
			str[i-1]='.';
			shift++;
			last_pos=i-1;
		}
		if(shift)
		str[i+shift-1]=str[i+shift];
	}}
	for(int i=1;i<n;i++)
	final[total_lines][i-1]=str[i-1];
	len[total_lines]=n-y;
	total_lines++;
	}
	for(int x=0;x<total_lines;x++)
	{	for(int i=0;i<len[x];i++)
		cout<<final[x][i];
		cout<<"\n";
	}
	return 0;
}
