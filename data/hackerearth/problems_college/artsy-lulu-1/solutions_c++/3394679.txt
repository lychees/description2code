#include <iostream>
using namespace std ; 
int main()
{
	char c ; 
	int n ;
	int  i; 
	int j ; 
	while ( cin >> c )
	{
		cin >> n ; 
	/*	if ( c <= 'a' && c >='z' && c <= 'A' && c <='A' )
		{
			
		}*/
		 if ( n <2)
		{
			cout << "INVALID" <<endl; 
		}
		else if ( c >= 'a' && c <='z')
		{
			if ( (int)(c - 'a')+ n >26 )
			{
				cout << "INVALID" <<endl ;
			}
			else
			{
				for (i = 1 ; i<=n ; i++ )
				{
				
					for ( j = 1 ; j<=i ; j++)
					{
						cout << c <<" " ; 
					}
						c = c+1 ; 
					cout << endl; 
				}
			}
		}
			else if ( c >= 'A' && c <='Z')
		{
			if ( (int)(c - 'A')+ n >26 )
			{
				cout << "INVALID" <<endl ;
			}
			else
			{
				for (i = 1 ; i<=n ; i++ )
				{
				
					for ( j = 1 ; j<=i ; j++)
					{
						cout << c <<" " ; 
					}
						c = c+1 ; 
					cout << endl; 
				}
			}
		}
		cout << endl; 
		
	}
	return 0 ; 
}
