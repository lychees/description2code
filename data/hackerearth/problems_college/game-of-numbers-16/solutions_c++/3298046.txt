#include<iostream>
using namespace std;
main()
{
    unsigned long long i,j,t,m,n,a,cnt=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>a;
        for(n=2;n<=a;n++)
        {
            for(m=1;m<n;m++)
            {
               if((m*n)%(m+n)==0)
               {
                cnt++;
               }
            }
        }
        cout<<cnt<<endl;
        cnt=0;
    }
}
