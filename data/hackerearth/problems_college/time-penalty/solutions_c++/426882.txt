#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int scanint(){
	register int c = getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

int main()
{
    int d,h,m,w;
    unsigned long long min=0;
    d=scanint();
    h=scanint();
    m=scanint();
    w =scanint();
    min=m+(h*60)+((d-1)*24*60)+w*20;
    cout<<min;
}
