#include <bits/stdc++.h>

using namespace std;

int main()
{
	string s;
    cin >> s;
    
    int sz = s.size();
    //int cntr = 0;
    int f;
    int sc;
    int t;
    int cntr = 0;
    f = sc = t = 0;
    
    int zloc = -1;
    
    for (int i = 0; i < sz; i++){
    	
    	if (i >= 3 && s[i-3] == '0')
    		zloc = i-3;
    		
    	t = sc;
    	sc = f;
    	f = s[i] - '0';
		
		if (f % 8 == 0)
			cntr++;
    	
    	if (i >= 1 && sc != 0) {
    		int num = sc*10 + f;
    		if (num % 8 == 0 && num > 0)
        		cntr++;
    	}
    	if (i >= 2 && sc != 0 && t != 0) {
    		int num = t*100 + sc*10 + f;
    		if (num % 8 == 0 && num > 0)
        		cntr = cntr + (i-2)+ 1 - zloc- 1;
    	}
    }
    cout <<cntr << endl;
    
    return 0;
}
