#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
#define MAX 210
#define ll long long
ll TC, n, s, total;
ll A[MAX];
 
bool solve() {
	if (n == 1) {
		if (A[0] && !(A[0] & (A[0] - 1)))
			return true;
		return false;
	}
	for (int i = 0; i < 32; i++) {
		s = total;
		for (int j = 0; j < n; j++) {
			if(A[j] & (1<<i))
				s = s & A[j];
		}
		if (s && !(s & (s - 1)))
			return true;
	}
	return false;
}
 
int main()
{
	total = 0;
	for (int i = 0; i < 32; i++) {
		total += (1<<i);
	}
	scanf("%lld", &TC);
	while (TC--) {
		scanf("%lld", &n);
		for (int i = 0; i < n; i++) {
			scanf("%lld", &A[i]);
		}
		if (solve() == true) {
			printf("YES\n");
		}
		else {
			printf("NO\n");
		}
	}
    return 0;
}
