#include <bits/stdc++.h>
using namespace std;
int m;
bool compr(string a, string b)
{
	int flag=0;
	for(int i=0;i<m;i++)
	{
		if(a[i]!=b[i])
		{
			if(a[i]<b[i])
			flag=1;
			else
			flag=2;
			break;
		}
	}
	if(flag<2)return true;
	else return false;
}
int main(void)
{
	int t;
	cin>>t;
	while(t--){
    vector<string>s;
    string s2;
    int n,k;
    cin>>n>>k>>m;
    for(int i=0;i<n;i++)
    {
    	cin>>s2;
    	s.push_back(s2);
    }
    sort(s.begin(),s.end(),compr);
    cout<<s[k-1]<<endl;}
}
