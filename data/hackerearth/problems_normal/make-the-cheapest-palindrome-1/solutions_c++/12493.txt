#include<iostream>
#include<string>
#include<cstdio>

using namespace std;

int main()
{
	int tests;
	scanf("%d", &tests);
	while(tests)
	{
		string str;
		cin >> str;
		int aCost, bCost;
		cin >> aCost;
		cin >> bCost;
		int i, j, min;
		i = 0, j = str.length() - 1;
		if(aCost < bCost)
			min = aCost;
		else
			min = bCost;
		int minCost = 0;
		while(i < j)
		{
			if(str[i] == '/' && str[j] == '/')
				minCost += 2*min;
			else if((str[i] == 'a' && str[j] == '/') || (str[i] == '/' && str[j] == 'a'))
				minCost += aCost;
			else if((str[i] == 'b' && str[j] == '/') || (str[i] == '/' && str[j] == 'b'))
				minCost += bCost;
			else if(str[i] != str[j])
			{
				minCost = -1;
				break;
			}
			i++;
			j--;
		}
		cout << minCost << endl;
		tests--;
	}
	return 0;
}
