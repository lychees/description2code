#include<stdio.h>
#include<limits.h>
#define N 501

int lengthOfLog(int a[],int n,int sum)
{
    //for(int i=0;i<n;i++)
    //printf("%d ",a[i]);
    int *Min=new int[N];
    int i,j;
    for(i=0;i<=sum;i++)
    Min[i]=10000;
    Min[0]=0;

    for(i=1;i<=sum;i++)
    for(j=0;j<n;j++)
    {
        if(i>=a[j])
        {
            if(Min[i-a[j]]+1<Min[i])
            Min[i]=Min[i-a[j]]+1;
        }
    }
    //for(i=0;i<=sum;i++)
    //printf("%d ",Min[i]);
    if(Min[sum]==10000)
    return -1;
    return Min[sum];
}

int main()
{
    int t,i,sum,n;
    int a[N];
    scanf("%d",&t);
    while(t--){
    scanf("%d%d",&sum,&n);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    printf("%d\n",lengthOfLog(a,n,sum));
    }
    return 0;
}
