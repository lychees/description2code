#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    char arr[1000000];
    cin>>arr;
    long long int len,half,i,ans=0;
    len=strlen(arr);
    if(len%2==0)
    {
    	half=len/2;
    }
    else
    {
    	half=(len+1)/2;
    }
    
    for(i=0;i<half;i++)
    {
    	int cost=0;
    	int first=int (arr[i]);
    	int last=int (arr[len-1]);
    	if(first>=last)
    	{
    		cost=first-last;
    		ans=ans+cost;
    	}
    	else
    	{
    		cost=last-first;
    		ans=ans+cost;
    	}
    	len--;
    }
    cout<<ans<<endl;
    }
    return 0;
}
