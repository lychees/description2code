#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    string s;
    map<string,int> m1;
    cin >> t;
    
   	long long  count = 0;
    while(t--)
    {
    	cin >> s;
    	count += m1[s];
    	
    	reverse(s.begin(),s.end());
    	
    	m1[s]++;
    	
    	
    }
    cout << count << endl;
    return 0;
}
