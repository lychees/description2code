#include <bits/stdc++.h>

using namespace std;
long long int prime[1002] = {0};
long long int res[1000] = {0};
long long fastpow(long long int x, long long int y)
{
	long long int boom = 1;
	 while (y > 0)
    {
       
        if (y & 1)
            boom = (boom*x);
        y = y>>1; 
        x = (x*x);
    }
    return boom;
}
int main()
{
        long long int i,j;
     
        for(i = 2;i <= 500;i++) {
            if(prime[i] == 0) 
                for(j = 2*i;j <= 500;j += i)
                    prime[j] = 1;
        }  
       
        res[1] = 0;
        res[2] = 1;
        long long int co = 0;
        for(i = 3;i <= 500;i++) {
            if(prime[i] == 0) {
                co++;
            }
            long long t1 = fastpow(2, co);
            res[i] = t1 + res[i - 1];
        }
           // res[i] %= MOD;
           int n;
           cin >> n;
           cout<<res[n]<<endl;
           return 0;
}
