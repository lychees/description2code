#include<iostream>
using namespace std;
int p[100001]={0};

void prime()
{
    p[1]=1;
 int i,j;
 for(i=2;i<=5000;i++)
 {
     for(j=i*2;j<=100000;j+=i)
     {
         if(!p[j])
         p[j]=1;
     }
 }
}
int main()
{
    prime();
    int t;
    cin>>t;

    while(t--)
    {
        int n;
        cin>>n;
        if(n==3)
        cout<<"3\n";
        else{
        if(p[n]==1)
            cout<<"1\n";

        else
        cout<<"2\n";
        }
    }
    return 0;
}
