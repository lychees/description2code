#include<bits/stdc++.h>
using namespace std;
int len(const char *str)
{
		int n = strlen(str);
		int i, j, cl;
		int L[n][n];  


		for (i = 0; i < n; i++)
				L[i][i] = 1;

		for (cl=2; cl<=n; cl++)
		{
				for (i=0; i<n-cl+1; i++)
				{
						j = i+cl-1;
						if (str[i] == str[j] && cl == 2)
								L[i][j] = 2;
						else if (str[i] == str[j])
								L[i][j] = L[i+1][j-1] + 2;
						else
								L[i][j] = max(L[i][j-1], L[i+1][j]);
				}
		}

		return L[0][n-1];
}
int main()
{
		int t;
		cin>>t;
		while(t--)
		{
				string str;
				cin>>str;
				int l;
				int max=-1;
				int curr;
				string nstr,tmp;
				for(l=2;l<str.size();++l)
				{
						tmp=str.substr(str.size()-l);
						reverse(tmp.begin(),tmp.end());
						nstr=str.substr(0,str.size()-l) + tmp;
						//cout<<nstr<<endl;
						curr=len(nstr.c_str());
						if(curr>max) max=curr;
				}
				tmp=str;
				reverse(tmp.begin(),tmp.end());
				curr=len(tmp.c_str());
				//cout<<tmp<<endl;
				if(curr>max) max=curr;
				cout<<max<<endl;			
		}
}
