#include<iostream>
using namespace std;

int main()
{
	string str;
	getline(cin,str);
	int n;
	cin>>n;
	int len = str.length();
	int i=0;
	while(len>=n)
	{
		string str1;
		str1.append(str,i,n);
		cout<<str1<<"\n";
		i = i+n;
		len = len-n;
	}
	if(len>0)
	{
		string str1;
		str1.append(str,i,len);
		cout<<str1<<"\n";
	}
	return 0;
}
