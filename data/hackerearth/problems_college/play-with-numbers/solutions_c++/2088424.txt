#include<cstdio>
long long int x, y, z, d, i, j, wynik;

int main()
{
  scanf("%lld", &d);
  
  while(d--)
  {
    scanf("%lld%lld%lld", &z, &x, &y);
    wynik = 0;
    
    for(i = 1; i <= x && i * i <= z; i++)
    {
      if(z % i == 0)
      {
	j = z / i;
	if (j <= y){wynik++;}
      }
    }
    for(i = 1; i <= y && i * i < z; i++)
    {
      if(z % i == 0)
      {
	j = z / i;
	if (j <= x){wynik++;}
      }
    }
    printf("%lld\n", wynik);
  }  
  
return 0;
}
