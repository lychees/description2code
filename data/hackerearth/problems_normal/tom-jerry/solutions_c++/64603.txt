#include<stdio.h>
#include<iostream>
#include<string>
#include<string.h>
#include<algorithm>
using namespace std ;

int n , flag[102][102]={0} ;
char a[102][102] ;
long long ans=0 ;

void DFS(int i,int j)
{
   if(a[i][j]!='0' || flag[i][j]) return ;	
   if(i==n-1&&j==n-1) { ans++ ; return ; }
   
   flag[i][j]=1 ;
   if(i+1<n) DFS(i+1,j) ;
   if(i-1>=0) DFS(i-1,j) ;
   if(j+1<n) DFS(i,j+1) ;
   if(j-1>=0) DFS(i,j-1) ;
   flag[i][j]=0 ;
   
}

int main()
{
  scanf("%d",&n) ; 
  for(int i=0 ; i<n ; i++) 
   for(int j=0 ; j<n ; j++) 
     cin >> a[i][j] ;
  DFS(0,0) ;   
  printf("%lld\n",ans) ; 

 return 0 ;
}
