#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,l,j,jp=0,i,k;
    string w;char c,temp;
    cin>>t;
    while(t--)
    {
        cin>>w;
        j=1,jp=0;
        l=w.length();
        while(1){
        c=w[l-j];
        for(i=l-j-1;i>=0;i--)
        {
            if(c>w[i])
            {
                for(k=0;k<j;k++)
                {
                temp=w[l-j+k];
                w[l-j+k]=w[i+k];
                w[i+k]=temp;
                }
                cout<<w<<endl;
                jp=1;break;
            }
        }
        if(jp==1)
            break;
        j++;
        if(j>l)
        {
            cout<<"no answer\n";
            break;
        }
        }

    }

}
