#include <bits/stdc++.h>
using namespace std;

#define lld long long int


lld subs[25][25], ninf=1e16;
int dig[25]; 
char arr[25];
void generate(int len) {
	for(int i=0;i<len;i++) subs[i][i]=dig[i];
	
	for(int st=0;st<len;st++)
		for(int en=st+1;en<len;en++)
			{
				subs[st][en]=subs[st][en-1]*10+dig[en];
			}
	
	for(int i=0;i<len;i++)
		for(int j=i+1;j<len;j++) {
			if(subs[i][j]>(lld)1e12) subs[i][j]=ninf;
		}
}

lld solve(int len) {
	generate(len);
	
	lld res=-1;
	for(int i=0;i<len-3;i++)
		for(int j=i+1;j<len-2;j++)
			for(int k=j+1;k<len-1;k++)
				if(!(dig[0]==0 && (i+1)>1) && !(dig[i+1]==0 && (j-i)>1) 
					&& !(dig[j+1]==0 && (k-j)>1) && !(dig[k+1]==0 && (len-1-k)>1))
					res=max(res, subs[0][i]+subs[i+1][j]+subs[j+1][k]+subs[k+1][len-1]);
	return res;
}

int main() {
	// your code goes here
	
	int n; scanf("%d", &n); ninf=-ninf;
	while(n-->0) {
		scanf("%s", arr); int len=strlen(arr);
		for(int i=0;i<len;i++) dig[i]=arr[i]-'0';
		lld res=solve(len);
		if(res<0) printf("unlucky\n");
		else printf("%lld\n",res);
	}
	return 0;
}
