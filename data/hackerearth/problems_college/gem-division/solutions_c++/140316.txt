#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
#include<iostream>
#include<cstring>
#define pp pair<int,int>
using namespace std;
int a[100];
vector<pp> v;
int solve(int y,int sum1,int n)
{
    int sum=sum1-a[y];
   // cout<<"a[y]="<<a[y]<<endl;
    //cout<<"sum="<<sum<<endl;
    int s[sum+10];
    s[0] = 1;
    for(int i=1; i<=sum; i++) s[i] = 0;
    for(int i=0; i<n; i++)
    {
        if(i!=y){
        for(int j = sum; j>=0; j--)
        {
            if(s[j])
            {
                s[j+a[i]]+=1;
            }
        }
        }
    }
    for(int j = sum/2; j>=1; j--)
        if(s[j])
        {
               // cout<<"j="<<j<<endl;
            return abs(sum-2*j);
        }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        v.clear();
        int n,sum=0;
        scanf("%d",&n);
        for(int i=0;i<n;i++){
                scanf("%d",&a[i]);
                sum+=a[i];}
        sort(a,a+n);
        if(n==2)
        {
                printf("%d %d\n",a[0],a[1]);
                continue;
        }
        int pos=0;
        int mindiff=10000000007;
        for(int i=0;i<n;i++)
        {
                int temp=solve(i,sum,n);
                if(temp<mindiff)
                {
                                pos=i;
                                mindiff=temp;
                }
        }
        printf("%d %d\n",mindiff,a[pos]);
    }
    return 0;
}
