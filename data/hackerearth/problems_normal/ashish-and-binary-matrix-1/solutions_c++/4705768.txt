#include <iostream>
#include<unordered_map>
using namespace std;

int main()
{

	int t;
	cin>>t;
	while(t>0)
	{
		int n,m;
		cin>>n>>m;
		string s;
		unordered_map<string,int> mymap;
		for(int i=0;i<n;i++)
		{
			cin>>s;
			mymap[s]=1;
		}
		if(mymap.size()==n)
		{
			cout<<"Yes\n";
		}
		else
			cout<<"No\n";
		t--;
	}
	
}
