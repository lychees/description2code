#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t, n, ans, tmp, cnt;
    cin >> t;
    while (t--) {
        cin >> n;
        ans = (log(2*acos(-1)*n)/2+n*(log(n)-1))/log(10)+1;
        tmp = 5;
        cnt = 0;
        while (n/tmp) {
            cnt += (n/tmp);
            tmp *= 5;
        }
        ans -= cnt;
        cout << ans << endl;
    }
    return 0;
}
