#include<iostream>
#include<string.h>
using namespace std;

int main()
{   int t,l,i,j,a=0,f=0;
    cin>>t;
    while(t)
    {   f=0;
        char w[110];
    	cin>>w;
    	l=strlen(w);
    	for(i=l-1;i>=0;i--)
    	{   if(f==1)
    	       break;
    		for(j=i-1;j>=0;j--)
    		{    
    			if(int(w[i])>int(w[j]))
    			{
    				char c=w[i];
    				w[i]=w[j];
    				w[j]=c;
    				f=1;
    				a=j+1;
    				break;
				}
			}
		}
		for(i=a;i<l;i++)
		{    int s=i;
			for(j=i+1;j<l;j++)
			{
				if(int(w[i])>int(w[j]))
				s=j;
			}
			if(s!=i)
			{
			char t=w[i];
			w[i]=w[s];
			w[s]=t;
			}
		}
		if(f==1)
		cout<<w<<endl;
		else
		cout<<"no answer"<<endl;
		t--;
	}
	
	
	return 0;
}
