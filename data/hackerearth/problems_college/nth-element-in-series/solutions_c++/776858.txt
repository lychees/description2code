void mul(ill a[5][5] , ill b[5][5]) {
    ill c[5][5];
    F(i, 0, 5) {
        F(j, 0, 5){
            c[i][j] = 0;
            F(k, 0, 5) {
                c[i][j]+= ((a[i][k]%mod)*(b[k][j]%mod))%mod;
                c[i][j] %= mod;
            }
        }
    }
    F(i, 0, 5) F(j, 0, 5) a[i][j] = c[i][j];
}
