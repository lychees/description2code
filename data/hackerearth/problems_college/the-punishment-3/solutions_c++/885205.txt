#include<iostream>
#include<string>
using namespace std;
int main()
{
	int testcases;
	cin>>testcases;
	while(testcases-->0)
	{
		int n;
		cin>>n;
		int k;
		cin>>k;
		int a[n+1];
		for(int i=1;i<n;i++)
		a[i]=0;
		a[0]=1;
		a[n]=1;
		int b[n+1];
		for(int i=0;i<n+1;i++)
		b[i]=0;
		b[k]++;
		b[n-k]++;
		int ans[n+1];
		for(int i=0;i<n+1;i++)
		{   
			if(a[i]-b[i]>=0)
			ans[i]=a[i]-b[i];
			else
			{
				ans[i]=10+a[i]-b[i];
				a[i+1]--;
			}
		}
		int start=0;
		for(int i=n;i>=0;i--){
			if(start!=1)
			{
				if(ans[i]!=0)
			{	start=1;
				cout<<ans[i];
			}}
			else
			cout<<ans[i];
		}
		cout<<endl;
	}}
