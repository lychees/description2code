#include <iostream>
using namespace std;

int main()
{
    int T, i;
    long long int N;
    int cost_arr[26] = {0};
    long long int sum_all = 0, quo, rem, sum_rem, result;
    
    cin>>T;
    while(T--)
    {
    	sum_all = sum_rem = result = 0;
    	quo = rem = 0;
    	cin>>N;
    	N *= 12;
    	for(i = 0; i<26; i++)
    	{
    		cin>>cost_arr[i];
    		sum_all += cost_arr[i];
    	}
    	quo = N/26;
    	rem = N%26;
    	for(i=0; i<rem; i++)
    		sum_rem += cost_arr[i] + quo;
    		
    	result = sum_rem + sum_all*quo + (quo*(quo-1))*13;
    	//cout<<N<<" "<<quo<<" "<<rem<<" "<<sum_all<<" "<<sum_rem<<endl;
    	cout<<result<<endl;
    	
    }
    return 0;
}
