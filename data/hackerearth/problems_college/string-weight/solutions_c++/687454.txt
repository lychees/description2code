#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
#include <string>
#include <set>
using namespace std;

typedef long long ll;

int main ()
{
	int t;
	scanf ("%d", &t);
	while (t--)
	{
		char s[10005];
		scanf ("%s", s);
		int len = strlen(s);
		int weights[26];
		for (int i = 1; i <= 26; ++i)
			weights[i-1] = i;
		int n;
		scanf ("%d", &n);
		getchar();
		char ss[100];
		scanf ("%[^\r\n]", ss);
		for (int i = 0; ss[i]!= '\0'; ++i)
		{
			if (ss[i] != ' ')
				weights[ss[i]-'a'] = 0;
		}
		int freq[26] = {0};
		for (int i = 0; i < len; ++i)
			freq[s[i]-'a']++;
		ll ans = 0;
		for (int i = 0; i < 26; ++i)
			ans += freq[i]*weights[i];
		printf ("%lld\n", ans);		
	} 
	return 0;
}
