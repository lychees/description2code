#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int n;
    	cin >> n ;
    	
    	int n5 = n;
    	int n3 = 0;
    	
    	while(n5 && n5 % 3 != 0)
    	{
    		n5 -= 5;
    		n3+=5;
    	}
    	
    	if(n5>=0)
    	{
    		while(n5--)
				cout << 5 ;
			while(n3--)
				cout << 3 ;
			cout << '\n';
    	}
    	else
    	{
    		cout << -1 << '\n';
    	}
    
    	
    
    }
    return 0;
}
