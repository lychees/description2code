#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
typedef vector<int> vi;
#define gc getchar_unlocked

int read_int(){
    register int x = 0, c = gc();
    int sign = 1;
    while(c!='-' && (c<48 || c>57) ) c = gc();
    if(c=='-')
        sign = -1, c = gc();
    while(c>=48 && c<=57) 
        x = (x<<1) + (x<<3) + c - 48, c = gc();
    return sign*x;
}

bool prime[100005];
void sieve()
{
    int i,m,k;
    int n=100001;
    for(i=0;i<=n;i++)
        prime[i]=true;
    prime[0]=false;
    prime[1]=false;
    for(i=2;i*i<=n;i++)
    {
        if(prime[i])
        {
            for(k=i*i;k<=n;k+=i)
                prime[k]=false;
        }
    }
}

int main()
{
    
    int t,n;
    t = read_int();
    sieve();
    while(t--)
    {
    	n = read_int();
    	if(n==0)
    	{
    		printf("4\n");
    		continue;
    	}
    	for(int i=n+1;;i++)
    	{
    		if(prime[i] == 0)
    		{
    			printf("%d\n",i);
    			break;
    		}
    	}
    	
    }
    return 0;
}
