int Z[2*MAX];

ll f[MAX][4][3][2][2];

void Zfunction(string S){
	int L, R, N;
	N = S.length();
	L = R = 0;
	for (int i = 1; i < N; i++){
		if (i > R){
			L = R = i;
			while (R < N && S[R - L] == S[R])
				R++;
			Z[i] = R - L;
			R--;
		}
		else{
			int k = i - L;
			if (Z[k] < R - i + 1)
				Z[i] = Z[k];
			else{
				L = i;
				while (R < N && S[R - L] == S[R])
					R++;
				Z[i] = R - L;
				R--;
			}
		}
	}
}

ll dp(int id, int two, int three, int five, int seven){
	if (id == a.size())
		return (two > 2 && three > 1 && five && seven);
	if (f[id][two][three][five][seven] != -1)
		return f[id][two][three][five][seven];

	ll &ret = f[id][two][three][five][seven];
	ret = 0;

	int pos = a[id];
	int _two = 0;
	while (!(pos&1)) _two++, pos /= 2;
	
	pos = a[id];
	int _three = 0;
	while (pos % 3 == 0) _three++, pos /= 3;

	pos = a[id];
	int _five = 0;
	while (pos % 5 == 0) _five++, pos /= 5;

	pos = a[id];
	int _seven = 0;
	while (pos % 7 == 0) _seven++, pos /= 7;

	ret = dp(id+1, two, three, five, seven);
	ret += dp(id+1, min(3, two + _two), min(2, three + _three), min(1, five + _five), min(1, seven + _seven));
	return ret %= MOD;
}
