#include<iostream>
#include<cstdio>
using namespace std;

int gcd(int a, int b)
{
    return (b==0)?a:gcd(b,a%b);
}

int main()
{
    int t,n,res;
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        res = 1;
        for(int i=2; i<n; ++i){
            if(gcd(i,n)==1){
                res++;
            }
        }
        printf("%d\n", res);
    }
    res;
}
