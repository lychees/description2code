N=int(raw_input())
for i in range(0,N):
	T=''.join(raw_input().strip().split('$')[1].split())
	for j in range(0,len(T)):
		if not T[j].isdigit():
			T=T[0:j]
			break;
	print('$'+str(int(T)))
