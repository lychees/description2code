#include <iostream>
#define MAX 100001
using namespace std;

int main()
{
	int t,c,i,l,h;
    long long int f[MAX];
    f[0]=0;
    f[1]=1;
    
    for(i=2;i<MAX;i++){
    	f[i] = f[i-1]+f[i-2];
    }
    
    cin>>t;
    while(t--){
    	c=0;
    	cin>>l>>h;
    	for(i=0;i<MAX;i++){
    		if(f[i]>h){
    			break;
    		}else if(f[i]>=l && f[i]<=h){
    			c++;
    		}
    	}
    	if(c%2==0){
    		cout<<"VALID"<<endl;
    	}else{
    		cout<<"INVALID"<<endl;
    	}
    }
    
    return 0;
}
