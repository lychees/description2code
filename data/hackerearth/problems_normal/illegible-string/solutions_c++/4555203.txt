#include <iostream>
using namespace std;

int main()
{
    int N;
	char A[1000000];
	cin >> N;
	int len = 0;
	for (int i = 0; i < N; i++)
	{
		cin >> A[i];
		len++;
	}

	int min = 0;
	int max = 0;

	for (int i = 0; i < N; i++)
	{
		if (A[i] == 'w')
			max++;


	}

	for (int i = 0; i < N; i++)
	{
		if (A[i] == 'v' )
		{
		//	cout << "inside i" << i << endl;
			for (int j = i + 1; j < N; j++)
			{
			//	cout << "value" << A[j] << endl;
				if (A[j] == 'v')
				{
				//	cout << "inside" << endl;
					min++;
					i = j;
					break;
				}

				if (A[j] == 'w')
				{
					continue;
				}
				else
				{
					break;
				}
			}
		}

	}


	cout << len - min << " ";
	cout << len + max << endl;
	
	return 0;
}
