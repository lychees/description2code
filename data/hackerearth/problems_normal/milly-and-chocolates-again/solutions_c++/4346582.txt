#include <iostream>
using namespace std;

int inBox(int queried_chocolate_index, int *cumulative_candy_count, int N) {
	int start_box_index = 1;
	int end_box_index = N;
	while(start_box_index<end_box_index) {
		int mid_box_index = (start_box_index+end_box_index)/2;
		if(cumulative_candy_count[mid_box_index-1]==queried_chocolate_index) {
			break;
		}
		if(cumulative_candy_count[mid_box_index-1]<queried_chocolate_index) {
			start_box_index = mid_box_index+1;
			continue;
		}
		end_box_index = mid_box_index;
	}
	return (start_box_index+end_box_index)/2;
}

int main() {
	int N;
	cin >> N;
	int *candy_count = new int[N];
	for(int i=0; i<N; i++) {
		int count;
		cin >> count;
		candy_count[i] = count;
	}
	int *cumulative_candy_count = new int[N];
	cumulative_candy_count[0] = candy_count[0];
	for(int i=1; i<N; i++) {
		cumulative_candy_count[i] = cumulative_candy_count[i-1] + candy_count[i];
	}
	int Q;
	cin >> Q;
	while(Q--) {
		int queried_chocolate_index;
		cin >> queried_chocolate_index;
		cout << inBox(queried_chocolate_index, cumulative_candy_count, N) << endl;
	}
	return 0;
}
