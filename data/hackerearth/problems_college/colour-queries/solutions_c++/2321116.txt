#include <iostream>
using namespace std;
int n,m;
int arr[105][105];
bool barr[105][105];
int count;
int d;

void recurse(int i,int j){
	if(i-1 >=0 && !barr[i-1][j] ){
		if(abs(arr[i][j]- arr[i-1][j]) <= d){
			count++;
			barr[i-1][j] = true;
			recurse(i-1,j);
		}
	}
	if(j-1 >=0 && !barr[i][j-1]){
		if(abs(arr[i][j]- arr[i][j-1]) <= d){
			count++;
			barr[i][j-1] = true;
			recurse(i,j-1);
		}
	}
	if(i+1 < n && !barr[i+1][j] ){
		if(abs(arr[i][j]- arr[i+1][j]) <= d){
			count++;
			barr[i+1][j] = true;
			recurse(i+1,j);
		}
	}
	if(j+1 < m && !barr[i][j+1] ){
		if(abs(arr[i][j]- arr[i][j+1]) <= d){
			count++;
			barr[i][j+1] = true;
			recurse(i,j+1);
		}
	}
}


int main() {
	// your code goes here
	int q;
	cin >> n >> m >> q;
	int i,j;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			cin >> arr[i][j];
		}
	}

	int x,y;
	int k;
	for(k=0;k<q;k++){
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				barr[i][j] = false;
			}
		}
		cin >> x >> y >> d;
		count = 0 ;
		barr[x-1][y-1] = true;
		count++;
		recurse(x-1,y-1);
		cout << count <<  endl;
	}
	return 0;
}
