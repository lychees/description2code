#include <bits/stdc++.h>
using namespace std;
const int sz=30;
char grid[sz][sz];

void flip(int r,int c){
	for(int i=0;i<=r;++i){
		for(int j=0;j<=c;++j){
			grid[i][j]=(grid[i][j]=='0')?'1':'0';
		}
	}
}

bool canbereduced(int r,int c){
	bool ans=0;
	for(int i=r;i>=0 && !ans;--i){
		for(int j=c;j>=0;--j){
			if(grid[i][j]=='0'){
				flip(i,j);
				ans=1;
				break;
			}
		}
	}
	return ans;
}
int main()
{
	int t,r,c;
	cin>>t;
	while(t--){
		cin>>r>>c;
		for(int i=0;i<r;++i){
			for(int j=0;j<c;++j)
			cin>>grid[i][j];
		}
		--r,--c;
		int res=0;
		while(canbereduced(r,c)) ++res;
		cout<<res<<endl;
	}
    return 0;
}
