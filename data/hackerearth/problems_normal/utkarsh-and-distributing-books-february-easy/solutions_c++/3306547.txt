#include<iostream>
using namespace std;
int m(int a, int b){
    return a < b ? a : b;
}
main()
{
   int t,n,i,j,r1,r2;
   cin>>t;
   for(i=0;i<t;i++){
    cin >> n;
    int arr[n];
    for(j=0;j<n;j++)
        cin>>arr[j];
    int minim = 10000;
    for(j=0;j<n;j++)
        minim = m(minim, arr[j]);

    r1 = minim-1;
    int sum=0;
    for(j=0;j<n;j++){
        sum+=arr[j];
        sum-=1;
    }
    r2 = sum;
    cout << r1 << " "<<r2<<endl;
   }

}
