#include <bits/stdc++.h>

using namespace std;

int main()
{
  int t;
  scanf("%d", &t);
  while(t--) {
    int n;
    scanf("%d", &n);
    
    int W[n], H[n];
    for(int i = 0; i < n; ++i) {
      scanf("%d", W + i);
    }
    for(int i = 0; i < n; ++i) {
      scanf("%d", H + i);
    }
    int id = max_element(H, H + n) - H;
    
    int tot = W[0] * H[id];
    swap(H[id], H[0]);
    sort(W + 1, W + n);
    sort(H + 1, H + n, greater<int>());
    bool can = true;
    for(int i = 1; i < n; ++i) {
      if(W[i] * H[i] >= tot) {
        can = false;
      }
    }
    printf("%s\n", can?"YES":"NO");
  }
  return(0);
}
