#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;

	while(t--) {
	int n, k;
	cin >> n >> k;

	int a[n];
	for(int i = 0; i < n; ++i) cin >> a[i];
	
	bool flag = 0;
	int idx = 0;

	while(k && idx < n) {
		int mini = a[idx], sidx = -1;
		for(int i = idx+1; i < min(idx+k+1, n); ++i) {
			if(mini > a[i]) {
				mini = a[i];
				sidx = i;
			}
		}

		if(sidx == -1) {idx++; continue;}
		
		k -= (sidx - idx);
		for(int i = sidx; i > idx; --i) {
			swap(a[i], a[i-1]);
		}
		idx++;
	}

	for(int i = 0; i < n; ++i) cout << a[i] << " "; cout << endl;
	}
	return 0;
}
