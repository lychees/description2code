#!/usr/bin/python

def pow(a, n):
    if n == 0:
        return 1
    if n == 1:
        return a
    ans = pow(a, n/2)
    ans = ans*ans
    if n % 2 == 1:
        ans = ans*a
    return ans


T = input()
while (T > 0):
    T -= 1
    In = raw_input()
    In = In.split()
    k = int(In[0])
    n = int(In[1])
    # print k,n;
    ans = (pow(k, n+1)-1)/(k-1)
    ans = str(ans)
    # print ans
    val = 0
    for i in ans:
        val += ord(i)-48
    print val
