#include <iostream>
using namespace std;
bool has(int n){
	int m=n;
	while(m!=0){
		int y=m%100;
		if(y==21)
			return true;
		m=m/10;
	}
	return false;
}
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int n;
    	cin>>n;
    	if(n%21==0)
    	{
    		cout<<"The streak is broken!"<<endl;
    	}
    	else if(has(n)){
    		cout<<"The streak is broken!"<<endl;
    	}
    	else
    	{
    		cout<<"The streak lives still in our heart!"<<endl;
    	}
    }
    return 0;
}
