#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define IN(A,B,C) (B)<=(A) && (A)<=(C)
#define AIN(A,B,C) assert(IN(A,B,C))
#define MAX 1000000
#define C(n) scanf("%lld",&n)

 
using namespace std;

ll arr[1000100];
ll b[1000100] = {0};

int main() {
	ll n,m,k;
	C(n);
	ll i,j;
	
	for(i = 1;i <= n;i++) {
		C(arr[i]);
	}
	C(m);
	while(m--) {
		ll l,r;
		C(l);
		C(r);
		C(k);
		b[l] += k;
		b[r + 1] -= k;
	}
	ll val = 0;
	for(i = 1;i <= n;i++) {
		val += b[i];
		cout << arr[i] + val << "\n";
	}
		
}
