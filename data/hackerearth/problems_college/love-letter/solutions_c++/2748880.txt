#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	long long k;
    	cin>>n>>k;
    	string s;
    	if(k==0)
    	s=s+'0';
    	while(k!=0)
    	{
    		if(k%2==0)
    		s=s+'0';
    		else
    		s=s+'1';
    		k=k/2;
    	}
    	int l=s.size();
    	if(l<n)
    	cout<<"NO"<<endl;
    	else
    	{
    		int chk=0;
    		for(int i=0;i<n;i++)
    		{
    			if(s[i]!='1')
    			{
    				cout<<"NO"<<endl;
    				chk=1;
    				break;
    			}
    		}
    		if(chk==0)
    		cout<<"YES"<<endl;
    	}
    }
    return 0;
}
