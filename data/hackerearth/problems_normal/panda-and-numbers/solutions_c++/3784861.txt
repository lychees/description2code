#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
bool ans[1000001];
int main()
{
   	int t, n, i, x; 
   	memset(ans, 0, sizeof ans);
   	ans[1]=ans[4]=ans[27]=ans[256]=ans[3125]=ans[46656]=ans[823543]=1;
   	for(i=2;i<1000001;i++)
   	{
   		n=i;
   		while(n)
   		{
   			x=n%10;
   			n/=10;
   			if(i-x*x>0)
   			ans[i]=ans[i]|ans[i-x*x];
   		}
   	}
   	scanf("%d", &t);
   	while(t--)
   	{
   		scanf("%d", &n);
   		if(ans[n])
   			printf("Yes\n");
   		else
   			printf("No\n");
   	}
    return 0;
}
