#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
bool isX(char s){
	return s=='X';
}
bool isY(char s){
	return s=='Y';
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	string s;
    	cin>>s;
    	int count = 0;
    	char prev = s[0];
    	for(int i = 1;i<s.length();i++){
    		if(isX(s[i]) && isX(prev)){
    			count++;
    			continue;
    		}
    		else if(isY(s[i]) && isY(prev)){
    			count++;
    			continue;
    		}
    		prev = s[i];
    	}
    	cout<<count<<endl;
    }
    return 0;
}
