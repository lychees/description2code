int n,v,t[N+5]={0};
void prepare(){
    FOR(i,2,N) if (t[i]==0)
        for(int j=i;j<=N;j+=i) t[j]=i;
}
int main(){
    prepare();
    scanf("%d",&n);
    int ans=0;
    while (n--){
        int cur=0;
        scanf("%d",&v);
        while (v>1) cur++,v/=t[v];
        ans^=cur;
    }
    if (ans>0) printf("ALICE");else printf("BOB");
