#include <iostream>
#include <algorithm>
#include <map>
using namespace std;


int main()
{
	int t,i,j,digit,val,k;
	map<int,string> m ;
	map<int,string>::iterator it ;
	cin>>t;
	int n[t];
	for(i=0;i<t;i++) {
		cin>>n[i];
		m.insert(pair<int,string>(n[i],""));
	} 
	string fact = "1";
	j=2;
	for(it=m.begin();it!=m.end();it++){
		//cout<<it->first;
		for(j;j<=it->first;j++){
			int len = fact.length();
			int quot = 0,rem=0;
			//cout<<"j:"<<j<<endl;
			for(k=0;k<len;k++){
				digit = (fact.at(k)) - '0';
				val = (digit*j)+quot;
				//cout<<digit<<","<<val<<endl;
				quot = val/10;
				rem = val%10;
				fact.replace(k,1,to_string(rem));
				//cout<<fact<<endl;
			}
			if(quot>0){
				string quotstr  = to_string(quot);
				reverse(quotstr.begin(),quotstr.end());
				fact = fact + quotstr;	
			}
			//cout<<fact<<endl;
		}
		//reverse(fact.begin(),fact.end());
		//cout<<fact<<endl;
		m[it->first] = fact;
	}
	
	for(i=0;i<t;i++){
		string fact = m.at(n[i]);
		reverse(fact.begin(),fact.end());
		int len= fact.length();
		int sum =0;
		for(j=0;j<len;j++){
			sum+= fact.at(j)-'0';
		}
		cout<<sum<<endl;
		//cout<<it->first<<"-"<<it->second<<endl;
	}	
    return 0;
}
