#include <bits/stdc++.h>
using namespace std;

double dp[1000005];
int main()
{
	int n,val;
	scanf("%d %d",&n,&val);
    double tmp=val*0.01;
    dp[0]=1.0;
    dp[1]=0.0;
    dp[2]=tmp;
    dp[3]=(1-tmp);
    for(int i=4;i<=n;i++)
    	dp[i]=dp[i-2]*tmp+dp[i-3]*(1-tmp);
    printf("%.6f\n",dp[n]);
    	return 0;
}
