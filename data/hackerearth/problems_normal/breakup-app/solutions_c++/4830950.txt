#include <iostream>
#include <string>
#include <algorithm>
#include <vector>


using namespace std;

int main()
{
	
	int n=0;
	cin>>n;
	string s="";
	int arr[32]={0};
	
while(n>0)
{
 getline(cin,s);
 string sp="";
 
 int length=0;
 length=s.length();
 int date=0;
 for(int i=0;i<length;i++)
 {
 	
 	if(s[i]>=48 && s[i]<=57)
 	{
 		sp+=s[i];
 	}
 	if(s[i+1]==32 ||i==(length-1))
 	
 	{
 		date = atoi(sp.c_str());
 	
 		
 		if(date!=0)
 		{
 		
 			if(s[0]=='G')
		{
 			arr[date%32]+=2;	
 		}
 	
 			else if(s[0]=='M')
 		{
 			arr[date%32]+=1;
    	}
 		}
 		sp="";
 	}
 	
	
 }
 
 //vector<int> v(arr,arr+32);
 //make_heap(v.begin(),v.end());
	n--;
}


 int max=0,nextmax=-1;
 for(int i=1;i<32;i++)
 {
 	//cout<<arr[i];
 	if(arr[i]>=arr[max])
 	{
 		nextmax=max;
 		max=i;
 		
 	}
 	
 }
 //cout<<max<<endl;
 //cout<<nextmax<<endl;
 


if(arr[max]==arr[nextmax])
 {
 cout<<"No Date"<<endl;	
 }
 else if(max==19 ||max==20)
 {
 	cout<<"Date"<<endl;
 }
 else
 cout<<"No Date"<<endl;



    return 0;
}
