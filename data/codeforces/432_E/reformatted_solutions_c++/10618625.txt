using namespace std;

const int maxn = 100 + 15;
int n, m, a[maxn][maxn];
int found(int x, int y) {
  if (a[x][y] != 0)
    return a[x][y];
  for (int i = 'A';; i++)
    if (a[x][y - 1] != i && a[x][y + 1] != i && a[x - 1][y] != i &&
        a[x + 1][y] != i)
      return i;
  return 0;
}
int main() {
  scanf("%d%d", &n, &m);
  for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
      int t = found(i, j);
      printf("%c", t);
      if (a[i][j] != 0)
        continue;
      int ed = 1;
      while (i + ed <= n && j + ed <= m && found(i, j + ed) == t)
        ed++;
      for (int k = i; k < i + ed; k++)
        for (int l = j; l < j + ed; l++)
          a[k][l] = t;
    }
    printf("\n");
  }
  return 0;
}
