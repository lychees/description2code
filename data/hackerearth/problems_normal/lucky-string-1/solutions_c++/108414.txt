#include<iostream>
#include<algorithm>
#include<utility>
#include<vector>
#include<string>
#include<cmath>
using namespace std;

int main()
{
  int i, j, t;
  cin>>t;
  while(t--)
    {
      long long n, power = 2, index, dindex, len=1;
      cin>>n;
      n--;
      while(n >= power*len)
	{
	  n -= power*len;
	  len++;
	  power *= 2;
	}
      index = n/len;
      dindex = n - index*len;
      power /= 2;
      while(dindex != 0)
	{
	  index %= power;
	  power /= 2;
	  dindex--;
	}
      cout<<((index /power)?"Earth\n":"Hacker\n");
    }
}
