#include <iostream>
#include <cstdio>
#include <queue>
#include <stack>
using namespace std;


long long A, B, C, D;
long long foo(long long t){
	return ((A*t*t*t)+(B*t*t)+ (C*t) +D);
}


int solve(long long K){
	long long start = 0;
	long long end = 10;
	
	while(foo(end)<=K){
		end = end*10;
	}
	if(end>0 && foo(end-1)<=K){
		return (end-1);
	}
	
	while(start<end){
		long long mid = start + ((end-start)/2);
		if(foo(mid) < K){
			if(foo(mid+1)>K){
				return mid;
			}
			start = mid + 1;
		}else if(foo(mid) == K){
			return mid;
		}else if(foo(mid) > K){
			if(foo(mid-1)<=K){
				return (mid-1);
			}
			end = mid - 1;
		}
	}
	return 0;
}

int main()
{
	int T;
    scanf("%d", &T);
    while(T--){
    	long long K;
    	long long ans;
    	scanf("%lld %lld %lld %lld %lld", &A, &B, &C, &D, &K);
    	if(foo(0)>=K){
    		ans = 0;
    		printf("%d\n", ans);
    	}else{
    		printf("%d\n", solve(K));
    	}
    }
    return 0;
}
