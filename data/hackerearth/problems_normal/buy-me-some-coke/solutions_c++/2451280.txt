#include<iostream>
using namespace std;
int dp[1000][155][55];
bool visited[1000][155][55];
int func(int curr,int n,int one,int five,int ten)
{
    if(curr==n)
        return 0;
    int ret=1e9;
    if(visited[one][five][ten]==true)
        return dp[one][five][ten];
    visited[one][five][ten]=true;
    if(ten>0)
        ret=min(ret,1+func(curr+1,n,one+2,five,ten-1));
    if(ten>0 && one>=3)
        ret=min(ret,4+func(curr+1,n,one-3,five+1,ten-1));
    if(five>=2)
        ret=min(ret,2+func(curr+1,n,one+2,five-2,ten));
    if(five>=1 && one>=3)
        ret=min(ret,4+func(curr+1,n,one-3,five-1,ten));
    if(one>=8)
        ret=min(ret,8+func(curr+1,n,one-8,five,ten));
    dp[one][five][ten]=ret;
    return dp[one][five][ten];
}
int main(void)
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0;i<1000;++i)
            for(int j=0;j<155;++j)
                for(int k=0;k<55;++k)
                    visited[i][j][k]=false;
        int n;
        cin>>n;
        int one,five,ten;
        cin>>one>>five>>ten;
        if(one+five*5+ten*10<n*8)
            cout<<"-1\n";
        else
            cout<<func(0,n,one,five,ten)<<endl;
    }
    return 0;
}
