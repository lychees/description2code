#include<iostream>
#include<fstream>
#include<list>
#include<algorithm>
using namespace std;

main(){
	ifstream abaris("ana.in");
	int num, counter=0;
	string word;
	cin>>num;
	list<string>A;
	
	for(int a=0;a<num;a++){
		cin>>word;
		sort(word.begin(),word.end());
		A.push_back(word);
	}
	
	while(!A.empty()){
		int tmp = count(A.begin(),A.end(),A.front());

	}

}
