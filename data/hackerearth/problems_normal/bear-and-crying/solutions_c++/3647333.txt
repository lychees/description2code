#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define MOD 1000000007
int main(){
	string s;int c=0;
	cin>>s;
	int j=s.size();
	long long int ans=0,te;
	for(int i=0;i<j;i++){
		if(s[i]==';'){
			c++;
			ans+=te;
			ans%=MOD;
		}
		else{
			te=te*2ll+c;
			te%=MOD;
		}
	}
	cout<<ans<<endl;
	
	return 0;
}
