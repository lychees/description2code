#include <iostream>
using namespace std;

int main()
{
	int tc;
	cin >> tc;
	while(tc--) {
		string str;
		int i;
		cin >> str;
		int l = str.length();
		int a[l];
		for(int i=0;i<l;i++){
			a[i] = (int) str[i] -49;
		}
		int sum=0,sum2=0;
		for (i=0 ;i <l/2; i++){
			sum = sum+ a[i];
		}
		for(i= i; i<l;i++){
			sum2 += a[i];
		}
		if(sum==sum2){
			cout << "YES" << endl;
		}
		else {
			cout << "NO" << endl;
		}
	}
    return 0;
}
