#include<iostream>
using namespace std;
int arr[1001];
int arr2[100001];
int main()
{
int a,l,r,q;
cin>>a>>l>>r>>q;
for(int i=0;i<a;i++)
    cin>>arr[i];
for(int i=l;i<=r;i++)
    arr2[arr[i]]=1;
while(q--)
{
    int p=-1,y;
    cin>>y;
    for(int i=y+1;i<100001;i++)
    {
        if(arr2[i]==1)
        {p=i;break;}
    }
    cout<<p<<endl;
}
}
