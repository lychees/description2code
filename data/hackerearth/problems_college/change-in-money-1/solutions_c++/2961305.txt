#include<bits/stdc++.h>
using namespace std;
	long long int temp[1000000];
long long int fun(long long int a[],long long int n,long long int m)
{

	
	temp[0]=1;
   long long int i,j;
   for(i=0;i<m;i++)
   {
   	for(j=a[i];j<=n;j++)
   	   temp[j]=temp[j]+temp[j-a[i]];
   	   //cout<<temp[j]<<endl;
   }
   return temp[n];
   
   
}

{
