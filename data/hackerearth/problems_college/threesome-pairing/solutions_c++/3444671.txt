#include <iostream>
using namespace std;

int main()
{
   long long int t=0;
   scanf("%lld",&t);
   while(t>0)
   {
   	
   	int n=0;
   	scanf("%d",&n);
   	long long int arr[101];
   	for(int i=0;i<n;i++)
   	scanf("%lld",&arr[i]);
   	long long int count=0;
   	for(int i=0;i<n;i++)
   	{
   		for(int j=i+1;j<n;j++)
   		{
   			if((arr[i]+arr[j])%3==0)count++;
   			
   		}
   		
   		
   	}
   	printf("%lld\n",count);
   	
   	
   	
   	
   	t--;
   }
}
