#include <bits/stdc++.h>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
    ll n,t,i;
    cin>>t;
    while(t--)
    {
        ll count=0;
        cin>>n;
        ll a[n],m;
        for(i=0;i<n;++i)
            cin>>a[i];
        sort(a,a+n);
        for(i=0;i<n;++i)
        {
            m=a[i];
            if(a[i+1]==m+1)
                i++;
            count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
