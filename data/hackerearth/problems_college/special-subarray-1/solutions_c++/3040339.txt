#include<bits/stdc++.h>
using namespace std;
 
long n,m,ar[200000],l,r,up[200000],down[200000];
int main(){
	ios_base::sync_with_stdio(0);
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		cin>>ar[i];
		if (i==1){
			up[i]=down[i]=1;
			continue;
			}
		if (ar[i]>=ar[i-1]){
			up[i]=up[i-1];
		}
		else  up[i]=i;
		if (ar[i]<=ar[i-1]){
			down[i]=down[i-1];
		}
		else down[i]=i;   
    }
    for(int i=1;i<=m;i++){
    	cin>>l>>r;
        r=down[r];
        r=up[r];
        if(r<=l)   cout<<"Yes"<<endl;
        else    cout<<"No"<<endl;
    }
    return 0;
}
