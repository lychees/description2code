#include<bits/stdc++.h>
using namespace std;
int prime[10000005];
void sieve(){
    int i,j;
    memset(prime,0,sizeof(prime));
    for(i=2;i<=sqrt(10000005);i++)
        {if(!prime[i])
        {
            for(j=i*i;j<=10000005;j+=i)
            {
                prime[j]=1;

                }
        }
        }

return;
}
int main(){
sieve();
long int n,x,s;
cin>>n;
while(n--){
    cin>>x;
    s=sqrt(x);
    if( !(prime[x]) )
        cout<<"NO\n";
    else if(s*s==x && (!prime[s]))
        cout<<"NO\n";
    else
        cout<<"YES\n";

    }


return 0;

}
