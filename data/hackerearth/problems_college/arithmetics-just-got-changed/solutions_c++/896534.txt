#include<iostream>
#include<cstdio>
using namespace std;

int revnum;
inline int revdig(long num)
{
    revnum = 0;
    while(num != 0)
    {
        revnum = (revnum * 10) + (num % 10);
        num /= 10;
    }
    return revnum;
}

int main()
{
	int T, A, B;
	cin>>T;
	while(T--)
	{
	 	scanf("%d %d", &A, &B);
	 	printf("%d\n",revdig(revdig(A) + revdig(B)));
	}
	return 0;
}
