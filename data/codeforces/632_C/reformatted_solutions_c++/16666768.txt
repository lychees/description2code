#include <algorithm>
#include <cstring>
#include <iostream>
using namespace std;
string s[500005];
bool cmp(string a, string b) { return (a + b) < (b + a); }
int main() {
  int i, n;
  cin >> n;
  for (i = 0; i < n; i++) {
    cin >> s[i];
  }
  sort(s, s + n, cmp);
  for (i = 0; i < n; i++) {
    cout << s[i];
  }
  return 0;
}
