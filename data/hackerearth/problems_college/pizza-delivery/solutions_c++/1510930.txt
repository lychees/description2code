#include <iostream>
using namespace std;
int main()
{
	int n,m,i,j,k,g,s,d;
	int T[101][101];
	cin>>n;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
			cin>>T[i][j];
	}

	for(k=0;k<n;k++)
	{
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				T[i][j]=min(T[i][j],T[i][k]+T[k][j]);
		}
	}
	cin>>m;
	while(m--)
	{
		cin>>s>>g>>d;
		cout<<T[s][g]+T[g][d]<<" "<<T[s][g]+T[g][d]-T[s][d]<<endl;
	}
	return 0;
}
