#include<bits/stdc++.h>
using namespace std;
int main()
{
    //freopen("in.txt","r",stdin);
    int n,m;
    scanf("%d %d",&n,&m);
    vector<string> arr(n+1);
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    vector<int> now(m,1);
    vector<int> pre(m,1);
    int res=0;
    for(int i=1;i<n;i++)
    {
        for(int j=1;j<m;j++)
        {
            if(arr[i-1][j]==arr[i][j-1] && arr[i-1][j]!=arr[i][j] && arr[i][j]==arr[i-1][j-1])
            {
                int x=min(now[j-1],min(pre[j-1],pre[j]));
                now[j]=x+1;
            }
            else
            {
                now[j]=1;
            }
            if(now[j]>=32 && arr[i-31][j-31]=='_') res++;
        }
        pre=now;
        now.clear();
        now.resize(m,0);
        now[0]=1;
    }
    cout<<res;
    return 0;
}
