#include<bits/stdc++.h>
using namespace std;
int isprime(int n)
{
	if(n==1)
	{
		return 0;
	}
	int i;
for(i=2;i<=sqrt(n);i++)
{
	if(n%i==0)
	{
		return 0;
	}
}
return 1;
}
int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else
	{
		gcd(b,a%b);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		int c=0;
		int n,i;
		cin>>n;
		
		if(n==1)
		{
				cout<<"Incorrect"<<"\n";
				continue;
		}
			for(i=1;i<n;i++)
			{
				if(gcd(i,n)==1)
				{
					c++;
				}
			}
			if(isprime(c))
			{
					cout<<"Correct"<<"\n";
			}
			else
			{
				cout<<"Incorrect"<<"\n";
			}
	}
}
