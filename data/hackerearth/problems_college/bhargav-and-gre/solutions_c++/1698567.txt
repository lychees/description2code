#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n;
	string s;
	cin>>s;
	int l;
	l=s.length();
	int f;
	f=s[0]-'0';
	
	int c=0,c1=0;
	cin>>n;
	while(n--)
	{
		string s1;
		cin>>s1;
		int len;
		len=s1.length();
		if(s1[0]-'0'==f && l==len)
		c++;
		if(abs((s1[0]-'0')-f)==3 && l==len)
		c1++;
	}
	cout<<c<<" "<<c1<<endl;
	
	
	
	return 0;
}
