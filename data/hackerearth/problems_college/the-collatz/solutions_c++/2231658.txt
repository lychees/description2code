#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	int t;
	ll n,o,e,steps;
	cin>>t;
	while(t--)
	{
		cin>>n;
		o=e=steps=0;
		while(n!=1)
		{
			if(n%2)
			{
				o+=n;
				n=(n*3)+1;
			}
			else
			{
				e+=n;
				n/=2;
			}
			steps++;
		}
		o+=1;
		if((e%steps)>(o%steps))
			cout<<"Even Rules\n";
		else if((e%steps)<(o%steps))
			cout<<"Odd Rules\n";
		else
			cout<<"Tie\n";
	}
	return 0;
}
