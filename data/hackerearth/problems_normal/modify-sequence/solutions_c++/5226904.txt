#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int n;
	cin >> n;
	
	long int arr[n];
	for(long int i=0;i<n;i++)
		cin >> arr[i];
	
	for(long int i=n-2;i>=0;i--)
	{
		arr[i] = arr[i] - arr[i+1];	
		arr[i+1]=0;
	}
	if(arr[0] == 0)
		cout << "YES\n";
	else
		cout << "NO\n";
	return 0;
}
