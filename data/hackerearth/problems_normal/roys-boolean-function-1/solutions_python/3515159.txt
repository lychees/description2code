def gcd(a, b):
    while b:
        a, b = b, a%b
    return a
TestCase=input()
for i in range(0,TestCase):
	Num=input()
	k=0
	f=0
	for j in range(1,Num+1):
		if (gcd(j,Num))==1:
			k+=1
	for s in range(1,k+1):
		if(k%s==0):
			f+=1
		if f>2:
			break
	if f!=2:
		print "FALSE"
	else:
		print "TRUE"
