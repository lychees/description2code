#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	  int t;
	  cin>>t;
	  while(t--)
	  {
	  	string str1;
	  	string str2;
	  	cin>>str1;
	  	cin>>str2;
	  	sort(str1.begin(),str1.end());
	  	sort(str2.begin(),str2.end());
	  	if(str1.compare(str2)==0)
	  	{
	  		cout<<"Yes"<<endl;
	  	}
	  	else
	  	{
	  		cout<<"No"<<endl;
	  	}
	  }
}
