int main()
{
    int t;
    si(t);
    while(t--)
    {
        int n;
        si(n);
        int a[n];
        for(int i=0;i<n;i++)
            si(a[i]);
        sort(a,a+n);
        for(int i=0;i<n/2;i++)
            a[i]+=a[n-i-1];
        sort(a,a+n/2);
        pi(a[n/2-1]-a[0]);
        pnew;
    }
