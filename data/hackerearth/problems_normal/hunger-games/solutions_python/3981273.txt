n = int(raw_input())
np = raw_input()
arr = [int(i) for i in np.split()]

arr.sort()

ans = 0
for j in range(n):
	if j + 2 >= n:
		break
	ans = max(ans, (arr[j+2] - arr[j]))
print ans
