#include <bits/stdc++.h>
using namespace std;

#define ll long long int

int main()
{
    ll t;
    cin>>t;
    
    while(t--) {
    	ll n;
    	cin>>n;
    	
    	ll a[1000005];
    	for(ll i=0;i<n;i++) {
    		cin>>a[i];
    	
    	}
    	
    	sort(a,a+n);
    	
    	ll ans=0;
    	for(ll i=0;i<n;i++) {
    		ll j=i;
    		while(a[j]==a[j+1]) {
    			ans++;
    			j++;
    		}
    	}
    	
    	cout<<ans  + n<<endl;
    	
    }
    
    return 0;
}
