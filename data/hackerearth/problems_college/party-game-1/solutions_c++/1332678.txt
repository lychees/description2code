#include <iostream>
#include <algorithm>
using namespace std;

int arr[10010];
int main()
{
    int t,x,y,n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d %d",&x,&y,&n);
    	for(int i=0;i<n;++i)
    	{
    		scanf("%d",&arr[i]);
    	}
    	sort(arr,arr+n);
    	long long sum=0;
    	int i;
    	for(i=0;i<n;++i)
    	{
    		sum+=arr[i];
    		if(sum<=x && sum<=y) continue;
    		else if(sum<=x && sum>y){ printf("Chef Won\n"); break; }
    		else if(sum>x && sum<=y){ printf("Jhool Won\n"); break; }
    		else if(sum>x && sum>y){ printf("Tie\n"); break;}
    	}
    	if(i==n) printf("Tie\n");
    }
    return 0;
}
