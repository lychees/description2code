#include<bits/stdc++.h>
using namespace std;
long long int arr[3003],arr1[3002],arr2[3003],n,i,j;
double ans[3003],res;
double max(double a,double b)
{
	if(a>b)
	return a;
	return b;
	
}
int main()
{
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lld %lld %lld",&arr[i],&arr1[i],&arr2[i]);
		ans[i]=-99999999999;
	}
	res=arr2[0];
	ans[0]=arr2[0];
	for(i=1;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			ans[i]=max(ans[i],ans[j]+(double)arr2[i]-sqrt((arr[i]-arr[j])*(arr[i]-arr[j])+(arr1[i]-arr1[j])*(arr1[i]-arr1[j])));
			if(ans[i]>res)
			res=ans[i];
		}
	}
	printf("%.6lf\n",ans[n-1]);
}
