#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
#define ll long long
int main()
{
    ll i,n,k,t,now,next;
    vector < ll > A;
    cin >> n >> k;
    for(i = 0;i < n;i++)
    {
    	cin >> t;
    	A.push_back(t);
    }	
    sort(A.begin(),A.end());
    for(i = 1;i < n;i++)  
      if(A[i] - A[i - 1] > k)
        break;
    cout << A[i - 1];
}
