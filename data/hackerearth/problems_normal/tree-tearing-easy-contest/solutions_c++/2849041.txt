#include<bits/stdc++.h>
using namespace std;

const int N=1e4+10;

int cnt=0;
int k,n;
vector<int>g[N];
int sz[N];
bool removeM[N];

void dfs(int u) {
  sz[u]=1;
  for(int i=0;i<g[u].size();i++) {
    int to=g[u][i];
    if(sz[to]==-1) {
      dfs(to);
    }
    sz[u]+=sz[to];
    if(removeM[u]==true) {
      sz[u]=0;
    }
    if(sz[u]>=k) {
      sz[u]=0;
      removeM[u]=true;
      cnt++;
    }
  }
  return;
}
int main() {
  //freopen("a.cpp","r",stdin);
  cin>>k>>n;
  if(k==1) {
    cout<<n<<endl;
    return 0;
  }
  for(int i=2;i<=n;i++) {
    int val;
    cin>>val;
    g[val].push_back(i);
  }
  memset(sz,-1,sizeof(sz));
  dfs(1);
  cout<<cnt<<endl;
  return 0;
}
