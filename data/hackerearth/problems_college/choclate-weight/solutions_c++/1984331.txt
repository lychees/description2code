#include <bits/stdc++.h>
using namespace std;
#define SIZE 12345678
int main(void) {
    int N = 0;
    vector<int> w;
    scanf("%d",&N);
    int i = 0, j = 0;
    for(; i < N; ++i) {
    	scanf("%d",&j);
    	w.push_back(j);
    }
    j = w.size();
    
    for(i = 0; i < j; ++i)
    	for(int k = i + 1; k < j; ++k)
    		w.push_back(w[i] + w[k]);
    
    scanf("%d",&N);
    for(i = 0; i < N; ++i) {
    	scanf("%d",&j);
    	if (find(w.begin(), w.end(), j) != w.end())
    		printf("I Like You\n");
    	else
    		printf("I Hate You\n");
    }
    return 0;
}
