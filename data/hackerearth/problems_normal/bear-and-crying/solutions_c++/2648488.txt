#define MAX 100005
#define INF 1e18
#define MOD 1000000007

long long power(long long n, long long m)
{
    if (m == 0) return 1;
    long long x = power(n, m / 2);
    if (!(m & 1)) return (x * x) % MOD;
    else return (((x * x) % MOD) * n) % MOD;
}

int main(){
    ios::sync_with_stdio(false);
    string s;
    cin >> s;
    int n = s.size();
    long long dp1[MAX];
    int semiColon = (s[0] == ';');
    dp1[0] = 0;
    for (int i = 1; i < n; i++) {
        if (s[i] == '_') {
            dp1[i] = 2*dp1[i-1] + semiColon;
            dp1[i] %= MOD;
        } else {
            dp1[i] = dp1[i-1];
            semiColon++;
        }
    //    cerr << dp1[i] << " ";
    }
    //cerr << endl;

    long long answer = 0;
    for (int i = 2; i < n; i++) {
        if (s[i] == ';') {
            answer += dp1[i-1];
            answer %= MOD;
        }
    }
    cout << answer << endl;
}
