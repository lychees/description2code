int row,column,arr[1005][1005],visit[1005][1005],count=0;

pair<int,int> adj[4]; 

void bfs(int x,int y)
{
	//cout<<"Count "<<count<<endl;
	pair<int,int> cur,nxt;
	queue< pair<int,int> > q;
	q.push(make_pair(x,y));
	arr[x][y]=2;
	//cout<<"Pushed "<<x<<" "<<y<<endl;
	//visit[x][y]=1;
	while(!q.empty())
	{
		cur=q.front();
		q.pop();
		count--;
		x=cur.first;
		y=cur.second;
		int newx,newy;
		for(int i=0;i<4;i++)
		{
			newx=x+adj[i].first;
			newy=y+adj[i].second;
			
			if(newx>=0 && newy>=0 && newx<row && newy<column)
			{
				//if(visit[newx][newy]==0)
				//{
				//	visit[newx][newy]=1;
					
					if(arr[newx][newy]==1)
					{
						//cout<<"Pushed "<<newx<<" "<<newy<<endl;
						q.push(make_pair(newx,newy));
						arr[newx][newy]=2;
						//count--;
					}
				//}	
			}

		}
	}


}

	}
	/*for(int i=0;i<row;i++)
