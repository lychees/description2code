#include <iostream>
#include <math.h>
#define MAX 1000001
using namespace std;

int arr[MAX] = {0};
const int mod = 1000000007;
 
void factors()
{
	int lmt = sqrt(MAX);
	for (int i = 2; i <= lmt; ++i)
	{
		if (arr[i] == 0)
		{			
			for (int j = i<<1; j < MAX; j += i)
			{
				if (arr[j] == 0)
					arr[j] = i;
			}
		}
	}
}

void cf()
{
	for (int i = 1; i < MAX; ++i)
	{
		arr[i] = (arr[i] + arr[i-1]) % mod;
	}	
}

int main()
{
	factors();
	cf();	
    int t;
    cin >> t;
    while (t--)
    {
    	int a, b;
    	cin >> a >> b;
    	    	
    	cout << (arr[b] - arr[a-1]) % mod << endl;
    	
    }
    return 0;
}
