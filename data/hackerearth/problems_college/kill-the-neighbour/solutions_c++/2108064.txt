#include <iostream>
#include <vector>
using namespace std;

bool isI(vector<int> A)
{
	bool b=true;
	for(int i=0;i<A.size()-1;i++)
	{
		if(A[i]>A[i+1])
		{
			b=false;
			break;
		}
	}
	return b;
}

int main()
{
    int n;
    cin >> n;
    
    vector<int> A;
    
    for(int i=0;i<n;i++)
    {
    	int x;
    	cin >> x;
    	
    	A.push_back(x);
    }
    
    int s=0;
    while(!isI(A))
    {
    	for(int i=A.size()-1;i>0;i--)
    	{
    		if(A[i-1]>A[i])
    		{
    			A[i]=-1;
    		}
    	}
    	
    	for(int i=0;i<A.size();i++)
    	{
    		if(A[i]==-1)
    		{
    			A.erase(A.begin()+i);
    			i--;
    		}
    	}
    	
    	s++;
    }
    
    cout << s;    
}
