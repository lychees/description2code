# define FORN(a,b,c) for ( int a = b ; a <= c ; ++ a )
# define mp make_pair
# define pb push_back
# define MOD 1000000007

const int MAX = 1000010 ;

using namespace std;

long long fibo [ MAX ] ;

int main()
{
    fibo [ 1 ] = 0 ;
    fibo [ 2 ] = 1 ;
    FORN ( i , 3 , 22 )
        fibo [ i ] = ( fibo [ i - 1 ] + fibo [ i - 2 ] ) % MOD ;
    int poz8 = 7 ;
    int poz144 = 13 ;
    int t ;
    cin >> t ;
    FORN ( k , 1 , t )
    {
        int a , b ;
        cin >> a >> b ;
        long long sol = 0 ;
        if ( a <= poz8 and poz8 <= b )
            sol = sol + fibo [ poz8 ] ;
        if ( a <= poz144 and poz144 <= b )
            sol = sol + fibo [ poz144 ] ;
        cout << sol % MOD << '\n' ;
    }
    return 0;
}
