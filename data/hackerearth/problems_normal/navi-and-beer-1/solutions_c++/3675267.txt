#include <iostream>
using namespace std;

int main()
{
	int tc;
	cin >> tc;
	long long cube[1003] = {};
	for(int i=1;i<1001;i++)cube[i] = i*i*i;
	for(int i=1;i<1001;i++)cube[i] += cube[i-1];
	while(tc--)
	{
		long long n,k,m;
		cin >> n >> k >> m;
		long long ans = cube[n]*k - m ;
		if(ans < 0) ans = 0;
		cout << ans << "\n";
	}
    return 0;
}
