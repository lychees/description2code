#define getcx getchar
inline void in(int &n)
{
    n=0; int ch = getcx(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
    n = n * sign;
}

{

    long long int ans = 0;
    for(int i=1; i<=n; i++)
    {
        int x,y;
        in2(x,y);
        ans += min(x,y);
        a[i] = abs(x-y);

    if(n%2==0)
    {
        bool fnd = true;
        for(int i=2; i<=n; i+=2)
        {
            if(a[i]!=a[i-1]) fnd = false;
            ans += a[i];
        }
        if(fnd==false)
        {
            ans = -1;
        }
    }
    else
    {
        bool fnd = true;
        for(int i=1; i<=n; i++)
        {
            if(a[i]!=0)
            {
                fnd = false;
            }
        }
        if(fnd==false)
        {
            ans = -1;
        }

    while(q--)
