x = input()
for _ in xrange(x):
    a = map(int, raw_input().split())
    area = abs(a[0]*(a[3]-a[5]) + a[2]*(a[5]-a[1]) + a[4]*(a[1]-a[3]))
    if area:
        print "NO"
    else:
        print "YES"
