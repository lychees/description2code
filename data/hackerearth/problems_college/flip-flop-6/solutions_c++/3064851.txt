#include<iostream>
using namespace std;
char next(char x)
{
	if(x=='X')
	return 'Y';
	else
	 return 'X';
}
int main()
{
	int i,j,k,n,t,pre,c;
	char s[100001];
	cin>>t;
	while(t--)
	{
	 cin>>s;
	 pre=0,c=0;
	 for(i=1;s[i]!='\0';i++)
      if(s[i]==next(s[pre]))
		 pre=i;
		else
		 c++;
	 cout<<c<<endl;
    }
	return 0;
}
