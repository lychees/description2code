#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int winner(int N)
{
    if(N==0)return 1;
      int p=pow(2,double(N));
    int** arr=new int*[p+1];
    for(int i=0;i<p+1;i++)
      arr[i]=new int[p+1];
      memset(arr,0,sizeof(arr));

            for(int i=1;i<p;i++)
            {
                for(int j=1;j<=i;j++)
                {
                    cin>>arr[i+1][j];
                }
            }

          int tmp[513];
          memset(tmp,0,sizeof(tmp));
           int l=0;
       for(int k=1;k<=pow(2,double(N-1));k++)
       {
           if(arr[2*k][2*k-1]==1)
           {
               tmp[l++]=2*k;
           }
           else{
                arr[2*k][2*k-1]=0;
                tmp[l++]=2*k-1;
           }
       }
         int m=l;
         while(m>1)
         {
             l=0;
             for(int i=0;i<m-1;i+=2)
             {
                 if(arr[tmp[i+1]][tmp[i]]==1)
                 {
                     tmp[l++]=tmp[i+1];
                 }
                 else{
                        tmp[l++]=tmp[i];
                      arr[tmp[i+1]][tmp[i]]=0;
                 }
             }
             m=l;

         }
         return tmp[0];
}
int main()
{
    int N;
    cin>>N;
    cout<<winner(N)<<endl;
    return 0;

}
