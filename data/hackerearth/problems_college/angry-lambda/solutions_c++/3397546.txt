#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k,i;
        scanf("%d %d",&n,&k);
        int arr[n],count = 0;
        for(i=0;i<n;i++)
        {
            scanf("%d",&arr[i]);
            if(arr[i]<=0)
            {
                count++;
            }
        }
        if(count<k)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");

        }
    }
    return 0;
}
