#include <fstream>
#include <iomanip>
#include <iostream>
#include <set>
#include <stdlib.h>
#include <string>
#include <vector>
using namespace std;

vector<int> g[100007];

double ans = 0;

void dfs(int p, int v, int d) {
  ans += 1.0 / (double)d;
  for (int i = 0; i < g[v].size(); i++)
    if (g[v][i] != p)
      dfs(v, g[v][i], d + 1);
}

int main() {

  int n, i, x, y;
  cin >> n;
  for (i = 1; i < n; i++) {
    cin >> x >> y;
    g[x].push_back(y);
    g[y].push_back(x);
  }
  dfs(0, 1, 1);
  cout << setprecision(10) << ans;

  return 0;
}
