#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,sum=0,max=0;
		cin>>n;
		int arr[n+1];
		for(i=1;i<=n;i++)
		{
			cin>>arr[i];
		}
		for(i=1;i<n;i++)
		{
			for(j=i+1;j<=n;j++)
			{
				if(arr[i]<arr[j] && arr[j]>max)
				{
					max=arr[j];
				}
		 	}
		if(max!=0)
			{
			
				sum=sum+max-arr[i];
			}
			max=0;
			
		}
		cout<<sum<<endl;
		sum=0;
	}
}
