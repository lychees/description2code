#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int d,n,w,h;
	cin>>d>>n;
	for(int i=0;i<n;i++){
		cin>>w>>h;
		if(w<d||h<d){
			cout<<"UPLOAD ANOTHER"<<endl;
		}
		else if(w>=d&&h>=d){
			if(w==h){
				cout<<"ACCEPTED"<<endl;
			}
			else{
				cout<<"CROP IT"<<endl;
			}
		}
	}
    return 0;
}
