#include <iostream>
#include <set>
using namespace std;

int main()
{
    set <string> s;
    int t=0;
    int n=0;
    int i=0;
    int j=0;
    string str;
    
    cin >> t;
    
    for(i=0;i<t;i++)
    {
    	cin >> n;
    	for(j=0;j<n;j++)
    	{
    		cin >> str;
    		s.insert(str);
    	}
    	
    		set <string> :: iterator it;
    		for (it = s.begin();it !=s.end();it++)
    		{
    			cout << *it << endl;
    		}
    		
    		while(!s.empty())
    		{
    			s.erase(s.begin());
    		}
    }
return 0;    
}
