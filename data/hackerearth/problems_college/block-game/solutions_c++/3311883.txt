#include<bits/stdc++.h>
#define ll long long

using namespace std;

ll h[10005], wt[10004], n;

ll cost(int m)
{
	ll i, c = 0;
	for(i = 0; i < n; ++i)
		c += abs(h[i] - m)*wt[i];
	return c;
}

int main()
{
    int t,i;
    
    cin >> t;
    
    while (t--) {
    	cin >> n;
    	ll l = INT_MAX, hi = INT_MIN;

    	for (i = 0; i < n; i++) {
    		cin >> h[i];
		    l = min(l, h[i]);
    		hi = max(hi, h[i]);
    	}
    	for (i = 0; i < n; i++) {
    		cin >> wt[i];
    	}    		
    	ll m,t1,t2,t3;
    	while((hi-l) > 1) {
    		m = (l+hi)/2;
    		t1 = cost(m-1), t2 = cost(m), t3 = cost(m+1);
    		
    		if(t2 < t1 && t2 < t3) {
    			l = m;
    			
    			break;
    		}
    		
    		if(t2 <= t1 && t2 >= t3)
    			l = m;
    		else
    			hi = m-1;
    	}
    	//cout << l << " " << m << endl;
    	cout << min(cost(l), min(cost(l - 1), cost(l + 1))) << endl;
    }
    
    return 0;
}
