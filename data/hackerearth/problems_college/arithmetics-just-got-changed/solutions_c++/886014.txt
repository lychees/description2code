#include<iostream>
#include<stdio.h>
using namespace std;

int reverse(int num){
    int r=0;
    while(num != 0){
        r = r*10 + num % 10;
        num /= 10;
    }
    return r;
}

int main()
{
    int num_1, num_2, test;
    scanf("%d", &test);
    while(test--)
    {
        scanf("%d %d", &num_1, &num_2);
        printf("%d\n", reverse(reverse(num_1)+reverse(num_2)));
    }
}
