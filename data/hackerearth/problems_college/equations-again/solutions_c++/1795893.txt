#include<iostream>
long long ExtEuqlid(long long a, long long b, long long &x, long long &y){
    if (a == 0) {
        x = 0; 
        y = 1;
        return b;
    }
    long long x1, y1;
    long long d = ExtEuqlid(b % a, a, x1, y1);
    x = y1 - (b / a) * x1;
    y = x1;
    return d;
}
int main(){
    long long a,b,c,x=0,y=0;
    std::cin>>a>>b>>c;
    if(c % ExtEuqlid(a,b,x,y)||(a==0&&b==0)){
        std::cout<<-1;
        return 0;
    }
    if(b == 0 && !(c%a)){
        std::cout<<-c/a<<" 0";
        return 0;
    }
    if (a == 0 && !(c%b)){
        std::cout<<"0 "<<-c/b;
        return 0;
    }
    long long a1=a, b1=b;
    a /= ExtEuqlid(a1, b1, x, y);
    b /= ExtEuqlid(a1, b1, x, y);
    c /= ExtEuqlid(a1, b1, x, y);
    ExtEuqlid(a,b,x,y);
    std::cout<<-c*x<<" "<<-c*y;
    return 0;
}
