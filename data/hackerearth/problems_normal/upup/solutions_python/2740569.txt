strg=''
s=raw_input()
l = len(s)
i=0
while i<l:
    if s[i]==' ':
        strg=strg + s[i]
        i+=1
    else:
        r=s.find(' ',i)
        if r==-1:
            r=l
        x=''
        a=ord(s[i])
        if a>=97 and a<=122:
            a=a-32
            x=chr(a)+s[i+1:r]
            
        else:
            x=s[i:r]
        strg = strg + x
        i=r
print strg
