#include <iostream>
using namespace std;
#include <stdio.h>
#define inp(t) scanf("%ld",&t)
int main()
{
    long t,n;
    inp(t);
    while(t--)
    {
    	long a[10001]={0};
    	inp(n);
    	long i,x,c=0;
    	for(i=0;i<n;i++)
    	{
    		inp(x);
    		a[x]++;
    	}
    	for(i=0;i<=10001;i++)
    	if(a[i]!=0)
    	c++;
    	printf("%ld\n",c);
    }
    return 0;
}
