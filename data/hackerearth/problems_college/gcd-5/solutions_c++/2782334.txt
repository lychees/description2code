#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<stdlib.h>
#include<cmath>
using namespace std;
int main()
{
	long long int t,n,*a,i;
	cin>>t;
	int flag=1;
	while(t--)
	{
		cin>>n;
		a=new long long int[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];		
		}//int count=0;
		for(i=0;i<n-1;i++)
		{
			if(a[i]<a[i+1])	
			{
				flag=1;
//	/			count++;
	//			cout<<count<<endl;
			}
			else
			{
				flag=0;
				break;
			}
		}
		if(flag==0)
			cout<<"NO"<<endl;
		else
			cout<<"YES"<<endl;
	}
	
}
