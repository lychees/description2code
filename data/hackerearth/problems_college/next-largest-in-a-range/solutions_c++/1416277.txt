#include <iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int n,l,r,q,x,a;
    vector< int > arr;
    scanf("%d%d%d%d",&n,&l,&r,&q);
    for(int i = 0;i < n; ++i){
    	scanf("%d",&a);
    	if(i >= l && i <= r)arr.push_back(a);
    }
    sort(arr.begin(),arr.end());
    for(int i = 0;i < q;++i){
    	scanf("%d",&x);
    	x++;
    	vector< int >::iterator  t = lower_bound(arr.begin(),arr.end(),x);
    	int l1 = t - arr.begin();
    	if(l1 > (r - l))
    		printf("-1\n");
    	else printf("%d\n",arr[l1]);
    }
    
    return 0;
}
