const long long MOD = 1E9 + 7;
long long bin(long long x, long long y) {
    if (y == 0) {
        return 1;
    }
    long long u = bin(x, y/2);
    if (y % 2 == 0) {
        return u*u % MOD;
    }
    else {
        return u*u % MOD * x % MOD;
    }
}

ll power(ll a, ll b) {
	//cout << a << " " << b << endl;
	ll ans = 1;
	while(b) {
		if(b&1) ans = (ans * a)%MOD;
		b >>= 1;
		a = (a*a)%MOD;
	}
 
	return ans;
}

    	ll ans = power(2, n - 1);
    	ans = power(ans, MOD - 2);
    	pl(ans);
    	pn;

    }
    return 0;
}
