#include <iostream>
using namespace std;
int jump(int A[], int n) {  
   int steps = 0;  
   for (int i=0, mx=0, next=0; i<n-1 && next<n-1; ++i) {  
     mx = max(mx, i+A[i]);  
     if (i == next) {  // ready to jump  
       if (mx == next) return -1; // unreachable  
       next = mx;  
       ++steps;  
     }  
   }  
   return steps;  
 }

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int A[n];
		for(int i=0; i<n; i++){
			cin>>A[i];
		}
		int mn = jump(A,n);
		cout<<mn-1<<endl;
	}
    return 0;
}
