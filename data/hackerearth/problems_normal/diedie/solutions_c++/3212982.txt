#include <iostream>

#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
ll power(ll a,ll b)
{  
	
	ll ans=1;
	while(b)
	{
		if(b&1)
		ans=(ans*a)%mod;
		a=(a*a)%mod;
		b/=2;
	}
	return ans;
}
ll sim_pow(ll a,ll b)
{  
	
	ll ans=1;
	while(b)
	{
		if(b&1)
		ans=(ans*a);
		a=(a*a);
		b/=2;
	}
	return ans;
}
