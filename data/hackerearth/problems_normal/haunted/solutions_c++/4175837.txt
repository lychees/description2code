#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,k=0;
    long int mk,temp;
    cin>>n>>mk;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	a[i]=0;
    }
    map <int,long int> m;
    priority_queue<pair<int, int> >pq;
    while(n--)
    {
    	cin>>temp;
    	if(m.count(temp)==0)
    	{
    		m[temp]=k;
    		a[k]++;
    		k++;
    	}
    	else
    	{
    		a[m[temp]]++;
    	}
    	pq.push(make_pair(a[m[temp]],temp));
    	cout<<pq.top().second<<" "<<pq.top().first<<endl;
    }
    return 0;
}
