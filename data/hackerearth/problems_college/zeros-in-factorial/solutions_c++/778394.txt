#include<stdio.h>
typedef long long int ll;
ll a[1000005]={0},sum[1000005]={0};
ll getSum(ll val)
{
    ll ans=0;
    while(val%5==0)
    {
        ans++;
        val=val/5;
    }
    return ans;
}
int main()
{
    ll i;
    a[0]=0;
    for(i=1;i<=1000000;i++)
    {
        a[i]=getSum(i);
        a[i]+=a[i-1];
    }
    for(i=1;i<=1000000;i++)
    {
        sum[i]=a[i]+sum[i-1];
        //if(i<=25)
            //printf("i=%lld sum=%lld a=%lld\n",i,sum[i],a[i]);
    }
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        ll a1,b1;
        scanf("%lld%lld",&a1,&b1);
        printf("%lld\n",sum[b1]-sum[a1-1]);
    }
    return 0;
}
