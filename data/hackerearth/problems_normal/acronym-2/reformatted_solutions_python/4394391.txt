K = int(input())
dislike = []
for k in range(K):
    dislike.append(raw_input())
N = int(input())
text = raw_input().split()
solution = ""
first = 1
for n in range(N):
    for i in range(K):
        if (text[n] == dislike[i]):
            break
    else:
        if (first):
            solution = text[n][0].upper()
            first = 0
        else:
            solution = solution+"."+text[n][0].upper()
print solution
