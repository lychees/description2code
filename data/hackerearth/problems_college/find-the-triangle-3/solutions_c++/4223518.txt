#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	vector<long long> v;
    	long long tp;
    	while(n--)
    	{
    		cin>>tp;
    		v.push_back(tp);
    	}
    	sort(v.begin(),v.end());
    	bool c=0;
    	if(v.size()<3)
    	cout<<"-1"<<endl;
    	else
    	{
    		long i;
    		for(i=v.size()-3;i>=0;i--)
    		{
    			if((v[i]+v[i+1])>v[i+2])
    			{
    				c=1;
    				break;
    			}
    		}
    		if(c==0)
    		cout<<"-1"<<endl;
    		else
    		cout<<v[i]<<" "<<v[i+1]<<" "<<v[i+2]<<endl;
    	}
    }
    return 0;
}
