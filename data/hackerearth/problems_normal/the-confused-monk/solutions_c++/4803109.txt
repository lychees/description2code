#include<bits/stdc++.h>
using namespace std;
typedef  long long int lu;
#define  od 1000000007

lu modi(lu n,lu x)
{
	if(x==0)
	return 1;
	lu ans=modi(n,x/2)%od;
	ans=(ans*ans)%od;
	
	if(x%2!=0)
	return (ans*n)%od;
	else 
	return ans;
}
lu gcd(lu a,lu b)
{
	if(a==0)
	return b;
	else
	return gcd(b%a,a);
}
int main()
{
	lu n,ans,f=1,g,i;
	scanf("%lld",&n);
	lu ar[n];
	for(i=0;i<n;i++)
	{
		scanf("%lld",&ar[i]);
		f=(f*ar[i])%od;
	}
	g=ar[0];
	for(i=1;i<n;i++)
	{
		g=gcd(g,ar[i]);
	}
	ans=modi(f,g);
	
	cout<<ans;
	
	
  return 0;
}
