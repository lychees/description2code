#include<iostream>
#include<limits.h>
using namespace std;
long int arr[10][10001];
int main()
{	
	long int t,n,limit,temp,heap[10],small,pos;
	cin>>t;
	for( int i=0 ; i<t ; i++ )
	{
		cin>>n;
		for( int j=0 ; j<n ; j++ )
		{
			int k;
			cin>>limit;
			for(k=0; k<limit ; k++ )
				cin>>arr[j][k];
			arr[j][k]=LONG_MAX;
		}
		for(int k=0;k<n;k++)
				heap[k]=0;
		for(;;)
		{
			small = arr[0][heap[0]];
			pos = 0;
			for( int k=0 ; k<n ; k++)
				if( small > arr[k][heap[k]] )
				{
					small = arr[k][heap[k]];
					pos=k;
				} 
			if( small == LONG_MAX )
				break;
			heap[pos]++;
			cout<<small<<" ";
		}	
		cout<<"\n";
	}
	return 0;
}
