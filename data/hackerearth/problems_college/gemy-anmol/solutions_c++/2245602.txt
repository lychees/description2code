#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    long long sum = 0,m;
		long long n,i;
		scanf("%lld",&n);
		scanf("%lld",&m);
		long long ar[n];
		for(i=0;i<n;i++)
		{
			scanf("%lld",&ar[i]);
		}
		sort(ar,ar+n);
		for(i=n-1;i>=0;i--)
		{
		    sum+=ar[i];
		    if(sum>=m){
                break;
		    }
		}
		if(sum>=m){
            printf("%lld\n",n-i);
		}
		else{
            printf("-1\n");
		}
	}
	return 0;
}
