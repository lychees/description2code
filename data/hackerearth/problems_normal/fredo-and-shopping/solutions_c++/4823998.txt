#include<bits/stdc++.h>
using namespace std;
#define ull long long int
ull n,BIT[1000006]={0};
void update(ull idx,ull val)
{
    for(;idx<=n;idx+=((idx)&(-idx)))
        BIT[idx]+=val;
    return;
}
ull query(ull idx)
{
    ull sum=0;
    for(;idx>0;idx-=((idx)&(-idx)))
        sum+=BIT[idx];
    return sum;
}
int main()
{
    ios::sync_with_stdio(false);
    ull i;
    cin>>n;
    ull arr[n+5],temp[n+5];
    for(i=0;i<n;i++){
        cin>>arr[i];
        temp[i]=arr[i];
    }
    sort(temp,temp+n);
    for(i=0;i<n;i++)
        arr[i]=lower_bound(temp,temp+n,arr[i])-temp+1;
    for(i=0;i<n;i++)
    {
        update(arr[i],1);
        cout<<query(arr[i]-1)<<" ";
    }
    return 0;
}
