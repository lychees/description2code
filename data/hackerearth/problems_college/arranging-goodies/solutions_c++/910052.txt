typedef long long int ll;

#define pb(n) push_back(n)

#define READ(x) freopen(x, "r", stdin)
#define WRITE(x) freopen(x, "w", stdout)
#define mp make_pair
typedef struct node{
	double a;
	int b;
}node;
bool cmp(node a,node b)
{
	if(a.a!=b.a) return a.a<b.a;
	else return a.b<b.b;
}
int main()
{
	ll n,m;
	cin>>n>>m;
	vector<node> v;
	int i;
	node temp;
	for(i=1;i<=m;i++) {
		temp.a=abs(1.0*((m+1)/2.00)-i);
		temp.b=i;
		v.pb(temp);
	}
	sort(v.begin(),v.end(),cmp);
//	for(i=0;i<v.size();i++) {
//		cout<<v[i].a<<" "<<v[i].b<<"\n";
//	}
	for(i=0;i<n;i++) {
		cout<<v[i%m].b<<"\n";
	}
}
