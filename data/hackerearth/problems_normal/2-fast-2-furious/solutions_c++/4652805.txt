#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t,i;
    long d[1000000],b[1000000],ds,bs;
    cin>>t;
    for(i=0;i<t;i++)
    	cin>>d[i];
    for(i=0;i<t;i++)
    	cin>>b[i];
    ds=fabs(d[0]-d[1]);
    for(i=1;i<t-1;i++)
    {	if(fabs(d[i]-d[i+1])>ds)
    		ds=fabs(d[i]-d[i+1]);
    }
    bs=fabs(b[0]-b[1]);
    for(i=1;i<t-1;i++)
    {	if(fabs(b[i]-b[i+1])>bs)
    		bs=fabs(b[i]-b[i+1]);
    }
    if(ds>bs)
    	cout<<"Dom"<<endl<<ds;
    else if(bs>ds)
    	cout<<"Brian"<<endl<<bs;
    else
    	cout<<"Tie";
    return 0;
}
