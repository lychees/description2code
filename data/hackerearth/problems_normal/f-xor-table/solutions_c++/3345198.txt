#include<bits/stdc++.h>
using namespace std;

char s[1000006],tmp[1000006];

void func(int tim, int p, int m,int inv)
{
    int i;
    for(i=0;i<m;i++)
        s[i]=s[i]-'0';
    while(tim)
    {
        if(tim&1)
        {
            for(i=0;i<m;i++)
                tmp[i]=(s[i]^s[(i-p+m)%m]);
            for(i=0;i<m;i++)
                s[i]=tmp[i];
        }
        tim/=2;
        p=(p*2)%m;
    }
    for(i=0;i<m;i++)
    {
        if(inv)   s[i]=s[i]^1;
        s[i]=s[i]+'0';
    }
}

int main()
{
    int t,n,m,p,x,cnt;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d%d",&n,&m,&p,&x);
        scanf("%s",s);
        func(n-1,p,m,x);
        printf("%s\n",s);
    }
    return 0;
}
