#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define MAX 100001

int f[MAX];

int calc(int n){
    int ret;
    while (n){
        ret += n & 1;
        n >>= 1;
    }
    return ret;
}

void init(){
    f[0] = 0;
    f[1] = 1;
    for (int i = 2; i < MAX; i++)
        f[i] = calc(f[i - 1]) > calc(i) ? f[i - 1] : i;
}

int main(){
    init();
    int t, n;
    scanf("%d", &t);
    while (t--){
        scanf("%d", &n);
        printf("%d\n", f[n]);
    }
    return 0;
}
