#include<bits/stdc++.h>

using namespace std;

#define all(v) v.begin() , v.end()
using ll = long long;

int main() {
//ios::sync_with_stdio(0), cin.tie(0), cout.tie(0);

	int T, a, b;
	scanf("%d", &T);
	while (T--) {
		scanf("%d%d", &a, &b);
		int res = a ^ b;
		res = __builtin_popcount(res);
		if (res & 1)
			puts("NO");
		else
			puts("YES");
	}
	return 0;

}
