    #include <iostream>
    #include <vector>
    #include <algorithm>
    using namespace std;
    vector<pair<int,int>>vec1,vec2;
    //vector<pair<float,pair<int,int>>>::iterator up,lo;
    int main()
    {
        int n;
        cin>>n;
        float a,b;
        float c;
        long m;
        for(int i=0;i<n;i++)
        {
        	cin>>a>>b;
        	c=(b-a);
        	b=(int)b;
        	a=(int)a;
        	if(c>=0)
        	vec1.push_back(make_pair(a,b));
        	else
        	vec2.push_back(make_pair(b,a));
        }
        sort(vec1.begin(),vec1.end());
        sort(vec2.begin(),vec2.end());
        reverse(vec2.begin(),vec2.end());
        m=0;
        long temp=m;
        //int g,inds,inde;
        for(int i=0;i<vec1.size();i++)
        {
        	temp=temp-vec1[i].first;
    		if(temp<0)
    		{
    			m=m+abs(temp);
    			temp=0;
    		}
    		temp=temp+vec1[i].second;
    		//cout<<vec1[i].first<<" "<<vec1[i].second<<"\n";
        }
        //temp=m;
        for(int i=0;i<vec2.size();i++)
        {
        	temp=temp-vec2[i].second;
    		if(temp<0)
    		{
    			m=m+abs(temp);
    			temp=0;
    		}
    		temp=temp+vec2[i].first;
    		//cout<<vec1[i].first<<" "<<vec1[i].second.first<<" "<<vec1[i].second.second<<"\n";
        }
        cout<<m;
        return 0;
    }
