void scan(int &x)
{
register int c = gx();

for(;(c<48 || c>57);c = gx());

{

}
}

int main()
{

ios_base::sync_with_stdio(false);
int t;
string s;
int ct[26],n;
cin>>t;
while(t--){
memset(ct,0,sizeof ct);
cin>>s;
n=s.length();
for(int i=0;i<n;i++)
ct[s[i]-'a']++;
int ii=1;
while(1){
int e=0,o=0,ne=0;
for(int i=0;i<26;i++){
if(ct[i]%2==0){
e++;
if(ct[i]>0)
ne++;
}	
else
o++;
}
if(e==26||(e==25 && o==1))
break;
else
ii*=-1;
if(e==24 && o==2 && ne>0){
for(int i=0;i<26;i++)
if(ct[i]%2==0 && ct[i]>0){
ct[i]--;break;
}
}
else{
for(int i=0;i<26;i++){
if(ct[i]%2){
ct[i]--;break;
}
}
}	
}
if(ii==1)
