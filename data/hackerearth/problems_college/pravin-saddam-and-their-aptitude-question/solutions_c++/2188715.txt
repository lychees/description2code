#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {int a,b;
   	cin>>a>>b;
   int sum=0;
   while(a<=b&&a<5)
   {if(a==1)
   sum+=1;
   else if(a==2)
   sum+=2;
   else if(a==3)
   sum+=6;
   else if(a==4)
   sum+=4;
   a++;
   }
   cout<<sum%10<<"\n";
   }
    return 0;
}
