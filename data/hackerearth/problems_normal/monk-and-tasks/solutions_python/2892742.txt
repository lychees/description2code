'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
	
for _ in range(input()):
	res = []
	final = []
	number_of_tasks = input()
	tasks = [int(x) for x in raw_input().split()]
	
	for i in range(number_of_tasks):
		res.append((i, "{0:b}".format(tasks[i]).count('1')))
	
	ans = sorted(res, key=lambda x: x[1])
	for k, v in ans:
		final.append(tasks[k])
	print " ".join(str(x) for x in final)
