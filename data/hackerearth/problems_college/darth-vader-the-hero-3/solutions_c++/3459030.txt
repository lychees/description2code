#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll maze[1003][1003];
ll vis[1003][1003];
ll freq[1000000+2];
ll n,m;
ll mark;
ll a[]={-1,-1,0,1,1,1,0,-1};
ll b[]={0,1,1,1,0,-1,-1,-1};
bool is_valid(ll a,ll b)
{
	if(a>=1 && a<=n && b>=1 && b<=m)
		return true;
	return false;
}
void solve(ll x,ll y)
{
	vis[x][y]=mark;
	for(ll i=0;i<8;++i)
	{
		ll px=x+a[i];
		ll py=y+b[i];
		if(is_valid(px,py) && vis[px][py]==0 && maze[px][py]==1)
			solve(px,py);
	}
	return;
}

		mark=1;
		for(ll i=1;i<=n;++i)
		{
			for(ll j=1;j<=m;++j)
			{
				if(vis[i][j]==0 && maze[i][j]==1)
				{
					solve(i,j);
					++mark;
				}
			}		
		}
		/*for(int i=1;i<=n;++i)
