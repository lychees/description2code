#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;

typedef long long ll;
const int mod = 1e7+7;
const int N=1005;
ll f[N],g[N];
ll inv(ll x,ll y,ll p,ll q,ll r,ll s)
{
    if(y==0) return (p%mod+mod)%mod;
    return inv(y,x%y,r,s,p-r*(x/y),q-s*(x/y));
}
ll comb(int n,int k)
{
    if(n<k)
        return 0;
    if(k==0||n==k)
        return 1;
    return f[n]%mod*g[k]%mod*g[n-k]%mod;
}


int main() {
    ll n;
    cin>>n;
    ll ans=1;
    f[0]=1;
    ll r;
    cin>>r;
    if(r>n) {
    	puts("-1");
    	return 0;
    }
    if(r==1) {
    	puts("1");
    	return 0;
    }
    for(int i=1;i<=n;i++) f[i]=f[i-1]*i%mod;
    for(int i=1;i<=n;i++) g[i]=inv(f[i],mod,1,0,0,1);
    cout<<comb(n-1,r-1)%mod<<endl;
    return 0;
}
