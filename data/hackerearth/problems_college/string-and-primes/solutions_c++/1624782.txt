#include <iostream>
#include<string.h>
using namespace std;


int a[125];
void prime()
{
int i,j;	
for(i=2;i<=62;i++)
{
	
	for(j=2;j<=(124/i);j++)
	{
		a[i*j]=-1;
	}
}	


}
int main()
{
	int t,i,j,len;
	
	char s[100001];
	prime();
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	len=0;
    	for(i=0;i<strlen(s);i++)
    	{
    		if(a[s[i]]==-1)
    		{cout<<s[i]; len++;}
    		
    	}
  		if(len==0)
  		cout<<"NULL"<<endl;
  		else
    	cout<<endl;
    	
    }
    return 0;
}
