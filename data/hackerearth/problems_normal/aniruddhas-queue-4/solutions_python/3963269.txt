for _ in range(int(raw_input())):
	n = int(raw_input())
	arr = map(int, list(raw_input().split()))
	sum1 = sum(arr)
	tar = int(raw_input())
	tar = tar%sum1 
	if tar == 0:
		for i in xrange(n-1,-1,-1):
			if arr[i] != 0:
				print i+1
				break
	else:
		cur = 0
		for i in range(n):
			if arr[i] > 0:
				cur += arr[i]
			if cur >= tar:
				print i+1
				break
