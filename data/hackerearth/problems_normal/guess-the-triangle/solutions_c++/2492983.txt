#include <iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
vector<pair<double,double> >ans;

int cmp(pair<double,double> a,pair<double,double> b)
{
   if(a.first!=b.first)
     return  a.first<b.first;
     else
     return a.second<b.second;
}
int main()
{
      double a,b,c,d,e,f,x1,x2,x3,y1,y2,y3;
      cin>>a>>b;
      cin>>c>>d;
      cin>>e>>f;
      x1=c-e+a;
      y1=d-f+b;
      ans.push_back(pair<double,double>(x1,y1));
      x2=e-c+a;
      y2=f-d+b;
      ans.push_back(pair<double,double>(x2,y2));
      x3=e-a+c;
      y3=f-b+d;
      ans.push_back(pair<double,double>(x3,y3));
      sort(ans.begin(),ans.end(),cmp);
      for(int i=0;i<3;i++)
        printf("%.4f %.4f\n",ans[i].first,ans[i].second);
      
      
}
