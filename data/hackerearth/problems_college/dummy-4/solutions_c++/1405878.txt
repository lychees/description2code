#include<iostream>
using namespace std;
int main(){
	long long int n,rem,sum=0,temp;
	cin>>n;
	while(n--){
		string a;
		cin>>a;
		for(int i=0;i<a.length();i++){
			if(a[i]=='0'){
				sum+=0;
			}
			else if(a[i]=='1'){
				sum+=1;
			}
			else if(a[i]=='2'){
				sum+=2;
			}
			else if(a[i]=='3'){
				sum+=3;
			}
			else if(a[i]=='4'){
				sum+=4;
			}
			else if(a[i]=='5'){
				sum+=5;
			}
			else if(a[i]=='6'){
				sum+=6;
			}
			else if(a[i]=='7'){
				sum+=7;
			}
			else if(a[i]=='8'){
				sum+=8;
			}
			else if(a[i]=='9'){
				sum+=9;
			}
		}
	}
	if(sum<=9)
	cout<<sum;
	else{
		while(sum>9){
	temp=sum;
	sum=0;
	while(temp>9){
		rem=temp%10;
		sum+=rem;
		temp/=10;
	}
	sum+=temp;}
	cout<<sum<<endl;
	}
	return 0;
}
