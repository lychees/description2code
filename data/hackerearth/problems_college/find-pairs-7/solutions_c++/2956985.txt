#include <bits/stdc++.h>
using namespace std;
#define ll long long

int a[1000000];

int main(){
	int t;
	scanf("%d",&t);
	while (t--){
		int n;
		scanf("%d",&n);
		for (int i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int i=0;
		ll int sum = 0;
		while (i<n){
			int j=i;
			while (a[i]==a[j]){
				if (j==n) break;
				j++;
			}
			int x = j-i;
			sum+=((x*(x+1))/2);
			i=j;
		}
		cout << sum << endl;
	}
	return 0;
}
