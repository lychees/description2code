#include <iostream>
using namespace std;

int main()
{
    
    int n;
    cin >> n;
    long long arr[n];
    for(int i=0;i<n;i++)
    cin >> arr[i];
    long long max1[n],max2[n];
    long long m =  arr[0];
    for(int i=1;i<n;i++)
	{
		max1[i] = m;
		if(m<arr[i])
		m = arr[i];
	}
	m = arr[n-1];
	for(int i=n-2;i>=0;i--)
	{
		max2[i] = m;
		if(m<arr[i])
		m = arr[i];
	}
	cout<<1<<" ";
	for(int i =1;i<n-1;i++)
	{
		if(max1[i]< arr[i] || max2[i]<arr[i])
		cout<<i+1<<" ";
	}
	
	cout<<n<<" ";
    return 0;
}
