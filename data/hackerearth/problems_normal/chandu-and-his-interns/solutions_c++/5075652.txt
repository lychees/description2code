#include <bits/stdc++.h>

using namespace std;

int main(int argc, char const *argv[])
{
	long long n,x,i;
	int count;
	bool flag = false;
	cin>>n;
	while(n--)
	{	count = 0;
		flag = false;
		cin>>x;
		for(long long i = 1;i*i < x ;++i)
		{
			if(x%i==0)
				count+=2;
			if(count>3)
				{cout<<"YES"<<endl; flag = true;
				break;
				}
		}
		if(flag != true)
			cout<<"NO"<<endl;
	}

	return 0;
}
