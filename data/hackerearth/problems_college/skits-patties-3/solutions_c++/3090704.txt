#include <iostream>
using namespace std;

int main()
{
    unsigned long long int a,b,c,t,n;
    bool found;
    cin>>t;
    while(t--)
    {
    	found=false;
    	cin>>n;
    	for(a=0;(6*a<=n)&&(!found);a++)
    	{
    		for(b=0;(9*b<=n)&&(!found);b++)
    		{
    			for(c=0;(20*c<=n)&&(!found);c++)
    			{
    				if((6*a+9*b+20*c)==n)
    				 found=true;
    			}
    		}
    	}
    	
    	if(found)
    	cout<<"True\n";
    	else
    	cout<<"False\n";
    }
    return 0;
}
