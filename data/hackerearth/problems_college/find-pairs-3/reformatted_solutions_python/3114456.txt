# print 'Hello World!'
t = input()
while t > 0:
    t -= 1
    n = input()
    l = [int(i) for i in raw_input().split()]
    c = 0
    for i in range(n):
        for j in range(n):
            if j >= i and l[i] == l[j]:
                c += 1
    print c
