n, m = map(int, raw_input().split())

arr = []
row_sum, col_sum = [0]*n, [0]*m

for i in xrange(n):
    arr.append(map(int, raw_input().split()))

for i in xrange(n):
    row_sum[i] = row_sum[i] + sum(arr[i])
    for j in xrange(m):
        col_sum[j] = col_sum[j] + arr[i][j]

# print row_sum, col_sum

maxor = 0
for i in row_sum:
    for j in col_sum:
        temp = i ^ j
        maxor = temp if temp > maxor else maxor

print maxor
