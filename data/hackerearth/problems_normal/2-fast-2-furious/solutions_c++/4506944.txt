#include <iostream>
#include <vector>
#include <cmath>

int main()
{
	int n, cur, last, max = 0;
	std::string winner;
	std::vector<long long int> speeds;
	std::cin >> n;
	winner = "Tie";
	for (int i = 0; i < n; i++) {
		std::cin >> cur;
		speeds.push_back(cur);
	}
	std::cin >> last;
	for (int i = 1; i < n; i++) {
		std::cin >> cur;
		if (abs(cur - last) > max) {
			winner = "Brian";
			max = abs(cur - last);
		}
		if (abs(speeds[i] - speeds[i - 1]) > max) {
			winner = "Dom";
			max = abs(speeds[i] - speeds[i - 1]);
		}
		last = cur;
	}
	std::cout << winner << std::endl;
	if (max) {
		std::cout << max;
	}
}
