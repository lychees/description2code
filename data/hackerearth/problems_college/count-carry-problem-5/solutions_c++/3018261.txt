#include <bits/stdc++.h>
using namespace std;
 
void print(long long int x) {
    if (x == 0) printf("No carry operation\n");
    
    else if (x == 1) printf("1 carry operation\n");
    
    else printf("%llu carry operations\n", x);
}
 
int main() {
    unsigned long long int t, a, b, c, count, x, y;
    scanf("%llu", &t);
    while (t--) {
    	scanf("%llu%llu", &a, &b);
        c = 0;
        count = 0;
        while (a != 0 || b != 0) {
            x = a % 10;
        	y = b % 10;
            if (x+y+c > 9) {
                count++;
            }
            c = (x+y+c)/10;
            
            a /= 10;
            
            b /= 10;
        }
        print(count);
    }
    return 0;
}
