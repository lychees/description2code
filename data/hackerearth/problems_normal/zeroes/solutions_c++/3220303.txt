#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

#define MAX 100010

bool prime[MAX];
int len, sp[MAX];

void sieve() {
	for(int i = 2; i < MAX; i += 2)	
		sp[i] = 2;
	sp[1] = 1;
	for(long long int i = 3; i < MAX; i += 2) {
		if(!prime[i]) {
			sp[i] = i;
			for(long long int j = i; (j*i) < MAX; j += 2) {
				if(!prime[j*i])	prime[j*i] = true;
				sp[j*i] = i;
			}
		}
	}
}

int lar[100001];
int a[100001];

int main(void) {

	sieve();
	
	int n;
	scanf("%d", &n);
	memset(lar, 0, sizeof(lar));

	for(int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		while(a[i] != 1) {
			lar[sp[a[i]]]++;
			a[i] /= sp[a[i]];
		}
	}

	int q;
	scanf("%d", &q);

	while(q--) {
		int num;
		scanf("%d", &num);

		int ans = 1000001;

		while(num != 1) {
			int val = 0;
			int pr = sp[num];
			while(num != 1 && (num%pr == 0)) {
				//cout<<sp[num]<<endl;
				num /= pr;
				val++;
			}
			if(val) {
				ans = min(ans, lar[pr]/val);
			}
		}
		if(ans == 1000001)	ans = 0;
		printf("%d\n", ans);
	}

	return 0;
}
