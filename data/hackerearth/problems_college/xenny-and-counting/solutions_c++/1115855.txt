#include <iostream>
using namespace std;

int main()
{
    // (k+2)C(n)
    int t;
    cin>>t;
    while(t--)
    {
    	int k, n;
    	cin>>n>>k;
    	k += n-1;
    	n--;
    	unsigned long long int ans = 1;
    	for(int i=0; i<n; i++)
    	{
    		ans = ans*(k-i)/(i+1);
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
