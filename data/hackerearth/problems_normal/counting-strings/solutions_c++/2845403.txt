#include <iostream>
using namespace std;

int main()
{

	int test ; 
	cin >> test ;
	
	while(test --)
	{
		string str ; 
		cin >> str ; 
		
		int len = str.length() ;
		long long int ans = 0 ;
		int count = 0 ; 
		for(int i = 0 ; i < len ; i++)
		{
			if(str[i] == 'a' || str[i] == 'z')
			{
				//cout << " check >> " << len-i << endl ;
				ans += (len-i) + (count*(len-i)) ;  
				count = 0 ; 
			}
			else 
			{
				count ++ ; 
			}
		}
		cout << ans << endl ; 
	}

    return 0;
}
