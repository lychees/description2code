#include <iostream>
#include<vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int ans=0;
        int res=0,flag=0;
    	string s;
    	cin>>s;
    	for(int i=0;i<s.size();i++)
    	{
    	   	if(s[i]=='_')
    	   	{res++;
    	   	}
    	   	else if(s[i]=='|')
    	   	{ans^=res;
    	   	flag=1;
    	   	res=0;}
    	}
    	ans^=res;
    	if(flag==0)
    	cout<<"Bob"<<endl;
   else if(ans)
    cout<<"Alice"<<endl;
    else
    cout<<"Bob"<<endl;
    }
}
