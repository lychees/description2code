#include <stdio.h>
#include <iostream>
#include <cstring>

using namespace std;


int memo[105][105][105][2];
int dp[105][105][105][2];
int foo(int i,int j,int k,int turn)
{
	if(i==0 && j==0 && k==0) return 1-turn;	
	else if(i<0||j<0||k<0) return -1;
	
	 if(memo[i][j][k][turn]) 
	{
		return dp[i][j][k][turn];
	}
	else {
	
	int t1 = foo(i-1,j-1,k-1,1-turn);
	int t2 = foo(i-1,j-1,k,1-turn);
	int t3 = foo(i-1,j,k-1,1-turn);
	int t4 = foo(i,j-1,k-1,1-turn);
	int t5 = foo(i-1,j,k,1-turn);	
	int t6 = foo(i,j-1,k,1-turn);
	int t7 = foo(i,j,k-1,1-turn);
	memo[i][j][k][turn]=1;
	if(t1==turn ||t2==turn ||t3==turn ||t4==turn ||t5==turn ||t6==turn ||t7==turn)
	{
		dp[i][j][k][turn]=turn;
	}
	else dp[i][j][k][turn]=1-turn;
	
	return dp[i][j][k][turn];
	
	}
		
}


int main()
{    
   
   	for(int i=0;i<=100;++i)
   	{
   		for(int j=0;j<=100;++j){
   			for(int k=0;k<=100;++k)
   			{
   				dp[i][j][k][0]=foo(i,j,k,0);
   				dp[i][j][k][1]=foo(i,j,k,1);
   			}
   		}
   	}
      
      int q; cin>>q;
      
      while(q>0)
      {
      	--q;
      	 int n1,n2,n3;
       cin>>n1>>n2>>n3;
       
       if(dp[n1][n2][n3][0]==0)
       cout<<"Ashima"<<endl;   
      else
       cout<<"Aishwarya"<<endl;  
       
      }
   
    return 0;
}
