for i in range(input()):

    string = raw_input()
    sub = 0
    count = 0
    lis = ['a', 'A', 'b', 'B', 'c', 'C']

    for char in string:
        if char not in lis:
            count += sub*(sub+1)/2
            sub = 0
        else:
            sub += 1
    count += sub*(sub+1)/2
    print count
