#include <bits/stdc++.h>
using namespace std;
 
int main() {
	long long int n;
	cin>>n;
//	string s;
//	cin>>s;
char ch;
	long long int q;
	long long int cnt[26];
		for(int i=0;i<26;i++) {
			cnt[i]=0;
		}
	for(int i=1;i<=n;i++) {
			cin>>ch;
			cnt[ch-'a']++;
	}
	cin>>q;
	while(q--) {
		char l,r;
		cin>>l>>r;
		if(l==r) {
			cout<<(cnt[l-'a']*(cnt[l-'a']-1))/2<<endl;
		}
		else cout<<cnt[l-'a']*cnt[r-'a']<<endl;
	}
	return 0;
}
