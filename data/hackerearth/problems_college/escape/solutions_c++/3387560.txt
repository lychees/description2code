#include<bits/stdc++.h>
using namespace std;
 
int a[20][20],m[20][20],c=0,n;
 
void route(int i,int j){
	if((i==n)&&(j==n)){
		c++;
		return;
	}
	m[i][j] = 1;
	if( (j+1)<=n && m[i][j+1] == 0 && a[i][j+1] == 0 )
		route(i,j+1);
	if( (i+1)<=n && m[i+1][j] == 0 && a[i+1][j] == 0 )
		route(i+1,j);
	if( (j-1)>=1 && m[i][j-1] == 0 && a[i][j-1] == 0 )
		route(i,j-1);
	if( (i-1)>=1 && m[i-1][j] == 0 && a[i-1][j] == 0 )
		route(i-1,j);
	m[i][j] = 0;
	return;
}
 
int main()
{
	int i,j,t;
	cin >> t;
	while(t--){
		cin >> n;
		for(i=1;i<=n;i++){
			for(j=1;j<=n;j++){
				cin >> a[i][j];
				m[i][j] = 0;
			}
		}
		c = 0;
		route(1,1);
		cout << c << endl;
	}
	return 0;
}
