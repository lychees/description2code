#include<iostream>
using namespace std;
int x[11],a[11];
int main(){
	cin.tie(NULL);
	ios::sync_with_stdio(false);
	int xor1,xor2,t,n;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n;
		xor1=0;
		xor2=0;
		for(int j=0;j<n;j++)
		{
			cin>>a[j];
		}
		for(int j=0;j<n;j++)
		{
			cin>>x[j];
			xor1^=(x[j]-1);
			xor2^=(a[j]-x[j]);
		}
		if((xor1^xor2)==0)
		{
			cout<<"Second\n";
		}
		else
		{
			cout<<"First\n";
		}
		continue;
		if(xor1==0)
		{
			if(xor2==0)
			{
				cout<<"Second\n";
			}
			else
			{
				cout<<"First\n";
			}
		}
		else
		{
			if(xor2==0)
			{
				cout<<"First\n";
			}
			else
			{
				cout<<"Second\n";
			}
		}
	}
	return 0;
}
