#include<bits/stdc++.h>
#define MOD 1000000009
#define ll long long int
using namespace std;
int len[]={1,2,3,2,3,4,3,2,4,2};
int pre[10][10]={{8},{2,4},{1,3,5},{2,6},{1,5,7},{2,4,6,8},{3,5,9},{4,8},{0,5,7,9},{6,8}};
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int i;
    ll dp[10009][20];
    int n,j,k;
    ll ans=0;
    for(i=0;i<=9;i++)
    dp[1][i]=1;
    cin>>n;
    for(i=2;i<=n;i++)
    {
        for(j=0;j<=9;j++)
        {
            dp[i][j]=dp[i-1][j];
            for(k=0;k<len[j];k++)
            {
                dp[i][j]=(dp[i][j]%MOD+dp[i-1][pre[j][k]]%MOD)%MOD;
            }
        }
    }
    for(i=0;i<=9;i++)
    ans=(ans%MOD+dp[n][i]%MOD)%MOD;
    cout<<ans<<endl;
    }
    return 0;
}
