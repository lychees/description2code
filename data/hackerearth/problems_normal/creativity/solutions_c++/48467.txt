#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
	int t,i;
	char s1[1001],s2[1001];
	cin>>t;
	getchar();
	while(t--)
	{
		gets(s1);
		gets(s2);
		int l1=strlen(s1),l2=strlen(s2),j=0;
		if(l1>l2)
			cout<<"not possible\n";
		else
		{
			for(i=0;s2[i]!='\0';i++)
			{
				j=0;
				if(s2[i]==s1[j] && s1[l1-1]==s2[i+(l1-1)])
				{
					for(j=0;s1[j]!='\0';j++,i++)
					{
						//cout<<s2[i]<<""<<s1[j]<<"\n";
						if(s1[j]!=s2[i])
							break;
					}
					if(j==l1)
						break;
				}
			}
		if(l1!=j)
			cout<<"not possible\n";
		else
			cout<<"possible\n";
		}		
	}
	return 0;
}
