#include<bits/stdc++.h>
#define lld long long  int
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        lld n,K;
        scanf("%lld %lld",&n,&K);
        int c1=0,c0=0;
        for(int i=0;i<n;i++)
        {
            int a;
            scanf("%d",&a);
            if(a==1)c1++;
            else c0++;
        }
        if(K<=c0)
        {
            c1+=K;
            printf("%d\n",c1);
        }
        else
        {
            c1+=c0;
            K-=c0;

            if(K%2==0)
            {
                printf("%d\n",c1);
            }
            else
            {
               printf("%d\n",c1-1);
            }
        }
    }
    return 0;
}
