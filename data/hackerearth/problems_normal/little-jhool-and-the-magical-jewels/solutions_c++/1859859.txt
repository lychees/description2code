#include <iostream>
using namespace std;
#include<climits>
int main()
{
   int t;
   cin>>t;
   
   while(t--)
   {
   	string s;
   	cin>>s;
   	int i=0;int h[26]={0};
   	while(s[i])
   	{
   	   h[s[i]-'a']++;i++;	
   	}
   	int m=INT_MAX;
   	int c[4];
   	  c[0]=h['b'-'a'];
   	  c[1]=h['r'-'a']; c[2]=h['u'-'a']; c[3]=h['y'-'a'];
   	  
   	  for(int k=0;k<4;k++)
   	  {
   	  	if(c[k]<m)
   	  	{
   	  		m=c[k];
   	  	}
   	  }
   	  cout<<m<<endl;
   }
   
    return 0;
}
