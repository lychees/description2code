#include <algorithm>

#include <set>
#include <string>
using namespace std;

const int maxn = 50000 + 15;
char st[maxn];
int n;
int main() {
  scanf("%s", st);
  n = strlen(st);
  for (int i = 1; i <= n / 2; i++) {
    int m = i, pp = 0;
    for (int j = i; j < n; j++) {
      if (st[j] == st[j - i])
        pp++;
      else
        pp = 0;
      if (pp == i) {
        m -= i;
        pp = 0;
      }
      st[m++] = st[j];
    }
    n = m;
  }
  for (int i = 0; i < n; i++)
    printf("%c", st[i]);
  printf("\n");
  return 0;
}
