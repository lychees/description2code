#include <iostream>
using namespace std;
long long int findgcd(long long int a,long long int b)
{
	if(a==0)
	return b;
	return findgcd(b%a,a);
}
int main()
{
   int test;
   scanf("%d",&test);
   while(test--)
   {
     long int n,count=0;
      long long int k,num;
      scanf("%ld",&n);
      scanf("%lld",&k);
      for(long int i=0;i<n;i++)
      {
      	scanf("%lld",&num);
      	if(num==k)
      	count++;
      }
      if(count<=2)
      printf("0/1");
      else if(count==n)
      printf("1/1");
      else
      {
      	long long int nume,deno, gcd;
      	nume=count*(count-1)*(count-2) / 6;
      	deno= n*(n-1)*(n-2) / 6;
      	gcd = findgcd(nume,deno);
      	nume=nume/gcd;
      	deno=deno/gcd;
      	printf("%lld",nume);
      	printf("/");
      	printf("%lld",deno);
      }
      printf("\n");
   }
}
