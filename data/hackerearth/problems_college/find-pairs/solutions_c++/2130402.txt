#include <iostream>
using namespace std;
long long sum[1000001];
long long freq[2000005];
int main()
{
	sum[0] = 0;
	for(long long i=1; i<1000001; i++) {
		sum[i] = sum[i-1] + i;
	}
	
	int t;
	long long count, x, n, i;
	cin>>t;
	while(t--) {
		count = 0;
		cin>>n;
		for(i=0; i<2000005; i++) {
			freq[i] = 0;
		}
		while(n--) {
			cin>>x;
			x += 1000000;
			freq[x]++;
		}
		for(i=0; i<2000005; i++) {
			count += sum[freq[i]];
		}
		cout<<count<<endl;
	}
    return 0;
}
