#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	string s;
	cin>>s;
	int len=s.size();
	int arr[30]={0};
	for(int i=0;i<len;i++)
	{
		arr[s.at(i)-'a']++;
	}
	
	int cnt=0;
	for(int i=0;i<26;i++)
	{
		if(arr[i]%2==1)
			cnt++;
	}
	if(cnt==1)
	{
		char ch;
		for(int i=0;i<26;i++)
		{
			if(arr[i]%2==1)
			{
				ch='a'+i;
				arr[i]--;
			}	
		}
		string s="";
		char tempch;
		for(int i=0;i<26;i++)
		{
			for(int j=0;j<arr[i]/2;j++)
			{
				tempch='a'+i;
				s+=tempch;
			}
		}
		string rev=s;
		reverse(rev.begin(),rev.end());
		s+=ch;
		s+=rev;
		cout<<s<<endl;
		
	}
	else if(cnt==0)
	{
		string s="";
		char tempch;
		for(int i=0;i<26;i++)
		{
			for(int j=0;j<arr[i]/2;j++)
			{
				tempch='a'+i;
				s+=tempch;
			}
		}
		string rev=s;
		reverse(rev.begin(),rev.end());
		s+=rev;
		cout<<s<<endl;
	}
	else
	{
		cout<<"-1\n";
	}
	
}
