			lens[i] = lens[i-1] * 2 + S[i].size();

			long long k;
			scanf("%lld", &k);
			char ans = '?';
			for(int i = N-1; i >= 0; -- i) {
				if(i == 0 || lens[i-1] * 2 <= k) {
					ans = S[i][(int)(k - (i == 0 ? 0 : lens[i-1] * 2))];
					break;
				}

					k = lens[i-1] * 2 - 1 - k;
			}
