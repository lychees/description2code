#include <iostream>
using namespace std;
int a[100001];

int main()
{
	int t;
	cin>>t;
	while(t--)
	{int n;
	cin>>n;
	int x;
	long sum=0;
	int flag=0;
	int i;
	
	for(i=0;i<=n-1;i++)
	{cin>>x; sum=sum + x;
	a[i]=x;
	
	}
	
	int sum1=a[0];
 
	for(i=1;i<=n-2;i++)
	{
		if(sum1 == (sum-sum1-a[i]))
		{
			flag=1;
			break;
			
		}
		sum1=sum1+a[i];
	}
	
	if(flag==1)
	cout<<"YES\n";
	else cout<<"NO\n";
	
		
		
		
		
	}
	
	
	
    return 0;
}
