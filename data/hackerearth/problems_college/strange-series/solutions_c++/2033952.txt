#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    	int a[200],t,temp=0,x,r,m=1,tr=0,i,j;
	scanf("%d\n",&t);
	for(i=1;i<=t;i++)
	{
	    a[0]=2;m=1;temp=0;
	    scanf("%d\n",&r);
	    for(j=2;j<=r;j++)
	    {    tr=0;
	        while(tr<m)
	        {
	           x=a[tr]*2+temp;
	           a[tr]=x%10;
	           temp=x/10;
	           tr++;
	        }
	        while(temp>0)
	        {
	            a[m++]=temp%10;
	            temp=temp/10;
	            
	        }
	    }
	    temp=0;
	    while(a[temp]==0){
	    	a[temp++]=9;
	    }
	    a[temp]-=1;
	    for(j=m-1;j>=0;j--)
	    {
        printf("%d",a[j]);
        a[j]=0;
	    }
	   printf("\n");
	        
	}
 
    return 0;
}
