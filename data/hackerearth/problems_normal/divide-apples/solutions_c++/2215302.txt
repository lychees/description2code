#include<iostream>
#include<algorithm>
#include<cmath>
#define max 10000
using namespace std;
typedef long long LL;
int main()
{
    int n,arr1[10000],avg;
    LL arr2[ max ],val=0,sum=0,mid;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>arr1[i];
        sum += arr1[i];
    }
    avg=sum/n;
    arr2[0]=0;
    for(int i = 0; i < n-1; i++){
		arr2[i+1] = arr2[i]+arr1[i]-avg;
	}
	sort(arr2,arr2+n);
	mid = -arr2[n/2];
	for(int i=0;i<n;i++)
    {
        val += abs(arr2[i]+mid);
    }
    cout<<val;
    return 0;
}
