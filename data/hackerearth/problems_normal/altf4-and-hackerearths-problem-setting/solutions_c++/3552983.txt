#include <cstdio>
#include <algorithm>
using namespace std;
const int inf=4009;
long long dp[inf];
pair<long long,long long>num[inf];//first price second task

int main()
{
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		for (int i=1;i<=n;++i)
			scanf("%lld",&num[i].first);
		for (int i=1;i<=n;++i)
			scanf("%lld",&num[i].second);
		sort(num+1,num+n+1);
		for (int i=1;i<=n;++i)
			num[i].second+=num[i-1].second;
		for (int i=1;i<=n;++i)
		{
			long long mx=9999999999999;
			for (int j=0;j<i;++j)
				mx=min(mx,dp[j]-num[i].first*num[j].second);
			dp[i]=mx+(k+num[i].second)*num[i].first;
		}
		printf("%lld\n",dp[n]);
	}
}
