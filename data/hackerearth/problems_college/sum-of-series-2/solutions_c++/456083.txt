#define N 2001000
#define P 1001000
#define ln 15

ll i,j,k,n,t,x[N],y[N],p,q,s,r;

ll power(ll p,ll q){
	ll ans=1;
	while(q){
		if(q&1) ans=(ans*p)%mod;
		p=(p*p)%mod;
		q/=2;
	}
	return ans;
}
int prime[2000000];
ll fact[N+5],ans[N+5];
int main(){
	
	
	fact[0]=fact[1]=1;
	
	rep(i,2,N){
		fact[i]=(fact[i-1]*i)%mod;
	}
	
	
	for(i=3;i<=P;i+=2) prime[i]=1;
	prime[2]=1;
	
	for(i=3;i<=sqrt(P);i+=2){
		if(prime[i]) for(j=i*i;j<=P;j+=i+i) prime[j]=0;
	}
	
	ans[0]=ans[1]=0;
	
	for(i=2;i<=1000001;i++){
		if(prime[i]){
			ll tt=(i*(i-1))%mod;
			ll pp=(fact[(2*i)-2]*power(((fact[i-2]*fact[i])%mod),mod-2))%mod;
			tt=(tt*pp)%mod;
			ll p1=(fact[(2*i)-1]*power(((fact[i-1]*fact[i]))%mod,mod-2))%mod;
			p1=(p1*i)%mod;
			ans[i]=(ans[i-1]+((tt+p1)%mod))%mod;
		}
		else{
			ans[i]=ans[i-1];
		}
	}
	
	inp(t);
	while(t--){
		inp(n);
		cout<<ans[n]<<'\n';
	}
	
	return 0;
}
