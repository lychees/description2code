n = input()
a = [map(int, raw_input().split()) for _ in range(n)]
h, k = {}, {}
for i, j in a:
    h[i] = h.get(i, 0)+1
    k[i] = k.get(j, 0)+1

for i, j in a:
    p = h.get(j, 0)
    print n-1+p, n-1-p
