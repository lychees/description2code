def isprime(p):
    c=0
    for i in xrange(p):
        if i!=0 and i!=1:
            if p%i==0:
                return 0;
    return 1;
a=[]
c=0
for i in xrange(501):
    if i==0:
        a.append(0)
    elif i==1:
        a.append(0)
    else:
        if isprime(i)==1:
            c=c+1
        a.append(a[i-1]+pow(2,c-1))
