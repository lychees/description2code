def fact(n):
    k = 1
    for i in range(1, n+1):
        k = k*i
    return k


t = int(raw_input())
for i in range(0, t):
    n = int(raw_input())
    ans = fact(n)
    copy = str(ans)
    ans1 = 0
    for j in copy:
        ans1 = ans1 + int(j)
    print ans1
