#include <iostream>
using namespace std;
#include<set>
#include<math.h>
int main()
{
    //cout << "Hello World!" << endl;
    int n,i,a;
    set<int>f;
    cin>>n;
    while(n--)
    {
    	cin>>a;
    	for(i=1;i<=sqrt(a);i++)
    	{
    		if(a%i==0)
    		{
    			f.insert(i);
    			f.insert(a/i);
    		}
    		if(f.size()>=4)
    		break;
    	}
    	if(f.size()>=4)
    	cout<<"YES\n";
    	else cout<<"NO\n";
    	f.clear();
    }
    return 0;
}
