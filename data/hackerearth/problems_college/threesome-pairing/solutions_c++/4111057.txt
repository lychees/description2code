#include <iostream>
using namespace std;

int main()
{
    int t,n,a[100],i,j,c;
    cin>>t;
    while(t--){
    	cin>>n;c=0;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	for(i=0;i<n;i++){
    		for(j=i+1;j<n;j++){
    			if((a[i]+a[j])%3==0)
    			c++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
