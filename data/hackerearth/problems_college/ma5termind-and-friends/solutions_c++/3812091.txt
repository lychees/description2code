ll arr[100005],brr[100005],crr[100005],ans[100005],temp[100005];

int main()
{
	ll t,n,c,q,l,r,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>c;
		
		for(i=0;i<=100000;i++)
		{
			arr[i]=0;
			brr[i]=0;
			crr[i]=0;
			ans[i]=0;
			temp[i]=0;
		}
		for(i=1;i<=c;i++)
		cin>>arr[i]>>brr[i]>>crr[i];
		
		cin>>q;
		
		for(i=1;i<=q;i++)
		{
			cin>>l>>r;
			temp[l]+=1;
			temp[r+1]-=1;
		}
		
		
		for(j=1;j<=c;j++)
		{
		//	if(crr[j]>=0)
			//{
				temp[j]=temp[j-1]+temp[j];
			ans[arr[j]]+=(temp[j]*crr[j]);
			ans[brr[j]+1]-=(temp[j]*crr[j]);
		  // }
		   /*else
		   {
		   	ans[arr[j]]-=(temp[j]*abs(crr[j]));
			ans[brr[j]+1]+=(temp[j]*abs(crr[j]));
		   }*/
		}
		
		for(i=1;i<=n;i++)
		{
			ans[i]=ans[i]+ans[i-1];
			cout<<ans[i]<<" ";
		}
		cout<<"\n";
	}
	return 0;
}
