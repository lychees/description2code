#include <iostream>
using namespace std;
#include<string.h>

int main()
{
    char str[101][101];
    int t,n,i,j,gasit;
    scanf("%d",&t);
    while(t){
    	t--;
    	scanf("%d",&n);
    	for( i = 0 ; i < n; i++)
    		scanf("%s",str[i]);
    	gasit = 0;
    	for( i = 0 ; i < n; i++)
    		for( j = 0 ; j < n ;j++)
    			if( j != i && strstr(str[i],str[j]) == str[i]){
    				gasit = 1;
    				i = n;
    				j = n;
    			}
    	if( gasit )
    		printf("NO\n");
    	else
    		printf("YES\n");
    				
    }
    return 0;
}
