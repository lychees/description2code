#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int N=0;
    cin >>N;
    vector <long long> v;
    for (int i=0; i<N; ++i)
    {
    	long long num;
    	cin >>num;
    	v.push_back(num);
    	
    }
    sort (v.begin(), v.end());
    
    for (int i=0; i<v.size(); ++i)
    	cout << v[i] << endl;
    	
    return 0;
}
