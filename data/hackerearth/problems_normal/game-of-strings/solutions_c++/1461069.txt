#include <bits/stdc++.h>

using namespace std;

#define f(i,a,n) for(int i = a; i < n; i++)
#define max1 5002

int dp[max1][max1];

int lcs(string f,string s) {
	int lf = f.size();
	int ls = s.size();
	
	string a="";
	f(i,0,lf) {
		if(f[i]=='a'||f[i]=='e'||f[i]=='i'||f[i]=='o'||f[i]=='u') {
			continue;
		}
		else {
			a.push_back(f[i]);
		}
	}
	
	string b="";
	
	f(j,0,ls) {
		if(s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u') {
			continue;
		}
		else {
			b.push_back(s[j]);
		}
	}
	int la = a.size();
	int lb = b.size();

	f(i,0,la+1) {
		dp[i][0] = 0;
	}
	
	f(j,0,lb+1) {
		dp[0][j] = 0;
	}
	
	f(i,1,la+1) {
		f(j,1,lb+1) {
			if(i==0||j==0) {
				dp[i][j]=0;
			}
			else {
				if(a[i-1]==b[j-1]) {
					dp[i][j] = 1 + dp[i-1][j-1];
		
				}
				else {
					dp[i][j] = max(dp[i][j-1] , dp[i-1][j]);
				}
			}
		}
	}
		
	return dp[la][lb];
}

int main()
{
	ios_base::sync_with_stdio(0);
	string a,b;
	
	cin>>a;
	cin>>b;
	cout<<lcs(a,b)<<endl;
	
	return 0;
}
