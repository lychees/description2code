#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	char str[110];
	scanf("%s",str);
	ll int ans=0;
	ll int mc=1;
	ll int i;
	for(i=1;str[i]!='\0';i++)
	{
		if(str[i]==str[i-1])
		{
			mc++;
		}
		else
		{
			if((mc%2)==0)
			{
				ans++;
			}
			mc=1;
		}
	}
	if((mc%2)==0) //As for last alphabet/sequence we also need to check
	{
		ans++;
	}
	printf("%lld\n",ans);
	return 0;
}
