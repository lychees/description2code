int main() {
	int t; cin >> t;
	while (t --) {
		int n; cin >> n;
		vector<string> s;
		
		int cnt = 0;
		for (int i = 0; i < n; i ++) {
			string ss; cin >> ss;
			for (int j = 0; j < n; j ++) if(ss[j] == '#') cnt ++;
			s.pb(ss);
		}
		int a = 0, b = 0;
		bool flag = false;
		for (int i = 0; i < n; i ++) {
			for (int j = 0; j < n; j ++) {
				if(s[i][j] == '#') a ++;
				if(s[j][i] == '#') b ++;
			}
			if(a == cnt - a or b == cnt - b) {
				puts("YES");
				flag = true;
				break;
			}
		}
		if(!flag) puts("NO");
	}
	return 0;
}
