#include <bits/stdc++.h>
using namespace std;
int n, d, a, b, ans=1024,curr;
bool used[1024];
vector<int> v[1024];

void dfs(int i)
{
	curr++;
	used[i]=true;
	for(int j=0;j<v[i].size();j++)
	{
		if(!used[v[i][j]])
		{
			dfs(v[i][j]);
		}
	}
}
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n>>d;
	while(d--)
	{
		cin>>a>>b;
		v[a].push_back(b);
	}
	for(int i=1;i<=n;i++)
	{
		memset(used,0,sizeof(used));
		curr=0;
		dfs(i);
		ans=min(ans,curr);
	}
	cout<<ans<<endl;
	return 0;
}
