#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<cmath>
#include<set>
using namespace std;
typedef long long int ll;
int main()
{
       cin.tie(NULL);
    ios::sync_with_stdio(false);
    ll T,n,m,i,j,ans=0,ans1=0,num=0,k;
    cin>>T;
    while(T--)
    {
        ans=0;
        num=0;
        ans1=0;
        j=0;
        cin>>n>>m;
        ll A[n];
        for(i=0;i<n;i++)
            cin>>A[i];
        sort(A,A+n);
        for(i=0;i<n;i++)
        {
            if((num+A[i])>m)
            {
               // cout<<"A"<<endl;
               for(k=j;k<=i;k++)
               {
                  // cout<<"B"<<endl;
                   if((num+A[i])>m)
                       num-=A[k];
                    else break;
               }
               if(k==(i+1))
               {
                 //  cout<<"C"<<endl;
                   if(ans1>ans)
                    ans=ans1;
                   break;
               }
               else{
                   // cout<<"D"<<endl;
                  if(ans1>ans)
                    ans=ans1;
                  ans1=i-k+1;
                  j=k;
                  num+=A[k];
               }
            }
            else {
                   // cout<<"E"<<endl;
                num+=A[i];
                ans1++;
            }
        }
        if(ans1>ans)
            ans=ans1;
        cout<<ans<<endl;
    }
    return 0;
}
