#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	long long int a[1000000],b[1000000],i;
	cin>>n;
	for(int i=0;i<n;i++){
	cin>>a[i];
	}
	sort(a,a+n);
	int c=0,j=n-1;
	for(i=0;i<n/2;i++){
		b[c++]=a[i];
		b[c++]=a[j];
		j--;
	}
	long long int sum=0;
	for(int i=0;i<n;i++){
		if(i==n-1){
		sum=sum+(abs(b[i]-b[0]));
	}
	else
		sum=sum+(abs(b[i]-b[i+1]));
	}
	cout<<sum<<endl;
return 0;
}
