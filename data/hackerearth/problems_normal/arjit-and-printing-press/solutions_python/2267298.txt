t = input()
for _ in xrange(t):
    s = raw_input()
    r = raw_input()
    R = [ord(i) for i in r]
    R.sort()
    S = [ord(i) for i in s]
    i = 0
    while(i<len(S) and len(R)!=0):
        if(S[i]>R[0]):
            S[i]=R[0]
            R.pop(0)
        i+=1
    ss = ""
    for i in S:
        ss+=chr(i)
    print ss
