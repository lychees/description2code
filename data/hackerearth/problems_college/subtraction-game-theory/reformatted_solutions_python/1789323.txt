def sod(a):
    t = 0
    for num in a:
        s = 0
        for i in num:
            s = s + int(i)
        t = t+s
    return t


def soall(a):
    to = 0
    for num in a:
        to = to + int(num)
    return to


inputs = input()
numbers = []
for i in range(0, inputs):
    numbers.append(raw_input())

sd = sod(['23', '43'])
sd1 = sd % 9
tol = soall(['23', '43'])
sd2 = tol % 9
