#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
typedef long long ll;
#define sz 24

ll gp,gv,gc,size;
ll p[sz],v[sz],c[sz];

ll fun(ll n,ll x,ll y,ll z){
	ll np,nv,nc,k;
	
	if(n==size)
		return 0;
	
	np=x+p[n];
	nv=y+v[n];
	nc=z+c[n];
	if(np==gp && nv==gv && nc==gc)
		return 1;
	if(np<=gp && nv<=gv && nc<=gc){
		k=fun(n+1,np,nv,nc);
		if(k==1)
			return 1;
	}
	k=fun(n+1,x,y,z);
	if(k==1)
		return 1;
	return 0;
}

int main(){
	ll t,i,z,n;
	cin>>t;
	for(z=1;z<=t;z++){
		cin>>gp>>gv>>gc;
		cin>>n;
		for(i=0;i<n;i++)
			cin>>p[i]>>v[i]>>c[i];
		 size=n;
		 if(fun(0,0,0,0)==1)
		 	cout<<"Case #"<<z<<": yes"<<endl;
		 else
		 	cout<<"Case #"<<z<<": no"<<endl;
	}
	return 0;
}
