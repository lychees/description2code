#include <stdio.h>
#include <iostream>
using namespace std ; 

int main()
{
    long long int n ; 
    cin >> n ; 
    long long int m ; 
    cin>> m ; 
    long long int a[100000] = {0} ; 
    long long int i ; 
    for ( i = 0; i< m ; i++)
    {
    	long long int q ; 
    	cin>>q; 
    	long long int b ; 
    	cin>> b ;
		long long int k ;  
		cin >> k ; 
		long long int p  ; 
    	for (long long  int j  = q-1 ; j < b  ; j++)
    	{
    		p = a[j] ; 
    		a[j] = p  + k ; 
		}
	}

	long long int y = 0; 
    for ( i  =0 ; i< n ; i++)
    {
    	y = y + a[i] ; 
	}

	cout << y / n <<endl; 
    
    return 0;
}
