for _ in range(int(raw_input())):
    num = raw_input()
    valid = True
    for i in range(1, len(num)):
        if num[i-1]+num[i] == '21':
            valid = False
            break
    num = int(num)

    if num % 21 == 0:
        valid = False
    if valid:
        print 'The streak lives still in our heart!'
    else:
        print 'The streak is broken!'
