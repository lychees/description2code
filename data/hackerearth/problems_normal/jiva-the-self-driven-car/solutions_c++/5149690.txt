int calculateFare(int arr[]){
	double fare=0;
	for(int i=0; i<100;i++){
		if(arr[i] == 1){fare += 10;}
			if(arr[i] == 2){fare += 19;}
	if(arr[i] > 2){fare += (10 * 0.93 * arr[i]);}
	}
	if(fare - floor(fare) <0.5){return floor(fare);}
	else{
return ceil(fare);}
}
 
void sort(int pin[][2], int n)
{
	int y,y1,d;
	for(int i=0;i<n-1;i++)
	{
		for(int j=0; j<n-1-i; j++)
		{
			if(pin[j+1][0] < pin[j][0]) // swap values
			{
				y = pin[j][0];
				pin[j][0] = pin[j+1][0];
				pin[j+1][0] = y;
 
				y1 = pin[j][1];
				pin[j][1] = pin[j+1][1];
				pin[j+1][1] = y1;			}
		}
	}
}
