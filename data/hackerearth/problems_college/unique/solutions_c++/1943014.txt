#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long a[n],i;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		int c=0;
		for(i=0;i<n;i++)
		{
		if(a[i]!=a[i+1])
		{
			c++;
		}
		}
		cout<<c<<"\n";
	}
}
