#include<bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--)
	{
		int n;
		cin >> n;
		vector<long long int> a(n),b(n);
		for(int i=0;i<n;i++)
			cin >> a[i];
		for(int i=0;i<n;i++)
			cin >> b[i];
		int low,high;
		int ans=0;
		for(int i=0;i<n;i++)
		{
			low=0,high=n-1;
			while(low<=high)
			{
				int mid=(low+(high-low)/2);
				if(b[i]>=a[mid])
				{
					ans=max(ans,i-mid);
					high=mid-1;
				}
				else
					low=mid+1;
			}
		}
		cout << ans << "\n";
	}
	return 0;
}
