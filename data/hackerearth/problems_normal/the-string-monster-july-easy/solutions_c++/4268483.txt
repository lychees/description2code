vector<int>v[35],final(26,0);
int n,ok;
set<vector<int> > st;
void solve1(int ind,vector<int>&send)
{
	if(ind == n/2)
	{
		if(send==final)ok=1;
		vector<int>tmp = send;
		st.insert(tmp);return;
	}
	solve1(ind+1,send);
	int i;
	for(i=0;i<26;++i)
		send[i]+=v[ind][i];

	solve1(ind+1,send);
	for(i=0;i<26;++i)
		send[i]-=v[ind][i];
	
}
void solve2(int ind,vector<int>&send)
{
	if(ind == n)
	{
		if(send==final)ok=1;
		vector<int>tmp(26,0);
		for(int i=0;i<26;++i)
		{
			tmp[i] = final[i]-send[i];
		}
		if(st.find(tmp)!=st.end())
			ok=1;
		return;
	}
	solve2(ind+1,send);
	int i;
	for(i=0;i<26;++i)
		send[i]+=v[ind][i];

	solve2(ind+1,send);
	for(i=0;i<26;++i)
		send[i]-=v[ind][i];
	
}

{

		for(i=0;i<=n+1;++i)v[i].clear();

			scanf("%s",s);
			vector<int>hsh(26,0);
			for(j=0;s[j];++j)
			{
				hsh[s[j]-'a']++;
			}
			v[i]=hsh;
		}
		char sex[905];
		scanf("%s",sex);
		for(i=0;i<26;++i)final[i]=0;
		for(i=0;sex[i];++i)
		{
			final[sex[i]-'a']++;
		}
		
		vector<int>send(26,0);
		solve1(0,send);

		send.clear();

		if(!ok)solve2(n/2,send);
