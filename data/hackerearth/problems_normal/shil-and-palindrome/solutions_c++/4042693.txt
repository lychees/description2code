#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
	char a[1000000],b[1000000];
	scanf("%s",a);
	long long int size = strlen(a),count1 = 0,j = 0,temp = 0,k,num,num1;
	sort(a,a+size);
	//scanf("%s",a);
	long long int count[26] = {0};
	for(int i = 0;i < size;i++)
	{
		count[a[i] - 97]++;
	}
	if(strlen(a) == 1)
	{
		printf("%s",a);
		return 0;
	}
	for(int i = 0;i < 26;i++)
	{
		//cout << count[i] << " ";
		if(count[i] % 2 !=0)
		{
			count1++;
		}
	}
	if(count1 > 1)
	{
		printf("-1\n");
		return 0;
	}
	for(k = 0;k < 26;k++)
	{
		if(count[k] == 0)
		{
			continue;
		}
		else if(count[k] % 2 != 0)
		{
			temp = k;
			/*while(count[k] != 1)
			{
				b[j] = (char)(k + 'a');j++;
				count[k] = count[k] - 2;
			}*/
			num = count[k]/2;
			while(num!= 0)
			{
				b[j] = (char)(k+'a');
				j++;
				num--;
			}
			
		}
		else if(count[k] % 2 == 0)
		{
			num1 = count[k]/2;
			while(num1!=0)
			{
				b[j] = (char)(k + 'a');j++;
				num1--;
			}
		}
	}
	if(temp != 0)
	{
		b[j] = (char)(temp + 'a');
	}
	printf("%s",b);
	for(int m = strlen(b)-1;m>=0;m--)
	{
		if(temp !=0)
		{
			temp = 0;
			continue;
		}
		else
		{
			printf("%c",b[m]);
		}
	}
	//printf("%c %d",b[j],j);
	return 0;
}
