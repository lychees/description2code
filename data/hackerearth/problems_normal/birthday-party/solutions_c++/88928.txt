#include<stdio.h>
#include<map>
#include<stdlib.h>
#include<string.h>
#include<iostream>
using namespace std;
bool checkSubset(int s,int *arr,int n)
{
	bool dp[s+1];
	memset(dp,false,sizeof(dp));
	dp[0]=true;
	int i,t;
	
	for(i=0;i<n;i++)
	{
		for(t=s;t>=arr[i];t--)
		{
			dp[t]=dp[t]|dp[t-arr[i]];
			if(dp[s]==true)
				return true;
		}
	}
	return false;
}

int main()
{
	int test;
	scanf("%d",&test);
	
	while(test>0)
	{
		map<int,int> m;
		map<int,int>::iterator it;
		m.clear();
		int i,num,val,s;
		scanf("%d%d",&num,&val);
		
		int arr[val];
		for(i=0;i<val;i++)
			scanf("%d",&arr[i]);
		
		
		for(i=0;i<num;i++)
		{
			scanf("%d",&s);
			m[s]++;
		}
		
		int cnt=0;
		for(it=m.begin();it!=m.end();it++)
		{
			s=(*it).second;
			if(checkSubset(s,arr,val))
				cnt++;
		}
		printf("%d\n",cnt);
		test--;
	}
}
