#include<stdio.h>
#include<vector>

using namespace std;

typedef long long ll;

const ll MOD = (1e+9)+7;
ll ncr[101][101];
ll n;

ll pow(ll a, ll m)
{
    ll res = 1;
    while(m)
    {
        if(m&1)
            res = (res*a)%MOD;
        a = (a*a)%MOD;
        m /= 2;
    }
    return res;
}

vector<ll> derangement()
{
    vector<ll> derag(n+1);
    derag[0] = 1;
    derag[1] = 0;
    for(ll i=2;i<=n;++i)
    {
        derag[i] = ((i-1)*(derag[i-1]+derag[i-2]))%MOD;
    }
    return derag;
}

void comb()
{
    for(ll i=0;i<=n;++i)
    {
        for(ll j=0;j<=i;++j)
        {
            if(i==j||j==0)
                ncr[i][j]=1;
            else
                ncr[i][j] = (ncr[i-1][j-1] + ncr[i-1][j])%MOD;

        }
    }
}

int main()
{
    ll l,r;

    scanf("%lld%lld%lld",&n,&l,&r);
    comb();
    vector<ll> derag = derangement();
    ll sum=0;
    for(ll i=l;i<=r;++i)
    {
        sum=(sum + derag[i]*ncr[n][i])%MOD;

    }
    ll q=1;
    for(ll i=2;i<=n;++i)
        q = (q*i)%MOD;
    ll ans = (sum*pow(q,MOD-2))%MOD;
    printf("%lld",ans);
    return 0;
}
