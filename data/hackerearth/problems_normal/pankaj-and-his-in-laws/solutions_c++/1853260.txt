#include<iostream>
#include<cstdio>
#include<cstring>
#include<bitset>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    int a[n+10];
    for(i=0;i<n;i++)
        cin>>a[i];
    int lis[n+10];
    for(i=0;i<n;i++)
        lis[i]=1;
    for(i=1;i<n;i++)
    {
        for(j=0;j<i;j++)
        {
            //cout<<lis[i]<<" ";
            if((a[i]>a[j])&&(lis[i]<(lis[j]+1)))
            lis[i]=1+lis[j];

        }

    }
    int max=1;
    for(i=0;i<n;i++)
    {
        if(max<lis[i])
            max=lis[i];
    }
    string binary=std::bitset<7>(max).to_string();
    for(i=0;i<binary.length();i++)
    {
        if(binary[i]=='0')
            continue;
        else
            break;
    }
    for(j=i;j<binary.length();j++)
        cout<<binary[j];


    return 0;
}
