#include <iostream>
#include<string>
using namespace std;

int main()
{
	string name;
	cin >> name;
	long long int q, a, b, len, c, d;
	len = name.length();
	cin >> q;
	for (int i = 0; i<q; i++) {
		cin >> a >> b;
		c = (a-1)%len;
		d = (b-1)%len;
		//cout << c << " " << d << endl;
		if (name[(c)] == name[d]) {
			cout << "Yes" << endl;
		}
		else {
			cout << "No" << endl;
		}
	}

	return 0;
}
