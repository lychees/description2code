#include <iostream>
using namespace std;
int main()
{
    int t , n , num_zeros , num_ones , k;
    bool status[102];
    
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	
    	status[0] = 1;
    	status[n+1] = 1;
    	
    	for(int i=1; i<=n ; i++){
    		cin>>status[i];
    	}
    	
    	bool temp_prev;
    	
    	for(int i=1; i<=k; i++){
    		
    		temp_prev = status[0];
    		
    		num_ones = num_zeros = 0;
    		for(int i=1; i<=n; i++){
    			if(temp_prev && status[i+1]){
    				temp_prev = status[i];
    				status[i] = 1;
    				num_ones += 1;
    			}else{
    				temp_prev = status[i];
    				num_zeros += 1;
    				status[i] = 0;
    			}
    		}
    		
    		if(num_ones ==n || num_zeros ==n){
    			break;
    		}
    	}
    	
    	for(int i=1; i<=n; i++)
    	  cout<<status[i]<<" ";
    	cout<<endl;  
    	
    }
    return 0;
}
