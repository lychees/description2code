#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int i;
    int a[n];
    for(i=0;i<n;i++)
    {
    	int tmp;
    	cin >> tmp;
    	a[i]=tmp;
    }
    int diff = a[0];

	if(n>1)
    {
    	diff=a[1] - a[0];
    	for(i=2;i<n;i++)
    	{
	    	diff= a[i] - diff;
    	}
    }
    
    if(diff)
    cout << "NO" << endl;
    else
    cout << "YES" << endl;
    return 0;
}
