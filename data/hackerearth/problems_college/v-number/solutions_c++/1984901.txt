#include <iostream>
#include<cstdio>
#include<cstring>

using namespace std;

int main() {
	int a[10000],b[10000],i,j,t,k,s1=0,s2=0;
	char str[10000];
	scanf("%d",&t);
	while(t--)
	{
		s1 = 0;
		s2 = 0;
		scanf("%s",str);
		k = strlen(str);
		for(i=0;i<k;i++)
		a[i] = str[i] - '0';
		for(i=0;i<k/2;i++)
		s1 = s1 + a[i];
		for(i=k/2;i<k;i++)
		s2 = s2 + a[i];
		if(s1==s2)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
