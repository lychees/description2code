#include <iostream>
#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;

int main()
{
    int t,p,x,y,u,v;
    scanf("%d",&t);
    
    while(t--) {
    	scanf("%d %d %d %d %d",&p,&x,&y,&u,&v);
    	if(x==u && y==v) {
    		if(p>0)
    		printf("Counter-Terrorists Win !\n");
    		else
    		printf("Terrorists Win !\n");
    	}
    	else if((abs(x)+abs(y))%2==(abs(u)+abs(v))%2) {
    		if(abs(abs(x)-abs(u)) == abs(abs(y)-abs(v))) {
    			if(p>1) {
    				printf("Counter-Terrorists Win !\n");
    			} else 
    			printf("Terrorists Win !\n");
    		} else {
    			if(p>2) 
    			printf("Counter-Terrorists Win !\n");
    			else
    			printf("Terrorists Win !\n");
    		}
    	} else 
    		printf("Terrorists Win !\n");
    	
    }
    
    
    return 0;
}
