#include<bits/stdc++.h>
using namespace std;
const int MAX_VALUE=10000002;
int arr[MAX_VALUE]={0},hsa[MAX_VALUE]={0};
vector<int>vec;

void sieve()
{
    arr[0]=1;
    arr[1]=1;
    for(int i=2;i<=sqrt(MAX_VALUE);i++)
    {
        if(arr[i]>0)continue;
        
        for(int j=2*i;j<MAX_VALUE;j+=i)
        {
            arr[j]=i;
        }

    }
    int val=3;
    for(int i=5;i<MAX_VALUE;i++)
    {
        if(arr[i]==0 )
        {
            if(hsa[i-val]==0)
                hsa[i-val]++;
            
            val=i;
        }
    }

    for(int i=0;i<1000;i++)
    {
        while(hsa[i])
            {
                vec.push_back(i);
                vec.push_back(i);
                hsa[i]--;
            }
    }
    
}


int main()
{
    sieve();
    sort(vec.begin(),vec.end());
    
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int N;
         cin>>N;
        
        int l=0;
        int r=vec.size()-1;
        
        int flag=0;
        while(l<r)
        {
            
            if(vec[l]+vec[r]==N)
            {
                flag=1;
                break;
            }
            else if((vec[l]+vec[r])<N)
            {
                l++;
            }
            else r--;
        }
        if(flag)printf("YES\n");
        else printf("NO\n");

    }
    
    
    return 0;
}
