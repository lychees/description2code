#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<map>
using namespace std;
long long int a[10005];

int main()
{
    int t;
    long long int i,j,k,chef,jhool,n,x,y;
    scanf("%d",&t);
    while(t--)
    {
        chef=0;jhool=0;

        scanf("%lld%lld%lld",&x,&y,&n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
        }
        sort(a,a+n);
        i=0;
        while( i<n &&  x>=a[i])
        {
            x-=a[i++];
            chef++;
        }
        i=0;
        while( i<n && y>=a[i])
        {
            y-=a[i++];
            jhool++;
        }
        if(chef>jhool)
            cout<<"Chef Won"<<endl;
        else if(chef<jhool)
            cout<<"Jhool Won"<<endl;
        else
            cout<<"Tie\n";

    }

}
