#include<string>
#include<iostream>
#include<algorithm>
 
using namespace std;
 
int main()
{
 
    int t;
    cin>>t;
    while(t--){
    string s;
    cin>>s;
    pair<int,int> p[s.length()+1];
    int flag[3]={0,0,0};
    int i;
    for(i=0;i<s.length();i++)
    {
        if((s[i]-97)<3)
        {
            flag[s[i]-97]++;
        }
        p[i]=make_pair(flag[1]-flag[0],flag[2]-flag[0]);
    }
    p[i]=make_pair(0,0);
    
 
    /*for(int i=0;i<s.length()+1;i++)
        cout<<p[i].first<<" "<<p[i].second<<endl;*/
 
    sort(p,p+s.length()+1);
 
    pair<int,int> last=make_pair(-10000,-10000);
    int sum=0,inc=1;
    for(int i=0;i<s.length();i++)
    {
        if(p[i]==p[i+1]) 
        {
            inc++;
        }
        else
        {
            sum+=((inc)*(inc-1))/2;
            inc=1;
        }
    }
    sum+=((inc)*(inc-1))/2;
 
    cout<<sum<<endl;}
    return 0;
}
