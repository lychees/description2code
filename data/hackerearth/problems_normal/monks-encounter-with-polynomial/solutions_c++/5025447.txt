#include<iostream>
#include<vector>
using namespace std;
int main()
{  int val;
	int n;
	cin>>n;
	while(n--)
	{
		int a,b,c,k;
		cin>>a>>b>>c>>k;
		int i=0,cnt=0;
		do
		{
			val=(a*i*i+b*i+c);
			if(val<k)
			{
			cnt++;
			}	
		
			
			i++;
		}
		while(val<k);
		cout<<cnt<<"\n";
	}
}
