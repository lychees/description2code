#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
 
using namespace std;
 
vector< pair<long int, int> >v;
 
int main()
{
    int N;
    cin >> N;
    long int a[N];
    cin >> a[0];
    v.push_back(make_pair(0 , 1));
    v.push_back(make_pair(a[0] , 2));
    for(int i=1; i<N; i++)
    {
    	cin >> a[i];
    	a[i] += a[i-1];
    	v.push_back(make_pair(a[i],i+2));
    }
    sort(v.begin(), v.end() );
    int m = v[0].second;
    int l=0, val=0, count=0;
       
    for(int i=1; i<=N; i++)
    {
    	int x = v[i].second;
    	val = max(0, x-m);
    	if(val > l)
    	{
    		l = val;
    		count = 1;
    	}
    	else if(val == l)
    		count++;
    	m = min(x,m);
    }
    if(l != 0)
    cout<<l<<" "<<count<<endl;
    else
    cout << -1<<endl;
    return 0;
}
