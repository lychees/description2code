#include<cstdio>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i,j,k=0,a[100100];
	for(i=2;i<=100000;i++)
	{
		n = i;
		int c=0;
		if(n%2==0)
		{
			while(n%2==0)
			{
				n = n/2;
			}
			c++;
		}
		for(j=3;j<=sqrt(n);j = j+2)
		{
			if(n%j==0)
			{
				while(n%j==0)
				{
					n = n/j;
				}
				c++;
			}
		}
		if(n>2)
		{
			c++;
		}
		if(c>=3)
		{
			a[k]=i;
			k++;
		}
	}
	
	int t;
	cin>>t;
	while(t--)
	{
		long long int n;
		cin>>n;
		cout<<a[n-1]<<"\n";
	}
}
