#include <bits/stdc++.h>
using namespace std;
     
int minm(string s)
{
    int i=0,k=0,j=1;
    s=s+s;
    while((i+k)<s.size() && (j+k)<s.size())
    {
    	if(s[i+k]==s[j+k])
    	k++;
    	else
    	if(s[i+k]<s[j+k])
    	{
    		i=i+k+1;
    		if(i<=j)
    		i=j+1;
    		k=0;
    	}
    	else
    
    	{
    		j=j+k+1;
    		if(j<=i)
    		j=i+1;
    		k=0;
    	}
    	}
   return min(i,j);
}
     
int main()
{
	ios_base::sync_with_stdio(0);
	int i;
    string s;
    cin>>s;
       
    int res=minm(s);
    for(i=res;i<s.size();i++)
       cout<<s[i];
       
     for(i=0;i<res;i++)
       cout<<s[i];
       return 0;
    }
