using namespace std;
#include <iostream>
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,x;
    	cin>>n>>x;
    	int a[1000002]={};
    	for(int i=0;i<n;i++)
    		cin>>a[i];
    	int curr_sum = a[0], start = 0, i;
    	int flag=0;
    	int sum=x;
    	for (i = 1; i <= n; i++)
    	{
            while (curr_sum > sum && start < i-1)
        	{
            	curr_sum = curr_sum - a[start];
            	start++;
       	 	}
 	        if (curr_sum == sum)
     	    {
        	    flag=1;
        	    break;
        	}
 			if (i < n)
          		curr_sum = curr_sum + a[i];
    	}
    	if(flag==1)	
    		cout<<"YES\n";
    	else
    		cout<<"NO\n";
    }
    return 0;
}
