n = int(raw_input())
losers = []
winners = []
for dummy in range((2 ** n) - 1):
    round = raw_input().split()
    losers.append(round[1])
    winners.append(round[0])

winners_set = set(winners)
losers_set = set(losers)
for winner in winners_set:
    if winner not in losers_set:
        print winner
        break
