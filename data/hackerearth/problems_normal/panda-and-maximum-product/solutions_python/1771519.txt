n=input()

l=map(int,raw_input().split())

lneg=[]
lpos=[]

pcount=0
ncount=0
zcount=0

for element in l:
	if element>0:
		lpos.append(element)
		pcount+=1
	elif element<0:	
		lneg.append(element)
		ncount+=1
	else:
		zcount+=1

lpos.sort()
lneg.sort()

if pcount>=2:
	if ncount>=2:
		print max(lneg[0]*lneg[1], lpos[-1]*lpos[-2])
	else:
		print lpos[-1]*lpos[-2]
		
elif ncount>=2:
	print lneg[0]*lneg[1]
	
else:
	if zcount>0:
		print 0
	else:
		print lpos[0]*lneg[-1]
