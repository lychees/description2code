#include<iostream>
#include<cstdio>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<limits.h>
using namespace std;
int get_ans(int i, int j)
{
    if(j == 1)
        return i-1;
    if(i == 1)
        return j-1;
    if(i == j)
       return 1000000;
    if(i % 2 == 0 && j % 2 == 0)
        return 1000000;
    if(i > j)
        return 1 + get_ans(i-j,j);
    else
        return 1 + get_ans(i, j-i);
}
int main()
{
  int t,n;
  scanf("%d",&t);
  while(t--)
  {
      scanf("%d",&n);
      if(n == 1)
        printf("0\n");
      else if(n  == 2)
        printf("1\n");
      else
      {
          int cnt = n-1;
          for(int i=1,j=n-1; i < j; i++,j--)
          {
               if(i % 2 == 0 && j % 2 == 0)
                continue;
               int tmp = 1 + get_ans(i,j);
               //printf("%d\n",tmp);
               cnt = min(tmp,cnt);
          }
          printf("%d\n",cnt);
      }
  }
  return 0;
}
