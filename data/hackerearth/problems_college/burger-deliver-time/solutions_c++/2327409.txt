#include<iostream>
#include<vector>
#include<map>
using namespace std;
int main(){
	int n,i,a,b,j=0;
	map<int,vector<int> > mp;
	map<int,vector<int> >::iterator imp;
	vector<int>::iterator iv;
	cin>>n;
	for(i=0;i<n;i++){
		cin>>a>>b;
		mp[a+b].push_back(++j);
	}
	
	for(imp=mp.begin();imp!=mp.end();imp++){
		for(iv=imp->second.begin();iv!=imp->second.end();iv++)
			cout<<*iv<<" ";
	}
	return 0;
}
