#include <iostream>
#include<stdio.h>
#include<vector>
#include<set>
#include<map>
using namespace std;

int main()
{
	int n ; scanf("%d",&n);
	set<int> m_set ;
	map<int , int> m_map ;
	
	
	for(int i=0 ; i<n ; i++)
	{
		int val ; scanf("%d",&val) ;
		m_map[val]++;
		m_set.insert(val);
	}
	cout<<n<<endl;
	
	for( set<int> :: iterator it = m_set.begin() ; it != m_set.end() ; it++)
	{
		int count = m_map[*it];
		cout<<n-count<<endl;
		
		n -= count;
		if(n==1)
			break;
	}
	
    return 0;
}
