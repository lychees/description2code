#include<bits/stdc++.h>
using namespace std;

const int N=1e5+10;
const int inf=0x3f3f3f3f;

int a[N];

int main() {
  int n,t;
  scanf("%d%d",&n,&t);
  for(int i=0;i<n;i++) {
    scanf("%d",a+i);
  }
  while(t--) {
    int minVal=inf;
    int l,r;
    scanf("%d%d",&l,&r);
    for(int i=l;i<=r;i++) {
      minVal=min(minVal,a[i]);
    }
    printf("%d\012",minVal);
  }
  return 0;
}
