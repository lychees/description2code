#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int o[100000];
    int i,test;
    o[0]=0;
    for(i=1;i<=n;i++)
    {
    	scanf("%d",&test);
		o[i]=o[i-1]+test;
    }
       int q;
    scanf("%d",&q);
    while(q--)
    {
    	int l,r;
    	scanf("%d%d",&l,&r);
    	printf("%d\n",o[r]-o[l-1]);
    }
    return 0;
}
