#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
    char a[120];
    scanf("%s\n",a);
    int i,c=0,j,s=0;
    int n=strlen(a);
    for(i=0;i<n;i++){c=1;
    for(j=i+1;j<n;j++){
    	if(a[i]==a[j])
    	{
    		c++;
    	}
    	else
    		break;
    }
    i=i+c-1;
    if(c%2==0){
    s++;
   
    }
    }
    printf("%d\n",s);
    return 0;
}
