#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int n1,n2;
    while(t--)
    {
    	cin>>n1>>n2;
    	long long int count=0;
    	while(n1!=0&&n2!=0)
    	{
    		long long int sum=0;
    		long long int r=n1%10;
    		long long int r1=n2%10;
    		sum=r+r1;
    		if(sum>=10)
    			count++;
    			
    		n1=n1/10;
    		n2=n2/10;
    	}
    	if(count==0)
    		cout<<"No carry operation"<<"\n";
    	else if(count==1)
    	   cout<<count<<" "<<"carry operation"<<"\n";
    	
    	else
    		cout<<count<<" "<<"carry operations"<<"\n";
    	
    }
    return 0;
}
