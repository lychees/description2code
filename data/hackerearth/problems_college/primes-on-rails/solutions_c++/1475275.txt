const int MAX = 100007;

int p[MAX];
int A[MAX];

int main()
{
  //  freopen("in.txt", "r", stdin);

    //freopen("input.txt", "r", stdin);
    //freopen("output.txt","w",stdout);

    p[1] = 1;
    for(int i = 2; i * i < MAX; ++i)
        if (!p[i])
            for(int j = i * i; j < MAX; j += i)
                p[j] = 1;

    int n;
    cin >> n;
    int res = 0;
    FOR(i,0,n)
    {
        int a , b , c;
        scanf("%d%d%d" , &a , &b , &c);
        int t = 3600 * a + 60 * b + c;
        int d = 2 * 60;
        if (!p[t])
        {
            d = 5 * 60;
        }
        else
        {
            if (t % 2 == 1)
            {
                d = 3 * 60;
            }
        }
        FOR(j,0,d)
        {
            A[t + j]++;
            res = max(res , A[t + j]);
        }
    }
    cout << res << endl;

    return 0;
}
