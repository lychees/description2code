N = int(raw_input())
S = map(int, raw_input().split())

seen = set()
total = 0

def traverse(path):
    global total
    for x in S:
        s = tuple(sorted(path + (x,)))
        if s in seen:
            continue
        seen.add(s)
        ss = sum(s)
        if ss == N:
            total += 1
            continue
        if ss > N:
            continue
        traverse(s)

traverse(())
print total
