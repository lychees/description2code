// A naive recursive C++ program to count number of ways to cover
// a distance with 1, 2 and 3 steps
#include<iostream>
using namespace std;
 
// Returns count of ways to cover 'dist'
long long int printCountRec(long long int dist)
{
    long long int dp[dist],i;
    dp[0]=1;
    dp[1]=2;
    for(i=2;i<dist;i++)
    dp[i]=dp[i-1]+dp[i-2];
    return dp[dist-1];
}
 
// driver program
int main()
{
    long long int n,tests;
    cin>>tests;
    while(tests>0)
    {
    cin>>n;
    cout << printCountRec(n)<<endl;
    tests--;
    }
    return 0;
}
