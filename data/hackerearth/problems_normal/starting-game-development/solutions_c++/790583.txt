#include <cstdio>
#include <cstdlib>
#include<iostream>
using namespace std;
#define M_N 105
#define M_M 5005
int mat[M_N][M_M];
int ar[M_N];
int M,N,Q;

int solve(){
	int ans = M-1;
	for(int i=0;i<N;i++){
		while(ar[i]<mat[i][ans])
			ans--;
	}
	return ans+1;
}
int main(void){
	scanf("%d %d %d", &N, &M, &Q);
	for(int i=0;i<N;i++)
		for(int j=0;j<M;j++)
			scanf("%d", &mat[i][j]);

	while(Q--){
		for(int i=0;i<N;i++)
			scanf("%d", &ar[i]);
		printf("%d\n", solve());
	}
	return 0;
}
