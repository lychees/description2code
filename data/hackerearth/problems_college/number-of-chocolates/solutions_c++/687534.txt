#define MAXN 1000010
#define LL long long
#define sd(x) scanf("%d", &x)
#define sld(x) scanf("%lld", &x)

LL a[MAXN], sum[MAXN];

inline void Solve(){
    LL n, i, ans, q;
    a[0] = 0;
    sum[0] = 0;
    sld(n);
    sld(q);
    for(i = 1; i <= n; i++){
        sld(a[i]);
        sum[i] = a[i] + sum[i - 1];
    }
    LL x1, y1, x2, y2, x, y;
    while(q--){
        sld(x1);sld(x2);sld(y1);sld(y2);
        x1++; y1++; x2++; y2++;
        if(x1 > x2){
            swap(x1, x2);
            swap(y1, y2);
        }
        if(x2 <= y1){
            x = min(x1, x2);
            y = max(y1, y2);
            ans = sum[y] - sum[x - 1];
        }
        else{
            ans = sum[y1] - sum[x1 - 1] + sum[y2] - sum[x2 - 1];
        }
        printf("%lld\n", ans);
    }
    return;
}

int main(){
    LL t = 1;
    //cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
