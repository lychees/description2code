#include <bits/stdc++.h>
using namespace std;

int main(){

	int t;
	scanf("%d",&t);
	while(t--){
		int n,k;
		scanf("%d %d",&n,&k);
		int barr[n],arr[k+1];
		memset(arr,0,sizeof(arr));
		for(int i=0;i<n;i++)
			{
			scanf("%d",&barr[i]);
			arr[barr[i]]++;
			}

		int max=-100000000;
		for(int i=1;i<=k;i++){
			if(arr[i]>max)max=arr[i];
			}

		printf("%d\n",(n-max));


	}





	return 0;
}
