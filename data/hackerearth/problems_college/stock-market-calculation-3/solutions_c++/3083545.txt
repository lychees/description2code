#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int test;
    cin>>test;
    while(test--)
    {
        lli n,k=0,p=0,ans=0;
        cin>>n;
        lli arr[n];
        for(lli i=0;i<n;i++)
            cin>>arr[i];
        lli c[n-1];
        c[n-2]=arr[n-1];
        for(lli i=n-2;i>0;i--)
        {
            c[i-1]=max(c[i],arr[i]);
            //cout<<c[i-1]<<endl;
        }
        for(lli i=0;i<n-1;i++)
        {
            if(arr[i]<c[i])
            {
                k++;
                p+=arr[i];
            }
            else
            {
                ans+=((k*arr[i])-p);
                k=0;
                p=0;
            }
        }
        ans+=((k*arr[n-1])-p);
        cout<<ans<<endl;
    }
    return 0;
}
