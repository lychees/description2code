#include <cstdio>
#include <iostream>

using namespace std;

int main() {
	
      string arr;
      cin>>arr;
      
     
     int cnt[26] = {0};
	  for(int i=0; arr[i]; i++) {
				if(arr[i]!=32 || arr[i]!=10)
					(cnt[arr[i]-97])++;
      }
      
      
		int max=cnt[0];
        for(int i=1; i<26; i++) {
			if(cnt[i]>max) {
				max=cnt[i];
				//cout<<max<<" "<<a<<endl;
			}
		}
		char ch;
		for(int i=0; i<27; i++) {
			if(cnt[i]==max) {
				ch=i+97;
				//cout<<ch<<endl;
				break;
			}
		}
		
		int flag=0;
		
		if( ch=='h'
		 || ch=='e'
		 || ch=='a'
		 || ch=='d'
		 || ch=='r'
		 || ch=='u'
		 || ch=='s') {
			 flag++;
		 }
		
		if(ch=='c'
		 || ch=='u'
		 || ch=='b'
		 || ch=='i'
		 || ch=='n'
		 || ch=='g'
		 || ch=='l'
		 || ch=='b') {
			 flag++;
			 flag++;
		 }
		 
		 if(flag==1) {
			 cout<<"Thanks head rush"<<endl;
		 } else if(flag==2) {
			 cout<<"Thanks cubing club"<<endl;
		 } else if(flag==3) {
			 cout<<"Thanks both"<<endl;
		 }else if(flag==0) {
			 cout<<"none"<<endl;
		 }
		 
    return 0;
}
