# print 'Hello World!'

n = int(raw_input())
f = 0
for i in range(0, n):
    a = raw_input()
    b = raw_input()
    for j in a:
        if j in b:
            f = 1
            break
    if f == 1:
        print "YES"
    else:
        print "NO"
    f = 0
