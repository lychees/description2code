#include <iostream>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n,count=0,i,j;
    	cin>>n;
    	long long int a[100001],b[100001];
    	
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	for(i=0;i<n;i++)
    		cin>>b[i];
    	
    	int ans=0,mid;	
    	for(i=0;i<n;i++)
    	{
    		int low=i,high=n-1;
    		while(low<=high)
    		{
    			mid=(low+high)/2;
    			if(b[mid]>=a[i])
    			{
    				j=mid;
    				low=mid+1;
    				if(ans<(j-i))
    				{
    					ans=(j-i);
    				}
    			}
    			else
    			{
    				high=mid-1;
    			}
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
