#include <iostream>
#include <stdlib.h>
using namespace std;

int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}


int main()
{
		int tc;
		cin>>tc;
		for (int i = 0; i < tc; ++i)
		{
			int no,k,min=100000;
			int arr[1000]={0};
			cin>>no>>k;
			for (int j = 0; j < no; ++j)
			{
				cin>>arr[j];
			}
				qsort (arr, no, sizeof(int), compare);
				
			for (int j = 0; j < no-k+1; ++j)			//sliding window
			{

					int start=arr[j];					//start and end number for each window
					int end=arr[k+j-1];

				int sli;
					for(sli=start; sli<=end;++sli)
					{
						int total=0;
						for (int c = j; c <j+k ; ++c)
						{
					
						if(arr[c]<sli)
						{
							total=total+(sli-arr[c])*3;

						}
						else if(arr[c]>sli)
						{
							total=total+(arr[c]-sli)*5;
						}
					}
					
				if (total<min)
				{
					min=total;
				}

				}

			}


			cout<<min<<"\n";





		}



	return 0;

}
