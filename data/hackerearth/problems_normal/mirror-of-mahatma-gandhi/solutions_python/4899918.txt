n = int(raw_input())
for i in range(n):
	s = raw_input()
	v = True
	if(len(s)%2 == 1):
		for j in range((len(s)-1)/2):
			if(s[j] != s[len(s)-1-j]):
				v = False
	else:
		for j in range(len(s)/2):
			if(s[j] != s[len(s)-1-j]):
				v = False
	iv = ['2','3','4','5','6','7','9']
	for e in iv:
		if e in s:
			v = False
	if v == False:
		print 'NO'
	else:
		print 'YES'
