#include <iostream>
using namespace std;
int main()
{
	int i,j=0,len,out=0,n;
	string s;
    cin>>n;
    int stud[n];
    for(i=0;i<n;i++)stud[i]=0;
    cin>>s;
    len=s.size();
    i=0;
    while(1)
    {
    	if(out==n-1)break;
    	if(s[i]=='a')
    	{
    		do
    		{	j++;
    			if(j>=n)j%=n;
    		}while(stud[j]==1);
    	}
    	else if(s[i]=='b')
    	{
    		stud[j]=1;
    		out++;
    		do
    		{	j++;
    			if(j>=n)j%=n;
    		}while(stud[j]==1);
    	}
    	else j=1;
    	i++;
    	if(i>=len)i%=len;
    }
    for(j=0;stud[j]==1;j++);
    cout<<j+1;
    return 0;
}
