#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define pb push_back
#define ll long long int
#define s second
#define f first
typedef pair<int,int> p;
typedef pair<int,p> pp;
int arr[100005];
int sz[100005];
vector< pp > v(100005);
int root(int a)
{
    while(arr[a]!=a)
    {
        a = arr[a];
        arr[a] = arr[arr[a]];
    }
    return a;
}
void unio(int i,int j)
{
    int x = root(i);
    int y = root(j);
    if(sz[x]<sz[y])
    {
      arr[x] = y;
      sz[y] = sz[y]+sz[x];
    }
    else
    {
        arr[y] = x;
        sz[x] = sz[x]+sz[y];
    }
}
int main()
{

int t,n,m;
int a,b,c;
cin>>t;
while(t--)
{
    for(int i=1;i<100005;i++)
    {
        sz[i]=1;
        arr[i]=i;
    }
  scanf("%d%d",&n,&m);
  int concomp = n;
  for(int i=0;i<m;i++)
  {
      scanf("%d%d%d",&a,&b,&c);
      v[i].f = c;
      v[i].s.f=a;
      v[i].s.s = b;
  }
  ll ans =1ll;
  sort(v.begin(),v.begin()+m);
  for(int i=0;i<m;i++)
  {
      if(root(v[i].s.f)!=root(v[i].s.s))
      {
          unio(v[i].s.f,v[i].s.s);
          ans = (ans*(v[i].f))%mod;
          
      }
  }
 printf("%lld\n",ans);
}
return 0;
}
