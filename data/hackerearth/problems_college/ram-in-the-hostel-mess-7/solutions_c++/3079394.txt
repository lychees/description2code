typedef long long int ll;

int main(){
//    ios_base::sync_with_stdio(0);
//    cin.tie(0);

    v[0] = 1;
    v[1] = 2;
    for(int i = 2; i < mx; i++){
        v[i] = v[i - 1] + v[i - 2];
    }
    cin >> t;
    while(t--){
        cin >> x;
        ans = v[x] * v[x];
        cout << ans << endl;
    }
    return 0;
}
