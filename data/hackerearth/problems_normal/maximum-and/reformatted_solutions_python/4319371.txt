def power(a):
    if a & (a-1) == 0:
        return True
    else:
        return False


for _ in range(input()):
    a, b = map(int, raw_input().split())
    c, d = a, b
    # p,q=map(int,raw_input().split())
    if power(b):
        b = b-1
    if b % 2 == 0:
        b = b-1
    a = b-1

    if a >= c:

        print a & b

    else:

        print max(0, c & d)
