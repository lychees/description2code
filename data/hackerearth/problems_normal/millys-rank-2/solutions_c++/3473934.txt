#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int n,x,y;
		cin>>n>>x>>y;
		ll int t1,t2;
		//no less than X people who are better --Atleast x people better than miley
		t1=x+1;
		//no more than Y people are worse than her --Atmost y people worse than miley
		t2=n-y;
		ll int ans=max(0ll,n-max(t1,t2)+1);
		cout<<ans<<endl;
	}
	return 0;
}
