def Solve():
	nm = raw_input().split(' ')
	n = int(nm[0])
	m = int(nm[1])
	c = input()
	ans = 0
	x = 1
	y = 1
	while c>0:
		c-=1
		dd = raw_input().split(' ')
		dx = int(dd[0])
		dy = int(dd[1])
		while 1:
			#print x,y 
			nexx = x + dx
			nexy = y + dy
			if nexx>=1 and nexx<=n and nexy>=1 and nexy<=m:
				x = nexx
				y = nexy
				ans+=1
			else:
				break
	print ans

t = input()
while t>0:
	t-=1
	Solve()
