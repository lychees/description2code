#include <iostream>

#include<string>

#include<algorithm>
using namespace std;
 
 
 
int main()
{
    int n,k;
    cin>>n>>k;
    int arr[n+1];
    int i;
    for(i=1;i<=n;i++)
    {
        cin>>arr[i];
    }
 
    sort(arr+1,arr+(n+1));
 
    for( i=1;i<n;i++)
    {
       if((arr[i+1]-arr[i])>k)
        {break;}
        else
            continue;
    }
    cout<<arr[i]<<endl;
 
    return 0;
}
