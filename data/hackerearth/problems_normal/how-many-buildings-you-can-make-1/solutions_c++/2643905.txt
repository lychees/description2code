#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int ans=0;
    	int i=1;
    	while(1)
    	{
    		int t=(n/i)-(i-1);
    		if(t>0) {ans+=t; i++;}
    		else break;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
