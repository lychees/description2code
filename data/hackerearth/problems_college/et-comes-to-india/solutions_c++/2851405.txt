#include <iostream>
#include <math.h>

using namespace std;
int main()
{
	long long int t;
	long long int k;
	int n;
	
	long long int from,to,total_val,mid;
	char mid_char;
	
	cin>>t;
	int i,j;
	
	for(i=0;i<t;i++)
	{
		mid_char='G';
		from=1;
		cin>>n>>k;
		total_val=pow(2,n)-1;
		to=total_val;
		if(k<=total_val)
		{
		for(j=0;;j++)
		{
			if(k==from){ cout<<"G"<<endl; break;}
			if(k==to) { cout<<"C"<<endl; break;}
			mid=(from+to)/2;
			if(k==mid){cout<<mid_char<<endl; break;}
			else if(mid>k){ to=mid-1; mid_char='G';}
			else if(mid<k) {from=mid+1; mid_char='C';}
		}
		
		}
		else cout<<"-1"<<endl;
	}
	
	return (0);
}
