void msl(string &s,string &s2)
{
	if(s.size()>s2.size()) 
	{
		int n = s.size()-s2.size() ; 
		for(int i=0;i<n;i++)
		s2 ="0"+s2 ; 
	}
	else if(s2.size()>s.size())
	 {
	 	int n = s2.size()-s.size() ; 
		for(int i=0;i<n;i++)
		s ="0"+s ;
	 }

}

int carry(string s,string s2) 
{
	
	msl(s,s2) ; 
	int n=s.size() ; 
	 
	//cout<<s<<" "<<s2<<endl; 
	
	int count =0 ;
	for(int i=n-1;i>=0;i--) 
	{
		
		int m = s[i]-'0'+s2[i]-'0' ; 
		if(m>=10) 
		{
			s[i-1] += 1; 
			s[i]    = m-10 +'0' ;
			count++;  
		}
	}
	 //if(s[0]!='0') 
	 //count ++ ; 
	 
	return count ;
}
int main()

	for(int i1=0;i1<t;i1++)
