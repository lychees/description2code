#include <iostream>
using namespace std;

int main()
{
    int t; cin>>t;
    while(t--)
    {
    	int n; cin>>n;
    	int hash[10]={0};
    	for(int i=0;i<n;i++)
    	{
    		int tt; cin>>tt;
    		hash[tt%10]++;
    	}
    	int ans=0;
    	for(int i=0;i<10;i++)
    	{
    		if(hash[i]>1) ans+=hash[i]-1;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
