#include <cstdio>
#include <iostream>
#include <map>
#include <vector>
#include <utility>
#include <cstring>
using namespace std;
int main()
{
	int t;
	scanf(" %d",&t);
	while(t--)
	{
		int n;
		scanf(" %d",&n);
		int i,j;
		long long a[100011];
		long long b[100011];
		long long dp[100011];
		for(i=0;i<n;i++)
		{
			scanf(" %lld",&a[i]);
		}
		for(i=0,j=n-1;i<n;i++,j--)
		{
			b[j] = a[i];
		}
		if(n==1)
		{
			dp[0] = b[0];
		}
		else
		{
			dp[0] = b[0];
			dp[1] = dp[0] + b[1];
			for(i=2;i<n;i++)
			{
				long long p = b[i] - dp[i-1];
				long long q = b[i] + b[i-1] - dp[i-2];
				if(p>q)
				{
					dp[i] = p;
				}
				else
				{
					dp[i] = q;
				}
			}
		}
		if(dp[n-1]==0)
		{
			printf("DRAW\n");
		}
		else if(dp[n-1]>0)
		{
			printf("LOSE\n");
		}
		else if(dp[n-1]<0)
		{
			printf("WIN\n");
		}
	}
	return 0;
}
