#include <iostream>
using namespace std;

void checkPassword(char a[], char b[]) {
	int i = 0, j = 0;
	while (a[i]!='\0'){
		i++;
	}
	while (b[j]!='\0'){
		j++;
	}
	if ( j != 2*i){
		cout << "Impossible\n";
		return;
	}
	int aHash[256] = {0}, bHash[256] = {0};
	i = 0, j = 0;
	while(a[i]!='\0'){
		aHash[a[i]]++;
		i++;
	}
	while(b[j]!='\0'){
		bHash[b[j]]++;
		j++;
	}
	i = 0;
	while (a[i]!='\0') {
		if(bHash[a[i]] != 2*aHash[a[i]]){
			// cout << a[i] << " "<< bHash[a[i]] << " ";
			cout << "Impossible\n";
			return;
		}
		i++;
		
	}
	cout << "Possible\n";
	
}

int main()
{
	int t;
	char a[40];
	char b[80];
    cin >> t;
    while (t) {
    	cin >> a;
    	cin >> b;
    	checkPassword(a,b);
    	t--;
    }
    return 0;
}
