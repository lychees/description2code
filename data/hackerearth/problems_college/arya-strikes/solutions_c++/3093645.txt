#include <bits/stdc++.h>
using namespace std;
int gr[55][55];
int visited[55][55];
int n;
bool issafe(int x, int y)
{
	if(x >= n || y >= n) return false;
	return true;
}

void dfs(int x, int y)
{
	visited[x][y] = 1;
	if(issafe(x + 1, y) && gr[x + 1][y] == 1 && !visited[x + 1][y]) {
		dfs(x + 1, y);
	}
	if(issafe(x, y + 1) && gr[x][y + 1] == 1 && !visited[x][y + 1]) {
		dfs(x, y + 1);
	}
}

int main()
{
	int t;
	cin >> t;
	while(t--) {
		memset(visited, 0, sizeof(visited));
		cin >> n;
		for(int i = 0; i < n; i++)
			for(int j = 0; j < n; j++)
				cin >> gr[i][j];
		
		dfs(0, 0);
		if(visited[n - 1][n - 1] == 1) puts("POSSIBLE");
		else puts("NOT POSSIBLE");
	}
    return 0;
}
