#include<iostream>
using namespace std;
#include<string>

int main() {
	string str;
	long l, i;
	
	while(getline(cin, str)) {
		l = str.length();
		for(i=0;i<l;i++) {
			if(str[i]=='/' && str[i+1]=='/')
				break;
			else if(str[i]=='-' && str[i+1]=='>') {
				str.erase(i+1, 1);
				str[i] = '.';
			}
		}
		cout<<str<<endl;
	}
	
	return 0;
}
