#include <iostream>
using namespace std;
long long combo(int n, int r)
{
	if(r>n/2) r=n-r;
	long long ans=1;
	for(int i=1;i<=r;i++)
	{
		ans*=n-r+i;
		ans/=i;
	}
	return ans;
}
int main()
{
    int n,r;
    cin>>n>>r;
    if(n<r)
    {
    	cout<<"-1"<<endl;
    	return 0;
	}
	else
	{
		long long a=combo((n-1),(r-1));
		cout<<a%10000007<<endl;
	}
    return 0;
}
