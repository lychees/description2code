#include <iostream>
#include<algorithm>
using namespace std;
#define ll long long int
#define max 1000005
int main()
{
    ll m,n;
    cin>>m>>n;
    ll a[max]={0};
	ll k=0;
	ll p;
    for(ll i=0;i<m;i++)
	{
		cin>>p;
		a[p]++;
		if(p>k)
		k=p;
	}
	ll c=0;
	for(ll i=k;i>0;i--)
	{
		if(a[i]!=0)
		{
			if(n<=a[i])
			{
				c=c+n*i;
				goto loop;
			}
			else
			{
				c=c+a[i]*i;
				a[i-1]+=a[i];
				n-=a[i];
			}
		}
	}
   loop: cout<<c;
    return 0;
}
