#include<bits/stdc++.h>
using namespace std;
long long int arr[1000009];
int main()
{
      int n,q,x1,y1,x2,y2;
      long long int p;
      scanf("%d%d",&n,&q);
      scanf("%lld",&arr[0]);
      for(int i=1;i<n;i++)
      {
         scanf("%lld",&p);
         arr[i]=arr[i-1]+p;
      }
      while(q--)
      {
      scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
      if(x2>=y1&&y2>=x1)
      {
         int lim1=min(x1,y1);
         int lim2=max(x2,y2);
         long long int ans=arr[lim2];
         if(lim1!=0)
            ans-=arr[lim1-1];
         printf("%lld\n",ans);
      }
      else
      {
         long long int ans=arr[x2]+arr[y2];
         if(x1!=0)
            ans-=arr[x1-1];
         if(y1!=0)
            ans-=arr[y1-1];
         printf("%lld\n",ans);
      }
   }
}
