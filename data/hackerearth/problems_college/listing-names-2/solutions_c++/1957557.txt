#include <iostream>
#include <vector>
#include<string>
#include <algorithm>
using namespace std;

int main()
{
    vector<string> ip;
    int t,c;
    string temp;
    cin>>t;
    while(t-->0)
    {
    	cin>>temp;
    	ip.push_back(temp);
    }
    sort(ip.begin(),ip.end());
    for(int i=0; i<ip.size();i++)
    {
    	cout<<ip[i];
    	c=count(ip.begin(),ip.end(),ip[i]);
    	cout<<" "<<c<<endl;
    	i=i+c-1;
    }
    return 0;
}
