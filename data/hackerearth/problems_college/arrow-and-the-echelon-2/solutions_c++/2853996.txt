#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, five, three;
   	cin >> t;
   	while(t--)
   	{
   		five = 0;
   		three = 0;
   		scanf("%d", &n);
   		int five = (n / 3) * 3;
   		int three = n - five;
   		while((five % 3) || (three % 5))
   		{
   			if(five < 0)
   				break;
   			else
   			{
   				five -= 3;
   				three += 3;
   			}
   		}
   		
   		if(five < 0)
   			cout << "-1";
   		else
   		{
   			for(int i = 0; i < five; i++)
   				cout << "5";
   			for(int i = 0; i < three; i++)
   				cout << "3";
   		}
   		cout << "\n";
   	}
    return 0;
}
