#include <iostream>
#include<cstdlib>
#include<cstring>
using namespace std;

int main()
{
 string str;
 int i,count;
 int sum=0;
 count=1;
 cin>>str;
 int len=str.length();
 for(int i=0;i<len;i++)
 {
 if(str[i]==str[i+1])
 {
  count++;
 }
 else
 {
 sum+=count*(count+1)/2;
 count=1;
 }
 }
 cout<<sum;
    return 0;
}
