#include <iostream>
#include <cmath> 

int main()
{
	int t, n, m, d, targx, targy, c;
	int shootx[1000], shooty[1000];
	std::cin >> t;
	for (int i = 0; i < t; i++) {
		std::cin >> n >> m >> d;
		c = 0;
		for (int j = 0; j < n; j++) {
			std::cin >> shootx[j] >> shooty[j]; 
		}
		for (int j = 0; j < m; j++) {
			std::cin >> targx >> targy;
			for (int k = 0; k < n; k++) {
				if ((std::abs(targx - shootx[k]) + std::abs(targy - shooty[k])) <= d) {
					c++;
					break;
				}
			}
		}
		if (c >= m / 2) {
			std::cout << "YES";
		}
		else {
			std::cout << "NO";
		}
		std::cout << std::endl;
	}
}
