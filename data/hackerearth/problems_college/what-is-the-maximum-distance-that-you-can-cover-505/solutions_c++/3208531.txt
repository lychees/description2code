#include<iomanip>
#include<iostream>
#include<stdio.h>
#include<math.h>

using namespace std;
int main()
{
	int size,i;
	double totalfuel,max;
	int *fuel=new int[size];
	int *speed=new int[size];
	double *km=new double[size];
	cin>>size;
	for(i=0;i<size;i++)
	{
		cin>>speed[i];
	}
	for(i=0;i<size;i++)
		{
			cin>>fuel[i];
		}
	cin>>totalfuel;
	for(i=0;i<size;i++)
			{
		        cout<<fixed;
				km[i]=(totalfuel/fuel[i])*speed[i];
			}
	max=km[0];
	for(i=1;i<size;i++)
				{
			        if(km[i]>max)
			        	max=km[i];
				}
	max=floor(1000*max)/1000;
	cout<<setprecision(3)<<max;



}
