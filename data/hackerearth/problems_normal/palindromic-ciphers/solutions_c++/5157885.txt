#include <iostream>
using namespace std;
int palindrome(string s){
	int i,j;
	for(i=0,j=s.size()-1;i<(s.size()/2) && i < j ; i++,j--){
		if(s[i]!=s[j])
			return 0;
	}
	return 1;
}
void product(string s){
	long long int ans = 1;
	for(int i=0;i<s.size();i++){
		ans = ans * (s[i]-'a'+1);
	}
	cout<<ans<<endl;
}
int main()
{
	int t;
	string s;
	cin>>t;
	while(t>0){
		cin>>s;
		if(palindrome(s) == 1)
			cout<<"Palindrome"<<endl;
		else
			product(s);
		t--;
	}
    return 0;
}
