#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    long int i,j,N,c;
    vector<long int> v;
    cin>>T;
    while(T--)
    {
        cin>>N;
        c=0;
        for(i=0;i<N;i++)
        {
            cin>>j;
            v.push_back(j);
        }
        for(i=0;i<N;i++)
        {
            for(j=i;j<N;j++)
            {
                if(v[i]==v[j])
                    c++;
            }
        }
        cout<<c<<"\n";
        v.clear();
    }
}
