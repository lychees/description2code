#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        double PI=3.14159265358979323;
        double E=2.71828182845904523536;
        double p;
        scanf("%lf",&p);
        if(p==0||p==1)
        {
            printf("1\n");
            continue;
        }
        double t=((log(2*PI*p)/2.0+p*(log(p)-log(E)))/log(10.0))+1;
        printf("%lld\n",(long long int)t);
    }
}
