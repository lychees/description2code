#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		long long int n, m, a, i ;
		scanf("%lld %lld %lld",&n,&m,&a);
		long long int ans = 0;
		for( i = 1; i*i <= (a); i++ ) {
			if( a % i == 0 ) {
				long long  int  d2 = a / i;
				if( i < n && d2 < m ) {
					ans = ans + (n -i)*(m - d2);
				}
				if( i != d2 ) {
					if( i < m && d2 < n) {
						ans  = ans + (m - i)*(n - d2);
					}
				}
			}
		}
		printf("%lld\n",ans);
	}

}
