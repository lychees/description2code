#include <iostream>
using namespace std;

int main()
{	
	int t,j,k,flag,count;
	cin>>t;
	int n,i=1;long long int *arr;
    while(i++<=t)
    {
    	cin>>n;
    	arr=(long long int*)malloc(sizeof(long long int)*n);
    	for(j=0;j<n;j++)
    	cin>>arr[j];
    	flag=0;count=0;
    	for(j=0;j<=n-1;j++)
    	{	
    		count=0;
    		for(k=0;k<n;k++)
    		{
    			if(arr[j]==arr[k]&&j!=k)
    			count=2;
    			if(count==2)
    			break;		
    			 			
    		}
    		if(count==0)
    	   flag=arr[j];
    	}
    	cout<<flag<<"\n";
    }
    return 0;
}
