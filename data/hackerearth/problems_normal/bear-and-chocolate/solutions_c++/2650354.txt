#include<iostream>
using namespace std;

int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
    	
    	cin>>n;
    	int cnt=0,temp;
    	char a[n][n];
    	int rs[n],cs[n];
    	for(i=0;i<n;i++)
    	{
    		rs[i]=0;
    		cs[i]=0;
    		for(j=0;j<n;j++)
    		{
    			cin>>a[i][j];
    			if(a[i][j]=='#')
    				cnt++;
    			
    		}
    	}

    	if(cnt%2 != 0)
    	{
    		cout<<"NO"<<endl;
    		continue;
    	}
   		else
    	{
    			
    			
    			for(i=0;i<n;i++)
    			{
    				for(j=0;j<n;j++)
    				{
    					if(a[i][j]=='#')
    					{ 
    						rs[i]++;
    						cs[j]++;
    					}
    				}
    			
    			}
    		for(i=1;i < n-1;i++)
    		{
    			rs[i]=rs[i]+rs[i-1];
    			cs[i]=cs[i]+cs[i-1];
    		}
    				
    		int temp=1;
    		for(i=0;i<n;i++)
    		{
    			if((2*rs[i]==cnt) || (2*cs[i]==cnt))
    			{

					temp=0;
				//	break;
    			}
    		}
    	if(temp==0)
    		cout<<"YES"<<endl;
    	if(temp==1)
    	cout<<"NO"<<endl;
    	
    	}
    }
    return 0;
}
