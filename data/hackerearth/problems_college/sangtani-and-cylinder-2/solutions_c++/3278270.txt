#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define pi 3.141592653589793238462643383279502884197169399375105820974944592307816406286
int main()
{
    double radius1,radius2,r,a1,a2;
    ll t,v;
	cin>>t;
	while(t--)
	{
		cin>>v;
		radius1=(double)pow((2*v),0.333333333333333333333333333333333333);
		radius2=(double)pow((4*pi),0.33333333333333333333333333333333333);
		r=(double)radius1/(double)radius2;
		a1=(double)2*v/(double)r;
		a2=(double)2*pi*r*r;
		printf("%.9f\n",(a1+a2));
	}
	return 0;
}
