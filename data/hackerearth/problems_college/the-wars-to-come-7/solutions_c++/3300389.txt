#include <iostream>
using namespace std;
 
int main()
{
    int n,a[1000],s;
    cin>>n;
    
    int i,j;
    for(i=0;i<n;i++)
    cin>>a[i];
    cout<<endl;
    cin>>s;
    int t;
    for(i=0;i<n;i++)
    {
    	for(j=0;j<n-i-1;j++)
    	{
    	  if(a[j]<a[j+1])	
    	  {
    	  	t=a[j];
    	  	a[j]=a[j+1];
    	  	a[j+1]=t;
    	  }
    	}
    }
    int count=0,x=0;
    for(i=0;i<n;i++)
    {
    	count++;
    	x+=a[i];
    	if(x>s)
    	break;
    }
    cout<<"\n"<<count;
}
