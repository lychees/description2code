    #include <iostream>
    #include<bits/stdc++.h>
    using namespace std;
     int t,i,j,n,prime[1000001],dp[1000001],a,b;
    
    void solve()
    {
    	for(i=2;i<1000001;i++)
    	{
    		if(!prime[i])
    		{
    			for(j=i;j<1000001;j+=i)
    			{
    				prime[j]=1;
    				dp[j]=dp[j/i]+1;
    			}	
    		}
    	}
    }
    
    int main() 
    {
    
    	cin>>t;
    	solve();
    	for(i=2;i<1000001;i++)
    	dp[i]+=dp[i-1];
    	while(t--)
    	{
    		scanf("%d%d",&a,&b);
    		printf("%d\n",dp[b]-dp[a-1]);
    	}
    	
    	return 0;
    }
