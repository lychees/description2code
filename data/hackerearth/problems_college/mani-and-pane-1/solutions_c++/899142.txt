#include <bits/stdc++.h>
using namespace std;

long long _gcd(long long a, long long b)
{
	if (b == 0)
		return a;
	else
		return _gcd(b, a%b);
}

int main()
{
	map<long long, long long> gcds;
	long long div = (100 * 99 * 98);
	for (int i = 1; i <= 100; i++) {
		for (int j = 1; j <= 100; j++) {
			for (int k = 1; k <= 100; k++) {
				if (i != j && j != k && i != k) {
					long long gcd = _gcd(i, _gcd(j, k));
					gcds[gcd]++;
				}
			}
		}
	}

	/*for (map<long long, int>::iterator it = gcds.begin(); it != gcds.end(); it++) {
		cout << it->first << ", " << it->second << "\n";
	}
*/
	int t;
	scanf("%d", &t);
	while (t--) {
		long long x, z;
		scanf("%lld %lld", &x, &z);

		long long cnt = 0;
		for (map<long long, long long>::iterator it = gcds.begin(); it != gcds.end(); it++) {
			if (abs(x - z) % (it->first) == 0)
				cnt += it->second;
		}

		printf("%0.6lf\n", (double)cnt / div);
	}
}
