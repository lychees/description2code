#include <bits/stdc++.h>
using namespace std;
int freq[100010][26];
int main()
{
	int n,q,l,r;
	string str;
	cin >>n;
	cin >> str;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<26;j++)
			freq[i][j]=0;
	}


	freq[0][str[0]-'a']=1;

	for(int i=1;i<n;i++)
	{
		int val  = str[i]-'a';
		for(int j=0;j<26;j++)
			freq[i][j] = freq[i-1][j];
		freq[i][val]++;
	}

	
	cin >> q;

	int odd,val,len;

	for(int i=1;i<=q;i++)
	{
		cin >> l >> r;
		
		len = r-l+1;

		l = l-1;
		r = r-1;
		
		odd=0;

		for(int j=0;j<26;j++)
		{
			if(l==0)
				val = freq[r][j];
			else
				val = (freq[r][j] - freq[l-1][j]);

			if(val%2)
				odd++;
		}
		if(len == 1)
		{
			cout << "NO\n";
		}

		else if(len%2 == 0)
		{
			if(odd ==0)
				cout << "YES\n";
			
			else if(odd == 2)
				cout << "YES\n";
			else
				cout << "NO\n";
		}
		else
		{
			if(odd == 1)
				cout << "YES\n";
			else
				cout << "NO\n";
		}
		
	}


	return 0;
}
