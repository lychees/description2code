#include <bits/stdc++.h>
 
using namespace std;
 
const int N = 1e6 + 10;
 
int a[N];
 
int main() {
	int n, k;
	scanf("%d %d", &n, &k);
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
	}
	sort(a, a + n);
	int res = a[0];
	for (int i = 1; i < n; i++) {
		if (a[i] - a[i - 1] <= k) {
			res = a[i];
		} else {
			break;
		}
	}
	printf("%d\n", res);
	return 0;
}
