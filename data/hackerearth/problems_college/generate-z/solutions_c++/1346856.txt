#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    for(i=1;i<=n;i++)
    {
    	cout<<"*";
    }
    cout<<endl;
    for(i=1;i<n-1;i++)
    {
    	for(j=1;j<=n-1-i;j++)
    	{
    		cout<<" ";
    	}
    	cout<<"*"<<endl;
    }
    for(i=1;i<=n;i++)
    {
    	cout<<"*";
    }
    return 0;
}
