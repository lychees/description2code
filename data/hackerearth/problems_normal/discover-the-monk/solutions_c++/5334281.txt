#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,q,i,x;
	vector<int> v;
	cin>>n>>q;
	v.resize(n);
	for(i=0;i<n;i++)
	{
		cin>>v[i];
	}
	sort(v.begin(),v.end());
	while(q--)
	{
		cin>>x;
		if(binary_search(v.begin(),v.end(),x))
			cout<<"YES\n";
		else
			cout<<"NO\n";
	}
	return 0;
}
