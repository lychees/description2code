#include<bits/stdc++.h>
using namespace std;
char str[10010];
int main()
{
int t;
cin>>t;
while(t--)
{
cin>>str;
int ans = 0,l = strlen(str);
int cr[26] = {0};
int  j=0;
for(int i = 0; i < l;i++)
{
if(cr[str[i]-65] == 1)
{
if(ans < i-j)
{
ans = i-j;
}
while(str[j] != str[i])
{
    cr[str[j]-65]--;
    j++;
}
cr[str[j]-65]--;
j++;
cr[str[i]-65]++;
if(ans < i-j+1)
{
ans = i-j+1;
}
}
else
{
    cr[str[i]-65]++;
    if(ans < i-j + 1)
        ans = (i-j+1);

}
}
if(ans < (l-1 - j + 1))
    ans = l-j;
cout<<ans<<endl;
}
}
