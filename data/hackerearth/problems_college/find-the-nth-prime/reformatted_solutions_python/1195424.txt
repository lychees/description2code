import sys
f = sys.stdin

MAX = 100001
P = [0]*(MAX)


def sieve():

    P[0] = P[1] = 1
    for i in range(2, MAX):
        if P[i] == 0:
            j = 2*i
            while j < MAX:
                P[j] = 1
                j += i


sieve()
N = int(f.readline())
for i in range(0, MAX):
    if P[i] == 0:
        N -= 1
    if N == 0:
        print i
        break
