#include <cmath>
#include <cstdio>
#include <map>
#include <iostream>
#include <algorithm>
using namespace std;
int a[64];
int b[64];
int ans[64];
long long as,i,j,t,x,k;
int main(){
	int p;
	p=0;
	cin>>t;
	
	while(t--){
		p++;
		cin>>x>>k;
		for( i=0;i<64;i++)
		a[i]=b[i]=ans[i]=0;
		i=0;
		while(x){
			a[i++]=x%2;
			x=x/2;
		}
		i=0;
		while(k){
			b[i++]=k%2;
			k=k/2;
		}
		 j=0;
		for( i=0;i<=63;i++){
			if(a[i]==1)
				ans[i]=0;
			else{
				ans[i]=b[j++];
			}
		}
		as=0;
		for(i=63;i>=0;i--){
            as=as*2+ans[i];
		}

     cout<<"Case #"<<p<<": "<<as<<endl;

	}

}
