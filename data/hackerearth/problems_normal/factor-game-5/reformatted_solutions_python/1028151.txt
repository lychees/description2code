sieve = {i: 0 for i in range(1, 100001)}
for i in range(2, 100001):
    if sieve[i] == 0:
        for j in range(1, 1+100000/i):
            k = i*j
            while k % i == 0:
                k /= i
                sieve[i*j] += 1
n = input()
ans = 0
s = map(int, raw_input().split())
for i in s:
    ans ^= sieve[i]
if ans == 0:
    print 'BOB'
else:
    print 'ALICE'
