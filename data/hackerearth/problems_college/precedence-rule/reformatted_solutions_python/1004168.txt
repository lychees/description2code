for g in range(input()):
    x = raw_input("")
    starter = int(x[len(x)-1])
    check = 0
    op = ""
    for g in xrange(len(x)-2, -1, -1):
        if (check == 0):
            op = x[g]
        else:
            k = int(x[g])
            if (op == '+'):
                starter += k
            elif (op == '*'):
                starter *= k
            else:
                starter -= k
        check = 1-check
    print(starter)
