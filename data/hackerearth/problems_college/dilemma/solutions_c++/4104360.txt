#include <iostream>
using namespace std;
int coeff(int n)
{
    return (n+1)*(n+2)/2;
}
int main() {
    int T,A,B,C,N,x;
    int power[8],coef[8]={1,-1,-1,1,-1,1,1,-1};
    cin>>T;
    while(T--)
    {
        cin>>N>>A>>B>>C;
        power[0]=0;
        power[1]=A+1;
        power[2]=B+1;
        power[3]=A+B+2;
        power[4]=C+1;
        power[5]=A+C+2;
        power[6]=B+C+2;
        power[7]=A+B+C+3;
        long long sum=0;
        for(int i=0;i<=N;i++)
        {
            for(int j=0;j<8;j++)
            {
                x=(i-power[j]);
                if((x)>=0 && (x)<=N)
                sum+=coeff(x)*coef[j];
            }
        }
        cout<<sum<<endl;
        
    }
	// your code goes here
	return 0;
}
