#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a;
    long long s1=0,s2=0;
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	{
    		cin>>a;
    		if(i==j)
    		s1+=a;
    		if(i+j==(n-1))
    		s2+=a;
    	}
    }
    cout<<abs(s1-s2);
    return 0;
}
