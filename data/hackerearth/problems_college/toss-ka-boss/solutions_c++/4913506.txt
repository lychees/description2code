#include <iostream>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    for(;t--;)
    {
    	string s;
    	cin>>s;
    	long int c=0,count=1,i;
    	for(i=0;i<s.length();i++)
    	{
    		if(s[i]=='T')
    		count=1;
    		else if(s[i]=='H')
    		{
    			c=c+count;
    			if(s[i+1]=='H')
    			count++;
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
