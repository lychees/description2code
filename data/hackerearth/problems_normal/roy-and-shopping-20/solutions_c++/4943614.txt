#include<bits/stdc++.h>
#define  den 1000001
typedef long long int lu;
using namespace std;
int main()
{
	lu i,j,x,ans,t,y;
	int ar[1000001]={0};
	for(i=2;i<den;i++)
	{
		if(ar[i]==0)
		{
			x=2;
			for(j=i;j<den;)
			{
				if(ar[j]==0)
				ar[j]=i;
			j=i*x;
			x++;}
		}
	}
	
	scanf("%lld",&t);
	while(t--)
	{
		ans=0;
		scanf("%lld",&y);
		if(y==1)
		{
			cout<<1<<endl;
			continue;
		}
		ans=y-ar[y];
		printf("%lld\n",ans);
	}
	return 0;
}
