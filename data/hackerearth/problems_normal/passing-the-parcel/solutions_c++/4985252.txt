#include<string.h>
#include <iostream>
using namespace std;

int main()
{
    int n,a[1000],g=0,i,k,j=0,flag=0;
    cin>>n;
    string str;
    cin>>str;
    int l=str.length();
    for(i=0;i<n;i++)
    {
    	a[i]=i+1;
    }
    for(i=0;i<n;)
    {
    	if(n>1)
    	{
    		if(j==l)
    		{
    			j=0;
    		}
    		if(g==n)
    		{
    			g=0;
    		}
 		   	if(str[j]=='b')
    		{
    			flag=1;
    			for(k=g+1;k<n;k++)
					a[k-1]=a[k];
					n--;
					g--;
    		}
    		j++;
    		g++;
    	}
    	else
    	break;
    }
    	cout<<a[0];
    
}
