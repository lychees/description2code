#include <iostream>
using namespace std;

int main()
{
    int t,i,j;
    cin>>t;
    while(t--)
    {
    	
    	int n,cnt=0;
    	cin>>n;
    	for(i=1;i<n;i++)
    	{
    		for(j=i+1;j<=n;j++)
    		{
    			if((i*j)%(i+j)==0)
    				cnt++;
    		}
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
