import sys
for a in range(input()):
	N,M,D=map(int,raw_input().split())
	ax=[0]*N; ay=[0]*N
	for i in range(N):
		st=raw_input().split()
		ax[i]=int(st[0]); ay[i]=int(st[1])
	result=0;index=0	
	for i in xrange(M):
		x1,y1=map(int,sys.stdin.readline().split())
		index=0
		for index in range(N):
			res=abs(x1-ax[index])+abs(y1-ay[index])
			if(res<=D):
				result+=1; break;
	if(result>M/2):
		print 'YES'
	else:
		print 'NO'
