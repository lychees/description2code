#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t;

   scanf("%d",&t);

   while(t--)
    {
      int n,a,k,i;
      scanf("%d%d%d",&a,&k,&n);

      int arr[n+2];

      arr[1] = a;

      for(i = 2; i<=n; i++)
        arr[i] = arr[i-1]+k;

       cout<<arr[n]<<endl;

    }


return 0;
}
