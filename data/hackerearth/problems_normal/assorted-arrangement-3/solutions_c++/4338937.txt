#include <iostream>
using namespace std;

int main()
{
	int n,m;
	cin>>n;
	cin>>m;
	int a[n+1];
	int c[m+1];
	for(int i=1;i<=m;i++)
	cin>>c[i];
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	
	int sets[n+1];
	sets[0]=c[a[1]]-1;
	//cout<<sets[0]<<endl;
	for(int i=1;i<=n;i++){
		int count=0;
		int start=c[a[i]]-(sets[i-1]%c[a[i]]);
		for(int j=sets[i-1]+start;;j=j+c[a[i]]){
			bool isd=true;
			//if(j>sets[i-1]){
				for(int k=a[i]+1;k<=m;k++){
					if(k!=a[i]){
						if(j%c[k]==0){
							isd=false;
							break;
						}
					}
				}
				if(isd){
					sets[i]=j;
					count++;
				}
				if(count==1)
				break;
			//}
		}
	}
	
	int result=sets[1];
	for(int i=2;i<=n;i++){
		if(result<sets[i]){
			result=sets[i];
		}
	}
	cout<<result<<endl;

    return 0;
}
