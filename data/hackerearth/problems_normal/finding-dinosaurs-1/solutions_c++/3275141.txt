#include <bits/stdc++.h>
const long long  mod = 1.e9 + 7;
using namespace std;

int main()
{
    map<long long int,string> m;
    long long int n,q;
    int k,l,x,c=0;
    cin>>n>>k>>l>>q;
    string s;
    long long int  mul;
    while(mul < 5) mul = (rand() << 10) ^ rand();

    for(int i=0;i<n;i++){
        cin>>s;
        long long int h=0;
        for(int j=0;j<k;j++){
            cin>>x;
             h = (h * mul + x) % mod;
            }
            m[h]=s;
    }
    while(q--){
        long long int h=0;
        for (int i=0;i<k;i++){
            cin>>x;
             h = (h * mul + x) % mod;
        }
        if(m.count(h)){
            cout<<m[h]<<endl;
        }
         else{ 
         	cout << "You cant fool me :P" << endl;
		 }
   }
    //cout << "Hello world!" << endl;
    return 0;
}
