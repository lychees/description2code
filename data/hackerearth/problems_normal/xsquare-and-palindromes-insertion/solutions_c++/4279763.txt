#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int i,j,n,t,ans,f;
	cin>>t;
	
	cin.ignore(256,'\n');
	while(t--)
	{
		ans=0,f=0;
		char s[10000];
		int cnt[26]={0};
		cin>>s;
		for(i=0;s[i]!='\0';i++)
		cnt[s[i]-'a']++;
		for(i=0;i<26;i++)
		{
			if(cnt[i]%2!=0)
			{
			ans++;
			f=1;
			}
		}
		if(f==0)
		cout<<"0"<<endl;
		else
		cout<<ans-1<<endl;
		
	}
	return 0;
}
