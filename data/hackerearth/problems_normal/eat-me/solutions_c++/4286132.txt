#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
	long long n,k,i,j;
	cin>>n>>k;
	long long dp[n+1][k+1];
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=k;j++)
		{
			if(i==0||j==0)
			dp[i][j]=0;
		}
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=k;j++)
		{
			dp[i][j]=(((dp[i-1][j]%mod)+(dp[i][j-1]%mod)%mod)+1)%mod;
		}
	}
	cout<<dp[n][k]%mod<<endl;
    return 0;
}
