#include <iostream>
#include <cstring>

using namespace std;

const int M = 1000000007;

long long x;
int dp[60][3][2];

int go(int bi, int carry, int done, long long m) {
	if (bi == 60) return (done && (carry == 0)) ? 1 : 0;
	int& ret = dp[bi][carry][done];
	if (ret >= 0) return ret;
	int tgt = (m & x) ? 1 : 0;
	ret = 0;
	for (int a = 0; a < 2; a++) for (int b = 0; b < 2; b++) for (int c = 0; c < 2; c++) {
		int s = carry + a + b + c;
		if ((s & 1) == tgt) {
			ret += go(bi+1, s >> 1, done | (a & b & c), m << 1);
			if (ret >= M) ret -= M;
		}
	}
	return ret;
}

int main()
{
	int t; cin >> t;
	while (t--) {
		cin >> x;
		memset(dp, 0xff, sizeof(dp));
		cout << go(0, 0, 0, 1ll) << endl;
	}
    return 0;
}
