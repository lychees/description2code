#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	t--;
    	long long x;
    	cin>>x;
    	long long i;
    	for(i=1;i*i<=x;i++); // Incrementing number to reach the closest next square
    	if((i-1)*(i-1)==x)
    	{
    		cout<<"YES"<<'\n';
    		continue;
    	}
    	else
    	{
    	long long a=(i*i - x),b=(x - (i-1)*(i-1));
    	if(a<b)
    	{
    		cout<<'+'<<a<<'\n';
    		
    	}
    	else
    	cout<<'-'<<b<<'\n';
    	}
    }
 return 0;
}
