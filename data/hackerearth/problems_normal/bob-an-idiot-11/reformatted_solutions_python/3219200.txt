n = input()
orig = dict()
a = ord('a')
A = ord('A')
for i in range(26):
    c = chr(a+i)
    orig[c] = c
    c = chr(A+i)
    orig[c] = c
    orig[' '] = ' '
for j in range(n):
    x, y = raw_input().split()
    orig[x], orig[y] = orig[y], orig[x]
    x = x.swapcase()
    y = y.swapcase()
    orig[x], orig[y] = orig[y], orig[x]
mp = dict()
for c in orig:
    mp[orig[c]] = c
s = raw_input()
s = map(lambda x: mp[x], s)
print "".join(s)
