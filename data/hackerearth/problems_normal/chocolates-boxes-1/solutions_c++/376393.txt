#include <iostream>
using namespace std;

int main()
{
    int t,n,k,a[100009];
    cin>>t;
    while(t--)
    {
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>a[i];
    cin>>k;
    int count[109]={0};    
    long long res=0;
    for(int i=0;i<n;i++)
    {
    if(k-a[i]>=0 && k-a[i]<=100)
    res+=count[k-a[i]];
    count[a[i]]++;
    }
    cout<<res<<"\n";
    }
    return 0;
}
