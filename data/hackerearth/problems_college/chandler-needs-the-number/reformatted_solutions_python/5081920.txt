from sys import stdin


def gcd(a, b):
    while a % b:
        t = a % b
        a = b
        b = t
    return b


def lcm(a, b):
    return (a*b)/gcd(a, b)

b = stdin.readlines()
for i in b:
    c = map(int, i.split())
