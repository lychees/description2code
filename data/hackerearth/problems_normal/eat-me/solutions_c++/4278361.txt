long long int power(long long int a,long long int b)
{
	if(b==0)
	return 1;
	long long int temp=power(a,b/2)%M;
	if(b%2)
	return ((a%M*temp%M)%M*temp%M)%M;
	else
	return (temp%M*temp%M)%M;
}
