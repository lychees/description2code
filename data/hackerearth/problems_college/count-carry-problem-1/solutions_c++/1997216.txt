#include <iostream>
#include <cstdlib>
#include <string>
using namespace std;

int main(){
	long long T, a, b, ans, c;
	cin >> T;
	while(T--){
		ans = 0;
		c = 0;
		cin >> a >> b;
		while(a&&b){
			if(a%10 + b%10 + c>9){
				ans++;
				c = 1;
			}
			a = a/10;
			b = b/10;
		}
		while(a){
			if(a%10 + c>9){
				ans++;
				c = 1;
			}
			a = a/10;
		}
		while(b){
			if(b%10 + c>9){
				ans++;
				c=1;
			}
			b = b/10;
		}
		if(ans==0){
			cout << "No carry operation" << '\n';
		}
		else{
			cout << ans << ' ' <<  "carry operation";
			if(ans!=1){
				cout << 's' << '\n';
			}
			else{
				cout << '\n';
			}
		}
	}
	return 0;
}
