for _ in xrange(input()):
    r, k, v = map(int, raw_input().split())
    a = [r, k, v]
    a = sorted(a, reverse=True)
    if a[0] != 0 and a[1] != 0:
        if a[0] != a[1]:
            print a[1]*2 + 1
        else:
            print a[1]*2
    elif a[0] == 0:
        print 0
    elif a[1] == 0:
        print 1
