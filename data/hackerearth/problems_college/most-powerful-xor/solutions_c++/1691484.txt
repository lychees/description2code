#include <cstdio>
#include <algorithm>
using namespace std;

int l, r, a, b, ret;

int main () {
	int t;
	scanf("%d",&t);
	while(t--) {
    scanf("%d %d", &l, &r);
    ret = -1;
    for(a = l; a <= r; a++)
        for(b = a; b <= r; b++)
            ret = max(ret, a ^ b);
    printf("%d\n", ret);
	}
    return 0;
}
