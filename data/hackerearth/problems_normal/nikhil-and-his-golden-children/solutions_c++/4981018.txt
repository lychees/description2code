#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007



const int nax = 1e6 + 5;


using namespace std;
ll arr[nax];

ll modPow(ll a,ll b,ll MOD)
{
ll res=1;
 a%=MOD;
 for(;b;b>>=1)
    {
        if(b&1)res=res*a%MOD;
          a=a*a%MOD;
    }
 return res;
}
ll modInverse(ll a){
    return modPow(a,mod-2,mod);
    }
int main()
{

 ll t,k,m,ans;
 cin>>t;
 while(t--)
 {
     ll sum=0;
     cin>>k>>m;
     for(int i=0;i<k;i++)
     {


     cin>>arr[i];
     sum+=arr[i];
     }
     if(sum> m)
        cout<<0<<"\n";
     else
     {
         m-=sum;
         ll x=m+k-1;
         ll y=k-1;

         ll ans=1;
         for(int i=x;i>y;i--)
         ans=(ans*i)%mod;

         for(int i=1;i<x-y+1;i++)
         ans=(ans *modInverse(i))%mod;
         cout<<ans<<"\n";
     }

 }


}
