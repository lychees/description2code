#include <bits/stdc++.h>
#define ll long long int
#define re(t) int t; cin>>t; while(t--)
#define CLEAR(a) memset((a),0,sizeof(a))
#define mod 1000000007
using namespace std;

ll DP[10004];

int main()
{
    ll n;
    int p;
    re(t){
        cin>>n>>p;
        CLEAR(DP);
        DP[0]=1;
        for(int i=1;i<=n;i++){
            for(int j=i-1;(j>=0) && (i-j<=p);j--){
                DP[i]+=DP[j];
                DP[i]=DP[i]%mod;
            }
        }
        cout<<DP[n]<<endl;
    }
    //cout << "Hello world!" << endl;
    return 0;
}
