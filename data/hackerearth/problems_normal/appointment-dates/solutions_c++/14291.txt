# include <iostream>
#include <cstdio>

using namespace std;

int leapdaystill(int y) {
    int start = 2000;
    int ct = 0;
    while (start < y) {
        if ((start%4 == 0 && start%100!=0) || (start%400 ==0)) ct++;
        start++;
    }
    return ct;
}

int isleap(int start) {
    if ((start%4 == 0 && start%100!=0) || (start%400 ==0))
    return 1;
    return 0;
}
