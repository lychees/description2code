#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;

int cases, n;
ll P[500010];

ll solve(){
	reverse(P, P+n);
	ll last = P[0];
	ll ans = 0;
	for(int i = 0; i < n; i++){
		if(P[i] > last)
			last = P[i];
		else
			ans += (last-P[i]);
	}
	return ans;
}

int main() {
	scanf("%d", &cases);
	while(cases--){
		scanf("%d", &n);
		for(int i = 0; i < n; i++)
			scanf("%lld", &P[i]);
		printf("%lld\n", solve());
	}
    return 0;
}
