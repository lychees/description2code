#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;

int main()
{
    int N,K;
    int price[100];
    scanf("%d %d\n",&N,&K);
    for(int i = 0; i < N; i++)
    	scanf("%d",&price[i]);
    sort(price,price+N);
    long sum = 0;
    int multiplier = 1;
    int counter = 0;
    for(int i = N-1; i>=0;i--){
    	counter++;
    	sum += price[i]*multiplier;
    	if(counter == K){
    		counter = 0;
    		multiplier++;
    	}
    }
    printf("%ld\n",sum);
    return 0;
}
