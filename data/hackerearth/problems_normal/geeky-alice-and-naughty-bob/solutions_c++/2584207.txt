#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int fac[7]={0,1,2,6,6,3};
    while(t--)
    {
        int a,b;
        cin>>a>>b;
        long long  int sum=0;

        if(a<6&&b<6)
        {
            for(int i=a;i<=b;i++)
                sum+=fac[i];
        }
        else if(a<6&&b>=6)
        {
            while(a<6)
            {
                sum+=fac[a];
                a++;
            }
            sum+=(long long)(b-a+1)*9;
        }
        else
        {
            sum+=(long long)(b-a+1)*9;
        }
        cout<<sum<<endl;
    }
}
