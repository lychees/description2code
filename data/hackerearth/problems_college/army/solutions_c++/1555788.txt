#include <iostream>
#include <map>
using namespace std;

int main()
{
	long long n,count;
	int q,i,j,m,l,t;
	cin>>n;
	cin>>q;
	for(i=0;i<q;i++)
	{
		count=0;
		int k[10]={0};
		cin>>m;
		for(j=0;j<m;j++)
		{
			cin>>l;
			k[l]=1;
		}

    for(t=1; t<=n; t++)
    {
        int temp = t;
        while(temp > 0)
        {
            if(k[temp%10] == 1)
            {
                count++;
                break;
            }
            temp=temp/10;
        }
    }
    cout<<count<<endl;
	}
    return 0;
}
