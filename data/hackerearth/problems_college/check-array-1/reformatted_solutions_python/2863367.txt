t = int(raw_input())
while t > 0:
    n = input()
    a = map(int, raw_input().split())
    b = []
    c = []
    s = 0
    r = 0
    m = 0
    for i in range(0, len(a)):
        b.append(s)
        s += a[i]
    a.reverse()
    for i in range(0, len(a)):
        c.append(r)
        r += a[i]
    c.reverse()
    for i in range(0, len(c)):
        if c[i] == b[i]:
            m += 1

    if m >= 1:
        print 'YES'
    else:
        print 'NO'
    t -= 1
