#include <iostream>
using namespace std;

long int match(long long int n){
	long int count=0;
	if(n==0)
	return 0;
	else
	while(n!=1){
		if(n%2==0)
		{
			count+=(n/2);
			n=n/2;
		}
		else{
			count+=(n/2);
			n=(n/2)+1;
		}
	}
	return count;
}

int main()
{
    std::ios::sync_with_stdio(false);
    int t;
    cin>>t;
    long long int n;
    while(t--){
    	cin>>n;
    	cout <<match(n)<< endl;
    }
    return 0;
}
