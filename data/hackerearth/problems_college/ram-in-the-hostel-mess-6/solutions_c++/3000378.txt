#include<iostream>
using namespace std;

long factorial(unsigned long long int n) {
   int c;
   long result = 1;

   for (c = 1; c <= n; c++)
      result = result*c;

   return result;
}
long find_ncr(unsigned long long int n, unsigned long long int r) {
   long result;

   result = factorial(n)/(factorial(r)*factorial(n-r));

   return result;
}


int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        unsigned long long int N,i,ans=0,mid,final_ans;
        cin>>N;
        if(N%2==0)
            mid = N/2;
        else
            mid = N/2 + 1;
        for(i=0;i<=mid;i++)
        {
            ans = ans + find_ncr(N-i+1,i);
        }
        final_ans=ans*ans;
        cout<<final_ans<<endl;
    }
}
