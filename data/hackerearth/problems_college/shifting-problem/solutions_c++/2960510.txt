#include<iostream>
using namespace std;

long solve(int n){
	if (n == 1 || n == 2 || n == 3) return n;
	long a = 1;
	long b  = 2;
	long temp = 0;
	int count = 2;
	while (count++ < n){
		temp = b;
		b += a;
		a = temp;
		
	}
	return b;
}

int main(void){
	int t,n;
	cin >> t;
	while (t-- > 0){
		cin >> n;
		cout << solve(n) << endl;
	}
}
