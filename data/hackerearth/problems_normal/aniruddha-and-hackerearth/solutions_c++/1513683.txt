using namespace std;
#define MOD 1000000007
int fib[1000006];
void precalc()
{
	int i;
	fib[0] = fib[1] = 1;
	FOR(i, 2, 1000005)
		fib[i] = (fib[i-1] + fib[i-2])%MOD;
	FOR(i,0,1000000)
		fib[i] = (fib[i] + fib[i+1])%MOD;
	FOR(i,1,1000000)
		fib[i] = (fib[i] + fib[i-1])%MOD;
}

    	assert(n>=1 && n<=1000000);
    	cout<<fib[n-1]<<endl;
    }
    return 0;
}
