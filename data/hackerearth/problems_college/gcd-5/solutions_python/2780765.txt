t = int(raw_input())
while t>0 :
	n = int(raw_input())
	l = raw_input()
	l = l.split()
	count = 0
	for i in range(0,len(l) - 1):
		if int(l[i]) > int(l[i+1]) :
			print "NO"
			break
		count = count + 1
	if count == n - 1 :
		print "YES"
	t = t - 1
