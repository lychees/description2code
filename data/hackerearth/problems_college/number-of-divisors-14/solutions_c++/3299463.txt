#include<stdio.h>
#include<stdlib.h>
#define MAX 1000001
long long int sieve[MAX],prime[MAX];
void create_sieve(){
 long long int i,j,c=0;;
    for(i=0;i<MAX;i++)
        sieve[i]=0;
    for(i=2;i*i<MAX;i++)
        if(sieve[i]==0)
        {   prime[c++]=i;
            for(j=i*i;j<MAX;j+=i)
                if(!sieve[j])
                    sieve[j]=i;
        }
    for(;i<MAX;i++)
        if(sieve[i]==0)
            prime[c++]=i;
}

long long int lprime(long long int n){ long long int i;
    if(n<MAX){
    if(sieve[n]==0)
        return n;
    return sieve[n];
    }
    else{
        for(i=0;i<MAX;i++)
           if(n%prime[i]==0)
                return prime[i];

    }

}
int main(){long long int t,n,p,c,factor[2],temp;
    create_sieve();
    scanf("%lld",&t);
    while(t--)
    {   c=1;temp=1;
        scanf("%lld",&n);
        p=n;
        while(p!=1)
        {
            factor[1]=lprime(p);
            if(p==n){factor[0]=factor[1]; temp++; }
            else if(factor[1]==factor[0]) temp++;
            else {c=c*temp;
            temp=2;factor[0]=factor[1];
            }
            p/=factor[1];

        }
        c=c*temp;
        printf("%lld\n",c);
    }
    return 0;
}
