#include<bits/stdc++.h>
using namespace std;
long long int hcf(long long int a,long long int b)
{
	if(a==0)
	return b;
	return hcf(b%a,a);
}
int main()
{
	int d;
	long long int x,y,i,k,gcd,ct;
	cin>>d;
	while(d--)
	{
		ct=0;
		cin>>x>>y>>k;
		gcd=hcf(x,y);
		for(i=1;i<=(gcd);i++)
		{
			if(gcd%i==0)
			ct++;
			if(ct==k)
			break;
	
		}
		if(ct<k)
		cout<<"No crymeth today"<<endl;
		else
		cout<<gcd/i<<endl;
	}
	return 0;
}
