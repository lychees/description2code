#include <iostream>
#include <string>
using namespace std;
int main()
{

    int x,y;
    cin >> x >> y;
    string a;
    cin >> a;
    a.resize(x);
    while(y--)
    {

        int q;
        char w;
        cin >>q>> w;
        a[q-1]=w;
        int z=0;
        int t=0;
        for(int d=0;d<a.size();d++)
        {

            if(a[d]=='.')
            {

                z++;

            }
            else
            {
                if(z!=0)
                {
                        t+=z-1;
                }


                z=0;
            }
        }
        if(z!=0)
        {

            t+=z-1;
        }
        cout << t << endl;
    }
}
