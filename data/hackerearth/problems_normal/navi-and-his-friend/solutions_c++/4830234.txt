#include<bits/stdc++.h>
using namespace std ;
long long p[20] , w[20] , max_weight ;
int n , max_items ;
long long getans ( int index , int mask ) {
    long long ans = -1 ;
    for ( int mask = 0 ; mask < (1<<n) ; mask++ ) {
        long long weight = 0 , price = 0 ;
        int cnt = 0 ;
        for ( int i = 0 ; i < n ; i++ ) {
            if ( (1<<i)&mask ) {
                cnt++ ;
                price += p[i] ;
                weight += w[i] ;
            }
        }
        if ( cnt <= max_items && weight <= max_weight )
            ans = max ( ans , price ) ;
    }
    return ans ;
}
int main () {
    int d ;
    scanf ( "%d" , &d ) ;
    for ( int day = 1 ; day <= d ; day++ ) {
        scanf ( "%d" , &n ) ;
        for ( int i = 0 ; i < n ; i++ ) {
            scanf ( "%lld %lld" , &p[i] , &w[i] ) ;
        }
        scanf ( "%lld %d" , &max_weight , &max_items ) ;
        printf ( "For Day #%d:\n" , day ) ;
        if ( getans ( 0 , 0 ) == 0 )
        	printf ( "-1\n" ) ;
        else 
        
        	printf ( "%lld\n" , getans ( 0 , 0 ) ) ;
    }
    return 0 ;
}
