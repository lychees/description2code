#include <iostream>
using namespace std;

int main()
{
    int t,n,k,val;
    cin>>t;
    while(t--)
    {
    	int count=0;
    	cin>>n>>k;
    	int z=0,o=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>val;
    		if(val)
    			o++;
    		else
    			z++;
    	}
    	count=o;
    	int m=k-z;
    	if(z>=k)
    		count+=k;
    	else
    	{
    	    if(m%2 == 0)
			{
				count=n;
			}
			else
			{
				count=n-1;
			}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
