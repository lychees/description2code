void mul(ll a[2][2], ll b[2][2], ll mod)
{
    ll x = (a[0][0] * b[0][0] %mod + a[0][1] * b[1][0] % mod ) %mod;
    ll y = (a[0][0] * b[0][1] %mod + a[0][1] * b[1][1] % mod ) %mod;
    ll z = (a[1][0] * b[0][0] %mod + a[1][1] * b[1][0] % mod ) %mod;
    ll u = (a[1][0] * b[0][1] %mod + a[1][1] * b[1][1] % mod ) %mod;
    a[0][0] = x;
    a[0][1] = y;
    a[1][0] = z;
    a[1][1] = u;
}  
void mat_mul(ll mat[2][2], ll n,ll mod)
{
    if(n == 0 || n == 1) {
        return;
    }
    mat_mul(mat,n/2,mod);
    ll m[2][2] = {{1,1},{1,0}};
    mul(mat,mat,mod);
    if(n % 2 != 0) {
        mul(mat,m,mod);
    }
}
