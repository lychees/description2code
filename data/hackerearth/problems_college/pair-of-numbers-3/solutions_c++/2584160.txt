#include <bits/stdc++.h>

using namespace std;
 
int main()
{
    int tc,N,i,j,k,a;
    cin>>tc;
    for(i=0;i<tc;i++)
    {
     a=0;
     cin>>N;
     for(j=1;j<N;j++)
     {
      for(k=j+1;k<=N;k++)
    	{
    	 if(((j*k)%(j+k))==0)
    	  {
    	   a++;
    	  }
    	 }
      }
    	cout<<a<<"\n";
    	
    }
    return 0;
}
