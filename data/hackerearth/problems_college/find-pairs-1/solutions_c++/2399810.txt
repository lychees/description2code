#include <iostream>
using namespace std;

int main()
{
   
   int t,i,j;
   long int count,a[1000],n;
   cin>>t;
    while(t--)
    {
        count=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	 cin>>a[i];
    	 for(i=0;i<n;i++)
    	  {
    	  	for(j=i;j<n;j++)
    	  	{
    	  		if(a[i]==a[j])
    	  		count++;
    	  	}
    	  }
    	  cout<<count<<endl;
    }
return 0;
}
