from fractions import gcd

def main():
    T = int(raw_input())
    while T>0:
        N,X = map(int, raw_input().strip().split(" "))
	num_list = map(int,raw_input().strip().split(" "))
        valid = []
	for i in num_list:
            if gcd(i,X)==1:
		valid.append(str(i))
	if valid == []:
            print -1
	else :
            print " ".join(valid)

	T = T-1
	
main()
