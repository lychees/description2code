#include<bits/stdc++.h>
#define ll long long
using namespace std;
//g++ -std=c++11 your_file.cpp

ll a[1000006],freq[(1<<20)+2];
ll dp[(1<<20)+2][21];

int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);

	ll t,n,i;
	cin >> t;
	while(t--){
		cin >> n;
		for(i=0;i<(1<<20);i++) freq[i]=0;
		for(i=0;i<n;i++){
			cin >> a[i];
			freq[a[i]]++;
		}
		for(ll mask=0;mask<(1<<20);mask++){
			dp[mask][0] = freq[mask];
			if(mask&1) dp[mask][0] += dp[mask^1][0];
			for(i=1;i<=20;i++){
				dp[mask][i] = dp[mask][i-1];
				if(mask&(1<<i)){
					dp[mask][i] += dp[mask^(1<<i)][i-1];
				}
			}
		}
		ll ans=0;
		for(i=0;i<n;i++){
			ans += dp[((1<<20)-1)^a[i]][20];
			freq[a[i]]=0;
		}
		cout << ans << endl;
	}

	return 0;
}
