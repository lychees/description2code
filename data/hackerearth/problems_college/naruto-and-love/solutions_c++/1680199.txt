#include<bits/stdc++.h>
using namespace std;
long long A[150000+2],B[150000 +2];
int n,m;
bool satisfy(long long  value)
{ int pos1=0,pos2=0;
    while(pos1<n&&pos2<m)
     {
         if(abs(A[pos1]-B[pos2])<=value)
         {
             pos1++;
             pos2++;
         }
         else
            pos2++;
     }
     if(pos1==n)
        return true;
     else
        return false;

}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    { long long sum=-1,lo,high,midi;
        scanf("%d %d",&n,&m);
        for(int i=0;i<n;i++)
            scanf("%lld",&A[i]);
        for(int i=0;i<m;i++)
            scanf("%lld",&B[i]);
            sort(A,A+n);
            sort(B,B+m);
        lo=0;
        high=1e18;
        while(lo<high)
        { midi=(lo+high)/2;
            if(satisfy(midi))
                high=midi;
           else
            lo=midi+1;

        }
        printf("%lld\n",lo);

    }
    return 0;

}
