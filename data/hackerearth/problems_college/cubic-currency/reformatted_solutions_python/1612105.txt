t = int(raw_input())
n = 10009
li = [(i+1)*(i+1)*(i+1) for i in range(0, 21)]

ans = [[0 for j in range(0, 22)] for i in range(0, n + 1)]

for i in range(0, 21):
    ans[0][i] = 1
for i in range(1, n+1):
    for j in range(0, 21):
        if i - li[j] >= 0:
            ans[i][j] = ans[i-li[j]][j]
        if j >= 1:
            ans[i][j] += ans[i][j-1]

while t > 0:
    t -= 1
    x = int(raw_input())
    print ans[x][20]
