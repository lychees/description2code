#include <cstdio>
#include <algorithm>
#define N 201
int main()
{
	int sno,i,m,s,c,stalls[N] = {0},min,empty[N],cen,si,ei,max;
	scanf("%d %d %d", &m, &s, &c);
	min = s;
	cen = 0;
	for(i=0;i<c;i++) {
		scanf("%d", &sno);
		stalls[sno] = 1;
	}
	// left trim
	for(i=1;i<=s && stalls[i] == 0;i++)
		min--;
	si=i;
	// right trim
	for(i=s;i>0 && stalls[i] == 0;i--)
		min--;
	ei=i;
	for(i=si;i<=ei;i++) {
		if (stalls[i] == 0) {
			empty[cen] = 0;
			for (;i<=ei && stalls[i]==0;i++) {
				empty[cen]++;
			}
			cen++;
		}
	}
	sort(empty, empty+cen, std::greater<int>());
	for(i=0;i<m-1 && i<cen;i++) {
		min -= empty[i];
	}
	printf("%d", min);
    return 0;
}
