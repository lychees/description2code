#include <iostream>
using namespace std;
typedef long long int lli;
int main()
{
	int n;
	cin>>n;
	int arr[n];
	lli sum = 0;
	int min = -1;
	for(int i=0; i<n; i++){
		cin>>arr[i];
		sum += arr[i];
	}
	if(sum%2 == 0){
		for(int i=0; i<n; i++){
			if(arr[i]%2 == 1){
				if(min == -1){
					min = arr[i];
				} else if(arr[i]<min){
					min = arr[i];
				}
			}
		}
	}
	
	if(sum%2 == 1){
		cout<<sum;
	} else if(min == -1) {
		cout<<":(";
	} else {
		cout<<sum-min;
	}
	
    return 0;
}
