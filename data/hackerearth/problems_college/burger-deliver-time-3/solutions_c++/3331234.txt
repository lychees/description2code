#include <iostream>
#include<algorithm>
using namespace std;

struct day{
	int in,sum;
};

int cmp(struct day d1,struct day d2)
{
	if(d1.sum<d2.sum)
		return 1;
	else if(d1.sum==d2.sum && d1.in<d2.in)
		return 1;
	return 0;	
}
int main()
{
    int n,i,d,t;
    cin>>n;
    struct day arr[n];
    for(i=0;i<n;i++)
    {
    	scanf("%d %d",&d,&t);
    	arr[i].in=i+1;
    	arr[i].sum=d+t;
    	
    }
    
    sort(arr,arr+n,cmp);
    for(i=0;i<n;i++)
    cout<<arr[i].in<<" ";
    
    return 0;
}
