#include <iostream>
using namespace std;

bool prime(int x){
	
	if(x==2){
		return true;
	}
if(x%2==0){
	return false;
	
}	
for(int i=3;i<x/2;i=i+2){
	if(x%i==0){
		return false;
	}
}

return true;	
}



int main()
{
    int cases;
    cin>>cases; 
    
   while(cases--){
    	int frscr=0,ans1=0,ans2=0,flag=0;
    	
    	cin>>frscr;
    	
    	int x;
    	if(frscr%2==0){
    		x=(frscr/2)-1;
    	}
    	else{
    		x=frscr-(frscr/2);
    	}
    //	x=(frscr/2)-1;
    	for(;x>=2;x--){
    		
    		bool pr = prime(x);
    		
    		bool pr2 = prime(frscr-x);
    			
    		  if(pr==true && pr2==true){
    			
    			ans1=x;
    			ans2=frscr-x;
    			flag=1;
    			
    			break;}
    		
    	}
    	
    	if(flag==0){
    		cout<<"-1 -1";
    	}
    	else{
    	 if(ans1>ans2){
    	 	cout<<ans2<<" "<<ans1;
    	 }
    		else{
    			cout<<ans1<<" "<<ans2;
    	}
    		
    	}
    	
    	cout<<endl;
    	
    	
    	
    	
    	
    }
    return 0;
}
