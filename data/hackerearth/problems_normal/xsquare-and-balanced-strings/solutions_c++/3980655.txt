#include <iostream>
using namespace std;
int main()
{
//    printf("Hello World!\n");
int t;
cin>>t;
while(t)
{
	string s;
	int hash[26];
	for(int i=0;i<26;i++)hash[i]=0;
	cin>>s;
	int n=s.length();
	for(int i=0;i<n;i++)
	{
		hash[s[i]-'a']++;
	}
//	for(int i=0;i<26;i++)
//	cout<<hash[i]<<" ";
	int flag=0;
	for(int i=0;i<26;i++)
	{
		if(hash[i]%2){flag=1;break;}
	}
	if(flag)
	cout<<"-1\n";
	else
	cout<<"1\n";
	t--;
}
    return 0;
}
