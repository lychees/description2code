#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,dig,i,x;
    cin>>t;
    while(t--)
    {
    	x=0;
    	cin>>dig;
    	for(x=0;x<=dig;x++)
    	{
    		if((dig-x)%3==0&&x%5==0)
    		break;
    	}
    	if(x==dig+1)
    	{
    	cout<<"-1"<<endl;
    	continue;
    	}
    	for(i=0;i<dig-x;i++)
    	cout<<"5";
    	for(i=0;i<x;i++)
    	cout<<"3";
    	cout<<endl;
    }
    
    return 0;
}
