#include <iostream>
#include <algorithm>
using namespace std;
int abc[100000];
int main()
{
    int t,n,c,a,i;
    cin>>t;
    while(t--){
    	cin>>n;
    	for(i=0;i<n;i++)
    		cin>>abc[i];
    	sort(abc,abc+n);
    	c=1;a=1;
    	for(i=0;i<n-1;i++){
    		if(abc[i]==abc[i+1])
    			c++;
    		else{
    			if (c > a )
    			a=c;
    			c=1;
    		}
    	}
    	if (c > a )
    			a=c;
    	cout<<a<<endl;
    }
}
