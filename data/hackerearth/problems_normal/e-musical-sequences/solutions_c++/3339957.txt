#include <iostream>
#include <stdio.h>
#include <stdint.h>
#include <vector>

using namespace::std;

int main()
{
	int T;
	scanf("%d", &T);

	for (int t = 0; t < T; t++)
	{
		int n, m;
		scanf("%d", &n);
		scanf("%d", &m);

		int64_t z;
		scanf("%lld", &z);

		vector<int> a(n + 1, 0);
		for (int i = 0; i < n; i++)
		{
			int x;
			scanf("%d", &x);

			a[i] = x;
		}

		for (int i = 1; i <= n; i++)
		{
			if (i & 1)
			{
				a[n] += a[n - i];
				if (a[n] >= m)
					a[n] -= m;
			}
			else
			{
				if (a[n] >= a[n - i])
					a[n] -= a[n - i];
				else
					a[n] += m - a[n - i];
			}
		}

		int64_t q = z / (int64_t) (n + 1);
		int64_t r = z % (int64_t) (n + 1);

		if ((n % 2 == 0) && (q % 2 == 1))
		{
			int ans = (a[r] == 0 ? 0 : (m - a[r]));
			printf("%d\n", ans);
		}
		else
			printf("%d\n", a[r]);
	}
	
	return 0;
}
