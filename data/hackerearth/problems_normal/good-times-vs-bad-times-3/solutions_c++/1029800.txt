pair <int,int> arr[60][60][60] ;
 
int gcd(int m, int n){
	return n == 0 ? m : gcd(n,m%n);
}
 
int gcdtriplet(int m, int n, int o){
	if(arr[m][n][o].second != -1){
		return arr[m][n][o].first;
	}
	return gcd(m,gcd(n,o));
}

}
