ll MOD = 1000000007;
ll _MOD = 1000000009;
double EPS = 1e-10;
int INF = INT_MAX / 2;

ll extgcd(ll a, ll b, ll& x, ll& y) {
	if (b == 0) {
		x = (a >= 0) ? 1 : -1;
		y = 0;
		return abs(a);
	}
	else {
		ll res = extgcd(b, a % b, y, x);
		y -= (a / b) * x;
		return res;
	}
}

ll mod_inverse(ll a, ll m) {
	ll x, y;
	extgcd(a, m, x, y);
	return (m + x % m) % m;
}
 
ll c(ll n, ll k) {
	ll res = 1;
	for (int i = 0; i < k; i++)
		res = (res * ((n - i) % 1000000007)) % 1000000007;
	for (int i = 1; i <= k; i++)
		res = (res * mod_inverse(i, 1000000007)) % 1000000007;
	return res;
}

int main() {
	int T; cin >> T;
	while (T--) {
		int N, Q; cin >> N >> Q;
		vector<i_i> v;
		v.push_back(i_i(0, 0));
		v.push_back(i_i(N - 1, N - 1));
		while (Q--) {
			int a, b; cin >> a >> b;
			v.push_back(i_i(a, b));
		}
		sort(v.begin(), v.end());
		v.erase(unique(v.begin(), v.end()), v.end());
		ll ans = 1;
		for (int i = 0; i + 1 < v.size(); i++) {
			int dx = v[i + 1].first - v[i].first;
			int dy = v[i + 1].second - v[i].second;
			ans = ans * c(dx + dy, dx) % MOD;
		}
		cout << ans << endl;
	}
}
