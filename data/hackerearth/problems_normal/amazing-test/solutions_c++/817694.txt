#include <iostream>
#include<queue>
using namespace std;

int main()
{
    int n,i,t,g,j;
	cin>>t;
	while(t--){
		priority_queue<int>a;
		cin>>n>>g;
		for(i=0;i<n;i++){
			cin>>j;
			a.push(j);
		}
		a.push(0);
		i=0;
		int tim=0;
		int val,val1;
		while(1){
			val=a.top();
			if(val==0)
			break;
			a.pop();
			val1=a.top();
			a.pop();
			if(val1==0){
				tim+=val;
				break;
			}
			tim+=val1;
			val-=val1;
			a.push(val);
		}
		if(tim<=g)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
    return 0;
}
