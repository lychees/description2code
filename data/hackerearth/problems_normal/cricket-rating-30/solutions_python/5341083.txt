n=int(raw_input())
def max_subarray(A):
    max_end=max_so_far=A[0]
##    print max_end,max_so_far
    for x in A[1:]:
##        print x
        max_end=max(x,max_end+x)
##        print 'xxx'+str(max_end)
        max_so_far=max(max_so_far,max_end)
##        print str(max_so_far)+'zzz'
    return max_so_far


if n>0:
    runs=map(int,raw_input().split())
    if max_subarray(runs)<0:
        print '0'
    else:
        print max_subarray(runs)
else:
    print '0'
