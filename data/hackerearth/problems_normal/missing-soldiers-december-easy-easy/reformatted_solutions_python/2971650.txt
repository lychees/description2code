l = []
for _ in range(int(raw_input())):
    x, y, d = map(int, raw_input().split())
    l.append((x, d))
l.sort(key=lambda x: x[0])
start, d = l.pop(0)
s = d+1
last = start + d
for start, d in l:
    if start > last:
        s += d+1
        last = start + d
    elif start <= last and start+d > last:
        s += start + d - last
        last = start + d
print s
