#include <iostream>
using namespace std;
struct point
{
	int x,y;
};
int abs(int a)
{
	if(a<0)
		return -1*a;
	return a;
}
int main()
{
	int t,n,m,d,i,j,count,dist,hit[1005];
	point pos[1005],target[1005];
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>n>>m>>d;
		for(i=0;i<n;i++)
			cin>>pos[i].x>>pos[i].y;
		for(i=0;i<m;i++)
		{
			cin>>target[i].x>>target[i].y;
			hit[i]=0;
		}
		for(i=0;i<m;i++)
		{
			if(hit[i]==0)
			{
				for(j=0;j<n;j++)
				{
					dist=abs(pos[j].x-target[i].x)+abs(pos[j].y-target[i].y);
					if(dist<=d)
					{
						hit[i]=0;
						count++;
						break;
					}
				}
			}
		}
			if(count>m/2)
				cout<<"YES"<<endl;
			else
				cout<<"NO"<<endl;
	}
    return 0;
}
