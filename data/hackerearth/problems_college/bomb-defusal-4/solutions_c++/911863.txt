#include <iostream>
#include <math.h>
#include <cstdlib>
#include <cstdio>
using namespace std;

#define gc getchar_unlocked
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
    int t,x,y,a,b,p,st,fi;
    scanint(t);
    //cin >> t;
    while(t--){
        scanint(p);
        scanint(x);
        scanint(y);
        scanint(a);
        scanint(b);
        //cin >> p >> x >> y >> a >> b;
        st = a - b;
        fi = x - y;
        if (x == a && y == b){
            cout << "Counter-Terrorists Win !" << endl;
        } else {
        if (st == fi){
            if (p > 1 ){
            cout <<"Counter-Terrorists Win !"<<endl;
            }
            else {
                cout <<"Terrorists Win !"<<endl;
            }
        }
        else {
            if (abs(st)%2 == abs(fi)%2){
                if (p > 2){
                    cout << "Counter-Terrorists Win !"<<endl;
                }
                else {
                    cout <<"Terrorists Win !"<<endl;
                }
            }
            else{
                cout << "Terrorists Win !"<<endl;
            }
        }
    }
    }
    return 0;
}
