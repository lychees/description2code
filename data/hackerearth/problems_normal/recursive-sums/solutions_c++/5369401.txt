#include <bits/stdc++.h>
using namespace std;

long long unsigned int fun(long long unsigned int x)
{
	if(x<=9)
	return x;
	int sum=0;
	while(x>0)
	{
		int d=x%10;
		sum+=d;
		x/=10;
	}
	return sum;
}

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	cin>>n;
    	long long unsigned int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		long long unsigned int a,b;
    		cin>>a>>b;
    		sum+=(a*b);
    		sum=fun(sum);
    	}
    	
    	sum=fun(sum);
    	
    	cout<<sum<<"\n";
    }
    return 0;
}
