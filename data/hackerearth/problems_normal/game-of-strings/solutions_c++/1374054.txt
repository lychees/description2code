#include <bits/stdc++.h>
using namespace std;
int a[5001][5001];
int main()
{
    string s,s1;
    cin>>s>>s1;
    int m=s.size();
    int n=s1.size();
    int ans=0,i=0,j=0;
    for(i=0;i<=m;i++)
    {
    	for(j=0;j<=n;j++)
    	{
    		if(i==0 || j==0)
    		a[i][j]=0;
    		
 else if(s[i-1]==s1[j-1] && s[i-1]!='a' && s[i-1]!='e' && s[i-1]!='i' && s[i-1]!='o' && s[i-1]!='u')
    		a[i][j]=a[i-1][j-1]+1;
    			
    			else
    			a[i][j]=max(a[i][j-1],a[i-1][j]);
    		
    		
    		
    	}
    }
    cout<<a[m][n]<<endl;
    
    return 0;
}
