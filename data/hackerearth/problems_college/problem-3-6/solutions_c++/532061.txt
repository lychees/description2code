#include <iostream> 
using namespace std; 
int main() 
{ int n; cin>>n; int x[n],y[n],z[n],w[n]; for(int i=1;i<=n;i++) { cin>>x[i]; cin>>y[i]; } for(int g=1;g<=n;g++) { z[g]=n-1; w[g]=n-1; } for(int j=1;j<=n;j++) { for(int k=1;k<=n;k++) { if(x[j]==y[k]) { z[k]=z[k]+1; w[k]=w[k]-1; } } } for(int f=1;f<=n;f++) { cout<<z[f]<<" "<<w[f]; cout<<endl; } return 0; }
