#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	int count=0;
    	int n=str.length();
    	for(int i=0;i<str.length()/2;i++)
    	{
    		count+=abs(str[i]-str[n-1-i]);
    	}
    	cout<<count<<endl;
    }
    return 0;
}
