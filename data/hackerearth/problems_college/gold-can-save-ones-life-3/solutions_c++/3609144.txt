#include<bits/stdc++.h>
using namespace std;
#define ll long long int
vector<ll> arr(10001);
vector<ll> status(10001);
int main()
{
    ll num,i,j,test,k;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld%lld",&num,&k);
        for(i=0;i<num;i++)
         scanf("%lld",&arr[i]);
        if(arr[0]>0)
            status[0]=arr[0];
        else
            status[0]=0;
        for(i=1;i<=k;i++)
            status[i]=max(arr[i],status[i-1]);
        for(i=k+1;i<num;i++)
        {
            if(arr[i]>0)
                status[i]=max(arr[i]+status[i-k-1],status[i-1]);
            else
                status[i]=status[i-1];
        }
        printf("%lld\n",status[num-1]);
    }
}
