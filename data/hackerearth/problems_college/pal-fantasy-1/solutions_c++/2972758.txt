#include <iostream>
using namespace std;
#include<cstdio>
bool palindrome(char*s,int n){
	for(int i=0;i<=n/2;i++){
		if(s[i]!=s[n-i-1])
			return false;
	}
	return true;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	char s[51];
    	scanf("%s",s);
    	int len;
    	for(len=0;s[len]!='\0';len++);
    	for(int i=0;s[i]!='\0';i++){
    		if(palindrome(s+i,len-i)){
    			printf("%d\n",len+i);
    			break;
    		}
    	}
    	
    }
    //cout << "Hello World!" << endl;
    return 0;
}
