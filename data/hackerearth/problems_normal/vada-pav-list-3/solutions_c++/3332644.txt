#include <iostream>
#define BLANK 96
#define ASCII_a 97
using namespace std;


bool mHash[27][27][27][27][27] = {false};

int main()
{
	std::ios_base::sync_with_stdio(false);
    // cout << "Hello World!" << endl;
    int N;
    int count = 0;
    string name;
    int h[5];
    bool * val;
    
    cin >> N;
    while(N--){
    	cin >> name;
    	for(int i = 0; i < 5; i++){
    		if(i >= name.length()) h[i] = 0;
    		else h[i] = ((int) name[i]) - ASCII_a + 1;
    	}
    	val = &(mHash[h[0]][h[1]][h[2]][h[3]][h[4]]);
    	if(!(*val)){
    		count++;
    		*val = true;
    	}
    }
    
    cout << count << endl;
    for(int i = 0; i < 27; i++){
    	for(int j = 0; j < 27; j++){
    		for(int k = 0; k < 27; k++){
    			for(int l = 0; l < 27; l++){
    				for(int m = 0; m < 27; m++){
    					if(mHash[i][j][k][l][m]){
    						cout << (char) (i + ASCII_a - 1);
    						
    						if(j) cout << (char) (j + ASCII_a - 1);
    						else{
    							cout << endl;
    							continue;
    						}
    						
    						if(k) cout << (char) (k + ASCII_a - 1);
    						else{
    							cout << endl;
    							continue;
    						}
    						
    						if(l) cout << (char) (l + ASCII_a - 1);
    						else{
    							cout << endl;
    							continue;
    						}
    						
    						if(m) cout << (char) (m + ASCII_a - 1);
    						
    						cout << endl;
    					}
    				}
    			}
    		}
    	}
    }
    
    return 0;
}
