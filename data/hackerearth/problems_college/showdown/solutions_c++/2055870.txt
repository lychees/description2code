const int mod = 1000000000 + 7 ;
const int maxn = 100000 + 10 ;
const int inf = 2*1e9 ; 


int n , A[55] , dp[55][55] , B[55]; 

int solve(int i,int j){
	if(i <= 0 || j <= 0)
		return 0 ;
	int &ret = dp[i][j] ;
	if(ret != -1)
		return ret ;
	ret = 0 ;
	if(A[i] > B[j])
		ret = max(ret,100+solve(i-1,j-1)) ;
	else if(A[i] == B[j]){
		ret = max(ret,max(solve(i,j-1),50+solve(i-1,j-1))) ;
	}else{
		ret = max(ret,solve(i,j-1)) ;
	}
	return ret ;
}

int main(){

	cin >> n ;
	for(int i=1;i<=n;i++) cin >> A[i] ;
	for(int i=1;i<=n;i++) cin >> B[i] ;
	sort(A+1,A+1+n) ;
	sort(B+1,B+1+n) ;
	memset(dp,-1,sizeof dp) ;
	int ans = 0 ;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			ans = max(ans,solve(i,j)) ;
		}
	}
	cout << ans << endl ;
	return 0 ; 
}
