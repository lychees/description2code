#include <iostream>
using namespace std;

int main()
{
	long unsigned int a,b,n,i=2,temp=0;
    cin>>a>>b>>n;
    if(n==1)
    	cout<<a;
    else if(n==2)
    	cout<<b;
    else
    	{
    	while(i<n)
    	{
    		temp=(b*b)+a;
    		a=b; b=temp;
    		i++;
    	}
    	cout<<b;
    	}
    return 0;
}
