#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
short  dp[10000][10000];
 lli arr[100000];
 int solve(int st,int en)
  {
  	//cout<<"st "<<st<<" en"<<en<<endl;
  	     if(st==en) return 1;
		else if(st>en)return 0;
  	   else if(dp[st][en]!=-1)
  	    {
  	  	return dp[st][en];
		}
		else if(arr[en]<=arr[st]*3)
		{
			// cout<<" here "<<endl;
			
			return 0;
		}
		
		else
		{
			 dp[st][en]=min(solve(st+1,en),solve(st,en-1))+1;
			 return dp[st][en];
		}
  }
int main()
 {
 	int n;
 	 cin>>n;
 	 for(int i=0;i<n;i++)
 	  {
 	  	 cin>>arr[i];
	   }
	   sort(arr,arr+n);
	   
	   for(int i=0;i<10000;i++)
	   {
	   	for(int j=0;j<10000;j++)
	   	 {
	   	 	 dp[i][j]=-1;
			}
	   }
	   int ans=solve(0,n-1);
	   cout<<ans<<endl;
 	
 }
