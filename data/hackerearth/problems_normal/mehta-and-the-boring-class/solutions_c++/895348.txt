#include <bits/stdc++.h>
using namespace std;
vector<bool> p(1000001,1);
vector<long long> v(1000000),ans(1000000,0);
void pre()
{
	long long i,j;
	v[0]=0;v[1]=0;
	for(i=2;i<=1000000;i++)
	{
		v[i]=i;
	}
	p[0]=0;
	p[1]=0;
	for(i=2;i<=1000000;i++)
	{
		if(p[i]==1)
		{
			ans[i]=1;
			for(j=i*2;j<=1000000;j+=i)
			{
				p[j]=0;
				while(v[j]%i==0)
				{
					if(v[j]==0)
						break;
					ans[j]++;
					v[j]=v[j]/i;
				//	cout<<i<<" "<<v[j]<<" "<<ans[j]<<" "<<j<<endl;
				}
			}
		}
	}
	
}
int main()
{
    pre();
    long long i,t,x,y;
    vector<long long> sum(1000000,0);
    for(i=1;i<=1000000;i++)
    {
    	sum[i]=sum[i-1]+ans[i];
    }
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld %lld",&x,&y);
    /*	for(i=x;i<=y;i++)
    		cout<<ans[i]<<" ";*/
    	printf("%lld\n",sum[y]-sum[x-1]);
    }
    return 0;
}
