#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long
#define sd(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)

int main()
{ 
	int t;
	sd(t);
	while(t--)
	{
		ll n, i;
		sll(n);
		ll s = 0;
		for(i=1;i*i<n;++i)
		{
			if(n%i==0)
			{
				if(i%4)
					s += i;
				if((n/i)%4)
					s += (n/i);
			}
		}
		if(i*i==n and i%4)
			s += i;
		s *= 8;
		cout << s << endl;
	}	
	return 0;
}
