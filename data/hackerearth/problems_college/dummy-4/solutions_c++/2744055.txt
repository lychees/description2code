#include <iostream>
using namespace std;

int main()
{
    int t,sum=0,temp,i;
    string s;
    
    cin>>t;
    while(t--){
    	cin>>s;
    	for(i=0;i<s.size();i++){
    		sum+=s[i]-'0';
    	}
    }
    temp=sum;
    sum=0;
    while(temp){
    	sum+=temp%10;
    	temp/=10;
    	
    	if(temp==0 && sum>9){
    		temp=sum;
    		sum=0;
    	}
    }
    cout<<sum<<endl;
    return 0;
}
