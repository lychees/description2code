#include <iostream>
#include <string>
using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
  
    	string str;
    	int cntr = 0;
    	cin >> str;
    	int i, length = str.size();//l as length s as str
    	if(length>=2 && str[0]=='S' && str[1]=='-')
    	{
    	  str[1]='A';
    	  //cout << l << endl;
    	}
    	for(i=1;i<length;i++)
    	{
    		if(i+1<length && str[i-1]=='-' && str[i]=='S' && str[i+1]=='-')
    		{
    			str[i-1]=str[i+1]='A';
    		}
    		else if(i-2>=0 &&  str[i-2]=='-' && str[i-1]=='-' && str[i]=='B')
    		{
    			str[i-1]=str[i-2]='A';
    		}
    			else if(i-2>=0 &&  str[i-2]=='-' &&  str[i]=='B')
    		{
    		            str[i-2]='A';
    		}
    		else if(i+1<length && str[i]=='S' && str[i+1]=='-')
    		{
    			str[i+1]='A';
    		}
    		else if((str[i-1]=='-' && str[i]=='S') || (str[i-1]=='-' && str[i]=='B'))
    		{
    			str[i-1]='A';
    		}
    	}
    	
    	for(i=0;i<length;i++)
    	{
    		if(str[i]=='-')
    			cntr++;
    	}
    	
    	cout << cntr << endl;
    }
    return 0;
}
