ll power(ll b, ll e) {
    ll p = 1;
    while (e > 0) {
        if(e&1) {
            p = (p*b)%mod;
        }
        e = e>>1;
        b = (b * b)%mod;
    }
    return p;
}

ll fact[200001],ifac[200001];
void compute()
{
    ll i;
    fact[0]=ifac[0]=fact[1]=ifac[1]=1;
    for(i=2;i<200001;i++)
    {
        fact[i]=(fact[i-1]*i)%mod;
        ifac[i]=(ifac[i-1]*power(i,mod-2))%mod;
    }
}
ll a[100001],c[100001];
ll ncr(ll n,ll r)
{
    if(r>n)return 1;
    return (fact[n]*((ifac[n-r]*ifac[r])%mod))%mod;
}
