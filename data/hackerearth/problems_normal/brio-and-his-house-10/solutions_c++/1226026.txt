#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main()
{
    int t;
    long double a,b,c,p,area,rad,area_cir;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c;
    	
    	p=(a+b+c)/2.0;
    	area=sqrt(p*(p-a)*(p-b)*(p-c)*1.0);
    	
    	rad=(a*b*c)/(4.0*area);
    	
    	area_cir=3.1415*rad*rad;
    	
    	cout<<fixed<<setprecision(4)<<area_cir<<"\n";
    }
}
