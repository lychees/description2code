#include<bits/stdc++.h>
using namespace std;

const int n=20000010;
int f[n];

int main()
{
    f[1]=0;
    for(int i=2; i<n; i++)
    {
        f[i]=f[i-1]+1;
    if(i%2==0)
        f[i]=min(f[i],f[i/2]+1);
    if(i%3==0)
        f[i]=min(f[i],f[i/3]+1);
    }
    int t;
    cin>>t;
    while(t--)
    {
        int num;
        cin>>num;
        cout<<f[num]<<endl;
    }
    return 0;
}
