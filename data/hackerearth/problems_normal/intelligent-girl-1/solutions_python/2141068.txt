def main():
	l = ['2','4','6','8'] 
	s = raw_input()
	rev_s = s[::-1]
	
	opt = ""
	c = 0
	
	for i in rev_s:
		if i in l:
			c = c+1 
		opt = str(c)+" "+opt
		
	
	print opt
	
	
if __name__=="__main__":
	main()
