#include <iostream>
using namespace std;

int main()
{
	int t;scanf("%d",&t);
	while(t--){
		int n; scanf("%d",&n); long long a[n+1],b[n+1],c[n+1],sum[n+1],mn;
		cin>>a[0];
		sum[0]=a[0];
		for(int i=1;i<n;i++){
			scanf("%lld",&a[i]);
			sum[i]=sum[i-1]+a[i];
		}
		for(int i=0;i<n;i++){   
			b[i]=sum[i];
			c[i]=sum[n-1]-b[i];
		} mn=abs(c[0]-b[0]);
		for(int i=1;i<n;i++){
			if(abs(c[i]-b[i])<mn) mn=abs(c[i]-b[i]);
			
		}
		printf("%lld\n",mn);
	}
    //cout << "Hello World!" << endl;
    return 0;
}
