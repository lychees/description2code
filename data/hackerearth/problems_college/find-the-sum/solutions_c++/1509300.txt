LL f(LL n)
{
	LL i=0LL;
	for(i=0;i<=2000005;i+=1LL)
	{
		if(((i*(i+1LL))/2LL)>n)break;
	}
	i--;
	LL sum=0;
	for(LL j=1LL;j<=i;j+=2LL)
	{
		sum+=j;
	}
	n-=(i*(i+1LL))/2LL;
	if(n%2)
	sum+=(i+1LL);
	return sum;
}
