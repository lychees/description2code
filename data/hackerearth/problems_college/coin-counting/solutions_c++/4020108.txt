
#include <bits/stdc++.h>
using namespace std;

int coinc(int a[], int n, int sum)
{
	int dp[8][sum + 1];
	int i;
	int j;
	for(i = 0 ; i < n ; i++)
	{
		dp[a[i]][0] = 1;
		//cout<<a[i]<<endl;
	}
	
	for(j = 0 ; j <= sum ; j++)
	{
		dp[a[0]][j] = 1;
	}
	int k;
	for(i = 1 ; i < n ; i++ )
	{
		
		for(j = 0 ; j <= sum ; j++)
		{
			if(j < a[i])
			{
				dp[a[i]][j] = dp[a[i - 1]][j];
			}
			else
			   dp[a[i]][j] = dp[a[i - 1]][j] + dp[a[i]][j - a[i]];
		}
	}
	return dp[a[n - 1]][sum];
}
int main()
{
  int t;
  cin >> t;
  int a[3] ={1,3,5};
  while(t--)
  {
  	int n;
  	cin >> n;
  	cout<<coinc(a,3,n)<<endl;
  }
  return 0;
}
