#include <iostream>
using namespace std;




int main()
{
	int t,n,x,i,sum;
	cin>>t;
	for(i=1;i<=t;i++)
	{sum=0;
		cin>>n;
		while(n!=1)
	{
		if(n%3==0)
		{
			sum++;
			n=n/3;
		}
		else
		if(n%2==0)
		{	
			sum++;
			n=n/2;
		}
		else
		{
			sum++;
			n=n-1;
		}
	}
		cout<<sum;
		cout<<endl;
	}
    return 0;
}
