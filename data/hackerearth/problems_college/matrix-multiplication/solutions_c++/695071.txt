#include <iostream>

using namespace std;

int main()
{
        int test_cases;
        cin>>test_cases;
        while(test_cases > 0)
        {
                int X1,X2,Y1,Y2;
                cin>>X1>>Y1>>X2>>Y2;
                int result = 0;
                if(X1==Y2){
                	result++;
                }
                if(Y1==X2){
                	result++;
                }
                cout << result << endl;
                test_cases --;
        }
        return 0;
}
