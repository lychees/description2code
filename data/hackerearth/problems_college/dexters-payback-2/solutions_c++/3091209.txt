#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int t,a,b,i,j,k,s;
    sd(t);
    while(t--)
    {
        sd(a);
        sd(b);
        k=0;s=0;
        for(i=0;i<a;i++)
        {
            if(k<2)
                s+=2*(b/4)+min(2,b%4);
            else
                s+=2*(b/4)+max(0,(b%4)-2);
            k=(k+1)%4;
        }
        printf("%d\n",s);
    }
    return 0;
}
