#include <bits/stdc++.h>
using namespace std;

int main()
{int t;
cin>>t;
while(t--){
	int n;
	cin>>n;
	int ar[2*n];
	for(int i=0;i<(2*n);++i)
		cin>>ar[i];
	sort(ar,ar+(2*n));
	int min1=0;
	for(int i=1;i<=n;++i)
	{
		min1=max(min1,ar[i-1]+ar[2*n-i]);
		
	}
	cout<<min1<<endl;
}
    //cout << "Hello World!" << endl;
    return 0;
}
