#include <iostream>
using namespace std;

int main()
{
    int t,i;
    int flag;
    char c[100001];
    cin>>t;
    while(t--)
    {
    
    	i=0;flag=0;
    	cin>>c;
    	while(c[i]!='\0')
    	{
    		if(c[0]=='Y')
    		{
    			flag=1;
    			break;
    		}
    		if((c[i]!='R'&&c[i]!='Y'))
    	{flag=1;break;}
    	
    	if(c[i]=='Y')
    	{
    		if(c[i+1]!='\0'&&c[i+1]=='Y'&&c[i+2]!='\0'&&c[i+2]!='R')
    	{	flag=1;
    		break;}
    		
    		
    	}
    	i++;
    	}
    	if(flag==1)
    	cout<<"NO\n";
    	else 
    	cout<<"YES\n";
    	
    }
}
