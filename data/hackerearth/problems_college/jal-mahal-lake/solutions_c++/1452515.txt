#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	long long int n;
    	scanf("%d",&n);
    	vector < pair <long long int,long long int> > p;
    	long long int a[n],b[n],i,x=0,y=0,s=0,minm=99999999999999999;
    	for(i=0;i<n;i++)
    	scanf("%lld",&a[i]);
    		for(i=0;i<n;i++)
    	scanf("%lld",&b[i]);
    	for(i=0;i<n;i++)
    	p.push_back(make_pair(a[i],a[i]+b[i]));
    	sort(p.begin(),p.end());
    	for(int j=0;j<n;j++){
    		y=0;
    		x=0;
    	for(int k=0;k<j;k++){
    		x=x+(p[k+1].first-p[k].second)*(k+1);
    		
    		y=x;
    		//cout<<x<<" "<<k<<" ";
    	}
    	//cout<<endl;
    	s=x;
    	x=0;
    	y=0;
    	int c=1;
    	for(i=n-1;i>j;i--){
    		x=x+(p[i].first-p[i-1].second)*c;
    		
    		y=p[i].first-p[i-1].second;
    		c++;
    	//	cout<<x<<" "<<j<<" ";
    		
    	}
    	minm=min(minm,s+x);
    //	cout<<endl;
    	}
    	cout<<minm<<endl;
    }
    return 0;
}
