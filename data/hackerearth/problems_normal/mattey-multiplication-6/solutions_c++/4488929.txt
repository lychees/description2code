#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
bool pos[67];
int main(){
    int tc;scanf("%d",&tc);
    while(tc--){
        lli a,b;scanf("%lld%lld",&a,&b);
        lli num=b;
        int p=0;
        int cnt1=0;
        memset(pos,false,sizeof pos);
        while(num>0){
            if(num&1){
                pos[p]=true;
                cnt1++;
            }
            p++;
            num>>=1;
        }
        int cnt2=0;
        for(int i=62;i>=0;--i){
            if(pos[i] ){
                cnt2++;
                if(cnt2!=cnt1)
                printf("(%lld<<%d) + ",a,i);
                else
                printf("(%lld<<%d)\n",a,i);
            }
        }
    }
    return 0;
}
