raw_input()
lst = map(int, raw_input().split())
fSum = 0
sSum = 0
tSum = 0
y = 0
for x in xrange(len(lst)):
    if y == 0:
        fSum += lst[x]
        y = 1
    elif y == 1:
        sSum += lst[x]
        y = 2
    else:
        tSum += lst[x]
        y = 0
print str(fSum) + " " + str(sSum) + " " + str(tSum)
