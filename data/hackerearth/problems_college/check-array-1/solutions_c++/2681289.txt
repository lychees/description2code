#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int A[100005];
		long long lpart[100005]={0},rpart[100005]={0};
	    int n;
	    cin >> n;
	    for(int i = 1 ;i <= n ; ++i)
	    	cin >> A[i];
	    for(int i = 1,j=n;i<=n; i++,j--)
	    {
	    	lpart[i]=lpart[i-1]+A[i];
	    	rpart[j]=rpart[j+1]+A[j];
	    }
	    int k;
	    for(k = 1 ; k <= n ;k++)
	    	if(lpart[k] == rpart[k])
	    	{
	    		cout << "YES\n";
	    		break;
	    	}
	    if(k==n+1)
	    	cout << "NO\n";
	}
	return 0;
}
