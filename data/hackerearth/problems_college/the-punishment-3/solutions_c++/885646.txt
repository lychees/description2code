#include <iostream>
#define MAX 100001
using namespace std;

int main() {
    
    int t;
    cin>>t;
    while (t--)
    {
        char ans[MAX];
        int n,k;
        cin>>n>>k;
        int a=k;
        int b= n-k;
        ans[a+b]='1';
        for (int i=0;i<a+b;i++)
            ans[i]='0';
        if (a>b)
        {
            for (int i=b;i<a+b;i++)
                ans[i]='9';
            ans[a]='8';
        }
        else
        {
            for (int i=a;i<a+b;i++)
                ans[i]='9';
            ans[b]='8';
        }
        ans[0]='1';
        for (int i=a+b-1;i>=0;i--)
            cout<<ans[i];
        cout<<endl;
    }
    return 0;
}
