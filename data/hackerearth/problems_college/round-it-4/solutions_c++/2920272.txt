#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n, b;
        scanf("%d %d", &b, &n);
        int c1 = ((b/n)*n)-b;
        int c2 = ((b/n)*n)+n-b;
        int c3 = b;
        //cout<<c1<<" "<<c2<<" "<<c3<<endl;
        int ans;
        if(abs(c1) == c2 && abs(c1)<=c3)
            ans = b+c2;
        else if(c3<=abs(c1) && c3<=c2)
            ans = 0;
        else if(c2<abs(c1))
            ans = b+c2;
        else
            ans = b+c1;
        cout<<ans<<endl;
    }
    return 0;
}
