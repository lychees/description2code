#include <iostream>
using namespace std;

void print_color(int x){
	if(x<10){
		cout<<x;
	}
	else
	{
		switch(x){
			case 10:cout<<"U";
					break;
			case 11:cout<<"V";
					break;
			case 12:cout<<"W";
					break;
			case 13:cout<<"X";
					break;
			case 14:cout<<"Y";
					break;
			case 15:cout<<"Z";
					break;
		}
	}
}

void convert(int x)
{
	if(x<16)
	{
		print_color(x);
	}
	else
	{
		convert(x/16);
		print_color(x%16);
	}
}
int main()
{
    int t,r,g,b;
    int i=1;
    cin>>t;
    while(t--){
    	cout<<"Case "<<i<<":"<<" ";
    	cin>>r>>g>>b;
    	if(r<16){
    		cout<<"0";
    		convert(r);
    	}
    	else
    	{
    		convert(r);
    	}
    	cout<<" ";
    	if(g<16){
    		cout<<"0";
    		convert(g);
    	}
    	else{
    		convert(g);
    	}
    	cout<<" ";
    	if(b<16){
    		cout<<"0";
    		convert(b);
    	}
    	else{
    		convert(b);
    	}
    	cout<<" "<<"\n";
    	i=i+1;
    	
    }
    return 0;
}
