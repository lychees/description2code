#define MOD 1000000007

int re[1111];
int a[1111];
int cal(int x)
{
	int re=0;
	while(x)
	{
		re+=x%10;
		x/=10;
	}
	return re;
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n;
	int jin,t;
	cin>>T;
	re[1]=1;
	a[1]=1;
	n=1;
	re[0]=1;
	for(i=2;i<=1000;i++)
	{
		jin=0;
		for(j=1;j<=n;j++)
		{
			t=jin+i*a[j];
			jin=t/1000;
			a[j]=t%1000;
		}
		if(jin)
		{
			n++;
			a[n]=jin;
		}
		re[i]=0;
		for(j=1;j<=n;j++)
		{
			re[i]+=cal(a[j]);
		}
	}
	while(T--)
	{
		cin>>i;
		cout<<re[i]<<endl;
	}
//	system("pause");
	return 0;
}
