#include<bits/stdc++.h>
using namespace std;
long long LCA(long long n,long long a,long long b)
{
    if(n>a && n>b)return 0;
    if(n==a||n==b)return n;
    long long left=LCA(2*n,a,b);
    long long right=LCA(2*n+1,a,b);
    if(left&&right)
        return n;
    else
        return (left?left:right);
}
int main()
{
    int T;
    long long int a,b,ans;
    cin>>T;
    while(T--)
    {
        scanf("%lld %lld",&a,&b);
        printf("%lld\n",LCA(1,a,b));
    }
}
