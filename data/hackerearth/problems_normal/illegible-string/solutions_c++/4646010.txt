#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string s;
	cin>>s;
	int num=0;
	int maximum=0;
	int minimum=0;
	int cnt=0;
	for(int i=0;i<n;i++)
	{
		if(s.at(i)=='v')
			cnt+=1;
		else if(s.at(i)=='w')
		{
			cnt+=2;
		}
		else
		{
			maximum+=cnt;
			minimum+=((cnt+1)/2);
			num++;
			cnt=0;
		}
	}
	maximum+=cnt;
	minimum+=((cnt+1)/2);
	cout<<num+minimum<<" "<<num+maximum<<endl;

}
