void mulMat(ll A[2][2], ll B[2][2])
{
  ll x =  (A[0][0]*B[0][0] + A[0][1]*B[1][0])%mod;
  ll y =  (A[0][0]*B[0][1] + A[0][1]*B[1][1])%mod;
  ll z =  (A[1][0]*B[0][0] + A[1][1]*B[1][0])%mod;
  ll w =  (A[1][0]*B[0][1] + A[1][1]*B[1][1])%mod;
  A[0][0] = x;  A[0][1] = y;  A[1][0] = z;  A[1][1] = w;
}
 
void mul(ll M[2][2],ll x, ll ret[2][2]){
    rep(i,2)rep(j,2)ret[i][j]=(i==j);
 
    while(x > 0){
        if(x&1){
            x--;
            mulMat(ret,M);
        }
        else{
            x>>=1;
            mulMat(M,M);
        }
    }
}
 
ll solve(ll n){
    if(n<=1) return n;
    ll M[2][2]={{1,1},{1,0}};
    ll ret[2][2];
    mul(M,n-1,ret);
   
    ll fn = ret[0][0];
    ll fn_1 = (ret[0][0]+ret[1][0])%mod;
    return (fn * fn_1)%mod;
}

        printf("%lld\n",solve(n));
