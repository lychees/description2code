typedef long long ll;
#include <iostream>
using namespace std;
ll mod=10000007,fac[107],inv[107];
void euclid(ll a,ll b,ll& x,ll& y)
{
	if(a%b==0)
	{
		x=0;
		y=1;
		return;
	}
	euclid(b,a%b,x,y);
	ll temp=x;
	x=y;
	y=temp-y*(a/b);
}
ll inverse(ll a,ll m)
{
	ll x,y;
	euclid(a,m,x,y);
	if(x<0)x+=m;
	return x;
}

ll compute(ll n,ll r)
{
	if(n<r || n==0)
	return 0;
	if(r==0)
	return 1;
	ll res;
	res=(fac[n]*inv[n-r])%mod;
	res=(res*inv[r])%mod;
	return res;
}
int main()
{
	ll i,n,r;
	fac[0]=fac[1]=inv[0]=inv[1]=1;
	for(i=2;i<=101;i++)
	{
		fac[i]=(i*fac[i-1])%mod;
		inv[i]=(inverse(fac[i],mod))%mod;
	}
	scanf("%lld %lld",&n,&r);
	if(n<r)
	printf("-1\n");
	else
	{
		printf("%lld\n",compute(n-1,r-1));
	}
    
    return 0;
}
