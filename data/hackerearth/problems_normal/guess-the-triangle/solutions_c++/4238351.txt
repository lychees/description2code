#include <bits/stdc++.h>
using namespace std;

bool myfunc(pair<float, float> a, pair<float, float> b)
{
	if(a.first != b.first)
		return a.first < b.first;
	else
		return a.second < b.second;
}

int main()
{
    vector<pair<float, float>> mid(3);
    vector<pair<float, float>> vert(3);
    
    cin >> mid[0].first >> mid[0].second;
    cin >> mid[1].first >> mid[1].second;
    cin >> mid[2].first >> mid[2].second;
    
 	vert[0].first = mid[1].first + mid[2].first - mid[0].first;
 	vert[1].first = mid[2].first + mid[0].first - mid[1].first;
 	vert[2].first = mid[0].first + mid[1].first - mid[2].first;
 	vert[0].second = mid[1].second + mid[2].second - mid[0].second;
 	vert[1].second = mid[2].second + mid[0].second - mid[1].second;
 	vert[2].second = mid[0].second + mid[1].second - mid[2].second;
 	
 	sort(vert.begin(), vert.end(), myfunc);
 	printf("%.4f %.4f\n", vert[0].first, vert[0].second);
 	printf("%.4f %.4f\n", vert[1].first, vert[1].second);
 	printf("%.4f %.4f\n", vert[2].first, vert[2].second);
}
