#include <iostream>
#include <cmath>
using namespace std;


int** splendmatrix(int **ptr, int n, int s, int a, int b)
{
	if(n==1)
	{
		ptr[a][b]=s;
		ptr[a][b+1]=s+1;
		ptr[a+1][b]=s+2;
		ptr[a+1][b+1]=s+3;
		
	}
	
	else if(n>1)
	{
	
	int p=pow(2,n-1)*pow(2,n-1);
	int k=pow(2,n)/2;
		splendmatrix(ptr, n-1,s,a,b);
	    splendmatrix(ptr, n-1,s+p,a,b+k);
	    splendmatrix(ptr, n-1,s+p*2,a+k,b);
	    splendmatrix(ptr, n-1,s+p*3,a+k,b+k);
		
		
	}
	
	return ptr;
}
int main()
{
    int n;
    cin>>n;
    int par=pow(2,n);
    
    int** mat = new int*[par];
	for(int i = 0; i < par; ++i)
    mat[i] = new int[par];
    
    mat=splendmatrix(mat,n,1,0,0);
   
    for(int i=0; i<par; i++)
    {
    	for(int j=0; j<par; j++)
         cout<<mat[i][j]<<" ";
         
         cout<<"\n";
    }
    
    return 0;
}
