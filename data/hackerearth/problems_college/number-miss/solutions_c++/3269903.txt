#include<stdio.h>
int main(){
	int n, m, i; scanf("%d", &n); int a[n + 1], ai[100001] = {0}, bi[100001] = {0};		
	for(i = 1; i <= n; scanf("%d", &a[i]), ai[a[i]]++, i++);
	scanf("%d", &m); int b[m + 1];
	for(i = 1; i <= m; scanf("%d", &b[i]), bi[b[i]]++, i++);
	for(i = 1; i <= 10000; i++)	if(ai[i] > 1 && bi[i] > 1 && ai[i] != bi[i])	printf("%d ", i);
	printf("\n");
	return 0;
}
