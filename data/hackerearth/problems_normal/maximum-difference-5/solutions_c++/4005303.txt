#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int com1 (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int com2 (const void * a, const void * b)
{
  return ( *(int*)b - *(int*)a );
}
int main()
{
	ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    //cout << "Hello World!" << endl;
    long long int n,i,c=0,m=0,j,m1=0;
    cin>>n;
     int a[n],b[n];
    for(i=0;i<n;++i)
    {
    	cin>>a[i];
    	b[i]=a[i];
    }
    qsort(a,n,sizeof(int),com1);
    qsort(b,n,sizeof(int),com2);
    i=0;j=0;
    while(c<n-1)
    {
    	m=m+abs(a[i]-b[j]);
    	if(c%2==0)
    	++i;
    	else
    	++j;
    	++c;
    }
    i=0;j=0;c=0;
    while(c<n-1)
    {
    	m1=m1+abs(a[i]-b[j]);
    	if(c%2==0)
    	++j;
    	else
    	++i;
    	++c;
    }
    if(m>m1)
    cout<<m;
    else
    cout<<m1;
    return 0;
}
