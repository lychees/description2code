#include <iostream>
using namespace std;
#include <algorithm>

int main()
{
	int t;
	cin >> t;
	while(t--){
		int n,x,y;
		cin >> n >> x >> y;
		int arold[n];
		int arnew[n];
		for(int i = 0; i < n; i++){
			cin >> arold[i];
			cin >> arnew[i];
		}
		sort(arold,arold+n);
		sort(arnew,arnew + n);
		long int cost1 = 0;
		for(int i = 0; i < n ; i++){
			if(arold[i] < arnew[i]) cost1 += x*(arnew[i] - arold[i]);
			else if(arold[i] > arnew[i]) cost1 += y*(arold[i] - arnew[i]);
		}
		
		cout << cost1 << endl;
	}
    
    return 0;
}
