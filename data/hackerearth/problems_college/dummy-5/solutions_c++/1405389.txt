#include<bits/stdc++.h>
#define LL long long int
#define P(N) printf("%d\n",N);
#define S(N) scanf("%d",&N);
#define pnl printf("\n");
using namespace std;
int arr[25][4];
int GP,GC,GF;
bool solve(int N)
{
    int x,y,z;
    int set_size= pow(2,N);
 //   P(set_size)
    for(int i=0;i<set_size;i++)
    {
        x=0,y=0,z=0;
        for(int j=0;j<N;j++)
        {
            if(i &(1<<j) )
            {
            //	cout<<j<<"dsadnsa ";
                x+=arr[j][0]; y+=arr[j][1]; z+=arr[j][2];
                if(x>GP || y>GC || z>GF)
                    break;
                if(x==GP && y==GC && z==GF)
                {
                    return true;
                }
            }
        }

    }
    return false;
}
int main()
{
      int i,j,t;
      S(t);
      int N;
      for(int tc=1;tc<=t;tc++)
      {
         cin>>GP>>GC>>GF;
         S(N)
         for(i=0;i<N;i++)
         {
             scanf("%d %d %d",&arr[i][0],&arr[i][1],&arr[i][2]);
         }
         bool x= solve(N);
         printf("Case #%d: ",tc);
         if(x)
            printf("yes\n");
         else printf("no\n");
      }
return 0;
}
