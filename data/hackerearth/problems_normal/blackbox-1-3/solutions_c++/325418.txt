using namespace std;

#define ll long long
#define vt vector

ll gcd(ll a,ll b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main()
{
	ll t;
	vector<ll> vals(3,0);
	cin>>t;
	while(t--)
	{
		scanf("%lld %lld %lld",&vals[0],&vals[1],&vals[2]);
		sort(vals.begin(),vals.end());
		if(vals[2]*vals[2]==vals[1]*vals[1]+vals[0]*vals[0] && gcd(vals[0],vals[1])==1 && gcd(vals[1],vals[2])==1 && gcd(vals[0],vals[2])==1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
