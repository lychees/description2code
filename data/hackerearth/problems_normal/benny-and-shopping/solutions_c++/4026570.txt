#include<stdio.h>
#include<iostream>
#include<sstream>
#include<string.h>

using namespace std;

int main()
{
   int no_lines;
   string input_line;
   
   getline(cin, input_line);
   stringstream input_buf(input_line);
   input_buf >> no_lines;   
   
   char ch;
   bool dollar_seen;   
   bool leading_zeros_gone;
   
   for(int i=0; i<no_lines; i++)
   {
      getline(cin, input_line);
      
      input_buf.str(input_line);      

      dollar_seen = false;
      leading_zeros_gone = false;
      stringstream output_buf;      

      for(int j=0; input_line[j]!='\0'; j++)
      {        
         ch = input_line[j];
         
         if(true == dollar_seen)
         {
            if(((ch >= 'a') && (ch <= 'z')) || ((ch >= 'A') && (ch <= 'Z')))
            {             
               break;
            }
            
            if(ch != ' ')
            {
               if((ch > '0') && (ch <= '9'))
               {
                  leading_zeros_gone = true;
               }
               if(true == leading_zeros_gone)
               {
                  output_buf << ch;
               }
            }
         }
         
         if(ch == '$')
         {
            dollar_seen = true;
            leading_zeros_gone = false;
            output_buf << ch;
         }         
      }
      if(false == leading_zeros_gone)
      {
         output_buf << '0';
      }
      cout << output_buf.str() << endl;
   }  
	return 0;
}
