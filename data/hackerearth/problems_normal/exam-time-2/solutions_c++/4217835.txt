int etf(int n){
	float ans=n;
	for(int i=2;i*i<=n;i++){
		if(n%i==0){
			while(n%i==0)n/=i;
			ans*=(1.0-(1.0/(float)i));
		}
	}
	if(n>1)ans*=(1.0-(1.0/(float)n));
	return (int)ans;
}

		inint(k);
		cout<<etf(k)<<"\n";
	}

}
