#include<iostream>
#include<algorithm>
#include<string>
#include<stdlib.h>
using namespace std;
main()
{

    string c,s;
    int i,j,t,f=0;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>c;
        s=c;
        reverse(c.begin(),c.end());

        for(j=0;j<c.length()-1;j++)
        {

            if(abs(c[j+1]-c[j])==abs(s[j+1]-s[j]))
            {
                f=1;
            }
            else{
                f=0;
                cout<<"Not Funny"<<endl;
                goto q;
            }
        }
        if(f==1)
        {

            cout<<"Funny"<<endl;
        }
        q:;
    }
}
