#include<iostream>
using namespace std;
#include<cstdlib>
#include<string.h>

int main()
{
	int tc,count=0,ans=0;
	cin>>tc;
	string s[tc];
	char c[tc][100];
	int x,y[tc][100];
	for(int i=0;i<tc;i++)
	{
		for(int j=0;j<100;j++)
		y[i][j]=0;
	}
	for(int i=0;i<tc;i++)
	{
		cin>>s[i];
		strcpy(c[i],s[i].c_str());
		
	}
	
	
	for(int i=0;i<tc;i++)
	{
		for(int j=0;j<strlen(c[i]);j++)
		{
			x=c[i][j];
			count=0;
			for(int k=0;k<100;k++)
			{
				if(x>=97 && x<=122)
				if(x==y[i][k])
				{
					count++;
				}
			}
			if(count==0)
			{
				y[i][j]=x;
			}
		}
}
		
		for(int i=0;i<tc;i++)
	{
		ans=0;
		for(int j=0;j<100;j++)
		{
			if(y[i][j]!=0)
				{
					ans++;
				}
			//cout<<y[i][j]<<"\t";
		}
			if(ans==26)
			cout<<"YES\n";
			else
			cout<<"NO\n";
		//	cout<<y[i]<<"\n";
	}
	
	return 0;
	
}
