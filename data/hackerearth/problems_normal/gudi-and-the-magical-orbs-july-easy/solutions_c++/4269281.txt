#include <bits/stdc++.h>
using namespace std;
int dp[111][111][555];
int a[111][111];
int main()
{
	int i,j,k,n,m,t,K;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d %d",&n,&m,&K);
		for(i=0;i<111;i++)
			for(j=0;j<111;j++)
				for(k=0;k<555;k++)
					dp[i][j][k]=0;
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
				scanf("%d",&a[i][j]);
		}
		dp[0][0][a[0][0]]=1;
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)	
			{
				for(k=0;k<=K;k++)
				{
					if(i==0)
					{
						if(dp[i][j-1][k-a[i][j]])
							dp[i][j][k]=1;
					}
					else if(j==0)
					{	
						if(dp[i-1][j][k-a[i][j]])
							dp[i][j][k]=1;		
					}
					else if(dp[i-1][j][k-a[i][j]] || dp[i][j-1][k-a[i][j]] || dp[i-1][j-1][k-a[i][j]])
						dp[i][j][k]=1;
				}
			}
		}
		int f=0,max;
		for(k=0;k<=K;k++)
		{
			if(dp[n-1][m-1][k])
			{
				f=1;
				max = k;
			}
		}
		if(f)
			printf("%d\n",max);
		else
			printf("-1\n");
	}
	return 0;
}
