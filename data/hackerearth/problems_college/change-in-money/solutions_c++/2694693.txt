typedef long long ll;

ll count( int S[], int m, int n )
{
	ll table[n+1];

	memset(table, 0ll, sizeof(table));

	table[0] = 1ll;
	for(int i=0; i<m; i++) {
		for(int j=S[i]; j<=n; j++)
			table[j] += table[j-S[i]];
//		printf("%lld",table[i]);sp;
	}

	return table[n];
}
