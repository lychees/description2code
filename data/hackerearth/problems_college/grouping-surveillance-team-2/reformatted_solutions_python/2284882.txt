num = input()
groupSize = dict()


def gs(n, m):
    if m < 2:
        return 0
    if (m == n):
        return 1
    if n < m:
        return 0
    if (n, m) in groupSize:
        return groupSize[(n, m)]
    res = 1
    for i in range(m, n-1):
        res += gs(n-i, i)
    groupSize[(n, m)] = res
    return res


for _ in range(num):
    n = input()
    print gs(n, 2)
