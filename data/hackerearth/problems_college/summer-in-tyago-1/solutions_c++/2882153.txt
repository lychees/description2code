#include <iostream>
#include <algorithm>
using namespace std;
 
int main()
{
	int n,input[200000],max,pos = 0;
	cin>>n;
	for(int i =0;i<n;i++)
		cin>>input[i];
	pos = 0;
	for(int j = 0; j< n; j++)
	{
		for(int i = n-1-j; i> 0; i--)
		{
			if(input[i]<input[i-1])	
			{
				input[i-1]--;
				input[i]++;
				int k = input[i];
				input[i] = input[i-1];
				input[i-1] = k;
			}
		}
	}
	if(!is_sorted(input,input+n))
		cout<<":(";
	else
		for(int i =0;i<n;i++)
			cout<<input[i]<<" ";
    return 0;
}
