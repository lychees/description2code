#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int n,h,i,mul=1;
    	scanf("%lld %lld",&n,&h);
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&a[i]);
    		mul = ((mul%h)* (a[i]%h))%h; 
    	}
    	if(mul%h==0)
    	{
    		printf("YES\n");
    	}
    	else
    	{
    		printf("NO\n");
    	}
    }
    return 0;
}
