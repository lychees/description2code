import sys

x1d , y1d = map(float,sys.stdin.readline().split())
x2d , y2d = map(float,sys.stdin.readline().split())
x3d , y3d = map(float,sys.stdin.readline().split())

x3 = x2d - x1d + x3d
x1 = 2*x3d -x3
x2 = 2*x1d - x1

y3 = y2d - y1d + y3d
y1 = 2*y3d - y3
y2 = 2*y1d - y1

ans = [(x1,y1),(x2,y2),(x3,y3)]
ans.sort(key=lambda x:x[0])
for i in ans :
	print "%.4f %.4f"%(i[0],i[1])
