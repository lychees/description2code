#include <iostream>
using namespace std;
int min(int a,int b)
{ 
return a < b ?a:b ;
	}
int max(int a,int b){
	return a>b ?a:b ;
}	
int main()
{
   int test;
   cin>>test;
   while(test--)
   {
   	int food,gold,samrui,paladin,champion,pw,s,p,c,temp=0;
   	cin>>food>>gold>>s>>p>>c>>pw;
   	for(samrui=0;samrui*100<=food;samrui++)
   	{
   		for(paladin=0;samrui*100+paladin*125<=food&&paladin*50<=gold;paladin++)
   		{
   			int f_left=(food-100*samrui-125*paladin);
   			int g_left=(gold-50*paladin);
   			champion=min(f_left/50,g_left/100);
   			temp=max(temp,s*samrui+p*paladin+c*champion);
   		}
   	}
   		if(temp>pw)
   		cout<<temp-pw<<endl;
   		else
   		cout<<"-1"<<endl;
   	
   }
   
   
    return 0;
}
