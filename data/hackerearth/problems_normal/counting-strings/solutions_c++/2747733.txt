#include <iostream>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int T;
    long ctr,ctr1;
    cin  >>T;
    string s;
    while(T--)
    {
    cin >> s;
    long lens =s.length();
    ctr =0;
    ctr1=0;
    for(int i=0;i<lens;i++)
    {
    	ctr1++;
    	if((s[i]=='a')||(s[i]=='z'))
    	{
    		ctr += (ctr1)*(lens-i);
    		ctr1=0;
    		
    	}
    	
    	
    }
    cout << ctr <<"\n";
    	
    	
    }
    return 0;
}
