#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s;
    	int a,b,c;
    	
    	cin>>s;
    	
    	if(s.length()%3!=0)
    		cout<<"Not OK"<<endl;
    	else
    	{
    		set<char>st;
    		bool f=true;
    		
    		for(int i=0;i<s.length()/3;i++)
    		{
    			st.insert(s[i]);
    		}
    		
    		if(st.size()!=1)
    		{
    			cout<<"Not OK"<<endl;
    			f=false;
    		}
    		
    		
    		for(int i=s.length()/3;i<2*s.length()/3;i++)
    		{
    			st.insert(s[i]);
    		}
    		
    		if(st.size()!=2&&f)
    		{
    			cout<<"Not OK"<<endl;
    			f=false;
    		}
    		
    		
    		for(int i=2*s.length()/3;i<s.length();i++)
    		{
    			st.insert(s[i]);
    		}
    		
    		if(st.size()!=3&&f)
    		{
    			cout<<"Not OK"<<endl;
    			f=false;
    		}
    		
    		if(f)
    			cout<<"OK"<<endl;
    	}
    }
    return 0;
}
