#include <bits/stdc++.h>

using namespace std;

void counting_sort( int a[], int n ){
	int max=0;
	int i;
	for( i = 0; i < n; i++ ){
		if( a[i] > max )
		max = a[i];
	}
	int c[max+1];
	for(  i = 0; i <= max; i++ )
		c[i] = 0;
	for( i = 0; i < n; i++ ){
		c[a[i]]++;
	}
	for( i = 1; i <= max; i++){
		c[i] = c[i] + c[i-1]; 
	}
	int b[n];
	for( i = n-1; i >= 0; i-- ){
		b[c[a[i]]-1] = a[i];
		c[a[i]]--;
	}
	for( i = 0; i < n; i++ )
		a[i] = b[i];
}

int main()
{
	int t,n,k,i,j;
	long long ans;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&n,&k);
		int a[n];
		for (i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		counting_sort(a,n);
		reverse(a, a + n);
		ans = 0;
		for (i = 0; i < k && i < n; i ++) {
			ans += a[i];
		}
		printf("%lld\n",ans);
	}
	
	return 0;
}
