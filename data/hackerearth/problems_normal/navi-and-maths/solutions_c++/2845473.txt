#include <bits/stdc++.h>
#define ll long long
using namespace std;
#define mod 1000000007


ll fast_exp(ll base, ll exp)
{
    ll res=1;
    while(exp>0)
    {
       if(exp%2==1) res=(res*base)%mod;
       base=(base*base)%mod;
       exp/=2;
    }
    return res;
}

int main()
{
    ll t,n,arr[17];
    cin>>t;
    ll k=1;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++) cin>>arr[i];
    	ll ans=0;
    	for(int i=1;i<(1<<n);i++)
    	{
    		ll p=1;
    		ll s=0;
    		ll count=0;
    		for(int j=0;j<n;j++)
    		{
    			if(i&(1<<j)) 
    			{
    				p*=arr[j];
    				s+=arr[j];
    				p%=mod;
    				s%=mod;
    				count++;
    			}
    		}
    		if(count<2) continue;
    		p%=mod;
    		s%=mod;
    		s=fast_exp(s,mod-2);
    		p=(p*s)%mod;
    		p%=mod;
    		if(p>ans) ans=p;
    	}
    	printf("Case #%lld: ",k++);
    	cout<<ans<<endl;
    }
    return 0;
}
