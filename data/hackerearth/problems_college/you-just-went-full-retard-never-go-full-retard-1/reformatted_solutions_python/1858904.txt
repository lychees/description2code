def isp(n):
    if n == 1 or n == 0:
        return False
    for i in xrange(2, int(n**.5)+1):
        if n % i == 0:
            return False
    return True


for _ in xrange(input()+1):
    n = raw_input()
    if n != "":
        if isp(int(n)):
            print "YES"
        else:
            print "NO"
