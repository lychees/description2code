#include <bits/stdc++.h>

using namespace std;

map<long, long long> mem;
long long MOD = 1e9 + 7;
long long modpowIter(long long a, long long b, long long c) {
    long long ans = 1;
    while (b != 0) {
        if (b % 2 == 1)
            ans = (ans * a) % c;

        a = (a * a) % c;
        b /= 2;
    }
    return ans;
}
long long split(long i) {
    if (i <= 3)
        return 1;
    if (mem.count(i))
        return mem[i];
    long im = i - 1;
    long long sum = 1;
    for (int j = 2; j <= sqrt(im); ++j) {
        if (im % j == 0) {
            sum += modpowIter(split(j),im/j,MOD);
            if(j*j!=im)sum += modpowIter(split(im/j),j,MOD);
            sum %= MOD;
        }
    }
    mem.insert(make_pair(i, sum));
    return mem[i];
}

int main(void) {
    long n;
    scanf("%ld", &n);
    cout << split(n) << endl;
}
