#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    int k[1002],n,s,i,min,max;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&k[i]);
    	}
    	sort(k,k+n);
    	max = 0;
    	min = INT_MAX;
    	for(i=0;i<n/2;i++)
    	{
    		s = k[i] + k[n-1-i];
    		if(max<s)
    		max = s;
    		if(min>s)
    		min = s;
    	}
    	printf("%d\n",max-min);
    }
    return 0;
}
