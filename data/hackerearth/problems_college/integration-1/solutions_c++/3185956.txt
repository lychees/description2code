#include <bits/stdc++.h>
using namespace std;
long long int n , p , u , r , s ;
int inc()
{
	int c= 0;
	while(n)
	{
		if( p + r <= u )
		{
			p += r ;
			n--;
			++c;
		}
		else break;
	}
	if( c ) return 1;
	else return 0;
}
int dec( )
{
	int c = 0 ;
    if( (p-s) >= 0 )
    {
    	p -= s ;
    	n--;
    	return 1 ;
    }
    else return 0;
}
int main()
{
	int t ;
	cin >> t ;
	for( int i = 1 ; i <= t ; i++ )
	{
		cin >> n >> p >> u >> r >> s ;
	    while( n )
	    {
	    	//cout << p << " " << n << endl;
	    	if( !inc() )
	    	{
	    		if( !dec() )
	    			break;
	    	}
	    }
	    printf("Case #%d: %lld\n",i , p );

	}
	return 0;
}
