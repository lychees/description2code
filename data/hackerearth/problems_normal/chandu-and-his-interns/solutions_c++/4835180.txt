#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	while(n--){
		long long int x,count=0,i;
		cin >>x;
		for(i=1;i*i<=x;i++){
			if(x%i==0){
				if(i*i==x)
				count++;
				else
				count+=2;
				if(count>=4)
				break;
			}
		}
		if(count>=4)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
