#include<stdio.h>
#include<iostream>
#include<stdlib.h>

using namespace std;


long int max(long int ,long int );
long int findLarge(long int[] ,long int );


long int findLarge(long int arr[],long int n ){



	long int val[n+1];
	val[0]=0;
	long int i,j,mv=0;
	
	for(i=1; i<n+1; i++){
		mv=0;
		for(j=0; j<i; j++){
			
			mv=max(mv,arr[j]+val[i-j-1]);
		
		}

		val[i]=mv;	
	}
	
	return val[n];

}


long int max(long int a, long int b){

	return (a>b)? a:b;

}


int main(){

	int t,i;
	long int n,j,val;
	long int *arr;
	
	
	
	scanf("%d",&t);
	for(i=0; i<t; i++){
		scanf("%ld",&n);
		arr=new long int[n];
		for(j=0; j<n; j++)
			scanf("%ld",&arr[j]);
		val=findLarge(arr,n);
		printf("%ld\n",val);
		delete(arr);
	}	




	return 0;
}
