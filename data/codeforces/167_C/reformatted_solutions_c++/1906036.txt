#include <algorithm>

const int N = 100010;
const int inf = 0x3f3f3f3f;

using namespace std;

typedef long long ll;

bool gcd(ll a, ll b) {
  if (a < b)
    swap(a, b);
  if (b == 0)
    return 0;
  if (!gcd(b, a % b))
    return 1;
  return a / b % (b + 1) % 2 == 0;
}

int main() {
  ll a, b;
  int ca;
  for (cin >> ca; ca--;) {
    cin >> a >> b;
    if (gcd(a, b))
      puts("First");
    else
      puts("Second");
  }
  return 0;
}
