#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

inline int fastread()
{
    int i=0;
    char c=getchar_unlocked();
    while(c<'0')
    c = getchar_unlocked();
    while(c>='0' && c<='9')
    {
        i = (i<<3)+(i<<1)+c-'0';
        c = getchar_unlocked();
    }
    return i;
}

int main()
{
    int t=fastread();
    while(t--)
    {
		long long int w=fastread(),n=fastread(),hgt=0,prev=0;
		for (int i=1;i<=n;i++)
		{
			if (fastread()==1) hgt=(2*i);
			else hgt=(prev+1);
			prev=hgt;
		}
		long long int ans=hgt+n*w;
		printf("%lld\n",ans);
    }
    return 0;
}
