#include <iostream>
#define ll long long int 
using namespace std;

ll f(int n,ll *a )
{  
	if(n==0)
	{
		return 1;
	}
	if(n<0)
	{
		return 0;
	}
	if(a[n]!=0)
	{
		return a[n];
	}
	ll b=0,c=0,d=0;
	    b=f(n-1,a);
		c=f(n-2,a);
	d=b+c+d;
	a[n]=d;
	return d;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	ll a[n];
    	for(int i=0;i<=n;i++)
    	{
    		a[i]=0;
    	}
    	cout<<f(n,a )<<endl;
    }
    return 0;
}
