double dp	[1242422];

	solve();

}

void solve()
{
	int n;double p;
	cin>>n>>p;	
	dp[0]=(double)1;
	for(int i=0;i<n;i++)
	{
		dp[i+2] += dp[i]*(p/(double)100);
		dp[i+3] += dp[i]*((double)1-(p/(double)100));
	}
	printf("%0.6lf\n",dp[n] );
	return;
}
