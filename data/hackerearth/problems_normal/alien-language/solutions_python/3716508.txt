noOfTestCases = int(raw_input())
def isPattern(s1, s2):
	#print s1, s2
	
	chrOfFirst = {}
	
	for c in s1:
		if(chrOfFirst.has_key(c) == False):
			chrOfFirst[c] = 1
		else:
			chrOfFirst[c] += 1
	atLeastOne = False
	
	for c in s2:
		if(chrOfFirst.has_key(c) == True):
			atLeastOne = True
	if(atLeastOne == True):
		print "YES"
	else:
		print "NO"

for caseNo in range(1, noOfTestCases + 1):
	s1 = raw_input()
	s2 = raw_input()
	isPattern(s1, s2)
