#include <bits/stdc++.h>
using namespace std;

long long fi(int n) 
     { 
       int result = n; 
       for(int i=2;i*i <= n;i++) 
       { 
         if (n % i == 0) result -= result / i; 
         while (n % i == 0) n /= i; 
       } 
       if (n > 1) result -= result / n; 
       return result; 
     } 

int main() {
	long long sieve[10001];
	for(int i=1;i<10001;++i)
		sieve[i] = fi(i);
	for(int i=2;i<10001;++i)
		sieve[i] += sieve[i-1];
	int t,n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		cout << sieve[n]-1 << endl;
	}
	return 0;
}
