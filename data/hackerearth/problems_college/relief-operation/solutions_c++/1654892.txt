#include <iostream>
using namespace std;
int max(int a,int b)
{
	 return a>b ? a :b;
}
int maxRelief(int w,int wt[],int val[],int n)
{
	int k[n+1][w+1];
	int i,j;
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=w;j++)
		{
			if(i==0 || j==0)
			  k[i][j]=0;
			else if(wt[i-1]<=j)
			 k[i][j]=max(val[i-1]+k[i-1][j-wt[i-1]],k[i-1][j]);
			else
			 k[i][j]=k[i-1][j];  
		}
	}
	return k[n][w];
}
int main()
{
    int t,m,l,wt[2001],w[2001],i;
    cin>>t;
    while(t--)
    {
    	cin>>m>>l;
    	for(i=0;i<m;i++)
    	  cin>>wt[i];
    	for(i=0;i<m;i++)
    	  cin>>w[i];
    	cout<<maxRelief(l,wt,w,m)<<endl;  
    }
    return 0;
}
