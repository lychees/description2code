#include <iostream>
using namespace std;

int main()
{
	int T, N, max;
	cin>>T;
	int arr[1001];
	for(int t=0;t<T;t++){
		cin>>N;
		for(int n=0;n<1001;n++){
			arr[n]=0;
		}
		for(int n=0;n<N;n++){
			cin>>max;
			arr[max]++;
		}
		long int Total=1;
		for(int n=1,i=0;n<1001;n++){
			if(arr[n]>0){
				max =arr[n];
				while(max>0){
					Total=(Total*(n-i))%1000000007;
					max--;
					i++;
				}	
			}
		}
		cout<<Total<<endl;
	}	
    return 0;
}
