#include <iostream>
#include<cmath>
using namespace std;
long long int check(long long int n)
{
  long long int i, flag=0;
  if(n==1)
  return 0;
  for(i=2;i<=sqrt(n);i++)
  {
      if(n%i==0)
      {
          flag=1;
          break;
      }
  }
  if (flag==0)
      return 1;
  else
     return 0;
}
int main()
{
  //  cout << "Hello World!" << endl;
    long long int t,n;
    cin>>t;
    while(t--)
    {
     cin>>n;
     long long int rem,sum=0,res;
     while(n>0)
     {
     	rem=n%10;
     	sum=sum+rem;
     	n=n/10;
     }
     res=check(sum);
     if(res==1)
     cout<<"YES"<<endl;
     else
     cout<<"NO"<<endl;
    }
     return 0; 
}
