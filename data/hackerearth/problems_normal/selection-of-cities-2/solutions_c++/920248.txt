#include <bits/stdc++.h>

using namespace std;

#define ull unsigned long long
#define mod 1000000007
ull fast_pow(ull a) 
{

	ull res = 1;
	ull y = 2;
	while (a) {
		if (a % 2 == 1) {
			res = (res*y)%mod;
		}
		y = (y*y)%mod;
		a = a>>1;
	}
	return res;
}
int main()
{
    int test;
    cin >> test;
    
    while (test--) {
    	
    	ull n;
    	cin >> n;
    	
    	ull ans = (fast_pow(n) - 1 + mod)%mod;
    	cout << ans << endl;
    }
    return 0;
}
