#include <iostream>
#include<math.h>
#include<stdio.h>
using namespace std;

int main()
{
    double x[501],y[501],z[501],vx[501],vy[501],vz[501],ttim;
    double vxx,vyy,vzz,xx,yy,zz,x0,y0,z0,v;
    long long t,n;
    cin>>t;
    for(int p=1;p<=t;p++)
    {
    	cin>>n;
    	vxx=0,vyy=0,vzz=0;
    	xx=0,yy=0,zz=0;
    	for(int i=0;i<n;i++)
    	{
    	cin>>x[i]>>y[i]>>z[i]>>vx[i]>>vy[i]>>vz[i];
    	vxx+=vx[i];
    	vyy+=vy[i];
    	vzz+=vz[i];
    	xx+=x[i];
    	yy+=y[i];
    	zz+=z[i];
    	
    	}
    	double min=10000000000,pre=1000000000000;
    	int tim=0;
    for(tim=0;;tim++){
    	vxx=0,vyy=0,vzz=0;
    	for(int i=0;i<n;i++)
    	{
    		
    		vxx+=x[i]+tim*vx[i];
    		vyy+=y[i]+tim*vy[i];
    		vzz+=z[i]+tim*vz[i];
    		
    	}
    	xx=(vxx*vxx)+(vyy*vyy)+(vzz*vzz);
    	
       	xx=sqrt(xx);
       	xx/=n;
       //cout<<xx<<"  ";
       	if(tim>0)
       	{
       		if(xx>=pre)
       		break;
       	}
       	ttim=tim;
       	pre=xx;
       
       
       	
    }
    	double tt=tim-1;
    	cout<<"Case #"<<(p)<<": ";
    	printf("%.8f %.8f\n",pre,tt);
  
    }
}
