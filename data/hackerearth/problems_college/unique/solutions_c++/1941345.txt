using namespace std;

#include <bits/stdc++.h>

long long int arr[100000];

int main() {
    int t;
    scanf("%d", &t);
    while(t--) {
        long long int n, i, result=1;
        scanf("%lld", &n);
        for(i=0; i<n; i++) {
            scanf("%lld", &arr[i]);
        }

        sort(arr, arr+n);

        for(i=0;i<n-1;i++) {
            if(arr[i]!=arr[i+1]) {
                result++;
            }
        }
        printf("%lld\n", result);
    }
    return 0;
}
