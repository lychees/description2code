#include<bits/stdc++.h>
using namespace std;
int t,n,k,q,m;

void dfs(int u,int c,int gr[1009][1009],int vis[])
{
	vis[u] = 1;
	for(int i=1;i<=n;i++)
	{
		if(gr[u][i]!=0 && !vis[i] && gr[u][i]<c)
		{
			dfs(i,c,gr,vis);
		}
	}
}
int gcd(int a,int b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}
int main()
{
	cin>>t;
	int a,b,c;
	int gr[1009][1009],vis[1009];
	while(t--)
	{
		cin>>n>>k;
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=n;j++)
				gr[i][j] = 0;
		}
		for(int i=1;i<=k;i++)
		{
			cin>>a>>b>>c;
			gr[a][b] = c;
			gr[b][a] = c;
		}
		int cnt = 0;
		cin>>q;
		for(int i=1;i<=q;i++)
		{
			cin>>a>>b;
			for(int j=1;j<=n;j++)
			{
				vis[j] = 0;
			}
			if(gr[a][b]!=0)
			{
				dfs(a,gr[a][b],gr,vis);
				if(vis[b]!=1)
					cnt++;
			}
			
		}
		int g=gcd(cnt,q);
		cout<<cnt/g<<"/"<<q/g<<endl;
	}
}
