#include <iostream>
#include <cstring>
#include <stdio.h>
using namespace std;

int main()
{
    int ch[26];
    int N;
    char in[100];
    cin>>N;
    for(int j=0;j<N;j++)
    {
    	fflush(stdin);
    	for(int i=0;i<26;i++)
    	{
    	ch[i]=0;
    	}
    	scanf("%s",in);
    	for(int i=0;i<strlen(in);i++)
    	{
    		ch[in[i]-97]+=1;
    	}
    	int i=0;
    	while(i<26&&ch[i]>0){i++;}
    	if(i==26)cout<<"YES"<<endl;
    	else cout<<"NO"<<endl;
    }
    return 0;
}
