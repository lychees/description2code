#include <iostream>
using namespace std;

int main()
{

		long test; cin>>test;
		for(int j=1;j<=test;j++)
		{
			long n; cin>>n;
			int c = 0;
			int sum =0;
			while(n>0)
			{
				sum+=n%10;
				n/=10;
			}
			if(sum==2)cout<<"YES"<<endl;
			else if (sum==1||sum==0)cout<<"NO"<<endl;
			else
			{
				for(int i = 2;i*i<=sum;i++)
				{
					if(sum%i==0)
					{
						c=1;
						break;
					}
				}
				if(c==1)cout<<"NO"<<endl;
				else cout<<"YES"<<endl;
			}
		}
    return 0;
}
