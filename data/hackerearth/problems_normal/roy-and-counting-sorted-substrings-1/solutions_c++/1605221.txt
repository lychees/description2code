#include <iostream>
#include <cstdio>
using namespace std;
char str[1000001];
int main()
{
    int t,n;
    long long int sol;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	scanf("%s",str);
    
    	sol = n ;
    	int count=0;
    	for(int i=1;i<n;++i)
    	{

    		
    		if(str[i]<str[i-1])
    		{
    			
    			sol += (((long long)count*(long long)(count+1))>>1);
    			count = 0 ;
    		}
    		else
    		{	
    			count++;
    		}
    	}
    	sol += (((long long)count*(long long)(count+1))>>1);

    	printf("%lld\n",sol);
    }
    
}
