#include <vector>
#include <map>
#include <iostream>
#include <fstream>
#include <algorithm>
#include <string>
using namespace std;

main()
{
	ifstream raj("pairsum.in");
	int ctr, sum, tmp;
	vector <int> num;

	cin >> ctr >> sum;
	for (int t = 0; t<ctr; t++)
	{
		cin >> tmp;
		num.push_back(tmp);
	}
	sort(num.rbegin(), num.rend());
	int dif = sum - num[0], x = 0;
	bool tama = false;
	//cout<<num[0]<<" . "<<dif<<endl;;

	while (!tama && dif <= num[x])
	{
		dif = sum - num[x];
		if (num[x]<sum){
			//cout<<sum<<" "<<num[x]<<" "<<dif<<endl;
			int finder = (find(num.begin() + (x + 1), num.end(), dif) != num.end());
			//cout<<finder<<endl;
			//cout<<num[finder]<<endl;
			if (finder>0 || (dif == 0))
			{
				tama = true; break;
			}
			else { x++; }
		}
		else x++;
	}
	if (tama) cout << "YES" << endl;
	else cout << "NO" << endl;
}
