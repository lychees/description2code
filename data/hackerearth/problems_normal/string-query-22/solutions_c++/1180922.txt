inline void inp(int &n ) {//fast input function
      n=0;
      int ch=getchar(),sign=1;
      while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getchar();}
      while( ch >= '0' && ch <= '9' )
          n=(n<<3)+(n<<1)+ ch-'0', ch=getchar();
      n=n*sign;
}

int main () {
  string str;
  cin >> str;
  vector <int> vec[26];
  int s = str.size();
  map<int,char> m;

  for (int i = 0; i < s; i++) {
    vec[str[i]-'a'].push_back(i);
    m[i] = str[i];
  }

  int q;
  inp(q);

  while (q--) {
    int k;
    char ch;
    inp(k); cin >> ch;
    int tmp = vec[ch-'a'][k-1];
    vec[ch-'a'].erase(vec[ch-'a'].begin()+k-1);
    m.erase(tmp);
  }
  map<int,char>::iterator it;
  string ans = "";
  for (it=m.begin(); it != m.end(); it++) {
    ans += (*it).second;
  }
  printf("%s\n",ans.c_str());
}
