#include<stdio.h>
#include<iostream>
#include<string.h>

using namespace std;

long long l,i,count;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
        string s;
        count=0;

		cin>>s;
		//l=strlen(s);

		for(i=0;s[i]!='\0';i++)
		{
			if((s[i]=='X'|| s[i]=='Y') && (s[i+1]==s[i]))
                count++;
		}
		printf("%lld\n",count);
	}
	return 0;
}
