#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
	int t;
	ll ans;
	double n,k;
    cin>>t;
    while(t--)
   	{
   		cin>>n>>k;
   		ll val=(floor(floor(n/2)/k));
   		if(val>0)
   			val=1;
   		else if(val==0)
   			val=0;
   		else
   			val=-1;
   			
   		ll v1=floor(n/(2*n-2*k+1));
   		ll v2;
   		if(v1==0)
   		v2=1;
   		else
   		v2=floor(log2(v1))+1;
   		ll v3=pow(2,v2);
   	//	cout<<"v1="<<v1<<" v2="<<v2<<" v3="<<v3<<endl;
   	//	cout<<"val="<<val<<endl;
   		ans=2*n+1-(2*n-2*k+1)*(v3-val);
   	//
   //	cout<<ans<<endl;
   		ans=(ans-1);
   		if(ans==0)
   			ans=n;
   			cout<<ans<<endl;
   	}
    return 0;
}
