#include <iostream>
#include <cmath>
#define ll long long
using namespace std;
bool isprime(ll x)
{
	if(x == 1)
	  return false;
	ll i,s = sqrt(x);  
	for (i= 2;i <= s;i++)
	  if(x % i == 0)
	    break;
	if(i == s + 1)
	  return true;
	return false;  
}	
int main()
{
    ll t,n,sum;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	sum = 0;
    	while(n)
    	{
    		sum += n % 10;
    		n /= 10;
    	}
    	if(isprime(sum))
    	  cout << "YES\n";
    	else
    	  cout << "NO\n";
    }
    return 0;
}
