#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,a;
    cin>>t;
    int arr[t];
    cin>>arr[0];
    for(int i=1;i<t;i++)
    {
    	cin>>a;
    	arr[i]=arr[i-1]+a;
    }
    
    int q;
    cin>>q;
    for(int i=1;i<=q;i++)
    {
    	cin>>a;
    	int l=0,h=t-1,m,flag=0;
    	while(l<=h)
    	{
    		m=l+(h-l)/2;
    		if(arr[m]==a)
    		{
    			flag=1;
    			break;
    		}
    		else if(arr[m]<a)
    		l=m+1;
    		else
    		{
    			if(m==0 || arr[m-1]<a)
    			{
    				flag=1;
    				break;
    			}
    			else
    			h=m-1;
    		}
    	}
    	if(flag==1)
    	cout<<m+1<<endl;
    	else 
    	cout<<-1<<endl;
    }
    return 0;
}
