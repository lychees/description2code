#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstdlib> 
using namespace std;
#define MAX_L 100000000000
#define LL long long
 

long long MAX = MAX_L;
int main()
{
    int t;
    cin>> t;
 
    while(t--)
    {
        int n, m, a;
        cin>> n>> m>> a;
 
 
        LL p = a*a;
        LL pp =  p* a;
        LL vol = n+m*pp;
        LL best = MAX_L;
 
        for(long r = a; r*r*r<=best; r++)
        {
            long R = r/a;
            for(long c = r; r*c *c<=best; c++)
            {
                LL C= c/a;
                LL h=(m+R*C-1)/(R*C)*a;
                
                LL V = r*c*h;
                LL h2 = max(0LL, vol- V);
                h +=(h2+r*c-1)/(r*c);
                best = min(best, r*c*h);
            }
        }
        cout << (int)best<< endl;
    }
 
    return 0;
}
