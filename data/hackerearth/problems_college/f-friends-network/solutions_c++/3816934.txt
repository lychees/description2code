#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
typedef long long ll;
ll a[505][505];

int main()
{
    ll t,n,i,k,j;
    ios_base::sync_with_stdio(false);
    cin>>t;
  while(t--)
  {
      cin>>n;
      memset(a,0,sizeof(a));
      for(i=1;i<=n;i++)
      {
          for(j=1;j<=n;j++)
          {
              cin>>a[i][j];
          }
      }
      ll flag=0;
      for(i=1;i<=n;i++)
      {
      	for(j=i+1;j<=n;j++)
      	{
      		for(k=j+1;k<=n;k++)
      		{
      		if(a[i][j] && a[j][k] && a[k][i])
      		{
      			flag=1;
      			break;
      		}
      		}
      		if(flag==1)
      		break;
      	}
      	if(flag==1)
      	break;
      }
    
      
      
      if(flag==1)
      cout<<"Aye"<<endl;
      else
      cout<<"Nae"<<endl;
  }
}
