#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

#define mod 1000000007
#define MX 100010
#define MX1 1001
#define get getchar//_unlocked

int main()
{
    int t, i, j, k;
    cin >> t;
    string s, p;
    int A[26] = {0};
    vector<pair<int, char> > v;
    while(t--) {
        v.clear();
        cin >> s;
        for(i = 0 ; i < 26; i++) {
            A[s[i] - 97] = i + 1;
        }
        cin >> p;
        for(i = 0; i < p.length(); i++) {
            v.pb(mp(A[p[i] - 97], p[i]));
        }
        sort(v.begin(), v.end());
        for(i = 0; i < v.size(); i++) {
            cout << v[i].second;
        }
        cout<<endl;
    }

    return 0;
}
