namespace test {
	void end_test() {
		int val;
		if (cin >> val) {
			exit(1);
		}
	}
	void range_test(int t, int l, int r) {
		if (t < l || r < t) {
			exit(1);
		}
	}
}
//o(n^2) solution
int n;
int a[102];
void all_different() {
	set<int> s;
	for (int i = 0;i < n;i++) {
		if (s.count(a[i])) {
			exit(1);
		}
		s.insert(a[i]);
	}
}
int main() {
	scanf("%d", &n);
	test::range_test(n, 1, 100);
	for (int i = 0;i < n;i++) {
		scanf("%d", &a[i]);
		test::range_test(n, 1, 1000000);
	}
	test::end_test();
	all_different();
	bool out = false;
	for (int i = 0;i < n;i++) {
		bool ok = false;
		for (int j = 0;j < n;j++) {
			if (i != j) {
				if (a[i] % a[j] == 0) {
					ok = true;
					break;
				}
			}
		}
		if (!ok) {
			if (out) {
				printf(" ");
			}
			out = true;
			printf("%d", a[i]);
		}
	}
	puts("");
	return 0;
}
