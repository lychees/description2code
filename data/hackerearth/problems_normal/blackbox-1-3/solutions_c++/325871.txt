#include<bits/stdc++.h>
using namespace std;


int gcd(int a, int b)
{
  if ( a==0 ) return b;
  return gcd( b%a, a );
}







main()

{
    int t;
    cin>>t;
    while(t--)
    {
        int a,b,c,flag=0;
        cin>>a>>b>>c;
        int k=gcd(a,b);
        int l=gcd(b,c);
        int m=gcd(c,a);
        int max=a;
        if(b>a)
            swap(a,b);
            if(c>a)
                swap(a,c);
            if((a*a)==((b*b)+(c*c)))
                flag=1;



        if(k==1 && l==1 && m==1 && flag==1)

        else cout<<"NO"<<endl;
    }
}
