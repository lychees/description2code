#include<bits/stdc++.h>
#define fr(i,a,b) for(i=a;i<b;i++)
#define ll long long

using namespace std;
int a[100005];
bool check(int mid,int k,int n)
{
    int i;int cur=1,prev=0;int move=0;
    fr(i,0,n)
    {
        while(cur!=n && a[cur]-a[prev]<=mid)
            cur++;
        move++;
        if(cur==n)
            break;
        prev=cur-1;
        
    }
    if(cur!=n)
        return false;
    if(move<=k)
        return true;
    else
        return false;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n,k;
        cin >> n >> k;int i;
        
        fr(i,0,n)
        {
            cin >> a[i];
        }
        int l=0;int ans;
        l=a[n-1];
        int low=0,high=l;
        while(low<=high)
        {
            int mid=(low+high)/2;
            if(check(mid,k,n))
            {
                ans=mid;
                high=mid-1;
            }
            else
                low=mid+1;
        }
        cout << ans << "\n";
    }
    return 0;
}
