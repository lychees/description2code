#include <iostream>
#include <cstring>
using namespace std;

int sum[52][52][52],ar[52][52][52];

int main() 
{
	int t,n,i,j,k;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			for(k=0;k<n;k++)
			cin>>sum[i][j][k];
		}
		
		ar[0][0][0]=sum[0][0][0];
		for(i=1;i<n;i++)
		{
			ar[i][0][0]=sum[i][0][0]-sum[i-1][0][0];
			ar[0][i][0]=sum[0][i][0]-sum[0][i-1][0];
			ar[0][0][i]=sum[0][0][i]-sum[0][0][i-1];
		}
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		{
			ar[i][j][0]=sum[i][j][0]-sum[i-1][j][0]-sum[i][j-1][0]+sum[i-1][j-1][0];
			ar[0][i][j]=sum[0][i][j]-sum[0][i-1][j]-sum[0][i][j-1]+sum[0][i-1][j-1];
			ar[i][0][j]=sum[i][0][j]-sum[i-1][0][j]-sum[i][0][j-1]+sum[i-1][0][j-1];
		}
		for(i=1;i<n;i++)
		for(j=1;j<n;j++)
		for(k=1;k<n;k++)
		{
			ar[i][j][k]=sum[i][j][k]-sum[i-1][j][k]-sum[i][j-1][k]-sum[i][j][k-1]+sum[i-1][j-1][k]
			+sum[i][j-1][k-1]+sum[i-1][j][k-1]-sum[i-1][j-1][k-1];
		}
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		{for(k=0;k<n;k++)
		cout<<ar[i][j][k]<<" ";
		cout<<"\n";}
	}
	return 0;
}
