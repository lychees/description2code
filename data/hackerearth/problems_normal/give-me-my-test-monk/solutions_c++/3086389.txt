#include <iostream>
#include<cstdlib>
#include<cmath>
using namespace std;

#define llu unsigned long long 
#define mod 1000000007


llu calc(llu a ,llu b , llu m)
{
	llu r=1;
	while(b)
	{
		if(b&1)
		 r=(r*a)%m;
		
		b=b>>1;
		a=(a*a)%m;
		
	}
	
	return r;
	
	
}
int main()
{
    llu n,l,r,x,i,j,*da,**memo,res,wa,nf;
    cin>>n>>l>>r;
    
    // constructing dearrangements table
    da=(llu *)malloc(sizeof(llu)*(n+1));
    da[0]=1;
    da[1]=0;
    for(i=2;i<=n;i++)
     {
     	da[i]=((i-1)*(da[i-1]+da[i-2]))%mod;
     }
     
    // constructing nCr table
     memo=(llu **)calloc(n+1,sizeof(llu *));
     for(i=0;i<n+1;i++)
      memo[i]=(llu *)calloc(n+1,sizeof(llu));
      
    for(i=0;i<=n;i++)
     memo[i][i]=1;
    
    for(i=1;i<=n;i++) 
     memo[i][0]=1;
    
    
    for(i=1;i<=n;i++)
    {
    	for(j=1;j<i;j++)
     	 {
     	 	
     	 	memo[i][j]=(memo[i-1][j]+memo[i-1][j-1])%mod;
     	 	
     	 }
    }
    
    res=0;
    nf=0;
   for(i=0;i<=n;i++)
     {
     	wa=(da[i]*memo[n][i])%mod;
     	if(i>=l&&i<=r)
     	 res=(res+wa)%mod;
      
      nf=(nf+wa)%mod;	
     	 
     }
     
    cout<<(res*calc(nf,mod-2,mod))%mod<<endl; 
    
    
   
    
    
     
     

     
    
    return 0;
}
