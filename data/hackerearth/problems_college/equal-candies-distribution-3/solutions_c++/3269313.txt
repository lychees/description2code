#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a[10000],t,n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		for(int i=0; i < n; i++)
		{
			cin >> a[i];
		}
		int mn=a[0];
		for(int i=1; i < n; i++)
		{
			if(a[i] < mn)
			mn=a[i];
		}
		int step=0,diff=0;
		for(int i=0; i < n; i++)
		{
			if(a[i] > mn)
			{
				diff=a[i]-mn;
				step+=(diff/5)+(diff%5)/2+(diff%5)%2;
			}
		}
		cout << step <<"\n";
	}
    //cout << "Hello World!" << endl;
    return 0;
}
