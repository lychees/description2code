#include <bits/stdc++.h>
using namespace std;

int main()
{
	long long int t,n;
	cin>>t;
	while(t--){
		cin>>n;
		long long int a[n];
	long long	int sum=0;
		for(int i=0;i<n;i++){
			cin>>a[i];
	sum+=a[i];
	}
		if(sum%n==0){
			cout<<"YES"<<"\n";
			cout<<sum/n<<" "<<"0"<<"\n";
		}else{
			cout<<"NO"<<"\n";
			cout<<sum/n<<" "<<sum%n<<"\n";
		}
	}
    return 0;
}
