#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	long long a,m=LONG_MAX,count;
    	for(int i=0;i<n;i++){
    		cin>>a;
    		count=0;
    		while(a){
    			a=a&(a-1);
    			count++;
    		}
    		if(count<m){
    			m=count;
    		}
    	}
    	cout<<m<<endl;
    }
    return 0;
}
