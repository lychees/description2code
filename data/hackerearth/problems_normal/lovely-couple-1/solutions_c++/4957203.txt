#include <iostream>
#include<math.h>
using namespace std;


int checkPrime( int n)
{
    int i;
    for(i=2;i<=sqrt(n);i++)
        if(n%i==0)
            break;

    if(i>sqrt(n))
        return 1;
    else
        return 0;

}

int prime( int n)
{
    int c=0;
    while (n%2 == 0)
    {
        c=1;
        n/=2;
    }

    for (int i = 3; i < n; i = i+2)
    {
        if(n%i==0)
        {
            c++;
        }
        while (n%i == 0)
        {
            n = n/i;
        }
    }

    if (n > 2)
        c++;
        
    return c;

}

int gcd(int a, int b)
{
    if (a == 0)
        return b;
    return gcd(b%a, a);
}
int main()
{
    int t,l;
    int a,b,j,c;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        c=0;
    	cin>>a>>b;

        l = a*b/gcd(a,b);
        
        c = prime(l);

    	if(c>1 && checkPrime(c))
    		cout<<"Yes\n";
    	else
    		cout<<"No\n";
    }
    return 0;
}
