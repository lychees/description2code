#include<iostream>
#include <math.h>
#include <stdlib.h>
using namespace std;

int main()
{
    int n,maxx,i,re;
    cin>>n;
    long int a,b;

    while(n--)
    {
        cin>>a>>b;
        maxx=max(log10(a)+1 ,log10(b)+1);
        int ar1[maxx],ar2[maxx];
        for(i=0;i<maxx;i++)
        {
            ar1[i]=ar2[i]=0;
        }
        i=0;
        while(a)
        {
            ar1[i]=a%10;
            a/=10;
            i++;
        }
        i=0;
        while(b)
        {
            ar2[i]=b%10;
            b/=10;
            i++;
        }
        re=0;
        for(i=0;i<maxx;i++)
        {
            if(ar1[i]+ar2[i] > 9)
            {
                re++;
            }
        }

        if(re==1)
        {
            cout<<"1 carry operation"<<endl;
        }
        else if(re==0)
        {
            cout<<"No carry operation"<<endl;
        }
        else{
            cout<<re<<" carry operations"<<endl;
        }
    }
}
