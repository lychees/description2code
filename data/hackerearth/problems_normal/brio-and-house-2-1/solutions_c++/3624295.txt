#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    for(cin>>t;t--;){
    	int x1,x2,y1,y2,r;
    	cin>>x1>>y1>>x2>>y2>>r;
    	double m,d;
    	d=abs(y1*(x2-x1)-(y2-y1)*x1)/sqrt((y2-y1)*(y2-y1)+(x2-x1)*(x2-x1));
    	if(d<r)
    	cout<<"REPLANNING"<<endl;
    	else if(d==r)
    	cout<<"JUST MISSED"<<endl;
    	else
    	cout<<"SAFE"<<endl;
    }
    return 0;
}
