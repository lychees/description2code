#include <iostream>
#include <bits/stdc++.h>
using namespace std;
typedef map<string,int> mp;
int main()
{
    int i,k=0,n,t;
    mp m;
    cin>>t;
    string j;
    while(t--)
    {
        string s;
        cin>>s>>n;
        m[s]+=n;
        if(m[s]>k)
        {
            k=max(m[s],k);
            j=s;
        }
    }
    cout<<j;
    return 0;
}
