#include <iostream>
using namespace std;

struct node {
	int a;
	int b;
};

node dp[1000001];

int main()
{
	dp[0].a = 1;
	dp[0].b = 0;
	dp[1].a = 1;
	dp[1].b = 0;
	dp[2].a = 2;
	dp[2].b = 1;
	
	for (int i = 3; i < 1000001; i++) {
		dp[i].a = dp[i - 2].a + dp[i - 3].a;
		
	}
	int n;
	cin >> n;
	cout << dp[n].a << endl;
    return 0;
}
