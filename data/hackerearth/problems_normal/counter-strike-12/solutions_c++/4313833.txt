#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--) {
		long long n, m, d;
		cin >> n >> m >> d;
		long long a[n][2], b[m][2], shot[m];
		
		for(int i = 0; i < n; i++) {
			cin >> a[i][0] >> a[i][1];
		}
		for(int i = 0; i < m; i++) {
			cin >> b[i][0] >> b[i][1];
			shot[i] = 0;
		}
		
		int flag = 0, count = 0;
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < m; j++) {
				long long md = abs(a[i][0] - b[j][0]) + abs(a[i][1] - b[j][1]);
				if(md <= d && !shot[j]) {
					count++;
					shot[j] = 1;
				}
			}
			if(count > m/2) {
				flag = 1;
				break;
			}
		}
		if(flag)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
    return 0;
}
