#include <iostream>
using namespace std;

int main()
{
    int t, a, b, ans, i,j;
    cin >> t;
    while(t--){
    	cin >> a >> b;
    	ans = 0;
    	for(i = a; i <= b; i++){
    		for(j = i ; j <= b; j++){
    			ans = max(ans, i ^ j);
    		}
    	}
    	cout << ans<< endl;
    }
    return 0;
}
