	char s[100];
	int hash[26]={0};
	int n,cnt=0,k;
	cin>>n>>k;
	for(int i=1;i<n+1;i++)
	{
		cin>>s;
		for(int j=0;s[j]!='\0';j++)
		{
			for(int k=j+1;s[k]!='\0';k++)
			{
				if(s[j]==s[k])
				s[k]='0';
			}
		}
		for(int j=0;s[j]!='\0';j++)
		{
			if(s[j]!='0')
			{
				int x=s[j]-'a';
				hash[x]++;
			}
		}
	}
	for(int i=0;i<26;i++)
	{
		if(hash[i]>=n)
		cnt++;
	}
	if(cnt>=k)
	cout<<":-)"<<endl;
	else
	cout<<":-("<<endl;
