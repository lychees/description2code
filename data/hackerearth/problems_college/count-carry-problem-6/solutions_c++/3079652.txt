#include<bits/stdc++.h>
using namespace std;
int main()
{
        int t;
        char str1[100],str2[100];
        cin >> t;
        while(t--)
        {
                int carry=0;
                scanf("%s",str1);
                scanf("%s",str2);
                int l1=strlen(str1);
                int l2=strlen(str2);
                int i=l1-1,j=l2-1,ans=0;
                while(i >=0 || j  >=0)
                {
                        if(carry)
                         ans++;
                        int val=(i >=0 ? str1[i]-'0' : 0)+(j>=0 ? str2[j]-'0': 0)+ carry;
                        carry=val/10;
                        i--,j--;

                }
                if(carry)
                        ans++;
                //cout << ans <<"\n";
                if(ans==0)
                        cout <<"No carry operation\n";
                else
                {
                        if(ans==1)
                        cout << ans << " carry operation\n";
                        else
                         cout << ans << " carry operations\n";
                }

        }
}
