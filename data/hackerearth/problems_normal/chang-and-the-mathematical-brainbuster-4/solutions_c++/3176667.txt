    #include <bits/stdc++.h>
    #include <algorithm>
     
    using namespace std;
     
    int a [100001] = {0};
     
    int main ()
    {
    	int i, count, j, k, n, mini, flag;
    	
    	scanf ("%d", &n);
    	
    	for (i = 0; i < n; i++) {
    		scanf ("%d", & a [i]);
    	}
    	
    	sort (a, a + n);
    	flag = 0;
    	
    	mini = 100001;
    	for (i = n - 1; i >= 0; i--) {
    		flag = 0;
    		for (j = 0; j < i; j++) {
    			if ((a [i]) - 3 * a [j] <= 0) {
    				flag = 1;
    				break;
    			}
    		}
    		if (flag)
    		mini = min (mini, n - 1 - i + j);
    	}
    	
    	printf ("%d\n", mini);
    	
    	return 0;
    }
