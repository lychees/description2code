#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll tc,n,k,i,j,temp,mn,mx;
vector<ll> t,tme;
vector<ll>::iterator it1,it2;
int main()
{
    scanf("%lld",&tc);
    while(tc--)
    {
        t.clear();
        tme.clear();
        scanf("%lld %lld",&n,&k);
        t.resize(n);
        tme.resize(n/2);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&temp);
            t[i]=temp;
        }
        sort(t.begin(),t.end());
        mx=0;mn=INT_MAX;
        for(i=0;i<n/2;i++)
            tme[i]=t[i]+t[n-i-1];
        sort(tme.begin(),tme.end());
        mx=tme[(n/2)-1];
        mn=*tme.begin();
        if(n==2)
        {
            printf("0\nChick magnet Jhool!\n");
            continue;
        }
        if((mx-mn)>k)
            printf("%lld\nNo more girlfriends!\n",mx-mn);
        else if((mx-mn)==k)
            printf("%lld\nLucky chap!\n",mx-mn);
        else
            printf("%lld\nChick magnet Jhool!\n",mx-mn);
    }
}
