#include <iostream>
#include <string>
#include <cstring>
using namespace std;

int main()
{
   string name;
    cin >> name;
    char name1[200];
    strcpy(name1,name.c_str());
    for(int i=0; i<name.length();i++){
        if(name1[i] != '\0'){
            if(name1[i] > 96){
                name1[i] -= 32;
                cout << name1[i];
            }
            else if(name1[i] > 64 && name1[i] < 91){
                name1[i] += 32;
                cout << name1[i];
            }
        }
    }
    return 0;
}
