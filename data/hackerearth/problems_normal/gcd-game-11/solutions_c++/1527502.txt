#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
int dp[1003][1002][2],GCD[1002][1002];

int solve(int a,int b,int flag)
{
	//cout<<"a="<<a<<"b="<<b<<"f="<<flag<<endl;
	if(a==1 && b==1) return 2;
	if(a==1 && b!=1) return 0;
	if(a!=1 && b==1) return 1;
	if(dp[a][b][flag]!=-1) return dp[a][b][flag];
	int x,y,res;
	if(!flag)
	{
		x=solve(a,b-1,!flag);
		if(GCD[a][b]>1)
		{
			y=solve(a,b/GCD[a][b],!flag);
			if(x==1) res=x;
			else if(y==1) res=y;
			else if(x==2 || y==2) res=2;
			else res=0;
		}
		else res=x;		
	}
	else
	{
		x=solve(a-1,b,!flag);
		if(GCD[a][b]>1)
		{
			y=solve(a/GCD[a][b],b,!flag);
			if(x==0) res=x;
			else if(y==0) res=y;
			else if(x==2 || y==2) res=2;
			else res=1;
		}	
		else res=x;
	}
	return dp[a][b][flag]=res;
}	
int main()
{
	int a,b,t;
	for(int i=1;i<=1000;i++)
		for(int j=1;j<=i;j++)
			GCD[i][j]=GCD[j][i]=__gcd(i,j);
	cin>>t;
	while(t--)
	{
		memset(dp,-1,sizeof(dp));
		cin>>a>>b;
		if(solve(a,b,0)==2)
			cout<<"Draw"<<endl;
		else if(solve(a,b,0)==1)
			cout<<"Arjit\n";
		else cout<<"Chandu Don\n";	
	}
	return 0;
}
