#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	long long int a[100001],n,k,max,x;
	int i,j,t;
	scanf("%d",&t);
	for(i=1;i<=t;i++)
	{
		scanf("%lld%lld",&n,&k);
		for(j=0;j<n;j++)
		{
			scanf("%lld",&a[j]);	
			a[j]=k/a[j];
		}	
		max=0;
		for(j=0;j<n;j++)
		{
			scanf("%lld",&x);	
			x=x*a[j];
			if(x>max)
				max=x;
		}
		printf("%lld\n",max);
	} 
}
