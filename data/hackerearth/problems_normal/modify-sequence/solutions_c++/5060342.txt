#include <iostream>
using namespace std;

int main()
{
	int N,Flag = 0;
	cin>>N;
	int a[N];
	for(int i=0;i<N;i++)
    {
    	cin>>a[i];
    }
    for(int i=0;i<N-1;i++)
    {
    	if(a[i]<=a[i+1])
    	{
    		a[i+1] = a[i+1] - a[i];
    		a[i] = 0;
    	}
    	else
    	{
    		Flag = 1;
    		break;
    	}
    		
    }
    if(Flag == 1)
    {
    	cout<<"NO";
    }
    else
    {
    	for (int i = 0; i < N; ++i)
    	{
    		/* code */
    		if(a[i]!=0)
    		{
    			Flag = 1;
    			break;
    		}
    	}
    	if(Flag == 1)
    	{	
    	cout<<"NO";
   	 	}
   	 	else
    	cout<<"YES";
	}
    return 0;
}
