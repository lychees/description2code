#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test,n,k,x;
	vector<int> vec;

	cin >> test;
	while(test--)
	{
		cin >> n >> k;
		vector<int> sum(n+1,0) ;
		for(int i=1;i<=n;i++)
		{
			cin >> x;
			vec.push_back(x);
		}

		int end = n,counter=0;
		while(counter<=k)
		{
	//		cout << sum[end-counter]<<" "<< vec[end-(counter+1)] << endl;;
			sum[end-(counter+1)] = max(sum[end-counter],vec[end-(counter+1)]);
			++counter;
		}
		end = n-1;
		for(int i=end-(k+1);i>=0;i--)
			sum[i] = max(sum[i+k+1],max(sum[i+1],sum[i+k+1]+vec[i]));
		cout << sum[0]<<endl;
		sum.clear();
		vec.clear();
		

	}
	return 0;
}
