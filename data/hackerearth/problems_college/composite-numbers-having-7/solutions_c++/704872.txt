#include<cstdio>
#include<algorithm>
using namespace std;
#define M 1000001
#define Int long long
int primes[M];
void sieve()
{
   unsigned long long int i,j,count=0;
  for (i=2;i<M;i++)
        primes[i]=1;

    for (i=2;i<M;i++)
        {
            if (primes[i])
            {
                for (j=i*i;j<M;j+=i)
                primes[j]=0;
            }
        }
}
int count(int n)
{
    while(n)
    {
        if(n%10==7)return 1;
        n=n/10;
    }
    return 0;
}
int main()
{
    sieve();
    int t;
    scanf("%d",&t);getchar();
    while(t--)
    {
        int nn,mm,sum=0;
        scanf("%d %d",&nn,&mm);
        for(int i=nn;i<=mm;i++)
        {
            if(!primes[i])
            {
                sum+=count(i);
            }
        }
        if(sum!=0)printf("%d\n",sum);
        else printf("-1\n");
    }
    return 0;
}
