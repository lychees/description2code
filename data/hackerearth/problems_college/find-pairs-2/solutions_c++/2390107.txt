#include<iostream>
#include<string>
using namespace std;
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        int n;
        cin>>n;
        int arr[2000090];
        for(int i=0;i<2000090;i++){
            arr[i]=0;
        }
        int ans=0;
        int temp;
        for(int i=0;i<n;i++){
            cin>>temp;
            temp+=1000000;
            ans++;
            ans+=arr[temp];
            arr[temp]++;
        }
        cout<<ans<<endl;
    }


    return 0;

}
