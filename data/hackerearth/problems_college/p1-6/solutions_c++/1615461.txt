#include<iostream>
#include<cstring>
using namespace std;

char str[1000007];
int main()
{long long int ans,len,ctr,one,i;

    cin>>str;
    len=strlen(str);
    ctr=0;
    one=0;
    ans=0;
    for(i=len-1;i>=0;i--)
    {
        if(str[i]=='0')
        {
            ctr=ctr+one;
        }
        else
        {one++;
          ans=ans+ctr;
        }

    }
    cout<<ans<<endl;

    return 0;

}
