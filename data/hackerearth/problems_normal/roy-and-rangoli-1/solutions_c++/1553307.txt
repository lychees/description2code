#include <iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define mod 1000000007
bool prime[2000005]={0};
int main()
{
	int i,j,n,l,ans;
    for(i=2;i<=sqrt(2000004);i++)
    {
    	if(!prime[i])
    	{
    	    for(j=i*i;j<=2000004;j+=i)
    	    {
    	    	prime[j]=1;
    	    }
    	}
    }
    scanf("%d",&n);
    l=2*(n-1);
    ans=0;
    for(i=2;i<=l;i++)
    {
    	if(!prime[i])
    	{
    		if(i<=l/2)
    		ans=(ans+i+1)%mod;
    		else
    		ans=(ans+l-i+1)%mod;
    	}
    }
    printf("%d\n",ans);
    return 0;
}
