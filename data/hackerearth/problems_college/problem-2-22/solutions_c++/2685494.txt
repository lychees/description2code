#include <cstring>

#include <cmath>
using namespace std;



int main()
{
	int T;
	cin >> T;
	for (; T--;) {
		int r, xc, yc;
		cin >> r >> xc >> yc;
		bool ok = 1;
		for (int i = 0; i < 4; ++i) {
			int x, y;
			cin >> x >> y;
			if (1ll * (x - xc)*(x - xc) + (y - yc)*(y - yc) > r*r)
				ok = 0;
		}
		cout << (ok ? "Yes" : "No") << endl;
	}


	return 0;
}
