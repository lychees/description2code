int main(){
	int t;
	cin >> t;
	while(t--){
		int n;
		cin >> n;
		int a[n];
		int total=0;
		for(int i=0;i<n;i++){
			cin >> a[i];
			total+=a[i];
		}
		int b[total+1];
		for(int i=1;i<=total;i++)
			b[i]=-1;
		b[0]=0;
		for(int i=0;i<n;i++){
			for(int j=total;j>=0;j--){
				if(b[j]!=-1){
					b[j+a[i]]=0;
				}
			}
		}
		int ans=0;
		for(int i=0;i<=total;i++){
			if(b[i]!=-1)
				ans++;
		}
		cout << ans << endl;
	}


		return 0;
	}
