#include <iostream>
using namespace std;

int main()
{  int a[100],n,j=0,e=0,k=0,t,i=0;
   cin>>t;
   
   while(i<t)
   { k=0;
     e=0;
   	 cin>>n;
   	 for(j=0;j<n;j++)
     cin>>a[j];
     for(j=0;j<n,e<n;e++,j=a[j]) 
      {if(a[j]==-1) k=1;
       else if(a[j]==j) break;      
      }
     if(k==1) cout<<"Win"<<endl;
     if(k==0) cout<<"Lose"<<endl;
       
     i++;
    }
   return 0;
}
