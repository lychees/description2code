#include <iostream>
using namespace std;

int main()
{int t;
cin>>t;
while(t--)
{   int flag=0;
	string s;
	cin>>s;
	for(int i=0;i<s.length();i++)
	{
		if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='A'||s[i]=='I'||s[i]=='O'||s[i]=='U'||s[i]=='E')
		{
		flag=1;
		cout<<s[i];
	}
	}
	
	if(flag==0)
   cout<<"No"<<endl;
   else
   cout<<endl;
}
   
    return 0;
}
