#include <iostream>
using namespace std;

int main()
{
    long long int t,a,b,c;
   cin>>t;
   while(t--)
   {
   	cin>>a>>b>>c;
   	int flag=0;
   	if(a+b>c && b+c>a && c+a>b){
   	if(a==b && b==c )
   	{
   		cout<<"YES. It is a equilateral triangle."<<endl;
   	}
   	else if(a!=b && b!=c && c!=a)
   	{
   		cout<<"YES. It is a scalene triangle."<<endl;
   	}
   	else
   	{
   		cout<<"YES. It is a isosceles triangle."<<endl;
   	}
   	}
   	else
   	{
   		cout<<"NO"<<endl;
   	}
   }
   
    return 0;
}
