#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		int i,len = str.length();
		if(len%2==0)
		{
			int sum=0;
			for(i=0;i<len/2;i++)
			{
				if(str[i]>str[len-i-1] || str[i]<str[len-i-1])
				{
					sum = sum + abs(str[i]-str[len-i-1]);
				}
			}
			cout<<sum<<"\n";
		}
		else
		{
			int sum =0;
			for(i=0;i<len/2;i++)
			{
				if(str[i]>str[len-i-1] || str[i]<str[len-i-1])
				{
					sum = sum + abs(str[i]-str[len-i-1]);
				}
			}
			cout<<sum<<"\n";
		}
	}
}
