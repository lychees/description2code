import sys
hashmap = {}


def count(n):
    if n <= 5:
        return n
    elif n in hashmap.keys():
        return hashmap[n]
    else:
        k = count(n / 2) + count(n / 3) + count(n / 4)
        if k > n:
            hashmap[n] = k
            return k
        else:
            hashmap[n] = n
            return n


for line in sys.stdin:
    result = count(int(line))
    print(result)
