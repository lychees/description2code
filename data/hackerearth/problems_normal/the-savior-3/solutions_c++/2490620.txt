#include <iostream>
#include <string>
#include <cmath>
using namespace std;

int main()
{
    int n, t;
    cin >> t;
    for(int k=0; k<t;k++)
    {
        cin >> n;
        int a[n], ans=0;
        for(int l=0; l<n;l++)
        {
            cin >> a[l];
        }
        for(int i=0;i<n-1;i++)
        {
            for(int j=i+1; j<n;j++)
            {
                //cout << a[i]<<":"<<a[j]<<endl;
                if((a[i]+a[j])%2==0 && a[i]!=a[j])
                {
                    ans=ans+1;
                }
            }
        }
        cout << ans<<endl;
    }
    return 0;
}
