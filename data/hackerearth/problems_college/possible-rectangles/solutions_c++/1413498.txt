#include<bits/stdc++.h>
using namespace std;
long long fast_int()
{
        long long n = 0;
        char c = getchar_unlocked();
        while (!('0' <= c && c <= '9'))
        {
                c = getchar_unlocked(); 
        }
        while ('0' <= c && c <= '9')
        {
                n = (n<<3)+(n<<1) + c - '0';
                c = getchar_unlocked();
        }
        return n;
}
void fast_write(long long n)
{
    long long temp,zero=0;
    temp=n;
    if(n==0)
    { 
        putchar_unlocked('0');
        putchar_unlocked('\n');
        return;
    }
    while((temp%10)==0)
    { 
        zero++; 
        temp/=10;
    }
    temp=0;
    while(n) 
    { 
        temp=(temp<<3)+(temp<<1)+ n%10; 
        n/=10;
    } 
    while(temp) 
    { 
        putchar_unlocked(temp%10+'0'); 
        temp/=10;
    }
    while(zero--) 
        putchar_unlocked('0');
    putchar_unlocked('\n');
}
int main()
{

	int t;
	t=fast_int();
	while(t--)
	{
		long long m,n,a;
		m=fast_int();
		n=fast_int();
		a=fast_int();
		long long i,ans=0;
		for(i=1;i*i<=a;i++)
		{
		    if(a%i!=0)
		        continue;
		    long long one=i;
		    long long two=a/i;
		    if(m>one && n>two)
		    ans+=((m-one)*(n-two));
		  
		    if(m>two && n>one && one!=two)
		    ans+=((m-two)*(n-one));
		    
		}    
		fast_write(ans);
	}
	return 0;
}
