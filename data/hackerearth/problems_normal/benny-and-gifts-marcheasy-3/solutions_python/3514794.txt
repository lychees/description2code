x, y = 0, 0
visited = set()
S = raw_input()
falls = 0

for s in S:
    visited.add((x,y))

    if s == 'L':
        y -= 1
    if s == 'R':
        y += 1
    if s == 'U':
        x -= 1
    if s == 'D':
        x += 1
    if (x,y) in visited:
        falls += 1

print falls
