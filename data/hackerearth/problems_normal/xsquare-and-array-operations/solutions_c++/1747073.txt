#include <bits/stdc++.h>

using namespace std;

namespace stuff {
typedef long long ll;
const int MOD = int(1e9) + 7;

void solve(int test_num) {
  int N;
  cin >> N;
  vector<int> a(N);
  for (int i = 0; i < N; i++)
    cin >> a[i];
  ll res = 0;
  for (int i = 1; i < N; i++)
    res += max(a[i - 1], a[i]);
  cout << res << endl;
}
