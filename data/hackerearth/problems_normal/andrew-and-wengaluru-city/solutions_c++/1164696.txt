#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007

void solve(){
	int n;
	cin>>n;
	long long a[n], fm[n], bm[n];
	for(int i = 0; i<n; i++) cin>>a[i];
	fm[0] = a[0];
	for(int i = 1; i<n; i++){
		if(a[i]>fm[i-1]) fm[i] = a[i];
		else fm[i] = fm[i-1];
	}
	bm[n-1] = a[n-1];
	for(int i = n-2; i>=0; i--){
		if(a[i]>bm[i+1]) bm[i] = a[i];
		else bm[i] = bm[i+1];
	}
	long long ans = 0;
	for(int i = 0; i<n; i++){
		ans += min(fm[i],bm[i]) - a[i];
		ans %= mod;
	}
	cout<<ans<<endl;
}

int main(){
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
