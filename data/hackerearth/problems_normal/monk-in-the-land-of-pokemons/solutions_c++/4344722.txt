#include<bits/stdc++.h>
using namespace std;
int t,n,a,b,ans,v[1000001],i;
int main()
{
    cin>>t;
    while(t--)
    {
        cin>>n;
        while(n--)
        {
            cin>>a>>b;
            if(a!=b)
            {
                if(v[b]>=1)
                    v[b]--;
                else
                    ans++;
                v[a]++;
            }
        }
        cout<<ans<<endl;
        ans=0;
        for(i=0;i<=1000000;i++)
            v[i]=0;
    }
}
