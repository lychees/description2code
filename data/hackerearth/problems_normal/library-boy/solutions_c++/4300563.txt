#include <string>
#include <iostream>
using namespace std;
int main()
{

    //97
    int x;
    cin >> x;
    int b[26];
    for(int d=0;d<26;d++)
    {

        b[d]=0;
    }
    while(x--)
    {
        string a;
        cin >> a;
        b[a[0]-97]++;

    }
    int t=0;
    for(int d=0;d<26;d++)
    {
        if(b[d]!=0){
        if(b[d]<=10)
        {

            t++;
        }
        else
        {

            if(b[d]%10==0)
                
            {
                
                t+=b[d]/10;
            }
            else 
            {
                
                t+=b[d]/10+1;
            }
        }
        }

    }
    cout << t << endl;
}
