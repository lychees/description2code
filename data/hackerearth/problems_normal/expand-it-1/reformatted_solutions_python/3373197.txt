input_string = raw_input()
t = input()
l_str = []
x = {}
repetitions = ""
character = ""
for ch in input_string:
    if ch in "abcdefghijklmnopqrstuvwxyz":
        if not character == "":
            l_str.append((character, int(repetitions)))
            x[character] = x.get(character, 0) + int(repetitions)
        character = ch
        repetitions = ""
    else:
        repetitions += ch
l_str.append((character, int(repetitions)))
x[character] = x.get(character, 0) + int(repetitions)
sk = sorted(x.keys())
for _ in xrange(t):
    k = input()
    i = 0
    while k > 0 and i < (len(sk)):
        k -= x[sk[i]]
        i += 1
    if k <= 0 and i <= (len(l_str)):
        print sk[i-1]
    else:
        print -1
