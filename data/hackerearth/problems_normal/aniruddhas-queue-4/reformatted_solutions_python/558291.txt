t = int(raw_input())
for _ in range(t):
    n = int(raw_input())
    x = map(int, raw_input().split())
    m = int(raw_input())
    if m == 0:
        print 1
        continue
    s = sum(x)
    l = len(x)
    if m > s:
        temp = m % s
        if temp == 0:
            m = s
        else:
            m = temp
    t = 0
    for j in range(l):
        t += x[j]
        if t >= m:
            print j+1
            break
