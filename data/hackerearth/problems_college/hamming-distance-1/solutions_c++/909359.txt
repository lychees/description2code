#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    cout<<"\n";
    while(t--)
    {
        unsigned int x,y;
        unsigned int dist,hm=0;
        cin>>x;
        cin>>y;
        dist=x^y;
       // cout<<dist;
        while(dist)
        {
            hm=hm+1;
            dist=dist& (dist-1);
        }
        cout<<hm<<"\n";
    }
}
