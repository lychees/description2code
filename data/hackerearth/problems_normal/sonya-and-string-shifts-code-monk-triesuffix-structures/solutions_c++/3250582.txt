#include<stdio.h>
#include<stdlib.h>

char *s;
int *res;
int n,smallest,q,x;

int main()
{
    scanf("%d",&n);
    s = new char[n+n];
    res = new int[n];
    scanf("%s",s);

    for(int i=0;i<n;++i)
    {
        s[n+i]=s[i];
    }
    //s[n+n] = '\0';
    res[0]=0;
    smallest=0;
    for(int i=1;i<n;++i)
    {
        for(int j=0;j<n;++j)
        {
            if(s[smallest+j] > s[i+j])
            {
                smallest = i;
                break;
            }
            else if(s[smallest+j] < s[i+j])
                break;
        }
        res[i]=smallest;
    }

    scanf("%d",&q);

    while(q--)
    {
        scanf("%d",&x);
        smallest = (x>res[x])?res[x]:x;
        printf("%d\n",smallest);
    }
    return 0;
}
