#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll modpow(ll a,ll b)
{
	ll res=1;
	while(b)
	{
		if(b&1)
		 res=(res*a)%1000000007;
		 
		 a=(a*a)%1000000007;
		 b>>=1;
	}
	return res;
}
int main()
{
	ll n,i,val,xorval=0,ans=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>val;
		xorval=xorval^val;
	}
	if(xorval!=0)
	 cout<<"0"<<endl;
	 else
	 {
	 	ans=modpow(2,n-1);
	 	cout<<ans-1<<endl;
	 }
	
	
}
