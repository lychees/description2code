#include <bits/stdc++.h>
using namespace std;

long int modu(long int a)
{
	if (a<0)return -a;
	return a;
}

int main()
{
	long int t, tmp1, tmp2, diff=INT_MIN, dif=INT_MIN;
	cin>>t;
	tmp1=0;
	for (int i=0; i<t; i++)
	{
		cin>>tmp2;
		diff = max(modu(tmp2-tmp1), diff);
		tmp1=tmp2;
	}
	tmp1=0;
	for (int i=0; i<t; i++)
	{
		cin>>tmp2;
		dif = max(modu(tmp2-tmp1), dif);
		tmp1=tmp2;
	}
	string ans = (diff>dif)?"Dom":"Brian";
	diff = (diff>dif)?diff:dif;
	cout<<ans<<endl<<diff<<endl;
    return 0;
}
