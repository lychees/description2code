#include <iostream>
#include<cmath>
using namespace std;

int main() 
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long int a[n],i,r=0,l=0;
		for(i=0;i<n-1;i++)
		{
			cin>>a[i];
			l+=a[i];
		}
		cin>>a[n-1];
		r=a[n-1];
		long int j,min;
		min=abs(l-r);
		for(i=n-2;i>0;i--)
		{
			r+=a[i];
			l-=a[i];
			j=abs(l-r);
			if(j<min)
				min=j;
		}
		cout<<min<<endl;
	}
	return 0;
}
