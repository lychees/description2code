#include<iostream>

using namespace std;
#include<math.h>
#include<stdio.h>
#include<set>
int main()
 {
 	int t,m,nn,k,i,j,n,kk;
 	scanf("%d",&t);
 	while(t--)
 	  { //map<pair<int,int>,int > mp;
	   double mp[52][10501]={{0}};
 	   double num=0,den=0;
 	    mp[0][0]=1.0;
 	  	scanf("%d%d%d",&m,&n,&kk);
 	  	set<int> a;
 	  	set<int> b;
 	  	a.insert(0);
 	  	for(i=1;i<=m;i++)
 	  	  {
 	  	  	for(set<int>::iterator it=a.begin();it!=a.end();it++)
 	  	  	 {k=*it;
 	  	  	   if(mp[i-1][k])
 	  	  	  	   {for(j=1;j<=n;j++)
 	  	  	  	      {
 	  	  	  	      mp[i][k+j]+=mp[i-1][k];
 	  	  	  	      b.insert(k+j);
 	  	  	  	      }
 	  	  	       }
						 // mp[make_pair(i,(it->first).second+j)]+=it->second;
 	  	  	  }
 	  	  	  a.clear();
 	  	  	  a.swap(b);
 	  	  	  //b.clear();
 	  	  }
 	  	for(set<int>::iterator it=a.begin();it!=a.end();it++)
 	  	   {k=*it;
			  if(mp[m][k])
 	  	      den+=mp[m][k];
 	  	   }
 	  	if(kk<=m*n && kk>=m)
 	  	num=mp[m][kk];
 	  	else
 	  	 num=0;
 	  	double ans=num*1.0/den;
 	  	int count=0;
 	  while(ans<1 && ans!=0.0)
 	  	   {ans*=10;
 	  	    count++;
 	  	   }
 	  	printf("%0.3lf %d\n",ans,count);
	  
 	  }
 }
