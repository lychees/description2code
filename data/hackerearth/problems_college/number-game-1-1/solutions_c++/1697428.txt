#include <iostream>
using namespace std;

int main()
{
    int T;
    long long N,count;
    cin>>T;
    while(T--)
    {
    	cin>>N;
    	count=0;
    	while(N!=1)
    	{
    		if(N%3==0)
    		{
    			N=N/3;
    			count++;
    		}
    		else if(N%2==0)
    		{
    			N=N/2;
    			count++;
    		}
    		else
    		{
    			N--;
    			count++;
    		}
    			
    	}
    	cout<<count<<endl;
    }
    return 0;
}
