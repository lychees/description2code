t = int(raw_input())
while t:
    t -= 1
    n = int(raw_input())
    l = [int(x) for x in raw_input().split()]
    count = 0
    for i in range(n):
        for j in range(i, n):
            # print l[i],l[j]
            if l[i] != l[j] and (l[i]+l[j]) % 2 == 0:
                count += 1
    print count
