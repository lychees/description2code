#include <iostream>
#include <vector>
#include <string>
#include <queue>
#include <algorithm>
#include <cstdio>

using namespace std;



int main(){
	int T;
	cin >> T;
	static int N = 5*1000000+1;
	vector<bool> P(N, true);
	P[0] = P[1] = false;
	for(int i=2; i<N; i++){
		if(P[i] == false) continue;
		for(int j=i+i; j<N; j+=i){
			P[j] = false;
		}
	}

	vector<long long> val(N);
	for(int i=1; i<N; i++){
		val[i] = (P[i]?-1:1) * i;
	}
	
	for(int t=0; t<T; t++){
		int x,y;
		scanf("%d%d", &x, &y);

		if(x==y){
			printf("%d\n", val[x]);
			continue;
		}
		
		long long ans = val[x] + val[y];
		
		while(x!=y){
			if(x>y){
				x = (x>>1);
				if(x!=y) ans += val[x];
			}else{
				y = (y>>1);
				if(x!=y) ans += val[y];
			}
		}
		printf("%lld\n", ans);
	}
	return 0;
}
