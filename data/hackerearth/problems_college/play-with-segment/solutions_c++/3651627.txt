#include<bits/stdc++.h>
using namespace std;
static int f[1000001],a[500001];
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		for(int i=0;i<=1000000;i++)
		    f[i]=0;
		int n,ma=0,mi=INT_MAX,s=0,x,ans=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++){
		scanf("%d",&x);
		a[i]=x;
		f[x]++;
		ma=max(ma,x);
		mi=min(mi,x);
		if(i==0)
		    continue;
		 //ma=max(ma,x);
		 //mi=min(mi,x);
		 while(s<=i&&(ma-mi)>1&&(f[mi]>0&&f[ma]>0)){
		 //	printf("i=%d\n",i);
		 	f[a[s]]--;s++;
		 	if(f[mi]==0){
		 		mi=mi+1;break;
		 	} else
		 	if(f[ma]==0){
		 		ma--;break;
		 	}
		 }
		 ans=max(ans,i-s+1);
		}
		printf("%d\n",ans);
	}
	return 0;
}
