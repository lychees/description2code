#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int k,n;
    	cin >> k >> n;
    	if(k > n) {
    		cout << "1.000000000\n";
    		continue;
    	}
    	double ans = 1.0;
    	for(int i=0;i<k;i++) {
    		ans *= (double)(n-i)/(double)n;
    	}
    	
    	printf("%.9lf\n",1.0 - ans);
    }
}
