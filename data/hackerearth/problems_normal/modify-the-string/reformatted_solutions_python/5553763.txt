input_string = raw_input()
output_string = ''
for i in range(len(input_string)):
    if input_string[i] >= 'a' and input_string[i] <= 'z':
        output_string = output_string + input_string[i].upper()
    else:
        output_string = output_string + input_string[i].lower()

print output_string
