#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1009;

int runs, balls;
double dp[MAXN][MAXN], p;

int score[] = {0, 0, 1, 2, 3, 4, 6};

double calcProb(int r, int b) {	
	if(r < 1) return 1;
	double& res = dp[r][b];
	if(res != -1) {
		return res;
	}
	if(b == 0) {
		return 0;
	}
	res = 0;
	for(int i=0; i<7; i++) {
		res += p * calcProb(r-score[i], b-1);
	}
	return res;
}

int main(int argc, char const *argv[]){
	
	int cases;
	scanf("%d", &cases);
	for(int i=0; i<1009; i++) {
		for(int j=0; j<1009; j++) {
			dp[i][j] = -1;
		}
	}
	p = 1/double(7);
	while(cases--) {
		scanf("%d%d", &runs, &balls);
		printf("%.6lf\n", calcProb(runs, balls));
	}
	return 0;
}
