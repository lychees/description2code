#include <bits/stdc++.h>
using namespace std;

int main()
{
    int64_t n;
    vector<int64_t> a;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	int64_t k;
    	cin>>k;
    	a.push_back(k);
    }
    sort(a.begin(),a.end());
    for(int i=0;i<n;i++)
    {
    	cout<<a[i]<<endl;
    }
    return 0;
}
