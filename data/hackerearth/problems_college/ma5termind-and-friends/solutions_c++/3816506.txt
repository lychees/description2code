#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll l[100005],r[100005],c[100005],a[100005],ans[100005];
int main()
{
  ll t,n,i,m,q,x,y;
  ios_base::sync_with_stdio(false);
  cin>>t;
  while(t--)
  {
      cin>>n>>m;
      for(i=1;i<=m;i++)
      cin>>l[i]>>r[i]>>c[i];
      cin>>q;
      memset(a,0,sizeof(a));
      memset(ans,0,sizeof(ans));
      for(i=0;i<q;i++)
      {
          cin>>x>>y;
          a[x]++;
          a[y+1]--;
      }
      for(i=2;i<=m;i++)
      {
        a[i]=a[i]+a[i-1];
      }
      for(i=1;i<=m;i++)
      {
          ans[l[i]]+=a[i]*c[i];
          ans[r[i]+1]-=a[i]*c[i];
      }
      cout<<ans[1]<<" ";
       for(i=2;i<=n;i++)
      {
        ans[i]+=ans[i-1];
        cout<<ans[i]<<" ";
      }
      cout<<endl;

  }

}
