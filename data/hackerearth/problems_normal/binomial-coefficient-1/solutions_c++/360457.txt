#include<stdio.h>
#include<iostream>
#include<algorithm>
#define ll long long
#define MAX 1000005
using namespace std;
long long int mod = 1000000007;
ll pow(ll b,ll n)
{
if (n==1)
return b;
else if(n==0)
return 1;
ll x=pow(b,n>>1);
if(n&1)
return ((b*x)%mod)*x%mod;
else
return (x*x)%mod;
}
ll fact[MAX]={0};
long long int inv(long long int x)
{

    return pow(x,mod-2)%mod;
}
long long int GCD(long long int a,long long int b)
{
     if (b == 0) 
     {
        return a;
     }
    return GCD(b, a % b);
}
long long int nck(long long int n,long long int k)
{
	if(k>n-k)
	k=n-k;
	long long int i=0;
	long long int ans=1;
	for(i=0;i<k;i++)
	{
		long long int n1=n-i;
		long long int k1=k-i;
		long long temp=GCD(n1,k1);
		n1=n1/temp;
		k1=k1/temp;
		ans=((ans*n1)%mod*inv(k1))%mod;
	}
	return ans;
	
}
int main()
{
long long int n,k,i,a,b;
scanf("%lld %lld %lld %lld",&a,&b,&n,&k);

printf("%lld\n",((pow(a,k)*pow(b,n-k))%mod*nck(n,k))%mod);
return 0;
}
