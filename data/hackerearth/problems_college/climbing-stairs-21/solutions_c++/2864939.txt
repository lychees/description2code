#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
    	int n;
    	cin >> n;
    	if (n == 1) {
    		cout << "1" << endl;
    		continue;
    	}
    	long long dp[n];
    	dp[n-1] = 1;
    	dp[n-2] = 2;
    	for (int i = n-3; i >= 0; i--) {
    		dp[i] = dp[i+1] + dp[i+2];
    	}
    	cout << dp[0] << endl;
    }
    return 0;
}
