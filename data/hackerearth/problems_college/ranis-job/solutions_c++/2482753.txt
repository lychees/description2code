#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	vector<long long> sw(n);
    	long long total=0,in,ind=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>in;
    		total+=in;
    	}
    	for(int i=0;i<n;i++)
    	{
    		cin>>sw[i];
    	}
    	sort(sw.begin(),sw.end(),greater<long long>());
    	
    	for(int i=0;i<n;i++)
    	{
    		total-=sw[i];
    		if(total<=0)
    		{
    			cout<<i+1<<endl;
    			break;
    		}
    	}
    }
    return 0;
}
