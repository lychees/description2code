#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define MAX 100005
#define ll long long int
ll fact[MAX];
ll ifact[MAX];
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		if(b&1) res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
int main()
{
	int i;
	int j;
	fact[0]=1;
	for(i=1;i<MAX;++i)
	{
		fact[i] = (fact[i-1]*i)%MOD;
	}
	for(i=0;i<MAX;++i)
		ifact[i]=modexp(fact[i],MOD-2);
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,k;
    	cin>>n>>k;
    	cout<<(fact[k]*ifact[k-n])%MOD<<endl;
    }
    
    return 0;
}
