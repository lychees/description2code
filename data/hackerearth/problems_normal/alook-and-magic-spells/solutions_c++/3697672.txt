#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define bitcount __builtin_popcountll
#define ll long long
ll int mod=1000000007;
ll int a[4][4],b[4][4],c[4][4];
void power(int n)
{
	int i,j,k;
	if(n==1)
		return;
	if(n%2==0)
	{
		power(n/2);
		for(i=1;i<=3;i++)
		{
			for(j=1;j<=3;j++)
			{
				b[i][j]=0;
				for(k=1;k<=3;k++)
				{
					b[i][j]=(b[i][j]+(a[i][k]*a[k][j]+mod*mod)%mod)%mod;
				}
			}
		}
		for(i=1;i<=3;i++)
		{
			for(j=1;j<=3;j++)
				a[i][j]=b[i][j];
		}
	}
	else
	{
		for(i=1;i<=3;i++)
		{
			for(j=1;j<=3;j++)
				c[i][j]=a[i][j];
		}
		power(n/2);
		for(i=1;i<=3;i++)
		{
			for(j=1;j<=3;j++)
			{
				b[i][j]=0;
				for(k=1;k<=3;k++)
				{
					b[i][j]=(b[i][j]+(a[i][k]*a[k][j]+mod*mod)%mod+mod)%mod;
				}
			}
		}
		for(i=1;i<=3;i++)
		{
			for(j=1;j<=3;j++)
			{
				a[i][j]=0;
				for(k=1;k<=3;k++)
				{
					a[i][j]=(a[i][j]+(b[i][k]*c[k][j]+mod*mod)%mod+mod)%mod;
				}
			}
		}
	}
}
int main()
{
	int i,j,k;
	int t,n;
	sd(t);
	while(t--)
	{
		sd(n);
		if(n==1)
			printf("2\n");
		else if(n==2)
			printf("14\n");
		else
		{
			n-=2;
			a[1][1]=3;
			a[1][2]=2;
			a[1][3]=-4;
			a[2][1]=1;
			a[2][2]=0;
			a[2][3]=0;
			a[3][1]=0;
			a[3][2]=1;
			a[3][3]=0;
			power(n);
			printf("%lld\n",((a[1][1]*14)%mod+(a[1][2]*2)%mod)%mod);
		}
	}
	return 0;
}
