#include <bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{
	ll t;
	cin >> t;
	ll a[4] = {2, 3, 5, 7};
	ll v[1000001];
	v[0] = 0;
	for(ll i = 1; i < 1000001; i++)
	{
		v[i] = INT_MAX;
		for(ll j = 0; j < 4; j++)
		{
			if(a[j] <= i)
				v[i] = min(v[i], 1 + v[i - a[j]]);
		}
	}
	while(t--)
	{
		ll n;
		 cin >> n;
		 if(v[n] != INT_MAX)
		 	cout << v[n] << endl;
		 else
		 	cout << -1 << endl;
	}
    return 0;
}
