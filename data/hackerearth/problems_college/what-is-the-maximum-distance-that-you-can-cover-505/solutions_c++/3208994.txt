#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	int n;
	cin>>n;
	double prop[n],max = 1000000000,fuel;
	for(int i = 0 ; i < n ; i++){
		cin>>prop[i];
	}
	for(int i = 0 ; i < n ; i++){
		double temp;
		cin>>temp;
		prop[i] = temp / prop[i];
		if(max > prop[i])
			max = prop[i];
	}
	cin>>fuel;
	//cout<<fuel/max;
	printf("%.3f",floor(1000 * (fuel/max)) / 1000);
    //cout << "Hello World!" << endl;
    return 0;
}
