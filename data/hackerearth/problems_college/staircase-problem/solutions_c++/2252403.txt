int toInt(string s){int sm;stringstream ss(s);ss>>sm;return sm;}
long long int toLlint(string s){long long int sm;stringstream ss(s);ss>>sm;return sm;}

int solve(int n) {
    int i = 0, ans = 0, cnt = 0;
    while (ans < n) {
        i++;
        ans += i;
        cnt++;
    }

    ans = (ans == n) ? cnt : (--cnt);
    cout << ans << endl;

}

int main()
{
    int t, n;
    cin >> t;
    while (t--) {
        cin >> n;
        solve(n);
    }

    return 0;
}
