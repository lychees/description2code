#include <bits/stdc++.h>

using namespace std;

int m = 1000000007;

void fib(int a[],int n,int l)
{
    int i = l;
	while(i < n)
	{
		a[i] = (a[i-1]+a[i-2])%m;
		i++;
	}
	return;
}

int main()
{
	int a[1000000];
	a[0] = 2;
	a[1] = 3;
	int t,l=2;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		if(n <= l)
		{
			printf("%d\n", ((a[n-1]-1)+m)%m);
		}
		else
		{
			fib(a,n,l);
			l = n;
			printf("%d\n", ((a[n-1]-1)+m)%m);
		}
	}
	return 0;
}
