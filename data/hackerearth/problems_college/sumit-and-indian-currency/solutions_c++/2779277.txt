#include <iostream>
using namespace std;

int main()
{
    int t,i,c,fi,f,k;
    string s;
    
    cin>>t;
    while(t--){
    	cin>>s;
    	
    	if(s.size()<=3){
    		cout<<s;
    	}else{
	    	c = s.size()-3;
	    
	    	if(c%2==0){
	    		fi=2;
	    	}else{
	    		fi=1;
	    	}
	    	k=0;
	    	for(i=0;i<s.size()-3;i++){
	    		
	    		if(fi==1){
	    			cout<<s[i]<<",";
	    			fi=2;	
	    		}else{
	    			if(k%2==0){
	    				cout<<s[i];
	    				k++;
	    			}else{
	    				cout<<s[i]<<",";
	    				k++;
	    			}
	    		}    		
	    	}
	    	
	    	for(i=s.size()-3;i<s.size();i++){
	    		cout<<s[i];
	    	}
    	}
    	cout<<endl;
    }
    
    return 0;
}
