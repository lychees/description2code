#include <bits/stdc++.h>
using namespace std;
long long ans[10007][10007];
int main()
{
	int n,s,m,p;
	cin >> n >> s >> m >> p;
	long long a[n];
	for(int i=0;i<n;i++)
	   cin >> a[i];
	for(int i=0;i<n;i++) {
		if(a[0] < a[i]) 
			ans[0][i]=-p*abs(a[i]-a[0]);
		else
		   ans[0][i]=-m*abs(a[i]-a[0]);
	} 
	for(int i=1;i<n;i++) {
		long long val=ans[i-1][0];
		for(int j=1;j<n;j++) {
			val=max(val,ans[i-1][j]);
		}
		for(int j=0;j<n;j++) {
			if(a[i] < a[j]) 
			   ans[i][j]=-p*abs(a[i]-a[j]);
		    else
		       ans[i][j]=-m*abs(a[i]-a[j]);
		    ans[i][j] += max(val,s+ans[i-1][j]);
		}
	}
	long long res=0;
	for(int i=0;i<n;i++) {
		res=max(res,ans[n-1][i]);
	}
	cout << res << endl;
}
