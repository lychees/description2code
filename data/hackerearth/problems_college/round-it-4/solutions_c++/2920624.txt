#include<iostream>
#include<cmath>
using namespace std;
int main()
{long long t;
cin>>t;
while(t--)
    {long long n,k,ans;
    cin>>n>>k;

if(n%k==0)
    ans=(n/k)*k;
   else
            if(abs(n-((n/k)*k))<abs(n-(((n/k)+1)*k)))
            ans=(n/k)*k;
   else
   ans=((n/k)+1)*k;


    cout<<ans<<endl;

    }

return 0;

}
