#include<iostream>
#include<string>
using namespace std;
int main()
{
	int tc=0;
	cin>>tc;
	while(tc-->0)
	{
		string s;
		cin>>s;
		int arr[256]={0};
		for(int i=0;s[i];i++)
		{
			arr[s[i]]++;
		}
		int count=0;
		for(int i=0;i<256;i++)
		{
			if(arr[i]>0)
				count++;
		}
		cout<<(count*(count+1)/2)<<endl;
	}
	return 0;
}
