#include<bits/stdc++.h>
using namespace std;

#define fastScan ios_base::sync_with_stdio(0); cin.tie(NULL);

int main()
{
    fastScan;
    int t;
    cin>>t;
    while(t--){
        long long int n,i,x,ans;
        cin>>n;
        for(i=0;pow(2,i)<=n;i++);
        x=pow(2,i-1);
        ans=1+2*(n-x);
        cout<<ans<<endl;
    }
    return 0;
}
