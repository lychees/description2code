#include <cstdio>
#include <cstring>

int n;
char s[100010];
int main()
{
    int t;
    scanf("%d", &t);
    while(t--){
      scanf("%s", s);
      n = strlen(s);
      int a = 0;
      for(int i=0;i<n;i++){
        while(i+1<n && s[i]==s[i+1])i++, a++;
      }
      printf("%d\n", a);

    }
    return 0;
}
