#include<bits/stdc++.h>
using namespace std;

//char a[1000000000]={'\0'};
//char b[1000000000]={'\0'};
string s1;
string s2;
string s3;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long int len1,len2;
		long long int a=0,b=0;
		cin>>s1>>s2;
		len1=s1.length();
		len2=s2.length();
		for(int ik=0;ik<len1;ik++)
		{
			if(ik+len2<=len1)
			{
				s3=s1.substr(ik,len2);
			//	cout<<s3<<"\n";
				
			}
			
			for(int i=0;i<len2;i++)
			{
				a=a+(pow(10,len2-i-1)*(s3.at(i)-48));
				//cout<<a<<" ";
			}
			
			for(int i=0;i<len2;i++)
			{
				b=b+(pow(10,len2-i-1)*(s2.at(i)-48));
			}
			
			if(ik+len2<=len1)
			cout<<a*b<<"\n";
			a=0;
			b=0;
			
		}
		
		
	}
	
}
