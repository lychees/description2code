#include<bits/stdc++.h>
#include<string>
using namespace std;
int func(string s,int l);
int main()
{
string str;
int l,count;
cin>>str;
l=str.length();
count=func(str,l);
cout<<count<<"\n";
return 0;
}
int func(string s,int l)
{
int count,i;
count=0;
for(i=0;i<l;i++)
{
if(s[i]=='0')
count+=6;
else if(s[i]=='1')
count+=2;
else if(s[i]=='2')
count+=5;
else if(s[i]=='3')
count+=5;
else if(s[i]=='4')
count+=4;
else if(s[i]=='5')
count+=5;
else if(s[i]=='6')
count+=6;
else if(s[i]=='7')
count+=3;
else if(s[i]=='8')
count+=7;
else if(s[i]=='9')
count+=6;
else
continue;
}
return count;
}
