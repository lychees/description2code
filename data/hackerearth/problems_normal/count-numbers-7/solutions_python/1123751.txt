k,t=map(int,raw_input().split())
klis=list(map(int,raw_input().split()))

for _ in xrange(t):
	ans=0
	a,b=map(int,raw_input().split())
	for i in xrange(1,2**k):
		l,r=a-1,b
		par=0
		for j in xrange(k):
			if(i&(1<<j)):
				par+=1
				l/=klis[j]
				r/=klis[j]
		if(par&1):
			ans+=(r-l)
		else:
			ans-=(r-l)
	print ans
