#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll GCD(ll x,ll y)
{
    if(y==0)
        return x;
    return GCD(y,x%y);
}
main()
{
    ll count=0,n,i=0,j;
    cin>>n;
    vector<int>A(n);
    while(i<n &&  cin>>A[i++]);
    for(i=0;i<A.size();i++)
        for(j=i+1;j<A.size();j++)
         count+=(i<j && A[i]>A[j] && (GCD(A[i],A[j])==1))?1:0;
    cout<<count;
}
