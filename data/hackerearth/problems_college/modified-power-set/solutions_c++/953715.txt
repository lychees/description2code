#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string input;
    	vector <bool> ele(128, true);
    	cin>>input;
    	int i,c=0;
    	for(i=0; i<input.size(); i++)
    		if(ele[input[i]]){
    			c++;
    			ele[input[i]]=false;
    		}
    	cout<<(c*(c+1))/2<<endl;
    	
    }
    return 0;
}
