noc = input()
a = [0] * noc
b = [0] * noc
for x in range(noc):
    ai, bi = map(int, raw_input().split())
    a[x] = ai
    b[x] = bi

pairs = 0
for i in range(noc):
    mh = -10000000
    for j in range(i+1, noc):
        slope = ((b[j] - b[i]) * 1.0) / (a[j] - a[i])
        if slope >= mh:
            pairs += 1
            mh = slope

print pairs
