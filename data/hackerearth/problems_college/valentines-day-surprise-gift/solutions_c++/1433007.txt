#define ll long long int
#define mod 1000000007 


long long int gcd ( long long int a,long long  int b )
{
  if ( a==0 ) return b;
  return gcd ( b%a, a );
}

ll power(ll base, ll exp) {
  base %= mod;
  ll result = 1;
  while (exp > 0) {
    if (exp & 1) result = (result * base) % mod;
    base = (base * base) % mod;
    exp >>= 1;
  }
  return result;
}
