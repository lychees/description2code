#include <cstdio>
#include <vector>
#include <bitset>
#include <cstring>

#define si(x) scanf("%d", &x);

#define vi vector<int>
#define lli long long int

char S[1000004];
long long out[26][26];
int num[26];

int main(){
	int n;
	si(n);
	scanf("%s\n", S);
	int q;
	si(q);
	memset(num, 0, sizeof(int)*26);
	memset(out, 0, sizeof(long long)*26*26);
	for(int i=0; i<n; i++) num[S[i]-'a']++;
	
	for(int i=0; i<n; i++){
		for(int j=0; j<26; j++){
			if (S[i]-'a' == j) out[S[i]-'a'][j] += num[j]-1;
			else out[S[i]-'a'][j] += num[j];
		}
		num[S[i]-'a']--;
	}
	for(int i=0; i<q; i++){
		char x, y;
		scanf("\n%c %c", &x, &y);
		if(x==y)
		printf("%lld\n",out[x-'a'][y-'a']);
		else
		printf("%lld\n",out[x-'a'][y-'a']+out[y-'a'][x-'a']);
	}
	return 0;
}
