#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int t,n,i,j,x;
    sd(t);
    while(t--)
    {
        x=0;
        sd(n);
        n*=3;
        for(i=0;i<n;i++)
        {
            sd(j);
            x=(x^j);
        }
        if(x==0)
            printf("Lannister\n");
        else
            printf("Stark\n");
    }
    return 0;
}
