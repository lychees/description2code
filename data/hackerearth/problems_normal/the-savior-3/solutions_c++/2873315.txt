#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++){
    	int n;
    	cin>>n;
    	int a[n];
    	for(int j=0;j<n;j++){
    		cin>>a[j];
    	}
    	int count=0;
    	for(int k=0;k<n-1;k++){
    		for(int j=k+1;j<n;j++){
    			if(a[k]%2==0){
    				if(a[j]%2==0 && a[j]!=a[k]){
    					count++;
    				}
    			}
    			else{
    				if(a[j]%2==1 && a[j]!=a[k]){
    					count++;
    				}
    			}
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
