#include <bits/stdc++.h>
using namespace std;
char a[1024][1024];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		int i,j;
		for(i=0;i<n;i++) 
		{
			for(j=0;j<m;j++)
			{
				cin>>a[i][j];
			}
		}
		int cnt=0;
		for(i=0;i<n-1;i++)
		{
			for(j=0;j<m-1;j++) 
			{
				if(a[i][j]=='/' && a[i][j+1]=='\\' && a[i+1][j]=='\\' && a[i+1][j+1]=='/') 							      {
					cnt++;
				}
			}
		}
		cout<<cnt<<endl;
	}
}
