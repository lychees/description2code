#include<iostream>

#include<bits/stdc++.h>

using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n],i,j; int b[n];
    for(i=0;i<n;i++)
        {cin>>a[i]; b[i]=a[i];}
        std::sort(a,a+n);

    for(i=0;i<n;i++)

    { j=0;

        while(j<n&& (a[j]<=b[i]/2))
        {
            if(b[i]%a[j]==0&&b[i]!=a[j])
                break; j++;

        }
        if(j==n)
            cout<<b[i]<<" ";
            else if(a[j]>b[i]/2)
                cout<<b[i]<<" ";
        }
        cout<<endl;
}
