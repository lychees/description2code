#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main(){
	ios::sync_with_stdio(false);
	int N=0;
	int T=0;
	
	cin >> N >> T;
	
	int arr[N];
	
	for(int i=0;i<N;i++)
	 arr[i] = 0;
	
	for(int i=0;i<N;i++)
	 cin >> arr[i];
	
	while(T--)
	{
	 int first=0;
	 int last=0;
	 
	 cin >> first >> last;
	 
	 vector<int> vec;
	 
	 for(int i=first;i<=last;i++)
	  vec.push_back(arr[i]);
	 
	 sort(vec.begin(),vec.end());
	 
	 cout << vec.at(0) << endl;
	 
	 vec.clear();
	}
	
	return 0;
}
