#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int  main()
{
    int t,n,p;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vector<int>animals;
        vector<int>swords;
        for(int i =0;i<n;i++)
        {
            cin>>p;
            animals.push_back(p);
        }
        for(int i =0;i<n;i++)
        {
            cin>>p;
            swords.push_back(p);
        }
        sort(animals.begin(),animals.begin()+n);
        sort(swords.begin(),swords.begin()+n);
             for(int i =0;i<n;i++)
        {
            if(animals.at(i)<swords.at(i))
            {
                cout<<"NO"<<endl;
                break;
            }
            else if(i ==n-1)
            {
                cout<<"YES"<<endl;
            }
        }
    }
}
