#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n1, n;
    	long long min, a, sum=0;
    	min = 10000000001;
    	cin>>n1;
    	n=n1;
    	while(n--){
	    	cin>>a;
	    	sum+=a;
	        min = a<min?a:min;
    	}
    	sum+=((n1-2)*min);
    	cout << sum << endl;
    }
    return 0;
}
