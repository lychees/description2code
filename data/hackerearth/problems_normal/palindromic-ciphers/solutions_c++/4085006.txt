#include <iostream>
using namespace std;

int main()
{
	int t;
    cin >> t;
    while(t--)
    {
    	int i,ctr=0;
    	long long int p =1;
     	string s;
     	cin >> s;
     	for(i=0;i<s.size()/2;i++)
     	{
     		if(s[i] == s[s.size() - 1 -i])
     		{
     			ctr++;
     		}
     		p = p*(s[i] - 96 )*(s[s.size() - 1 -i] - 96);
     	}
     	if(ctr == s.size()/2 )
     	{
     		cout << "Palindrome" << endl;
     	}
     	else if(s.size()%2 == 0)
     	{
     		cout << p  << endl;
     	}
     	else
     	{
     		cout << p*(s[s.size()/2] - 96) << endl;
     	}
    }
    return 0;
}
