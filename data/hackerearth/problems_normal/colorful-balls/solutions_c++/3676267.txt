#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int dp[12][12][12][12][5];
//0 red ball,1 blue ball,2 green ball,3 yellow ball
ll int myfind(ll int r,ll int b,ll int g,ll int y,ll int p)
{
	if(r==0 && b==0 && g==0 && y==0)
	return 1ll;
	if(dp[r][b][g][y][p]==-1)
	{
		ll int ans=0;
		if(p==0)
		{
			if(b>0)
			ans+=myfind(r,b-1,g,y,1);
			if(g>0)
			ans+=myfind(r,b,g-1,y,2);
			if(y>0)
			ans+=myfind(r,b,g,y-1,3);
		}
		else if(p==1)
		{
			if(r>0)
			ans+=myfind(r-1,b,g,y,0);
			if(g>0)
			ans+=myfind(r,b,g-1,y,2);
			if(y>0)
			ans+=myfind(r,b,g,y-1,3);
		}
		else if(p==2)
		{
			if(r>0)
			ans+=myfind(r-1,b,g,y,0);
			if(b>0)
			ans+=myfind(r,b-1,g,y,1);
			if(y>0)
			ans+=myfind(r,b,g,y-1,3);
		}
		else if(p==3)
		{
			if(r>0)
			ans+=myfind(r-1,b,g,y,0);
			if(b>0)
			ans+=myfind(r,b-1,g,y,1);
			if(g>0)
			ans+=myfind(r,b,g-1,y,2);
		}
		else
		{
			if(r>0)
			ans+=myfind(r-1,b,g,y,0);
			if(b>0)
			ans+=myfind(r,b-1,g,y,1);
			if(g>0)
			ans+=myfind(r,b,g-1,y,2);
			if(y>0)
			ans+=myfind(r,b,g,y-1,3);
		}
		dp[r][b][g][y][p]=ans;
	}
	return dp[r][b][g][y][p];
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int a,b,c,d;
	cin>>a>>b>>c>>d;
	ll int i,j,k,l,m;
	for(i=0;i<=a;i++)
	{
		for(j=0;j<=b;j++)
		{
			for(k=0;k<=c;k++)
			{
				for(l=0;l<=d;l++)
				{
					for(m=0;m<5;m++)
					dp[i][j][k][l][m]=-1ll;
				}
			}
		}
	}
	cout<<myfind(a,b,c,d,4)<<endl;
	return 0;
}
