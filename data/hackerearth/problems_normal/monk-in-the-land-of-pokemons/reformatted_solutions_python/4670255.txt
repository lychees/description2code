t = int(raw_input())

for i in range(t):
    n = int(raw_input())
    count = 0
    foods = {}
    for j in range(n):
        (a, b) = map(int, raw_input().split(' '))
        if a != b:
            if b not in foods or foods[b] == 0:
                count += 1

            else:
                foods[b] -= 1

            if a not in foods:
                foods[a] = 1
            else:
                foods[a] += 1

    print count
