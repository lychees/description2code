string add(string s, string t){
	int l1 = s.length() - 1, l2 = t.length() - 1;
	string res = "";
	stack<char> st;
	int tot = 0, car = 0;
	
	while(l1 >= 0 && l2 >= 0){
		tot = (s[l1] - 48) + (t[l2] - 48) + car;
		car = tot/10;
		st.push((tot % 10) + 48);
		l1--;
		l2--;
	}
	
	while(l1 >= 0){
		tot = (s[l1] - 48) + car;
		car = tot/10;
		st.push((tot % 10) + 48);
		l1--;
	}
	
	while(l2 >= 0){
		tot = (t[l2] - 48) + car;
		car = tot/10;
		st.push((tot % 10) + 48);
		l2--;
	}
	
	while(car){
		st.push((car % 10) + 48);
		car /= 10;
	}
	
	while(!st.empty()){
		res.append(1, st.top());
		st.pop();
	}
	
	return res;
}

int main(){
	
	vector<string> v(10001);
	v[1] = "2";
	v[2] = "3";
	for(int i=3;i<v.size();i++){
		v[i] = add(v[i-1], v[i-2]);
	}
	int t, n;
	cin >> t;
	while(t--){
		cin >> n;
		cout << v[n] << endl;
	}
    return 0;
}
