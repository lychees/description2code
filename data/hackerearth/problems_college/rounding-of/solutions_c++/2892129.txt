#include <iostream>
using namespace std;
int main()
{
    long long T,i,j,k,t,p;
    cin>>T;
    while(T--){
    	cin>>j>>k;
    	t=j/k;
    	p=j%k;
//    	cout<<p<<endl;
    	if(k%2==1){
    		if(p<=k/2) cout<<t*k<<endl;
    		else cout<<k*(t+1)<<endl;
    	}
    	else{
    		if(p<k/2) cout<<t*k<<endl;
    		else cout<<k*(t+1)<<endl;
    	}
    }
    return 0;
}
