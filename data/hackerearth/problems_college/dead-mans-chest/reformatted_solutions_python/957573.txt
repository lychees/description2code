def py_2():
    t = input()
    for tt in xrange(t):
        freq = [0]*26
        s = raw_input()
        slen = len(s)
        for i in xrange(slen):
            # print ord(s[i])-97
            freq[ord(s[i])-97] += 1
        c = 0
        for i in xrange(26):
            if freq[i] % 2:
                c += 1
        if c == 0:
            print 'JACK'
        elif c % 2:
            print 'JACK'
        else:
            print 'DAVY JONES'


py_2()
