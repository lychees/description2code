#include <bits/stdc++.h>

using namespace std;

#define MAX 11
#define INF 100000000
#define MOD 1000000007
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define ll long long

int cnt[MAX][26], n, k;
string s;
int dp[MAX][1<<MAX];

int solve(int mask, int index){
	if(index == k)
		return 1;
	if(dp[index][mask] != -1)
		return dp[index][mask];
	
	dp[index][mask] = 0;

	for(int i=0 ; i<n ; i++)
		if((mask & (1 << i)) == 0)
			dp[index][mask] = (dp[index][mask] + 1ll * cnt[i][s[index] - 'a'] * solve(mask | (1 << i), index + 1)) % MOD; 

	return dp[index][mask];
}

int main(){
	cin>>n>>k;

	memset(dp, -1, sizeof(dp));

	for(int i=0 ; i<n ; i++){
		for(int j=0 ; j<6 ; j++){
			char c;
			cin>>c;
			cnt[i][c - 'a']++;
		}
	}

	cin>>s;

	cout<<solve(0, 0)<<"\n";

	return 0;
}
