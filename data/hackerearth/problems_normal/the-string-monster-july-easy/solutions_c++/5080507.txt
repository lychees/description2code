int N;

int find(int pos, vector< vector<int> > &strings, vector< int > &target, vector<int> so_far){
  if(pos >= N) return 0;

  int t_res = find(pos+1, strings, target, so_far);
  if(t_res) return 1;

  int eq = 0;
  for(int i=0; i<26; i++){
    so_far[i] += strings[pos][i];
    if(so_far[i] > target[i]) return 0;
    if(so_far[i] == target[i]) eq++;
  }
  if(eq == 26) {
    return 1;
  }

  return find(pos+1, strings, target, so_far);
}

int main(){
  std::ios::sync_with_stdio(false);

  int t;
  int len;
  char s[910];

  scanf("%d", &t);
  while(t--){
    scanf("%d", &N);

    vector< vector<int> > strings(N, vector<int>(26, 0));
    for (int i = 0; i < N; ++i)
    {
      scanf("%s", s);
      len = strlen(s);
      for(int j=0; j < len; j++){
        strings[i][ s[j] - 'a' ]++;
      }
    }

    vector<int> target(26, 0);
    vector<int> so_far(26, 0);
    scanf("%s", s);
    len = strlen(s);
    for (int i = 0; i < len; ++i)
    {
      target[ s[i] - 'a' ]++;
    }
    if(find(0, strings, target, so_far)) printf("YES\n");
    else printf("NO\n");

  }

  return 0;
}
