#include <iostream>
using namespace std;

int main()
{
	int t,n,d,temp;
	cin>>t;
	while(t--)
	{
		temp=0;
		cin>>n>>d;
		while(n&&d)
		{
			n--;
			d=d-2;
			temp++;
			if(temp%3==0)
			{
				n=n+2;
			}
		}
		if(d<=0)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
   // cout << "Hello World!" << endl;
    return 0;
}
