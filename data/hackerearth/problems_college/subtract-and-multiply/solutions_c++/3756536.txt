#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,x,y,i,j,k,l,c;
    cin>>t;
    while(t--){
    	cin>>x>>y;
    	if(y==0){
    		cout<<4<<endl;
    	}
    	else{
    	c=0;
    	for(i=1;i<=sqrt(y);i++){
    		if(y%i==0){
    		l=i;
    		k=y/i;
    		j=abs(l-k);
    		if(j==x){
    			c++;
    			if(l!=k){
    			c++;
    		}
    		}
    		
    	}
    	}
    	cout<<2*c<<endl;
    }
    }
    return 0;
}
