typedef pair<int, int> pii;
typedef vector<int> vi;


const ll MOD = 1000000007;
const int MAX = 100005;
pair<ll, ll> v[MAX];
bool A[MAX];

void s()
{
    ll i, j;
    for(i = 1;i < MAX;++i)
        v[i] = make_pair(1, i);
    for(i = 2;i < MAX;++i)
    {
        if(A[i] == false)
        {
            v[i].first *= i;
            for(j = 2*i;j < MAX;j += i)
            {
                A[j] = true;
                v[j].first *= i;
            }
        }
    }
}


int main()
{
	//freopen("input.txt", "r", stdin);
	s();
	sort(v, v + 100001);
	int t, n;
	for(ini(t);t--;)
	{
        ini(n);
        cout << v[n].second << endl;
	}
	return 0;
}
