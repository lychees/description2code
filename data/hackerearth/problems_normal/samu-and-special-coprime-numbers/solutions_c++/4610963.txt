#include <iostream>
#include <map>
#include <cstdio>
#include <cstring>
#include <algorithm>
#define sc(x) scanf("%lld",&x)
#define pr(x) printf("%lld\n",x)
#define ll long long int
using namespace std;

ll dp[20][200][2000];
ll d[20];

ll solver(ll pos, ll s, ll sq, ll check)
{
	if(pos==-1)
		return __gcd(s,sq)==1;
	if(check==0 && dp[pos][s][sq]!=-1)
		return dp[pos][s][sq];
	ll curr,ans=0;
	if(check==1)
		curr=d[pos];
	else
		curr=9;
	for(int i=0;i<=curr;i++)
		ans+=solver(pos-1,s+i,sq+i*i,check && curr==i);
	if(check==0)
		return (dp[pos][s][sq]=ans);

	return ans;
}

ll solve(ll n)
{
	ll len=0;
	if(n==0)
        return 0;
	while(n>0)
	{
		d[len++]=n%10;
		n/=10;
	}
	return solver(len-1,0,0,1);
}


int main()
{
	ll t;
	sc(t);
	memset(dp,-1,sizeof(dp));
	while(t--)
	{
		ll l,r;
		sc(l);
		sc(r);
		ll ans=solve(r)-solve(l-1);
		pr(ans);
	}
	return 0;
}
