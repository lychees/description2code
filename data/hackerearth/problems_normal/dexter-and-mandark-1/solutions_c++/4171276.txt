using namespace std;
int n,m;
int dp[1005][1005];
int func(int index,int dex,int man)
{
    if(dex<(man*m))
        return 0;
    if(index>=n)
    {
        if(dex>man)
            return 1;
        else
            return 0;
    }
    if(dp[dex][man]!=-1)
        return dp[dex][man];
    return dp[dex][man]=(func(index+1,dex+1,man)%mod+func(index+1,dex,man+1)%mod)%mod;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        memset(dp,-1,sizeof(dp));
        cout<<func(0,0,0)<<"\n";
    }
}
