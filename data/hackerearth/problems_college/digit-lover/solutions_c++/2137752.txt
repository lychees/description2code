#include <iostream>
#include <string>
using namespace std;

void baby(long int r)
{
    int m;
    long int S=0;
    while(r!=0)
    {
        S=S+(r%10);
        r=r/10;
    }
    if(to_string(S).size()>1)
        baby(S);
    else cout<<S<<"\n";

}
int main()
{
    int t;
    long int n;
    cin>>t;
    while(t--){
        cin>>n;
        if((n/10)!=0)
            baby(n);
        else cout<<n<<"\n";

}
}
