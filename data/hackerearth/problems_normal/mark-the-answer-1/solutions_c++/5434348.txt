#include <bits/stdc++.h>
using namespace std;

int main(){
    int n, x, v;
    cin>>n>>x; int arr[n];
    for(int i=0;i<n;i++){
    	cin>>v;
        arr[i]=(v<=x);
    }
    int ans=0, z=2;
    for(int i=0; i<n && z; i++){
        if(arr[i]==0) z--;
        ans+=arr[i];
    }
    cout<<ans;
    return 0;
}
