#include <iostream>
using namespace std;

int main()
{
    long long int a[100001],i,n,q,no,pro=1;
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    cin>>q;
    while(q--)
    {
    	cin>>no;
    	pro=pro*no;
    }
    if(pro!=0)
    {
    	for(i=0;i<n;i++)
    	a[i]=(a[i]/pro);
    
    for(i=0;i<n;i++)
    cout<<a[i]<<" ";
    }
    else
    {
    	for(i=0;i<n;i++)
    	cout<<"0"<<" ";
    }
    return 0;
}
