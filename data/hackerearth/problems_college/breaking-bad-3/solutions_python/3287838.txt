import sys
f=sys.stdin
x=int(f.readline())
array=[]
score=0
k=1
for d in range(0,x):
    s=int(f.readline())
    line=f.readline()
    for x in line.split():
        array.append(int(x))
    array.sort()
    while k<s:
        score=score+array[k-1]*array[k]
        k=k+2
    sys.stdout.write(str(score)+'\n')
    array[:]=[]
    k=1
    score=0
