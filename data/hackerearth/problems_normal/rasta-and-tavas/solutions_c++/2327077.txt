#include <bits/stdc++.h>
using namespace std;
#define int long long
int sume [1000005]; 
main(){
	ios_base::sync_with_stdio(0); 
	for (int g=2; g<=1000000; g++){
		if (sume[g]!=0) continue; 
		for (int y=g; y<=1000000; y+=g) sume[y]+=g; 
	}
	int n, k; cin >> n >> k; 
	int ans = 0; 
	for (int g=1; g<=n; g++) if (sume[g]==k) ans++;
	cout << ans;
	return 0;
}
