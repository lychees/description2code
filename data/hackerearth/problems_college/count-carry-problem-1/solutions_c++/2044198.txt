#include <stdio.h>
int main()
{
    int t,flag,carry;
    long a,b;
    scanf("%d",&t);
    while(t--)
    {
    	flag=carry=0;
    	scanf("%ld%ld",&a,&b);
    	while(a>0 && b>0)
    	{
    		if(a%10+b%10+carry>=10)
    		{
    			a=a/10;
    			b=b/10;
    			carry=1;
    			flag++;
    		}
    		else 
    		{
    			carry=0;
    			a=a/10;
    			b=b/10;
    		}
    	}
    	if(flag==0)
    		printf("No carry operation\n");
    	else if(flag==1)
    		printf("1 carry operation\n");
    	else
    		printf("%d carry operations\n",flag);
    }
    return 0;
}
