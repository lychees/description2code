#include <iostream>
#include<algorithm>
 
using namespace std;
 
int main()
{
  int testcase, cases = 0;
    
    scanf("%d", &testcase);
    while(testcase--) {
        long long int ret = 0;
        int n, i, j;
        scanf("%d", &n);
    
        //sort(A, A+n);
        int front = 1, rear = n-1;
        int l = (n-1)/2, r = (n-1)/2;
        int B[100010] = {};
        B[l] = n;
        while(front <= rear) {
            B[++r] = front++;
            if(front > rear)    break;
            B[--l] =front++;
            if(front > rear)    break;
            B[++r] = rear--;
            if(front > rear)    break;
            B[--l] = rear--;
        }
        for(i = 1; i < n; i++)
            ret += abs(B[i]-B[i-1]);
 
        if(n==1)
        printf("1\n");
        else
        printf("%lld\n", ret);
    }
}
