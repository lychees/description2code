t = int(raw_input())
for i in range(t):
	s = list(raw_input().strip())
	#print(s)
	reserve = raw_input().strip()
	a = [0] * 26
	for ch in reserve:
		a[ord(ch)-97] += 1
	#print(a)
	for k in range(len(s)):
		char = ord(s[k])-97
		for p in range(26):
			if char > p and a[p] > 0:
				a[p] -= 1
				s[k] = chr(97+p)
				break
				
	print("".join(s))
