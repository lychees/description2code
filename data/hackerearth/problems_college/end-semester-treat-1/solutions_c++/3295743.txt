#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll av[10005],fl[10005];
vector<ll> pr[10005],req[10005];
int main()
{
   // printf("Hello World!\n");
   int n;
   int t,m;
   scanf("%d",&t);
   while(t--)
   {
   	int done = 0;
   	scanf("%d",&n);
   //	printf("%d res\n",n);
   	int i,j;
   	ll k;
   	for(i=0;i<n;i++)
   	scanf("%lld",&av[i]);
   	
   	scanf("%d",&m);
   	for(i=0;i<m;i++){fl[i] = 0;
   	pr[i].clear();
   	req[i].clear();
   	}
  // 	printf("now %d process\n",m);
   	for(i=0;i<m;i++)
   	{
   	//	printf("%d procwss has \n",i);
   		for(j=0;j<n;j++)
   	{
   		scanf("%lld",&k);
   	//	printf("%d ",k);
   		pr[i].push_back(k);
   	}
   //	printf("\n");
   	}
   		for(i=0;i<m;i++)
   	{
   	//	printf("%d procwss has \n",i);
   		for(j=0;j<n;j++)
   	 	{
   		scanf("%lld",&k);
   		//	printf("%d ",k);
   		req[i].push_back(k);
   	}
   		//	printf("\n");
   	}
   	while(1)
   	{
   	int	found = 0;
  
   	for(i=0;i<m;i++)
   	{
   		if(fl[i]==0){
   		int valid = 1;
   		for(j=0;j<n;j++)
   		{
   		if(req[i][j] > pr[i][j] + av[j]){valid = 0;
   		//	printf("required %d is more than %d\n",req[i][j],av[j]);
   			break;
   		}
   	
   		}
   		if(valid)
   		{
   		//	printf("we can finish %d\n",i);
   			found = 1;
   			fl[i] = 1;
   			done++;
   			for(j=0;j<n;j++)
   			av[j]+=pr[i][j];
   		}
   		}
   		
   	}
   		
   		if(found==0)break;
   	}
   	done = 0;
   	for(i=0;i<m;i++)
   	done+=fl[i];
   	if(done==m)printf("YES\n");
   	else printf("NO\n");
   }
    return 0;
}
