n = int(raw_input())
for i in range(n):
    lis, add = raw_input().split()
    lis = map(int, lis[-1::-1])
    add = map(int, add[-1::-1])
    count = 0
    for j in range(len(lis)):
        if (lis[j]+add[j]) > 9:
            count += 1
    if count == 0:
        print "No carry operation"
    elif count == 1:
        print count, "carry operation"
    else:
        print count, "carry operations"
