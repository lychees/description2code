#include <bits/stdc++.h>
#define m 1000000007
using namespace std;
long long int Rem(long long int a)
{
	long long int ans=1,y=m-2;
	while(y)
	{
		if(y%2)
		{
		ans=(ans*a)%m;
		}
		a=(a*a)%m;
		y/=2;
	
	}
	return ans;
}
int main()
{
    int n,t;
    scanf("%d",&t);
    for(int p=1;p<=t;p++)
    {
    	printf("Case #%d: ",p);	   
	    scanf("%d",&n);
	    long long int arr[n];
	    for(int i=0;i<n;i++)
	    {
	    	scanf("%lld",&arr[i]);
	    }
	    int tk[100];int t=2;long long int total=LLONG_MIN;
	    for(int i=3;i<(1<<n);i++)
	    {
	    	int cnt=0;
	    	for(int j=0;j<t;j++)
	    	{
	    		if(i&(1<<j))
	    		{
	    			tk[cnt++]=j;
	    		}
	    	}
	    	if(i==(1<<t)-1)
	    	t++;
	    	if(cnt<2)
	    	continue;
	    	long long int ans=1,sum=0,tmp;
	    	for(int j=0;j<cnt;j++)
	    	{
	    		long long int a=arr[tk[j]];
	    		ans=(ans*a)%m;
	    		sum+=a;
	    	}
	    	tmp=ans;
	    	long long int d=Rem(sum);
	    	tmp=(ans*d)%m;
	    	total=max(total,tmp);
	    }
	    printf("%lld\n",total);
    }
}
