using namespace std;

#define LMT 1001

long long a[1000000];
int total;
int flag[MAX];

void sieve()
{
	//cout<<"In seive\n";
	int i, j, k;
	for(i=3; i<LMT; i+=2)
		if(!flag[i])
			for(j=i*i,k=i<<1; j<MAX; j+=k)
				flag[j]=1;
	a[0] = -1;
	a[1] = 2;
	for(i=3,j=2; i<MAX; i+=2)
		if(!flag[i])
			a[j++] = i;
	total = j;
	//cout<<total<<endl;
}

int prime_ndig(long long n)
{
	int t,nd=0;
	while(n)
	{
		t=n%10;
		if(t==2 ||  t==5 || t==7 || t==3)
			nd++;
		n/=10;
	}
	if(nd==2 || nd==3 || nd==5 || nd==7 || nd==11 || nd==13)
		return 1;
	else
		return 0;
}
int seive[MAX];
int main()
{
	int t;
	long long i,j,n,m,k,x;
	int ans=0;
	sieve();
//cout<<"Hello\n";
	in(t);

		inll2(m,n);
		ans=0;
		for(i=0;i<=n-m;i++)
			seive[i]=0;
		for(x=1;a[x]*a[x]<=n;x++)
		{
			//cout<<x<<endl;
			i=a[x];
			k=(m/i)*i;
			if(k<m)

				j=k-m;
			//printf("j=%d\n",j);

			for(;j<=n-m;j+=i)
				if(j>=0)
					seive[j]=1;
			if(m==1)
				seive[0]=1;
		}
		
		for(i=0;i<=n-m;i++)
			   if(seive[i]==0)
