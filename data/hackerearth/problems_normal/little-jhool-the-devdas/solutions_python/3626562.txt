n=int(raw_input())
lst=raw_input().split()
nist=[int(w) for w in lst]
nist.sort()

s=0
odd=[]

for j in range(0,n):
    if(nist[j]%2!=0):
        odd.append(nist[j])
        odd.sort()
        

for i in range(0,n):
    s=s+nist[i]
if (s%2==0 and len(odd)==0):
    print(":(")

if (s%2==0 and len(odd)!=0):
    ans=s-odd[0]
    if(ans%2==0):
        print(":(")
    else:
        print(ans)
if (s%2!=0):
    print(s)
