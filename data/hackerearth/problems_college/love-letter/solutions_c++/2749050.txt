#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,k;
    	int f[32];
    	memset(f,0,sizeof(f));
    	scanf("%d %d",&n,&k);
    	int j=1;
    	while(k)
    	{
    		f[j++]=k%2;
    		k=k/2;
    	}
    	int an=1;
    	for(int i=1;i<=n;i++)
    	 if(f[i]==0)
    	  an=0;
    	if(an)
    	 printf("YES\n");
    	else printf("NO\n"); 
    }
    return 0;
}
