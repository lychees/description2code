def main():
    N, X = map(int,raw_input().strip().split(' '))
    A = map(int, raw_input().strip().split(' '))
    skip = False
    solved = 0
    
    for i in A:
        if i<=X:
            solved += 1
        elif i>X:
            if skip==False:
                skip=True
            else:
                break
    print solved


main()
