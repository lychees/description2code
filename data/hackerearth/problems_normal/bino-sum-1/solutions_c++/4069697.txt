#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
	long long a,b,sum,t,n=1000,i,j,ans;
	long long ncr[n+1][n+1];
	for(i=0;i<=n;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(j==0||i==j)
				ncr[i][j]=1;
			else
				ncr[i][j]=((ncr[i-1][j-1])%mod+(ncr[i-1][j])%mod)%mod;
		}
	}
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>a>>b;
		for(i=0;i<=b;i++)
		{
			sum=((sum)%mod+(ncr[a][i])%mod)%mod;
		}
		cout<<sum<<endl;
	}
}
