#include<bits/stdc++.h>
using namespace std;
long int a[1000005],b[1000005],c[1000005];;
int main()
{
	long long int n,q,i,l,r;
	scanf("%lld%lld",&n,&q);
	a[0]=0;
	b[0] = 0;
	c[0] = 0;
	for(i = 1; i < n+1; i++)
	{
		scanf("%lld",&a[i]);
		b[i] = a[i];
		c[i] = a[i];
	}
	for(i =2; i<n+1; i++)
	{
		a[i] = a[i] + a[i-1];
	}
	for(i = n-1; i >=0; i--)
	{
		b[i] = b[i+1]-b[i];
	}
	for(i = 0; i < q; i++)
	{
		scanf("%lld%lld",&l,&r);
		printf("%lld %lld\n",a[r]-a[l-1],b[l]-b[r]+c[r]);
	}
}
