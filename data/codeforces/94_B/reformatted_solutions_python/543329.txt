#!/usr/bin/python
import sys


def gcd(a, b):
    while b > 0:
        a, b = b, a % b
    return a


def read_ints():
    return map(int, sys.stdin.readline().split())


def read_line():
    return sys.stdin.readline().strip()


n = read_ints()[0]
edge = {}
for i in range(n):
    a, b = read_ints()
    edge[(a-1)*100+(b-1)] = 1
    edge[(b-1)*100+(a-1)] = 1

for a in range(5):
    for b in range(a):
        for c in range(b):
            x = (a*100+b) in edge
            y = (b*100+c) in edge
            z = (c*100+a) in edge
            if x == y and y == z:
                print 'WIN'
                sys.exit()
print 'FAIL'
