#include <bits/stdc++.h>
 
using namespace std;

int arr[100005];
 
void solve()
{
    int i;
       arr[0]=0;arr[1]=1;arr[2]=0;arr[3]=1;arr[4]=1;
       
           for(i=5;i<=100000;i++)
           {
               if((arr[i-1]==1)&&(arr[i-4])==1)
               arr[i]=0;
               else
               arr[i]=1;
              
           } 
}

 

 
using namespace std;
 
int main() {
	
	ios_base::sync_with_stdio(false);
	cin.tie(0),cout.tie(0);
		solve();
	int t,n,i;
	cin>>t;

    while(t--)
    {
      
       cin>>n;
       if(arr[n]==1)
       cout<<"Yes";
       else
       cout<<"No";
       cout<<"\n";
    }
	return 0;
}
