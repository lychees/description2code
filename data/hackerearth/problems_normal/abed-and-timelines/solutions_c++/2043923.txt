#include <iostream>
using namespace std;

int main()
{	long long int n,z,a,b,x,y,p,q,c,d,e,f;
    cin>>n;
    while(n--)
    {	cin>>a>>b>>x>>y>>p>>q>>z;
    	if(x>p)
    		c=x-p;
    	else
    		c=p-x;
    	d=a-c;
    	if(c>d)
    		c=d;
    	if(y>q)
    		e=y-q;
    	else
    		e=q-y;
    	f=b-e;
    	if(e>f)
    		e=f;
    	if(((e+c)*z)>1000)
    		cout<<(e+c)*z<<"\nLet go of the group.\n";
    	else
    		cout<<(e+c)*z<<"\nYou saved the group.\n";
    }
    return 0;
}
