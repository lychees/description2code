#include <iostream>
using namespace std;
long int FindNumber(int hash[]) {
    int i;
	for(i=1;i<10;i++) {
		if(hash[i] == 0)
			return i;
	}
	int mi = 100;
	int in = 0;
	for(i=1; i<10; i++)
		if(hash[i] < mi ) {
			mi = hash[i];
			in = i;
		}
	int ans = mi;
	int res = 0;
	while(ans--) {
		res = (res+in)*10;
	}
	res =  (res+in);
	return res;
}
int main()
{
 	int t;
 	scanf("%d",&t);
 	while(t--) {
 		int arr[10];
 		for(int i=0; i<10; i++)
 			scanf("%d",&arr[i]);
		long int number = FindNumber(arr);
		printf("%ld\n",number);
 	}
    return 0;
}
