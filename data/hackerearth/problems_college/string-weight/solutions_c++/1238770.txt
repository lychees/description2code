#include <iostream>
#include<stdbool.h>
using namespace std;

int main()
{
    int t,n,weight,i;
    bool a[26];
    string w;
    char c;
    cin>>t;
    while(t--)
    {
    	cin>>w;
    	cin>>n;
    	for(i=0;i<=26;i++)
    	   a[i]=false;
    	while(n--)
    	{
    		cin>>c;
    		a[c-'a']=true;
    	}
    	weight=0;
    	for(i=0;i<w.size();i++)
    	{
    		if(!a[w[i]-'a'])
    		  weight+=w[i]-'a'+1;
    	}
    	cout<<weight<<endl;
    }
    return 0;
}
