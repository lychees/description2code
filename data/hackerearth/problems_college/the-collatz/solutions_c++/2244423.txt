#include<bits/stdc++.h>
 
using namespace std;
 
int main(){
	int n,a,l=0,r=0,step;
	cin >> n;
	while(n--){
		l=0;r=0;
		cin >> a;
		step=0;
		while(a!=1){
			if(a&1){
				r+=a;
				a=3*a+1;
			}
			else{
				l+=a;
				a/=2;
			}
			step++;
//			cout << a << endl;
		}
		r++;
//		cout << step << endl;
//		cout << l << ' ' << r << endl;;
		if(l%step<r%step)
			cout << "Odd Rules" << endl;
		else if(l%step>r%step)
			cout << "Even Rules" << endl;
		else
			cout << "Tie" << endl;
	}
	return 0;
}
