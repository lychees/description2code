#include<bits/stdc++.h>
using namespace std;

int main() {
    int i,j,k,t,ht[26],hp[26];
    cin >> t;
    while(t--) {
    	string text,pattern;
    	cin >> text >> pattern;
        for(i=0;i<26;i++) 
        	ht[i]=hp[i]=0;
        j=text.size();
        k=pattern.size();
        for(i=0;i<j;i++) {
        	ht[text[i]-'a']++;
        }
        for(i=0;i<k;i++) {
        	hp[pattern[i]-'a']++;
        }
        for(i=0;i<26;i++) {
        	if(ht[i] && hp[i]) {
        		cout << "YES" << endl;
        		break;
        	}
       }
       if(i==26)
       	 cout << "NO" << endl;
    } 

	return 0;
}
