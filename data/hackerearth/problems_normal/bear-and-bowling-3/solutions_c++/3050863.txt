ll binpow(ll a, ll b)
{
	if (b <= 0)
	{
		return 1ll;
	}
	ll local_res = 1;
	while (b)
	{
		if (b & 1)
		{
			local_res *= a;
			local_res %= mod;
		}
		a *= a;
		a %= mod;
		b >>= 1;
	}
	return local_res;
}
