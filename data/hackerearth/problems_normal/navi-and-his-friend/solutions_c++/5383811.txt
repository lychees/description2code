#include <bits/stdc++.h>
using namespace std;
int gcd(int a, int b) 
{
    return b == 0 ? a : gcd(b, a % b);
}
int a[17],b[17];
int main()
{	
	int x=1;
	int n,t;
	int d,w,c;
	cin >> d;
	for(int r=1;r<=d;r++)
	{
		cin >> n;
		t=x << n;
		memset(a,0,sizeof a);
		memset(b,0,sizeof b);
		for(int i=0;i<n;i++)
		cin >> a[i] >> b[i];
		cin >> w >> c;
		int ans=0,temp1,temp2,ct;
		for(int i=0;i<t;i++)
		{
			temp1=0,temp2=0,ct=0;
			for(int j=0;j<n;j++)
			{
				if(i&(x<<j))
				{
					temp1+=a[j];
					temp2+=b[j];
					ct++;
					if(ct>=c)
						break;
				}
			}
			if(temp2<=w && ct<=c)
				ans=max(ans,temp1);
		}
		if(ans==0)
			ans=-1;
		printf("For Day #%d:\n%d\n",r,ans);
	}
    return 0;
}
