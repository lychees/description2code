#include <iostream>
#include <string.h>
using namespace std;
 
int main()
{
	string str;
	cin>>str;
	int lenstr = str.length();
	long long int noofcases;
	cin>>noofcases;
	for(long long int i=0;i<noofcases;i++)
	{
		long long int a, b;
		cin>>a>>b;
		a=a%lenstr;
		b=b%lenstr;
		if(a==0)
		{
			a=lenstr;
		}
		if(b==0)
		{
			b=lenstr;
		}
		if(str[a-1]==str[b-1])
		{
			cout<<"Yes"<<endl;
		}else
		{
			cout<<"No"<<endl;
		}
	}
    
    return 0;
}
