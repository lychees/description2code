#include <iostream>
#include <cstdio>
#define ss(n) scanf("%lld", &n)
#define ssf(n) scanf("%lf", &n)
#define ll long long
using namespace std;

int main()
{
    ll t, i, sqsum;
    double ans=1, n, r, s;
    ss(t);
    while(t--)
    {
    	ssf(n);
    	ssf(r);
    	sqsum=n*(n+1)*(2*n+1)/6;
    	s=(n*(n+1)/2)*(n*(n+1)/2);
    	s/=r;
    	s/=sqsum;
    	ans*=s;
    }
    printf("%0.4lf", ans);
    return 0;
}
