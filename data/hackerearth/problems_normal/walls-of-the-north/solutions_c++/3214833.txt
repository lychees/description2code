using namespace std;

int main(){
	int t,n;
	cin>>t;
	
	while(t--){
		cin>>n;		
		int lh=1,ll=1,ch,cl,x,y;
		
		cin>>x;
		n--;
		
		while(n--){
			cin>>y;
			if(y>x){
				ch=max(lh,ll+1);
				cl=ll;
			}else if(y==x){
				ch=max(lh,ll+1);
				cl=max(ll,lh+1);
			}else{
				ch=lh;
				cl=max(ll,lh+1);
			}
			ll=cl;
			lh=ch;
			x=y;
		}
		
		cout<<max(ll,lh)<<endl;
	}
	
	return 0;
}
