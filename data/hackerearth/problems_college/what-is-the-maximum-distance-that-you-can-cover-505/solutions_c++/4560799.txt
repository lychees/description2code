#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    cin >> n;
    double s[n],c[n],fuel;
    double dist,m=99999999,temp;
    for(i=0;i<n;i++) {
    	cin >> s[i];
    }
    
    for(i=0;i<n;i++) {
    	cin >> c[i];
    }
    
    cin >> fuel;
    
    for(i=0;i<n;i++) {
 		temp = c[i]/s[i];
 		m = m<temp?m:temp; 
    }
    cout.setf(std::ios::fixed);
    cout.precision(3);
    dist = (int)((fuel/m)*1000);
    dist /= 1000;
    cout << dist;
    return 0;
}
