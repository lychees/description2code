
def hardness(s):
    consective_consonants = 0
    consective_condition = False
    consonants, vowels = 0, 0
    for c in s:
        if c not in ['a', 'e', 'i', 'o', 'u', 'y']:
            consonants += 1
            consective_consonants += 1
            if consective_consonants == 3:
                consective_condition = True
        else:
            vowels += 1
            consective_consonants = 0
    # print(consonants, vowels, consective_condition)
    if (consonants > vowels) or consective_condition:
        return 'hard'
    return 'easy'


def main():
    T = int(raw_input().strip())
    for t in range(T):
        s = raw_input().strip()
        ans = hardness(s)
        print ans


if __name__ == '__main__':
    main()
