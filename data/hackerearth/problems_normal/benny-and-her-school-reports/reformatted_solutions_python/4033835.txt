# print 'Hello World!'

K = raw_input()
K = int(K)
while (K > 0):
    N, M, X = map(int, raw_input().split())
    arr = map(int, raw_input().split())
    ans = (N+1)*X - sum(arr)
    if (ans <= M):
        print min(max(ans, 1), M)
    else:
        print 'Impossible'
    K = K-1
