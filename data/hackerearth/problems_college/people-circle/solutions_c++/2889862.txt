#include <stdio.h>
#include <stdlib.h>

int main()
{
int t;
scanf("%d", &t);
while(t--)
{
int m, f, k, i, j=-1, x;
scanf("%d %d %d", &m, &f, &k);
int n = m+f;
char a[n];
for(i =0; i<n; i++)
{
a[i] = 'M';
}
for(i=0;i<f;i++)
{
x = k;
while(x)
{
++j;
j = j%n;
if(a[j]!='F') x--;
}
a[j] = 'F';
}
for(i =0; i<n; i++)
{
printf("%c", a[i]);
}
printf("\n");

}
return 0;
}
