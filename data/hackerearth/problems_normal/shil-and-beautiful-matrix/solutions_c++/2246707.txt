#include <iostream>
#include <cassert>

using namespace std;

int n, m;
int mat[1001][1001];
long long fact[1000001];
int freq[1000001];

int main()
{
	cin >> n;
	cin >> m;
	
	fact[0] = 1;
	long long mod = 1000000007LL;
	for (int i = 1 ; i<1000001 ; i++) {
		fact[i] = (i*fact[i-1])%mod;	
	}
	
	for (int i=1; i<=n ; i++) {
		for (int j=1 ; j<=m ; j++) {
			freq[i^j]++;
		}
	}
	
	long long ans = 1;
	for (int i = 0 ; i<1000001 ; i++) {
		ans = (ans * fact[freq[i]])%mod;	
	}
	
	assert(ans >= 0);
	printf("%lld\n", ans);
    return 0;
}
