using namespace std;
#define MAX 1000001
bool use[MAX];
long long int a[MAX];
long long int MOD = 1e9 + 7;
int main(){
	for (int i = 2; i < MAX; i++){
		if (!use[i]){
			use[i] = true;
			for (int j = i * 2; j < MAX; j += i){
				if (a[j] == 0){
					a[j] = i;
				}
				use[j] = true;
			}
		}
	
		a[i] += a[i - 1];
		a[i] %= MOD;
	}
	int t;
	scanf("%d", &t);
	while (t--){
		int l, r;
		scanf("%d%d", &l, &r);
		long long int ans = a[r] - a[l - 1];
		while (ans <0){
			ans += MOD;
		}
		ans %= MOD;
		printf("%lld\n", ans);
	}
	return 0;
}
