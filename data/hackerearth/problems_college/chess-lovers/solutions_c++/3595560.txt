#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int d1,d2,ans,t,x1,x2,y1,y2;
	cin>>t;
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2;
		d1=abs(x1-y1);
		d2=abs(x2-y2);
		if((d1+d2)%2)
		ans=-1;
		else if(abs(x1-x2)==abs(y1-y2))
		ans=0;
		else
		ans=1;
		cout<<ans<<"\n";
	}


return 0;
}
