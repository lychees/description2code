#include <iostream>
using namespace std;

int main() {
	long long int t,a,b,x,y,q,w;
	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b;
		if(b%2==0)
		{
			x=a/b;
			y=a%b;
			if(y>=b/2)
			{
				cout<<(b*(x+1))<<endl;
			}
			else
			cout<<b*x<<endl;
		}
		else
		{
			q=a/b;
			w=a%b;
			if(w>b/2)
			cout<<(b*(q+1))<<endl;
			else
			cout<<b*q<<endl;
		}
	}
	return 0;
}
