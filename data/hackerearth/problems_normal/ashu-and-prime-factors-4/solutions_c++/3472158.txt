#include<bits/stdc++.h>

const int MAXN = 10+1000000;

int minP[MAXN], cnt[MAXN];
void init(){
	memset(minP, 0, sizeof(minP));
	memset(cnt, 0, sizeof(cnt));

	minP[1] = 1;
	for (int i = 1; i <= int(1e6); i++)
		if (!minP[i]){
			minP[i] = i;
			int j = i+i;
			while (j <= int(1e6)){
				if (!minP[j]){
					cnt[i]++;
					minP[j] = i;
				}
				j += i;
			}
		}
}

int main(){
#ifndef ONLINE_JUDGE
	freopen("aa.inp","r",stdin);
#endif
	std::ios::sync_with_stdio(false);

	init();

	int test, n;
	std::cin >> test;
	//std::cout << minP[2];
	while (test--){
		std::cin >> n;
		std::cout << cnt[n]+1 << std::endl;
	}

	return 0;
}
