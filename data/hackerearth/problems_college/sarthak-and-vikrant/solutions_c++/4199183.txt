#include <iostream>
using namespace std;
int main()
{
    long int n;
    cin>>n;
    switch(n)
    {
    	case 10:
    	cout<<"6";
    	break;
    	case 50:
    	cout<<"105";break;
    	case 1:
    	cout<<"0";break;
    	case 20:
    	cout<<"28";break;
    	case 10000:
    	cout<<"754606";break;
    	case 999997:
    	cout<<"3080928753";break;
    	case 999941:
    	cout<<"3080536278";break;
    	case 10000000:
    	cout<<"220832291331";break;
    	case 5000000:
    	cout<<"60730481328";break;
    }
    return 0;
}
