
#include <iostream>
#include <string.h>
using namespace std;
int main()
{
    int t=0;
    cin>>t;
    while(t--)
    {
        char s[21]={'\0'};
        cin>>s;
        int len = strlen(s);
        int i = len;
        int ans =-1;
        for(i;i>0;--i)
        {
            ans =-1;
            int begin =0;
            int end = begin + i -1;
            while(end<len)
            {
                int j = begin;
                int l = end;
                bool btrue = true;
                while(btrue && j<=l)
                {
                    if(s[j]!=s[l])btrue =false;
                    ++j;
                    --l;
                }
                if(btrue)
                {
                    ans = i;
                    break;
                }
                    
                ++begin;
                ++end;// = begin + i -1;
            }
            if(ans!=-1)
                break;
        }
        cout<<ans<<endl;
    }
    return 0;
    
}
