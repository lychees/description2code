#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(;t;t--)
    {
    	long long val[100005],n;
    	val[1]=1;
    	val[2]=2;
    	val[3]=4;
    	for(int i=4;i<=100000;i++)
    	{
    		val[i]=(val[i-1]+val[i-2]+val[i-3])%1000000007;
    	}
    	cin>>n;
    	cout<<val[n]<<endl;
    }
}
