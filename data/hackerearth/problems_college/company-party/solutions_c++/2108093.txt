#include <bits/stdc++.h>

using namespace std;

int n,i,p[2010],m[2010],max_;

inline int manager_seq(int j)
{
    if(p[j] == -1)
        return 1;
    if(m[j])
        return m[j];
    m[j] = 1 + manager_seq(p[j]);
    return m[j];
}

int main()
{
    cin >> n;
    for(i=1;i<=n;++i)
        cin >> p[i];
    for(i=1;i<=n;++i)
        m[i] = manager_seq(i);
    max_ = 0;
    for(i=1;i<=n;++i)
        max_ = max(max_,m[i]);
    printf("%d\n",max_);
    return 0;
}
