#include<iostream>
#include<math.h>
using namespace std;

int main(){
   int t;
   cin>>t;
    long n;
    double a;
   while(t--){
    cin>>n;
    a=(-1+sqrt(1+8*n))/2;
    a=ceil(a);
    if((int)a%3==1) cout<<"LB"<<endl;
    else if((int)a%3==2) cout<<"BS"<<endl;
    else
    cout<<"SL"<<endl;

   }
    return 0;
}
