#include<iostream>
#include<cstdio>
using namespace std;
#define MAXN 1000001
#define MOD 1000000007
long long int dp[MAXN];
int main(){
	dp[0]=1;
	dp[1]=1;
	for(int i=2;i<=MAXN;i++){
		dp[i]+=dp[i-1]+dp[i-2]*(i-1);
		dp[i]%=MOD;
	}
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		printf("%lld\n",dp[n]);
	}
}
