	int t;

	cin>>t;

long long	int f[100];

long long	int a[1000001]={0};

	a[0]=1;

	a[1]=1;

	f[0]=0;

	for(int i=2;i<=31;i++)

{	f[i]=f[i-1]+f[i-2];

if(f[i]<=1000000)

a[f[i]]=1;



}

a[0]=0;

a[2]=0;

a[3]=0;



for(int i=4;i<=1000000;i++)

	if(a[i]==0)

		a[i]=a[i-1]+i;

	a[i]=a[i-1];

	

}



//for(int i=4;i<=100;i++)

while(t--)

{

	int n;

	long long int sum=a[n-1];

	int count=0;

long long int ans=sum,k=sqrt(ans);

//cout<<ans<<endl;



for(int i=2;i<=k;i++)

{

	{

		count=1;

	}

}

cout<<"Tic"<<endl;

else

}



}

}
