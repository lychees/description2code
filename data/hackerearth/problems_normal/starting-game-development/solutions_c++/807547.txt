int binSearch(int a[],int l,int r,int val)
{
	int mid;
	int found=-1;
	while(l<=r)
	{
		mid=(l+r)/2;
		//if(a[mid]==val)	return mid;
		if(a[mid]<=val)
		{
			l=mid+1;
				
		}
		else
		{
			r=mid-1;
		}
	}
	return l-1;
}
