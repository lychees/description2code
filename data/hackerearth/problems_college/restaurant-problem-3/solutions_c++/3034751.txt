#include <bits/stdc++.h>
using namespace std;
#define limit 1000006
int zero[limit];

int main()
{
    string str;
    cin >> str;
    int l = str.size();
    //calculating num of zeros
    for(int i = 0; i < l; i++) {
    	if(i == 0) {
    		if(str[i] == '0') {
    			zero[i] = 1;
    		}
    		else {
    			zero[i] = 0;
    		}
    	}
    	else {
    		if(str[i] == '0') {
    			zero[i] = zero[i - 1] + 1;
    		}
    		else {
    			zero[i] = zero[i - 1];
    		}
    	}
    }
    int ans = 0;
    //length 1
    for(int i = 0; i < l; i++) {
    	if((str[i] - '0') % 8 == 0) {
    		ans++;
    	} 
    }
    //length 2
    for(int i = 1; i < l; i++) {
    	int num = (str[i - 1] - '0') * 10 + (str[i] - '0');
    	if((str[i - 1] != '0') && num % 8 == 0) ans++;
    }
    //length 3
    for(int i = 2; i < l; i++) {
    	int num = (str[i - 2] - '0') * 100 + (str[i - 1] - '0') * 10 + (str[i] - '0');
    	if(str[i - 2] != '0' && num % 8 == 0) {
    		ans++;
    		ans += (i - 2);
    		if(i - 3 >= 0) {
    			ans -= zero[i - 3];
    		}
    	}
    }
    cout << ans << endl;
    return 0;
}
