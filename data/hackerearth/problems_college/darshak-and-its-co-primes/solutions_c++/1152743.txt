#include<bits/stdc++.h>
using namespace std;
bool RelativelyPrime (int a, int b) { // Assumes a, b > 0
   return (a<b) ? RelativelyPrime(b,a) : !(a%b) ? (b==1) : RelativelyPrime (b, a%b);
}
int main()
{
	int t;
	long long a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld %lld",&a,&b);
		if(RelativelyPrime(a,b))
			printf("Is a Co-Prime\n");
		else
			printf("Not a Co-Prime\n");
	}

	return 0;
}
