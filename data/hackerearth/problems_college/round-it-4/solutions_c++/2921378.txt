#include <iostream>
#include <string>
using namespace std;
int main()
{
    int t=0;
    int i=0,j=0,h=0;
    cin>>t;
    for(h=0;h<t;h++)
    {
       int n=0,b=0,temp=0;
       cin>>n>>b;
       if(n%b==0)
       {
           cout<<n<<"\n";
       }
       else
       {
           int temp1=0;
           temp=n;
           while(temp%b!=0)
           {
               temp--;
           }
           temp1=n;
           while(temp1%b!=0)
           {
               temp1++;
           }
          if(n-temp==temp1-n)
          {
              cout<<temp1<<"\n";
          }
          else
          {
              if(n-temp<temp1-n)
              {
                  cout<<temp<<"\n";
              }
              else
              {
                  cout<<temp1<<"\n";
              }
          }
           
       }
    }  
    return 0;
}
