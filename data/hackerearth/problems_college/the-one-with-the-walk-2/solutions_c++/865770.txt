#include <iostream>
using namespace std;
#define MAX9 1000000007

long long dp[1005][1005];
int a, b, c, d;

void preCalc() {
	int i, j;
	for(i=b; i<=d; i++) {
    	dp[a][i] = 1;
    	dp[0][i] = 0;
    }
	for(i=a; i<=c; i++) {
    	dp[i][b] = 1;
    	dp[i][0] = 0;
    }
	for(i=a+1; i<=c; i++) {
    	for(j=b+1; j<=d; j++) {
    		
    		dp[i][j] = dp[i-1][j]+dp[i][j-1]+dp[i-1][j-1];
    		if(dp[i][j] >= MAX9) {
    			dp[i][j] %= MAX9;
    		}
    	}
    }
}

int main()
{
	
	int i, j, t;
    a=b=1; c=d=1004;
    preCalc();
    long long ans;
    cin>>t;
    while(t--) {
    	scanf("%d%d%d%d", &a, &b, &c, &d);
    	if(a==c || b==d) {
    		printf("1\n");
    	}
    	else {
    		ans = dp[c - a + 1][d - b + 1];
	    	printf("%lld\n", ans);
    	}
    }
    return 0;
}
