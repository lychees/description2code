#include <iostream>
#include<iomanip>
using namespace std;

int sum(int n)
{
	int t=0;
	while(n!=0)
	{
		t+=(n%10);
		n/=10;
	}
	return t;
}

int main()
{
 	int n,x;
 	double y;
 	cin>>n;
 	int a[n],l,b,tot=0;
 	for(int i=0;i<n;i++)
 		cin>>a[i];
 	cin>>l>>b;
 	for(int i=0;i<n;i++)
 	{
 		x=sum(a[i]);
 		y=(x*l*b*a[i]);
 		if(a[i]>0&&a[i]<11)
 			y=y*(0.75);
 		else if(a[i]>10&&a[i]<21)
 			y*=(0.7);
 		else
 			y*=(0.8);
 		tot+=y;
 	}
 	cout<<tot<<endl;
    return 0;
}
