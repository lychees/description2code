#include <bits/stdc++.h>
using namespace std;

bool tc()
{
	int n, l;
	cin >> n >> l;
	vector<pair<int, int>> a(n);
	for (int i = 0; i < n; i++)
		cin >> a[i].first >> a[i].second;
	sort(a.begin(), a.end());
	map<int, int> v;
	for (auto &ai : a) {
		if (v.find(ai.first) == v.end())
			v[ai.first] = ai.first;
		for (auto vit = v.begin(); vit != v.end(); vit++) {
			if (vit->second >= ai.first && ai.second > vit->second &&
				ai.second - vit->first <= l) {
				vit->second = ai.second;
				if (vit->second - vit->first == l)
					return true;
			}
		}
	}
	return false;
}

int main()
{
	ios::sync_with_stdio(false), cin.tie(nullptr);
	int t;
	cin >> t;
	for (int i = 0; i < t; i++)
		cout << (tc() ? "Yes\n" : "No\n");
}
