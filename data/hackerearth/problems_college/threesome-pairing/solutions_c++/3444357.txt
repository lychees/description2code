#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n],s1=0,s2=0,s3=0,i;
    	for(i=0;i<n;i++) {cin>>a[i];
    	if(a[i]%3==0) s1++;
    	else if(a[i]%3==1) s2++;
    	else s3++;}
    	cout<<(s2*s3)+(s1*(s1-1))/2<<endl;
    }
    return 0;
}
