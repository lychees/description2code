#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	string s;
    	cin>>s;
    	long long sum=0;
    	int k=0;
    	
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='H')
    		k=k+1;
    		else
    		k=0;
    		sum+=k;
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
