#include<bits/stdc++.h>
#define ll long long
#define MAX 50000
#define rep(i,a,b) for(ll i=a;i<b;i++)
#define IOS ios::sync_with_stdio(false)
#define vi vector < ll >
#define pb push_back
#define mod 1000000007
using namespace std;

vi a,dp;

int main()
{
    IOS;
    a.pb(-50);dp.pb(-50);
    rep(i,1,115)
    {
        a.pb(i);
    }
    rep(i,1,115)
    {
        if(a[i]%12==0)
        {
            ll s=a[i]/12;
            dp.pb(s);
        }
        else if(a[i]%10==0)
        {
            ll s=a[i]/10;
            dp.pb(s);
        }
        else dp.pb(-1);
    }
    rep(i,13,115)
    {
        if(a[i]%2!=0)
            continue;
        if(dp[i-10]==-1&&dp[i-12]==-1)
            continue;
        if(dp[i-10]==-1||dp[i-12]==-1)
            dp[i]=max(dp[i-10],dp[i-12])+1;
        else dp[i]=min(dp[i-10],dp[i-12])+1;
    }
    ll t;
    cin>>t;
    rep(i,0,t)
    {
        ll n;
        cin>>n;
        cout<<dp[n]<<endl;
    }
    return 0;
}
