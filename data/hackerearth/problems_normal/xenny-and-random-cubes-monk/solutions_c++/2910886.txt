#include<iostream>
using namespace std;


int c[11][150];
long long int ans=0;
int n, k;
char temp;
char s[20];
bool visited[10];

void fun(int index, int j, int total)
{
	if(index>n)
		return;
	if(j==k)
	{
		ans+=total;
		ans%=1000000007;
		visited[index]=false;
		return;
	}

	/*for(int i=index; n-i >= k-j ; i++)
	{
		if(c[i][s[j]]==0)
			continue;
		fun(i+1, j+1, total*c[i][s[j]]);
	}*/
	for(int i=0; i<n; i++)
    {
        if(visited[i]==true || c[i][s[j]]==0)
            continue;
        //cout<<i<<" "<<j<<endl;
        visited[i]=true;
        fun(i, j+1, total*c[i][s[j]]);
    }
    visited[index]=false;
	return;
}


int main()
{
	cin>>n>>k;

	for(int i=0; i<10; i++)
        visited[i]=false;

	for(int i=0; i<11; i++)
		for(int j=0; j<130; j++)
			c[i][j]=0;

	for(int i=0; i<n; i++)
		for(int j=0; j<6; j++)
		{
			cin>>temp;
			c[i][temp]++;
		}

	for(int i=0; i<k; i++)
		cin>>s[i];

	fun(0, 0, 1);
	cout<<ans<<endl;

	return 0;
}
