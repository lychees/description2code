#include <iostream>
using namespace std;

int main()
{
	int n,count,s,a,p,t;
    cin>>n;
    for(n;n>0;n--)
    {
    	cin>>s;
    	int a[s];
    	cin>>a[0];
    	count=1;
    	t=1;
    	for(int i=1;i<s;i++)
    	{
    		cin>>p;
    		for(int j=0;j<t;j++)
    		{
    			if(a[j]==p)
    			{
    				count++;
    				goto z;
    			}
    		}
    		a[t]=p;
    		t++;
    		
    		z:;
    	}
    	cout<<t<<endl;
    }
    return 0;
}
