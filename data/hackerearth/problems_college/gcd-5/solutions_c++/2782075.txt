#include<bits/stdc++.h>
using namespace std;
int main()
{   
	int t;
    cin>>t;
    while(t--)
    {int n,f;
    f=0;
    cin>>n;
    long long a[n+1];
    for(int i=0;i<n;i++)
    {cin>>a[i];
    }
    for(int i=0;i<n-1;i++)
    {if(a[i]<=a[i+1])
       f++;
    }
     if(f==n-1)
    cout<<"YES"<<endl;
    else 
    cout<<"NO"<<endl;
    }
    return 0;
}
