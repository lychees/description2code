#include <iostream>
using namespace std;

int main()
{
     int t;
     cin>>t;
    for(int i=0;i<t;i++){
    	string a, b;
    	cin>>a;
    	int flag=1;
    	cin>>b;
    	string t1=a+b;
    	int l=0,r=t1.length()-1;
    	while(l<r){
    		if(t1[l]==t1[r]) {l++;r--;}
    		else { flag=0; break;}
    	}
    	if(flag==1) {cout<<"YES"<<endl;continue;}
    	t1=b+a;
    	l=0,r=t1.length()-1;
    	flag=1;
    	while(l<r){
    		if(t1[l]==t1[r]) {l++;r--;}
    		else { flag=0; break;}
    	}
    	if(flag==0) cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    }
    return 0;
}
