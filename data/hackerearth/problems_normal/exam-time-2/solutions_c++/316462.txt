#include <iostream>
using namespace std;
#define lm int

 
 int scan()
{
int t=0;
char c;
c=getchar_unlocked();
while(c<'0' || c>'9')
c=getchar_unlocked();
while(c>='0' && c<='9')
{
t=(t<<3)+(t<<1)+c-'0';
c=getchar_unlocked();
}
return t;
}
int main()
{
	int q;
    lm n;
	q = scan();

	for (int J=0;J<q;J++)
	{
        n = scan();
        int  ans = n;
		int i;
	for(i=2;i*i <= n;i++)
	{
		    if (n % i == 0)
	       	ans -= ans / i;
  	       	while (n % i == 0)
     	    n /= i;
	}
		    if (n > 1)
	{
			ans -= ans / n;
	}
			printf("%d\n",ans);

	}

    
    return 0;
}
