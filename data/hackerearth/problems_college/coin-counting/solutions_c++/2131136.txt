#include <iostream>
using namespace std;

int count(int *a,int m,int n)
{
	if(n==0)
	return 1;
	else if (n<0)
	return 0;
	else if (n>0 && m<=0 )
	return 0;
	else
	{
		return (count(a,m-1,n)+count(a,m,n-a[m-1]));
	}
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	int n;
   	cin>>n;
   	int arr[]={1,3,5};
   	int sum=n;
   	cout<<count(arr,3,sum)<<endl;
   }
    return 0;
}
