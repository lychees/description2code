#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long a,b;
        cin>>a>>b;
        int ans[b-a+1];
        for(int i=0;i<=b-a;i++)ans[i]=0;
        int temp;
        for(int i=1;i<=10000;i++)
        {
            temp=a/i;
            if(a%i!=0)temp++;
            for(int j=temp;j*i<=b;j++)
            {
                if(j>i)ans[i*j-a]+=2;
                if(i==j)ans[i*j-a]++;
            }
        }
        int max=0;
        for(int i=0;i<=b-a;i++)
        {
            if(ans[i]>max)max=ans[i];
        }
        int count=1;
        for(int i=0;i<=b-a;i++)
        {
            if(ans[i]==max)count++;
        }
        cout<<count<<endl;
    }
}
