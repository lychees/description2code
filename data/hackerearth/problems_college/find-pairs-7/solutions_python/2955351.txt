def pairs(lis,l):
	c=0
	i=0
	while i<l :
		j=i
		while j<l and j>=i:
			if lis[i] == lis[j]:
				c =c+1
			j=j+1
		i=i+1
	return c
	
	
t= input()
re=[]
for i in range(t):
	l=input()
	s=raw_input()
	s=s.split()
	re.append(pairs(s,l))
