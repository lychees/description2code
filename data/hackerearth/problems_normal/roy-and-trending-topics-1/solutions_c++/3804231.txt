#include<iostream>
#include<list>
#include<map>
#include<algorithm>
#include<string>
#include<utility>
#include<queue>
using namespace std;
int main()
{
map<long long int,long long int> m;
priority_queue< pair<long long int,long long int> > prio;

long long int n,presx,p,l,c,s,id;
cin>>n;
for(long i = 0;i<n;i++)
{
cin>>id>>presx>>p>>l>>c>>s;
long long int newx = (p*50)+(l*5)+(c*10)+(s*20);
long long int change = (newx-presx);
pair<long long int,long long int> p;
p.first=change;
p.second=id;
prio.push(p);
m[id]=newx;
}
for(int i = 0;i<5;i++)
{
pair<long long int,long long int> p = prio.top();
cout<<p.second<<" "<<m[p.second]<<endl;
prio.pop();
if(prio.size()==0)
break;
}
return 0;
}
