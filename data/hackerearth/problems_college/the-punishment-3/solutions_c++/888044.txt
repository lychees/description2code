#include <bits/stdc++.h>
using namespace std;

void solve()
{
    int n;
    int k;
    scanf("%d%d", &n, &k);
    int arr[n + 1];
    if(k > n / 2) {
        k = n - k;
    }

    for(int i = 1; i < k; i++) {
        arr[i] = 9;
    }
    arr[k] = 8;
    for(int i = k + 1; i <= n - k; i++) {
        arr[i] = 9;
    }
    for(int i = n - k + 1; i < n; i++) {
        arr[i] = 0;
    }
    arr[n] = 1;
    for(int i = 1; i <= n; i++) {
        printf("%d", arr[i]);
    }
    printf("\n");
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        solve();
    }
    return 0;
}
