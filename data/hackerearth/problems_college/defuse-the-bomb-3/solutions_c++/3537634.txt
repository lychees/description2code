#include<bits/stdc++.h>
using namespace std;
#define ll long long
string a,b;
ll lena,lenb;
ll solve(int i)
{
	ll temp=0;
	for(int j=i;j<=i+lenb-1;++j)
	{
		temp=10*temp+(a[j]-'0');
	}
	return temp;
}
int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		cin>>a>>b;
		lena=a.length();
		lenb=b.length();
		ll aa=0;
		ll bb=0;
		for(int i=0;i<lenb;++i)
		{
			bb=10*bb+(b[i]-'0');
		}
		for(int i=0;i+lenb-1<=lena-1;++i)
		{
			aa=solve(i);
			cout<<aa*bb<<endl;
		}
	}
	return 0;
}
