#include <bits/stdc++.h>
using namespace std;

int dp[109];

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	for(int i=0;i<n+1;i+=1)
    		dp[i]=0;
    	//memset(dp,0,sizeof(dp));
    	dp[0]=1;
    	dp[6]=1;
    	dp[8]=1;

		for(int i=9;i<=n;i+=1){
			if(dp[i-6]!=0) dp[i]=dp[i-6]+1;
			if(dp[i-8]!=0) dp[i]=dp[i-8]+1;
		}
		if(dp[n]==0) cout<<"-1"<<endl;
		else cout<<dp[n]<<endl;
    }
    return 0;
}
