#include <bits/stdc++.h>
using namespace std;

int lis( int arr[], int n )
{
    int *lis, i, j, max = 0,val1,val2;
    lis = (int*) malloc ( sizeof( int ) * n );
 
    
    for (i = 0; i < n; i++ )
        lis[i] = 1;
 
    
    int sgn1,sgn2;
    for (i = 1; i < n; i++ ){
        for (j = 0; j < i; j++ ){
            val1=abs(arr[i]); sgn1=arr[i]/val1;
            val2=abs(arr[j]); sgn2=arr[j]/val2;
            if ( val1 > val2 && sgn1!=sgn2 && lis[i] < lis[j] + 1)
                lis[i] = lis[j] + 1;
        }
    }
    
    for (i = 0; i < n; i++ )
        if (max < lis[i])
            max = lis[i];
 
    
    free(lis);
 
    return max;
}
 

int main()
{
    int a[5002];
    int n,i;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>a[i];
    }
    printf("%d\n", lis( a, n ) );
    return 0;
}
