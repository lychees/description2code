#include <bits/stdc++.h>

using namespace std;

typedef long long ll ;

int main()
{
//    freopen("in.txt", "r", stdin) ;
    ios_base ::sync_with_stdio(false) ;
    cin.tie(false), cout.tie(false) ;
    ll m, n, tst, k, l, x, i, a[100010] ;
    cin >> tst ;
    while( tst-- )
    {
        ll ans = 0 ;
        cin >> n >> x ;
        for( i = 0; i < n; i++ )
            cin >> a[i] ;
//        for( i = 0; i < n; i++ )
//            cout << a[i] << "   " ;
//        cout << endl ;
        for( i = 1; i < n; i++ )
        {
            if( a[i] <= a[i-1] )
            {
                k = llabs(a[i] - ( a[i-1] - 1 )) ;
                m = k / x ;
                if( k % x )
                    m++ ;
                ans += m ;
                k = x * m ;
                a[i] += k ;
            }
        }
        cout << ans << endl ;
    }
    return 0 ;
}
