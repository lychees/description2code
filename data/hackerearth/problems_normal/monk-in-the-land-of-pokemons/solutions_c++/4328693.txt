#include <cstdio>
#include <vector>



typedef long long ll;

const int NMAX = (int)1e5 + 3;

std::vector<int> hv[NMAX];

ll hash(int s)
{
	ll res = ((ll)s * (s+3)) % NMAX;
	return res;
}

void insert(int s)
{
	int key = hash(s);
	hv[key].push_back(s);
}

bool search(int s) 
{
	int key = hash(s);
	for(int i = 0; i < (int)hv[key].size(); i++) {
		if(hv[key][i] == s) {
			hv[key].erase(hv[key].begin()+i);
			return true;
		}
	}

	return false;
}

int main()
{
	int t,n,a,b,res;
	scanf("%d", &t);
	while(t--) {
		res = 0;
		scanf("%d", &n);
		for(int i = 0; i < NMAX; i++)
			hv[i].clear();

		for(int i = 0; i < n; i++) {
			scanf("%d %d", &a, &b);
			insert(a);
			if(!search(b))
				res++;
		}

		printf("%d\n", res);
	}
}
