#include <iostream>
using namespace std;

int main()
{
    string s;
    cin >> s;
    int k = s.length()/2;
   // cout << k;
    int n = s.length();
   // cout << n;
    for(int i=0; i<k; i++){
    	if(s[i] != s[n-i-1]){
    		cout << "NO"; 
    		break;
    	}
    	else if(i==k-1){
    		cout << "YES";
    	}
    }
    return 0;
}
