#include <bits/stdc++.h>
using namespace std;
#define max(a,b)  ((a)>(b)?(a):(b))
#define min(a,b)  ((a)<(b)?(a):(b))
int main()
{
    //cout << "Hello World!" << endl;
    int t,s,d,x,y;
    cin>>t;
    while(t--)
    {
    	cin>>s>>d;
    	if(s>d){
    	x=(s+d)/2;
    	y=(s-d)/2;
    	if(x+y==s && abs(x-y)==d)
    	{
    	cout<<max(x,y)<<" "<<min(x,y)<<endl;
    	}
    	else
    	cout<<"impossible"<<endl;
    	}
    	else
    	cout<<"impossible"<<endl;
    }
    return 0;
}
