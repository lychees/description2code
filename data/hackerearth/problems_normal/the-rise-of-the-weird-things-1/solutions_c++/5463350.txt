#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n; cin >> n;
	vector<int> e; vector<int> o;
	int so = 0, se = 0;
	for(int i=0;i<n;i++)
	{
		int x; cin >> x;
		if(x%2 == 0)
		{
			se += x;
			e.push_back(x);
		}
		else
		{
			so += x;
			o.push_back(x);
		}
	}
	sort(e.begin(), e.end());
	e.push_back(se);
	
	sort(o.begin(), o.end());
	o.push_back(so);
	
	for(int i=0;i<o.size();i++)
	{
		e.push_back(o[i]);
	}
	
	for(int i=0;i<e.size();i++)
	{
		cout << e[i] << " ";
	}
}
