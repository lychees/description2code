#include<bits/stdc++.h>
using namespace std;
int fun(int n)
{
	unsigned int count = 0;
	while (n)
	{
		n &= (n-1) ;
		count++;
	}
	return count;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int n,k,i;
	cin>>n>>k;
	vector< pair<int,int> >a;
	a.resize(n);
	for(i=0;i<n;i++)
	{
		int x;
		cin>>x;
		a[i].second=x;
		a[i].first=fun(x);
	}
	sort(a.begin(),a.end());
	int j=n-1;
	int s=0;
	while(k--)
	{
		s=s+a[j].first;
		j--;
	}
	cout<<s<<endl;
	}	
	return 0;
}
