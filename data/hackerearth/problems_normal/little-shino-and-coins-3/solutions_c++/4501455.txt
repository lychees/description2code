#include <bits/stdc++.h>
using namespace std;
int main()
{
  int n,k;
  string s;
  set< char > st;
  cin>>k>>s;
  n = s.length();
  map<char,int> ct;
  int l = 0,r=0,cnt = 0;
  while(l<n)
  {
    r = l;
    st.clear();
    while(r<n)
    {
      st.insert(s[r]);
      if(st.size()==k)
        cnt++;
      r++;
    }
   l++;

  }
  cout<<cnt<<endl;

}
