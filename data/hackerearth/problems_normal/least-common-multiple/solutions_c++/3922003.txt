using namespace std;
typedef pair<ll, ll>pll;

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0);
	ll n;
	while (cin >> n)
	{
		ll m = n / 2 + 1;
		if (n == 1)
		{
			cout << 1 << endl;
			return 0;
		}
		while (m <= n)
		{
			ll temp = m;
			for (ll i = 2; i * i <= m; i++)
			{
				if (m % i == 0)
				{
					temp = i;
					break;
				}
			}
			ll x = m;
			while (x % temp == 0)
			{
				x /= temp;
			}
			if (x == 1)
				break;
			m++;
		}
		cout << m << endl;
	}
	return 0;
}
