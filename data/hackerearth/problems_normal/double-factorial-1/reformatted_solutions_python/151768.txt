def fac(i):
    a = 1
    for i in xrange(1, i+1):
        a = a*i
    return a


def calc(n, m):
    return fac(fac(n)) % (10**m)


t = input()

while t:
    t -= 1
    strr = raw_input()
    (n, m) = [int(i) for i in strr.split(" ")]

    if n >= 5:
        print 0
    else:
        print calc(n, m)
