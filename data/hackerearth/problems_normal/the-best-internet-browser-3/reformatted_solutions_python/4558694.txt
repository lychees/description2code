t = int(raw_input())

for n in xrange(1, t+1):
    site = raw_input()
    actual = out = len(site)
    if 'www.' in site:
        site = site.replace("www.", '')
        out -= 4
    if '.com' in site:
        site = site.replace(".com", '')
    for i in site:
        if i in 'aeiou':
            out -= 1
    print "%s/%s" % (out, actual)
