tc = int(raw_input())
for i in range(tc):
	n = int(raw_input())
	sum1 = 0
	sum2 = 0
	angersum = 0
	for j in range(n):
		favour = 0
		anger = 0
		favour, anger = raw_input().split()
		favour = long(favour)
		anger = long(anger)
		angersum = angersum + anger
		total = favour+anger
		if total >= sum1:
			sum2 = sum1
			sum1 = total
		elif total >= sum2:
			sum2 = total
	print ((sum1+sum2) - angersum)
