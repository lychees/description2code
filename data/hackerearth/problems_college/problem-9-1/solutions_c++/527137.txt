#include <iostream>
using namespace std;

int main()
{
    int N, Q;
    cin>>N>>Q;
    bool a[N];
    for(int i=0; i<N; i++)
    {a[i]=false;}
    for(int i=0; i<Q; i++)
    {
    	int w,A,B;
    	cin>>w>>A>>B;
    	if(w==0)
    	{
    		for(int j=A;j<=B; j++)
    		{
    			if(a[j])
    			{a[j]=false;}
    			else
    			{a[j]=true;}
    		}
    	}
    	else
    	{
    		int b=0;
    		for(int j=A; j<=B; j++)
    		{
    			if(a[j])
    			{b++;}
    		}
    		cout<<b<<endl;
    	}
    }
    return 0;
}
