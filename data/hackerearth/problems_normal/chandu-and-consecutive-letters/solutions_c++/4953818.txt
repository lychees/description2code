#include <iostream>
#include <string>
using namespace std;

int main()
{
    int n;
    cin >>n;
    while(n--){
    	string str;
    	string str1;
    	cin>>str;
    	int l= str.length();
    	for(int i=0;i<l;i++){
    		if(str[i]!=str[i+1]){
    			 str1+=str[i];
    			
    		}
    	}
    	cout<<str1<<endl;
    	
    }
    return 0;
}
