#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		vector<string>  friends;
		for(int i=0;i<n;i++){
			string x;
			cin>>x;
			friends.push_back(x);
		}
		
		vector<int> mask;
		for(int i=0;i<n;i++){
			int a=0;
			for(int j=0;j<k;j++){
				if(friends[i][j]=='1')
				{
					a=a | (1<<(k-1-j));
				}
			}
			mask.push_back(a);
		}
		int ans=k;
		for(int i=1;i<(1<<k);i++){
			bool isPossible=true;
			for(int j=0;j<n;j++){
				if((mask[j] & i)==0)
				{
					isPossible=false;
				}
			}
			if(isPossible==true){
				int counter=0;
				for(int m=0;m<k;m++){
					if((1<<m)&i)
					{
						counter++;
					}
				}
				ans=min(ans,counter);
			}
		}
		cout<<ans<<"\n";
	}
}
