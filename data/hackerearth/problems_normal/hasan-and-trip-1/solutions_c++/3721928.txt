#include <bits/stdc++.h>
#define ll long long int
#define si(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define MOD (int)1e6+7
using namespace std;
double dist(double a, double b, double x, double y) {
	double dx = abs(a - x);
	double dy = abs(b - y);
	dy *= dy;
	dx *= dx;
	return sqrt(dy + dx);
}
int main() {
	//START
	int n;
	si(n);
	double x[n], y[n], h[n];
	for(int i = 0; i < n; i++) {
		scanf("%lf%lf%lf", &x[i], &y[i], &h[i]);
	}
	double maxh[n];
	maxh[0] = h[0];
	for(int i = 1; i < n; i++) maxh[i] = -10000000000.0;
	for(int i = 1; i < n; i++) {
		for(int j = 0; j < i; j++) {
			maxh[i] = max(maxh[i], maxh[j] - dist(x[i], y[i], x[j], y[j]) + h[i]);
		}
	}
	printf("%.6lf\n", maxh[n-1]);
	return 0;
}
