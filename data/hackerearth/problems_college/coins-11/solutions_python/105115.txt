def isprime(x):
    for i in range(2, x-1):
        if x % i == 0:
            return False
    else:
        return True

n = int(raw_input(''))

while n:
	x = int(raw_input(''))
	if (x == 1):
		print 1
	elif (not isprime(x)):
		print 1
	elif (x == 3):
		print 3
	else:
		print 2
	n -= 1
