#include<bits/stdc++.h>
using namespace std;
#define pb push_back
vector<int> v[1002];
int arr[100005];
int dp[100005];
int hash1[100005];
int main()
{
    int n,t,i,j;
    cin>>n;
    int sum=0;
    int cnt=0;
    for(i=1;i<=n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }
    dp[1]=arr[1];
    hash1[arr[1]]=1;
    for(i=2;i<=n;i++)
    {
        dp[i]=dp[i-1]+arr[i];
        hash1[dp[i]]=i;
    }
    int maxi=-1;
    for(i=1;i<=sum;i++)
    {
        cnt=0;
        int k=-1;
        for(j=1;j<=n;j++)
        {
            int x=dp[j]-i;
            if(x==0)
            {
                cnt++;
                k=j;
            }
            if(x>0&&hash1[x]!=0&&hash1[x]>=k)
            {
                cnt++;
                    k=j;

            }
        }
        v[cnt].pb(i);

        maxi=max(maxi,cnt);
    }
    cout<<maxi<<endl;
    sort(v[maxi].begin(),v[maxi].end());
    for(i=0;i<v[maxi].size();i++)
    {
        cout<<v[maxi][i]<<" ";
    }

}
