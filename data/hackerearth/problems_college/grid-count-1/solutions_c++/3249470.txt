# include<bits/stdc++.h>
using namespace std;
int A[1000][1000];
int main()
{
    for(int i=0;i<1000;i++)
        for(int j=0;j<1000;j++)
            A[i][j]=0;
    int n,k;
    long long ans,sum,counter=0,inter;
    cin>>n>>k;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        {
            cin>>((A[i][j]));
        }
    for(int i=0;i<=n-k;i++)
    {
        sum=0;
        for(int a=i;a<i+k;a++)
        {
            for(int b=0;b<k;b++)
                sum+=A[a][b];

        }
        inter=cbrt(sum);
        if(inter*inter*inter==sum||(inter+1)*(inter+1)*(inter+1)==sum)
            counter++;
        for(int j=k;j<n;j++)
        {
            for(int l=i;l<i+k;l++)
            {
                sum+=A[l][j];
                sum-=A[l][j-k];
            }
           inter=cbrt(sum);
        if(inter*inter*inter==sum||(inter+1)*(inter+1)*(inter+1)==sum)
            counter++;

        }

    }
    cout<<counter<<"\n";
}
