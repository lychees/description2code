const int MAX = 1e5 + 10;
using namespace std;
vector<int>v[MAX];
int vis[MAX];
int ans = 0;
void dfs(int x)
{
	vis[x] = 1;
	ans++;
	for (int i = 0; i < v[x].size(); i++)
	{
		int cur = v[x][i];
		if (!vis[cur])
			dfs(cur);
	}
}

		for (int i = 1; i <= r; i++)
		{
			cin >> x >> y;
			v[x].pb(y);
			v[y].pb(x);
		}
		ll cnt = 1;
		for (int i = 1; i <= n; i++)
		{
			if (!vis[i])
			{
				ans = 0;
				dfs(i);
				if (ans > 1)
					cnt = (cnt*((ans * (ans - 1)) / 2) % mod)%mod;
			}
		}
