#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long long int dp[1005];
    memset(dp, 0, sizeof(dp));
    dp[0] = 1;
    dp[1] = 1;
    for(int i=2; i<1005; i++)
        dp[i] = dp[i-1] + dp[i-2];
    while(t--)
    {
        int n;
        scanf("%d",&n);
        printf("%lld\n",dp[n]);
    }
    return 0;
}
