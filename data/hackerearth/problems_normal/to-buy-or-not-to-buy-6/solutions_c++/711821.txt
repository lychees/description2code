vi mark;
vector<vi> A;
int n,m;

void dfs(int in,int cost)
{
    mark[in]=1;
    int i;
    FORI(i,1,n+1)
    {
        if(A[in][i]==-1) continue;
        if(mark[i]==0 && A[in][i]<cost)
            dfs(i,cost);
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int cost,query;
        int i,j,k;
        int a,b;
      //  cout<<"test"<<endl;
        A = vector<vi> (1001,vector<int>(1001,-1));
        cin>>n>>m;
        //cout<<"test"<<endl;
        FOR(i,m)
        {
            scanf("%d%d%d",&a,&b,&cost);
            A[a][b]=A[b][a]=cost;
        }
        //cout<<"test"<<endl;
        cin>>query;
        int ans=0;
        int p,q;
        FOR(i,query)
        {
            cin>>p>>q;
            mark = vi(1005, 0);

            dfs(p,A[p][q]);
            if(mark[q]==0){
              ans++;
            }
        }

        int tgcd=__gcd(query,ans);
        ans/=tgcd;
        query/=tgcd;


        cout<<ans<<"/"<<query<<"\n";
    }
    return 0;
}
