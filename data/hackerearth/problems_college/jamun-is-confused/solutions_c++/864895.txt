#include <iostream>
#include <stdio.h>
#include <vector>

using namespace std;

typedef long long int Lint;

Lint GCD( Lint a, Lint b ) {
	
	if( ( a == 1 ) || ( b == 1 ) )
		return 1;
	if( ( a % b ) == 0 )
		return b;
	else
		return GCD( b, ( a % b ) );
}

int main()
{
	Lint t, n, i, gcd, flag;
	
	scanf("%lld", &t );
	
	while( t-- ) {
			
		scanf("%lld", &n );
		vector<Lint> A( n );
		
		flag = 0;
		for( i = 0;i < n;i++ ) {
			scanf("%lld", &A[ i ] );
			if( A[ i ] == 0 )
				flag = 1;
		}
		
		if( flag )
			printf("1\n");
		else {
			gcd = A[ 0 ];
			
			for( i = 1;i < n;i++ )
				gcd = GCD( A[ i ], gcd );
			
			printf("%lld\n", gcd );
		}
	}
	
	return 0;
}
