#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n;
   cin>>n;
   int arr[100012];
   long long int sum =0;
   for(int i=0;i<n;i++)
   {
   	cin>>arr[i];
   }
  long long  int mx =0;
   for(int i=0;i<n;i++)
   {
   	sum = sum+arr[i];
   	if(sum<0) sum=0;
   	if(mx<sum) mx = sum;
   }
   
   
   cout<<mx<<endl;
    return 0;
}
