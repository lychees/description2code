#include<bits/stdc++.h>
using namespace std;
int main()
{
	char s[109];
	int n=0,i,j,k,c=0;
	scanf("%[^\n]",&s);
//	printf("%s\n",s);
	for(i=0;s[i]!='\0';i++)
	{
		j=(int)s[i];
		c+=j;
		n+=1;
	}
//	printf("%d\n",c);
	k=c/n;
//	printf("%d\n",k);
	if(k&1)
	{
		int a[299]={0};
		for(i=0;i<n;i++)
		{
			j=(int)s[i];
			a[j]+=1;
		//	printf("%c %d\n",s[i],a[j]);
		}
		int m=-1;
		char t;
		for(i=0;i<299;i++)
		{
			if(a[i]>m)
			{
				m=a[i];
				t=(char)i;
			//	printf("%d %c\n",m,t);	
			}
		}
		printf("%c",t);
	}
	else
	{
		for(i=n-1;i>=0;i--)
		{
			printf("%c",s[i]);
		}
//		printf("\n");
	}
	return 0;
}
