#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t,n,i,j,p,x;
    cin>>t;
    while(t--)
    {
        cin>>n;
        vector<ll> v;
        p=0;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        for(i=0;i<n;i++)
        {
            for(j=i;j<n;j++)
            {
                if(v[i]==v[j])
                    p++;
            }
        }
        cout<<p<<'\n';
    }
    return 0;
}
