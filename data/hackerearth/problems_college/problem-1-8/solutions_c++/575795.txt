#include <iostream>
#include <string>

int main()
{
	std::string partial;
	
	std::cin >> partial;
	
	int words;
	std::cin >> words;
	
	std::string shortestMatch;
	std::string word;
	
	while (words-- > 0) {
		std::cin >> word;
		
		// Do we already have a shorter word?  If so, fast-fail this one.
		if (shortestMatch.size() != 0 && shortestMatch.size() < word.size()) {
			continue;
		}
		
		// Is the word shorter than the partial word entered?  If so, fast-fail.
		if (word.size() < partial.size()) {
			continue;
		}
		
		// All we have to do now is verify that the input word starts with the partial word.
		if (partial.compare(0, partial.size(), word, 0, partial.size()) == 0) {
			shortestMatch = word;
		}
	}
	
	std::cout << (shortestMatch.size() != 0 ? shortestMatch : partial) << std::endl;
	
    return 0;
}
