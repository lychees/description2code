#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t;
    cin>>t;
    cout.setf ( ios::fixed ); 
    while(t--)
    {
    	double p;
    	int tm;
    	cin>>p>>tm;
    	
    	double ans = 0.0;
    	ans = 1 - pow(1 - p, (double)tm/1440);
    	
    	cout.precision(4);
    	
    	cout<<ans<<endl;
    }
    return 0;
}
