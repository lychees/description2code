#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n ,k ;
	cin >> n >> k ;
	vector<int> a , b ;
	for(int i=1;i<=n;i++) a.push_back(i) ;
	int ans = -1 ;
	for(int i=1;i<=n;i++){
		int rm ;
		rm =(k-1) % a.size() ;
		ans = a[rm] ;
		for(int j=rm+1;j<a.size();j++) b.push_back(a[j]) ;
		for(int j=0;j<rm;j++) b.push_back(a[j]) ;
		a.clear() ;
		for(int j=0;j<b.size();j++) a.push_back(b[j]) ;
		b.clear() ;
	//	for(int j=0;j<a.size();j++) cout << a[j] << " " ;
	//	cout << endl ;
	}
	cout << ans << endl ;
    return 0;
}
