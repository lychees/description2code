#include<bits/stdc++.h>
using namespace std;

vector<vector<int> >adj;
int vertices,edges;
vector<bool>visited;

bool cycle(int u,int p=-1) {
  visited[u]=true;
  for(int i=0;i<adj[u].size();i++) {
    int v=adj[u][i];
    if(visited[v]==false) {
      if(cycle(v,u)) {
        return true;
      }
    } else if(v!=p) {
      return true;
    }
  }
  return false;
}
int main() {
  //freopen("a.cpp","r",stdin);
  scanf("%d%d",&vertices,&edges);
  adj.resize(vertices);
  for(int i=0;i<edges;i++) {
    int foo,bar;
    scanf("%d%d",&foo,&bar);
    foo--,bar--;
    adj[foo].push_back(bar);
    adj[bar].push_back(foo);
  }
  visited.resize(vertices,false);
  for(int i=0;i<vertices;i++) {
    if(visited[i]==false) {
      if(cycle(i)) {
        puts("FHTAGN!");
        return 0;
      }
    }
  }
  puts("NO");
  return 0;
}
