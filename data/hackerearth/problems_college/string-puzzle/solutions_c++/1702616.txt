#include <bits/stdc++.h>
using namespace std;
void swap(char *a,char *b)
{
 char temp=*a;
 *a=*b;
 *b=temp;	
}	
void per(char *a,long int s,long int e)
{
	if(s==e)
	 cout<<a<<endl;
	else
	  {
	  	 for(long int j=s;j<=e;j++)
	  	  {
	  	  	 swap(a[j],a[s]);
	  	  	 per(a,s+1,e);
	  	  	 swap(a[j],a[s]);
	  	  }
	  }
}
int main()
{
   char a[100001],b[100001];
    scanf("%s",&a);
    long int t;
    cin>>t;
    while(t--)
   {
   	 long int s,e;
   	 cin>>s>>e;
   	 int len=strlen(a);
   	 for(int i=0;i<len;i++)
   	 b[i]=a[i];
   	 per(a,s,e);
   	 cout<<endl;
   } 
    return 0;
}
