#include <iostream>
#include <string>
using namespace std;


int main()
{
    int T;
    cin>>T;
    
    while (T--) {
    	string initial, final;
    	
    	cin>>initial>>final;
    	int pos1 = 0;
    	int pos2 = 0;
    	int len = initial.length();
    	
    	bool flag = true;
    	while (true) {
    		for (; initial[pos1]=='#' && pos1<len; pos1++);
    		for (; final[pos2]=='#' && pos2<len; pos2++);
    		
    		if (pos1>=len || pos2>=len)
    			break;
    		
    		if (initial[pos1] != final[pos2]) {
    			flag = false;	break;
    		}
    		
    		if ((initial[pos1]=='A' && pos1>=pos2) || (initial[pos1]=='B' && pos1<=pos2)) {
    			pos1++;		pos2++;
    			continue;
    		}
    		else {
    			flag = false;	break;
    		}
    		
    	}
    	
    	string res = (flag)?"Yes":"No";
    	cout<<res<<endl;
    	
    }
    return 0;
}
