n = raw_input()
strs = raw_input().split(' ')
num = []
count = []
for i in strs:
    num.append(int(i))
    count.append(1)
num.sort()
best = 0
for i in xrange(1, len(num)):
    for j in xrange(0, i):
        if num[i] % num[j] == 0:
            count[i] = max(count[i], 1+count[j])
    best = max(best, count[i])
if best == 1:
    print -1
else:
    print best
