#include <iostream>
using namespace std;
long long int z[10010],o[100010];
int main()
{
    int i,t,n;
    z[0]=0;o[0]=1;
    for(i=1;i<1000;i++)
   	{
   		z[i]=z[i-1]+o[i-1];
   		o[i]=z[i-1];
   	}
   	
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<o[n]<<" "<<z[n]<<endl;	
    }
    return 0;
}
