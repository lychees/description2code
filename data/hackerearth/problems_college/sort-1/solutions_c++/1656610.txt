#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,size[100000],i,j;
    cin>>t;
    while(t>0)
    {
    	cin>>n;
    	i=0;
    	while(n>0)
    	{
    		cin>>size[i];
    		i++;
    		n--;
    	}
    	sort(size,size+i);
    	
    	for(j=0;j<i;j++)
    	cout<<size[j]<<" ";
    	
    	cout<<"\n";
    	
    	t--;
    }
    return 0;
}
