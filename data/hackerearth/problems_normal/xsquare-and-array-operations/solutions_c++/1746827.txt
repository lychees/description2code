#define MAXN 100010
#define ls (node<<1)
#define rs ((node<<1)+1)
#define ii pair<int,int>

vector<ii> vp;

int N;

int arr[MAXN];

int skipf[MAXN],skipb[MAXN];

bool done[MAXN];

inline void ReadInput(void){
	si(N);
	for(int i=1;i<=N;i++){
		si(arr[i]);
		vp.pb(ii(arr[i],i));
	}
}

inline void solve(void){
	arr[0]=mod;
	arr[N+1]=mod;
	ll ans=0;
	int steps=0;
    sort(vp.begin(),vp.end());
	for(int i=0;i<N-1;i++){
		int pos1,pos2,v;
		v=vp[i].second;
		pos1=skipf[v];
		pos2=skipb[v];
        //cout<<v<<" "<<pos1<<" "<<pos2<<endl;
		if(arr[pos1]<arr[pos2]){
			ans+=(arr[pos1]);
			skipf[skipb[v]]=skipf[v];
			skipb[skipf[v]]=skipb[v];
			done[v]=true;
		}else{
			ans+=(arr[pos2]);
			skipf[skipb[v]]=skipf[v];
			skipb[skipf[v]]=skipb[v];
		}
        //cout<<ans<<endl;
	}
	printf("%lld\n",ans );
	vp.clear();
}

inline void Refresh(void){
	for(int i=1;i<=N;i++){
		skipf[i]=i+1;
		skipb[i]=i-1;
		done[i]=false;
	}
}

int main()
{	
	ios_base::sync_with_stdio(false);
	int t; si(t);
	while(t--){
		ReadInput();
		Refresh();
		solve();
	}
    return 0;
}
