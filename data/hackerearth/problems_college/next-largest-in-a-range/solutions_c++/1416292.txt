#include<bits/stdc++.h>
using namespace std;

int main(){
	int N,L,R,Q;
	cin>>N>>L>>R>>Q;
	vector<int> arr(N);
	for(int i=0;i<N;i++){
		cin>>arr[i];
	}
	while(Q--){
		int X;
		cin>>X;
		int ans=INT_MAX;
		for(int i=L;i<=R;i++){
			if(arr[i]>X && arr[i]<ans){
				ans=arr[i];
			}
		}
		if(ans==INT_MAX)ans=-1;
		cout<<ans<<"\n";
	}
}
