#include <iostream>
#define ll long long int
using namespace std;

int main()
{
	int n,l,r,q;

	cin >> n;
	int a[n+1];
	ll sum[n+1];
	sum[0]=0;
	for(int i=1;i<=n;i++){
		cin >> a[i];
		sum[i]=sum[i-1]+a[i];
	}
	cin >> q;
	for(int i=0;i<q;i++){
		cin >> l >> r;
		cout << sum[r] -sum[l-1] << endl;
	}
    return 0;
}
