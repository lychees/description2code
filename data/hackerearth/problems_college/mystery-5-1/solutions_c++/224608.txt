#include<cstdio>
#include<cmath>
#include<iostream>
#include<vector>
#include<map>
#include<stack>
using namespace std;
int main()
{
int T;
char s[10001];
scanf("%d\n",&T);
while(T--)
	{
	int a[26]={0};
	gets(s);
	for(int i=0;s[i];i++)
		{
		a[s[i]-'a']++;
		}
	int max=0;
	for(int i=0;i<26;i++)
		{
		if(max<a[i])
			max=a[i];
		}
	printf("%d\n",max);
	}
return 0;
}
