tc=int(raw_input())
array=[]
for i in range(tc):
    Text=raw_input()
    Pattern=raw_input()
    temp_array=[]
    temp_array.append(Text)
    temp_array.append(Pattern)
    array.append(temp_array)


def chunkstring(string, length):
    return (string[0+i:length+i] for i in range(0, len(string), length))

for i in array:
    count = 0
    something= list(chunkstring(i[1],1))
    for j in something:
        result=i[0].find(j)
        if result != -1:
            count +=1
            break
    if count != 0:
        print ('YES')
    else:
        print ('NO')
