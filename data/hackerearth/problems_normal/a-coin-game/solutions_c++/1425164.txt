#include<iostream>
using namespace std;
int main()
{
	 int i,j=0;
	 int k=1;
	 long int n1,n2;
	 
	 cin>>j;
	 
	 int arr[1000000]={0};
	 
	 for(i=1;i<1000000;i++)
	 {
	 	 if(arr[i]==0)
	 	 {
	 	 	arr[i]=i+k;
	 	 	
	 	 	if(i+k<1000000)
	 	 	  arr[i+k]=i;
	 	 	  
	 	 	  k++;
	 	 }
	 }
	 
	 while(j--)
	 {
	 	cin>>n1>>n2;
	 	if(arr[n1]==n2)
	 	 cout<<"Don't Play \n";
	 	else
	 	 cout<<"Play \n";
	 	
	 }
	 
}
