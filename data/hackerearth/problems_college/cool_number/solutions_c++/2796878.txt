#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int bin[106];
void swap(int *a,int *b)
{
	int t=*a;
	*a=*b;
	*b=t;
}
void rev(int a[],int l,int h)
{
	for(int i=l,j=h; i < j; i++,j--)
	{
		swap(&a[i],&a[j]);
	}
}
int count(int a[],int l,int h)
{
	if(h < 3)
	return 0;
	int c=0;
	for(int i=l; i < h;)
	{
		if(a[i]==1 && a[i+1]==0 && a[i+2]==1)
		{
			c++;
			i=i+2;
		}
		else
		i++;
	}
	return c;
}
int convert_and_count(int n)
{
	memset(bin,0,sizeof(bin));
	int i=0;
	while(n)
	{
		bin[i++]=n%2;
		n=n/2;
	}
	rev(bin,0,i);
	return count(bin,0,i);
}
int main()
{
	int r,k,t,bin[106];
	cin >> t;
	while(t--)
	{
		int c=0;
		cin >> r >> k;
		for(int i=1; i <=r; i++)
		{
			if(convert_and_count(i) >=k)
			{
				c++;
			}
		}
		cout << c <<"\n";
	}
	
   // cout << "Hello World!" << endl;
    return 0;
}
