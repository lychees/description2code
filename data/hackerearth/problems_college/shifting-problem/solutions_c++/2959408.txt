using namespace std;
 
ll ans[1001];
 
void calc()
{
    ll i,j;
    ans[0]=1;
    ans[1]=1;
    for(i=2;i<=1000;i++) {
        ans[i]=ans[i-1]+ans[i-2];
    }
}
 
int main()
{
    ll t,n,i,j,k,l,w;
    s(t);
    calc();

    }

}
