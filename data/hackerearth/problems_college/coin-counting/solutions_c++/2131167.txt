#include <bits/stdc++.h>
using namespace std;
 
int coin(int S[], int m, int n )
{
    if (n == 0)
        return 1;
    if (n < 0)
        return 0;
 	if (m <=0 && n >= 1)
        return 0;
 	return coin( S, m - 1, n ) + coin( S, m, n-S[m-1] );
}
 
int main()
{
    int t;
	cin>>t;
    int a,arr[3] = {1, 3, 5};
 	while(t--) {
		cin>>a;	
	    cout<<coin(arr, 3, a)<<"\n";
	}
    return 0;
}
