    #include <bits/stdc++.h>
    using namespace std;
    #define mod 1000000007
    long long int dp[10004];
    int main()
    {
    int t;
    cin>>t;
    while(t--)
    {
    int n,p;
    cin>>n>>p;
    memset(dp,0,sizeof(dp));
    dp[0]=1;
    for(int i=1;i<=n;i++)
    {
    for(int j=i-1;(j>=0)&&(i-j<=p);j--)
    {
    dp[i]+=dp[j];
    dp[i]%=mod;
    }
    }
    cout<<dp[n]<<endl;
    }
    return 0;
    }
