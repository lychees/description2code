#include<bits/stdc++.h>

using namespace std;

//#define test

int solve(int a, int b) {
    int s = 0, tmp;
    while(b != 0) {
        s += a / b;
        tmp = a % b;
        a = b;
        b = tmp;
    }
    if(a == 1) return s;
    return INT_MAX;
}

int main() {
    std::ios::sync_with_stdio(false);
#ifdef test
	freopen("a.in","rt",stdin);
	freopen("a.out","wt",stdout);
#endif
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        if(n == 1) {
            cout << 0 << endl;
            continue;
        }
        int ans = INT_MAX;
        for(int i = 1; i <= n / 2; i++) {
            ans = min(ans, solve(n - i, i));
        }
        cout << ans << endl;
    }
}
