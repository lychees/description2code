#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
    for(int i=0;i<t;i++){
    	int n;
    	int l=INT_MIN;
    	cin>>n;
    	map<int,int>m;
    	for(int j=0;j<n;j++){
    		int h;
    		cin>>h;
    		m[h]+=1;
    	}
    	map<int,int>::iterator it;
    	for(it=m.begin();it!=m.end();it++){
    		if(it->second>l){
    			l=it->second;
    		}
    	}
    	vector<int>v;
    	for(it=m.begin();it!=m.end();it++){
    		if(l==it->second){
    			v.push_back(it->first);
    		}
    	}
    	for(int z=(v.size())-1;z>=0;z--){
    		cout<<v[z]<<" ";
    	}
    	cout<<""<<endl;
    }
    return 0;
}
