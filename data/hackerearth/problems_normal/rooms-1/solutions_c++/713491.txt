#include <cstdio>
#include <utility>
#include <algorithm>
#include<iostream>
using namespace std;


typedef long long ll;
#define MAX_N 100005

ll arrive[MAX_N],depart[MAX_N];
pair<ll, int> event[2*MAX_N];
int main(){
	int T, N;

	scanf("%d", &T);
	while(T--){
		scanf("%d", &N);
		for(int i=0; i<N; i++)
			scanf("%lld", &arrive[i]);
		for(int i=0; i<N; i++)
			scanf("%lld", &depart[i]);
        //event.clear();
		for(int i=0; i<N; i++){
			event[2*i]=make_pair(arrive[i], 1);
			event[2*i+1]=make_pair(arrive[i]+depart[i], -1);
		}
		sort(event, event+2*N);
		int guests=0, answer=0;
		for(int i=0; i<2*N; i++){
			guests+=event[i].second;
			answer=max(answer, guests);
		}
		printf("%d\n", answer);
	}
	return 0;
}
