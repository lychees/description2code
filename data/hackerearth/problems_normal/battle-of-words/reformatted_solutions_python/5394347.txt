def makeFreq(s):
    f = {}
    for c in s:
        if c not in f:
            f[c] = 0
        f[c] += 1
    return f


def positiveLength(f):
    for v in f.itervalues():
        if v > 0:
            return True
    return False


for _ in xrange(input()):
    alice = ''.join(raw_input().split())
    bob = ''.join(raw_input().split())
    t = makeFreq(alice)
    for c in bob:
        if c in t:
            t[c] -= 1
    ares = positiveLength(t)
    t = makeFreq(bob)
    for c in alice:
        if c in t:
            t[c] -= 1
    bres = positiveLength(t)
    if ares and bres:
        print 'You draw some.'
    if ares and not bres:
        print 'You win some.'
    if not ares and bres:
        print 'You lose some.'
    if not ares and not bres:
        print 'You draw some.'
