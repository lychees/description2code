#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    if(cin>>t){
            char c;
    while (cin.get(c) && c != '\n')
        if (!std::isspace(c))
        {
            std::cerr << "ERROR unexpected character '" << c << "' found\n";
            exit(EXIT_FAILURE);
        }
    while(t--)
    {
        string s;
        while(getline(cin,s))
        {
        //cout<<s<<endl;
        int len=s.size();
        //cout<<len<<endl;
        string c="$";
        for(int i=0;i<len;i++)
        {
            if(s[i]=='$')
            {
                int flag=1,j;
                for(j=i+1;j<len;j++)
                {
                        if(flag == 1 && s[j]=='0')continue;
                        else if(s[j]==' ')continue;
                        else if(s[j]>=48 && s[j]<=57)
                        {
                            flag=0;
                            //cout<<"1"<<endl;
                            c+=s[j];
                        }

                        else break;
                }
                if(c=="$")c+='0';
                i=j;
            }
        }
    cout<<c<<endl;
        }
}
    }
return 0;
}
