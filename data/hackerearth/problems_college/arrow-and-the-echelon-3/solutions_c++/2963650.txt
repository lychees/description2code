#include <bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int n3=0,n5=n;
    	int f=0;
    	while(n5>=0&&f==0&&n3<=n)
    	{
    		if(n3%5==0&&n5%3==0)
    		f=1;
    		else
    		{
    			n5-=5;
    			n3+=5;
    		}
    	}
    	if(f==1)
    	{
    		for(int i=0;i<n5;i++)
    		cout<<"5";
    		for(int i=0;i<n3;i++)
    		cout<<"3";
    		cout<<endl;
    	}
    	else
    	{
    		cout<<"-1"<<endl;  
  		}
    }
    return 0;
}
