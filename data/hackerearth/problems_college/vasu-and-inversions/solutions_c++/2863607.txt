#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    int t, n, k, v;
    
    scanf("%d", &t);
    
    while(t--)
    {
    	scanf("%d", &n);
    	scanf("%d", &k);
    	
    	int ans = 0;
    	for(int i = 0; i<n; i++)
    	{
    		scanf("%d", &v);
    		if(v == 0)
    		{
    			if(k>0)
    			{
    				k--;
    				ans++;
    			} 
    		}
    		else
    			ans++;
    	}
    	
    	if(k > 0)
    		if(k % 2 != 0)
    			ans--;
    	
    	if(ans < 0)
    		ans = 0;
    		
    	printf("%d\n", ans);
    }
    
    return 0;
}
