#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        bool mark[n+1];int arr[n];
        mark[n]=0;
        int f=0;
        for(int i=0;i<n;i++){
            scanf("%d",&arr[i]);
            mark[i]=0;
        }
        int i=0;
        mark[i]=1;
        while(1){
            if(arr[i]==-1){
                f=1;
                break;
            }
            else if(!mark[arr[i]]){
                mark[arr[i]]=1;
                i=arr[i];
            }
            else break;
        }
        if(f==0)
            printf("Lose\n");
        else printf("Win\n");
    }
    return 0;
}
