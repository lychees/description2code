#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    string a;
    long long int cnt,s;
    while(t--){
    	cin>>a;
    	cnt =0;
    	s=0;
    	for(int i = 0 ; i< a.size();i++){
    		if(a[i]=='H')
    		{
    			cnt++;
    		
    		}
    		else
    		cnt=0;
    			s=s+cnt;
    	}
    	cout<<s<<endl;
    }
    return 0;
}
