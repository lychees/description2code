#include <iostream>
using namespace std;

int main()
{
    int t=100000,i,j,n;
    char s;
    while(t--)
    {
    	s=0;
    	cin>>s>>n;
    	if(n<=1||n>26&&s>0) cout<<"INVALID\n";
    	else if(s>='A'&&s<='Z') {
    		if('Z'-s<n-1) cout<<"INVALID\n";
    		else {
    			for(i=1;i<=n;i++) {
    				for(j=1;j<=i;j++) {
    					cout<<s<<" ";
    				}
    				cout<<endl;
    				s=s+1;
    			}
    		}
    	}
    	else if(s>='a'&&s<='z') {
    		if('z'-s<n-1) cout<<"INVALID\n";
    		else {
    			for(i=1;i<=n;i++) {
    				for(j=1;j<=i;j++) {
    					cout<<s<<" ";
    				}
    				cout<<endl;
    				s=s+1;
    			}
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
