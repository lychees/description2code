#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int a[10001],b[105],i;
    int t;
    cin>>t;
    for(;t--;)
    {
    	for(i=0;i<10001;i++)
    	{
    		a[i]=0;
    	}
    	int n,j,count=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>b[i];
    	}
    	a[0]=1;
    	for(i=0;i<n;i++)
    	{
    		for(j=10000;j>=0;j--)
    		{
    			if((j-b[i])>=0)
    			a[j]|=a[j-b[i]];
    		}
    	}
    	for(i=0;i<10001;i++)
    	{
    		if(a[i])
    		count++;
    	}
    	cout<<(count)<<endl;
    }
    return 0;
}
