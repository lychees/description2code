#include <iostream>
using namespace std;
#define MOD 1000000009
long long int power(long long int a,long long int b){
	long long int res = 1 ;
	while(b){
		if(b%2){
			res = res*a ;
			res %= MOD ;
		}
		b /= 2 ;
		a = a*a ;
		a %= MOD ;
	}
	
	return res ;
}



int main()
{
	long long int t, n, i ,j ,k, r, ans;

	   {

    	r=(n+1)/2;
    
    	cout<<power(26,r)<<endl;
    }
    return 0;
}
