#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <stdio.h>
using namespace std;
int main() {
  // freopen("input.txt","r",stdin);
  // freopen("output.txt","w",stdout);
  int q, w, e, r, t;
  cin >> q;
  for (w = 1; w < q; w++) {
    for (e = 1; e < q; e++) {
      if (w * e >= q)
        cout << w * e / q;
      cout << (w * e) % q << " ";
    }
    cout << "\n";
  }
  return 0;
}
