#include<stdio.h>

int partitionindex(int a[], int n){
	int sum = 0, leftsum = 0, i;
	for(i = 0; i < n; sum += a[i++]);
	for(i = 0; i < n; i++){
		sum -= a[i];
		if(leftsum == sum)	return i;
		leftsum += a[i];
	}
	return -1;
}

int main(){
	int t, n; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		int a[n], i;
		for(i = 0; i < n; scanf("%d", &a[i++]));
		(partitionindex(a, n) != -1) ? printf("YES\n") : printf("NO\n");
	}		
	return 0;
}
