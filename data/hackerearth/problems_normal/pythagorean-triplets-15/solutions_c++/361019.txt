#include <cstdio>
#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;

int func(int t)
{
	int i,j,a,b,c,r=0;
	for (i = 2; i <= (sqrt(t) + 1); i++)
		for (j = 1; j < i; j++)
			if ((__gcd(i,j) == 1) && ((i-j) % 2) && ((i*i + j*j) <= t))
			{
				a = i * i - j * j;
				b = 2 * i * j;
				c = i * i + j * j;
				r=r+t/c;
			}
	return r;
}

int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		printf("%d\n",func(n));
	}
	return 0;
}
