#include<iostream>
using namespace std;

int is_prime(int num)
{
	if(num==2 || num==3)
		return 1;
	if(num%2==0)
		return 0;
	if(num%3==0)
		return 0;
	int i=5,w=2;
	while(i*i<=num)
	{
		if(num%i==0)
		   return 0;
		i+=w;
		w=6-w;
	}
	return 1;
}

int main()
{
   int t,m,n,i,cnt;
   cin>>t;
   while(t--)
   {
	cnt=0;
	cin>>m>>n;
	if(m==1)
	   m++;
	for(i=m;i<=n;i++)
		if(is_prime(i))
			cnt++;
	cout<<cnt<<endl;
   }
   return 0;
}
