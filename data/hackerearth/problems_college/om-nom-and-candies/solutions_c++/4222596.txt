#include <bits/stdc++.h>
using namespace std;
long long arr[100010];
long long dp[100010];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%lld",&arr[i]);
		dp[i] = arr[i];
	}
	for(int i=1;i<=n;i++)
	{
		for(int k=2;i*k<=n;k++)
		{
			dp[i*k] = max(dp[i*k],arr[i*k]+dp[i]);
		}
	}
	for(int i=1;i<=n;i++)
	{
		printf("%lld ",dp[i]);
	}
}
