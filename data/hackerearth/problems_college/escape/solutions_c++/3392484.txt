#include<bits/stdc++.h>

#define ff first
#define ss second
#define mp make_pair
#define loop_p(i,a,b) for(i=a;i<=b;i++)
#define mem(n,m) memset(n,m,sizeof(n))
#define ll long long
#define ull unsigned long long
#define boost ios_base::sync_with_stdio(0)
using namespace std;

ll cell[22][22];
bool visited[22][22]={false};
int cnt,n;

void lc(ll a,ll b)
{
    if(a==n && b==n)
    {
        ++cnt;return;
    }
    visited[a][b]=true;
    if(cell[a][b+1]==0 && visited[a][b+1]==false)
        lc(a,b+1);
    if(cell[a+1][b]==0 && visited[a+1][b]==false)
        lc(a+1,b);
    if(cell[a-1][b]==0 && visited[a-1][b]==false)
        lc(a-1,b);
    if(cell[a][b-1]==0 && visited[a][b-1]==false)
        lc(a,b-1);
    visited[a][b]=false;
}

int main()
{
	boost;
	ll t,i,j;
	cin>>t;
	while(t--)
    {
        cin>>n;
        mem(cell,1);
        mem(visited,false);
        loop_p(i,1,n)loop_p(j,1,n)cin>>cell[i][j];
        if(cell[1][1]==1 && cell[n][n]==1)
        {
            cout<<"0"<<endl;
        }
        else
        {
            cnt=0;
            lc(1,1);
            cout<<cnt<<endl;
        }
    }
	return 0;
}
