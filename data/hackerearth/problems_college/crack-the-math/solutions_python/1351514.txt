t=input()
ch=' '
S=""
A=[]
if t>=1 and t<=1000000:
    for i in range(0,t):
        ch=' '
        S=""
        a=raw_input()
        b=raw_input()
        if len(a)==len(b) and len(a)<=100 and len(b)<=100:
           for j in range(0,len(a)):
               if a[j]!=b[j]:
                   ch='1'
               else:
                   ch='0'
               S=S+ch
           A.append(S)
    for i in range(0,t):
        print(A[i])
