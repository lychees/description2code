#include <bits/stdc++.h>
#define fast ios_base::sync_with_stdio(false);cin.tie(NULL)
#define endl '\n'

using namespace std;

int find_ring(int n, int swap) {
	while (swap--) {
		if (n == 0) n = 1;
		else if (n == 1) n = 0;
		else if (n == 2) n = 1;
	}
	return n;
}

int main() {
	fast;
	int t, index, n;
	cin >> t;
	while (t--) {
		cin >> index >> n;
		cout << find_ring(index, n) << endl;
	}
	return 0;
}
