#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
#define max2 1000006
typedef pair<long long,int> ii;
vector<long long> h(max2);
int ans[max2];
vector<pair<long long,int> > v1;
int main()
{
      int tc;
      scanf("%d",&tc);
      while(tc--)
      {
        int n,k;
        long long a;
        scanf("%d %d",&n,&k);
        //long long h[k];
        for(int i=0;i<n;i++)
        {
           scanf("%lld",&a);
           v1.push_back(ii(a,i));
        }
        for(int i=0;i<k;i++)
        scanf("%lld",&h[i]);
        sort(v1.begin(),v1.end());
        int start =0;
        int j;
        fill(ans,ans+n,0);
        for(int i=0;i<n;i++)
        {
                j=start;
                if(i>0)
                ans[v1[i].second]=ans[v1[i-1].second];
                else
                ans[v1[i].second]=0;
                while(j<k && h[j]<=v1[i].first)
                {
                  j++;
                  ans[v1[i].second]++;
                }
                start=j;
        }
        int max1 =-1,ind;
        for(int i=0;i<n;i++)
        {
           if(ans[i]>max1)
           {
              max1=ans[i];
              ind = i;
           }
        }
        cout<<ind<<endl;
        h.clear();
        v1.clear();
      }
      return 0;
}
