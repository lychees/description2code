#include <iostream>
using namespace std;
 
int main()
{
    int t,i,max;
    string a,b;
    
    cin>>t;
    while(t--){
    	cin>>a>>b;
    	max = (a.size()>b.size())?a.size():b.size();
    	if(a=="0"&&b=="0")  
        {   break;  }  
    	int s[100000] = {0};
    	int c = 0;
    	for(i=0;i<max;i++){
    		if(i!=0){
	    		if(a[i]-'0'+b[i]-'0'+s[i-1]>=10){
	    			c++;
	    			s[i]=1;
	    		}
    		}else{
    			if(a[i]-'0'+b[i]-'0'>=10){
    				c++;
    			}
    		}
    	}
    	
    	if(c==0){
    		cout<<"No carry operation"<<endl;
    	}else if(c==1){
    		cout<<"1 carry operation"<<endl;
    	}else{
    		cout<<c<<" carry operations"<<endl;
    	}
    	
    }
    return 0;
}
