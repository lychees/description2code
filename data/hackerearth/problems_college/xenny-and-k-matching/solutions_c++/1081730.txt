#include <iostream>
using namespace std;

int main()
{
	int N,k;
    cin>>N>>k;
    int arr[N+1];
    int count=0,i,j;
    
    for(i=0;i<N;i++)
    	cin>>arr[i];
    
    for(i=0;i+k<N;i++)
    {
    	if((arr[i]+arr[i+k])%k==0)
    		count++;
    	else if(abs(arr[i]-arr[i+k])%k==0)
    		count++;
    }
    
    cout<<count<<endl;
    return 0;
}
