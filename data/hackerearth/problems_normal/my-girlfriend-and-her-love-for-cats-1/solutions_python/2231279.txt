import sys

def maxsum(n,s,c):

    s.sort(reverse = True)
    c.sort(reverse = True)
    tot = 0
    for x in xrange(n):
        tot += s[x] * c[x]
    return tot

if __name__ == '__main__':
    n = sys.stdin.readline()
    s = map(int, sys.stdin.readline().split())
    c = map(int, sys.stdin.readline().split())
    print(maxsum(int(n),s,c))
