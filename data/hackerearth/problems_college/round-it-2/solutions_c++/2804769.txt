#include <iostream>
using namespace std;

int main()
{
 	int t,j=1,i,n[1000],b[1000];
 	
 	cin>>t;
 	
 	for(i=0;i<t;i++){
 		cin>>n[i];
 		cin>>b[i];
 	}
 	
 	for(i=0;i<t;i++){
 		for(j=0 ; b[i]*j<n[i] ; j++);
 		
 		if(b[i]*j -n[i] > n[i]-b[i]*(j-1)){
 			cout<<b[i]*(j-1)<<endl;
 		}
 		else{
 			cout<<b[i]*j<<endl;
 		}
 	}
    return 0;
}
