#include<bits/stdc++.h>
using namespace std;
#define gc getchar
#define ll long long
#define v vector
#define pr pair<int,int>
#define sd second
#define ft first
#define pb push_back
ll  fr()
{
    char ch;
    ll t=0,flag=0;
    ch=gc();
    while(ch<33)
        ch=gc();
    if(ch=='-')
        {
            ch=gc();
            flag=1;
        }
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=gc();
    }
    if(flag)
        return -t;
    else
    return t;
}
int main()
{
    ll  i,j,k,n,fac;
    string ans[5]={ "Sheldon", "Leonard", "Penny", "Rajesh", "Howard" };
    ll a[50]={0};
    a[0]=0;
    k=5;
    for(i=1;i<50;i++)
    {
        a[i]+=a[i-1]+k;
        k=k*2;
    }
    n=fr();
    for(i=1;i<50;i++)
    {
        if(n>a[i-1]&&n<=a[i])
            break;
    }
    n=n-a[i-1];
    fac=pow(2,i-1);
    for(j=0;j<5;j++)
    {
        if(n<=fac)
            {cout<<ans[j];
            break;}
            n-=fac;
    }
    return 0;
}
