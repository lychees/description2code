#define ll int64_t

using namespace std;

ll gcd(ll a, ll b) {
    return b == 0 ? a : gcd(b, a % b);
}

int main() {
	// your code goes here
	int n;
	cin>>n;
	ll x[n],y[n];
	for(int i=0;i<n;i++)
	cin>>x[i]>>y[i];
	ll count=n;
	for (int i=0;i<n;i++)
	{
		ll a=abs(x[i]-x[(i+1)%n]);
		ll b=abs(y[i]-y[(i+1)%n]);
		count+=(gcd(a,b)-1);
	}
	double area=0;
	for (int i=0;i<n;i++)
	{
		area=area+(x[i]*y[(i+1)%n]-x[(i+1)%n]*y[i]);
	}
	area=abs(area)/2;
	//cout<<area<<" "<<count<<" ";
	ll ans=area-count/2+1;
	cout<<ans;
	return 0;
}
