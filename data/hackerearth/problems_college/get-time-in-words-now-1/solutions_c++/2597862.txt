#include <iostream>
using namespace std;
int main()
{	int h;
	int m;
	char s[29][100]={"one","two","three","four","five","six","seven","eight","nine","ten", 			"eleven","twelve","thirteen","fourteen","fifteen","sixteen","seventeen","eighteen","nineteen","twenty one","twenty two","twenty three","twenty four","twenty five","twenty six","twenty seven","twenty eight","twenty nine"};
	cin>>h>>m;
	if(m>0&&m<30&&m!=15)
		cout<<s[m*1]<<" minutes past "<<s[h-1];
	else if(m==0)
		cout<<s[h-1]<<" o' clock" ;
	else if(m==30)
		cout<<"half past "<<s[h-1];
	else if(m==45)
		cout<<"quarter to "<<s[h];
	else if(m==15)
		cout<<"quarter past "<<s[h-1];
	if(m>30&&m<60&&m!=45)
		cout<<s[60-m-1]<<" minutes to "<<s[h];
	return 0;
}
