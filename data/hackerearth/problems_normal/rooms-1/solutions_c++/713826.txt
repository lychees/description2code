#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
        int n, mm = 0;
        cin>>n;
        int a[n];
        int b[n];
        for(int i = 0; i < n; i++) cin>>a[i];
        for(int i = 0; i < n; i++) { cin>>b[i]; b[i] += a[i]; }
        sort(a, a + n);
        sort(b, b + n);
        int aaa = 0, bbb = 0, cc = 0;
        while(aaa < n) {
            if(a[aaa] < b[bbb]) {
                cc++;
                aaa++;
            } else {
                bbb++;
                cc--;
            }
            if(cc > mm) mm = cc;
        }
        cout<<mm<<endl;
    }
    return 0;
}
