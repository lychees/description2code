    #include <stdio.h>
     
    using namespace std;
     
    int main()
    {
    	int n;
    	scanf("%d", &n);
    	int arr[n + 1];
    	for(int i=0;i<n+1;i++)
    		arr[i] = 0;
    	int conn;
    	scanf("%d", &conn);
    	for(int i=0;i<conn;i++)
    	{
    		int p, q;
    		scanf("%d%d", &p, &q);
    		arr[q] = p;		
    	}
    	int qu;
    	scanf("%d", &qu);
    	for(int i=0;i<qu;i++)
    	{
    		int p, q;
    		scanf("%d%d", &p, &q);
    		if(arr[q] == 0)
    			printf("No Connection\n");
    		else
    		{
    			if(arr[q] == p)
    				printf("Direct Connection\n");
    			else 
    			{
    				int temp = arr[q];
    				while(true)
    				{
    					if(arr[temp] != p)
    					{
    						temp = arr[temp];
    					}
    					else if(temp == 0)
    					{
    						printf("No Connection\n");
    						break;
    					}
    					else	
    					{
    						printf("Indirect Connection\n");
    						break;
    					}				
    				};		
    			}
    		}
    	}	
    }
