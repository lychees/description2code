#include <iostream>
#include<math.h>
using namespace std;



int main()
{
    long n,c,arr[100000],q,m,d,i,wait,temp[100000],r,j;
    cin>>n>>c;
    
    for(i=0;i<n;i++)
    {cin>>arr[i];
    temp[i]=ceil((float)arr[i]/5);}  
    
    
    cin>>q;
    
    for(i=0;i<q;i++)
    {
    	cin>>m>>d;
    	wait=0;
    	if(m==1)
    	{	for(j=0;j<d-1;j++)
    		{
    		if(arr[j]<c)
    			wait=wait+arr[j];
    		else
    			wait=wait+c;
    		}
    		cout<<wait<<"\n";
    	}
    		
    	else
    	{r=1;
    		    		 	
    		 for(j=0;j<n;j++)
    		 {  
    		 	if(temp[j]<ceil((float)arr[d-1]/5))
    		 	r++;
    		 	
    		 	else if(temp[j]==ceil((float)arr[d-1]/5))
    		 	{
    		 		if(j<(d-1))
    		 		r++;
    		 	}
    		 }
    		 
    		 
    			
    	cout<<r<<"\n";
    
    
    }
    	
    	
    	
    
    	}}
