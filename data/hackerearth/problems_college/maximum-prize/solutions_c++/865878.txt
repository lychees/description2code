#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    long long int t;
    cin >> t;
    while(t--){
    	long long int n,k;
    	cin >> n >> k;
    	vector<long long int> num;
    	while(n--){
    		long long int c;
    		cin >> c;
    		num.push_back(c);
    	}
    	sort(num.begin(),num.end());
    	long long int a = num.size()-1;
    	long long int tot = 0;
    	while(k--){
    		if(a >= 0)
    		{tot += num[a];
    		a--;
    		}
    	}
    	cout << tot << endl;
    }
    return 0;
}
