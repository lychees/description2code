ar = [int(i) for i in raw_input().split()]
ar.sort(reverse=True)
ans = 0
n = int(input())
li = sorted([int(i) for i in raw_input().split()])
c = 0
while sum(li):
    i = 0
    c += 1
    l = len(li)
    while i < l:
        if len(ar) > 0:
            ans += ar[i]*c
            ar.pop(i)
            l -= 1
        else:
            break
    li = [i-1 for i in li if i > 1]
if len(ar) != 0:
    print("-1")
else:
    print(ans)
