
class TreeNodes():
    def __init__(self, value):
        self.value = value
        self.left = None
        self.right = None


def main():
    T = int(raw_input())
    if 1 <= T <= 10:
        count = 0
        result = []
        while count < T:
            N, M = [long(i) for i in raw_input().split(' ')]
            if 1 <= N <= 100000 and 1 <= M <= 100000:
                candies = [long(i) for i in raw_input().split(' ')]
                root = None
                output = []
                for candy in candies:
                    if root is None:
                        root = TreeNodes(candy)
                        output.append('NO')
                    else:
                        temp = root
                        while True:
                            if temp.value < candy:
                                if temp.right is None:
                                    temp.right = TreeNodes(candy)
                                    output.append('NO')
                                    break
                                else:
                                    temp = temp.right
                            elif temp.value > candy:
                                if temp.left is None:
                                    temp.left = TreeNodes(candy)
                                    output.append('NO')
                                    break
                                else:
                                    temp = temp.left
                            else:
                                output.append('YES')
                                break

                result.append(output[N:])
            count += 1
        return result


if __name__ == '__main__':
    output = main()
    for out in output:
        for i in out:
            print i
