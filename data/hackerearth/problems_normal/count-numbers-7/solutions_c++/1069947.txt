#include <iostream>
#include <cmath>
using namespace std;
int main(){
    int k,t;
    long long int a,b,ac,bc;
    long long int count,ans;
    int p[10];
   
        cin>>k>>t;
        for (int i = 0; i<k; i++) {
            cin>>p[i];
        }
    
    

        while (t--) {
            cin>>a>>b;
            ans = 0;
            for (int i=1; i<=pow(2, k)-1; i++) {
                count = 0;
                ac=a-1,bc=b;
                for (int j=0; j<=k-1; j++) {
                    if((i & (int)pow(2, j)) != 0){
                        
                        count++;
                        ac/=p[j];
                        bc/=p[j];
                    }
                }
                if (count%2 ==0) {
                    ans-=bc-ac;
                }
                else{
                    ans+=bc-ac;
                }
            }
            cout<<ans<<endl;
        }
    
}
