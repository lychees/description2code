#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int t,i,x,p,count,res;
    scanf("%d",&t);
    while(t--)
    {
    	count=0;
    	scanf("%d%d",&x,&p);
    	res=x;
    	while(1)
    	{
    		if(res%p==0)
    		{
    			res=res/p;
    			count++;
    		}
    		else
    			break;
    	}
    	if(res==1)
    	res=0;
    	printf("%d %d\n",count,res);
    }
    return 0;
}
