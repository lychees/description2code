#include<iostream>
#include<math.h>

using namespace std;

int main()
{
	int t,p,x,y,u,v,i;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>p>>x>>y>>u>>v;
		x=x-u;
		y=y-v;
		if(x<0)x=-1*x;
		if(y<0)y=-1*y;
		if(x==0&&y==0&&p>0)cout<<"Counter-Terrorists Win !"<<endl;
		else if(p==2){if(x!=0&&y!=0&&y==x)cout<<"Counter-Terrorists Win !"<<endl;else cout<<"Terrorists Win !"<<endl;}
		else if(p>=3){if((x+y)%2==0)cout<<"Counter-Terrorists Win !"<<endl;else cout<<"Terrorists Win !"<<endl;}
		else{ cout<<"Terrorists Win !"<<endl;}
	}
}
