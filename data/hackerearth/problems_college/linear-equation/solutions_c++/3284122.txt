#include <bits/stdc++.h>
using namespace std;
long long int  a[500006];
long long int sum;
int main(){
long long int b;

int t;
cin>>t;
int n;
while(t--){
    cin>>n;
    sum = 0;
    for(int i = 0; i < n; i++){
        cin>>a[i];
        sum = sum + a[i];
    }
    if(n == 2){
    	cout<<a[1]<<" "<<a[0]<<endl;
    	continue;
    }
    sum = sum/(n-1);
    long long int ans;
    for(int i = 0; i < n; i++){
        ans = (sum - a[i]);
        cout<<ans<<" ";
    }
    cout<<endl;
}
}
