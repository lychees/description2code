#include <iostream>
using namespace std;

int matrix[102][102]={0};

int main()
{
    int T, N,C;
    cin>>T;
    for( int i=0; i<T;i++){
    	cin>>N>>C;
    	for(int j=1;j<=N;j++){
    		for(int k=1;k<=N;k++){
    			cin>>matrix[j][k];
    			(matrix[j-1][k] >= matrix[j][k-1])&&(matrix[j][k]=matrix[j][k]+matrix[j-1][k]);
    			(matrix[j-1][k] < matrix[j][k-1])&&	(matrix[j][k]=matrix[j][k]+matrix[j][k-1]);
    			}
    		}
    		(matrix[N][N] >= C)&&cout<<matrix[N][N] - C<<endl;
    		(matrix[N][N] < C)&&cout<<-1<<endl;
    	}
    
    return 0;
}
