#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MOD 1000000007;
string s;
int countt[50][50];

int main(){
   	int n,k;
   	ll ans = 0;
   	string s;
   	cin >> n >> k;
   	char ch;
   	for(int i=0;i<50;i++){
   		for(int j=0;j<50;j++){
   			countt[i][j] = 0;
   		}
   	}
   	for(int i=0;i<n;i++){
   		for(int j=0;j<6;j++){
   			cin >> ch;
   			countt[i][ch-'a']++;
   		}
   	}
   	cin >> s;
   	
   	
	for(int i=0;i<(1<<n);i++){
		if(__builtin_popcount(i)==k){
			vector<int> data;
			int t = i;
			int ind = 0;
			while(t){
				if(t&1) data.push_back(ind);
				ind++;
				t = t>>1;
			}
			sort(data.begin(),data.end());
			//cout << data << endl;
			do{
				ll ways = 1;
				for(int j=0; j<k; j++){
						ways*=countt[data[j]][s[j]-'a'];
						ways%=MOD;
				}
				ans+=ways;
				ans%=MOD;
			} while(next_permutation(data.begin(), data.end()));
		}
	}
	ans = ans%MOD;
   	cout << ans << endl;
    return 0;
}
