#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int t,i,n,cnt;
    char a[100000];
    cin>>t;
    while(t--)
    {
    	cnt=0;
    	cin>>a;
    	n=strlen(a);
    	for(i=0;i<n-1;i++)
    	{
    	if(a[i]==a[i+1])
    	cnt++;
    	}
    	cout<<cnt<<endl;
    }
    return 0;
}
