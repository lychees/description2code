from math import sqrt

PI = 3.1415
testcases = input()

while testcases != 0:
	try:
		a, b, c = [float(f) for f in raw_input().split(" ")]
	except EOFError:
		pass
	radius = (a*b*c)/ (sqrt((a+b+c)*(a+b-c)*(b+c-a)*(a+c-b)))
	print("%.4f" % (PI * radius * radius))
	testcases -= 1
