#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int dp[1008][1008];

void preProcess()
 {
 	int i,j;
   for(i=0;i<1008;i++)
    { 
    	dp[i][0]=1;
        dp[0][i]=1;
    }
   for(i=1;i<1008;i++)
     for(j=1;j<1008;j++)
        dp[i][j]=(dp[i-1][j]%mod+dp[i][j-1]%mod+dp[i-1][j-1]%mod)%mod;
 }
 
int main()
{
	int test;
	scanf("%d",&test);
	long long int a,b,c,d,i,j;
	preProcess();
	while(test--)
	 {
	   scanf("%lld%lld%lld%lld",&a,&b,&c,&d);
	    cout<<dp[c-a][d-b]<<endl;
	 }
	 return 0;
}
