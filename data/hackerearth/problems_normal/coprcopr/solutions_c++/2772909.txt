#include<bits/stdc++.h>
#define sc(n) scanf("%d",&n);
#define pr(n) printf("%lld\n",n);
#define ll long long
using namespace std;
ll int ph[10000001];
bool v [10000001];
int main()
{
    for(int i=1;i<=10000000;i++)
        ph[i]=i;
    int num=2,k=10000000/2,t,n;
    while(num<=k)
    {
        for(int i=num*2;i<=10000000;i+=num)
        {
            ph[i]-=ph[i]/num;
            v[i]=1;
        }
        num++;
        while(v[num])
            num++;
    }
    for(int i=2;i<=10000000;i++)
    {
        if(v[i]==0)
            ph[i]--;
    }
    for(int i=1;i<=10000000;i++)
        ph[i]+=ph[i-1];
    sc(t);
    while(t--)
    {
        sc(n);
        pr(ph[n]);
    }
    return 0;
}
