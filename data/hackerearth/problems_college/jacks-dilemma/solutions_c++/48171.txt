#include<iostream>
using namespace std;

int main()
{
	int t,n,j,l,tm;
	char d;
	cin>>t;
	while(t--)
	{	cin>>n>>j>>l>>d;
		if(d=='D')
		{
			if(l>=j)
			{	tm=5*l+20;
			}
			else
			{	tm=j*10 +1000;
			}
		}
		if(d=='U'&& n>j)
		{	tm=5*(2*n-l)+30;
		}
		if(d=='U' && n==j)
		{	tm=5*(2*n-l)+20;
		}
		if(tm<=j*10)
		cout<<"Lift "<<tm<<endl;
		else
		cout<<"Stairs "<<j*10<<endl;
		tm=0;
	}


return 0;
}
