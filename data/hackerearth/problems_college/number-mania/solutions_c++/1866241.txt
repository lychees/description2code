#include<iostream>
using namespace std;
int prime(int n)
{
    int i,f=0;
    for(i=2;i<=n/2;++i)
    {
      if(n%i==0)
      {
          f=1;
          break;
      }
	}
    if(f==0)
	return 1;
	else 
	return 0;
}
int main()
{
int i,j,n,k,c,arr[200]={1},h,p;
cin>>n;
for(i=2;i<=n;i++)
{
    if(prime(i)==1)
    {
    	k=0;c=0;
    	for(p=0;p<200;p++)
    	arr[p]=1;
        for(h=1;h<=i;h++)
        {
            for(j=0;j<=k;j++)
            {
                arr[j]=arr[j]*h+c;
                c=arr[j]/10;
                arr[j]=arr[j]%10;
                
            }
            while(c)
            {
                k++;
                arr[k]=c%10;
                c/=10;
                
            }
        }
        for(p=k;p>=0;p--)
        cout<<arr[p];
        cout<<endl;
    }
}
return 0;
}
