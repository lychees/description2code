#include <iostream>
using namespace std;
#define sz 1002

int a[sz][sz],sol[sz];

int main()
{
    int n,i,j;
    cin>>n;
    for(i=0;i<n;i++)
    	for(j=0;j<n;j++)
    		cin>>a[i][j];
    if(n==1){
    	cout<<a[0][0]<<endl;
    	return 0;
    }
    for(i=0;i<n;i++)
    	sol[i]=a[i][n-1];
    for(i=n-2;i>=0;i--){
    	sol[0]+=a[0][i];
    	for(j=1;j<n;j++)
    		sol[j]=min(sol[j-1],sol[j])+a[j][i];
    	for(j=n-2;j>=0;j--)
    		sol[j]=min(sol[j],sol[j+1]+a[j][i]);
    }
    int min=1000000007;
    for(i=0;i<n;i++)
    	if(sol[i]<min)
    		min=sol[i];
    cout<<min<<endl;
    return 0;
}
