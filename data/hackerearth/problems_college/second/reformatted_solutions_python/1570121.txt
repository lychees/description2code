n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
a = sorted(a)
if (n == 1):
    if (a[0] == m):
        print 1
    else:
        print 0
else:
    i = 0
    ans = 0
    current = m
    while i < len(a):
        # print i,a[i],current
        if (a[i] > current):
            break
        elif (a[i] == current):
            ans = current
            current += m

        i += 1

    print ans
