#include <iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,k;
    	cin>>n>>k;
    	ll i,a[100099];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	cout<<a[k-1]<<endl;
    }
    return 0;
}
