#include <iostream>

using namespace std;
long long int fact[19];
long long int  fact1(int a)
{
    if(a==0 || a==1)
    return 1;
    else return a*fact1(a-1);
}

int main()
{
    for(int i=1;i<=18;++i)
    {
        fact[i]=fact1(i);
    }
    int t,a,b;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        long long int ans=0;
        for(int i=a;i<=b;++i)
        {
            ans=ans+fact[i];
        }
        int r= ans%10;
        cout<<r<<"\n";
    }
    return 0;
}
