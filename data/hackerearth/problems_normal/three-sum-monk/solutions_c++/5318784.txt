#define lli long long int

using namespace std;

int main()
{
    lli t;
    cin>>t;
    lli x=0,cube[25];
    while(1)
    {
        cube[x]=(x+1)*(x+1)*(x+1);
        if(cube[x]<=6000)
        {
            x++;
        }
        else
            break;
    }
    //cout<<x<<endl;
    while(t--)
    {
        lli i,j,n;
        cin>>n;
        lli a[n];
        lli m[6100];
        fill_n(m,6100,0);
        //cout<<x<<endl;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            m[a[i]]++;
        }
        //cout<<x<<endl;
        lli count=0;
        for(i=0;i<n-2;i++)
        {
            m[a[i]]--;
           // cout<<m[a[i]]<<endl;
            for(j=i+1;j<n-1;j++)
            {
                m[a[j]]--;
                for(lli l=0;l<x;l++)
                {
                    if(cube[l]-a[i]-a[j]<0)
                    {

                    }
                    else
                    {
                        if(m[cube[l]-a[i]-a[j]]!=0)
                        {
                            count=count+m[cube[l]-a[i]-a[j]];
                            //cout<<count<<endl;
                        }
                    }
                }

            }
            for(j=i+1;j<n-1;j++)
            {
                m[a[j]]++;
            }

        }
        //cout<<count<<endl;
        cout<<fixed<<setprecision(9)<<(count*6.0)/((n)*(n-1)*(n-2))<<endl;

}
