#include <bits/stdc++.h>

using namespace std;

int main()
{
	int ts;
	cin>>ts;
	while(ts--){
		int d,wt,fare=0,h=0;
		cin>>d>>wt;
		if(d>=2){
		h = d-2;}
		if(d<2){
			h=0;}
		if(d==0){
			cout<<"0"<<endl;}
		if(d>0 && d<=60){
			fare= 50 + h*12+wt*2;
		cout<<fare<<endl;
		}
		else{
			cout<<"NA"<<endl;}
	}
		return 0;
}
