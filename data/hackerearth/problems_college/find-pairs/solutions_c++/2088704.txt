#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,i,j,count=0;
    cin>>t;
    int arr[100000];
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		
    		cin>>arr[i];
    		
    	}
    	
    	
    	for(i=0;i<n;i++)
    	{
    		
    		for(j=i;j<n;j++)
    		 {
    		 	
    		 	if(arr[j]==arr[i])
    		 	 count++;
    		 }
    	}
    	cout<<count<<endl;
    	count=0;
    	
    }
    return 0;
}
