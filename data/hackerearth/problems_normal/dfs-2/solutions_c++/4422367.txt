#include<bits/stdc++.h>
using namespace std;
#define l1 long long int
int main()
{
	l1 i,j,n,q,l,m,s;
	cin>>n;
	l1 vec[n+9];
	for(i=0;i<n;i++)
	cin>>vec[i];
	stable_sort(vec,vec+n);
	cin>>q;
	while(q--)
	{
		cin>>l>>m;
		l1*p=lower_bound(vec,vec+n,l);
		l1 *q=upper_bound(vec,vec+n,m);
		s=int(q-p);
		cout<<s<<endl;
	}
	return 0;
}
