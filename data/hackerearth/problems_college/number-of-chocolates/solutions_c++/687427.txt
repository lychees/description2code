#include <bits/stdc++.h>

using namespace std;

long long A[1000005];
long long dp[1000005];

int main()
{
	int n,m,g1,p1,g2,p2;
	scanf("%d%d", &n, &m);
	dp[0] = 0;
	for ( int i = 1; i <= n; i++ ) {
		scanf("%lld", &A[i]);
		dp[i] = dp[i-1] + A[i];
	}
	while ( m-- ) {
		scanf("%d%d%d%d", &g1, &p1, &g2, &p2);
		g1++, g2++, p1++, p2++;
		long long ans = dp[g2]-dp[g1-1] + dp[p2]-dp[p1-1];
		int lf = max(g1,p1);
		int rt = min(g2,p2);
		if ( rt >= lf ) ans -= (dp[rt]-dp[lf-1]);
		printf("%lld\n", ans);
	}
	return 0;
}
