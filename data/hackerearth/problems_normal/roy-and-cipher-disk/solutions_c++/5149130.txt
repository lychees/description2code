#include <iostream>
using namespace std;

int main()
{
	int t;
	string s;
	cin >>t;
	while(t>0){
		cin>>s;
		char curr = 'a';
		for(int i=0;i<s.size();i++){
			if( curr <=  s[i] ){
				if(s[i]-curr <= 13)
					cout << s[i] - curr << " " ;
				else
					cout << "-" << 26-s[i]+curr << " ";
			}
			else{
				if(curr-s[i] < 13 )
					cout << "-" << curr-s[i] << " " ;
				else
					cout << 26 + s[i] - curr << " " ;
			}
			curr = s[i];
		}
		cout<<endl;
		t--;
	}
    return 0;
}
