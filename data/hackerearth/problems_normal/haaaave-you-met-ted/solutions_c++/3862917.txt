#include <iostream>
#include <bitset>
#include <iostream>
#include <climits>
#include <stdio.h>
#include <algorithm>
typedef unsigned long long int ll;
using namespace std;

size_t popcount(size_t ans) {
    std::bitset<sizeof(size_t) * CHAR_BIT> b(ans);
    return b.count();
}

int main() {
    //freopen ("met.txt" ,"r" ,stdin);
    int T;
    cin >> T;
    while ( T-- ) {
        ll n ,a[100000],b ,min1=100000;
        cin >> n;

        for ( ll i = 0; i < n; i++ ) {
            cin >> a[i];
            b = popcount(a[i]);
            min1=min(min1,b);
        }
        cout << min1 <<endl;
    }
}
