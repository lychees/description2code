import sys
t = int(input())
while t:
    n = int(input())
    if (n >= 25):
        print "00000"
    else:
        ans = 1
        while (n > 1):
            ans = (ans*n) % 100000
            n -= 1
        ans = str(ans)
        for i in xrange(5-len(ans)):
            sys.stdout.write('0')
        print ans
    t -= 1
