#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	
	int t;
	cin>>t;
	
	for(int j=0;j<t;++j) {
		
		int n,l;
		
		cin>>n;
		vector<int> a(n);
		
		
		for(int i=0;i<n;++i) {
			cin>>a[i];
		}
		cin>>l;
		
		vector<int> temp = {a[0]};
		for(int i=1;i<n;++i) {
			
			
			if(a[i] > temp.back()) temp.push_back(a[i]);
			
			*lower_bound(temp.begin(),temp.end(),a[i]) = a[i];
			
		}
		
		if(temp.size() >= l)		
			cout<<temp[l-1];
		else
			cout<<"-1";
		
		cout<<endl;
		
	}
	
    return 0;
}
