#include <iostream>
#include<string>
using namespace std;
class node
{
public:
int but;
	int pos;
	node():but(0),pos(0)
	{
		
	}
};
int main()
{
    int t;
    cin>>t;
    node arr[128];
    int i,j=97,k=3,p;
    for(i=2;i<=9;i++)
    {
    	if(i==7||i==9)
    	k=4;
    	for(p=1;p<=k;p++,j++)
    	{
    		arr[j].but=i;
    	   arr[j].pos=p;
    	}
       k=3;
    }
  for(i=48;i<=57;i++)
     {
     	arr[i].but=i-48;
     	if(i==55||i==57||i==49)
     	arr[i].pos=5;
     	else if(i==48)
     	arr[i].pos=2;
          else
          arr[i].pos=4;
     }
   arr['_'-'\0'].but=0;	
   arr['_'-'\0'].pos=1;
  arr['.'-'\0'].but=1;	
  arr['.'-'\0'].pos=1;
    arr[','-'\0'].but=1;	
    arr[','-'\0'].pos=2;
   arr['?'-'\0'].but=1;	
    arr['?'-'\0'].pos=3;
     arr['!'-'\0'].but=1;	
    arr['!'-'\0'].pos=4;
    string s;
    int pb;
    long long tm;
    while(t--)
    {
    	tm=0;
    	cin>>s;
    	pb=1;
    	for(i=0;i<s.length();i++)
    	{
    		if(pb!=arr[s[i]-'\0'].but)
    	        tm++;
    	  pb=arr[s[i]-'\0'].but;
    	  tm+=arr[s[i]-'\0'].pos;    
    	}
    cout<<tm<<endl;
    }
    
  return 0;
}
