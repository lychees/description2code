#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    
    while (t--) {
    	int n, c, m;
    	scanf("%d %d %d", &n, &c, &m);
    	int toffe = n/c;
    	int x = n/c;
    	while (x >= m) {
    		toffe += x/m;
    		x = x/m + x%m;
    	}
    	
    	printf("%d\n", toffe);
    }
    
    return 0;
}
