#include <iostream>
using namespace std;
bool vis[1000005];
int main()
{
    int arr[1000005];
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	for(int i=0;i<n;i++) 
    	{
    		scanf("%d",&arr[i]);
    		vis[i]=0;
    	}
    	vis[n-1]=1;
    	int ans=1;
    	for(int i=n-2;i>=0;i--)
    	{
    		int k=i+1;
    		int flag=0;
    		while(arr[i]>arr[k]&&k<n)
    		{
    			if(vis[k]) 
    			{
    				vis[i]=1;
    				ans++;
    				flag=1;
    				break;
    			}
    			k++;
    		}
    		if(k==n&&!flag)
    		{
    			vis[i]=1;
    			ans++;
    		}
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
