#include <bits/stdc++.h>
using namespace std;

int main()
{
   long long int m,n,x,i;
   scanf("%lld%lld",&m,&n);
   std::priority_queue<int>q;
   while(m--){
   	scanf("%lld",&x);
   	for(i=1;i<=x;i++)
   	q.push(i);
   }x=0;
   for(i=0;i<n;i++)
   {x=x+q.top();q.pop();}
   printf("%lld\n",x);
    return 0;
}
