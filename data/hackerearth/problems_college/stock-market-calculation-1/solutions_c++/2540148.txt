#include <iostream>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
       cin>>n;
       int arr[n];
       for(int i=0;i<n;i++)
       {
       	cin>>arr[i];
       }
       int ans=0;
       int d=arr[0];
       int count=1;
       for(int i=1;i<n;i++)
       {
       	  while(arr[i]>arr[i-1]&&i<n) 
       	  {
       	  	d+=arr[i];
       	  	count++;
       	  	i++;
       	  }
       	  ans+=count*arr[i-1] - d;
       	  d=arr[i];
       	  count=1;
       }
       cout<<ans<<endl;
    }
   // return 0;
}
