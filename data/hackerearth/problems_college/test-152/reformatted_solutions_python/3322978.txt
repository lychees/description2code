t = int(raw_input())
for x in range(t):
    n = int(raw_input())
    s = raw_input()
    ar = []
    for y in s:
        ar.append(y)
    l = ['a', 'b', 'c', 'd', 'e', 'f', 'g', 'h', 'i', 'j', 'k', 'l', 'm',
         'n', 'o', 'p', 'q', 'r', 's', 't', 'u', 'v', 'w', 'x', 'y', 'z',]
    c = []
    for z in ar:
        if z in l:
            l.remove(z)
            c.append(ar.count(z))
    if max(c) % 2 == 0:
        print "Yes"
    else:
        print "No"
