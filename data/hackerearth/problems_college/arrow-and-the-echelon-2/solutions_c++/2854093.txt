#include <bits/stdc++.h>
using namespace std;
 //k
int main()
{
   int i,t,x,n1,n2,rem;
   set<int> s;
   s.insert(0);
   for(i=30;i<=100000;i=i+30)
   	s.insert(i);
   	set<int>::iterator it;
   vector<pair<int,int> > v(100001);
   v[3].first=3;
   v[3].second=0;
   v[5].first=0;
   v[5].second=5;
   v[6].first=6;
   v[6].second=0;
   for(i=8;i<=100000;i++)
   {
   		if(i%3==0)
   		{
   			v[i].first=i;
   			v[i].second=0;
   		}
   		else if(i%5==0)
   			{
   				it=s.lower_bound(x);
   				if((it==s.end())||((*it)!=x))
   					it--;
   				n1=(*it)/2;
   				n2=(*it)/2;
   				rem=x-(*it);
   				n2=n2+rem;
   				v[i].first=n1;
   				v[i].second=n2;
   			}
   		else
   		{
   			v[i].first=v[i-3].first+3;
   			v[i].second=v[i-3].second;
   		}
   }
   scanf("%d",&t);
   while(t--)
   {
   		scanf("%d",&x);
   		if((x<3)||(x==4)||(x==7))
   			printf("-1\n");
   		else
   		{
   			
   			n1=v[x].first;
   			n2=v[x].second;
   				for(i=1;i<=n1;i++)
   				printf("5");
   			for(i=1;i<=n2;i++)
   				printf("3");
   			printf("\n");
   		}
   }
    return 0;
}
