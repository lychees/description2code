#include <iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;


struct stu
{
	stu(string n,int ma)
	{
		name = n;
		marks = ma;
	}
	string name;
	int marks;
};

int main()
{
    //cout << "Hello World!" << endl;
    vector<stu> students;
    int nostud;
    
    cin >>nostud;
    string name;
    int marks;
    for(int i=1 ; i <= nostud ; i++)
    {
    	cin >>name>>marks;
   		students.emplace_back(stu(name,marks));
    }
    
     std::sort(students.begin(),students.end(),[] (const  stu &obj1,const  stu &obj2){
    	
    	if( obj1.marks == obj2.marks)
    	{
    		return obj1.name < obj2.name;
    	
    	}
    	else
    	{
     	return(obj1.marks > obj2.marks);
    		
     	}
     });
    
    for( auto o : students)
    {
    	cout <<o.name<< " "<<o.marks;
    	cout<<"\n";
    }
   		
    return 0;
}
