using namespace std;

#define ll long long

#define inf 1000000000
#define mod 1000000007

typedef pair<ll, ll> pll;

ll dp[2001], ctask[2001];
pll setter[2001];

bool compare(pll a, pll b)
{
    return a.first < b.first;
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t-- > 0)  
    {
        int n, k;
        scanf("%d %d", &n, &k);
        for(int i = 0; i < n; i++)
            scanf("%lld", &setter[i].first);
        for(int i = 0; i < n; i++)
            scanf("%lld", &setter[i].second);
        sort(setter, setter+n, compare);
        ctask[0] = setter[0].second;
        for(int i = 0; i < n; i++)
            ctask[i] = ctask[i-1] + setter[i].second;
        for(int i = 0; i < n; i++)
        {
            dp[i] = setter[i].first * (ctask[i]+k);
            for(int j = 0; j < i; j++)
                dp[i] = min(dp[i], dp[j] + (ctask[i] - ctask[j] + k) * setter[i].first);
        }
        printf("%lld\n", dp[n-1]);
    }
	return 0;
}
