#include <iostream>
#include <cmath>
#include <math.h>
using namespace std;

double process(string bin) {
	double val = 0;
	for(int i=7;i>=0;i--) {
		val += bin[i] * pow(2,(7-i));
	}
	if(bin[0] == 1) {
		return 256-val;
	}
	else {
		return val;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--) {
		string a, b;
		a.clear();
		b.clear();
		cin>>a;
		cin>>b;
		double A, B;
		A = process(a);
		B = process(b);
		//cout<<A<<" "<<B<<endl;
		double LA, LB;
		LA = log(A);
		LB = log(B);
		//cout<<LA<< " "<<LB<<endl;
		if((B/A) <= (LB/LA)) cout<<"The Hound\n";
		else cout<<"The Mountain\n";
		
	}
    return 0;
}
