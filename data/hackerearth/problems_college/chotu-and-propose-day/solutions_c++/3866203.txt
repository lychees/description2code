#include <bits/stdc++.h>
#include <iostream>
using namespace std;

int main()
{
    int t, caa, cbb, cab, cba, i;
    scanf("%d", &t);
    while(t--){
    	scanf("%d %d %d %d", &caa, &cbb, &cab, &cba);
    	for(i = 0; i < caa / 2; i++) {
    		printf("aa");
    	}
    	int count = cab;
    	if(cba < count) count = cba;
    	if(caa != 0 && cba != 0) {
    		for(i = 0; i < count; i++) {
    			printf("ab");
    		}
    	}
    	for(i = 0; i < cbb / 2; i++) {
    		printf("bb");
    	}
    	if(caa % 2 != 0) {
    		printf("aa");
    	} else if(cbb % 2 != 0) {
    		printf("bb");
    	}
    	for(i = 0; i < cbb / 2; i++) {
    		printf("bb");
    	}
    	if(caa != 0 && cba != 0) {
    		for(i = 0; i < count; i++) {
    			printf("ba");
    		}
    	}
    	for(i = 0; i < caa / 2; i++) {
    		printf("aa");
    	}
    	printf("\n");
    }
    return 0;
}
