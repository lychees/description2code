#include <iostream>
#define ll long long
using namespace std;
const int mod=1e9+7;
int arr[20];
long long power(long long base, long long exponent, long long modulus)
{
    long long result = 1;
    while (exponent > 0) 
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{
	int T;
	cin>>T;
	for(int t=0;t<T;t++)
	{
		int n;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>arr[i];
		// this is a set of size n
		ll ans=1;// jab ek hi set bit ho
		for( ll i=1 ; i < (1<<n) ; i++ )
		{
			ll mul=1;
			ll add=0;
			int number_of_set_bits=0;
			for(int j=0;j<n;j++)
			{
				if( i&(1<<j) )
				{
					number_of_set_bits++;
					mul=(mul*arr[j])%mod;
					add=(add+arr[j])%mod;
				}
			}
			if(number_of_set_bits>1)
			{
				ll temp=power(add,mod-2,mod);
				temp=(temp*mul)%mod;
				ans=max(ans,temp);
			}
		}
		cout<<"Case #"<<t+1<<": "<<ans<<endl;
	}
    return 0;
}
