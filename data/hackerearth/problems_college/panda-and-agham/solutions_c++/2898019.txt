# include <iostream>
# include <algorithm>
using namespace std;

int n;
int ar[100000];
int res;
int cows;
bool isarrang (int x) {
	int start = 0;
	int count = 1;
	for (int i = 1; i < n; i++) {
		if (ar[i] - ar[start] >= x){
			count ++;
			start = i;
		}
		if (count == cows)
			return 1;
	}

		return 0;
}
void disc_binary(int low, int high) {
	int mid;
	if  (low > high) 
		return;

		mid = (low + high)/2;
		if (isarrang (mid) ) {
			res = mid;
			disc_binary (mid+1, high);
		}
		else
			disc_binary (low, mid - 1);
}
int main () {
	int t;
	cin >> t;
	while  (t--) {
	cin >> n;
	cin >>cows;
	//int ar[100];
	for (int i = 0; i < n; i++)
		cin >> ar[i];
	sort(ar, ar+n);
	//`:wq
	//`:wq
	//cout <<isarrang (2);
	disc_binary (0, ar[n-1]);
	cout <<res<<endl;}
	return 0;
}
