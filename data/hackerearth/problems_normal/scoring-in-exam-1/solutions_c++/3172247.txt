#include<bits/stdc++.h>
using namespace std;

int main()
{
	int N,Q,s;
    cin>>N>>Q;
    int times[N+1],score[N+1];
    for(int i=0;i<N;i++)
    	cin>>times[i];
    for(int i=0;i<N;i++)
    	cin>>score[i];
    sort(times,times+N);
    int total[N+1];
    total[0] = times[0];
    for(int i=1;i<N;i++)
    	total[i] = total[i-1] + times[i];
    while(Q--)
    {
    	cin>>s;
    	if(s == N)
    		cout<<total[N-1];
    	else
    		cout<<total[N-1]-total[N-s-1];
    	cout<<endl;
    }
    return 0;
}
