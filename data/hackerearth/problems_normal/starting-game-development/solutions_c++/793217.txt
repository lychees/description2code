#include <bits/stdc++.h>
using namespace std;

int arr[100][5000];

int main()
{
    int n,m,q,a;
    scanf("%d %d %d",&n,&m,&q);
    
    
    for(int i = 0; i<n;i++){    	
    	for(int j = 0; j<m;j++){
    		scanf("%d", &arr[i][j]);
    	}    	
    }
    
    while(q--){
    	
    	int min = 10000000;
    	
    	for(int i = 0; i<n; i++){
    		
    		scanf("%d", &a);
    		
    		int j = m-1;
    		
    		while(a < arr[i][j] && j > 0) j--;   		
    		
    		
    		if(min > j) min = j;
    		    	
    	}
    	printf("%d \n", min+1);
    	
    	
    }
    return 0;
}
