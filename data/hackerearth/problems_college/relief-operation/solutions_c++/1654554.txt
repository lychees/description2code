#include <bits/stdc++.h>
#include <bits/stdc++.h>
#define si(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define ss(n) scanf("%s",n)
#define fr(i, n) for(int i = 0; i < n; i++)
#define REP(i, a, b) for (int i = int(a); i <= int(b); i++)
#define REN(i, a, b) for (int i = int(a); i >= int(b); i--)
#define ms(i, n) memset(i, n, sizeof(i))
#define INF 1002000000
#define MOD 1000000007
typedef long long LL;
using namespace std;
int dp[2003][2003], m, w[2003], c[2003], l;
int travel(int curr, int sum) {
    if(curr == m)
        return 0;
    if(dp[curr][sum] != -1)
        return dp[curr][sum];
    int maxi = 0;
    maxi = max(maxi, travel(curr + 1, sum));
    if(sum - w[curr] >= 0)
        maxi = max(maxi, travel(curr + 1, sum - w[curr]) + c[curr]);
    return dp[curr][sum] = maxi;
}
int main ()
{
	//freopen ("input.txt","r",stdin);
	//freopen ("output.txt","w",stdout);
	int t;si(t);
	while(t--) {
        si(m);si(l);
        fr(i, m) {
            fr(j, l + 1) {
                dp[i][j] = -1;
            }
        }
        fr(i, m) {
            si(w[i]);
        }
        fr(i, m) {
            si(c[i]);
        }
        cout<<travel(0, l)<<endl;
	}
	return 0;
}
