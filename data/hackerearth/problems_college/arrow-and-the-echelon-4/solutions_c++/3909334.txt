#include <stdio.h>
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int x,i,j,check,k;
    	scanf("%d",&x);
    	
    	for(i=0;i<=x/5;i++)
    	{
    		 check=0;
    	    for(j=0;j<=x/3;j++)
    	    if(x==3*j+5*i){check=1;break;}
    	    
    	    if(check) break;
    	}
           if(check==0) printf("-1\n");
            else
              {
                 for(k=0;k<3*j;k++) printf("5");
                 for( k=0;k<5*i;k++) printf("3");
                 printf("\n");
               }
    }
    return 0;
}
