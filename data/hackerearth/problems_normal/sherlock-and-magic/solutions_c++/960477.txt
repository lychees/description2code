#include<bits/stdc++.h>

using namespace std;
struct criminals
{
    int x,y;
    char d;
    bool exist;
};

int left_criminals(criminals C[],int N)
{
    for(int t=1;t<4004;t++)
    {
        for(int i=0;i<N;i++)
        {
            if(C[i].exist){
                switch(C[i].d)
                {
                    case 'N': ++(C[i].y); break;
                    case 'S': --(C[i].y); break;
                    case 'W': --(C[i].x); break;
                    case 'E': ++(C[i].x); break;
                }
            }
        }
        for(int i=0;i<N-1;i++)
        {
            bool flag = false;
            if(!C[i].exist)
                continue;
            for(int j=i+1;j<N;j++)
            {
                if(C[j].exist && C[i].x == C[j].x && C[i].y == C[j].y)
                {
                    flag = true;
                    C[j].exist = false;
                }
            }
            if(flag)
                C[i].exist = false;
        }
    }
    int count = 0;
    for(int i=0;i<N;i++)
    {
        if(C[i].exist)
            count = count+1;
    }
    return count;
}
int main()
{
    int tests,N;
    criminals C[50];
    cin>>tests;
    while(tests--)
    {
        cin>>N;
        for(int i=0;i<N;i++)
        {
            cin>>C[i].x;
            C[i].x = 2*C[i].x;
            cin>>C[i].y;
            C[i].y = 2*C[i].y;
            cin>>C[i].d;
            C[i].exist = true;
        }
        int count = left_criminals(C,N);
        cout<<count<<"\n";
    }
}
