#define f(i, a, b) for(int i = a; i < b; i++)
#define rep(i, n) f(i, 0, n)

typedef long long ll;
typedef pair<int, int> pii;

static pii p[10000005];

int main() {
  int T; scanf("%d", &T); while(T--) {

    int n;
    scanf("%d", &n);

    rep(i, n) {
      scanf("%d", &p[i].first);
      p[i].second = 1;
    }

    rep(i, n) {
      scanf("%d", &p[i + n].first);
        p[i+n].first+=p[i].first;
      p[i + n].second = -1;
    }

    sort(p, p + 2 * n);

    int res = 0;
    int ct = 0;
    rep(i, 2 * n) {
      ct += p[i].second;
      res = max(res, ct);
    }

    printf("%d\n", res);

  }
}
