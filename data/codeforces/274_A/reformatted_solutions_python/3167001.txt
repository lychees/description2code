n, k = map(int, raw_input().split())
d = sorted(map(int, raw_input().split()))
h = {}
l = [0 for i in range(n)]
ans = 0
for i in range(n):
    h[d[i]] = i
for i in range(n):
    if l[i] == 0:
        l[i] = 1
    if l[i] % 2 == 1:
        ans += 1
    t = h.get(d[i]*k)
    if t is not None:
        l[t] = l[i]+1
print(ans)
