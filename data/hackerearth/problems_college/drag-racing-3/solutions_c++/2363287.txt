#include<cstdio>
#include<iostream>
#include<map>
using namespace std;
map<string,int> mp;
int main()
{
	string a,b;
	int N; cin>>N;
	for(int i=1;i<(1<<N);i++)
	{
	
		cin>>a>>b;
	if(mp[a]==0) mp[a]=1;
	mp[b]=2;
		
	}
	
	map<string,int> :: iterator it;
	it=mp.begin();
	for(;it!=mp.end();++it)
	{
		int x=it->second;
		if(x==1) cout<<it->first;
	}
}
