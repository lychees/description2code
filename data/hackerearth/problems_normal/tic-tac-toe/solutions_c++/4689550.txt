#include <bits/stdc++.h>
using namespace std;
#define g 1000000007
 
int main()
{
	unsigned long long int t,n,in,ans,co,c,reco,dire,pr;
	scanf("%llu",&t);
	while(t--)
	{
		ans=0;
		in=0;
		scanf("%llu",&n);
		c=n-1;
		
		
			in=( ( ( (c * (c+1))%g * (((2*c)+1)%g) ) % g ) * 166666668 ) % g;
		
		co=(n*(n-1))/2;
		
		reco=co%n;
		dire=co/n;
		pr=(dire*dire)%g;

		
			ans=(ans+(pr*(n-reco))%1000000007)%1000000007;
		
		pr=((dire-1)*(dire-1))%g;
		
		
			ans=(ans+(pr*reco)%1000000007)%1000000007;
		
		
		cout<<ans<<" "<<in<<endl;
	}
    return 0;
}
