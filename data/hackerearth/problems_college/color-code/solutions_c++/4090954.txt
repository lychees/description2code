#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    int i=1;
    while(t--)
    {
        int r,b,g;
        scanf("%d%d%d",&r,&b,&g);
        string red,blue,green;
        int a,d,c;
        char temp;
        a=r/16;
        if(a<10)
            temp=(char)(a+48);
        else
            temp=(char)(85+(a%10));
        //cout<<temp<<"\n";
        red=temp;
        //cout<<"red="<<red<<"\n";
        a=r%16;
        if(a<10)
            temp=(char)(48+a);
        else
            temp=(char)(85+(a%10));
         //cout<<temp<<"\n";
        red=red+temp;
        //cout<<"red="<<red<<"\n";
        d=b/16;
        if(d<10)
            temp=(char)(d+48);
        else
            temp=(char)(85+(d%10));
        //cout<<temp<<"\n";
        blue=temp;
        //cout<<"blue="<<blue<<"\n";
        d=b%16;
        //cout<<"b%16="<<d<<"\n";
        if(d<10)
            temp=(char)(48+d);
        else
            temp=(char)(85+(d%10));
        //cout<<temp<<"\n";
        blue=blue+temp;
        //cout<<"blue="<<blue<<"\n";
        c=g/16;
        if(c<10)
            temp=(char)(c+48);
        else
            temp=(char)(85+(c%10));
        //cout<<temp<<"\n";
        green=temp;
        //cout<<"green="<<green<<"\n";
        c=g%16;
        if(c<10)
            temp=(char)(48+c);
        else
            temp=(char)(85+(c%10));
        //cout<<temp<<"\n";
        green=green+temp;
        //cout<<"green="<<green<<"\n";
        cout<<"Case "<<i<<":"<<" "<<red<<" "<<blue<<" "<<green<<"\n";
        i++;
    }
    return 0;
}
