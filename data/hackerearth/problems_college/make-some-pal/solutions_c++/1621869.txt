#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char s[55];
        scanf("%s",s);
        int i,ans=0,l,j;
        l=strlen(s);
        for(i=0;i<l;i++)
        {
            int b=0;

                int j=i,m=l-1;
                while(j<=m)
                {
                    if(s[j]!=s[m])
                    {
                        b=1;
                        break;
                    }
                    else
                    {
                        j++;m--;
                    }
                }

            if(b==1)
                continue;
            else
            {
                ans=l+i;
                printf("%d\n",ans);
                break;
            }
        }
    }
    return 0;
}
