for _ in range(int(raw_input())):
    n=int(raw_input())
    i=0
    x=n
    a=[] 
    while(x>=3):
        x=n-(5*i)
        if x%3==0:
            for i in range(x):
                a.append('5')
            for i in range(n-x):
                a.append('3')
            break
        i+=1
    if len(a)==0:
        print -1
    else:
        print ''.join(a)
