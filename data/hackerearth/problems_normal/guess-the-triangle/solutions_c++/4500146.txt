using namespace std;
vector<double> func(vector<double> v)
{
	double sum=v[0]+v[1]+v[2];
	vector<double> ret;
	ret.resize(3);
	ret[0]=sum-2*v[1];
	ret[1]=sum-2*v[2];
	ret[2]=sum-2*v[0];
	return ret;
}
int main()
{
	vector<double> midx,midy;
	vector<pair<double,double> > snorlex;
	double a,b;
	for(int i=0;i<3;i++)
	{
		cin>>a>>b;
		midx.push_back(a);
		midy.push_back(b);
	}
	vector<double> v1=func(midx);
	vector<double> v2=func(midy);
	for(int i=0;i<3;i++)
	{
		snorlex.push_back(make_pair(v1[i],v2[i]));
	}
	sort(snorlex.begin(),snorlex.end());
	cout<<fixed;
	cout<<setprecision(4);
	for(int i=0;i<3;i++)
	{
		cout<<snorlex[i].first<<" "<<snorlex[i].second<<endl;
	}
	return 0;
}
