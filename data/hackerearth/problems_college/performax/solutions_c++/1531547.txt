#include <iostream>
using namespace std;

int main()
{
    int n;
    long long int d,a,b,ans=0,i;
    cin>>n;
    cin>>d;
    for(i=0;i<n;i++)
    {
    	if(i==0)
    	{cin>>a;b=a;}
    	if(i>0)
    	{
    		
    		cin>>a;
    		if(a<b)
    		{ans+=((b-a)/d)+1;
    		b=((((b-a)/d)+1)*d)+a;
    		}
    		else
    		{
    			b=a;
    		}
    		
    	}
    }
    cout<<ans<<endl;
    return 0;
}
