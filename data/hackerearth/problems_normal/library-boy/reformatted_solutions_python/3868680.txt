import collections
N = input()

for i in range(N):

    book_name = raw_input()
    books.append(book_name)

books.sort()

shelves = collections.defaultdict(dict)
for i in range(len(books)):
    shelf_char = books[i][0:1]

    shelves[shelf_char][i] = books[i]

total_shelves = shelves.keys()
shelf = 0
for i in range(len(total_shelves)):
    add = (len(shelves[total_shelves[i]]))
    if (add > 10):
        if ((add % 10) > 0):
            shelf = shelf + int(add/10) + 1
        else:
            shelf = shelf + int(add/10)
    else:
        shelf = shelf + 1
