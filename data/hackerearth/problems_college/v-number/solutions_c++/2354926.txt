#include <bits/stdc++.h>
using namespace std;

int main()
{  int t;
cin>>t;
while(t--)
{ int i,j,c=0,d=0,n=0;
    char a[10000];
  cin>>a;
  n=strlen(a);
  for(i=0,j=(n/2);i<n/2;i++,j++) {c=c+a[i];d=d+a[j];}
  if(c==d) cout<<"YES"<<endl;
  else cout<<"NO"<<endl;
}

    return 0;
}
