#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long n,i,x,m,r,j,mc,rc;
    cin>>t;
    while(t--){
    	vector<long long> v;
    	m=r=mc=rc=0;
    	cin>>n;
    	for(i=0;i<n;i++){
    		cin>>x;
    		v.push_back(x);
    	}
    	for(i=0,j=n-1;i<=j;){
    		if(m<=2*r){
    			m+=v[i];
    			mc++;
    			i++;
    		}
    		else{
    			r+=v[j];
    			rc++;
    			j--;
    		}
    	}
    	cout<<mc<<" "<<rc<<endl;
    }
    return 0;
}
