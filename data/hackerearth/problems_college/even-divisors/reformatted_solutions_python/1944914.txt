T = int(raw_input())
for loop in xrange(T):
    N = int(raw_input())
    if N == 2:
        print 1
        continue
    ans = 0
    i = 2
    while i * i <= N:
        cnt = 0
        while N % i == 0:
            cnt += 1
            N /= i
        if i == 2:
            ans += cnt
        else:
            ans *= cnt + 1
        i += 1
    if N > 1:
        ans *= 2
    print ans
