#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int n,count=0,m;
		
		cin>>n;
		m=n-1;
		if(n==1)
		{
			cout<<"0 1\n";
			continue;
		}
		if(m==0)
			count=1;
		
		while(m)
		{
			m=m/2;
			count++;
		}
		m=0;
		while(count)
		{
			m=m<<1;
			m++;
			count--;
		}
		cout<<m<<" "<<(m+1)-2*(m-n+1)<<"\n";
		
	}
	return 0;
}
