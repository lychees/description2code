int main()
{
	int n, q; cin >> n >> q;
	vector<int> arr(n);
	for(int i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	sort(arr.begin(), arr.end());
	while(q--)
	{
		int x; cin >> x; bool flag = false;
		if(x < arr[0] || x > arr[n-1]) cout <<  "NO";
		else
		{	
			int low = 0; int high = n-1;
			while(low<=high)
			{
				int mid = (low+high)/2;
				if(arr[mid] == x) 
				{	
					flag = true;
					break;
				}
				else if(arr[mid] > x) high = mid-1;
				else low = mid+1;
			}
			if(flag) cout << "YES";
			else cout << "NO";
		}
		cout << endl;
	}
}
