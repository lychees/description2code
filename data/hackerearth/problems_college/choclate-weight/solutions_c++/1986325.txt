#include <iostream>
#include<algorithm>
using namespace std;
long long int binarySearch(long long int arr[], long long int l, long long int r, long long int x)
{
   if (r >= l)
   {
        long long int mid = l + (r - l)/2;
 
       
        if (arr[mid] == x)  return 1;
 
        
        if (arr[mid] > x) return binarySearch(arr, l, mid-1, x);
 
        
        return binarySearch(arr, mid+1, r, x);
   }
 
   
   return -1;
}

bool isSum(long long int A[], long long int N, long long int sum)
{
    long long int l, r;
 
    
    sort(A, A+N);
    
 
   
    l = 0;
    r = N-1; 
    while(l < r)
    {
         if(A[l] + A[r] == sum)
              return 1; 
         else if(A[l] + A[r] < sum)
              l++;
         else 
              r--;
    }    
    return 0;
}
 
 

int main()
{
    
    long long int n;
    scanf("%lld", &n);
    long long int a[n];
    for(long long int i=0; i<n; i++){
    	scanf("%lld", &a[i]);
    }
    
    long long int t;
    scanf("%lld", &t);
    int result;
    while(t--){
    	long long int x;
    	scanf("%lld", &x);
        result = binarySearch(a, 0, n-1, x);
        
       
       if(result==1||isSum(a, n, x))
           printf("I Like You\n");
       else
            printf("I Hate You\n");
        
        
    }
    
    
    return 0;
}
