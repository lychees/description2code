T = int(raw_input())
while (T):
    T -= 1
    N = int(raw_input())
    hires, unassigned = 0, 0
    while (N):
        N -= 1
        X = raw_input()
        if (X[0] == 'N'):
            if (hires > 0):
                hires -= 1
            else:
                unassigned += 1
        else:
            hires += int(X)
    print unassigned
