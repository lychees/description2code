#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const ll N = 1e4 + 5;
const ll MOD = 1e9 + 7;
ll dp[N][15];
int n, p;

ll calc(int i, int qtd)
{
	if (i == n)
		return 1;
	
	ll &ans = dp[i][qtd];
	if (ans == -1)
	{
		if (!i || qtd == p - 1)
			ans = calc(i + 1, 0) % MOD;
		else
			ans = (calc(i + 1, qtd + 1) + calc(i + 1, 0)) % MOD;
	}
	return ans;
}

int main()
{
	ios_base :: sync_with_stdio(0);
	cin.tie(0);
	
	int cases;
	
	cin >> cases;
	
	while (cases--)
	{
		cin >> n >> p;
		memset(dp, -1, sizeof dp);
		
		cout << calc(0, 0) << '\n';			
	}
}
