#include<bits/stdc++.h>

using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define fu(i,a,b) for(int i=a;i<=b;i++)
#define fd(i,a,b) for(int i=a;i>=b;i--)
#define vi vector<int>
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define ll long long

#define mod 1000000009

ll dpow[2][2];
ll zero[2][2];
ll M[2][2]={{1,1},{1,0}};

void power(ll x[2][2],ll n)
{
    ll iden[2][2]={{1,0},{0,1}};
    for(int i=0;i<2;i++)
    for(int j=0;j<2;j++)
        dpow[i][j]=iden[i][j];
    while(n)
    {
        if(n&1)
        {
            ll res[2][2];
            rep(i,2) rep(j,2) res[i][j]=0;

            for(int i=0;i<2;i++)
            for(int j=0;j<2;j++)
            for(int k=0;k<2;k++)
                    res[i][j]+=(dpow[i][k]*x[k][j])%mod;

            rep(i,2) rep(j,2) dpow[i][j]=res[i][j]%mod;
        }
        ll res2[2][2];
        rep(i,2) rep(j,2) res2[i][j]=0;

        for(int i=0;i<2;i++)
        for(int j=0;j<2;j++)
        for(int k=0;k<2;k++)
                res2[i][j]+=(x[i][k]*x[k][j])%mod;

        rep(i,2) rep(j,2) x[i][j]=res2[i][j]%mod;

        n>>=1;
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    ll n;
    while(t--)
    {
        scanf("%lld",&n);
        ll x[2][2]={{1,1},{1,0}};
        power(x,n);
        ll ans=(dpow[0][0]*dpow[0][1])%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
