#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int i=1;
    long dp[100+1];
    for(i=1;i<=8;i++)
    {
    	dp[i]=INT_MAX;
    }
    dp[6]=dp[8]=1;
    for(i=9;i<=100;i++)
    {
    	long tmp;
    	tmp=dp[i-8]+1;
    	tmp=min(tmp,dp[i-6]+1);
    	dp[i]=tmp;
    }
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(dp[n]>=INT_MAX)
    	  cout<<"-1\n";
    	else
    	  cout<<dp[n]<<"\n";
    }
    return 0;
}
