int main() {
ios_base::sync_with_stdio(false); cin.tie(0);
int t;
cin>>t;
while(t--)
{
	string a;
	cin>>a;
	int y = a.length();
	int A[y];
	int count = 0;
	mem(A,0);
	for(int i=0;i<y;i++)
	{
		if(a[i]=='D')
	{
		A[i]=1;
	}
	else if(a[i]=='B')
	{
		A[i]=1;
		A[i-1]=1;
		A[i-2] = 1;
	}
	else if(a[i]=='S')
	{
		A[i]=1;
		A[i-1]=1;
		A[i+1]=1;
	}
	}
	for(int i=0;i<y;i++)
	{
		if(A[i]==0)
		{
			count++;
		}
	}

}

}
