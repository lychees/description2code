#include<iostream>
using namespace std;
int main()
{
	long long int num,cnt=0,t;
	cin>>t;
	while(t--)
	{
	cin>>num;
	while(1)
	{
			if(num==1)
		break;
		
		if(num%3==0)
		{
			cnt++;
			num/=3;
		}
		
	   else if(num%2==0)
	   {
	   	cnt++;
	   	num/=2;
	   }
		else
		{
		
		num=num-1;
		cnt++;
	}
	}
		cout<<cnt<<endl;
	cnt=0;
}

	return 0;
}
