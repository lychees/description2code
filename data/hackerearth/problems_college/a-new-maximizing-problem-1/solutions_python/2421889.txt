MOD=1000000007
t=input()
while t:
	t-=1
	n=input()
	arr=[]
	arr=map(int, raw_input().split())
	B=0
	for i in range(n):
		if arr[i]<0:
			B=B+((-1)*arr[i])
		else:
			B=B+arr[i]
	A=0
	arr.sort()
	i,j=0,n-1
	while i<=j:
		A=A-arr[i]
		if j>i:
			A=A+arr[j]
		i+=1
		j-=1
	
	C=((A%MOD)*(B%MOD))%MOD
	print C
