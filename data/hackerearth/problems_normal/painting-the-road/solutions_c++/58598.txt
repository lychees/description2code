#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;

struct number
{
	int x,y,cost;
}numbers[515];

bool sorter(struct number const& l,struct number const& r)
{
	if(l.x<r.x)
	return 1;
	return 0;
}

int main() {
	// your code goes here
	int t,l,k;
	cin>>t;
	while(t--)
	{
		int dp[515];
		dp[0]=0;
		cin>>l>>k;
		for(int i=1;i<=l;i++)
		dp[i]=1000000;
		for(int i=0;i<k;i++)
		{
			cin>>numbers[i].x>>numbers[i].y>>numbers[i].cost;
		}
		sort(numbers,numbers+k,sorter);
		if(numbers[0].x!=0)
		{
			cout<<"-1"<<endl;
			continue;
		}
		//cout<<dp[l]<<endl;
		
		for(int i=0;i<k;i++)
		{
			int idx=0;
			for(int j=numbers[i].x+1;j<=numbers[i].y;j++)
			{
				if(dp[j]>dp[numbers[i].x]+numbers[i].cost)
				{
					dp[j]=dp[numbers[i].x]+numbers[i].cost;
					idx=j+1;
					break;
				}
			}
			if(idx)
			for(int j=idx;j<=numbers[i].y;j++)
			{
				dp[j]=dp[idx-1];
			}
		}
		//cout<<dp[l]<<endl;
		if(dp[l]==1000000){
			cout<<"-1"<<endl;
			continue;
		}
		cout<<dp[l]<<endl;
	}
	return 0;
}
