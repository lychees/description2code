n = int(input())
candidates = []
losers = []
total_candidates = (2 ** n) - 1
while (total_candidates > 0):
    x, y = raw_input().split()
    candidates.append(x)
    candidates.append(y)
    losers.append(y)
    total_candidates -= 1

candidates = set(candidates)
losers = set(losers)
winner = list(candidates.symmetric_difference(losers))[0]
print winner
