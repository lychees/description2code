#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
    int t, i, j, n;
    char s[1000][1000];
    int cherriesRows[1000], cherriesCols[1000];

    cin >> t;

    while (t--)
    {
        cin >> n;

        int currcherries, totalcherries = 0;

        for (i = 0; i < n; i++)
        {
            currcherries = 0;
            for (j = 0; j < n; j++)
            {
                cin >> s[i][j];
                if (s[i][j] == '#')
                    currcherries++;
            }
            cherriesRows[i] = currcherries;
            totalcherries += currcherries;
            if (i > 0)
            {
                cherriesRows[i] += cherriesRows[i - 1];
            }
        }

        for (j = 0; j < n; j++)
        {
            currcherries = 0;
            for (i = 0; i < n; i++)
            {
                if (s[i][j] == '#')
                    currcherries++;
            }
            cherriesCols[j] = currcherries;
            if (j > 0)
            {
                cherriesCols[j] += cherriesCols[j - 1];
            }
        }

        /*
        cout << "\nCHERRIES IN ROWS: \n";
        for (i = 0; i < n; i++)
            cout << cherriesRows[i] << " ";

        cout << "\nCHERRIES IN COLS: \n";
        for (i = 0; i < n; i++)
            cout << cherriesCols[i] << " ";
        */

        if (totalcherries % 2 == 1)
        {
            cout << "NO\n";
        }
        else
        {
            int eachPart = totalcherries / 2;

            if (binary_search(cherriesRows, cherriesRows + n, eachPart))
            {
                cout << "YES\n";
            }
            else if (binary_search(cherriesCols, cherriesCols + n, eachPart))
            {
                cout << "YES\n";
            }
            else
            {
                cout << "NO\n";
            }
        }

    }
    return 0;
}
