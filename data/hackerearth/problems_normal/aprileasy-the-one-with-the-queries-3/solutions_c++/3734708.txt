#include<iostream>
#include<string>
#include<cstdio>
using namespace std;
bool comp(string a,string q)
{
    long i,j,x=0;
    for(j=0;j<q.length();j++)
    {
        long y = a.find(q[j],x);
        if(y== string::npos)
            return false;
        x=y+1;
    }
    if(j==q.length())
        return true;

}
int main()
{
    long n;
    string s,q;
    getline(cin,s);
    cin>>n;
    for(long i=0;i<n;i++)
    {
      cin>>q;
      if(comp(s,q))
        cout<<"Yes"<<endl;
      else
        cout<<"No"<<endl;
    }

}
