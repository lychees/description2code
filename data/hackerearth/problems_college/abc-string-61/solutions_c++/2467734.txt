LL mpow(LL a, LL n)
{
    LL ret=1;
    LL b=a;
    while(n)
    {
        if(n&1)
        ret=(ret*b)%MOD;
        b=(b*b)%MOD;
        n>>=1;
    }
    return (LL)ret;
}
