#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
int arr[100002][26];
int main() 
{
    int t;
    scanf("%d",&t);
    char s[100002];
    char s1[100002];
    while(t--) {
    	scanf("%s%s",s,s1);
    	int i,j; 	
   		int a[26]={0};
    	int b[26]={0};
    	if(strlen(s1) > strlen(s)) {
    		puts("NO");
    		continue;
    	}
    	for(i = 0; i < strlen(s1); i++) {
    		b[s1[i]-97]++;
    	}
    	int l = strlen(s1);
    	bool flag = false;
    	int x = 0;
    	for(i = 0; i < strlen(s); i++) {
    		a[s[i]-97]++;
    		for(j = 0; j < 26; j++) {
    			arr[i][j] = a[j];
    		}
    		x = 0;
    		if(i == l-1) {
    			for(j = 0; j < 26; j++) {
    				if(arr[i][j] != b[j]) {
    					x = 1;
    					break;
    				}
    			}
    			if(x == 0) {
    				flag = true;
    				break;
    			}
    		}
    		else if(i >= l) {
    			for(j = 0; j < 26; j++) {
    				if(arr[i][j]-arr[i-l][j] != b[j]) {
    					x = 1;
    					break;
    				}
    			}
    			if(x == 0) {
    				flag = true;
    				break;
    			}
    		}
    	}
    	puts(flag?"YES":"NO");
    }
	return 0;
}
