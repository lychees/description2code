#include <iostream>
#include <vector>
using namespace std;

int main()
{
	 long long int T,i,j,N,K,P;
    cin>>T;
    vector<long long int>answer;
    for(i=0;i<T;i++)
    {
        long long int small,Size,num;
        cin>>N>>K>>P;
        small=P;
        Size=N;
        for(j=0;j<K;j++)
        {
            cin>>num;
            if(num<=small)
                small++;
            Size--;
        }
        if(Size>0&&small<=N)
            answer.push_back(small);
        else
            answer.push_back(-1);
    }
    for(i=0;i<T;i++)
        cout<<answer[i]<<endl;
    return 0;
}
