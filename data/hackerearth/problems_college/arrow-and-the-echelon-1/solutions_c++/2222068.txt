#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int temp=0,temp1=0;
        if(n%3==0)
            temp=n;
        else
        {
            int tt,t1=5;
            do
            {
                tt=n-t1;
                if(tt%3==0)
                {
                    temp=tt;
                    temp1=t1;
                    break;
                }
                t1+=5;
            }while(tt>=-5);
        }
        if(temp<0 || temp1<0)
            cout<<-1;
        else
        {
        for(int i=1;i<=temp;i++)
            cout<<5;
        for(int i=1;i<=temp1;i++)
            cout<<3;
        }
        cout<<endl;
    }
    return 0;
}
