#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
long long int dp[506][105];
int a[505];
int main()
{
//	freopen("input.txt","r",stdin);
        // freopen("out.txt","w",stdout);
	int n,i,j,k;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);

	for(i=0;i<n;i++)
	{
		dp[i][a[i]]++;
		for(j=0;j<i;j++)
		{
			if(a[j]<a[i])
			{
				for(k=1;k<=102;k++)
				{
                   dp[i][__gcd(a[i],k)]=(dp[i][__gcd(a[i],k)]+dp[j][k])%mod;
				}
			}
		}
	}
	long long int ans=0;
	for(i=0;i<n;i++)
		ans=(ans+dp[i][1])%mod;
	cout<<ans;
}
