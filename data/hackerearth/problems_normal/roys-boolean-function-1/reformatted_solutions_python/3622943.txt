T = int(raw_input())


def prime(num):
    if num == 1:
        return 0
    for i in range(2, num):
        if num % i == 0:
            return 0
    return 1


def gcd(p, q):
    while (q != 0):
        temp = q
        q = p % q
        p = temp
    return p


for t in range(T):
    m = 1
    N = int(raw_input())
    for j in range(2, N):
        if gcd(N, j) == 1:
            m += 1
    if prime(m):
        print "TRUE"
    else:
        print "FALSE"
