#include <iostream>
#include <iomanip>
using namespace std;
static int a[1000001],count1[1000001];
void prime(int n)
{
	int i=2,k=0;
	count1[0]=0;
	count1[1]=0;
	for (i=2;i<=n;i++) a[i]=0;
	i=2;
	while (i<=n)
	{
		if (a[i]==0)
		{
			count1[i]=count1[i-1]+1;
			if (i>10000)
			{
				i+=1;
				continue;
			}			
			int j=i*i;
			while (j<=n)
			{
				a[j]=1;
				j+=i;
			}
		}
		else
		{
			count1[i]=count1[i-1];
		}
		i+=1;
	}
}
int main()
{
	int t,l,r,pr;
	prime(1000000);
	cin>>t;
	while (t--)
	{
		cin>>l>>r;
		pr=count1[r]-count1[l-1];
		cout<<fixed<<setprecision(6);
		cout<<(float)pr/(r-l+1)<<endl;
	}	
	return 0;
}
