t = int(raw_input())
for _ in range(0, t):
    s = raw_input()
    num = 0
    for i in s:
        if (i.isdigit()):
            num += int(i)
    result = ''.join([i for i in s if not i.isdigit()])
    gg = sorted(result.split())
    for i in range(0, len(gg)):
        if (num - len(gg[i]) > 0):
            print gg[i], num - len(gg[i]),
        else:
            print gg[i], -1*(num - len(gg[i])),
        num -= len(gg[i])
    print ""
