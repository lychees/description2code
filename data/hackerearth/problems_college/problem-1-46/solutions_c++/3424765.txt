#include <bits/stdc++.h>

using namespace std;

vector<map<char, int> > v;

int main()
{
    int x, n, l, r;

    scanf("%d", &x);

    string str;

    cin >> str;

    map<char, int> m;
    v.push_back(m);

    for(int i = 0; i < str.size(); i++) {
        m[str[i]]++;
        v.push_back(m);
    }

    scanf("%d", &n);

    while(n--) {

        int cnt = 0;

        scanf("%d", &l);
        scanf("%d", &r);
			
		if(l == r) {
			printf("NO\n");
			continue;
		}	
			
        for(char ch = 'a'; ch <= 'z'; ch++) {
            if( (v[r][ch] - v[l-1][ch]) % 2 == 1) {
                cnt++;
              //  cout <<ch<<endl;
            }
        }
      //  cout <<" "<<cnt <<endl;
        if(cnt > 2)
            printf("NO\n");
        else
            printf("YES\n");

    }
}
