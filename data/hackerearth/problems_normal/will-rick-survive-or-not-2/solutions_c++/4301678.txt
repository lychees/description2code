#include<bits/stdc++.h>
using namespace std;
int main()
{	
	ios::sync_with_stdio(false);
	int i,j,n,t,f=0,c;
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		vector<int>num(n);
		for(i=0;i<n;i++)
		cin>>num[i];
		f=0,c=0;
		sort(num.begin(),num.end());
		for(i=0;i<n;i++)
		{
			if(i%6==0&&i!=0)
			{
				c++;
				if(num[i]-c<=0)
				{
				f=1;
				//cout<<num[i]<<endl;
				break;	
				}
				c++;
			}
			else
			{
				if(num[i]-c<=0)
				{
					f=1;
					//cout<<num[i]<<endl;
					//cout<<i<<endl;
					break;
				}
				c++;
			}
		}
		if(f==1)
		{
		cout<<"Goodbye Rick"<<endl;
		cout<<i<<endl;
		}
		else
		cout<<"Rick now go and save Carl and Judas"<<endl;
		num.clear();		
	}
	
}
