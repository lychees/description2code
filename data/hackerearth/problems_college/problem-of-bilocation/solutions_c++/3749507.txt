#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c,d,s;
		cin>>a>>b>>c>>d>>s;
		
		if(c-s<= b && c-s>=a)
		{
			cout<<c-s<<"\n";
		}
		else
		if(c-s>=b && c-s>=a){
			cout<<b<<"\n";
		}
		else
		{
			cout<<"STAY IN LONDON"<<"\n";
		}
	}
}
