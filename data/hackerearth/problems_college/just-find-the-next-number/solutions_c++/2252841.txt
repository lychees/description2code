#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    long long int t,i,n,j,pos,x,num;
    string bin;
    cin>>t;
    for(i=0;i<t;i++)
    {
        pos=0;
        cin>>n;
        num=n;
        bin.clear();
        while(n>0)
        {
            bin.push_back(n%2+48);
            n=n/2;
        }
       //out<<"hello "<<bin;
        pos=bin.find("11");
        if(pos==string::npos)
        {
            cout<<num<<endl;
        }
        else
        {
            //cout<<"else ";
            for(j=num+1;;j++)
            {
                bin.clear();
                x=j;
                pos=0;
                while(x>0)
                {
                    bin.push_back((x%2)+48);
                    x=x/2;

                }
                //cout<<"bin ";
                pos=bin.find("11");
                if(pos==string::npos)
                {
                    cout<<j<<endl;
                    break;
                }

            }
        }


    }
    return 0;
}
