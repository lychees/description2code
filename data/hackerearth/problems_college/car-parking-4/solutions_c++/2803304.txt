#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e6+123;

char ar[maxn];

int main(){
	
	int t;
	
	scanf("%d",&t);
	
	while( t-- ){
		scanf("%s",ar+1);
		int n = strlen(ar + 1);
		int ans = 0;
		
		for( int i=1 ; i <= n ; i++ ){
			bool flag = ar[i] == '-';
			if( i-1 >= 1 and ar[i-1] == 'S' )
				flag = 0;
			if( i+1 <= n and (ar[i+1] == 'S' or ar[i+1] == 'B') )
				flag = 0;
			if( i+2 <= n and ar[i+2] == 'B' )
				flag = 0;
			ans += flag;
		}
		
		printf("%d\n",ans);
	}
	
	return 0;
}
