#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int mo=1000000007;
void mul(long long int a[][55],long long int b[][55],long long int c[][55],int n)
{
	int i,j,k;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			c[i][j]=0;
			for(k=0;k<n;k++)
			{
				c[i][j]+=a[i][k]*b[k][j];
				c[i][j]%=mo;
			}
		}
	}
}
void poww(long long int a[][55],long long int b,long long int ans[][55],int n)
{
	int i,j;
	if(b==0)
	{
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if(i==j)
				{
					ans[i][j]=1;
				}
				else
				{
					ans[i][j]=0;
				}
			}
		}
		return;
	}
	long long int tmp[55][55];
	poww(a,b/2,tmp,n);
	if(b%2==0)
	{
		mul(tmp,tmp,ans,n);
	}
	else
	{
		mul(tmp,tmp,ans,n);
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				tmp[i][j]=ans[i][j];
			}
		}
		mul(a,tmp,ans,n);
	}
}
int main()
{
    int t,i,j,n,m;
    cin>>t;
    long long int a[55][55],ans[55][55];
    while(t--)
    {
    	cin>>n>>m;
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			cin>>a[i][j];
    		}
    	}
    	poww(a,m,ans,n);
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<n;j++)
    		{
    			cout<<ans[i][j]<<" ";
    		}
    		cout<<endl;
    	}   	
    }
    return 0;
}
