primes = []

n = int(raw_input())
ans = 0


def isprime(x):
    i = 2
    while i*i <= x:
        if x % i == 0:
            return 0
        i += 1
    return 1


for i in range(2, n+1):
    if isprime(i):
        primes.append([i, 1])
    else:
        for j in primes:
            if i % j[0] == 0:
                j[1] += 1
    temp = 1
    for j in primes:
        temp *= 2
    ans += temp

print ans/2
