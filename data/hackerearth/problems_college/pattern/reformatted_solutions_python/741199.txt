import sys

n = int(raw_input())

for i in range(0, n):
    if ((i == 0) or (i == n-1)):
        for j in range(0, n):
            sys.stdout.write('*'),
        print ''
    if ((i > 0) and (i < n-1)):
        for j in range(1, n-i):
            print '',
        print '*'
