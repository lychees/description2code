#include <iostream>
#include <math.h>
using namespace std;

int main()
{
        long n,d,r;
        int t;
        
        cin>>t;
        
        while(t--)
        
{
        
    	cin>>n>>d>>r;
    	
    	long ans=0;
    	bool start = 0;
  		while(r)
    	{
         ans=0;       
         if(n>d)
         {
         ans+=n/d;
         n=fmod(n,d);
         start=1;
         }             
         
         n*=10;
        
        if(start)
                 r--;
                 
        start=1;
        }
        
        cout<<ans<<endl;
}

}
