#include<iostream>
#include<cmath>
using namespace std;
#define ll long long
const ll mod=1000000007;
ll sum(ll l,ll r){
	l--;
	ll tmp1=l*(l+1)/2;
	tmp1%=mod;
	ll tmp2=r*(r+1)/2;
	tmp2%=mod;
	return ((tmp2-tmp1)+mod)%mod;
}
int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	int t;
	cin>>t;
	while(t--){
		ll n;
		cin>>n;
		ll ans=0;
		ll l,r,tmp;
		l=1;
		while(l<=n){
		    tmp=n/l;
			r=n/tmp;
			ans=(ans+(tmp*sum(l,r))%mod)%mod;
			l=r+1;
	    	//if(tmp==1)break;
		}
		cout<<ans<<'\n';
	}
    return 0;
}
