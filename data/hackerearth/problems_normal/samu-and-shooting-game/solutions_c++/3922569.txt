#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x, y, n, w, p1, p2;
		cin>>x;
		cin>>y;
		cin>>n;
		cin>>w;
		cin>>p1;
		cin>>p2;
		double a[1009+1][1009+1];
		memset(a, 0, sizeof(a));
		int i, j;
		
		for(i=1;i<=w;i++)
		{
			a[0][i]=0;
		}
		for(i=0;i<=n;i++)
		{
			a[i][0]=1;
		}
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=w;j++)
			{
				a[i][j] = max(p1*0.01*a[i-1][max(j-x,0)] + (100-p1)*0.01*a[i-1][max(j,0)],0.01*p2*a[i-1][max(j-y,0)] + (100-p2)*0.01*a[i-1][max(j,0)]); 
			}
		}
		printf("%.6f\n",a[n][w]*100);
	}
	return 0;
}
