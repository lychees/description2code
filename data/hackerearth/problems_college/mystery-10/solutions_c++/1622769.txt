#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
main()
{
	int T,i,l;
	char S[105],TE[27]="qwertyuiopasdfghjklzxcvbnm";
	cin>>T;
	cin.ignore();
	while(T--)
	{
		gets(S);
		l=strlen(S);
		for(i=0;i<l;i++)
		{
			if(S[i]==' ')
			{
				cout<<" ";
			}
			else
			cout<<TE[S[i]-'a'];
		}
		cout<<"\n";
	}	
}
