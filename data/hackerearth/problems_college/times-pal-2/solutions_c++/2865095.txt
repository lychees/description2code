#include <bits/stdc++.h>
using namespace std;

int main()
{
	cin.sync_with_stdio(false);
	cin.tie(false);
    string a, b;
    int t;
    
    cin >> t;
    
    while(t--)
    {
    	int count = 0;
    	cin >> a; 
    	cin >> b;
    	for(int i = 0; i <= a.length(); i++)
    	{
    		
    		string c = a;
    		c.insert(i, b);
    	//	cout << c << '\n';
    		string d = c;
    		reverse(c.begin(), c.end());
    		if(d == c)
    			count++;
    	}
    	cout << count << '\n';
    }
    
   // cout << "Hello World!" << endl;
    return 0;
}
