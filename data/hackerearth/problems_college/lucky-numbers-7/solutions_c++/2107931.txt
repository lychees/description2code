bool issame(int a,int b)
{
    int c=0,d=0;
    while(a)
    {
        int p=a%10; if (p==4 || p==7) c+=1;
        a/=10;
    }
    while(b)
    {
        int p=b%10; if (p==4 || p==7) d+=1;
        b/=10;
    }
    if (c==d)
    {
        return true;
    } else return false;
}

int main()
{
    int a,l; cin>>a>>l; int ans=a+10;
    for (int i=a+1; i<a+50000; i++)
    {
        int b=i; int flag=0;
        for (int j=a; j<a+l; j++)
        {
            if (!issame(j,b))
            {
                flag=1;
                break;
            }
            b++;

        }
        if (flag==0)
        {
            ans=i;
            break;
        }
    }
    cout<<ans<<endl;



}
