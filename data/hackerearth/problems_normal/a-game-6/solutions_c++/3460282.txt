#include <iostream>
using namespace std;
typedef long long LL;

int is_prime(int j)
{
	if(j < 2)
		return 0;
	else if(j == 2)
		return 1;
	else if(j%2 == 0)
		return 0;
	for(int i = 3; 1LL*i*i <= j; i++)
        if(j%i == 0)
            return 0;
    return 1;
	
}

int main()
{
	int p,q,s,t = 0;
	cin >>t;
	
	while (t > 0)
	{
		cin >> p>>q>>s;
		if(p%2 == 0)	
			cout << "Havanese" << endl;
		else
		{
			p %=2;
			if(s >= q)
				cout << "Havanese" << endl;
			else
			{
				int win=0;
				if(q%s == 0)
					win=1;
				else
				{
					for(int div=2;div <= 40000 && win == 0; div++)
					{
						if(q%div) 
							continue;
                        int f = q/div;
                        
                        if(f < s) 
                        	continue;
                        if(f < (2*s)) 
                        	win = 1;
                        if(is_prime(f))
                        	win = 1;
					}
				}
				if(win)
					cout << "Bernard" << endl;
				else
					cout << "Havanese" << endl;
			}
		}
		t--;
	}
    return 0;
}
