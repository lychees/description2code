#include <iostream>
#include <algorithm>
using namespace std;

unsigned count_bits(unsigned num) {

  int count = 0;

  while(num != 0) {
    count++;
    num &= num - 1;
  }

  return count;
}

int main() {

  int n, t, num;
  cin >> t;

  for(int testCase = 0; testCase < t; testCase++) {

    cin >> n;
    unsigned dist = -1;
    for(int i = 0; i < n; i++) {
      cin >> num;
      dist = min(dist, count_bits(num));
    }

    cout << dist << endl;
  }
  return 0;
}
