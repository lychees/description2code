#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	int n = s.length();
    	int ch[n];
    	int rCn = 0;
    	int kCn = 0;
    	for(int i = 0;i < n;i++) {
    		if(s[i] == 'R') {
    			rCn++;
    		}else {
    			kCn++;
    		}
    	}
    	int r = 0;
    	int k = 0;
    	int maxi = 0;
    	for(int i=0;i<n;i++) {
    		if(s[i] == 'R'){
    			r++;
    			maxi = max( 2 * min(r, kCn - k), maxi);
    		}else {
    			maxi = max( 2 * min(r, kCn - k),maxi);
    			k++;
    		}
    	}
    	cout<<maxi<<endl;
    }
    return 0;
}
