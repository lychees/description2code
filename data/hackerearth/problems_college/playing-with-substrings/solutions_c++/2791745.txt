bool check(string s, string t)
{
	for (int i = 0; i + t.length() - 1 < s.length(); i++)
	{
		if (s.substr(i, t.length()) == t) return true;
	}
	return false;
}
int main()
{
	string s, t;
	cin >> s >> t;
	set<string>go;
	for (int i = 0; i < s.length(); i++)
	{
		string tmp;
		for (int j = i; j < s.length(); j++)
		{
			tmp += s[j];
			if (check(tmp, t)) go.insert(tmp);
		}
	}
	printf("%d\n", go.size());
}
