#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int a[27];
int main()
{
    long long int t,max,i,n;
    char b[10005];
    scanf("%lld",&t);
    while(t--)
    {
    	max=0;
    	memset(a,0,sizeof(a));
    	scanf("%s",b);
    	n=strlen(b);
    	for(i=0;i<n;i++)
    	{
    		a[b[i]-97]++;
    		if(max<a[b[i]-97])
    		{
    			max=a[b[i]-97];
    		}
    	}
    	printf("%lld\n",max);
    }
    return 0;
}
