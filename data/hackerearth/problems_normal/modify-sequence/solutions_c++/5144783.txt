#include <iostream>
using namespace std;

int main()
{
    long int n,min;
    cin>>n;
    long int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    if(n==1)
    if(a[0]==0)
    cout<<"YES\n";
    else
    cout<<"NO\n";
    else
    if(n==2)
    if(a[0]-a[1]==0)
    cout<<"YES\n";
    else
    cout<<"NO\n";
    else
    {
    	min=a[0];
    	for(int i=1;i<n;i++)
    	{
    		min=a[i]-min;
    	}
    	if(min==0)
    	cout<<"YES\n";
    	else
    	cout<<"NO\n";
    }
    return 0;
}
