#include <iostream>
#include <string.h>
using namespace std;

int main() {
	
	int t,i,j,k;
	char str[100000];
	long long int count;
	scanf("%d",&t);
	
	for(i=0;i<t;i++)
	{
		cin>>str;
		count=0;
		j=1;
		k=0;
		while(str[j]!='\0')
		{
			if(str[k]==str[j])
				{
					count++;
					j++;
					
				}
		
			else
				{
					k=j;
					j++;
				}
		}
		printf("%lld\n",count);
	}
	
	
	return 0;
}
