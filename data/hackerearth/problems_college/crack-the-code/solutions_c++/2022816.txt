#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    vector<int>vec;
    vec.resize(26);
    vec[0]=0,vec[1]=1;
    for(int i=2;i<26;i++)
    {
    	vec[i]=vec[i-1]+vec[i-2];
    }
    while(t--)
    {
    	int a,b,count=0;
    	scanf("%d%d",&a,&b);
    	for(int i=0;i<26;i++)
    	{
    		if(a<=vec[i]&&vec[i]<=b)
    		{
    			count++;
    		}
    	}
    	if(count%2==0)
    	{
    		printf("VALID\n");
    	}
    	else
    	{
    		printf("INVALID\n");
    	}
    }
}
