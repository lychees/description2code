#include <bits/stdc++.h>

using namespace std;
int a[100007];
int ct[100007];
int main()
{
    ct[0]=0;
    a[0]=0;
    for(int i=1;i<100005;i++)
    {
        ct[i]=__builtin_popcount(i);
        if(ct[i]>=ct[a[i-1]]) a[i]=i;
        else a[i]=a[i-1];
    }
    //for(int i=1;i<20;i++)cout<<a[i]<<endl;
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        cout<<a[n]<<endl;
    }
    return 0;
}
