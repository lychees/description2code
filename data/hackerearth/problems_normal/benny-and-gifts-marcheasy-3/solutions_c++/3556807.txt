#include <iostream>
#include <cstring>
#include <vector>
#include <queue>
#include <stack>
#include <unordered_map>
#include <cmath>
#include <map>

#define lop(i, n) for (int (i) = 0; (i) < (n); (i)++)
#define lopi(i, s, e) for (int (i) = s; (i) <= (e); (i)++)

using namespace std;

map<pair<int, int>, int> umap;

char inp[100001];

int main() {
    int x = 0;
    int y = 0;
    int count = 0;
    
    scanf("%s", inp);
    int len = strlen(inp);
    
    umap[make_pair(0, 0)] = 1;
    
    lop(i, len) {
        switch (inp[i]) {
                case 'U':
                y++;
                break;
                
                case 'D':
                y--;
                break;
                
                case 'L':
                x--;
                break;
                
                case 'R':
                x++;
                break;
        }
        //string s = getstring(x, y);
        pair<int, int> p = make_pair(x, y);
        umap[p]++;
        if (umap[p] > 1) count++;
        //cout<<" "<<s<<" "<<mmap[s]<<endl;
        //ch = getchar();
    }
    
    cout<<count;
}
