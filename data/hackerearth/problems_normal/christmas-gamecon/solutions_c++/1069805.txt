#include <iostream>
using namespace std;
int a[100002];
int main()
{
    int t,n,i,ans;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		a[i]%=5;
    	}
    	ans=0;
    	
    	for(i=0;i<n;i++)
    	ans^=a[i];
    	
    	if(ans)
    	cout<<"Aniruddha\n";
    	else
    	cout<<"Andrew\n";
    	
    }
    return 0;
}
