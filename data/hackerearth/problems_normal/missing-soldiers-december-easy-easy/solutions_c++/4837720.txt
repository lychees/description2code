#include <iostream>
using namespace std;

int main()
{
    int n;
    cin >> n;
    long long l = 1000000000;
    long long h = 0;
    for(int i=0; i<n; i++)
    {
    	long long x,y,d;
    	cin >> x >> y >> d;
    	long long x1 = x + d;
    	if( x1 > h )
    		h = x1;
    	if( x < l)
    		l = x;
    }
    if(h-l == 11)
    	cout << h-l << endl;
    else
    	cout << h-l+1 << endl;
    return 0;
}
