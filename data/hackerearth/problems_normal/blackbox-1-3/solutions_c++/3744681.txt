#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
	int t;
	cin>>t;
	while(t--){
		ll a,b,c;
		cin>>a>>b>>c;
		ll hcf=__gcd(a,__gcd(b,c));
		if(hcf==1){
			vector<ll> v;
			v.push_back(a);
			v.push_back(b);
			v.push_back(c);
			sort(v.begin(),v.end());
			if((v[0]*v[0]+v[1]*v[1])==v[2]*v[2])
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
		}else{
			cout<<"NO"<<endl;
		}
	}
	return 0;
}
