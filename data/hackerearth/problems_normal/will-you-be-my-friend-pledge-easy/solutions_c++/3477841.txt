#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(a==0)
	return b;
	
	return gcd(b%a,a);
}

int main()
{
	int a,n;cin>>a>>n;
	int count=0,ar[n];	
	for(int i=0;i<n;i++)
	{
		cin>>ar[i];
		int d=ar[i];
		if(gcd(a,d)==1)
		count++;
	}	
	
	cout<<n-count;
	
    //cout << "Hello World!" << endl;
    return 0;
}
