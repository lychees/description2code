#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int i,f,t;
    char a[50];
    cin>>t;
    while(t!=0)
    {
    	cin>>f;
    	cin>>a;
    	for(i=0;a[i]!=NULL;i++)
    	{
    		if((a[i]-f)>=65)
    			a[i]=a[i]-f;
    		else
    		{
    			a[i]=90-(f-(a[i]-64));
    		}
    	}
    	cout<<a;
    	cout<<endl;
    	t--;
    }
    return 0;
}
