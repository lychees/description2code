int main()
{
    ios_base::sync_with_stdio(false);cin.tie(0);

    int test=1,i,j,n,cnt;
    string patrn,text;

    cin>>test;
    while(test--){
        int hashP[26],hashT[26],correct[26];
        SetA(hashP,0);
        SetA(hashT,0);
        SetA(correct,0);
        cnt=0;
        cin>>patrn>>text;
        For(i,0,patrn.length()-1){
            hashP[patrn[i]-'a']++;
            hashT[text[i]-'a']++;
        }
        For(i,0,25)
            if(hashP[i]==hashT[i]){
                correct[i]=1;
                cnt++;
            }
        i=0;
        j=patrn.length();
        while(cnt!=26 && j<text.length()){
            if(--hashT[text[i]-'a']==hashP[text[i]-'a']){
                correct[text[i]-'a']=1;
                cnt++;
            }
            else if(correct[text[i]-'a']){
                cnt--;
                correct[text[i]-'a']=0;
            }
            if(++hashT[text[j]-'a']==hashP[text[j]-'a']){
                correct[text[j]-'a']=1;
                cnt++;
            }
            else if(correct[text[j]-'a']){
                cnt--;
                correct[text[j]-'a']=0;
            }

            i++;
            j++;
        }
        if(cnt==26)
        cout<<"YES"<<endl;
        else
        cout<<"NO"<<endl;
    }
//end of test
    return 0;
}
