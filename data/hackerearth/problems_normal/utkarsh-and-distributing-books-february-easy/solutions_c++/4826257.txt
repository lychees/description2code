#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
	int t,no,*books,i;
	int uCount,sCount;
	cin>>t;
	while(t--)
	{
		uCount = INT_MAX;
		sCount = 0;
		cin >> no;
		books = new int[no];
		for(i=0;i<no;i++)
		{	
			cin>>books[i];
		}
		
		for(i=0;i<no;i++)
		{
			if(books[i]-1 < uCount)
				uCount = books[i]-1;
				
			sCount += books[i]-2 + 1;
		}
		
		cout<<uCount<<" "<<sCount<<endl;
	}
    return 0;
}
