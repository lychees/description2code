using namespace std;

int main()
{

	{

 			 {
				  cout<<"YES. It is a equilateral triangle.\n";
 			 }

 			 {

 			 	if(a==b)
 			 	{
 			 		if((a+c)>a&&2*a>c)
 			 			f=1;
 			 		else
 			 			f=0;
 			 	}
 			 	else if(b==c)
				{
 			 		if((b+a)>b&&2*b>a)
 			 			f=1;
 			 		else
 			 			f=0;
 			 	}
 			 	else
 			 	{
 			 		if((c+b)>c&&2*c>b)
 			 			f=1;
 			 		else
 			 			f=0;
 			 	}
 			 	if(f==1)
				  cout<<"YES. It is a isosceles triangle.\n";
				else
					cout<<"NO\n";
 			 }
 			else if(((a+b)>c)&&((b+c)>a)&&((c+a)>b))
		
 				 cout<<"YES. It is a scalene triangle.\n";
 		
  			else
   				cout<<"NO\n";
	}
return 0;
}
