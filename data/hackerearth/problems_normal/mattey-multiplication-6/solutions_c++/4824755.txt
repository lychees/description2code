#include <bits/stdc++.h>
using namespace std;

int main()
{ 
	long long t,n,m,flag,i,a[1000000],l;
    scanf("%lld",&t);
    while(t--){
    	scanf("%lld %lld",&n,&m);
    	flag=0;l=0;
    while(m!=0){
    	a[l++]=m%2;
    	m=m/2;
    }
    for(i=l-1;i>=0;i--){
    	if(a[i]==1&&flag==0){
    		printf("(%lld<<%lld)",n,i);
    	
    		flag=1;
    	}
    	else if(a[i]==1&&flag==1)
    	printf(" + (%lld<<%lld)",n,i);
    }
    	printf("\n");
    }
    return 0;
}
