using namespace std;
map<string, int> fi, fo, n;
string sub;
int a, b, f, F;
int main() {
  while (cin >> sub) {
    F = 0;
    while (1) {
      int x = sub.find_last_of('\\');
      if (x == 2)
        break;
      sub = sub.substr(0, x);
      int f = fo[sub];
      fi[sub] += F;
      fo[sub]++;
      if (fi[sub] > a)
        a = fi[sub];
      if (fo[sub] > b)
        b = fo[sub];
      if (!f)
        F++;
    }
  }
  cout << a << ' ' << b;
  return 0;
}
