#include<iostream>
#include<cstdio>
using namespace std;

int ar[10000000+7],ans[10000000+7];
void gen()
{
ar[0]=1;
ar[1]=1;
for(int i=2;i<=10000000;i++)
ar[i]=i;
for(int i=2;i<=10000000;i++)
if(ar[i]!=1)
{
for(int j=i*2;j<=10000000;j+=i)
{
ans[j]++;
ar[j]=ar[j]/ar[i];
}
ans[i]++;
}
for(int i=1;i<=10000000;i++)
ans[i]=ans[i-1]+ans[i];
}

int main()
{
//ios_base::sync_with_stdio(false);
//freopen("in5.txt","r",stdin);
//freopen("out5.txt","w",stdout);
gen();
int t;
scanf("%d",&t);
while(t--)
{
int a,b;
scanf("%d",&a,&b);
printf("%d\n",ans[a]);
}
return 0;
}
