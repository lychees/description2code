#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,n,i,maxh,h;
    int *p;
    scanf("%d",&t);
    while(t--)
    {
    	maxh=0;
    	h=0;
    	scanf("%d",&n);
    	p=new int[n];
    	for(i=0;i<n;i++)
    	scanf("%d",&p[i]);
    	
    	for(i=0;i<n;i++)
    	{
    		h+=p[i];
    		if(h>maxh)
    		maxh=h;
    		if(h<0)
    		h=0;
    	}
    	
    	printf("%d\n",maxh);
    	
    }
    return 0;
}
