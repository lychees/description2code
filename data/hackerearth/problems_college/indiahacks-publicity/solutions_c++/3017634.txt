#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
int visited[MAX];
void fun(vector<int>p[],int i)
{
	vector<int>::iterator it;
	visited[i]=1;
	it=p[i].begin();
	while(it!=p[i].end())
	{
		if(visited[*it]==0)
			fun(p,*it);
		it++;
	}
}

int main()
{
	int i,j,count,k,m,n,t,e,a,b,l;
	si(t);
	while(t--)
	{
		si(n);
		vector<int>p[n];
		si(e);
		f(i,0,e)
		{
			si(a);
			si(b);
			p[a].pb(b);
			p[b].pb(a);
		}
		f(i,0,n)
			visited[i]=0;
		count=0;
		f(i,0,n)
		{
			if(visited[i]==0)
			{
				fun(p,i);
				count++;
			}
		}
		printf("%d\n",count);
	}
}
