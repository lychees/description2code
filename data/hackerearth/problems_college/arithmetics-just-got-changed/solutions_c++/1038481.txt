#include <iostream>
using namespace std;
int rev(long long int n){
	long long temp=0;
	while(n){
		temp=temp*10+n%10;
		n=n/10;
	}
	return temp;
}
int main()
{
	int n;
	cin>>n;
	while(n--){
		long long int i1,i2;
		scanf("%lld %lld",&i1,&i2);
		i1=rev(i1);
		i2=rev(i2);
		long long ans=i1+i2;
		printf("%lld\n",rev(ans));
	}
    return 0;
}
