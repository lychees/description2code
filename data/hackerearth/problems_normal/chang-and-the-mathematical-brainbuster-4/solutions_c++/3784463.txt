#include <bits/stdc++.h>
using namespace std;
int arr[10010];
int n;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    int j,ans=1e9;
    for(int i=n-1;i>=0;i--)
    {
    	int flag=0;
    	for(j=0;j<i;j++)
    	{
    		if(arr[i]-3*arr[j]<=0)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag)
    	ans=min(ans,n-(i-j+1));
    }
    cout<<ans<<"\n";
    return 0;
}
