#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while(t--)
    {
    	string s,p;
    	cin >> s >> p ;
    	
    	int j=0;
    	for(int i=0;i<s.length() && j<p.length();i++)
    	{
    		if(s.at(i)==p.at(j))
    			j++;
    	}
    	
    	if(j==p.length())
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
}
