#include <iostream>
using namespace std;

int main()
{
	int n,i;
	int count[5];
	char *a;
	cin>>n;
	a = (char *)malloc(sizeof(char)*n);
	
	for(i=0;i<n;i++)
		cin>>a[i];
		
	for(i=0;i<n;i++)
	{
		if(a[i]=='a'){
			count[0]=1;
		}else if(a[i]=='e'){
			count[1]=1;
		}else if(a[i]=='i'){
			count[2]=1;
		}else if(a[i]=='o'){
			count[3]=1;
		}else if(a[i]=='u'){
			count[4]=1;
		}		
	}
	
	int flag=1;
	for(i=0;i<5;i++)
	{
		if(count[i]!=1)
		{
			flag=0;
			break;
		}
	}
	
	if(flag==1)
		cout<<"YES"<<endl;
	else
		cout<<"NO"<<endl;
		
	return 0;
}
