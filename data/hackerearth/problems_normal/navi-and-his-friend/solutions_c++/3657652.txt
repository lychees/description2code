#include <bits/stdc++.h>
using namespace std;
int wt[100];
int price[100];
int main()
{
	int t;
	int n;
	scanf("%d",&t);
	long long int wtmax,cmax;
	for(int test=1;test<=t;test++)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&price[i],&wt[i]);
		}
		cin>>wtmax>>cmax;
		long long int ans=-1;
		for(int i=0;i<pow(2,n);i++)
		{
			int c=0;
			long long int cost=0;
			long long int weight=0;
			for(int j=0;j<n;j++)
			{
				if(i&(1<<j))
				{
					cost=cost+price[j];
					c++;
					weight=weight+wt[j];
				}
			}
			if(c>0&&c<=cmax)
			{
				if(weight<=wtmax)
				ans=max(ans,cost);
			}
		}
		printf("For Day #%d:\n%d\n",test,ans);
		
	}
	return 0;
}
