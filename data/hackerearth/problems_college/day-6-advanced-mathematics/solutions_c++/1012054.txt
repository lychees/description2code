#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	string str;
	bool flag;
	while(t){
	cin >> str;
	flag = true;
	for(int i=0; i<str.length()-1; i++){
		if(str[i]==str[i+1]){
			flag = false; 
			break; 
		}
	}
	if(flag) cout << "KISS\n";
	else cout << "SLAP\n";
	t--;
	}
    return 0;
}
