for i in range(int(raw_input(""))):

    a = set([])
    maxi = 0
    count = 0
    n = int(raw_input(""))
    for l in range(n):
        s, r = [int(k) for k in raw_input().split()]

        if r == 0:

            count = 0
        else:
            if s not in a:
                a.update(set([s]))
                count = count + 1
                if count >= maxi:
                    maxi = count
    print maxi
