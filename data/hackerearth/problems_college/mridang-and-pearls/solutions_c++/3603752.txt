#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    //cout << "Hello World!" << endl;
    ll n,num,pos=1,cnt=0;
    vector<ll> a;
    vector<ll> b;
    map<ll,ll> m;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cin>>num;
    	m[num]++;
    	if(m[num]==2)
    	{
    		cnt++;
    		a.push_back(pos);
    		b.push_back(i);
    		pos=i+1;
    		m.clear();
    	}
    }
    if(pos<(n+1) && cnt)
    {
    	b.pop_back();
    	b.push_back(n);
    }
    if(cnt)
    {
    	cout<<cnt<<endl;
    	for(int i=0;i<cnt;i++)
    	{
    		cout<<a[i]<<" "<<b[i]<<endl;
    	}
    }
    else cout<<"-1\n";
    return 0;
}
