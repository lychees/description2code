#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define MOD		1000000007
vector<LL> T(100001);
void getvector(vector<int> &n,int length){int m1;for(int i=0;i<length;i++){cin>>m1;n.push_back(m1);}}
int N,M;
int daysRequired(LL daily)
{
	int reqd = 1;
	LL currSum = 0;
	for(int i = 1; i <= N; i++)
	{
		if(currSum + T[i] > daily)
		{
			reqd += 1;
			currSum = T[i];
		}
		else
			currSum += T[i];
	}
	return (reqd <= M) ? 1 : 0;
}
int main()
{
	cin >> N >> M;
	LL lo = 1, hi = 1LL * MOD * MOD;
	for(int i = 1; i <= N; i++)
	{
		cin >> T[i];
		lo = max(lo , T[i]);
	}
	while(lo < hi)
	{
		LL mid = (lo + hi) >> 1;
		int p = daysRequired(mid);
		if(p)
			hi = mid;
		else
			lo = mid + 1;
	}
	cout << lo << "\n";
	return 0;
}
