#include <iostream>
#include <vector>
#include <string>
#include <string>
#include <cmath>
using namespace std;

int main() {
  int t;
  cin >> t;
  string fights[] = {"SL", "LB", "BS"};
  long num;
  int battle;
  while(t--) {
    battle = 1;
    cin >> num;
    while(battle < num) {
      num = num - battle;
      battle++;
    }
    cout << fights[battle%3] << endl;
  }

  return 0;
}
