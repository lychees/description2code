#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod = 1e6+7;
int prime[mod];
 
void sieve()
{
	for(int i=2;i<mod;i++)
	{
		if(prime[i]==0)
		{
			for(int j=i;j<mod;j+=i)
			{
				if(prime[j]==0)	prime[j]=i;
			}
		}
	}
}
 
int main()
{
   
   
    int t;
    scanf("%d",&t);
    sieve();
    
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	printf("%d\n",n-prime[n]);
    }
    return 0;
}
