#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
long long int arr[1000009];
long long int fact[100009];
long long int invfact[100009];
long long int mod_expo(long long int a,long long int b,long long int p)
{
   long long int ans=1;
   a%=p;
   while(b)
   {
      if(b&1)
      ans=(ans*a)%p;
      b>>=1;
      a=(a*a)%p;
   }
   return ans;
}
void pre_calculate()
{
   fact[1]=1;
   invfact[1]=1;
   for(int i=2;i<100009;i++)
   {
      fact[i]=(fact[i-1]*i)%MOD;
      invfact[i]=mod_expo(fact[i],MOD-2,MOD);
   }
}
int main()
{
   pre_calculate();
   int test;
   scanf("%d",&test);
   while(test--)
   {
      int n,m;
      scanf("%d%d",&n,&m);
      int arr[n];
      int tot=0;
      for(int i=0;i<n;i++)
      {
         scanf("%d",&arr[i]);
         tot+=arr[i];
      }
      if(tot>m)
         printf("0\n");
      else
      {
         long long int num=m-tot;
         num+=n-1;
         long long int den=n-1;
         long long int ans=1;
         for(long long int i=num;i>num-den;i--)
         {
            ans*=i;
            ans%=MOD;
         }
         ans*=invfact[den];
         ans%=MOD;
         printf("%lld\n",ans);
      }
   }
}
