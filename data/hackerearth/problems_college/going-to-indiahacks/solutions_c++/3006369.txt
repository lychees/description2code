#include<iostream>
#include<math.h> 
using namespace std;
int main()
{ 
    long t,i,j,dp[1005][1005],a,b,c,d;
    for(i=1;i<1005;i++)
    {
	dp[i][1]=1;
	dp[1][i]=1;
    }  
    for(i=2;i<1005;i++)
    	for(j=2;j<1005;j++)
    		dp[i][j]=(dp[i-1][j-1]+dp[i][j-1]+dp[i-1][j])%1000000007;
    cin>>t;
    while(t--)
    { 
    	cin>>a>>b>>c>>d;
    	cout<<dp[1+abs(a-c)][1+abs(b-d)]<<"\n";
    }
    return 0;
}
