#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
     cin>>a[i];
    sort(a,a+n);
    int ans = n;
    for(int i=0;i<n;i++)
     {
     	int k = upper_bound(a+i+1,a+n,3*a[i])-a;
     	ans = min(ans,n-k+i);
     }
    cout<<ans<<endl;
    return 0;
}
