#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    int lens;
    
    getline(cin, s);
    lens = s.length();
    
    int i, wt = 0;
    for (i = 0; i < lens; i++) {
    	wt += s[i];
    }
    wt /= lens;
    
    if (wt % 2 == 1) {
    	map<char, int> m;
    	for (i = 0; i < lens; i++) {
    		m[s[i]]++;
    	}
    	map<char, int>::iterator it;
    	char ans;
    	int maxi = INT_MIN;
    	for (it = m.begin(); it != m.end(); it++) {
    		if (it->second > maxi) {
    			maxi = it->second;
    			ans = it->first;
			}
		}
		cout << ans << endl;
    } else {
    	reverse(s.begin(), s.end());
    	cout << s << endl;
    }
    
    return 0;
}
