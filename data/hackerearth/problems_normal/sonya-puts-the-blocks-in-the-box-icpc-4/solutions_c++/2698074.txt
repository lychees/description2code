#define N 100005

template <class T,class U>inline void Max(T &a,U b){if(a<b)a=b;}
template <class T,class U>inline void Min(T &a,U b){if(a>b)a=b;}

int a[N];ll dp[1005],f[1005];
int main(){
    int T,i,j,k,n,m;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++){
        scanf("%d",&k);
        for(j=1;j<=k;j++)scanf("%d",&a[j]);
        for(j=1;j<=m;j++)dp[j]=0;
        for(j=1;j<=k;j++){
            ll s=0;
            for(int r=0;r<m;r++){
                if(j-r<0)break;
                s+=a[j-r];
                Max(dp[r+1],s);
            }
        }
        for(j=m-1;j>=0;j--)
            for(k=m-j;k>0;k--){
                Max(f[j+k],f[j]+dp[k]);
            }
    }
    ll ans=0;
    for(i=1;i<=m;i++)Max(ans,f[i]);
    printf("%lld\n",ans);
    return 0;
}
