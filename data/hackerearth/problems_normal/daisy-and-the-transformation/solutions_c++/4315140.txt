#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	 getchar();
    	 string str,str2;
    	 cin>>str>>str2;
    	 int n;
    	 cin>>n;
    	 int i,cnt=0,cntr=0;
    	 for(i=0;i<str.size();i++) {
    	 	if(str[i]=='M' && str2[i]=='D') cnt++;
    	 	else if(str[i]=='D' && str2[i]=='M') cntr++;
    	 }
    	 if((cnt+cntr)==n ||(cnt+cntr<n && (n-cnt-cntr)%2==0)) cout<<"Yes\n";
    	 else cout<<"No\n";
    }
    return 0;
}
