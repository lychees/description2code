#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int t,tp,tc,min,p,c,var;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&tc,&tp);
        if(tc==1)
         printf("0\n");
         else
        {
        min=1;p=1;c=1;
        long long int tempp=p;long long int tempc=c;
        while(p+tempc<=tp && c+tempp<=tc)
        {
            p+=tempc;
            c+=tempp;
            min++;
            tempp=p;
            tempc=c;
        }

        if(c==tc)
        {
           printf("%lld\n",min);
        }
        else
        {
            if(p+tempc>tp && c+tempp>tc)
            {
                p=tp;
                c=tc;
                min++;
                printf("%lld\n",min);
            }
            else if(p+tempc>tp)
            {
                p=tp;
                c+=tempp;
                min++;
                var=tc-c;

                if(var%p==0)
                    min+=var/p;
                else
                    min+=var/p + 1;
                 printf("%lld\n",min);

            }
            else if(c+tempp>tc)
            {
                c=tc;
                p+=tempc;
                min++;
                 printf("%lld\n",min);
            }
        }
        }
    }
    return 0;
}
