#include<stdio.h>
#include<string>
#include<iostream>
#include<algorithm>
#define ll long long
#define MAX 1000005
using namespace std;
long long int  mod = 1000000007;
long long int fact[MAX];
long long int inv_fact[MAX];
long long int pow(long long int  b,long long int  n)
{
if (n==1)
return b;
else if(n==0)
return 1;
ll x=pow(b,n>>1);
if(n&1)
return ((b*x)%mod)*x%mod;
else
return (x*x)%mod;
}
 
long long int inv(long long int x)
{
 
    return pow(x,(mod-2))%mod;
}
int main()
{
 
fact[0]=1;

for(long long int  i=1;i<MAX;i++)
fact[i]=(i*fact[i-1])%mod;
int t;
scanf("%d",&t);
while(t--)
{
    string a;
    cin>>a;
int n =a.length();
int i=0;
long long int scount[26]={0};
long long int c=0,v=0;
for(i=0;i<n;i++)
{
    if(a[i]=='a'||a[i]=='e'||a[i]=='i'||a[i]=='o'||a[i]=='u')
        v++;
    else
        c++;
    scount[a[i]-97]++;
}

if(c+1<v)
    cout<<"-1"<<endl;
else
{

    long long int temp=(fact[c]*fact[c+1])%mod;
    temp= (temp*inv(fact[c+1-v]))%mod;
    for(i=0;i<26;i++)
    {
        temp=(temp*inv(fact[scount[i]]))%mod;
   }
    cout<<temp<<endl;
}
 
}
 
return 0;
}
