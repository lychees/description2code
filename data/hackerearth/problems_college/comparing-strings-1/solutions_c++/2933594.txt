#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t, found, n1, n2;
    char s1[10000], s2[10000];
    cin>>t;
    while(t--)
    {
    	cin>>s1>>s2;
    	n1=strlen(s1);  //hello
    	n2=strlen(s2);  //world
    	found=0;
    	for(int i=0; i<n1-1; i++)
    	{
    		for(int j=0; j<n2-1; j++)
    		{
    			if(s1[i]==s2[j])
    			{
    				cout<<"YES"<<endl;
    				found=1;
    			}
    		if(found==1)
    			break;
    		}
    	if(found==1)
    		break;
    	}
    if(found==0)	
    	cout<<"NO"<<endl;
    }
    return 0;
}
