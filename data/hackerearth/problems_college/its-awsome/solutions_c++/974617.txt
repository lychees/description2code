#include <iostream>
using namespace std;

int main()
{
	int a;
	cin>>a;
	while(a--)
	{
		string b;
		cin>>b;
		int len = b.size();
		int cnt = 0;
		for(int i=0;i<len;i++)
		{
			if(b[i]=='4' || b[i]=='7'){cnt++;}
		}
		if(cnt==4 || cnt==7)cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
    return 0;
}
