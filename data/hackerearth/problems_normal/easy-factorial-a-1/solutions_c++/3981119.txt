#include <iostream>
using namespace std;


long long int factorial(long long int n, long long int m)
{
	int answer = 1;
	for(long long int counter = 1 ; counter <= n ; counter++)
		answer = (answer*counter) % m;
	return answer;
}

int main()
{
	long long int n, m;
	while(cin >> n >> m)
		cout << factorial(n , m) << endl;
    return 0;
}
