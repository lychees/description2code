def prime(n):
    x = int(n**0.5)
    for i in range(2, x+1):
        if n % i == 0:
            return False
    return True


n = input()
sum = 0
count = 0
for i in range(2, n+1):
    if prime(i):
        count = count+1
        sum = sum+2**(count-1)
    else:
        sum = sum+2**(count-1)
print sum
