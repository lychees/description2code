#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    //freopen("int.txt","r",stdin);
    int T;
    cin >> T;
    while (T--)
    {
        long long n, q, tmp, cost = 0;
        vector<long long>height;
        cin >> n >> q;
        for (int i = 0; i < n; ++i)
        {
            cin >> tmp;
            height.push_back(tmp);
        }
        if (n == 1) cout << "0" << endl;
        else
        {
            sort(height.begin(),height.end());

            cost += q*abs(height[0] - height[n-1]);

            cout << cost << endl;
        }


    }
    return 0;
}
