#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;
#define LL long long
int main()
{
	ios::sync_with_stdio(0);
	int t;
	cin>>t;
	LL F,G,A,B,C,pow;
	while(t--)
	{
		cin>>F>>G>>A>>B>>C>>pow;
		LL maxi=0;
		for(int z=0;100*z<=G;++z)
		{
			int yo=(G-100*z)/50;
			for(int y=0;y<=yo;++y)
			{
				if(125*y+50*z <=F)
				{
					int x=(F-125*y-50*z)/100;
					maxi=max(maxi,A*x + B*y +C*z);
				}
			}
		}
		cout<<(maxi-pow >0 ? maxi-pow : -1)<<endl;
	}
	
	
	
	return 0;
}
