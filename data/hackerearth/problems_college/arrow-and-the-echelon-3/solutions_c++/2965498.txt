#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll t,x,i,j;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>x;
        for(i=x;i>=0;i--)
        {
            if(i%3==0 && (x-i)%5==0)
            {
               s="";
               for(j=1;j<=i;j++)
               {
                   s+='5';
               }
               for(j=1;j<=x-i;j++)
               {
                   s+='3';
               }
               break;
            }
        }
        if(s=="")
            cout<<-1<<'\n';
        else
            cout<<s<<'\n';
    }
    return 0;
}
