n = int(raw_input())
ans  = 0
for i in range(n):
	s = []
	w = raw_input()
	for c in w:
		if c == 'A':
			if len(s) > 0 and s[-1] == 'A':
				s.pop()
			else:
				s.append('A')
		if c == 'B':
			if len(s) > 0 and s[-1] == 'B':
				s.pop()
			else:
				s.append('B')
	if len(s) == 0:
		ans += 1
print ans
