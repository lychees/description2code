#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>

using namespace std;

inline int fastread(){
  register int a=0;
  register char c=getchar_unlocked();
  
  while(c<48 || c>57) c=getchar_unlocked();

  while(c>=48 && c<=57){
    a=(a<<3)+(a<<1)+c-'0';
    c=getchar_unlocked();
  }
  
  return a;
}

int main(){
  int t, m, n,k;
  int min, max;
  double prob;

  t=fastread();
  while(t--){
    m=fastread(); n=fastread(); k=fastread();
    double a[m+1][k+1];

    memset(a, 0, sizeof(double)*(m+1)*(k+1));
    
    double d=pow(n,m);
   
    for(int i=1; i<=n && i<=k; i++){
      a[1][i]=(1.0)/d;
    }
    
    for(int i=2; i<=m ; i++){
      for(int j=i; j<=k && j<=n*i ; j++){
	for(int l=1; l<=n && l<=j; l++){
	  a[i][j]+=a[i-1][j-l];
	}
      }
    }
    
    prob=a[m][k];
    
    if(prob==0){
      printf("0.000 0\n");
    }else{
      int e=0;
      while(prob<1){
	e++;
	prob*=10;
      }
      printf("%.3lf %d\n",prob, e);
    }
  }
    
  return 0;
}
