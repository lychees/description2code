#include <iostream>
#include <bits/stdc++.h>
#define upperlimit 100
#define ll long long
using namespace std;
int dv[upperlimit+1];
int cnt[upperlimit+1];
int gcd(int num1,int num2){
	if(num1%num2==0)return num2;
	return gcd(num2,num1%num2);
}
int main() {
	// your code goes here
	int t;
	ll x,y;
	int a,b,c,tot=0,temp,i;
	double res;
	for(a=1;a<=upperlimit;a++)for(b=a+1;b<=upperlimit;b++)for(c=b+1;c<=upperlimit;c++){
		tot++;
		cnt[gcd(a,gcd(b,c))]++;
	}
	cin>>t;
	while(t--){
		cin>>x>>y;
		temp=0;
		x-=y;
		if(x<0)x*=-1;
		for(i=1;i<=upperlimit;i++)if(x%i==0)temp+=cnt[i];
		res=((double)(temp))/tot;
		printf("%.6f\n",res);
	}
	return 0;
}
