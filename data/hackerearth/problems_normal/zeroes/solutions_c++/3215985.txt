#include<bits/stdc++.h>
using namespace std;
#define ll long long

ll a[100010]={0};

void convert(ll n)
{
    while (n%2 == 0)
    {
        a[2]++;
        n = n/2;
    }
    for (ll i = 3; i <= sqrt(n); i = i+2)
    {
        while (n%i == 0)
        {
            a[i]++;
            n = n/i;
        }
    }
    if (n > 2)
        a[n]++;
}

ll convertp(ll n)
{
	ll minx=10001000;
	ll cnt=0;
    while (n%2 == 0)
    {
        n = n/2;
        cnt++;
    }
    if(cnt!=0)
    	minx=min(minx,(a[2]/cnt));
    if(n==0)
    	return minx;
    	
    cnt=0;
    for (ll i = 3; i <= sqrt(n); i = i+2)
    {
    	cnt=0;
        while (n%i == 0)
        {
            n = n/i;
            cnt++;
        }
        if(cnt!=0)
        	minx=min(minx,(a[i]/cnt));
    }
    cnt=0;
    if (n > 2){
        cnt++;
    	minx=min(minx,(a[n]/cnt));
    }
    return minx;
}


int main()
{
    ll t,n,m,i,j,k,x;
    cin>>n;
    memset(a,0,sizeof(a));
    for(i=0;i<n;i++)
    {
    	cin>>k;
    	convert(k);
    }
    cin>>m;
    while(m--)
    {
    	cin>>x;
    	ll flag=convertp(x);
    	cout<<flag<<endl;
    }
    return 0;
}
