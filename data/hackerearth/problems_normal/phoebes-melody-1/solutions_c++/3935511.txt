#include<bits/stdc++.h>
using namespace std;
int main(){
   int t;
   cin>>t;
   while(t--){
   	int n,k,i,j,k1;
   cin>>n>>k;
   int a[n],dp[n];
   for(i=0;i<n;i++){
   	cin>>a[i];
   	dp[i]=0; 
   }
   
   for(i=0;i<n;i++){
   	k1=k;
   	for(j=i;k1<n;k1++){
   		if(j+k1<n){
   		
   		if(a[j+k1]<a[i] ){
   			dp[i]=abs(j+k1-i);
   			break;
   		}
   	}
   	    if(j-k1>=0){
   	    
   		if(a[j-k1]<a[i]  ){
   			dp[i]=abs(k1);
   			break;
   		}
   	}
   	}
   }
   int sum =0;
   for(i=0;i<n;i++) sum=sum +dp[i];
   cout<<sum<<endl;
   }
}
