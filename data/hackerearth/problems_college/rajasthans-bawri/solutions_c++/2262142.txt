#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,px,py,bx,by;
    cin>>t;
    while(t-->0){
    	cin>>px>>py>>bx>>by;
    	if(px<py){
    		swap(px,py);
    	}
    	if(bx<by){
    		swap(bx,by);
    	}
    	if(px>=bx && py>=by ){
    		cout<<"Possible"<<endl;
    	}else{
    		cout<<"Not Possible"<<endl;
    	}
    }
    return 0;
}
