#include <iostream>
#define ll long long
using namespace std;
ll a[100001], u[100001], d[100001];
int main()
{
	ll n, q, i, li, ri;
    cin>>n>>q;
    for(i=1;i<=n;i++)
    	cin>>a[i];
    u[n]=n;d[1]=1;
	for(i=n-1;i>=1;i--)
	{
		if(a[i]<=a[i+1])
			u[i]=u[i+1];
		else u[i]=i;	
	}
	for(i=2;i<=n;i++)
	{
		if(a[i-1]>=a[i])
			d[i]=d[i-1]; 
		else d[i]=i;
	}
    while(q--)
    {
    	cin>>li>>ri;
    	if(u[li]>=d[ri]) cout<<"Yes"<<endl;
    	else cout<<"No"<<endl;
    }
    return 0;
}
