#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int n,k,v[1000005],temp;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
    	cin>>v[i];
    }
    sort(v,v+n);
    temp=v[0];
    for(int i=1;i<n;i++)
    {
    	if(v[i]-temp>k)
    	break;
    	temp=v[i];
    	
    }
    cout<<temp;
    return 0;
}
