#include <iostream>
using namespace std;
typedef long long relong;

int main() 
{	
	relong T, N, n, dist, min;	
	cin >> T;
	while (T--) {		
		cin >> N;
		min = 2147483648;
		for (relong i = 0 ; i < N; ++i) {
			cin >> n;
			dist = 0;
			while (n) {				
				if (n&1) {
					dist++;					
				}
				n >>= 1;
			}
			if (dist < min)
				min = dist;					
			
		}
		cout << min << endl;
	}

	return 0;
}
