#include<bits/stdc++.h>
#define ll long long int
using namespace std;


int main()
{
    int t;
    cin>>t;
    int flag=1;
    while(t--)
    {
        int n;
        cin>>n;
        int arr[n];
        for(int i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        ll ans;
        ans=arr[0];
        ll incl=arr[0];
        ll excl=0;
        ll exclnew=0;
        for(int i=1;i<n;i++)
        {   exclnew=((incl>excl)?incl:excl);
            incl=arr[i]+excl;
            excl=exclnew;
            ans=max(incl,excl);
        }
        cout<<"Case "<<flag<<": "<<ans<<endl;
        flag++;


    }



}
