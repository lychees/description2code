#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int t,a[26]={0},m,i;
	char s[100000];
	
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	
    	for(i=0;i<strlen(s);i++)
    	{
    		
    		a[s[i]-'a']++;
    		
    	}
    	m=a[17];
    //	cout<<a[1]<<a[17]<<a[20]<<a[24];
    	for(i=0;i<26;i++)
    	{
    		if(i==20)
    		m=min(m,a[i]);
    		
    		if(i==24)
    		m=min(m,a[i]);
    		
    		if(i==1)
    		m=min(m,a[i]);
    		
    		a[i]=0;
    		
    	}
    	
    	cout<<m<<endl;;
    	
    	
    }
    return 0;
}
