#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	string str;
	while(t--) {
		cin>>str;
		int a[257]={0};
		long long int l = str.size(),i;
		for(i = 0; i < l; i++) {
		   a[str[i]]++;       
		}
		l = 0;
		for(i = 0; i < 257; i++) {
		    if(a[i])
		    l++;
		}
		l = (l*(l+1))/2;
		cout<<l<<endl;
	}
    return 0;
}
