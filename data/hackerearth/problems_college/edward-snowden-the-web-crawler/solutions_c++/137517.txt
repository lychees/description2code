#include<stdio.h>
#include<iostream>
#include<string>
#include<vector>
using namespace std;
int main()
{
	int i, len1, len2, j, found;
	string str,s1,temp;
	vector<string> v;
	cin>>s1;
	while(1)
	{
		cin>>str;
		if(str=="exit")
			break;
		found = str.find(s1);
		 if(found!=string::npos)
		  {
  			j=found;
  			temp="";
  			while((j+1)<str.size()&& str[j+1]!='>')
  			{
  				temp+=str[j];
  				j++;
  			}
  			v.push_back(temp);
  		}
  		
	}
	for(i=0;i<v.size();i++)
	{
		cout<<v[i]<<endl;
	}
	return 0;
}
