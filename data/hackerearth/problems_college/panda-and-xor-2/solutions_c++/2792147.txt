#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mod 1000000007
#define ll long long int
int main()
{
    int t,n,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        int c=0;
        while(n>0&&m>0)
        {
            int v=n&1;
            int u=m&1;
            if(u!=v)
                c++;
            n>>=1;
            m>>=1;
        }
        while(n>0)
        {
            if(n&1)c++;
            n>>=1;
        }
        while(m>0)
        {
            if(m&1)c++;
            m>>=1;
        }
        if(c&1)cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
