#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main()
{
    long long t, i, j, x, y, c, a, b;
    cin>>t;
    while(t--)
    {
        cin>>x>>y;
        c=i=0;
        while(x>0 or y>0)
        {
            a=x%10;
            b=y%10;
            if(a+b+i>=10)
            {
                c++;
            }
            j=(a+b+i)/10;
            i=j;
            x/=10;
            y/=10;
        }
        if(c==0)
            cout<<"No carry operation\n";
        else if(c==1)
            cout<<"1 carry operation\n";
        else
            cout<<c<<" carry operations\n";


    }

    return 0;
}
