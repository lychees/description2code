import sys

for line in sys.stdin:
    ll = list(line)

    for i in range(0, len(ll)):
        if ll[i] == '/' and i+1 < len(ll) and ll[i+1] == '/':
            break

        if ll[i] == '-' and i+1 < len(ll) and ll[i+1] == '>':
            ll[i] = '.'
            ll[i+1] = ''

    sys.stdout.write(''.join(str(element) for element in ll))
