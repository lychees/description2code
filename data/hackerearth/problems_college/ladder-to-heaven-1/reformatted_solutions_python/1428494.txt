#!/usr/bin/env python

tc = int(raw_input())

outputs = []

for i in xrange(tc):

    N = int(raw_input())

    even_sum = 2

    curr_num = 0
    prev1, prev2 = 1, 2

    while True:

        curr_num = prev1 + prev2

        if curr_num > N:
            break

        if curr_num % 2 == 0:

            even_sum += curr_num

        prev1 = prev2
        prev2 = curr_num

    outputs.append(even_sum)

for i in outputs:
    print i
