#include<stdio.h>
int main()
{
 long long unsigned int i=1,j,n,c,t;
 scanf("%llu",&n);
	while(i++)
		{
			c=0;
			for(j=1;j<=i;j++)
			{
			if(i%j==0)
			c++;
			if(c>=3)
			break;
			}
		if(c==2)
			{
				t=i;
				n--;
				if(n==0)
				break;
			}	
		}
		printf("%d",t);
 return 0;
}
