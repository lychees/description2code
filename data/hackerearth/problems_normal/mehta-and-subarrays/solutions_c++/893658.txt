#include <vector>
#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
vector < pair<long long, int> > v;
int n;
long long int kari;
int main(){
	cin>>n;
	v.push_back(make_pair(0,1) );
	long long int sum=0;
	for(int i=1;i<=n;i++){
		cin>>kari;
		sum+=kari;
		v.push_back(make_pair(sum,i+1) );
	}
	sort(v.begin(),v.end());
	int b=v[0].second;
	int countt=0;
	int val=0;
	for(int i=1;i<v.size();i++){
		int len=max(0,v[i].second-b);
		if(len){
		if(len>val){
			val=len;
			countt=1;
		}
		else{
		if(len==val){
			countt++;
		}
		}
		}
		b=min(b,v[i].second);
	}
	if(val==0){
		cout<<-1<<endl;
		return 0;
	}
	else{
		cout<<val<<" "<<countt<<endl;
	}
	return 0;
}
