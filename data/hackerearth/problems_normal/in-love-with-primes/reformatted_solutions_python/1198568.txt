from math import *


def is_prime(numb):
    if numb == 0 or numb == 1:
        return False
    for i in range(2, int(sqrt(numb)+1)):
        if numb % i == 0:
            return False
    return True


t = input()
for _ in range(t):
    numb = input()
    flag = False
    for i in range((numb/2)+1):
        if is_prime(i) and is_prime(numb-i):
            flag = True
            break
    if flag:
        print "Deepa"
    else:
        print "Arjit"
