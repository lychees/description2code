#include <bits/stdc++.h>
using namespace std;
//long prime[1000000000];//Global variable

int main()

{	
	int t;
	cin>>t;
	string s;
	int a[26];
	for(int j=0;j<t;j++)
	{		
		for(int i=0;i<26;i++)
		{a[i]=0;}
		cin>>s;
		int l=s.length();
		for(int i=0;i<l;i++)
		{
			a[int(s[i])-65]++;
		}
		
	//	for(int i=0;i<26;i++)
	//	{cout<<a[i]<<" ";}
		int i;
		for(i=0;i<26;i++)
		{
			if(a[i]!=26-i&&a[i]>0)
			break;
		}
		if(i!=26)
		cout<<"No\n";
		else
		cout<<"Yes\n";
	}
}
