#include<iostream>
#include<cmath>
#include<cstdio>
#include<cstring>
#define LL long long
using namespace std;
int Max(int x, int y)
{ return (y > x)? y : x; }

int MaxSubArraySum(int a[], int size)
{
   int Max_so_far = a[0], i;
   int curr_Max = a[0];

   for (i = 1; i < size; i++)
   {
        curr_Max = Max(a[i], curr_Max+a[i]);
        Max_so_far = Max(Max_so_far, curr_Max);
   }
   return Max_so_far;
}
int A[1000141];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);

        int i;
        for(i=0;i<n;i++)
            scanf("%d",&A[i]);
        int ans=MaxSubArraySum(A,n);
        printf("%d\n",ans);
    }
    return 0;
}
