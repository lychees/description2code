#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,h,m,ans;
    cin>>t;
    while(t--)
    {
    	cin>>h>>m;
    	if(h==12)	h=0;
    	h=0.5*(h*60+m);
    	m=6*m;
    	ans=abs(h-m);
    	(360-ans)<ans?cout<<360-ans:cout<<ans;
    	cout<<endl;
    	
    }
    return 0;
}
