#include<iostream>

using namespace std;
int main()
{
	int t;
	cin>>t;
	string s;
	while(t--)
	{
		int p=0;
		int hash[26]={0};
		cin>>s;
		for(int i=0;i<s.size();i++)
		{
			hash[s[i]-'a']=1;
		}
		for(int i=0;i<26;i++)
		{
			if(hash[i]==1)
			p++;
		}
		if(p==26)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
return 0;
}
