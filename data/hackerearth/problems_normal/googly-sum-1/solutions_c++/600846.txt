#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define SH 100001
bool add(int a)
{
    if(a<=100000&&a>=-100000)
        return true;
    return false;
}
int arr[200005]={0};
int main()
{
    ios_base::sync_with_stdio(false);
    int n,m[5005],counter=0;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>m[i];
    if(add(m[0]<<1))
        arr[(m[0]<<1)+SH]=1;
    for(int i=1;i<n;i++)
    {
        int as;
        for(int j=0;j<i;j++)
        {
            int ask=m[i]-m[j];
          if(add(ask)&&arr[ask+SH])
          {
              counter++;
              break;
          }
        }
        for(int j=0;j<=i;j++)
        {
            as=m[i]+m[j];
            if(add(as))
            {
                as+=SH;
                arr[as]=1;
            }
        }
    }
    cout<<counter;
}
