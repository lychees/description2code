#include <iostream>
#include<algorithm>
#define ll long long int 
using namespace std;
ll rem_packet(ll arr[],ll n)
{
	ll i;
	sort(arr,arr+n);
	ll tmp,max;
	tmp=max=1;
	for(i=1;i<n;i++)
	{
		if(arr[i-1] == arr[i])
		{
			tmp++;
			if(tmp > max)
			max=tmp;
		}
		else
		tmp=1;
	}
	return max;
}

int main()
{
   int t;
   ll arr[100000],n,i;
   cin >> t;
   while(t--)
   {
   	 cin >> n;
   	 for(i=0;i<n;i++)
   	 cin >> arr[i];
   	 cout << rem_packet(arr,n) <<"\n";
   }
  
    return 0;
}
