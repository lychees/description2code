from operator import xor
import itertools

l=int(raw_input(''))
computed={}
for i in range (0,l):
	n = int(raw_input(''))
	if n not in computed:
		count=0
		lst=range(1,n+1)
		s=list(itertools.combinations(lst,2))

		for items in s:
			if xor(items[0],items[1])<=n:
				count = count+1
		
		computed[n]=count
		print count

	else:
	    c=computed[n]
	    print c
