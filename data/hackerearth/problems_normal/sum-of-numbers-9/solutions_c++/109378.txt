#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int check(int *arr,int start,int currSum,int n, int sum)
{
	if(start==n)
    {
		if(currSum==sum)
            return 1;
		return 0;
	}
	return check(arr,start+1,currSum+arr[start],n,sum)||check(arr,start+1,currSum,n,sum);
}

int main()
{
	int t,i,n;
	cin>>t;
	while(t--)
    {
		cin>>n;
        int arr[n],sum,sum1=0;
		for(i=0;i<n;i++)
            {cin>>arr[i];sum1+=arr[i];}
		cin>>sum;
		int tmp=0;
		if(sum==0||sum==sum1)
            tmp=1;
        else
            tmp=check(arr,0,0,n,sum);
        if(tmp)
            cout<<"YES"<<endl;
		else
            cout<<"NO"<<endl;
	}
	return 0;
}
