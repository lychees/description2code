#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    
while(t--)
    {unsigned char s[105];
    	cin>>s;
    	int i=0;
    	while(s[i]!='\0')
    	{   int j=i;
    	     j%=26;
    	    
    		s[i]+=j;
    	
    		if(s[i]>122)
    		 s[i]=s[i]-26;
    		 
    		i++;
    	}
    
    	cout<<s<<"\n";
    }
}
