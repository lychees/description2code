#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b;
		cin>>a>>b;
		if(a>b)
		{
			int sum=a+b;
			if(sum%2==1)
			cout<<"impossible"<<endl;
			else
			{
				int ans1=sum/2;
				cout<<ans1<<" ";
				int ans2=a-ans1;
				cout<<ans2<<endl;
			}
		}
		else 
		cout<<"impossible"<<endl;
	}
}
