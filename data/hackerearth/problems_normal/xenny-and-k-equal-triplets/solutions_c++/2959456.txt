#include <iostream>
#include <stdio.h>
#define gc getchar_unlocked
#define FOR(i,n) for(ll i=0;i<n;i++)
using namespace std;
typedef long long int ll;
ll get() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
ll fun(ll n){
	ll ans=1;
	if(n%6==0){
		ans*=(n/6);
		ans*=(n-1);
		ans*=(n-2);
	}
	else if((n-1)%6==0){
		ans*=((n-1)/6);
		ans*=(n);
		ans*=(n-2);
	
	}
	else if((n-2)%6==0){
		ans*=((n-2)/6);
		ans*=(n-1);
		ans*=(n);
	
	}
	else if(n&1){
		ans *= ((n-1)/2);
		if(n%3==0){
			ans *= (n/3);
			ans *= (n-2);
		}
		else if((n-2)%3==0){
			ans *= ((n-2)/3);
			ans *= n;
		}
	}
	else {
		ans *= (n/2);
		if((n-1)%3==0){
			ans *= ((n-1)/3);
			ans *= (n-2);
		}
		else {
			ans *= ((n-2)/3);
			ans *= (n-1);
		}
	}
	return ans;
}
ll gcd(ll x, ll y){
    ll wk;
    if(x<y){ wk=x;x=y;y=wk; }
    while(y){
        wk = x%y;
        x=y;
        y=wk;
    }
    return x;
}
int main()
{
    ll n,k,x,t,num,den,y;
    t=get();
    while(t--){
    	n=get();k=get();
    	num=0;den=fun(n);y=0;
    	while(n--){
    		x=get();
    		if(x==k)num++;
    		/*else {
    			if(num>=3)
    			y+=fun(num);
    			num=0;
    		}*/
    	}
    	num=fun(num);
    	ll g = gcd(num, den);
    	if(num==0)
    	cout<<"0/1\n";
    	else{
    	num/=g;
    	
    	den/=g;
    	
    	cout<<num<<"/"<<den<<"\n";}
    }
    return 0;
}
