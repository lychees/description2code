#include <cstdio>
#include <cstdlib>
#include <cstring>

using namespace std;

#define alc(a, n) memset((a), n, sizeof (a))
#define MX 212345
#define MOD 1000000007

int N;
int dec[MX], num[MX], rlt;
char s[MX];

int main() {
	
//	freopen("in.txt", "r", stdin);
//	freopen("out.txt", "w", stdout);
	
	int i, j, k;
	
	dec[0] = 1;
	num[0] = 1;
	for (i = 1; i < 200000; i++) {
		dec[i] = 1LL * dec[i - 1] * 10 % MOD;
		num[i] = (num[i - 1] + dec[i]) % MOD;
	}
	scanf("%s", s);
	N = strlen(s);
	rlt = 0;
	for (i = 0; i < N; i++) {
		rlt = (rlt + (s[i] - '0') * 1LL * (i + 1) * num[N - i - 1] % MOD) % MOD;
	}
	printf("%d\n", rlt);	
	return 0;	
}
