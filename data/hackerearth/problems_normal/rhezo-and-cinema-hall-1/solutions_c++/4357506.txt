#include <bits/stdc++.h>

using namespace std;

const int MAXN = 501;
const int mod = 1000000007;

int n, m, k;

int memo[MAXN][11][MAXN][2][3];

int bt(int i, int j, int cnt, bool prev, int t) {
	cnt = min(cnt, k);
	if (i == n - 1 && j == m) {
		return cnt >= k && t == 2;
	}
	if (j == m) {
		if (t < 2) return 0;
		return bt(i + 1, 0, cnt, false, 0);
	}
	int &ret = memo[i][j][cnt][prev][t];
	if (ret == -1) ret = 0;
	else return ret;
	if (!prev)
		ret += bt(i, j + 1, cnt + 1, true, min(t + 1, 2)), ret %= mod;
	ret += bt(i, j + 1, cnt, false, t), ret %= mod;
	return ret;
}

int main() {
	memset(memo, -1, sizeof memo);
	cin >> n >> m >> k;
	cout << bt(0, 0, 0, 0, 0) << endl;
}
