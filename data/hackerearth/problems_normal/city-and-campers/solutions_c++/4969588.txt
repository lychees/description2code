#include <iostream>
#include<cstdio>
using namespace std;
int parent[100005],sz[100005],mx=1,mn=100005,n;
int find1(int i)
{
    while(parent[i]!=i)
    {
        parent[i]=parent[parent[i]];
        i=parent[i];
    }
    return i;
}
int union1(int x,int y)
{
    int rootx=find1(x);
    int rooty=find1(y);
    if(rootx!=rooty)
    {
        if(sz[rootx]>=sz[rooty])
        {
            parent[rooty]=rootx;
            sz[rootx]+=sz[rooty];
            if(sz[rootx]>mx)
                mx=sz[rootx];
            //cout<<"Fuckx ";
        }
        else
        {
            parent[rootx]=rooty;
            sz[rooty]+=sz[rootx];
            if(sz[rooty]>mx)
                mx=sz[rooty];
            //cout<<"Fucky ";
        }
        /*if(mx==n)
            return 0;*/
        mn=mx;
        for(int i=1;i<=n;i++)
        {
            if(parent[i]==i)
            {
                if(sz[i]<mn)
                    mn=sz[i];
                //cout<<"Ghusa ";
            }
            if(mn==1)
                break;
        }
    }
    if(mx==n)
        return 0;
    return (mx-mn);
}
int main()
{
    int q,i;
    scanf("%d%d",&n,&q);
    for(i=1;i<=n;i++)
    {
        parent[i]=i;
        sz[i]=1;
    }
    while(q--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        int diff=union1(x,y);
        printf("%d\n",diff);
    }
    return 0;
}
