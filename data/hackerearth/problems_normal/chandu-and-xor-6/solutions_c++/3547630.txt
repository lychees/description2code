{
	fast;

	rep(i , n) {
		rep(j , m) {
			cin >> A[i][j];
			row[i] += A[i][j];
			col[j] += A[i][j];
		}
	}

	ll ans = 0;
	rep(i , n) {
		ll v = row[i];
		rep(j , m) {

		}
	}
