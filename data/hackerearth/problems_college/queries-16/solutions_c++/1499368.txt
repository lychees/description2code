using namespace std;

int cnt[10000005] = {0};

void solve() {
    int t; scanf("%d",&t);
    int s;
    if(!t) {
        scanf("%d",&s);
        while(s) {
            cnt[s]++;
            s>>=1;
        }
        cnt[0]++;
    }
    else if(t == 1) {
        scanf("%d",&s);
        while(s) {
            cnt[s]--;
            s>>=1;
        }
        cnt[0]--;
    }
    else {
        scanf("%d",&s);
        printf("%d\n",cnt[s]);
    }
        
    
}
int main() {
    int t; cin>>t;
    while(t--) {
        solve();
    }
    
    
}
