#include <iostream>
using namespace std;

int find_max(int a,int b)
{
	if(a>b)
		return a;
	else return b;
}
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    int max=0,max_now=0;
    for(int i=0;i<n;i++)
    {
    	max=find_max(max+a[i],a[i]);
    	if(max<0)
    		max=0;
    	max_now=find_max(max,max_now);
    }
    cout<<max_now;
    return 0;
}
