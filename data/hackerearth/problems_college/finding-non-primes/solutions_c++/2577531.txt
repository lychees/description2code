#include <iostream>
using namespace std;

int main()
{
    int arr[10001];
    for(int i =1;i<10001;i++)
    	arr[i]=1;
    for(int i =2;i<10001;i++)
    	if(arr[i] == 1)
    		for(int j = i+1 ;j<10000;j++)
    			if(arr[j] != 0 && j%i == 0)
    				arr[j]=0;
    	
    int n;
    cin>>n;
    for(int i =4;i<=n;i++)
    	if(arr[i] == 0)
    		cout<<i<<endl;
}
