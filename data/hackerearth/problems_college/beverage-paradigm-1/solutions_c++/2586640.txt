#include <iostream>
using namespace std;
 
int main()
{long long int z,n,i,a[300000],j,k,flag=0;
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    cin>>k;
    for(i=0;i<n-2;i++)
    {
    	for(j=i+1;j<n-1;j++)
    	{
    		    	for(z=j+1;z<n;z++)
    		    	{
    		if(a[i]+a[j]+a[z]==k)
    		{
    			flag=1;
    			break;
    	}
    		else flag=0;
    	}
    	if(flag==1)
    	{
    		j=n;
    		i=n;
    	}
    	}
    	
    }
    if(flag==0)
    cout<<"False"<<endl;
    else 
      cout<<"True"<<endl;
    return 0;
}
