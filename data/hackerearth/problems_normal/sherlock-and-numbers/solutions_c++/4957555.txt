#include <bits/stdc++.h>

using namespace std;

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n, k, p, res;
		set <int> removed;
		scanf("%d %d %d", &n, &k, &p);
		res = p;
		for (int i = 0; i < k; i++) {
			int num;
			scanf("%d", &num);
			removed.insert(num);
			if (num <= res) {
				do {
					res++;
				} while (removed.count(res));
			}
		}
		printf("%d\n", (res > n) ? -1 : res);
	}
	return 0;
}
