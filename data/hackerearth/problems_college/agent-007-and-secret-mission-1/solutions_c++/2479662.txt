#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s1;
    int i,j,c=0,t,l;
    cin>>t;
    while(t--)
    {
    	cin>>s1;
    	l=s1.length();
    	cin>>i;
    	j=0;
    	c=0;
    	while(s1[j]!=NULL)
    	 {
    		if(s1[j]=='1')
    		c++;
    		j++;
    	 }
    	 
    	 
    	 i=(i*(i+1))/2;
     if(i%2==0)
        cout<<c<<endl;
        else
        cout<<l-c<<endl;
   }
    return 0;
}
