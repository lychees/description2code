using namespace std;
int t,n,i,a[100003];

int main(){
    cin>>t;
    while(t--){
        cin>>n;
        for(i=0;i<n;i++)
        cin>>a[i];
        
        for(i=1;i<n;i++)
        a[i]=a[i]+a[i-1];
        
        for(i=1;i<n;i++){
            if(a[i-1]==(a[n-1]-a[i])){
                cout<<"YES"<<endl;
                break;
            }
            if(i==n-1)
            cout<<"NO"<<endl;
        }
    }
}
