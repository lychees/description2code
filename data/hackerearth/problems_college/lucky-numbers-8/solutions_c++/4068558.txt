#include <iostream>
using namespace std;

int main()
{
    long long int i,a[1000000],max,t,count=0,n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	for(i=0;i<n;i++)
   	cin>>a[i];
   i=n-1;
   max=a[i];
   count=0;
   //cout<<max;
   i--;
   count++;
   while(i>=0)
   {
   	if(max<a[i])
   	{
   		max=a[i];
   		count++;
   	}
   	i--;
   }
   	cout<<count<<endl;
   }
    return 0;
}
