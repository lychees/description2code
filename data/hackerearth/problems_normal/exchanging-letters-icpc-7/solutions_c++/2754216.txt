#include <bits/stdc++.h>

using namespace::std;
#define ll long long int

const int MAX=2002,MOD=1000000007;

ll ncr[MAX+1][MAX+1];

void mod_ncr()
{
	ll i,j;
	for(i=0;i<=MAX;i++)
	{
		ncr[i][i]=1;
		ncr[i][0]=1;
		for(j=i+1;j<=MAX;j++)
		ncr[i][j]=0;	
	}	

	for(i=1;i<=MAX;i++)
	{
		for(j=1;j<i;j++)
		ncr[i][j]=(ncr[i-1][j] + ncr[i-1][j-1])%MOD;	
	}
}


int main()
{
	ll i,j,k,n,t;

	ll a[10000];

	cin>>n>>k>>t;

	ll hash[10000];

	ll ans1=0,ans2=0;
	for(i=0;i<k;i++)
	{
		cin>>j;
		ans1+=(j-1);
		ans2+=(j+1);
	}	

	if(t<ans1 || t>ans2)
	{
		cout<<"0\n";
		return 0;
	}
	mod_ncr();

	ll val = t-ans1;

	if(val%2!=0)
	cout<<"0\n";
	else
	{
		//cout<<"hi";
		cout<<ncr[k][val/2]<<endl;
	}	

}
