#!/usr/bin/env python
import sys


def run(inputStream):
    num, numBubly = 0, 0
    for line in inputStream:
        line = line.rstrip("\n")
        stack, lenToCheck = [], 0

        if num == 0:
            num = int(line)
        else:
            length = len(line)
            if length % 2 != 0:
                continue

            for i, char in enumerate(line):
                if i == 0 or lenToCheck == 0:
                    stack.append(char)
                    lenToCheck = 1
                    continue

                if char == stack[lenToCheck-1]:
                    stack.pop(lenToCheck-1)
                    lenToCheck -= 1
                else:
                    stack.append(char)
                    lenToCheck += 1

            if lenToCheck == 0:
                numBubly += 1
    return numBubly


if __name__ == '__main__':
    data = sys.stdin
    print run(data)
