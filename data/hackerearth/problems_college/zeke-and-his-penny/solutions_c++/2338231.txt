#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int a[3],t,temp1,temp2,temp3;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&a[0],&a[1],&a[2]);
        sort(a,a+3);
        temp1=a[0]%1000000007;
        temp2=(((a[1]-1)%1000000007)*temp1)%1000000007;
        temp3=(((a[2]-2)%1000000007)*temp2)%1000000007;
        printf("%lld\n",temp3);
    }
    return 0;
}
