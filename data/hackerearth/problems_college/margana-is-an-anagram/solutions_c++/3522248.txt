#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int a[26],t,i,j,k,flag=0;
    char str1[1000],str2[1000];
    cin>>t;
    for(i=0;i<t;i++){
    	for(j=0;j<26;j++){
    		a[j]=0;
    	}
    	cin>>str1>>str2;
    	int l1=strlen(str1);
    	int l2=strlen(str2);
    	if(l1!=l2)
    	cout<<"No";
    	else{
    	for(j=0;j<l1;j++){
    		int l=(int)str1[j];
    		a[l%97]=1;
    	}
    	for(j=0;j<l2;j++){
    		int p=(int)str2[j];
    		if(a[p%97]==1)
    		  flag=1;
    		else{
    			flag=0;
    			break;
    		}
    	}
    	/*for(j=0;j<26;j++){
    		cout<<a[j];
    	}*/
    	if(flag==1)
    	cout<<"Yes";
    	else
    	cout<<"No";
    }
    	cout<<"\n";
    }
}
