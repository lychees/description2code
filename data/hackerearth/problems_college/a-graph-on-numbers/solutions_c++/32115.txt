#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;
int a[10000001],b[10000001];
int gcd(int a,int b)
{
    if(a==0)return b;
    return gcd(b%a,a);
}

int solve(int n)
{
    return b[n];
}

void init()
{
    int i,j,k;
    a[1]=1;

    for(i=2;i*i<=10000000;i++)
    if(!a[i])
    {
    for(j=i*i;j<=10000000;j+=i)
    a[j]=i;
    }

    b[1]=0;b[2]=b[3]=1;
    for(i=4;i<=10000000;i++)
    {
        if(!a[i]){b[i]=1;continue;}

        b[i]=b[a[i]]+b[i/a[i]];
    }

}
int main()
{
    int t,n, i,j,k;
    int ans;
    scanf("%d",&t);
    init();
    while(t--)
    {
        scanf("%d %d",&i,&j);

        n=min(i,j);
        j=max(i,j);
        i=n;

        if(j%i==0)
        ans=b[j/i];
        else
        {
            n=gcd(i,j);
            ans=b[i/n];
            ans+=b[j/n];
        }

            printf("%d\n",ans);

            }

    return 0;
}
