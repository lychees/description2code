#include <iostream>
using namespace std;

int main()
{
    int t;
    unsigned long long n, x, y;
    
    cin >> t;
    
    while(t--) {
    	cin >> n >> x >> y;
    	long long start = (x + 1 > n - y) ? (x + 1) : (n - y);
    	if(n == 0)cout << "0\n";
    	else if(start <= n)cout << (n - start + 1) << endl;
    	else cout <<"0\n";
    }
}
