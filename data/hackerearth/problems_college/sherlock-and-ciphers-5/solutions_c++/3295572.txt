#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)
#define pb(x) push_back(x)
#define MAXN 100010

char str[500];

int main()
{
	ll i,j,n,l,q;

	scanf("%lld", &n);
	
	getchar();
	while(n--) {
		cin.getline(str,500);

		l = strlen(str);

		q = sqrt(l);	

		for(i=0;i<q;i++) {
			for(j=i;j<l;j+=q) {
				printf("%c", str[j]);
			}
		}
		printf("\n");
	}


	return 0;
}
