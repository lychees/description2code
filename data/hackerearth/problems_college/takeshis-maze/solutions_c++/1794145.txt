#include <iostream>
using namespace std;
int a[100][100];
long mincost[100][100];

int n;
 
long max(long a,long b)
{
if(a>b)	return a;
else return b;
}

long sumf=0;
long findmax(int i,int j)
{ //cout<<i<<j<<" "<<sum<<"\n";
	
	if(i<0|j<0)
	{return 0;}
 	if(mincost[i][j]==0)
	 {
		mincost[i][j]=a[i][j] + max(findmax(i-1,j),findmax(i,j-1));			
 	 }
 	 
	   return mincost[i][j];
	
 }

 
 
int main()
{int t,i,j;
 long c;
    cin>>t;
    while(t--)
    {sumf =0;
       cin>>n>>c;
       for(i=0;i<n;i++)
       {
        for(j=0;j<n;j++)
        {
         cin>>a[i][j];
		 mincost[i][j]=0;               
 		}
       }    

       long x = findmax(n-1,n-1);
        //cout<<"sumf="<<sumf;
      	
        if(c<=x)
        {cout<<x-c<<"\n";}
        else
        {cout<<-1<<"\n";}      
              
              
    }
    
     return 0;
}
