#include <iostream>
using namespace std;

int main()
{
    int n,k,a[100][26],i,j,c=0;
    char str[100000];
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
    	cin>>str;
    	for(j=0;str[j]!='\0';j++)
    	{
    	  a[i][str[j]-97]=1;	
    	    	}
    }
  	for(j=0;j<26;j++)
    {
    	int f=1;
    	for(i=0;i<n;i++)
    	 if(a[i][j]!=1)
    	 {
    	 	f=0; break;
    	 }
    	if(f)
    	 c++;
    }
    //cout<<k<<c;
    if(c==k)
     cout<<":-)";
    else
     cout<<":-(";
    return 0;
}
