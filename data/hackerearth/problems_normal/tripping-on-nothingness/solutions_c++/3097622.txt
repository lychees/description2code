#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
int main()
{ll s,q,x;
    map<ll,ll> m;
    m.clear();
    cin>>s>>q;
    while(q--)
    {
    	cin>>x;
    	if(m[__gcd(x,s)]!=-1) cout<<__gcd(x,s)<<endl,m[__gcd(x,s)]=-1;
    	else cout<<"-1\n";
    	
    	
    }
    return 0;
}
