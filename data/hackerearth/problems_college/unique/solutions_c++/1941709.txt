#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t; cin >> t;
    while(t--)
    {
    	int n; cin >> n;
    	int a[n],i,ans=0;
    	for(i=0;i<n;i++)
    	    cin >> a[i];
    	map <int,int> mymap;
    	for(i=0;i<n;i++)
    	{
    		if(mymap[a[i]]==0)
    		{
    			ans++;
    			mymap[a[i]]++;
    		}
    	}
    	cout << ans << endl;
    }
    return 0;
}
