#include <iostream>
#include <string>
#include<utility>
#include <algorithm>
#include <cstddef>
using namespace std;

int main()
{
    string s;
    pair<char, int> cur_max('\0', 0);
    size_t i=0;
    int t;
    getline(cin, s);
    sort(s.begin(), s.end());
    i = s.find_last_of(s[0]);
    if(cur_max.second<i+1)
    {
    	cur_max.first=s[0];
    	cur_max.second=i+1;
    }
    while(i != string::npos && i<s.length()-1)
    {
    	s=s.substr(i+1);
    	i = s.find_last_of(s[0]);
    	if((i+1)>cur_max.second)
    	{
    		cur_max.first=s[i];
    		cur_max.second=i+1;
    	}
    }
    cout<<cur_max.first<< " "<<cur_max.second;
    return 0;
}
