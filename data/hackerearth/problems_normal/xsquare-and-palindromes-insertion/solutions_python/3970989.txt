from collections import Counter
for i in range(int(raw_input())):
    flag = 1
    strn = raw_input()
    if strn == strn[::-1]:
        flag=0
    else:
        numEven=0
        numOdd=0
        c = Counter(strn)
        DictNew = c.items()
        #print (DictNew)
        for k,v in DictNew:
            if v%2 == 0:
                numEven +=1
            else:
                numOdd +=1
        if numOdd == 1 and numEven > 0:
            flag = 0
        elif numOdd == 0 and numEven > 0:
            flag = 0
        elif numOdd > 0:
            flag = numOdd-1
    print (flag)
