ll solve(ll x)
{
	if(dp[x]!=-1)
		return dp[x];

	if(x<0)
		return 0;
	if(x==0)
	{
		// cout << "\tYES" << endl; 
		return 1;
	}
	ll sum=0;
	sum+=solve(x-1);
	sum = sum % mod;
	sum+=solve(x-2);
	sum = sum % mod;
	sum+=solve(x-3);
	sum = sum % mod;
	return dp[x]=sum;
}
