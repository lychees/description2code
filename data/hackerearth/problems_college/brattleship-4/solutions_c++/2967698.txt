#include <iostream>
using namespace std;

int main()
{
    int t,r,c,w,i,j,ans;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>r>>c>>w;
    	if(r==1)
    	{
    	ans=(c-w)*r+w;
    	if(c-w!=0)
    	ans--;
    	}
    	else
    	{
    		ans=(c-w+1)*r+w-1;
    	}
    	
    	cout<<"Case #"<<i+1<<": "<<ans<<endl;
    }
    return 0;
}
