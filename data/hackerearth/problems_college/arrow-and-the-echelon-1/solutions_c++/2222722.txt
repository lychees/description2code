#include <iostream>
using namespace std;

int main()
{
   int a[]={0,10,5};
   int t,i,j,n;
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	i=n%3;
   	j=n/3;
   	if(i==0)
   	  {
   	  	for(int k=0;k<j*3;k++)
   	  	 	   cout<<"5";
   	  	 	   cout<<"\n";
   	  }
   	  else if(i==1)
   	  {
   	  	j=j-3;
   	  	if(j>=0)
   	  	 {
   	  	 	for(int k=0;k<j*3;k++)
   	  	 	   cout<<"5";
   	  	 	 	for(int k=0;k<a[i];k++)
   	  	 	   cout<<"3";  
   	  	 	   cout<<"\n";
   	  	 }
   	  	 else
   	  	 cout<<"-1\n";
   	  }
   	  else
   	   {
   	   		j=j-1;
   	  	if(j>=0)
   	  	 {
   	  	 	for(int k=0;k<j*3;k++)
   	  	 	   cout<<"5";
   	  	 	 	for(int k=0;k<a[i];k++)
   	  	 	   cout<<"3";  
   	  	 	   cout<<"\n";
   	  	 }
   	  	 else
   	  	 cout<<"-1\n";
   	   }
   	  
   }
    return 0;
}
