#include<iostream>
#include<vector>
#include<algorithm>
#include<string>

int main()
{
    int n,k,i,j,a,b;

   while(t--)
    {
	   for(i =0;i<k;i++)

           str[i].clear();

         }
       }
       for(i =0;i<k;i++)
       {
           sort(str[i].begin(),str[i].end());
       }
           for(i = 0;i<k;i++)

              a = n/2;

          }
