#include<bits/stdc++.h>

using namespace std;

long long arr[1000001];
int main()
{
long long n,k,i,j,ans=0,sum=0;

cin>>n>>k;
for(i=0;i<n;i++)
    cin>>arr[i];

sort(arr,arr+n);

for(i=0;i<n;i++)
{
    if(sum+arr[i]<=k)
    {ans+=1;
    sum+=arr[i];
    }
    else
        break;
}

cout<<ans<<endl;
}
