#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
bool isPrime[1000005];
bool visited[1000005];
ll arra[1000005];
ll arrb[1000005];
void calc()
{
    int i,j;
    for(i=0;i<1000005;i++)
    {
        arra[i]=0;
        arrb[i]=i;
    }
    for(i=2;i<1000005;i++)
    {
        if(arra[i]==0)
        {
            arra[i]=1;
            arrb[i]=i-1;
            for(j=2*i;j<1000005;j=j+i)
            {
                arra[j]=arra[j]+1;
                arrb[j]=arrb[j]-(arrb[j]/i);
            }
        }
    }
}
int main() {
    calc();
    for(int i=1;i<1000005;i++) arrb[i]+=arrb[i-1];
    int t;
    scanf("%d",&t);
    int n,m;
    while(t--) {
        scanf("%d%d",&n,&m);
        printf("%lld\012",arrb[n/m]);
    }
}
