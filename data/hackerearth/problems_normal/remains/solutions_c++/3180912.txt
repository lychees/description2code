#include <iostream>
using namespace std;

int main()
{
	long long int i,j,k,n,T,ans,a,b,c;
	scanf("%lld",&T);
	while(T--){
		scanf("%lld%lld%lld",&a,&b,&n);
		ans=a+b;
		for(i=3;i<=n;i++){
			if(a<b){
				c=b-a;
				ans+=c;
			}		
			else{
				c=a-b;
				ans+=c;
			}
			a=b;
			b=c;
			if(b==0){
				ans+=(n-i)/3*a*2+(((n-i)%3)*a);
				break;
			}
				
		}
		printf("%lld\n",ans);
	}
    
    return 0;
}
