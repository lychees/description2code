#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t,n,count,flag,i,k;
    cin >> t;
    while(t--)
    {
    	cin >> n;
    	int arr[n];
    	count=flag=0;
    	k=1;
    	for(i=0;i<n;i++)
    		cin >> arr[i];
    	
    	while(1)
    	{
    		sort(arr,arr+n);
    		for(i=1;i<=k;i++)
    		{
    			if(arr[n-i]>0)
    				arr[n-i]-=1;
    			else
    			{
    				flag=1;
    				break;
    			}
    		}
    		if(flag==1)
    			break;
    		count++;
    		k++;
    	}	
    	cout << count << endl;
    }
    return 0;
}
