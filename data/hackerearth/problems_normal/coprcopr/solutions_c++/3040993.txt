#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define pb push_back

const int MAX_SIZE = 10000007;
int fs[MAX_SIZE];
void gen()
{
  int i,j;
  fs[1]=1;
  for(i=3;i<MAX_SIZE;i+=2)
    fs[i]=i,fs[i-1]=2;
  for(i=3;i<MAX_SIZE;)
    {
      for(j=2*i;j<MAX_SIZE;j+=i)
	if(fs[j]>i)
	  fs[j]=i;
      i+=2;
      while(i<MAX_SIZE and fs[i]<i)
	i+=2;
    };
};

int phi(int N)
{
  int r = N,p=fs[N];
  if(fs[N]==0){
    printf(">>N is : %d\n",N);
  };
  r/=p;r*=(p-1);N/=p;
  while(N>1)
    {
      if(fs[N]-p)//new prime factor
	{
	  p=fs[N];
	  r/=p;r*=(p-1);
	};
      N/=fs[N];
    };
  return r;
};

ll SumPhi[MAX_SIZE];

int main()
{
  gen();
  int T,N,i,M=-1;
  cin >> T;
  vector<int> INPUT;
  while(T--)
    {
      cin >> N;
      M = max(M,N);
      INPUT.pb(N);
    };
  SumPhi[0]=1;
  for(i=1;i<=M;i++)
    {
      SumPhi[i]=phi(i)+SumPhi[i-1];
    };
  for(i=0;i<INPUT.size();i++)
    {
      cout << SumPhi[INPUT[i]] << endl;
    };
  return 0;
};
