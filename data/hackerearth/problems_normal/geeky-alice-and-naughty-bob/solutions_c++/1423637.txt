#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	ll t,a,b,i,sum,tmp;
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld%lld",&a,&b);
		if(a>=6)
			sum=(b-a+1)*9;
		else{
		for(i=a;i<=b&&i<6;i++)
		{
			if(i==0)
				sum+=0;
			else if(i==1)
				sum+=1;
			else if(i==2)
				sum+=2;
			else if(i==3||i==4)
				sum+=6;
			else if(i==5)
				sum+=3;
			else
				sum+=9;
		}
		if(b>=6){
			tmp=b-i+1;
			sum+=tmp*9;
		}
		}
		cout<<sum<<endl;
	}
	return 0;
}
