#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int input;
	int one=0,two = 0,three = 0,four=0;
	
	for(int i=0;i<n;i++) {
		scanf("%d",&input);
		
		if(input ==1)
			one++;
		else if(input ==2)
			two ++;
		else if (input == 3)
			three ++;
		else 
			four++;
	}
	
	int ans = 0;
	ans += four;
	
	if(three >= one) {
		ans += three;
		one = 0;
		three = 0;
	} else {
		ans += three;
		one = one-three;
		three = 0;
	}
	
	if(one) {
		
		if( two >= ((one/2)+one%2)) {
			ans += ((one/2)+one%2);
			two -= ((one/2) + one%2);
			one = 0;
		} else {
			ans += two;
			one -= (2*two);
			two = 0;
		}
		
		if(two) {
			ans += (two/2 + two%2);
			two = 0;
		}
		
	} else {
		ans += (two/2 + two%2);
		two = 0;
		
	}
	
	if(one) {
		ans += (one/4);
		one%=4;
		if(one)
			ans++;
	}
	
	cout<<ans;
	
    return 0;
}
