#include<bits/stdc++.h>
#define FOR(i,a,b) for(i=(a);i<(b);i++)
#define pb(x) push_back(x)
#define mp(x,y) make_pair((x),(y))
#define mset(m,v) memset(m,v,sizeof(m))
#define dbg(x)	{cerr<< #x << " = " << x <<endl;}
#define gc getchar_unlocked
#define N		2000100

typedef long long ll;
typedef unsigned long long ull;
 
using namespace std;

ll a[N],b[N],tree[N];

ll read(ll idx){
	ll sum = 0;
	while (idx > 0){
		sum += tree[idx];
		idx -= (idx & -idx);
	}
	return sum;
}

void update(ll idx ,ll val){
	while (idx <= N){
		tree[idx] += val;
		idx += (idx & -idx);
	}
}

int main(){
	ll i,n,ans;
	#ifndef ONLINE_JUDGE
		freopen("inpu.txt","r",stdin);
	#endif
	cin>>n;
	FOR(i,0,n)	cin>>a[i];
	FOR(i,0,n)	cin>>b[i];
	ans=0;
	for(i=n-1;i>=0;i--){
		ans=(ans+read(a[i]-1));
		update(b[i],1);
	}
	//cout<<ans<<endl;
	if(ans==1){
		cout<<"SIMPLE INVERSION\n";
		return 0;
	}
	bool f=true;
	for(i=2;i*i<=ans;i++){
		if(ans%i==0){
			f=false;
			break;
		}
	}
	if(f){
		cout<<"MAGIC INVERSION\n";
	}	
	else{
		cout<<"SIMPLE INVERSION\n";
	}
	return 0;
}
