#include <iostream>
#include<math.h>
using namespace std;

int main()
{   long long num;
    int t;
    cin>>t;
    while(t--)
    {
    	cin>>num;
    	long long m=sqrt(num);
    	if(m*m==num)
    	  cout<<"YES\n";
    else
    {
    	long long a=abs(num-(m+1)*(m+1));
    	long long b=abs(num-(m)*(m));
    	if(a<b)
    	  cout<<"+"<<(m+1)*(m+1)-num<<endl;
    	else
    	  cout<<"-"<<num-(m)*(m)<<endl;
    	
    }
    	
    	
    }
   
    return 0;
}
