#include<iostream>
#include<stdio.h>

using namespace std;

int gcd(int a, int b)
{
	int tmp;
	if(a<b)
		return gcd(b,a);
	while(b!=0)
	{
		tmp = b;
		b = a%b;
		a = tmp;
	}
	return a;
}

int main()
{
	int t,d,g;
	long long n;
	scanf("%d",&t);
	for(int i=1;i<=t;i++)
	{
		scanf("%lld %d %d",&n,&d,&g);
		printf("#%d:",i);
		if((g==100 && d<100)||(g==0 && d>0))
			printf("Broken\n");
		else
		{
			int x = gcd(100-d,d);
			int d = d/x;
			int y = (100-d)/x;
			if(n<(d+y))
				printf("Broken\n");
			else
				printf("Possible\n");
		}
	}
	return 0;
}
