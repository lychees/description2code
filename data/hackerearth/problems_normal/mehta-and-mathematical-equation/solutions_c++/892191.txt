#include <bits/stdc++.h>

using namespace std;

const int MOD=1000000007;
int T;
long long N;
int S[61];
int dp[65][6][2];

int rec(int pos, int prev, int good)
{
    if(pos==61)
        return prev==0 && good;
    int& ret=dp[pos][prev][good];
    if(ret!=-1)
        return ret;
    ret=0;
    int n=prev*2+S[pos];
    for(int i=0; i<2; i++)
        for(int j=0; j<2; j++)
            for(int k=0; k<2; k++)
            {
                int v=n-i-j-k;
                if(0<=v && v<=5)
                    ret=(ret+rec(pos+1, v, good || (i && j && k)))%MOD;
            }
    return ret;
}

int main()
{
    scanf("%d", &T);
    while(T--)
    {
        scanf("%lld", &N);
        memset(dp, -1, sizeof dp);
        for(int i=60; i>=0; i--)
            S[60-i]=(N>>i)&1;
        printf("%d\n", rec(0, 0, 0));
    }
    return 0;
}
