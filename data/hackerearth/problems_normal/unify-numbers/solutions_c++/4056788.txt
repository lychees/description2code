#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   long long int t,n,sum=0,i,j;
    cin>>t;
    while(t--)
    {
    	sum=0;
    	cin>>n;
    	long long int a[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum+=a[i];
    	}
    	// j=sum/n;
    	sort(a,a+n);
    	if(n%2!=0)
    	{
    		j=a[n/2];
    	}
    	else
    	{
    		j=a[n/2-1]+a[n/2];
    		j/=2;
    	}
    	 long long int s=0;
    	for(i=0;i<n;i++)
    	{
    		s+=abs(a[i]-j);
    	}
    	cout<<s<<endl;
    	
    }
    
}
