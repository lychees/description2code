#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    for(int i=0; i<t; i++)
    {
    	string w;
    	cin >> w;
    	int l = w.length();
    	for(int j=0; j<l; j++)
    	{
    		int p = int(w[j]);
    		p = p + j;
    		if(p>122)
    		{
    			p = (p % 122) + 96;
    		}
    		char a = char(p);
    		cout << a;
    	}
    	cout << endl;
    }
    return 0;
}
