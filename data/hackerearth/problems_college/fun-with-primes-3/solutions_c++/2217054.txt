#include<bits/stdc++.h>
using namespace std;
int len(int n)
{
    int ret=0;
    while(n>0)
    {
        n/=10;
        ret++;
    }
    return ret;
}
int sumdig(int n)
{
    int lgth,i,ret=0;
    lgth=len(n);
    while(lgth--)
    {
        ret+=(n%10);
        n/=10;
    }
    return ret;
}
int main()
{
    int t,n,i,j,temp,temp1,tmp;
    cin>>t;
    while(t--)
    {
        cin>>tmp;
        n=sumdig(tmp);
        temp=0;
        temp1=1;
        while(n%2==0)
        {
            n/=2;
            temp++;
        }
        temp1=(temp+1);
        for(i=3;i<=sqrt(n);i+=2)
        {
            temp=0;
            while(n%i==0)
            {
                n/=i;
                temp++;
            }
            temp1*=(temp+1);
        }
        if(n>1)
            temp1*=2;
        if(temp1==2)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
