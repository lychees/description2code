(n, m) = map(int, raw_input().split())
list = [-1] * n
list[0] = 0
tree = []
for i in range(n):
	tree.append([])
for i in range(m):
	(a, b) = map(int, raw_input().split())
	tree[a].append(b)
	tree[b].append(a)

queue = [0]
while(queue):
	current = queue.pop(0)
	for next in tree[current]:
		if list[next] < 0:
			list[next] = list[current] + 1;
			queue.append(next)

for i in range(n):
	if i > 0:
	  	print(list[i])
