#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,m;
	set<string> s;
	string str;
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&n,&m);
		for (int i = 0; i < n; i++) {
			cin >> str;
			s.insert(str);
		}
		if (s.size() == n) {
			printf("Yes\n");
		} else {
			printf("No\n");
		}
		s.clear();
	}
	
	return 0;
}
