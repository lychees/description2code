#include <iostream>
#include<stack>
#include<cstring>
using namespace std;

struct X
{
	char c;
	int ind;

};

char s[1000011],a[1000011];
int main()
{
    stack <X> st;
    X temp;
    int t,flag = 0;

    cin >> t;

    while(t--)
    {   flag = 0;

    	cin >> s >> a;
		for(int i = 0; s[i] ; i++)
    	{
    		if(s[i] == 'A' || s[i] == 'B')
    		{
    			temp.c = s[i];
    			temp.ind = i;
    			st.push(temp);
    		}

    	}

        int l = strlen(a);
        l--;
    	while(st.size() && l>=0)
		{
            if(a[l] == '#')
                l--;
            else
            {   if(a[l] == st.top().c && a[l] == 'A')
                {
                    if( l > st.top().ind )
                    {   flag = 1;
                        break;
                    }
                    else
                        {   l--;
                            st.pop();
                        }
                }
                else
                    if(a[l] == st.top().c && a[l] == 'B')
                    {
                        if( l < st.top().ind )
                        {   flag = 1;

                            break;
                        }
                        else
                        {   l--;
                            st.pop();
                        }
                    }
                    else
                    {
                        flag = 1;
                        break;
                    }
            }
		}

		if(flag == 1  )
            cout<<"No\n";
        else
            cout<<"Yes\n";

    }
    return 0;
}
