#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define sc(n) scanf("%lld",&n)
#define mod 1000000007
ll exp(ll a,ll b)
{
	if (b==0)
		return 1;
	else if (b==1)
		return a;
	else if(b%2==0)
		return exp((a*a)%mod,b/2);
	else
		return (a*(exp((a*a)%mod,(b-1)/2)))%mod;
}
int main()
{
	ll a,b,i,j,k,arr[1000005],sum=0,flag=0,t;
	sc(t);
	for (i=2;i<=1000000;i++)
			arr[i]=1;
	for (i=2;i<=1000;i++)
		{
			if (arr[i]==1)
			{
				ll sq=exp(i,2);
				k=i;
			//	printf("sq i%lld %lld\n",sq,i);
				for (j=sq;j<=1000000;j=j+k)
				{
		//			printf("false %lld ",j);
					arr[j]=0;
				}
			}
		}
	while (t--)	
	{	
		sum=0;
		flag=0;
		sc(a);
		sc(b);
		for(i=a;i<=b;i++)	//TLE
		{
			if(arr[i]==1)
				sum=sum+i;
		}
		ll newsq=sqrt(sum);
		for(i=2;i<=newsq;i++)
		{
			if(sum!=0 && sum%i==0)
			{	flag=1;
				break;
			}
		}
		if (sum==0 || flag==1)
				printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
