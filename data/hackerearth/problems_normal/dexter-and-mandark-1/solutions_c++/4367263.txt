#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define pll pair<ll,ll>
#define mp make_pair
#define ff first
#define ss second
#define pb push_back
#define pf pop_front
#define IOS ios::sync_with_stdio(false)
#define MOD 1000000007
int t , n , m ;
int dp[1005][1005];
int main(){

	int i , j , k ;
    cin >> t;
    while(t--){
        cin >> n >> m ;
        ll  res = 0;
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        for(i = 1 ; i <= n;i++){
            dp[i][0]=1;
        }
        for(i = 1 ;i <= n ; i++){
            for(j = 1 ; j <= n ; j++){
                if(i >= j*m)
                dp[i][j]=(dp[i-1][j]+dp[i][j-1])%MOD;
            }
        }
        for(i = 0 ; i <= n ; i++){
            if(n-i>i  && n-i>=i*m)
                res+=(dp[n-i][i]);
            res%=MOD;
        }
        cout << res << endl;
    }
	return 0;
}
