#include<bits/stdc++.h>
using namespace std;
void last_m(char s[],int m)
{
	int i,n=strlen(s);
	if(m>=n)
		cout<<s<<endl;
	else if(m)
		cout<<atol(s+n-m)<<endl;
	else
		cout<<"0\n";
}
int main()
{
    int t,n,m;
    char a[][20]={"1","1","2","720","8401733239439360000"};
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	if(n>=5)
    		cout<<"0\n";
    	else
    		last_m(a[n],m);
    }
    return 0;
}
