n = input()
arr = raw_input().split(" ")
res = str()
for i in range(len(arr)):
    flag = 1
    for j in range(len(arr)):
        if i == j:
            continue
        elif int(arr[i]) % int(arr[j]) == 0:
            flag = 0
            break
    if flag == 1:
        res += " %s" % arr[i]
print res
