#include <iostream>
#include <cstring>
using namespace std;
int L;
long long calc (int N, int K, long long  cache[][202]) {
	//cout<<b<<" "<<e<<" "<<N<<endl;
	if (N == 0 ) return 1;
	if (K>N) return 0;
    if (cache[N][L-K]) return cache[N][L-K];
	cache[N][L-K] =  calc(N-K,K, cache) + calc(N, K+1, cache);
    return cache[N][L-K];
}
int main()
{
    // cout << "Hello World!" << endl;
    int T,N,K,i;
    long long cache[202][202];
    cin>>T;
    while (T--) {
    	cin>>N>>K;
        L = K;
        memset (cache, 0, sizeof(cache));
        for (i=0;i<202;i++) cache[0][i] = 1;
    	cout<<calc(N,K,cache)<<endl;
    }
    return 0;
}
