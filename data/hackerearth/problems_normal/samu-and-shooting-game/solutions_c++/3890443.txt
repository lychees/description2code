#include<bits/stdc++.h>
using namespace std;
#define MAX 1010
double dp[MAX][MAX];
int main(){
	int ts;
	cin>>ts;
	for(int inst=0; inst<ts; inst++){
		int x,y,n,w, P1,P2;
		cin>>x>>y>>n>>w>>P1>>P2;
		for(int t=0;t<=n;t++){
			for(int p=0;p<=w;p++){
				dp[t][p]=0;
			}
		}
		double p1 = P1/100.0;
		double p2 = P2/100.0;
		for (int t=0;t<=n;t++)
      		dp[t][0] = 1;
    	for (int p=1;p<=w;p++)
      		dp[0][p] = 0;
    	for (int t=1;t<=n;t++){
      		for (int p=1;p<=w;p++) {
        		dp[t][p] = max(p1*dp[t-1][max(p-x,0)] + (1-p1)*dp[t-1][max(p,0)], p2*dp[t-1][max(p-y,0)] + (1-p2)*dp[t-1][max(p,0)]);
      		}
      	}
      	cout.precision(6);
      	cout << setiosflags (ios::fixed);
        cout<<dp[n][w]*100<<"\n";
	}
	return 0;
}
