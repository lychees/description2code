#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <string>
using namespace std;

struct Node
{
	char ch;
	struct Node *child[26];
};

int cnt=1;

void insertIntoTrie(Node *root, string str)
{
	int l = str.length();
	for(int i=0; i<l; ++i)
	{
		char ch = str[i];
		if(root->child[ch-97] == NULL)
		{
			root->child[ch-97] = (Node *)calloc(1, sizeof(Node));
			++cnt;
		}
		root = root->child[ch-97];
	}
}

int main()
{
    int n;
    cin >> n;
   
   	Node *root = (Node *)calloc(1, sizeof(Node));
   	
    while(n--)
    {
    	string str;
    	cin >> str;
    	insertIntoTrie(root, str);
   	}
   	
   	cout<<cnt<<endl;
   	
    return 0;
}
