#include <iostream>
using namespace std;
#include <vector>
#include <map>
#include <algorithm>

int main()
{
	int N,Q;
	vector<int> time;
	vector<int> score;

	cin >> N >> Q;
	int K[Q];
	int temp;

	for( int i = 0; i < N; i++ )
	{
		cin >> temp;
		time.push_back(temp);
	}

	for( int i = 0; i < N; i++ )
	{
		cin >> temp;
		score.push_back(temp);
	}
	
	for( int i = 0 ; i < Q; i++ )
		cin >> K[i];


	sort(time.begin(),time.end());
	sort(score.begin(),score.end());

	for( int i = 0 ; i < Q; i++ )
	{
		int res = 0;
		for( int j = 0 ; j < K[i]; j++ )
		{
			res = res + time[N-1-j];
		}
		cout << "\n" << res;

	}


	return 0;
}
