#include<bits/stdc++.h>
using namespace std;
#define MAX 1000007
int lp[MAX] , primes[MAX];
int pr;
long long int sum[MAX];
void siever()
{
		int i;
		int j;
		pr=0;
		for(i=2;i<MAX;++i)
		{
				if(lp[i] == 0)
				{
						primes[pr++]=i;
						for(j=i;j<MAX;j+=i)
								lp[j]=i;
				}
		}
}
int main()
{
	ios_base::sync_with_stdio(false);
		siever();
		int i;
		sum[0]=primes[0];
		for(i=1;i<pr;++i)
		{
				sum[i] = sum[i-1]+primes[i];
		}
		int t;
		cin>>t;
		while(t--)
		{
				int l,r;
				cin>>l>>r;
				int a,b;
				a = lower_bound(primes,primes+pr,l) - primes;
				b = lower_bound(primes,primes+pr,r) - primes;
				//cout<<"a="<<a<<" b="<<b<<endl;
				if(primes[b] != r) b--;
				if(a == 0) cout<<sum[b]<<endl;
				else cout<<sum[b]-sum[a-1]<<endl;

		}
}
