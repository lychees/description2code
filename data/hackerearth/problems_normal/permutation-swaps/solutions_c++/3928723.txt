using namespace std;
int parent[100005];
void ini()
{
    for(int i=0;i<=100000;i++)
        parent[i]=i;
}
int root(int a)
{
    while(a!=parent[a])
    {
        parent[a]=parent[parent[a]];
        a=parent[a];
    }
    return a;
}
void uni(int a,int b)
{
    int p=root(a);
    int q=root(b);
    if(p==q)
    return;
    parent[p]=q;
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
   int t;
   cin>>t;
   while(t--)
   {
       ini();
       int n,m;
       cin>>n>>m;
       int arr[n+1],brr[n+1];
     
       for(int i=1;i<=n;i++)
        cin>>arr[i];
       for(int i=1;i<=n;i++)
        cin>>brr[i];
       for(int i=1;i<=m;i++)
       {
       	int a,b;
           cin>>a>>b;
           uni(arr[a],arr[b]);
       }
       int flag=0;
       for(int i=1;i<=n;i++)
       {
 
           if(root(arr[i])!=root(brr[i]) )
           {
               flag=1;
               break;
           }
       }
        if(flag==0)
            cout<<"YES\n";
        else
            cout<<"NO\n";
   }
}
