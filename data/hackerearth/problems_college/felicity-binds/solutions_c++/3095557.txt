#include <bits/stdc++.h>
#define M 1000000007
#define ll long long

using namespace std;

int main() {
    ll t, n, fs;
    cin >> t;
    while(t--) {
        cin >> n;
        fs = 0;
        int m[n], w[n];
        for (int i = 0; i < n; i++) {
            cin >> m[i];
        }
        for (int i = 0; i < n; i++) {
            cin >> w[i];
        }
        sort(m, m + n);
        sort(w, w + n);
        for (int i = 0; i < n; i++) {
            fs += m[i] * w[i];
        }
        cout << fs << endl;
    }
    return 0;
}
