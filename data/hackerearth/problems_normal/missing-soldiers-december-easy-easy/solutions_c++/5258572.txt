#include <iostream>
#include <algorithm>
#define in(n) int n; cin>>n;
#define ll long long int
#define For(a,b) for(int i=a;i<b;i++)
using namespace std;

int main()
{
    in(n)
    vector<pair<int,int>>vec;
    For(0,n){
    	in(a) in(b) in(c)
    	vec.push_back(make_pair(a,a+c));
    }
    sort(vec.begin(),vec.end());
    int ans=0,len=vec.size(),temp=1;
    for(int i=0;i<len;i++){
    	temp=max(temp,vec[i].first);
    	if(vec[i].second>=temp){
    	ans+=(vec[i].second-temp+1);
    	temp=vec[i].second+1;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
