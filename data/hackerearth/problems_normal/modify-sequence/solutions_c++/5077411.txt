#include <iostream>
using namespace std;
 
int main()
{
	int t,i;
	cin >> t;
	int arr[t],odd=0,even=0;
	for(i=0;i<t;i++)
		cin >> arr[i];
	for(i=0;i<t;i++)
	{
		if(i%2==0)
		even=even+arr[i];
		else
		odd=odd+arr[i];
	}
	if(even==odd)
    cout << "YES" << endl;
    else
    cout << "NO" << endl;
    return 0;
}
