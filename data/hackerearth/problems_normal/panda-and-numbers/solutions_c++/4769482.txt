int ans[maxn];

     int cur = i;

     {
        int d = cur%10;
        cur/=10;
        if(i-d*d>=1 && ans[i-d*d]==1)
        {
            ans[i] = 1;
            break;
        }
     }
  }

  {
     cin>>n;

  }
