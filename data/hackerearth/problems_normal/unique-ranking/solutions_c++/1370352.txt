#include<bits/stdc++.h>
using namespace std;
int A[1000+5][20+2],B[20+2],C[1000+10]={0};
vector<int > zz;
int main()
{
    int n,k,q,s;
    scanf("%d %d",&n,&k);
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<k;j++)
        {
            scanf("%d",&A[i][j]);
        }
    }
    scanf("%d",&q);
    while(q--)
    {   for(int i=0;i<=n;i++)
            C[i]=-1;
        int mini;
        scanf("%d",&s);
        for(int i=0;i<s;i++)
            scanf("%d",&B[i]);
        for(int j=0;j<s;j++)
        { mini=INT_MAX;
            int k=B[j]-1;
            for(int i=0;i<n;i++)
            {      if(C[i]!=-1)
                        continue;
                 if(A[i][k]==mini)
                {
                    zz.push_back(i);
                }
                else if(A[i][k]<mini)
                {
                    mini=A[i][k];
                    zz.clear();
                    zz.push_back(i);
                }
            }
            for(int m=0;m<zz.size();m++)
            {
                C[zz[m]]--;
            }
            for(int i=0;i<n;i++)
                C[i]++;
            zz.clear();
        }
        for(int i=0;i<n;i++)
        {
            if(C[i]==-1)
            {
                printf("%d\n",i+1);
                break;
            }
        }

    }



return 0;
}
