#include <iostream>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
    int N,C;
    cin>>N>>C;
    C--;
    vector<int> v;
    int i=1;
    while(i<=N)
    {
       v.push_back(i++);
    }
    
     i=0;
    
    while(v.size()!=1)
    {
    	i=(i+C)%v.size();
    	//cout<<v[i];
    	v.erase(v.begin()+i);
    	i=i%v.size();
    	
    }
    cout<<v.front();
    return 0;
}
