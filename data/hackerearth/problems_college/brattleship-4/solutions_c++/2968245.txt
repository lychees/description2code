#include<iostream>
using namespace std;
int main()
{
    int t, tt;
    cin>>t;
    tt=t;
    while(t--)
    {
       	int r, c, w;
    	cin>>r>>c>>w;
       	if(w==1 || w==c)
    	cout<<"Case #"<<tt-t<<": "<<r*c<<endl;
    	else if(w==(c/2) || w>(c/2))
    	cout<<"Case #"<<tt-t<<": "<<r*(w+1)<<endl;
    	else if(c%2!=0 && w<(c/2))
    	cout<<"Case #"<<tt-t<<": "<<r*((c/w)+w)<<endl;
    	else if(c%2==0 && c%w==0)
    	cout<<"Case #"<<tt-t<<": "<<r*((c/w)+w-1)<<endl;
    }
return 0;
}
