n=int(raw_input())
si=map(int,raw_input().split())
tot = sum(si)
batch=int(raw_input())
z,inc=0,0
for i in xrange(batch):
    b=map(int,raw_input().split())
    p=b[b[0]+1]
    old=0
    z=z+p
    for j in b[1:b[0]+1]:
        old = old + si[j-1]
    if(old<p):
        inc=p-old+inc        
if(z>tot):
    print z-tot
else:
    print inc
