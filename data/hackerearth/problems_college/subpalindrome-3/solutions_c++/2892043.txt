#include <bits/stdc++.h>
using namespace std;
#define ll long long 

ll ans;
bool ispalin(string s) {
	int i=0;
	int j=s.length()-1;
	while(i<j && s[i++] == s[j--]);
	if(i==j)
		return true;
	else
		return false;
}
void palin(string s,string q,int i) {
	
	if(i==s.length()) {
		if(ispalin(q))
			ans++;
		return ;
	}
	palin(s,q+s[i],i+1);
	palin(s,q,i+1);
}

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string s;
    	cin>>s;
    	ans=0;
    	ans = s.length();
    	ll dp[ans+1][ans+1];
    	memset(dp,0,sizeof(dp));
    	
    	for(int i=0;i<=ans;i++)
    		dp[i][i] = 1;
    		
    	for(int len=2;len<=s.length();len++) {
    		for(int j = len-1;j<s.length();j++) {
    			int i = j-len+1;
    			if(len==2) {
    				dp[i][j] = 2;
    				if(s[i]==s[j])
    					dp[i][j] ++;
    				//cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
    			}else {
    				dp[i][j] = dp[i][j-1] + dp[i+1][j] - dp[i+1][j-1];
    				if(s[i] == s[j])
    					dp[i][j] += (dp[i+1][j-1] +1);
    				//cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
    			}
    		}
    	}
    	
    	cout<<dp[0][ans-1]<<endl;
    }
    return 0;
}
