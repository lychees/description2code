#include<vector>
#include<algorithm>
#include <iostream>
using namespace std;

int main()
{
    int i,n,q,x,check;
    int arr[10000000];
    cin>>n>>q;
    for(i=0;i<n;i++)
    {
    	cin>>arr[i];
    	
    }
    vector<int>v(arr,arr+n);
    sort(v.begin(),v.end());
    for(i=0;i<q;i++)
    {
    	cin>>x;
    		if(binary_search(v.begin(),v.end(),x))
    		{
    			check=1;
    		}
    		else
    		{
    			check=0;
    		}
    	if(check==1)
    	{
    			cout<<"YES"<<endl;
    	}
    	else if(check==0)
    	{
    			cout<<"NO"<<endl;
    	}
    }
    return 0;
}
