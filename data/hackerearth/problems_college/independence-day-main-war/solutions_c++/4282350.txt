#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		float x,y,h=0,k=0;
		int n,nc;
		cin>>n;
		nc=n;
		while(nc--)
		{
			cin>>x>>y;
			h+=x;
			k+=y;
		}
		cout<<"("<<fixed<<setprecision(2)<<h/n<<" , "<<k/n<<")"<<endl;
	}
    return 0;
}
