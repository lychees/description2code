#include <iostream>
using namespace std;
#include<vector>
#include<map>
#include<algorithm>


int is_conso(char i)
{
		if(i=='a' || i=='e' || i=='i' || i=='o' || i =='u' || i=='y')
		return 0;
		return 1;
}


int main() 
{
	int t;
	string n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int cons=0;
		int vowels =0;
		int flag=0;
		for(int i=0;i<n.length();i++)
		{
		        if(!is_conso(n[i]))
				vowels++;
			else
			{
				cons++;
				if(i>0 && is_conso(n[i-1]) && i<n.length()-1 && is_conso(n[i+1]))
				{
					flag=1;
					break;
				}
			}
		}
		if(flag==1 || cons > vowels)
		cout<<"hard"<<endl;
		else
		cout<<"easy"<<endl;

		
	}
	// your code goes here
	return 0;
}
