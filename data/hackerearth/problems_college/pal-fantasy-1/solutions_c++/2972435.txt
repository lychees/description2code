  #include <iostream>
    using namespace std;
    bool dp[51][51];
    int main() {
    	int t;
    	cin >> t;
    	while(t--){
    		string s;
    		cin>>s;
    		int l=s.length();
    		for (int i = 0; i < l ; ++i)
    		{dp[i][i] = true;}
    		for (int i = l; i >= 0 ; i--)
    		{for (int j = i+1; j < l; ++j)
    			{   				if(s[i] == s[j]){
    					if(i+1 == j) dp[i][j] = true;
    					else{
    						dp[i][j] = dp[i+1][j-1] ;
    					}
    				}    				else{
    					dp[i][j] = false;
    				}
    			}	
    		}
    		int i;
    		for ( i = 0; i < l; ++i){
    			if(dp[i][l-1]){
    				break;
    			}}cout << l + i << endl;
    	}	return 0;
    }
