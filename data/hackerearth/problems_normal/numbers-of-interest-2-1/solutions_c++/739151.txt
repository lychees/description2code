long long int prime[1000002],prime2[1000002],a[1000002],b[1000002];
int sieve()
{
	long long int i,j,k,nu=0;;
	long long int m,n,u,l;
	n=1000002;
	for(i=0;i<n;i++)
	{
		prime[i]=0;
	}
    
		for(i=2;i*i<=n;i++)
		{
			if(!prime[i])
			{
				for(j=i*i;j<=n;j=j+i)
				{
					if(!prime[j])
					prime[j]=i;					
				}
			}
		}
		for(i=2;i<n;i++)
		{
			if(!prime[i])
			prime[i]=i;
			
		}
/*		for(i=0;i<20;i++)
		printf("%d ",prime[i]);
		printf("\n");
		
	*/	
	return nu;
}
