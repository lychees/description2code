#include <iostream>
using namespace std;

int main()
{
    int t;
    
    scanf("%d",&t);
    while(t--)
    {
    	long long int n,a1=1,a2=2,ans,i;
    	scanf("%lld",&n);
    	if(n==1) ans=1;
    	else  if(n==2) ans=2;
    	else
    	{
    		for(i=3;i<=n;i++)
    		{
    			ans=a1+a2;
    			a1=a2;
    			a2=ans;
    		}
    	}
    	printf("%lld\n",ans);
    }
    return 0;
}
