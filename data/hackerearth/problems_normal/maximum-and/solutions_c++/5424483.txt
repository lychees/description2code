#include <iostream>
using namespace std;
#define ll long long int
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll a,b;
    	cin>>a>>b;
    	ll ans=0;
    	if(b-a==1)
    	{
    		ans = a&b;
    	}
    	
    	else{
    		ll temp=b&(b-1)	;
    		if(temp==0)
    		{
    		//	cout<<"hi"<<endl;
    			ans = (b-1) & (b-2); 
    		}
    		else
    		{
    		//	cout<<"hi"<<endl;
    			ans = max((b)&(b-1),(b-1)&(b-2));
    		}
    	}cout<<ans<<endl;
    	
    	
    }
    return 0;
}
