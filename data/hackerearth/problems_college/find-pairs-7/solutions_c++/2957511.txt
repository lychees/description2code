#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	do
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		int count=0;
		for(int i=0;i<n;i++)
		{
			for(int j=i;j<n;j++)
			{
				if(a[i]==a[j])
				count++;
			}
		}
	    cout<<count<<"\n";
		t--; 
	}while(t>0);
}
