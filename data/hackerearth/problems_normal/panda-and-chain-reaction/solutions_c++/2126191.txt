
#include <iostream>
#include<stdio.h>
using namespace std;
const int MAX = 1000003;
int main()
{
	long fact_hash[MAX+1];
	fact_hash[0]=1;
	for(int i=1;i<=MAX;i++){
		fact_hash[i] = (i*fact_hash[i-1])%MAX;
	}
	
	int t;
	scanf("%d",&t);
	while(t--){
		
		unsigned long long n,x;
		scanf("%lld %lld",&n,&x);
		if(n >=MAX)
			printf("0\n");
		else{
		x = x %MAX;
		
		printf("%lld\n",(fact_hash[n]*x)%MAX);
		}
		
	}
    
    return 0;
}
