#include <bits/stdc++.h>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {   int flag=0;
    	string ch;
    	cin>>ch;
    	string ch2=ch;
    	reverse(ch2.begin(),ch2.end());
    	for(int i=1;i<ch2.length();i++)
    	{
    		if(abs(ch[i]-ch[i-1])!=abs(ch2[i]-ch2[i-1]))
    		{flag=1;break;}
    	}
    	if(flag)cout<<"Not Funny"<<endl;
    	else cout<<"Funny"<<endl;
    	
    }
    return 0;
}
