#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int tc;
	cin >> tc;
	while(tc-- > 0)
	{
		long long int a,b,c;
		cin >> a >> b >> c;
		double x = pow(a,b);
		long long int closest = floor(x);
		long long int ans = closest-closest%c;
		
		double rem = fmod(x,c);
		if(2*rem <= c)
			ans = x - rem;
		else
			ans = x - rem + c;
		
		cout << ans << endl;
	}
    return 0;
}
