#include <bits/stdc++.h>
using namespace std;

int main() {
  int tc;
  cin >> tc;
  while (tc--) {
    long long int n;
    cin >> n;
    long long int i = 1, ct = 0;
    while (n > 0) {
      i = 1;
      while (n > i) {
        i = i * 2;
      }
      if (n == i) {
        ct++;
        break;
      } else {
        i = i / 2;
        n = n - i;
        ct++;
      }
    }
    cout << ct << endl;
  }
  return 0;
}
