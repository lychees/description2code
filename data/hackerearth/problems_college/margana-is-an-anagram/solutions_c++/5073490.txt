#include "bits/stdc++.h"
using namespace std;
bool func(string a,string b)
{
	vector<long int> word1(26,0),word2(26,0);
	for(long int i=0;i<a.length();i++)
		word1[a[i]-97]++;
	for(long int i=0;i<b.length();i++)
		word2[b[i]-97]++;
	for(long int i=0;i<26;i++)
		if(word1[i]!=word2[i])
			return false;
	return true;
}
int main()
{
	long int tc;
	cin >> tc;
	while(tc--)
	{
		string a,b;
		cin >> a >> b;
		if(func(a,b))
			cout << "Yes\n";
		else
			cout << "No\n";
	}
	return 0;
}
