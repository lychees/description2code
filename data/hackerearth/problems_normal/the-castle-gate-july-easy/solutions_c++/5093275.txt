#include <iostream>
using namespace std;

int main()
{
	int count = 0,t, n;
	cin>>t;
	for(int i = 1; i<=t; i++)
	{
		cin>>n;
		for (int j =1; j<=n; j++)
		{
			for( int k = j+1; k<=n; k++)
			{
				if( (j^k) <= n )
				{
				   count++;	
				}
				
			}
		}
		cout<<count<<endl;
		count = 0;
	}
    return 0;
}
