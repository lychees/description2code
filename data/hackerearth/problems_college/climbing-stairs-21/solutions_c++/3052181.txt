#include <iostream>
using namespace std;

unsigned long long int lk[1002];

unsigned long long int fib(int n){
	
	if(n<=1) return 1;
	if(lk[n]!=0) return lk[n];
	lk[n]=fib(n-1)+fib(n-2);
	return lk[n];
}	
int main()
{
 	int t,n;
 	cin >> t;
 	while(t--){
 		cin >> n;
 		cout << fib(n) << endl;
 	}
}
