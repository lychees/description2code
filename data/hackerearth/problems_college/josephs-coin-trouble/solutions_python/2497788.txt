n = int(raw_input())
a = map(int, raw_input().split())
s = [[0 for i in range(len(a))] for i in range(n+1)]
for i in range(len(a)):
	s[0][i] = 1
x = 0
y = 0
for i in range(1, n+1):
	for j in range(len(a)):
		if i - a[j] >= 0:
			x = s[i-a[j]][j]
		else:
			x = 0
		if j >= 1:
			y = s[i][j-1]
		else:
			y = 0
		s[i][j] = x+y

print s[n][len(a)-1]
