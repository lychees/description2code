#include <iostream>
using namespace std;

int main()
{
	unsigned int test_cases=0;
	cin>>test_cases;
	unsigned int heap=0;
	unsigned int value=0;
	while(test_cases--)
	{
	unsigned int mini = 1001, maxi = 0;
			cin>>heap;
			unsigned int temp = heap;
			while(temp--)
			{
				cin>>value;
				mini = min(mini,value);
				maxi += value;
			}
		cout<<mini-1<<" "<<(maxi-heap)<<endl;

	}
}
