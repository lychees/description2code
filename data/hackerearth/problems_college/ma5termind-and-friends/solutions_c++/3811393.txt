#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{	
    int t;
    cin>>t;
    while(t--)
    {
    vector< pair< pair<int,int> , int> > v(100001); 
    int n,q;
    cin>>n>>q;
    for(int i=1;i<=q;i++)
    {
        int l,r,c;
        cin>>l>>r>>c;
        v[i]={{l,r},c};
    }
    int q2;
    cin>>q2;
    int q1[100001]={0};
    while(q2--)
    {
       int l,r;
       cin>>l>>r;
       q1[l]++;
       q1[r+1]--;
    }
    for(int i=1;i<=q;i++)
        q1[i]+=q1[i-1];
    ll ans[100001]={0};
    for(int i=1;i<=q;i++)
    {
        int l=v[i].first.first;
        int r=v[i].first.second;
        int c=v[i].second;
        ans[l]+=1LL*c*q1[i];
        ans[r+1]-=1LL*c*q1[i];
    }
    for(int i=1;i<=n;i++)
        ans[i]+=ans[i-1];
    for(int i=1;i<=n;i++)
        cout<<ans[i]<<" ";
    cout<<endl;
    }
    return 0;
}
