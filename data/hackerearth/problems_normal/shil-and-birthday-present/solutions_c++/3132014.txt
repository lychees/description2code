

    #include <iostream>
    #include<algorithm>
    using namespace std;
     
    int main()
    {
        long long int n;
        cin>>n;
        long long int a[n],i;
        for(i=0;i<n;++i)
        cin>>a[i];
        sort(a,a+n);
        long long int ctr=1;
        for(i=1;i<n;++i)
        {
        	if(a[i]!=a[i-1])
        	++ctr;
        }
        cout<<(ctr*(ctr+1))/2-ctr<<"\n";
        return 0;
    }
