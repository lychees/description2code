#include <iostream>
using namespace std;
#include <algorithm>
 
struct user{
    long int start;
    long int time1;
    int call;
};
 
bool compare(user a, user b){
    return a.start<b.start;
}
 
int main()
{
    int n,m;
    cin >> n >> m;
    long int texi[m];
    for(int i = 0; i < m; i++){
        texi[i] = 0;
    }
    user* ar=new user[n];
    for(int i = 0; i < n; i++){
        long int start11, time11;
        cin >> start11 >> time11;
        ar[i].start = start11;
        ar[i].time1 = time11;
        ar[i].call = i;
    }
    sort(ar,ar+n,compare);
    int print1[n];      // to store the indices to print
 
    for(int i = 0; i < n; i++){
        long int start11, time11;
        int call;
        start11 = ar[i].start;
        time11 = ar[i].time1;
        call = ar[i].call;
 
        int index;
        bool flag = false;
        
        for(int j = 0; j < m; j++){
            if(texi[j] < start11){flag = true; texi[j] = start11 + time11-1;index = j+1;break;} 
        }
        if(flag) print1[call] = index;
        else print1[call] = -1;
 
    }
    
    for(int i = 0; i < n; i++){
    	cout << print1[i] << " ";
    }
cout << endl;
    
    
    return 0;
}
