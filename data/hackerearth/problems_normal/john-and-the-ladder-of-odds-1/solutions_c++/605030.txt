#include<bits/stdc++.h>

using namespace std;
#define ull unsigned long long
#define FOR(i,a,n)for(i=a;i<n;i++)
#define FORE(i,a,n)for(i=a;i<=n;i++)
#define MOD 1000000009
#define ll long long
ull fibo(long long int n)
{
    ull ans[2][2],result[2][2]={{1,0},{0,1}},fibo[2][2]={{1,1},{1,0}};
    int i,j,k;
    while(n)
    {
        memset(ans,0,sizeof(ans));

        if(n&1)
        {
            FOR(i,0,2)
                FOR(j,0,2)
                	//cout<<"x="<<ans[i][j]<<" "<<endl;
                    FOR(k,0,2)
                        {
                            ans[i][j]=(ans[i][j]+result[i][k]*fibo[k][j])%MOD;

                        }
            FOR(i,0,2)
                FOR(j,0,2)
                    result[i][j]=ans[i][j];
        }
         memset(ans,0,sizeof(ans));
        FOR(i,0,2)
                FOR(j,0,2)
                    FOR(k,0,2)
                        {
                            ans[i][j]=(ans[i][j]+fibo[i][k]*fibo[k][j])%MOD;
                        }
        FOR(i,0,2)
                FOR(j,0,2)
                    fibo[i][j]=ans[i][j];
        n/=2;
    }

    return result[0][1]%MOD;
}
int main()
{
	int t;
	long long n,max=1000000009;

	while(t--)
	{
		scanf("%lld",&n);
		long long  x=fibo(n);
		long long y=fibo(n+1);
		x*=y;
		x%=MOD;
		printf("%lld\n",x);
	}
    return 0;
}
