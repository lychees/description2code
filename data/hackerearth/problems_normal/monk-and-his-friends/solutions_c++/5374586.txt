#include<iostream>
#include<map>
using namespace std;
map<long long,long long> ma;
long long t,n,m,a;
int main()
{   cin>>t;
    while(t--)
    {ma.clear();
    	cin>>n>>m;
      for(int i=1;i<=n;i++)
      {cin>>a;
        ma.insert(pair<long long,long long>(a,i));
      }
      map<long long,long long>::iterator it;
      for(int i=0;i<m;i++)
      {cin>>a;
       it=ma.find(a);
       if(it->first==a&&it!=ma.end())
       cout<<"YES"<<endl;
       else
       {cout<<"NO"<<endl;
       ma.insert(pair<long long,long long>(a,i));
       }
      }
    }
    return 0;
}
