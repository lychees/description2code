#include <iostream>

using namespace std;

int main(){
    int n,a[100][26];
    string s;
    cin>>n;
    for(int i=0;i<n;i++){
        for(int j=0;j<26;j++){
            a[i][j]=0;
        }
    }
    for(int i=0;i<n;i++){
        cin>>s;
        int len = s.size();
        for(int j=0;j<len;j++){
            a[i][(int)s[j]-97]++;
        }
    }
    int count=0;
    for(int i=0;i<26;i++){
        int cc=0;
        int x=a[0][i];
        if(x>0){
            cc=1;
            for(int j=1;j<n;j++){
                if(a[j][i]>0){
                    cc++;
                }
            }
        }
        if(cc==n){
            count++;
        }
    }
    cout<<count<<"\n";
    return 0;
}
