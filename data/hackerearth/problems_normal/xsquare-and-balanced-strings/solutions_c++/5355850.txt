#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int hash[26];
    while(t--){
    	string s;
    	cin>>s;
    	int len = s.length();
    	memset(hash,0,sizeof(hash));
    	if(len%2==1) cout<<"-1"<<endl;
    	else
    	{
    	   for(int i=0;i<len;i++)
    	     {
    		  hash[s[i]-'a']++;
    	     }
    	  int no=1;
    	  for(int i=0;i<26;i++)
    	     {
    		   if(hash[i]%2){
    		   	no = 0;
    			cout<<"-1"<<endl;
    			break;
    		  }
    	    }
    	   if(no) cout<<"1"<<endl;
    	}
    }
    return 0;
}
