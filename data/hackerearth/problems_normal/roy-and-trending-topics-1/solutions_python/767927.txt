n = int(raw_input())
ans = []
for i in range(n):
	lst = map(int,raw_input().split())
	ans.append([lst[2]*50+lst[3]*5+lst[4]*10+lst[5]*20-lst[1],lst[0],lst[1]])
ans = sorted(ans,reverse = True)
if n<5:
	for a in ans:
		print a[1],a[2]+a[0]
else:
	for i in range(5):
		print ans[i][1],ans[i][2]+ans[i][0]
