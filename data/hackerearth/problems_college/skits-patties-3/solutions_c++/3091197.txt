#include<bits/stdc++.h>
using namespace std;

int main()
{
    //6 9 20
    ios::sync_with_stdio(false);
    long long a,b,c,u,left;
    int t;
    bool flag;
    cin>>t;
    while(t--)
    {
        cin>>u;
        flag=false;
        for(a=0;a<=u;a+=6)
        {
            for(b=0;a+b<=u;b+=9)
            {
                left = u-a-b;
                if(left%20==0)
                {
                    flag=true;
                    break;
                }
                if(flag)
                    break;
            }
            if(flag)
                break;
        }
        if(flag)
            cout<<"True\n";
        else
            cout<<"False\n";
    }
}
