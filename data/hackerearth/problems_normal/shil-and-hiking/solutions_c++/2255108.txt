#include <iostream>
using namespace std;

long long x[1010],y[1010];
int main()
{
    int n,ans=0,pos;
    cin>>n;
    for(int i=1;i<=n;i++) cin>>x[i]>>y[i];
    for(int i=1;i<=n-1;i++) {
    	pos = i+1;ans++;
    	for(int j=i+2;j<=n;j++) {
    		if(((y[pos]-y[i])*(x[j]-x[pos])-(x[pos]-x[i])*(y[j]-y[pos]))<=0) {
    			ans++;pos=j;
    		}
    	}
    }
    cout<<ans;
    //cout << "Hello World!" << endl;
    return 0;
}
