#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>
using namespace std;

int main()
{
    int n,qq,k,i,j;
    scanf("%d%d",&n,&qq);
    long long a;
    long long p[n],q[n];
    long long S[n+1];
    S[0]=0;
    vector<pair<long long, long long> > P;
    for(i=0;i<n;i++)
    	scanf("%lld",&p[i]);
    for(i=0;i<n;i++)
    	scanf("%lld",&q[i]);
    for(i=0;i<n;i++)
    	P.push_back(make_pair(p[i],q[i]));
    sort(P.begin(),P.end());
    for(i=0;i<n;i++)
    	S[i+1]=S[i]+P[n-i-1].first;
    while(qq--)
    {
    	scanf("%d",&k);
    	printf("%lld\n",S[k]);
    }
    
    return 0;
}
