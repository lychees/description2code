#include<cstdio>
#define MOD 1000000007
int main(){
    int cs;
    scanf("%d",&cs);
    while(cs--){
        long long  n,k;
        scanf("%lld%lld",&n,&k);
        long long ans=1;
        for(int i=k;i>=k-n+1;i--){
            ans=(ans*i)%MOD;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
