#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        int l=s.size();
        int a[l];
        for(int i=0;i<l;i++)
        {
            a[i]=s[i]-'a';
        }
        int p;
        for(int i=0;i<l;i++)
        {
            p=a[i]+i;
            p=p%26;
            cout<<char('a'+p);
        }
        cout<<endl;
    }
    return 0;
}
