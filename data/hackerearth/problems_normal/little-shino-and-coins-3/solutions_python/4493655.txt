K = int(raw_input())
S = raw_input()
Str = []
for i in xrange(len(S)): Str.append(ord(S[i]) - 97)

Ret = 0
for i in xrange(len(S)):
    Flag = [0] * 26
    Cnt = 0
    for j in range(i,len(S)):
        if Flag[Str[j]] == 0:
            Cnt += 1
            Flag[Str[j]] = 1
        if Cnt > K: break
        elif Cnt == K: Ret += 1
print Ret
