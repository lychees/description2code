#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	int a[100];
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    sort(a,a+n);
    	int j=0;
    	int count=0;
    	for(int i=0;i<n;i=i+j)
    	{
    		if(i==n-1)
    		{
    			count++;
    			break;
    		}
    		else if(a[i]+1==a[i+1] && i!=n-2)
    		{
    			count++;
    			j=2;
    		}
    		else if(a[i]+1==a[i+1] && i==n-2)
    		{
    			count++;
    			break;
    		//	j=1;
    		}
    		else
    		{
    			count++;
    			j=1;
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
