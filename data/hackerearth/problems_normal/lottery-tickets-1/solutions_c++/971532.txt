#include <iostream>
#include<algorithm>
using namespace std;
 
int start[10000],finish[10000],profit[10000],f[10000];
int main()
{
   // cout << "Hello World!" << endl;
   int t,i,j;
   cin>>t;
   while(t--)
   {
   	int n;
   	float result=0.0;
   	cin>>n;
   	for(i=0;i<n;i++)
   	{
   		cin>>start[i]>>finish[i]>>profit[i];
   		
   	}
   	for(i=0;i<n;i++)
   	{
   		for(j=i+1;j<n;j++)
   		if(finish[i]>finish[j])
   		{
   			swap(finish[j],finish[i]);
   			swap(start[j],start[i]);
   			swap(profit[j],profit[i]);
   		}
   	}
   	for(i=0;i<n;i++)
   	{ f[i]=profit[i];
   		for(j=0;j<i;j++)
   		{
   			if(start[i]>=finish[j] && f[j]+profit[i]>f[i])
   			f[i]=f[j]+profit[i];
   		}
   	if (f[i]>result) 
                result=f[i]; 
   		
   	}
printf("%.2lf\n",double(result)/100.0); 
   	result=0;
   	
   }
   
   
   
   
   return 0;
}
