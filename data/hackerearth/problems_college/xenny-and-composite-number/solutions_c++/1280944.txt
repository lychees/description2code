#include <iostream>
using namespace std;
#define primemax 1000000
long long int prime[primemax];
void cprime(){	for(int i=2;i<primemax;i++)	if(!prime[i])	for(int j=2*i;j<primemax;j+=i)	prime[j]=1;}
int main()
{
    int t;
    cin>>t;
    cprime();
    while(t--)
    {
    	long no;
    	cin>>no;
    	for(int i=no+1;i<primemax;i++)
    		if(prime[i]){
    			cout<<i<<"\n";
    			break;
    		}
    }
}
