#include<bits/stdc++.h>
using namespace std;;

int a[5000];

int cf[1000005]={0};

int main()
{
	int n,k;
	int i,j,x,ctr;
	cin>>n>>k;
	for(i=0;i<n;i++)
	  {
	  	cin>>x;
	  	a[i]=x;
	  	cf[x]++;
	  }
   for(i=1;i<=1000001;i++)
     cf[i]+=cf[i-1];
    long long ans=0;
    
   for(i=0;i<n;i++)
    {
    	for(j=i+1;j<n;j++)
    	  {
    	  	int up=k/(a[i]*1LL*a[j]);
    	  	ctr=cf[up];
    	  	if(a[i]<=up)
    	  	   ctr--;
    	  	if(a[j]<=up)
    	  	    ctr--;
    	  	ans+=ctr;
    	  }
    }
   cout<<ans/3<<endl; 
    
}
