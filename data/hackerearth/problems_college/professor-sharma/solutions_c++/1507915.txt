#include <bits/stdc++.h>

int main()
{
	int testcase, count=1;
	scanf("%d", &testcase);
	while(testcase)
	{
		long long int X, Y;
		scanf("%ld %ld", &X, &Y);
		double result, val;
		long long int ans, answer;
		int i=1;
		int digits = log10(Y) + 1;
		while(1)
		{
			result = i*log10(X);
			ans = result;
			val = result - ans;
			answer = pow(10, val)*pow(10, digits-1);
			if(answer == Y)
			{
				printf("Case %d: %d\n", count, i);
				break;
			}
			i++;
		}
		count++;
		testcase--;
	}
	return 0;
}
