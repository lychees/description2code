#include <iostream>
#include <algorithm>
using namespace std;
int arr[100001];
void disp(int N){
	for(int i=0;i<N;i++)
		printf("%d ",arr[i]);
	printf("\n");	
}
int main(){
	int T;scanf("%d",&T);
	int N;
	while(T--){
		scanf("%d",&N);
		for(int i=0;i<N;i++)
			scanf("%d",&arr[i]);
		std::sort(arr,arr+N);
		disp(N);
	}
    return 0;
}
