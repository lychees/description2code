#include <bits/stdc++.h>
#define gc getchar_unlocked
using namespace std;
void slli(long long &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
    long long n,q,in,temp=0,query;
    slli(n);
    map<long long,long long>m;
    map<long long,long long>::iterator it;
    for(int i=1;i<=n;i++){
        slli(in);
        temp+=in;
        m[temp]=i;
    }
    slli(q);
    for(int i=0;i<q;i++){
        slli(query);
        it=m.lower_bound(query);
        cout<< it->second<<endl;
    }
    return 0;
}
