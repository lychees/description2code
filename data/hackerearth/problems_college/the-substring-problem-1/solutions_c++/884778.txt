#include <iostream>
using namespace std;
#include<cstring>
#include<cstdio>
int main()
{
	int t,n;
	cin>>t;
	int count[27];
	int status[27];
	char a[105];
	while(t--)
	{
		for(int i=0;i<27;i++)
		{
		count[i]=0;
		status[i]=0;
		}
		cin>>n;
		while(n--)
		{
			scanf("%s",a);
			for(int i=0;i<27;i++)
	      	status[i]=0;
			int l=strlen(a);
			for(int i=0;i<l;i++)
			{
				int j=a[i];
				j=j-97;
				if(status[j]==0)
				{
					status[j]=1;
					count[j]+=1;
				}
			}
		}
		int max=0;
		int r;
		for(int i=0;i<27;i++)
		{
			if(count[i]>max)
			{
				max=count[i];
				r=i;
			}
		}
		r=r+97;
		char q;
		q=r;
		cout<<q<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
