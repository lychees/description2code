vector< pair<int,pii> > v;
int pwr[MAX][MAX];
void pre(){
	int p ;
	for(int i=0;i<=100;i++)
	{
		p = 1;
		pwr[i][0] = 1;
		for(int j=0;j<100;j++){
			p *= i;
			if(p > 100)	break;
			pwr[i][j+1] = p;
		}
	}	
}
int main()
{
	memset(pwr,31,sizeof pwr);
	pre();
	ios::sync_with_stdio(false);

	{
		cin>>n;
		v.clear();

			for(int j=0;j<=n;j++){
				//if(i > j)
				if(pwr[i][j] + pwr[j][i] == n)
					v.pb(mp((i-j),mp(i,j)));	
			}
		}
		sort(all(v));
		int size = v.size();
		cout<<v[size-1].second.first<<" "<<v[size-1].second.second<<endl;

	}
	return 0;
