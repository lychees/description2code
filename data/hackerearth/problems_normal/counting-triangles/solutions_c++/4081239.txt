#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	ll n,a,b,c;
	vector<ll> v;
	map<ll,ll> mp;
	scanf("%lld",&n);
	for(ll i=0;i<n;i++)
	{
		scanf("%lld%lld%lld",&a,&b,&c);
		v.clear();
		v.push_back(a);
		v.push_back(b);
		v.push_back(c);
		sort(v.begin(),v.end());
		ll num=(v[0]*100)+(v[1]*10)+v[2];
		mp[num]++;
	}
	ll cnt=0;
	map<ll,ll>::iterator itr;
	for(itr=mp.begin();itr!=mp.end();itr++)
	{
		if(itr->second==1)
		{
			cnt++;
		}
	}
	printf("%lld\n",cnt);
	return 0;
}
