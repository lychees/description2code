#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;
int main()
{
    string a="ABCDEFGHIJKLMNOPQRSTUVWXYZ";
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string str;
        int n;
        scanf("%d",&n);
        cin>> str;
        for(int i=0;i<str.length();i++)
            cout<< (a[((str[i]-'A')+26-n)%26]);
        printf("\n");
    }
    return 0;
}
