	if (y < x) x = y;
}

const int MAXN = 222222;
int n, m;
set<int> con[MAXN];
int a[MAXN];
int val[MAXN], vn;

int main() {

	int ncase;
	for (scanf("%d", &ncase); ncase--; ) {
		scanf("%d", &n);
		for (int i = 0; i < n; i++) scanf("%d", a + i);
		for (int i = 1; i <= n; i++) con[i].clear();
		scanf("%d", &m);
		while (m--) {
			int u, v;
			scanf("%d%d", &u, &v);
			con[u].insert(v);
			con[v].insert(u);
		}
		vn = 0;
		for (int i = 0; i < n - 1; i++) {
			if (con[a[i]].count(a[i + 1])) {
				val[vn++] = i;
			}
		}
		LL ret = 0;
		for (int i = 0; i < n - 1; i++) {
			if (con[a[i]].count(a[i + 1])) {
				ret += (n - i - 1);
			} else {
				int p = lower_bound(val, val + vn, i) - val;
				if (p == vn) continue;
				ret += (n - val[p] - 1);
			}
		}
		printf("%lld\n", ret);
	}

	return 0;
}
