#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	 int T;
	 cin>>T;

	 if(!((T>=1)&&(T<=100)))	return -1;

	 int i;
	 long ARR[100];
	 long SOL[100];
	 for(i=0;i<T;++i)
	 cin>>ARR[i];

	 for(i=0;i<T;++i)
	 {
		if(ARR[i]>2)
		SOL[i]=(6*ARR[i]*(ARR[i]-2))+8;
		else
		SOL[i]=pow(ARR[i],3);
	 }

	 for(i=0;i<T;++i)
	 cout<<SOL[i]<<endl;

	 return 0;
}
