#include<bits/stdc++.h>
using namespace std;

string arr[31];
int yes;
int n;

bool chk(int co[])
{
	for(int i=0;i<=26;i++)
	{
		if(co[i]!=0)
		return false;
	}
	yes=1;
	return true;
}

void calc(int index,int cnt[])
{
	if(yes)
	return;
	
	if(index>=n)
	return;
	
	int co[27];
	for(int i=0;i<27;i++)
	{
		co[i]=cnt[i];
	}
	
	int no=0;
	for(int i=0;i<arr[index].length();i++)
	{
		co[arr[index][i]-97]--;
		if(co[arr[index][i]-97]<0)
		no=1;
	}
	
	if(no==0)
	{
		chk(co);
		calc(index+1,co);
	}
	calc(index+1,cnt);
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int cnt[27]={0};
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		string x;
		cin>>x;
		for(int i=0;i<x.length();i++)
		{
			cnt[x[i]-97]++;
		}
		calc(0,cnt);
		if(yes)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		
		yes=0;
	}
	return 0;	
}
