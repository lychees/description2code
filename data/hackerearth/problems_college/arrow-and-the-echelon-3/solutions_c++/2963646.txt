#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
using namespace std;


int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	int x;
    	scanf("%d",&x);
//    	cout<<(double)(4*x/(double)3)<<endl;
    	int fact= ceil((double)(4*x/(double)3));
    	int a = 21*x - 15*fact;
    	int b = 15*fact - 20*x;
//    	cout<<fact<<" "<<a<<" "<<b<<endl;
    	if(a<0){
    		printf("-1\n");
    	}
    	else{
    		string ans("");
    		for(int i = 0;i<a;i++){
    			ans+='5';
    		}
    		for(int i = 0;i<b;i++){
    			ans+='3';
    		}
    		cout<<ans<<endl;
    	}
    }
    return 0;
}
