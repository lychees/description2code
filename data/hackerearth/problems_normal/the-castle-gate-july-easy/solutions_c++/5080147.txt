#include<iostream>
using namespace std;
int main()
{
    int N,t;
    cin>>t;
    while(t--)
    {
        cin>>N;

    int c=0;
    for(int i=1;i<N;i++)
    {
        for(int j=i+1;j<=N;j++)
        {
            if((i^j)<=N)
            {
                c++;
            }
        }
    }
    cout<<c<<"\n";
        
    }
    
    
}
