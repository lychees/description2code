using namespace std;
int main(){
	long long int t,n,m,i,j,count;
	cin>>t;
	while(t--){
		cin>>n>>m;
		i=0;
		string a;
		while(m!=1){
			a[i]=m%2;
			m=m/2;
			++i;
		}
		a[i]=m;
		for(j=-1,count=0;j!=i;i--){
			if(a[i]==1){
			if(count==10)
			cout<<" + ";
			cout<<"("<<n<<"<<"<<i<<")";
			count=10;	
			}
		}
		cout<<endl;
	}
	return 0;
}
