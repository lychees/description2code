#include <iostream>
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll sum(ll n)
{
	if(n<=0)return 0;
	ll s,x,ans;
	ans = s = 0;
	for(x=1;x<=n;x++)
	{
		if(s>=n)return ans;
		if(s+x <=n)
		{
			s+=x;
			if(x%2==1)ans+=x;
		}
		else {
			ll mi = min(x,n-s);
			if(mi%2==1)ans+=x;
			s+=mi;
		}
	}
	return ans;
}
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
	ll l,r;
	scanf("%lld%lld",&l,&r);
	ll tm;
	if(l>r){
		tm = l; l=r; r = tm;
	}
	printf("%lld\n",sum(r)- sum(l-1));
}
    return 0;
}
