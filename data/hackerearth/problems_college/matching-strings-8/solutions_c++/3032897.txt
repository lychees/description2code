
#include <bits/stdc++.h>
//kise
using namespace std;
 
int main()
{
	int t, i, c;
	string s1, s2;
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>s1>>s2;

    	if(s1.length()<s2.length())
    		swap(s1,s2);
    	for(i=0;i<s2.length();i++)
    		if(s1[i]==s2[i])
    			c++;
    	cout<<c<<endl;
    }
    return 0;
}
