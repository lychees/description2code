#include <bits/stdc++.h>
using namespace std;

/*
*/

int a[200005];
int main()
{
	int n, m;
	scanf("%d %d", &n, &m);
	for (int i = 0; i < n; i++)
		scanf("%d", a+i);
	for (int i = 0; i < n; i++)
		scanf("%d", a+n+i);
	sort(a, a+2*n);
	int i = 0;
	for (i = 0; i < n; i++)
	{
		if (m >= a[i])
		{
			m -= a[i];
		}
		else
			break;
	}
	printf("%d\n", i);
}
