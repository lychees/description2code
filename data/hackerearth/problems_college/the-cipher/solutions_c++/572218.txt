#include <iostream>
#include<vector>
using namespace std;
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main() {
	int t;
	scanint(t);
	while(t--)
	{
		int n;
		vector <int> ar;
		scanint(n);
		ar.resize(n);
		for(int i=0;i<n;i++)
			scanint(ar[i]);
		for(int i=0;i<n;i++)
			printf("%c",255-ar[i]);
		printf("\n");
	}
	return 0;
}
