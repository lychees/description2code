#include <iostream>
using namespace std;

int a,b,c,d,e,f;

bool fun(){
	int d1,d2,d3,sum,max;
	d1=(a-c)*(a-c)+(b-d)*(b-d);
	d2=(c-e)*(c-e)+(d-f)*(d-f);
	d3=(e-a)*(e-a)+(f-b)*(f-b);
	if(d1==0 || d2==0 || d3==0)return false;
	if(d1>d2){
		max=d1;
		sum=d2+d3;
	}
	else{
		max=d2;
		sum=d1+d3;
	}
	if(max<d3){
		max=d3;
		sum=d1+d2;
	}
	//cout<<max<<" "<<sum<<endl;
	if(max==sum && max!=0){
		return true;
	}
	else
		return false;
}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	cin>>a>>b>>c>>d>>e>>f;
    	if(fun()){cout<<"YES\n";continue;}
    	a++;if(fun()){cout<<"YES\n";continue;}a--;
    	a--;if(fun()){cout<<"YES\n";continue;}a++;
    	b++;if(fun()){cout<<"YES\n";continue;}b--;
    	b--;if(fun()){cout<<"YES\n";continue;}b++;
    	c++;if(fun()){cout<<"YES\n";continue;}c--;
    	c--;if(fun()){cout<<"YES\n";continue;}c++;
    	d++;if(fun()){cout<<"YES\n";continue;}d--;
    	d--;if(fun()){cout<<"YES\n";continue;}d++;
    	e++;if(fun()){cout<<"YES\n";continue;}e--;
    	e--;if(fun()){cout<<"YES\n";continue;}e++;
    	f++;if(fun()){cout<<"YES\n";continue;}f--;
    	f--;if(fun()){cout<<"YES\n";continue;}f++;
    	cout<<"NO\n";
    }
    return 0;
}
