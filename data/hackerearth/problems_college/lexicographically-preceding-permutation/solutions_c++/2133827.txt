#include <iostream>
#include <algorithm>

using namespace std;

int __, n, p[1111111];

int main()
{
    scanf("%d", &__);
    for (int _ = 0; _ < __; ++_) {
    	scanf("%d", &n);
    	for (int i = 0; i < n; ++i) {
    		scanf("%d", p + i);
    	}
    	if (!prev_permutation(p, p + n)) {
    		next_permutation(p, p + n);
    	}
    	for (int i = 0; i < n; ++i) {
    		printf("%d%c", p[i], " \n"[i == n - 1]);
    	}
    	
    }
    return 0;
}
