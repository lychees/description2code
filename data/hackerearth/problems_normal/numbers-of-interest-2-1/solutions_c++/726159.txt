void precompute()
{
    for(int i = 1; i < 1000005; i++) 
    {
        factor[i] = i;
    }
    
    for(int i = 2; i * i < 1000005; i++) 
    {
            if (factor[i] == i) 
            {
                    for(int j = i + i; j < 1000005; j += i) 
                    {
                           if(factor[j]==j)factor[j] = i;
                    }
            }
    }
    factor[1]=0;
    for(int i=1;i<1000005;i++)
    {
                factor[i]=factor[i-1]+factor[i];
    }
}
