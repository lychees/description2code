#include<bits/stdc++.h>
#define ll long long int
#define fastScan ios_base::sync_with_stdio(0); cin.tie(NULL);
using namespace std;
typedef pair <ll, ll> pll;

ll X, Y, N, W, P1, P2;
vector<pair<int, int> > visited;


double dp[1001][1001];
bool vis[1001][1001];

double func(ll n, ll w) {
  if(w <= 0) return 100.000;
  if(n == 0) return 0.00000;
  if(vis[n][w]) return dp[n][w];
  vis[n][w] = 1;
  visited.push_back(make_pair(n, w));
  double & ans = dp[n][w];
  ans = max(func(n - 1, w - X) * P1 + (100 - P1) * func(n - 1, w), func(n - 1, w - Y) * P2 + (100 - P2) * func(n - 1, w));
  ans = ans / 100.000;
  return ans;  
}


int main() {

    fastScan;

    ll t;
    cin >> t;

    while(t--) {
      cin >> X >> Y >> N >> W >> P1 >> P2;
      printf("%.6lf\n", func(N, W));
      for(int i = 0; i < visited.size(); i++) vis[visited[i].first][visited[i].second] = false;
      visited.clear();
    }
     
    return 0;
}
