#include<bits/stdc++.h>
using namespace std;

#define M 100

long long gcd(long long a,long long b)
{
	if(!b)
		return a;
	return gcd(b,a%b);
}

long long pp(long long a)
{
	return (a>0)?a:-a;
}

int main()
{
	long long a[M+1]={0};
	long long i,j,k,p,t;
	p=0;
	for(i=1;i<=M;i++)
	{
		for(j=1;j<i;j++)
		{
			for(k=1;k<j;k++)
			{
				a[gcd(gcd(i,j),k)]++;
				p++;
			}
		}
	}
	cin>>t;
	while(t--)
	{
		long long m,n;
		scanf("%lld%lld",&m,&n);
		k=pp(m-n);
		long long r=0;
		for(i=2;i<=M;i++)
		{
			if(k%i)
			{
				r+=a[i];
			}
		}
		printf("%0.6lf\n",(double)(p-r)/double(p));
	}
}
