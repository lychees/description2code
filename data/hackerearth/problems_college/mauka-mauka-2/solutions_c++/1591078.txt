int A[100005];
int S[100005];
int C[100005];
//map<int, vector<int> > M;
int main()
{
    int n;
    scanf("%d",&n);
    memset(C,0,sizeof C);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",&A[i]);
     // A[i] = i;
        //M[A[i]] = i;
    }
    int y = A[1];
    S[1] = 1;
    for(int j=2;j<=sqrt(y);j++)
    {
        if(y%j==0)
        {
            C[j] = max(C[j],S[1]);
        }
    }
    C[y] = S[1];
    for(int i=2;i<=n;i++)
    {
        S[i] = 1;
        int x = A[i];
        for(int j=2;j<=sqrt(x);j++)
        {
            if(x%j==0)
            {
               // cout << i << " " << S[i] << " " << j << " " << C[j] << endl;
                S[i] = max(S[i], C[j]+1);
            }
        }
        for(int j=2;j<=sqrt(x);j++)
        {
            if(x%j==0)
            {
                C[j] = max(C[j],S[i]);
            }
        }
        C[x] = S[i];
    }
    int ans = 0;
    for(int i=1;i<=n;i++)
        ans = max(ans,S[i]);
    cout << ans;
return 0;
}
