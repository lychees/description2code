ll fact[maxn],ifact[maxn];

ll bpow(ll x,ll n)
{
	ll ans=1;
	while(n>0)
	{
	    if(n&1) ans*=x;
	    x*=x;
	    ans%=mod;
	    x%=mod;
	    n/=2;
	}
	return ans;
}
void pre_calc(ll N)
{
    fact[0]=ifact[0]=1;
    for(ll i=1; i<=N; ++i)
    {
        fact[i]=(fact[i-1]*i)%mod;
        ifact[i]=bpow(fact[i],mod-2);
    }
}
ll ncr(int n,int r)
{
    if(r<0 or n<r)
        return 0;
    ll ans=fact[n];
    ans=(ans*ifact[r])%mod;
    ans=(ans*ifact[n-r])%mod;
    return ans;
}
