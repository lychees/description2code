#include <iostream>
#include <cmath>
#include <cstdlib>

int main(void)
{
    short T;
    long N;
    short B;
    
    std::cin>>T;
    
    while (T--)
    {
        std::cin>>N>>B;
        
        long div  = N/B;
        
        long first = div*B;
        long sec = (div+1)*B;
        
        if (std::abs(first-N) == std::abs(sec-N))
        {
            if (first > sec)
            {
                std::cout<<first<<std::endl;
            }
            else
                std::cout<<sec<<std::endl;
        }
        else
        {
            if (std::abs(first-N) <std::abs(sec-N))
            {
                std::cout<<first<<std::endl;
            }
            else
            {
                std::cout<<sec<<std::endl;
            }
        }
        
    }
    return 0;
}
