#include <bits/stdc++.h>
using namespace std;

bool isVowel(char c){
	c = toupper(c);
	if(c == 'A' || c  == 'E' || c == 'I' || c == 'O' || c == 'U')
		return true;
	return false;
}

int main(){
	
	string s;
	while(getline(cin, s)){
		if(s == "") {
			cout << "NO\n";
			continue;
		}
		istringstream ss(s);
		string token;
		int a = 0, b = 0;
		int t1 = 0, t2 = 0;
		while(getline(ss, token, ' ')){
			for(int i=0; i<token.size(); i++){
				if(!isalpha(token[i])){
					token.erase(token.begin()+i);
				}
			}
			
			for(int i=0; i<token.size()-1; i++){
				if(isVowel(token[i]) && isVowel(token[i+1]) && (!isVowel(token[i-1]) || !isVowel(token[i+2])))
					a=1, t1++;
				if(!isVowel(token[i]) && !isVowel(token[i+1]))
					b=1, t2++;
			}
		}
		if(a == 1 && b == 1){
			if( t1 > 1 || t2 > 1){
				cout << "YES\n";
			}
			else
			cout << "NO\n";
		}
		else
			cout << "NO\n";
		
	}
	
	return 0;
}
