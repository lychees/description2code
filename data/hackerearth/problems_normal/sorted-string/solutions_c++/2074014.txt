#include <bits/stdc++.h>
#define ll long long
using namespace std;
typedef struct node
{
	int c;//count
	char a;//alphabet
}node;
node r[26];
bool my(node x,node y)
{
	if(x.c<y.c)
	return true;
	if(x.c==y.c)
	{
		if(x.a<y.a)
		return true;
	}
	return false;
}
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int i;
		for(i=0;i<26;i++)
		{
			r[i].a=('a'+i);
			r[i].c=0;
		}
		char str[110];
		scanf("%s",str);
		for(i=0;str[i]!='\0';i++)
		{
			(r[str[i]-'a'].c)++;
		}
		sort(r,r+26,my);
		for(i=0;i<26;i++)
		{
			while(r[i].c!=0)
			{
				printf("%c",r[i].a);
				(r[i].c)--;
			}
		}
		printf("\n");
	}
	return 0;
}
