#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin >>t;
    for(int i =0 ;i<t;i++)
    {
    	int n;
    	cin >> n;
    	vector<int> v(n);
    	for(int i=0;i<n;i++){
    		cin >> v[i];	
    	}
    	sort(v.begin(),v.end());
    	vector<int>sum(n/2);
    	for(int i=0;i<n/2;i++){
    		sum[i] = v[i] + v[n-1-i];
    	}
    	sort(sum.begin(),sum.end());
    	
    	cout<<sum[n/2-1]-sum[0]<<endl;
    	
    	
    }
}
