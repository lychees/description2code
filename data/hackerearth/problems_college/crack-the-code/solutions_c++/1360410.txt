#include <iostream>
#include <math.h>
using namespace std;

int main()
{
    long long t=0;
    long long x,y;
    cin>>t;
    while(t--)
    {
    	long long a=0;
    	long long b=1;
   	 	long long c=0;
    	cin>>x>>y;
    	long long count=0;
	    while(a<x)
	    {
	    	c=a+b;
	    	a=b;
	    	b=c;
	    }
	    while(a<=y)
	    {
	    	count++;
	    	c=a+b;
	    	a=b;
	    	b=c;
	    }
	    if(count%2==0)
	    cout<<"VALID"<<endl;
	    else
	    cout<<"INVALID"<<endl;
    }
}
