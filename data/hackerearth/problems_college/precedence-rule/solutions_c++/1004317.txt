# include<iostream>
# include<cstring>
using namespace std;
int main()
{
	long long t;
	cin>>t;
	while(t--)
	{
		string a;
		cin>>a;
		long long l=a.length(),i,x,y;
		x=a[l-1]-48;
	//	cout<<x<<"   ";
		char c;
		for(i=l-2;i>=0;i-=2)
		{
			c=a[i];
			y=a[i-1]-48;
		//	cout<<c<<"   "<<y;
			if(c=='+')
			x=x+y;
			else if(c=='*')
			x*=y;
			else if(c=='-')
			x=x-y;
		}
		cout<<x<<endl;
	}
}
