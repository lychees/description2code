#include <iostream>
#include<cstdio>
#include<cstring>

using namespace std;

const long long int N=10000000;

long long int ph[N+10],prime[N+10];


void phi_all() ///get phi(n) from 1 to N in liner time
{
    memset (ph, 0, sizeof (long long) * (N + 1));
    prime[0] = 0;
    for (long long i = 2; i <= N; i++)
    {
        if (!ph[i])
            prime[++prime[0]] = i, ph[i] = i - 1;
        for (long long j = 1; prime[j] <= N / i && j <= prime[0]; j++)
        {
            if (i % prime[j])
                ph[prime[j]*i] = ph[prime[j]] * ph[i];
            else
            {
                ph[prime[j]*i] = ph[i] * prime[j];
                break;
            }
        }
    }
}


int main() {
	
	
	phi_all();
	ph[1]=1;
	
	long long int temp=0,i,ans,n,t;
	cin>>t;
	
	for(i=2;i<=N;i++)
	 ph[i]+=ph[i-1];
	
	while(t--)
	{
	   cin>>n;
	   cout<<ph[n]<<endl;
	}
	
	return 0;
}
