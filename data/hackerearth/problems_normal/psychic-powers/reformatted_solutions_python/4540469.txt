# print 'Hello World!'
s = raw_input()
one, zero = 0, 0
flag = False
for bin in s:
    if zero == 6 or one == 6:
        flag = True
        break
    if bin == '0':
        zero += 1
        one = 0
    else:
        one += 1
        zero = 0
if (flag):
    print "Sorry, sorry!"
else:
    print "Good luck!"
