#include <iostream>
using namespace std;
#include<vector>
#include<utility>
#include<cstring>
#include<queue>
#define ll long long
int main()
{
    int t,i,j,n,m,k,l,x,y,c;
    cin>>t;
    while(t--){
    	cin>>n>>m;
    	vector<vector<int> > g(n+1);
    	for(i=0;i<m;i++){
    		cin>>x>>y;
    		g[x].push_back(y);
    		g[y].push_back(x);
    	}
    	
    	priority_queue<pair<int,int>, vector<pair<int,int>>, greater<pair<int,int>> > q;
    	q.push(make_pair(0,1));
    	vector<int> v(n+1,0);
    	vector<int> d(n+1,0);
    	while(!q.empty()){
    		if(v[q.top().second]==1)
    			{
    				q.pop();
    				continue;
    			}
    		v[q.top().second]=1;
    		x=q.top().second;
    		y=q.top().first;
    		d[x]=y;
    		q.pop();
    		for(i=0;i<g[x].size();i++){
    			if(v[g[x][i]]==0){
    				//cout<<"pushing "<<g[x][i]<<endl;
    				q.push(make_pair(y+1,g[x][i]));
    			}
    		}
    	}
    	cout<<d[n]<<endl;
    }
    return 0;
}
