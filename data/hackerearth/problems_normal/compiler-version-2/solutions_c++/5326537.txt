#include <bits/stdc++.h>
using namespace std;

int main()
{
	string str;
    while(getline(cin,str))
    {
    	//int n=str.length();
    	for(int i=0;i<str.length()-1;i++)
    	{
    		if(str[i]=='/' && str[i+1]=='/') break;
    		if(str[i]=='-' && str[i+1]=='>')
    		{
    			str.replace(i,2,".");
    		}
    	}
    	cout<<str<<endl;
    }
}
