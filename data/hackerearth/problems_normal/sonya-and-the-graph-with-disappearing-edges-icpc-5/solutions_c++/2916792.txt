#define MOD 1000000007
int a[111111];
vector<int> v[111111];
vector<int> vt[111111];
int t[111111];
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int i,j,n,m,k;
	int x,y,T;
	queue<int> q;
	cin>>n>>m>>k;
	for(i=1;i<=m;i++)
	{
		cin>>x>>y;
		v[x].push_back(y);
		v[y].push_back(x);
		vt[x].push_back(i);
		vt[y].push_back(i);
	}
	while(k--)
	{
		cin>>T>>x;
		t[x]=T;
	}
	for(i=1;i<=n;i++)
		a[i]=-1;
	a[1]=0;
	q.push(1);
	while(!q.empty())
	{
		x=q.front();
		//cout<<x<<endl;
		q.pop();
		m=v[x].size();
		for(i=0;i<m;i++)
		{
			y=v[x][i];
			if(a[y]==-1&&(t[vt[x][i]]>a[x]+1||t[vt[x][i]]==0))
			{
				a[y]=a[x]+1;
				q.push(y);
			}
		}
	}
	//for(i=1;i<=n;i++)
		cout<<a[n]<<endl;
//	system("pause");
	return 0;
}
