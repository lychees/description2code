#include <iostream>
using namespace std;

int main()
{
    long int n,t,area,x,y,c,i;
    cin>>t;
    for(i=0;i<t;i++)
    {
    	cin>>n;
    	if(n%2!=0)
    		n=n-1;
    	if(n%4==0)
    		x=y=n/4;
    	else
    	{	
    		x=y=n/4;
    		y++;
    	}
    	area=x*y;
    	cout<<area<<endl;
    }
    return 0;
}
