noOfTestCases = int(raw_input())

def round(no, roundTo):
	no = int(no)
	roundTo = int(roundTo)
	
	lowerLimit = (no / roundTo) * roundTo
	upperLimit = ((no / roundTo) + 1) * roundTo
	
	if((no - lowerLimit) ==  (upperLimit - no)):
		return upperLimit
	elif((no - lowerLimit) < (upperLimit - no)):
		return lowerLimit
	else:
		return upperLimit
		

for i in range(1, noOfTestCases + 1):
	no , roundTo = raw_input().split(" ")
	print round(no, roundTo)
