using namespace std;

typedef pair<int,int>pii;
typedef long long LL;

bool cmp(const pii &left,const pii &right)
{
	return left.second<right.second;
}
int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else
		return gcd(b,a%b);
}
int main()
{
	int t;
	scan(t);
	while(t--)
	{
		LL n;
		scanl(n);
		LL d=n/2;
		
		d--;
		if(n==1) 
		{
			cout<<"0\n";
			continue;
		}
		cout<<(n*(n-1))/2+d<<endl;
	}
}
