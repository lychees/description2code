#include<stdio.h>
int main()
{
	long int T;
	for(scanf("%ld",&T);T>0;T--)
	{
		long long int a,b,n;
		float c;
		scanf("%lld%lld",&n,&b);
            a=n%b;
            c=(float)b/2;
           if(a<c)
			printf("%lld\n",n-a);
		else if(a>=c)
			printf("%lld\n",n+b-a);
	}
	return 0;
}
