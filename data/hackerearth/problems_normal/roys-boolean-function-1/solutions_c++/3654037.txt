#include <iostream>
using namespace std;


int gcd(int a, int b)
{
	if(b==0)
		return a;
	else
		return(gcd(b,a%b));	
}
bool IsPrime(int a)
{
	bool ret = false;
	if(a==1 || (a!=2) && (a%2 == 0))
	{
		ret = false;
		return ret;
	}
	else if (a == 2)
	{
		return true;
	}
	else
	{
		for(int i=3; i<a; i+=2)
		{
			if(a%i == 0)
			{
				ret = false;
				break;
			}
			else
			{
				ret = true;
			}
		}
	}
	return ret;
}


int main()
{
    int t = 0;
    cin>>t;
    for(int i = 0; i<t; i++)
    {
    	int n=0;
    	cin>>n;
    	int j = 1;
    	int primes=0;
    	while(j<n)
    	{
    		if(gcd(n,j)==1)
	    	{
	    		primes++;
	    	}
	    	j++;
	    	
    	}
    	bool flag = IsPrime(primes);
    	if(flag)
    	{
    		cout<<"TRUE"<<endl;
    	}
    	else
    	{
    		cout<<"FALSE"<<endl;
    	}
    }
    return 0;
}
