long long mulmod(long long a,long long b,long long c){
  LL x = 0,y=a%c;
    while(b > 0){
        if(b%2 == 1){
            x = (x+y)%c;
        }
        y = (y*2)%c;
        b /= 2;
    }
    return x%c;
}
LL modulo(LL a,LL b,LL c){
    LL x=1,y=a; // long long is taken to avoid overflow of intermediate results
    while(b > 0){
        if(b%2 == 1){
            x = mulmod(x,y,c);
        }
         y = (y*y)%c ;// squaring the base
        b /= 2;
    }
    return x%c;
}
