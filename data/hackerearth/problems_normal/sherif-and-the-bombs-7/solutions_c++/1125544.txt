#include <iostream>
#include <vector>
using namespace std;

int main(){

    int N,cont = 0;
    int X,Y,R;
    int nPontos = 0;
    vector <vector <int> > matriz;
    matriz.resize(1001);

    for (int i = 0 ; i < 1001 ; i++){
        matriz[i].resize(1001,0);
    }

    cin >> N;
    while(cont < N){

        cin >> X >> Y >> R;
            for(int j = 1  ; j <= Y+R ;j++){
                for(int i =1  ; i <= X+R  ; i++){
                    if(((i-X)*(i-X) + (j-Y)*(j-Y) <= R*R) && (i <= 1000 && j <= 1000)) matriz[i][j] += 1;
            }
        }
        cont++;
    }

    for(int i = 1 ; i <=1000; i++){
        for(int j = 1 ; j <= 1000 ; j++){
            if((matriz[i][j]) > 1) nPontos++;
        }
    }

    cout << nPontos << endl;

    return 0;
}
