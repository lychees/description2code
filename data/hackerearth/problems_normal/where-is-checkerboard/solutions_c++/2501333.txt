#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define ios ios_base::sync_with_stdio(false)
#define fcin freopen("stdin.input","r",stdin)
#define fcout freopen("out.txt","w",stdout)
#define rep(i,n) for(ll i=0;i<n;i++)
#define reep(i,m,n) for(ll i=m;i<=n;i++)
#define vec vector<ll>
#define vv vector<vector<ll> >
#define pb push_back
#define pp pair<ll,ll>
#define mp make_pair
#define precisionfixed std::cout << std::fixed;std::cout.precision(3)
#define bp __builtin_popcountll
#define MOD 1000000007

int main()
{
    ios;
    //fcin;
    //fcout;
    //precisionfixed;
    ll n,m;cin>>n>>m;
    if(n<32 or m<32) cout<<0;
    vv v;v.resize(n);
    rep(i,n)
    {
        rep(j,m)
        {
            char c;cin>>c;if(c=='#') v[i].pb(1);else v[i].pb(0);
        }
    }
    bool cur=0;
    unsigned int w[32]={0};
    unsigned int comp[32]={0};
    //rep(i,32) w[i]=0;
    //rep(i,32) cout<<w[i]<<endl;
    rep(i,32)
    {
        rep(j,32)
        {
            if(cur==0)
            {
                //w[i].pb(0);
                cur=1;
            } 
            else
            {
                //w[i].pb(1);
                w[i] |= 1<<(31-j);
                cur=0;
            } 
        }
        cur=!cur;
    }
    
    //rep(i,32) cout<<w[i]<<endl;
    rep(i,32)
    {
        rep(j,32)
        {
            //cout<<v[i][j];
            if(v[i][j]==1)
                comp[i] |= 1<<(31-j);
        }
        //cout<<endl;
    }
    //rep(i,32) cout<<w[i]<<" "<<comp[i]<<endl;
    ll Hlimit=m-32+1;
    ll Vlimit=n-32+1;
    ll count=0;
    
    rep(i,Vlimit)
    {
        rep(o,32)
        {
            comp[o]=0;
            rep(p,32)
            {
                //cout<<v[i][j];
                if(v[o+i][p]==1)
                    comp[o] |= 1<<(31-p);
            }
            //if(i==0) cout<<comp[o]<<"P";
            //if(i==0) cout<<endl;
        }
        ll proxy=0;
        rep(j,Hlimit)
        {
            //rep(i,32) comp[i]=0;
            //cout<<i<<","<<j<<endl;
            bool valid=1;
            if(proxy)
            {
                rep(k,32)
                {
                    comp[k] = comp[k]<<1;
                    comp[k] |= v[i+k][j+31];
                }
            }
            
            rep(z,32)
            {
                //if(i==5 and j==5) cout<<comp[z]<<endl;
                if( (comp[z] != w[z]) )
                {
                    valid=0;
                    break;
                }
            }
            //cout<<"###################################"<<endl;
            if(valid==1) count++;
            proxy=1;
        }
    }
    cout<<count;
    return 0;
}
