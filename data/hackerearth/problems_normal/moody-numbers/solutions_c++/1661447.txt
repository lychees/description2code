#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define pb push_back
#define mp make_pair
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}

int digit_sum(long long int n){
    int sum=0;
    while(n>0){
        sum=sum+n%10;
        n=n/10;
    }
    return sum;
}

int lookup[163];
void precompute(){
    memset(lookup,0,sizeof(lookup));
    lookup[1]=1;
    lookup[2]=1;
    lookup[3]=-1;
    lookup[5]=-1;
    lookup[4]=1;
    lookup[9]=-1;
    lookup[13]=-1;
    for(int i=3;i<163;i++){
        int temp=i;
        if(i%3==0)
           lookup[i]=0;
        else if(lookup[i]==0)
        {
            while(lookup[temp]==0)
            {
                temp=temp*temp;
                temp=digit_sum(temp);    
            }
            lookup[i]=lookup[temp];
        }         
    }
}
int main(){
    int t=scan();
    precompute();
    while(t--){
        long long int n=scan();
        if(n==4){
            printf("YES\n");
        }
        else
        {
            n=n*n;
            n=digit_sum(n);
            if(lookup[n]==1)
                printf("YES\n");
            else
                printf("NO\n");    
        }
    }
}
