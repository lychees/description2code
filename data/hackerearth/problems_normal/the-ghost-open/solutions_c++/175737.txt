#include<iostream>
using namespace std;

main(){
    int t;
    long long n,s,temp;
    cin>>t;
    while(t--){
        s=0;
        cin>>n;
        if(n<=1)
        {
            cout<<0<<"\n";
            continue;
        }
        while(n!=1){
            temp=n>>1;
            s+=temp;
            n=(temp+n%2);
        }
    cout<<s<<"\n";
    }
}
