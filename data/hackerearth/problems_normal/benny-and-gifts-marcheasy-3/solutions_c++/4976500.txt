#include <iostream>
#include <bits/stdc++.h>

using namespace std;


int main()
{
    string st;
    cin >> st;
    int x = 0, y = 0;
    set<pair<int, int> > Po;
    int ans = 0;
    for (int i = 0; i <= st.size(); ++i)
    {

        if (Po.find(make_pair(x,y)) != Po.end()) ++ans;
        Po.insert(make_pair(x,y));
        if (st[i] == 'U') --x;
        if (st[i] == 'D') ++x;
        if (st[i] == 'L') --y;
        if (st[i] == 'R') ++y;
    }
    cout << ans << endl;
    return 0;
}
