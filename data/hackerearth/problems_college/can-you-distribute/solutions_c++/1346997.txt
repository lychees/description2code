#include<bits/stdc++.h>
using namespace std;
void EE(long long int a, long long int b, long long int& x, long long int& y)
{
    if(a%b == 0)
    {
        x=0;
        y=1;
        return;
    }
    EE(b,a%b,x,y);
    long long int temp = x;
    x = y;
    y = temp - y*(a/b);
}
long long int inverse(long long int a, long long int m)
{
    long long int x,y;
    EE(a,m,x,y);
    if(x<0) x += m;
    return x;
}
long long int mod = 10000007;
int main()
{
	long long int n, r, i, j, k, m, l;
	scanf("%lld%lld", &n, &r);
	l = 1;
	if (r > n)
		printf("-1\n");
	else {
		for (i = 2; i <= n - 1; i++) {
			l = (l * i) % mod;	
		}
		for (i = 2; i <= r - 1; i++) {
			k = inverse(i, mod);
			l = (l * k) % mod;
		}		
		for (i = 2; i <= n - r; i++) {
			k = inverse(i, mod);
			l = (l * k) % mod;
		}
		printf("%lld\n", l);
	}
	return 0;
}
