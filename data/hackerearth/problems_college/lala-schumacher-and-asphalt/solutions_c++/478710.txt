#include<stdio.h>
#include<stdlib.h>

int main()
{
	int t, n;
	long long int boost, normal, fx, fy;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		int nx[n], ny[n];
		for (int i = 0; i < n; i++)
		{
			scanf("%d", &nx[i]);
		}
		for (int i = 0; i < n; i++)
		{
			scanf("%d", &ny[i]);
		}
		long long int distance, total_time;
		scanf("%llu %llu", &fx, &fy);
		scanf("%llu %llu", &normal, &boost);
		distance = (abs(fx) + abs(fy)) * normal;
		for (int i = 0; i < n; i++)
		{
			total_time = (abs(nx[i]) + abs(ny[i])) * normal;
			total_time += (abs(nx[i] - fx) + abs(ny[i] - fy)) * boost;
			if(distance > total_time)
				distance = total_time;
		}
		printf("%llu\n", distance);
	}
	return 0;
}
