#include <bits/stdc++.h>
using namespace std;
 
#define pb push_back
#define sn(n) scanf("%d",&n);
#define pn(n) printf("%d\n",n);
#define sln(n) scanf("%lld",&n);
#define pln(n) printf("%lld\n",n);
#define MOD 1000000007
#define ll long long int

int gcd(int a, int b) { int temp; while(b!=0){ temp=b; b=a%b; a=temp; } return a;  }
 
 
ll INIT_MAX=100000000;
ll  INIT_MIN=-1; 
 
int main()
{
	ll t,x,n,credit=0,debit=0,mint,minx;
	mint=INIT_MAX;
	minx=mint;
	sln(n);
	while(n--)
	{
		sln(t);
		sln(x);
		if(t<=x)
		{
			credit=credit+x-t;
			mint=min(mint,t);
		}
		else
		{
			debit=debit+t-x;
			minx=min(minx,x);
		}
		
	}
	if(mint==INIT_MAX)
	mint=0;
	if(minx==INIT_MAX)
	minx=0;
	
	if(credit+mint>debit+minx)
	{pln(mint);}
	else 
	{pln((debit-credit+minx));}
	
	
   return 0;
}
