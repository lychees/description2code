using namespace std;

int main()
{
	int t,n,m;
	cin>>t;
	while(t--){
		cin>>n>>m;
		int a[5005], dp[5005]={0};
		for(int i=0 ; i<n ; i++){
			double d;
			cin>>a[i]>>d;
		}	
		for(int i=0 ; i<n ; i++){
			int x=a[i]-1;
			for(int j=x ; j>=0; j--)
				dp[x]=max(dp[x],1+dp[j]);
		}
		int ans=0;
		for(int i=0 ; i<n ; i++)
			ans=max(ans,dp[i]);
		ans=n-ans;
		cout<<ans<<"\n";
	}
	return 0;
}
