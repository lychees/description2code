b=['a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z']
s=raw_input()
a=[0]*30
le=len(s)
for i in range(0,le):
	a[ord(s[i])-ord('a')]=a[ord(s[i])-ord('a')]+1
final=""
temp=""
for i in range(0,26):
	if(a[i]%2==0):
		final=final+b[i]*(a[i]/2)
	else:
		final=final+b[i]*(a[i]/2)
		temp=temp+b[i]
		
final=final+temp+final[::-1]	
flag=0
for i in range(0,le):
	if(final[i]!=final[le-i-1]):
		flag=1
		break
if(flag):
	print -1
else:
	print final
