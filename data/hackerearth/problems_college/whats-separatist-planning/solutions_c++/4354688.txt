#include <bits/stdc++.h>
using namespace std;

vector<string> S;

int main()
{
    int L, D, N;
    scanf("%d %d %d", &L, &D, &N);
    
    for(int i=1; i<=D; i++)
    {
    	string tmp;
    	cin >> tmp;
    	
    	S.push_back(tmp);
    }
    
    while(N--)
    {
    	string A;
    	cin >> A;
    	
    	int ans = 0;
    	
    	for(int i=0; i<D; i++)
    	{
    		int flag = 0, k = 0;
    		
    		for(int j=0; j<S[i].size(); j++)
    		{
    			int tmp = 0;
    			
    			if(A[k]=='(')
    			{
    				while(A[k]!=')')
    				{
    					if(S[i][j]==A[k])
    						tmp = 1;
    				
    					k++;
    				}
    				
    				k++;
    			}
    			else
    			{
    				if(S[i][j]==A[k])
    					tmp = 1;
    				
    				k++;
    			}
    			
    			if(!tmp)
    			{
    				flag = 1;
    				break;
    			}
    		}
    		
    		if(!flag)
    			ans++;
    	}
    	
    	printf("%d\n", ans);
    }
    
    return 0;
}
