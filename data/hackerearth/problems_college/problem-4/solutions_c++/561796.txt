#include <iostream>
#include<string.h>
using namespace std;

int main()
{
	int len, i, j;
	char temp, string[10000];
    cin>> string;
    len= strlen(string);
    for(i=0;i < len;i++)
    {
    	for(j=i+1;j < len;j++)
    		if(string[i] > string[j])
    		{
    			temp = string[j];
    			string[j] = string[i];
    			string[i] = temp;
    		}
    }
    cout<< string;
    return 0;
}
