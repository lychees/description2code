#include <iostream>
using namespace std;

int main()
{
	int n;
	int N,k;
	cin>>n;
	
	for(int i=0;i<n;i++)
	{
		cin>>N>>k;
		int a[N];int count=0;
		for(int j=0;j<N;j++)
		{
	
			cin>>a[j];
			if(a[j]<=0)
				count++;
		}
		if(count>=k)
			cout<<"NO"<<"\n";
		else
			cout<<"YES"<<"\n";
	}
    return 0;
}
