#include<bits/stdc++.h>
using namespace std;
#define ld long double
#define ll long long int
#define MOD 1000000007
#define MAX 3005
#define eps 1e-11
ll x[MAX],y[MAX],f[MAX];
ld dp[MAX];
ld dis(int i,int j)
{
	ld ans;
	ans = (x[i]-x[j])*(x[i]-x[j]) + (y[i]-y[j])*(y[i]-y[j]);
	ans = sqrtl(ans);
	return ans;
}
int main()
{
	int n;
	int i,j;
	cin>>n;
	for(i=0;i<n;++i)
	{
		cin>>x[i]>>y[i]>>f[i];
		dp[i] = -LDBL_MAX;
	}
	//for(i=0;i<n;++i)
	//	cout<<dp[i]<<" ";
	//cout<<endl;
	dp[0] = f[0];
	for(i=1;i<n;++i)
	{
		for(j=0;j<i;++j)
		{
			dp[i] = max(dp[i],dp[j]+f[i]-dis(i,j));
		}
	}
	cout<<fixed;
	cout<<setprecision(6);
	//for(i=0;i<n;++i)
	//	cout<<dp[i]<<" ";
	//cout<<endl;
	cout<<dp[n-1]<<endl;
	
	
	
    return 0;
}
