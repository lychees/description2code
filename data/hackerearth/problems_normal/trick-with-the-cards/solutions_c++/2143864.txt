#include<bits/stdc++.h>
using namespace std;
int main()
{

    int t;
    cin >> t;
    while(t--)
    {

         long long  n , val ;
         cin >> n;
           if(n == 0)
           val = 0;
           else
           if( n == 1 || n == 2)
           val = 1;
           else
           if(n%3 == 0)
             val = n/3;
             else
             val = n;
             cout << val << endl;

    }

}
