ll gcd(ll a, ll b) {

	else return gcd(b, a % b);
}

int main() {
	int t; cin >> t;
	while (t--) {
		int H; cin >> H;
		map<int, int> m;
		for (int i = 2; i * i <= H; i++) {
			while (H % i == 0) {
				H /= i;
				m[i]++;
			}
		}
		if (H > 1) m[H]++;

		for (map<int, int>::iterator it = m.begin(); it != m.end(); it++)
