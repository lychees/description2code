#include <iostream>
#include <string>
#include <stack>
using namespace std;

int main()
{
	long long int t,count=0,flag=0;
	cin>>t;
	while(t--)
    {	stack <string> st;
    	string s,temp="-";
    	cin>>s;
    	for(int i=0;i<s.length();i++)
    	{  
    	    temp[0]=s[i];	
    		if(!st.empty())
    		{
    			if(st.top()==temp)
    				st.pop();
    			else
    				st.push(temp);
    		}
    		else
    		{
    			st.push(temp);
    		}
    	
    	}
    	if(st.empty())
    		count++;
    }
    cout<<count;
    return 0;
}
