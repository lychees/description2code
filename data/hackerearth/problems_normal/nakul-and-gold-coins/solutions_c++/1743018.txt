#define ull unsigned long long
#define inf 1000000007 
#define mod 1000000007
#define all(x) x.begin(),x.end() 
#define N 1000001
#define pi acos(-1.)
#define DBG(x) cerr<<#x<<"="<<x<<"\n"; 
#define FOREACH(it,x) for(__typeof(x.begin()) it=x.begin();it!=x.end();it++) 
using namespace std; 

int p[N],q[N],f[10010];
vi v[10010];
int find(vi &v,int n){
    return lower_bound(all(v),n)-v.begin();
}
int main()
{
    int T,i,j,k,ca=0,n,m;
    int t=0;
    for(i=2;i<N;i++){
        if(!p[i])q[t++]=i;
        for(j=0;j<t;j++){
            if(1LL*q[j]*i>=N)break;
            p[i*q[j]]=1;
            if(i%q[j]==0)break;
        }
    }
    //DBG(t)
    int sz=10000,cnt=0;
    for(i=0;i<t;i++){
        for(j=i+1;j<t;j++){
            ll x=1LL*q[i]*q[j];
            if(x>1e8)break;
            f[x/sz]++;
            v[x/sz].pb(x);
            cnt++;
        }
    }
    for(i=0;i<sz+2;i++)sort(all(v[i]));
    //DBG(cnt)
    scanf("%d",&T);
    while(T--){
        int l,r;
        scanf("%d%d",&l,&r);
        int x=l/sz,y=r/sz,ans=0;
        //cerr<<x<<" "<<y<<"\n";
        if(x==y){
            x=find(v[x],l);
            y=find(v[y],r+1)-1;
            if(x<=y)ans=y-x+1;
        }
        else{
            for(i=x+1;i<y;i++)ans+=f[i];
            i=find(v[x],l);
            if(i!=v[x].size())ans+=v[x].size()-i;
            j=find(v[y],r+1)-1;
            if(j>=0)ans+=j+1;
        }
        printf("%d\n",ans);
    }
    return 0;
}
