template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

int ans[100005], a[100005], p[100005];
bool used[100005];

int main()
{	
	int n;
	ll t;
	
	cin >> n >> t;
	
	for (int i = 1; i <= n; i++)
		scanf("%d", a + i);
		
	for (int i = 1; i <= n; i++)
		scanf("%d", p + i);
		
	for (int i = 1; i <= n; i++) {
		if (!used[i]) {
			int len = 0, start = i, cur = p[start];
			
			used[i] = 1;
			
			while (cur != start) {
				used[cur] = 1;
				len++;
				cur = p[cur];
			}
			
			len++;
			
			int steps = t % len;
			
			int pos = i;
			
			for (int j = 1; j <= steps; j++)
				pos = p[pos];
				
			cur = i;
			
			for (int j = 1; j <= len; j++) {
				ans[cur] = a[pos];
				cur = p[cur];
				pos = p[pos];
			}
		}
	}
	
	for (int i = 1; i <= n; i++)
		printf("%d ", ans[i]);
	
	return 0;	
}
