#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e5+10;
pair<int,int>d[mx];
bool cmp(pair<int,int>a,pair<int,int>b){
   return a.second-a.first<b.second-b.first;
}
int main(){
   int n,q;
   scanf("%d %d",&n,&q);
   for(int i=0;i<n;i++){
      scanf("%d %d",&d[i].first,&d[i].second);
      if(d[i].first>d[i].second)swap(d[i].first,d[i].second);
   }
   sort(d,d+n,cmp);
   uli x=0;
   bool ok=true;
   for(int i=0;i<n;){
      int l=i;
      while(i<n && d[i].second-d[i].first==d[l].second-d[l].first){
         i++;
      }
      if((i-l)%2==1){
         ok=false;
         break;
      }
      for(int j=l+1;j<i;j+=2){
         x+=uli(d[j-1].first+d[j].second);      
      }
   }
   if(!ok)x=-1;
   while(q--){
      uli v;
      scanf("%lld",&v);
      if(v==x)puts("YES");
      else puts("NO");
   }
   return 0;
}
