#include <iostream>
#include<map>
using namespace std;
int main() {
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	map<string,int>m;
	int n;
	cin>>n;
	int k=(1<<n)-1;
	while(k--) {
		string a, b;
		cin>>a>>b;
		m[a]++;
	}
	map<string,int>::iterator it=m.begin();
	int ma=it->second;
	string p=it->first;
	for(; it!=m.end(); it++)
		if(it->second>ma) {
			ma=it->second;
			p=it->first;
		}
	cout<<p<<"\n";
	return 0;
}
