#include <bits/stdc++.h>
using namespace std;

int f1(int n)
{
	if( n == 0)
		return 1;
	if(n <0)
		return 0;
	return f1(n - 1) + f1(n - 2) + f1(n-3);
}

int f2(int n)
{
	if( n == 0)
		return 1;
	if(n <0)
		return 0;
	return f2(n - 1) + f2(n - 2);
}

int main()
{
    int n;
    cin>>n;
    if(n > 0)
    {
    	cout<<"Jack-"<<f1(n)<<endl;
    	cout<<"Jill-"<<f2(n)<<endl;
    }
    else cout<<"ERROR\n";
    return 0;
}
