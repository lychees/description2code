#define abs(x) ((x)>0?(x):-(x))
#define MOD 1000000007
#define LL signed long long int

#define print(x) printf("%d\n",x)
#define scanll(x) scanf("%lld",&x)

vector<int> G[2*100000+5];
LL pow(LL base, LL exponent,LL modulus)
{
	LL result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}

LL fact[1000000+5],ifact[1000000+5];
void pre_calc(int N)
{
    fact[0]=ifact[0]=1;
    for(int i=1; i<=N; ++i)
    {
        fact[i]=(fact[i-1]*i)%MOD;
        ifact[i]=pow(fact[i],MOD-2,MOD);
    }
}
LL ncr(int n,int r)
{
	if(n<0 or r<0 or n<r)
		return 0;
    LL ans=fact[n];
    ans=(ans*ifact[r])%MOD;
    ans=(ans*ifact[n-r])%MOD;
    return ans;
}
int N,K;
int isVowel[3005];
int dp[3005][3005][4];
int rec(int i,int L,int p){
	if(p==3 or L<0)
		return 0;
	if(i==N+1){
		if(L==0)
			return 1;
		return 0;
	}
	if(dp[i][L][p]!=-1)
		return dp[i][L][p];
	int &ret = dp[i][L][p];
	ret = 0;
	int bit = isVowel[i];
	if(bit==0){
		ret += rec(i+1,L-1,0);	//put a one
		ret += rec(i+1,L,p+1);	//put a zero
	}
	else{
		ret += rec(i+1,L,bit==0?p+1:0);	//put a zero
	}
	ret %= MOD;
	return ret;
}
int main(){
	//fre;
	pre_calc(3000);
	memset(dp,-1,sizeof dp);
	string S;
	cin>>N>>K>>S;
	int V = 0;
	rep(i,0,N-1){
		if(S[i]=='a' or S[i]=='e' or S[i]=='i' or S[i]=='o' or S[i]=='u' or S[i]=='y' ){
			isVowel[i+1] = 1;
			++V;
		}
	}
	LL good = 0;
	for(int i=0;i<=K;++i){
		LL temp = rec(1,i,0) * ncr(V,K-i) % MOD;
		int one = V + i;
		int cons = N-one;
		if(one >= cons)
			good = (good + temp) % MOD;
	}
	LL ans = ncr(N,K);
	//cout<<ans<<' '<<good<<endl;
	ans = (ans + MOD - good) % MOD;
	printll(ans);
}
