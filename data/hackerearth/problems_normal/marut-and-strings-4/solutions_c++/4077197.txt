#include <iostream>
#include<string.h>
using namespace std;
int main()
{int i;
   int t;
   int u=0;
   int l=0;
   char c[100];
   cin>>t;
   if(t>=1&& t<=10)
   {
  
   
  
   while(t!=0)
   {
cin>>c;
if(strlen(c)>100)
{
cout<<"Invalid Input \n";
}
else{
for(i=0;i<strlen(c);i++)
{
   if(c[i]>=65 && c[i]<=90)
   {

      u++;
   }
   else if(c[i]>=97 &&c[i]<=122)
   {

      l++;
   }

}
if(u==0 && l==0)
{

 cout<<"Invalid Input"<<"\n";
}
else if(u<l)
{
   cout<<u<<"\n";
}
else if(l<u)
{
    cout<<l<<"\n";
}
else if(l==u)
{

 cout<<l<<"\n";
}
l=0,u=0;
}
      t--;
   }
   }
   else{
   	 	cout<<"Invalid Test \n";
   }
    return 0;
}
