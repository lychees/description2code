#include<stdio.h>
#include<malloc.h>
long long int a[1000002];
int main()
{
	long long int flag,t,i,j,n,c;
	scanf("%lld", &t);
	while (t--)
	{
		c = 0;
		scanf("%lld", &n);
		//long long int *a = new long long int[n];
		for (i = 0; i < n; i++)
		{
			scanf("%lld", &a[i]);
		}
		for (i = 0; i < n - 1; i++)
		{
			flag = 0;
			for (j = i + 1; j < n; j++)
			{
				
				if (a[j]>=a[i])
				{
					flag = 1;
					break;
				}
				
			}
			if (flag == 0)
				c++;
			//free(a);
		}
		printf("%lld\n", c + 1);
	}
	return 0;
}
