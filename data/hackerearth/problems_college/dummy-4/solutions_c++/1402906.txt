#include<iostream>
#include<string>
using namespace std;
int main()
{
	  int n,rem,ans;
	  cin >> n;
	  while(n--)
	  {
  		string x;
  		cin >> x;
  		 rem=0;
  		for( int i = 0 ; i < x.size() ; ++i ){
		  	rem = (rem*10+(x[i]-'0'))%9;
		  }
		  ans = (ans + rem)%9;
  	}
  	cout << ans << endl; 
  	   return 0;
}
