#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
bool search(vector<ll>& arr,int val)
{
  ll n=arr.size();
  int i=0,j=n-1;
  ll mid=0;
  bool flag=false;
  while(i<j)
  {
      mid=(i+j)/2;
      if(arr[mid]==val)
             return true;
      else if(val>arr[mid])
             i=mid+1;
      else
            j=mid;
  }
  return false;
}
int main()
{
   ll n,a,aux;
  cin>>n;
   vector<ll> store;
   for(int i=0;i<n;i++)
   {
        cin>>a;
       store.push_back(a);
   }
  sort(store.begin(),store.end());
  ll m;
  cin>>m;
  bool flag=false;
  for(int i=0;i<m;i++)
  {
    flag=false;
    cin>>a;
    if(search(store,a))
      {
         cout<<"I Like You"<<endl;
         continue;
      }
    else
     {
       for(int i=0;i<n;i++)
       {
           if(store[i]<a)
                {
                   aux=a-store[i];
                   if(search(store,aux)) {flag=true;break;}
                      
                }
            
          }
       
     }
     if(flag)
                    cout<<"I Like You"<<endl;
            else
                            cout<<"I Hate You"<<endl;
  }
}
