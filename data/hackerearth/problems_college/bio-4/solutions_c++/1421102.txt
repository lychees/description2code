#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long t,n,m,i,j,k,ans;
    string s1,s2;
    cin>>t;
    while(t--)
    {
    	cin>>s1;
    	cin>>s2;
    	n=s1.size();
    	m=s2.size();
    	ans=1000000000000;
    	for(i=0;i<n-m+1;i++)
    	{
    		k=0;
    		for(j=0;j<m;j++)
    		{
    			if(s1[i+j]!=s2[j])
    				k++;
    		}
    		if(k<ans)
    			ans=k;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
