#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,a[100000],sum,maxsum,i,j;
    cin>>t;
    while(t--){
    	cin>>n;
    	n=2*n;
    	for(i=0;i<n;i++){
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	maxsum=0;
    	for(i=0,j=n-1;i<j;i++,j--){
    		sum=a[i]+a[j];
    		if(sum>maxsum){
    			maxsum=sum;
    		}
    	}
    	cout<<maxsum<<endl;
    }
    return 0;
}
