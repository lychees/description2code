#include <iostream>

using namespace std;

int main()
{
    int n;
    cin >> n;
    for (int i = 0; i < n; i++){
    	string s1, s2;
    	cin >> s1 >> s2;
    	int is_set = 0;
    	for (string::iterator it = s1.begin(); it != s1.end(); it++){
    		int index_char = *it-'a';
    		is_set = is_set | (1 << index_char);
    	}
    	bool is_common = false;
    	for (string::iterator it = s2.begin(); it != s2.end(); it++){
    		int index_char = *it-'a';
    		if (is_set >> index_char & 1){
    			is_common = true;
    			break;
    		}
    	}
    	
    	if (is_common) cout << "Yes" << endl;
    	else cout << "No" << endl;
    }
    return 0;
}
