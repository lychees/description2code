#include<bits/stdc++.h>
using namespace std;
bool visited[1001];
int sum;
vector<int> v[1001];
void dfs(int s)
{
    visited[s]=true;
    for(int i=0;i<v[s].size();i++)
    {
        if(!visited[v[s][i]])
        {
            sum++;
            dfs(v[s][i]);
        }
    }
}

int main()
{
    int n,d,mini=10e6,a,count,b;
    cin>>n>>d;
    for(int i=0;i<d;i++)
    {
        cin>>a>>b;
        v[a].push_back(b);
    }
    for(int i=1;i<=n;i++)
    {
        sum=1;
        memset(visited,false,sizeof(visited));
        dfs(i);
        mini=min(mini,sum);
    }
    cout<<mini<<endl;
    return 0;
}
