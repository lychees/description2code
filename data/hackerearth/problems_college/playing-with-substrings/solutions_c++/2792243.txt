#include<iostream>
#include<string>
#include<set>

using namespace std;

bool isSubstring(string s, string t)
{
    size_t found = s.find(t);
    if(found!=std::string::npos)
        return true;
    else
        return false;
}

int main()
{
    string t, s;
    cin>>s;
    cin>>t;
    set<string> str;
    for(int i=1;i<=s.length();i++)
    {
        for(int j=0;j<s.length();j++)
            str.insert(s.substr(j,i));
    }
    /*cout<<str.size()<<endl;
    //set<string> iterator::it;
    for(set<string>::iterator it=str.begin();it!=str.end();it++)
        cout<<*it<<" ";*/
    int c=0;
    set<string>::iterator it;
    for(it=str.begin();it!=str.end();it++)
    {
        if(isSubstring(*it,t))
            c++;
    }
    cout<<c;

    return 0;
}
