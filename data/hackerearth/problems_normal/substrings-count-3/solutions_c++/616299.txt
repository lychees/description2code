#include <iostream>
#include <algorithm>
#include <vector>
#include <unordered_map>
using namespace std;

int main()
{
    int n,q,l,r;
    string s;
    cin>>n;
    unordered_map<string, vector<int> > p;
    unordered_map<string, int> dict;
    for(int i=0;i<n;i++){
    	cin>>s;
    	int ln = s.size();
    	string x;
    	for(int j=0;j<ln;j++){
    		x.clear();
    		for(int len=1;j+len-1<ln;len++){
    			x.push_back(s[j+len-1]);
    		    if(p.find(x)==p.end())
    		      p[x].push_back(i);
    		    if(p[x].back()!=i)
    		      p[x].push_back(i);    		    
    		}
    	}
    }
    cin>>q;
    for(int i=0;i<q;i++){
    	cin>>l>>r;
    	l--;r--;
    	cin>>s;
    	if(p.find(s)!=p.end()){
    		cout<<(upper_bound(p[s].begin(), p[s].end(), r) - lower_bound(p[s].begin(), p[s].end(), l))<<endl;
    	}else
    		cout<<0<<endl;
    }
    return 0;
}
