int ef[101];
	int ti[101];
	int di[101];
int mindistruct(int ce,int ct,int n,int i)
{
	int minc =0;
	int mexc =0;
	if(i>n)
		return 0;
	if(i<=n)
	{
		if(ce >= ef[i] && ct>=ti[i])
		{
			mexc =  mindistruct(ce,ct,n,i+1);
			minc = di[i] + mindistruct(ce-ef[i],ct-ti[i],n,i+1);
			//cout<<"minc :: "<<minc<<endl;
			//cout<<"mexc :: "<<mexc<<endl;
			if(minc >= mexc)
				return minc;
			else
				return mexc;
		}
		else
		{
			return mindistruct(ce,ct,n,i+1);
		}
		
	}
	
}	
int main()
{
	int n;
	int e;

	cin>>n;

	int k =1;
	int total = 0;
	for(k;k<=n;++k)
	{
		cin>>ef[k];
	}
	for(k=1;k<=n;++k)
	{
		cin>>ti[k];
	}
	for(k=1;k<=n;++k)
	{
		cin>>di[k];
		total += di[k];
	}
	cout<<total - mindistruct(e,t,n,1)<<endl;
	return 0;
}
