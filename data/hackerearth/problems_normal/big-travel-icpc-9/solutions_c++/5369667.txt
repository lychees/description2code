#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n,i,j,ans=0;
    scanf("%ld",&n);
    long int a[n],b[n];
    for(i=0;i<n;i++)
    scanf("%ld %ld",&a[i],&b[i]);
    sort(a,a+n);
    sort(b,b+n);
    for(i=0;i<n-1;i++)
    {
    	ans=(ans+(a[i+1]-a[i])*((i+1)*n-(i+1)*(i+1)))%1000000007;
    	ans=(ans+(b[i+1]-b[i])*((i+1)*n-(i+1)*(i+1)))%1000000007;
    }
    printf("%ld",ans);
    return 0;
}
