#include <iostream>
#include <cmath>
using namespace std;
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		long long n,m1,m2,n1,n2;
		double x;
		cin>>n;
		x = sqrt(n);
		if(x==(int)sqrt(n))
			cout<<"YES"<<endl;
		else
		{
			m1 = (int)sqrt(n);
			m2 = m1+1;
			n1=m1*m1;
			n2=m2*m2;
			n1 = n - n1;
			n2 = n2 - n;
			if(n1<n2)
				cout<<"-"<<n1<<endl;
			else if(n1>n2)
				cout<<"+"<<n2<<endl;
		}
	}
	return 0;
}
