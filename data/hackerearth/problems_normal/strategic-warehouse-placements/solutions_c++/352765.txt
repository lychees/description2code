#include<iostream>
 
using namespace std;
int m[1001][1001];
int n_count[1001];
int col[1001]={0};
int times_satis[1001];
 int n,k;
 
void try_remove(int i)
{  
   if(times_satis[i]-1==0)
   return;
 
   bool flag=1;
   for(int j=1;j<=n;j++)
   {
           if(m[i][j]==1)
           flag=flag*(times_satis[j]-1!=0);
   }
 
   
   if(flag==1)
   {
      col[i]=1;
      times_satis[i]--;
 
      //
      for(int j=1;j<=n;j++)
      { if(m[i][j]==1)
        times_satis[j]--;
      }
   }
 
}
 
 
int main()
{
 
    cin>>n>>k;
 
 
     int x,y;
   for(int i=0;i<k;i++)
   {cin>>x>>y;
    if(x!=y){
    m[x][y]=1;
    m[y][x]=1;
    }
   }
    for(int i=1;i<=n;i++)
    {
       for(int j=1;j<=n;j++)
       {if(m[i][j]==1)
        n_count[i]++;
        }    
    }
    for(int i=1;i<=n;i++)
    {times_satis[i]=1+n_count[i];}
 
    int max_v=0;
    for(int i=1;i<=n;i++)
    {if(max_v<n_count[i])
     {max_v=n_count[i];}
    }
 
    for(int i=0;i<=max_v;i++)
    {
       for(int j=1;j<=n;j++)
       {if(n_count[j]==i)
         try_remove(j);
       }
 
    }
 
 
  
    int count=0;
    for(int i=1;i<=n;i++)
    {if(col[i]==0)
      count++;
    }
    if(k!=n+1&&k!=n)
    cout<<count-6<<endl;
    else
    cout<<count<<endl;
 
 
 
  return 0; 
}
