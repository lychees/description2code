#include <iostream>
#include <iomanip>
#include <cmath>
#define float double
using namespace std;

float area(float a,float b,float c)
{
	float at,s,ans;
	
	s= (a+b+c)/2;
	
	at = sqrt(s*(s-a)*(s-b)*(s-c));
	
	ans= 3.1415 * (a*b*c)/(4*at) * (a*b*c)/(4*at);
	
	return ans;
	
}

int main()
{
    int tno;
    float a,b,c,ans;
    
    cin>>tno;
    
    while(tno--)
    {
    	cin>>a>>b>>c;
    	
    	ans = area(a,b,c);
    	
    	
    	cout<<setprecision(4)<<fixed<<ans<<endl;
    }
    return 0;
}
