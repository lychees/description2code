#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t, n, len, i, sum, count, ans;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	cin>>n;
    	count = 0;
    	len = s.length();
    	for(i=0;i<len;i++){
    		if(s[i]=='1'){
    			count++;
    		}
    	}
    	sum = n*(n+1)/2;
    	if(sum%2==0){
    		ans = count;
    	}
    	else{
    		ans = len-count;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
