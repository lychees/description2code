#include <iostream>
using namespace std;

int main()
{
   int t,i,j,a,b,c;
   cin>>t;
   while(t--)
   { 
   	a=0;
   	b=0;
   	c=3;
   	string s;
   	cin>>s;
   	for(i=0;i<s.length();i++)
   	{
   		if(s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u'||s[i]=='y'){
  	   a++;
  	   c=3;  }
  	      else
  	      {
  	      	b++;
  	      	c--;
  	      	if(c==0){
  	      	cout<<"hard\n";
  	      	break;
  	      	}
  	      }
   	}
   	if(c!=0)
   	{
   		if(b>a)
   		cout<<"hard\n";
   		else
   		cout<<"easy\n";
   	}
   }
   return 0;
}
