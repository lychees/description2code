#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	int flag=0,n,i,x,j;
    	cin >> n;
    	for(i=0 ; i<=n ; i++)
    	{
    		x=n-i;
    		if(x%3==0 && i%5==0)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==1)
    	{
    		for(j=0 ; j<x ; j++)
    		cout << 5;
    		for(j=0 ; j<i ; j++)
    		cout << 3;
    	} 
    	else
    	cout << "-1";
    	
    	cout << "\n";
    }
    return 0;
}
