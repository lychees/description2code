#include <iostream>
#include <cmath>
using namespace std;
#define MAX 1000001
int main()
{
    long long int t,n,m,i,j,s;
    int *p = (int *)malloc(sizeof(int)*MAX);
    
    p[0]=p[1]=0;
    for(i=2;i<MAX;i++){
    	p[i]=1;
    }
    
    for(i=2;i<MAX;i++){
    	if(p[i]){
    		for(j=i*i;j<MAX;j+=i){
    			p[j]=0;
    		}
    	}
    }
    
    
    cin>>t;
    while(t--){
    	cin>>m>>n;
    	s=0;
    	for(i=m;i<=n;i++){
    		if(p[i]){
    			s+=i;
    		}
    	}
    	
    	int f=1;
    	if(s<MAX){
    		f=p[s];
    	}else{
    		for(i=2;i<=sqrt(s);i++){
    			if(s%i==0){
    				f=0;
    				break;
    			}
    		}
    	}
    	
    	if(f){
    		cout<<"YES"<<endl;
    	}else{
    		cout<<"NO"<<endl;
    	}
    }
    
    return 0;
}
