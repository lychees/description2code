#include <bits/stdc++.h>

#define MOD 1000000007
bool isprime[2000005];

using namespace std;

int main()
{
   long long int n, ans, i, j;
   cin >> n;
   for( i = 0; i <= 2*(n-1); i++ )
   {
   		isprime[i] = 1;
   }
   isprime[0] = 0;
   isprime[1] = 0;
   for( i = 2; i * i <= 2*(n-1); i++ )
   {
   		if( isprime[i] == 1 )
   			for( j = i + i; j <= 2*(n-1); j += i )
   			{
   				isprime[j] = 0;
   			}
   }
   ans=0;
   for( i = 0; i <= n - 1; i++ )
   	if( isprime[i] == 1 )
   	{
   	//	cout << i+j <<" i\n";
   		ans = (ans + i + 1) % MOD;
   	}
   for( i = n ,j = (n - 1); i <= 2 * (n - 1); i++ ,j-- )
   {
   		if( isprime[i] == 1 )
   			ans = (ans + j) % MOD;
   }
   cout << ans;
}
