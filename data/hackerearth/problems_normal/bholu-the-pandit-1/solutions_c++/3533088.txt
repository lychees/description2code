#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,a,b,c,i,ans,p=1000000007;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld %lld %lld",&a,&b,&c);
    	ans=__gcd(a,__gcd(b,c));
    	a=a/ans;b=b/ans;c=c/ans;
    	printf("%lld %lld\n",ans,(((a%p)*(b%p))%p*(c%p))%p);
    }	
    return 0;
}
