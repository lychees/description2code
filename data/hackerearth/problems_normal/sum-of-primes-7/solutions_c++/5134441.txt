#include <bits/stdc++.h>

#define pb push_back
#define fr first
#define se second
#define mp make_pair
#define memo(a,b) memset(a,b,sizeof(a))
#define MAX 1000001

using namespace std;
typedef long long ll;
ll n,t,v,s,x,y,a[MAX];
bool f;
int main()
{
  for(ll i = 2;i*i<MAX;++i){
    if(!a[i]){
      for(ll j = i*i;j<MAX;j+=i)
	a[j] = 1;
    }
  }
  for(ll i = 2;i<MAX;++i){
    if(!a[i])
      a[i] = i;
    else
      a[i] = 0;
    a[i]+=a[i-1];
  }
  scanf("%lld",&t);
  while(t--){
    scanf("%lld%lld",&x,&y);
    s = a[y]-a[x-1];
    printf("%lld\n",s);
  }
}
