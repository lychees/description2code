#include <iostream>
using namespace std;

int n;
int a[100][100];
int solve(int i , int j){
	if(i==n-1 && j==n-1 && a[i][j]==1)
		return 1;
	//else if(i==n-1 && j==n-1 && a[i][j]!=0)
	//	return 0;
	
	if(a[i][j]==0)
		return 0;
	
	if(i+1>n || j+1>n)
		return 0;
	
	return solve(i+1,j)||solve(i,j+1);
	
}
int main()
{
	int t;
	cin>>t;
	
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				cin>>a[i][j];
		
/*
		for(int i=0;i<n;i++)
			{for(int j=0;j<n;j++)
				cout<<a[i][j]<<" ";
			cout<<endl;		
			}
*/			
		if(!solve(0,0))
			cout<<"NOT POSSIBLE\n";
		else
			cout<<"POSSIBLE\n";
	}
	return 0;
}
