#include <iostream>
#include <stdio.h>

using namespace std;

typedef unsigned long long int Lint;

Lint GCD( Lint a, Lint b ) {
	
	if( ( a % b ) == 0 )
		return b;
	else
		return GCD( b, a % b );
}

int main()
{
	
	Lint t, a, b;
	
	scanf("%llu", &t );
	
	while( t-- ) {
		
		scanf("%llu %llu", &a, &b );
		
		if( GCD( a, b ) == 1 )
			printf("Is a Co-Prime\n");
		else
			printf("Not a Co-Prime\n");
	}
	return 0;
}
