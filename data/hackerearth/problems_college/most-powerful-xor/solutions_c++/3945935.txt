#include <bits/stdc++.h>
//zone
using namespace std;

int main()
{
    int i,j;
    int t;
    cin >> t;
    int l,r;
    while(t--)
    {
    	cin >> l;
    	cin >> r;
    	int maxi = 0;
    	for(i = l ; i <= r ;i++)
    	{
    		for(j = l + 1; j<=r; j++)
    		{
    			maxi = max(maxi,i^j);
    		}
    	}
    	cout<<maxi<<endl;
    }
    
    return 0;
}
