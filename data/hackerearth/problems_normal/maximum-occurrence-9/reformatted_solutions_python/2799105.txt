l = raw_input()
l = sorted([x for x in l])
maxchar = l[0]
maxcount = 1

currentchar = l[0]
currentcount = 1

length = len(l)

for i in xrange(1, length):

    if l[i] == l[i-1]:
        currentcount += 1
        if currentcount > maxcount:
            maxcount = currentcount
            maxchar = currentchar
    else:
        currentchar = l[i]
        currentcount = 1
print maxchar, maxcount
