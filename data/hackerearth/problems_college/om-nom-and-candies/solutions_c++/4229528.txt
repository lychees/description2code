#include<bits/stdc++.h>
#define ll long long
#define fo(i,n) for(i=0;i<n;i++)
#define forr(i,n) for(i=n-1;i>=0;i--)
using namespace std;
int main(){
 ll n, i, j, x;
 long double d;
 cin>>n;
 ll a[n+1];
 for(i=1;i<=n;i++)
    cin>>a[i];
 ll s[n+1];
 s[1] = a[1];
 vector<ll>divisor;
 for(i=2;i<=n;i++)
    s[i] = a[1] + a[i];
 for(i=2;i<=n;i++){
    divisor.clear();
    for(j=1;j<sqrt(i);j++){
        if(i%j==0)
        {
            divisor.push_back(j);
            divisor.push_back(i/j);
        }
    }
    d = sqrt ( i );
    if(d==int(d)){
        x = d;
        if(i%x==0)
            divisor.push_back(sqrt(i));
    }
    sort(divisor.begin(),divisor.end());

    for(j=1;j<divisor.size()-1;j++){
            if(s[divisor[j]]+a[i]>s[i])
                s[i] = s[divisor[j]] + a[i];
    }
 }
 for(i=1;i<=n;i++)
    cout<<s[i]<<" ";
 cout<<endl;
 return 0;

}
