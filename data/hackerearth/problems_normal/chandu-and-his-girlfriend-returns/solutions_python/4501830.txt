from __future__ import print_function
import bisect
tc=input()

while tc>0:

    n,m=tuple(map(int, raw_input().split()))
    narr=map(int, raw_input().split())[::-1]

    for i in map(int, raw_input().split())[::-1] :
        bisect.insort(narr, i)
        

    print(*narr[::-1], sep=" ") 
    tc=tc-1
