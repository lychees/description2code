#include <iostream>
using namespace std;

int main()
{
    int n,i,j;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    int q;
    cin>>q;
    while(q--)
    {
    	int t;
    	cin>>t;
    	int f=0;
    	for(i=0;i<n;i++)
    	{
    		if(t==a[i])
    		{
    			f=1;
    			break;
    		}
    		else
    		{
    			for(j=i+1;j<n;j++)
    			{
    				if(a[i]+a[j]==t)
    				{
    					f=1;
    					break;
    				}
    			}
    		}
    	}
    	if(f==0)
    	cout<<"I Hate You"<<endl;
    	else
    	cout<<"I Like You"<<endl;
    }
    return 0;
}
