using namespace std;

int flag[1000001];
int temp[1000001];
int k=0;
void sieve(int n)
{
	int i,j;
  for(i=2;i<=n;i++)
  {
  	if(flag[i]==0)
  	temp[k++]=i;
  	for(j=i;j<=n;j+=i)
  	flag[j]=1;	
  }
	
}


int main()
{
	int x1,x2;
	int i,j,t,sum,n,h=0;
	sieve(1000000);
	cin>>t;
	while(t--)
	{
		cin>>n;
		i=0;
		j=k-1;
		h=0;
		x1=-1;
		x2=-1;
		while(i<j)
		{
			sum=temp[i]+temp[j];
			if(sum==n)
			{
				h=1;
				x1=temp[i];
				x2=temp[j];
				i++;
				j--;
			}
			else
			if(sum>n)
			j--;
			else
			if(sum<n)
			i++;
		}
		if(h==0)
		cout<<"-1 -1\n";
		else
		cout<<x1<<" "<<x2<<"\n";
	}
	return 0;
	
}
