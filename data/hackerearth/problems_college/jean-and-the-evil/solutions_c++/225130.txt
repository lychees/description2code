#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
    long long int sum, t, n, *arr, i, ans=0;
    scanf("%lld", &t);
    while (t--)
    {
        scanf("%lld", &n);
        arr=new long long int[n];
        for (i=0; i<n; i++)
        {
            scanf("%lld", &arr[i]);
        }
        sum=0;
        ans=arr[0]>0?arr[0]:0;
        for (i=0; i<n; i++)
        {
            if ((arr[i]+sum)>0)
            {
                arr[i]=arr[i]+sum;
                sum=arr[i];
            }
            else
            {
                sum=0;
                arr[i]=0;
            }
            if (arr[i]>ans)
            {
                ans=arr[i];
            }
        }
        printf("%lld\n", ans);
        //delete [] arr;
    }
    return 0;
}
