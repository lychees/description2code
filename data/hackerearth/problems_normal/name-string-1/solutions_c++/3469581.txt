#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    
    int t;
    cin>>t;
    while(t--)
    {
    	
    	int a,b,c;
    	cin>>a>>b>>c;
    	int mx = max(a,max(b,c));
    	int mn = min(a,min(b,c));
    	a = (a+b+c)-mx-mn;
    	if(mx==a)
    		cout<<2*a<<endl;
    	else	
    		cout<<2*a+1<<endl;
    	
    }
    
    return 0;
}
