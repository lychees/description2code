#include<iostream>
using namespace std;
#define ll long long
int main()
{
	ll n,count=0,i,l;
	string s1,s2;
	cin>>s1>>s2;
	l=s1.size();
	for(i=0;i<l;i++)
	{
		if(s1[i]!=s2[i])
			count++;
		if(count>1)
			break;
	}
	if(count>1)
		cout<<"loses";
	else
		cout<<"wins";
	return 0;
}
