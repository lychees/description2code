#include<bits/stdc++.h>
using namespace std;
vector<pair<long long int,int>>aray;
int main()
	{
		int N,mn,val,len,cnt;
		cin>>N;
		long long int x,sum;
		aray.push_back(make_pair(0,1));
		sum=0;
		for(int i=1;i<=N;i++)
			{
				scanf("%lld",&x);
				sum=sum+x;
				aray.push_back(make_pair(sum,i+1));
			}
		sort(aray.begin(),aray.end());
		mn=aray[0].second;
		len=0;
		val=0;
		cnt=0;
		for(int i=1;i<=N;i++)
			{
				val=max(0,aray[i].second-mn);
				if(val>len)
					{
						len=val;
						cnt=1;
					}
				else if(val==len) cnt++;
				mn=min(mn,aray[i].second);
			}
		if(len==0)
			cout<<"-1"<<endl;
		else
			cout<<len<<" "<<cnt<<endl;
	}
