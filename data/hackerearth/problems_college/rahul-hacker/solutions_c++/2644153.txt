#include<iostream>
#include<stdlib.h>
using namespace std;
	long long int fun(	long long int n){
			long long int sum=0,r;
		while(n!=0){
			r=n%10;
			sum=sum*10+r;
			n=n/10;
		}
		return sum;
	}
int main(){
	long long int t;
	cin>>t;
	while(t--){
				long long int a,b;
				cin>>a>>b;
					long long int k=fun(a)+fun(b);
					
					cout<<fun(k)<<"\n";
	}
return 0;
}
