def rev(string):
    return string[::-1]


answers = []
t = int(raw_input())
for i in range(t):
    word = raw_input()
    reverse = rev(word)
    limit = len(word)
    flag = 0
    for i in range(1, limit):
        if abs(ord(word[i]) - ord(word[i-1])
               ) != abs(ord(reverse[i]) - ord(reverse[i-1])):
            flag = 1
            break
    if flag == 0:
        answers.append("Funny")
    else:
        answers.append("Not Funny")
for i in answers:
    print i
