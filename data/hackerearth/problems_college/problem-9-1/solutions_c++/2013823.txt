#include <iostream>

using namespace std;

int main()
{
	bool flag;
	int a,b,i,n,Q,heads;
	cin>>n;
	cin>>Q;
	bool coin[n];
	for(i=0;i<n;i++)
	 coin[i]=0;
	while(Q--)
	{
		cin>>flag;
		cin>>a>>b;
		if(!flag)
		 for(i=a;i<=b;i++)
		  coin[i]=!coin[i];	
		else
		{
			heads=0;
			for(i=a;i<=b;i++)
			 if(coin[i])
			  heads++;
			cout<<heads<<endl; 
		}
	}	
    return 0;
}
