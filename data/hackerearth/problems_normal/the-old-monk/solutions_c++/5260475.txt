#include <iostream>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	long a[n], b[n];
    	for(int i = 0; i < n; i++){
    		cin>>a[i];
    	}
    	for(int i = 0; i < n; i++){
    		cin>>b[i];
    	}
    	
    	int ans = 0; 
    	
    	for(int i = 0; i<n; i++){
    	//	int max = 0;
    		for(int j = i; j<n; j++){
    			if(a[i]>b[j]){
    				break;
    			}
    			if(ans<(j-i)){
    				ans=j-i;
    			}
    		}
    	}
    	cout<<ans<<endl;
    }
}
