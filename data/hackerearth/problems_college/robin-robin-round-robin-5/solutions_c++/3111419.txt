#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n, c, q,t= 0,i;
    cin >> n >> c;
    vector <long long int> a(n),b(n), robin(n);
    vector <pair <long long int,long long int> > roundrob(n);
    for (i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] % c != 0) 
        roundrob[i].first = (a[i]/c) + 1;
        else
        roundrob[i].first = a[i]/c;
        roundrob[i].second = i;
        robin[i] = t;
        t=t+min(c, a[i]);
    }

    sort(roundrob.begin(),roundrob.end());
    for (i = 0; i < n; i++)
    {
    	b[roundrob[i].second] = i+1;
}
    cin >> q;
    while (q--) {
        int m, d;
        cin >> m >> d;
        if (m == 1)
        cout << robin[d-1] << endl;
        else
        cout << b[d-1] << endl;
    }
    return 0;
}
