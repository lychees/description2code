#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

int main()
{
    int T, i, n;
    string s;
    cin>>T;
    while(T--)
    {
    	cin>>s;
    	n=s.length();
    	int flag=0;
    	for(i=1; i<n; ++i)
    	{
    		if(abs(s[i]-s[i-1])!=abs(s[n-i-1]-s[n-i]))
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    		cout<<"Funny\n";
    	else
    		cout<<"Not Funny\n";
    	s.clear();
    }
    return 0;
}
