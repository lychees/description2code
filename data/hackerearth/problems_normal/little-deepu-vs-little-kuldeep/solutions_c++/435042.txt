#define N 2000500
#define ln 15

ll fact[N],n;
int i,j,k,t;

ll power(ll p,ll q){
	ll ans=1;
	while(q){
		if(q&1) ans=(ans*p)%mod;
		p=(p*p)%mod;
		q/=2;
	}
	return ans;
}

int main(){
	fact[0]=fact[1]=1;
	rep(i,2,N){
		fact[i]=(fact[i-1]*i)%mod;
	}
	inp(t);
	while(t--){
		inpll(n);
		ll temp=(fact[n]*fact[n])%mod;
		temp=(temp*(n+1))%mod;
		temp=(fact[2*n]*(power(temp,mod-2)))%mod;
		cout<<temp<<'\n';
	}
	return 0;
}
