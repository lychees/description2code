t = input();
while(t > 0):
	n = input();
	a = [int(i) for i in raw_input().split()]; b = [int(i) for i in raw_input().split()];
	ans = 0;
	for i in range(0, n):
		l = 0; h = n - 1; p = -1;
		while(l <= h):
			m = (l + h) / 2;
			if(b[m] >= a[i]):	p = m; l = m + 1;
			else:	h = m - 1;
		ans = max(ans, p - i);
	print ans;
	t -= 1;
