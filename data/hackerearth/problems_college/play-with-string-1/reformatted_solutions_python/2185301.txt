for _ in xrange(input()):
    s = raw_input()
    r = s[::-1]
    l = len(s)
    flag = 0
    for i in xrange(1, l):
        if abs((ord(s[i])-ord(s[i-1]))) != abs((ord(r[i])-ord(r[i-1]))):
            flag = 1
            break
    if flag == 1:
        print "Not Funny"
    else:
        print "Funny"
