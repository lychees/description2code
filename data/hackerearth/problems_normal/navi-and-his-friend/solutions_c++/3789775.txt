#include <bits/stdc++.h>
using namespace std;
int n;
long long w[17], c[17], W;
int C;
long long dp[(1<<17)];
long long func(int x, int w1, int c1){
	if(c1==C){
		return 0;
	}
	if(dp[x]!=-1){
		return dp[x];
	}
	long long ans=0;
	for(int i=0; i<n; i++){
		if(!(x&(1<<i))){
			if(w1+w[i]<=W){
				ans = max(ans, func(x|(1<<i), w1+w[i], c1+1)+c[i]);
			}
		}
	}
	dp[x]=ans;
	return ans;
}

int main()
{
    int t, d=1;
    cin>>t;
    while(t--){
    	memset(dp, -1, sizeof(dp));
    	cin>>n;
    	for(int i=0; i<n; i++){
    		cin>>c[i]>>w[i];
    	}
    	cin>>W>>C;
    	bool check=true;
    	for(int i=0; i<n; i++){
    		if(w[i]<=W){
    			check=false;
    			break;
    		}
    	}
    	cout<<"For Day #"<<(d++)<<":"<<endl;
    	if(check){
    		cout<<"-1\n";
    	}else{
    		cout<<func(0, 0, 0)<<endl;
    	}
    }
    return 0;
}
