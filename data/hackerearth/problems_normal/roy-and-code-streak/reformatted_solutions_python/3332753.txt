T = input()
for _ in xrange(T):
    N = input()
    maxStreak = 0
    curStreak = 0
    ids = set()
    for __ in xrange(N):
        pId, result = map(int, raw_input().strip().split())
        if result == 0:
            curStreak = 0
        elif pId not in ids:
            ids.add(pId)
            curStreak += 1
            if curStreak > maxStreak:
                maxStreak = curStreak
    print maxStreak
