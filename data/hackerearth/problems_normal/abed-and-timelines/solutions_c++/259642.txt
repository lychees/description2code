#include <iostream>
using namespace std;

#define ll long long
#define vt vector

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x1,y1,x2,y2,p,n,m;
		cin>>n>>m;
		cin>>x1>>y1>>x2>>y2>>p;
		int x = min(abs(x1-x2),n-abs(x1-x2));
		int y = min(abs(y1-y2),m-abs(y1-y2));
		cout << (x+y)*p << endl;
		if((x+y)*p<=1000)
			cout << "You saved the group." << endl;
		else
			cout << "Let go of the group." << endl;
	}
	return 0;
}
