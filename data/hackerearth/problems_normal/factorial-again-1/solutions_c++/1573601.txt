ll powe(int x, int n, int m)
{
	if(!n)
		return 1;
	ll ans = powe(x, n/2, m);
	ans = (ans * ans) % m;
	if(n & 1)
		ans  = (ans * x) % m;
	return ans;
}

			ll ans = 1;
			for (ll i = 3 *n + 1; i < p; i++) {
				ans = (ans * i) % p;
			}
			ans = powe(ans, p - 2, p);
			ans = (-1 * ans + p) % p;
			ll den  = powe(6, n, p);
			den = powe(den, p - 2, p);
			cout << (ans * den) % p << "\n";
		}
	}
    return 0;
}
