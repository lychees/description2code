#include <iostream>
#include<stdio.h>
#define LL long long int
using namespace std;
#define gc getchar_unlocked()  
inline void inp(LL &x) 
{
    register LL c = gc ;
    x = 0;
    for(; ((c<48 || c>57)); c = gc );
    for(; c>47 && c<58 ; c = gc )x = (x<<1) + (x<<3) + c - 48;
}
LL a[25]={0};
void pre()
{
a[0]=0;
LL i,m=1;
for(i=0;i<=19;i++)
{
a[i]=(a[i-1]*6)+m*4;
m=m*10;
}
}

LL fnc(LL x)
{
	if(x<2)return 0;
	LL x2=x/10,d=1,ld=0,e,f,ans=0;
	while(x2)
	{
	d=d*10;
	ld++;
	x2=x2/10;
	}
	e=x/d;
	f=0;
	if(e>2)f++;
	if(e>3)f++;
	if(e>5)f++;
	if(e>7)f++;
	ans=f*d;
	if(ld>0)ans+=(e-f)*a[ld-1];
	if((e==2 || e==3 || e==5 || e==7))
	ans+=(x%d+1);
	else
	if(x>10)
	{
	ans+=fnc(x%d);
	}
	return ans;
}

int main()
{
	pre();
	LL t;
	inp(t);
	while(t--)
	{
	LL x;
	inp(x);
	printf("%lld\n",fnc(x));
	}
	return 0;
}
