#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int players;
    	cin>>players;
    	int matches=0;
    	while(players>1)
    	{
    		matches+=players/2;
    		int tempplayers=(players/2)+(players%2);
    		players=tempplayers;
    	}
    	cout<<matches<<endl;
    }
    return 0;
}
