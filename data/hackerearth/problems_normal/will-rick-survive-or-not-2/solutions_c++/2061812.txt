#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int a[50001]={0}; //to count occurence at same distance
		int n;
		scanf("%d",&n);
		long int i,temp;
		for(i=0;i<n;i++)
		{
			scanf("%ld",&temp);
			a[temp]++;
			
		}
		int flag=1; //to indicate whether walker reached to rick or not
		int j=0;	//to count distance of current walker;
		int res=0; //to count number walkers killed
		int dcount=0;  //to count distance travelled by walkers
		for(i=1;i<=n;i++)  //every walker may be at different positions
		{
			while(a[j]==0)
			{
				j++;
			}
			a[j]--; //walker at pos j is killed
			if(j-dcount<=0) //to check if any walker reached to rick or not
			{
				flag=0;
				break;
			}
			if(i%6==0)
			{
				dcount++;
			}
			res++;
			dcount++;
		}
		if(flag)
		cout<<"Rick now go and save Carl and Judas"<<endl;
		else
		cout<<"Goodbye Rick\n"<<res<<endl;
	
		
		
	}
}
