#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	long long int s=0,a,min=1000000007;
    	for(int i=0;i<n;i++){
    		cin>>a;
    		if(a<min)
    			min=a;
    		s+=a;
    	}
    	if(n>1)
    		s+=(n-2)*min;
    	cout<<s<<endl;
    }
    return 0;
}
