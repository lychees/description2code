#include <iostream>
using namespace std;

long long dp[1234];
const int mod = 1e9 + 7;

long long CT(int n) {
	if(n <= 1)
		return 1;
	if(dp[n] != -1) {
		return dp[n];
	}
	long long ans = 0;
	for(int i = 0; i<n; i++) {
		ans = (ans + (CT(i) * CT(n-i-1)) % mod) % mod;
	}
	return dp[n] = ans;
}

int main() {
    int testCase;
    fill(dp, dp+1234, -1);
    for(cin>>testCase; testCase; testCase--) {
    	int N; cin>>N;
    	cout<<CT(N)<<'\n';
    }
    return 0;
}
