noc = input()
while noc:
    noc -= 1
    s = raw_input()
    sl = len(s)
    i = 0
    ans = "Funny"
    while i < sl//2:
        ld = abs((ord(s[i])) - ord(s[i+1]))
        rd = abs((ord(s[sl - i - 1])) - ord(s[sl-i-2]))
        if ld != rd:
            ans = "Not Funny"
            break
        i += 1
    print ans
