#define MOD 1000000007
#define INF 1000000000000000000
#define PI acos(-1)

using namespace std;

long long GCD (long long a,long long b) {
	if (b == 0) return a;
	return(a % b == 0 ? b : GCD(b,a % b));
}

long long POW (long long base,long long exp) {
	long long val;
	val = 1;
	while (exp > 0) {
		if (exp % 2 == 1) {
			val = (val * base) % MOD;
		}
		base = (base * base) % MOD;
		exp = exp / 2;
	}
	return val;
}

struct ch {
	char c;
	int freq;
};

char s[105];

bool cmp (ch a, ch b) {
	if (a.freq != b.freq) {
		return (a.freq < b.freq);
	} else {
		return (a.c < b.c);
	}
}

int main()
{
	int t;
	scanf("%d",&t);
	while (t --) {
		int i;
		scanf("%s",s);
		vector <ch> v(26);
		for (i = 0; i < 26; i++) {
			v[i].c = 'a' + i;
			v[i].freq = 0;
		}
		for (i = 0; s[i] != '\0'; i++) {
			v[s[i] - 'a'].freq++;
		}
		sort(v.begin(),v.end(),cmp);
		for (i = 0; i < 26; i++) {
			while (v[i].freq --) {
				printf("%c",v[i].c);
			}
		}
		printf("\n");
	}
	return 0;
}
