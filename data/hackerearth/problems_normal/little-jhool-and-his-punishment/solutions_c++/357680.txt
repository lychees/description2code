#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>

using namespace std;

int optimum(int n, int b1, int g1, int start)
{
	int prev = start;
	int score1 = 0;
	if(prev)
		g1--;
	else
		b1--;
	for(int i=1; i<n; i++)
	{
		if(i%2 == 0)
		{
			if(prev)
			{
				if(g1)
				{
					g1--;
					score1++;
				}
				else
				{
					b1--;
					prev = !prev;
				}
			}
			else
			{
				if(b1)
				{
					b1--;
					score1++;
				}
				else
				{
					g1--;
					prev = !prev;
				}
			}
		}
		else
		{
			if(prev)
			{
				if(b1)
				{
					b1--;
					prev = !prev;
				}
				else
				{
					g1--;
					score1++;
				}
			}
			else
			{
				if(g1)
				{
					g1--;
					prev=!prev;
				}
				else
				{
					b1--;
					score1++;
				}
			}
		}
	}
	return score1;
}

int main()
{
	int t, n, b, g, b1, g1;
	cin>>t; 
	while(t--)
	{
		scanf("%d%d%d", &n, &b, &g);
		b1 = b;
		g1 = g;
		int score1 = optimum(n, b, g, 0); //Start with a Boy
		int score2 = optimum(n, b, g, 1); //Start with a Girl
//		cout<<score1<<" "<<score2<<endl;
		if(score1>((n-1)/2) || score2>((n-1)/2))
			cout<<"Little Jhool wins!\n";
		else
			cout<<"The teacher wins!\n";
	}
	return 0;
}
