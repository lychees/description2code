#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
unsigned int v[100001];
int main()
{
    unsigned int n, temp;
    cin >> n;
    for(unsigned int i = 0; i < n; ++i){
    	cin >> v[i];
    }
    sort(v, v + n);
    unsigned int ans = 1, current = v[0];
    for(unsigned int i = 1; i < n; ++i){
    	if(v[i] > current + 4){
    		++ans;
    		current = v[i];
    	}
    }
    cout << ans << endl;
    return 0;
}
