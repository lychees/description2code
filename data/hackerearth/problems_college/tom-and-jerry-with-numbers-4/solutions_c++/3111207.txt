#include <iostream>
using namespace std;
int main()
{
	int t,a,b,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n>>a>>b;
		int arr[n+2];
		for(i=0;i<a;i++)
			arr[i]=0;
		arr[a]=1;
		
		if(b>n && a>n)
			arr[n]=0;
		else if(b>n && a<=n)
			for(i=a+1;i<=n;i++)
			{
			
				if(arr[i-a]==0)
					arr[i]=1;
				else
					arr[i]=0;
			}
		else
			for(i=a+1;i<=n;i++)
			{
				if(arr[i-a]==0 ||arr[i-b]==0)
					arr[i]=1;
				else
					arr[i]=0;
			}
			cout<<arr[n]<<endl;
	}
			return 0;
	}
