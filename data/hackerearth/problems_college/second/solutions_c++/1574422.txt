#include<iostream>
#include<stdio.h>

#define MAX 100000

int main()
{
	int n,i,ans=0,k;
	int temp;
	int a[MAX];
	scanf("%d%d",&n,&k);
	temp=k;
	for(i=0; i<n; i++)
	{
		scanf("%d", &a[i]);
	}
	 
       for(i=0; i<n; i++)
       {
	if(a[i]==temp)
         { 
	 temp=temp+k;	
	 ans=a[i];	 
     i=0;
         }
       }

	 printf("%d\n",ans);
	return 0;
}
