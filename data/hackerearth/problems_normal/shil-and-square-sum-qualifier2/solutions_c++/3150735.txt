#include <bits/stdc++.h>

using namespace std;

const int mod = 1e9 + 7;

inline void add(int &a, int b) {
  a += b;
  if (a >= mod) a -= mod;
}

inline void sub(int &a, int b) {
  a -= b;
  if (a < 0) a += mod;
}

inline int mul(int a, int b) {
  return 1LL * a * b % mod;
}

int a[1234567];
int s[1234567];

int main() {
  // freopen("in", "r", stdin); freopen("out", "w", stdout);
  int n, k;
  scanf("%d %d", &n, &k);
  s[0] = 0;
  for (int i = 1; i <= n; ++i) {
    scanf("%d", a + i);
    s[i] = s[i - 1];
    add(s[i], a[i]);
  }
  int sum = 0, val = 0;
  for (int i = 1; i <= k; ++i) {
    add(sum, mul(mul(i, i), a[i]));
    add(val, mul(2 * i - 1, a[i]));
  }
  printf("%d", sum);
  //cout << endl;
  for (int i = 2; i <= n - k + 1; ++i) {
    sub(sum, val);
    //cout << "k + i - 1 = " << k + i - 1 << endl;
    add(sum, mul(mul(k, k), a[k + i - 1]));
    sub(val, a[i - 1]);
    sub(val, mul(2, s[k + i - 2]));
    add(val, mul(2, s[i - 1]));
    add(val, mul(2 * k - 1, a[k + i - 1]));
    //cout << "sum = " << sum << endl;
    //cout << "val = " << val << endl;
    //cout << endl;
    printf(" %d", sum);
  }
  printf("\n");
  return 0;
}
