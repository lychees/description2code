#include <bits/stdc++.h>
using namespace std;
int p(long long int x)
{
	for(int i=2;i<=sqrt(x);i++)
	{
		if(x%i==0)
		return 0;
	}
return 1;	
}
int main()
{
    int t;
    int a[100000];
    long long int n;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
    	int u=0;
    	cin>>n;
    	int k=0;
    	for(int j=2;j<=n;j++)
    	{
    		if(p(j)==1)
    		{
    			a[k]=j;
    			k++;
    		}
    	}
    	int min=n;
    	int x,y;
    	x=y=0;
    	for(int j=0;j<k;j++)
    	{
    		if((p(n-a[j])==1)&&((n-a[j])-a[j]>0))
    		{
    			u=1;
    			int f=abs(n-a[j]-a[j]);
    			if(f<min)
    			{
    				x=a[j];
    				y=n-a[j];
    			min=f;
    			}
    		}
    	}
    	if(u==0)
    	cout<<"-1 -1";
    	else
    	cout<<x<<" "<<y;
    	cout<<endl;
    }
    return 0;
}
