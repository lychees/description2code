#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    
    int n=s.size();
    int dp[n];
    
    if((s[n-1]-'0')%2==0){
    		dp[n-1]=1;
    	}
    	else
    		dp[n-1]=0;
    
    for(int i=n-2;i>=0;i--){
    	if((s[i]-'0')%2==0){
    		dp[i]=1+dp[i+1];
    	}
    	else{
    		dp[i]=dp[i+1];
    	}
    	
    }
    for(int i=0;i<n;i++)
    cout<<dp[i]<<" ";
    
    return 0;
}
