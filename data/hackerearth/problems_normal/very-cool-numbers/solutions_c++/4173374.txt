

    #include <bits/stdc++.h>
    using namespace std;
    int counter[100005];
    int main()
    {
    for (int g=1; g<=100000; g++)
    {
    for (int y=0; y<20; y++)
    {
    if (((1<<y)&g) && (!((1<<(y+1))&g)) && ((1<<(y+2))&g))counter[g]++;
    }
    }
    int T; scanf("%d", &T);
    for (int g=0; g<T; g++)
    {
    int R,K;scanf("%d %d", &R, &K);
    int ans=0;
    for (int y=1; y<=R; y++)
    {
    ans+=(counter[y]>=K);
    }
    printf("%d\n", ans);
    }
    return 0;
    }
