#include<iostream>
#include<vector>
using namespace std;
static int prime[1000001];
int upto[1000001];
int check(int n){
	while(n>0){
		if(n%10==7){
			return 1;
		}
		n/=10;
	}
	return 0;
}
int sieve()
{
	upto[0]=upto[1]=upto[2]=0;
	for(int i=2;i<=1000;i++)
	{
		if(prime[i]==0)
		{
			for(int j=i*i;j<=1000000;j+=i)
			{
				prime[j]=1;
			}
		}
	}
	for(int i=3;i<=1000000;i++)
	{
		if(prime[i]==1)
		{
			if(check(i)){
				upto[i]=1;
			}
		}
		upto[i]+=upto[i-1];
	}
	return 0;
}
int main()
{
	sieve();
	int t;
	cin>>t;
	int m,n;
	while(t--){
		cin>>m>>n;
		int x=(upto[n]-upto[m-1]);
		if(!x){
			cout<<"-1"<<endl;
			continue;
		}
		cout<<x<<endl;
	}
	return 0;
}
