#include<iostream>
#include <bits/stdc++.h>

using namespace std;

int n;
int answer = 1e9;
int a [12 + 10];
int maxi(int a,int b)
{
    if(a>b)
        return a;
    else
        return b;
}
int mini(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}

void fun(int L1, int L2, int L3, int position){

    if(position == n){

        int maxim = maxi(L1, maxi(L2, L3) );
        answer = mini(maxim, answer);
        return;
    }

    fun(L1 + a[position], L2, L3, position + 1);
    fun(L1, L2 + a[position], L3, position + 1);
    fun(L1, L2, L3 + a[position], position + 1);

}

int main()
{

    cin >>n;

    for(int i = 0; i < n; i++)
        cin >> a[i];

    fun(0, 0, 0, 0);

    cout << answer << endl;

    return 0;
}
