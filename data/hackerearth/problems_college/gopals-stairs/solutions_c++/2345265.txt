#include <bits/stdc++.h>
using namespace std;
#define lli long long int   
#define MOD 1000000007LL
long long a[100005];
int main()
{
	a[1]=1;
	a[2]=2;
	a[3]=4;
	for(int i=4;i<=100000;i++)
		a[i]=( (a[i-1]%MOD) + ((a[i-2]%MOD + a[i-3]%MOD)%MOD) )%MOD;
		
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		printf("%lld\n",a[n]);
	}
	return 0;
}
