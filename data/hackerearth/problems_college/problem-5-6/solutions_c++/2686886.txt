#include <iostream>
using namespace std;

int main()
{
    int T,N,X;
    cin>>T;
    for(int i=0;i<T;i++){
    	cin>>N;
    	cin>>X;
    	int* P=new int[N];
    	for(int i=0;i<N;i++)
    		cin>>P[i];
    	int* Table=new int[N];
    	Table[0]=P[0]-X;
    	//cout<<Table[0]<<endl;
    	for(int k=1;k<N;k++){
    		Table[k]=P[k]-X;
    		for(int j=0;j<k;j++){
    			//cout<<j<<","<<k<<","<<Table[j]+Table[k-j-1]<<endl;
    			Table[k]=max(Table[k],Table[j]+Table[k-j-1]);
    		}
    	}
    	cout<<Table[N-1]<<endl;
    }
    return 0;
}
