#include <iostream>
#include<cmath>
#include<stdlib.h>
using namespace std;
long long int power(long long int x, long long int y,long long int k)
{
    if( y == 0)
        return 1;
    else if (y%2 == 0)
        return (power(x, y/2,k)%k*power(x, y/2,k)%k)%k;
    else
        return ((x%k*power(x, y/2,k)%k)%k*power(x, y/2,k)%k)%k;
 
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long long int x,y,k;
    	cin>>x>>y>>k;
    	long long int m=1;
         for(int i=0;i<k;i++){
         	m=m*10;
         }long long int f=power(x,y,m);
          
         int s=0;
         while(f>0){
         	s=s+f%10;
         	f=f/10;
         }
    	cout<<s<<endl;
    }
    return 0;
}
