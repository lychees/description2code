#include <iostream>
using namespace std;

const long long mod = 1000000007LL;

long long fast_pow(long long a,long long b){
	long long ans = 1LL;
	while(b > 0){
		if (b & 1)
			ans = (ans * a)%mod;
		a = (a*a)%mod;
		b >>= 1;
	}
	return ans;
}

int main()
{
	int t;
	cin >> t;
	while(t--){
		long long num,ans;
		cin >> num;
		ans = fast_pow(2LL,num);
		ans = (ans - 1LL);
		if (ans < 0) ans += mod;
		cout << ans << "\n";
	}
}
