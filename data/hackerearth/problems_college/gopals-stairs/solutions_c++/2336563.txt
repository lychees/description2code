#include <iostream>
using namespace std;

int main()
{
    int t,n,i;
    long long one,two,three,temp;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	if(n<=2)
    	{
    		cout<<n<<endl;
    		continue;
    	}
    	one=1;
    	two=2;
    	three=4;
    	for(i=4;i<=n;i++)
    	{
    		temp=one+two+three;
    		temp%=1000000007;
    		one=two;
    		two=three;
    		three=temp;
    		
    	}
    	cout<<three<<endl;
    	
    }
    return 0;
}
