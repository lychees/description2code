#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
#define mod	1000000007
double find_ways(int m, int n, int x)
{
	double table[n+1][x+1];
	memset(table,0,sizeof(table));
	for(int i=1;i<=m && i<=x; ++i)
		table[1][i]=1;
	for(int i=2;i<=n;++i)
	{
		for(int j=1;j<=min(x,i*m);++j)
		{
			for(int k=1; k<=m && k<j; ++k)
			{
				table[i][j]+=table[i-1][j-k];
			}
		}
	}
	return table[n][x];
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,k;
		cin>>n>>m>>k;
		double ways=find_ways(m,n,k);
		//cout<<ways<<endl;
		double total_ways=1;
		for(int i=1;i<=n;++i)
			total_ways*=m;
		//cout<<total_ways<<endl;
		double ans=(ways*1.0)/total_ways;
		int x=0;
		if(ans!=0)
		{
			while(ans<1)
			{
				ans*=10;
				x++;
			}
		}
		printf("%.3lf %d\n",ans,x);
	}
	return 0;
}
