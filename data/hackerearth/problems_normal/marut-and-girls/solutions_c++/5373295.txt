#include <iostream>
#include<map>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    map<int,int> mymap;
    int a;
    for(int i=0;i<n;i++)
    {
    	cin>>a;
    	mymap[a]++;
    }
    int t;
    cin>>t;
    int ans=0;
    while(t--)
    {
    	int count=0;
    	int a;
    	char b;
    	while(scanf("%d%c",&a,&b)>0)
    	{   //cout<<val;
    		//cin>>val;
    		
    		if(mymap.find(a)!=mymap.end())
    			count++;
    		if(b=='\n')
    			break;
    	}
    //	cout<<endl;
    	if(count>=n)
    		ans++;
	 	
    	
    }
    cout<<ans;	
    return 0;
}
