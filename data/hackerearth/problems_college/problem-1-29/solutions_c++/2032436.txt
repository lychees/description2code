#include<bits/stdc++.h>
using namespace std;
//#defien N 100
int mat[100][100];
//bool f=false;
bool  solve(int x,int y,int N)
{
	if(x > N || y > N || x < 0 || y < 0 || mat[x][y]==0 )
	{
		return false;
	}
	if(x==N-1 && y==N-1)
	{
		//f=true;
		return true;
	}
	//mat[x][y]='#'
		return (solve(x,y+1,N) || solve(x+1,y,N));
}
int main()
{
	
	int t;
	cin >> t;
	while(t--)
	{
	int n;
	cin >> n;
	for(int i=0; i < n; i++)
	{
		for(int j=0; j < n; j++)
		{
			cin >> mat[i][j];
		}
	}
	bool t=solve(0,0,n);
	if(t)
	printf("POSSIBLE\n");
	else
	printf("NOT POSSIBLE\n");
	}	
}
