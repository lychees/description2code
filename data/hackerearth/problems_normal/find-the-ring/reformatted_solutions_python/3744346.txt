def find_ring(position, swaps):
    if swaps == 0:
        return position
    if position == 0 or position == 2:
        if swaps % 2 == 0:
            return 0
        else:
            return 1
    else:
        if swaps % 2 == 0:
            return 1
        else:
            return 0


t = int(raw_input())
for i in range(0, t):
    data = raw_input().split()
    position = int(data[0])
    swaps = int(data[1])
    print find_ring(position, swaps)
