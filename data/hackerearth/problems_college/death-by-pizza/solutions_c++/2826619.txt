#include<iostream>
using namespace std;

int main()
{
    int pizza[5]={16,9,26,26,1};
    long n,q,l,x,r,i;
    cin>>n;
    int arr[n];
    cin>>q;
    for(i=0;i<n;i++)
        cin>>arr[i];
    for(i=1;i<=q;i++)
    {
        cin>>x;
        cin>>l;
        cin>>r;
        if(x==0)
        {
            for(long j=l-1,k=0;j<r;j++,k++)
            {
                arr[j]=arr[j]+pizza[k%5];
            }
        }
        else
        {
            long sum=0;
            for(long j=l-1;j<r;j++)
                sum=sum+arr[j];
            cout<<sum<<"/"<<(r-l+1)<<endl;
        }
    }

    return 0;
}
