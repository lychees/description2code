    #include<bits/stdc++.h>
    using namespace std;
    long long int arr[100000],num;
    int arr_lr(long long int sum)
    {
        long long int i,j,sum_left,sum_right;
        if(num==3)
        {
            if(arr[0]==arr[2])
                return 1;
            else
                return 0;
        }
        sum_left=arr[0];
        sum_right=sum-arr[0]-arr[1];
        for(i=1;i<=num-2;i++)
        {
            if(sum_left==sum_right)
                return 1;
            sum_left=sum_left+arr[i];
            sum_right=sum_right-arr[i+1];
        }
        return 0;
    }
    int main()
    {
        long long int test,i,j,ans,sum;
        scanf("%lld",&test);
        while(test--)
        {
            sum=0;
            scanf("%lld",&num);
            for(i=0;i<num;i++)
            {
                scanf("%lld",&arr[i]);
                sum=sum+arr[i];
            }
            ans=arr_lr(sum);
            if(ans==1)
                printf("YES\n");
            else
                printf("NO\n");
        }
    }
