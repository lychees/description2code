// Monk and the collision

#include<iostream>
#include<cstdio>

using namespace std;

int a[10];

void initialize(){
     for( int i = 0; i < 10; i++ )
        a[i] = 0;
     }

int main()
{
    int t, n, i, ans;
    long int num;
    
    cin>>t;
    
    while( t-- ){
           
           initialize();           
           ans = 0;
           cin>>n;
           for( i = 0; i < n; i++ ) 
           {
                cin>>num;
                num %= 10;
                
                if( a[num] == 0 )
                   a[num]++;
                
                else
                   ans++;   
                }   
           
           cout<<ans<<endl;
           
           }


    return 0;
}
