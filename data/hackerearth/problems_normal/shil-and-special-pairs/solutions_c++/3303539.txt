#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define ull unsigned long long
#define ii pair<int,int>
#define iii pair<ii, int>
#define pll pair<ll,ll>
#define ppll pair<pll, ll>

#define fi first
#define se second
#define mp make_pair
#define pb push_back
#define ep emplace_back
#define sz(a) (int) a.size()
#define cl(a) a.clear()

#define vi vector<int>
#define vii vector<ii>

#define FOR(x,a,b) for (int x=a;x<=b;x++)
#define FOD(x,a,b) for (int x=a;x>=b;x--)
#define REP(x,a,b) for (int x=a;x<b;x++)
#define RED(x,a,b) for (int x=a;x>b;x--)

const int MAX = 1e5 + 10;
const int MAXN = 1e4 + 10;
const int MOD = 1e9 + 7;
const int inf = 1e9;
const double pi = acos(-1.0);
const double eps = 1e-6;

ll f[MAX], res[MAX], pos[MAX];
vector<ppll> q;
int a[MAX];
int n, m, d;

void update(int id){
	while(id<=n){
		f[id]++;
		id += id & -id;
	}
}

ll query(int id){
	ll ans=0;
	while(id>0){
		ans+=f[id];
		id -= id & -id;
	}
	return ans;
}

bool cmp(ppll a,ppll b){
	return a.fi.se < b.fi.se;
}

int main(){
	// freopen("test.in", "rt", stdin);
	// freopen("test.ou", "wt", stdout);
	ios::sync_with_stdio(false);

	cin >> n >> m >> d;
	
	FOR(i,1,n){
		cin >> a[i];
		pos[a[i]] = i;
	}

	int l, r;
	REP(i,0,m){
		cin >> l >> r;
		q.pb(mp(mp(l,r), i));
	}
	sort(q.begin(), q.end(), cmp);
	l = 0;
	FOR(i,1,n){
		FOR(j,max(1,a[i]-d),min(n,a[i]+d))
			if (pos[j] <= i)
				update(pos[j]);
		while (l < q.size() && q[l].fi.se == i){
			res[q[l].se] = query(q[l].fi.se) - query(q[l].fi.fi - 1);
			l++;
		}
	}
	REP(i,0,m) cout << res[i] << endl;

	return 0;
}
