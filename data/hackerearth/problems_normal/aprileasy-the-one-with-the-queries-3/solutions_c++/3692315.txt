#include <bits/stdc++.h>
using namespace std;
#define ll long long
vector<ll> a[100005];
ll ar[1000005];
int main()
{
	string s,sub;
	ll i,q,curr,j;
	bool flag , f;
    cin>>s;
    for(i=0;i<s.length();i++)
    {
    	a[s[i]-'a'].push_back(i);
    }
    cin>>q;
    while(q--)
    {
    	for(i=0;i<27;i++)
    	{
    		ar[i]=0;
    	}
    	cin>>sub;
    	curr = -1;
    	f = true;
    	for(i=0;i<sub.length();i++)
    	{
    		if(!a[sub[i]-'a'].empty())
    		{
    			flag = false;
    			for(j=ar[sub[i]-'a'];j < a[sub[i]-'a'].size() ; j++)
    			{
    				if(a[sub[i]-'a'][j] > curr)
    				{
    					curr = a[sub[i]-'a'][j];
    					flag = true;
    					ar[sub[i]-'a'] = j+1;
    					break;
    				}
    			}
    			if(!flag)
    			{
    				f = false;
    				break;
    			}
    		}
    		else
    		{
    			f = false;
    			break;
    		}
    	}
    	if(!f)
    	{
    		cout<<"No\n";
    	}
    	else
    	{
    		cout<<"Yes\n";
    	}
    }
    return 0;
}
