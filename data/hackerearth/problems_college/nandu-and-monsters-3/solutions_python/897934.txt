	if inp[0] >= inp[1]:

		i = inp[1]

	temp = i
	root = 1
	path = []
	while (temp >= root and monster[temp-1] != '1'):
		path.append(temp)
		temp = temp/2
	
	temp = j
	while(temp >= root and monster[temp-1] != '1'):
		if temp in path:
			print 'Yes'

		temp = temp/2
