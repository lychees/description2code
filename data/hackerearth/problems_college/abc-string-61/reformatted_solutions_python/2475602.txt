mod = 1000000007


def f(b, e):
    if e == 0:
        return 1
    if e % 2 == 0:
        temp = f(b, e/2)
        return (((temp % mod)*(temp % mod)) % mod)
    return (((b % mod)*((f(b, e-1) % mod))) % mod)


t = input()
for k in range(0, t):
    n = input()
    if n < 3:
        print 0
    else:
        x = ((f(3, n) % mod)+3-((f(2, n) % mod)*3) % mod) % mod
        print x
