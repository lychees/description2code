#include <bits/stdc++.h>
using namespace std;
typedef unsigned int uint;

int main()
{
    uint n;
    cin>>n;
    ++n;
    while(--n)
    {
    	string s;
    	cin>>s;
    	
    	uint len=s.size(),ans=0;
    	for(uint i=0;i!=len;)
    	{
    		char a=s[i];     
    		uint j=0;
    		while(a==s[++i])
    		 ++j;
    		ans+=j;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
