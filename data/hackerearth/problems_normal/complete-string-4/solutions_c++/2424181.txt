#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    
    for(int i=1;i<=T;i++) {
    	bool value[26] ={false};
    	string s;
    	cin>>s;
    	const char *j =s.c_str();
    	if(s.size() <26){
    		cout<<"NO\n";
    		continue;
    	}
    	while(*j!='\0') {
    		value[(*j-'a')] = true;
    		j++;
    	}
    	bool flag = false;
    	for(int i =0; i<26;i++) {
    		if(!value[i]) {
    			cout<<"NO\n";
    			flag = true;
    			break;
    		}
    	}
    	if(!flag) {
    		cout<<"YES\n";
    	}
    }
    return 0;
}
