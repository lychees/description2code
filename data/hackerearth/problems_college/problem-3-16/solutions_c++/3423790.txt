#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		int num[k];
		memset(num,0,k*sizeof(int));
		for(int i=0;i<n;i++)
		{
			int temp;
			cin>>temp;
			num[temp-1]++;
		}
		int max1=-1;
		for(int i=0;i<k;i++)
		{
			max1=max(max1,num[i]);
		}
		cout<<n-max1<<endl;
	}
	return 0;
}
