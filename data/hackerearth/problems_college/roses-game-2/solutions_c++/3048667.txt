#include<bits/stdc++.h>
#include<cstring>
using namespace std;
int arr[100001];
int visit[100001];
int main()
{
    int t,n,i,temp=0;
    cin>>t;
    while(t--)
    {
        memset(visit,0,sizeof(visit));
        cin>>n;
        for(i=0;i<n;i++)
            cin>>arr[i];
        temp=arr[0];
        while(1)
        {
            if(visit[temp])
            {
                cout<<"Lose\n";
                break;
            }
            if(temp==-1)
            {
                cout<<"Win\n";
                break;
            }
            visit[temp]=1;
            temp=arr[temp];
        }
    }
    return 0;
}
