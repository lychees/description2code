def find(num, m):
	digits = []
	while(num != 0):
		digits.append(num % 10)
		num /= 10
	k = 0
	sum = 0
	for i in digits:
		if i < 3:
			sum += (9 ** k) * i
		else:
			sum += (9 ** k) * (i - 1)
		k += 1
	return sum
