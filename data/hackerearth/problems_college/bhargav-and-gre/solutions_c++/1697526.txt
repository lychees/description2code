#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
	char str[100];
	gets(str);
	int testCases;
	scanf("%d\n", &testCases);
	int ant = 0, syn = 0;
	for(int c=0;c<testCases;c++)
	{
		char temp[100];
		gets(temp);
		if(strlen(str) == strlen(temp))
		{
			if(temp[0] == str[0])
				syn++;
			else if((char)((int)temp[0] - 3) == str[0])
				ant++;
		}
	}
	printf("%d %d", syn, ant);
    return 0;
}
