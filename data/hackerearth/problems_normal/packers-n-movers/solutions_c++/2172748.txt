#include <iostream>
using namespace std;
int dp[1001][1001];
int max(int a,int b)
{
	return (a>b)?a:b;
}
int main() 
{
    int m,p;
    cin>>m>>p;
    int pck[p+1];
   int maxi=0;
    for(int i=0;i<p;i++)
    {
        cin>>pck[i];
    }
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<p;j++)
        {
            if(i==0)
            {   
                if(j==0) 
                 dp[i][j]=pck[j];
                else
                 dp[i][j]=dp[i][j-1]+pck[j];
            }
            else
            {   
                if(j==0)
                {
                    dp[i][j] = pck[j];
                }
                else
                {
                int temp = max(dp[i-1][j-1],pck[j]);
                long long sum=pck[j];
                for(int k=j-1;k>0;k--)
                {   
                    sum+=pck[k];
                    if(max((sum),dp[i-1][k-1])<temp)
                     temp = max((sum),dp[i-1][k-1]);
                }
                if(temp>(sum+pck[0]))
                 temp=(sum+pck[0]);
                dp[i][j] = temp;
                }
            }
        }
    }
    /*for(int i=0;i<m;i++)
    {
        for(int j=0;j<p;j++)
        {
            cout<<dp[i][j]<<" ";
        }
        cout<<endl;
    }*/
    cout<<dp[m-1][p-1]<<endl;
	return 0;
}
