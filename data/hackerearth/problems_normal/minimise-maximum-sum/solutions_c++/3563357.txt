#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long int n;
    	cin>>n;
    	long int i,a[100000], b[100000];
    	
    	for(i=0;i<(2*n);i++)
    	   cin>>a[i];
    
         sort(a,a+(2*n));
   
           for(i=0;i<n;i++)
            {
   	
               	b[i] = a[i]+a[(2*n)-i-1];
             }
             sort(b,b+n);
   
           cout<<b[n-1]<<endl;
    }
    return 0;
}
