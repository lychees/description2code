#include <bits/stdc++.h>

using namespace std;

int a[109];

int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
	int N;
	cin>>N;
	for(int i=0;i<N+1;i+=1)
	a[i]=0;
	a[0]=1;
	a[6]=1;
	a[8]=1;

	for(int i=9;i<=N;i+=1)
	{
	if(a[i-6]!=0) 
	a[i]=a[i-6]+1;
	if(a[i-8]!=0) 
	a[i]=a[i-8]+1;
	}
	if(a[N]!=0)
	cout<<a[N]<<endl;
	else cout<<"-1"<<endl;
	}
	return 0;
}
