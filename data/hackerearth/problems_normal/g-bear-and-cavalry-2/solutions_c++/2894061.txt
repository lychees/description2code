#include <algorithm>
#include <iostream>
using namespace std;

bool fun(int i, int j){return i>j;}

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int w[n],h[n];
    	for(int i=0; i<n; i++){
    		cin>>w[i];
    	}
    	for(int i=0; i<n; i++){
    		cin>>h[i];
    	}
    	sort(h,h+n,fun);
    	int max = w[0]*h[0];
    	sort(w+1,w+n,fun);
    	sort(h+1,h+n);
    	int i;
    	for(i=1; i<n; i++){
    		//cout<<w[i]<<","<<h[i]<<"; ";
    		if(w[i]*h[i] >= max){
    			cout<<"NO"<<endl;
    			break;
    		}
    	}
    	if(i==n){
    		cout<<"YES"<<endl;
    	}
    }
    
    return 0;
}
