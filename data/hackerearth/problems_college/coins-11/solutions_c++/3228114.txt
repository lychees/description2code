#include <iostream>
#include <cstdio>
#include <cmath>

#define EPS 1e-7
#define LL long long int
using namespace std;



bool isPrime (LL x)
{
  if(x<=1) return false;
  if(x<=3) return true;
  if (!(x%2) || !(x%3)) return false;
  LL s=(LL)(sqrt((double)(x))+EPS);
  for(LL i=5;i<=s;i+=6)
  {
    if (!(x%i) || !(x%(i+2))) return false;
  }
  return true;
}

int main()
{
	LL T,num,i=0,ans=0;
	
	scanf("%lld",&T);
	for(i=1; i<=T; i++){
		scanf("%lld",&num);
		{
			ans=0;
			
			if(isPrime(num)){
				if(num==3)
				printf("3\n");
				
				else{
				ans=2;
				printf("%lld\n",ans);
				}
			}
			else{
				ans=1;
				printf("%lld\n",ans);
			}
		}
	}
    return 0;
}
