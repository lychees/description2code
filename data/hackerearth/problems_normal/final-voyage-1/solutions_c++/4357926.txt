#include<iostream>
#include<vector>
#include<stdio.h>
#include<stdlib.h>

using namespace std;

int max(int a,int b){
	if(a>b){
		return a;
	}
	return b;
}

int main()
{
    int t,n,ml,w[1000],v[1000],M[1000][1001],i,j,maxim;
    cin>>t;
    while(t--){
    	cin>>n;
    	cin>>ml;
    	for(i=0;i<n;i++){
    		cin>>w[i];
    	}
    	for(i=0;i<n;i++){
    		cin>>v[i];
    	}
    	
    	for(i=0;i<n;i++){
    		for(j=0;j<=ml;j++){
    			M[i][j]=0;
    		}
    	}
    	maxim=0;
    	if(ml>=w[0]){
    		M[0][ml-w[0]]=v[0];
    		maxim=v[0];
    	}
    	
    	for(i=1;i<n;i++){
    		for(j=ml;j>=0;j--){
    			if((M[i-1][j]!=0||j==ml)&&j>=w[i]){
    				M[i][j-w[i]]=max(M[i][j-w[i]],M[i-1][j]+v[i]);
    			}
    			M[i][j]=max(M[i-1][j],M[i][j]);
    			maxim=max(maxim,M[i][j]);
    		}
    	}
    	cout<<maxim<<"\n";
    }
    return 0;
}
