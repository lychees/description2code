#include <iostream>
#include <algorithm>
#include <stdio.h>

using namespace std;

int main()
{
	int n;
	int a[10001];
	int min;
	int i;
	int j;
	int f;
	cin >> n;
    //scanf("%d", &n);
    for(i = 0; i < n; i++)
    	//scanf("%d", a[i]);
    	cin >> a[i];
    sort(a,a+n);
    
    min = 10001;
    
    for(i = n-1; i >= 0; i--) {
    	f = 0;
    	for(j = 0; j < i; j++) {
    		if(a[i] <= 3*a[j]) {
    			f = 1;
    			break;
    		}
    	}
    	if(f == 1)
    	min = min>(n-i+j-1)?(n-i+j-1):min;
    }
    cout << min;
    //printf("%d", min);
    return 0;
}
