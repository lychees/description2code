#include <bits/stdc++.h>
 
using namespace std;
 
int arr[10];
 
int main(int argc, char const *argv[])
{
	int n;
	cin>>n;
	int max1,max2,max3;
	long long prod = 1;
	if(n<3)
	{
		while(n--)
			cout<<-1<<endl;
	}
	else
	{
	cin>>arr[1]>>arr[2];
	cout<<-1<<endl<<-1<<endl;	
	cin>>arr[3];
	if(arr[1]>=arr[2])
		{if(arr[1]>=arr[3])
			{max1 = arr[1];
				if(arr[2]>=arr[3])
					{max2 = arr[2]; max3 = arr[3];}
				else
					{max2 = arr[3]; max3 = arr[2];}
			}
		else
			{max1 = arr[3];
				if(arr[2]>=arr[1])
					{max2 = arr[2]; max3 = arr[1];}
				else
					{max2 = arr[1]; max3 = arr[2];}
			}
		}
	else if(arr[2]>=arr[3])
			{max1 = arr[2];
				if(arr[1]>=arr[3])
					{max2 = arr[1]; max3 = arr[3];}
				else
					{max2 = arr[3]; max3 = arr[1];}
			}
	else
		{max1 = arr[3];
			if(arr[2]>=arr[1])
					{max2 = arr[2]; max3 = arr[1];}
				else
					{max2 = arr[1]; max3 = arr[2];}
		}
 
 
	prod*=max1;
	prod*=max2;
	prod*=max3;
 
	cout<<prod<<endl;
 
	for (int i = 4; i <= n; ++i)
	{
		cin>>arr[4];
 
		if(max3>arr[4])
			cout<<prod<<endl;
		else
		{	
			prod = 1;
			if(arr[4]>=max1)
			{
				max3 = max2;
				max2 = max1;
				max1 = arr[4];
			}
			else if(arr[4]>=max2)
			{
				max3 = max2;
				max2 = arr[4];
			}
			else 
			{
				max3 = arr[4];
			}
 
			prod*=max1;
			prod*=max2;
			prod*=max3;
 
			cout<<prod<<endl;
 
		}
	}
	}
	return 0;
}
