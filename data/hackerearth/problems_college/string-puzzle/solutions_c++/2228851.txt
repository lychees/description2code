#include <iostream>
using namespace std;

#include <stdio.h>
void perform(char *str,int a,int b)
{
	int j;
	if(a == b)
	{ 
	  cout << str << endl;
	  return;
	}
	
	 for(j=a;j<=b;j++)
	 {
	 	str[a]=(str[a]+str[j])-(str[j]=str[a]);
	 	perform(str,a+1,b);
	 	str[a]=(str[a]+str[j])-(str[j]=str[a]);
	 }
}
int main()
{
    int t, a, b;
    char str[1000];
   cin >> str >> t;
    while(t--)
    {
    	cin >> a >> b;
    	perform(str,a,b);
    	cout << endl;
    }
    return 0;
}
