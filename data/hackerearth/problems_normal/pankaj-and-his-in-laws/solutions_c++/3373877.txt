#include <bits/stdc++.h>
using namespace std;

void rec(int a){
	if(a<1)return;
	int t=a%2;
	rec(a/2);
	cout<<t;
	return;
}

int main()
{
    int t;
    cin>>t;
    int a[t],L[t];
    for(int i=0;i<t;i++){
    	cin>>a[i];L[i]=1;
    }
    for(int i=1;i<t;i++){
    	for(int j=0;j<i;j++){
    		if(a[i]>a[j] && L[i]<L[j]+1)L[i]=L[j]+1;
    	}
    }
    sort(L,L+t);
    int aa=L[t-1];
    rec(aa);
    cout<<endl;
    return 0;
}
