#include<iostream>
using namespace std;

int main()
{
   long long int a,b,n,ans=0,cnt=0;
    cin>>a;
    cin>>b;
    cin>>n;
    while(ans<n)
    {
        ans=ans+a;
        cnt++;
        if(ans>=n)
        {
            break;
        }
        ans=ans-b;
    }
cout<<cnt<<endl;
}
