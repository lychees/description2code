#include <bits/stdc++.h>
#define PB push_back
#define MP make_pair
#define LL long long
#define MAXN 100001
#define M 1000000007
#define F first
#define S second
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

LL dp[112345][2], dp1[112345];

void pre(){
    int i;
    dp[0][0] = 1;
    dp[0][1] = 1;
    dp[1][0] = 1;
    dp[1][1] = 1;
    dp1[1] = 2;
    dp1[0] = 1;
    for(i=2;i<=100000;i++){
        dp[i][1] = dp[i - 1][1] + dp[i - 1][0];
        dp[i][0] = dp[i - 1][1];
        dp[i][1] %= M;
        dp[i][0] %= M;
        dp1[i] = dp[i][0] + dp[i][1];
    }
}

void solve(int n){
    LL ans = 2 * dp1[n - 3];
    for(int i=2;i<=n - 2;i++){
        ans += (dp1[i - 2] * dp1[n - (i + 2)]);
        ans %= M;
    }
    cout<<ans<<endl;
}

int main(){
    int t, i, j;
    pre();
    int n;
    cin>>t;
    while(t--){
        cin>>n;
        if(n == 1)
            cout<<0<<endl;
        else if(n == 2)
            cout<<1<<endl;
        else if(n == 3)
            cout<<2<<endl;
        else if(n == 4)
            cout<<5<<endl;
        else
        solve(n);
    }
}
