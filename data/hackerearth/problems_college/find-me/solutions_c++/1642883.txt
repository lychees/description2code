#include <iostream>
using namespace std;

int main()
{
	int t,p,n;
    cin>>t;
    int arr[1000000];
    int i;
    while(t)
    {
      cin>>n>>p;
    for(i=0;i<n;i++)
    cin>>arr[i];
    int flag=0;int count=0;
    for(i=0;i<n;i++)
    {
    	if(arr[i]>p)count++;
    	else if(arr[i]==p)flag=1;
    }
    if(flag)cout<<"YES "<<count<<"\n";
    else cout<<"NO "<<0<<"\n";
    t--;
    }
    return 0;
}
