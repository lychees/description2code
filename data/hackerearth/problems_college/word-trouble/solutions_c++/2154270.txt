#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int cs,n,m,s,i;
char x[101],y[101];
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%s%s",x,y);
    n=strlen(x);
    m=strlen(y);
    if(n<m)
    {
      swap(x,y);
      swap(n,m);
    }
    s=n-m;
    for(i=0;i<m;i++)
      s+=x[i]!=y[i];
    if(s)
      printf("Non Exact %d\n",s);
    else
      printf("Exact\n");
  }
  return 0;
}
