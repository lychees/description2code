for _ in range(int(raw_input())):
	n= map(int,raw_input().split())
	b = n[0];w=n[1]
	r = map(int,raw_input().split())
	br = r[0]
	wr=r[1]
	tr=r[2]
	if br == wr:
		print (b*br + w*wr)
	elif br > wr+tr :
		print (w*wr + b*(wr+tr))
	elif wr > br+tr :
		print (b*br + w*(br+tr))
	else:
		print (w*wr+b*br)
