#include<iostream>
#include <stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
#define SCAN(a) scanf("%ld",&a)
#define PRINT(a) printf("%ld",a)
#define iloop(a,b) for(i=a;i<b;i++)
typedef long int li;
int main()
{
	  li i,n,k,a[100005],cnt=0,sum=0;
	  SCAN(n);SCAN(k);
	  iloop(0,n)
	  	SCAN(a[i]);
	  sort(a,a+n);
	  iloop(0,n)
	  {
	  	 
	  	 if(a[i]<k)
	  	 {
	  	 	sum+=a[i];
	  	 	cnt++;
	  	 }
	  	 if(sum>=k)
	  	 break;
	  }
	  PRINT(cnt-1);
	  return 0;
}
