#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    { map<int,int>arr;
      map<int,int>dep;
    	int n,i;
    	cin>>n;
    	int a[n],d[n],maxi=0,mini=INT_MAX;
    	long long sum=0,count=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		arr[a[i]]=arr[a[i]]+1;
    		
    		if(a[i]<mini)
    		mini=a[i];
    	}
    
    	for(i=0;i<n;i++)
    	{  int num;
    		cin>>num;
    		d[i]=a[i]+num;
    		dep[d[i]]=dep[d[i]]+1;
    		if(d[i]>maxi)
    		maxi=d[i];
    	}
    	
    for(i=mini;i<=maxi;i++)
    {       

    	sum=sum+arr[i];
    	sum=sum-dep[i];
    	if(sum>count)
    	count=sum;
    	
    }
   cout<<count<<"\n";
    }
    return 0;
}
