#include <iostream>
#include<stdlib.h>
using namespace std;

void city()
{
    int n;
    cin>>n;
   // cout<<"\n";
    string arr[n];
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    //	cout<<"\n";
    }
    int y=0;
    for(int i=0;i<n;i++)
    {
    	char ch=arr[i][0];
    //	cout<<ch;
    	for(int j=0;j<n;j++)
    	{
    		if(i==j)
    		  continue;
    		else
    		{
    		  if(ch==arr[j][0])
    		  {
                  y++;
    		  }
    		}

    	}

    }
    if(y==0)
    {
      cout<<"YES";
      
    }
    else
    {
    cout<<"NO";
    }
}

int main()
{
    int testcase;
    cin>>testcase;
    cout<<"\n";
    for(int i=0;i<testcase;i++)
    {
    city();
    	cout<<"\n";
    }
}
