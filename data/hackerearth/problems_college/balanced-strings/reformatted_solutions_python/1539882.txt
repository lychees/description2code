'''

'''
ts = int(input())
for i in range(ts):
    s = raw_input()
    l = s.count('a')
    o = 0
    for k in range(97, 123):
        z = s.count(chr(k))
        if z == l:
            o = 0
            continue
        else:
            o = 1
            break

    if o == 0:
        print 'Yes'
    else:
        print 'No'
