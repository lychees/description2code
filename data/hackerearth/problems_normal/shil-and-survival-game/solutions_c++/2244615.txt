char _;

#define int long long
#define rep(i, s, e) for(int i = s; i < e; i ++)
typedef double ld;

const int maxN = 100*1000 + 5;
const int mod = 1000*1000*1000+7;
const int base = 701;
const int maxL = 14;
typedef long long ll;
#define X first
#define Y second
int a[maxN];

int L[maxN], R[maxN];
main()
{
    ios::sync_with_stdio(0); cin.tie(0);
    int n; cin >> n;
    rep(i,0,n) cin >> a[i];

    L[0] = a[0];
    rep(i,1,n) 
    {
	  L[i] = mod;
	  if(i > 1)
		L[i] = max(max(a[i], a[i-1]), L[i-2]);
	  L[i] = min(L[i], max(a[i], L[i-1]));
    }
    
    R[n-1] = a[n-1];
    for(int i = n-2; i >= 0; i --)
    {
	  R[i] = mod;
	  if(i < n-2) 
		R[i] = max(max(a[i], a[i+1]), R[i+2]);
	  R[i] = min(R[i], max(a[i], R[i+1]));
    }

    vector<int> ans;
    rep(i,0,n) 
    {
//	  if(i) cout << L[i-1] << ' ' << R[i+1] << endl;
	  if(i == 0) 
		ans.push_back(i);
	  else
		if(min(R[i+1], L[i-1]) < a[i]) ans.push_back(i);
    }

    rep(i,0,ans.size()) cout << ans[i]+1 << ' '; cout << endl;

    return 0;
}
