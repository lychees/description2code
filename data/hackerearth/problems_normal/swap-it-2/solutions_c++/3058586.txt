#include <bits/stdc++.h>
using namespace std;
int a[1005];
int main()
{
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(int i=0;i<n;i++)
		{
			int min1=a[i];
			int index=i;
			for(int j=i+1;j<n;j++)
			{
				if(a[j]<a[i]&&a[j]<min1&&(j-i)<=k)
				{
					min1=a[j];
					index=j;
				}
			}
			if(min1!=a[i])
			{
				k=k-(index-i);
				int temp=a[index];
				int temp1=a[i];
				for(int j=index-1;j>=i;j--)
				{
					a[j+1]=a[j];
				}
				a[i]=temp;
			}
			
		}
		for(int i=0;i<n;i++)
		cout<<a[i]<<" ";
		cout<<"\n";
	}
}
