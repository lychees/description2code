void word(int x)
{
	if(x==0)
	return;
	if(x<=20)
	print(x);
	else
	{
		if(x>50)
		{
			cout<<"fifty ";
			x-=50;
			word(x);
		}
		else if(x>40)
		{
			cout<<"forty ";
			x-=40;
			word(x);
		}
		else if(x>30)
		{
			cout<<"thirty ";
			x-=30;
			word(x);
		}
		else
		{
			cout<<"twenty ";
			x-=20;

	}
}
