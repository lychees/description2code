#include <iostream>
using namespace std;
void swap(int *a,int *b){int t=*a;*a=*b;*b=t;}
void fun(int *arr,int n,int k)
{
	int pos;
	for(int i=0;i<n;i++)
	{
		 pos=i;
		 for(int j=i+1;j<n;j++)
		 {
		      if(j-i>k)break;
		 	if(arr[pos]>=arr[j])
		 	pos=j;
		 }
		if(pos!=i)
		{
			
			for(int x=pos;x>i;x--)
			swap(&arr[x-1],&arr[x]);
			k=k-(pos-i);
		}
		
	}
	
}
int main()
{
   int t,i;i=0;
   cin>>t;
   int len,k;
   while(i<t)
   {
   	 cin>>len>>k;
   	 int *arr= new int[len];
   	
   	 for(int j=0;j<len;j++)
   	{ cin>>arr[j];}
   	 fun(arr,len,k);
   	 for(int j=0;j<len;j++)
   	 cout<<arr[j]<<" ";cout<<endl;
   	 
   	i++;
   	
   	
   }
    return 0;
}
