#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin >> t;
	vector<string> arr;
	while (t--)
	{
		int n,k;
		cin >> n >> k;		
		arr.clear();
		std::vector<std::vector<char>> chararray;
		string outputstr;
		for (int i = 0; i < k; i++)
		{
			vector<char> indi;
			chararray.push_back(indi);
		}
		while (n--)
		{
			string str;
			cin >> str;
			arr.push_back(str);
			for(int i = 0; i < k; i++)
			{
				chararray[i].push_back(str[i]);
			}
		}
		for (int j = 0; j < chararray.size(); j++)
		{
			sort(chararray[j].begin(), chararray[j].end());
			int mid = (chararray[j].size() % 2) == 0 ? chararray[j].size() / 2 : (chararray[j].size() + 1) / 2;
			outputstr = outputstr + chararray[j][mid - 1];
		}
		cout << outputstr << "\n";
	}
	return 0;
}
