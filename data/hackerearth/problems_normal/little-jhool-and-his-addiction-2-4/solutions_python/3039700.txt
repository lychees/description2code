t = int(raw_input())
while t:
	t = t - 1
	n,k = map(int, raw_input().split())
	li = map(int, raw_input().split())
	li.sort()
	minp = maxp = li[-1] + li[0]
	i = 0
	while i < n/2:
		s = li[i] + li[n-i-1]
		if s > maxp:
			maxp = s
		elif s < minp:
			minp = s
		i = i + 1
	ans = maxp-minp	
	print ans
	if ans > k:
		print "No more girlfriends!"
	elif ans == k:
		print "Lucky chap!"
	else:
		print "Chick magnet Jhool!"
