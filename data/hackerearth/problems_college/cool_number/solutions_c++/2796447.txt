#include<iostream>
using namespace std;
int tmp[27];
int coolness(int n)
{
	for (int i = 1; i <= 26; i++)
	{
		tmp[i] = n % 2;
		n /= 2;
	}
	int ans = 0;
	for (int i = 1; i + 2 <= 26; i++)
	{
		if (tmp[i] == 1 && tmp[i + 1] == 0 && tmp[i + 2] == 1)
		{
			ans++;
		}
	}
	return ans;
}
int ans[100007];
int main()
{
	for (int i = 1; i <= 100000; i++)
	{
		ans[i] = coolness(i);
	}
	int t;
	cin >> t;
	for (int tt = 1; tt <= t; tt++)
	{
		int r, k;
		cin >> r >> k;
		int a = 0;
		for (int i = 1; i <= r; i++)
		{
			if (ans[i] >= k) a++;
		}
		cout << a << endl;
	}
}
