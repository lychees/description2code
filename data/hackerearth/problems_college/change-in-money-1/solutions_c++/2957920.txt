#include<iostream>
typedef long long int ll;
using namespace std;
ll count( ll  S[], ll  m,ll n )
{
    ll i, j, x, y;
    ll table[n+1][m];
    
   
    for (i=0; i<m; i++)
        table[0][i] = 1;
 
    
    for (i = 1; i < n+1; i++)
    {
        for (j = 0; j < m; j++)
        {
            
            x = (i-S[j] >= 0)? table[i - S[j]][j]: 0;
 
            y = (j >= 1)? table[i][j-1]: 0;
 
            
            table[i][j] = x + y;
        }
    }
    return table[n][m-1];
}
int main()
{
	
	
	ll n,m;
	cin>>n>>m;
	ll ar[m];
	for(int i=0;i<m;i++)
	cin>>ar[i];
	ll z=sizeof(ar)/sizeof(ar[0]);
	cout<<count(ar,z,n)<<"\n";
}
