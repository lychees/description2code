#include <iostream>
typedef unsigned long int lint;
using namespace std;

void answer(lint p, lint  m){
 cout << 2*(p+m) << endl;
}
int main(){
    lint t;
    cin >> t;
    lint p,m;
    for(int i=0;i<t;i++){
        cin >> p;
        cin >> m;
        answer(p,m);
    }
    return 0;
}
