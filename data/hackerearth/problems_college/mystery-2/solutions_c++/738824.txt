#include <iostream>
using namespace std;



int main()
{
     int t,n,m;
     int a[100];
    cin>>t;
    while(t--)
    {   int i=0;
    	cin>>n>>m;
    	while(n>=1)
    	{
    		a[i++]=n%m;
    		n=n/m;
    	}
    	i--;
      while(i>=0)
        cout<<a[i--];
        cout<<endl;
  }
    return 0;
}
