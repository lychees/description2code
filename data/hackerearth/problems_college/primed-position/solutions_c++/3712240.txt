#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MOD 1000000007
int main()
{
    vector<ll> primes;
    vector<int> A(3000001,1);
    primes.resize(10000);
    A[1] = 0;
    int primeC = 0;
    int n = 0;
    for(int i = 2; i <=3000000; i++){
    	if(A[i]){
    		primeC++;
    		for(int j = 2*i; j <= 3000000; j += i){
    			A[j] = 0;
    		}
    		if(A[primeC] == 1){
    			if(n < 10000) primes[n++] = (ll)i;
    		}
    	}
    	if(n == 10000) break;
    }
    //cout << "n is : " << n << endl;
    //cout << primes[9999] << "\n";
    int t; int a,m;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&a,&m);
    	ll ans = ((primes[a-1] % MOD) *(primes[m-1] % MOD) ) % MOD;
    	printf("%lld\n",ans);
    }
    return 0;
}
