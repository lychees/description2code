LL BigMod(LL a,LL b,LL mod)
{
      if(!b) return 1;
      LL x = BigMod(a,b/2,mod);
      x = (x*x)%mod;
      if(b%2)
            return (x*a)%mod;
      else
            return x;
}

LL modinv(LL a,LL mod)
{
      return BigMod(a,mod-2,mod);
}
LL mod = 1000000007;
LL fact[1000001] = {0};

LL ncr(LL a,LL b)
{
      LL ret = (fact[a]*modinv(fact[b],mod))%mod;
      ret = (ret*modinv(fact[a-b],mod))%mod;
      return ret;
}
