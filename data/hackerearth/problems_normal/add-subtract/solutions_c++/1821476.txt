#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;
int min(int a, int b)
{
	return (a>=b)?b:a;
}	
int main()
{
    int t;
    cin>>t;
    while(t--)
   { 
     int n,k,f=0;
     cin>>n>>k;
     int count[101]={0};
     
     int *a = new int[n];
     
    for(int i=0; i<n; i++)
    {    
       cin>>a[i];
       count[a[i]]++;
       if(count[a[i]]>f)
         f=count[a[i]];
    }
    
    if(f>=k)
    { cout<<0<<endl;
      continue;
    }
    
    int aux[100][100]={0};
    
    for(int i=0; i<n; i++)
     for(int j=0; j<n; j++)
      aux[i][j]=(a[i]>a[j])?(a[i]-a[j])*3:(a[j]-a[i])*5;
      
    for(int i=0; i<n; i++)  
      sort(aux[i],aux[i]+n);
      
    long int m=INT_MAX;
    
    for(int i=0; i<n; i++)
    {
    	 int c=count[a[i]];
    	 long int sum=0;
    	 for(int j=0; j<n; j++)
    	 {
    	 	if(aux[i][j]!=0)
    	 	 {
    	 	 	sum+=aux[i][j];
    	 	 	c++;
    	 	 } 
    	 	if(c>=k)
    	 	  break;
    	 }
      m=min(sum,m);	
    }
      
    cout<<m<<endl; 
   } 
    //cout << "Hello World!" << endl;
    return 0;
}
