from math import sqrt


sieve = [False for i in xrange(10**6 + 2)]
count = [0 for i in xrange(10**6 + 2)]

for i in xrange(2, int(sqrt(10**6 + 2))):
    for j in xrange(2, 10**6 + 2):
        if i*j > 10**6:
            break

        if not sieve[i*j]:
            sieve[i*j] = True
            count[i] += 1

for t in xrange(int(raw_input())):
    p = int(raw_input())
    print(count[p] + 1)
