#include<iostream>
#include<string>
using namespace std;

int main()
{
	
	int test;
	string name;
	//cout<<name.max_size();
	cin>>test;
	while (test--)
	{
		cin>>name;
		int len=name.size();
		int i;
		int count=0;
		int total=0;
		for (i=0;i<len;i++)
		{
			while (name[i]==name[i+1])
			{
				i++;
				count++;
			}
			total+=count;
			count=0;
		}
		cout<<total<<endl;
	}
	return 0;
}
