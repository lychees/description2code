#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int main() {
    int n, flag = 0, max_cnt = 0, min_cnt = 0, max_fin = 0, min_fin = 0, v_cnt = 0, w_cnt = 0;
    cin>>n;
    string S;
    cin>>S;
    for(int  i = 0; S[i] != '\0'; i++) {
        if(((S[i]-97) == 21) || ((S[i]-97) == 22)) {
            flag = 1;
            if((S[i]-97) == 21)
                v_cnt++;
            else
                w_cnt++;
        }
        else {
            if(flag) {
                max_cnt = (w_cnt * 2) + v_cnt;
                min_cnt = (max_cnt / 2) + (max_cnt % 2);
                max_fin += max_cnt;
                min_fin += min_cnt;
                w_cnt = 0;
                v_cnt = 0;
            }
            max_fin++;
            min_fin++;
            flag = 0;
        }
    }
    if(flag) {
        max_cnt = (w_cnt * 2) + v_cnt;
        min_cnt = (max_cnt / 2) + (max_cnt % 2);
        max_fin += max_cnt;
        min_fin += min_cnt;
    }
    cout<<min_fin<<" "<<max_fin;
    //char c = 97+22;
    //cout<<c;
    return 0;
}
