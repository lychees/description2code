#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin>>n;
    string s,c;
    cin>>s;
    c=s;
    reverse(s.begin(),s.end());
    for(int i=0;i<n-1;i++){
    	cin>>s;
    	c+=s;
    	reverse(s.begin(),s.end());
    	size_t t1=c.find(s);
    	if(t1!=string::npos){
    		//cout<<c<<endl;
    		cout<<s.length()<<" "<<c[t1+s.length()/2]<<endl;
    		break;
    	}
    }
    return 0;
}
