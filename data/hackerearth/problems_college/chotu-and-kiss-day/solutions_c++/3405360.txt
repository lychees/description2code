#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
lli arr[3005];
int main()
{
	lli i,j,k,m,n,t,index,sum,flag=0;
	slli(n);
	f(i,0,n)
	slli(arr[i]);

	slli(k);

	sort(arr,arr+n);

	f(i,0,n)
	{
		if(arr[i]>k)
			break;
		f(j,i+1,n)
		{
			if(arr[j]>k)
				break;
			sum=arr[i]+arr[j];
			if(k > sum)
			{
				index=binary_search(arr,arr+n,k-sum);
				if(index)
				{
					printf("YES\n");
					flag=1;
					break;
				}
			}
		}
		if(flag==1)
			break;
	}
	if(flag!=1)
	printf("NO\n");
}
