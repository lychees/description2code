#define MOD 1000000007     // 10^9+7
#define INF 100000008     // 10^9+9

bool prime[100000008];
void sieve()
{
  ll k,i,j;
   k=sqrt(INF);
   memset(prime,true,sizeof prime);
   prime[0]=prime[1]=false;
   rep(i,2,k+1)
   {
     if(prime[i]){
     for(j=i+i;j<INF;j+=i)
      prime[j]=false;
      }
   }
}

int main()
{
ll t,m,n,ans,i;
   sieve();
   rd(t);
   while(t--)
   {
     rd(m);
     rd(n);
     ans=0;
     rep(i,m,n+1)
      if(prime[i])
       ans++;
    wr(ans);
   }

   return 0;
}
