#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,sum,l;
	string str,s;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		sum=0;
		s="";
		char ch;
		getline(cin,str);
		l=str.length();
		int count=0;
		map<string,int>my;
		map<string,int>::iterator it;
		int i;
		for(i=0;i<l;i++)
		{
			if(str[i]>='0'&&str[i]<='9')
			sum+=str[i]-'0';
			else if(str[i]==32)
			{
			my[s]=count;
			s="";
			count =0;
			}
			else
			{
				s+=str[i];
				count++;
			}
		}
			my[s]=count;
		//cout<<"4";
		for(it=my.begin();it!=my.end();it++)
		{
			cout<<it->first<<" ";
			sum=sum-(it->second);
			if(sum<0)
			{
				cout<<(-1)*sum<<" ";
			}
			else
			cout<<sum<<" ";
		}
		cout<<"\n";
		
	}
	return 0;
}
