#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int i,a[n],dp[n],j,maxi=1;
    memset(dp,0,sizeof(dp));
    for(i=0;i<n;i++) cin>>a[i];
    dp[0]=1;
    for(i=1;i<n;i++) {
    	int temp=0;
    	for(j=i-1;j>=0;j--) {
    		if(a[i]>a[j] && temp<dp[j]) temp=dp[j];
    	}
    	dp[i]=temp+1;
    	if(dp[i]>maxi) maxi=dp[i];
    //	cout<<dp[i]<<" ";
    }
   //cout<<maxi<<endl;
    stack<int> st;
    while(maxi>0) {
    	st.push(maxi%2);
    	maxi=maxi/2;
    }
    while(!st.empty()) {
    	cout<<st.top();
    	st.pop();
    }
    return 0;
}
