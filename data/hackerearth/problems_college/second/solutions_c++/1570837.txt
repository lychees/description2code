#include <iostream>
#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<string.h>
#define MAX 604800
using namespace std;
int main()
{
	  int n,a[100000],x,ans=0;
	  cin>>n>>x;
	  for(int i=0;i<n;i++)
	  cin>>a[i];
	  for(int i=0,j=x;i<n;i++)
	  {
	  	if(a[i]==j)
	  	{
	  		j+=x;
	  		ans=a[i];
	  		i=0;
	  	}
	  }
	  cout<<ans<<endl;
}
