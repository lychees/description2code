def inputnumber():
    num = raw_input().split(' ')
    return [int(n) for n in num]


def balanced(numbers):
    for pivot in range(len(numbers)):
        left_total = sum(numbers[:pivot])
        right_total = sum(numbers[pivot:])
        if left_total == right_total:
            return True
    return False
