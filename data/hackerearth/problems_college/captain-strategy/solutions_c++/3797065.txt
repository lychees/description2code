#include<iostream>
using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int e[n+1];
	int ans[m+1];
	for(int i=1;i<n+1;i++){
		cin>>e[i];
	}
	int x,y;
	int d[n+1];
	for(int i=1;i<n;i++){
		cin>>x>>y;
		d[y]=x;
	}
	d[1]=0;
	/*for(int i=1;i<n+1;i++){
		cout<<e[i]<<" ";
	}
	cout<<endl;
	for(int i=1;i<n+1;i++){
		cout<<d[i]<<" ";
	}*/
	for(int i=1;i<=m;i++){
		char a;
		cin>>a;
		int x,y;
		cin>>x;
		if(a=='Q'){
			
			int sum=0;
			if(x!=1){
				
			for(int i=2;i<=n;i++){
				if(i!=x){
					int t=d[i];
					while(t!=x){
						t=d[t];
						if(t==0) break;
					}
					if(t==x) sum+=e[i];
				}
			}
			sum+=e[x];
			
		}
		else {
			for(int i=1;i<=n;i++) sum+=e[i];
		}
		ans[i]=sum;
	}
		else{
			cin>>y;
			e[x]=y;
			ans[i]=-1;
		}
	}
	for(int i=1;i<=m;i++){
		if(ans[i]!=-1) cout<<ans[i]<<endl;
	}
	return 0;
}
