from collections import Counter
temp_shelf = []
n = int(raw_input().strip())
for i in range(n):
    title = str(raw_input().strip())
    temp_shelf.append(title[0])
temp_shelf = dict(Counter(temp_shelf))
shelf = sum([(i//10)+1 if i % 10 > 0 else (i//10)
            for i in temp_shelf.values()])

print shelf
