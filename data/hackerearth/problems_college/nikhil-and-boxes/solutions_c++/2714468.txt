#include<iostream>
#define ll long long
using namespace std;
int main()
{
	ll t,l,b,h;
	
	cin>>t;
	while(t--)
	{
		scanf("%lld%lld%lld",&l,&b,&h);
		if(b<l)
		{
			if(b<h)
			{
				printf("%.1f\n",(double)b/2);
			}
			else
			{
				printf("%.1f\n",(double)h/2);
			}
		}
		else
		{
			if(l<h)
			{
				printf("%.1f\n",(double)l/2);
			}
			else
			{
				printf("%.1f\n",(double)h/2);
			}
		}
	}
	return 0;
}
