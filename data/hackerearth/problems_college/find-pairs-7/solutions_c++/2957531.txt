#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	long long n,i,c,j;
	scanf("%d",&t);
	while(t--)
	{
		c=0;
		scanf("%lld",&n);
		long long a[n];
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		//sort(a,a+n);
		
		for(i=0;i<n;i++)
		{
			for(j=i;j<n;j++)
			{
			if(a[i]==a[j])
			c++;
			
			}
			}
		
		printf("%lld\n",c);
	}
}
