#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll int fib[1000001];
ll int sum_array[1000001];

bool isprime(ll int x)
{
    for(ll int i=2;i<=sqrt(x);i++)
    {
        if(x%i==0)
            return false;
    }
    return true;
}

int main()
{
    fib[0]=0;
    fib[1]=1;
    sum_array[0]=0;
    sum_array[1]=1;
    ll int sum=1;
    for(ll int i=2; i<1000001; i++)
    {
        fib[i]=fib[i-1]+fib[i-2];
        sum=sum+fib[i];
        sum_array[i]=sum;
    }

    ll int t;
    cin>>t;
    while(t--)
    {
        ll int n;
        cin>>n;
        ll int x;
        for(ll int i=0; i<1000001; i++)
        {
            if(fib[i]>n)
            {
                x=i;
                break;
            }
        }
        x--;
        ll int ans=((n+1)*n)/2-sum_array[x]-n+1;

        if(isprime(ans))
        {
            cout<<"Tic\n";
            continue;
        }

        else if(ans%2==0)
        {
            cout<<"Tac\n";
            continue;
        }
        else
        {
            cout<<"Toe\n";
            continue;
        }

    }
    return 0;
}
