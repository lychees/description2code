# print 'Hello World!'

T = int(raw_input())
for t in range(T):
    N = int(raw_input())
    food = dict()
    count = 0
    for i in range(N):
        Ai, Bi = map(int, raw_input().split())
        if Ai == Bi:
            continue
        else:
            if food.get(Bi) and food.get(Bi) > 0:
                food[Bi] -= 1
            else:
                count = count+1
            if food.get(Ai):
                food[Ai] += 1
            else:
                food[Ai] = 1
    print count
