typedef map<string, int> msi;
// To simplify repetitions/loops, Note: define your loop style and stick with it!
#define s(i) scanf("%d",&i)
#define sl(i) scanf("%ld",&i)
#define sll(i) scanf("%lld",&i)
#define REP(i, a, b) \
for (int i = int(a); i <= int(b); i++) // a to b, and variable i is local!
#define TRvi(c, it) \
for (vi::iterator it = (c).begin(); it != (c).end(); it++)
#define TRvii(c, it) \
for (vii::iterator it = (c).begin(); it != (c).end(); it++)

int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","r",stdout);
    int t;s(t);
    double phi = (1.00 + sqrt(5.00))/2;
    while(t--)
    {
        ll ans = 0;
        ii p,q;
        s(p.first);s(p.second);s(q.first);s(q.second);
       // cout << floor(p.first * phi) + 1 << endl;
        REP(i,p.first,p.second)
        {
            int val = floor(double(i) * phi) + 1;
            ll add = q.second - max(val,q.first) + 1;
            //cout << add << endl;
            if(add <= 0)
                break;
            ans += add;
        }
        REP(i,q.first,q.second)
        {
            int val = floor(double(i) * phi) + 1;
            ll add = p.second - max(val,p.first) + 1;
            //cout << add << endl;
            if(add <= 0)
                break;
            ans += add;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
