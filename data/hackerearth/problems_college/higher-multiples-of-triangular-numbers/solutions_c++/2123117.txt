#include <iostream>
using namespace std;
int series(int);
int count(int);
int main()
{
    int t,n,i,c,b;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int i=1;
    	for(i=1;i<=1000;i++)
    	{
    		
    		c=series(i);
    		b=count(c);
    		
    		if(b==n)
    			continue;
    		if(b>n)
    			break;
    	}
    	cout<<c<<endl;
    }
    return 0;
}
int series(int n)
{
	return (n*(n+1)/2);
}
int count(int c)
{
	int count=0,i;
	for(i=1;i<=c;i++)
	{
		if(c%i==0)
		{
			count++;
		}
	}
	return count;
}
