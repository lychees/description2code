import sys
def xqrt(n):
    lo,hi = 0,1000000000
    while(lo<hi):
	mid=(lo+hi+1)/2
	if(mid*mid<=n):
	    lo=mid
	else:
	    hi=mid-1
    return lo
for _ in xrange(input()):
    L,R=map(int,raw_input().split())
    x,y=-1,-1
    rt=xqrt(L)
    if(rt*rt == L):
	x=rt
    elif(rt+1)*(rt+1) <=R :
	x=rt+1
    rt=xqrt(R)
    if(rt*rt>=L):
	y=rt
    if(x==-1 and y==-1):
	print "0"
    else:
	print y-x+1
