#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    char str[100];
    int T,l,i=0,f;
    cin>>T;
    while(T--)
    {
    	f=0;
    	i=0;
    	cin>>str;
    	l=strlen(str)-1;
    	while(i<=(l/2))
    	{
    		if(str[i]!=str[l])
    		{
    			cout<<"NO\n";
    			f=1;
    			break;
    		}
    		l--;
    		i++;
    	}
    	if(!f)
    	{
    		cout<<"YES\n";
    	}
    }
    return 0;
}
