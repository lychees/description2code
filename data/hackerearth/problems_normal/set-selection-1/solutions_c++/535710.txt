#include<iostream>
using namespace std;
int p,q,r;
bool is_or_not[1000001];  
   
   int main(){
   cin >> p; 	
   int a[p];
   for(int i = 0; i < p ; i++){
   cin  >> a[i];
   }
   
   is_or_not[0] = true;
   for(int i = 1; i <1000001;i++){
   is_or_not[i] = false;
   }
   
   
   for(int i = 0; i <p;i++ ){
   	for(int j = 1000001-a[i]; j>=0 ; j-- ){
   	  if(is_or_not[j]){
   	  	is_or_not[j+a[i]] = true;
   	  }
	 }
   }
   
   
   cin >> q;
   while(q--){
   cin >> r;
   if(is_or_not[r]){
   cout << "Yes" << endl;
   }
   else{
   cout << "No" <<endl;
   }
   
   } 
   
   
   	return 0;
}
