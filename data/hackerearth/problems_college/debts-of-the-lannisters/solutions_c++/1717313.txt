#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char a[11];
	cin>>a;
	int t;
	cin>>t;
	while(t--){
		char c[11];
		strcpy(c,a);
		int g=0,s=0;
		char b[11];
		cin>>b;
		for(int i=0;a[i];i++){
			if(a[i] == b[i])
			{	
				g++;
				b[i]='Q';
				a[i]='D';
			}	
		}
		
		for(int i=0;a[i];i++)
			for(int j=0;a[j];j++)
				if(a[i]==b[j])
				{
					s++;
					//cout<<i<<','<<j<<','<<a[j]<<' ';
					a[i] = 'Z';
					b[j] = 'E';
				}
				
		//cout<<'\n';
		cout<<g<<' '<<s<<'\n';
		strcpy(a,c);
	}
}
