#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	ll t,n,i;
	vector<ll> v(20000001);
	v[1]=0;
	v[2]=1;
	v[3]=1;
	for(i=4;i<=20000000;i++)
	{
		v[i]=1+v[i-1];
		if(i%3==0)
		{
			v[i]=min(v[i],1+v[i/3]);
		}
		 if(i%2==0)
		{
			v[i]=min(v[i],1+v[i/2]);
		}
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		printf("%lld\n",v[n]);
	}
    return 0;
}
