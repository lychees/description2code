#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int max=100000000000000000,i,n,k,arr[100005];
    cin>>n>>k;
    for(i=0;i<n;i++)
    cin>>arr[i];
    sort(arr,arr+n);
    for(i=0;i<=n-k;i++){
    	if(arr[k+i-1]-arr[i]<max)
    	max=arr[k+i-1]-arr[i];
    }
    cout<<max<<endl;
    
    return 0;
}
