#include <bits/stdc++.h>
using namespace std;
int DP[10005];
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
    	int n;
    	int sum=0;
    	cin>>n;
    	int arr[100];
    	
    	for(int i=0; i<n; i++)
    	{
    		cin>>arr[i];
    		sum += arr[i];
    	}
    	
    	memset(DP, 0, sizeof(DP));
    	DP[0]=1;
    	int counter = 1 ;
    	for(int i=0; i<n; i++)
    	{
    		for(int j=sum; j>=0; j--)
    		{
    			if(DP[j])
    			{
    				if(!(DP[j+arr[i]]))
    				{
    					DP[j+arr[i]] = 1;
    					counter++;
    					//cout<<j+arr[i]<<"\t";
    				}
    			}
    		}
    		//cout<<"\n";
    	}
    	cout<<counter<<"\n";
    }	
    return 0;
}
