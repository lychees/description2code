#include<iostream>
using namespace std;
long long arr[100000]={0};
long long brr[100000]={0};
void fun();
void fun()
{
	long long k=0,l=0,e=0,o=0;
	for(int i=1;i<=100000;i++)
	{
		if(i%2==0)
		e+=i;
		else
		o+=i;
		arr[k++]=e;
		brr[l++]=o;
	}
}
int main()
{
	fun();
	int t;
	cin>>t;
	while(t--)
	{
		int r,c;
		scanf("%d%d",&r,&c);
		long long o=0,e=0;
		if(r<=c)
		printf("%lld %lld\n",arr[r-1],brr[r-1]);
	   else
	   {
	   	for(int i=c+1;i<=r;i++)
	    {
	    	if(i%2==0)
	    	e+=c;
	    	else
	    	o+=c;
	    }
	    printf("%lld %lld\n",arr[c-1]+e,brr[c-1]+o);
       }
    }
    return 0;
}
