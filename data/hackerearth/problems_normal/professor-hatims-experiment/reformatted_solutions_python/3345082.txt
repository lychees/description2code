interactions = {}
color = []
color_dict = {"Black": "Grey", "Grey": "Black"}


def bfs(lizard1):
    queue = [lizard1]
    susp = False
    while len(queue) != 0 and not suspicious:
        lizard1 = queue.pop(0)

        for lizard in interactions[lizard1]:
            if color[lizard] == color[lizard1]:
                susp = True
                break
            elif color[lizard] == "White":
                color[lizard] = color_dict[color[lizard1]]
                queue.append(lizard)
    return susp


tc = int(raw_input())

for i in range(tc):
    N, M = map(int, raw_input().split(" "))
    interactions = dict((i, []) for i in range(N+1))

    for _ in range(M):
        a, b = map(int, raw_input().split(" "))
        interactions[a].append(b)
        interactions[b].append(a)

    color = ["White" for i in range(N+1)]

    for i in range(N):
        suspicious = False
        if color[i] == "White":
            color[i] = "Black"
            suspicious = bfs(i)
            if suspicious:
                break

    if suspicious:
        print "Suspicious lizards found!"
    else:
        print "No suspicious lizards found!"
