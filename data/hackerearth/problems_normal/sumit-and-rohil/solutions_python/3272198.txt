def checkIfTwoStringsAreAnagrams(str1,str2):
    str1List=[character for character in str1 ]
    str2List=[character for character in str2 ]

    for character in str2List:
         if character not in str1List:
             return False
         else:
             str1List.remove(character)

    if not str1List:
        return True
    else:
        return False



N=int(raw_input())

names=[raw_input() for x in xrange(0,N)]
solDict={}

for name in names:
    key=name[0]+name[-1]
    list=solDict.get(key,[])
    list.append(name)
    solDict[key]=list


inputStack=[]
processedStack=[]


for value in solDict.itervalues():
    inputStack.append(value)

while (inputStack):
    group=inputStack.pop()
    firstString=group[0]
    notAnagramsWithFirstString=[]
    tempList=[x for x in group]
    for entry in tempList:
        if  not checkIfTwoStringsAreAnagrams(firstString,entry):
            group.remove(entry)
            notAnagramsWithFirstString.append(entry)
    if notAnagramsWithFirstString:
        inputStack.append(notAnagramsWithFirstString)

    processedStack.append(group)

print len(processedStack)
