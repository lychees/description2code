#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
int check(int n)
{
    int l=sqrt(n),c=0;
    for(int i=1;i<=l;++i)
    {
        if(n%i==0)
        {
            if(i%2==0)
            c++;
            if(n/i>l && n/i%2==0)
            c++;
        }
    }
    return c;
}
int main()
{
    int n,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",check(n));
    }

}
