#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n],i=0,j=0,cnt=0,ch=0;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;)
    	{
    		j=a[i];
    		i=j;
    		if(i==-1)
    		{
    			ch=1;
    			break;
    		}
    		cnt++;
    		if(cnt>n)
    		{
    			ch=2;
    			break;
    		}
    	}
    	if(ch==1)
    	{
    		cout<<"Win"<<endl;
    	}
    	else
    	cout<<"Lose"<<endl;
    }
    return 0;
}
