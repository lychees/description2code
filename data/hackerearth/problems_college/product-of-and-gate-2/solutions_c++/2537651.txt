#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin >> t;
    while(t--)
    {
    	long long int a,b,x=pow(2,64)-1;
    	cin >> a >> b;
    	for(long long int i=a; i<=b; i++)
    	{
    		x=(i & x);
    	}
    	cout << x <<"\n";
    }
    return 0;
}
