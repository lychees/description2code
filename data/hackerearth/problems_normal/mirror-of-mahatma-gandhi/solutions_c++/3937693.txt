#include <bits/stdc++.h>
using namespace std;
int pal(string n){
string k=n;
reverse(n.begin(),n.end());
if(k==n) return true;
else return false;
}
int main() {
int t;
cin>>t;
while(t--){
string n;
cin>>n;
int flag=0;
for(int i=0;i<n.size();i++){
if(!pal(n)) {flag=0; break;}
if(n[i]=='0' or n[i]=='1' or n[i]=='8'){ flag=1;}
else {flag=0; break;}
}
if(flag) cout<<"YES"<<endl;
else cout<<"NO"<<endl;
}
return 0;
}
