#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char input_str[100];
    int i;
    cin>>input_str;
    for(i=0;i<strlen(input_str);i++)
    {
      if(input_str[i]>64 && input_str[i]<97)
      {
      	input_str[i]=input_str[i]+32;
      }
      else
      
      if(input_str[i]>96)
      {
      	input_str[i]=input_str[i]-32;
      }
    }
      cout<<input_str;
    return 0;
}
