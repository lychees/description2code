#include <bits/stdc++.h>
using namespace std;

int main()
{
	 long a[100][100];
	unsigned int i,j,n;
	long s1,s2;

    for(i=1;i<=n;i++)
    { for(j=1;j<=n;j++)
    cin>>a[i][j];
    	
    }

 	s2=s2+a[i][n-i+1];
 }

}
