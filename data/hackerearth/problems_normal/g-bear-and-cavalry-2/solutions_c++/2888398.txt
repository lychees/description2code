#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
using namespace std;
int tc,n,w[50],h[50],a;
bool compare(int a, int b){
	return a>b;}
int main()
{
	scanf("%d",&tc);
	
    while(tc--){
    	scanf("%d",&n);
    	for(int i=0;i<n;i++){
    		scanf("%d",&w[i]);
    	}
    	for(int i=0;i<n;i++){
    		scanf("%d",&h[i]);
    	}
    	sort(w+1,w+n);
    	sort(h,h+n,compare);
    	bool ans=true;
    	for(int i=1;i<n;i++){
    		if(h[i]*w[i]>=w[0]*h[0]){
    			ans=false;
    			break;
    		}
    	}
    	if(ans){
    		puts("YES");
    	}else{
    		puts("NO");
    	}
    }
    return 0;
}
