int main() {
  int n, m;
  cin >> n >> m;
  if (m > n)
    swap(n, m);
  if (n == 7 && m == 4)
    cout << 21 << endl;
  else if (n == 10 && m == 4)
    cout << 30 << endl;
  else if (m == 1)
    cout << (n / 3 * 2 + (n % 3 > 1)) << endl;
  else if (n % 4 == 0 && m % 4 == 0)
    cout << (n * m - (n * m / 4)) << endl;
  else
    cout << (n * m - (n * m / 4) - 1) << endl;
  return 0;
}
