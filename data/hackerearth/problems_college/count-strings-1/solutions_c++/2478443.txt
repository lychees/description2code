#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
char s[1000010],t[1000010];
long long a[26],b[26],x[1000010];

long long powmod(long long g)
{
    long long d=1,h=mod-2;
    while(h){
        if(h&1){
            d=(d*g)%mod;
        }
        g=(g*g)%mod;
        h=h/2;
    }
    return d%mod;
}
long long funcb(long long g,long long h)
{
    long long p,q,r,d;
    p=x[g];
    q=x[g-h];
    d=(q)%mod;
    d=powmod(d);
    return (p*d)%mod;
}
int main()
{
    int z;
    x[0]=1;
    long long i,j,k,l1,l2,n,m;
    for(i=1;i<1000010;i++){
        x[i]=(x[i-1]*i)%mod;
    }
    scanf("%d",&z);
    while(z--){
        scanf("%s %s",s,t);
        l1=strlen(s);
        l2=strlen(t);
        for(i=0;i<l1;i++){
            a[s[i]-'a']++;
        }
        for(i=0;i<l2;i++){
            b[t[i]-'a']++;
        }
        j=0;
        for(i=0;i<26;i++){
            if(a[i]>=b[i]){
                j++;
            }
        }
        long long ans=1;
        if(j!=26){
            printf("0\n");
        }
        else{
            for(i=0;i<26;i++){
                ans=(ans*funcb(a[i],b[i]))%mod;
            }
            printf("%lld\n",ans);
        }
        memset(a,0,sizeof(a));
        memset(b,0,sizeof(b));
    }
}
