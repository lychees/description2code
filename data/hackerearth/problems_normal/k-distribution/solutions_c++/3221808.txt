#include<bits/stdc++.h>
typedef long long int lli;
lli arr[101];
using namespace std;
#define mod 1000000007
lli dp[102][500010];
int main()
 {
 	 int n,k;
 	  cin>>n>>k;
 	  lli sum=0;
 	  for(int i=1;i<=n;i++) cin>>arr[i],sum+=arr[i];
 	  if(sum<2*k)// special case 
 	  {
 	  	cout<<0<<endl;
 	  	return 0;
	   }
 	 dp[0][0]=1;
 	 for(int i=1;i<=n;i++)
 	  {
 	  	 if(arr[i]>k) 
 	  	 {
 	  	 	 for(int j=0;j<=k;j++) dp[i][j]=dp[i-1][j];
 	  	 	 
			}
 	  	 else
 	  	 {
 	  	 	for(int j=0;j<=k;j++)
 	  	 	 {
 	  	 	 	dp[i][j]=(dp[i][j]+dp[i-1][j])%mod;
 	  	 	 	if(j-arr[i]>=0)
 	  	 	 	 if(dp[i-1][j-arr[i]]>=1)
 	  	 	 	  {
 	  	 	 	  	dp[i][j]=(dp[i][j]+dp[i-1][j-arr[i]])%mod;
					   }
				 }
			}
	   }
	   
	   //cout<<dp[n][0]<<endl;
	   lli ans=1;
	   for(int i=1;i<=n;i++)
	    {
	    	ans=(ans*2)%mod;
		}
		
		for(int i=0;i<k;i++)
		 {
		 	 if(dp[n][i]!=0)
		 	  {
		 	  	 ans=ans-2*dp[n][i];
		 	  	 if(ans<0) ans+=mod;
			   }
		 }
 	cout<<ans<<endl;
 }
