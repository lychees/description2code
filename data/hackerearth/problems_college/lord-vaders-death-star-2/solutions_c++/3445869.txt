#include <bits/stdc++.h>
using namespace std;
map<int,int> mp;
int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
    	int b;
    	cin>>b;
    	if(mp.find(b)==mp.end()){
    		mp[b]=1;
    	}
    	else
    		mp[b]++;
    }
    map<int,int>::reverse_iterator it=mp.rbegin();
    int flg=0;
    for(;it!=mp.rend();it++){
    	if(it->second==1){
    		flg=1;
    		break;	
    	}
    }
    if(flg){
    	cout<<it->first<<endl;
    }
    else
    	cout<<"-1";
    return 0;
}
