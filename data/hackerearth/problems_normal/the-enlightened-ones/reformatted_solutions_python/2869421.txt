'''

name = raw_input('What is your name?\n')
print 'Hi, %s.' % name
'''
n, k = map(int, raw_input().split())
arr = sorted(map(int, raw_input().split()))
l, h = 0, 10**7

while (l < h):
    mid = l + (h-l)//2
    count, pos = 1, arr[0] + mid
    for i in xrange(1, n):
        if arr[i] - pos > mid:
            count += 1
            pos = arr[i] + mid
    if count > k:
        l = mid + 1
    else:
        h = mid
print l
