#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	cin>>a>>b;
    	int h1[26]={0},h2[26]={0},i,count=0;
    	for(i=0;i<a.length();i++)
    	{
    		h1[a[i]-'a']++;
    	}
    	for(i=0;i<b.length();i++)
    	{
    		h2[b[i]-'a']++;
    	}
    	for(i=0;i<26;i++)
    	{
    		if(h1[i] && h2[i])
    		  count++;
    	}
    	if(count)
    	cout<<"Yes\n";
    	else
    	cout<<"No\n";
    }
    return 0;
}
