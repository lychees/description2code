#include<bits/stdc++.h>

using namespace std;
int n,k;
long long dp[210][210];
long long rec(int SumLeft,int minTake)
{
    if(SumLeft-minTake==0)
        return 1;
    //Sumleft<0
    if(SumLeft<minTake || SumLeft<k  )
        return 0;
    if(dp[SumLeft][minTake]!=-1)
        return dp[SumLeft][minTake];

   return dp[SumLeft][minTake]=rec(SumLeft-minTake,minTake)+rec(SumLeft,minTake+1);
}

int main ()
{
    int t; scanf("%d",&t);
    while(t--)
    {
        memset(dp,-1, sizeof dp);
         scanf("%d%d",&n,&k);
         long long ans=rec(n,k);
         printf("%lld\n",ans);
    }

  return 0;
}
