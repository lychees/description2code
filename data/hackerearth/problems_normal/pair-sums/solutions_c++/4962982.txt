#include "bits/stdc++.h"
using namespace std;
typedef long long ll;
int main()
{
	ios::sync_with_stdio(0);cin.tie(0);cout.tie(0);
    //cout << "Hello World!" << endl;
    ll n,k,i,x;
    cin>>n>>k;
    ll a[n];
    map<ll,ll> m;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	m[a[i]]=i;
    }
    for(i=0;i<n;i++)
    {
    //	cout<<i<<" "<<k-i<<"\n";
    //	cout<<a[i]<<" "<<a[k-i]<<"\n";
    x=k-a[i];
    if(m.find(x)!=m.end())
    {
    	if(m[x]!=i)
    	{
    		cout<<"YES\n";
    		goto p;
    	}
    }
    }
    cout<<"NO\n";
    p:;
    return 0;
}
