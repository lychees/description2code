#include<iostream>
using namespace std;
const int MAX=(int)(3*1e5+1e1);
int ar[MAX],br[MAX],ct1[MAX],ct2[MAX],n;
int main()
{
	cin>>n;
	for(int i=1;i<=n;++i)
	{
		cin>>ar[i]>>br[i];
		ct1[ar[i]]++;
		ct2[br[i]]++;
	}
	for(int i=1;i<=n;++i)
	{
		int b=br[i];
		b=ct1[b];
		int a1=n-1,a2=n-1;
		a1+=b;
		a2-=b;
		cout<<a1<<" "<<a2<<endl;
	}
	return 0;
}
