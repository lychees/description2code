#include <iostream>
using namespace std;

int main()
{
       int t;
       cin>>t;
       for(int k=0;k<t;k++)
       {
       	char s[1000000];
       	cin>>s;
       	int i=0;
       	int j=i+1;
       		while(s[j]!='\0')
       		{
       			if(s[i]!=s[j])
       			{
       				cout<<s[i];
       				i=j;
       				j++;
       			}
       			else{j++;}
       		}
       		cout<<s[j-1]<<endl;
       	}

           return 0;
}
