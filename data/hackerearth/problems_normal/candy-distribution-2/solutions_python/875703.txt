test = int(raw_input())
for i in range(test):
    n = int(raw_input())
    line = raw_input()
    ints_list = []
    sum = 0
    ints_list = line.split(' ')
    for i in ints_list:
        sum += int(i)
    sum = sum - n + 1
    print(sum)
