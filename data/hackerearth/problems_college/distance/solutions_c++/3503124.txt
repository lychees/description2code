#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int gx,gy,sx,sy;
    float d;
    scanf("%d %d %d %d",&gx,&gy,&sx,&sy);
    d=sqrt(((gx-sx)*(gx-sx))+((gy-sy)*(gy-sy)));
      cout.setf(ios::fixed,ios::floatfield);
    cout.precision(5);
    cout<<d;
    return(0);
    }
