#include <iostream>
#include <cstring>
#include <string.h>
using namespace std;

int main()
{
	int t,i;
	cin>>t;
	while(t--){
		char str[10000];
		cin>>str;
		
		int flag=1;
		for(i=0;i<strlen(str)/2;i++){
			if(abs(str[i]-str[i+1])!=abs(str[strlen(str)-1-i]-str[strlen(str)-2-i])){
				flag=0;
				break;
			}
		}
		if(flag==1){
			cout<<"Funny"<<endl;
		}else{
			cout<<"Not Funny"<<endl;
		}
	}
	return 0;
}
