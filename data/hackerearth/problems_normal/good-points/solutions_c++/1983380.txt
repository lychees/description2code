#include <iostream>
using namespace std;

int main()
{
    int N=0;
    scanf("%d\n",&N);
    if(N==4)
     {
     	printf("YES\n0 0\n0 1\n1 1\n1 0\n");
     }
     else if(N==1)
     {
     	printf("YES\n0 0\n");
     }
     else if(N==2)
     {
     	printf("YES\n0 0\n1 0\n");
     }     
     else
     {
     	printf("NO\n");
     }
    return 0;
}
