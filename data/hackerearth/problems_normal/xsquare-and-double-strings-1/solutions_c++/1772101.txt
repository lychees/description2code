#include<iostream>
#include<cstdio>
#include<map>
#include<string>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        map<char,int>m;
        m.clear();
        int sz=s.size();
        int j=0;
        for(int i=0;i<sz;i++)
        {
            if(m[s[i]]!=0)
            {
                cout<<"Yes"<<endl;
                j=1;
                break;
            }
            else
            {
                m[s[i]]++;
            }
        }
        if(j==0)cout<<"No"<<endl;
    }
}
