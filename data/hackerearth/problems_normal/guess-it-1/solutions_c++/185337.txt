using namespace std;
int main(){
	int t,sol[100];
	char ch[200];
	int s;
	cin>>t;
	for(int i=0;i<t;i++)
	{
	cin>>ch;
    s=0;
	for(int a=0;ch[a]!='\0';a++)
    {
    	if(ch[a]!='_')
    s=s+ch[a]-48;
}
sol[i]=s%9;

}
	for(int i=0;i<t;i++)
	{if(sol[i])
	 cout<<9-sol[i]<<endl;
	 else
	 cout<<"0 or 9"<<endl;
	}


}
