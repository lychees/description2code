#include<bits/stdc++.h>
using namespace std;
int INF = 1e7;
int memo[100001];
int dp(int x)
{
	int arr[] = {2,3,5,7};
	if(memo[x])
		return memo[x];
	for(int i=0;i<4;i++)
	{
		int res = dp(x-arr[i]);
		if(res!=-1)
		{
			if(!memo[x])
			{
				memo[x] = res+1;
			}
			else
				memo[x] = min(res + 1,memo[x]);	
		}
	}
	if(!memo[x])
		return memo[x] = -1;
	else 
		return memo[x];
}
int main()
{
	int t,x;
	cin>>t;
	memset(memo,0,sizeof(memo));
	memo[0] = -1;
	memo[1] = -1;
	memo[2] = 1;
	memo[3] = 1;
	memo[4] = -1;
	memo[5] = 1;
	memo[6] = 2;
	memo[7] = 1;
	memo[8] = 2;
	memo[9] = 3;
	while(t--)
	{
		cin>>x;
		cout<<(dp(x)>=INF?-1:dp(x))<<endl;
	}
}
