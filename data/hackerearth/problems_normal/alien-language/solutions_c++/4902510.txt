#include<bits/stdc++.h>
using namespace std;
bool func(string x,string y)
{
	map<char,bool> mp;
	for(int i=0;i<x.size();i++)mp[x[i]]=true;
	for(int i=0;i<y.size();i++)
	{
		if(mp[y[i]])return true;
	}
	return false;
}
int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		string s1,s2;
		cin >> s1;
		cin >> s2;
		if(func(s1,s2))cout<<"YES"<<endl;
		else cout << "NO"<<endl;
	}
	return 0;
}
