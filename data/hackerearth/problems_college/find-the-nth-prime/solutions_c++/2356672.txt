#include <iostream>
#include<cmath>
using namespace std;
bool a[8001];
int b[1001];
int main()
{
	int i,j,n;
	for(i=2;i<=8000;i++)
	a[i]=true;
	int k=sqrt(8000);
	for(i=2;i<=k;i++)
	{
		if(a[i])
		{
			for(j=2;i*j<=8000;j++)
			{
				a[j*i]=false;	
			}
		}
		
	}
	int cnt=1;
	for(i=2;i<8000;i++)
	{
		if(a[i])
		b[cnt++]=i;
	}
	cin >> n;
	cout << b[n]  << endl;
	
	
	return 0;
}
