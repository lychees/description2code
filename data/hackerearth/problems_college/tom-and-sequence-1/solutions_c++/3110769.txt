#include <iostream>
#define mod 1000000007
using namespace std;
bool isPrime(long int n)
{
	if(n<=1)
		return false;
	if(n==2)
		return true;
	for(int j=2;j*j<=n;j++)
	{
		if(n%j==0)
			return false;
	}
	return true;
}
int main()
{
   	long long int a[100005];
   	long long int p=2,k=1;
   	a[1]=1;
   	a[0]=1;
   	while(k<=100000)
   	{
   		if(isPrime(p))
   		{
   			a[k]=( (p%mod) * (a[k-1]%mod) )%mod;
   			k++;
   		}
   		p++;
   	}
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		long int n;
   		cin>>n;
   		cout<<a[n]<<endl;
   	}
    return 0;
}
