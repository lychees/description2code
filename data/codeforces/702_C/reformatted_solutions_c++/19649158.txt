#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <map>
#include <queue>
#include <stack>
#include <utility>
#include <vector>
using namespace std;
const int qq = 1e5 + 10;
int num[qq], dp[qq];
int n, m;
int main() {
  scanf("%d%d", &n, &m);
  for (int i = 1; i <= n; ++i)
    scanf("%d", &num[i]);
  for (int i = 1; i <= m; ++i)
    scanf("%d", &dp[i]);
  int t = 1, r = 0;
  for (int i = 1; i <= n; ++i) {
    while (abs(dp[t] - num[i]) >= abs(dp[t + 1] - num[i]) && t < m)
      ++t;
    r = max(r, abs(dp[t] - num[i]));
  }
  printf("%d\n", r);
}
