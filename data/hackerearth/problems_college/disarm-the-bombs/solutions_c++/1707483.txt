#include <iostream>
using namespace std;
int min(int a,int b)
{
	return a<b ? a :b;
}
int mindest(int n,int E,int T,int e[],int t[],int d[])
{
	if(n==0) return 0;
	if(t[n-1]>T || e[n-1]>E)
	  return d[n-1]+mindest(n-1,E,T,e,t,d);
	return min(d[n-1]+mindest(n-1,E,T,e,t,d),mindest(n-1,E-e[n-1],T-t[n-1],e,t,d));
}
int main()
{
    int N,E,T,e[101],t[101],d[101],i;
    cin>>N>>E>>T;
    for(i=0;i<N;i++)
     cin>>e[i];
    for(i=0;i<N;i++)
     cin>>t[i];
     
     for(i=0;i<N;i++)
     cin>>d[i];
     
    cout<<mindest(N,E,T,e,t,d)<<endl;
    return 0;
}
