midpts = []

for index in range(3):
	x1,y1 = map(float,raw_input().split())
	midpts.append([x1,y1])
	
#print midpts

sumx = midpts[0][0] + midpts[1][0] + midpts[2][0]
sumy = midpts[0][1] + midpts[1][1] + midpts[2][1]

pts = []
pts.append([sumx-2*midpts[0][0],sumy-2*midpts[0][1]])
pts.append([sumx-2*midpts[1][0],sumy-2*midpts[1][1]])
pts.append([sumx-2*midpts[2][0],sumy-2*midpts[2][1]])

#print pts

pts.sort(key = lambda x: [x[0],x[1]])

for pt in pts:
	print "%0.4f %0.4f" % (pt[0], pt[1])
