#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,a,b,c;
		cin>>n>>a>>c>>b;
		long long sum=0;
		for(int i=0;i<=min(a,n);i++){
		for(int j=0;j<=min(n-i,b);j++){
			for(int k=0;k<=min(n-(i+j),c);k++){
				sum++;
			}
		 }
		}
		cout<<sum<<endl;
	}
}
