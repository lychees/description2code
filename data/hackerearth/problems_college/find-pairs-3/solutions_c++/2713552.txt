#include <iostream>
using namespace std;

int main()
{
   int t,n,a[1000],i,j,count;
   cin>>t;
   while(t-->0)
    {
    	count=0;
    	cin>>n;
    	for(i=0;i<n;i++)
    	   cin>>a[i];
    	for(i=0;i<n;i++)
    	  {
    	  	count++;
    	  	for(j=i+1;j<n;j++)
    	  	   if(a[i]==a[j])
    	  	      count++;
    	  }
        cout<<count<<endl;	  
    }
}
