#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll gcd(ll a, ll b){
	if(a==0)
		return b;
	return gcd(b%a,a);
}
#define s 1000010
int arr[s];
void prime(){
	int i,j;
	for(i=2;i*i<=s;i++){
		for(j=2;j*i<s;j++)
			arr[i*j]=1;
	}

	// 		cout<<i<<" ";
	// cout<<endl;
}
int main(){
	ll n,t,i,a,b;
	memset(arr,0,s);

		ll g=gcd(a,b);
		//cout<<g<<endl;
		ll l=(a*b)/g;
		ll flag=0,pre=-1,count=0;
		//cout<<"HCM "<<l<<endl;
		a=2;
		while(l!=1){
			if(l%a == 0){
				l=l/a;
				if(pre==-1)
				{
					pre=a;
					count=1;
