#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    
    while (t--) {
    	string a, b;
    	cin >> a;
    	cin >> b;
    	
    	for (int i = 0; i < a.length(); i++) {
    		putchar(a[i] == b[i] ? '0' : '1');
    	}
    	
    	putchar('\n');
    }
    
    return 0;
}
