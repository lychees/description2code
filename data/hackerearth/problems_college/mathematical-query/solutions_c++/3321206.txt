#include <bits/stdc++.h>
using namespace std;
#define m 10000003
int main()
{
	
    long long t,i,n;
   
    cin>>t;
    while(t--)
    {
    	vector<long long>a;
    	cin>>n;
    	a.push_back(0);
    	a.push_back(1);
    	a.push_back(1);
    	for(i=3;i<=n;i++)
    	a.push_back((a[i-2]+a[i-3]+2*i)%m);
    	
    	cout<<a[n]<<endl;
    }
    return 0;
}
