#include <bits/stdc++.h>
using namespace std;

int a[5010][5010];

bool check(char x) {
	if (x == 'a' || x == 'e' || x == 'i' || x == 'o' || x == 'u')
		return true;
	return false;
}

int main()
{
	string pat, text;
	cin >> text;
	cin >> pat;
	
	int i,j;
	
	for (i = 1; i <= text.size(); i++) {
		for (j = 1; j <= pat.size(); j++) {
			if (text[i-1] == pat[j-1] && !check(text[i-1])) {
				a[i][j] = a[i-1][j-1] + 1;
			}
			a[i][j] = max(a[i][j], max(a[i-1][j], a[i][j-1]));
		}
	}
	
	cout <<a[text.size()][pat.size()]<<endl;
	return 0;
}
