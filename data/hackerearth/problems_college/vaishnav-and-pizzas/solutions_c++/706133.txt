#include<stdio.h>
#define max 10000
int a[max]={0};
int s[max]={0};
int fi(int n)
     { long  i;
       int result = n;
       for(i=2;i*i <= n;i++)
       {
         if (n % i == 0) result -= result / i;
         while (n % i == 0) n /= i;
       }
       if (n > 1) result -= result / n;
       return result;
     }


void calc()
{
    int sum=0;
    s[1]=0;
    for(int i=1;i<=max;i++)
    {
        a[i]=fi(i);

        if(i>1)
        {
            sum+=a[i];
            s[i]=sum;
        }
    }
}
int main()
{
