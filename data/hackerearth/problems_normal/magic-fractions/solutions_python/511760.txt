def isprime(n):
    a=int(pow(n,0.5))
    flag=0
    for i in range(2,a+1):
        if n%i==0:
            flag=1
            return 0
    return 1
arr=[0]*510

for i in range(3,505):
    if isprime(i):
        cnt=cnt*2
