#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int c=0;
    for(int i=0;i<s.size();i++)
    {
    	if(s[i]=='8')
    	c+=7;
    	else if(s[i]=='0'||s[i]=='6'||s[i]=='9')
    	c+=6;
    	else if(s[i]=='2'||s[i]=='3'||s[i]=='5')
    	c+=5;
    	else if(s[i]=='4')
    	c+=4;
    	else if(s[i]=='7')
    	c+=3;
    	else
    	c+=2;
    	
    }
    cout<<c<<endl;
    
    return 0;
}
