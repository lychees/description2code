int rev(int x)
{
    int rr = 0;
    while(x!=0)
    {
        rr = rr*10+x%10;
        x/=10;
    }
    return rr;
}
int dig(int x)
{
    int rr = 0;
    while(x!=0)
    {
        rr++;
        x/=10;
    }
    return rr;
}
