#include <iostream>
using namespace std;

int getDivisorsCount(long long num){
	int count=2;
	int i=2;
	while(i*i<num){
		if(num%i==0)
			count+=2;
		i++;
	}
	if(i*i==num)
		count++;
	return count;
}


int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	
    	long long num,i=2; 
    	while(1){
    		num=(i*(i+1))/2;
    		int count=getDivisorsCount(num);
    		if(count>n)
    			break;
    		i++;
    	}
    	cout<<num<<endl;
    }
    return 0;
}
