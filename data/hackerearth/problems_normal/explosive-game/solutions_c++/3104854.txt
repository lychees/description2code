#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
	cout.sync_with_stdio(false);
	cin.tie(0);
	int n, q;
	cin >> n >> q;
	long long s = 0;
	long long s2 = 0;
	vector<int> sd(n);
	for (int i = 0; i < n; ++i) {
		int a, b;
		cin >> a >> b;
		if (a > b) 
			swap(a, b);
		s += a;
		s2 += (b - a);
		sd[i] = (b - a);
	}
	sort(sd.begin(), sd.end());
	bool ok = [&]()->bool {
		for (int i = 0; i < n - 1; i += 2)
			if (sd[i] != sd[i + 1])
				return false;
		return true;
	}();
	s += s2 / 2;
	for (int i = 0; i < q; ++i) {
		long long x;
		cin >> x;
		cout << (ok && x == s ? "YES\n" : "NO\n");
	}
	return 0;
}
