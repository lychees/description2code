#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test; cin>>test;
	while(test--)
	{
		string s; cin>>s;
		int ans ;
		int c=0;
		for(int i =0,j=s.size()-1;i<=j;i++,j--)
		{
			if(s[i]!=s[j])
			{
				if(s[i]==s[j-1])ans=j;
				else ans =i;
				c=1;
				break;
			}
		}
		if(c)cout<<ans<<endl;
		else cout<<-1<<endl;
	}
}
