#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	int t,n,a[105],i,j;
	cin >> t;
	while(t--){
		cin >> n;
		for(i=0; i<n; i++)
		 cin >> a[i];
		 int count = 0;
		 for(i=0; i<n; i++)
		  for(j=i+1; j<n; j++)
		   if((a[i]+a[j])%3==0)
		     count++;
		     
		    cout << count << endl; 
	}
	return 0;
}
