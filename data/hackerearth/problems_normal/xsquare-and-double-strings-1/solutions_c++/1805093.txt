#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string s;
    	cin>>s;
    	int dp[30]={0};
    	for(int i=0;i<s.size();i++) {
    		dp[s[i]-'a']++;
    	}
    	int f=0;
    	for(int i=0;i<30;i++) {
    		if(dp[i]>=2) {
    			f=1;
    			break;
    		}
    	}
    	cout<<(f==1?"Yes\n":"No\n");
    }
    return 0; 
}
