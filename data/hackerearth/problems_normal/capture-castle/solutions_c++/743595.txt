#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		int a,b;
   		cin>>a>>b;
   		int p[a][b];
   		int f[a][b];
   		for(int i=0;i<a;i++)
   		{
   			for(int j=0;j<b;j++)
   			{
   				cin>>p[i][j];
   				f[i][j] = 9999;
   			}
   		}
   		int x,y,t;
   		cin>>x>>y>>t;
   		x--;
   		y--;
   		f[0][0] = p[0][0];
   		queue< pair< int,int > >q;
   		int x1[] = {0,0,1,-1};
	    int y1[] = {-1,1,0,0};
   		q.push(make_pair(0,0));
   		while(!q.empty())
   		{
   			int lx = q.front().first;
   			int ly = q.front().second;
   			q.pop();
   			for(int i=0;i<4;i++)
   			{
   				int tx = lx + x1[i];
   				int ty = ly + y1[i];
   				if(tx>=0 && tx<a && ty>=0 && ty<b)
   				{
   					if(f[tx][ty]>f[lx][ly]+p[tx][ty])
   					{
   						f[tx][ty]=p[tx][ty]+f[lx][ly];
   						q.push(make_pair(tx,ty));
   					}
   				}
   			}
   		}
   		if(f[x][y]<t)
   		{
   			cout<<"YES"<<endl;
   			cout<<t-f[x][y]<<endl;
   		}
   		else
			cout<<"NO"<<endl;
   	}
    return 0;
}
