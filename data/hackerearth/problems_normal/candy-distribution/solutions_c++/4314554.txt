typedef long long ll;
ll MOD=1000000007;
ll pw(ll a,ll b){
  ll r=1;
  while(b>0){
     if(b&1){
       r=(r*a)%MOD;
        }
     b=b>>1;
     a=(a*a)%MOD;
  }
return r;
}
map<ll,ll>cache;
ll fun(ll n){
  if(n==1||n==2)return 1;

  if(cache[n])return cache[n];
    n--;
    ll sum=1;
  for(ll i=2;i*i<=n;i++){
    if((n%i)==0){
       ll b=n/i;
        sum=(sum+ pw(fun(i),b)%MOD )%MOD;
       if(b!=i)
        sum=(sum+ pw(fun(b),i)%MOD )%MOD;
    }
  }
 return  cache[n+1]=sum;
}
