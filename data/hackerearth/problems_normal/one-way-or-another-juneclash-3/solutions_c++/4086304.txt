int main(){

    int t;
    cin>>t;

    while(t--){

      int n;
      cin>>n;

      ll y[10005],x[10005],val[10005];

      for(int i=0;i<n;++i){

        cin>>y[i]>>x[i]>>val[i];
      }

      // Considering odd se start

      bool flag1=true;

      for(int i=0;i<n;++i){

         if((y[i]-1+x[i]-1)%2==0){

             if(val[i]%2==0)
               flag1=false;

         }

         else{

            if(val[i]%2==1)
               flag1=false;


         }


      }

      // Considering even se start

      bool flag2=true;

      for(int i=0;i<n;++i){

         if((y[i]-1+x[i]-1)%2==0){

             if(val[i]%2==1)
                flag2=false;

         }

         else{

            if(val[i]%2==0)
               flag2=false;


         }


      }

      if(flag1==true || flag2==true)
         cout<<"Yes\n";
      else
         cout<<"No\n";



    }
