#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long int n1,n2;
    while(t--)
    {
    	scanf("%ld %ld",&n1,&n2);
    	int ans = 0,c = 0;
    	while(n1>0||n2>0)
    	{
    		c+=(n1%10) + (n2%10);
    		
    		if(c>=10)
    		ans++;
    		
    		n1/=10;
    		n2/=10;
    		c/=10;
    	}
    	if(ans==0)
    	printf("No carry operation\n");
    	else if(ans==1)
    	printf("1 carry operation\n");
    	else
    	printf("%d carry operations\n",ans);
    }
    return 0;
}
