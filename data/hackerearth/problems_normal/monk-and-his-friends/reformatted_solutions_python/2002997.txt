for tc in range(input()):
    N, M = map(int, raw_input().split())
    A = list(map(int, raw_input().split()))
    candy = set()
    for i in range(N):
        candy.add(A[i])

    for i in range(N, N+M):
        if A[i] in candy:
            print "YES"
        else:
            print "NO"
            candy.add(A[i])
