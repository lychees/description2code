import sys
t = input()
for _ in range(t):
    n = input()
    arr = [int(e) for e in raw_input().split()]
    s = sum(arr)
    dp = [0]*(s+1)
    dp[0] = 1
    for i in xrange(n):
        for j in xrange(s, arr[i]-1, -1):
            dp[j] = dp[j] | dp[j-arr[i]]
    print sum(dp)
