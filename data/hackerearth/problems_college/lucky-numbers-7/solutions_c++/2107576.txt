#include <iostream>
using namespace std;
 #include <stdio.h>
 int f(int n){
 	int t=0;
 	while(n>0){
 		if(n%10==4||n%10==7)
 			t++;
 		n=n/10;
 	}
     return t;
 }
int main()
{
   	int a,l;
   	scanf("%d%d\n",&a,&l);
   	int i,c=0,j;
   	for(i=a+1;;i++){
   		for(j=0;j<l;j++){
   		//	printf("%d %d\n",f(a+j),f(i+j));
   			if(f(a+j)!=f(i+j))
   				break;
   		}
   		if(j==l)
   		{
   			c=i;
   			break;
   		}
   	}
   	printf("%d\n",c);
    return 0;
}
