#!/usr/bin/env python

import math


def isPrime(n):

    n_root = int(math.sqrt(n))

    if (n == 1):
        return False
    if (n == 2 or n == 3):
        return True
    for i in range(2, n_root+1):
        if n % i == 0:
            return False

    return True


tc = input()
for i in range(tc):
    n = input()
    if n == 3:
        print "3"
        continue
    if isPrime(n):
        print "2"
    else:
        print "1"
