#include <iostream>
#include<limits.h>
using namespace std;

int main()
{
    int n,i;
    cin>>n;
    int a[n],b[n],c[n];
    
    for(i=0;i<n;i++)
    {
    	cin>>c[i]>>b[i];
    }
   	for(i=0;i<n;i++)
   	{
   		a[i]=c[i]+b[i];
   	}
   int count=0;
  int pos;
   while(count<n)
   {
   	 int min=INT_MAX;
   	for(i=0;i<n;i++)
   	{
   		if(a[i]<min)
   		{
   			min=a[i];
   			pos=i;
   		}	
   	}
   cout<<pos+1<<" ";
   count++;
   a[pos]=INT_MAX;
   }
    return 0;
}
