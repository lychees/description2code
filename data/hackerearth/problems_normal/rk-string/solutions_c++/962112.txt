#include <iostream>
#include<algorithm>
#include<cstdio>
#include<string>
using namespace std;

#define LL long long

int a[1000001],b[1000001],c[1000001],d[1000001];

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	
    	int m,n;
    	scanf("%d%d",&m,&n);
    	string str;
    	cin >> str;
    	for(int i=0;i<=str.length();i++)
    		a[i]=b[i]=c[i]=d[i]=0;
   		c[0]=d[0]=1;
   		for(int i=1;i<=str.length();i++){
   			if(str[i-1]=='R')
   				a[i]=a[i-1]+1;
   			else a[i]=a[i-1];
   			if(str[i-1]=='K')
   				b[i]=b[i-1]+1;
   			else b[i]=b[i-1];
   			c[a[i]]++;
   			d[b[i]]++;
   		}
   		if(m){
   			LL result=0;
   			for(int i=1;i<=str.length();i++){
   				if(a[i]-m>=0)
   					result+=c[a[i]-m];
   			}
   			printf("%lld ",result);
   		}
   		else{
   			LL result=0,temp=0;
   			for(int i=1;i<=str.length();i++){
   				if(b[i]>b[i-1])
   					temp++;
   				else {
   					result+=temp*(temp+1)/2;
   					temp=0;
   				}
   			}
   			if(temp)
   				result+=temp*(temp+1)/2;
   			printf("%lld ",result);
   		}
		if(n){
			LL result = 0;
			for(int i=1;i<=str.length();i++){
				if(b[i]-n>=0)
					result+=d[b[i]-n];
			}
			printf("%lld\n",result);
		}
		else{
			LL result=0,temp=0;
			for(int i=1;i<=str.length();i++){
				if(a[i]>a[i-1])
					temp++;
				else{
					result+=temp*(temp+1)/2;
					temp=0;
				}
			}
			if(temp)
				result+=temp*(temp+1)/2;
			printf("%lld\n",result);
		}
    	
    }
    return 0;
}
