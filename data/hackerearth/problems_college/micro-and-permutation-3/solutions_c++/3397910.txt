#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll comp(string a, string b)
{
string s1=a.append(b);
string s2=b.append(a);
if(s1.compare(s2)>0)
return 1;
else
return 0;
}
void maximum(string a[], ll n)
{
sort(a,a+n,comp);
for(ll i=0;i<n;i++)
cout<<a[i];
}
int main()
{
ll n, i, x;
cin>>n;
string a[n];
for(i=0;i<n;i++)
{
cin>>x;
string st="";
while(x!=0)
{
st=((char)(x%10+48))+st;
x/=10;
}
a[i]=st;
st="";
}
maximum(a,n);
return 0;
}
