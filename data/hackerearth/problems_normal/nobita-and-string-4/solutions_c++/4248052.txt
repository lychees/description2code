#include <iostream>
#include <cstring>
using namespace std;
#define NUM 100002

int main() {
    int t=0,i,j=0;
    cin>>t;
    t++;
    while(t--)
    {
        char a[NUM];
        if(cin.getline(a,NUM))
        {
            int l=strlen(a);
            for(i=l-1,j=i+1;i>0;i--)
            {
                if(a[i]==' ')
                {
                    for(int k=i+1;k<j;k++)
                        cout<<a[k];
                    cout<<" ";
                    j=i;
                }
            }
            for(i=0;i<j;i++)
                cout<<a[i];
            cout<<endl;   
        }
    }
    return 0;
}
