// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%d", &x)
#define sd(x) scanf("%lf", &x)
#define sl(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define mod 1000000007

int main()
{
    ll n, i;
    sl(n);
    ll a[n+1];
    ll dp[n+1];
    for (i = 1; i <= n; i++)
        sl(a[i]);
    dp[0] = INT_MIN;
    ll mx = 0;
    for (i = 1; i <= n; i++) {
        dp[i] = max(a[i], dp[i-1]+a[i]);
        mx = max(mx, dp[i]);
    }
    printf("%lld\n", mx);
    return 0;
}
