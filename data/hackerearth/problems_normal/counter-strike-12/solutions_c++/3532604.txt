#include <bits/stdc++.h>

using namespace std;

int x11 [1000 + 10];
int x22 [1000 + 10];
int y11 [1000 + 10];
int y22 [1000 + 10];

int main()
{
    ios::sync_with_stdio(false);cin.tie(0);         // decrease time of cin, cout

    int n , m , t , d;
    string s , c;

    cin >>      t       ;

    while(t--){

        cin >> n >> m >> d;

        int ans = 0;

        for(int i = 0; i < n; i++){

            cin >> x11[i] >> y11[i];
        }

        for(int i = 0; i < m; i++){

            cin >> x22[i] >> y22[i];
        }

        for(int i = 0; i < m; i++){

            for(int j = 0; j < n; j++){

               int dist = abs(x11[j] - x22[i]) + abs(y11[j] - y22[i]);          // calculating the Manhattan distance 
               if(dist <= d){

                    ans++;                                                      // increase number of reachable targets
                    break;
               }
            }
        }

        if(ans > m / 2 + 1) cout << "YES\n";    
        else cout << "NO\n";
    }

    return 0;
}
