#include <iostream>
using namespace std;

#define p 1000000007

long long modexp(long long x, long long y){
	long long ans = 1;
	while(y){
		if(y&1){
			ans = (ans*x)%p;
		}
		x = (x*x)%p;
		y >>= 1;
	}
	return ans;
}

int main()
{
    long long t,n;
    cin >> t;
    while(t--){
    	cin >> n;
    	cout << (2*modexp(modexp(2,n),p-2))%p << endl;
    }
    return 0;
}
