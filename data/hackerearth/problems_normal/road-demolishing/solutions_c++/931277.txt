LANG: C++
Direct using formula given in http://en.wikipedia.org/wiki/Tur%C3%A1n's_theorem won't give correct ans which Wolfram alpha also uses.
For example consider this case. n=12,q=9
Formula will give ans as 63
But best you can do is 62 by diving nodes into 8 parts as following
4 parts with 1 nodes
4 parts with 2 nodes
Now, make 8-partite graph on these parts. We will get 62 as ans which when subtracted from max no. of edges yield 4 as ans.
Thanks to azukun for bringing the flaw into notice.
We will use the left side of the inequality given in http://en.wikipedia.org/wiki/Tur%C3%A1n_graph to exactly find the number of edges.
*/
#include<bits/stdc++.h>
using namespace std;
#define ll	long long int
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	assert(t>=1 && t<=100000);
	while(t--)
	{
		ll n,q;
		cin>>n>>q;
		assert(n>=3 && n<=1000000);
		assert(q>=3 && q<=n);
		q--;
		ll ans=n*n-(n%q)*((n+q-1)/q)*((n+q-1)/q);
		ans=ans-(q-n%q)*(n/q)*(n/q);
		ans/=2;
		ans=(n*(n-1))/2-ans;
		assert(ans>0);
		cout<<ans<<endl;
	}
	return 0;
}
