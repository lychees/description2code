int main()
{
    string str;
    int t , len , one , two ;
    cin >> t ;
    while(t--)
    {
        one = 0 , two = 0;
        cin >> str;
        len = str.size();
        for(int i = 0 ; i < len ; ++i)
        {
        	if(str[i] == '1') ++one;
        	else ++two;
        }
        cout << len - (min(one , two) << 1) << endl;
    }
    return 0;
}
