#include<bits/stdc++.h>
using namespace std;
struct database
{
    string name;
    int cnt;
};
struct database arr[10000];
int push(string str,int count)
{
    if(count==0)
    {
        arr[0].name=str;
        arr[0].cnt=1;
        return 1;
    }
    for(int i=0;i<count;i++)
    {
        if(arr[i].name==str)
        {
            arr[i].cnt+=1;
            return count;
        }
    }
    arr[count].name=str;
    arr[count++].cnt=1;
    return count;
}
struct myfunc
{
    bool operator()(struct database a,struct database b)
    {
        return (a.name<b.name);
    }
};
int main()
{
    string str;
    int test,num;
    int count;
    //scanf("%d",&test);
        count=0;
        scanf("%d",&num);
        for(int i=0;i<num;i++)
        {
            cin>>str;
            count=push(str,count);
        }
        sort(arr,arr+count,myfunc());
        //printf("%d\n",count);
        for(int i=0;i<count;i++)
        {
            cout<<arr[i].name<<" "<<arr[i].cnt<<endl;
        }
}
