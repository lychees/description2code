#include<iostream>
#include<cstring>
using namespace std;
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string str;
    int s = 0;
    cin>>str;
for(int i = 0 ; i < str.length() ; i++){
        if(str[i] == '0'){
            s += 6;
        }
    else if(str[i] == '1'){
        s += 2;
    }else if(str[i]=='2'){
     s += 5;
    }else if(str[i] == '3'){
      s += 5;
    }else if(str[i] == '4'){
     s += 4;
    }
    else if(str[i] == '5'){
     s += 5;
    }else if(str[i] == '6'){
     s += 6;
    }else if(str[i] == '7'){
      s += 3;
    }else if(str[i] == '8'){
     s += 7;
    }else if(str[i] == '9'){
      s += 6;
    }
}cout<<s<<endl;

return 0;
}
