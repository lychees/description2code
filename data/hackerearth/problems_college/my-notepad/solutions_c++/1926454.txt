#include <bits/stdc++.h>

using namespace std;

#define pb push_back

int main()
{
    int t, n, i, j, tmp, top, x;
    string s;
    cin >> t;
    while (t--) {
        cin >> n;
        vector < pair <int, pair <char, string> > > v(n);
        for (i = 0; i < n; i++) {
            cin >> v[i].first >> s >> v[i].second.second;
            v[i].second.first = s[0];
        }
        sort(v.begin(), v.end());
        char a[n];
        top = 0;
        vector <string> ans;
        ans.pb("");
        for (i = 0; i < n; i++) {
            if (v[i].second.first == 't') {
                ans.pb(ans[ans.size()-1]);
                ans[ans.size()-1].pb(v[i].second.second[0]);
            }
            else {
                x = v[i].first-atoi(v[i].second.second.c_str());
               // cout << x << endl;
                top = ans.size()-1;
                for (j = i-1; j >= 0 && v[j].first >= x; j--)
                    top--;
                ans.pb(ans[top]);
            }
        }
        cout << ans[ans.size()-1] << endl;
    }
    return 0;
}
