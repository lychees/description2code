#include <bits/stdc++.h>
using namespace std;

int n, value;
int bit[1000100], arr[1000100], arr2[1000100];

void update(int x)
{
	for(; x <= n; x += x & -x)
		bit[x]++;
}

int query(int x)
{
	int sum = 0;
	for(; x; x -= x & -x)
		sum += bit[x];
	return sum;
}


int main()
{
    scanf("%d", &n);
    for(int i = 0; i < n; i++)
    {
    	scanf("%d", arr + i);
    	arr2[i] = arr[i];
    }
    
    unordered_map<int, int> mp;
    value = 1;
    sort(arr2, arr2 + n);
    for(int i = 0; i < n; i++)
    	if(!mp[arr2[i]])
    		mp[arr2[i]] = value++;
    
    for(int i = 0; i < n; i++)
    	arr[i] = mp[arr[i]];
    
    
    for(int i = 0; i < n; i++)
    {
    	int ans = query(arr[i] - 1);
    	if(i)
    		printf(" ");
    	printf("%d", ans);
    	update(arr[i]);
    }
    return 0;
}
