bool prime(int n)
{
	int x;
	
	for(x=2;x<=sqrt(n);x++)
	if(n%x == 0) return 0;
	
	return 1;
	
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    int v[1000000]={0};
    
    int t;
    cin>>t;
    
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    
    	
    	mt(v,0);
    	
    	int x,c=0,k=0;
    	
    		if(prime(n)) {cout<<2<<"^"<<0<<"\n"; continue; }
    	
    	for(x=2;x*x<=n;x++)
    	{
    		if(n%x==0)
    		{
    			while(n%x==0)
    			{
    				v[x]++;
    				n/=x;
    			}
    			
    			c++;
    		}
    	}
    	
    	if(n>1) { v[n]++; c++; }
    	
    	//cout<<c<<" ";
    	
    	
    	
    	if(v[2]>0 && k<c-1){ cout<<2<<"^"<<v[2]<<"*"; k++; }
    	else if(v[2]>0 && k==c-1) { cout<<2<<"^"<<v[2]<<"\n"; continue;}
    	else if(!v[2]) cout<<2<<"^"<<v[2]<<"*";
    	
    	
    	
    	for(x=3;x<=999999;x++)
    	{
    	//	cout<<k<<" ";
    		if(v[x]>0 && k<c-1) { cout<<x<<"^"<<v[x]<<"*"; k++; }
    		else if(v[x]>0 && k==c-1) {cout<<x<<"^"<<v[x]; break;}
    	}
    	
    	cout<<"\n";
    	
    }
    
    return 0;
    
}
