#include <bits/stdc++.h>
using namespace std;
char check(int x)
{
	switch(x)
	{
		case 10: return 'U';
		case 11: return 'V';
		case 12: return 'W';
		case 13: return 'X';
		case 14: return 'Y';
		case 15: return 'Z';
		default: return x+48;
	}
}
int main()
{
    int t,r,g,b,rem,i,j;
    char a[2];
    memset(a,48,sizeof(a));
    cin>>t;
    for(i=1;i<=t;i++)
    {
    	cin>>r>>g>>b;
    	j=1;
    	cout<<"Case "<<i<<": ";
    	while(r)
    	{
    		rem=r%16;
    		a[j]=check(rem);
    		r/=16;
    		j--;
    	}
    	cout<<a[0]<<a[1]<<" ";
    	memset(a,48,sizeof(a));
    	j=1;
    	while(g)
    	{
    		rem=g%16;
    		a[j]=check(rem);
    		g/=16;
    		j--;
    	}
    	cout<<a[0]<<a[1]<<" ";
    	memset(a,48,sizeof(a));
    	j=1;
    	while(b)
    	{
    		rem=b%16;
    		a[j]=check(rem);
    		b/=16;
    		j--;
    	}
    	cout<<a[0]<<a[1]<<endl;
    	memset(a,48,sizeof(a));
    }
    return 0;
}
