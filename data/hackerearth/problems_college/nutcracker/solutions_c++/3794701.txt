inline ll powr (ll a, ll  b)
{
    if (b == 0)
        return 1;
    long long int x = powr(a, b/2);

        return (x*x)%MOD;
    else
        return (((x*x)%MOD)*a)%MOD;
}
 
inline ll inv(ll x) {

{
