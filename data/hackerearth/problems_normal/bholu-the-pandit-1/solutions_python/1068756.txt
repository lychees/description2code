
def fact(n):
    d={}
    for i in xrange(1,int(n**0.5)+1):
         if n%i==0:
             d[i]=1
             d[n/i]=1
    return d

t=input()



while t>0:
    a,b,c=map(int,raw_input().split())
    d1={}
    d2={}
    d3={}
    d1=fact(a)
    d2=fact(b)
    d3=fact(c)
    l=[]
    for i in d1:
        if i in d2 and i in d3:
            l.append(i)
    mx=max(l)
    print mx,((a*b*c)/(mx**3))%1000000007
    t=t-1
