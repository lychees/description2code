#include <iostream>
#include <cmath>
using namespace std;

int main()
{
   long long tc, N;
   cin >> tc ;
   while(tc--) { 
   		cin >> N;
   		int cnt = 0;
   		long long val = 1; 
   		while(true) {
   			val *= 2;
   			if(val <= N+1)
   			cnt++;
   			else break;
   		}
   		cout << cnt - 1<< endl;
   }
    return 0;
}
