#include<bits/stdc++.h>
using namespace std;
#define long long long
const long M = 1000000007;
map<long, long> F;
 
long f(long n) {
	if (F.count(n)) return F[n];
	long k=n/2;
	if (n%2==0) {
		return F[n] = (f(k)*f(k) + f(k-1)*f(k-1)) % M;
	} else { 
		return F[n] = (f(k)*f(k+1) + f(k-1)*f(k)) % M;
	}
}
 
int main(){
	int t;
	cin>>t;
	while(t--){
	long n,k;
	F[0]=F[1]=1;
	cin>>n;
	k=(n==0 ? 0 : f(n-1));
	cout<<k<<endl;
	}
	return 0;
}
