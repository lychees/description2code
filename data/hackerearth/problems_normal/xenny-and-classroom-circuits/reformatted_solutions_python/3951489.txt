T = int(raw_input())
for i in range(T):
    N = int(raw_input())
    line = str(raw_input())
    bg_boxes = 0
    gb_boxes = 0
    for j in range(N):
        if (line[2*j: 2*j+2] == "BG"):
            bg_boxes += 1
        elif (line[2*j: 2*j+2] == "GB"):
            gb_boxes += 1

    result = min(bg_boxes, gb_boxes) + (N-bg_boxes - gb_boxes)/2
    print(result)
