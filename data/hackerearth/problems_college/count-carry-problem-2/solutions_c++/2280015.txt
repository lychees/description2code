#include <iostream>
using namespace std;

int main()
{
    long int x,y,n,count,carry;
    cin>>n;
    while(n--)
    {
    	cin>>x>>y;
    	carry=0;count=0;
    	while(x&&y)
    	{
    		carry+=x%10+y%10;
    		carry/=10;
    		x/=10;
    		y/=10;
    		if(carry)
    		count++;
    	}
    	if(x)
    	{
    		while(x)
    		{
    			carry+=x%10;
    			carry/=10;
    			x/=10;
    			if(carry)
    			count++;
    		}
    	}
    	else if(y)
    	{
    		while(y)
    		{
    			carry+=y%10;
    			carry/=10;
    			y/=10;
    			if(carry)
    			count++;
    		}
    	}
    	
    	if(count>1)
    	cout<<count<<" "<<"carry operations"<<endl;
    	else if(count==1)
    	cout<<count<<" "<<"carry operation"<<endl;
    	else
    	cout<<"No carry operation"<<endl;
    }
    return 0;
}
