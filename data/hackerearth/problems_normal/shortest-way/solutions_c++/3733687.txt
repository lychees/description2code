#include<bits/stdc++.h>
#define ll long long 
using namespace std;
int main()
{
	int t;
	ll a,b=1e18;
	cin>>t;
	while(t--)
	{
		 ll  result =0;
		cin>>a>>b;
		while(a>1||b>1)
		{
			if(a>b)
			{
				result= result+a/b;
				a=a%b;
			}
			else
			{
				result=result+b/a;
				b=b%a;
				
			}
		}
		if(result>0) --result;
		cout<<result<<endl;
	}
	return 0;
	
}
