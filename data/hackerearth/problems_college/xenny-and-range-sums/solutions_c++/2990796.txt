#include<stdio.h>
int main(){
	long long int n, q, lr, rl, i, j, l, r; scanf("%lld%lld", &n, &q); int a[n + 1];
	for(i = 1; i <= n; i++)	scanf("%d", &a[i]);
	while(q-- && scanf("%lld%lld", &l, &r)){
		lr = 0, rl = a[r] * 2;
		for(i = l, j = r; i <= r, j >= l; i++, j--)	lr += a[i], rl -= a[j];
		printf("%lld %lld\n", lr, rl);
	}return 0;}
