#include <bits/stdc++.h>
using namespace std;

#define ll long long int

int main()
{
    int t;
    cin >> t;
    while(t--){
        ll n, m, mod;
        cin >> n >> m;
        mod = (ll) pow(10, m);
        if(n > 4 || m == 0)
            cout << "0" << endl;
        else if(n == 0 || n == 1)
            cout << "1" << endl;
        else if(n == 2)
            cout << "2" << endl;
        else if(n == 3)
            cout << 720 % mod << endl;
        else if(n == 4){     //  (4!)! = 620448401733239439360000%mod = 8401733239439360000
            if(m == 19)
                cout << 8401733239439360000LL << endl;
            else
                cout << 8401733239439360000LL % mod << endl;
        }
        else
            cout << "0" << endl;
    }
    return 0;
}
