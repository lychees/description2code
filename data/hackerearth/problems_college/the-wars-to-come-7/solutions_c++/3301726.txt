#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	cin>>N;
	vector<int>v(N);
	for(int i=0;i<N;i++)
		cin>>v[i];
	sort(v.rbegin(),v.rend());
	int num=0,sum=0,min;
	cin>>min;
	for(int i=0;i<N;i++){
		if(sum>=min)
			break;
		else{
			sum+=v[i];
			num++;
		}
	}
	cout<<num<<endl;
	return 0;
}
