#include <iostream>
#include <bits/stdc++.h>
using namespace std;
//int hashm(int a[][]);
map<int,pair<int,int>> hm;

int main()
{
	int n,m,a[1000][1000];
//	memset(hasht,0,sizeof(hmap));
	cin>>n;
	cin>>m;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			cin>>a[i][j];
			hm[a[i][j]]=make_pair(i,j);
		}
	}
	int q;
	cin>>q;
	while(q--)
	{
		int x;
		cin>>x;
		if(hm[x].first==0)
		{
		cout<<"-1"<<" "<<"-1"<<endl;
		continue;
		}
		cout<<hm[x].first-1<<" "<<hm[x].second-1<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
