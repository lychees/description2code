#include <bits/stdc++.h>

#define ll long long int
#define SZ 1234567
#define MOD 1000000007LL

using namespace std;

ll int_mod(ll a, ll b) {		return ( ( (a%b) + b) % b);		}


ll modulo(ll a, ll b , ll c)
{
    ll x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
             x=(x*y)%c;
        }
        y = (y*y)%c;
        b /= 2;
    }
    return x%c;
}

int main()
{
	ios::sync_with_stdio(false);
	
	ll t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		
		ll ans = (modulo(3LL, n, MOD)%MOD + (MOD + 3 - 3*modulo(2LL,n,MOD)%MOD))%MOD;
		cout<<ans%MOD<<endl;
	}
	
	return 0;
}
