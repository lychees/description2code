

    #include <iostream>
    #include <cstdio>
    #include <cstring>
    #include <algorithm>
    #include <string>
    #include <fstream>
    using namespace std;
    int n;
    
    int main()
    {
        scanf("%d",&n) ;
        
        int ans = 0 ; 
        
        for(int i = 0 ; i < n ; i++)
        {
        	int temp ;
        	scanf("%d",&temp) ;
        	ans = ans | temp ;
        }
        
        cout << ans ; 
        
        return 0;
    }
