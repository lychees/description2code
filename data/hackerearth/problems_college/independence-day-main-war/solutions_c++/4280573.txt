#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		long double sum1 = 0.0 , sum2 = 0.0;
		for(int i=0;i<n;i++){
			long double x,y;
			scanf("%Lf %Lf",&x,&y);
			sum1+=x,sum2+=y;
		}
		printf("(%.2Lf , %.2Lf)\n",sum1/n,sum2/n);
	}
	return 0;
}
