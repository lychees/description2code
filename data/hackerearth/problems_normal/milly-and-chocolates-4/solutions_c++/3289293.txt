#include <bits/stdc++.h>
using namespace std;
int main() {
	int t,n;
	cin >> t;
	
	int temp,ans;
    double final;
	while(t--) {
		ans = 0;
		cin >> n;
		for(int i = 0;  i < n; i++) {
			cin >> temp;
			ans += temp;
		}
		final = (double)ans/n;
		printf("%0.8lf\n",final);
	}
	return 0;
}
