#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long n;
    	cin>>n;
    	long x=sqrt(n), c=0;
    	if (x*x!=n)x++;
    	for(int i=1; i<x; i++)if(n%i==0){c+=2;if(c>=4)break;}
    	if(c<4) cout<<"NO"<<endl;
    	else cout<<"YES"<<endl;
    }
    return 0;
}
