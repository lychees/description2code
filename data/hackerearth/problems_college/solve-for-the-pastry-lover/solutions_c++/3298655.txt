#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll n,p=1000000001,k,i,j;
        cin>>n;
        string z;
        cin>>z;
        for(i=0;i<z.length();++i)
        {   k=0;
            for(j=i+1;j<z.length() && k==0;++j)
                if(z[i]==z[j])
                    if(j-i<p)
                    {   p=j-i;
                        k=1;
                    }
        }
        if(p==1000000001)
            cout<<"-1\n";
        else
            cout<<p<<"\n";
    }
    return 0;
}
