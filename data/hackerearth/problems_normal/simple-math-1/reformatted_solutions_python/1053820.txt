from sys import stdin
fa = [1]
for x in xrange(1, 205):
    fa.append(x*fa[x-1])
a = stdin.readline().split()
n = int(a[0])
r = int(a[1])
if n < r:
    print -1
elif n == r:
    print 1
else:
    n = n-r
    r = r-1
    n = n+r
    fir = fa[n]
    sec = fa[r] * fa[n-r]
    fir = fir/sec
    sec = 10000007
    print fir % sec
