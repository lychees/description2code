#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
        long double X=0,Y=0;
    	for(int i=0;i<n;i++){
    		long double x,y;
    		cin>>x>>y;
    		X = X + x;
    		Y = Y + y;
    	}
    	X = X/n; Y = Y/n;
    	cout<<fixed<<setprecision(2)<<"("<<X<<" , "<<Y<<")"<<endl;
    }
}
