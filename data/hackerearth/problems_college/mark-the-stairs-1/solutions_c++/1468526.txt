#include <bits/stdc++.h>
 using namespace std;
int main()
{
    int t,a,b;
    scanf("%d",&t);
    while (t--) {
    	scanf("%d%d",&a,&b);
    	if ((a+b)%2 ==0&&a%2!=0) {
    		if (a >= b && (a-b) == 2 || (a-b) == 0) 
    			printf("%d\n",(a+b-1));
    		else
    			printf("No Number\n");
    	}
    	else if ((a+b)%2== 0&&a%2==0) {
    		if (a >= b && (a-b) == 2 || (a-b) == 0)
    			printf("%d\n",(a+b));
    		else
    			printf("No Number\n");
    	}
    	else {
    		printf("No Number\n");
    	}
    }
    return 0;
}
