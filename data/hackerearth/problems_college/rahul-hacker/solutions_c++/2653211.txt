#include <iostream>
using namespace std;

int flip(int num)
{
  int rem;
  int newnum=0;
  while(num!=0)
  {
  	rem=num%10;
  	newnum=newnum*10+rem;
  	num=num/10;
  }
  return newnum;
}
int main()
{
    int n1,n2,t;
    cin>>t;
    while(t--)
    {
    	cin>>n1>>n2;
    	n1=flip(n1);
    	n2=flip(n2);
    	int res=n1+n2;
    	cout<<flip(res)<<"\n";
    }
    
    return 0;
}
