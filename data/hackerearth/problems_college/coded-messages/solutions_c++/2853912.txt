#include <bits/stdc++.h>

using namespace std;
int romanToInt(string s) {
    
    map<char,int> m;

	m['A']=1;
	m['B']=5;
	m['C']=10;
	m['D']=50;
	m['E']=100;
	m['F']=500;
	m['G']=1000; 


	int ans=0;
	for (int i = 0; i < s.size(); ++i)
	{
		if(i>0 && m[s[i]]>m[s[i-1]])
			ans+= (m[s[i]]) - 2*m[s[i-1]];
		else
			ans+=m[s[i]];
	}

	return ans;
}

string intToRoman(int A) {
    vector<int> a = {1000,900,500,400,100,90,50,40,10,9,5,4,1};
    vector<string> s = {"G","EG","F","EF","E","CE","D","CD","C","AC","B","AB","A"};
    string ans;
    for(int i=0;i<a.size();i++)
    {
        while(A>=a[i])
        {
            ans+=s[i];
            A-=a[i];
        }
    }
    return ans;
}



int main(int argc, char const *argv[])
{

	int t;
	cin >> t;
	while(t--)
	{
		string s;
		
		cin >> s;
		int num = romanToInt(s);

		cout << intToRoman(num) << '\n';

	}

	return 0;
}
