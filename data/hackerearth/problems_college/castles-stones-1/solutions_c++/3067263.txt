#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int n,maxi;
    	cin>>maxi;
    	cin>>n;
    	int a[n],i,j,count=0;
    	for(i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    //	for(i=0;i<n;i++)
    //	cout<<a[i]<<" ";
	//	cout<<endl;    	
    	for(i=0,j=n-1;i<=j;j--,i++)
    	{
    	//	cout<<"lol1"<<" "<<i<<" "<<j<<" "<<count<<a[i]<<" "<<a[j]<<endl;
    		if(a[j]>maxi)
    		{
    			i=i-1;
    		}
    		else if(a[i]>maxi)
    		break;
    		
    		else if(a[i]==maxi)
    		{
    			count++;
    			j=j+1;
    		}
    		
    		else if(a[j]==maxi)
    		{
    			count++;
    			i=i-1;
    		}
    		else if(a[i]+a[j]<=maxi)
    		count++;
    		
    		else if(a[j]+a[i]>maxi)
    		{
    			count++;
    			i=i-1;
    		}
    	//	cout<<"lol2 "<<i<<" "<<j<<" "<<count<<a[i]<<" "<<a[j]<<endl;
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
