no_cats = raw_input();
cats_string = raw_input();
strengths_string = raw_input();

cats = [];
strengths = [];
for i in cats_string.split():
	cats.append(int(i));
for i in strengths_string.split():
	strengths.append(int(i));

cats.sort();
strengths.sort();
total = 0;
while len(cats) > 0:
	total += cats.pop()*strengths.pop();
