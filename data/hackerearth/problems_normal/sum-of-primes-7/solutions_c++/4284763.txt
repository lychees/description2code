#include <bits/stdc++.h>
using namespace std;
bool prime[1000005];
long sum[1000005];
void SieveOfEratosthenes(int n)
{
    memset(prime, true, sizeof(prime));
 
    for (int p=2; p*p<=n; p++)
    {
        if (prime[p] == true)
        {
            for (int i=p*2; i<=n; i += p)
                prime[i] = false;
        }
    }
}
int main()
{
	long i,n,l,r;
	SieveOfEratosthenes(1000002);
	prime[1] = false;
	for(i = 1;i <= 1000002;i++){
		if(prime[i] == true){
			sum[i] = sum[i-1] + i;
		}
		else{
			sum[i] = sum[i-1];
		}
	}
    cin >> n;
    while(n--){
    	cin >> l >> r;
    	cout << sum[r] - sum[l-1] << endl;;
    }
    return 0;
}
