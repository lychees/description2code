#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
bool hashTable[10005];
int a[1000005];
int main() {
    int t;
    scanf("%d",&t);
    while(t--) {
        memset(hashTable,0,sizeof(hashTable));
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++) scanf("%d",&a[i]);
        int count=0;
        for(int i=0;i<n;i++) {
            if(hashTable[a[i]]==false) count++,hashTable[a[i]]=!hashTable[a[i]];
        }
        printf("%d\012",count);
    }
}
