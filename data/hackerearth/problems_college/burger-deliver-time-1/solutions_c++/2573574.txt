#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	multimap<int,int>mymap;
	cin>>t;
	for(int i=1;i<t+1;i++)
	{
		int x,y;
		
		cin>>x>>y;
		mymap.insert(pair<int,int>(x+y,i));
	}
	//sort(mymap.begin(),mymap.end());
	multimap<int,int>::iterator it;
	for(it=mymap.begin();it!=mymap.end();it++)
	{
		cout<<it->second<<" ";
	}
	//cout<<endl;
}
