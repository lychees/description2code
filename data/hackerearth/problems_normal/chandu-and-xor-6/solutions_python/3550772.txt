nm = raw_input().split()
matrix = []
rowSum = []
colSum = [0] * int(nm[1])
for i in range(int(nm[0])):
	row = raw_input().split()
	s = sum(map(int, row))
	rowSum.append(s)
	matrix.append(map(int, row))
	for j in range(int(nm[1])):
		colSum[j] += int(row[j])
maxXor = 0
for i in range(int(nm[0])):
	for j in range(int(nm[1])):
		xor = rowSum[i] ^ colSum[j]
		if xor > maxXor:
			maxXor = xor
print maxXor
