#include <iostream>
#include <string>
using namespace std;

int T;

void test(string &s)
{
	/*char ch = s[0]; */
	int i, count = 0;
	char ch = (s[0] == 'X')? 'X' : 'Y';
	//int f = 1; // 1 means X --> or some reason
	for (i = 1; i < s.length(); ++i)
	{
		if (s[i] == ch) //Previously (s[i] != ch)
			++count;
		else
		{
			if (ch == 'X')
			{
				ch = 'Y';
			}
			else
			{
				ch = 'X';
			}
		}
	}
	cout << count << endl;
}
int main()
{
	cin >> T;
	string *s = new string[T];
	for (int i = 0; T--; ++i)
	{
		cin >> s[i];
		test(s[i]);
	}
}
