#include<iostream>
#include<iomanip>
#include<math.h>
using namespace std;

int main(void)
{
	long long  t,z,m,n,cnt;
	cin >> t;
	while(t--)
	{
		cnt=0;
		cin >> z >> m >> n;
		for(int x=1;x<=sqrt(z);x++)
		{
			if(z%x==0)
			{
				if((x<=m&&(z/x)<=n))
					cnt++;
				if((x<=n&&(z/x)<=m))
					cnt++;
				if(x==(z/x))
					cnt--;
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
