import re

#print re.findall(r'[bcdfghjklmnpqrstvwxz]{3,}','qiewldoaa')

t=input()
while t:
    t-=1

    s = raw_input()

    cond1 = re.findall(r'[aeiouy]',s)
    no_of_consonants = len(s) - len(cond1)

    if no_of_consonants > len(cond1):
        print 'hard'

    else:
        cond2 = re.findall(r'[bcdfghjklmnpqrstvwxz]{3,}',s)

        if len(cond2)>0:
            print 'hard'
        else:
            print 'easy'
