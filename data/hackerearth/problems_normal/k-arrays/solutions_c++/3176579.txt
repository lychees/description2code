#include <iostream>
#define mod 1000000007
using namespace std;
typedef long long int64;

int64 mod_exp(int x,int p) {
	if(p==0)
		return 1;
	int64 res = mod_exp(x,p/2);
	res = (res*res)%mod;
	if(p&1) {
		return (res*x)%mod;
	}
	return res;
}

int main()
{
	int T,N,K;
	int imodt[20];
    cin>>T;
    for(int i=0;i<20;i++) {
    	imodt[i] = mod_exp(i,mod-2);
    	//cout<<imodt[i]<<endl; 
    }
    while(T) {
    	cin>>N>>K;
    	int arr[K][N];
    	for(int i=0;i<K;i++) {
    	  	for(int j=0;j<N;j++) {
    	  		cin>>arr[i][j];
    	  	}
    	}
    	int mxi=20,mxv=0,mni=0,mnv=mod;
    	for(int i=1;i<(1<<N);i++) {
    		int m = i;
    		int c = 0;
    		int x = 0;
    		int64 q = 1;
    		while(m>0) {
    			if(m&1) {
    				int64 s = 0;
    				for(int j=0;j<K;j++) {
    					s=(s+arr[j][c])%mod;
    				}    			
    				q = (q*s)%mod;
    				x++;
    			}
    			m=m>>1;
    			c++;
    		}
    		q = (q*imodt[x])%mod;
    		if(q>mxv) {
    			mxv = q;
    			mxi = x;
    		}
    		else if(q==mxv&&x<mxi) {
    			mxi = x;
    		}
    		if(q<mnv) {
    			mnv = q;
    			mni = x;
    		}
    		else if(q==mnv&&x>mni) {
    			mni = x;
    		}
    	}
    	cout<<(mnv^mxv)<<" "<<(mni^mxi)<<endl;
    	T--;
    } 
    return 0;
}
