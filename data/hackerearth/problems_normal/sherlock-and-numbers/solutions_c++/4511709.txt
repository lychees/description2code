#include <iostream>
#include <string>

using namespace std;
 
int main()
{
    int t;
    cin >> t;
    while(t--){
    	long int n,k,p,a;
    	cin >> n >> k >> p;			
    	for(long int i = 0; i < k ; i++){
    		cin >>a;
    		if(a <= p)
    			p++;
		}
		if(p <= n){
			cout << p ;
		}
		else{
			cout << "-1";
		}
		if(t-1 >= 0){
			cout << "\n";
		}
    }
    return 0;
}
