#include <bits/stdc++.h>
#define ll long long int
#define gt(i) scanf("%lld", &i)
using namespace std;

int main() {
    ll test;
    gt(test);
    while(test--) {
    unsigned int dig[10000], fir=0, lt=0, car, n, x, sum=0;
    dig[0] = 1;    
    ll no;
    gt(no);
    for(n=2; n <= no && n < 1000; n++) {
        car = 0;
        for(x=fir; x <= lt; x++) {
            car = dig[x]*n + car;
            dig[x] = car%100000;
            if(x == fir && !(car%100000)) fir++;
            car /= 100000; }
        if(car) dig[++lt] = car; }
    for(x=fir; x <= lt; x++)
        sum += dig[x]%10 + (dig[x]/10)%10 + (dig[x]/100)%10 + (dig[x]/1000)%10
            + (dig[x]/10000)%10;
    	cout << sum << endl; 
    }
}
