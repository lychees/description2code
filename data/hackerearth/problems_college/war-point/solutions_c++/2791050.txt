#include <bits/stdc++.h>
using namespace std;

int main()	{
    long long t,i,j,n;
    scanf("%lld",&t);
    while(t--)	{
    	scanf("%lld",&n);
    	long long mini=1000000000000000000,sl=0,sr=0;
    	long a[n];
    	sl=sr=0;
    	for(i=0;i<n;i++)	{
    		scanf("%lld ",&a[i]);
    		sr+=a[i];
    	}
    	i=0;
    	//printf("%lld\n",sr);
    	for(i=0;i<n;i++)	{
    		sr-=a[i];
    		sl+=a[i];
 			mini=min(mini,abs(sr-sl));
		}
		printf("%lld\n",mini);
	}
    return 0;
}
