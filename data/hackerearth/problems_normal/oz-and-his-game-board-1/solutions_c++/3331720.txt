#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	if(t<1 || t>100){return 0;}
	for(int i=0;i<t;i++){
		long n,m,a,b,c;
		cin>>n>>m>>a>>b>>c;
		if(n<1 || n>pow(10,9)){return 0;}
		if(m<1 || m>pow(10,9)){return 0;}
		if(a<(-(pow(10,4))) || a>pow(10,4)){return 0;}
		if(b<(-(pow(10,4))) || b>pow(10,4)){return 0;}
		if(c<(-(pow(10,4))) || c>pow(10,4)){return 0;}
		long long sum = (n*a)+(m*b);
		if(m>n){
			if((((m-n)*b)+(n*2*c)) > sum){
				sum = ((m-n)*b)+(n*2*c);
			}
		}
		if(m<n){
			if((((n-m)*a)+(m*2*c)) > sum){
				sum = ((n-m)*a)+(m*2*c);
			}
		}
		if(m==n){
			if((n*2*c)>sum){
				sum = n*2*c;
			}
		}
		cout<<sum<<endl;
	}
    return 0;
}
