t, ri, gi, bi = map(int, raw_input().split())
rs=gs=bs=True
r=g=b=y=c=m=w=bk=0
for i in xrange(t):
	if i%ri == 0:
		rs = not rs
	if i%gi == 0:
		gs = not gs
	if i%bi == 0:
		bs = not bs
	if rs and gs and bs:
		w += 1
	elif not rs and not gs and not bs:
		bk += 1
	elif rs and gs:
		y +=1
	elif gs and bs:
		c +=1
	elif bs and rs:
		m +=1
	elif rs:
		r +=1
	elif bs:
		b +=1
	elif gs:
		g +=1
print r, g, b, y, c, m, w, bk
