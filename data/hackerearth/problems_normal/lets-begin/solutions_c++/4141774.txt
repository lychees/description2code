#include <iostream>
using namespace std;
long long int dp[1000001]={0};
int main()
{
    long long int t,n,i,j,k,l;
    dp[0]=-1;
    dp[1]=-1;
    dp[2]=1;
    dp[3]=1;
    dp[4]=2;
    dp[5]=1;
    dp[6]=2;
    dp[7]=1;
    dp[8]=2;
    for(i=9;i<1000001;i++){
    	dp[i]=min(dp[i-2],min(dp[i-3],min(dp[i-5],dp[i-7])));
    	dp[i]+=1;
    }
    cin>>t;
    while(t--){
    	cin>>n;
    	cout<<dp[n]<<endl;
    }
    return 0;
}
