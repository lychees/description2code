#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int m;
	cin>>m;
	int i,j;
    long long a[m][m], suml=0, sumr=0;
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		scanf("%lld", &a[i][j]);
    		if(i==j)
    		suml+=a[i][j];
    	}
    }
    
    for(i=0,j=m-1;i<m,j>=0;i++,j--)
    	sumr+=a[i][j];
    //long long ans = ;
    printf("%lld", abs(sumr-suml));
     return 0;
}
