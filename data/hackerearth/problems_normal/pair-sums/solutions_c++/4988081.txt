#include <iostream>
#include <vector>
#include <algorithm>
#define MAX 1000000
using namespace std;

int main()
{
	int n, k, flag, i, j, a;
    cin>>n>>k;
    int arr[MAX];
    for(i=0; i<n; i++)
    {
    	cin>>arr[i];
    }
    sort(arr, arr+n);
    flag = 0;
    for(i=0, j = n-1; i<j;)
    {
    	if(arr[i]+arr[j] == k)
    	{
    		cout<<"YES\n";
    		flag = 1;
    		break;
    	}
    	else if(arr[i]+arr[j] < k)
    	{
    		i++;
    	}
    	else
    	{
    		j--;
    	}
    }
    if(!flag)
    {
    	cout<<"NO\n";
    }
    return 0;
}
