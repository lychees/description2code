#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int T,i;
    string s,t;
    cin>>T;
    while(T--)
    {
    	cin>>s>>t;
    	int a[26];
    	memset(a,0,sizeof(a));
    	int l=s.length(),m=t.length(),cnt=0;
    	for(i=0;i<m;i++)
    	a[t[i]-97]++;
    	for(i=0;i<l;i++)
    	a[s[i]-97]--;
    	for(i=0;i<26;i++){
    	if(a[i]>0)
    	{
    	cnt++;	
    	}}
    	
    	if(cnt>(m-l))
    	cout<<"-1"<<endl;
    	else
    	cout<<abs(cnt)<<endl;
    }
    return 0;
}
