#include <iostream>
using namespace std;

int main()
{
    int A[10001],t,n,c,i;
    A[0]=0;
    A[1]=0;
    A[2]=1;
    for(i=3;i<=10000;i++){
    		c=0;
    		if((i%3==1) || (i%3==2) )
    			c=1;
    		
    		A[i] = 1+A[(i/3)+c];
    	}
    cin>>t;
    while(t--){
    	scanf("%d",&n);
    	printf("%d\n",A[n]);
    }
}
