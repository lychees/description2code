#include <iostream>
using namespace std;

int main()
{
    int n,i,maxEndingHere=0,maxSoFar=0;
    cin>>n;
    int a[n];
    
    for(i=0;i<n;i++)
    {
    	cin>>a[i];	
    	maxEndingHere+=a[i];
    	
    	if(maxEndingHere<0)
    		maxEndingHere=0;
    		
    	if(maxEndingHere>maxSoFar)
    		maxSoFar=maxEndingHere;
    	
    }
    cout<<maxSoFar;
    
    return 0;
}
