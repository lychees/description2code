#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
char a[100005];
struct abc
{
    int pos;
    int l;
};
abc  b[100005];
bool cmp(abc c, abc d){
    if(c.l==d.l)
        return c.pos<d.pos;
    return c.l>d.l;
}
int main()
{
	int i,j,k;
	int n;
	sd(n);
	ss(a);
	for(i=0;i<n;i++)
    {
        b[i].pos=i;
        b[i].l=(int)(a[i]-'a');
    }
    sort(b,b+n,cmp);
    int posdone=-1,ldone=27;
    for(i=0;i<n;i++)
    {
        if(b[i].pos>posdone&&b[i].l<ldone)
        {
            printf("%c",b[i].l+'a');
            posdone=b[i].pos;
            ldone=b[i].l;
        }
    }
	return 0;
}
