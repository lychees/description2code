#include<bits/stdc++.h>
using namespace std;

long long pw1(long long ax ,long long bx)
{
    long long ans=1;
    while(bx--)
    {
        ans*=ax;
    }
    return ans;
}

long long solve(long long ax)
{
    vector<int> my;
    while(ax)
    {
        my.push_back(ax%10);
        ax/=10;
    }

    while(true)
    {
        if(my[0]==0)
        {
            my.erase(my.begin());
        }
        else break;
    }

    long long ans =0;

    for(int i=0;i<my.size();i++)
    {
        //cout<<my[i]<<endl;
        ans+=(my[i]*pw1(10,my.size()-i-1));
    }
    return ans;
}




int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       long long ax,bx;
       cin>>ax>>bx;
       //cout<<solve(ax)<<endl;
       cout<<solve(solve(ax)+solve(bx))<<endl;
   }
    return 0;
}
