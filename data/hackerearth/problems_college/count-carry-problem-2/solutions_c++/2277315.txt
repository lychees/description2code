#include <iostream>
using namespace std;

int main()
{
    long long int test;
    cin >> test;
    while(test--) {
    	long long int a, b, tt = 0, flag = 0;
    	cin >> a >> b;
    	while(a != 0 && b != 0) {
    		if((a % 10 + b % 10 + flag) / 10 != 0) {
    			tt++;
    			flag = 1;
    		}
    		else 
    			flag = 0;
    		a = a/10;
    		b = b/10;
    	}
    	if(tt == 0) 
    		cout <<	"No carry operation" << endl;
    	else if(tt == 1)
    		cout << tt <<" carry operation" << endl;
    	else 
    		 cout << tt <<" carry operations" << endl;
    }
    return 0;
}
