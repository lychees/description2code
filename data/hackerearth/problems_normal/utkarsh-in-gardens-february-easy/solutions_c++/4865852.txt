#include<iostream>
#include<bitset>
#include<bits/stdc++.h>
using namespace std;
bitset<2015>arr[2015];
int main()
{
int n,x;
cin>>n;
for(int i=0;i<n;i++)
for(int j=0;j<n;j++)
{
cin>>x;
arr[i][j]=x;	
}
long long int ans=0;
for(int i=0;i<n;i++)
for(int j=i+1;j<n;j++)
{
int cnt=(arr[i]&arr[j]).count();
ans+=(cnt*(cnt-1))/2;	
}
cout<<ans/2;	
}
