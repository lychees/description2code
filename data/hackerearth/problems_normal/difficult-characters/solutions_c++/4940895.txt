#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;
bool compare(const pair<int,int> &a,const pair<int,int> &b)
{
	if(a.first<b.first)
		return true;
    else if(a.first==b.first)
    	{
    		if(a.second>b.second)
    			return true;
    			else
    			return false;
    	}
    else
    	return false;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string str;
		cin>>str;
		vector<pair<int,int>> vec;
  		for(int i=0;i<26;i++)
  		{
  			vec.push_back(make_pair(0,i));
  		}
  		for(int i=0;i<str.length();i++)
  		{
  			vec[str[i]-'a'].first++;
  		}
  		sort(vec.begin(),vec.end(),compare);
  		vector<pair<int,int>> :: iterator it=vec.begin();
  		for(;it!=vec.end();it++)
  		{
  			cout<<(char)(it->second+'a')<<" ";
  		}
  		cout<<endl;
  	}
	
	return 0;
}
