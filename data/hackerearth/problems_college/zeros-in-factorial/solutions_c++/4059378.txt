#include<bits/stdc++.h>
using namespace std;
#define ll long long int
long long int answer[1000005];
void ans(){
	answer[0]=0;
	answer[1]=0;
	answer[2]=0;
	answer[3]=0;
	answer[4]=0;
	for(long long int i=5;i<=1000000;++i){
		long long int tmp=i;
		long long int cnt=0;
		long long int pw=5;
		while(pw<=tmp){
			cnt+=(tmp/pw);
			pw*=5;
		}
		answer[i]=cnt;
	}
	for(int i=1;i<=1000000;++i)
		answer[i]+=answer[i-1];
}
int main()
{
    ans();
    int t,a,b;
    scanf("%d",&t);
    while(t--){
    	scanf("%d%d",&a,&b);
    	cout<<answer[b]-answer[a-1]<<endl;
    }
    return 0;
}
