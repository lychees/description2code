#include<bits/stdc++.h>
using namespace std;

long long a[100][30];

int main()
{
    long long i , j;
    long long n = 100;
    long long k = 30;
    for (i = 0; i <= n; i++)
    {
        for ( j = 0; j <= k; j++)
        {
            if ( j == i|| j==0 )
                a[i][j] = 1;
            else
                a[i][j] = a[i-1][j-1] + a[i-1][j];
        }
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n,k;
        scanf("%lld%lld",&n,&k);
        printf("%lld\n",a[n+k-1][n-1]);
    }
}
