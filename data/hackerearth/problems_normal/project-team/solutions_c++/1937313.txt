#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    
    int T=0,num=0,min = INT_MAX,max = 0;
    cin >> T;
    
    while(T--){
    	cin >> num;
    	int *a = new int[num];
    	for(int i = 0; i < num; i++){
    		cin >> a[i];
    	}
    	sort(a,a+num);
    	priority_queue<int> pq;
    	for(int i = 0; i < num/2; i++){
    		int sum = a[num-i-1]+ a[i];
    		if(min > sum)min = sum;
    		if(max < sum)max = sum;
    	}
    	cout << max-min<< endl;
    	max = 0;min = INT_MAX;
    }
    return 0;
}
