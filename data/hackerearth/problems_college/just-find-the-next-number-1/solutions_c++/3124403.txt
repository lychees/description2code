#include <iostream>
using namespace std;
#define ll long long

int main()
{	ll x;
	cin>>x;
	while(x--)
	{	ll z,t,f=0;
		cin>>z;
		while(1)
		{	t=3;f=0;
			while(t)
			{	if((z&t)==t)
				{	f=1;
					break;
				}
				t=t<<1;
			}
			if(f==1)
				z++;
			else
			{	cout<<z<<"\n";
				break;
			}
		}
	}
    return 0;
}
