#include <iostream>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   		long long int n;
   		long long int arr[100000];
   		long long int sum=0;
   		cin>>n;
   		for(int i=0;i<n;i++)
   		{
   			cin>>arr[i];
   			sum+=arr[i];
   		}
   		if(sum%n==0)
   			cout<<"YES\n"<<sum/n<<" 0"<<endl;
   		else
   			cout<<"NO\n"<<sum/n<<" "<<sum%n<<endl;
   }
    return 0;
}
