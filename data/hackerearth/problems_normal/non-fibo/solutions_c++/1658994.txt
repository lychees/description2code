#include <iostream>
#include <algorithm>
#include <stdio.h>
using namespace std;

int main() {	
	long long int arr[75];
	arr[1]=1;
	arr[2]=2;
	for(int i=3;i<75;i++)
	{
		arr[i]=arr[i-1]+arr[i-2];
	}
	int t;
	scanf("%d",&t);
	while(t--){
		long long int n,temp,count;
		scanf("%lld",&n);
		temp=upper_bound(arr,arr+75,n)-arr;
		temp--;
		count=n+1;
		while(temp>0){
			if(binary_search(arr,arr+75,count)!=1)
			{
				temp--;
				if(temp==0)
					break;
			}
			count++;
		}
	printf("%lld\n",count);
	}
	return 0;
}
