#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    long t,n ,b,res ;
    scanf("%ld",&t);
    while(t--)
    {
        int i ;
        i=1;
        res=0;
        scanf("%ld%ld",&n,&b);
        if(n>b)
        {
            while(n>(b*i))
            {
                i++;
            }
            if(n-(b*(i-1))>(b*i)-n)
            {
               res= b*i;
            }
            else if(((b*i)-n) > (n-(b*(i-1))))
                    {
                        res=b*(i-1);
                    }
                    else
                        {
                            res = b*i ;
                        }

        }
        if(b%2!=0)
        {
            b++;
            if(n==b/2)
            {
                res=b-1;
            }
        }
        if(n==b/2)
            {
                res=b;
            }
            printf("%ld\n",res);


    }
    return 0;
}
