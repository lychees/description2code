#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int k = 0; k < t; k++ ){
			vector<pair< pair < long int , long int >, int > > v;
			int n;
			cin>>n;
			for(int i = 0; i < n; i++){
				int x,y,z;
				cin>>x>>y>>z;
				v.push_back(make_pair(make_pair(x,y),z));
			}
			sort(v.begin(),v.end());
			int dp[n+2];
			for(int i = 0; i <= n+1; i++){
				dp[i] = 0;
			}
			dp[n-1] = v[n-1].second; 
			for(int i = n-2; i >= 0; i--){
				if(v[i].first.second <= v[i+1].first.first){
					dp[i] = dp[i+1] + v[i].second;
				}
				else{
					int j = i+1;
					while(j < n && v[i].first.second > v[j].first.first){
						j++;
					}
					dp[i] = max(dp[i+1],v[i].second+dp[j]);
				}	
			}
		
			
			float ans = (float)dp[0]/100;
			cout<<fixed<<setprecision(2)<<ans<<endl;
		
	}
    return 0;
}
