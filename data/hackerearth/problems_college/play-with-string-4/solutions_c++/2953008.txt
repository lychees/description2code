#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;cin >> t;
    while(t--){
    	string ch, ch1;
    	int ok=0;
    	cin >> ch;
    	ch1 = ch;
    	reverse(ch1.begin(),ch1.end());
    	for(int i=1;i<ch.length();i++){
    		if(abs(ch[i]-ch[i-1])!=(abs(ch1[i]-ch1[i-1]))){
    			ok=1;break;
    		}
    	}
    	if(ok) cout << "Not Funny" <<endl;
    	else cout << "Funny" <<endl;
    }
    return 0;
}
