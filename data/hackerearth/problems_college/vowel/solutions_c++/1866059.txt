#include <stdio.h>

int main()
{
	int testCases, flag;
	scanf("%d", &testCases);
	for(int c=0;c<testCases;c++)
	{
		char str[100];
		scanf("%s", str);
		flag = 0;
		for(int i=0;str[i]!= '\0';i++)
		{
			if(str[i]=='A' || str[i]=='E' || str[i] == 'I' || str[i] == 'O' || str[i] == 			'U' || str[i] == 'a' || str[i] == 'i' || str[i] == 'e' || str[i] == 'o' || str[i] == 'u')
			{
				printf("%c", str[i]);
				flag = 1;
			}
		}
		if(flag == 0)
			printf("No");
		printf("\n");
	}
	return 0;
}
