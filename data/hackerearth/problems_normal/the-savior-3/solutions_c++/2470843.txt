#include<iostream>
using namespace std;
int main()
{
	int t,i,j,n,k,l,count;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		int a[n];
		for(j=0;j<n;j++)
		cin>>a[j];
		count=0;
		for(k=0;k<n;k++)
		{
			for(l=k+1;l<n;l++)
			{
				if(a[k]!=a[l])
				{
					if((a[k]+a[l])%2==0)
					count++;
			    }
			}
		}
		cout<<count<<"\n";
	}
	return 0;
}
