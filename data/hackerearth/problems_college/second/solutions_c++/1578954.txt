#include <iostream>
#include<algorithm>
using namespace std;
	int binarysearch(int *a,int x,int t){
	int b=0,e=t-1;
	int mid;
	while(b<e){
		mid=(b+e)/2;
		if(a[mid]==x)
		return 1;
		else if(a[mid]>x)
		e=mid-1;
		else if(a[mid]<x)
		b=mid+1;
	}
	if(a[(b+e)/2]==x)
	return 1;
	
	return 0;
	}
