#include <iostream>
using namespace std;

int main()
{
    long long int t,n,a,b,c;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cin>>a>>b>>c;
    	long long int x[n],y[n],i,ctr1=0,ctr2=0;
    	for(i=0;i<n;++i)
    	{
    		cin>>x[i]>>y[i];
    	}
    	for(i=0;i<n;++i)
    	{
    		if(a*x[i]+b*y[i]+c>0)
    		++ctr1;
    		else if(a*x[i]+b*y[i]+c<0)
    		++ctr2;
    	}
    	if(ctr1>ctr2)
    	cout<<ctr1<<"\n";
    	else cout<<ctr2<<"\n";
    }
    return 0;
}
