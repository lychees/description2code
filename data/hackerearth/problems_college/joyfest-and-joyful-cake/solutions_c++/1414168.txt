#include <iostream>
using namespace std;
#include<cstdio>
#include <sstream>
#define gc getchar_unlocked

void scanint(long long &x)
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
long long array[2000000];
int main()
{
    int i;
    array[0] = 1;
    for(i = 1; i < 2000000; i++) {
    	array[i] = array[i - 1] + i;
    }
    int t;
    cin>>t;
    while(t--) {
    	register long long c = gc();
    	long long x = 0;
	    for(;(c<48 || c>57);c = gc());
	    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    	int start = 0, end = 1999999;
    	int mid;
    	while(start <= end) {
    		mid = (start + end) / 2;
    		if(array[mid] < x)
    			start = mid + 1;
    		else if(array[mid] > x)
    			end = mid - 1;
    		else
    			break;
    	}
    	if(array[mid] >= x)
    		cout<<mid;
    	else if(array[mid] < x)
    		cout<<mid+1;
    	cout<<"\n";
    }
    return 0;
}
