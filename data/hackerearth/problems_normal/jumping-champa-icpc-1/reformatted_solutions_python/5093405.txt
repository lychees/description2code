t = input()
for x in xrange(0, t):
    nq = raw_input().split()
    nq = map(int, nq)
    cities = raw_input().split()
    cities = map(int, cities)
    cities.sort(reverse=True)
    sum = 0
    for y in xrange(0, nq[0]-1):
        # print(cities[y],cities[y+1])
        sum += abs(cities[y]-cities[y+1])
    sum *= nq[1]
    print(sum)
