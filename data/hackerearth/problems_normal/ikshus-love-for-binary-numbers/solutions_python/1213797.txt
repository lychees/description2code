from fractions import gcd
def simp(a, b):
    x = gcd(a, b)
    return (str(a//x) + '/' + str(b//x))
a, b = map(int, raw_input().split(' '))
denom = 2**a
dp = [0] * (a+1)
for i in range(1, b):
    dp[i] = 2**i
dp[b] = 2**b-1
for i in range(b+1, a+1):
    dp[i] = sum(dp[i-b:i])

print(simp(2**a-dp[-1], 2**a))
