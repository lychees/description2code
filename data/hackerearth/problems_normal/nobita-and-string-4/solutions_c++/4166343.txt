#include<bits/stdc++.h>
using namespace std;
vector<string> vt;
string str;
int main()
{
		//cout << "Hello World!" << endl;
		int t;
		cin>>t;
		getchar();
		while(t--)
		{
				getline(cin,str);
				int i;
				string str1;
				for(i=0;i<str.size();++i)
				{
						str1 = "";
						while(i<str.size() && str[i]!=' ')
						{
								str1 += str[i];
								++i;
						}
						vt.push_back(str1);
				}
				//cout<<"tokenization done\n";
				int n = vt.size();
				int j;
				for(i = 0,j = n - 1; i<n && j>=0 && i < j;++i,--j)
				{
						//cout<<vt[j]<<" "<<vt[i]<<" ";
						swap(vt[i],vt[j]);
				}

				//if(i == j) cout<<vt[j]<<endl;
				//else cout<<endl;
				for(i=0;i<vt.size();++i)
						cout<<vt[i]<<" ";
				cout<<endl;
				vt.clear();
		}
		return 0;
}
