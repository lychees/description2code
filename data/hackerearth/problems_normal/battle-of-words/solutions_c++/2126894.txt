#include<iostream>
#include<cstring>
#include<stdio.h>
#include<cstdio>
using namespace std;
int main()
{
    int t,key1[26],key2[26],j,i,flag1,flag2,l1,l2;
    char str1[100001],str2[100001];
    scanf("%d",&t);
    cin.ignore();
    while(t--)
    {
        flag1=flag2=0;
        for(i=0;i<26;i++)
        {
            key1[i]=key2[i]=0;
        }
        gets(str1);
        gets(str2);
        l1=strlen(str1);
        l2=strlen(str2);
        for(int k=0;k<l1;k++)
        {
            if(str1[k]!=' ')
                key1[str1[k]-97]++;
        }
        for(int k=0;k<l2;k++)
        {
            if(str2[k]!=' ')
                key2[str2[k]-97]++;
        }
        for(i=0;i<26;i++)
        {
            if(key1[i]>key2[i])
                flag1=1;
            else if(key1[i]<key2[i])
                flag2=1;
        }
        if(flag1==1&&flag2==1)
            printf("You draw some.\n");
        else if(flag1==1)
            printf("You win some.\n");
        else if (flag2==1)
            printf("You lose some.\n");

    }
    return 0;
}
