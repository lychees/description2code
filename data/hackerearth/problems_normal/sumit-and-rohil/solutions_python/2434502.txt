from collections import Counter as mset

if __name__ == "__main__":

    T = int(raw_input())

    pair = {}
    CT = 0

    for t in xrange(T):

        w = raw_input()
        k = (w[0], w[-1])

        L = pair.get(k, None)
        if L is None:
            pair[k] = [mset(w)]
            CT += 1
        else:
            S = mset(w)
            if S not in pair[k]:
                CT += 1
                pair[k].append(S)

    print CT
