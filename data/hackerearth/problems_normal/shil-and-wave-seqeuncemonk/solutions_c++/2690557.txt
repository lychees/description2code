const int maxN = 100000 + 2;
const int mod = 1000*1000*1000 + 7;

int fen[2][maxN];

void add(int ps, int n, int k)
{
    for(; n < maxN; n += n & (-n))
	  fen[ps][n] = (fen[ps][n] + k) % mod;
}
int get(int ps, int n)
{
    int res = 0;
    for(; n > 0; n -= n & (-n))
	  res = (res + fen[ps][n]) % mod;
    return res;
}

int a[maxN];
int dp[2][maxN];

	  dp[1][i] += get(0, maxN) - get(0, a[i]);
	  
	  dp[0][i] %= mod;
	  dp[1][i] %= mod; dp[1][i] += mod; dp[1][i] %= mod;

	  add(0, a[i], dp[0][i]);
	  add(1, a[i], dp[1][i]);
    }

//    rep(i,0,2) cout << dp[0][i] << ' ' << dp[1][i] << endl;
