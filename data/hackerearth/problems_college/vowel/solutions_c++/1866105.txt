#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,i,j,ans;
    char s[100000];
    scanf("%d",&t);
    while (t --) {
    	scanf("%s",s);
    	int len = strlen(s);
    	string res = "";
    	int c = 0;
    	for (i = 0; i < len; i++) {
    		if (s[i] == 'a'||s[i]=='A'||s[i]=='e'||s[i]=='E') {
    			res += s[i];
    			c++;
    		}
    		if (s[i]=='i'||s[i]=='I'||s[i]=='o'||s[i]=='O'){
    			res += s[i];
    			c++;
    		}
    		if (s[i]=='u'||s[i]=='U') {
    			res += s[i];
    			c++;
    		}
    	}
    	if (c >= 1) {
    		cout << res << endl;
    	} else {
    		cout << "No" << endl;
    	}
    }
    
    return 0;
}
