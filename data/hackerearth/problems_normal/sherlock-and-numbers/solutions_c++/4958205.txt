#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 10;

int a[N];

int main() {
	int t;
	scanf("%d", &t);
	for (int tt = 1; tt <= t; tt++) {
		int n, k, p;
		scanf("%d %d %d", &n, &k, &p);
		for (int i = 0; i < k; i++) {
			scanf("%d", &a[i]);
		}
		sort(a, a + k);
		int pos = 0, st = 0;
		do {
			pos = upper_bound(a + st, a + k, p + st) - (a + st);
			st += pos;
		} while (pos != 0);
		int res = p + st;
		printf("%d\n", (res > n) ? -1 : res);
	}
	return 0;
}
