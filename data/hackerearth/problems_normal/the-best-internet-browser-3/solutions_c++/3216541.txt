#include <iostream>
#include <string>

using namespace std;

int main()
{

	string web;
	int tc, count, len;
	
	cin >> tc;
	
	while(tc--)
	{
		cin >> web;
		
		count = 0;
		
		len = web.size();
		
		for(int i = 4; i < len; i++)
			if(web[i] != 'a' && web[i] != 'e' && web[i] != 'i' && web[i] != 'o' && web[i] != 'u')
				count++;
		
		cout << count+1 <<"/"<< len << endl;
	}
    return 0;
}
