#!/usr/bin/env python
# encoding: utf-8

fa = []


def findfa(x):
    if x == fa[x]:
        return x
    else:
        fa[x] = findfa(fa[x])
        return fa[x]


if __name__ == '__main__':
    a = map(int, raw_input().split())
    N, K = 0, 0
    if len(a) == 1:
        N = a[0]
        K = int(raw_input())
    else:
        N, K = a[0], a[1]
    fa = [i for i in xrange(N + 1)]
    for _ in xrange(K):
        i, j = map(int, raw_input().split())
        fi, fj = findfa(i), findfa(j)
        if fi != fj:
            fa[fj] = fi

    ans = 0
    for i in xrange(1, N + 1):
        if i == findfa(i):
            ans += 1
    print ans
