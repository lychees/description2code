#include<stdio.h>
#define SIZE 12
int troop[SIZE] = {0};
int ans = 9999,N;

int max(int v1,int v2,int v3)	{
	int big = v1;
	if(v2>big)	big = v2;
	if(v3>big)	return v3;
	return big;
}
void arrangeTroops(int s1,int s2,int s3,int p)	{
	if(p>=N)	{
		int maxVal = max(s1,s2,s3);
		if(maxVal<ans)	ans = maxVal;
	}
	else	{
		arrangeTroops(s1+troop[p],s2,s3,p+1);
		arrangeTroops(s1,s2+troop[p],s3,p+1);
		arrangeTroops(s1,s2,s3+troop[p],p+1);
	}
}

int main()	{
	scanf("%d",&N);
	for(int i=0;i<N;i++)
		scanf("%d",&troop[i]);
	arrangeTroops(0,0,0,0);
	printf("%d",ans);
	return 0;
}
