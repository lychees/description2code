#include<iostream>
using namespace std;
int main()
{
	long long int k,m,n,j,i,a,b,count=0;
	char s[100000];
		cin>>s;
	cin>>m;
		for(i=0;s[i]!='\0';i++)
	{
		count++;
	}
for(k=0;k<m;k++)
{
	cin>>a>>b;
	a=a%count;
	b=b%count;
	if(a%count==0)
	{
		a=count;
	}
	if(b%count==0)
	{
		b=count;
	}	
	for(i=0;i<count;i++)
	{
		if(s[a-1]==s[b-1])
		{
			cout<<"Yes"<<endl;
			break;
		}
		else 
		{
			cout<<"No"<<endl;
			break;
		}
	}
}
}
