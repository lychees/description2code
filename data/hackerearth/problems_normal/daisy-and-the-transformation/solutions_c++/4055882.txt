#include <iostream>
using namespace std;

int main()
{
    long int t,n;
    string s1,s2;
    cin>>t;
    while(t--){
    	s1.clear();
    	s2.clear();
    	cin>>s1>>s2>>n;
    	for(int i=0;i<s1.length();i++){
    		if(s1[i]!=s2[i])
    			n--;
    	}
    	if(n>=0){
    	if(n%2==0)
    	cout<<"Yes\n";
    	else{
    		cout<<"No\n";
    	}
    	}
    	else
    	cout<<"No\n";
    }
    return 0;
}
