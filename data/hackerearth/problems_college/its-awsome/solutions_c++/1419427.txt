#include <iostream>
using namespace std;


long long int calc(long long int n)
{
	long long int d,c1=0;
	while(n>0)
	{
		d=n%10;
		if(d==4||d==7)
		c1++;
		n=n/10;
	}
	return c1;
}

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    long long 	int n,x;
    	cin>>n;
    x=	calc(n);
    
    if(x==4||x==7)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    	
    }
}
