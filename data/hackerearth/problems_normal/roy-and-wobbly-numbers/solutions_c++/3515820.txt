#include <iostream>
using namespace std;

int main()
{
   int t,n,k,i,a[100],p=10,a1,a2;
    for(i=1;i<82;i++)
    {
    	if((p)/10==(p)%10) {p++;}
    	a[i]=p;
    	p++;
    }
    cin>>t;
    while(t)
    {
    	cin>>n>>k;
    	if(k>81) {cout<<"-1";}
    	else{
    		a1=a[k]/10;
    		a2=a[k]%10;
    		for(i=1;i<=n;i++)
    		{
    			if(i%2==0) {cout<<a2;}
    			else {cout<<a1;}
    		}
    	}
    	cout<<"\n";
    	t--;
    }
    return 0;
}
