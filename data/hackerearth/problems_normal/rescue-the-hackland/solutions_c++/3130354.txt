#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
		int t;
	   cin >> t;
	   
	   while (t--) {
	   		int L, R, K;
	   		cin >> L >> R >> K;
			int max = INT_MIN;
			int i = 0;
			for (i = L; i < L + K; ++i) {
				int c = 0;
				int j = i;
				while (j <= R) {
					++c;
					j = j * K + 1;
					
				}
				
				if (c > max) {
					max = c;
				}
			}
			
			cout << max << endl;
	   }
}
