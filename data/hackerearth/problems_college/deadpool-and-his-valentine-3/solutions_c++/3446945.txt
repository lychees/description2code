#include<iostream>
using namespace std;

int main()
{
    bool arr[110];
    for(int i=0; i<110; i++)
        arr[i] = false;
    int k, m;
    cin>>k>>m;
    while(k--)
    {
        int n;
        cin>>n;
        for(int i=0; i<n; i++)
        {
            int temp;
            cin>>temp;
            arr[temp] = true;
        }
    }

    bool flag = true;
    for(int i=1; i<=m; i++)
        if(arr[i] == false)
    {
        flag = false;
        break;
    }

    if(flag == true)
        cout<<"YES\n";
    else
        cout<<"NO\n";

    return 0;
}
