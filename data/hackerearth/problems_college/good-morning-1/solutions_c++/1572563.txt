#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,d;
	long long ans[101];
	ans[1] = 2;
	ans[2] = 4;
	for (i = 3; i < 102; i++) {
		ans[i] = ans[i-1] + 6;
	}
	scanf("%d",&t);
	while (t --) {
		scanf("%d %d",&n,&d);
		if (ans[n] >= d) {
			printf("Yes\n");
		} else {
			printf("No\n");
		}
	}
	
	return 0;
}
