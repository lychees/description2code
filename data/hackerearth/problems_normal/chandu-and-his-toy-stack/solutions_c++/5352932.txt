#include <bits/stdc++.h>
using namespace std;
// code monk greedy

int main()
{
	int t,n,x,y,i,a[100000],b[100000];
	long long sum;

	cin>>t;
	while(t--)
	{
		cin>>n>>x>>y;
		sum=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i]>>b[i];			
		}
		sort(a,a+n);
		sort(b,b+n);
		for(i=0;i<n;i++)
		{
			if(a[i]>b[i])
				sum+=(a[i]-b[i])*y;
			else
				sum+=(b[i]-a[i])*x;
		}
		cout<<sum<<"\n";
	}
	return 0;
}
