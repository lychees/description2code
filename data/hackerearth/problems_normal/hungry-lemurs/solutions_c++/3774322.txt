

#include<iostream>
#include<limits.h>

using namespace std;

int main()
{
    int N,K,mb,ma;
    cin>>N>>K;
    if(N<=K)
        cout<<min(K-N,N);
    else
    {
        int btim=INT_MAX;
        for(int i=1;i<=N;i++)
        {
           int tmin;
           if(K-i >= 0)
                tmin=K-i;
           else
            tmin=(-(K-i));
           mb=(N/i)*i;
           ma=mb+i;
           tmin+=min(N-mb,ma-N);
           if(tmin < btim)
              btim=tmin;
        }
        cout<<btim;
    }
    return 0;
}
