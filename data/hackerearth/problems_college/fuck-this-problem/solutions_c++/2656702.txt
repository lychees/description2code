#include <iostream>
#include<string.h>
#include<vector>
using namespace std;

int main()
{
    vector<string> result;
    char s[10000];
    string temp;
    int t,num=0;
    gets(s);
    cin>>t;
    for(int i=0;i<strlen(s);i++){
    	temp += s[i];
    	num++;
    	if(num==t){
    		result.push_back(temp);
    		temp.clear();
    		num=0;
    	}
    }
    result.push_back(temp);
    for(int i=0;i<result.size();i++){
    	cout<<result[i]<<endl;
    }
    return 0;
}
