def suture(s1, s2):
    for n in xrange(len(s2), 0, -1):
        if s1.endswith(s2[:n]):
            return s1 + s2[n:]
    return s1 + s2

import itertools
strings = []
for _ in xrange(input()):
    strings.append(raw_input())
best = 10**100
for perm in itertools.permutations(strings):
    line = ''
    for s in perm:
        line = suture(line, s)
    if len(line) < best:
        best = len(line)
print best
