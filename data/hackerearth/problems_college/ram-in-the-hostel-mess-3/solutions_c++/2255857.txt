template < typename T > T abs(T x)
{
	return x > 0 ? x : -x;
}

template < typename T > T sqr(T x)
{
	return x * x;
}

ull f[60];

int main()
{	
	int t;
	
	cin >> t;
	
	f[0] = 1; f[1] = 2;
	
	for (int i = 2; i <= 60; i++)
		f[i] = f[i - 1] + f[i - 2];
	
	while (t--) {
		int n;
		
		cin >> n;
		
		cout << f[n] * f[n] << "\n";
	}
	
	return 0;
}
