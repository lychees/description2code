#include <iostream>
#include <algorithm>
using namespace std;

void solve(){
	long long g,m;
	cin>>g>>m;
	long long groups[g];
	for(int i = 0; i<g; i++){
		cin>>groups[i];
	}
	sort(groups,groups+g);
	long long i = 0;
	long long sum = 0;
	while(i<g && sum+groups[i]<=m){
		sum += groups[i];
		i++;
	}
	cout<<i<<endl;
}

int main(){
    int d;
    cin>>d;
    while(d--) solve();
    return 0;
}
