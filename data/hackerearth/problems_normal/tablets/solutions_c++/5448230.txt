#include <bits/stdc++.h>
using namespace std;
int a[100005];
int candy[100005];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>a[i];
    }
    candy[0]=1;
    for(int i=1;i<n;i++){
        if(a[i]>a[i-1]){
            candy[i]=candy[i-1]+1;
        }
        else{
            candy[i]=1;
        }
    }
    for(int i=n-2;i>=0;i--){
        if(a[i]>a[i+1]){
            candy[i]=max(candy[i],candy[i+1]+1);
        }
        else{
        //
        }
    }
    long long int ans=0;
    for(int i=0;i<n;i++){
        ans+=candy[i];
    }
    cout<<ans<<endl;
}
