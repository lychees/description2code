S = raw_input()
q = int(raw_input())

for i in xrange(q):

    L1, R1, L2, R2 = map(int, raw_input().split(" "))

    len1 = R1-L1+1
    len2 = R2 - L2 + 1

    sub1 = S[L1-1:R1]
    sub2 = S[L2-1:R2]
    if sub1 == sub2:
        print "Yes"

        print "No"
