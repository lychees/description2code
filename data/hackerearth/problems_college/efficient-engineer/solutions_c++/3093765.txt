#include <bits/stdc++.h>
#define ll long long
#define LINF 9999999999999
#define INF 999999999

using namespace std;
int main(){
	ll N,K,D;
	cin >> N >> K >> D;
	vector<ll> v;
	vector<ll> f;
	for(ll i=0;i<N;i++){
		ll x; cin >> x;
		v.push_back(x);
	}
	ll sum = 0;
	ll ans = 0;
	int flag = true;
	for(ll w =0;w<N-K+1;){
		ll sum = 0;
		//cout << "w " << w << endl;
		for(ll i=0;i<K;i++){
			//cout << "i " << w+i << endl;
			sum+=v[w+i];
		}
		if(sum == 0){
			flag = false;
			break;
		}
		if(sum < D){
			ans++;
			ll j = w+K-1;
			ll idx;
			for(ll i=0;i<K;i++){
				if(v[j-i] !=0){
					idx = j-i;
					break;
					
				}
			}

		w = idx+K;
		}else{
			w++;
		}
//		cout << endl;
	}
	if(flag)
		cout << ans << endl;
		else
		cout << "-1" << endl;
	return 0;
}
