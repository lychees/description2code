#include <iostream>
using namespace std;
#define ll long long 
ll int S[25];
ll int fun( ll int n, ll int m )
{
   ll int i, j, x, y;
 
   ll  int table[n+1][m];
    
    for (i=0; i<m; i++)
        table[0][i] = 1;
 
    for (i = 1; i < n+1; i++)
    {
        for (j = 0; j < m; j++)
        {
            x = (i-S[j] >= 0)? table[i - S[j]][j]: 0;
 
            y = (j >= 1)? table[i][j-1]: 0;
 
            table[i][j] = x + y;
        }
    }
    return table[n][m-1];
}
int main()
{
    //printf("Hello World!\n");
  ll int t,i,k=0,j;
   cin>>t;
   for(i=1;i<=21;i++)
   	S[k++]=i*i*i;
   while(t--)
   {
   	ll int x;
   	cin>>x;
   	ll int ans=fun(x,21);
   	printf("%lld\n",ans);
   }
   return 0;
}
