#define si(a)     scanf("%d", &a)
#define sl(a)     scanf("%lld", &a)
#define pi(a)     printf("%d", a)
#define pl(a)     printf("%lld", a)
#define pm(str)   printf("%s\n",str)
#define pn        printf("\n")


const long long MOD = 1E9 + 7;
ll power(ll a, ll b) {
	//cout << a << " " << b << endl;
	ll ans = 1;
	while(b) {
		if(b%2) ans = (ans * a)%MOD;
		b = b/2;
		a = (a*a)%MOD;
	}
 
	return ans;
}

int main() {
    int t;
    si(t);
    while (t--) {
    	ll n;
    	sl(n);
    	ll ans = power(2, n - 1);
    	ans = power(ans, MOD - 2);
    	pl(ans);
    	pn;
    
    }
    return 0;
}
