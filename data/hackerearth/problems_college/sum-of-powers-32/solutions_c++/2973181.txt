#include <iostream>
using namespace std;

int main()
{
    int tc,ar[1000005]={0};
    ar[1]=1;
    for(long i=2;i<1000002;i++)
    {
    	if(ar[i]==0)
    		for(long j=i*i;j<1000002;j*=i)
    			ar[j]=1;
    }
    cin>>tc;
    while(tc--)
    {
    	long n,i;
    	cin>>n;
    	for(i=1;i<n;i++)
    		if(ar[n-i] && ar[i])
    		{
    			cout<<"Yes\n";
    			break;
    		}
    	if(i==n) cout<<"No\n";
    }
    return 0;
}
