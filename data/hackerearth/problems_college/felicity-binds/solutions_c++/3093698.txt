#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int men[n],wom[n];
		for(int i=0;i<n;i++)
			cin>>men[i];
		for(int i=0;i<n;i++)
			cin>>wom[i];
		sort(men,men+n);
		sort(wom,wom+n);
		long long ans=0;
		for(int i=0;i<n;i++)
		{
			ans+=(men[i]*wom[i]);
		}
		cout<<ans<<endl;
	}
    return 0;
}
