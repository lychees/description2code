
t = int(raw_input(""))
for n in range(0, t):

    x = str(raw_input(""))

    l = []

    l.append(x[0])

    for i in range(1, len(x)):
        if x[i] == l[-1]:
            continue
        else:
            if x[i] == x[i-1]:
                l.append(x[i])
            else:
                l.append(x[i])

    s = ''.join(l)
    print(s)
