#include <bits/stdc++.h>

using namespace std;

int comp(string a,string b)
{
    string ab = a.append(b);
    string ba = b.append(a);

    if (ab > ba)
        return 1;
    else
        return 0;
}

int main()
{
    int tc;
    cin >> tc;

    vector<string > pr;
    vector<string> :: iterator it;
    string n;
    while(tc--)
    {
        cin >> n;
        pr.push_back(n);
    }

    sort(pr.begin(),pr.end(),comp);

    for (int i = 0;i < pr.size();i++)
        cout << pr[i];

    cout << endl;

    return 0;
}
