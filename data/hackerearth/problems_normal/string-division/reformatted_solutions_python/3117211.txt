T = input()
for i in range(T):
    string = raw_input()
    l = len(string)
    ls = len(set(string))
    if l < 4:
        print 'NO'
    elif l > 9:
        print 'YES'
    elif l == 4 and ls > 3:
        print 'YES'
    elif l > 4 and ls > 2:
        print 'YES'
    elif l > 6 and ls > 1:
        print 'YES'
    else:
        print 'NO'
