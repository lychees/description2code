template<int R, int C>
class MAT{
  typedef unsigned T;
  typedef vector<T> V;
  typedef MAT<R,C> mat;
  public:
  int r, c;
  int m[R][C];
  MAT(int r, int c):r(r), c(c){memset(m,0,sizeof(m));}
  MAT mul(const MAT &other){
    assert(c==other.r);
    int rr=r, rc=other.c;
    MAT res(rr,rc);
    for(int i=0;i<rr;i++)for(int j=0;j<rc;j++)
    for(int k=0;k<c;k++)res.m[i][j]+=m[i][k]*other.m[k][j];
    return res;
  }
  void pow(ll b){
    assert(r==c);
    MAT res(r,c);
    for(int i=0;i<r;i++)res.m[i][i]=1;
    while(b){
      if(b&1)res=res.mul(*this);
      *this=mul(*this);
      b/=2;
    }
    *this = res;
  }
};
void doit(){
  int t, tt;
  ll n;
  unsigned res=0;
  typedef MAT<252,252> mat;
  mat v(MODV,MODV);
  for(int i=0;i<MODV;i++){
    t=i*10;
    for(int j=1;j<10;j++)
      if((t+j)%j==0){
        tt=(t+j)%MODV;
        v.m[i][tt]++;
      }
  }
  n=getll();
  v.pow(n);
  for(int i=0;i<MODV;i++){
    res += v.m[0][i];
  }
  cout<<res<<endl;
}
