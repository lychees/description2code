#include<bits/stdc++.h>
using namespace std;
int main()
{
    int k,i,j,n,f=0;
    string hate,str;
    cin>>k;
    map<string,int> m;
    for(i=0;i<k;i++)
    {
        cin>>hate;
        m[hate]=1;
    }
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>str;
        char c=str[0];
        c=toupper(c);
        if(m[str]!=1)
        {
            if(f==1)
                cout<<".";
            else f=1;
            cout<<c;
        }
    }
}
