#include<iostream>
#include<stdio.h>
#include<math.h>
#include<assert.h>
#include<string.h>
#include<algorithm>
#include<vector>

using namespace std;
int main()
{
	int t, alpha[26], even, odd;
	cin >> t;
	string s[t];
	for (int i = 0; i < t; i++)
	{
		cin >> s[i];
	}

	for (int i = 0; i < t; i++)
	{
		fill_n(alpha, 26, 0);
		even = odd = 0;
		for (int j = 0; j < s[i].length(); j++)
		{
			assert(s[i].at(j) >= 97 && s[i].at(j) <= 122);
			alpha[s[i].at(j) - 97] ++;
		}
		for (int k = 0; k < 26; k++)
		{
			if ( alpha[k] % 2 == 0)
			{
				even ++;
			}
			else
				odd ++;

		}
		if(odd == 0)
			cout << odd << endl;
		else if(odd == 1)
			cout << "0" << endl;
		else
			cout << odd - 1 << endl;
		
	}
}
