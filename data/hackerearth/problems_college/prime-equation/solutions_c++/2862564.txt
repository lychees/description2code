void getPrimes(int n,vector<int> &v)
{
    ll i,j;
    for(i=3;i<=n;i+=2)
    {
        if(grid[i])
            continue;
        v.pb(i);

        for(j=i*i;j<=n;j+=i+i)
            grid[j]=1;
    }
}
