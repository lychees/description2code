#include <bits/stdc++.h>
using namespace std;
int main()
{
    int a[100005];
    int n;
    int b[100005];
    int i;
    int m;
    m = 0;

    cin>>n;
    for(i = 1; i <= n; i++) {
        cin>>a[i];
    }
    b[0] = 0;
    for(i = 1; i <= n; i++) {
        b[i] = max(b[i-1]+a[i],a[i]);
    }
    for(i = 0; i <= n; i++) {
        if(b[i] > m) {
            m = b[i];
        }
    }
    cout<<m;

}
