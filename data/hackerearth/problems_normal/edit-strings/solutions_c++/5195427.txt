#include <iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
using namespace std;

int main()
{
    int i,j,k,q,num,n,l;
    cin>>l>>q; string str; cin>>str;
    i=0; int arr[l]; memset(arr,0,sizeof(arr)); int count=0;
    while(str[i]!='.'  &&  i!=str.size())
    {i++;}
    if(i!=str.size())
    { i++;
    for(;i<str.size();i++)
    {
    	if(str[i]=='.')
    	{ if(str[i-1]=='.')
    		arr[i]=arr[i-1]+1;
    		else
    		arr[i]=0;
    	}
    	else{
    		count+=arr[i-1];
    	}
    }
    count+=arr[l-1];
    } char c; 
    for(num=0;num<q;num++)
    {
    	cin>>n>>c; n=n-1;
    	if(str[n]!='.' && c=='.')
    	{ str[n]='.';
    		if(n>0)
    		{
    			if(str[n-1]=='.')
    			count++;
    		}
    		if(n<str.size()-1)
    		{
    			if(str[n+1]=='.')
    			count++;
    		}
    	}
    	else if(str[n]=='.' && c!='.')
    	{
    		 str[n]=c;
    		if(n>0)
    		{
    			if(str[n-1]=='.')
    			count--;
    		}
    		if(n<l-1)
    		{
    			if(str[n+1]=='.')
    			count--;
    		}
    	}
    	cout<<count<<endl;
    }
    
    return 0;
}
