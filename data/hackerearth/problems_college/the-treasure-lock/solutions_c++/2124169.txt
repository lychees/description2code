#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int T;
    int N;
    cin>>T;
    while(T--){
    	cin>>N;
    	if(N<3 && N<5){
    		cout<<"-1"<<endl;
    	}
    	else if(N%3==0){
    		for(int i=0;i<N;i++){
    			cout<<"5";
    		}
    		cout<<endl;
    	}
    	else if(N%5==0){
    		for(int i=0;i<N;i++){
    			cout<<"3";
    		}
    		cout<<endl;
    	}
    	else {
    		int y = N;
    		while(y){
    			if(y%3!=0 || (N-y)%5!=0){
    				y--;
    			}
    			else 
					break;
    		}
    		//cout<<y<<endl;
    				int p = max(y,N-y);
    				//cout<<p<<endl;
    				for(int i=0;i<p;i++){
    					cout<<"5";
    				}
    				for(int i=0;i<N-p;i++){
    					cout<<"3";
    				}
    				cout<<endl;
    			
    	}
    }
    return 0;
}
