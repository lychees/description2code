#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int key;
		scanf("%d",&key);
		string s;
		cin>>s;
		int i;
		for(i=0;i<s.length();i++)
		{
			if(s.at(i)>='a' && s.at(i)<='z')
			s.at(i)=('a'+(s.at(i)-'a'+ key)%26);
			else
			s.at(i)=('A'+(s.at(i)-'A'+ key)%26);
		}
		cout<<s<<endl;
	}
	return 0;
}
