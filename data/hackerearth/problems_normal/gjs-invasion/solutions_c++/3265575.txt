#include <bits/stdc++.h>
using namespace std;
long long int ways[1<<18];
long long int s[1<<18];
long long int fun(int n)
{
	for(int i=0;i<(1<<n);i++)
	{
		s[i]=0;
		ways[i]=0;
	}
	ways[0]=1;
	for(int mask=0;mask<pow(2,n);mask++)
	{
		for(int i=0;i<n;i++)
		{
			if(!(mask&(1<<i)))
			{
				if(__builtin_popcount(mask)!=i)
				{
					long long int bigger=0;
					for(int j=i+1;j<n;j++)
					{
						if(mask&(1<<j))
						bigger++;
					}
						ways[mask|(1<<i)]=ways[mask]+ways[mask|(1<<i)];
						s[mask|(1<<i)]=s[mask|(1<<i)]+bigger*ways[mask]+s[mask];
					}
				}
			}
		}
	
	return s[(1<<n)-1];
}
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		long long int ans=fun(n);
		cout<<ans<<"\n";
	}
	return 0;
}
