#include <iostream>
#include <map>
#define ll long long int
#define x 1000000007
using namespace std;
map <ll,ll> m;
map <ll,ll> :: iterator it;
int main()
{
    ll number,n,ans,i;
    cin>>number;
    while(number--)
    {
    	cin>>n;
    	for(i=2; i*i<=n;)
    	{
    		if(n % i == 0)
    		{
    			m[i]++;
    			n = n/i;
    		}
    		else
    		{
    			i++;
    		}
    	}
    	if(n > 1)
    	{
    		m[n]++;
    	}
    }
    ans=1;
    for(it=m.begin(); it!=m.end(); it++)
    {
    	ans = ((ans%x)*(it->second + 1)%x)%x;
    }
    cout<<ans;
    return 0;
}
