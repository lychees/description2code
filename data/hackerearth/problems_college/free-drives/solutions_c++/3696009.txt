#include<iostream>
#include<map>
using namespace std;
int main()
{
    map<int,int> m;
    m[0] = 0;
    m[1] = 0;
    for(int i = 1; i <= 100000; i++)
    {
        m[i] = (i-1) + m[i-1];
    }
    int t;
    cin >> t;
    while(t--)
    {
        int a;
        cin >> a;
        cout << m[a] << endl;
    }
    return 0;
}
