#include<iostream>
using namespace std;

int arr[1000000];

int main()
{
	int num,test;
	cin>>test;
	while (test--)
	{
		cin>>num;
		int i,j;
		int pair=0;
		for (i=0;i<num;i++)
		{
			cin>>arr[i];
			for (j=i;j>=0;j--)
				if (arr[i]==arr[j]) pair++;
		}
		cout<<pair<<endl;
	}
	return 0;
}
