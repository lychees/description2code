#include<iostream>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;

main()
{
    long long int n,i,j;
    cin>>n;
    for(i=4; i<=n ;i++)
    {
        int c=0;
        for(j=2; j<=sqrt(i);j++)
        {
            if(i%j==0) {  c=1; break;  }
        }
        if(c==1) cout<<i<<' ';
    }
    return 0;
}
