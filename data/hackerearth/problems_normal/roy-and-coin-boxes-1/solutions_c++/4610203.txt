void fs(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int n,m;


int tree[4000000];
int lazy[4000000];
int cnt[1000001];

void update(int node, int st, int en, int l, int r)
{
    if(l>en || r<st) return;
    if(l<=st && en<=r)
    {
        lazy[node]++;
        return;
    }
    update(node*2, st, (st+en)/2, l, r);
    update(node*2+1, (st+en)/2+1, en, l , r);
}

void flat(int node, int st, int en)
{
    if(st!=en)
    {
        lazy[node*2]+=lazy[node];
        lazy[node*2+1]+=lazy[node];
        lazy[node]=0;
        flat(node*2, st, (st+en)/2);
        flat(node*2+1, (st+en)/2+1, en);
        return;
    }
    cnt[lazy[node]]++;
    
}
