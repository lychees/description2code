#include <iostream>
#include <unordered_map>
#include <vector>
#include <algorithm>
using namespace std;

int n, t;
unordered_map <long long, short int> Repetitions;
vector <long long> Ramanujan;

int main() {
    ios_base::sync_with_stdio(0);
    for (int i = 1; i <= 1000; i++) {
        for (int j = i + 1; j <= 1000; j++) {
            long long id = (i * i * i) + (j * j * j);
            Repetitions[id]++;
            if (Repetitions[id] == 2) {
                Ramanujan.push_back(id);
            }
        }
    }
    sort(Ramanujan.begin(), Ramanujan.end());
    cin >> t;
    for (int i = 0; i < t; i++) {
        cin >> n;
        cout << Ramanujan[n - 1] << endl;
    }
    return 0;
}
