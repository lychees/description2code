#include <iostream>
using namespace std;

int gcd(int a, int b)
{
	if (a<b) return gcd(b, a);
	return (b == 0 ? a : gcd(b, a%b));
}

void print_factors(int n)
{
	for (int i = 2; i <= n / 2; i++)
	{
		if (n%i == 0)
			cout << i << " ";
	}
	cout << n;
}

int main()
{
	int n; cin >> n;
	int a, b, g = 0; cin >> a;
	for (int i = 1; i<n; i++)
	{
		cin >> b; g = gcd(g, abs(b - a)); a = b;
	}
	print_factors(g);
	return 0;
}
