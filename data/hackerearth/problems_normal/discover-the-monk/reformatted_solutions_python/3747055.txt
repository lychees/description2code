a = raw_input()
a = a.split()
a = map(int, a)
b = raw_input()
b = sorted(map(int, b.split()))


def search(n):
    low = 0
    high = len(b)-1
    while high >= low:
        mid = (high+low)/2
        if b[mid] == n:
            return True
        elif b[mid] < n:
            low = mid+1
        elif b[mid] > n:
            high = mid-1
    return False


while a[1] > 0:
    c = input()
    if search(c):
        print 'YES'
    else:
        print 'NO'
    a[1] = a[1]-1
