#include <iostream>
#include <iomanip>

using namespace std;

#define zero 0.0000000000

double speed[1001], fuel[1001];

int main()
{
    
    int n;
    cin>>n;
    
    for(int i = 0; i < n; ++i)
    	cin>>speed[i];
    for(int i = 0; i < n; ++i)
    	cin>>fuel[i];
    double total;
    cin>>total;
    
    double ans = zero;
    
    for(int i = 0; i < n; ++i) {
    	ans = max(ans, speed[i]*(total/fuel[i]));
    }
    
    int anss = ans * 10000.0;
    anss /= 10;
    
    ans = (double)anss;
    ans /= 1000.0;
    cout<<fixed<<setprecision(3)<<ans;
    
    return 0;
}
