#include<iostream>
#include<sstream>
#include<string.h>
using namespace std;

string longToString(long n)
{
    stringstream mystream;
    mystream << n;
    return (mystream.str());
}

long stringToLong(string str)
{
    long n;
	stringstream sstr(str);
    sstr >> n;
    return n;
}
long reverse(long n)
{
    string sumString = longToString(n);   
    char temp;
    int len = sumString.length();
    for(int i=0 ; i<len/2 ; i++)
    {
            temp = sumString[i];
            sumString[i]=sumString[len-1-i];
            sumString[len-1-i]=temp;
    }
    return (stringToLong(sumString));
}

int main()
{
    long *numbers,n,sum;
    stringstream mystream;
    cin>>n;
    numbers = new long[2*n];
    //long *flippedNumbers = new long[n];
    for(int i =0; i<2*n; i+=2)
    {
            cin>>numbers[i]>>numbers[i+1];
            numbers[i] = reverse(numbers[i]);
            numbers[i+1] = reverse(numbers[i+1]);
            sum = numbers[i]+numbers[i+1];
            cout<<reverse(sum)<<"\n";
    }
}
