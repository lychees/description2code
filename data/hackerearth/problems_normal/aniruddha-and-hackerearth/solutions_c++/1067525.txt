#include <iostream>
#define MOD 1000000007
using namespace std;

long long sum[1000001];
int main()
{
    long long preve=1,prevh=1,newe,newh;
    sum[1]=2;
    for(int i=2;i<=1000000;++i)
    {
    	newh=(preve+prevh)%MOD;
    	newe=prevh;
    	sum[i]=(sum[i-1]+newh+newe)%MOD;
    	preve=newe;
    	prevh=newh;
    }
    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<sum[n]<<endl;
    }
    return 0;
}
