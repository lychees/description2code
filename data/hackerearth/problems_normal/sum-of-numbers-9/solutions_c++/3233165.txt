#include <iostream>
using namespace std;
long a[15];
int sum(int su,int s,int n )
{
	int temp=0;
	if(su==s)
		return 1;
	if(n==-1)return 0;
	
	//else if(su>s)
	//	return 0;
	else 
	{
		
		temp = sum(su+a[n],s,n-1);
		if(temp==1)return 1;
		temp = sum(su,s,n-1);
		if(temp==1)return 1;
		return 0;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		
		int n;
		cin>>n;
		for(int i = 0;i<n;i++)
		{
			cin>>a[i];
		}
		int s;
		cin>>s;
		if(sum(0,s,n-1))
			cout<<"YES"<<endl;
		else 
			cout<<"NO"<<endl;
	}
	return 0;
}
