#include<bits/stdc++.h>
using namespace std;
int pr[100002];
int sum[100002];
void prime()
{
pr[1]=1;
for(int i=2;i<=sqrt(100002);i++)
{
   if(pr[i]==0)
   {
     for(int j=2*i;j<100002;j+=i)
      pr[j]=1;

   }
}

}

void calsum()
{
   sum[1]=1;
  for(int i=2;i<100002;i++)
  {
     if(pr[i]==0)
       sum[i]=sum[i-1];
     else
      sum[i]=sum[i-1]+1;


  }


}



main()
{
prime();
calsum();

int t;
cin>>t;
for(int z=0;z<t;z++)
{
int n,m;
cin>>n>>m;
int ar[100002];
for(int i=0;i<n;i++)
cin>>ar[i];
sort(ar,ar+n);
int avail[100002];
for(int i=0;i<n;i++)
   {avail[i]=sum[m]-sum[ar[i]-1];
    //cout<<avail[i]<<endl;
    }

   long long int ans=1;
   int co=0;
   for(int i=n-1;i>=0;i--)
   {
      ans=ans*(avail[i]-co)%1000000007;
      co++;
   }

   cout<<ans<<endl;

}

}
