#include<bits/stdc++.h>
using namespace std;
int dp[1000]= {0};

int fn(int n)
{
    if(n<=3)return n;
    if(dp[n]!=0)return dp[n];
    dp[n]=fn(n-1)+fn(n-2);
    return dp[n];
}
int main()
{
    int t;
    cin>>t;
    dp[1]=1;
    dp[2]=2;
    dp[3]=3;
    while(t--)
    {
        int n;
        cin>>n;

        if(n<=3)
        {
            cout<<n<<endl;
            continue;
        }
        int res=fn(n);
        cout<<res<<endl;
    }
}
