#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{

	char inp[150];

		{
			int p=inp[0];
			printf("%d\n",p);
		}
		else
		{
		
		while(i<(length/2))
		{
			//printf("%d %d %c %c\n",i,j,inp[i],inp[j]);

			{
				flag=0;
				break;
			}
			else
			{
				flag=1;
			}
			i++;
			j--;
		}
		if(flag==1) 
		{
			int k;
			k=inp[length/2];
			printf("%d\n",k);
		}//printf("yesy\n");

}
