#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int a, b, c=0, tt=0, i, j;
    	cin>>a>>b;
    	for(;a>0,b>0;a=a/10,b=b/10)
    	{
    		if(c==0)
    		{
    			i=a%10;
    			j=b%10;
    			//cout<<"A = "<<i<<endl;
    			//cout<<"B = "<<j<<endl;
    			if(i+j>=10)
    			{
    				tt++;
    				c=1;
    			}
    			else 
    			c=0;
    			//cout<<"TT = "<<tt<<endl;
    			//cout<<"C = "<<c<<endl;
    		}
    		else if(c==1)
    		{
    			i=a%10;
    			j=b%10;
    			//cout<<"A = "<<i<<endl;
    			//cout<<"B = "<<j<<endl;
    			if(a+b+1>=10)
    			{
    				//cout<<"Inside Carry"<<endl;
    				tt++;
    				c=1;
    			}
    			else 
    			c=0;
    			//cout<<"TT = "<<tt<<endl;
    			//cout<<"C = "<<c<<endl;
    		}
    	}
    	if(tt==0)
    	cout<<"No carry operation"<<endl;
    	else if(tt==1)
    	cout<<tt<<" carry operation"<<endl;
    	else
    	cout<<tt<<" carry operations"<<endl;
    }
return 0;
}
