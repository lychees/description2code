#include <stdio.h>
#include<math.h>
#include<iostream>
using namespace std;
long long a[10000001]={}, d[10000001]={};
int main() {
	// your code goes here
	long long i, j, k, n, x, l, c=0;
//	cout<<sqrt(10000000);//3162.8
	for(i=2; i<=3162; i++){
		k=i*i;
		if(a[k]==0)
		for(j=k; j<=10000000; j+=k){
			a[j]=1;
		}
	}
//	for(i=0; i<=100; i++)cout<<i<<" "<<a[i]<<endl;
	scanf("%lld", &n);
	for(i=0; i<n; i++){
		scanf("%lld", &x);
		if(d[x]!=0)c+=d[x];
		else
		{
			k=sqrt(x);
			for(j=1; j<=k; j++){
				if(x%j==0){
					if(a[j]==0)d[x]++;
					l=x/j;
					if(l!=j&&a[l]==0)d[x]++;
				}
			}
			c+=d[x];
		}
	}
	printf("%lld", c);
	return 0;
}
