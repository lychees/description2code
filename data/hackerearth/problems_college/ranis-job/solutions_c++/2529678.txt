#include <iostream>
#include<bits/stdc++.h>

using namespace std;
bool fun(int i,int j)
{
	return (i>j);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,i;
    	int sum=0;
    	cin>>n;
    	int a[n];
    	int c[n];
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    		sum=sum+a[i];
    	}
    	for(i=0;i<n;i++)
    	cin>>c[i];
    	sort(c,c+n,fun);
    	int count=1;
    	for(i=0;i<n;i++)
    	{
    		sum=sum-c[i];
    		if(sum>0)
    		count++;
    		else
    		break;
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
