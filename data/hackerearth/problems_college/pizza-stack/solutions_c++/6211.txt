#include <stdio.h>
#include<vector>
using namespace std;
 
//int level;
 
 
int mod=1000000007;
 
int main() {
long long stack[1000][1000];
int Test,i,N,j;
long long sum;
 
 
//int stack[1000][1000];
for(i=0;i<1000;i++)
for(j=0;j<1000;j++)
stack[i][j]=0;
for(i=0;i<1000;i++)
stack[i][0]=1;
for(i=1;i<1000;i++)
stack[i][i]=1;
for(i=1;i<1000;i++)
{
for(j=1;j<i;j++)
{
stack[i][j]=((stack[i-1][j-1]+(j+1)*stack[i-1][j]))%mod;
}
}
scanf("%d",&Test);
for(i=0;i<Test;i++)
{
scanf("%d",&N);
sum=0;
for(j=0;j<N;j++)
sum=(sum+stack[N-1][j])%mod;
//printf("%ld\n",sum);
sum=sum%mod;
printf("%ld\n",sum);
}
return 0;
}
