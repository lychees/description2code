#include <iostream>
#include <string>
#include <cstring>

#define sz(a) int(a.size())
using namespace std;

char vowel[] = "aeiouy";
string s;
int consonants, vowels, slozhna;


bool consonant (char x) {
	for (int i = 0;i < 6;i ++)
		if (vowel[i] == x)	return 0;
	return 1;	
}	
bool condition1 () {
	for (int i = 0;i < sz(s) - 2;i ++)
		if (consonant (s[i]) && consonant(s[i + 1]) && consonant (s[i + 2]))
			return 1;
	return 0;
}
int main()
{
	int t;	cin >> t;
	while (t --) {
		cin >> s;
		slozhna = 0;
		consonants = 0, vowels = 0;
		slozhna |= condition1 ();
		for (int i = 0;i < sz(s);i ++) {
			if (consonant (s[i]))	consonants ++;
					else	vowels ++;
		}
		slozhna |= (consonants > vowels);
		cout << (slozhna ? "hard" : "easy") << endl;
	}
   return 0;
}
