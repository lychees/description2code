#include<iostream>
#include<cstdio>


#include<map>
using namespace std;

string a,b;
void change(int n)
{
     int i;
     string c;
     c.resize(n);
     for(i=0;i<n/2;i++)
     {
        c[(i<<1)] = a[i];
        c[(i<<1)+1] = a[n-i-1];
     }
     if(n%2==1)
     {
        c[n-1] = a[n/2];
     }
     a = c;
    
}
int main()
{
    int k,n;
    scanf("%d",&k);
    cin>>a;
    b = a;
    n = a.size();
    change(n);
    int p = 1;
    while(a!=b)
    {
       p++;
       change(n);
    }
    k=(p-k%p);
    a = b;
    while(k--)
    {
       change(n);
    }
    cout<<a<<endl;
   
    return 0;
}
