#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int f=0;f<t;f++)
    {
    	string a;
    	cin>>a;
    	int n=a.length();
    	int b[26];
    	for(int i=0;i<26;i++)
    	   b[i]=0;
    	for(int i=0;i<n;i++)
    	{
    		b[a[i]-97]++;
    	}
    	int count=0;
    	for(int i=0;i<26;i++)
    	{
    		if(b[i]%2)
    		{
    			cout<<-1<<endl;
    			count=-1;
    			break;
    		}
    	}
    	if(count==-1)
    	  continue;
    	else
    	  cout<<1<<endl;
    }
    return 0;
}
