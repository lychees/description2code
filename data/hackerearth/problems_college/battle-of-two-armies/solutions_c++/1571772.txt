#include<bits/stdc++.h>
using namespace std;
int prime(int n) {
	for(int i=2;i<=sqrt(n);i++) {
		if(n%i==0) {
			return 0;
		}
	}
	return 1;
	
}
int main() {
	int t,n;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		int p[n],c[n],i;
		for(i=0;i<n;i++) {
			scanf("%d",&c[i]);
		}
		for(i=0;i<n;i++) {
			scanf("%d",&p[i]);
		}
		for(i=0;i<n;i++) {
			int x=prime(c[i]);
			int y=prime(p[i]);
			if((x==0 &&  y==1) || (x==1 && y==0)) {
				if(c[i]>p[i]) {
					c[i]-=p[i];
					p[i]=0;
				}
				else {
					p[i]-=c[i];
					c[i]=0;
				}
			}
			else if(x==0) {
				p[i]=0;
			}
			else if(y==1) {
				c[i]=0;
			}
		}
		int sc=0,sp=0;
		for(i=0;i<n;i++) {
			sc+=c[i];
		}
		for(i=0;i<n;i++) {
			sp+=p[i];
		}
		if(sc>sp) {
			printf("The Composites\n");
		}
		else if(sp>sc) {
			printf("The Primes\n");
			
		}
		else {
			printf("Tie\n");
		}
	}
	return 0;
}
