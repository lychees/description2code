#include <iostream>
#include <string>

using namespace std;

int main()
{
    int t,d[62],i;static int j;
    i='a';
    while(i<='z')
    {d[j]=i;
       i++;j++;
    }
    i='A';
    while(i<='Z')
    {d[j]=i;
       i++;j++;
    }
    i='0';
    while(i<='9')
    {d[j]=i;
       i++;j++;
    }

  cin>>t;
    while(t--)
    {string a;int l,c=0,k;
    cin>>a;
    l=a.length();
    for(k=0;k<62;k++)
    for(i=0;i<l;i++)
    {if(d[k]==a[i])
       {
        c++;break;
       }

    }
    cout<<(c*(c+1))/2<<endl;

    }
    return 0;
}
