#include <iostream>
using namespace std;
int ar[100001];
int fun(int x, int lo,int hi)
{
	if(hi<lo)
	{cout<<"impossible";}
	if(lo==hi&&x<ar[lo])
	return hi;
	
	if(x==ar[lo])
	return lo;
	
	if (x==ar[hi])
	return hi;
	
	if(hi==lo+1&&ar[hi]>x&&ar[lo]<x)
	{
		return hi;
	}
	
	int mid=lo+(hi-lo)/2;
	if(x>ar[mid]&&x<ar[hi])
	{
		fun(x,mid,hi);
	}
	else if(x<ar[mid]&&x>ar[lo])
	{
		fun(x,lo,mid);
	}
	else if(x==ar[mid])
	{
		return mid;
	}
	
}
int main()
{
    int n,q;
    cin>>n;
    ar[0]=0;
    for(int i=1;i<=n;i++)
    {
    	int x;
    	cin>>x;
    	ar[i]+=ar[i-1]+x;
    }
    //for(int i=0;i<n;i++)
    //cout<<ar[i]<<" ";
    //cout<<endl;
    cin>>q;
    while(q--)
    {
    	int x;
    	cin>>x;
    	cout<<fun(x,0,n)<<endl;
    }
}
