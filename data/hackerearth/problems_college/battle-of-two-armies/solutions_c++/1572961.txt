bool isprime(int n)
{
    int val = n,i;
    for(i=2;i<n;++i)
    {
        if(val%i==0)
        {
            return false;
        }
    }
    return true;
}

int main()
{
    int t,n,c[200],p[200],w[200],i,fl;
    cin>>t;
    while(t--)
    {
        memset(w,0,sizeof(w));
        cin>>n;
        for(i=0;i<n;++i)
        {
            cin>>c[i];
        }
        for(i=0;i<n;++i)
        {
            cin>>p[i];
        }

        for(i=0;i<n;++i)
        {
            fl = 0;
            if(!isprime(c[i]))
            {
                fl ++;
            }
            if(isprime(p[i]))
            {
                fl--;
            }

            if(fl==1)
            {
                w[i] = 3;
            }
            else if(fl==-1)
            {
                w[i] = 4;
            }
            else
            {
                if(p[i]>c[i])w[i]=1;
                else if(p[i]<c[i]) w[i] = 2;
                else w[i] = -1;
            }
        }
        int cntc,cntp;
        cntc=cntp=0;
        for(i=0;i<n;++i)
        {
            if(w[i]==2)cntc+=c[i]-p[i];
            else if(w[i]==1) cntp+=p[i]-c[i];
            else if(w[i]==3) cntc+=c[i];
            else if(w[i]==4) cntp+=p[i];
        }
        //cout<<cntc<<" "<<cntp<<"\n";
        if(cntp<cntc)
        {
            cout<<"The Composites\n";
        }
        else if(cntp>cntc)
        {
            cout<<"The Primes\n";
        }
        else
        {
            cout<<"Tie\n";
        }
    }
    return 0;
}
