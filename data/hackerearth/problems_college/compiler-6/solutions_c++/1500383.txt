string s;
LL solve(int start , int last)
{
	LL ret = 0;
	int i =0;
	for(int i=start ; i<=last;i++)
	{
		if(s[i]=='A')
		{
			ret++;
			while(s[i]!=';')
			 i++;
		}

		if(i==s.size()-1)
		{
		break;
		}
		if(s[i]=='l')
		{
			while(s[i] != '(')
			 i++;
			 i++;
			LL tval =0;

			while(s[i] != ')')
			{
				tval = tval*10 + (s[i]-'0');
				i++;
			}
			i++;i++;
			
			LL opb = 1 , clb = 0;
			string passtr = "";
			int ss = i;
			while(1)
			{
				if(s[i] ==  '{')
					opb++;
				if(s[i] == '}' && opb-clb == 1)
				{
					break;
				}
				else if(s[i] == '}')
				{
					clb++;
				}
				passtr += s[i];
				i++;
			}
			int ee = i-1;
			if(tval != 0)
			{
				ret = ret + tval*solve(ss,ee);
			}
		}
	}
	return ret;
}
