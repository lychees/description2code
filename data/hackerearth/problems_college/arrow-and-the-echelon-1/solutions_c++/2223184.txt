#include<iostream>
#include<bits/stdc++.h>

using namespace std;


int main ()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int a,i,j,k=0,l=0;
		cin>>a;
		for(i=0;i<=a/5;i++)
		{
			for(j=0;j<=a/3;j++)
			{
				if(i*5+j*3==a)
				{
						for(k=0;k<j;k++)
					     cout<<"555";
					for(k=0;k<i;k++)
					     cout<<"33333";
				
					
					     cout<<endl;
					     k=-1;
					     l=1;
					     break;
					
				}
				
			}if(l==1)
			break;
		}
		if(k!=-1)
		{
			cout<<"-1\n";
		}
	}



	
	return 0;
}
