#include<bits/stdc++.h>
using namespace std;
int numOfBits(int n){
	return (n==0)? 0:(n%2)+numOfBits(n/2);
}
int main(){
	int t; cin >> t; while(t--){
		int n; cin >> n;
		int arr[n]; for(int i=0;i<n;i++) cin >> arr[i];
		sort(arr,arr+n);
		bool ans=false;
		for(int i=0;i<n;i++){
			int currN=arr[i];
			int currBits = numOfBits(arr[i]);
			for(int j=i;j<n;j++){
				int temp = numOfBits((currN&arr[j]));
				if(temp==1){
					ans=true;
					break;
				}else if(temp<=currBits && temp!=0){
					currBits=temp;
					currN=(currN&arr[j]);
				}
			}
		}
		(ans)?cout<<"YES"<<endl:cout<<"NO"<<endl;
	}
	return 0;
}
