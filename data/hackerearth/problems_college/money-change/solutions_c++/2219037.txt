#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	long long sum[251];
	int coins[51],n,m;
	memset(sum,0,sizeof(sum));
    sum[0]=1;
    cin>>n>>m;
    for(int i=0;i<m;i++)
    	cin>>coins[i];
 
    	
    for(int i=0;i<m;i++){
    	for(int j=coins[i];j<=n;j++)
    		sum[j]+=sum[j-coins[i]];
    }
    
    cout<<sum[n];
return 0;
}
