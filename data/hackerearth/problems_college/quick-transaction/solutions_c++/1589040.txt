#include<iostream>
#include<map>
#include<cstdio>
using namespace std;
#define tr(c,it) \
for(typeof(c.begin())it=c.begin();it!=c.end();it++)
int main()
{
    map<int,int> m;
    int t,r,countn;
    long long x;
    scanf("%d",&t);
    while(t--)
    {
        countn=0;
        scanf("%lld",&x);
        r=x/1000;

        m[1000]=r;
        x=x-r*1000;
        r=x/500;

        m[500]=r;
        x=x-r*500;
        r=x/100;
        m[100]=r;
        x=x-(r*100);
        r=x/50;

        m[50]=r;
        x=x-r*50;
        r=x/20;
        m[20]=r;
        x=x-r*20;
        r=x/10;

        m[10]=r;
        x=x-r*10;
        r=x/5;

        m[5]=r;
        x=x-r*5;
        r=x/2;

        m[2]=r;
        x=x-r*2;
        r=x/1;

        m[1]=r;
        for(map<int,int>::iterator it=m.begin();it!=m.end();it++)
        {
            countn+=it->second;
        }
        printf("%d\n",countn);
    }
    return 0;
}
