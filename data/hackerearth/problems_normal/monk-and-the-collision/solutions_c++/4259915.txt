#include <bits/stdc++.h>

using namespace std;



int main(){
	int t;
	cin >> t;
	if(t >10 ) return 0;
	while(t--){
		int arr[10005] = {0};
	int n,x;
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> x;
		int index = x % 10;
		arr[index]++;
	}
	int count =0;
	for(int i=0;i<10005;)
	{
		if(arr[i]>1){
			count++;
			arr[i]--;
		}
		else
			i++;
	}
	cout << count << endl;
}
	return 0;
}
