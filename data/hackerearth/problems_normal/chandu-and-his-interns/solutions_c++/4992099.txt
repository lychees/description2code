bool isPrime(int n)
{
    bool flag= true;
    double d=sqrt(n);
    int a=(int)d;
    if((n==1)||(n==2))
        return true;
    else
    {
        for(int i=2;i<=a;i++)
        {
            if(n%i==0)
            {
                flag= false;
                break;
            }
        }
    }
    return flag;
}
bool isPerfectSquare(int n)
{
    bool fire = false;
    double f=sqrt(n);
    if(f==((int)f))
    {
        if(isPrime((int)f))
        {
            fire = true;
        }
    }
    return fire;

}

int main( )
