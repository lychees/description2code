#include <iostream>
using namespace std;
#include<algorithm>
bool comp(int i,int j){return (i>j);}
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    cin>>a[i];
    int m;
    cin>>m;
    sort(a,a+n,comp);
    int ans=0;int x=0;
    for(int i=0;i<n;i++)
    {
    	if(x>m)
    	break;
    	
    	x=x+a[i];
    	ans++;
    }
    
    cout<<ans<<endl;
    
    return 0;
}
