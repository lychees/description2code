#include <iostream>

using namespace std;

int main()
{
    long long int i,k,l,h,m,s,n,q,a[100005],x,y;
    cin>>n;
    s=0;
    for(i=0;i<n;i++)
    {
        cin>>x;
        s=s+x;
        a[i]=s;
    }
    cin>>q;
    for(i=0;i<q;i++)
    {
        cin>>k;
        if(k>s)
        {
            cout<<"-1\n";
            continue;
        }
        else
        {
        l=0;h=n-1;
        while(l<=h)
        {
            m=(l+h)/2;
            if(a[m]==k)
            {
                y=m;
                break;
            }
            else
            {if(a[m]<k)
                l=m+1;
                else
                {
                    y=m;
                    h=m-1;
                }

        }
        }

    }
    cout<<y+1<<"\n";
    }
    return 0;
}
