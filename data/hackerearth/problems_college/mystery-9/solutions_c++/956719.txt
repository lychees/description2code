#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int mon[12] = {31,28,31,30,31,30,31,31,30,31,30,31};
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int sum = 0;
    	for(int i=0;i<12;i++)
    	{
    		sum = sum + mon[i];
    		if(sum>=n)
    		{
    			printf("%d\n",i+1);
    			break;
    		}
    	}
    }
    return 0;
}
