n = input()
s = map(int, raw_input().split())
q = input()
for _ in xrange(q):
    k = input()
    love = False
    if k in s:
        love = True
    else:
        for i in range(len(s)-1):
            # print "i=",i
            # print "minus=",k-s[i],"s[i:]=",s[i:]
            if k-s[i] in s[i+1:]:
                love = True
                break
    if love:
        print "I Like You"
    else:
        print "I Hate You"
