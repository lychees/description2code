
#include <bits/stdc++.h>


using namespace std;

int cow[100],diff[100];

main()
{
    ios_base::sync_with_stdio(false);
    int m,s,c,i,stalls,k;
    cin>>m>>s>>c;
    for(i=0;i<c;i++)
    {
        cin>>cow[i];
    }

    sort(cow,cow+c);
    if(c<=m)
        cout<<c;
    else{
    for(i=0;i<c-1;i++)
    {
        diff[i]=cow[i+1]-cow[i];
    }
    sort(diff,diff+(c-1));
    stalls=cow[c-1]-cow[0]+1;
    //cout<<stalls<<endl;
    k=c-2;
    for(i=1;i<m;i++)
    {
        stalls=stalls-diff[k]+1;
        k--;
    }
    cout<<stalls;
    }
	return 0;
}
