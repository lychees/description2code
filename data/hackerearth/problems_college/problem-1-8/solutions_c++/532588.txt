#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	string s, substring;
	int n, ans=-1, x, y, flag;
	
	cin>>s;
	cin>>n;
		
	if(n==0){
		cout<<s<<endl;
		return 0;
	}
	
	string k[n];
		
	for(int i=0 ; i<n ; i++){
		cin>>k[i];
	}
	
	
	ans = 0;
	
	for(int i=0 ; i<n ; i++){
		if(i==ans){
			continue;
		}
		
		for(int j=s.length() ; j<k[i].length() ; j++){
			x = k[i][j];
			y = k[ans][j];
		
			if(x < y){
				ans = i;
				break;
				flag = 0;
			}
		}
	}

	substring = k[ans].substr(0, s.length());
	
	if(s.compare(substring)!=0){
		cout<<s;
	} else {
		cout<<k[ans];
	}
	
	return 0;
}
