#include <iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;

int main()
{
	char st[1000000];
    int t , temp,found=0 , length;
    cin>>t;
    cin.ignore(1);
    while(t--)
    {
    	found=0;
    	cin.getline(st,1000000);
    	length=strlen(st);
    	temp=atoi(st);
    	for(int i=0;i<length;i++)
    	{
    		if(st[i]=='2')
    		if(st[i+1]=='1')
    		found=1;
    	}
    	if(temp%21==0)
    	found=1;
    	
    	if(found==1)
    	cout<<"The streak is broken!"<<endl;
    	else
    	cout<<"The streak lives still in our heart!"<<endl;
    }
    return 0;
}
