#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 101000
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18)
#define mod 1000000007
using namespace std;
long long int  arr[1050][1050];
int main(int argc, char const *argv[])
{
  int t;int a,b,c,d;
  cin>>t;
  arr[1][1]=1;
  for(int i=1;i<=1006;i++)
  {
    for(int j=1;j<=1006;j++)
    {
      arr[i][j] = (arr[i][j]+arr[i-1][j]+arr[i][j-1]+arr[i-1][j-1])%mod;
    }
  }
  ll ans;
  while(t--)
  {  ans=0;
     scanf("%d%d%d%d",&a,&b,&c,&d);
     c-=a-1;
     d-=b-1;
     if(c<0||d<0)
      ;
    else
      ans = arr[c][d];
    printf("%lld\n",ans);
  }
  return 0;
}
