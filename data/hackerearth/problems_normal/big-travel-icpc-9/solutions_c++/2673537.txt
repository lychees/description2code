#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>
using namespace std;

int main()
{
    long int n,i,j,p;
    cin>>n;
    
    long long int a[n],b[n];
    
    long long int sum=0,k=1000000007;
    
    for(i=0;i<n;i++)
    {
    	cin>>a[i]>>b[i];
    }
    sort(a,a+n);
    sort(b,b+n);
    p=n-1;
    for(i=n-1;i>=0;i--)
    {
    	sum=(sum+p*a[i])%k;
    	sum=(sum+p*b[i])%k;
    	p=p-2;
    }
    cout<<sum;
    return 0;
}
