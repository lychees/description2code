#include <iostream>
#include <map>
using namespace std;
map<string, long long > m;

int main()
{
    int n,q;
    cin>> n>>q;
    while(n--){
    	string s;long long cost;
    	cin>> s;
    	cin>> cost;
    	for(int i=1;i<=s.size();i++){
    	string sb = s.substr(0,i);
    	m[sb]= max(m[sb],cost);
    	}
    }
    while(q--){
    	string s;
    	cin>>s;
    	if(m.find(s)!= m.end()) cout << m[s]<<endl;
    	else cout<<-1<<endl;
    }
    return 0;
}
