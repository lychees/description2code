#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,i,j,k,l;
	scanf ("%d %d",&n,&k);
	int arr[n];
	for (i=0;i<n;++i)
		scanf ("%d",&arr[i]);
	for (i=0;i<n;++i)
	{
		int fmax=i;
		for (j=i+1;j<n;++j)
			if (arr[j]>=arr[fmax])
				fmax=j;
		if (fmax!=i)
		{
			l=arr[i];
			arr[i]=arr[fmax];
			arr[fmax]=l;
			k--;
		}
		if (k==0)
			break;
	}
	for (i=0;i<n;++i)
		printf ("%d ",arr[i]);
	printf ("\n");
    return 0;
}
