 int mini(int x, int y, int z) 
    {
       return min(min(x, y), z);
    }
int edit(string word1,string word2,int m ,int n )
    {
        int dp[m+1][n+1];
        for(int i = 0; i <= m; i++ )
        {
            for(int j = 0; j <= n; j++)
            {
                if(i==0)
                dp[i][j] = j;
                else if(j==0)
                dp[i][j] = i;
                else if(word1[i-1]==word2[j-1])
                dp[i][j] = dp[i-1][j-1];
                else
                {
                dp[i][j] =  1+ mini((dp[i][j-1]),dp[i-1][j-1],dp[i-1][j]);
                }
            }
        }
        return dp[m][n];
    }
