#include<iostream>
#include<vector>

using namespace std;


int main() {
  int t;
  int n;
  int m;
  vector<int> first;
  vector<int> second;

  cin >> t;
  while (t--) {
    cin >> n >> m;

    //first.clear();
   // second.clear();
    first.resize(n);
    second.resize(n);
    for (int i = 0; i < n; i++) {
      int tmp;
      cin >> tmp;
      first[i] = tmp;
    }
    int index = 0;
    for (int i = 0; i < m; i++) {
      for (int j = 0; j < n; j++) {
        if (index == 0) {
          if ((j == 0&&first[j+1]==1)||
            (j!=0&&j!=n-1&&first[j-1]==1&&first[j+1]==1)||
            (j==n-1&&first[j-1]==1)) {
            second[j] = 1;
          }
          else {
            second[j] = 0;
          }
        }
        else {
          if ((j == 0 && second[j + 1] == 1) ||
            (j == n - 1 && second[j - 1] == 1) ||
            (j != 0 && j != n - 1 && second[j - 1] == 1 && second[j + 1] == 1)) {
            first[j] = 1;
          }
          else {
            first[j] = 0;
          }

        }

      }
      index = (index + 1) % 2;
      bool consted = true;
      for (int k = 0; k < n; k++) {
        if (first[k] != second[k]) {
          consted = false;
          break;
        }
      }
      if (consted)
        break;
    }

    for (int i = 0; i < n; i++) {
      if (index == 0) {
        cout << first[i] << " ";
      }
      else {
        cout << second[i] << " ";
      }
    }
    cout << endl;
  }


  return 0;
}
