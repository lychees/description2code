def solution(ip):
    # ip = raw_input()
    ascii_for_z = int(ord('z'))
    ascii_for_a = int(ord('a'))
    result = []
    ip = list(ip)
    count = {}
    alphabet_ascii = ascii_for_z
    while alphabet_ascii >= ascii_for_a:
        if chr(alphabet_ascii) in ip:
            if chr(alphabet_ascii) not in count.keys():
                count[chr(alphabet_ascii)] = ip.count(chr(alphabet_ascii))
        else:
            result.append(chr(alphabet_ascii))
        alphabet_ascii = alphabet_ascii - 1
    count_values = count.values()
    count_values = list(set(count_values))
    count_values = sorted(count_values)
    while count_values:
        minimum_count = count_values.pop(0)
        keys = find_keys(minimum_count, count)
        keys = sorted(keys, reverse=True)
        for key in keys:
            result.append(key)
    op = ""
    for i in result:
        op = op + str(i) + " "
    print op


def find_keys(value, d):
    res = []
    for key in d.keys():
        if d[key] == value:
            res.append(key)
    return res


t = int(raw_input())

input_now = []

for i in range(0, t):
    input_now.append(raw_input())

for i in input_now:
    solution(i)
