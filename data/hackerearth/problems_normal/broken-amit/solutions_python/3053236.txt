n=input()
love=map(int,raw_input().split())
decr=map(int,raw_input().split())
for i in range(len(decr)): decr[i]*=i
decr.sort()
decr.reverse()
tDecr=[0]*n
for i in range(len(decr)):
	cDecr=0
	tDec=0
	for j in range(i):
		tDec+=cDecr
		cDecr+=decr[i]
	tDecr[i]=tDec+(n-i)*cDecr	
su=sum(love)*n-sum(tDecr)
print su
