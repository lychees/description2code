#include<cstdio>
using namespace std;
int main()
{
   int i,n,test;
   int arr[10009];
   int dp[10009];
   arr[1]=0;
   dp[0]=0;
   for(int j=2;j<=10001;j++)
   {
      n=j;
      int result=n;
      for(i=2;i*i<=n;i++)
      {
         if(n%i==0)
         {
            result-=result/i;
            while(n%i==0)
            n/=i;
         }
      }
      if(n>1)
      result-=result/n;
      arr[j]=result;
   }
   for(int i=1;i<=10001;i++)
      dp[i]=arr[i]+dp[i-1];
   scanf("%d",&test);
   while(test--)
   {
      scanf("%d",&n);
      printf("%d\n",dp[n]);
   }
   return 0;
}
