#include <bits/stdc++.h>
#include <math.h>
using namespace std;
const int sz=3010;
double cache[sz];
struct cord{
	int x,y,f;
};
cord s[sz];

double dist(cord a,cord b){
	double x,y;
	x = a.x-b.x;
	y = a.y-b.y;
	return sqrt(x*x+y*y);	
}
int main()
{
	int x,y,f,n;
	cin>>n;
	for(int i=0;i<n;++i){
		cin>>s[i].x>>s[i].y>>s[i].f;
	}
	fill(cache,cache+sz,INT_MIN);
	cache[0]=s[0].f;
	double res=cache[0];
	for(int i=1;i<n;++i){
		for(int j=0;j<i;++j){
			cache[i] = max(cache[i],s[i].f-dist(s[i],s[j])+cache[j]);
		}
		//cache[i]+=s[i].f;
		res = max(res,cache[i]);
	}
	printf("%.6lf",cache[n-1]);
    return 0;
}
