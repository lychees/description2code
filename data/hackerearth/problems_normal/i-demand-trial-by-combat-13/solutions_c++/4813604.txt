#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,x;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        vector<int> v;
        for(int i=0;i<n;i++)
        {
            cin>>x;
            v.push_back(x);
        }
        vector<int> pre,pre2;
        for(int k=1;k<=m;k++)
        {
            vector<int> ans;
            ans.push_back(v[1]);
            for(int i=1;i<n-1;i++)
            {
                if(v[i-1]==1&&v[i+1]==1)
                ans.push_back(1);
                else
                ans.push_back(0);
            }
            ans.push_back(v[n-2]);
            v=ans;
            if(ans==pre)
                break;
            else if(ans==pre2)
            {
                if((m%2==0&&k%2==0)||(m%2&&k%2))
                   break;
                else
                {
                    v=pre;
                    break;
                }
            }
            pre2=pre;
            pre=ans;
        }
        for(int i=0;i<v.size();i++)
            cout<<v[i]<<" ";
        cout<<endl;
    }
    return 0;
}
