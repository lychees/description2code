#include <iostream>
#include<string>
using namespace std;

int main()
{
	string line;
	int i,temp,n,temp2;
    while(getline(cin,line))
    {
    	temp = 0;
    	temp2 = 0;
    	n = line.length();
    	for(i=0; i < n; i++)
    	{
    		if(line[i] == '/' && line[i+1] == '/')
    		{
    			temp=1;
    		}
    		if((i < n-1) && (temp != 1) && line[i] == '-' && line[i+1] == '>' )
    		{
    			line[i]='.';
    			temp2 = 1;
    		}
    		cout << line[i];
    		if(temp2 == 1)
    		i++;
    		temp2=0;
    	}
    	cout << endl;
    }
    
    return 0;
}
