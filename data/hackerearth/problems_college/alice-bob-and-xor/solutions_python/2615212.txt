def f(n):
	if n%4 == 0: return n
	if n%4 == 1: return 1
	if n%4 == 2: return n+1
	return 0
	
t = int(raw_input())
while t > 0:
	L, R = map(int, raw_input().split())
	print f(R) ^ f(L-1)
	t -= 1
