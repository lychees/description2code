#include<iostream>


using namespace std;

int main()
{
	int n,q,m;

	cin>>n>>m>>q;

	//vector<int> a(n);
	//vector<int> left(n);
	//vector<int> right(n);

	int *a = new int[n];
	int *left = new int[n];
	int *right = new int[n];
	int l(1), r(1);
	for(int i=0; i<n;i++)
	{
		cin>>a[i];
	}

	for(int i=0; i<n;i++)
	{
		l= ( (l%m) * (a[i]%m) ) % m;
		r = ( (r%m) * (a[n-1-i]%m) ) % m ;
		left[i] = l;
		right[n-1-i] = r;
	}

	for(int i=0; i<q; i++)
	{
		int index;
		cin>>index;
		
		if(index== 1)
		{
			cout<<right[1]<<endl;
		}
		// Product excluding the last element will be stored at left[n-2]
		else if(index == n)
		{
			cout<<left[n-2]<<endl;
		}
		else
		{
			// multiply product before the current number ie on the left side and after the no ie on the right side
			cout<<((right[index]%m)* (left[index-2]%m) )%m<<endl;
		}
	}
	return 0;
}
