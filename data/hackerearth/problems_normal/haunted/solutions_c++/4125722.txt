#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,x;
    cin>>n>>m;
    priority_queue<pair<int,int>,vector<pair<int,int>>> pq;
    map<int,int> mp;
    for(int i=0;i<n;i++)
    {
        cin>>x;
        mp[x]++;
        pq.push({mp[x],x});
        printf("%d %d\n",pq.top().second,pq.top().first);
    }
}
