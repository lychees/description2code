#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    cout<<setprecision(2)<<fixed;
    while(t--)
    {
    	cin>>n;
    	long double x,y,cx=0,cy=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>x>>y;
    		cx+=x;
    		cy+=y;
    	}
    	cx/=n;
    	cy/=n;
    	cout<<"("<<cx<<" , "<<cy<<")"<<endl;
    }
    return 0;
}
