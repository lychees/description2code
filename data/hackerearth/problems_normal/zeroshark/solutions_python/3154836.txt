lista = []
for i in range(int(input())):
    n = int(input())
    dp = [0, 0, 1] + [0] * (n-2)
    fib = [0, 1, 1] + [0] * (n-3)
    for i in range(3, n):
        fib[i] = (fib[i-1] + fib[i-2]) % (10**9+7)
    for i in range(3, n+1):
        dp[i] = (dp[i-1] + dp[i-2] + fib[i-1]) % (10**9+7)
    lista.append(dp[n])
for i in lista:
    print(i)
