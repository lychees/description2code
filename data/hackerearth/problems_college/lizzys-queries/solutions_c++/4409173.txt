#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
	char s[100005];
    int n,q,i,ch,id,j,l,r;
    char x;
    scanf("%d%d",&n,&q);
    scanf("%s",s);
    while(q--)
    {
    	scanf("%d",&ch);
    	if(ch==0)
    	{
    		scanf("%d %c",&id,&x);
    		s[id-1]=x;
    	}
    	else
    	{
    		scanf("%d%d",&l,&r);
    		l--;
    		id=1;
    		for(i=l,j=r-1;i<=j;i++,j--)
    		{
    			if(s[i]!=s[j])
    			{
    				id=0;
    				break;
    			}
    		}
			if(id)
			{
				printf("Yes\n");
			}
			else
			{
				printf("No\n");
			}
    	}
    }
    return 0;
}
