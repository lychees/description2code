#include <bits/stdc++.h>
#define ull unsigned long long int
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		ull a,b,res=0;
		cin>>a>>b;
		while(a>1||b>1){
			if(a>b){
				res=res+a/b;
				a=a%b;
			}
			else{
				res=res+b/a;
				b=b%a;
			}
		}
		if(res>0)
			--res;
		cout<<res<<endl;
	}
    return 0;
}
