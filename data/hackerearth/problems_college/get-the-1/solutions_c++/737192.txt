#include<iostream>
#include<map>
#include<vector>

using namespace std;


map < long , bool > m;
vector < bool > ans;


int calc(long no)
{
    int ans=0;
    int t;
    
    while(no>0)
    {
               t=no%10;
               t*=t;
               ans+=t;
               no/=10;
    }
    
    return ans;
}
               


int main()
{
    int T;
    
    cin>>T;
    
    long i,j,k,no;
    
    for(i=0;i<T;i++)
    {
                    m.clear();
                    
                    cin>>no;
                    
                    
                    
                    
                    while(1)
                    {
                            m[no]=1;
                    
                            k=calc(no);
                            
                    if(k==1)
                    {
                            ans.push_back(1);
                            break;
                    }
                    
                    else if(m.find(k)!=m.end())
                    {
                         ans.push_back(0);
                         break;
                         
                    }
                    
                    no=k;
                    
                    }
                    
    }
          
          
    for(i=0;i<T;i++)
    {
                    switch(ans[i])
                    {
                                  case 0:
                                       cout<<"Case #"<<i+1<<":NO\n";
                                       break;
                                  default:
                                       cout<<"Case #"<<i+1<<":YES\n";
                                       break;
                    }
                    
    }
    
        
    return 0;
}
