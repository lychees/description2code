#include <bits/stdc++.h>
using namespace std;

int main()
{
  //  cout << "Hello World!" << endl;
  long long int  n,i,ans=0,j,k=0;
  cin>>n;
  int arr[n],a[n];
  for(i=0;i<n;i++)
  	cin>>arr[i];
  	sort(arr,arr+n);
  	for(i=0,j=n-1;k<n;){
  		if(ans%2==0)
  			a[k++]=arr[i++];
  		else
  			a[k++]=arr[j--];
  			ans++;
  	}
  	ans=0;
  	for(i=0;i<n;i++){
  		ans+=abs(a[i]-a[(i+1)%n]);
  	}
  	cout<<ans;
    return 0;
}
