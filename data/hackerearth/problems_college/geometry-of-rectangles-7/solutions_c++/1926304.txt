// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

int a[1001][1001];

int main()
{
    int t, x1, x2, x3, x4, y1, y2, y3, y4, i, j, cnt;
    cin >> t;
    while (t--) {
        cin >> x1 >> y1 >> x2 >> y2;
        memset(a, 0, sizeof(a));
        for (i = x1; i <= x2; i++) {
            for (j = y1; j <= y2; j++)
                a[i][j]++;
        }
        cin >> x3 >> y3 >> x4 >> y4;
        vector <int> v;
        for (i = x3; i <= x4; i++) {
            cnt = 0;
            for (j = y3; j <= y4; j++)
                cnt += (a[i][j] == 1);
            if (cnt)
                v.push_back(cnt);
        }
        if (!v.size())
            printf("Apart\n");
        else if (v.size() == 1 && v[0] == 1)
            printf("Point\n");
        else if (v.size() == 1)
            printf("Line\n");
        else if (v.size() > 1 && v[0] > 1)
            printf("Rectangle\n");
        else
            printf("Line\n");
    }
    return 0;
}
