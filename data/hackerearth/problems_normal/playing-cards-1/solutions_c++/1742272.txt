#include <bits/stdc++.h>
using namespace std;

const int mx = 55 * 52;
const int mod = 1000000007;

int C[mx][mx];

int main()
{
	for(int i = 0; i < mx; ++i) {
		C[i][0] = C[i][i] = 1;
		for(int j = 1; j < i; ++j) {
			C[i][j] = (C[i - 1][j] + C[i - 1][j - 1]) % mod;
		}
	}
    int t, n, m;
    scanf("%d", &t);
    while(t--) {
    	scanf("%d %d", &n, &m);
    	if(m % 2 == 1) {
    		printf("0\n");
    		continue;
    	}
    	
    	int f = 6 * n, r = 18 * n;
    	int ans = 0;
    	
    	int x = m / 2;
    	for(int i = 0; i <= x; ++i) {
    		int cur =  1ll * C[f][i] * C[r][x - i] % mod;
    		cur = 1ll * cur * C[f][x - i] % mod * C[r][i] % mod;
    		ans = (ans + cur) % mod;
    	}
    	printf("%d\n", ans);
    	
    }
    return 0;
}
