t=input()
for i in range(t):
	index,n=map(int,raw_input().split())
	if index==0:(a,b,c)=(1.0,0.0,0.0)
	if index==1:(a,b,c)=(0.0,1.0,0.0)
	if index==2:(a,b,c)=(0.0,0.0,1.0)
	for j in range(n):
		x=0.5*b
		y=a+c
		z=0.5*b
		a=x
		b=y
		c=z
	if a>=b and a>=c:print 0
	if b>=c and b>a:print 1
	if c>a and c>b:print 2
