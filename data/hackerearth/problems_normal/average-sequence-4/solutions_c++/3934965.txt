#include<iostream>
#define ll long long int
using namespace std;

int main()
{
	int n;
	cin>>n;
	ll a[n],b[n],c=0;
	for(int i=1;i<=n;i++)
	{
		cin>>b[i];
	}
	a[1]=b[1];
	for(int i=2;i<=n;i++)
	{
		c+=a[i-1];
		a[i]=(b[i]*i)-(c);
	}
	for(int i=1;i<=n;i++)
	{
		cout<<a[i]<<" ";
	}
	return 0;
}
