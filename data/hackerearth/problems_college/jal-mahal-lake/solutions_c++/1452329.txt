#include <bits/stdc++.h>
#define si(n) scanf("%d",&n)
#define sl(n) scanf("%lld",&n)
#define ss(n) scanf("%s",n)
#define fr(i, n) for(int i = 0; i < n; i++)
#define REP(i, a, b) for (int i = int(a); i <= int(b); i++)
#define REN(i, a, b) for (int i = int(a); i >= int(b); i--)
#define ms(i, n) memset(i, n, sizeof(i))
#define INF 1002000000
#define MOD 1000000007
typedef long long LL;
using namespace std;
LL dp[101][101];
struct stu {
    LL pos;
    LL len;
}arr[101];
int t, n;
bool srrt (stu s1, stu s2) {
    return s1.pos < s2.pos;
}
LL calc(int i, int j) {
    return arr[j].pos - arr[i].pos - arr[i].len;
}
LL memoize(LL st, LL en) {
    if(st == en)
        return 0;
    if(dp[st][en] != -1)
        return dp[st][en];
    LL m = n;
    LL ans = min(memoize(st + 1, en) + (st + 1) * calc(st, st + 1), memoize(st, en - 1) + (m - en) * calc(en - 1, en));
    return dp[st][en] = ans;
}
int main ()
{
	//freopen ("input.txt","r",stdin);
	//freopen ("output.txt","w",stdout);
	si(t);
	while(t--) {
        si(n);
        fr(i, n)
            fr(j, n)
                dp[i][j] = -1;
        fr(i, n)
            sl(arr[i].pos);
        fr(i, n)
            sl(arr[i].len);
        sort(arr, arr + n, srrt);
        printf("%lld\n", memoize(0, n - 1));
	}
	return 0;
}
