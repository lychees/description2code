#hackerearth problem

test = input()
i=0
while i!=test:
	a,b = map(int,raw_input().split())
	#print a,b
	final = []
	final = [b]
	#print final
	for j in range(a):
		query = raw_input().split()
		#print "Query", query
		if query[0] == 'P':
			nt = int(query[1])
			final.append(nt)
		elif query[0] == 'B':
			final.append(final[len(final)-2])
			
		#print final

	print "Player",final[len(final)-1]
	i = i+1
