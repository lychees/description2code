#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main ()
{

		int a,i,j=5,count=0,c,k;
	cin>>a;
	if(a==1)
	cout<<"2\n";
else	if(a==2)
	cout<<"3\n";
	else
	{
		count=2;
		for(i=4;i<j;i++)
	{
		
		c=0;
	for(k=2;k<=sqrt(i);k++)	
	{
		if(i%k==0)
		{
			c=1;
			break;
		}
	}
	if(c==0)
	{
		count++;
		if(count==a)
		{
		cout<<i<<endl;
		break;
		}
	}
		j++;
	}
	}
	

	return 0;
}
