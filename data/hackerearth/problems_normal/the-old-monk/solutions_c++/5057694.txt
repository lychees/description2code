#include <iostream>
#include<queue>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	long int a[n],b[n];
    	for(int i =0;i<n;i++)
    	cin>>a[i];
    	for(int j =0;j<n;j++)
    	cin>>b[j];
    	//priority_queue<int> pq;
    	//pq.push(0);
    	int temp =0;
    	
    	for(int i=0,j=0;!(i==n||j==n);)
    	{
    		if(b[j]>=a[i]&&j>=i)
    		{
    			int y = j-i;
    			if(temp<y)
    			temp=y;
    			
    			//pq.push(y);
    			j++;
    			
    		}
    		else if(b[j]<a[i]&&i==j)
    		{
    			i++;
    			j++;
    		}
    		else 
    		i++;
    		
    		
    	}
    	cout<<temp<<endl;
    	
    	
    	
    	
    	
    }
    
    return 0;
}
