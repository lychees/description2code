#include <iostream>
#include <stdio.h>
#include <string.h>
#include <stdlib.h>
using namespace std;

int main()
{
  int t;
  
  scanf("%d",&t);
  while(t--)
  {
  	char str[10000]="";
  	scanf("%s",str);
  	int tab[26]={0};
  //	memset(0,tab,sizeof(tab));
  //	int *tab = (int *)calloc(26,sizeof(int));
  	int len = strlen(str);
  	for(int i=0;i<len;i++)
  	{
  		tab[str[i]-'a']++;
  	}
  	int max = -1000;
  	for(int i=0;i<26;i++)
  	{
  		if(tab[i]>=max)
  		 max = tab[i];
  	}
  	cout<<max<<endl;
  }
}
