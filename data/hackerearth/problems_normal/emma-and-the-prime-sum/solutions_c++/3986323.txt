#include <iostream>
using namespace std;
typedef long long int ll;
ll a[1000002];
int main()
{
    ll max = 1000000;
    ll i;
    ll j;
    a[0] = 0;
    a[1] = 0;
    for(i = 2; i*i <= max; i++) {
    	if(a[i] == 0) {
    		for(j = i * i; j <= max; j+=i) {
    			a[j] = 1;
    		}
    	}
    }
    for(i = 2; i <= max; i++) {
    	if(a[i] == 0) {
    		a[i] = i + a[i - 1];
    	}	
    	else {
    		a[i] = a[i - 1];
    	}
    }
    ll t;
    cin >> t;
    while(t--) {
    	ll x, y;
    	cin >> x >> y;
    	ll ans = a[y] - a[x - 1];
    	cout << ans << endl;
    }
    return 0;
}
