#include <bits/stdc++.h>
using namespace std;

typedef long long int64;

int64 fun(int64 m, char c, int64 x)
{
	if(c == 'N')
		return -m;
	else if(c == '+')
		return m+x;
	else if(c == '-')
		return m-x;
	else if(c == '*')
		return m*x;
	else if(c == '/')
		return m/x;
	else
		return 1;
}

int main()
{
	int64 t;
	cin >> t;
	while(t--)
	{
		int64 n;
		cin >> n;
		char c[n];
		int64 mi = 1, ma = 1;
		for(int64 i=0;i<n;i++)
		{
			cin >> c[i];
			int64 x;
			if(c[i] != 'N')
				cin >> x;
			vector<int64> maybe;
			maybe.push_back(mi);
			maybe.push_back(ma);
			maybe.push_back(fun(mi,c[i],x));
			maybe.push_back(fun(ma,c[i],x));
			sort(maybe.begin(),maybe.end());
			mi = maybe[0];
			ma = maybe.back();
		}
		cout << ma << endl;
	}
	return 0;
}
