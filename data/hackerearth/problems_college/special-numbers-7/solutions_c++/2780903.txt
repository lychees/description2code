




						#include<bits/stdc++.h>
using namespace std;
int c[100009] = {0};
void sieve()
{
	int i,j;
	for( i=2;i<=100;i++)
	{
		if(c[i]==0)
		{
			for( j=i*i;j<=10000;j+=i)
				c[j] = 1;
		}
	}
}
int main()
{
	sieve();
	vector<int> v;
	c[1] = 1;
	int i,j,temp,count,t,n;
	for(j=2;j<100000;j++){
		count = 0;
		for(i=1;i*i<=j;i++){
			if(j%i == 0){
				temp = j/i;
				if(c[i] == 0)
					count++;
				if(c[temp] == 0)
					count++;
			}
			if(count > 2)
				break;
		}
		if(count > 2)
			v.push_back(j);
	
	}
	/*for(i=0;i<1000;i++)
		printf("%d ",v[i]);
	printf("\n");*/
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		printf("%d\n",v[n-1]);
	}
	return 0;
}
