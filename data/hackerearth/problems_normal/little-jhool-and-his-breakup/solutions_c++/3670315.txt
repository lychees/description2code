#include <iostream>
using namespace std;

int main()
{
	char s[110];
	int e=-1,o=-1,l=-1,i,v=-1;
	i=0;
	scanf("%s",s);
	while(s[i]!='\0')
	{
		if(s[i]=='l' && l==-1)
		l=i;
		if(s[i]=='o' && o==-1 && l!=-1)
		o=i;
		if(s[i]=='v' && v==-1 && l!=-1 && o!=-1)
		v=i;
		if(s[i]=='e' && e==-1 && l!=-1 && o!=-1 && v!=-1)
		e=i;
		
		i++;
	}
	if(l<o && l<v && l<e && o<v && o<e && v<e)
	printf("I love you, too!\n");
	else
	printf("Let us breakup!\n");
	
    //cout << "Hello World!" << endl;
    return 0;
}
