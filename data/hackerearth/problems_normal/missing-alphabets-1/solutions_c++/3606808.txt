char c[333];
struct data
{
	string s,s2;
}d[111];
bool cmp(data d1,data d2)
{
	return d1.s2<d2.s2;
}
int main(void){
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n,m;
	string s;
	cin>>T;
	while(T--)
	{
		cin>>s;
		for(i=0;i<26;i++)
		{
			c[s[i]]=i+'a';
		}
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>d[i].s;
			m=d[i].s.size();
			d[i].s2.clear();
			for(j=0;j<m;j++)
			{
				d[i].s2+=c[d[i].s[j]];
			}
			//cout<<d[i].s2<<endl;
		}
		sort(d,d+n,cmp);
		for(i=0;i<n;i++)
			cout<<d[i].s<<endl;
	}
//	system("pause");
	return 0;
}
