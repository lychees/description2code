int main(int argc, char *argv[]) {
  int T;
  scanf("%d", &T);
  while(T--) {
    int A, B;
    scanf("%d%d", &A, &B);
    int ans = 0;
    if (A <= 7 && 7 <= B) ans += 8;
    if (A <= 13 && 13 <= B) ans += 144;
