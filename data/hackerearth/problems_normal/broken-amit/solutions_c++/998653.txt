#include <cstdio>
#include <iostream>
#include <algorithm>
#include <functional>
using namespace std;
typedef long long ll;


pair<int,pair<int, pair<int,int> > > value[18];

int main() {
	int n,l,d,i,temp,j;
	ll ans=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&temp);
		value[i].second.first=i;
		value[i].second.second.first=temp;
		ans+=temp;
	}
	for(i=0;i<n;i++)
	{
		scanf("%d",&temp);
		value[i].second.second.second=temp;
		value[i].first=(i*temp*n);
	}
	sort(value,value+n,greater<pair<int,pair<int, pair<int,int> > > >());
	for(i=1;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			ans+=value[j].second.second.first;
		}
		for(j=i;j<n;j++)
		{
			value[j].second.second.first=value[j].second.second.first-(value[j].second.first*value[j].second.second.second);
			ans+=value[j].second.second.first;
		}
	}
	printf("%lld",ans);
	return 0;
}
