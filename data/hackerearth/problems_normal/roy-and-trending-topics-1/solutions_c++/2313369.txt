#include <bits/stdc++.h>
using namespace std;
inline bool comp(const vector<long long> &a ,const vector<long long> &b ){
	if(a[1]==b[1])
	return b[0]<a[0];
	else
	return b[1]<a[1];
	
}
int main()
{
 //   cout << "Hello World!" << endl;
 long long i,j,id,z,p,l,c,s,n;
 cin>>n;
 
 vector< vector<long long > >v;
 
 for(i=0;i<n;i++){
 	cin>>id>>z>>p>>l>>c>>s;
 	vector<long long> v1;
 	v1.push_back(id);
 	v1.push_back(p*50+l*5+c*10+s*20-z);
 	v1.push_back(p*50+l*5+c*10+s*20);
 	v.push_back(v1);
 	
 }
 sort(v.begin(),v.end(),comp);
 for(i=0;i<5;i++){
 	cout<<v[i][0]<<" "<<v[i][2]<<"\n";
 }
    return 0;
}
