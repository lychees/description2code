#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int main()
	{
		int t;
		cin>>t;
		int n,a,b;
		vector<pair<int,int>>v;
		while(t--)
			{
				cin>>n;
				for(int i=0;i<n;++i)
					{
						cin>>a>>b;
						v.push_back(make_pair(a,b));
					}
				sort(v.begin(),v.end());
				//for(int i=0;i<n;++i)
				//	cout<<v[i].first<<" "<<v[i].second<<endl;
				long long int dp[n+1];
				dp[0]=v[0].second;
				long long int ans=0;
				for(int i=1;i<n;++i)
					{
						long long int r=v[i].first;
						long long int h=v[i].second;
						int flag=0;
						dp[i]=0;
						for(int j=i-1;j>=0;--j)
							{
								
								if(r>v[j].first && h>v[j].second){
								flag=1;
								//cout<<h<<endl;
								dp[i]=max(h+dp[j],dp[i]);
								//cout<<j<<" "<<dp[i]<<endl;
								}
							}
						if(flag==0)
							dp[i]=h;
						ans=max(ans,dp[i]);
					}
				cout<<ans<<endl;
				v.clear();
			}
		return 0;
	}
