#include <iostream>
#include <vector>

using namespace std;


int main(){
	

	int num;

	char a;


	while(cin >> a >> num)
	{
		if ( num < 2 || num > 26)
			cout << "INVALID" << endl;
		else if ((a + num - 1) > 122)
			cout << "INVALID" << endl;

		else
			for(int i = 1; i <= num; i++)
			{
				for(int j = 1; j <= i; j++ )
				{
					cout << char(a+i-1) << " ";
				}
				cout << endl;
			}
				
	cout << endl;

	}



	return 0;
}
