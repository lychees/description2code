#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int a[100005],n,t,k,i,h,l,v,m,ans,cnt;
	
	cin>>t;
	while(t--)
	{
			cin>>n>>k;
	
	for(i=0;i<n;i++)
	cin>>a[i];
	
	sort(a,a+n);
	
	l=0;
	h=a[n-1];
	
	while(h-l>1)
	{
		m = (l+h)/2;
		cnt=1;
		v=a[0];
		for(i=1;i<n;i++)
		if(a[i]-v>=m)cnt++,v=a[i];
		
		if(cnt>=k)l=m;
		else h=m-1;
		
	}
	
	cnt=1;
	v=a[0];
	m=h;
	for(i=1;i<n;i++)
	if(a[i]-v>=m)cnt++,v=a[i];
		
	if(cnt>=k)ans=h;
	else ans=l;
	
    cout << ans<<endl;
	}

    return 0;
}
