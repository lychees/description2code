using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int cnt=0;
        int n,m;
        cin>>n>>m;
        char arr[n+1][m+1];
       // memset(a,0,sizeof(a));
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                cin>>arr[i][j];
               
      
       for(int i = n - 1; i >= 0; i--)
       {
            for(int j = m - 1; j >= 0; j--)
            {

                if(arr[i][j] == '0')
                {                 

                    cnt++;                          

                    for(int p = i; p >= 0; p--)
                    {        
                        for(int k = j; k >= 0; k--)
                        {

                            if(arr[p][k] == '0')
     
                            arr[p][k] = '1';      
                            else 
                            arr[p][k] = '0';
                        }
                    }
                }
            }
        }


     
        
        cout<<cnt<<endl;
    }


   return 0;
}
