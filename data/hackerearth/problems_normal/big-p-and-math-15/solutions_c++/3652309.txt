#include <iostream>
using namespace std;

int main()
{
    int a,b;
    cin>>a>>b;
    int sumMax[10],sumMin[10],A[10]={0},B[10]={0};
    int top =9;
    int max=0,min=0;
    while(a){
    	A[top]= a%10;
    	a= a/10;
    	top--;
    }
    top = 9;
    while(b){
    	B[top]= b%10;
    	b= b/10;
    	top--;
    }
    for (int i=0;i<10;i++){
    	if (A[i]==6)
    		A[i]=5;
    	if(B[i]==6)
    		B[i]=5;
    	sumMin[i] =A[i]+B[i];
    	if(A[i]==5)
    		A[i]=6;
    	if(B[i]==5)
    		B[i]=6;
    	sumMax[i] =A[i]+B[i];
    	max = max*10 + sumMax[i];
    	min = min*10 + sumMin[i];
    }
    cout<<min<<" "<<max;
    
    return 0;
}
