#include<iostream>
using namespace std;
int main()
{
    int t,count;
    long long int no;
    cin>>t;
    while(t--)
    {
        cin>>no;
        count=0;
        while(no!=0)
        {
            if(no%2==1)
                count++;
            no=no/2;
        }
        if(count%2==0)
            cout<<"even"<<endl;
        else
            cout<<"odd"<<endl;

    }
}
