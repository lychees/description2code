#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	unsigned long long int T, N, no_of_bits, ans, max;
	cin>>T;
	while(T--){
		cin>>N;
		no_of_bits = ceil(log(N)*1.44269504088896340736);
		ans = pow(2,no_of_bits);
		max = ans-1;
		ans -= 2*(ans-N);
		cout<<max<<" "<<ans<<endl;
	}
    return 0;
}
