#include <iostream>
#include <map>
#include <climits>
using namespace std;

int main()
{
	map<string,int> M;
	int m,n,t,q;
	string s,s2;
	cin>>n>>m;
	for(int i=0;i<n;++i)
	{
		cin>>s;
		M[s]=i;
	}

	long long mat[100][100];
	for(int i=0;i<100;++i)
	for(int j=0;j<100;++j)
	mat[i][j]=INT_MAX;
	for(int i=0;i<m;++i)
	{
		
		cin>>s>>s2>>t;
	
		mat[M[s]][M[s2]] = t;
		mat[M[s2]][M[s]] = t;
		
	}
	
	
	for(int e=0;e<100;++e)
	for(int i=0;i<n;++i)
	for(int j=0;j<n;++j)
	for(int k=0;k<n;++k)
	{
		if(mat[i][j]>mat[i][k]+mat[k][j])
		{
	            	mat[i][j]=mat[i][k]+mat[k][j];
		}
	}
	cin>>q;
	for(int i=0;i<q;++i)
	{
		cin>>s>>s2;
		cout<<mat[M[s]][M[s2]]<<endl;
	}


}
