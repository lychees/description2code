#include <iostream>
#include <stack>

using namespace std;

int main()
{
	int Q, type, cost;
	cin >> Q;

	stack<int> pile;
	
	while (Q--)
	{
		cin >> type;
		if (type == 1)
		{
			if (pile.empty())
			{
				cout << "No Food" << endl;
			}
			else
			{
				cost = pile.top();
				pile.pop();
				cout << cost << endl;
			}
		}
		else
		{
			cin >> cost;
			pile.push(cost);
		}
	}

	return 0;
}
