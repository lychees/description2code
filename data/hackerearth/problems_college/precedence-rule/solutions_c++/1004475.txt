#include <iostream>
#include <algorithm>
#include <cstring>
#define $ while
using namespace std;

int main() {
	int t,l,i;
	long long a; 
	cin>>t;
	char s[31];
	$(t--)
	{
		cin>>s;
		l=strlen(s);
		if(l>2)
		{	
			a=s[l-1]-48;
			l-=3;
			for(i=l;i>=0;i-=2)
			{
				if(s[i+1]=='*')
				a*=s[i]-48;
				else if(s[i+1]=='+')
				a+=s[i]-48;
				else if(s[i+1]=='-')
				a-=s[i]-48;
			}
			cout<<a<<endl;
		}
		else
		cout<<s<<endl;
	}
	return 0;
}
