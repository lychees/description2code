def count(S, m, n):
    if (n == 0):
        return 1

    if (n < 0):
        return 0

    if (m <= 0) and (n >= 1):
        return 0
    # print 'm',m
    # print 'n',n
    return count(S, m - 1, n) + count(S, m, n-S[m-1])


t = input()
for _ in range(0, t):
    n = input()

    m = 3
    print count(arr, m, n)
