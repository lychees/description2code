#include <iostream>
#include <vector>
#define add push_back
#include <algorithm>
using namespace std;
int main()
{
    int t=0;
    int i=0,j=0,h=0;
    cin>>t;
    for(h=0;h<t;h++)
    {
        int n=0, temp=0, sum=0;
        vector <int> men,women;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>temp;
            men.add(temp);
        }
        for(i=0;i<n;i++)
        {
            cin>>temp;
            women.add(temp);
        }
        sort(men.begin(),men.end());
        sort(women.begin(),women.end());
        for(i=0;i<n;i++)
        {
            sum+=men[i]*women[i];
        }
        cout<<sum<<"\n";
    }
    return 0;
}
