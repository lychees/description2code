#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool comp(int a, int b) {
	return a>b;
}

int main()
{
    int T,N,K,num;
    cin>>T;
    while(T--) {
    	cin>>N>>K;
    	K=min(N,K);
    	vector<int> arr;
    	while(N--) {
    		cin>>num;
    		arr.push_back(num);
    	}
    	sort(arr.begin(),arr.end(),comp);
    	long long sum=0;
    	for(int i=0;i<K;i++) {
    		sum+=arr[i];
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
