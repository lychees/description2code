#include <iostream>
#define ll long long
using namespace std;
int main()
{
    ll n,q,x,aa,b,i;
    scanf("%lld", &n);
    int a[n];
    for(i=0;i<n;i++)
    	a[i]=0;
    scanf("%lld", &q);
    while(q--)
    {
    	scanf("%lld%lld%lld", &x,&aa, &b);
    	if(x==0)
    	{
    		for(i=aa;i<=b;i++)
    		{
    			if(a[i]==0)
    				a[i]=1;
    			else
    				a[i]=0;
    		}
    	}
    	else
    	{
    		ll count=0;
    		for(i=aa;i<=b;i++)
    		{
    			if(a[i]==1)
    				count++;
    		}
    		printf("%lld\n", count);
    	}
    }
    return 0;
}
