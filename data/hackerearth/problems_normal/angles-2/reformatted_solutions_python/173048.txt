n, k = map(int, raw_input().split(" "))
n1 = map(int, raw_input().split(" "))
k1 = map(int, raw_input().split(" "))
r1 = [""]*k
for i in range(n):
    j = 0
    p = n1[i]
    for j in range(100):
        q = (p % 360)
        for s in range(k):
            if k1[s] == q and r1[s] == "":
                r1[s] = "YES"
        p += n1[i]
for i in range(k):
    if r1[i] == "":
        print "NO"
    else:
        print "YES"
