#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int
#define s(x) scanf("%d", &x)
#define sl(x) scanf("%lld", &x)
#define sd(x) scanf("%lf", &x)
#define mod 1000000007
#define get getchar_unlocked

ll power(ll a, ll b)
{
    if (!b)
        return 1;
    ll x = power(a, b>>1);
    if (b & 1)
        return (((x*x)%mod)*a)%mod;
    return (x*x)%mod;
}

int main()
{
    int n, x, i;
    ll ans1 = 1, ans2 = 1;
    s(n);
    for (i = 0; i < n; ++i) {
        s(x);
        ans1 = (ans1 * x)%(mod-1);
    }
    for (i = 0; i < n; ++i) {
        s(x);
        ans2 = (ans2 * x)%(mod-1);
    }
    --ans2;
    printf("%lld\n", ((power(2, ans1)-1)*power(power(2, ans1+ans2), mod-2))%mod);
    return 0;
}
