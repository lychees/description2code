#include <bits/stdc++.h>
#define ll long long
#define pb push_back
#define mk make_pair
#define pi pair<int, int>
#define fi first
#define se second
using namespace std;
const int N = 1003;
int a[N][N], b[N][N];
int r[N][N], c[N][N];
int main() {
	int t;
	cin >> t;
	while(t--) {
		int n, m;
		scanf("%d%d", &n, &m);
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < m; j++) {
				scanf("%d", &a[i][j]);
				r[i][j] = 0;
				c[j][i] = 0;
			}
		}
		for(int i = 0; i < m; i++) {
			int ans = 0;
			for(int j = 0; j < n; j++) {
				if(ans < a[j][i]) {
					ans = a[j][i];
				}
			}
			for(int j = 0; j < n; j++) {
				if(a[j][i] == ans) {
					c[i][j] = 1;
				}
			}
		}
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < m; j++) {
				scanf("%d", &b[i][j]);
			}
		}
		for(int i = 0; i < n; i++) {
			int ans = 0;
			for(int j = 0; j < m; j++) {
				if(ans < b[i][j]) {
					ans = b[i][j];
				}
			}
			for(int j = 0; j < m; j++) {
				if(b[i][j] == ans) {
					r[i][j] = 1;
				}
			}
		}
		int ct = 0;
		for(int i = 0; i < n; i++) {
			for(int j = 0; j < m; j++) {
				if(r[i][j] == 1 && c[j][i] == 1) {
					ct++;	
				}
			}
		}
		printf("%d\n", ct);
	}
	return 0;
}
