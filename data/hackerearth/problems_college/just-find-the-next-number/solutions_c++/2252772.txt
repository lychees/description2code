#include <iostream>
using namespace std;


int checkNum(int n) {

	int m  = n/2;
	while(m) {
		
		if( (m & 1) && (n & 1))
			return 0;
		
		else{
			n = m;
			m = n/2;
		}
	}
	return 1;
}
int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin >> t;
    
    while(t--) {
    	
    	int num;
    	cin >> num;
    while(1) {
    	
    	if(checkNum(num))
    		break;
    	++num;
    	}
    	cout << num<<endl;
    }
    return 0;
}
