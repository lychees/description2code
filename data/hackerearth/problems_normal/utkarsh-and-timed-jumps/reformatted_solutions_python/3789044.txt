# Given a number N, find a number Y such as Y > N and Y is sum of first K natural numbers.
# K is the answer to the problem.

from math import sqrt, floor

t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    x = n * 2
    y = int(floor(sqrt(x)))
    k = (y * (y + 1)) / 2
    if k >= n:
        print y
    else:
        y += 1
        k = (y * (y + 1)) / 2
        if k >= n:
            print y
        else:
            print y + 1
