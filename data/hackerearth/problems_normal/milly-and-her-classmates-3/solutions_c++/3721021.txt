#include <bits/stdc++.h>
using namespace std;
int a[10005],b[10005];
int main()
{
   // cout << "Hello World!" << endl;
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	int i;
   	for(i=0;i<n;i++)
   	{
   		scanf("%d",&a[i]);
   		b[i] = a[i];
   	}
   	sort(a,a+n);
   	int found = 0;
   	for(i=0;i<n;i++)
   	if(a[i]!=b[i]){
   		found = 1;
   		 break;
   	}
   	if(found)
   	{
   	//	printf("solve karo beta\n");
   	int j;
   	for(j=i+1;j<n;j++)
   	if(a[i]==b[j])break;
   	int k;
   	int l = j-i+1;
   //	printf("from %d to %d of len %d\n",i,j,l);
   	for(k=0;k<l/2;k++)
   	{
   		int tmp = b[i+k];
   		b[i+k] = b[i+l-1-k];
   		b[i+l-1-k] = tmp;
   	//	printf("swapping %d and %d\n",i+k,i+l-1-k);
   	}
   	//printf("reverse from %d to %d\n",i+1,j+1);
   	for(k=0;k<n;k++)
   	if(a[k]!=b[k])found = 0;
   	if(found)printf("%d %d\n",i+1,j+1);
   	else printf("-1 -1\n");
   	}
   	else printf("-1 -1\n");
   }
    return 0;
}
