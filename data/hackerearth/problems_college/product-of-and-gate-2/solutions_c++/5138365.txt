#include <bits/stdc++.h>

using namespace std;

long solve(long l, long r) {
	long ll, lr;
	asm volatile ("bsrq %%rdi, %0" : "=r"(ll));
	asm volatile ("bsrq %%rsi, %0" : "=r"(lr));
	if (lr > ll) {
		return 0;
	}
	return l & r;
}

int main() {
	int t;
	long l, r;
	scanf("%i", &t);
	while (t--) {
		scanf("%li %li", &l, &r);
		printf("%li\n", solve(l, r));
	}
	return 0;
}
