#include <iostream>
#include <vector>
using namespace std;
vector<int> v1,v2;
int main()
{
    int t,n,p1,p2,carry,start,result;
    cin>>t;
    v1.resize(1005,0);
    while(t--)
    {
    	cin>>n;
    	v2.clear();
    	start=0;
    	result=0;
    	v2.push_back(1);
    	if(v1[n])
    	{
    		cout<<v1[n]<<"\n";
    		continue;
    	}
    	for(p1=2;p1<=n;p1++)
    	{
    		carry=0;
    		for(p2=start;p2<v2.size();p2++)
    		{
    			carry=v2[p2]*p1+carry;
    			v2[p2]=carry%100000;
    		//	if(p2==start&&!(carry%100000))
    		//	start++;
    			carry/=100000;
    		}
    		if(carry)
    		v2.push_back(carry);
    	}
    	for(p2=start;p2<v2.size();p2++)
    		result+=v2[p2]%10+(v2[p2]/10)%10+(v2[p2]/100)%10+(v2[p2]/1000)%10+(v2[p2]/10000)%10;
    		v1[n]=result;
    	cout<<v1[n]<<"\n";
    }
    return 0;
}
