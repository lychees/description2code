#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int a[100000], i, j, c, n, r1 ,r2;
int main()
{
    cin>>n;
    for(j=0;j<n;j++)
    {
    	cin>>a[j];
    }
    sort(a, a+n);
    r1=a[n-1]*a[n-2];
    r2=a[0]*a[1];
    if(r1>r2)
    {
    	cout<<r1<<endl;
    }
    else
    {
    	cout<<r2<<endl;
    }
    return 0;
}
