#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T;
	long int i,j,k,A[1001],N,K;
	cin>>T;
	while(T--)
	{
		cin>>N>>K;
		for(i=0;i<N;i++)
		scanf("%ld",&A[i]);
		
		sort(A,A+N);
		int c=0;
		i=0;
		while(A[i]<=0)
		{
			c++;
			i++;
		}
		if(c>=K)
		cout<<"NO\n";
		else
		cout<<"YES\n";
	}
}
