#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
#define mod 1000000007
#define lim 1000
#define ll long long
ll i,cs,res,rf,rr,bf,br,n,m,j;
ll C[lim][lim];
int main()
{
  for(i=0;i<lim;i++)
    for(C[i][0]=C[i][i]=j=1;j<i;j++)
      C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
  cin>>cs;
  while(cs--)
  {
    scanf("%d%d",&n,&m);
    for(res=rf=0;rf<=6*n;rf++)
      for(rr=0;rr<=18*n;rr++)
        for(bf=0;bf<=6*n;bf++)
        {
          br=m-rf-rr-bf;
          if(br>=0 && br<=18*n && rf+rr==bf+br && rf+bf==rr+br)
            res=(res+C[6*n][rf]*C[18*n][rr]%mod*C[6*n][bf]%mod*C[18*n][br])%mod;
        }
    cout<<res<<"\n";
  } 
  return 0;
}
