#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	int b[n],s=0;
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	for(int j=0;j<n;j++)
    	cin>>b[j];
    	sort(a,a+n);
    	sort(b,b+n);
    	for(int k=0;k<n;k++)
    	{
    		s+=(a[k]*b[k]);
    	}
    	cout<<s<<endl;
    }
    return 0;
}
