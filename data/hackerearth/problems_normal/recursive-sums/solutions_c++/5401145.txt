#include <bits/stdc++.h>
using namespace std;
#define ull unsigned long long
int f(ull ans)
{
	if(ans/10 == 0) return ans;
	int res = 0;
	while(ans > 0)
	{
		res += ans%10;
		ans = ans/10;
	}
	return f(res);
}
int main()
{
	int t; cin >> t;
	while(t--)
	{
		int m; cin >> m;
		unsigned long long int ans = 0;
		for(int i=0;i<m;i++)
		{
			long long int x, y; cin >> x >> y;
			ans += (x*y);
			ans = f(ans);
		}
		
		unsigned long long int res = 0;
		while(1)
		{
			while(ans > 0)
			{
				int x = ans%10;
				res += x;
				ans = ans/10;
			}
			if(res/10 == 0) break;
			else
			{
				ans = res;
				res = 0;
			}
		}
		cout << res << endl;
	}
}
