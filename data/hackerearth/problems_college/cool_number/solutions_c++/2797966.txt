    	noofbits=0;
    	bitset<16> o(l);
    	noofbits=o.size();
    	for(int i=0;i<noofbits;i++){
    		int b;
    		
    		b=a&7;

    		
    			if(b==5){
    				count++;
    			}
    			a=a>>1;
    	}

    		}
    	}

    }
    return 0;
}
