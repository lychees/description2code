#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
typedef unsigned long long int ulli;
lli a[1000005];
lli dp[100000];
int n;

lli f(int i);
int main(){
int t;
	cin >>t;
	while(t>0){
	lli incl,excl,temp;
	cin >>n;
	memset(dp,-1,sizeof(dp));
	
	for(int i=0;i<n;i++){
	cin >>a[i];
	}
	
	lli num =f(0);
	cout << num <<"\n";
	t--;}
	
	
return 0;
}

lli f(int i){
if(i>=n){
	return 0;}
 if(dp[i]!=-1){
	return dp[i];
	}
if(i==n-1){
	return dp[i]=a[i];
	}
return dp[i]=max(f(i+1),a[i]+f(i+2));	
}
