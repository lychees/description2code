    #include <iostream>
    #include <cstring>
    #include <algorithm>
    using namespace std;
    typedef long int li;
    li getCoins( li *a, int n,int m){
    	li ans[n+1];
    	memset(ans,0l,sizeof(ans));
    	ans[0] = 1l;
    	
    	for(int i = 0;i<m;i++)
    		for(int j = a[i];j<=n;j++)
    			ans[j] += ans[j-a[i]];
    	return ans[n];
    }
     
    int main()
    {
       	int n,m;
       	cin>>n>>m;
       	li ar[m];
       	
       	for(int i = 0;i<m;i++)
       		cin>>ar[i];
       	sort(ar,ar+m);
       	cout<<getCoins(ar,n,m)<<endl;
        return 0;
    }
