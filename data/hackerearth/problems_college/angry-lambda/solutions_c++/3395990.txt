#include <bits/stdc++.h>
using namespace std;
int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int N,K;
		cin>>N>>K;
		int count = 0;
		for(int i=0;i<N;i++)
		{
			int Num;
			cin>>Num;
			if(Num<=0)
				count++;
		}
		if(count>=K)
		cout<<"NO"<<endl;
		else
		cout<<"YES"<<endl;
	}
	return 0;
}
