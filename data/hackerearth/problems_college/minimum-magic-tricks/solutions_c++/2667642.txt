#include <iostream>
#define m 1000000007 
using namespace std;

int main()
{
	ios::sync_with_stdio(false);
    int t;
    int ans[1001];
    ans[0] = 0;
    ans[1] = 1;
    ans[2] = 1;
    for(int i=3;i<1001;i++){
    	ans[i] = (ans[i-1]%m + (2*ans[i-2])%m)%m;
    }
    cin >> t;
    while(t--){
    	int n;
    	cin >> n;
    	
    	cout << ans[n] << endl;
    }
    return 0;
}
