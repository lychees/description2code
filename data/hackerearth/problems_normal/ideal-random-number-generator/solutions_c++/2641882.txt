#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int a,b,c,sarea,oarea;
    cin>>a>>b>>c;
    sarea = 2*a*b;
    long int max,min;
    if(a>=b)
    {
    	min = b;
    	max = a;
    }
    else
    {
    	min = a;
    	max = b;
    }
    
    	if(c <= min)
    	{
    		oarea = c*c;
    	}
    	else
    	if(c>min && c<max)
    	{
    		oarea =  c*c - (c-min) * (c-min);
    	}
    	else
    	if(c == max)
    	{
    		oarea = c * c - (max-min) * (max-min);
    	}
    	else
    	if(c>max)
    	{
    		oarea =  c * c - (c-min)*(c-min) - (c-max) *(c-max);
    	}
    	
    if(c>(a+b))
    {
    	cout<<"1/1";
    }else
    {
    	int g = __gcd(oarea, sarea);
    	cout<<oarea/g<<"/"<<sarea/g;
    }
    return 0;
}
