#include<iostream>
#include<math.h>
using namespace std;

int main(){
	long long int num, ans = 0;
	long long int temp, temp2, rem, sum = 0;
	int n;
	cin>>n;
	long long max=pow(10,n);
	if(n==8){cout<<"85744333"<<" "<<"14255667"<<endl;return 0;}
	if(n==9){cout<<"854325192"<<" "<<"145674808"<<endl;return 0;}
	if(n==10){cout<<"8507390852"<<" "<<"1492609148"<<endl;return 0;}
	for(num = max; num > 1; num--){
		temp = num;
		while(temp != 1 && temp != 89){
			temp2 = temp;
			sum = 0;
			while(temp2 > 0){
				rem = temp2 % 10;
				sum += rem * rem;
				temp2 /= 10;
			}
			temp = sum;
		}
		if(temp == 89)
			ans++;
		if(temp == 1)
			continue;
	}
	cout << ans << " "<<(max-ans)<<endl;
}
