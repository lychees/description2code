#include <iostream>
#include <stdio.h>


using namespace std;

int main()
{
    int t, n;
    long v;
    
    scanf("%d", &t);
    
    
    for(int i= 1; i<=t; i++)
    {
    	scanf("%d", &n);
    	
    	scanf("%ld", &v);
    	
    	long long a = v;
    	long long b = 0;
    	long long c = 0;
    	
    	for(int j = 1; j<n; j++)
    	{
    		scanf("%ld", &v);
    		c = (a > b)?a:b;
    		
    		a = b+v;
    		b = c;
    		
    	}
  
    	if(a > b)
    		cout<<"Case "<<i<<": "<<a<<endl;
    	else
    		cout<<"Case "<<i<<": "<<b<<endl;
    }
    return 0;
}
