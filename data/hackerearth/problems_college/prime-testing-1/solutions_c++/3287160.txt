#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)

using namespace std;

int t, n;
int a[1000005];

int main()
{
	vector<int> v;
	v.push_back(2);
	v.push_back(3);
	
	for (int i = 4; i <= 1000; i++) {
		int flag = 0;
		for (int j = 2; j <= sqrt(i); j++) {
			if (i % j == 0) {
				flag = 1;
				break;
			}
		}
		if (flag == 0) {
			v.push_back(i);
		}
	}
	
	cin >> t;
	
	while (t--) {
		cin >> n;
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		int cnt = 0;
		for (int i = 0; i < n; i++) {
			if (binary_search(v.begin(), v.end(), a[i])) {
				cnt++;
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
