#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-->0){
    	long n,m,a,b,c,max0,max1,mx;
    	cin>>n>>m>>a>>b>>c;
    	max0=(n*a)+(m*b);
    	
    	if(n>=m){
    		max1=2*c*m+(n-m)*a;
    	}
    	else{
    		max1=2*n*c+(m-n)*b;
    	}
    	
    	
    	if(max0>=max1){
    		mx=max0;
    	}
    	else{
    		mx=max1;
    	}
    	cout<<mx<<endl;
    	
    }
    return 0;
}
