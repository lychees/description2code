#include<bits/stdc++.h>
#define ll long long int
using namespace std;

ll a[100010],n,k,mx1,mx2,mx,tc,temp,*it;

bool check(ll diff,ll k){
	ll lo = 1;
	ll up = n;
	ll i;
	bool flag=1;
	ll sum;
	while(k--){
		if(flag){
			sum = 0;
			for(i=lo+1;i<=up;i++){
				sum = sum+a[i]-a[i-1];
				if(sum > diff)
					break;
			}
		//	cout<<i<<" "<<up<<" a\n";
			if(i > up)
				return 1;
			else if(i == lo+1)
				return 0;
			else
				lo = i-1;
			
		}
		else{
			sum = 0;
			for(i=up-1;i>=lo;i--){
				sum = sum+a[i+1]-a[i];
				if(sum > diff)
					break;
			}
		//	cout<<i<<" "<<lo<<" b\n";
			if(i < lo)
				return 1;
			else if(i == up-1)
				return 0;
			else
				up = i+1;
		}
		flag = !flag;
	}
	return 0;
}
ll bsearch(ll lo,ll hi,ll k){
	bool flag;
	
	while(lo < hi){
		ll mid = (lo+hi)/2;
		flag = check(mid,k);
		if(flag){
			hi = mid;
		}
		else{
			lo = mid+1;
		}
	}
	
	return lo;	
}

	ll i;

	cin>>tc;
	while(tc--){
		cin>>n>>k;
		for(i=1;i<=n;i++)
			cin>>a[i];
	//	cout<<check(4,2);
		cout<<bsearch(1,a[n]-a[1],k)<<"\n";
	}
	return 0;
}
