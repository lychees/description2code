#include <iostream>

using namespace std;

int main()
{
	int testcase;
	cin>>testcase;
	while(testcase--)
	{
		int n;
		cin>>n;
		bool temp_1 = true;
		bool temp_2 = true;
		char** arr = new char*[n];
		for(int i=0; i<n; i++)
		{
			arr[i] = new char[n];
			for(int j=0; j<n; j++)
			{
				cin>>arr[i][j];
			}
		}

		for(int i=0; i<n/2; i++)
		{
			if (temp_1 == false)
				break;
			for(int j=0; j<n; j++)
			{
				if (arr[i][j] != arr[n-i-1][j])
				{
					temp_1 = false;
					break;
				}
			}
		}


		for(int j=0; j<n/2; j++)
		{
			if (temp_2 == false)
				break;
			for(int i=0; i<n; i++)
			{
				if (arr[i][j] != arr[i][n-j-1])
				{
					temp_2 = false;
					break;
				}
			}
		}

		if (temp_1 && temp_2)
			cout<<"BOTH"<<"\n";
		else if (temp_1)
			cout<<"HORIZONTAL"<<"\n";
		else if (temp_2)
			cout<<"VERTICAL"<<"\n";
		else
			cout<<"NO"<<"\n";
	}
	return 0;
}
