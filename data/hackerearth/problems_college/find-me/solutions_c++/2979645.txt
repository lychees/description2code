#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,p;
		cin>>n>>p;
		int a[n],f=-1,c=0;
		for(int i=0;i<n;i++){cin>>a[i];}
		for(int i=0;i<n;i++){
			if(a[i]==p){f=1;}
			else if(a[i]>p){c++;}
		}
		if(f==1){cout<<"YES "<<c<<endl;}
		else{cout<<"NO "<<0<<endl;}
	}
//    cout << "Hello World!" << endl;
    return 0;
}
