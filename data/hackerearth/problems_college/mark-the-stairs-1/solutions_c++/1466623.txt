#include <iostream>

typedef long long ll;

int main() {
  int T;
  std::cin >> T;
  while (T--) {
    ll i;
    std::cin >> i;

    ll j;
    std::cin >> j;

    if (i==j && j%2==0)
      std::cout << 4 * (j/2) << '\n';
    else if (i==j && j%2==1)
      std::cout << 1 + 4 * (j/2) << '\n';
    else if (i==j+2 && j%2==0)
      std::cout << 2 + 4 * (j/2) << '\n';
    else if (i==j+2 && j%2==1)
      std::cout << 3 + 4 * (j/2) << '\n';
    else
      std::cout << "No Number\n";
  }
}
