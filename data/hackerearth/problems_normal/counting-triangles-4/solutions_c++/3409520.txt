#include <iostream>
using namespace std;
typedef long long int lli;
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	lli num;
    	cin>>num;
    	lli ans = 8;
    	ans = ans * num;
    	num = num - 1;
    	num = num * 6;
    	ans = ans + num;
    	cout<<ans<<endl;
    }
    return 0;
}
