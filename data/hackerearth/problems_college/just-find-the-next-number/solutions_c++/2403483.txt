#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T,x,N;
	cin>>T;
	while(T--)
	{
		scanf("%d",&N);
		int flag=0;
		while(1)
		{
			x=3;
			flag=0;
			while(x)
			{
				if((N&x)==x)
				{
					flag=1;
					break;
				}
				x=x<<1;
			}
			if(flag==1)
			N++;
			else
			{
				printf("%d\n",N);
				break;
			}
		}
	}
}
