#include<iostream>
#include<utility>
using namespace std;
#define  max  100006
#define   pii pair<long ,long >
int main()
{
	int t;
	cin>>t;
	long int m,n,g,h,good,great;
	pii u[max],w[max];
	while(t--)
	{
		good=0;great=0;
		cin>>m>>n;
		for(long i=0;i<m;i++)
			cin>>u[i].first>>u[i].second;
		for(long i=0;i<n;i++)
			cin>>w[i].first>>w[i].second;
		cin>>g>>h;
		for(long i=0;i<m;i++)
		{
			for(long j=0;j<n;j++)
			{
				if(u[i].first==w[i].first)
				{
					good++;
					if(u[i].second==u[i].second);
					{
						great++;
						break;
					}
				}
			}
		}
		if(good==g)
		{
			if(great==h)
				cout<<"Great"<<endl;
			else cout<<"Good"<<endl;
		}
		else cout<<":("<<endl;
	}
	return 0;
}
