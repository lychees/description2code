#include<bits/stdc++.h>
#define MAX 10000001
using namespace std;
bool beer[MAX];

void sieve()
{
	for(int i=0;i<MAX;i++)
		beer[i]=true;
	beer[0]=false;
	for(int i=2;i<sqrt(MAX);i++)
	{
		if(beer[i]==true)
		{
			for(int j=2;i*j<MAX;j++)
			{
				beer[i*j]=false;
			}
		}
	}
	for(int i=sqrt(MAX)-1;i>=2;i--)
	{
		if(beer[i]==true)
		{
			beer[i*i]=true;
		}
	}
}

int main()
{
	ios_base::sync_with_stdio(false);
	sieve();
	int n;
	cin >> n;
	while(n--)
	{
		int x;
		cin >> x;
		if(!beer[x])
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
