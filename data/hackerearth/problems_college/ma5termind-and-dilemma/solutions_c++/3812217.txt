#include<bits/stdc++.h>
#define sz 100001
using namespace std;
int main()
{
	int t,n,i,j;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>n>>s;
		int l=0,r=n-1,f=0,g=0;
		while(l<r)
		{
			if(s[l]=='0')
				l++;
			if(s[r]=='1')
				r--;
			if(s[l]=='1' && s[r]=='0' && l<r)	
			{
				f++;
				l++;
				r--;
			}
		}
	//	cout<<s<<endl;
		l=0;r=n-1;
		while(l<r)
		{
			if(s[l]=='1')
				l++;
			if(s[r]=='0')
				r--;
			if(s[l]=='0' && s[r]=='1' && l<r)	
			{
				g++;
				l++;
				r--;
			}
//			cout<<l<<" "<<r<<endl;
		}
//		cout<<f<<" "<<g<<endl;
		cout<<min(f,g)<<endl;
	}
	return 0;
}
