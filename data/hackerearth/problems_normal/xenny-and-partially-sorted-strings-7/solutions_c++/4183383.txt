#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,n,t,k,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>k>>m;
		string s,s1;
		map<string,string>mp;
		while(n--)
		{
			cin>>s;
			s1=s.substr(0,m);
			mp[s1]=s;
		}
		j=0;
		map<string,string> ::iterator it;
		for(it=mp.begin();it!=mp.end();it++)
		{
			j++;
			if(j==k)
			{
				cout<<it->second<<endl;
				break;
			}
		}
		
	}
	return 0;
}
