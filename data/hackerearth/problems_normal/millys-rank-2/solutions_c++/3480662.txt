#include <iostream>
using namespace std;

int main()
{
   int t;
   long long int n,x,y,leftx,lefty;
   cin>>t;
   while(t--){
   	cin>>n>>x>>y;
   	leftx=x+1;
   	// rightx=n;
    lefty=n-y;
    // righty=n;
    if(leftx>lefty){
    cout<<n-leftx+1<<endl;	
    } 
   else{
   	cout<<n-lefty+1<<endl;
   }	
   
   	
   }
    return 0;
}
