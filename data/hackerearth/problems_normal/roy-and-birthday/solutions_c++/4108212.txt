#include <iostream>
#include <cstring>
#include <map>
using namespace std;
typedef long long int ll;
ll calc(ll a, ll b, ll mod) {
	ll ans = 1;
	while(b > 0) {
		if(b%2 == 1) {
			ans = ((ans % mod) * (a % mod)) % mod;
		}
		a = ((a%mod)* (a%mod)) % mod;
		b/=2;
	}
	return ans;
}
ll f[1000002];
int main()
{
	ll i;
	f[0] = 1;
	ll mod = 1000000007;
	for(i = 1; i <= 1000002; i++) {
		f[i] = (i * (f[i - 1] % mod)) % mod;
	}
	ll t;
    cin >> t;
    while(t--) {
    	string s;
    	map<char, ll> m;
    	map<char, ll> :: iterator it;
    	cin >> s;
    	ll c1 = 0, c2 = 0;
    	ll l = s.size();
    	for(i = 0; i < l; i++) {
    		if(s[i] == 'a' || s[i] == 'e' || s[i] == 'o' || s[i] == 'i' || s[i] == 'u') {
    			c1++;
    		}
    		else {
    			c2++;
    		}
    		m[s[i]]++;
    	}
    	if(c2 < c1 - 1) {
    		cout << "-1\n";
    	}
    	else {
    		ll ans = f[c2];
    		ans = ((ans%mod) * f[c1])%mod;
    		for(it = m.begin(); it != m.end(); it++) {
    			ans = ((ans%mod) * calc(f[it->second], mod - 2, mod))%mod;
    		}
    		ll ans2 = calc(f[c1], mod - 2, mod);
    		ll ans3 = calc(f[c2 + 1 - c1], mod - 2, mod);
    		ans = (((ans % mod) * (ans2 % mod))%mod) * (ans3 % mod);
    		ans = ans % mod;
    		ans = ans * f[c2 + 1];
    		ans = ans % mod;
    		cout << ans << endl;
    	}
    }
    return 0;
}
