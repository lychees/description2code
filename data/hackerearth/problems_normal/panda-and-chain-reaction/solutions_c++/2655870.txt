#include<bits/stdc++.h>
#define ll unsigned long long int
#define inp(x) scanf("%llu",&x)
using namespace std;
const ll M = 1000003;
static ll f[M+2];
inline void pre(){
	f[0]=1;
	for(ll i=1;i<=M;++i){
		f[i] = (i*f[i-1])%M;
	}
	return ;
}
int main(){
	ll t,n,x;
	pre();
	inp(t);
	while(t--){
		inp(n);inp(x);
		if(n>=M){
			printf("0\n");
		}else{
			printf("%llu\n",(x*f[n])%M);
		}
	}
	return 0;
}
