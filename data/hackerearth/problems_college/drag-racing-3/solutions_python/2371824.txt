import sys
dDrivers = {}

iN = int( sys.stdin.readline().strip() )

for iI in range(pow(2,iN) -1):
    lMatchSummary = sys.stdin.readline().strip().split()
    sWinner = lMatchSummary[0]
    sLoser = lMatchSummary[1]
    if not dDrivers.has_key(sWinner):
        dDrivers[sWinner] = 1 # 1 means victory
    
    dDrivers[sLoser] = 0
    
for key,value in dDrivers.iteritems():
    if value == 1:
