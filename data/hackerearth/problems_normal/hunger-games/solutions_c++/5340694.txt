#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int *arr=new int[n];
    int *ans=new int[n];
    for(int i=0;i<n;i++)
     cin>>arr[i];
     
     sort(arr,arr+n);
     int l=0;
    int tt=n-1;
    int flag=0;
     for (int i=n-1;i>=0;i--)
       {
       	if(flag==0)
       	{
       	  ans[l]=arr[i];
       	   flag=1;
       		l++;
       	}
       	else
       	{
       		ans[tt]=arr[i];
       		tt--;
       		flag=0;
       	}
       }
       int MMax= INT_MIN;
       int tem;
       for(int i=0;i<n-1;i++)
       {
       	tem= abs(ans[i]-ans[i+1]);
       	if(tem>MMax)
       	 MMax=tem;
       }
       cout<<MMax<<endl;
    return 0;
}
