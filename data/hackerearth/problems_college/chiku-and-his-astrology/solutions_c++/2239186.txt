#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s;
    int l,e,o,sum;
    while(t--)
    {
    	cin>>s;
        l=s.length();
        o=0;
        e=0;
        for(int i=0;i<l;i++)
        {
            if(s[i]=='1')
                o++;
            else
                e++;
        }
        sum=abs(o-e);
        printf("%d \n",sum);
    }
return 0;
}
