#include<iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        string s1, s2;
        cin>>s1>>s2;
        int arr[300];
        for(int i=0; i<300; i++)
            arr[i] = 0;
        for(int i=0; s1[i] !='\0'; i++)
        {
            arr[s1[i]-'0']++;
            arr[s2[i]-'0']--;
        }

        int len = s1.length();
         bool ans = true;

        for(int i=len; s2[i]!='\0'; i++)
        {
            arr[s2[i]-'0']--;
            arr[s2[i-len]-'0']++;

            ans = true;
            for(int i=0; i<300; i++)
            {
                if(arr[i] != 0)
                {
                    ans = false;
                    break;
                }
            }

            if(ans == true)
                break;

        }

        for(int i=0; i<300; i++)
            {
                if(arr[i] != 0)
                {
                    ans = false;
                    break;
                }
            }


        if(ans == true)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }

    return 0;
}
