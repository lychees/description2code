long long int cutMilk(long long int price[], int n)
{
   long long int val[n+1];
   val[0] = 0;
   long long int i, j;
   for (i = 1; i<=n; i++){
       long long int maxVal = -1;
       for (j = 0; j < i; j++)
       maxVal = max(maxVal, price[j] + val[i-j-1]);
       val[i] = maxVal;
   }
   return val[n];
}

int main()
{	int T=0;
	long int L;
	long long int prices[10000],temp;
	cin >> T;
	while(T--){
		//resetPrices(prices);
		cin >> L;
		for(int i=0;i<L;i++){
			cin >> temp;
			prices[i] = temp;
		}
		cout << cutMilk(prices,L)<<endl;
	}
    return 0;
}
