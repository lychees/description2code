#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	getchar();
	//fflush(stdin);
	while(t--)
	{

		string s;
		getline(cin,s);
		vector<string> v;

		string tmp;
		for(int i=0;i<s.size();i++)
		{
			if(s[i]!=' ')
				tmp+=s[i];
			else
			{
				v.push_back(tmp);
				tmp="";
			}
		}
		v.push_back(tmp);

		for(int i=v.size()-1;i>=0;i--)
			cout<<v[i]<<" ";
		cout<<endl;

	}
	return 0;
}
