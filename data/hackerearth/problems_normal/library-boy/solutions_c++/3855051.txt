#include <bits/stdc++.h>
using namespace std;
int main()
{
	int N;
	cin>>N;
	int hashchar[26];
	memset(hashchar,0,sizeof(hashchar));
	string books[N+1];
	for(int i=0;i<N;i++)
	{
		cin>>books[i];
	}
	for(int i=0;i<N;i++)
	{
		int index=books[i][0]-'a';
		hashchar[index]++;
	}
	int res=0;
	for(int i=0;i<26;i++)
	{
		res+=ceil(float(float(hashchar[i])/10.0));
	}
	cout<<res<<endl;
	return 0;
}
