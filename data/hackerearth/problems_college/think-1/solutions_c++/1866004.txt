#include <iostream>
using namespace std;

int main()
{
	int tt; cin >> tt;
	while(tt--) {
		int n; cin >> n;
		int t = 0;
		int x = n;
		while(x > 0) {
			t = t * 10 + x%10;
			x /= 10;
		}
		if(n == t) {
			cout << "No\n";
			continue;
		}	
		string s = to_string(1ll * n * n);
		string r = to_string(1ll * t * t);
		for(int i = 0, j = r.size() -1;i < j;i++,j--)
		   swap(r[i],r[j]) ;  
		//cout << s << " " << r << endl;
		if(s == r) cout << t << endl;
		else cout << "No\n";
	}
		
  	return 0;
}
