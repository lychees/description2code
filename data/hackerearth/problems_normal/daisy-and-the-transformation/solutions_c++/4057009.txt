#include <iostream>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    
    int t;
    cin>>t;
    while(t--)
    {
    	int x=0;
    	string s1;
    	string s2;
    	int n;
    	cin>>s1>>s2>>n;
    	for(int i=0;i<s1.length();i++)
    	{
    		if (s1[i]!=s2[i])
    		x++;
    	}
    	if(x==n)
    	cout<<"Yes"<<endl;
    	else{
    		if(x<n && (n-x)%2==0)
    		cout<<"Yes"<<endl;
    		else
    		cout<<"No"<<endl;
    	}
    }
    return 0;
}
