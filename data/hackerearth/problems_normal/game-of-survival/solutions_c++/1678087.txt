#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{

    int t,n,an,sw,i;

    cin>>t;
    while(t--)
    {
         vector<int>a,s;

        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>sw;
            s.push_back(sw);

        }

        for(i=0;i<n;i++)
         {
             cin>>an;
             a.push_back(an);
         }

         sort(s.begin(),s.end());
         sort(a.begin(),a.end());

         int flag=0;

         for(i=n-1;i>=0;i--)
         {
             if(s[i]<a[i])
             {
                 flag=1;
                 break;
             }

         }
         if(flag==0)
            cout<<"YES"<<endl;
         else
            cout<<"NO"<<endl;

    }

    return 0;
}
