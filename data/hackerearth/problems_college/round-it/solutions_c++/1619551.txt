#include <iostream>
#include <stdio.h>

using namespace std;

typedef long long int Lint;

int main()
{
	
	Lint t, n, b, a, v, d1, d2;
	
	scanf("%lld", &t );
	
	while( t-- ) {
		
		scanf("%lld %lld", &n, &b );
		a = ( n / b ) * b;
		v = a + b;
		d1 = n - a;
		d2 = v - n;
		if( d1 >= d2 )
			printf("%lld\n", v );
		else
			printf("%lld\n", a );
	}
	
	return 0;
}
