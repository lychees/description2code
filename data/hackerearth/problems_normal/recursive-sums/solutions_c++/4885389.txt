#include<bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	cin>>T;
	
	while(T--)
	{
		int M,i;
		cin>>M;
		
		unsigned long long int Sum=0;
		
		vector<unsigned long long >V(M);
		vector<int >D(M);
		
		for(i=0;i<=M-1;i++)
		{
			cin>>V[i]>>D[i];
			
			if(D[i]!=0)
			{
				if(D[i]%9==0 || V[i]%9==0 || (V[i]*D[i]%9==0) )
				{
					Sum+=9;
				}
				else
				{
					Sum+=(D[i]*V[i])%9;
				}
			}
		}
		
		if(Sum==0)
		{
			cout<<"0\n";
		}
		else
		{
			if(Sum%9==0)
			{
				cout<<"9\n";
			}
			else
			{
				cout<<Sum%9<<"\n";
			}
		}	
		
	}
	

 return 0;	
}
