#include <bits/stdc++.h>
using namespace std;
#define elif else if
#define int64 long long int
int main()
{
    int64 n,c,i,l;string s;char prev;
    cin>>n;
    while(n--)
    {
        cin>>s;c=0;
        l=s.length();prev=s[0];
        for(i=1;i<l;i++)
        {
            if(s[i]==prev)
                c++;
            else
                prev=s[i];
        }
        cout<<c<<endl;
    }
    return 0;
}
