#include <bits/stdc++.h>
using namespace std;

int main()
{
    int hash[200]={0};
    char  str[1005];
    gets(str);
    for(int i=0;i<strlen(str);i++){
    	int x=str[i];
    	
    	hash[x]++;
    }
    
   
    int maxm=0;
    char ch;
    for(int i=0;i<200;i++){
    	if(hash[i]>maxm){
    		ch=i;
    		maxm=hash[i];
    	}
    }
    cout<<ch<<" "<<maxm<<endl;
    return 0;
}
