#include <iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
{
   // cout << "Hello World!" << endl;
   long long t,a,i,n,k,count,max,num,j,rem,sum,index;
   cin>>t;
   assert(1<=t<=10);
   while(t--)
   {
   		cin>>n>>k;
   		assert(1<=n<=103);
   		assert(1<=k<=n);
   		sum=0;
   		long long hash[106];
   		for(i=0;i<n;i++)
   		{
   			cin>>num;
   			assert(0<=num<=105);
   			hash[i]=__builtin_popcount(num);
   		}
   		sort(hash,hash+n);
    	reverse(hash,hash+n);
    	for(a=0;a<k;a++)
    	sum+=hash[a];
    	cout<<sum<<endl;
   }
    return 0;
}
