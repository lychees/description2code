#include <iostream>
using namespace std;

int main()
{
    int cases,a,b,c,d,x,y,count;
    cin >> cases;
    while(cases--)
    {
    	cin >> a >> b >> c >> d;
    	if(b<a){a=b+a; b=a-b; a=a-b;}
    	if(c>d){c=d+c; d=c-d; c=c-d;}
    	count=0;
    	for(x=a;x<=b;x++)
    	{
    		for(y=c;y<=d;y++)
    		 {
    		 	if(x==y)
    		 	count++;
    		 }	
    	}
    	if(count==0)
    	 cout << "Line" << endl;
    	else if(count==1)
    	 cout << "Point" << endl;
    	else
    	 cout << "Nothing" <<endl;
    }
    return 0;
}
