#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q;
    cin>>n>>q;
    string S;
    cin>>S;
    while(q--)
    {
        int a,b;
        cin>>a>>b;
        bool bb=true;
        while(a!=b and bb)
        {
            if(S[a-1]==49 or S[b-1]==49) bb=false;
            if(bb and a<b) b/=2;
            else if(bb and b<a) a/=2;
        }
        if(S[a-1]==49 or S[b-1]==49) bb=false;
        if(bb) cout<<"Yes\n";
        else cout<<"No"<<endl;

    }
}
