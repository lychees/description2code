#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int n;
    static int prime[4000000];
    unsigned long long int sum=2;
    cin>>n;
    if(n<=2)
    {
    	cout<<0<<endl;
    	return 0;
    }
    int sq=sqrt(n);
    for(int i=3;i<=sq;i+=2)
	{
		if(prime[i]==0)
		{
			for(int j=i*i;j<=n;j+=i)
			{
				prime[j]=1;
			}
		}
	}
	for(long long int i=3;i<n;i+=2)
	{
		if(prime[i]==0)
		{
			sum=sum+i;
		}
	}
	cout<<sum<<endl;
    return 0;
}
