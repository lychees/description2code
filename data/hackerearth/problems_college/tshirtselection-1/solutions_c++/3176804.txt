#include<bits/stdc++.h>
using namespace std;

int main() {
	int t,i,j,k,n;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		int ar[n];
		int cur=0,a=0,b=0,j=0;
		map<int,int>mp;
		for(i = 0; i < n; i++) {
			scanf("%d",&ar[i]);
			k = ar[i];
			if(mp[k] == 0) {
				mp[k]++;
				cur++;
			}
			else {
				if(b-a < cur) {
					b = i;
					a = j;
				}
				while(j < i) {
					cur--;
					j++;
					mp[ar[j-1]]--;
					if(ar[j-1] == k) {
						mp[ar[j-1]]++;
						cur++;
						break;
					}
				}
			}//cout<<cur<<" ";
		}
		if(b-a < cur) {
			b = i;
			a = j;
		}
	//	b++;
		a++;
		cout<<a<<" "<<b<<endl;
	}
	return 0;
}
