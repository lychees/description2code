#include<iostream>
#include<string.h>
using namespace std;
int main()
{		int ta;
		cin>>ta;
		
		while(ta--)
		{
		char p[1000],t[100000];
		cin>>p>>t;
		int plen=strlen(p);
		int tlen=strlen(t);
		int parr[26]={0},tarr[26]={0};
		for(int i=0;i<plen;i++)
		{
				parr[p[i]-'a']++;
				tarr[t[i]-'a']++;
		}
		int yes=0;
		for(int i=0;i<=tlen-plen;i++)
		{		int flag=0;
				for(int j=0;j<26;j++)
				{
					if(tarr[j]!=parr[j])
					{
						flag=1;
					}
				}
				if(flag==0)
				{
						cout<<"YES\n";
						yes=1;
						break;
						
				}
				tarr[t[i]-'a']--;
				tarr[t[plen+i]-'a']++;
				
		}
		if(yes==0)cout<<"NO\n";
		}
		return 0;
}
