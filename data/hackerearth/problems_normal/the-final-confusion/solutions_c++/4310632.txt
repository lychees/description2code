#define vll vector<vector<long long> >

long long func(vll &dp,int n,int m)
{

	if(dp[n][m]!=-1)
		return dp[n][m];
	if(n==0)
		return dp[n][m]=1;
	dp[n][m]=func(dp,n-1,m+1);
	if(m>0)
		dp[n][m]+=func(dp,n,m-1);
	return dp[n][m];
}

		vector<vector<long long> > dp(n+1,vector<long long>(2*n+1,-1));
		cout<<(long long)func(dp,n,0)<<endl;
	}
	return 0;
}
