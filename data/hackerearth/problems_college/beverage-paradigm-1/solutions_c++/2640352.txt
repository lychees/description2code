#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int n,i,j,k,a[100001],X,sum;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	cin>>X;
	int f=0;
	sort(a,a+n);
	for(i=0;i<n-2;i++)
	{
		j=i+1;
		k=n-1;
		while(j<k)
		{
			sum=a[i]+a[j]+a[k];
			if(sum==X)
			{
				cout<<"True\n";
				f=1;
				break;
			}
			else if(sum>X)
				k--;
			else
				j++;
		}
		if(f==1)
			break;
	}
	if(f==0)
		cout<<"False\n";
}
