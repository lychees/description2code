#include <iostream>
#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int main()
{
    ll n,g,s,q,ans=0,p=0;
    cin>>n;
    ll arr[1000005]={0};
    for(ll i=0;i<n;i++){
        cin>>s>>q;
        arr[s]=q;
    }
    for(ll i=1000000;i>0;i--){
     if(p<arr[i])
     p=arr[i];
     arr[i]=p;
    }
    cin>>g;
    while(g--){
        cin>>s;
        if(s>0)
        ans+=(arr[s]/s)*100;
    }
    cout<<ans<<endl;
    return 0;
}
