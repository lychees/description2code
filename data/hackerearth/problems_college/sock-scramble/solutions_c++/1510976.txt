#include <iostream>
using namespace std;

int main()
{
	int t, n, *a;
	
	cin >> t;
	while(t>=1 && t<=20){
		cin >> n;
		a = new int[n];
		int b[10000] = {0};
		int i = n;
		
		if(n < 1 && n > 50) break;
		
		for(int i = 0; i < n; i++){
			cin >> a[i];
			b[a[i]]++;
		}
		
		for(int i = 1; i < 10000; i++){
			if(b[i]%2 == 1){
				cout << i << endl;
			}
		}
		
		t--;
	}
	return 0;
}
