#include<bits/stdc++.h>
using namespace std;
int main()
{
    int ceo,coo,cto;
    cin>>ceo>>coo>>cto;
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    }
    sort(a,a+n);
    long int sum=0;
    if(n%2==0)
    {
    	for(int i=1;i<n/2;i++)
    	{
    		sum+=abs(a[i]-a[i-1]);
    	}
    	sum+=(abs(ceo-a[n/2]));
    	sum+=(abs(ceo-a[n/2-1]));
    	for(int i=n/2+1;i<n;i++)
    	{
    		sum+=abs(a[i]-a[i-1]);
    	}
    	if(coo<cto)
    	{
    		sum+=abs(coo-a[0]);
    		sum+=abs(cto-a[n-1]);
    	}
    	else
    	{
    		sum+=abs(coo-a[n-1]);
    		sum+=abs(cto-a[0]);
    	}
    }
    else
    {
    	for(int i=1;i<=n/2;i++)
    	{
    		sum+=abs(a[i]-a[i-1]);
    	}
    	sum+=(abs(ceo-a[n/2+1]));
    	sum+=(abs(ceo-a[n/2]));
    	for(int i=n/2+2;i<n;i++)
    	{
    		sum+=abs(a[i]-a[i-1]);
    	}
    	if(coo<cto)
    	{
    		sum+=abs(coo-a[0]);
    		sum+=abs(cto-a[n-1]);
    	}
    	else
    	{
    		sum+=abs(coo-a[n-1]);
    		sum+=abs(cto-a[0]);
    	}
    }
    cout<<sum<<endl;
}
