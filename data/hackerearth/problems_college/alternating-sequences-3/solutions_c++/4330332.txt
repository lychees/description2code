#include <stdio.h>
#include <cstdlib>

long long opp (long long a, long long b)
{
	long long x = a/abs(a);
	long long y = b/abs(b);

	if (x*y == -1) return 1;
	return 0;
}

long long lis(long long arr[], long long n )
{
	long long *lis, i, j, max = 0;
	lis = (long long*) malloc ( sizeof(long long) * n );

	for (i = 0; i < n; i++ )
		lis[i] = 1;

	for (i = 1; i < n; i++ )
		for (j = 0; j < i; j++ )
			if (abs(arr[i]) > abs(arr[j]) && lis[i] < lis[j] + 1 && opp(arr[i],arr[j]))
				lis[i] = lis[j] + 1;

	for (i = 0; i < n; i++ )
		if (max < lis[i])
			max = lis[i];

	free(lis);

	return max;
}

int main ()
{
	long long n;
	scanf ("%lld",&n);
	long long arr[5005];
	for (long long i = 0 ; i < n ; i++) scanf ("%lld",&arr[i]);

	long long ans = lis (arr, n);
	printf ("%lld\n",ans);
	return 0;
}
