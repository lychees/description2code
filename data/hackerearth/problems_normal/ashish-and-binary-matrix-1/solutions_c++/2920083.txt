#include <bits/stdc++.h>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--){
    	set<string> st;
    	int n,m;
    	string s;
    	cin>>n>>m;
    	for(int i=0;i<n;i++){
    		cin>>s;
    		st.insert(s);
    	}
    	if(st.size()==n){
    		cout<<"Yes\n";
    	}else{
    		cout<<"No\n";
    	}
    }
    return 0;
}
