#include <iostream>
#define MAX 1000000
using namespace std;
void sieve(int *prime) {
	prime[0] = prime[1] = 0;
	prime[2] = 1;
	prime[3] = 1;
	for(long i=4;i<=MAX;i++) prime[i] = 1;
	for(long i=2;i<=MAX;i+=1) {
		if(prime[i] == 1) {
			for(long j=i+i;j<=MAX;j+=i) {
				prime[j] = 0;
			}
		}
	}
}
int main()
{
	long n;
	cin>>n;
	int prime[MAX+1];
	sieve(prime);
	long x[n];
	
	for(long i=0;i<n;i++) cin>>x[i];
	long largest_prime = 0;
	for(long i=0;i<n;i++) {
		if(prime[x[i]] == 1 && x[i] > largest_prime) {
			largest_prime = x[i];
		}
	}
	cout<<largest_prime<<endl;
	long count = 0;
	for(long i=0;i<n;i++) {
		if(x[i] == largest_prime) count++;
	}
	cout<<count<<endl;
    return 0;
}
