#include <bits/stdc++.h>
#define PB push_back
#define MP make_pair
#define LL long long
#define MAXN 2123
#define M 1000000007
#define F first
#define S second
#define sc(x) scanf("%d",&x)
#define scl(x) scanf("%lld",&x)
using namespace std;

LL ncr[MAXN + 1][MAXN + 1];

void pre(){
    int i, j;
    for(i=0;i<=MAXN;i++){
        for(j=0;j<=MAXN;j++)
            ncr[i][j] = 0;
    }

    for(i=0;i<=MAXN;i++)
        ncr[i][0] = 1;

    for(i=1;i<=MAXN;i++){
        for(j=1;j<=i;j++){
            ncr[i][j] = ncr[i - 1][j] + ncr[i - 1][j - 1];
            ncr[i][j] %= M;
        }
    }
}

pair<int, int > p[MAXN];

int main(){
    int n, t, i, j, m;
    LL ans;
    pre();
    cin>>t;
    while(t--){
        cin>>n>>m;
        p[1] = MP(0, 0);
        for(i=2;i<=m+1;i++){
            cin>>p[i].F>>p[i].S;
        }
        p[m + 2] = MP(n - 1, n - 1);
        sort(p + 1, p + m + 2 + 1);

        ans = 1;

        for(i=2;i<=m + 2;i++){
            ans = ans * (ncr[p[i].F + p[i].S - p[i - 1].F - p[i - 1].S][p[i].F - p[i - 1].F]);
            ans %= M;
        }
        cout<<ans<<endl;
    }
}
