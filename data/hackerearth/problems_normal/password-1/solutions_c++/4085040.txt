#include <bits/stdc++.h>
using namespace std;

int ispalindrome(string s1,string s2)
{
	int l=s1.length(),i,j;
	if(l!=s2.length())
	{
		return 0;
	}
	for(i=0;i<l;i++)
	{
		if(s1[i]!=s2[l-i-1])
		{
			return 0;
		}
	}
	return 1;
}


int main()
{
	int n,i,j;
	cin>>n;
	string s[n];
	for(i=0;i<n;i++)
	{
		cin>>s[i];
		for(j=0;j<i;j++)
		{
			if(ispalindrome(s[i],s[j]))
			{
				int l=s[i].length();
				cout<<l<<" "<<s[i][(l-1)/2]<<endl;
			}
		}
	}
    return 0;
}
