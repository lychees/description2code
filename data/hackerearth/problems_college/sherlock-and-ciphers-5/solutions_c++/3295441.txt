#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,l,s,i,j;
    char msg[205];
    cin>>n;
    while(n--)
    {
        scanf(" %[^\n]s",msg);
        l=strlen(msg);
        s=sqrt(l);
        for(i=0;i<s;i++)
        {
            for(j=i;j<l;j+=s)
            {
                printf("%c",msg[j]);
            }
        }
        printf("\n");
    }
    return 0;
}
