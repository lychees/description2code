#include <bits/stdc++.h>
using namespace std;

int a[1000007];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	memset(a,0,sizeof a);
    	int n,p;
    	cin>>n>>p;
    	int ans=0;
    	for(int i=0;i<n;i++)
    	{
    		int d;
    		cin>>d;
    		a[d]++;
    		if(d>p) ans++;
    	}
    	if(a[p]>0)
    	{
    		cout<<"YES "<<ans<<endl;
    	}
    	else cout<<"NO "<<0<<endl;
    	
    }
    return 0;
}
