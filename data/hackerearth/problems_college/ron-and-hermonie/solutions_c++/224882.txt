#include <iostream>
#include <cstdio>
#include <cmath>
#include <string>
using namespace std;
typedef unsigned long long int ull;
int main()
{
    ull t;
    cin>>t;
    while(t--)
    {
    	ull b,e;
    	scanf("%llu %llu",&b,&e);
    	int mod=e%4;
    	if(mod==0)
    	mod=4;
    	int a=b%10;
    	long long p=pow(a,mod);
    	int tmp=p%10;
    	printf("%d\n",tmp);
    }
    return 0;
}
