#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;
int main(){
	string s;
	cin>>s;
	int n;
	scanf("%d",&n);
	int t=0,p=0;
	while(n--)
	{
		string a;
		cin>>a;
		if(a.length()==s.length()){
			if(s[0]==a[0])t=t+1;
			else if(a[0]-s[0]==3)p=p+1;
		}
	}
	printf("%d %d\n",t,p);
}
