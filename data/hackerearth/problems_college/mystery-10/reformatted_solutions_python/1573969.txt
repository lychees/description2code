times = input()
inp = 'abcdefghijklmnopqrstuvwxyz'
outp = 'qwertyuiopasdfghjklzxcvbnm'
for _ in range(times):
    in1 = raw_input()
    lst = list(in1)
    for i in range(len(lst)):
        char = lst[i]
        if char in inp:
            index = inp.index(char)
            lst[i] = outp[index]
    else:
        print "".join(lst)
