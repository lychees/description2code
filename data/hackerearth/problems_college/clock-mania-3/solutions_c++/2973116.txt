#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int h,m;
		cin>>h>>m;
	
    	if(h==12) h=0;
		if(m==60) m=0;
	
		int HA=0.5*(h*60+m);
		int MA=6*m;

		int angle=abs(HA-MA);

		angle=min(360-angle, angle);
		cout<<angle<<endl;
	}
	return 0;
}
