#include <iostream>
#include <map>
using namespace std;

int main()
{
    unsigned int days=0;
    cin>>days;
    unsigned int age=0;
    cin>>age;
    map<unsigned int,int>a;
    days--;
    unsigned int x;
    cin>>x;
    unsigned int consistent=x;
    a[x]++;
    cout<<x<<" "<<a[x]<<endl;
    while(days--){
    	cin>>x;
    	a[x]++;
    	if(a[x] > a[consistent] ||(a[x]==a[consistent] && x > consistent)){
    		consistent = x;
    	}
    	cout<<consistent<<" "<<a[consistent]<<endl;
    }
    return 0;
}
