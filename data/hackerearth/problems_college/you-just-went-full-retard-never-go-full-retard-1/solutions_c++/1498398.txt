#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;


int isPrime(long n){
    int i;

    if (n==2)
        return 1;

    if (n%2==0)
        return 0;

    for (i=3;i<=sqrt(n);i+=2)
        if (n%i==0)
            return 0;

    return 1;
}

int main(){
    int i,t;
    long n;
	cin>>t;
	while(t--){
		cin>>n;
        if(isPrime(n))
         	printf("YES\n");
        else 
        	printf("NO\n");
   	}
	return 0;
}
