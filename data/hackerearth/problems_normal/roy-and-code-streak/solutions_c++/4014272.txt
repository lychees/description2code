#include <iostream>
#include <bits/stdc++.h>
#include <unordered_map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	
    	int ma=0,c=0;
    	unordered_map< int, int > m;
    	for(int i=0;i<n;i++)
    	{
    		int a,b;
    		cin>>a>>b;
    		if(b==0)
    		c=0;
    		else
    		{
    			if(m[a]==0)
    			{
    				c++;
    				m[a]=1;
    			}
    		}
    		ma=max(ma,c);
    	}
    	cout<<ma<<endl;
    }
    return 0;
}
