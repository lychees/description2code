#include <iostream>
using namespace std;

long long int gcd(long long int a,long long int b)
{
	if(b == 0)
	return a;
	long long int t=b;
	b=a%b;
	a=t;
	return gcd(a,b);
}

int main()
{
    int t,k;
    long long int n;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	long long int ct=0;
    	int w;
    	for(int i=0;i<n;i++)
    	{
    		cin>>w;
    		if(w == k)
    			ct++;
    	}
    	//cout<<ct<<endl;
    	long long int numerator=(ct*(ct-1)*(ct-2))/6;
    	//cout<<numerator<<endl;
    	long long int denumerator =(n*(n-1)*(n-2))/6;
    	//cout<<denumerator<<endl;
    	long long int g=gcd(numerator,denumerator);
    	//cout<<g<<endl;
    	numerator/=g;
    	denumerator/=g;
    	cout<<numerator<<"/"<<denumerator<<endl;
    }
    return 0;
}
