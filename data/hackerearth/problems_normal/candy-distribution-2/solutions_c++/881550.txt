#include <iostream>
using namespace std;

int main()
{
	int T,N,i,j;
   cin>>T;
   unsigned long long int S[T];
   
   for(i=0;i<T;i++)
   {
   	S[i]=0;
   	cin>>N;
   	unsigned long long int A[N];
   	for(j=0;j<N;j++)
   	{
   	cin>>A[j];
   	S[i]=S[i]+A[j];
   	}
   	S[i]=S[i]-N+1;
   }
   
   for(i=0;i<T;i++)
   cout<<S[i]<<endl;
   
    return 0;
}
