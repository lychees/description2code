#include <bits/stdc++.h>
using namespace std;
int sieve[1000001];
int main()
{
	int i,j,n,max,count,A[100001];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>A[i];
	for(i=2;i<=1000000;i++)
	{
		j=2;
		while(i*j<=1000000)
		{
			sieve[i*j]=1;
			j++;
		}
	}
	max=0;
	count=0;
	for(i=0;i<n;i++)
	{
		if(A[i]>=4 && sieve[A[i]]==0)
		{
			if(max==A[i])
				count++;
			if(max<A[i])
			{
				count=1;
				max=A[i];
			}
		}
	}
	cout<<max<<endl;
	cout<<count<<endl;
	return 0;
}
