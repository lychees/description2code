#include <iostream>
using namespace std;



int pivot(int a[], string &s2 , int l , int h){
	int p = a[s2[h]];
	int i = l;
	for(int j = l ; j<h; j++){
	//	cout<<j<<endl;
		if(a[s2[j]] < p){
			char c;
			c = s2[i];
			s2[i] = s2[j];
			s2[j] = c;
			i = i+1;
		}
	}
	char t = s2[i];
	s2[i] = s2[h];
	s2[h] = t;
	return i;
}

void quicksort(int a[], string &s2, int l, int h){
	if(l < h){
		int p = pivot(a, s2, l, h);
		quicksort(a , s2 , l , p-1);
		quicksort(a , s2, p+1 , h);
		
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		int a[256];
		for(int i=0; i<256; i++){
			a[i] = 0;
		}
		
		string s1;
		string s2;
		cin>>s1;
		for(int i=0; i<26; i++){
			a[s1[i]] = i+1;
		}
		
		cin>>s2;
		quicksort(a,s2,0,s2.length()-1);
		cout<<s2<<endl;
	}
    return 0;
}
