#include <stdio.h>
#include <cstring>
using namespace std;

char num[105];

int main(){
	int t;
	scanf ("%d ", &t);
	while (t--){
		scanf ("%s", &num);
		int len=strlen(num);
		int i, j;
		for (i=0, j=len-1; i<=j; i++, j--){
			if ((num[i]=='1' || num[i]=='8' || num[i]=='0') && num[i]==num[j])
				continue;
			break;
		}
		if (i<=j)
			printf ("NO\n");
		else
			printf ("YES\n");
	}
    return 0;
}
