#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
	int n, k, l, q, p, i, j;
	string s;
	
	cin>>n>>k>>l>>q;
 
	map<vector<int>, string> m;
	
	vector<int> v;
	
	for (i = 0; i < n; i++) {
		s.clear();
		v.clear();
		cin>>s;
		for (j = 0; j < k; j++) {
			cin>>p;
			//cout<<p<<" ";
			v.push_back(p);
		}
		m[v] = s;
	}
 
	for (i = 0; i < q; i++) {
		v.clear();		
		for (j = 0; j < k; j++) {
			cin>>p;
			v.push_back(p);
		}
		if (m.count(v)==0) {
			printf("You cant fool me :P\n");
		} else {
			cout << m[v] << endl;
		}
	}
 
	return 0;
}
