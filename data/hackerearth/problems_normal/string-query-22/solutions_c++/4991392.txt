#define MAX 200005

int n, siz, a, b;
char ch;
string s;
vector<bool> ban(200005 , false);

int BIT[30][MAX];

int next(int i)
{
  return i + (i&(-i));
}

int parent(int i)
{
  return i - (i&(-i));
}

void update(int ent, int ind, int val)
{
  while(ind <= siz + 5)
  {
    BIT[ent][ind] += val;
    ind = next(ind);
  }
}

int query(int x, int y)
{
  int temp = 0;

  while(y > 0)
  {
    temp += BIT[x][y];
    y = parent(y);
  }

  return temp;
}

int find(int x, int c)
{
  int start = 1;
  int end = siz + 5;
  int ind;

  while(start <= end)
  {
    int mid = (start + end)>>1;

    if(query(c,mid) >= x)
    {
      ind = mid;
      end = mid - 1;
    }
    else
      start = mid + 1;
  }

  return ind;
}
