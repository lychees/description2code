#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,l,r,a[100002],i;
    long long sum;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
    }
    scanf("%d",&q);
    while(q--)
    {
        sum=0;
        scanf("%d%d",&l,&r);
        for(i=l;i<=r;i++)
        {
            sum=sum+a[i];
        }
        printf("%lld\n",sum);
    }
    return 0;
}
