#include <iostream>
#include <vector>
using namespace std;

const int M = 1e9 + 7;

int main()
{
    vector<int> v(95);
    v[0] = 1;
    v[1] = 0;
    for (int i = 2; i < 95; ++i) {
    	v[i] = ((long long int)(i - 1) * ((v[i-1] + v[i-2]) % M)) % M;
    }
    int n;
    string s;
    cin >> n;
    cin >> s;
    cout << ((n == 1) ? 1 : v[n]);
    return 0;
}
