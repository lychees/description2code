
t = int(raw_input())
for i in xrange(t):
    A, B, C, K = map(int, raw_input().split())
    left, right = 0, 100000
    ans = 0
    while left <= right:
        mid = (left + right)/2
        p = A*mid**2+B*mid+C
        if p >= K:
            ans = mid
            right = mid - 1
        else:
            left = mid + 1
    print ans
