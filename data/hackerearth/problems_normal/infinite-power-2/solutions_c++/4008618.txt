#include <bits/stdc++.h>
#define ll long long int
using namespace std;
ll high(ll a)
{
	ll x,q=1,t;
	x = a;
	while(x%2==0 && x)
		x /= 10;
	if(!x) return a;
	
	x = a;
	while(x)
	{
		x /= 10;
		q *= 10;
	}
	q /= 10;
	x = a;
	while(true)
	{
		t = (x/q)%10;
		if(t % 2)
		{
			if(t == 9)
			{
				q *= 10;
				return high((x/q)*q + 2*q);
			}
			else return ((x/q)*q + q);
			break;
		}
		q /= 10;
	}
	return a;
}
ll low(ll a)
{
	ll x,q=1,t;
	x = a;
	while(x%2==0 && x)
		x /= 10;
	if(!x) return a;
	
	x = a;
	while(x)
	{
		x /= 10;
		q *= 10;
	}
	q /= 10;
	x = a;
	while(true)
	{
		t = (x/q)%10;
		if(t % 2)
		{
		
			ll k=0;
			x = (x/q)*q - q;
			while(q)
			{
				q/=10;
				k += q*8;
			}
			return x + k;
			break;
		}
		q /= 10;
	}
	return a;
}
ll gcd(ll a, ll b)
{
	return b ? gcd(b,a%b) : a;
}
int main() 
{
	ll n,a,x,i,g,h,l;
	scanf("%lld",&n);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a);
		x = a;
		while(x%2==0 && x)
			x /= 10;
		if(!x) printf("Unlimited Power\n");
		else
		{
			h = high(a);
			l = low(a);
			g = gcd(h-a,a-l);
		//	cout<<a<<" "<<h<<" "<<l<<" ";
			printf("%lld/%lld\n",(h-a)/g,(a-l)/g);
		}
	}
	return 0;
}
