#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long int id,z,p,l,c,s;
    priority_queue<pair<long int, long int> > pq;
    map<long int, long int> m;
    while(n--)
    {
    	cin>>id>>z>>p>>l>>c>>s;
    	pq.push(make_pair((50*p)+(5*l)+(10*c)+(20*s)-z,id));
    	m[id]=(50*p)+(5*l)+(10*c)+(20*s);
    }
    for(int i=0;i<5;i++)
    {
    	cout<<pq.top().second<<" "<<m[pq.top().second]<<endl;
    	pq.pop();
    }
    return 0;
}
