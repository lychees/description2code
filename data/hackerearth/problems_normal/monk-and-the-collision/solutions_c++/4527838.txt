#include <iostream>
using namespace std;

int main()
{
    int t,n,x,ctr,hash[10];
    cin>>t;
   	while(t--)
   	{
   		ctr = 0;
   		for(int i=0;i<10;i++)
   			hash[i] = 0;
   		cin>>n;
   		for(int i=0;i<n;i++)
   		{
   			cin>>x;
   			hash[x%10]++;
   		}
   		for(int i=0;i<10;i++)
   			if(hash[i]>1)
   				ctr+= hash[i]-1;
   		cout<<ctr<<endl;
   	}
    return 0;
}
