#include <iostream>
using namespace std;

int main()
{
    int t;
    long int x;
    
    cin>>t;
    while(t--)
    {
    	cin>>x;
    	if(x==1||x==2)
    	cout<<"-1";
        else if(x%3==0)
        {  
          while(x--)
          {
          	cout<<"5";
          }
        }
        else if(x%3==2)
        {
        	while(x>5)
        	{
        		cout<<"5";
        		x--;
        	}
        	while(x--)
        	{
        		cout<<"3";
        	}
        }
        else if(x%5==0)
        {
        	while(x--)
        	{
        		cout<<"3";
        	}
        }
        else
        {
           cout<<"-1";
        }
    	cout<<endl;
    }
    return 0;
}
