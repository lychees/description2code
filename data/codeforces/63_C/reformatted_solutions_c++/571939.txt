using namespace std;
#define li int
#define rep(i, to) for (li i = 0; i < ((li)to); i++)
#define pb push_back
#define sz(v) ((li)v.size())

int main() {
  set<string> ans;
  string s, str;
  rep(i, 10) rep(j, 10) rep(k, 10) rep(l, 10) {
    if (i == j || i == k || i == l || j == k || j == l || k == l)
      continue;
    s = "0000";
    s[0] += i;
    s[1] += j;
    s[2] += k;
    s[3] += l;
    ans.insert(s);
  }
  li n, a, b;
  cin >> n;
  rep(i, n) {
    cin >> str >> a >> b;
    rep(i0, 10) rep(i1, 10) rep(i2, 10) rep(i3, 10) {
      s = "0000";
      s[0] += i0;
      s[1] += i1;
      s[2] += i2;
      s[3] += i3;
      li c = 0, d = 0;
      rep(j, 4) rep(k, 4) {
        if (s[j] == str[k]) {
          if (j == k)
            c++;
          else
            d++;
        }
      }
      if (c != a || d != b)
        ans.erase(s);
    }
  }
  if (1 < sz(ans)) {
    cout << "Need more data" << endl;
  } else if (sz(ans) == 0) {
    cout << "Incorrect data" << endl;
  } else {
    cout << *ans.begin() << endl;
  }
}
