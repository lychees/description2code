#include <iostream>
#define MOD 1000000007
#define MAX 1000000
using namespace std;

int main()
{
    int t,n,i,j;
    long long int num_lines,f;
    long long int fact[MAX+1];
    
    fact[0] = fact[1] = 1;
    
    for(i=2;i<=MAX;i++){
    	fact[i] = fact[i-1]*i % MOD;
    }
    
    scanf("%d",&t);
    while(t--){
    	scanf("%d",&n);
    	num_lines = 0;
    	
    	//num_lines = fact + 2 * (fact - 1) + 5;
    	num_lines = 3 * (fact[n] + 1) % MOD;
    	printf("%lld\n",num_lines);
    }
    
    return 0;
}
