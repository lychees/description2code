#include <iostream>
#include <string.h>
using namespace std;

int main()
{
	int i,j,c=0;
    char ch[10000];
    cin.getline(ch,10000);
    if(strlen(ch)<=100)
    {
    for(i=0,j=strlen(ch)-1;i<strlen(ch),j>=0;i++,j--)
    {
    	if(ch[i]==ch[j])
    	c++;
    }
    if(c==strlen(ch))
	cout<<"YES";
	else
	cout<<"NO";
    }
    return 0;
}
