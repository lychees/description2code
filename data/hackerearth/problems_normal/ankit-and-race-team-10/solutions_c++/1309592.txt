#define rep(i,a,b) for(int i = a; i < b; i++)
#define REP(i, n) for (int i = 0; i < (int)(n); ++i)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)

typedef vector<int> VI;
typedef pair<int,int> PII;
typedef vector<PII> VPII;
typedef long long LL;

const int MAXN = 100010;
#define MOD 1000000007
LL fact[MAXN], invfact[MAXN], a[MAXN];

int powmod(int base, int expo){
	if(expo==0)
		return 1;
	else if(expo&1)
		return (LL)base*powmod(base, expo-1)%MOD;
	else{
		int root=powmod(base, expo>>1);
		return (LL)root*root%MOD;
	}
}

int inverse(int x){
	return powmod(x, MOD-2);
}

void init(int total){
	fact[0]=1;
	for(int i=1; i<=total; i++)
		fact[i]=(LL)i*fact[i-1]%MOD;
	invfact[total]=inverse(fact[total]);
	for(int i=total; i>0; i--)
		invfact[i-1]=(LL)i*invfact[i]%MOD;
}

int nCr(int n, int r){
	if(r>n || r<0)
		return 0;
	return (LL)((LL)fact[n]*invfact[r]%MOD)*invfact[n-r]%MOD;
}

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("in.txt","r",stdin);
    #endif
    init(MAXN);

    int t;
    S(t);

    while(t--) {
        int n,k;
        scanf("%d%d",&n,&k);

        rep(i,1,n+1) scanf("%lld",&a[i]);
        sort(a+1,a+n+1);

        LL ans = 0;
        for(int i=(n-k+1); i>=1; i--) {
            ans = (ans + (a[i]*nCr(n-i,k-1))%MOD)%MOD;
        }

        printf("%lld\n",ans);
    }
    return 0;
}
