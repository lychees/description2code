#include <iostream>
#include <cmath>

using namespace std;

int main()
{
	int tests;
	cin >> tests;
	for(int test(0) ; test < tests ; ++test){
		int size;
	    cin >> size;
	    int row,col;
	    for(int i(0) ; i<size ; ++i){
	    	for(int j(0) ; j<size ; ++j){
	    		char ch;
	    		cin >> ch;
	    		if(ch == '*'){
	    			row = i;
	    			col = j;
	    		}
	    	}
	    }
	    int middle = size/2;
	    int a = middle - row, b = middle - col;
	    int steps = abs(a)+abs(b);
	    cout << steps << endl;
	}
    return 0;
}
