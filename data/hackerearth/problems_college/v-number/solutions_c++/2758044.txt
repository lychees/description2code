#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		cin>>str;
		int l=strlen(str);
		int sumhalfl=0,sumhalfu=0;
		for(int i=0;i<l/2;i++)
		{
			sumhalfl=sumhalfl+str[i]-48;
		}
		for(int i=l/2;i<l;i++)
		{
			sumhalfu=sumhalfu+str[i]-48;
		}
		if(sumhalfu==sumhalfl)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
}
