#include <bits/stdc++.h>
using namespace std;
//zone
int main()
{
   int n;
   cin >> n;
   vector<int> v;
   int dp[n + 1];
   int i;
   string s;
   cin >> s;
   for(i = 0 ; i < s.length() ; i++)
   {
   	v.push_back((s[i] - '0'));
    // cout<<s[i] - '0'<<endl;
   }
   dp[0] = v[0];
   for(i = 1 ; i < v.size() ; i++)
   {
   	dp[i] = dp[i - 1] + v[i];
   }
   int q;
   cin >> q;
   while(q--)
   {
   	int l,r;
   	cin >> l >> r;
   	l-- ;
   	r--;
   	int ans = dp[r] - dp[l-1];
   	if(ans % 9 == 0)
   	{
   		cout<<"SAMEER\n";
   	}
   	else
   	    cout<<"NO\n";
   }
   	
   return 0;
}
