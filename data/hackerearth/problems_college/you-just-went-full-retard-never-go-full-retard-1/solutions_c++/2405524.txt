#define MAX 10000
 
int is_prime(long long int n)
{
    
	if(n==1 || n==0)
	return 0;
	for(int i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
			return 0;
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
		while(t--)
	{
		long long int x;
		cin>>x;
	if(is_prime(x)==1)
	   cout<<"YES"<<endl;
	else cout<<"NO"<<endl;   
		
	}
    //cout << "Hello World!" << endl;
    return 0;
}
