# print 'Hello World!'
t = int(raw_input())
i = 1
vowels = ['a', 'e', 'i', 'o', 'u']
while i <= t:
    website = raw_input()
    length_normal = len(website)
    website = website[4:-4]
    length_jhool = len(website)+4
    for char in website:
        if char in vowels:
            length_jhool = length_jhool-1
    print "%d/%d" % (length_jhool, length_normal)
    i = i+1
