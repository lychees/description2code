#include <bits/stdc++.h>

using namespace std;
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int n,i,j;
	long long ans;
	scanint(n);
	int a[n];
	for (i = 0; i < n; i++) {
		scanint(a[i]);
	}
	sort(a, a + n);
	ans = 1;
	for (i = 0; i < n-1; i++) {
		if (a[i] != a[i+1]) {
			ans++;
		}
	}
	//printf("%lld\n",ans);
	ans = ans * (ans - 1);
	ans = ans / 2;
	printf("%lld\n",ans);
	
	return 0;
}
