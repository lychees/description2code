#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

int num,n,flag=0;
int a[20];
void check(int i,int sum){
	if(i>=n){
		if(sum==num){
			flag=1;
		}
		return;
	}
	check(i+1,sum);
	check(i+1,sum+a[i]);
}

int main(){
	int t;
	cin >> t;
	for(int i=0;i<t;i++){
		cin >> n;
		flag=0;
		for(int j=0;j<n;j++){
			cin >> a[j];
		}
		cin >> num;
		check(0,0);
		if(flag==1)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
