#include<bits/stdc++.h>
using namespace std;

int main() {
	int T; cin >> T;
	while (T--) {
		int R, K; cin >> R >> K;
		int ans = 0;
		for (int X = 1; X <= R; X++) {
			int cnt = 0;
			for (int x = X; x > 0; x /= 2)
				if (x % 8 == 5)
					cnt++;
			if (cnt >= K)
				ans++;
		}
		cout << ans << endl;
	}
}
