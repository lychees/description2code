#include <iostream>
using namespace std;

bool isPrime(int number);

int main()
{
	int testCases;
	cin>>testCases;
	for(int i=0;i<testCases;i++)
	{
		int number;
		cin>>number;
		number++;
		int flag = 10;
		do
		{
			if(!isPrime(number))
			{
				cout<<number<<"\n";
				flag = 1;
				break;
			}
			else
			{
				number++;
				flag = 0;
			}
		}while(flag != 1);
	}
	return 0;
}

bool isPrime(int number)
{
	int flag = 0;
	if(number == 2)
	{
		flag = 0;
		return true;
	}

	else
	{
		for(int i=2;i<number/2+1;i++)
		{

			if(number % i == 0)
			{
				flag = 1;
				break;
			}
			else
				flag = 0;
		}
		if(flag == 1)
			return false;
		else
			return true;
	}
}
