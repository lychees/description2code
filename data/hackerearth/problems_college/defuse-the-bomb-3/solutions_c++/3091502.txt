#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t>0)
    {
    	long int n1,n2,num,i,j,k,a=0,c=0;
    	int b[20];
    	cin>>n1;
    	cin>>n2;
    	num=n1;
    	for(i=0;num>0;i++)
    	{
    		b[i]=num%10;
    		num=(num/10);
    		a++;
    	}
    	num=n2;
    	for(i=0;num>0;i++)
    	{
    		c++;
    		num=(num/10);
    	}
    	for(i=(a-1);i>=(c-1);i--)
    	{
    		j=0;
    		for(k=0;k<c;k++)
    		{
    			j*=10;
    			j+=(b[i-k]);
    		}
    		
    		cout<<j*n2<<endl;
    	}
    	t--;
    }
    return 0;
}
