#include <iostream>
#include<bits/stdc++.h>
using namespace std;
map<int,int> m;
	int arr[100012];
int main()
{
    //cout << "Hello World!" << endl;
    int n,d;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    
    	cin>>arr[i];
    	m[arr[i]]++;
    }
    cin>>d;
    int f=0;
    int sum =0;
    for(int i=0;i<n;i++)
    {
    	for(int j=i+1;j<n;j++)
    	{
    		sum = arr[i]+arr[j];
    		if(sum<d)
    		{
    			int left = d-sum;
    			if(m[left]!=0)
    			{
    				if(left==arr[i]) left--;
    				if(left==arr[j]) left--;
    				if(left>0) {
    					f=1;
    					break;
    				}
    			}
    		}
    	}
    	if(f==1) break;
    }
    if(f==1) cout<<"True"<<endl;
    else cout<<"False"<<endl;
    return 0;
}
