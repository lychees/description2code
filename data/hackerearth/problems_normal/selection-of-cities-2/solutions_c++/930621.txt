#include <iostream>
#include<cstdio>
#define mod 1000000007
using namespace std;
long long power(long long a)
{
	if(a==0)
		return 1;
	long long temp=power(a/2);
	temp=(temp*temp)%mod;
	if(a&1)
		temp=(temp*2)%mod;
	return temp;	
}	
int main()
{
    //cout << "Hello World!" << endl;
    int t;
    long long n;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	printf("%lld\n",(power(n)+mod-1)%mod);
    }
    return 0;
}
