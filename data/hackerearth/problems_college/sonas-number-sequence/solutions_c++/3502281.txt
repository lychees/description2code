#include <iostream>
using namespace std;

int main()
{
	long long int B[100],A[100],dp[100];
   int i,n;
   cin>>n;
   for(i=0;i<n;i++)
   {cin>>B[i];}
   A[0]=B[0];
   dp[0]=A[0];
   for(i=1;i<n;i++)
   {A[i]=B[i]*(i+1)-dp[i-1];
    dp[i]=dp[i-1]+A[i];}
    
    for(i=0;i<n;i++)
   {cout<<A[i]<<" ";}
}
