#include<iostream>
#define max 1000000
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--){
	int i=0,count=0,f=0,k=0,a[max+1],x,y,j,l;
	int b[1000001]={0};
	cin>>x>>y;
	int mul,di,add,sub;
	cin>>mul>>di>>add>>sub;
	a[0]=x;
	b[x]=1;
	while(1)
	{
		j=i;
		while(k<=j)
		{
			l=a[k]*mul;
			if(l>=2 && l<=max && b[l]==0)
			{
				if(l==y)
				{
					f=1;
					count++;
					break;
				}
				else
				a[++i]=l;
				b[l]=1;
			}
			l=a[k]/di;
			if(l>=2 && l<=max && b[l]==0)
			{
				if(l==y)
				{
					f=1;
					count++;
					break;
				}
				else
				a[++i]=l;
				b[l]=1;
			}
			l=a[k]+add;
			if(l>=2 && l<=max && b[l]==0)
			{
				if(l==y)
				{
					f=1;
					count++;
					break;
				}
				else
				a[++i]=l;
				b[l]=1;
			}
			l=a[k]-sub;
			if(l>=2 && l<=max && b[l]==0)
			{
				if(l==y)
				{
					f=1;
					count++;
					break;
				}
				else
				a[++i]=l;
				b[l]=1;
			}
			k++;
		}
		if(f==1)
		break;
		else if(k>i)
		break;
		else
		count++;
	}
	if(f==0)
	cout<<"-1"<<endl;
	else
	cout<<count<<endl;
	}
	return 0;
}
