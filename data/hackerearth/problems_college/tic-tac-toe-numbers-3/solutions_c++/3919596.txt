#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
	   long n;
	   long long sum=0;
	   int tic=0,tac=0,toe=0;
	   cin>>n;
	   long long a=0,b=1,c=a+b;
	   int a1[2*n];
	   for(long i=0;i<n;i++)
	   {
	       a1[i]=0;
	   }
	   for(long i=2;i<n;i++)
	   {
	       if(c<n)
	       {
	      c=a+b;
	      a=b;
	      b=c;
	      a1[b]++;
	       }
	       else{
	           break;
	       }
	   }
	   for(long i=1;i<n;i++)
	   {
	       if(a1[i]==0)
	       {
	           sum=sum+i;
	           //cout<<i<<endl;
	       }
	   }
	   if(sum%2==0)
	   tac=1;
	   else{
	   for(long long i=3;i<sqrt(sum);i+=2)
	   {
	       if(sum%i==0)
	       {
	           if(i%2==0)
	           {
	               tac=1;
	               break;
	           }
	           else{
	               toe=1;
	               break;
	           }
	       }
	   }}
	   if(tac==1)
	   {
	       cout<<"Tac"<<endl;
	   }
	   else if(toe==1)
	   {
	       cout<<"Toe"<<endl;
	   }
	   else{
	       cout<<"Tic"<<endl;
	   }
	}
	return 0;
}
