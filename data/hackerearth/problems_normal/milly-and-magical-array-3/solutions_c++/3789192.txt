#include <iostream>
#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	ll t;
	cin >> t;
	
	while (t--) {
		ll n;
		ll x;
		
		cin >> n >> x;
		
		ll arr[100005];
		for (ll i = 0; i < n; i++) {
			cin >> arr[i];
		}
		
		ll ans = 0;
		for (ll i = 1; i < n; i++) {
			if (arr[i] > arr[i - 1]) {
				continue;
			} else if (arr[i] == arr[i - 1]) {
				ans++;
				arr[i] += x;
			} else {
				ll diff = arr[i - 1] - arr[i];
				ll tempo = (diff) / (x);
				ans += (tempo + 1);
				arr[i] += (tempo + 1) * x;
			}
		}
		
		cout << ans << endl;
	}
	
    return 0;
}
