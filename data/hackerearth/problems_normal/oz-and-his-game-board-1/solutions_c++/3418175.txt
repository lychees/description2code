#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,m,a,b,c;
    long long int ans = 0,x,y,z,ans1;
    cin>>t;
    while(t--)
    {
    	ans = ans1 = 0;
    	cin>>n>>m>>a>>b>>c;
    	x = n;
    	y = a;
    	x = x*y;
    	y = m;
    	z = b;
    	y = y*z;
    	ans = x + y;
    	
    	if(n>m)
    	{
    		x = m *2;
    		y = c;
    		x = x* y;
    		n = n - m;
    		y = n;
    		z = a;
    		y = y*z;
    		ans1 = x+y;
    	}
    	else
    	{
    		x = n *2;
    		y = c;
    		x = x* y;
    		m = m - n;
    		y = m;
    		z = b;
    		y = y*z;
    		ans1 = x+y;
    	}
    	ans = max(ans,ans1);
    	cout<<ans<<endl;
    }
    return 0;
}
