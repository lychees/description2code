#include<bits/stdc++.h>

using namespace std ;

int main()
{
	int t , n , ans , i , j , k , maxm , count ;
	cin>>t ;
	//long int arr , dep ;
	while(t--)
	{
		cin>>n ;
		long int arr[n] ;
		long int dep[n] ;
		maxm = 0 ; 
		count = 0 ;
//		v.clear() ;
		for(i = 0 ; i < n ; i++)
		{
			cin>>arr[i] ;
	//		v.push_back(make_pair(arr , 0)) ;
		}
		for(i = 0 ; i < n ; i++)
		{
			cin>>dep[i] ;
			dep[i]+=arr[i] ;
		//	v[i].second = v[i].first + dep ;
		}
		sort(arr,arr+n) ;
		sort(dep,dep+n) ;
		i = 0 ; j = 0 ;
		while(i < n)
		{
			if(arr[i] < dep[j])
			{
				count++ ;
				i++ ;
			}
			else if(arr[i] > dep[j])
			{
				count-- ;
				j++ ;
			}
			else
			{
				i++ ;
				j++ ;
			}
			if(count > maxm)
				maxm = count ;
		}
		cout<<maxm<<endl ;
	}
	return 0 ;
}
