#include <iostream>
using namespace std;

int main()
{
    int t,x,y;
    long long ans,a,b,min,max,p,q;
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>x>>y;
    	if(a>b)
    	{
    		max=a;
    		min=b;
    	}
    	else
    	{
    		min=a;
    		max=b;
    	}
    	ans=max;
    	while(ans%min!=0)
    	{
    		ans+=max;
    	}
    	p=x/ans;
    	q=y/ans;
    	if(x%ans==0)
    		q++;
    	cout<<q-p<<endl;
    //	cout<<ans<<endl;
    	
    }
    return 0;
}
