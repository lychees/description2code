#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
    	int n;
    	double x,y,sumx=0,sumy=0;
    	cin >> n;
    	for(int i=0;i<n;i++){
    		cin >> x >> y;
    		sumx+=x;
    		sumy+=y;
    	}
    	sumx/=n;
    	sumy/=n;
    	cout << "(";
    	cout <<setprecision(2) << fixed<< sumx << " , " <<setprecision(2) << fixed << sumy << ")"<< endl;
    }
    return 0;
}
