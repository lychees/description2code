#include <iostream>
    #include <string>
    #include <algorithm>
    using namespace std;
    int main()
    {
        int t;
        cin>>t;
        while(t--){
        	string str;
        	cin>>str;
        	int len = str.length();
        	if(len == 1){
        		cout<<"no answer\n";
        		continue;
        	}
        	if(len == 2 && (str[0] == str[1])){
        		cout<<"no answer\n";
        		continue;
        	}
        	int all_equal = 1;
        	
        	int a[len];
        	for(int i=0;i<len;i++){
        		a[i] = int(str[i]);
        	}
        	for(int i=0;i<len-1;i++){
        		if(a[i] != a[i+1]){
        			all_equal = 0;
        			break;
        		}	
        	}
        	if(all_equal){
        		cout<<"no answer\n";
        		continue;
        	}
        	int i;
        	for(i = len-1;i>0;i--){
        		if(a[i]<=a[i-1]){
        			continue;
        		}
        		else{
        			break;
        		}
        	}
        	if(i == 0){
        		cout<<"no answer\n";
        		continue;	
        	}
        	int p = i-1;
        	int min = 1000;int min_location;
        	for(int j=p+1; j<len; j++){
        		if(a[j] < min && a[j] > a[p]){
        			min = a[j];
        			min_location = j;
        		}
        	}
        	// cout<<min<<"-"<<min_location;
        	int temp = a[p];
        	a[p] = min;
        	a[min_location] = temp;
        	p++;
        	
        	sort(a+p,a+len);
        	for(int i=0;i<len;i++){
        		cout<<char(a[i]);
        	}
        	cout<<endl;
        }
        return 0;
    }
