#include <iostream>
#include <stack>
using namespace std;

int main()
{   int t;long long res;
    cin>>t;
    while(t--)
    {   stack <long long int> s;
    	long long int a,b;
    	cin>>a>>b;
    	s.push(b);
    	while(a--)
    	{
    		string a;
    		cin>>a;
    		if(a=="P")
    		{
    			long long int val;
    			cin>>val;
    			s.push(val);
    		}
    		else if(a=="B")
    		{
    			long long int top1,top2;
    			top1=s.top();
    			s.pop();
    			top2=s.top();
    			s.pop();
    			s.push(top1);
    			s.push(top2);
    		}
    	
    	}
  		
    	cout<<"Player "<<s.top()<<endl;
    }
	
    return 0;
}
