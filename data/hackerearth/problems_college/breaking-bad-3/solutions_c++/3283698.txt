#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int tc, n, i;
    
    cin >> tc;
    
    while (tc--) {
    	cin >> n;
    	long long int a[n];
    	
    	for (i = 0; i < n; i++) {
    		cin >> a[i];
    	}
    	
    	
    	sort(a, a + n);
    	
    	long long int sum = 0;
    	
    	for (i = n - 1; i >= 0; i -= 2) {
    		if (i != 0)
    		sum += a[i] * a[i - 1];
    		
    	}
    	
    	cout << sum << endl;
    	
    }
    
    return 0;
}
