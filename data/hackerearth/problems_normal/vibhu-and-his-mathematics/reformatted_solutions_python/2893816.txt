maxx = 10**6+7
modd = 10**9+7
ans = [0 for i in range(maxx)]
for i in range(3):
    ans[i] = i
for i in range(3, maxx):
    ans[i] = ((ans[i-1] % modd) + (((i-1)*ans[i-2]) % modd)) % modd
# t = 1
t = input()
for _ in range(t):
    # n = 3
    n = input()
    print ans[n]
