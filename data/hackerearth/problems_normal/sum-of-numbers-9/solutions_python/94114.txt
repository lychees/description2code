import math

memory = {}
def add(a, b):
    sum = ' '.join(str(x) for x in (a, b))

    if sum in memory:
        return memory[sum]

    c = a + b
    memory[sum] = c

    return c

def yesno(result, arr):
    sums = [0]

    for n in arr:
        new_sums = []

        for sum in sums:
            new_sum = add(sum, n)
            if new_sum == result:
                return "YES"
            else:
                new_sums.append(new_sum)

        sums += new_sums

    return "NO"
