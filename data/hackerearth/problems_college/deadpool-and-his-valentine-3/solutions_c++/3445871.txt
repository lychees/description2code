#include<iostream>
using namespace std;
int main()
{
    int k;
    cin>>k;
    int n;
    cin>>n;
    int arr[n+1];
    for(int i=0;i<k;i++)
    {
        int s;
        cin>>s;
        for(int j=0;j<s;j++)
        {
            int a;
            cin>>a;
            arr[a]=1;
        }
    }
    int i;
    for(i=1;i<=n;i++)
    {
        if(arr[i]==1)
        {
            continue;
        }
        else
            break;
    }
    if(i==n+1)
        cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
return 0;
}
