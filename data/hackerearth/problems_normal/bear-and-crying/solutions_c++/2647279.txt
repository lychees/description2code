lld powP(lld a,lld b){
    if(b==0) return 1%mod;
    lld k;
    k=powP(a,b/2);
    k=k*k%mod;
    if(b%2==0) return k;
    else return a*k%mod;
}

int main(){ 
    lld i,j,k,l,r,s,a,b,c,d,x,y,z,m,n,t,p,q,T,curr,sum,ans,pos,val,countA;
    string S;
    cin>>S;
    n=S.size();
    if(S[0]=='_') C[0]=1;
    else C[0]=0;
    for(i=1;i<n;i++){
    	if(S[i]=='_') C[i]=C[i-1]+1;
    	else C[i]=C[i-1];
    }
    B[n]=0;
    for(i=n-1;i>0;i--){
    	if(S[i]==';') B[i]=(B[i+1]+powP(2,C[i])+mod)%mod;
    	else B[i]=B[i+1];
    }
    ans=0;
    for(i=0;i<n-1;i++){
    	if(S[i]==';') ans=(ans+B[i+1]*powP(powP(2,C[i]),mod-2))%mod;
    }
    t=n-C[n-1];
    t=t*(t-1)/2%mod;
    ans=(ans-t+mod)%mod;
    prL(ans);
    return 0;
}
