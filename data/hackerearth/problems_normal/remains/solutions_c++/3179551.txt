#include <iostream>
#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main() 
{
// your code goes here
ll t;
cin>>t;
//	scanf("%lld",&t);
while(t--)
{
ll x,y,n,sum=0,flag=0,i,swag=0,temp;
cin>>x>>y>>n;
//scanf("%lld%lld%lld",&x,&y,&n);
sum=x+y;
temp=0;
for(i=1;i<=n-2;i++)
{
flag=y;
y=abs(x-y);
x=flag;
sum+=y;
if(y==0 ) 
{
temp=1;
swag=i-1;
break;
}

}
if(temp)
{
sum+=2*((n-2-swag)/3)*x;
if((n-2-swag)%3==2)
sum+=x;
}
cout<<sum<<endl;
// printf("%lld\n",sum);
}

return 0;
}
