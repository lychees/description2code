#include <iostream>
#include<cstdio>
#define gc getchar_unlocked

using namespace std;

void scanint(long long &x)
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}


int main()
{
    long long T;
    scanint(T);
    while(T--) {
    	long long N;
    	scanint(N);
  		long long even = 0, odd = 0;
  		long long n;
    	for(long long i = 0; i < N; i++) {
    		scanint(n);
    		if(n % 2 == 0)
    			even++;
    		else
    			odd++;
    	}	
		cout << even * odd << endl;
    }
    return 0;
}
