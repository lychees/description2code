#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <stdlib.h>

using namespace std;

void hack(int* s, int len)
{
    for(int i = len-1; i >= 0; i--)
    {
        s[i] = s[i] ^ (0xf0fffff0 | (0xffffffff & 0x0f00000f));
    }
    for(int i = 0; i < len; i++)
    {
        s[i] = s[i] ^ (0xf0ff0ff0 | (0xffffffff & 0x0f00f00f));
    }
    for(int i = len-1; i >= 0; i--)
    {
        s[i] = s[i] ^ (0xf0ff00f0 | (0xffffffff & 0x0f00ff0f));
    }
}

int main()
{
    char s[10000];
    scanf("%s", s);
    int arr[10000];
    int len = strlen(s);
    for(int i=0; i<len; i++)
    {
        arr[i] = (int) s[i];
    }
    hack(arr, len);
    for(int i=0; i<len; i++)
    {
        printf("%d", arr[i]);
        if(i<(len-1))
        {
            printf(" ");
        }
    }
    return 0;
}
