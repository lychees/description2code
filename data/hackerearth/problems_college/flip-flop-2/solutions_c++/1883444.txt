#include<iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100000];
		int flag_X=0,flag_Y=0,cnt=0;
		cin>>str;
		int len=strlen(str);
		for(int i=0;str[i]!='\0';i++)
		{
			if(str[i]=='Y')
				flag_Y=1;
			if(str[i]=='X')
				flag_X=1;
		}
		if(flag_X==0 || flag_Y==0){
			cout<<len-1<<endl;
			continue;
		}
		int j;
		for(int i=0;i<len;i=j)
		{
			if(str[i]=='X')
				{
					for(j=i+1;str[j]!='\0';j++)
					{
						if(str[j]=='Y')
							break;
						else
							cnt++;
					}
				}
				else
				{
					for(j=i+1;str[j]!='\0';j++)
					{
						if(str[j]=='X')
							break;
						else
							cnt++;
					}
				}	
		}
		cout<<cnt<<endl;
	}
}
