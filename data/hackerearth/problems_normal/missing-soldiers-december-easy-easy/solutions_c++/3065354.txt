#include<bits/stdc++.h>
using namespace std;
struct interval
{
    long long start;
    long long finish;
    bool operator <(const interval &i) const
    {
        return start<i.start;
    }
};
int main()
{
    long n;
    scanf("%ld",&n);
    interval arr[n];
    long long x,y,d;
    long i;
    for(i=0;i<n;i++)
    {
        scanf("%lld%lld%lld",&x,&y,&d);
        arr[i].start=x;
        arr[i].finish=x+d;
    }
    sort(arr,arr+n);
    long long ans=0,s,f;
    s=arr[0].start;
    f=arr[0].finish;
    ans+=(f-s)+1;
   // cout<<arr[0].start<<" "<<arr[0].finish<<" "<<ans<<endl;
    for(i=1;i<n;i++)
    {

        if(arr[i].start>=s&&arr[i].finish<=f)
        {
           // cout<<"case:Overlapping "<<arr[i].start<<" "<<arr[i].finish<<" "<<ans<<endl;
        }
        else if(arr[i].start>f)
        {

            s=arr[i].start;
            f=arr[i].finish;
            ans+=(f-s)+1;
           // cout<<"case: Disjoint "<<arr[i].start<<" "<<arr[i].finish<<" "<<ans<<endl;
        }
        else if(arr[i].start<=f)
        {
            ans+=(arr[i].finish-(f+1))+1;
            f=arr[i].finish;
           // cout<<"case: Partial "<<arr[i].start<<" "<<arr[i].finish<<" "<<ans<<endl;
        }
        //cout<<arr[i].start<<" "<<arr[i].finish<<" "<<ans<<endl;
    }
    printf("%lld\n",ans);
    return 0;
}
