#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
      long long n,q;
        cin>>n>>q;
        q--;
      long long int ans = n*n - (n%q)*((n+q-1)/q)*((n+q-1)/q) - (q - n%q)*(n/q)*(n/q);
        ans /= 2;
        ans = (n*(n-1))/2 - ans;
       cout<<ans<<endl;
    }
    return 0;
}
