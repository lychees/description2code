#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,t,check;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		vector<int> v(n),flag(n);
		for(i=0;i<n;i++)
			scanf("%d",&(v[i]));
		i=check=0;
		do
		{
			if(v[i]==-1)
			{
				check=1;
				break;
			}
			flag[i]++;
			if(flag[i]>1)
				break;
			i=v[i];
		}
		while(1);
		if(check)
			printf("Win\n");
		else
			printf("Lose\n");	
	}
	return 0;
}
