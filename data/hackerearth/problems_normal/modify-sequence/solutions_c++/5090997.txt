#include <iostream>
using namespace std;
 
int main()
{
    //cout << "Hello World!" << endl;
    int n,eSum=0,oSum=0;
    cin>>n;
    for(int i=0;i<n;i++) {
    	int x;
    	cin>>x;
    	if((i+1)%2==0) {
    		eSum=eSum+x;
    	}
    	else {
    		oSum=oSum+x;
    	}
    }
    if(eSum==oSum) {
    	cout<<"YES";
    }
    else {
    	cout<<"NO";
    }
    return 0;
}
