#include <iostream>
#include<cstdio>
using namespace std;

int main()
{
    int n,p;
    scanf("%d%d",&n,&p);
    
    double DP[n],two=p/100.0,three=(1-two);
  
    DP[0]=DP[1]=0.0;
    
    DP[2]=two,DP[3]=three;
    if(n>=4)
    {
  
    for(int i=4;i<=n;i++)  
    DP[i]=two*DP[i-2]+three*DP[i-3];
    
    
    }
    
    
    printf("%0.6lf",DP[n]);
    return 0;
}
