#include <cstdio>
#include <iostream>
#include <vector>
using namespace std;

namespace
{
    typedef vector<int> ValCollection;
    int maxSum(const ValCollection &vals)
    {
        if(vals.empty())
            return 0;
        int min = 0;
        int max = 0;
        int res = 0;
        int sum = 0;
        for(auto v : vals)
        {
            sum += v;
            if(sum < min)
            {
                min = sum;
                max = sum;
            }
            else if(sum > max)
            {
                max = sum;
                if(max - min > res)
                    res = max - min;
            }
        }
        if(max - min > res)
            res = max - min;
        return res;
    }
}

int main()
{
    int t = 0;
    cin>>t;
    while(t-- > 0)
    {
        int count = 0;
        cin>>count;
        ValCollection vals;
        while(count-- > 0)
        {
            int n = 0;
            scanf("%d", &n);
            vals.push_back(n);
        }
        cout<<maxSum(vals)<<endl;
    }
    return 0;
}
