# define done return 0;
string minute(int m)
    {
    string ret;
    if(m==1)
        ret="one";
    else if(m==2)
        ret="two";
    else if(m==3)
        ret="three";
    else if(m==4)
        ret="four";
    else if(m==5)
        ret="five";
    else if(m==6)
        ret="six";
    else if(m==7)
        ret="seven";
        else if(m==8)
        ret="eight";
    else if(m==9)
        ret="nine";
        else if(m==10)
        ret="ten";
    else if(m==11)
        ret="eleven";
        else if(m==12)
        ret="twelve";
    else if(m==13)
        ret="thirteen";
        else if(m==14)
        ret="fourteen";
    else if(m==15)
        ret="quarter";
        else if(m==16)
        ret="sixteen";
    else if(m==17)
        ret="seventeen";
        else if(m==18)
        ret="eighteen";
    else if(m==19)
        ret="nineteen";
        else if(m==20)
        ret="twenty";
    else if(m==21)
        ret="twenty one";
        else if(m==22)
        ret="twenty two";
    else if(m==23)
        ret="twenty three";
        else if(m==24)
        ret="twenty four";
    else if(m==25)
        ret="twenty five";
        else if(m==26)
        ret="twenty six";
    else if(m==27)
        ret="twenty seven";
        else if(m==28)
        ret="twenty eight";
    else 
        ret="twenty nine";
        return ret;
}
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int h,m;
    string ans;
    ans="";
    cin>>h;
    cin>>m;
    if(m==15)
        ans=minute(m)+" past "+minute(h);
    if(m==30)
        ans="half past "+minute(h);
    if(m==45)
        ans=minute(60-m)+" to "+minute(h+1);
    if((m<30 && m>15)||(m<15&&m>1))
        ans=minute(m)+" minutes past "+minute(h);
    if((m>30&&m<45)||(m>45&&m<59))
        ans=minute(60-m)+" minutes to "+minute(h+1);
    if(m==1)
        ans="one minute past "+minute(h);
    if(m==59)
        ans="one minute to "+minute(h+1);
    if(m==0)
        ans=minute(h)+" o' clock";
    cout<<ans;
    done
}
