

    #include <iostream>
    #include<stdio.h>
    #include<stdlib.h>
     
    using namespace std;
     
    int main()
    {
    char a[100][100];
    int n,t,x=0,y=0,res=0;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    for(int j=0;j<n;j++)
    {
     
    cin>>a[i][j];
    if(a[i][j]=='*')
    {
    x=i;y=j;
    res=abs(x-(n/2))+abs(y-(n/2));
    }
    }
    }
   
    printf("%d\n",res);
    x=0;y=0;res=0;
    }
     
    return 0;
    }
