#include <bits/stdc++.h>
using namespace std;

int main()
{

   int n,m,i,j,q,c=1,x,z;
   map <int,int> mp;
   map <int,int> :: iterator it1;
   cin>>n>>m;
   int a[n+1][m+1];
   for(i=1;i<=n;i++)
   {
   	 for(j=1;j<=m;j++)
   	 {
   	 	cin>>x;
   	 	a[i][j]=x;
   	 	mp[x]=i;

   	 }

   }
   cin>>q;

   for(i=0;i<q;i++)
   {
   	cin>>x;
   	it1=mp.find(x);
   //	cout<<it1->first<<endl;
   	if((it1->first)==(m*n))
   	{
   		cout<<"-1"<<" "<<"-1"<<endl;
   	}
   	else
   	{
   		z=it1->second;
   		for(j=1;j<=m;j++)
   		{
   			if(x==a[z][j])
   			{
   				cout<<(z-1)<<" "<<(j-1)<<endl;
   			}
   		}
   	}
   }
    return 0;
}
