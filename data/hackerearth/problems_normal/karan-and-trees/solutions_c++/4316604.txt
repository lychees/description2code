#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define MAX 100005
ll modexp(ll a, ll b)
{
	ll res = 1;
	while(b)
	{
		res=(res*a)%MOD;
		a=(a*a)%MOD;
		b/=2;
	}
	return res;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll a, b;
		cin>>a>>b;
		while(a != b)
		{
			if(a > b) a>>=1;
			else b>>=1;
		}
		cout<<a<<endl;
	}
    return 0;
}
