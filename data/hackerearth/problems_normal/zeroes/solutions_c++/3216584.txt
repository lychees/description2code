#include <bits/stdc++.h>
#define mp make_pair
using namespace std;
const int C = 100500;

vector<int> steps(C,0);

vector<pair<int, int>> get_fact(int n) {
	vector<pair<int, int>> res;
	for (int i = 2; i * i <= n; ++i) {
		if (n % i == 0) {
			int cnt = 0;
			while (n % i == 0) {
				++cnt;
				n /= i;
			}
			res.push_back(mp(i, cnt));
		}
	}
	if (n > 1) {
		res.push_back(mp(n, 1));
	}
	return res;
}

void solve() {
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		int cur;
		cin >> cur;
		vector<pair<int, int>> fact = get_fact(cur);
		for (auto item : fact) {
			steps[item.first] += item.second;
		}
	}
	int Q;
	cin >> Q;
	while (Q--) {
		int x;
		cin >> x;
		auto fact = get_fact(x);
		int res = 1e9;
		for (auto item : fact) {
			res = min(res, steps[item.first] / item.second);
		}
		cout << res << "\n";
	}

}
int main()
{
    
    	solve();
    return 0;
}
