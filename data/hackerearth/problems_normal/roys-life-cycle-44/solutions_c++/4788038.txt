#include <iostream>
#include <cstring>
#include <vector>
#include <cmath>
using namespace std;
int longStreak(string str)
{
	int maxm=0,count=0,i;
	for(i=0;i<str.length();i++)
	{
		if(str[i]=='C')
		{
			count=0;
			while(str[i]=='C')
			{
				count++;
				i++;
			}
			maxm=max(maxm,count);
		}
	}
	maxm=max(maxm,count);
	return maxm;
}
int main() {
	// your code goes here
	string s,tot;
	int N,i,ans1=0,ans2=0;
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>N;
	for(i=0;i<N;i++)
	{
		cin>>s;
		tot+=s;
		ans1=max(ans1,longStreak(s));
	}
	ans2=max(ans2,longStreak(tot));
	cout<<ans1<<" "<<ans2<<"\n";
	return 0;
}
