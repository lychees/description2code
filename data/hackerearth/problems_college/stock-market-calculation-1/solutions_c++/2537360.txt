#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
	long long i,j,k,n,m,p,t,a[50005],max,pos,ans;
	
	scanf("%ld",&t);
	
	while(t--)
	{
		ans=0;
		
		scanf("%ld",&n);
		for(i=0;i<n;i++)
		scanf("%ld",&a[i]);
		
		pos=0;
		p=0;
		
		while(1)
		{
		
		if(p>=n-1)
		break;
//		printf("%ld ",p);
		max=a[p];
		pos=p;
		for(i=p+1;i<n;i++)
		{
			if(a[i]>max)
			{
				max=a[i];
				pos=i;
			}	
		}
		
		for(i=p;i<pos;i++)
			ans+=(max-a[i]);
		
		p=pos+1;
		
		}
		printf("%ld\n",ans);
	}
    return 0;
}
