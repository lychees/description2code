#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		cin>>s;
		int len = s.length();
		string ans;
		ans.push_back(s[len-1]);
		if(len >= 2)
			ans.push_back(s[len-2]);
		if(len >= 3)
			ans.push_back(s[len-3]);
		if(len > 3)
			ans.push_back(',');
		int i = len-4;
		int x = 0;
		while(i >= 0)
		{
			x++;
			ans.push_back(s[i]);
			i--;
			if(x == 2 && i >= 0)
			{
				x = 0;
				ans.push_back(',');
			}
		}
		int len2 = ans.length();
		len2--;
		while(len2 >= 0)
		{
			cout<<ans[len2];
			len2--;
		}
		cout<<endl;
	}
	return 0;
}
