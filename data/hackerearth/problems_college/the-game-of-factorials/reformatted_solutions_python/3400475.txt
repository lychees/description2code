t = input()
for i in range(0, t):
    N = input()
    # print N
    fact = 1
    for j in range(1, N+1):
        fact = fact*j
    digit_sum = 0
    while (fact > 0):
        rem = fact % 10
        digit_sum += rem
        fact = fact/10
    print digit_sum
