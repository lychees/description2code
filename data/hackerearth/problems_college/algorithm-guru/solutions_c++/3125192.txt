#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int t;cin >> t;while(t--){
   int n,x;
   cin >> n >> x;
   vector<int> v(n);
   vector<int> w(x+1,0);
   for(int i=0;i<n;i++) cin >> v[i];
   for(int i=1;i<=x;i++){
   	for(int j=0;j<v.size();j++){
   		if(i-v[j]>=0)
   		w[i] = max(w[i],w[i-v[j]]+v[j]);
   	}
   //	cout << w[i] << " ";
   }
   int ans=0;
   for(int i=1;i<=x;i++) ans=max(ans,w[i]); 
   cout << ans << endl;}
    return 0;
}
