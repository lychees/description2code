#include <iostream>
#include <string>
using namespace std;
typedef unsigned long long int ull;
int main()
{
    ull t;
    cin>>t;
    while(t--)
    {
    	ull index[30]={0};	
    	string s;
    	cin>>s;
    	ull max=0;
    	for(ull i=0;i<s.length();i++)
    	{
    		index[s[i]-'a']++;
    		if(index[s[i]-'a'] > max)
    		max=index[s[i]-'a'];
    	}
    	
    	cout<<max<<endl;
    }
    return 0;
}
