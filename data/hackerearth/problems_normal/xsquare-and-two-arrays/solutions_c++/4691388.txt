#include <iostream>
#include<bits/stdc++.h>
using namespace std;



int main()
{
long long q,a[100000],b[100000],n;
cin>>n>>q;
long long p[q][3];
for(long long i=0;i<n;i++)
	cin>>a[i];
for(long long i=0;i<n;i++)
	cin>>b[i];
	
for(long long i=0;i<q;i++)
	for(long long j=0;j<3;j++)
		cin>>p[i][j];
		
for(long long i=0;i<n;i++)
{
	if(p[i][0]==1)
	{
		long long result=0,count=0;
		for(long long j=p[i][1]-1;j<p[i][2];j++)
		{
			if(count==0)
				result+=a[j];
			else
				result+=b[j];
						
			if(count==0)
				count=1;
			else
				count=0;		
		}	
		cout<<result<<endl;
	}
	else if(p[i][0]==2)
	{
		long long result=0,count=0;
		for(long long j=p[i][1]-1;j<p[i][2];j++)
		{
			if(count==0)
				result+=b[j];
			else
				result+=a[j];
			
			if(count==0)
				count=1;
			else
				count=0;		
		}
		cout<<result<<endl;		
	}		
}
	
		
    return 0;
}
