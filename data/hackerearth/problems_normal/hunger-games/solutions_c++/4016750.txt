#include <iostream>
using namespace std;
#include<stdio.h>
#include<algorithm>
#include<cmath>

int main()
{ 
	int n,i,a[1000],b[1000],max=-1,r,l,x,y;
	scanf("%d",&n);
	//	printf("%d ",n);
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
		//	printf("%d ",a[i]);
	}
    std::sort(a,a+n); 
    //  for(i=0;i<n;i++){
    //	printf("%d ",a[i]);
  //  }
    x=n/2;
    b[x]=a[0];
    r=1;
    l=1;
    for(i=1;i<n;i++){
    	if(i%2==0){
    		b[x+r]=a[i];
    		r++;
    	}
    	else{
    			b[x-l]=a[i];
    		l++;
    	}
    }
  
    for(i=0;i<n;i++){
    	if(i==(n-1)){
    		y=abs(b[i]-b[0]);
    		//printf("%d %d\n",b[i],b[0]);
    	}
    	else{
    	  y=abs(b[i]-b[i+1]);
    	  //	printf("%d %d\n",b[i],b[0]);
    	  }
    	  if(y>max){
    	  	max=y;
    	  //	pr;intf("%d %d\n",max)
    	  }
    }
    printf("%d",max);
    return 0;
}
