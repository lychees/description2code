nums = []
count = int(raw_input())
line = raw_input().split(" ")
for num in line:
    nums.append(int(num))
result = []
nums.sort()
sub = 0
for num in nums:
    num -= sub
    result.append(num)
    sub += 1
print ' '.join(map(str, result))
