#define MAXN 10000001
int phi[MAXN + 1], prime[MAXN/10], sz;
bitset <MAXN + 1> mark;

void set() {
	for (int i = 2; i <= MAXN; i++ ){
		if(!mark[i]){
			phi[i] = i-1;
			prime[sz++]= i;
		}
		for (int j=0; j<sz && prime[j]*i <= MAXN; j++ ){
			mark[prime[j]*i]=1;
			if(i%prime[j]==0){
				phi[i*prime[j]] = phi[i]*prime[j];
				break;
			}
			else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
		}
	}
}


int main()
{
	set();
	int i, t;
	scanf("%d", &t);
	for (; t; t--) {
		scanf("%d", &i);
		printf("%d\n", phi[i]);
	}
	return 0;
}
