#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main(){
	ll m,n,i;
	cin>>m>>n;
	ll len = 0;
	ll a[m+5],f[m+5];
	for(i=1; i<=n; i++){
	  cin>>a[i];
	  len = len+a[i];
	}
	f[1] = 1;
	f[2] = 1;
	ll sum = f[1]+f[2];
	for(i=3; i<=m;i++){
		f[i] = f[i-1]+f[i-2];
		sum = sum+f[i];
	}
	ll ans = sum-len;
	 cout<<ans << endl;
	 return 0;
}
