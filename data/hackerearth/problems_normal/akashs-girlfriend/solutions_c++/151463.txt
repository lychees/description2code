long long power(long long a,long long b,long long c)
{
	if(b==0)
	return 1;
	long long temp=power(a,b/2,c);
	temp=(temp*temp)%c;
	if(b&1)
	temp=(temp*a)%c;
	if(temp<0)
	temp+=c;
	return temp;
}
int main()
{
	int t;
	long long n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		long long x=(5*power(10,n-1,1000000009))%1000000009;
		if(x<0)
		x+=1000000009;
		//cout<<x<<endl;
		long long y=(4*power(8,n-1,1000000009))%1000000009;
		if(y<0)
		y+=1000000009;
		x=x-y;
		if(x<0)
		x+=1000000009;
		x=x%1000000009;
		cout<<x<<endl;
	}
	return 0;
}
