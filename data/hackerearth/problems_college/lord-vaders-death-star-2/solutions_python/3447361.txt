N = int(input())

arr = map(int, raw_input().split())
m = dict()
for i in arr:
    if(i not in m):
        m[i] = 1 
    else:
        m[i] += 1            
        
found = False
p=0
for i in m:
    if(m[i] == 1 and i != 0):
        found = True
        if(p < i):
            p = i

if(not found):
    print -1   
else:
    print p
