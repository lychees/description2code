#include <iostream>
using namespace std;

void fact(int n) {
	
	int data[200]={0},no=1,j,i,prod,car=0;
		data[0]=1;
	
	for(i=1;i<=n;i++) {
		
		for(j=0;j<no;j++) {
			
			prod = data[j]*i + car;
			car = prod/10;
			data[j] = prod %10;
			
		}
		
		while(car>0) {
			data[no] = car%10;
			car=car/10;
			no++;
		}
	}
	
	for(i=no-1;i>=0;i--) {
		cout<<data[i];
	}
	cout<<"\n";
}

int main()
{
    int N;
    cin>>N;
    for(int i=2;i<=N;i++) {
    	int count =0;
    	for(int j=2;j<i;j++) {
    		if(i%j == 0) count ++;
    	}
    	if(count ==0) fact(i);
    }
    return 0;
}
