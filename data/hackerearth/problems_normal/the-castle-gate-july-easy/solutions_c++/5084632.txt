#include <iostream>
using namespace std;

int main() {
	int sampleCount;
	cin >> sampleCount;
	int solutionArray[sampleCount];
	int tempSample;
	for(int i = 0; i < sampleCount; i++) {
		solutionArray[i] = 0;
		cin >> tempSample;
		for(int j = 1; j < tempSample; j++) {
			for(int k = j + 1; k <= tempSample; k++) {
				if((j ^ k) <= tempSample) {
					solutionArray[i]++;
				}
			}
		}
	}
	for(int i = 0; i < sampleCount; i++) {
		cout << solutionArray[i] << endl;
	}
	return 0;
}
