#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,i,cont=0,max=0;
	cin>>n;
	long long a[n];
	for(i=0; i<n; i++)
	cin>>a[i];
		for(i=1; i<n; i++)
		{
			if((abs(a[i]-a[i-1])==1))
				cont++;
			else
				cont=0;
			if(cont>max)
				max=cont;
		}
	cout<<max+1<<endl;
		
	return 0;
}
