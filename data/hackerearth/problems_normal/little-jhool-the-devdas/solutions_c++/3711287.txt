#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,odd_min=101,sum=0,flag=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		int x;
		cin>>x;
		if(x%2==1)
		{
			if(x<odd_min)
			odd_min=x;
			flag=1;
		}
		sum+=x;
	}
    if(sum%2==0&&flag==1)
    {
    	sum=sum-odd_min;
    	cout<<sum<<"\n";
	}
	else if(sum%2!=0)
	cout<<sum<<"\n";
	else
	cout<<":(\n";
	return 0;
	
    
}
