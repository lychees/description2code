#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
 int t;
 cin >> t;
 while(t--)
 {
 	string s;
 	cin >> s;
 	int hash[26]={0};
 		int f=0;
 	for(int i=0;i<s.length();i++)
 	{
 	
 		hash[s[i]-'a']++;
 		if(hash[s[i]-'a']==2)
 		{
 			cout << "Yes" << endl;
 			f=1;
 		}
 		if(f==1)
 		break;
 	}
 	if(f==0)
 	cout << "No" << endl;
 	
 }
    return 0;
}
