mx = 10**6+7
md = 10**9+7
ans = [0 for i in range(mx)]
for i in range(3):
    ans[i] = i
for i in range(3,mx):
    ans[i] = ((ans[i-1]%md) + (((i-1)*ans[i-2])%md))%md
t = input()
for _ in range(t):
    n = input()
    print ans[n]
