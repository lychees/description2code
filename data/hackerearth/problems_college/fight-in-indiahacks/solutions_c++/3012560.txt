#include <bits/stdc++.h>
using namespace std;

int M1, M2;
int dp[1000006];

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
    	int N;
    	scanf("%d %d %d", &N, &M1, &M2);
    	
    	dp[0] = 0;
    	
    	for(int i=1; i<=N; i++)
    	{
    		int x = 1;
    		
    		if(i-M1>=0)
    			x &= dp[i-M1];
    		if(i-M2>=0)
    			x &= dp[i-M2];
    		
    		dp[i] = (1^x);
    	}
    	
    	printf("%d\n", dp[N]);
    }
    
    return 0;
}
