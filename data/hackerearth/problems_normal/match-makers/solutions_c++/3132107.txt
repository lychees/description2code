#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main(){
	int t;
	cin >> t;
	for (int i = 0; i < t; i++){
		int n;
		cin >> n;
		vector <int> girls, boys;
		for (int i = 0; i < n; i++){
			int temp;
			cin >> temp;
			girls.push_back(temp);
		}
		for (int i = 0; i < n; i++){
			int temp;
			cin >> temp;
			boys.push_back(temp);
		}
		sort(girls.begin(), girls.begin() + girls.size());
		sort(boys.begin(), boys.begin() + boys.size());
		long long cnt = 0;
		for (int i = 0; i < n; i++){
			if (girls[i] % boys[n - 1 - i] == 0 || boys[n - 1 - i] % girls[i] == 0){
				cnt++;
			}
		}
		cout << cnt << endl;
	}
	return 0;
}
