#include <iostream>
#include <cstring>
using namespace std;

bool validateEmail(char *email){
	int len=strlen(email),i,flag=0;
	if(len>44)
		return false;
	for(i=0;i<len && i<16;i++){
		
		if((email[i]>='A' && email[i]<='Z') || (email[i]>='a' && email[i]<='z') || (email[i]>='0' && email[i]<='9') || email[i]=='_')
			continue;
		else if(email[i]=='@') break;
		else return false;
	}
	if(i>16 || i==0 || i>=len)
		return false;
	int j=0;
	i++;
	for(;i<len && j<16;i++,j++){
		if((email[i]>='A' && email[i]<='Z') || (email[i]>='a' && email[i]<='z')){
			flag=1;
			continue;
		}
		else if(email[i]=='-' || (email[i]>='0' && email[i]<='9'))
			continue;
		else if(email[i]=='.') break;
		else return false;
	}
	if(j>16 || i>=len || j==0 || flag!=1)
		return false;
	j=0;
	i++;
	for(;i<len && j<10;i++,j++){
		if(email[i]>='a' && email[i]<='z')
			continue;
		else
			return false;
	}
	if(j>=10)
		return false;
	return true;
}

int main()
{
    int t;
    char email[1000];
    cin>>t;
    while(t--){
    	cin>>email;
    	if(validateEmail(email))
    		cout<<"YES";
    	else
    		cout<<"NO";
    	cout<<endl;
    }
    return 0;
}
