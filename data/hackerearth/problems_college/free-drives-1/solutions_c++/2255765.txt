#include <iostream>
using namespace std;
#define MAX1 2001
#define MAX2 4002
#define MOD 1000000000
long long int c[MAX1][MAX2]={0};

long long int comb(long long int n,long long int r)
{
	if(c[n][r]) return c[n][r];
    if(n==r) return c[n][n-r] = c[n][r] = 1;
    if(r==0) return c[n][n-r] = c[n][r] = 1;
    if(r==1) return c[n][n-r] = c[n][r] = ((long long int)n % MOD);
    return c[n][r] = c[n][n-r] = (comb(n-1,r) + comb(n-1,r-1) % MOD);
}

int main()
{
    long long int i,j,k,l,n,t;
    cin >> t ;
    while(t--)
    {
    	cin >> n;
    	cout << n*(n-1)/2 << endl ;
    }
    return 0;
}
