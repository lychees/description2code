#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <stack>
#include <string>
#include <vector>
using namespace std;
bool fa[30];
int d[30];
int main() {
  int n;
  scanf("%d", &n);
  memset(fa, 0, sizeof(fa));
  for (int i = 0; i < n; i++)
    scanf("%d", &d[i]);
  sort(d, d + n);
  for (int i = 0; i < n; i++) {
    int k = 1, j;
    for (j = i; k < d[i] && j >= 0; j--) {
      if ((!fa[j]) && d[j] + 2 <= d[i] && d[j] + k <= d[i]) {
        k += d[j];
        fa[j] = 1;
      }
    }
    if (j < 0 && k < d[i]) {
      printf("NO\n");
      return 0;
    }
  }
  for (int i = 0; i < n - 1; i++) {
    if (!fa[i]) {
      printf("NO\n");
      return 0;
    }
  }
  printf("YES\n");
  return 0;
}
