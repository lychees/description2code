#define MOD       1000000007

#define MAXN      1000005

#define CASE(x)   cout << "Case #" << x << ": ";
int p[MAXN], n, a[MAXN], cnt[MAXN];
int main()
{
	int maxVal = -INF;
    for (int i = 2; i <= 1000005; i++)
		p[i] = 1;
	for (int i = 2; i <= sqrt(1000005); i++)
	{
		if (p[i] != 0)
			for (int j = i * i; j <= 1000005; j += i)
				p[j] = 0;
	}
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		maxVal = max(maxVal, a[i]);
		cnt[a[i]]++;
	}
	for (int i = maxVal; i >= 2; i--)
	{
		if (p[i] && cnt[i])
		{
			cout << i << endl << cnt[i] << endl;
			break;
		}
	}
    return  0;
}
