#include <iostream>
using namespace std;

int a[1000010],b[1000010];

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int m,n;
    	string str;
    	cin >> m >> n;
    	cin >> str;
    	int l = str.size();
    	int x,y;
    	x=y=0;
    	for(int i=0;i<l;i++) {
    		if(str[i] == 'R') a[x++] = i;
    		else b[y++] = i;
    	}
    	long long an1,an2;
    	an1=an2=0;
    	if(m>0) {
    		for(int i=0;i+m-1<x;i++) {
    			long long t1,t2;
    			t1 = a[i];
    			if(i-1>=0) t1 -= a[i-1]; else t1++;
    			t2 = -1*a[i+m-1];
    			if(i+m<x) t2 += a[i+m]; else t2+=l;
    			an1 += t1*t2;
    		}
    	} else {
    		for(int i=0;i<x;i++) {
    			long long t = a[i];
    			if(i-1>=0) t -= a[i-1]; else t++;
    			t--;
    			an1 += (t*(t+1))/2;
    		}
    		long long t = l;
    		if(x-1>=0) t -= a[x-1]+1;
    		an1 += (t*(t+1))/2;
    	}
    	if(n>0) {
    		for(int i=0;i+n-1<y;i++) {
    			long long t1,t2;
    			t1 = b[i];
    			if(i-1>=0) t1 -= b[i-1]; else t1++;
    			t2 = 0-b[i+n-1];
    			if(i+n<y) t2 += b[i+n]; else t2 += l;
    			an2 += t1*t2;
    		}
    		
    	} else {
    		for(int i=0;i<y;i++) {
    			long long t = b[i];
    			if(i-1>=0) t -= b[i-1]; else t++;
    			t--;
    			an2 += (t*(t+1))/2;
    		}
    		long long t = l;
    		if(y-1>=0) t -= b[y-1]+1;
    		an2 += (t*(t+1))/2;
    	}
    	cout << an1 << " " << an2 << endl;
    }
    return 0;
}
