#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a,b,c,sum,l,i,s,j,p,m;
ll nd(ll n)
{
   c=0,s=1;
   while(n%2==0)
   {
       c++;n/=2;
   }
   if(c>0)
   s*=(c+1);
   p=sqrt(n);
   for(j=3;j<=p;j+=2)
   {
       c=0;
       while(n%j==0)
       {
           c++;n/=j;
       }
       if(c>0)
        s*=(c+1);
   }
   if(n>1)
    s*=2;
   return s;
}
int main()
{
   scanf("%lld",&m);
   for(i=1;i<=m;i++)
   {
       cin>>a;

       if(nd(a)>=4)
        printf("YES\n");
       else
        printf("NO\n");
   }

}
