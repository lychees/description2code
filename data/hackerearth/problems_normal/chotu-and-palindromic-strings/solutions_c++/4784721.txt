#include <iostream>
using namespace std;

bool palindrome(string s1,string s2){
	string s3=s1+s2;
	int length=s3.length();
	int i=0,flag=0;
	length=length-1;
	while(i<length){
		if(s3[i++]!=s3[length--]){
			flag=1;
			return 0;
		}
	}
	if(flag==0){
		return 1;
	}
	
	
}
int main()
{
    int t;
    cin>>t;
    string s1,s2;
    while(t>0){
    	cin>>s1;
    	cin>>s2;
    	int check=(  palindrome(s1,s2) | palindrome(s2,s1) );
    	if(check){
    		cout<<"YES"<<endl;
    	}
    	else cout<<"NO"<<endl;
    	
    	t--;
    }
    return 0;
}
