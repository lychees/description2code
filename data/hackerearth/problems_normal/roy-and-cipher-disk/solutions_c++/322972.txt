#include <iostream>
#include <string>

using namespace std;

int main() {
    int n_tc = 0;
	cin >> n_tc;

	while (n_tc--) {
		string plaintext;
		cin >> plaintext;

		int cur = 0, nxt = 0;
		for (int i = 0; i < plaintext.length(); i++) {
			nxt = plaintext[i] - 'a';
			if (cur <= nxt) {
				if (nxt - cur <= 26 - (nxt - cur))
					cout << (nxt - cur) << " ";
				else
					cout << (nxt - cur - 26) << " ";
			}
			else {
				if (cur - nxt < 26 - (cur - nxt))
					cout << (nxt - cur) << " ";
				else
					cout << 26 - (cur - nxt) << " ";
			}
			cur = nxt;
		}
		cout << endl;
	}
	return 0;
}
