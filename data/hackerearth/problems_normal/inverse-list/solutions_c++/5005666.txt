#include <bits/stdc++.h>

using namespace std;

int org [100000 + 10];
int arr [100000 + 10];

int main()
{
    //ios::sync_with_stdio(false);cin.tie(0);       // decrease the time of cin, cout

    int n , t;

    cin >>      t          ;

    while( t-- ){

        cin >> n;

        for(int i = 1; i <= n; i++){

            cin >> org[i];                              
            arr[  org[i] ] = i;                 // set the index every number
        }

        bool revers = true;

        for(int i = 1; i <= n; i++){

            if(org[i] != arr[i]){               // this number isn't in the correct index

                revers = false;         
                break;
            }
        }

        if(revers) cout << "inverse\n";
        else cout << "not inverse\n";
    }

    return 0;
}
