#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	map<string,int>m;
	map<string,int>::iterator it;
	for(int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		it=m.find(s);
		if(it!=m.end())
		{
			it->second++;
		}
		else
		m.insert(pair<string,int>(s,1));
	}
	
	vector<pair<int,string> >v;
	map<string,int>::iterator p;
	for(p=m.begin();p!=m.end();p++)
	{
		v.push_back(make_pair(p->second,p->first));
	}
	sort(v.begin(),v.end());
	int l2=v.size();
	for(int i=l2-1;i>=0;i--)
	{
		cout<<v.at(i).second<<" "<<v.at(i).first<<endl;
	}
}
