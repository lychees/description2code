string rr(string x){

	int nn=x.size(),i=0;
	while(i<nn){
		char temp=x[i];
		  x[i]=x[nn-1];
		  x[nn-1]=temp;
		  i++,nn--;
	}
  return x;
}

int isReverse(string a, string b)
{
	if(a.length()!=b.length())
		return false;
	int len = a.length();
	for(int i=0;i<len;i++)
		if(a[i]!=b[len-1-i])
			return false;

	return true;

}
