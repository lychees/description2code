#include <bits/stdc++.h>
using namespace std;
int main()
{
    float t,r1,r2,x1,y1,x2,y2;
    float d;
    cin>>t;
    while(t--)
    {
        cin>>x1>>y1>>r1;
        cin>>x2>>y2>>r2;
        d=sqrtf(abs(x1-x2)*abs(x1-x2)+abs(y1-y2)*abs(y1-y2));
        if(r1>r2 && r1>=d+r2)
        {
             cout<<"C1CC2"<<endl;
        }
        else if(r1<r2 && r2>=d+r1)
        {
             cout<<"C2CC1"<<endl;
        }
        else if(r1==r2 && d==0)
        {
            cout<<"C2~C1"<<endl;
        }
        else
        {
            cout<<"NOT SATISFIED"<< endl;
        }
    }
   return 0;
}
