bool stringmatch( string a, string b)
{
	if (a[0]!=b[0])
		return 0;
	else if (a.back()!=b.back())
		return 0;
	else
		{
			sort(a.begin(), a.end());
			sort(b.begin(), b.end());
			if(a==b)
				return 1;
		
			else return 0;
		}
}

int main()
{
    int t;
    int group=1;
    cin>>t;
    string ara[t][t];
    string str;
    cin>>ara[0][0];
    
    for (int i=1;i<t;i++)
    {
    	cin>>str;
    	for (int j=0; j<group; j++)
    	{
    		if (stringmatch( ara[j][0], str ))
    			  				break;
    			
    		else if ( j==group-1)
    			{ara[j+1][0]= str;
    			group+=1;
    			break;}
    }

    }
        cout<<group;
    return 0;
}
