#include <bits/stdc++.h>
using namespace std;

#define iter(c) __typeof((c).begin())
#define tr(i, c) for (iter(c) i = (c).begin(); i != (c).end(); i++)
#define rep(i, a, b) for (int i = (a); i < (b); i++)
#define read(x) scanf("%lld", &(x))
#define debug(x) cout << #x << "  " << x << endl

#define mp make_pair
#define pb push_back
#define f first
#define s second

typedef long long ll;
typedef pair<int, int> ii;

const int MAXN = int(1e5) + 5;

int n, a[MAXN], q;

bool bs(int val) {
    int l = 0, h = n-1;
    while (l <= h) {
        int m = (h-l)/2 + l;
        if (a[m] == val) return true;
        else if (a[m] > val) h = m-1;
        else l = m+1;
    }
    return false;
}

int main() {
//    freopen("input.inp", "r", stdin);

    read(n), read(q);
    rep(i, 0, n) read(a[i]);

    sort(a, a+n);

    rep(i, 0, q) {
        int val; read(val);
        bool r = bs(val);
        printf(r ? "YES\n" : "NO\n");
    }
}
