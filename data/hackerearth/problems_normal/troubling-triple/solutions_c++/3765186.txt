#include<cstdio>
#include<vector>
#include<algorithm>

using namespace std;

vector<int>::iterator its;

vector<vector<int> > v;

int val[2001];

int main(){
    int n, k;
    scanf("%d %d", &n, &k);
    for(int i=0;i<n;i++){
        scanf("%d", &val[i]);
    }
    for(int i=0;i<n;i++){
        vector<int> temp(val+i, val+n);
        sort(temp.begin(), temp.end());
        v.push_back(temp);
    }
    int ans = 0;
    for(int i=0;i<n-2;i++){
        for(int j=i+1;j<n-1;j++){
            int sum = k/(val[i]*val[j]);
            if(sum != 0){
                vector<int> temp = v[j+1];
                its = upper_bound(temp.begin(), temp.end(), sum);
                ans += (its - temp.begin());
            }
        }
    }
    printf("%d\n", ans);
}
