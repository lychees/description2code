#include<iostream>
#include<math.h>
#include<map>
#include<vector>
#include<algorithm>
using namespace std;

void fun(vector<int> &vij)
{
	for(int i=0;i<91;i++)
	{
		vij[i]=i*i*i;
	}
}
void bat(vector<int> &vij,vector<int> &ans)
{
	map<int,bool> m;
	int sum;
	for(int i=1;i<vij.size();i++)
	{
		for(int j=i;j<vij.size();j++)
		{
			sum=vij[i]+vij[j];
			if(m.count(sum)>0) // already in map
			{
				ans.push_back(sum);
			}
			else
			{
				m[sum]=1;
			}
		}
	}
	ans.push_back(-1);
	sort(ans.begin(),ans.end());
}
void sample(vector<int> &ans,int n)
{
	int s=ans.size();
	int i,c;
	for(i=0;i<s;i++)
	{
		if(n<ans[i])
			i++;
		else
		{
		    c=i;	
		}
	}
	cout<<ans[c]<<endl;
}
int main()
{
	vector<int> vij(91,0);
	vector<int> ans;
	int t,c,e;
	int n=704977;
	scanf("%d",&t);
	c=int(pow(n,1.0/3));
	c+=1;
	fun(vij);
	bat(vij,ans);
	while(t--)
	{
		scanf("%d",&n);
		sample(ans,n);
	}
}
