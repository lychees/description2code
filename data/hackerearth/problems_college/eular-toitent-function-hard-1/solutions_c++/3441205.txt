#include<bits/stdc++.h>
#define MAX 10000000
using namespace std;
int prime[MAX] ;

void etf()
{
	//memset(prime,true,sizeof(prime)) ;
	//prime[0] = false ;
	//prime[1] = false ;
	for(int i=1;i<=MAX;i++)
		prime[i] = i ;
	for(int i=2;i<=MAX;i++)
	{
		if(prime[i]==i)
		{
			for(int j=i;j<=MAX;j+=i)
				prime[j] = (prime[j]/i)*(i-1) ;
		}
	}
}

int main()
{
	etf() ;
	int t,x ;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&x);
		printf("%d\n",prime[x]) ;
	}
	
	
	
}
