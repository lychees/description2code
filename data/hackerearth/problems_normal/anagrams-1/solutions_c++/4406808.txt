const int MAXN = 1e5 + 3;
int T;
string s;
int cnt[150];
void solve() {
	cin >> T; cin.ignore();
	while (T--) {
		memset(cnt, 0, sizeof cnt);
		getline(cin, s, '\n');
		forn(i, s.length()) { if (s[i] < 'a') s[i] += 'a' - 'A'; cnt[s[i]]++; }
		getline(cin, s, '\n');
		forn(i, s.length()) { if (s[i] < 'a') s[i] += 'a' - 'A'; cnt[s[i]]--; }
		int flag = 0;
		int res = 0; // -1: noluck, 0: yes, 1: found
		int id = -1;
		fore(i, 'a', 'z') {
			if (cnt[i] != 0) {
				if (!flag) { flag = cnt[i] < 0 ? -1 : 1; res = 1; }
				else if (flag != (cnt[i] < 0 ? -1 : 1)) { res = -1; break; }
				if (abs(cnt[i]) % 2) {
					if (id == -1) id = i;
					else {res = -1; break;}
				};
			}
		}
		if (res == 0) puts("YES");
		else {
			if (res == -1) puts("NO LUCK");
			else {
				string str = "";
				fore(i, 'a', 'z') {
					forn(j, abs(cnt[i]) / 2) printf("%c", i);
				}
				if (id != -1) printf("%c", id);
				for (int i = 'z'; i >= 'a'; --i) {
					forn(j, abs(cnt[i]) / 2) printf("%c", i);
				}
				printf("\n");
			}
		}
	}
}
