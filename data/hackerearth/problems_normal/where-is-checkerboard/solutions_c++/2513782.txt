const int mod = 1000000000 + 7;
const int maxn = 2500;
const int maxm = 10000 + 10;
const int inv2 = 500000004;
const int inv6 = 166666668;

char checker[maxn][maxn];
int dp[maxn][maxn];


int main() {

	// f_in("input.txt");
	int n, m; cin >> n >> m;
	for(int i=1; i<=n; i++) {
		for(int j=1; j<=m; j++) {
			cin >> checker[i][j];
		}
	}
	int ans = 0;
	for(int i=n; i>=1; i--) {
		for(int j=m; j>=1; j--) {
			if(checker[i][j] == '_') {
				dp[i][j] = 1;
				if(checker[i+1][j+1] == '_' && checker[i+1][j] == '#' && checker[i][j+1] == '#') {
					dp[i][j] += min(dp[i+1][j+1], min(dp[i+1][j], dp[i][j+1]));
				}
				if(dp[i][j] >= 32) ans ++;
			} else {
				dp[i][j] = 1;
				if(checker[i+1][j+1] == '#' && checker[i+1][j] == '_' && checker[i][j+1] == '_') {
					dp[i][j] += min(dp[i+1][j+1], min(dp[i+1][j], dp[i][j+1]));
				}
			}
		}
	}
	cout << ans << "\n";
	return 0;
}
