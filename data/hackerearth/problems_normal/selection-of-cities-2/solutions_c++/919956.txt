#include <iostream>
using namespace std;
#include<math.h>

int fastP(long long base,long long n,long long M){
	if(n==0)
	return 1;
	if(n==1)
	return base;
	long long halfn=fastP(base,n/2,M);
	if(n%2==0)
	return (halfn*halfn)%M;
	else
	return (((halfn*halfn)%M)*base)%M;

}
int main()
{
	long long n;
	cin>>n;
	long long *a=new long long [n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<n;i++){
		int  temp=fastP(2,a[i],1000000007);
		cout<<(temp-1);
		cout<<"\n";
	}
