#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int n,i,j,count;
	cin>>n;
	string s;
	cin>>s;
	bool a[n+1];
	memset(a,false,sizeof(a));
	for(i=0,j=1,count=n;i<s.length();i++,j++){
		if(j==n+1)
		j=1;
		if(s[i]=='b'){
			if(a[j]==false){
			a[j]=true;
			--count;
			}
			else
			--i;
		}
		else{
			if(a[j]==true)
			--i;
		}
		if(i==s.length()-1)
		i=-1;
		if(count==1)
		break;
	}
	for(i=1;i<=n;i++){
		if(a[i]==false)
		break;
	}
	cout<<i;
	return 0;
}
