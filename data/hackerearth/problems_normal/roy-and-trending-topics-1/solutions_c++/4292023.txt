#include <iostream>
#include <queue>
#include <map>
using namespace std;

int main()
{
    long long n,id;cin>>n;
    priority_queue<pair<long,long> > pr;
    map<long,long> ma;
    while(n--){
    	long long z,p,l,c,s;
    	cin>>id>>z>>p>>l>>c>>s;
    	ma[id]=z;
    	//cout<<z<<" ";
    	z=p*50+l*5+c*10+s*20-z;
    	pr.push(make_pair(z,id));
    }
    int i=0;
    while(i<5){
    	pair<long,long> x=pr.top();
    	cout<<x.second<<" "<<ma[x.second]+x.first<<endl;
    	pr.pop();
    	i++;
    }
    return 0;
}
