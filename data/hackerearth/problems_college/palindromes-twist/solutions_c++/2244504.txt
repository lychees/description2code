#include<stdio.h>
#include<string.h>
int main()
{
	int acost,bcost,t,i,j,n,mincost,cost,flag;
  char str[10010],c;
  scanf("%d",&t);
  while(t--)
  {	scanf("%s",str);
        scanf("%d%d",&acost,&bcost);
        n=strlen(str);
        i=0;j=n-1;
        mincost=acost<bcost?acost:bcost;
        if(mincost==acost) c='a';
        else c='b';
        cost=0;flag=1;
        while(i < j)
        { //printf("hello\n"); 
        if(str[i]=='/' && str[j]!='/')
          { str[i]=str[j];
            if(str[i]=='a')cost+=acost;
             else cost+=bcost;
          i++; j--;
          }
          else if(str[j]=='/' && str[i]!='/')
          { str[j]=str[i];
            if(str[i]=='a') cost+=acost;
            else cost+=bcost;
            i++;j--;
          }
           else if(str[i]=='/' && str[j]=='/')
              {  str[i]=str[j]=c;
                  cost+=2*mincost;
                  i++;j--;
              }
            else if(str[i]!=str[j])
            { printf("-1\n");flag=0;
              break;
            }
            else
            {i++;
             j--;
            }
        }                     
      if(flag==1)
      printf("%d\n",cost);
  }
return 0;
}
