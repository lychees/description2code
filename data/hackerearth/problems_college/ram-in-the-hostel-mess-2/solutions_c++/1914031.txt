
#include <iostream>

#include <math.h>
using namespace std;


int fact(int n){
	return (n==0 || n==1) ? 1 : fact(n-1)*n;
}

int ncr(int n,int r){
	int result=fact(n)/(fact(r)*fact(n-r));
	return result;
}

int main() {


	int test;
	cin>>test;
	int a;
	int result=0;
	int p,q;
	while(test--){
		cin>>a;

		p=1;q=2;
		if(a==1)
			cout<<"4"<<endl;
		else{
			a=a-1;
			while(a--){
				result=p+q;
				p=q;
				q=result;
			}
			cout<<result*result<<endl;

		}

	}

	return 0;

}
