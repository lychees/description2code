#include<cstdio>
#include<iostream>

using namespace std;

int main()
{
   int t,n,i,p,q,k,l,j,r,b[10000],c[10000],root,f;
   char a[100005];

     scanf("%d%d",&n,&t);
    scanf("%s",a);

    while(t--)
    {
           scanf("%d%d",&p,&q);
            f=1;
            k=p;
            i=0;
            while(k)
            {
                b[i++]=k%2;
                k/=2;
            }
             j=0;
             k=q;
            while(k)
            {
                c[j++]=k%2;
                k/=2;
            }
            k=i-1;
            l=j-1;
            root=0;
            while(b[k]==c[l]&&k>=0&&l>=0)
            {
                root=root*2+b[k];
                k--;
                l--;

            }

            if(root==p)
            {
                while(root<=q&&l>=-1)
                {
                    if(a[root-1]=='1')
                    {
                        f=0;
                        break;
                    }
                    root=2*root+c[l--];
                }
            }else
             if(root==q)
            {
                while(root<=p&&k>=-1)
                {
                    if(a[root-1]=='1')
                    {
                        f=0;
                        break;
                    }
                    root=2*root+b[k--];
                }
            }else
            {
                r=root;
                 while(r<=q&&l>=-1)
                {
                    if(a[r-1]=='1')
                    {
                        f=0;
                        break;
                    }
                    r=2*r+c[l--];
                }
                while(root<=p&&k>=-1)
                {
                    if(a[root-1]=='1')
                    {
                        f=0;
                        break;
                    }
                    root=2*root+b[k--];
                }

            }


             if(f)
                printf("Yes\n");
             else
                printf("No\n");

    }
    return 0;
}
