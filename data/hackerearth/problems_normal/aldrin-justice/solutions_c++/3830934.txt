#include<iostream>
using namespace std;

void solve()
{
	int a,b,c,d;
	cin >> a >> b >> c >> d;
	if(a > b)
		swap(a,b);
	if(c > d)
		swap(c,d);
 
	int x = max(a,c);
	int y = min(b,d);
	if(x == y)
		printf("Point\n");
	else if(x > y)
		printf("Line\n");
	else
		printf("Nothing\n");
}
 
int main()
{
	int t;
	cin >> t;
	while(t--)
		solve();
	return 0;
}
