using namespace std;

typedef pair<int,int> ii;
typedef long long ll;
typedef unsigned long long ull;




const int N=1e5+10;
set<ii>st;
int n,res,x,y;
char s[N];







int main()
{
	std::ios::sync_with_stdio(0);
	cin.tie(0);
	cin>>s;
	n=strlen(s);
	x=0;
	y=0;
	st.insert(make_pair(x,y));
	res=0;
	for(int i=0;i<n;i++)
	{
		if(s[i]=='L')y--;
		else if(s[i]=='R')y++;
		else if(s[i]=='U')x--;
		else if(s[i]=='D')x++;
		if(st.find(make_pair(x,y))!=st.end())res++;
		else st.insert(make_pair(x,y));
	}
	cout<<res<<"\n";
	return 0;
}
