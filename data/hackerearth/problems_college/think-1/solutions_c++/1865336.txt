#include <bits/stdc++.h>

using  namespace std;
long long int rev(long long int i )
{
    long long int x=0,rem;
    while(i)
    {
        rem=i%10;
        x=x*10+rem;
        i/=10;
    }
    return x;
}
int main()
{
    long long int t,n,rn;
    cin>>t;
    while(t--)
    {
        cin>>n;
        rn=rev(n);
        if(rn*rn==rev(n*n) && n*n==rev(rn*rn) && rn!=n )
        {
            cout<<rn<<endl;
        }
        else
        {
            cout<<"No"<<endl;
        }
    }
}
