#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    float start[n];
    float end[n];
    for(int i =0;i<n;i++)
    {
        float a,b,c,d;
        scanf("%f:%f-%f:%f",&a,&b,&c,&d);
        b=b/60;
        d=d/60;
        a=a+b;
        c=c+d;
        start[i]=a;
        end[i]=c;
    }
    int temp=0;

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(start[j]>start[i] && start[j]<end[i])
            {
                //cout<<"Will need a moderator!"<<endl;
                temp=1;
                goto here;
            }
            else if(end[j]>start[i] && end[j]<end[i])
            {
                //cout<<"Will need a moderator!"<<endl;
                temp=1;
                goto here;
            }
            else
            {
                temp=0;
            }
        }
    }
    
    here:
        if(temp==1)
        {
            cout<<"Will need a moderator!"<<endl;
        }
        else
        {
            cout<<"Who needs a moderator?"<<endl;
        }
        
}
