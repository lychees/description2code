#include <iostream>
using namespace std;

int main()
{
	int n,q,i,j,c,l,r;
	cin >> n >> q;
	long long int a[n],b[n];
	for(i=0;i<n;i++) cin >> a[i];
	for(i=0;i<n;i++) cin >> b[i];
	for(i=0;i<q;i++){
		long long int sum=0;
		cin >> c >> l >> r;
		if(c==1){
			for(j=l-1;j<r;j++){
				if((j-l+1)%2==0) sum = sum + a[j];
				else sum = sum + b[j];
			}
		}
		else{
			for(j=l-1;j<r;j++){
				if((j-l+1)%2==0) sum = sum + b[j];
				else sum = sum + a[j];
			}
		}
		cout << sum << endl;
	}
}
