#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;
int myCompare(string X, string Y)
{   string XY = X.append(Y);
    string YX = Y.append(X);
    return YX.compare(XY) > 0 ? 1: 0;
}
void printLargest(vector<string> arr)
{  sort(arr.begin(), arr.end(), myCompare);
   for (int i=0; i < arr.size() ; i++ )
        cout << arr[i];
}
int main()
{   vector<string> arr;
    int n;
    char k[15];
        cin>>n;
        for(int i=0;i<n;i++){
                cin>>k;
                
        arr.push_back(k);
        }
   printLargest(arr);
   cout<<endl;
        return 0;
}
