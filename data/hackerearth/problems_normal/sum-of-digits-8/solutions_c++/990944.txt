ll a,b,S;
ll dp[20][200];// sum j is formed using i number of digits
char c[20];

ll solve(int digits,int sum){
	if(digits==0)return sum == 0;
	if(dp[digits][sum]!=-1)return dp[digits][sum];
	ll ans=0;
	for(int i=0; i<=9; ++i){
		if(sum-i>=0)ans+=solve(digits-1,sum-i);
	}
	return dp[digits][sum]=ans;
}

ll f(ll d){
	sprintf(c,"%lld",d);
	int len=strlen(c);
	ll ans=0;
	int remain=len;
	int sum=S;
	for(int i=0;i<len;++i){
		remain--;
		int dig=c[i]-'0';
		for(int j=0; j<dig; ++j){
			if(sum-j>=0)ans+=solve(remain,sum-j);
		}
		sum-=dig;
	}
	return ans;
}
