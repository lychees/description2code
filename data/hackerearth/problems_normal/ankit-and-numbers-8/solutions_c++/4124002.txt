#include <bits/stdc++.h>
using namespace std;

#include <stdio.h>

int main()
{
	int t;
	cin >> t;
	
	for(int i= 0;i < t; i++) {
		int n;
		cin >> n;
		
		long long int sum = (n)*(n + 1);
		sum  = sum / 2;
		
		long long int nooftimes = pow(2, n - 1);
		
		cout << sum * nooftimes << endl;
		
	}
	return 0;
}
