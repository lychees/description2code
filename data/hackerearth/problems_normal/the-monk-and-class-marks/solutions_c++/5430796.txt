#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int main()
{
	int N, tmp;
	string str;
	cin >> N;
	vector<pair<int, string>> v;
	for(int i=0; i<N; i++){
		cin >> str >> tmp;
		v.push_back(make_pair(-tmp, str));
	}
	sort(v.begin(), v.end());
	for(int i=0; i<N; i++){
		cout << v[i].second << " " << -1*v[i].first << '\n';
	}
    return 0;
}
