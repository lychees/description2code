tc = int(raw_input())
for i in range(tc) :
	dicts = {'a':0,'b':0,'c':0,'d':0,'e':0,'f':0,'g':0,'h':0,'i':0,'j':0,'k':0,'l':0,'m':0,'n':0,'o':0,'p'				:0,'q':0,'r':0,'s':0,'t':0,'u':0,'v':0,'w':0,'x':0,'y':0,'z':0}
	line = raw_input()
	for letter in line :
		dicts[letter] += 1
	keys = dicts.keys()
	num = dicts[keys[0]]
	yes = True
	for key in keys :
		if dicts[key] != num :
			yes = False
			print 'No'
			break
	if yes :
		print 'Yes'
