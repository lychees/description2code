#include <iostream>
using namespace std;
typedef long long int ll;
int main()
{
    ll n;
    cin >> n;
    ll a[n + 1];
    ll arr[n + 1];
    ll i, j;
    for(i = 1; i <= n; i++) {
    	cin >> a[i];
    	arr[i] = 0;
    }
    ll val;
    arr[1] = a[1];
    for(i = 1; i <= n; i++) {
    	for(j = 2 * i; j <= n; j+=i) {
    		val = arr[i] + a[j];
    		if(val > arr[j]) {
    			arr[j] = val;
    		}
    	}
    }
    for(i = 1; i <= n; i++) {
    	cout << arr[i] << " ";
    }
    return 0;
}
