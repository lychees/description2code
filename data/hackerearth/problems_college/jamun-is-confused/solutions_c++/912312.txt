#include <iostream>
#include <stdio.h>
using namespace std;

long gcd(long a,long b)
{
	if((a = a%b) == 0)
		return b;
	long ans;
	if (a>b)
		ans = gcd(a,b);
	else 
		ans = gcd(b,a);
	return ans;
}

int main()
{
    int T,N;
    long final;
    long *S;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d",&N);
    	S = new long[N];
    	for(int i=0;i<N;i++)
    	{
    		scanf("%ld",&S[i]);
    	}
    	if(S[0] == 0)
    		final = S[1];
    	else 
    		final = S[0];
    	if(N == 1){
    		cout<<S[0]<<endl; break;}
    	for(int i=1;i<N;i++)
    	{
    		if(final == 1)
    			break;
    		if(S[i] == 0)
    			continue;
    		if(final <= S[i])
    			final = gcd(S[i],final);
    		else
    			final = gcd(final,S[i]);
    	}
    	printf("%ld\n",final);
    	delete []S;
    }
    return 0;
}
