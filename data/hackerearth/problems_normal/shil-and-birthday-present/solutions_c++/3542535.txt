#include<stdio.h>
#define ll long long
#define sll(x) scanf("%lld",&x)
ll  n;
ll  cnt[1<<20];
int main()
{
	sll(n);
	ll i;
	for(i=1;i<=n;++i)
	{
		ll temp;
		sll(temp);
		cnt[temp]=1;
	}	
	ll dinct=0;
	for(i=1;i<=n;++i)
	{
		if(cnt[i]==1)
			dinct++;
	}
	ll ans=((dinct)*(dinct-1))/2;
	printf("%lld\n",ans);

	return 0;
}
