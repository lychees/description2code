#include<iostream>
using namespace std;
int main()
{
   int n,i;
   cin>>n;
   int a[n];
   for(i=0;i<n;i++)
   {
     cin>>a[i];
   }
  int q;
  cin>>q;
  while(q--)
  {
    int l,r,sum=0;
     cin>>l;
     cin>>r;
     for(i=l-1;i<r;i++)
     {
       sum=sum+a[i];
     }
     cout<<sum<<"\n";
  }
}
