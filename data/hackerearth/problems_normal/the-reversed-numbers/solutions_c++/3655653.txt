#include <iostream>
#include <math.h>
using namespace std;

// Flip both numbers, add and flip again 
int flip(int num) {
	int flipped = 0, i = 0, rem;
	while(num > 0) {
		rem = num % 10;
		num = num / 10;
		flipped = flipped*10 + rem;
		i++;
	}

	return flipped;
}

int main()
{
	int T, num1, num2, i;
	scanf("%d", &T);
	while(T--) {
		scanf("%d %d\n", &num1, &num2);
		num1 = flip(num1);
		num2 = flip(num2);
		printf("%d\n",flip((num1+num2)));
	}
    return 0;
}
