

#include <iostream>
#include <cmath>

int main()
{
	int t;
	std::cin>>t;
	while(t--)
	{
		long long n;
		std::cin>>n;

		int digits=0;

		long long m=n;

		while(m)
		{
			m/=10;
			digits++;
		}
		
		long long ans=0;
		m=1;

		for(int i=1 ; i<digits ; ++i)
		{
			ans+=9*m*i;
			m*=10;
		}
		ans+=(n-(pow(10,digits-1))+1)*digits;
		std::cout<<ans<<"\n";
	}
}
