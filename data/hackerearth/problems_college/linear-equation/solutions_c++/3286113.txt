#include <stdio.h>

int main()
{
    long long int t, i, n, sum ;
    scanf("%lld", &t);
    while(t--){
    	scanf("%lld",&n);
    	long long arr[n];
    	sum = 0;
    	for(i = 0; i < n; i++)
    	{
    		scanf("%lld", &arr[i]);
    		sum += arr[i];
    	}
    	sum /= n-1;
    	for(i = 0; i < n; i++)
    		printf("%lld ", sum - arr[i]);
    	printf("\n");
    	
    }
    return 0;
}
