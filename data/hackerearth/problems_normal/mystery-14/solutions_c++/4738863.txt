#include<iostream>
#include<vector>
using namespace std;
int main(){
    int n,val,cases,sum = 0;
    vector<int> v;
    cin>>cases;
    while(cases--){
        sum = 0;
        cin>>n;
        while(n--){
            cin>>val;
        v.push_back(val);
        }
        for(vector<int>::iterator it = v.begin(); it != v.end(); it++){
            if((*it%2)!=0){
                sum += (*it);
            }   }
            v.clear();
    cout<<sum<<"\n";    
        }
    }
