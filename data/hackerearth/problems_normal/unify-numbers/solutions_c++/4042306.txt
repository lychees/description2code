#include <iostream>
#include <algorithm>
using namespace std;

long mod(int a) {
    return a < 0 ? -a : a;
}

int main()
{
	int t;
	cin>>t;
	for (; t > 0; t--) {
            int n;
            cin>>n;
            int a[n];
            for (int i = 0; i < n; i++) {
                cin>>a[i];
            }
  			sort(a, a+n);
            int mean = a[n >> 1];
            long sum = 0;
            for (int i = 0; i < n; i++) {
                sum += mod(mean - a[i]);
            }
            cout<<sum<<endl;
        }
    return 0;
}
