#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int t;
    unsigned long long int prod,a,b,n;
scanf("%lld",&t);
while(t>0)
{
    t--;
    scanf("%lld",&n);
    b=n;
    a=2;
    prod=1;
    while(1)
        {
            if(b%2==1)
                prod=(prod*a)%1000000007;
            b/=2;
            if(b==0)
                break;
            a=(a*a)%1000000007;
        }
        printf("%llu\n",prod-1);
}
}
