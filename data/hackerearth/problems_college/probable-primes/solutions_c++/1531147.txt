#include <iostream>
using namespace std;
#define MX 1000005
bool prime[1000005]={0};
int ans[1000005]={0};
void calc(){
for(int i=2;i<MX;i++){
	if(prime[i]==0){
		for(int j=2*i;j<MX;j+=i){
			prime[j]=1;
		}
	}
}
}
int main()
{
	calc();
	int T;
	for(int i=2;i<MX-3;i++){
		if(prime[i]==0){
			ans[i]=ans[i-1]+1;
		}
		else ans[i]=ans[i-1];
	}
	cin>>T;
	while(T--){
	int L,R;
	cin>>L>>R;
	float s=ans[R]-ans[L-1];
	s=s/(R-L+1);
	printf("%.6f\n",s);
	}
    return 0;
}
