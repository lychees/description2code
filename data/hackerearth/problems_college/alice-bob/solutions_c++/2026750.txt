#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int MOD = 1e9+7;
const double ESP = 1e-9;

int main(void)
{
    // double timeStart = clock();
    //freopen("input.txt","r",stdin);
    ios::sync_with_stdio(false);
    vector<ll>data;
    set<ll>s;
    set<ll>::iterator k;
    s.insert(0);
    ll i,j,n;
    cin >> n;
    for(i = 0; i < n; i++)
    {
        cin >> n;
        data.push_back(n);
        s.insert(n);
    }
    bool chk = true;
    while(true)
    {
        n = data.size();
        bool flag2 = false;
        for(i = 0; i < n; i++)
        {
            for(j = 0; j < n; j++)
            {
                if(i == j)
                    continue;
                ll res = abs(data[i]-data[j]);
                if(not binary_search(s.begin(),s.end(),res))
                {
                    data.push_back(res);
                    s.insert(res);
                    flag2 = true;
                    break;
                }
            }
            if(flag2)
                break;
        }
        chk = not chk;
        if(not flag2)
            break;
    }
    chk == 1 ? cout << "Alice" : cout << "Bob";
    // double timeEnd = clock();
    // cout << (timeEnd - timeStart)/(CLOCKS_PER_SEC) << " seconds" << '\n';
    return 0;
}
