n = int(raw_input())
call = range(n)
ideal = range(n)
call = map(int, raw_input().split())
ideal = map(int, raw_input().split())
ans = n
cur = 0
req = 0
for i in ideal:
    req = call.index(i)
    if cur < req:
        ans += (req-cur)
    elif cur > req:
        ans += (len(call)-cur+req)
    cur = req
    call.remove(i)
    if cur == len(call):
        cur = 0
print ans
