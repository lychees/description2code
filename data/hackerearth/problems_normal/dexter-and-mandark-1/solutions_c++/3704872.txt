using namespace std;
#define MOD 1000000007
long long dp[2][1111];
int main(void){
	//ifstream cin("test.txt");
	std::ios::sync_with_stdio(false);cin.tie(0);
	int T;
	int i,j,n,m;
	int flag;
	int re;
	cin>>T;
	while(T--)
	{
		cin>>n>>m;
		for(i=0;i<=n;i++)
			dp[0][i]=0;
		dp[0][0]=1;
		flag=0;
		for(i=0;i<n;i++)
		{
			flag=1-flag;
			for(j=0;j<=n;j++)
			{
				dp[flag][j]=0;
				if(j>=1)
					dp[flag][j]=(dp[flag][j]+dp[1-flag][j-1])%MOD;
				if(j+m<=n)
					dp[flag][j]=(dp[flag][j]+dp[1-flag][j+m])%MOD;
			}
		}
		re=0;
		for(i=0;i<=n;i++)
		{
			//cout<<i<<' '<<dp[flag][i]<<endl;
			re=(re+dp[flag][i])%MOD;
		}
		if(m==1)
			re=(re-dp[flag][0]+MOD)%MOD;
		cout<<re<<endl;
	}
//	system("pause");
	return 0;
}
