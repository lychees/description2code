#include <iostream>
using namespace std;

int main()
{int n,m,t,i,j,s;
cin>>t;
while(t--){
	s=0;
	cin>>n;
	cin>>m;
	char a[n][m];
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>a[i][j];
			for(i=0;i<n;i++)
				for(j=0;j<m-1;j++)
					if((a[i][j]=='/')&&(a[i][j+1]=='\\')&&(a[i+1][j]=='\\')&&(a[i+1][j+1]=='/'))
					s++;	
	cout<<s<<endl;
}
return 0;
}
