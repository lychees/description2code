#include <iostream>
using namespace std;


int gcd(int firstNumber, int secondNumber)
{
	if(secondNumber == 0)
		return firstNumber;
	else
		return gcd(secondNumber, firstNumber % secondNumber);
}

int main()
{
    int maxNumber = 0, kidsCount = 0, kidNumber = 0, friendsCount = 0;
    
    cin >> maxNumber >> kidsCount;
    
    for(int index = 0; index < kidsCount; index++)
    {
    	cin >> kidNumber;
    	if(gcd(maxNumber, kidNumber) > 1)    	
    		friendsCount++;
    }
    
    cout << friendsCount;
    
    return 0;
}
