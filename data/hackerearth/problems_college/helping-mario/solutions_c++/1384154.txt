int gcd(int a, int b)
{
	if(!b)
		return a;
	return gcd(b, a % b);
}
ll mp[55][100010];
int arr[55], l, r, n, vis[55][100010], tc;
ll find(int idx, int g) {
	if(vis[idx][g] == tc)
		return mp[idx][g];
	if (idx == n) {
		if (g >= l && g <= r)
			return 1;
		else
			return 0;
	}
	vis[idx][g] = tc;
	mp[idx][g] = find (idx + 1, g) + find (idx + 1, gcd (g, arr[idx]));
	return mp[idx][g];
}
