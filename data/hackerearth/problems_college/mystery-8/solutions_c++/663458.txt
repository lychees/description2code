#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int t,i,l,j,ans;
    char s[105];
    scanf("%d",&t);
    while(t-->=0)
    {
    	gets(s);
    	l=strlen(s);
    	i=0;
    	j=l-1;
    	while(i<j)
    	{
    		if(!isalpha(s[i])&&s[i]!='*')
    		{
    			i++;
    			continue;
    		}
    		if(!isalpha(s[j])&&s[j]!='*')
    		{
    		    j--;
    			continue;
    		}
    		if(isalpha(s[i])&&s[j]=='*')
    		{ 
    			ans=s[i];
    			break;
    		}
    		if(isalpha(s[j])&&s[i]=='*')
    		{
    			ans=s[j];
    			break;
    		}
    		i++;
    		j--;
    	}
    	printf("%c\n",ans);
    }
    return 0;
}
