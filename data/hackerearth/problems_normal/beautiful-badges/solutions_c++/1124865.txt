#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define REP(i,b) for(int i=0;i<(b);i++)

using namespace std;

int N;
vector<int> d1, d2;

int main(int argc, char *argv[]) {
  scanf("%d", &N);
  REP(i, N) {
    int c, d;
    scanf("%d%d", &c, &d);
    if (c == 1) d1.push_back(d);
    else        d2.push_back(d);
  }
  sort(d1.begin(), d1.end());
  sort(d2.begin(), d2.end());
  int ans = 0;
  while (d1.size() && d2.size()) {
    int D1 = d1[d1.size() - 1], D2 = d2[d2.size() - 1];
    if (D1 > D2) {
      if (D2 * D2 * 2 > D1 * D1) { ans++; d2.pop_back(); }
      d1.pop_back();
    }
    else {
      if (D1 * D1 * 2 > D2 * D2) { ans++; d1.pop_back(); }
      d2.pop_back();
    }
  }
  printf("%d\n", ans);

  return 0;
}
