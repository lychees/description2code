#include<bits/stdc++.h> 
using namespace std; 
int main(void) {
	int T,x;	//number of test cases		 
	scanf("%d",&T);	
	x=T;
	while(T--) {
		int n,p,u,r,s;
		scanf("%d%d%d%d%d",&n,&p,&u,&r,&s);
		int temp=u-p;
		int temp1=temp/r;
		if(temp1>=n)
			printf("Case #%d: %d\n",x-T,p+r*n);
		else{
			int temp2=n-temp1;	//remaining turn
			int sum=p+r*temp1;
			while(temp2!=0){
				if(sum+r>u && sum-s>=0){
					sum-=s;
					temp2--;
				}
				else if(sum+r<=u){
					sum+=r;
					temp2--;
				}
				else
					break;
				
			}
			printf("Case #%d: %d\n",x-T,sum);
		}

		}
	return 0;
	}
