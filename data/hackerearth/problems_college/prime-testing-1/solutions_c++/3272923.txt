#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

bool isPrime[1005];

void sieve() {
    
    isPrime[0] = false;
    isPrime[1] = false;
    for(int i = 2; i * i <= 1000; ++i) {
         if(isPrime[i] == true) {
             
             for(int j = i * i; j <= 1000 ;j += i)
                 isPrime[j] = false;
        }
    }
}

int main() {
	// your code goes here
	
	memset(isPrime,true,sizeof isPrime);
	sieve();
	int t,n,a;
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&n);
		int count = 0;
		for(int i=0;i<n;i++){
			scanf("%d",&a);
			if(isPrime[a])
				count++;
		}
		printf("%d\n",count);
	}
	
	return 0;
}
