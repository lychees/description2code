#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,i;
		cin>>n>>m;
	
		while(n--)
		{ float ans=100,m1=0;
		 for(i=0;i<m;i++)	
		 {
		  float a1,a2,a3;
		  cin>>a1>>a2>>a3;//cout<<ans<<"\n";
		  float dis=100;
		  dis=dis*(1-a1/100)*(1-a2/100)*(1-a3/100);
		 // cout<<"dis"<<dis<<"\n";
		 if(dis<ans){ans=dis;m1=i+1;}
		 }
		 cout<<m1<<" ";
		}
		cout<<endl;
	}
}
