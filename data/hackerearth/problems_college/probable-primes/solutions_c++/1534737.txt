#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdio>
#define num 1000001
typedef long long ll;
using namespace std;

bool m[num];
int cnt[num]={0};
void init()
{
register ll i,j;
m[1]=false;
cnt[0]=0;
cnt[1]=0;

memset(m,true,num);
for(i=2;i<num;i++)
{
if(m[i]==true)
{
cnt[i]=cnt[i-1]+1;
for(j=2*i;j<num;j+=i)
{
m[j]=false;

}
}
else
cnt[i]=cnt[i-1];
}

}

int main()
{
init();
long long int t,l,r;
float ans;
scanf("%lld",&t);
while(t--)
{
scanf("%lld %lld",&l,&r);

ans=float(cnt[r]-cnt[l-1])/float(r-l+1);

printf("%.6f\n",ans);
}

}
