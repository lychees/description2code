#include <bits/stdc++.h>
#define ll long long
using namespace std ;
const ll mod = 10000007 ;
map<ll,ll> A,B ;
int C[2050] ,D[2050];
int main()
{
    ll X ;
    int N ; scanf("%d %lld",&N,&X) ;
    ll temp ;
    for(int i=0;i<N;++i)
    {
        scanf("%lld",&temp) ;
        if(temp<=1024)
        {
            for(int j=0;j<=2048;++j) D[temp^j] += C[j] ;
            for(int j=0;j<=2050;++j) C[j] =(C[j] + D[j])%mod , D[j]=0;
            C[temp] = (C[temp] + 1)%mod;
        }
        else
        {
            for(map<ll,ll>::iterator it=A.begin();it!=A.end();++it) B[it->first^temp] += it->second ;
            for(map<ll,ll>::iterator it=B.begin();it!=B.end();++it) A[it->first] += it->second ,A[it->first] %= mod;
            A[temp] += 1 ;
            B.clear() ;
        }
    }
    ll ans=0 ;
    if(X<=1024) ans += C[X] ; ans += A[X] ; ans %=mod ;
    for(map<ll,ll>::iterator it=A.begin();it!=A.end();++it) if((it->first ^ X) <=2048) ans =(ans + (ll)C[(it->first ^ X)]*(it->second) )%mod ;
    printf("%lld\n",ans) ;
    return 0 ;
}
