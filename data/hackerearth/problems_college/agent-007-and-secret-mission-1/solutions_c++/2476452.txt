#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int n = 0;
	cin >> n;
	cin.ignore ();
    string s;
    int N;
    int total;
	while (n--)
	{
		cin.ignore ();
		getline (cin, s);
		cin >> N;
		
		total = (N *(N + 1))/2;
		
		if (total % 2 == 0)
		{
			 cout << std::count(s.begin(), s.end(), '1') << endl;
		}
		else
		{
			cout << std::count(s.begin(), s.end(), '0') << endl;
		}
		
	}
}
