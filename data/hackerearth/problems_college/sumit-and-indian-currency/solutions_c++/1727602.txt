#include <cstdio>
#include <cstring>

#define MAX 100010

int t;
char s[MAX];
int n;

int main(){
    scanf("%d", &t);
    while (t--){
        scanf("%s", s);
        n = strlen(s);
        for (int i = 0; i < n; i++){
            printf("%c", s[i]);
            if (n - i > 2 && (n - i - 1) & 1)
                printf(",");
        }
        printf("\n");
    }
    return 0;
}
