instructionString = raw_input()
visited_points = {}

curX, curY = 0,0

def changedCoordinates(x,y,ins):
	if(ins == 'L'):
		return x,y-1
	elif(ins == 'R'):
		return x,y+1
	elif(ins == 'U'):
		return x-1,y
	else:
		return x+1,y
	
fall_count = 0
coordinatedKey = str(curX) + " " + str(curY)
visited_points[coordinatedKey] = 1

for ins in instructionString:
	curX, curY = changedCoordinates(curX, curY,ins);
	coordinatedKey = str(curX) + " " + str(curY)
	if(visited_points.has_key(coordinatedKey) == True):
		fall_count += 1
	else:
		visited_points[coordinatedKey] = 1
print fall_count
