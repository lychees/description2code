#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases>0)
    {
        char str[101];
        gets(str);
        // your code here
        char pqr[101];
        int i,j;
        int pos;
        for(i=0,j=0;str[i];i++)
        	{
        	if(str[i]=='*')
        		{
        		pqr[j++]='*';
        		pos=j-1;
        		}
        	else if(str[i]<='z' && str[i]>='a')
        		pqr[j++]=str[i];
        	}
        printf("%c\n",pqr[j-pos-1]);
        test_cases--;
    }
    return 0;
}
