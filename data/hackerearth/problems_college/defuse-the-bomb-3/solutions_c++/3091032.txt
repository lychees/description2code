#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T=0;
    cin >> T;
    for(int t = 0; t < T; t++)
    {
    	long long int A,B;
    	cin >> A >> B;
    	int l_B = 0, t_B = B, l_A = 0, t_A = A;
    	int c_A[10];
    	while(t_A > 0)
    	{
    		c_A[l_A++] = t_A % 10;
    		t_A /= 10;
    	}    	
    	while(t_B > 0)
    	{
    		t_B/=10;
    		l_B++;
    	}
    	for(int i = l_A - l_B; i >= 0 ; i--)
    	{
    		
    		int temp = 0;
			for(int j = 0; j < l_B; j++)
			{
				temp = temp + c_A[i+j] * pow(10,j);
			}
    		cout << temp*B << endl;
    	}
    }
    return 0;
}
