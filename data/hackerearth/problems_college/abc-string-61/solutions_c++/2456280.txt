#include<bits/stdc++.h>
using namespace std;

long long MOD=1e9+7;

long long mulmod(long long a,long long b)
{
	if(b==0)
	 return 1;
	if(b==1)
	 return a;
	long long c=mulmod(a,b/2);
	c=(c*c)%MOD;
	if(b%2)
		c=(c*a)%MOD;
	return c;
}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		long long ans=mulmod(3,n)-3*mulmod(2,n)+3;
		ans%=MOD;
		if(ans<0)
		 ans+=MOD;
		cout<<ans<<endl;
		
	}
}
