#include <iostream>
#include <vector>
#define N 1005

using namespace std;

int main()
{
    vector<int> a[N];
    int c, d;
    
    a[0].push_back(1);
    a[1].push_back(1);
    a[2].push_back(2);
    a[3].push_back(6);
    
    for (int i = 4; i < N; i++) {
    	c = 0;
    	d = i;
    	for (int j = 0; j < a[i-1].size(); j++) {
    		d = (a[i-1][j]*i + c)%10;
    		c = (a[i-1][j]*i + c)/10;
    		a[i].push_back(d);
    	}
    	while(c) {
    		a[i].push_back(c%10);
    		c /= 10;
    	}
    }
    
   	int t;
   	int n;
   	
   	scanf("%d", &t);
   	int sum;
   	while (t--) {
   		scanf("%d", &n);
   		sum = 0;
   		for (int i = 0; i < a[n].size(); i++) sum += a[n][i];
   		printf("%d\n", sum);
   	}
   
    return 0;
}
