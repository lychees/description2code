#include <iostream>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--){
    	int n;
    	cin>>n;
    	if(n%3 == 0){
    		for(int i = 0;i < n;i++)
    			cout<<5;
    		cout<<endl;
    	}
    	else{
    		int initial = (n/3)*3;
    		int rem = n - initial;
    		while(initial >= 0){
    			if(rem%5 == 0){
    				for(int i = 0;i < initial;i++)
    					cout<<5;
    				for(int i = 0;i < rem;i++)
    					cout<<3;
    				cout<<endl;
    				break;
    			}
    			else{
	    			initial -= 3;
    				rem = n - initial;
    			}
    		}
    		if(initial < 0)
    			cout<<-1<<endl;
    	}
    }
    return 0;
}
