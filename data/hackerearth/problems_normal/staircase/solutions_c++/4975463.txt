#include <iostream>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    long long int n,i;
    cin>>n;
    long long int dp[n+1];
    dp[0]=0;
    dp[1]=1;
    dp[2]=2;
    dp[3]=4;
    if(n<0)
    cout<<"ERROR"<<endl;
    else if(n>3){
    	for(i=4;i<n+1;i++){
    		dp[i]=dp[i-1]+dp[i-2]+dp[i-3];
    	}
    	cout << "Jack-" <<dp[n]<< endl;
    	for(i=3;i<n+1;i++){
    		dp[i]=dp[i-1]+dp[i-2];
    	}
    	cout << "Jill-" <<dp[n]<< endl;
    }
    else{
    	cout << "Jack-" <<dp[n]<< endl;
    	cout << "Jill-" <<dp[n]-1<< endl;
    }
    return 0;
}
