char game(int x, int y){
    if(x==0 and y==0) return 'P';
    if(x==0 or y==0) return 'N';
    if(x==y) return 'N';
    rep(i,1,x+1) 
        if(game(x-i,y)=='P') return 'N';
    rep(i,1,y+1)
        if(game(x,y-i)=='P') return 'N';
    rep(i,1,min(x,y)+1)
        if(game(x-i, y-i)=='P') return 'N';
    return 'P';
}

const int MAX = 1e6;
vi M(MAX+5);
vi seen(MAX+5, -1);
void build(){
    M[0] = 0;
    seen[0] = 1;
    int prev = 1;
    rep(i,0,MAX+1){
        if(seen[i]==1) continue;
        seen[i] = 1;
        M[i] = prev+i;
        
        if(prev+i<MAX){
            seen[prev+i] = 1;
        }
        //if(i<10) cout<<i<<" "<<prev+i<<endl;
        prev++;
    } 
}

int main(){
    int tt;s(tt);
    int x, y;
    build();
    while(tt--){
        cin>>x>>y;
        if(M[min(x,y)]==max(x, y)) cout<<"Don't Play"<<endl;
        else cout<<"Play"<<endl;
    }
}
