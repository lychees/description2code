#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
  ll t,val,sum,i,n,prev;
  cin>>t;
  while(t--)
  {
  	cin>>n;
  	sum=val=2;
  	i=1;
  	prev=0;
  	while(sum<n)
  	{
  		prev=sum;
  		i++;
  		val=val<<1;
  		sum+=(i)*(val);
  	}
  	cout<<min(2*i-1,n-prev)<<endl;
  }	
}
