#include <bits/stdc++.h>
#define ll long long
using namespace std;
int h[100010],a[100010];//home & away kit color
pair<int,int> p[100010];
int main()
{
	memset(a,0,sizeof(a));
	memset(h,0,sizeof(h));
	int i,n;//No of teams
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&(p[i].first),&(p[i].second));
		h[(p[i].first)]++;
		a[(p[i].second)]++;
	}
	int x,y;
	for(i=0;i<n;i++)
	{
		x=n-1;
		y=(n-1)-h[p[i].second];
		x+=h[p[i].second];
		printf("%d %d\n",x,y);
	}
	return 0;
}
