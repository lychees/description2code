#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll int a[100010];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int n,k;
	cin>>n>>k;
	ll int i;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n);
	ll int ans=INT_MAX;
	for(i=0;i<n-k+1;i++)
	{
		ans=min(ans,a[i+k-1]-a[i]);
	}
	cout<<ans<<endl;
	return 0;
}
