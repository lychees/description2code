#include<bits/stdc++.h>
using namespace std;
long long int dp[1000001];
long long int mod (pow(10,9) + 7);

long long int find(long long int n)
{
if(dp[n] != -1)
return dp[n];
return dp[n] = find(n-1)%mod + (n-1)%mod*find(n-2)%mod;
}

int main()
{
	memset(dp,-1,sizeof(dp));
	dp[0] = 1;
	dp[1] = 1;
	int t;
	cin >> t;
	while(t--)
	{  
		long long int n;
		cin >> n;
		long long int ways = find(n);
		cout << ways%mod << endl;
	}
	return 0;
}
