#include <iostream>
#include<cmath>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int d[n],b[n];
    for(int i=0;i<n;i++){
    	cin>>d[i];
    }
    for(int i=0;i<n;i++){
    	cin>>b[i];
    }
    int max1=0,max2=0;
    for(int i=0;i<n-1;i++){
     int diff=abs(d[i+1]-d[i]);	
     if(max1<diff) max1= diff;
    }
    for(int i=0;i<n-1;i++){
     int diff=abs(b[i+1]-b[i]);	
     if(max2<diff) max2= diff;	
    }
    if(max1>max2) cout<<"Dom"<<endl<<max1;
    else if(max1<max2) cout<<"Brian"<<endl<<max2;
    else cout<<"Tie"<<endl<<max1;
    return 0;
}
