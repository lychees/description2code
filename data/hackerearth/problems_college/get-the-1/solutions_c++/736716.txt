#include <iostream>
#include <cstdio>
#include <vector>
#include <math.h>
#include <climits>
#include <algorithm>
#define ll long long
using namespace std;

ll f(ll x){
    ll sum = 0;
    while (x != 0){
        sum += (x%10)*(x%10);   
        x /= 10;
    }
    return sum;
}

int main(){
    int N; cin >> N;
    for (int i = 1; i <= N; i++){
        ll x; cin >> x;
        ll temp = f(x);    
        while(temp/10 != 0) temp = f(temp);
        if (temp == 1) cout << "Case #" << i  <<  ":YES\n";
        else cout << "Case #" << i << ":NO\n";
    }
}
