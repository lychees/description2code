#include <iostream>
#include <cstdio>
using namespace std;

int t;
int n, m;
int main()
{
    scanf("%d", &t);
    while(t--){
    	scanf("%d%d", &n, &m);
    	if(n > 60)puts("NA");
    	else {
    		int ans = 0;
    		if(n > 2){    			
    			ans += (50 + (n - 2) * 12);
    		}
    		else ans = 50;
    		ans += m * 2;
    		printf("%d\n", ans);
    	}
    	
    }
    return 0;
}
