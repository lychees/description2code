#include <iostream>
using namespace std;

int main()
{
    long long int n,t,x,f,i,j;
    int a[500001];
    for(i=0;i<500001;i++)
      a[i]=0;
    cin>>n>>t;
    a[1]=1;
    for(i=2;i<=500000;i++)
    {for(j=i+i;j<=500000;j+=i)
     {if(a[j]==0)
        a[j]=1;
     }
    }
    while(t--)
    {cin>>x;
     f=0;
     for(int i=1;i<=x;i++)
       if(a[i]==0)
          f++;
     cout<<f<<endl;
    }
     
    return 0;
}
