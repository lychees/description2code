import sys
import operator

date = [x for x in range(1,31)]
day_count = dict((d,0) for d in date)

N = int(raw_input())
for i in range(N):
	S = raw_input()
	days = [int(s) for s in S.split() if s.isdigit()]
	if S[0] == 'G':
		wt = 2
		for i in days:
			day_count[i]+=2
	else:
		wt = 1
		for i in days:
			day_count[i]+=1
	
day_decided = max(day_count.iteritems(), key=operator.itemgetter(1))[0]
if day_decided in [19,20]:
	print "Date"
else:
	print "No Date"
