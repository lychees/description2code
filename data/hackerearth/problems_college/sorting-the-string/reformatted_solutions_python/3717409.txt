testcases = int(raw_input())

for i in range(0, testcases):
    line = raw_input().split()
    while not line:
        line = raw_input().split()
        # else:
    string = line[0]
    order = int(line[1])
    finalstring = ""
    flag = 0
    # print "Printing Final String"
    for j in string:

        if j.isalpha() and j.isupper():
            finalstring += j
    # print finalstring

        if not string.isupper():
            flag = 1
    if order % 2 == 0:
        # sorting is even
        b = sorted(finalstring)
    else:
        # sorting is odd
        b = sorted(finalstring, reverse=True)

    if flag == 1:
        print "NULL"
    else:
        print ''.join(b)
