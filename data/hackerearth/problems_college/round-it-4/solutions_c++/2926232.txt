#include<stdio.h>
#include <iostream>
using namespace std;
int main()
{
    	   long result=0, N, B,T;
    	    scanf("%ld",&T);
    	    while(T)
    	    {
	     		result=0;
	     		scanf("%ld%ld",&N,&B);
	     		if(N>B)
	     		{     long i=1;
			 	 while(N>B*i)
		 		 i++;
		 		 if(N-(B*(i-1))<((B*i)-N))
		 		 result=(B*(i-1));
		 		 else if(N-(B*(i-1))>((B*i)-N))
		 		 result=(B*i);
		 		 else
		 		 result=(B*i);
                  }
                 if(B%2!=0){
			   	 B++;
                  if(N==B/2)
                       result=B-1;
                       }
                       if(N==B/2)
                       result=B;
                      
                  printf("%d\n",result);	 
			T--;
	    }
    	    return 0;
}
