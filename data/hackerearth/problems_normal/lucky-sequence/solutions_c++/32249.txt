#include <iostream>
#include <cstdio>
using namespace std;

typedef long long ll;
typedef unsigned long long ull;
int main()
{
    ll n,a;
    int cur = 0;
    char ans[68];
    scanf("%ld",&n);
    while(n--){
        scanf("%ld",&a);
        a += 2;
        ull i = 0;
        while((1ull<<i) < a){
            ++i;    
        }
        --a;
        --i;
        int temp = i;
        for(;i>0;--i){
            ans[cur++] = (a>>(i-1))&1?'7':'4';
        }
        for(int lol = 0;lol < temp; ++lol)
        printf("%c",ans[lol]);
        ans[cur] = '\0';
        if(n) printf("\n");
        cur = 0;
    }
    return 0;
}
