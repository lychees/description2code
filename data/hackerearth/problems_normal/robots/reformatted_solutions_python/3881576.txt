import collections
import math


def numberOfWaysToMark():
    N = int(raw_input())
    maxList = sorted([int(v) for v in raw_input().strip().split()])
    ways = 1
    for x in range(N):
        if maxList[x] < x+1:
            return 0
        else:
            ways *= (maxList[x]-x)
    return ways % 1000000007


for tc in range(int(raw_input())):
    print("%d" % (numberOfWaysToMark()))
