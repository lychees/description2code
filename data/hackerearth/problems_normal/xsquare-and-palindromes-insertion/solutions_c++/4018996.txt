#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string s;
        cin>>s;
        map<char,int> mp;
        map<char,int> :: iterator it;
        mp.clear();
        for(int i=0;i<s.length();i++)
        {
            char ch=s[i];
            it=mp.find(ch);
            if(it==mp.end())
            {
                mp[ch]=1;
            }
            else
            {
                it->second++;
            }
        }
        /*for(it=mp.begin();it!=mp.end();it++)
        {
            cout<<"mp["<<it->first<<"]="<<it->second<<"\n";
        }*/
        if(mp.size()==1)
            printf("0\n");
        else
        {
            int count=0;
            bool flag=true;
            for(it=mp.begin();it!=mp.end();it++)
            {
                if(it->second%2!=0)
                {
                    if(flag==true)
                    flag=false;
                    else
                    count++;
                }
            }
            printf("%d\n",count);
        }
    }
    return 0;
}
