#include<iostream>
#include<string.h>
using namespace std;
char arr[101];
char check[27] = "qwertyuiopasdfghjklzxcvbnm";
int main()
{

	int t;
	cin >> t;
	getchar();
	while (t--)
	{
		gets(arr);
		for (int i = 0; arr[i] != '\0'; i++)
		{
			if (isalpha(arr[i]))
				cout << check[arr[i] - 'a'];
			else
				cout << arr[i];
		}
		cout << "\n";
	}
	return 0;
}
