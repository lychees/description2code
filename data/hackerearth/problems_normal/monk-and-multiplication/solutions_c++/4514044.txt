#include <stdio.h>
#include <queue>
#include <iostream>

using namespace std ; 

priority_queue<int , vector<int> , less<int> > q; 
int n , v ; 

int main(int argc, char **argv)
{
    
	cin >> n ; 
    for(int i = 1 ; i <= n ; ++i){
        cin >> v ; 
        q.push(v) ;        
        if(i <= 2) 
            cout << "-1" << endl ; 
        else {
            int x , y , z ; 
            x = q.top() ;
            q.pop() ;  
            y = q.top() ; 
            q.pop() ; 
            z = q.top() ;
            q.pop() ; 
            cout << (1LL * x * y * z) << endl ;
            q.push(x) ; 
            q.push(y) ; 
            q.push(z) ; 
        }
        
    }
    
	return 0;
}
