#include <cstdio>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    long long int test;
    long long int temp,ans;
    while(t--)
    {
    	ans=0;
    	scanf("%lld",&test);
    	for(int i=0;i<test;i++)
    	{
    		scanf("%lld",&temp);
    		ans+=temp;
    	}
    	ans-=test;
    	ans++;
    	printf("%lld\n",ans);
    }
    return 0;
}
