n = input()
for i in range(n):
    l = []
    # l.append("$")
    temp = raw_input()
    ind = 0
    for i in range(len(temp)):
        if temp[i] == "$":
            ind = i
            break
    for i in range(ind, len(temp), 1):
        if ord(temp[i]) >= 48 and ord(temp[i]) <= 57:
            l.append(temp[i])
        if ord(temp[i]) >= 58:
            break
    print "$"+str(int("".join(l)))
