#include <iostream>
#include<cstdio>
using namespace std;
#define mod 1000000007
int dp[1005][1005];
int main()
{
    int t,m,n,i,j;
    long long int ans;
    scanf("%d",&t);
    while(t--)
    {
    	ans=0;
    	scanf("%d %d",&n,&m);
    	for(i=0;i<=n;i++)
    	{
    		for(j=0;j<=n;j++)
    		{
    			dp[i][j]=0;
    		}
    	}
    	//dp[0][0]=1;
    	for(i=1;i<=n;i++)
    	{
    		dp[i][0]=1;
    		dp[0][i]=0;
    	}
    	for(i=1;i<=n;i++)
    	{
    		for(j=1;j<=n;j++)
    		{
    			if(i>=(j*m))
    			{
    				dp[i][j]=(dp[i-1][j]+dp[i][j-1])%mod;
    			}
    		}
    	}
    	/*for(i=0;i<=n;i++)
    	{
    		for(j=0;j<=n;j++)
    		{
    			printf("%d ",dp[i][j]);
    		}
    		printf("\n");
    	}*/
    	for(i=0;i<=n && n-i>=(i*m);i++)
    	{
    		if(n-i>i)
    		  ans=(ans+dp[n-i][i])%mod;
    	}
    	
    	printf("%lld\n",ans);	
    }
    return 0;
}
