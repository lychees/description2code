#include <iostream>
using namespace std;
 
int main()
{
    int t, n = 1;
    cin >> t;
    while(t--)
    {
    	int r,c,w;
    	cin >> r>>c>>w;
    	int sum = w;
    	if(c == w)
    	{
    		cout << "Case #"<<n<<": "<< r*sum<<endl;
    	}
    	else if (w >= (c/2))
    	{
    		cout << "Case #"<<n<<": "<< (sum+1)*r<<endl;
    	}
    	else
    	{
    		sum += c-w-w+1;
    		cout << "Case #"<<n<<": "<< sum*r<<endl;
    	}
    	n++;
    }
    return 0;
}
