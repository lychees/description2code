#include<iostream>
#include<stdio.h>
using namespace std;

int a[105][5005];

int BinarySearch(int target,int i, int low, int high)
{
   while (low != high) {
    int mid = (low + high)/2;
    if (a[i][mid] <= target) {
        low = mid + 1;
    }
    else {
        high = mid;
    }
}
return high;
}

int main()
{
    int n,m,q;
   //n=readInt();m=readInt();q=readInt();
    cin>>n>>m>>q;
    for(int i=0;i<n;i++)
    {
    for(int j=0;j<m;j++)
        cin>>a[i][j];
    }
    int ans=100000;
    int num;
    int j;
    while(q--)
    {
    for(int i=0;i<n;i++)
    {
        cin>>num;
        j=BinarySearch(num,i,0,m-1);
        while(num >=a[i][j] && j<m)
            j++;
        if(j<ans)
            ans=j;
    }
    printf("%d\n",ans);
    ans=100000;
    }
    return 0;
}
