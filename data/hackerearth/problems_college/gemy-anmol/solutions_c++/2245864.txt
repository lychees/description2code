#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{   int t;
    long m,n,sum,c;
    long a[7];
    cin>>t;
    while(t--)
    {   
    	cin>>n>>m;
    	int flag=1;
    	for(int i=0;i<n;i++)
    	  cin>>a[i];
    	
    	sort(a,a+n);
        sum=0,c=0;
    	for(int i=n-1;i>=0&&flag;i--)
    	{ sum+=a[i];
    	  if(sum>=m)
    	    flag=0;
    	  c++;
    	}
        if(flag)
          cout<<-1<<endl;
        else
          cout<<c<<endl;
    }
   
    return 0;
}
