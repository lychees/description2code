#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
    	cin>>a[i];
    }
    int tc;
    cin>>tc;
    while(tc--){
    	int x,y,val,count=0;
    	cin>>x>>y>>val;
    	for( ;x<=y;x++){
    		if(a[x]==val)
    			count++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
