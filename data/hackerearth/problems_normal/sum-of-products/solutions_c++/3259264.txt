#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007LL
#define ll long long

int main()
{
	cin.tie(0);
	ios::sync_with_stdio(false);
	int t,n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		ll res = 0;
		for(int i=1;i<=n;)
		{
			int val, nxt;
			val = n/i;
			nxt = n/val;
			res += 1LL*(nxt-i+1)*(i+nxt)*val/2;
			res %= MOD;
			i = nxt+1;
		}
		cout << res << endl;
	}
	return 0;
}
