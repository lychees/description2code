    int t;

    int sum=0;;

bool isprime(int n)
{
	if(n==2)
	return true;
	
	if(n==1)
	return false;
	
	if(n==0)
	return false;
	
	for(int i=2;i<sqrt(n)+1;i++)
	{
		if(n%i==0)
		return false;
	}
	
	return true;
}
