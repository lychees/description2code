#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		int count=0;
		while(n!=1)
		{
			if(n%3==0)
			{
				while(n%3==0)
				{
					count++;
					n=n/3;
				}
			}
			if(n==1)
					break;
			if(n%2==0)
			{
				while(n%2==0)
				{
					count++;
					n=n/2;
				}
			}
			if(n==1)
					break;
			if(n%2!=0)
			{
				n=n-1;
				count++;
			}
		}
		cout<<count<<endl;
		
	}
}
