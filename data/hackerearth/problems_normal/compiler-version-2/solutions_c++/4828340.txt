#include<iostream>
using namespace std;

int main()
{
    string s,c;
    while(getline(cin,s))
    {
        c="";
        int l=s.size();
        for(int i=0;i<l;)
        {
            if(i+1<l && s[i]=='/' and s[i+1]=='/')
            {
                for(int j=i;j<l;j++)c+=s[j];
                break;
            }
            else if(i+1<l and s[i]=='-' && s[i+1]=='>')
            {
                c+='.';
                i+=2;
            }
            else {c+=s[i];i++;}
        }
        cout<<c<<endl;
    }
    return 0;
}
