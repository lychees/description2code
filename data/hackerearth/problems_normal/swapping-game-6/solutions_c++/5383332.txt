#include <iostream>
#include <cstring>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;

int main()
{
   int n,i,k,j,flag=0,x;
   string s,s1,s2,s3,s4;
   cin>>k>>s;
   n = s.length();
   s3 =s;
   s4 = s;
   x =k;
   while(k--)
   {
   		s.clear();
   		s = s3;
   		s1.clear();
   		for(i=0;i<n;i=i+2)
   		s1.push_back(s[i]);
   		s2.clear();
   		for(i=1;i<n;i=i+2)
   		s2.push_back(s[i]);
   		reverse(s2.begin(),s2.end());
   		s3.clear();
   		s3 = s1+s2;
   		if(s3.compare(s4)==0)
   		{
   			j = x-k;
            //cout<<j<<endl;
   			break;
   		}
   		else if(k==0)
            flag=1;
   }
   if(flag==1 || j==0)
    cout<<s3<<endl;
   else{
   	int p =k%j;
   while(p--)
   	{
   		s.clear();
   		s = s3;
   		s1.clear();
   		for(i=0;i<n;i=i+2)
   		s1.push_back(s[i]);
   		s2.clear();
   		for(i=1;i<n;i=i+2)
   		s2.push_back(s[i]);
   		reverse(s2.begin(),s2.end());
   		s3.clear();
   		s3 = s1+s2;
	 }
   	cout<<s3<<endl;
   }
   return 0;
}
