#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
	ios::sync_with_stdio(false);
	ll t;cin>>t;
	while(t--){
		ll n, k;cin>>n>>k;
		ll a[100100]={0};
		for(int i=0; i<n; i++)cin>>a[i];
		sort(a, a+n);
		ll sum=0;
		for(int i=n-1; i>=max(n-k,0LL); i--)
			sum+=a[i];
		cout<<sum<<endl;
	}
	return 0;
}
