#include <iostream>
#include<map>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int l[n];
    	for(int i=0;i<n;i++)
    	cin>>l[i];
    	map<int ,int>mp;
    	for(int i=0;i<n;i++)
    	mp[l[i]]++;
    	int count=0;
    	for(map<int,int>::iterator iter = mp.begin(); 
                                        iter != mp.end(); 
                                        ++iter)
    	{
    		if(iter->second>count)
    		count=iter->second;
    	}
    	cout<<count<<endl;
    }
}
