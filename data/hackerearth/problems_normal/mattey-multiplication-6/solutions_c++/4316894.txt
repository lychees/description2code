#include <iostream>
#include<vector>
using namespace std;
vector<int> res;
int main()
{	
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		long long int n,m;
		cin>>n>>m;
	
		
	//	cin >> n >> m;

        int bit = 0;
        res.clear();

        while(m != 0){

            if(m % 2 != 0){         // binary bit = 1

                res.push_back(bit);
            }

            m /= 2;                 // shift right
            bit++;                  
        }
		cout<<"("<<n<<"<<"<<res[res.size()-1]<<")";
       

        for(int i = res.size() - 2; i >= 0; i--){       //traverse in a reverse order
          
            cout<<" + "<<"("<<n<<"<<"<<res[i]<<")";
        }
		cout<<"\n";

	}
	return 0;
}
