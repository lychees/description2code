#include <bits/stdc++.h>

using namespace std;

const int N = 1000010;
const string win = "Play\n", loose = "Don't Play\n";
int n, t, a, b, s[N];
bool f[N];

void pre()
{
    int j = 0;
    for (int i = 0; i + j < N; i++) if (!f[i])
    {
        s[i] = i + j; s[i + j] = i;
        f[i] = f[i + j] = true;
        j++;
    }
}

void solve()
{
    cin >> a >> b;
    if (!f[a]) cout << win; else
    if (s[a] != b) cout << win; else
    cout << loose;
}

int main()
{
    pre();
    cin >> t;
    while (t--) solve();
    return 0;
}
