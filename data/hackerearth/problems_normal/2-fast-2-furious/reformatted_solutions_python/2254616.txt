dom = map(int, raw_input().split())
d = []
brian = map(int, raw_input().split())
b = []
for x in range(len(dom)-1):
    d.append(abs(dom[x+1]-dom[x]))
    b.append(abs(brian[x+1]-brian[x]))
if max(d) > max(b):
    print "Dom"
elif max(d) < max(b):
    print "Brian"
else:
    print "Tie"

print max(max(d), max(b))
