typedef long long ll;
//int dx[]= {-1,0,1,0};
//int dy[]= {0,1,0,-1};
#define MAX 100005
unsigned int phi[MAX];
void solve()
{
    unsigned int i,j;
    phi[1]=1;
    for(i=2;i<MAX;i++)
    {
        if(!phi[i])
        {
            phi[i]=i-1;
            for(j=(i<<1);j<MAX;j+=i)
            {
                if(!phi[j]) phi[j]=j;
                phi[j] = (phi[j]/i)*(i-1);
            }
        }
    }
}
int main()
{
	int t;
	solve();
	scanf("%d",&t);
	while(t--)
	{
        int a,b;
        scanf("%d %d",&a,&b);
        ll ans = 0;
        for(int i=1;i<=min(a,b);i++)
        {
            if(a%i==0 && b%i==0)
                ans+=phi[i];
        }
        printf("%lld\n",ans);
	}
return 0;
}
