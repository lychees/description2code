#include<iostream>
#include<cstdio>
#include<vector>
#include<utility>
#include<cmath>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--){
		int n, m, d;
		cin >> n>> m>> d;
		vector < pair<int, int> > v;
		for(int i=0; i<n; i++){
			int x, y;
			scanf("%d%d", &x, &y);
			v.push_back(make_pair(x, y));
		}
		vector < pair<int, int> > v2;
		for(int i=0; i<m; i++){
			int x, y;
			cin >> x >> y;
			v2.push_back(make_pair(x, y));
		}
		bool flag = false;
		int count = 0;
		for(int i=0; i<m; i++){
			for(int j=0; j<n; j++){
				int dis = abs(v2[i].first - v[j].first)+abs(v2[i].second - v[j].second);
				if(dis <= d){
					count ++;
					break;
				}
			}
		}
		if(count > (m/2)){
			cout << "YES\n";
		}else{
			cout << "NO\n";
		}
	}
	return 0;
}
