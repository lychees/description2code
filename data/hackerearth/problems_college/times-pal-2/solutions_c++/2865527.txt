    #include <bits/stdc++.h>
    using namespace std;
     
    int main()
    {
        int t=0;
        cin >> t;
        while(t--)
        {
        	string a,b;
        	cin>>a>>b;
        	int i=0,c=0;
        	for(i=0;i<=a.size();i++)
        	{
        		string s,s1;
        		s=s1=a;
        		s.insert(i,b);
        		s1.insert(i,b);
        		reverse(s1.begin(),s1.end());
        		if(s==s1)c++;
        	}
        	printf("%d\n",c);
        }
        return 0;
    }
