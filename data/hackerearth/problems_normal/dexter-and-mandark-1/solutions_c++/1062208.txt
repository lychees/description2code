ll MOD=(ll) 1e9+7;

int addmod(int a,int b)
{
	ll t=(ll)a+b;
	if (t>=MOD) t=t%MOD;
	return t;
}

int submod(int a,int b)
{
    //assuming both a and b are positive and <MOD
    int t=a-b;
    t=t<0?t+MOD:t;
    return t;
}

int mulmod(int a,int b)
{
	ll t=(ll)a*b;
	if (t>=MOD) t=t%MOD;
	return t;
}


int epowmod(int a,int p)
{
	if (p==0) return 1;
	int t=epowmod(a,p/2);
	t=mulmod(t,t);
	if (p&1) t=mulmod(t,a);
	return t;
}
