#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
long long power(long long a,long long n)
{
	if(n==0)
	return 1;
	long long p=power(a,n/2);
	if(n%2)
	return ((p * p)%mod * a)%mod;
	else
	return (p * p)%mod;
}
int main()
{
	ll fact[100001];
	fact[0]=1;
	for(int i=1;i<=100001;i++)
	fact[i]=(fact[i-1]*i)%mod;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int m,k,sum=0,a[100001];
		scanf("%d %d",&k,&m);
		for(int i=0;i<k;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		
			ll c=(fact[m-sum]*fact[k-1])%mod;
			ll b=fact[m-sum+k-1];
			ll d=(b * power(c,mod-2)%mod)%mod;
		printf("%lld\n",d);	
		
	}
}
