import fileinput
fin = fileinput.input()
K = int(fin.readline().strip())
# print K
coins = fin.readline().strip()
length = len(coins)
# print length
my_list = []
# print my_list
count = 0
# # print l.index("a")
for i in range(0, length):
    del my_list[:]
    for j in range(i, length):
        if coins[j] in my_list:
            if (len(my_list) < K):
        	    continue
            elif (len(my_list) > K):
                break
            else:    
        	    count = count + 1
        else:
            my_list.append(coins[j])
            if (len(my_list) < K):
        	    continue
            elif (len(my_list) > K):
                break
            else:    
        	    count = count + 1
