#include <iostream>
using namespace std;

int compare(const void* a, const void* b){
	return   *(int *)a - *(int *)b ;
}

int mini(int a[], int n, int start){
	int i=start+1, count=1;
	while(a[i]==a[start] && i<n){
		count++;
		i++;
	}
	return count;
}

int main()
{
    int n, a[1000];
    int i, count, numOfMins, index=0;
    cin>>n;
    for(i=0; i<n; i++)
    	cin>>a[i];
    
    count=n;
    qsort(a, n, sizeof(int), compare);
//    for(i=0; i<n; i++)
//    	cout<<a[i]<<"  ";
   // cout<<endl;
    while(count){
    	cout<<count<<endl;
    	int min=a[index];
    	numOfMins= mini(a,n, index);
    //	cout<<"numOfMins : "<<numOfMins<<endl;
    	index += numOfMins;
    //	cout<<"index : "<<index<<endl;
    	count = count- numOfMins;
    	for(i=index; i<n; i++)
    		a[i]= a[i]-min;
    }
    return 0;
}
