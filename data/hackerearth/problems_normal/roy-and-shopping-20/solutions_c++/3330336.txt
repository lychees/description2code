void primer()
{
	long int i,j;	
	for(i=2;i<=1000005;i++)
	{
		if(primes[i]==0)
		{
			for(j=i;j<=1000005;j+=i)
			{	
				if(primes[j] == 0)
				primes[j] = i;	
			}
		}
	}
}
int main()
{
	long int n,i,*r,j,*sum;
	cin >>n;
	primer();
	r = (long int *)malloc(n*sizeof(long int));
	sum = (long int *)malloc(n*sizeof(long int));
	for(i=0;i<n;i++)
	{	scanf("%ld",&r[i]);
		sum[i] = r[i]-primes[r[i]];
		printf("%ld\n",sum[i]);
	}
}
