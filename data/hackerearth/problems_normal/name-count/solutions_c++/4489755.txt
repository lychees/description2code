#define ll long long
ll mod=1e9+7;
ll fastPow(ll a,ll b,ll mod)
{
	if(b==0)
	return 1;
	if(b==1)
	return a;
	ll half=fastPow(a,b/2,mod);
	if(b%2==0)
	return ( half * half ) % mod;
	return (((half * half ) % mod )* a ) % mod;
}
ll mmi(ll n,ll m)
{
	return fastPow(n,m-2,mod);
}
