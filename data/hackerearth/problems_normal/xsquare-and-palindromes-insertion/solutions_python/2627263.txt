from collections import Counter
t=input()
while t>0:
	t-=1
	s=raw_input()
	s=list(s)
	c=Counter(s)
	l=[]
	f=0
	for key in c:
		l.append(c[key])
	for i in l:
		if i%2!=0:
			f+=1
	if f>1:
		print f-1
	else:
		print"0"
