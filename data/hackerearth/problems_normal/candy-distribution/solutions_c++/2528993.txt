typedef long long int ll;
typedef pair<int, int > pii;

#define MOD 1000000007
#define pb push_back

map<ll, ll> m;

ll power(ll a, ll b)
{
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a) % MOD;
		a = (a*a)%MOD;
		b >>= 1;
	}
	return res;
}

ll combos(ll n)
{
	if(n == 1 || n == 2)
		return 1;
	if(m.find(n) != m.end())
		return m[n];
	n--;
	ll sq = sqrt(n);
	ll ret = 0;
	for(int i=1;i<=sq;i++) {
		if(n%i == 0) {
			ll r = n/i;
			ret = (ret + power(combos(i), r)) % MOD;
			if(r*r != n && r != n)
				ret = (ret + power(combos(r), i)) % MOD;
		}
	}
	m[n+1] = ret;
	return ret;
}

void solve()

	cin>>n;
	cout<<combos(n)<<"\n";
}
