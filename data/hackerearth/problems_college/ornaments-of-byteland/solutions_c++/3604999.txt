#include <iostream>
#include<string>
using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
    int test;
    int n,i,cnt[26],ans,j;
    string str;
        cin>>n;
        for(i=0;i<26;i++){
            cnt[i]=0;
        }
        for(i=0;i<n;i++){
            cin>>str;
            for(j=0;j<str.length();j++){
                if(cnt[str[j]-'a']==i){
                    cnt[str[j]-'a']++;
                }
            }
        }
        ans=0;
        for(i=0;i<26;i++){
            if(cnt[i]==n){
                ans++;
            }
        }
        cout<<ans;
    return 0;
}
