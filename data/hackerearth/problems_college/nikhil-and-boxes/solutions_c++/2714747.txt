#include <iostream>
using namespace std;
long long int min(long long int n1,long long int n2)
{
	if(n1<n2) return n1;
	return n2;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long int l,b,h;
    	cin>>l>>b>>h;
    	long long int res = min(l,min(b,h));
    	res = res *10;
    	res = res/2;
    	long long int t= res/10;
    	long long int r = res%10;
    	if(r==5)
    	cout<<t<<".5"<<endl;
    	else
    	cout<<t<<".0"<<endl;
    }
    return 0;
}
