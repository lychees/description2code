#include<iostream>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,a[2000],x[2000];
		int count=0,temp;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
	
			x[i]=a[i]%10;
			
		}
		for(int i=1;i<n;i++) 
   		{
      		for(int j=0;j<n-1;j++) 
	  		{
         		if(x[j]>x[j+1]) 
		 	{
            	temp=x[j];
            	x[j]=x[j+1];
            	x[j+1]=temp;
         	}
      	}
      }
		for(int i=0;i<n;i++)
		{
			if(x[i]==x[i+1])
			{
				count++;
				continue;
			}
		}
		cout<<count<<endl;
		
	}
	return 0;
}
