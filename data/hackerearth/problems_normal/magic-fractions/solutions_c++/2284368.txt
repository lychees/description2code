#include <bits/stdc++.h>
using namespace std;
#define ll long long int
vector<int> prime(600,0);
ll mod_exp(ll x,ll y)//x^y..//fast
{
	ll ans=1;
	while(y)
	{
		if(y%2==1)//odd power
		{
			ans=ans*x;
		}
		x=x*x;
		y=y/2;
	}
	return ans;
}
void sieve()
{
	for(int i=2;i*i<=600;i++)
	{
		if(prime[i]==0)//not visited
		{
			for (int j = i*i; j<=600 ; j+=i)
			{
				prime[j]=1;//not prime	
			}
		}
	}
}
int main(int argc, char const *argv[])
{
	sieve();
	int n;cin>>n;
	ll cnt=0;
	ll sum=0;
	for (int i = 2; i<=n ; ++i)
	{
		if(prime[i]==0)//prime
		{
			cnt++;
		}
		sum=sum+mod_exp(2,cnt-1);
	}
	cout<<sum<<endl;
	return 0;
}
