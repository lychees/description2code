#include <bits/stdc++.h>
#define ll long long
using namespace std;
char str[1000010];
int ispalin(int l,int r)
{
	int flag=1;
	while(l<r)
	{
		if(str[l]!=str[r])
		{
			flag=0;
			break;
		}
		l++;r--;
	}
	return flag;
}
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int l,p,q,ans;
		scanf("%s",str);
		l=strlen(str);
		if(ispalin(0,l-1))
		{
			ans=-1;
		}
		else
		{
			p=0;q=l-1;
			while(p<q)
			{
				if(str[p]!=str[q])
				{
					if(ispalin(p,q-1))
					{
						ans=q;
						break;
					}
					else
					{
						ans=p;
						break;
					}
				}
				p++;
				q--;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
