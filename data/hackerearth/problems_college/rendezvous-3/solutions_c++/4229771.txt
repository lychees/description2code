#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define maxl 100005
#define f first
#define s second
ll w[maxl];
pair<ll,pair<ll,ll>> p[maxl];

ll get(ll x){
	if(w[x]==x) return x;
	else return w[x]=get(w[x]);
}

void merge(ll a,ll b){
	a=get(a);
	b=get(b);
	w[b]=a;
}

int main()
{
	ll n,m,a,b,c,i;
	cin >> n >> m;
	for(i=0;i<maxl;i++){
		w[i]=i;
	}
	for(i=0;i<m;i++){
		cin >> a >> b >> c;
		p[i] = {c,{a,b}};
	}
	ll sum=0;
	sort(p,p+m);
	for(i=0;i<m;i++){
		a=p[i].s.f;
		b=p[i].s.s;
		c=p[i].f;
		if(get(a)!=get(b)){
			sum+=c;
			merge(a,b);
		}
	}
	cout << sum << endl;
    return 0;
}
