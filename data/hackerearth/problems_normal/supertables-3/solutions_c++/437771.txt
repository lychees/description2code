#include <iostream>
using namespace std;

#define lli long long int 

lli larger(lli a, lli b)
{
	return a>b?a:b;
}

lli smaller(lli a, lli b)
{
	return a<b?a:b;
}

lli gcd(lli a,lli b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}

lli lcm(lli a, lli b)
{
	return a*b/gcd(a,b);
}
lli getIndexOf(lli mid, lli l, lli s, lli mul)
{
	return mid/l+mid/s-mid/mul;
}

int main()
{
	lli t,l,s,a,b,n,mul,left,right;
	cin>>t;
	while(t)
	{
		cin>>a>>b>>n;
		l = larger(a,b);
		s = smaller(a,b);
		mul = lcm(a,b);
		left = 1;
		right = l*n;
		//binarySearch()
		while(left<=right)
		{
			lli mid = (left+right)/2;

			lli index = getIndexOf(mid,l,s,mul);
			//cout<<"mid"<<mid<<"index"<<index<<"left"<<left<<"right"<<right<<endl;
			//cin>>a;
			if(index<n)
			{
				left = mid+1;
			}
			else if(index>n)
			{
				right = mid-1;
			}
			else
			{
				lli ans = larger((mid/l)*l,(mid/s)*s);
				cout<<ans<<endl;
				break;
			}
		}
		t--;
	}
	return 0;
}
