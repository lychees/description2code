#include<iostream>
#include<vector>
using namespace std;

bool remdata(char c,int b)
{
	if(c == 'a') return true;
	else return false;
}
int main()
{
	string str;
	int a,b=0,ctr=0;
	vector<int> data;
	cin>>a;
	cin>>str;
	for(int x=0; x<a; x++) data.push_back(x+1);
	while(data.size()!=1)
	{
		if(remdata(str[ctr],b) == true)
		{
			if(b >= data.size()-1) b=0;
			else b++;
		}
		else 
		{
			data.erase(data.begin()+b);
			if(b > data.size()-1) b=0;
		}
		if(ctr < str.size()-1) ctr++;
		else ctr=0;
	}
	cout<<data[0];
	return 0;
}
