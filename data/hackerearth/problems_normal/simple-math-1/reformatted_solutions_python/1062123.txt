import math

[n, r] = [int(x) for x in raw_input().split()]
n, r = n-1, r-1

if r > n:
    print -1
elif r == n:
    print 1
else:
    i = math.factorial(n)
    j = math.factorial(n-r)
    k = math.factorial(r)
    print(i/(j*k)) % 10000007
