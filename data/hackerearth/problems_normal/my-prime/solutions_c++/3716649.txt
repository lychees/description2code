#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int arr[n],c[n];
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
        c[i]=0;
    }
    for(int i=0;i<n;i++)
    {
    	for(int j=0;j<n;j++)
    	if(arr[j]%arr[i]==0 && i!=j)
    	c[j]=c[j]+1;
    }
    for(int i=0;i<n;i++)
    {
        if(c[i]==0)
        cout<<arr[i]<<" ";
    }
    return 0;
}
