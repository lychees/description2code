#include <iostream>

using namespace std;

int main()
{
    int n,l,r;
    int q;

    cin>>n;

    int *arr = new int[n];
    for(int i=0; i<n; i++){
        cin>>arr[i];
    }

    cin>>q;
    while(q--){
        int sum = 0;
        cin>>l>>r;
        for(int i=(l-1); i<r; i++){
            sum = sum + arr[i];
        }
        cout<<sum<<endl;
    }
    return 0;
}
