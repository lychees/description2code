from fractions import gcd
#print gcd(60,6)
t=input()
while(t>0):
	t-=1
	n,k=map(int,raw_input().split())
	lst=map(int,raw_input().split())
	c=0
	for i in range(len(lst)):
		if lst[i]==k:
			c+=1
	
	#print (c*(c-1)*(c-2))
	#print (n*(n-1)*(n-2))
	c=(c*(c-1)*(c-2))
	n=(n*(n-1)*(n-2))
	div=gcd(n,c)
	#print div
	s=str(c/div)+'/'+str(n/div)
	print s
