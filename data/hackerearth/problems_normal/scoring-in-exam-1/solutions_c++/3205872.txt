#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
struct st{
	long long t,s;
};
bool com(st a,st b){
	return a.s>b.s;
}
int main()
{
	int n,q;
	cin>>n>>q;
	struct st a[n];
	
	for(int i = 0; i < n; i++){
		cin>>a[i].t;
		
	}
	for(int i = 0; i < n; i++){
		cin>>a[i].s;
		
	}
	
	sort(a,a+n,com);
	
	for(int i = 1; i<n;i++){
			a[i].t += a[i-1].t;
		}
	
	while(q--){
		int k;
		cin>>k;
		long long ans = a[k-1].t;
	
	cout<<ans<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
