#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int arr[n];
		bool visited[n];
		for(int i=0;i<n;i++){
			scanf("%d",&arr[i]);
			visited[i]=false;
		}
		int i=0;
		bool win=false;
		while(1){
			if(arr[i]==-1){
				win=true;
				break;
			}
			else if(visited[i]==false)
				visited[i]=true;
			else
				break;
			i=arr[i];
		}
		if(win==true)
			cout<<"Win\n";
		else
			cout<<"Lose\n";
	}
}
