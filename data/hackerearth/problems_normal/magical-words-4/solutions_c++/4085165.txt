#include <bits/stdc++.h>
using namespace std;

bool mat[1010][1010];
int main() {
	int t,len;
	long long count,ans;
	char str[1010];
	scanf("%d", &t);
	while(t--) {
		scanf("%s", str);
		len = strlen(str);
		memset(mat,0,sizeof(mat));
		for(int i = 0; i < len; i++) {
			mat[i][i] = true;
		}
		ans = len;
		for(int i=1; i < len; i++) {
			count = 0;
			for(int j = 0; j+i < len; j++) {
				if(str[j] == str[j+i]) {
					mat[j][j+i] = i == 1?true:mat[j+1][j+i-1];
				} else {
					mat[j][j+i] = false;
				}
				if(mat[j][j+i] == true) count++;
			}
			if(count > 0) {
				ans += (i+1)*(i+1)*count;
			}
		}
		printf("%lld\n", ans);
	}
    return 0;
}
