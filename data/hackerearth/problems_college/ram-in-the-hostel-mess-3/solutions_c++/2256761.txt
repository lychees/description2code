#include <bits/stdc++.h>
using namespace std;

int main()
{

    int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	vector<int> v;
    	
    	v.push_back(0);
    	v.push_back(2);
    	v.push_back(3);
    	for(int i=3;i<=n;i++)
    	{
    		v.push_back(v[i-1]+v[i-2]);
    		
    	}
    
    	cout<<(v[n]*v[n])<<endl;
    		
    	
  	
    }
    return 0;
}
