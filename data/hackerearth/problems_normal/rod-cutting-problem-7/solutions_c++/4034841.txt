#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll largest_power(ll N)
{
    //changing all right side bits to 1.
    N = N| (N>>1);
    N = N| (N>>2);
    N = N| (N>>4);
    N = N| (N>>8);
    N = N| (N>>16);
    N = N| (N>>32);
      
    return (N+1)>>1;

}
int main(){
ll T,count,i,n;
cin>>T;
while(T--){
cin>>n;
count=0;
if(!((n)&(n+1)))
	n++;
ll temp=largest_power(n);
temp--;
for(i=0;;i++){
if( temp & (1 << i) )
	count++;
else
	break;
}
cout<<(count-1)<<endl;
}
return 0;
}
