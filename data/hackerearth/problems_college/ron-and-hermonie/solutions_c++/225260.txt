unsigned long long power(unsigned long long x, unsigned long long y)
{
    if( y == 0)
        return 1;
    unsigned long long temp =  power(x, y/2);
    if (y%2 == 0)
        return ((temp )*(temp )) % 10;
    else
        return ((x )*(temp )*(temp ))%10;

}
