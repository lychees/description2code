import fileinput as f
import math
def computeValue(n,r):
	n = n-1
	r = r-1
	if n>0 and r>0 and n >= r:
		print (math.factorial(n)/(math.factorial(n-r) * math.factorial(r))) % 10000007
	else:
		print -1

vals=iter(f.input())
for line in vals:
	nr = line.split(' ')
	n = int(nr[0])
	r = int(nr[1])
	computeValue(n,r)
