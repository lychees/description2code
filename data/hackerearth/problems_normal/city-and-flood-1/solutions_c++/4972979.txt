#include<bits/stdc++.h>
using namespace std;
int n,k;
int par[100005];
int find(int x)
{   if(x!=par[x])
   return find(par[x]);
   else
   return x;
}

int main()
{    scanf("%d %d",&n,&k);
     for(int i=1;i<=n;i++)
     par[i]=i;
     for(int i=1;i<=k;i++)
     {    int x,y;
          scanf("%d %d",&x,&y);
          par[y]=x;
	 }
	 int meghans=0;
	 for(int i=1;i<=n;i++)
	 {   if(par[i]==i)
	      meghans++;
	 }
	 printf("%d\n",meghans);
	 return 0;
}
