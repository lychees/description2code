#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    long long Ar[300]={0};
    memset(Ar,0,sizeof(Ar));
    long long C[300]={0};
    memset(C,0,sizeof(C));
    int N,M;
    cin>>N>>M;

    for(int i=0;i<M;i++)
        cin>>C[i];

    for(int i=0;i<=N;i+=C[0])
        Ar[i]++;

    for(int i=1;i<M;i++)
    {
        long long Br[300]={0};
        memset(Br,0,sizeof(Br));
        for(int k=0;k<=N;k++)
        {
            if(Ar[k])
            {
                for(int j=k+C[i];j<=N;j+=C[i])
                    Br[j]+=Ar[k];
            }
        }
        for(int l=1;l<=N;l++) Ar[l]+=Br[l];
    }
    cout<<Ar[N]<<"\n";
    return 0;
}
