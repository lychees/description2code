#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll i,t,n,k,ans=1000000000,mini,maxi;
    int a[100000];
    cin>>n;
    cin>>k;
    
    for(i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    for(i=0;i<=n-k;i++)
    {
    	mini=a[i];
    	maxi=a[i+k-1];
    	if(ans>maxi-mini)
    		ans=maxi-mini;
    }
    cout<<ans;
    return 0;
}
