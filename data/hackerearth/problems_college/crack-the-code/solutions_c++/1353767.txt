#include<bits/stdc++.h>
using namespace std;
int main(){
	int fib[100001];
	fib[0]=1;
	fib[1]=1;
	int k=1;
	for(int i=2; fib[i]<=100000; i++){
		fib[i]=fib[i-1]+fib[i-2]; k++;
	}
	int t;
	cin>>t;
	while(t--){
		int x,y;
		cin>>x>>y;
		int count=0,ind=0;
		while(1){
			if(ind>k) break;
			if(fib[ind]>y) break;
			else{
				if(fib[ind]>=x && fib[ind]<=y){
					count++;
				} 
			}
			ind++;
		}
		if(count%2) cout<<"INVALID"<<endl;
		else cout<<"VALID"<<endl;
	}
}
