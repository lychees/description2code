import math
time = raw_input()

hour = int(time[0:2])
minute = int(time[3:5])

angle = 360 / 60

hr_angle = angle*hour*5 + (5/60.0) * minute * angle
min_angle = angle*minute

f_angle = hr_angle-min_angle
f_angle = math.copysign(f_angle, +0.0)

f_angle = f_angle if (f_angle <= 180) else (360-f_angle)

print '{0:.6f}'.format(f_angle)
