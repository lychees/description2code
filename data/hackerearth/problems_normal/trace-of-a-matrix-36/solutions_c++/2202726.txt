#include <iostream>
#include <vector>
using namespace std;

int main()
{
   int t;
   cin>>t;
   
   for(int i=0;i<t;i++)
   {
   	
   	long long int n_row,n_col;
   	
   	cin>>n_row>>n_col;
   	
   
   	
   
  
   
   
   
   
   
   
   
   	long long int sum=0;
   	sum=0;
   	int k=n_row+1;
   	int l=n_col+1;
   long long int n=min(n_row,n_col);
   
   sum=sum+(n+(((n*(n+1))/2)-n)*k);
   sum=sum+(n+(((n*(n+1))/2)-n)*l);
   
   cout<<sum<<endl;
   
   }
   
   
   
   
    return 0;
}
