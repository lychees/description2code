#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n,flag=0;
    cin>>n;
    int *a = new int[n];
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(a[i]>0)
    	flag=1;    	
    }
    if(flag==0)
       {
       	cout<<0<<endl;
       
       }
    else
    {
    	int cs=0,ps=0;
    	for(int i=0;i<n;i++)
    	{
    		cs+=a[i];
    		if(cs<0)
    		 cs=0;
    		 if(cs>ps)
    		 ps=cs;
    		 
    	}
    	 cout<<ps<<endl;
    }
   
    return 0;
}
