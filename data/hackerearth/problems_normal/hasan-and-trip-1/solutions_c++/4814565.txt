#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,i,j;
	cin>>n;
	double dp[n+1];
double x[n+1],y[n+1],f[n+1];
	for(i=1;i<=n;i++)
	{
		cin>>x[i]>>y[i]>>f[i];

	}
	for (i=1;i<=n;i++)
	dp[i]=-1000000;
	dp[1]=0;
	for(i=1;i<=n;i++)
	{
		dp[i]=dp[i]+f[i];
		for(j=i+1;j<=n;j++)
		{
			double D=sqrt((x[j]-x[i])*(x[j]-x[i])+(y[j]-y[i])*(y[j]-y[i]));
			dp[j]=max(dp[j],dp[i]-D);
		}
		cout.precision(6);
		
	}
	cout<<fixed<<dp[n]<<endl;
}
