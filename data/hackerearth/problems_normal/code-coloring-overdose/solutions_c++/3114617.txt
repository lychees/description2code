#include<bits/stdc++.h>
using namespace std;
long long mod = 1000000007;
vector<long long> pow2;
int main()
{
    long k,n;
    int m;
    long c;
    cin>>n>>k>>m;
    pow2.push_back(1);
    vector<long> has(k+1,0);
    for(long cn=0;cn<n;cn++){
        pow2.push_back((pow2[cn]*2)%mod);
        cin>>c;
        has[c]++;
    }
    vector<long long> dp(m+1,0);
    dp[0] = 1;
    for(int j=0; j < has.size();j++)
        if(has[j] != 0){
            for(int i=m;i > 0;i--){
                dp[i] += (pow2[has[j]]-1)*dp[i-1];
                dp[i] %= mod;
                // cout<<dp[i]<<" ";
            }
            // cout<<dp[0]<<endl;
        }
    long long ans = 0;
    for(int i = m;i >= 0;i--){
        ans += dp[i];
        ans %= mod;
    }
    cout<<ans;
	return 0;
}
