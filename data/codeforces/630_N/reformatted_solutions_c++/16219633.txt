#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
double a, b, c;
int main() {
  cin >> a >> b >> c;
  if (a < 0)
    a = -a, b = -b, c = -c;
  double rt = sqrt(b * b - 4 * a * c);
  printf("%.9f\n", (-b + rt) / (2 * a));
  printf("%.9f\n", (-b - rt) / (2 * a));
}
