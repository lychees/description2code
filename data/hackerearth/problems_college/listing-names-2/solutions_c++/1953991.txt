#include <iostream>
#include<map>
using namespace std;

int main()
{
   int t;
   cin>>t;
   map<string,int> mapi;
   map<string,int>::iterator it;
   while(t--)
   {
   	string S;
   	cin>>S;
   	it=mapi.find(S);
   	if(it!=mapi.end())
   	mapi[S]++;
   	else
   	mapi[S]=1;
   	
   	
   }
   for(it=mapi.begin();it!=mapi.end();it++)
   cout<<it->first<<" "<<it->second<<endl;
    return 0;
}
