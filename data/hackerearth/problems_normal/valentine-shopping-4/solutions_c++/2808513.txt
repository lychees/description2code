		{
			//vector<long double> val(m,0) ; 
			
			long double  min = numeric_limits<long double> :: max() ;

				long double a , b , c ;
				scanf("%Lf%Lf%Lf",&a,&b,&c) ;
				
				rem = rem - ((rem*a)/100.0) ;
				rem = rem - ((rem*b)/100.0) ;
