#include <iostream>

using namespace std;

int main()
{
    int l,n,a[10000],b[10000],i;
    cin>>l;
    cin>>n;
    if(n>1000&&n<1)
        cout<<"wrong input";
    for(i=0;i<n;i++)
    {
        cin>>a[i]>>b[i];
    }
    for(i=0;i<n;i++)
    {
    	if(a[i]==b[i]&&a[i]>l&&b[i]>l)
        cout<<"ACCEPTED\n";
        else 
        	if(a[i]<l||b[i]<l)
            cout<<"UPLOAD ANOTHER\n";
        	else if(a[i]>l||b[i]>l)
            cout<<"CROP IT\n";
        	else if(a[i]==l&&b[i]==l)
            cout<<"ACCEPTED\n";
        
    }
    return 0;
}
