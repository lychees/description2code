#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	int b[n];
    	int maxi=a[n-1];
    	b[n-1]=maxi;
    	for(int i=n-2;i>=0;i--)
    	{
    		if(a[i]>maxi)
    		{
    			maxi=a[i];
    		}
    		b[i]=maxi;
    	}
    	int d[n];
    	for(int i=0;i<n;i++)
    	{
    		d[i]=b[i]-a[i];
    	}
    	long int sum=0;
    	for(int i=0;i<n;i++)
    	{
    		sum+=d[i];
    	}
    	cout<<sum<<endl;
    }
}
