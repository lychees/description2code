#include<iostream>
#include<string>
#include<vector>

using namespace std;
static bool arr[100000];

int max(int a,int b)
{
    return(a>b)?
    a:b;
}

int check(int a,int b)
{
    if(arr[a-1]||arr[b-1])return 0;
    if(a==b)return 1;
    if(a>b)return check(a/2,b);
    else return check(a,b/2);
}


int main()
{
    int n,q;
    cin>>q>>n;
    string s;
    cin>>s;
    for (int i=0;i<q;i++)
    {
        arr[i]=s[i]-'0';
       //cout<<s[i]<<arr[i];

    }
    while(n--)
  {
    int a,b;
    cin>>a>>b;
    int l=check(a,b);
    if(l)cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
  }
  return 0;
}
