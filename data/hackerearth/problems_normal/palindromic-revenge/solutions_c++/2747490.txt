#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int t,c,i;
    string s;
    cin>>t;
    while(t--){
    	cin>>s;
    	c=0;
    	for(i=0;i<s.size()/2;i++){
    		if(s[i]!=s[s.size()-i-1]){
    			c+=abs(s[i]-s[s.size()-i-1]);
    		}
    	}
    	cout<<c<<endl;
    }
    return 0;
}
