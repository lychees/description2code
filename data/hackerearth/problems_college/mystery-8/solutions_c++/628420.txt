#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int test_cases;
    cin>>test_cases;
    getchar();
    while(test_cases--)
    {
        char str[101], ans;
        gets(str);
        // your code here
        int i=0, j=strlen(str)-1;
        		while(i<j)
		{
			
			if(!isalpha(str[i]) && str[i]!='*')
			{
				i++;
				continue;
			}
			if(!isalpha(str[j]) && str[j]!='*')
			{
				j--;
				continue;
			}
			
			if(str[i]=='*')
			{
				ans=str[j];
				break;
			}
			if(str[j]=='*')
			{
				ans=str[i];
				break;
			}
			i++;j--;
		}
cout<<ans<<"\n";
       
    }
    return 0;
}
