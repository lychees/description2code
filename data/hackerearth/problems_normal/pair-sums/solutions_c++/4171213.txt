#include <iostream>
using namespace std;
const int maxA = 1000005;
int main()
{
	int N, K;
	cin >> N >> K;
	int *A = new int [maxA];
	int *B = new int [maxA];
	for(int i = 0; i < maxA; ++i)
	{
		A[i] = B[i] = 0;
	}
	bool isExist = false;
	for(int i = 0; i < N; ++i)
	{
		int Q;
		cin >> Q;
		if (K > Q && (K - Q) < maxA)
		{
			++A[Q];
			if (A[Q]*B[Q] > 0)
			{
				isExist = true;
				cout << "YES" << endl;
				break;
			}
			++B[K-Q];
			if ((K != 2*Q) && (A[K-Q]*B[K-Q]) > 0)
			{
				isExist = true;
				cout << "YES" << endl;
				break;
			}
		}
	}
	if (!isExist)
	{
		cout << "NO" << endl;
	}
	
    //cout << "Hello World!" << endl;
    return 0;
}
