//Included Libraries
#include <iostream>
#include <cstdio>
#include <math.h>
#include <vector>
#include <utility>

//General Macros
#define S(a) 		scanf("%lld", &a)

using namespace std;

int main()
{
	pair<long long, long long> x;

	long long a, b, mid, N, T;
	double m;

	S(T);
						//cout << T << endl;
	while (T--)
	{
		x.first = 0;
		x.second = 1000000000;
		
		S(N);
						//cout << N << endl;
		S(a);
		for (int i = 1; i < N; i++ ){
			S(b);
						//cout << a << '\t' << b << '\t';
			m = (double)(a+b)/2;
						//cout << m << endl;
			if (a < m){
				mid = floor(m);
						//cout << mid << endl;
				x.second = min(x.second, mid);
						//printf("%lld %lld\n", x.first, x.second);
				if (x.second < x.first){
					break; 
				}
			}
			if (a > m){
				mid = ceil(m);
						//cout << mid << endl;
				x.first = max(x.first, mid);
						//printf("%lld %lld\n", x.first, x.second);
				if (x.second < x.first){
					break; 
				}
			}

			a = b;
		}
		
		if (x.second < x.first){
			printf("-1\n");
		}
		else
			printf("%lld %lld\n", x.first, x.second);
	}
	return 0;

}
