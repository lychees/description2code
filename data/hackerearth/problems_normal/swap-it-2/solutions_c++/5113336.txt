#include <iostream>
using namespace std;

void swap(long &a, long &b)
{
	long t = a;
	a = b;
	b = t;
}

int main()
{
	ios::sync_with_stdio(0);
	cin.tie(0);
	cout.tie(0);
    long n,i,k,t,a[1000],j,last;
    cin >> t;
    while (t--)
    {
    	cin >> n >> k;
    	for (i = 0; i < n; i++)
    		cin >> a[i];
    	for (i = 0; i < n; i++)
    	{
    		last = i;
    		for (j = i + 1; j < n; j++)
    		{
    			if (j - i > k) break;
    			if (a[j] < a[last])
    			   last = j;
    		}
    		k -= last - i;
    		for (j = last; j > i; j--)
    		  swap(a[j],a[j-1]);
    		if (k == 0) break;
    	}
    	for (i = 0; i < n; i++)
    	 cout << a[i] << " ";
    	cout << "\n";
    }
    return 0;
}
