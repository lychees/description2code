#include<bits/stdc++.h>
using namespace std;
typedef pair<long long,int>pii;
bool vis[100001];
vector<pii>adj[100001];
long long minicost;
long long mst(int u){
priority_queue<pii , vector<pii> , greater<pii> >Q;
pii p;
int x;
Q.push(make_pair(0,1));
while(!Q.empty()){
    p = Q.top();
    Q.pop();
    x= p.second;
    if(vis[x]==true)
        continue;
    minicost+=p.first;
    vis[x]=true;
    for(int i=0;i<adj[x].size();i++){
        if(!vis[adj[x][i].second])
            Q.push(adj[x][i]);

    }

}
}
int main(){
ios_base::sync_with_stdio(0);
int n,m;
memset(vis,false,sizeof(vis));
long long cost;
int i,x,y;
cin>>n>>m;
for(i=1;i<=m;i++){

    cin>>x>>y>>cost;
    adj[x].push_back(make_pair(cost,y));
    adj[y].push_back(make_pair(cost,x));
    }
minicost=0;
mst(1);
cout<<minicost<<endl;
}
