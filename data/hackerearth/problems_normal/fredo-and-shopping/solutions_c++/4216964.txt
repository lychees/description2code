#define ll long long int

ll a[1000008],n,b[1000008],bit[1000008];

void update(ll idx,ll val,ll me)
{
	for(;idx<=me;idx+=idx&(-idx))
	{
		bit[idx]+=1;
	}
}

ll query(ll x)
{
	ll s=0;
	for(;x>=1;x-=x&(-x))
	{
		s+=bit[x];
	}
	return s;
}
