#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		int n,flag=0;
		cin>>n;
	for(int i=0;i<=n;i++){
		for(int j=0;j<=n;j++){
			for(int k=0;k<=n;k++){
			if(6*i+9*j+20*k==n){
				flag=1;
				cout<<"True\n";
				i=n;j=n;k=n;
			}
			}
		}
	}
	if(flag==0)
	cout<<"False\n";
	}
	return 0;
}
