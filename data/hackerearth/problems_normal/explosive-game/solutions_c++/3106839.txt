#include <iostream>
#include <vector>
#include <map>
#include <set>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <fstream>
#define For(i, a, b) for(int i = a; i < b; ++i)
#define Out(i, a, b) for(int i = a - 1; i >= b; --i)
#define pb push_back
#define point pair <int, int>
#define x first
#define y second

using namespace std;

int n, q;
set <long long> b;
long long cnt = 0;

int main() {
    ios::sync_with_stdio(false);
    cin >> n >> q;
    For(i, 0, n) {
        long long L, R; cin >> L >> R;
        cnt += min(L, R);
        if (R != L) {
            if (b.count(abs(R - L)) == 1) {
                b.erase(abs(R - L));
                cnt += abs(R - L);
            } else {
                b.insert(abs(R - L));
            }
        }
    }
    For(i, 0, q) {
        long long a; cin >> a;
        if (a == cnt && b.size() == 0) cout << "YES" << endl;
        else cout << "NO" << endl;
    }
}
