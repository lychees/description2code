def func(x,y):
	count=0
	c=0
	if x>y:
		large=x
		small=y
	else:
		large=y
		small=x
	while large>0:
		d1=large%10
		d2=small%10
		s=d1+d2+c
		if s>9:
			c=s/10
			count+=1
		else:
			c=0
		large/=10
		small/=10
	return count

li=[]
t=int(input())
for i in range(t):
	a,b=raw_input().split()
	a=int(a)
	b=int(b)
	r=func(a,b)
	li.append(r)
for i in range(t):
	if li[i]==0:
		print "No carry operation"
	elif li[i]==1:
		print "1 carry operation"
	else:
		print str(li[i]) + " carry operations"
