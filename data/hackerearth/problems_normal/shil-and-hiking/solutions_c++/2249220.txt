#define N	1010

double a[N],b[N];
 
int main(){
	/*#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
	#endif*/
	int n,i,j;
	cin>>n;
	rep(i,n){
		cin>>a[i]>>b[i];
	}
	int ans = 0;
	FOR(i,1,n){
		double m = INF;
		for( int j = i-1 ; j >= 0 ; j-- ){
			double cur = (b[j] - b[i])/(a[j] - a[i]);
			if( cur <= m ){
				m = cur;
				ans++;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
