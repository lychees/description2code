#include <iostream>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else 
	return gcd(b,a%b);
}
int main()
{
    int a,b,c;
    cin>>a>>b>>c;
    int p,q;
    q=a*b;
    if(a>=c && b>=c)
    {
    	p=c*c;
    	q*=2;
    }
    else if(a<c && b<c)
    {
    	if(c>=(a+b))
    	{
    		p=a*b;
    	}
    	else
    	{
    		p=2*a*b-(c-a-b)*(c-a-b);
    		q*=2;
    	}
    }
    else if(a<c && b>=c)
    {
    	p=2*a*(c-a)+a*a;
    	q*=2;
    }
    else
    {
    	p=2*b*(c-b)+b*b;
    	q*=2;
    }
   // p=min(a,c)*min(b,c);
   int x=gcd(p,q);
   
    
    printf("%d/%d",p/x,q/x);
    return 0;
}
