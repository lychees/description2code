#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin >> t;
	
	while (t--) {
		int n, k;
		cin >> n >> k;
		int a[n];
		for (int i = 0; i < n; i++) {
			cin >> a[i];
		}
		sort(a, a + n);
		
		int ans = 1000000;
		for (int i = 0; i < n; i++) {
			int b[n];
			int c = 0;
			for (int j = 0; j < n; j++) {
				if (a[j] > a[i]) {
					b[j] = 5 * (a[j] - a[i]);
				} else {
					b[j] = 3 * (a[i] - a[j]);
				}
			}
			sort(b, b + n);
			for (int j = 0; j < k; j++) {
				c = c + b[j];
			}
			if (c < ans) {
				ans = c;
			}
		}
		cout << ans << endl;
	}
    return 0;
}
