#include<iostream>
#include<cstdio>
using namespace std;
bool seven(int i)
{
	if(i==0) return 0;
	if(i%10==7) return 1;
	return seven(i/10);
}
int main()
{
	int i,j;
	bool prime[1000001];
	int sev[1000001];
	for(i=1;i<=1000000;i++) {prime[i]=1;sev[i]=0;}
	prime[1]=1;
	i=2;
	
	while(i<=1000000)
	{
		//cout<<i<<endl;
		j=i*2;
		while(j<=1000000) {prime[j]=0;j+=i;}
		i++;
		while((prime[i]==0)&&i<=1000000) i++;
	}
	for(i=1;i<=1000000;i++)
	{
		if((prime[i]!=1) && (seven(i)==1)) sev[i]=1;
	}
	sev[0]=0;
	for(i=2;i<=1000000;i++) sev[i]+=sev[i-1];
	int t,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		int ans=sev[m]-sev[n-1];
		if(ans==0) ans=-1;
		printf("%d\n",ans);
	}
}
