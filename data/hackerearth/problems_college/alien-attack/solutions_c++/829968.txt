#include <iostream>
#define MOD 1000000007
using namespace std;

long long int a[2][2],res[2][2];
void pre(long long n){
	long long e=n;
	long long int b[2][2];
	a[0][0]=a[1][0]=1;
	a[0][1]=2;a[1][1]=0;
	res[0][0]=1;
	res[1][1]=1;
	res[1][0]=res[0][1]=0;
	while(e){
		if(e%2==1){
			b[0][0]=res[0][0];
			b[0][1]=res[0][1];
			b[1][0]=res[1][0];
			b[1][1]=res[1][1];
			res[0][0] = (b[0][0]*a[0][0] + b[0][1]*a[1][0])%MOD;
			res[0][1] = (b[0][0]*a[0][1] + b[0][1]*a[1][1])%MOD;
			res[1][0] = (b[1][0]*a[0][0] + b[1][1]*a[1][0])%MOD;
			res[1][1] = (b[1][0]*a[0][1] + b[1][1]*a[1][1])%MOD;
		}
		e=e>>1;
			b[0][0]=a[0][0];
			b[0][1]=a[0][1];
			b[1][0]=a[1][0];
			b[1][1]=a[1][1];
			a[0][0] = (b[0][0]*b[0][0] + b[0][1]*b[1][0])%MOD;
			a[0][1] = (b[0][0]*b[0][1] + b[0][1]*b[1][1])%MOD;
			a[1][0] = (b[1][0]*b[0][0] + b[1][1]*b[1][0])%MOD;
			a[1][1] = (b[1][0]*b[0][1] + b[1][1]*b[1][1])%MOD;
			//cout<<a[0][0]<<" "<<a[0][1]<<" "<<a[1][0]<<" "<<a[1][1]<<endl;
			
	}
}
int main()
{
    int t;
    cin>>t;
    long long n;
    
    while(t--){
    	cin>>n;
    	if(n==1)
    		cout<<2<<endl;
    	else if(n==2)
    		cout<<7<<endl;
    	else{
    		pre(n-2);
    		//cout<<res[0][0]<<" "<<res[0][1]<<endl;
    		long long val = (res[0][0]*7 + res[0][1]*2)%MOD;
    		cout<<val<<endl;
    	}
    		
    }
    return 0;
}
