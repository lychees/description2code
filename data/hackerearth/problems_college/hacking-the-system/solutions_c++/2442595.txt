#include<iostream>
#include<string>
using namespace std;
int main(){
	string x,r;
	int flag,len;
	cin>>x;
	cin>>r;
	if(x.compare(r)==0){
		cout<<"wins\n";
	}
	else{
		flag=0;
		len=x.length();
		len--;
		while(len>=0){
			if(x[len]!=r[len]){
				flag++;
			}
			len--;
		}
		if(flag<=1){
			cout<<"wins\n";
		}
		else{
			cout<<"loses\n";
		}
	}
	return 0;
}
