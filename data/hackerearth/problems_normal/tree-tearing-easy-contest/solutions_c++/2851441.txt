#include <vector>
#include <iostream>
#include <cstring>
using namespace std;
int deg[10010];
vector<int> vec[10010];
int ans = 0;
int vis[10010];
int k;
int dfs(int ver) {
	if(vis[ver])	return 0;
	vis[ver] = 1;
	int count = 0;
	int n = vec[ver].size();
	for(int i = 0; i < n; ++i) {
		if(vis[vec[ver][i]])	continue;
		count += dfs(vec[ver][i]);
	}
	count++;
	if(count >= k)	 {
		ans++;
		count = 0;
	}
	return count;	
}

int main()
{
    
    int n;
    cin>>k>>n;
    memset(deg, 0, sizeof(deg));
    for(int i = 2; i <= n; ++i) {
    	int b;
    	cin>>b;
    	vec[i].push_back(b);
    	vec[b].push_back(i);
    }
    memset(vis, 0, sizeof(vis));
    dfs(1);
    cout<<ans;
    
    return 0;
}
