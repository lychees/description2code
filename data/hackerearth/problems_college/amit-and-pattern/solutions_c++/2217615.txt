#include <bits/stdc++.h>
#include<stdio.h>
using namespace std;

int main() {
	// your code goes here
	int n,m;
	scanf("%d %d",&n,&m);
	int a[105][105];
	int mark[105][105];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			scanf("%d",&a[i][j]);
		}
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			mark[i][j]=0;
		}
	}
	queue< pair<int,int> > Q;
	Q.push(make_pair(0,0));
	Q.push(make_pair(-1,-1));
	while(Q.size()!=1)
	{
		pair<int,int> temp=Q.front();
		Q.pop();
		int x=temp.first;
		int y=temp.second;
		if(temp.first!=-1)
		{
			printf("%d ",a[x][y]);
			
			if(mark[x][y+1]==0&&(y+1<m)&&x<n)
			{
			Q.push(make_pair(x,y+1));
			mark[x][y+1]=1;
			}
            if(mark[x+1][y]==0&&(x+1<n)&&y<m)
			{
			Q.push(make_pair(x+1,y));
			mark[x+1][y]=1;
			}
			
		}
		else
		{
			printf("\n");
			Q.push(make_pair(-1,-1));
		}
		
	}
	
	
	return 0;
}
