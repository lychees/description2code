#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t, m, n, arr[1005], i, j, count;
    bool isAdded[1005];//is included
    scanf("%d", &t);
    while(t--)
    {
    	count = 0;
    	scanf("%d %d", &m, &n);
    	for(i=0; i<n; i++)
    	{
    		scanf("%d", &arr[i]);
    		isAdded[i]=false;
    	}
    	sort(&arr[0], &arr[n]);
    	for(i=n-1; i>=0; i--)
    	{
    		if(arr[i]<=m)
    		{
    			break;
    		}
    	}
    	for(; i>=0; i--)
    	{
    		if( ! (isAdded[i]) )
    		{
    			count++;
    			int temp = m - arr[i];
    			for(j=i-1; j>=0; j--)
	    		{
	    			if(arr[j]<=temp && (!isAdded[j]) )
	    			{
	    				isAdded[j] = true;
	    				break;
	    			}
	    		}
    		}
    	}
    	printf("%d\n", count);
    }
    return 0;
}
