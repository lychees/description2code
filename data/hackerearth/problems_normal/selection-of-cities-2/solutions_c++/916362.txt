unsigned long long int bigmod(unsigned long long int b,unsigned long long int p ,unsigned long long int m)
{
	if(p==0)return 1;
	else if(p%2==0)
	return square(bigmod(b,p/2,m))%m;
	else return ((b%m)*bigmod(b,p-1,m))%m;
	}//end of the big mod function
