def recur_factorial(n):
   if n == 1:
       return n
   else:
       return n*recur_factorial(n-1)

res=0
t=int(input())
while t>0 :
    sum=0
    num = int(input())
    res=recur_factorial(num)
    while res>0 :
        sum=sum+int(res%10)
        res=int(res/10)
    print(sum)
    t=t-1
