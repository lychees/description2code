#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
	
	int n;
	long int a[100000],x,q,l,r,i;
	
	cin>>n>>l>>r>>q;
	
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	
	sort(a+l,a+r+1);
	
	while(q--)
	{
		cin>>n;
		
		int flag=0;
		for(i=l;i<=r;i++)
		{
			if(a[i]>n)
			{
				cout<<a[i]<<endl;
				flag=1;
				break;
			}
		}
		
		if(flag==0)
		{
			cout<<"-1"<<endl;
		}
	}
 	 
    return 0;
}
