#include <bits/stdc++.h>
#include<vector>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
 int n,k,m,j=0,i;
scanf("%d",&n);
vector<int> v;
for(i=0;i<n;i++)
{
 scanf("%d",&m);
 v.push_back(m);
}
vector<int>::iterator it=v.begin();
vector<int>::iterator it1=v.end();
vector<int>::iterator it2;
sort(it,it1);
for(it2=it;it2<it1;it2++)
{
if(*it2==*(it2+1))
j++;
}
printf("%d\n",j);
}
return 0;
}
