#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
 
 
int main() 
{
  long long int n,k;
	int a[100000],i,j,count=0,sum,temp;
	cin>>n>>k;
	
	for(i=0;i<n;i++)
	cin>>a[i];
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n-i-1;j++)
		{
			if(a[j]>a[j+1])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	
	sum=a[0];
	i=1;
	
	while(sum<=k)
	{
		count++;
		sum=sum+a[i];
		i++;
	}
	
	cout<<count;
    return 0;
}
