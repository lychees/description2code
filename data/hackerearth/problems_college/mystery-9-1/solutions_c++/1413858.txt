#include <iostream>
#include <stdio.h>
#include <string>

using namespace std;

int main()
{
	
	int t, n, i, flag;
	
	cin >> t;
	
	while( t-- ) {
		
		string s;
		cin >> s;
		n = s.length();
		
		flag = 1;
		for( i = 0;i < n/2;i++ )
			if( s[ i ] != s[ n - i - 1 ] ) {
				flag = 0;
				break;
			}
		
		if( flag )
			cout << (int)s[ n / 2 ] << endl;
		else
			cout << "-1\n";
	}
	return 0;
}
