#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
	int t,digit[10]={6,2,5,5,4,5,6,3,7,6};
	cin>>t;
	while(t--)
	{
		int count=0;
		char str[1000];
		cin>>str;
		for(int i=0;i<strlen(str);i++)
		{
			int x=str[i]-'0';
			count+=digit[x];
		}
		cout<<count<<endl;
	}
}
