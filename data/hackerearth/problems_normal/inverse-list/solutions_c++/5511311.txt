#include <iostream>
#include<map>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--){
    	long int N;
    	cin>>N;
    	map<int,int> M;
    	int flag=0;
    	for(long int i=0;i<N;i++){
    		int t;
    		cin>>t;
    		M.insert(make_pair(t,i+1));
    	}
    	for(auto a:M){
    		if(a.first!=M.at(a.second)){
    			flag=1;
    			break;}
    	}
     	if(flag)
     		cout<<"not inverse"<<endl;
     	else
     		cout<<"inverse"<<endl;
    }
    return 0;
}
