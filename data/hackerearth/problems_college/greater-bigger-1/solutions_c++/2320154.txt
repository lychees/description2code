#include<bits/stdc++.h>
using namespace std;
int main()
{

    int t,flag,i,l;
    char str[110000];
    cin>>t;
    while(t--)
    {
//    	getchar();
        cin>>str;
        l=strlen(str);
        flag=0;
        for(i=l-1;i>0;i--)
        {
            if(str[i]>str[i-1])
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
        {
            int j,x=i;
            char temp;
            for(j=i;j<l;j++)
            {
                if(str[j]>str[i-1] && str[j]<str[x])
                        x=j;
            }
            temp=str[i-1];
            str[i-1]=str[x];
            str[x]=temp;
            sort(str+i,str+l);
            cout<<str<<"\n";
        }
        else
            cout<<"no answer\n";
    }
    return 0;

}
