N = int(raw_input())
choco_array = []
prev = 0
for i in raw_input().split():
    prev += int(i.strip())
    choco_array.append(prev)
Q = int(raw_input())
indexes = []
for i in range(Q):
    indexes.append(int(raw_input().strip()))
sorted_indexes = sorted(indexes)
ind_map = {}
ind = 1
for elemnt in sorted_indexes:
    while elemnt > choco_array[ind-1] and ind <= N:
        ind += 1
    if ind > N:
        break
    ind_map[elemnt] = ind
for elemnt in indexes:
    print ind_map[elemnt]
