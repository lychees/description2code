#include <bits/stdc++.h>

using namespace std;

long long divsum[1000005];
long long smallest[1000005];

bool vis[1000005];

void pre()
{
	for ( int i = 1; i <= 1000000; i++ ) {
		for ( int j = i; j <= 1000000; j += i ) {
			divsum[j] += i;			
		}
	}
	divsum[0] = divsum[1] = smallest[0] = smallest[1] = 0;
	
	for ( int i = 2; i <= 1000000; i++ ) {
		if ( !vis[i] ) {
			for ( int j = i; j <= 1000000; j += i ) {
				if ( !vis[j] ) smallest[j] = i, vis[j] = true;
			}
		}				
	}
	for ( int i = 2; i <= 1000000; i++ ) {
		divsum[i] = divsum[i-1] + divsum[i]-i;
		smallest[i] += smallest[i-1];
	}
	return;
}
int main()
{
	pre();
	int t,n;
	cin >> t;
	while ( t-- ) {
		cin >> n;
		cout << (divsum[n] + smallest[n])%n << endl;
	}
	return 0;
}
