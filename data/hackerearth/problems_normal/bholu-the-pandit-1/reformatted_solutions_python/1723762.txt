def gcd(a, b):
    if (b == 0):
        return a
    else:
        return gcd(b, a % b)


t = input()
for i in range(t):
    s = raw_input().split()
    l, b, h = int(s[0]), int(s[1]), int(s[2])
    edge = gcd(max(l, b, h), gcd(
        l+b+h-max(l, b, h)-min(l, b, h), min(l, b, h)))
    print edge,
    print((l*b*h)/(edge**3)) % 1000000007
