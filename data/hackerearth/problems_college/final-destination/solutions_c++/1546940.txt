#include <cstdio>
#include<cassert>
#include<queue>
#define MAX 1000000
int main()
{
   int t=0;
   scanf("%d",&t);
   while(t--)
   {
   	int a,b,mul,div,add,sub,i,temp;
   	scanf("%d %d",&a,&b);
   	scanf("%d %d %d %d",&mul,&div,&add,&sub);
   	std::queue<int> q;
   	int steps[MAX+1];
   	for(i=0;i<=MAX;i++)
   	{
   		steps[i]=-1;
   	}
   	steps[a]=0;
   	q.push(a);
   	while(!q.empty() && steps[b]==-1)
   	{
   		a=q.front();
   		q.pop();
   		
   		temp=mul*a;
   		if(temp<=MAX && temp>1 && steps[temp]==-1)
   		{
   			q.push(temp);
   			steps[temp]=steps[a]+1;
   		}
   			temp=a+add;
   		if(temp<=MAX && temp>1 && steps[temp]==-1)
   		{
   			q.push(temp);
   			steps[temp]=steps[a]+1;
   		}
   			temp=a/div;
   		if(temp<=MAX && temp>1 && steps[temp]==-1)
   		{
   			q.push(temp);
   			steps[temp]=steps[a]+1;
   		}
   			temp=a-sub;
   		if(temp<=MAX && temp>1 && steps[temp]==-1)
   		{
   			q.push(temp);
   			steps[temp]=steps[a]+1;
   		}
   		
   		
   	}
   	
   	printf("%d\n",steps[b]);
   }
    return 0;
}
