vowels = "aeiouy"
consonants = "bcdfghjklmnpqrstvwxz"
test_cases = int(raw_input())
for i in range(test_cases):
    data = raw_input()
    data_vow_cons = [True if i in vowels else False for i in data]
    prev = data_vow_cons[0]
    len_vow =  1 if prev else 0#data_vow_cons.count(True)
    len_con = 1 if not prev else 0#data_vow_cons.count(False)    
    count_consec = 1
    for i in range(1,len(data_vow_cons)):
        curr = data_vow_cons[i]
        if not curr:
            if curr == prev:
                count_consec += 1
            else:
                count_consec = 1
            len_con += 1
        else:
            len_vow += 1
        prev = curr
        if count_consec >= 3:
            break
    if len_con>len_vow or count_consec>=3:
        print 'hard'
    else:
        print 'easy'
