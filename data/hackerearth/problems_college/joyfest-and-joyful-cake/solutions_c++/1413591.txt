LL next_int, sign;
char in_char;
LL Wt=0;
inline LL rd(){
    in_char = ' ';
    sign = 1;
    while((in_char < '0') || (in_char > '9')){
        in_char = getchar_unlocked();
        if(in_char == '-'){
            sign = -1;
        }
    }
    next_int = 0;
    while((in_char >= '0') && (in_char <= '9')){
        next_int *= 10;
        next_int += in_char - 48;
        in_char = getchar_unlocked();
    }
    return next_int * sign;
}

int yes(LL m,LL x)
{
	LL chk = (m*m + m + 2)/2;
	if(chk >= x) return 1;
	else return 0;
}

int main()
{
	LL t = rd();
	while(t--)
	{
		LL x = rd();
		LL lo = 0;
		LL hi = 1 + sqrt(2*x);
		while(lo < hi)
		{
			LL mid = (lo + hi)/2;
			if(yes(mid,x)) hi = mid;
			else lo = mid + 1;
		}
		printf("%lld\n",lo);
	}
	return 0;
}
