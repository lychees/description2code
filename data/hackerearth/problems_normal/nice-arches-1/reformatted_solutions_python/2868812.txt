t = input()
c = 0
for j in range(t):
    a = raw_input()
    s = []
    f = 0
    for i in range(len(a)):
        if len(s) == 0:
            s.append(a[i])
            f = f + 1
        elif s[f-1] == a[i]:
            s.pop(f-1)
            f = f - 1
        else:
            s.append(a[i])
            f = f + 1
    if len(s) == 0:
        c = c + 1

print c
