#include <iostream>
using namespace std;
int  ncr(int n,int r)
{
	 int ans=1,i;
	for(i=0;i<r;i++) ans*=(n-i);
	for(i=r;i>1;i--) ans/=i;
	return ans;
}
long long fact(int n)
{
	long long int ans;
	if(n==1 || n==0) return 1;
	else
	{ans=n*fact(n-1);
	return ans;}
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int con,vow;
    	long long an;
    	cin>>con>>vow;
    	an=ncr(7,con)*ncr(5,vow)*fact(con+vow);
    	cout<<an<<endl;
    }
    
    return 0;
}
