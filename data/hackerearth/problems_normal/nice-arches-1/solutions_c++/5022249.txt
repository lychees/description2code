#include<iostream>
#include<stack>
#include<cstring>

using namespace std;

int main()
{

stack<char> Stack;
string str;
int T;
int count=0;
cin>>T;
for( int j=0 ; j<T ; j++)
{
cin >>str;
while(!Stack.empty())
{
Stack.pop();
}
for(int i = 0 ; i < str.length() ; i++)
{
//char s = Stack.top();
//cout<<str[i];
if(Stack.empty())
{
Stack.push(str[i]);
continue;
}


if(Stack.top() == str[i])
Stack.pop();

else
Stack.push(str[i]);


}

if(Stack.empty())
count=count+1;
else
continue;

}


cout<<count;
/*

cout<<Stack.top();
*/
return 0;
}
