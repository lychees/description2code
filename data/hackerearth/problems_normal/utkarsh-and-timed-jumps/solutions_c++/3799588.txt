#include <iostream>
using namespace std;
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		unsigned long long mid,n,ans,l,u;
		cin>>n;
		ans=10000000000;
		l=0,u=1000000000;
		while(l<=u)
		{
			mid=(l+u)/2;
			if((mid*(mid+1)/2)>=n)
			{
				ans=mid;
				u=mid-1;
			}
			else
			{
				l=mid+1;
			}
			
		}
		cout<<ans<<endl;
	}
    return 0;
}
