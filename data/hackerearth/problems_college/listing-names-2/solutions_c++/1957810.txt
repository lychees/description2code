#include<bits/stdc++.h>
using namespace std;
int main(void){
	int t;cin>>t;
	string arr[t];for(int i=0;i<t;i++) cin>>arr[i];
	sort(arr,arr+t);
	int count=1;
	for(int i=0;i<t;i++)
		if(arr[i]==arr[i+1])
		count++;
		else{
			cout<<arr[i]<<" "<<count<<endl;
			count=1;
		}
	return 0;
}
