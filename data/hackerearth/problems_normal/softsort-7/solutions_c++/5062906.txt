#include <iostream>
using namespace std;

int main()
{
	long long int fact[10000001], i, MOD = 1000000007;
	fact[0] = 1;
	for(i=1; i<1000000; i++)
		fact[i] = ( fact[i-1] * i ) % MOD;
	
    long int T;
    cin >> T;
    while(T--)
    {
    	unsigned long long int N, ans;
    	cin >> N;
    	ans = ( 3 * fact[N] + 3 ) % MOD;
    	cout << ans <<endl;
    }
    return 0;
}
