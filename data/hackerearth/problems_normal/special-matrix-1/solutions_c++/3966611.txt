#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		int n;
   		cin>>n;
   		int pos1,pos2;
   		
   		for(int i=0;i<n;i++)
   		{
   			for(int j=0;j<n;j++)
   			{
   				char c;
   				cin>>c;
   				if(c=='*')
   				{
   					pos1=i;
   					pos2=j;
   				}
   			}
   		}
   		
   		n=n/2;
   		cout<<abs(n-pos1)+abs(n-pos2)<<endl;
   	}
    return 0;
}
