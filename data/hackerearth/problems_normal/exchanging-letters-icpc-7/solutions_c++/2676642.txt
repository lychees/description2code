const long long mod = 1e9 + 7;
const long long INF = 1LL << 57LL;
const bool JUDGE = false;
int C[2005][2005];
void binomialCoeff(int n, int k)
{

    int i, j;
 
    // Caculate value of Binomial Coefficient in bottom up manner
    for (i = 0; i <= n; i++)
    {
        for (j = 0; j <= min(i, k); j++)
        {
            // Base Cases
            if (j == 0 || j == i)
                C[i][j] = 1;
 
            // Calculate value using previosly stored values
            else
                C[i][j] = C[i-1][j-1] + C[i-1][j];
			if (C[i][j] >= mod) C[i][j] -= mod;
        }
    }
 
}
int main(){
	csl;

	}
	cin >> n >> k >> t;
	ll sum = 0;
	binomialCoeff(2000, 2000);
	for (int i = 0; i < k; ++i) {
		cin >> x;
		sum += x;
	}
	if (abs(sum - t) > k || abs(sum - t) % 2 != k % 2) {
		cout << 0 << '\n';
	}
	else {
		int x = k - abs(sum - t);
		x /= 2;
		cout << C[k][x] << '\n';
	}
	return 0;
}
