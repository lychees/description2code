#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define ff first
#define ss second
#define rz resize
#define len length
#define nl '\n'

typedef long long ll;
typedef pair<int,int> pii;
typedef vector<int> vi;
typedef vector<pii> vpii;

ll n,x[4000],y[4000];
double f[4000],dp[4000];

inline double dist(int&i,int&j){
  return sqrt( (x[i]-x[j])*(x[i]-x[j]) + (y[i]-y[j])*(y[i]-y[j]) );
}

int main(){

  cin>>n;
  int i,j;
  for(i=1;i<=n;i++)
    scanf("%lld%lld%lf",x+i,y+i,f+i);

  dp[1]= f[1];

  for(i=2;i<=n;i++){

    double op=-1e18;
    for(j=1;j<i;j++){
      op = max(op,dp[j]-dist(i,j));
    }

    dp[i] = f[i]+op;
  }

  printf("%.6lf\n",dp[n]);

  return 0;
};
