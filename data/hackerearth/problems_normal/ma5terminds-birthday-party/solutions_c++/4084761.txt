ll fast_exp(ll a, ll b)
{
    ll x=1,y=a;
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>=MOD) x%=MOD;
        }
        y = (y*y);
        if(y>=MOD) y%=MOD;
        b /= 2;
    }
    return x;
}

 ll GCD(ll a, ll b)
{
    if (a == 0)
        return b;
    return GCD(b%a, a);
}

l pcount[1000005];

l is_prime[1000005];
void sieve(ll N)
{
    MEM(is_prime,1);
    is_prime[1]=is_prime[0]=0;
    l i,j;
    for(i=2;i*i<=N;++i)
    {
        if(is_prime[i])
        {
            for(j=2*i;j<=N;j+=i)
                is_prime[j]=0;
        }
    }
    for(i=2;i<=N;++i)
        if(is_prime[i])
            pcount[i]=pcount[i-1]+1;
        else
            pcount[i]=pcount[i-1];
}
