#include <iostream>
#include <string>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int main()
{
    int n;
    string s;
    set<string> group;
    
    cin >> n;
    while (n--)
    {
    	cin >> s;
    	sort(s.begin() + 1, s.end() - 1);
    	group.insert(s);
    }
    
    cout << group.size() << endl;
    
    return 0;
}
