#include <stdio.h>
#include<bits/stdc++.h>
using namespace std;
#define ll long long int 
ll mod = 1000003ll;
ll modexp(ll a,ll b)
{
	if(b==0ll)
	return 1ll;
	else if(b==1ll)
	return a ;
	else if(b&1)
	return (a*modexp(a,b-1))%mod;
	else
	return modexp((a*a)%mod,b/2);
}
ll modmul(ll a,ll b)
{
	if(b==0ll)
	return 0ll;
	else
	if(b==1)
	return a;
	else
	if(b&1)
	return (a+modmul(a,b-1))%mod;
	else
	return 
	modmul((2*a)%mod,b/2);
	
}
ll prefact[100004];
int main()
{
	prefact[0] =1ll;
for(ll i=1;i<=100003;i+=1ll)
prefact[i] = (prefact[i-1]*i)%mod;
int t;
cin>>t;
ll n,k;
while(t--)
{cin>>n>>k;
if(n>=1000003)
cout<<"0"<<endl;
else
cout<<modmul(prefact[n],k)<<endl;	
}
    return 0;
}
