#include <stdio.h>
#include <stdlib.h>
#include <string.h>

using namespace std;

long long getDigitSum(long long x)
{
	long long res = 0;
	while(x)
	{
		res += x%10;
		x/=10;
	}
	return res;
}

int main()
{
	int t;
	long long n, m;
	long long f[202];
	bool res[202];
	bool temp[202];

	scanf("%d", &t);

	for(long long i = 0; i<=200; i++)
	{
		long long sqr = i*i;
		f[i] = getDigitSum(sqr);
	}

	for(int i = 0; i<=200; i++)
	{
		memset(temp, false, sizeof(bool)*202);
		m = i;
		temp[m] = true;
		while(m!=1 && m!=4)
		{
			m = (int)f[m];
			if(temp[m])
				break;
			temp[m] = true;
		}

		if( m==1 || m==4)
			res[i] = true;
		else
			res[i] = false;
	}

	while(t--)
	{
		scanf("%lld", &n);
		if( n==1 || n==4)
		{
			printf("YES\n");
			continue;
		}
		long long sqr = n*n;
		m = getDigitSum(sqr);
		
		if(res[m])
			printf("YES\n");
		else
			printf("NO\n");
	}

	return 0;
}
