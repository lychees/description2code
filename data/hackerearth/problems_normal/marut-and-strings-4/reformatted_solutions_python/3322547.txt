import sys
test = raw_input()
try:
    test = int(test)
except ValueError:
    print "Invalid Test"
    sys.exit()
if test not in range(1, 11):
    print "Invalid Test"
else:
    for i in range(test):
        s = raw_input()
        if len(s) not in range(1, 101):
            print "Invalid Input"
        elif any(c.isalpha() for c in s):
            tot = sum(1 for c in s if c.isupper())
            cou = sum(1 for c in s if c.islower())
            if tot < cou:
                print tot
            else:
                print cou
        else:
            print "Invalid Input"
