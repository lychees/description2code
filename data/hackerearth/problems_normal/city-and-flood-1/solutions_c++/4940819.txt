#include <bits/stdc++.h>

using namespace std;

int root(int arr[], int i){
while(i!=arr[i])
i=arr[i];
return i;
}

void u(int Arr[ ] ,int A ,int B)
{
    int root_A = root(Arr, A);
    int root_B = root(Arr, B);
    Arr[ root_A ] = root_B ;
}

bool find(int arr[], int A,int B)
{
    if( root(arr, A)==root(arr, B) )
    return true;
    else
    return false;
}


int main(){

int n,m;

cin>>n>>m;


int v[n];

for(int i = 0;i<n;++i){
    v[i] = i;

}

for(int i = 0;i<m;++i){
int a, b;
cin>>a>>b;
u(v,a-1,b-1);
}

set<int> s;

for(int i = 0;i<n;++i)
s.insert(root(v,i));


cout<<s.size()<<endl;

return 0;
}
