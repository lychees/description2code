#include<iostream>
#include<vector>
#include<cmath>
using namespace std;
void getSumMatrix(vector<vector<long long int> > &v,vector<vector<long long> > &sumMatrix,long long int n,long long int k)
{
   for (int j=0; j<n; j++)
   {
       long long int sum = 0;
       for (int i=0; i<k; i++)
          sum += v[i][j];
       sumMatrix[0][j] = sum;
       for (int i=1; i<n-k+1; i++)
       {
            sum += (v[i+k-1][j] - v[i-1][j]);
            sumMatrix[i][j] = sum;
       }
   }
}
long long int getCount(vector<vector<long long int> > &sumMatrix,long long int n,long long int k)
{
   long long int cnt = 0;
   for (int i=0; i<n-k+1; i++)
   {
      long long int sum = 0;
      for (int j = 0; j<k; j++)
           sum += sumMatrix[i][j];
      long long int a = cbrt(sum);
      if(a*a*a == sum || (a-1)*(a-1)*(a-1)==sum || (a+1)*(a+1)*(a+1)==sum)
        ++cnt;
      for (int j=1; j<n-k+1; j++)
      {
         sum += (sumMatrix[i][j+k-1] - sumMatrix[i][j-1]);
        long long int a = cbrt(sum);
        if(a*a*a == sum || (a-1)*(a-1)*(a-1)==sum || (a+1)*(a+1)*(a+1)==sum)
         ++cnt;
      }
   }
   return cnt;
}
int main()
{
    long long int n,k;
    cin>>n>>k;
    vector<vector<long long int> > v(n+1,vector<long long int>(n+1,0));
    for(int i = 0; i < n; ++i)
    {
        for(int j = 0; j < n; ++j)
            cin>>v[i][j];
    }
    vector<vector<long long int> > sumMatrix(n+1,vector<long long int>(n+1,0));
    getSumMatrix(v,sumMatrix,n,k);
    cout<<getCount(sumMatrix,n,k);
    return 0;
}
