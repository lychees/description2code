#include<iostream>
#include<algorithm>
#include<vector>
#include<string>
#include<cstring>
#include<memory>
#include<math.h>
#include<iso646.h>

using namespace std;

int main()
{
    long int n,m,out=0;
    cin>>n>>m;
    vector<long int> a(n);
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a.begin(),a.end());
    int i=1;
    while(true)
    {
        if(binary_search(a.begin(),a.end(),(m*i)))
            {
                out=(m*i);
                i++;
            }
        else{
            break;
        }
    }
    cout<<out<<endl;
return 0;
}
