#include <iostream>
using namespace std;

int main()
{
    int n,h[26]={0},count=0;
    cin>>n;
    while(n--)
    {
    	string s;
    	cin>>s;
    	h[s[0]-'a']++;
    }
    for(int i=0;i<26;i++)
    {
    	if(h[i]!=0)
    	{
    		count++;
    		if(h[i]>10)
    		{
    			while(h[i]>10)
    			{
    				count++;
    				h[i]-=10;
    			}
    		}
    	}
    }
    cout<<count<<endl;
    return 0;
}
