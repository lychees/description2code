#include<bits/stdc++.h>
using namespace std;
int ar[100005];
int h[100005];
int mod=1e9+7;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        long long total=0;
        memset(h,0,sizeof(h));
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&ar[i]);
        int maxi=0;
        for(i=0;i<n;i++)
        {
            if(ar[i] > maxi) maxi = ar[i];
            h[i] = maxi;
        }

maxi = 0;
for(i=n-1;i>=0;i--)
{
    // If the ar is greater than or equal to the max, all water escapes.
    if(ar[i] >= maxi)
    {
        maxi = ar[i];
    }
    else
    {
        if(maxi > h[i])
        {
            // Make sure we don't run off the other side.
            if(h[i] > ar[i])
            {
                total =(total%mod+ (h[i] - ar[i])%mod)%mod;
            }
        }
        else
        {
            total =(total%mod+ (maxi - ar[i])%mod)%mod;
        }
    }
}
        printf("%lld\n",total);
    }
    return 0;
}
