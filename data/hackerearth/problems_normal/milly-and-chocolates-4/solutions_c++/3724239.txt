    #include <iostream>
    using namespace std;
     
    int main()
    {
        int T;
        cin>>T;
        while(T--){
        	int N;
        	cin>>N;
        	int sum =0;
        	for(int i=0;i<N;i++){
        		int a;
        		cin>>a;
        		sum=sum+a;
        	}
        	 double ans = ( double) sum/( double)N;
        	printf("%.8lf\n",ans);
        }
        return 0;
    }
