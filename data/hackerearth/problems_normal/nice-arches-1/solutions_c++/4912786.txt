#include <bits/stdc++.h>
using namespace std;

int main()
{
    int m, i, j;
    cin>>m;
    string s;
    int count = 0;
    
    for(j=0; j<m; j++)
    {
    	cin>>s;
    	stack<char> st;
    	
    	for(i=0; i<s.length(); i++)
    	{
    		
    		if(s[i]=='A')
    		{
    			if(st.empty())
    				st.push('A');
    				
    			else if(st.top()=='A')
    				st.pop();
    			
    			else
    				st.push('A');
    		}
    		
    		else
    		{
    			if(st.empty())
    				st.push('B');
    			
    			else if(st.top()=='B')
    				st.pop();
    			else
    				st.push('B');
    		}
    	}
    	
    	if(st.empty())
    		count++;
    }
    
    cout<<count<<endl;
}
