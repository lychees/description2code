t = input()
for _ in xrange(t):
    n, m = map(int, raw_input().strip().split())
    candies = map(int, raw_input().strip().split())
    in_class = set(candies[:n])
    for ele in candies[n:]:
        if ele in in_class:
            print "YES"
        else:
            print "NO"
        in_class.add(ele)
