using namespace std;



int n,m;
int A[ 131072 ], B[ 131072 ];

bool f(int lim)
{
    int j = 0;
    for(int i=0; i<n;i++)
    {
        if(j>=m)return false;
        if(B[j]-A[i]>lim)return false;
        while(j<m && A[i]-B[j]>lim)j++;
        if(j>=m)return false;
        if(B[j]-A[i]>lim)return false;
        j++;
    }
    return true;
}


int main()
{
    DRT()
    {
        si(n); si(m);
        for(int i=0; i<n;i++)si(A[i]);
        for(int i=0; i<m;i++)si(B[i]);
        sort(A,A+n);
        sort(B,B+m);
        int L = 0; 
        int R = 1000000000;
        while(R-L>1)
        {
            int M = (L+R)/2;
            if(f(M))R = M; else L = M+1;
        }
        while(!f(L))L++;
        cout<<L<<endl;
    }
    return 0;
}
