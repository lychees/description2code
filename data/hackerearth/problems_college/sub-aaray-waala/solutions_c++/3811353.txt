#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define M 1000000007
LL power(LL a,LL n)
{
	LL b=1;
	while(n)
	{
		if(n&1)
		{
			b=(b*a)%M;
		}
		a=(a*a)%M;
		n=n/2;
	}
	return b;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		LL n,temp,dis=0,ans;
		scanf("%lld",&n);
		set<LL> s;
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&temp);
			s.insert(temp);
		}
		set<LL>::iterator it;
		for(it=s.begin();it!=s.end();it++)
		{
			dis++;
		}
		ans=power(n,dis);
		ans=ans%M;
		printf("%lld\n",ans);
	}
	return 0;
}
