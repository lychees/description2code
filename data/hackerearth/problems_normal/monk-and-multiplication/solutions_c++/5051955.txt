#include<bits/stdc++.h>
using namespace std;

int main()
{
    long int n,i,s,j,item;
    priority_queue <long int> pq;
    cin>>n;
    long int a[n],b[3]={0};
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    s=1;
    for(i=0;i<n;i++)
    {
        pq.push(a[i]);
        if((i==0)||(i==1))
        {
            cout<<"-1"<<endl;
        }
        else
        {
            b[0]=pq.top();
            pq.pop();
            b[1]=pq.top();
            pq.pop();
            b[2]=pq.top();
            s=b[0]*b[1]*b[2];
            cout<<s<<endl;
            pq.push(b[0]);
            pq.push(b[1]);
        }
    }

    return 0;
}
