#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main(){
	
	int t;
	
	scanf("%d",&t);
	
	while(t--){
		
		long long int n;
		
		scanf("%lld",&n);
		
	    //a(n) = floor(n^2/2) - 1.
	    
	    if(n==1) {printf("0\n"); continue;}
	    
	    
	    
	    cout<<(n*n/2)-1<<"\n";
	    
	    
		
	}


return 0;
}
