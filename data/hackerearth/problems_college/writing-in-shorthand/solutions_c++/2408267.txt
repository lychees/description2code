#include <iostream>
#include<vector>
#include<map>
#include<string>
using namespace std;

int main()
{
   int m,n,i;
   cin>>m>>n;
   map<string,string> v;
   map<string,string>::iterator it;
   string a,b;
   
   while(n--)
   {
   	cin>>a>>b;
   v.insert(pair<string,string>(a,b));
   	
   }
   
   vector<string> s; 
   while(m--)
   {
   	cin>>a;
    s.push_back(a);
   	
   
   	
  
   }
   
  for(i=0;i<s.size();i++)
  {
  	b=s[i];
  	for(it=v.begin();it!=v.end();it++)
  	{
  		
  		if((it->first)==b)
  		{
  			a=(it->second);
  			if(b.length()<=a.length())
  			cout<<b<<" ";
  			else
  			cout<<(it->second)<<" ";
  		}
  	}
  }
    return 0;
}
