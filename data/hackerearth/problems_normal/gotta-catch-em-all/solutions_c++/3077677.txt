#include <iostream>
using namespace std;

#define s(n) scanf("%ld",&n);
#define p(n) printf("%ld",n);

void mergesort(long a[],long low,long high);
void merge(long a[],long low,long mid,long high);

int main()
{
    long k,i,a[100000];
    s(k);
    for(i=0;i<k;i++)
   	s(a[i]);
    mergesort(a,0,k-1);	
    for(i=0;i<k;i++)
    a[i]=a[i]+i+2;
    long max=a[0];
    for(i=0;i<k;i++)
    if(a[i]>max) max=a[i];
    p(max);
    return 0;
}

void mergesort(long a[],long low,long high)
{
	long mid;
	if(low<high)
	{
		mid=(low+high)/2;
		mergesort(a,low,mid);
		mergesort(a,mid+1,high);
		merge(a,low,mid,high);
	}
}

void merge(long a[],long low,long mid,long high)
{
	long k,i,l,m,temp[100000];
	i=l=low;
	m=mid+1;
	
	while(l<=mid && m<=high)
	{
		if(a[l]>=a[m])
		temp[i++]=a[l++];
		else 
		temp[i++]=a[m++];
	}
	
	if(l>mid)
	{
		while(m<=high)
		temp[i++]=a[m++];
	}
	else if(m>high)
	{
		while(l<=mid)
		temp[i++]=a[l++];
	}
	
	for(k=low;k<=high;k++)
	a[k]=temp[k];
}
