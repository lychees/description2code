#include <iostream>
using namespace std;
int main()
{
    int n=0;
    cin>>n;
    for(int i =0;i<n;i++){
    	long long co;
    	cin >> co;
    	long long answer=0;
    	for(long long k=2;k<=co;k=k*2){
    		for(long long j=1;j<k;j=j*2){
    			if(k+j<=co){
    				long long temp = (k+j)%1000000007;
    				answer=(answer+temp)%1000000007;
    			}
    		}
    	}
    	cout<<answer<<endl;
    }
}
