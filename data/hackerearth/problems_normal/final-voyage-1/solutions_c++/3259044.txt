#include <iostream>
#include <vector>
using namespace std;
vector<vector<int> > dp;
vector<int> weight,value;
int main()
{
    int t,n,p1,p2,w;
    cin>>t;
    while(t--)
    {
    	dp.clear();
    	value.clear();
    	weight.clear();
    	cin>>n;
    	dp.resize(n+1);
    	value.resize(n+1);
    	weight.resize(n+1);
    	cin>>w;
    	for(p1=0;p1<=n;p1++)
    		dp[p1].resize(w+1,0);
    	for(p1=1;p1<=n;p1++)
    		cin>>weight[p1];
    	for(p1=1;p1<=n;p1++)
    		cin>>value[p1];
    	for(p1=1;p1<=n;p1++)
    		for(p2=1;p2<=w;p2++)
    			if(p2>=weight[p1])
    				dp[p1][p2]=max(dp[p1-1][p2],dp[p1-1][p2-weight[p1]]+value[p1]);
    			else
    				dp[p1][p2]=dp[p1-1][p2];
    	cout<<dp[n][w]<<endl;
    }
    return 0;
}
