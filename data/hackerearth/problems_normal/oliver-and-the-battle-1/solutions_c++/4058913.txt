#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define inf (1<<28)
#define pi 3.141593
#define pb push_back
#define mp make_pair
typedef long long ll;
int main()
{
	ios_base::sync_with_stdio(false);
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,m;
		cin>>n>>m;
		ll arr[n+2][m+2];
		memset(arr,0,sizeof(arr));
		ll i,j,k,l;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				cin>>arr[i][j];
			}
		}
		ll no_troop=0,troops=0,temp;
		ll visit[n+2][m+2];
		memset(visit,0,sizeof(visit));
		queue<pair<ll,ll> > q;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				if(!visit[i][j] && arr[i][j])
				{
					no_troop++;
					//cout<<no_troop<<" troop"<<i<<" "<<j<<endl;
					q.push(mp(i,j));
					visit[i][j]=1;
					temp=1;
					while(!q.empty())
					{
						//cout<<"Inside While\n";
						ll x=q.front().first,y=q.front().second;
						q.pop();
						for(k=-1;k<=1;k++)
						{
							for(l=-1;l<=1;l++)
							{
								if(!visit[x+k][y+l] && arr[x+k][y+l])
								{
									temp++;
									visit[x+k][y+l]=1;
									q.push(mp(x+k,y+l));
								}
							}
						}
						//cout<<temp<<endl;
					}
					troops=max(troops,temp);
				}
			}
		}
		cout<<no_troop<<" "<<troops<<endl;
	}
    return 0;
}
