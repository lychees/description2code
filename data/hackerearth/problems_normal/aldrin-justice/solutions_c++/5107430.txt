#include <iostream>
using namespace std;

int T;
int bt1;
int bt2;
int mt1;
int mt2;

int main()
{
    cin >> T;
    
    int a, b, c, d;
    for (int kase = 0; kase < T; kase++) {
    	cin >> a >> b >> c >> d;
    
        bt1 = min(a, b);
    	bt2 = max(a, b);
    	mt1 = min(c, d);
    	mt2 = max(c, d);
    
    	if (bt1 > mt2 || mt1 > bt2) cout << "Line" << endl;
    	else if (bt1 == mt2 || mt1 == bt2 || bt1 == bt2 || mt1 == mt2) cout << "Point" << endl;
    	else cout << "Nothing" << endl;
    }
    
    return 0;
}
