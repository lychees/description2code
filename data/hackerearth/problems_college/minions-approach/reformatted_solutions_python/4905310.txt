def gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a


def prime(x):
    if x <= 1:
        return False
    if x == 2 or x == 3:
        return True
    for i in range(2, int(x**0.5)+1):
        if x % i == 0:
            return False
    return True


for _ in range(input()):
    n = input()
    count = 0
    for item in range(1, n):
        if gcd(item, n) == 1:
            count += 1
    if prime(count):
        print 'Correct'
    else:
        print 'Incorrect'
