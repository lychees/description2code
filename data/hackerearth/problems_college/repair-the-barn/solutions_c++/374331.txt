#include<stdio.h>
#include<iostream>
#include<list>
using namespace std;
int main()
{
list<int> l;    
list<int>::iterator it;    
    
int maxb,maxg,s,c,x,i;
int min=201,max=0; 
bool *arr; 
cin>>maxb;
maxg=maxb-1;
cin>>s;
arr=new bool[s+1];
cin>>c;
x=c;

while(x--)
{
    cin>>i;
    arr[i]=true;
    if(i<min)
      min=i;
    if(i>max)
      max=i; 
} 

//cout<<min<<":"<<max<<":"<<maxg<<":"<<s<<":"<<c<<endl;
//cin>>c;

c=0;
for(i=min;i<=max;i++)
 {     
   if(arr[i]==true)
    {   
        if(c!=0)
         {
           l.push_back(c);
           c=0;                  
          }
        else
          continue;
    }         
    else
    { 
      c++;                
    } 
 }
 
 l.sort();
 l.reverse();
 
/* for (it=l.begin();it!=l.end();it++)
  { cout<<*it<<",";}
 cout<<endl;*/
 
 int cnt=0;
 int sum=0;
 for (it=l.begin();it!=l.end()&&cnt<maxg;it++,cnt++)
  {
   sum+=*it;
  } 
//cout<<min<<":"<<max<<":"<<maxg<<":"<<sum<<":"<<cnt<<endl;
  
  
cout<<max-min+1-sum<<endl;
 
 
delete[] arr; 
return 0;
}
