#include <bits/stdc++.h>

using namespace std;

int search(int my_numbers[],int N, int key) {
	int left = 0, right = N-1, mid;
	while (left <= right) {
		mid = (int) ((left + right) / 2);
		if (key == my_numbers[mid]) {
			return mid;
		}
	else if (key > my_numbers[mid])
		left = mid + 1;
	else
		right = mid - 1;
	}
	return -1;
}

int main()
{	
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
    int N,Q;
    cin>>N>>Q;
    int A[N],i,j;
    i = 0;
    while(i < N)
    	cin>>A[i++];
    sort(&A[0],&A[N]);
    while(Q--) {
    	cin>>j;
    	if(j > A[N-1] || j < A[0])
    		cout<<"NO";
    	else if(j == A[N-1] || j == A[0])
    		cout<<"YES";
    	else if(search(A,N,j) == -1)
    		cout<<"NO";
    	else
    		cout<<"YES";
    	cout<<"\n";
    }
}
