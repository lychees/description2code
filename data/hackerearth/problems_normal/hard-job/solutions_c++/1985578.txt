#include<bits/stdc++.h>
using namespace std;
int bit[1500010];
int ind[500010];

int read(int idx){
    int sum = 0;
    while (idx > 0){
        sum += bit[idx];
        idx -= (idx & -idx);
    }
    return sum;
}
void update(int idx ,int val){
int MaxVal = 1500000;
    while (idx <= MaxVal){
        bit[idx] += val;
        idx += (idx & -idx);
    }
}


int main()
{
        int n,m,x;
        cin>>n>>m;
        int lft = 500000;
        int rght = lft+n;
        for(int i = 0;  i < n;i++)
        {
            cin>>x;
            ind[x] = lft + i;
            update(ind[x],1);
        }
        char c;
        long long ans = 0;
        while(m--)
        {
            cin>>x>>c;
            int y = read(ind[x]);
            ans = ans +  min(y - 1,n-y);
            update(ind[x],-1);
            if(c == 'l')
            {
                lft--;
                ind[x] = lft;
                update(lft,1);
            }
            else
            {
                    ind[x] = rght;
                    rght++;
                    update(ind[x],1);
            }

        }
        cout<<ans<<endl;

}
