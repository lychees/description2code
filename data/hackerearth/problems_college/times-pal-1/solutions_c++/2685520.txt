#include <bits/stdc++.h>
using namespace std;

bool isPalindrome(string str)
{

	int i=0,j = str.size();j--;
	while(i<j)
	{

		//cout<<str<<" "<<j<< " " ;
		if(str[i]!=str[j])return false;
		i++; j--;
	}
	return true;
}


int main()
{
int t;
	cin>>t;
	while(t--)
	{

		string a, b;
		cin>>a>>b;
		int n = a.size();
        int count = 0;
		string str;
		string x, y;
		for(int i=0; i<=n; i++)
		{

			x = a.substr(0,i);
			y = a.substr(i,n-i);
			str = x+b+y;
			//cout<<str<<" ";
		if(isPalindrome(str))count++;
		}
		cout<<count<<endl;
		
	}
	return 0;
}
