#include<iostream>
#include<algorithm>
using namespace std;
int n,a[50][50];
bool getpath(int i,int j)
{   if(i==n-1&&j==n-1)
	return true;
if(i>n-1||j>n-1)
	return false;
    if(a[i][j]==0)
       return false; 	
	
	if(getpath(i+1,j)||getpath(i,j+1))
	return true;
	return false;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		
		for(int i=0;i<n;i++)
		{for(int j=0;j<n;j++)
		{
			cin>>a[i][j];
			
		}
     	}
       	
		if(getpath(0,1)||getpath(1,0))
		cout<<"POSSIBLE"<<endl;
		else
		cout<<"NOT POSSIBLE"<<endl;
	}
	
	}
