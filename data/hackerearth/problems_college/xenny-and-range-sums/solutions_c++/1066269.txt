#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int n,q,arr[1000003];
	cin>>n>>q;
	for(int i=1;i<=n;i++) {
		cin>>arr[i];
	}
	while(q--) {
		int L,R;
		cin>>L>>R;
		long long sum=arr[R],diff=arr[R];
		for(int i=L;i<R;i++) {
			sum+=arr[i];
			diff-=arr[i];
		}
		cout<<sum<<" "<<diff<<endl;
		
	}
	return 0;
}
