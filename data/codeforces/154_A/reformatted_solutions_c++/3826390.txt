#include <algorithm>
#include <cctype>
#include <cmath>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <math.h>
#include <stdio.h>
#include <string>
using namespace std;
char s[100100];
int main() {
  int k, i;
  long long int j, aa = 0, bb = 0, ans = 0;
  char a, b;
  cin >> s >> k;
  for (i = 1; i <= k; i++) {
    cin >> a >> b;
    for (j = 0; j < strlen(s); j++) {
      aa = 0;
      bb = 0;
      for (; s[j] == a || s[j] == b; j++)
        (s[j] == a) ? aa++ : bb++;
      ans = ans + min(aa, bb);
    }
  }
  cout << ans;
  return 0;
}
