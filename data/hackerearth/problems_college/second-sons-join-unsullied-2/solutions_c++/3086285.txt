#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pi;

#define MP make_pair
#define PB push_back
#define F first
#define S second
#define MOD 1000000007
#define INF (1 << 30)

int main()
{
	ios::sync_with_stdio(false);
	int t, i;
	cin >> t;
	while(t--)
	{
		string s;
		cin >> s;
		int n = s.length();
		string s1, s2;
		int flag = 0;
		int u = 0, S = 0;
		for(i = 0; i < n; i++)
		{
			if(s[i] == 'U')
				u++;
			else
				S++;
		}
		if(abs(u - S) > 1)
		{
			cout << "-1" << endl;
			continue;
		}
		for(i = 0; i < n; i++)
		{
			s1 += flag ? 'U' : 'S';
			s2 += flag ? 'S' : 'U';
			flag ^= 1;
		}
		// cout << s1 << endl;
		// cout << s2 << endl;
		int ans = 0;
		if(u > S)
		{
			for(i = 0; i < n; i++)
			{
				if(s[i] != s2[i])
					ans++;
			}
			cout << ans/2 << endl;
		}
		else if(S > u)
		{
			for(i = 0; i < n; i++)
			{
				if(s[i] != s1[i])
					ans++;
			}
			cout << ans/2 << endl;
		}
		else
		{
			int ans1 = 0;
			int ans2 = 0;
			for(i = 0; i < n; i++)
			{
				if(s[i] != s1[i])
					ans1++;
			}
			for(i = 0; i < n; i++)
			{
				if(s[i] != s2[i])
					ans2++;
			}
			cout << min(ans1/2, ans2/2) << endl;;
		}
	}
	return 0;
}
