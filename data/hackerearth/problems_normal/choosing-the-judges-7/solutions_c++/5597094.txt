#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,j;
	cin>>t;
	for(j=1;j<=t;j++)
	{
		int n,i;
		cin>>n;
		int arr[n];
		for(i=0;i<n;i++)
		cin>>arr[i];
		int dp[n];
		int maxi;
		dp[0]=arr[0];
		maxi=arr[0];
		for(i=1;i<n;i++)
		{
			if(i==1)
			{
				dp[i]=arr[i];
			}
			else
			{
				dp[i]=arr[i]+maxi;
			}
			maxi=max(maxi,dp[i-1]);
		}
		int ans=0;
		for(i=0;i<n;i++)
		{
			ans=max(ans,dp[i]);
		}
		cout<<"Case "<<j<<": "<<ans<<endl;
	}
	return 0;
}
