#include <iostream>
#include <cmath>
#include <cstdlib>
#define EPS 1e-7
using namespace std;

typedef long long LL;

bool IsPrimeSlow (LL x)
{
  if(x<=1) return false;
  if(x<=3) return true;
  if (!(x%2) || !(x%3)) return false;
  LL s=(LL)(sqrt((double)(x))+EPS);
  for(LL i=5;i<=s;i+=6)
  {
    if (!(x%i) || !(x%(i+2))) return false;
  }
  return true;
}

int main()
{
      LL gridSize;
      cin >> gridSize;
      
      if(gridSize == 2)
      {
      	cout << 1 << endl;
      	return 0;
      }

      LL i;
      LL j;
      LL total  = 0;

      LL sum = 2*gridSize - 2;
      LL multiply = 3;
      LL start = 2;
      bool flag = true;
      while( start <= sum)
      {
            if( IsPrimeSlow(start))
            {
                  total = total + multiply * 1;
            }
            
            if( multiply <gridSize && flag == true)
            {
                  multiply++;
            }
            else
            {
                  multiply--;
                  flag = false;
            }

			
            start = start + 1;
      }

      cout << total % 1000000007 << endl;
      return 0;

}
