#include <iostream>
#include <map>
using namespace std;

int main()
{
    int t,n,i,j;
    long ans=0;
    cin>>t;
    map<int,int> mapp;
    while(t--)
    {
    	cin>>n;
    	ans=0;
    	mapp.clear();
    	for(i=0;i<n;i++)
    	{
    		cin>>j;
    		if(mapp.find(j)==mapp.end())
    		{
    			mapp[j]=1;
    		}
    		else
    		{
    			ans=ans+mapp[j];
    			mapp[j]++;
    		}
    	}
    	cout<<ans+n<<endl;
    }
    return 0;
}
