int n, k, t, a[2222], s;
const int mod = 1000000007;
int c[2222][2222];
int main() {

	cin >> n >> k >> t;

	for(int i = 0; i <= n; ++i) {
		//cerr << i << endl;
		c[i][0] = 1;
		for(int j = 1; j <= i; ++j) {
			c[i][j] = c[i - 1][j - 1] + c[i - 1][j];
			if (c[i][j] >= mod) c[i][j] -= mod;
		}
	}
	//cerr << "Here\n";
	for(int i = 0; i < k; ++i) {int x; cin >> x; s += x;}
	//cerr << "Here\n";

	int ans = 0;
	for(int pos = 0; pos <= k; ++pos) {
		if (s + pos - (k - pos) == t) {
			ans += c[k][pos];
			if (ans >= mod) ans -= mod;
		}
	}
	cout << ans << endl;
}
