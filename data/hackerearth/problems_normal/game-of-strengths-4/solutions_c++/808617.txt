#include <iostream>
#include <cmath>
#include <math.h>
#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

int main()
{
    long long int *arr,i,j,n,d,maxm;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&n);
        arr=(long long int*)malloc(n*(sizeof(long long int)));
        for(i=0;i<n;i++)
        {
            scanf("%lld",&arr[i]);
        }
        d=0;
        sort(arr,arr+n);
        maxm=arr[n-1];
        i=n-1;
        j=i;
        while(i>=0)
       {
           d=d+(i*(arr[i]-arr[j-i]))%1000000007;
           i--;
       }

         d%=1000000007;
         maxm%=1000000007;
         printf("%lld\n",((d*maxm)%1000000007));
    }

    return 0;
}
