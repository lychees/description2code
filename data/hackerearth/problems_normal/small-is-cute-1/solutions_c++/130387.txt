#include <iostream>
using namespace std;
typedef unsigned long long int ull;
int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}	
int main()
{
    int a,b,n;
    cin>>n>>a>>b;
    int modA=n%a;
    int modB=n%b;
    int lastA=n-modA;
    int lastB=n-modB;
    int hcf=gcd(a,b);
    ull tmp=(a*b);
    ull lcm=tmp/hcf;
    int modL=n%lcm;
    int lastL=n-modL;
    ull noA=n/a;
    ull noB=n/b;
    ull noL=n/lcm;
    if(modA==0)
    {
    	lastA=lastA-a;
    	noA--;
    }	
    if(modB==0)
    {
    	lastB-=b;
    	noB--;
    }	
    if(modL==0)
    {
    	lastL-=lcm;
    	noL--;
    }	
    ull sumA=((noA)*(a+lastA))/2;
    ull sumB=((noB)*(b+lastB))/2;
    ull sumL=((noL)*(lcm+lastL))/2;
    //cout<<"noA="<<noA<<"lastA="<<lastA<<"sumA="<<sumA<<"noB="<<noB<<"lastB="<<lastB<<"sumB="<<sumB;
    ull sum=sumA+sumB-sumL;
    cout<<sum<<endl;
    return 0;
}
