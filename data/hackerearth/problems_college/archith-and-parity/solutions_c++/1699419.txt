#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	long long int n;
    	cin>>n;
    	long long int sum=0;
    	while(n)
    	{
    		if(n%2!=0)
    		{
    			sum++;
    		}
    		n=n/2;
    	}
    
    if(sum%2==0)
    cout<<"even"<<endl;
    else
    cout<<"odd"<<endl;
    }
    return 0;
}
