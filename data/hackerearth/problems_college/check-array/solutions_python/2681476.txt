T=input()
for j in xrange(T):
    n=input()
    s=map(int,raw_input().split())
    a=s[:]
    for i in xrange(1,n-1):
        a[i]+=a[i-1]
    b=s[:]
    for i in xrange(n-2,-1,-1):
        b[i]+=b[i+1]
    for i in xrange(1,n-1):
        if a[i]==b[i]:
            print "YES"
            break
    else:
        print "NO"
