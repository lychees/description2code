#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <ctime>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <sstream>
#include <string.h>
#include <string>
#include <utility>
#include <vector>
using namespace std;

int used[10000];

int main() {
  int i, j, n, s = 0;
  string del, t;
  cin >> n;
  vector<string> ss, res;
  for (i = 0; i < n; i++) {
    cin >> t;
    s += t.size();
    ss.push_back(t);
  }
  cin >> del;
  s /= n / 2;

  sort(ss.begin(), ss.end());

  for (i = 0; i < n; i++) {
    if (used[i])
      continue;
    for (j = i + 1; j < n; j++)
      if (!used[j] && ss[i].size() + ss[j].size() == s)
        break;
    used[j] = 1;
    string s1 = ss[i] + del + ss[j], s2 = ss[j] + del + ss[i];
    res.push_back(s1 < s2 ? s1 : s2);
  }

  sort(res.begin(), res.end());

  for (i = 0; i < res.size(); i++)
    cout << res[i] << endl;

  return 0;
}
