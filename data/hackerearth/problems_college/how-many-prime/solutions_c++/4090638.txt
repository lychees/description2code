#include <iostream>
using namespace std;

int main()
{
    int t,a,i,j,b,k,d,e,c;
    cin>>t;
    while(t--){
    	cin>>a;
    	for(i=(a/2);i>1;i--)
    	{
    		c=0;
    		for(j=2;j<i;j++)
    		{
    			if(i%j==0)
    			{c=1;break;}
    		}
    		if(c==0){
    		b=a-i;
    		d=0;
    		if(b==i){}
    		else{
    		for(k=2;k<b;k++){
    		if(b%k==0)
    		{d=1;break;}
    		}
    		if(d==0){
    			e=i;
    			break;
    		}}
    	}
    	}
    	if(d==1)
    	cout<<-1<<" "<<-1<<endl;
    	else
    	cout<<e<<" "<<b<<endl;
    }
    return 0;
}
