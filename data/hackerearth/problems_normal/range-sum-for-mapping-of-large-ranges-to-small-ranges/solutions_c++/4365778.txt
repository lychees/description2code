#include <bits/stdc++.h>
using namespace std;
#define ll long long
const int MOD = (int) 1e9 + 7;

vector<vector<int> > adj;
bool visit[100002];
int n,m,total;


void dfs(int s){
	visit[s]=true;
	total += 1;
	for(int j=0;j<adj[s].size();j++){
		if(!visit[adj[s][j]]){
			dfs(adj[s][j]);
		}
	}
}

int main(){
	int T;
	cin>>T;
	while(T--){
		cin>>n>>m;
		adj.clear();
		adj.resize(n+2);

		for(int i=0;i<m;i++){
			int x,y;
			cin>>x>>y;
			adj[x].push_back(y);
			adj[y].push_back(x);
		}

		
		memset(visit,false,sizeof(visit));

		ll ans=1;
		for(int i=1;i<=n;i++){
			if(!visit[i]){
				total=0;
				dfs(i);
				if(total>1)
					ans = (ans * (((ll)total*(total-1))%MOD)/2)%MOD;
			}
		}

		cout<<ans<<endl;
	}
	return 0;
}
