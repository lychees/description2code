		cin>>n;
		set< pair< pair<ll,ll> ,ll > > s;
		map< pair<ll,ll> ,ll > m;
		forp(i,1,n){
			cin>>a>>b>>c;
			x=__gcd(a,b);
			x=__gcd(x,c);
			a/=x;
			b/=x;
			c/=x;
			temp=s.size();
			s.insert(mp(mp(a,b),c));
			if(temp!=s.size())
			{
				long long z=__gcd(a,b);
				a/=z;
				b/=z;
				m[mp(a,b)]++;
			}
		}
		ll ans=0;
		tr(m, it){
			ans=max(ans,it->ss);
		}
		cout<<ans<<endl;
	}
	return 0;
}
