#include <iostream>
#include <vector>
using namespace std;
 
vector<unsigned> ans(1000001, 0);
 
void prepare() {
	vector<unsigned> primes;
	primes.reserve(300);
	vector<int> sieve(1000, 0);
	primes.push_back(2);
	for (unsigned i = 3; i < 1000; i += 2)
		if (sieve[i] == 0) {
			primes.push_back(i);
			for (unsigned j = i * i; j < 1000; j += 2 * i)
				sieve[j] = 1;
		}
		
	
	unsigned count = 0;
	ans[1] = count;
	
	for (unsigned i = 2; i <= 1000000; ++i) {
		unsigned j = i;
		for (unsigned p: primes) 
			while (j % p == 0) {
				++count;
				j /= p;
			}
		if (j > 1)
			++count;
		ans[i] = count;
	}
}
 
int main() {
	ios_base::sync_with_stdio(0);
    cin.tie(0);
	
	prepare();
	int t;
	cin >> t;
	while (t--) {
		unsigned x;
		cin >> x;
		cout << ans[x] << '\n';
	}
    return 0;
}
