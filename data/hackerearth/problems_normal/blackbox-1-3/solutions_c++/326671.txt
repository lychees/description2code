
#include<iostream>
#include<limits.h>
using namespace std;

void coprime(int i, int j, int k);

int main()
{
	int a,b,c,T;
	long long int max, min1, min2;
	cin >> T;
	while(T)
	{
		
		cin >> a >> b >> c;
		
		max = a>b? a:b;
		min1 = a>b? b:a;
		min2 = max>c? c: max;
		max = max>c? max:c;
		
		if(min1*min1 == (max+min2)*(max-min2))
			coprime(a,b,c);
		else
			cout << "NO" << endl;
		T--;
	}

	return 0;
}

void coprime(int i, int j, int k)
{
	int min;
	min = INT_MAX;
	
	while(min)
	{
		min = min>i ? i:min;
		min = min>j?j:min;
		min = min>k?k:min;
		
		i = (i==min)?i:(i-min);
		j = (j==min)?j:(j-min);
		k = (k==min)?k:(k-min);
		
		if(i==j && j==k)
			min = 0;		
	}
	
	if(i==1)
		cout << "YES" << endl;
	else
		cout << "NO" << endl;
}
