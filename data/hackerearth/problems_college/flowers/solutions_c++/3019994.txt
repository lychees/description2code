#include <iostream>
#include <algorithm>
using namespace std;

void print(int C[], int N){
	for (int i=0; i<N; i++){
		cout << C[i] << " ";
		}
	cout << "\n";
	}

int cost(int C[], int N, int K){
	//Sort the costs in decreasing order
	sort(C, C+N, greater<int>());
	
	//Calculate the costs for the K friends
	int result=0, evendist, remainingflowers;
	//Calculate the least number of flowers that each friend gets
	evendist = N/K; remainingflowers = N%K;
	
	//Now add the cost for the evenly distributed flowers
	for (int i=0; i<evendist; i++){
		for (int j=0; j<K; j++){
			result += (i+1)*C[j+K*i];
			}
		}

	//Finally add the cost of the remaining flowers
	for (int i=0; i<remainingflowers; i++){
		result += (evendist+1)*C[i+evendist*K];
		}
	return result;
	}

int main(){
	int N, K;
	cin >> N >> K;
	int C[N];

	for (int i=0; i<N; i++){
		cin >> C[i];
		}
	cout << cost(C, N, K); 
	return 0;
	}
