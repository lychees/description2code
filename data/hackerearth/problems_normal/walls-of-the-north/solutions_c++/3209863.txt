int bit[2][100005];
int n;
int UU;

void upd(int ii, int x, int v) {
  if (ii) x = UU+1-x;
  for(int i = x; i <= UU; i += (i&-i)) {
    MAX(bit[ii][i], v);
  }
}

int que(int ii, int x) {
  if (ii) x = UU+1-x;
  int res = 0;
  for(int i = x; i > 0; i -= (i&-i)) {
    res = max(res, bit[ii][i]);
  }
  return res;
}

int x[100005];

int main() {
  int tc;
  scanf("%d",&tc);
  while(tc--) {
    scanf("%d",&n);
    int ans = 0;
    vector<int> v;
    FOR(a,1,n) {
      scanf("%d",&x[a]);
      v.pb(x[a]);
      bit[0][a] = bit[1][a] = 0;
    }
    SORT(v);
    UNIQUE(v);
    FOR(a,1,n) {
      x[a] = lower_bound(v.begin(), v.end(), x[a])-v.begin()+1;
    }
    UU = SIZE(v);

    FOR(a,1,n) {
      int val1 = que(1, x[a]) + 1;
      int val0 = que(0, x[a]) + 1;
      upd(1, x[a], val0);
      upd(0, x[a], val1);
      MAX(ans, val1);
      MAX(ans, val0);
      //cout << x << " " << val1 << " " << val0 << endl;
    }
    printf("%d\n", ans);
  }
}
