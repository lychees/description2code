#include <iostream>
using namespace std;

int main()
{
    int t;
        cin >> t;
        while (t--) {
            string str;
            cin >> str;
            int len = str.length();
            int res = len - 4;
            for (size_t i = 4; i < len - 4; i++) {
                if (str[i] == 'a' || str[i] == 'e' || str[i] == 'i' || str[i] == 'o' || str[i] == 'u') {
                    res--;
                }
            }
            cout << res << "/" << len << endl;
        }
    return 0;
}
