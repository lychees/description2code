#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[200005];
ll ways(int n)
{
	ll fir,sec;
   	fir = n;
   	sec = n-1;
   	if(n%2==0)fir/=2;
   	else sec/=2;
   	return fir*sec;
}
int main()
{
   // cout << "Hello World!" << endl;
   int t;
   scanf("%d",&t);
   while(t--)
   {
   	int n;
   	scanf("%d",&n);
   	
   	ll total = ways(n);
   	int i;
   	for(i=0;i<n;i++)
   	scanf("%d",&a[i]);
   	set< pair<int,int> > bad;
   	int m,x,y;
   	scanf("%d",&m);
   	for(i=0;i<m;i++)
   	{
   		scanf("%d%d",&x,&y);
   		bad.insert(make_pair(x,y));
   		bad.insert(make_pair(y,x));
   	}
   	ll good = 0;
   	int j;
   	j = 0;
   	for(i=1;i<n;i++)
   	{
   		pair<int,int> cur = make_pair(a[i],a[i-1]);
   		if(bad.find(cur)!=bad.end())
   		{
   			int l  = i-j;
   		//	printf("")
   			good += ways(l);
   			j = i;
   		}
   	}
   int	l = i-j;
   	good+=ways(l);
   	printf("%lld\n",total-good);
   }
    return 0;
}
