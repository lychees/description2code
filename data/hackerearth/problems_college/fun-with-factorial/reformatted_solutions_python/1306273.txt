def fact(n):
    f = 1
    if (n >= 25):
        return "00000"
    else:
        for i in range(1, n+1):
            f = f*i
        if (len(str(f)) > 5):
            return str(f)[len(str(f))-5:len(str(f))]
        else:
            return f


for o in range(input()):
    print str(fact(input())).rjust(5, "0")
