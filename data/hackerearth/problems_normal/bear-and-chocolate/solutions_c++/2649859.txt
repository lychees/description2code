#include <iostream>
using namespace std;
char mat[1002][1002];
int main()
{
	int t;
	cin >>t;
	int n;
	while(t--)
	{
		int count=0,ans=0;
		cin >>n;
		int temp=0;
		for(int i=0;i<n;i++)
			cin >>mat[i];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(mat[i][j]=='#')
					count++;
			}
		}
		temp=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(mat[i][j]=='#')
				temp++;
			}
			if(temp*2==count)
				ans=1;
		}
		temp=0;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(mat[j][i]=='#')
					temp++;
			}
			if(temp*2==count)
				ans=1;
		}
		if(ans==1)
			cout << "YES"<<endl;
		else
			cout <<"NO"<<endl;
	}
	
	return 0;
}
