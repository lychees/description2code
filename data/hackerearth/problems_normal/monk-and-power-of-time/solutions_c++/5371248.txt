#include <iostream>
#include<queue>
using namespace std;

int main()
{
	int N;
	cin>>N;
	queue<int> call;
	queue<int> ideal;
	int x;
	for(int i = 0 ; i < N  ; i++ ){
		cin>>x;
		call.push(x);
	}
	for(int i = 0 ; i < N ; i++ ){
		cin>>x;
		ideal.push(x);
	}
	int count = 0;
	while( ! call.empty() ){
		while( call.front() != ideal.front() ){
			x = call.front();
			call.pop();
			call.push(x);
			count++;
		}
		ideal.pop();
		call.pop();
		count++;
	}
	cout<<count;
    return 0;
}
