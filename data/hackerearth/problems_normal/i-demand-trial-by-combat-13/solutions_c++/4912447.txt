#include <iostream>
using namespace std;

int main()
{
    int t, i, j;
    cin>>t;
    while(t--)
    {
    	int n, hours;
    	cin>>n>>hours;
    	int arr[n];
    
    	for(i=0; i<n; i++)
    		cin>>arr[i];
    
    	int prev;
    	bool check;
    	
    	for(i=1; i<=hours; i++)
    	{
    		check = false;
    		
    		prev = arr[0];
    		if(arr[1] == 1)
    		{
    			arr[0] = 1;
    			check = true;
    		}
    		else
    			arr[0] = 0;
    		
    		for(j=1; j<=n-2; j++)
    		{
    			if(prev==1 && arr[j+1]==1)
    			{
    				prev = arr[j];
    				check = true;
    				arr[j] = 1;
    			}
    			else
    			{
    				prev = arr[j];
    				arr[j] = 0;
    			}
    		}
    		if(prev==1)
    		{
    			arr[j] = 1;
    			check = true;
    		}
    		else 
    			arr[j] = 0;
    		
    		if(!check)
    			break;
    	}
    	
    	for(i=0; i<n; i++)
    		cout<<arr[i]<<" ";
    	cout<<endl;
    }
}
