#include <bits/stdc++.h>
 
using namespace std;
#define LL_max 200000000000
#define mod 1000000007
 
#define LL long long
#define mp make_pair
#define pb push_back
 
 
 
int main()
{
    int t;
    cin>> t;
 
    while(t--)
    {
        int n, m, a;
        cin>> n>> m>> a;
 
 
        LL p = a*a;
        LL pp =  p* a;
        LL vol = n+m*pp;
        LL best = LL_max;
 
        for(long r = a; r*r*r<=best; r++)
        {
            long R = r/a;
            for(long c = r; r*c *c<=best; c++)
            {
                LL C= c/a;
                LL h=(m+R*C-1)/(R*C)*a;
                LL V = r*c*h;
                LL h2 = max(0LL, vol- V);
                h +=(h2+r*c-1)/(r*c);
                best = min(best, r*c*h);
            }
        }
        cout << (int)best<< endl;
    }
 
    return 0;
}
