#include<bits/stdc++.h>
#define lli long long int
#define MAX 100005
#define MOD 1000000007
lli arr[MAX];
using namespace std;
int main()
{
        lli i,j,k,m,n,t;
        arr[1]=1;
        arr[2]=2;
        arr[3]=4;
        for(i=4;i<MAX;i++)
                arr[i]=(arr[i-1]+arr[i-2]+arr[i-3])%MOD;

        scanf("%lld",&t);
        while(t--)
        {
                scanf("%lld",&n);
                printf("%lld\n",arr[n]);
        }
}
