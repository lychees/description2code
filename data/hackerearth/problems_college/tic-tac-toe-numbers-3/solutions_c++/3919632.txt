#include <iostream>
#include <cmath>
using namespace std;
bool isPr(long n){
    long i;
    int flag=0;
    for(i=2;i<=sqrt(n);++i)
    {
        if(n%i==0)
        {
          flag=1;
          break;
        }
    }
    if (flag==0)
      return 1;
    else
      return 0;
}
bool isPerSq(long x)
{   long s = sqrt(x);
    return (s*s == x);
}
bool isFib(long n)
{   return isPerSq(5*n*n + 4) || isPerSq(5*n*n - 4);
}
int main()
{	
    int i,j,T;
    long n,sum;
    cin>>T;
    for(i=0;i<T;i++){
        sum=0;
        cin>>n;
        for(j=0;j<n;j++){
            if(!isFib(j))
                sum+=j;
        }
        if(isPr(sum))
            cout<<"Tic\n";
        else if(sum%2==0)
            cout<<"Tac\n";
        else
            cout<<"Toe\n";
    }
}
