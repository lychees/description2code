#include <bits/stdc++.h>
#define ll long long
#define int ll
using namespace std;

int cases, m, num;	
int BIT[1000000];

void update(int i, int val){
	for(i; i < cases; i|=i+1)
		BIT[i]+=val;
}

int sum(int n){
	int x = 0;
	while(n >= 0){
		x += BIT[n];
		n &= n+1;
		n--;
	}
	return x;
}

struct q{
	int l, r;
	q(){
		l = r = 0;
	}
};

main(){
	scanf("%lld %lld", &cases, &m);
	for(int i = 0 ; i < cases; i++){
		scanf("%lld", &num);
		update(i,num);
	}
	while(m--){
		q Q[2];
		scanf("%lld %lld", &Q[0].l, &Q[1].l);
		scanf("%lld %lld", &Q[0].r, &Q[1].r);
		for(int i = 0 ; i < 2; i++){
			if(Q[i].l > Q[i].r)
				swap(Q[i].l, Q[i].r);
		}
		if(Q[0].l > Q[1].l)
			swap(Q[0], Q[1]);
		int ans = sum(Q[0].r) - sum(Q[0].l-1);
		int last = Q[0].r;
		if(last < Q[1].l)
			ans += sum(Q[1].r) - sum(Q[1].l-1);
		else if(last < Q[1].r)
			ans += sum(Q[1].r) - sum(last);
		printf("%lld\n", ans);
	}
}
