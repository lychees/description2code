#include<iostream>
#include<stdio.h>
using namespace std;
char board[2000][2000];
int first[2000][2000],second[2000][2000];
int main()
{
	int n,m;
	cin>>n>>m;	
	int size=32;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
		 cin>>board[i][j];	
		}
		
	}
	for(int i=0;i<n;i++)
	{
		first[i][0]=1;
	}
	for(int i=0;i<m;i++)
	{
		second[0][i]=1;
	}
	for(int i=1;i<n;i++)
	{
		if(board[i][0]!=board[i-1][0])
		second[i][0]=second[i-1][0]+1;
		else second[i][0]=1;
	}
	for(int i=1;i<m;i++)
	{
		if(board[0][i]!=board[0][i-1])
			first[0][i]=first[0][i-1]+1;
			else 
			first[0][i]=1;
	}
	
	for(int i=1;i<n;i++)
	{
		for(int j=1;j<m;j++)
		{
			if(board[i][j]!=board[i][j-1])
			first[i][j]=first[i][j-1]+1;
			else 
			first[i][j]=1;
			if(board[i][j]!=board[i-1][j])
			second[i][j]=second[i-1][j]+1;
			else
			second[i][j]=1;	
		}		
	}
	long long int count=0;
	int flag=0,temp=0;
	for(int i=size-1;i<n;i++)
	{
		for(int j=size-1;j<m;j++)
		{
			temp=0;flag=0;
			if(first[i][j]>=size&&second[i][j]>=size)
			{
				//cout<<"ok ";
				for(int k=i,l=j;k>i-size&&l>j-size;k--,l--)
				{
					if(board[k][l]!='_'||first[k][l]<size-temp||second[k][l]<size-temp)
					{
						flag=1;
						break;
					}
					temp++;
				}
				if(board[i][j]=='_'&&flag==0)
				 count++;
			}			
		}
	}
	cout<<count;
	return 0;
}
