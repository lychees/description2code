long long int gcd(long long int a,long long int b)
{
    if(a<b)
        swap(a,b);
    if(a%b==0)
        return b;
    else
        return gcd(b,a%b);
}
long long int lcm(long long int a,long long int b)
{
    long long int k=gcd(a,b);
    long long int lc=a*b/k;
    return lc;
}
