#include <iostream>
#include <string>
#include <algorithm>
#include <iomanip>
using namespace std;

// the minute hand moves 360 degrees in 60 minutes, or 6 degrees per minute.
//
// the hour hand moves 360 degrees in 12 hours, or 30 degrees per hour.  It therefore moves 0.5 degrees
// per minute.

int main()
{
	string time;
	cin >> time;
	
	int hour, minute;
	hour=stoi(time.substr(0, time.find(':',0)));
	time.erase(0, time.find(':',0)+1);
	minute=stoi(time);
//    cout << hour << endl << minute <<endl;
    
    if(hour==12)hour=0;
    if(minute==60)minute=0;
    
    float hourangle = (hour*60 + minute) * 0.5;
    float minuteangle = minute * 6;
    
//    cout << hourangle << endl << minuteangle << endl;
    
    float anglediff = max(hourangle, minuteangle) - min(hourangle, minuteangle);
    
//    cout << anglediff << endl;
    
    if(anglediff>180.0)anglediff-=180.0;
    
    cout << fixed;
    cout << std::setprecision(6);
    cout << anglediff << endl;
    
    return 0;
}
