#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,cnt,num;
    cin >> t;
    while(t--)
    {
    	cnt=0;
    	map<int,int>m;
    	cin >> n;
    	for(int i=0;i<n;i++)
    	{
    		cin >> num;
    		m[num]++;
    		cnt+=m[num];
    	}
    	cout << cnt << endl;
    }
    return 0;
}
