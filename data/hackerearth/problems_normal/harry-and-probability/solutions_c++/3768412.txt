double F(LL len, double theta)
{
	return len * cos(theta);
}

int main()
{
	int t;
	cin(t);
	while(t--)
	{
		LL a,b,c,d;
		int itr = 100;
		scanf("%lld%lld%lld%lld",&a,&b,&c,&d);
		if((b-a+d-c) <= (c-b))
			printf("%.6lf\n",0);
		else
		{
			double P = c-b;
			P /= (b+d-a-c);
			double ans = acos(P) / acos(-1);
			printf("%.6lf\n",ans);
		}
	}
	return 0;
}
