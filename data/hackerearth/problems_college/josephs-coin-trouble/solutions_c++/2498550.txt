#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll n,m=3,i,j,p,q;
    cin>>n;
    ll a[m];
    for(i=0;i<m;i++)
    {
        cin>>a[i];
    }
    ll dp[n+1][m];
    for(i=0;i<m;i++)
    {
        dp[0][i]=1;
    }
    for(i=1;i<n+1;i++)
    {
        for(j=0;j<m;j++)
        {
            p=(i-a[j])>=0?dp[i-a[j]][j]:0;
            q=j>=1?dp[i][j-1]:0;
            dp[i][j]=p+q;
        }
    }
    cout<<dp[n][m-1];
    return 0;
}
