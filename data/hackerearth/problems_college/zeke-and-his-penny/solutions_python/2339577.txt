t = int(input())
for i in range(t):
	a,b,c=map(int,raw_input().split())
	maxx=max(a,b,c)
	minn=min(a,b,c)
	mid=a+b+c-maxx-minn
	res=(a*b*c)-(minn*maxx)-(2*minn*mid)+(2*minn)
	res=res%1000000007
	if(res>0):
		print res
	else:
		print 0
