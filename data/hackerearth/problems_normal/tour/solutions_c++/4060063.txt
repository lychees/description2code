#include<iostream>
#include<map>
#include<string>
#include<vector>
using namespace std;
#define ll long long
ll cost[1002][1002];
map<string,int>mp;
vector<string>v;
int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;++i)
    {
        string s;
        cin>>s;
        mp[s]=i;
    }
    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
            cin>>cost[i][j];
    int q;
    cin>>q;
    ll ans=0;
    for(int i=1;i<=q;++i)
    {
        string s;
        cin>>s;
        v.push_back(s);
    }
    vector<string>:: iterator it;
    it=v.begin();
    string prev=*it;
    it++;
    ans+=cost[1][mp[prev]];
    for(;it!=v.end();++it)
    {
        string pres=*it;
        ans+=cost[mp[pres]][mp[prev]];
        prev=pres;
    }
    cout<<ans<<endl;
    return 0;
}
