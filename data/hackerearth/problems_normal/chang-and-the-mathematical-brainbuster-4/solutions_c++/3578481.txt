#include<iostream>
#include<bits/stdc++.h>
using namespace std ;

int main()
{
	int n , a[10000] ;
	int mini = 1000000 ;
cin>> n ;
for(int i=0 ;i<n ;i++)
{
	cin>>a[i] ; 
}

sort(a,a+n) ;

for(int i=n-1 ;i>=0 ;i--)
{
//flag = 0 ;

for(int j= 0; j<i ;j++)
{
	
if(a[i] <= 3*a[j])
{
	mini = min (mini , n-(i-j+ 1));
	break ;
}
}
}

cout <<mini ;
return 0 ;

}
