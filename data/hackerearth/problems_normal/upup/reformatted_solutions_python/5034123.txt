arr = [each for each in raw_input().split(" ")]
for index, each in enumerate(arr):
    if each and ord(each[0]) >= 97 and ord(each[0]) <= 123:
        arr[index] = chr(ord(each[0])-32)+each[1:]
print " ".join(arr)
