
#include<iostream>
#include<bits/stdc++.h>
#include<stdio.h>
#include<string>
#include<vector>
#include<set>

#define lld long long int

using namespace std;

int fast(){
    char c=getchar();
    while(c<'0' || c>'9')
        c=getchar();
    int num=0;
    while(c>='0' && c<='9'){
        num=num*10 + c - 48;
        c=getchar();
    }
    return num;
}

int main()
{
    //ios_base::sync_with_stdio(false);
    lld t,i,num;
    lld n,a[100000];

    t=fast();
    while(t--)
    {
        lld count=0;
        n=fast();
        for (lld i = 0; i < n; ++i)
        {
            a[i]=fast();
        }
        for (int i = 0; i < n; ++i)
        {
           for (int j = i; j < n; ++j)
           {
               if(a[i] == a[j] ) 
                 count++;
           }
        }
        printf("%lld\n",count );
    }
    return 0;
}
