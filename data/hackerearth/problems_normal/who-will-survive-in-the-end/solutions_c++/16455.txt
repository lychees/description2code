#include<iostream>
#include<stdio.h>
using namespace std;
long rem;
long solution(int N)
{
     if(N==0)
     return 0;
     
     if(N==1)
     return 1;
     
     if(N==2)
     return 1;
     
     if(N==3)
     return 3;
     
     if(N==4)
     return 1;
     
     if(N==5)
     return 3;
     
     if(N%2==0)
     {
               return 2*solution(N/2)-1;
     }
     else
     {
         rem = solution(N- (N/2));
         if(rem==1)
         return N;
     
         return 2*rem-3;
     
     }
}
int main()
{

    
    int N,T;
    scanf("%d",&T);
    
    while(T--)
    {
    scanf("%d",&N);
    
    printf("%ld\n",solution(N));
    }
    return 0;
}
