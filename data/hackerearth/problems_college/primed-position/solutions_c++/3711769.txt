#include<bits/stdc++.h>

using namespace std;

int main()
{

	vector<int> p;
	vector<int> pr;
	p.push_back(1);
	pr.push_back(1);
	char *s=(char*)calloc(10000001,sizeof(char));
	for(int i=2;i<=(int)sqrt(10000000);i++)
	{
		if(s[i]=='f')
		continue;
	
		int m=i;
		while((m*i) <= 10000000)
		{
			s[m*i]='f';
			m++;
		}
	}
	
	for(int i=2;i<=10000000;i++)
	{
		if(s[i]!='f')
		p.push_back(i);
	}

	for(int i=2;i<p.size();i++)
	{
		if(s[i] !='f')
		pr.push_back(p[i]);
	}	
	int t,n,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		printf("%d\n",((long long int)pr[n]*pr[m])%1000000007);
	}
	return 0;
}
