#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
struct car
{
	int num;
	int idx;
}cars[100005];
bool mycomp(struct car i,struct car j)
{
	if(i.num==j.num)
		return i.idx<j.idx;
	return (i.num<j.num);	
}
int arr[100005];
int main()
{
        int t,maxi=0,n,temp;
        scanf("%d",&t);
        while(t--)
        {
        	scanf("%d",&n);
        	for(int i=0;i<n;i++)
        	{
        		scanf("%d",&arr[i]);
        		cars[i].num=arr[i];
        		cars[i].idx=i;
        	}
        	sort(cars,cars+n,mycomp);
        	maxi=0;
        	for(int i=0;i<n;i++)
        	{
        		temp=cars[i].idx-i;
        		if(temp<0)
        			temp=temp*-1;
        		if(maxi<temp)
        			maxi=temp;
        	}
        	printf("%d\n",maxi);
        }
        return 0;
}
