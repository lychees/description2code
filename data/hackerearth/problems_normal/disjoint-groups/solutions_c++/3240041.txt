#include<bits/stdc++.h>
using namespace std;
const int mod =1e9+7;

int main()
 {
   int n ,ex ;
   cin >> n ;
	int a[n+1] ;
   int ans = 0  ; 
   a[0]=1;
   for (int i=1;i<n;i++)
   a[i]=(a[i-1]<<1) % mod;

   for (int i=0;i<n;i++)
   {
       cin >> ex ;
       ans ^= ex  ;   
   }

   if (ans!=0) 
   cout <<"0" ;
   else
   cout <<((a[n-1] + mod - 1 ) % mod);

   return 0;
}
