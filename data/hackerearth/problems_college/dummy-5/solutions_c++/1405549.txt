#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int arr[2500][500];
int gp,gv,gc,n;

bool fun(int a,int b,int c,int i)
{
	if(i>=n)
		return 0;
	if(a<0 || b<0 || c<0)
		return false;
	if(a==0 && b==0 && c==0)
		return true;
	bool x,y;
	x=fun(a-arr[i+1][0],b-arr[i+1][1],c-arr[i+1][2],i+1);
	y=fun(a,b,c,i+1);
	if(x==true || y==true)
		return true;
	else 
		return false;
}

int main()
{
	int t,j;
	cin>>t;
	while(t--)
	{
		//memset(dp,-1,sizeof(dp));
		cin>>gp>>gv>>gc;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
		}
		fun(gp,gv,gc,0);
		fun(gp-arr[0][0],gv-arr[0][1],gc-arr[0][2],0);
		if(fun(gp,gv,gc,0)==true || fun(gp-arr[0][0],gv-arr[0][1],gc-arr[0][2],0)==true )
			cout<<"Case #"<<j+1<<": yes"<<endl;
		else
			cout<<"Case #"<<j+1<<": no"<<endl;
		j++;
	}
return 0;
}
