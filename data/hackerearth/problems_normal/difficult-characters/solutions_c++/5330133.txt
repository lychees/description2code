#include <iostream>
#include<map>
using namespace std;
string s;
multimap<int,int> m;
int count[26],t;
char a;
int main()
{
    cin>>t;
    while(t--)
    {cin>>s;
    m.clear();
    for(int i=0;i<26;i++)
    count[i]=0;
    for(int i=0;i<s.length();i++)
    count[s[i]-'a']++;
    for(int i=25;i>=0;i--)
     if(count[i]==0)
    {a=i+'a'; 
    cout<<a<<" ";
    }
    else
    m.insert(pair<int,int>(count[i],i));
    map<int,int>::iterator it;
    for(it=m.begin();it!=m.end();it++)
    {a=it->second+'a';
    cout<<a<<" ";
    }
    cout<<endl;
    
    
 
    }
    return 0;
}
