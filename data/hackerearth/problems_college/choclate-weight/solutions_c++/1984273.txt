#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n; cin >> n;
    int a[n];
    for(int i=0;i<n;i++)
    	cin >> a[i];
    int Q; cin >> Q;
    sort(a,a+n);
    while(Q--)
    {
    	int x,flag=0; cin >> x;
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]==x)
    		{
    			flag=1;break;
    		}
    		if(binary_search(a+i+1,a+n,x-a[i]))
    		{
    			flag=1;break;
    		}
    	}
    	cout << (flag==0?"I Hate You":"I Like You") << endl;
    }
    return 0;
}
