#include <iostream>
using namespace std;

void count(long long int n)
{
	long long int digits[10], j, multiplier = 1, m=n;
	for (j=0; j< 10; j++)
		digits[j] = 0;

	if (n == 0)
		digits[0] = 1;

	while (n>0)
	{
		int current = n % 10;
		for (j=0;j<10;j++)
			digits[j] += (n/10) * multiplier + (j < current)*multiplier;
		digits[current] += (m % multiplier)+1;
		if (multiplier > 1)
			digits[0] -= multiplier;
		n=n/10;
		multiplier *= 10;
	}
    long long int sum=0;
	for (j=0; j<10; j++)
	       sum+=digits[j];
   printf("%lld\n",sum-1);
   return;
}

int main()
{
    //cout << "Hello World!" << endl;
    int t;
    scanf("%d", &t);
    while(t--){
    	long long int n;
    	scanf("%lld", &n);
    	count(n);
    }
    return 0;
}
