mod = 10 ** 9 + 7
test = int(input())
for i in range(test):
    n = int(input())
    t1 = 1
    t2 = 2
    t3 = 4
    if n == 1:
        print(t1)
    elif n == 2:
        print(t2)
    elif n == 3:
        print(t3)
    else:
        result = 0
        for j in range(n - 3):
            result = (t1 + t2 + t3) % mod
            t3, result = result, t3
            t2, result = result, t2
            t1, result = result, t1
        print(t3)
