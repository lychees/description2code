def start():
    n = input()
    i = []
    while n != 0:
        l = raw_input().split(" ")
        l[1] = int(l[1])
        i.append(l)
        n -= 1
    i = sorted(i, key=lambda i: i[1])
    i.sort(key=lambda i: i[0])
    i.sort(key=lambda i: i[1], reverse=True)
    for x in range(len(i)):
        print i[x][0], i[x][1]


if __name__ == "__main__":
    start()
