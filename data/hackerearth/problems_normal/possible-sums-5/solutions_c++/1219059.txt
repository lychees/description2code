#include <iostream>
#include<cstdio>
#include<cstring>
using namespace std;
bool h[10005];
int a[105];
int main()
{
	int tot,ans,t,i,j,n;
	scanf("%d",&t);
	while(t--)
	{
		tot=0;
		ans=1;
		memset(h,0,sizeof(h));
		h[0]=1;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			tot+=a[i];
		}
		for(i=0;i<n;i++)
		{
			for(j=tot;j>=a[i];j--)
			{
				if(h[j-a[i]]==1 && h[j]==0)
				{
					h[j]=1;
					ans++;
				}
			}
		}
		printf("%d\n",ans);
		
	}
    return 0;
}
