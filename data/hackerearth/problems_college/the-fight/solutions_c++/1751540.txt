typedef long long ll;

int main(){
    //ios::sync_with_stdio(false);
    #ifdef DEBUG
    freopen("input.txt", "r", stdin);
    #endif // DEBUG

    int N,M,I,J,K;
    scanf("%d", &N);
    ll *arr, *cnt;
    arr = new ll[N];
    cnt = new ll[N];
    for(int i=0;i<N;i++){
        scanf("%lld", &arr[i]);
        cnt[i] = 0;
    }
    scanf("%d", &M);
    while(M--){
        scanf("%d %d %d", &I, &J, &K);
        cnt[I-1] += (ll)K;
        if(J<N)
            cnt[J] -= (ll)K;
    }
    for(int i=0;i<N;i++)
        cnt[i] += (i==0)?0:cnt[i-1];
    for(int i=0;i<N;i++){
        printf("%lld\n", arr[i]+cnt[i]);
    }
    delete[] arr;
    delete[] cnt;

    return 0;
}
