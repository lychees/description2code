#include <bits/stdc++.h>

using namespace std;

namespace stuff {
typedef long long ll;

void solve(int test_num) {
  int N, Q;
  cin >> N >> Q;
  vector<string> str(N);
  for (int i = 0; i < N; i++)
    cin >> str[i];
  vector<ll> len(N);
  len[0] = str[0].size();
  for (int i = 1; i < N; i++)
    len[i] = 2 * len[i - 1] + str[i].size();
  string res = "";
  while (Q-- > 0) {
    ll pos;
    cin >> pos;
    int i = 0;
    while (pos >= len[i])
      i++;
    while (true) {
      if (i == 0 || pos >= 2 * len[i - 1]) {
        if (i > 0)
          pos -= 2 * len[i - 1];
        res += str[i][(int) pos];
        break;
      } else if (pos < len[i - 1]) {
        i--;
      } else {
        pos = len[i - 1] - (pos - len[i - 1]) - 1;
        i--;
      }
    }
  }
  cout << res << endl;
}

void solve() {
#ifdef AZN
  double start_t = clock();
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
//freopen("azn.txt", "w", stderr);
#endif
  ios::sync_with_stdio(false);
  cin.tie(NULL);
  int T = 1;
//scanf("%d", &T);
  cin >> T;
  for (int t = 1; t <= T; t++)
    solve(t);
#ifdef AZN
  cerr << "Took: " << ((clock() - start_t) / CLOCKS_PER_SEC);
#endif
}
}

int main() {
  stuff::solve();
  return 0;
}
