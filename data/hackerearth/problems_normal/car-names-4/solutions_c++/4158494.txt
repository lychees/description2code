#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int i,c=0,k,flag = 0,cr=0,ctr=0,h[26]={0},l=0;
		string s;
		cin >> s;
		for(i=0;i<s.size();i++)
		{
			h[s[i]-97]++;
			
		}
		for(i=0;i<26;i++)
		{
			if(h[i]>0)
			{
				l++;
			}
		}
		if(l != 3)
		{
			cout <<"Not OK"<< endl;
			continue;
		}
		for(i=0;i<s.size()-1;i++)
		{
			if(s[i] == s[i+1])
			{
				c++;
			}
			else
			{
				if(flag == 0)
				{
					k = c;
					flag=1;
				}
				else
				{
					if(k !=c)
					{
						cout << "Not OK" << endl;
						cr = 1;
						break;
					}
					
				}
				c = 0;
				ctr++;
			}
		//	cout << c <<" "<< ctr<<endl ;
		}
		if(ctr != 2 && cr == 0)
		{
			cout << "Not OK" <<endl;
			cr = 2;
		}
		if(ctr == 2 && cr !=1 && cr != 2)
		{
			cout << "OK"<<endl;
		}
	}
    
    return 0;
}
