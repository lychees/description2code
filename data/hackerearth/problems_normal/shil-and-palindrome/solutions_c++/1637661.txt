#include<bits/stdc++.h>

using namespace std;

int ar[26];


int main(){

    memset(ar,0,sizeof(ar));
    char str[100002];
    scanf("%s",str);
    //cout<<str<<endl;
    int len=strlen(str);

    for(int i=0;i<len;i++)
        ar[str[i]-'a']++;

       int cnt=0;   int idx=-1;

    for(int i=0;i<=25;i++){

        if(ar[i]&1){
            cnt++;
            idx=i;
        }
    }

    if(cnt>1){
        cout<<"-1"<<endl;
    }
    else{
        int i,j;

        for(j=0;j<=25;j++){

            i=ar[j]/2;
                for(int k=1;k<=i;k++)
                    cout<<(char)(j+'a');

        }

        if(idx!=-1)
            cout<<(char)(idx+'a');

        for(j=25;j>=0;j--){

            i=ar[j]/2;
                for(int k=1;k<=i;k++)
                    cout<<(char)(j+'a');

        }


    }


    return 0;
}
