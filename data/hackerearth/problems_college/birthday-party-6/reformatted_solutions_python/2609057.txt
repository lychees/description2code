
t = input()

for j in range(0, t):

    str_arr = raw_input().split(' ')
    arr = [int(num) for num in str_arr]
    m = arr[0]
    n = arr[1]

    if (n > m+1):
        print '-1'
    else:
        ans = 1
        i = 1
        for i in range(1, m+1):
            ans = ans*i
        ans = ans * ans
        ans = ans*(m+1)

        i = 1
        ans1 = 1
        for i in range(1, m-n+1+1):
            ans1 = ans1*i
        ans = ans/ans1
        ans = ans % 1000000007
        print ans
