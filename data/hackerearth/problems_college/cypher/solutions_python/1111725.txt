#print 'Hello World!'
import sys

n = int(raw_input())
i = 0

while (i < n):
	tc = raw_input()
	ii = 0
	
	while (ii < len(tc)):
		if (ord(tc[ii]) + ii > 122):
			sys.stdout.write(chr((ord(tc[ii]) + ii) % 122 + 96))
		else:
			sys.stdout.write(chr(ord(tc[ii]) + ii))
		
		ii += 1
		
	print ""
	i += 1
