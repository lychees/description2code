using namespace std;

long long gcd(long long a, long long b)
{
	if (a == 0)
	{
		return b;

	}
	else if (b == 0)
	{
		return a;
	}
	else{
		return gcd(b, a%b);
	}
}
int main()
{
	int t;
	scanf("%d", &t);

	while (t--)
	{
		int n, k;
		scanf("%d%d", &n, &k);

		//vector<long long > v;

		long long x;

		long long int c = 0;

		for (int i = 0; i < n; i++)
		{
			scanf("%d", &x);
			if (x == k)
			{
				c++;
			}
		//	v.push_back(x);
		}
		
		//cC3/nC3

		long long numer = (c*(c - 1)*(c - 2)) / 2;

		long long denom = ((long long)n*(long long)(n - 1)*(long long)(n - 2))/ 2;

		long long newnum = numer / gcd(numer, denom);
		long long newden = denom / gcd(denom, numer);

		printf("%lld/%lld\n", newnum, newden);
		//cout << newnum << "/" << newden << endl;
	}
	return 0;
}
