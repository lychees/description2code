#include<bits/stdc++.h>
#define s(n) scanf("%d",&n)
#define ps(s) scanf("%s",&s);
#define sll(n) scanf("%lld",&n)
#define sl(n) scanf("%ld",&n)
#define p(n) printf("%d\n",n)
#define pll(n) printf("%lld\n",n)
#define pl(n) printf("%ld\n",n)
#define py() printf("YES\n")
#define pn() printf("NO\n")
#define fp(i,a,n) for((i) = ( a); (i) <(n); (i)++)
#define fn(i,n,a) for((i) = ( n); (i) >=(a); (i)--)
#define ll long long int
#define l long int
ll i,j,k;
using namespace std;

int main()
{
    int t;
    s(t);
    string s;
    while(t--)
    {
        cin>>s;
        int c=0,cv=0,cc=0;
        fp(i,0,s.size())
        {
            if(s[i]-'a'==0 || s[i]-'a'==4 || s[i]-'a'==8 || s[i]-'a'==14 || s[i]-'a'==20 || s[i]-'a'==24)
            {
                cv++;
                c=0;
            }
            else
            {
                cc++;
                c++;
                if(c==3)
                    break;
            }
        }
        if(c==3)
            printf("hard\n");
        else if(cc > cv)
                printf("hard\n");
            else
                printf("easy\n");
    }
    return 0;
}
