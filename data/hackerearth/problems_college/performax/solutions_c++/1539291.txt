#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long d;
	cin>>d;
	long long a[t];
	for(int i=0;i<t;i++){
		cin>>a[i];
	}
	int prev=a[0];
	long long k;
	int ans=0;
	for(int i=1;i<t;i++){
		if(a[i]>prev) {}
		
		else{
			k=ceil((float)(prev-a[i])/d);
			if((k*d)+a[i]>prev){
				ans+=k;
				a[i]=k*d+a[i];
			}
			else{
				ans+=k+1;
				a[i]=(k+1)*d+a[i];
			}
		}
	//	cout<<a[i]<<" "<<ans<<endl;
		prev=a[i];
		
	}
	cout<<ans<<endl;
	
    return 0;
    
}
