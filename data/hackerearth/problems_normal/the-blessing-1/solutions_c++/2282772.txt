#include<cstdio>
#include<iostream>
#include<cmath>
typedef long long int i64;
using namespace std;
int main()
{
	i64 tc,inpop,r,wh;
	double res,pfood,incf;
	scanf("%lld",&tc);
	while(tc--)
	{
		scanf("%lf %lf",&pfood,&incf);
		res=0;
		inpop=1;
		wh=0;
		while(wh<=100)
		{
			res+=(pfood/inpop);
			inpop*=2;
			pfood+=incf;
			wh=inpop/pfood;
		}
		r=ceil(res);
		printf("%lld\n",r);
	}
	return 0;
}
