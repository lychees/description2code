int phi(int n)
{
    int result=n;
    if(n==1) return n;
    if(n%2==0)
    {
        result-=result/2;
        while(n%2==0) n/=2;
    }
    for(int i=3; i*i<=n; i+=2)
    {
        if(n%i==0)
        {
            result-=result/i;
            while(n%i==0) n/=i;
        }
    }
    if(n>1) result-=result/n;
    return result;
}

int main()
{
    int T,n,i,result,dp[MAX+1];
    dp[0]=0;
    dp[1]=0;
    dp[2]=1;
    for(i=3; i<=MAX; i++)
        dp[i]=phi(i)+dp[i-1];
    // for(i=0; i<=MAX; i++)   cout<<dp[i]<<" ";
    cin>>T;
    while(T--)
    {
        cin>>n;
        cout<<dp[n]<<endl;
    }
    return 0;
}
