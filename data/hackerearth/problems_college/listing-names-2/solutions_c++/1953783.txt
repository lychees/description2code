#include<iostream>

#include<vector>
#include<algorithm>

using namespace std;
 
#define all(a)                      a.begin(),a.end()
#define pb                          push_back
#define sz(a)                       sizeof(a)
#define mp                          make_pair
#define fir                         first
#define sec                         second
#define ll                          long long
#define pii                         pair<int,int>
#define ge(a)                       scanf("%d",&a)
#define pu(a)                       printf("%d",a)
#define ges(a,n)                    for(int i=0;i<n;i++)    ge(a[i])
#define pus(a,n)                    for(int i=0;i<n;i++)    printf("%d ",a[i])

int main()
{
ios::sync_with_stdio(0);
cin.tie(0);
map<string,int> m;
int t;
string s;
cin>>t;
vector<string > v;
while(t--)
{
	cin>>s;
	v.pb(s);
	m[s]++;
}
sort(all(v));
for(int i=0;i<v.size();)
{
	cout<<v[i]<<" "<<m[v[i]]<<"\n";
	i+=m[v[i]];
}
return 0;
}
