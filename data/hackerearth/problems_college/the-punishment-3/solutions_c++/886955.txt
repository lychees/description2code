#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{

int t;
scanf("%d",&t);
static int arr1[100000]={0},sum[100000]={0};
while(t--)
{
    

    
    int n,k;
    scanf("%d%d",&n,&k);
    arr1[0]=1;
    int l;
  if(n>2*k) { sum[0]=1; sum[n-2*k]++;  l=n-k;}
 else { sum[0]=1; sum[2*k-n]++;  l=k;  }
arr1[n]++;

//for(int i=0;i<l;i++) cout<<sum[i];
//cout<<endl;
for(int i=n,j=l;i>=0;i--,j--)
{

if(j<0)
{
    for(;i>=1;i--)
    {
        if(arr1[i]<0) arr1[i]+=10;
        arr1[i-1]--;
        
    }
  break;  
}
    if(arr1[i]>=sum[j]) continue;
    arr1[i]=arr1[i]-sum[j]+10;
    arr1[i-1]--;


}

int start=0;
if(arr1[0]==0) start++;
for(int i=start;i<=n;i++) printf("%d",arr1[i]);
printf("\n");

for(int i=0;i<100001;i++)
{
arr1[i]=0;
sum[i]=0;    
    
}

}

}
