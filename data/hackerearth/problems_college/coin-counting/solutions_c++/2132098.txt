#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int n,m,a[55],dp[255][55],i,j,t;
	scanf("%lld",&t);
	a[0]=1;
	a[1]=3;
	a[2]=5;
	while(t--)
	{
	scanf("%lld",&n);
	for(i=0;i<3;i++)
	{
		dp[0][i]=1;
	}
	for(i=1;i<=n;i++)
	{
		for(j=0;j<3;j++)
		{
			if((i-a[j])>=0)
				dp[i][j]=dp[i-a[j]][j];
			else
				dp[i][j]=0;

			if(j>0)
				dp[i][j]+=dp[i][j-1];
		}
	}
	printf("%lld\n",dp[n][2]);
	}
	return 0;
}
