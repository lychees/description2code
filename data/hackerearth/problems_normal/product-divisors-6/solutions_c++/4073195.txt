#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007

#define ll long long 

#define pb(a) push_back(a)
#define rrep(i, b, a) for (int i = b; i > a; i--)
#define all(v) v.begin(), v.end()

#define sz(x) ((int) (x).size())
#define rep(i, n) for (int i = 0; i < (n); i++)
#define sqrt(x) sqrt(abs(x))
#define unq(x) (x.resize(unique(all(x)) - x.begin()))

int n;
map<int, int> m;
vector<int> v;

void parse(int x){
	for(int i=2; i*i<=x; i++)
		while(x%i==0){
			x/=i;
			v.pb(i);
			m[i]++;
		}
	if(x>1){
		v.pb(x);
		m[x]++;
	}
}
int main(){
	
	cin>>n;
	rep(i,n){
		int x;
		cin>>x;
		parse(x);
	}
	sort(all(v));
	unq(v);
	ll ans = 1;
	rep(i,sz(v))
		ans=ans*(m[v[i]]+1)%mod;
	cout<<ans<<endl;
	return 0;
}
