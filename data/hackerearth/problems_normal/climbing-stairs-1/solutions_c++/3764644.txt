#include <iostream>
using namespace std;

int main()
{
	uint64_t A,B,N;
	cin>>A>>B>>N;
	
	//if B is greater than N, will never reach to the top
	//oh and thats why you have put B<A
	uint64_t nDaysRequired = 0;
	uint64_t nCurrentStep = 0;
	while(nCurrentStep <= N)
	{
		nCurrentStep += A;
		nDaysRequired++;
		if( nCurrentStep >= N)
			break;
		nCurrentStep -= B;
	}
	cout<<nDaysRequired<<endl;
    return 0;
}
