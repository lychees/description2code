#include <bits/stdc++.h>
using namespace std;
#define LL long long

void solve()
{
	LL a, b;
	scanf("%lld%lld", &a, &b);
	if(b >= a) {
		printf("%lld\n", b);
		return;
	}
	LL rem = a % b;
	if(b - rem <= rem - 0) {
		printf("%lld\n", a + (b - rem));
	}	
	else {
		printf("%lld\n", a - rem);
	}
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
