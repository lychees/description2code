#include <iostream>
using namespace std;

int count_zeroes(int num)
{
	int i,count=0;
for(i=5;num/i>=1;i=i*5)
{
	count=count+num/i;
}
return count;
}

void  create_arr(long int arr[])
{
	int i;
arr[0]=arr[1]=arr[2]=arr[3]=arr[4]=0;
for(i=5;i<=1000000;i++)
{
	arr[i]=count_zeroes(i)+arr[i-1];
}
}

int main()
{
 long int arr[1000001]={0};
int t,a,b;
cin>>t;
create_arr(arr);
while(t--)
{
cin>>a>>b;
cout<<arr[b]-arr[a-1]<<"\n";

}	
return 0;
}
