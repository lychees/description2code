#include <iostream>
#include<algorithm>
using namespace std;
typedef long long int ll;
ll mat[1000][1000];
int search( int n, int m,int x)
{
   int i = 0;int j = m-1;  //set indexes for top right element
   while ( i < n && j >= 0 )
   {
      if ( mat[i][j] == x )
      {
         printf("%d %d\n", i, j);
         return 1;
      }
      if ( mat[i][j] > x )
        j--;
      else //  if mat[i][j] < x
        i++;
   }
 
   printf("-1 -1\n");
   return 0;  // if ( i==n || j== -1 )
}
int main()
{
    ll i,j,n,m;
    cin>>n>>m;
  
    for(i=0;i<n;i++)
    { for(j=0;j<m;j++)
    	{cin>>mat[i][j];
    	}		
    }
   ll t;
   cin>>t;
   while(t--)
   {
   	ll key;
   	cin>>key;
   	search(n,m,key);
   }
    
  
    return 0;
}
