#include <iostream>
using namespace std;

int main()
{
    unsigned long long int cases,x,count,y[1000500],i,re=1000000007;
    y[1]=1;
    for(i=2;i<=1000000;i++)
    	y[i]=(y[i-1]*i)%re;
    	
    cin>>cases;
    while(cases--)
    {
    	cin>>x;
    	count=(y[x]*3+3)%re;
    	cout << count << endl;
    }
    return 0;
}
