#include<bits/stdc++.h>
using namespace std;
int main()
{
    bool a[26];
    memset(a,false,sizeof(a));
    int n;
    scanf("%d",&n);
    string s;
    cin>>s;
    stack <char> st;
    st.push(s[0]);
    a[s[0]-97]=true;
    if(n==1)
        cout<<s;
    else
    {
        for(int i=1;i<n;i++)
        {
            //cout<<st.top()<<"  "<<s[i]<<"\n";
            if(s[i]>st.top())
            {
                //cout<<a[s[i]-97]<<" i="<<i<<"\n";
                if(a[s[i]-97]==false){
                while(st.size()>0&&s[i]>st.top())
                {
                    a[st.top()-97]=false;
                    //cout<<st.top()<<"\n";
                    st.pop();
                }
                st.push(s[i]);
                a[s[i]-97]=true;
                }
            }
            else if(s[i]<st.top()){
                st.push(s[i]);
                a[s[i]-97]=true;
            }
        }
        string s1;s1.clear();
        while(!st.empty())
        {
            //cout<<st.top();
            s1+=st.top();
            st.pop();
        }
        for(int i=s1.size()-1;i>=0;i--)
            cout<<s1[i];
    }
    return 0;
}
