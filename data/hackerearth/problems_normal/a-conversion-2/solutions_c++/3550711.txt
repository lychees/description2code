#include <bits/stdc++.h>
using namespace std;
string s1,s2;
int ans;
void fun(string s1,int idx,int c)
{
	if(idx==s1.length())
	{
		ans=min(ans,c);
		return;
	}
	if(s1[idx]==s2[idx])
	{
		fun(s1,idx+1,c);
		return;
	}
	int d;
	if(s1[idx]>s2[idx])
		{
			fun(s1,idx+1,c+s1[idx]-s2[idx]);
		}
		if(s2[idx]>s1[idx])
		{
			fun(s1,idx+1,c+s2[idx]-s1[idx]);
		}
	for(int i=idx+1;i<s1.length();i++)
	{
		if(s1[idx]==s1[i])
		continue;
		swap(s1[idx],s1[i]);
		d=abs(s1[idx]-s2[idx]);
		fun(s1,idx+1,c+1+d);
		swap(s1[idx],s1[i]);
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--){
		ans=INT_MAX;
    cin>>s1>>s2;
    fun(s1,0,0);
	cout<<ans<<endl;
	}
    return 0;
}
