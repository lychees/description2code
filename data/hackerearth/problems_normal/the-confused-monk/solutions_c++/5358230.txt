#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef long long int ll;
typedef long double ld;
ll mexp(ll a,ll b,ll c)
{
if(b==0) return 1;
if(b==1) return a%c;
else if(b%2==0) return mexp((a*a)%c,b/2,c);
else return (a*mexp((a*a%c),b/2,c))%c;
}
ll gcd(ll a,ll b)
{
if(b==0) return a;
else return gcd(b,a%b);
}
int main()
{
ios_base::sync_with_stdio(false);
cin.tie(0);
cout.tie(0);
ll n,i,g=0,mul=1,ans=0;
ll dp[52];
memset(dp,0,sizeof(dp));
cin>>n;
for(i=0;i<n;i++)
{
cin>>dp[i];
}
for(i=0;i<n;i++)
{
g=gcd(g,dp[i]);
g=g%MOD;
}
//cout<<gcd<<"\n";
for(i=0;i<n;i++)
{
mul=(mul%MOD*dp[i]%MOD);
mul=mul%MOD;
}
//cout<<mul<<"\n";
ans=mexp(mul,g,MOD);
cout<<ans<<"\n";
return 0;
}
