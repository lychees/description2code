#include<bits/stdc++.h>
using namespace std;
vector <int> v,sum;
int main()
{
    int n,i,t;
    cin>>t;
    while(t--)
    {
    cin>>n;
    bool cond=false;
    v.resize(n+2);
    sum.resize(n+3);
    for(i=1;i<=n;i++)
    cin>>v[i];
    sum[0]=0;
    for(i=1;i<=n;i++)
        sum[i]=sum[i-1]+v[i];
    for(i=1;i<=n;i++)
    {
        if(sum[i-1]==sum[n]-sum[i])
        {
            cond=true;
            break;
        }
    }
    if(cond)
        cout<<"YES"<<endl;
    else
        cout<<"NO"<<endl;
    }
}
