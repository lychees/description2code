#include <iostream>
using namespace std;

int main()
{

   int test_case;
   cin>>test_case;
   long int p[test_case],q[test_case],n[test_case];
   for(int i=0;i<test_case;i++)
   {
   		cin>>p[i];
   		cin>>q[i];
   		cin>>n[i];
   		cout<<"\n";
   }
   for(int i=0;i<test_case;i++)
   {
   	if(p[i]*q[i]==n[i])
   	{
   		cout<<"\nYES";
   	}
   	else
   	{
   		cout<<"\nNO";
   	}
   }
    return 0;
}
