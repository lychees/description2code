#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	int t,n,i,j,cnt,result;
	double f;
	cin>>t;
	for(j=1;j<=t;j++){
		result=0;f=0;
		cin>>n;
		for(i=1;i<=n;i++){
			f=f+log10(i);
		}
		for(i=5;n/i>=1;i*=5){
			result+=n/i;
		}
		cnt=(int)ceil(f)-result;
		cout<<cnt<<"\n";
	}
    return 0;
}
