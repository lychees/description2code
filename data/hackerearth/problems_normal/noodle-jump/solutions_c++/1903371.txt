#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{	long long c=1,n,k,a[1000000],i,j;
    cin>>n>>k;
    for( i=0; i<n; i++ )
    	cin>>a[i];
    sort(a,a+n);
    for( i=0; i<n-1; i++ )
    {	if ( a[i+1]-a[i] > k )
    	{	cout<<a[i]<<endl;
    		break;
    	}
    }
    if( i == n-1 )
    	cout<<a[n-1]<<endl;
    return 0;
}
