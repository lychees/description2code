#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	char a[10000],b[10000],d[10000];
    	int i=0,c=0;
    	scanf("%s",a);
    	scanf("%s",b);
    	c=strlen(a);
    	while(a[i]!='\0'){
    		if(a[i]==b[i])
    			d[i]='0';
     		else
    			d[i]='1';
    		i++;
    	}
    	for(i=0;i<c;i++){
            printf("%c",d[i]);
    	}
    	printf("\n");
    }
    return 0;
}
