#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	unsigned long long int n=0,sum=0;
    	unsigned int total = 0;
    	unsigned long long int p = 0;
    	int k=0;
    	int extra = 0;
    	long long int q=0;
    	long long int r =0;
    	
    	cin>>n;
    	
    	//p = 18*n-6;
    	p = 12*n;
    	int c[26] = {0};	
    	for(int i=0;i<26;i++){
    		cin>>c[i];
    		total=total+c[i];		
    	}


    	q=p/26;
    	r=p%26;
    	
    	if(q>0){
    		sum=total*q+(q-1)*(q)*13;
    		extra = q;
    	}
    	k=0;
    	sum=sum+r*extra;
    	while(r--){
    		sum=sum+c[k];
    		k++;
    		
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
