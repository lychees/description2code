#include<bits/stdc++.h>
using namespace std;
void sol()
{
    long long n,m,t;
    cin>>n>>m;
    stack<int> s;
    while(m)
    {
        t=m&(-m);
        t=log2(t);
        s.push(t);
        m=m&(m-1);
    }
    while(!s.empty())
    {
        t=s.top();
        cout<<'('<<n<<"<<"<<t<<')'<<' ';
        s.pop();
        if(!s.empty()) cout<<'+'<<' ';
    }
    cout<<endl;
}
int main()
{
    int t;
    cin>>t;
    while(t--) sol();
    return 0;
}
