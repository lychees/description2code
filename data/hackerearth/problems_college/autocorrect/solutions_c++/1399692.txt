#define Max(a, b) (a > b) ? a : b

#define gcd __gcd
#define FOR(i, a, n) for (i = a; i < n; ++i)
#define FORe(i, a, n) for (i = a; i <= n; ++i)
#define FORi(i, a, n) for (int i = a; i < n; ++i)
#define FORie(i, a, n) for (int i = a; i <= n; ++i)		
#define FastIO ios_base::sync_with_stdio(false); cin.tie(0);
#define endl '\n'
using namespace std;
typedef long long ll;
typedef unsigned long long ull;

bool isRightTriangle(ll A[6]) {
	ll z = A[0] * (A[3] - A[5]) + A[2] * (A[5] - A[1]) + A[4] * (A[1] - A[3]);
	if (z == 0) return false;
	ll B[6] = { A[0] - A[2], A[2] - A[4], A[4] - A[0], A[1] - A[3], A[3] - A[5], A[5] - A[1] };
	FORi(i, 0, 6) B[i] *= B[i];
	if ((B[0] + B[3] + B[1] + B[4] == B[2] + B[5])
		|| (B[0] + B[3] + B[2] + B[5] == B[1] + B[4])
		|| (B[2] + B[5] + B[1] + B[4] == B[0] + B[3])) return true;
	return false;
}

bool allVariantsCheck(ll A[6]) {
	if (isRightTriangle(A)) return true;
	FORi(i, 0, 6) {
		A[i] += 1;
		if (isRightTriangle(A)) return true;
		A[i] -= 2;
		if (isRightTriangle(A)) return true;
		A[i] += 1;
	}
	return false;
}

int main() {
	FastIO
	ll N;
	cin>>N;
	while (N--) {
		ll A[6];
		FORi(i, 0, 6) cin>>A[i];
		cout<<((allVariantsCheck(A)) ? "YES\n" : "NO\n");
	}
	return 0;
}
