#include <cstdio>
#include <iostream>
#include <string>

using namespace std;

bool Process(int n,int k)
{
	
	if((n-1) % (k+1))
        return true;
    return false;
}

int main()
{
    int T,n,k;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&n,&k);
        Process(n,k) ? printf("Rani\n") : printf("Nandu\n");
    }
    return 0;
}
