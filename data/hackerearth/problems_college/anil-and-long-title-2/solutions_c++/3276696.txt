#include <iostream>
using namespace std;

string capitalize (string &s)
{
    bool cap = true;

    for(unsigned int i = 0; i <= s.length(); i++)
    {
        if (isalpha(s[i]) && cap == true)
        {
            s[i] = toupper(s[i]);
            cap = false;
        }
        else if (s[i]=='_')
        {
            cap = true;
        }
    }
    return s;
}
int main()
{
    string cc;
    cin>>cc;

	cout << capitalize(cc);

	return 0;
}
