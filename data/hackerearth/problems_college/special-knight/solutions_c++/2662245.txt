#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
    {
    	cout<<"Case #"<<i<<": ";
    	int h,w,r;
    	cin>>h>>w>>r;
    	bool b[h+1][w+1];
    	for(int i=1;i<=h;i++)
    		for(int j=1;j<=w;j++)
    		{
    			b[i][j]=true;
    		}
    	for(int i=1;i<=r;i++)
    	{
    		int x,y;
    		cin>>x>>y;
    		b[x][y]=false;
    	}
    	int ans[h+1][w+1];
    	for(int i=1;i<=h;i++)
    	for(int j=1;j<=w;j++)
    	  ans[i][j]=0;
    	ans[1][1]=1;
    	for(int i=1;i<=h;i++)
    	{
    		for(int j=1;j<=w;j++)
    		{
    			if(i+2<=h && j+1<=w && b[i+2][j+1])
    			 ans[i+2][j+1]=(ans[i+2][j+1]+ans[i][j])%10007;
    			if(i+1<=h && j+2<=w && b[i+1][j+2])
    			 ans[i+1][j+2]=(ans[i+1][j+2]+ans[i][j])%10007;
    		}
    		
    	}
    	cout<<ans[h][w]%10007<<endl;
    	
    }
    return 0;
}
