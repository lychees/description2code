#include <iostream>
#include <cstdio>
#include <vector>

#define FOR0(i,n) for(i=0;i<n;++i)
#define FOR1(i,n) for(i=1;i<=n;++i)
using namespace std;

typedef long long LL ;

LL dp[20000][40], cube[40], x, i, j, n ;

int main()
{
    FOR1(i,21)
        cube[i] = i*i*i ;
    dp[0][0] = 1 ;
    FOR0(i,10001)FOR1(j,21) {
        dp[i][j] = dp[i][j-1] ;
        if( i >= cube[j] )
            dp[i][j] += dp[i-cube[j]][j] ;
    }
    int __T ;
    cin >> __T ; while( __T-- ) {
        cin >> n ;
        cout << dp[n][21] << endl ;
    }
    return 0;
}
