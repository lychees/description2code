#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
#define ull unsigned long long

int t;
ull n, m, res;

int main(){
    scanf("%d", &t);
    while (t--){
        scanf("%llu%llu", &n, &m);
        res = n/m * (m*(m - 1)/2);
        n %= m;
        res += n*(n + 1) / 2;
        printf("%llu\n", res);
    }
    return 0;
}
