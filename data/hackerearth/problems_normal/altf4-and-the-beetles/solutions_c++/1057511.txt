using namespace std;
int t;
int n;
int a;
int b;
int c;
long long int sum = 0;
int main(){
	scanf("%d", &t);
	while (t--){
		sum=0;
		cin >> n >> a >> b >> c;
		priority_queue<pair<int, int> ,vector<pair<int,int > >,greater<pair<int,int> > > q;
		for (int i = 0; i < n; i++){
			int kari;
			int kari1;
			cin >> kari >> kari1;
			kari1++;
			q.push(make_pair(kari1, -b + c));
			if (0 < kari){
				sum += a;
				q.push(make_pair(kari, -a + b));
			}
			if (kari == 0){
				sum += b;
			}
		}
		long long int ans = sum;
		while (!q.empty()){
			//cout<<sum<<endl;
			//cout<<sum<<endl;
			//cout<<q.top().first<<endl;
			int time1 = q.top().first;
			while (!q.empty()){
				if (time1 == q.top().first){
					sum += q.top().second;
					q.pop();
				}
				else{
					break;
				}
			}
			ans = max(ans, sum);
		}
		printf("%lld\n", ans);
	}
	return 0;
}
