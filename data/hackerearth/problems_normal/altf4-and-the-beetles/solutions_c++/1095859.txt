#include<bits/stdc++.h>
using namespace std;

long long max(long long a, long long b)
    {
        return a>b?a:b;
    }

long long P[20001], Q[20001], A, B, C;

int main()
    {
        long long cur, ans;
        int test, N, i, j;
        cin>>test;
        while(test--)
            {
                ans = 0;
                cin>>N>>A>>B>>C;
                for(i = 0; i<N; i++)
                  {
                      cin>>P[i]>>Q[i];
                  }
                P[N] = Q[N] = INT_MAX;
                sort(P, P+N);
                sort(Q, Q+N);

                i = 0, j = 0;
                cur = N*A;
                ans = cur;
                while(i<N && j<N)
                    {
                        if(P[i] <= Q[j])
                            {
                                cur += B-A;
                                i++;
                            }
                         else
                            {
                                cur += C-B;
                                j++;
                            }
                        ans = max(ans, cur);
                    }
                 cout<<ans<<endl;
            }
         return 0;
    }
