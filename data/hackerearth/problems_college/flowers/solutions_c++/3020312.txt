#include<bits/stdc++.h>
#include<iostream>

using namespace std;

int main() {
	int n, k;
	scanf("%d %d", &n, &k);
	int c[102];
	int nos[102] = {0};
	int sum = 0;
	for (int i = 0; i < n; ++i)
	{
		scanf("%d", &c[i]);
	}
	if (n == 1) {
		printf("%d", c[0]);
		return 0;
	}
	if (n <= k) {
		for (int i = 0; i < n; ++i)
		{
			sum += c[i];
		}
		printf("%d", sum);
		return 0;
	}
	sum=0;
	sort(c, c + n);
	int j = n - 1;
	int b = k - 1;
	while (j >= 0) {
		nos[j]++;
		sum += c[j--];
		b--;
		if (b == -1)
			break;
	}
	//d
	//cout<<sum<<endl;
	int minindex = k - 1;
	while (j>=0) 
	{
		for (int i = 0; i < k; ++i)
		{
			if(((nos[minindex]+1)*c[j]) > ((nos[i]+1)*c[j]))
			{
				minindex = i;
			}
		}
		nos[minindex]++;
		//cout<<minindex<<":"<<nos[minindex]<<endl;
		sum+=((nos[minindex]+1)*c[j]);
		j--;
	}
	printf("%d", sum);
	return 0;
}
