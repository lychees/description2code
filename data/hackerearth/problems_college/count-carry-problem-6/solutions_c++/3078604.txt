#include<bits/stdc++.h>
using namespace std;
set <string> v;
map <string,int> map1;
int main()
{
	ios_base::sync_with_stdio(false);
	long t,carry,a,b,count;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		carry=0;count=0;
		while(a!=0&&b!=0)
		{
			carry=a%10+b%10+carry;a/=10;b/=10;carry/=10;
			if(carry)
				count++;
		}
		if(count)
			cout<<count<<" ";
		else
			cout<<"No"<<" ";
		cout<<"carry operation";
		if(count>1)
			cout<<"s"<<endl;
		else
			cout<<endl;
	}
	return 0;
}
