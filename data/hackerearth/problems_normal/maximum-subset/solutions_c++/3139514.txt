#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define ll long long


ll power(ll a,ll b,ll mode)
{
    ll x=1,y=a;
    while(b>0)
    {
        if(b%2)
        {
            x=(x*y)%mode;
        }
        y=(y*y)%mode;
        b/=2;
    }
    return x;
}

ll modinverse(ll num,ll mode)
{
    return power(num,mode-2,mode);
}

int main()
{
    ll t;
    cin>>t;
    while(t--){
    	ll n,i,j,k,tmp,cnt,ans,res,mx,num,denom;
    	cin>>n;
    	ll arr[n];
    	for(i=0;i<n;i++) cin>>arr[i];
    	sort(arr,arr+n);
    	ans=0;
    	for(i=0;i<(1<<n);i++){
    		vector<ll> v;
    		for(j=0;j<n;j++){
    			if(i&(1<<j)){
    				v.push_back(arr[j]);
    			}
    		}
    		
    		if(!v.empty() && (v.size()%2==0)){
				num=denom=1;
			
				for(j=0;j<(v.size())/2;j++){
					denom=(denom*v[j]+mod)%mod;
				}
				
				for(j=(v.size()/2);j<v.size();j++){
					num=(num*v[j]+mod)%mod;
				}
				tmp = (num*modinverse(denom,mod)+mod)%mod;
				ans=max(ans,tmp);
    		}
    	}
    	
    	cout<<ans<<endl;
    }
    return 0;
}
