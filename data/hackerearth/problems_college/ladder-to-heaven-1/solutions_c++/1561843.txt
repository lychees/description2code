#include <iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;
int main()
{
	int t;
	long long int a[100000];
	a[0]=1;
	a[1]=2;
	for(int i=2;i<100000;i++)
	{
		a[i]=a[i-1]+a[i-2];
	}
	cin>>t;
	while(t--)
	{
		
		long long int n;
		cin>>n;
		long long int sum=0;
		for(int i=1;i<100000;i=i+3)
		{
			if(a[i]>n)
			break;
			else if(a[i]%2==0)
			sum+=a[i];
			
		}
		cout<<sum<<endl;
	}
}
