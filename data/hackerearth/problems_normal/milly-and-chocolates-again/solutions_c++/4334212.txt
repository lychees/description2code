#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	int box[n];
	int sum=0;
	for(int i=0;i<n;i++){
		cin>>box[i];
		sum+=box[i];
	}
	int arr[sum+1];
	int k=1;
	for(int i=0;i<n;i++){
		for(int j=0;j<box[i];j++){
			arr[k]=i+1;
			k+=1;
		}
	}
	int q;
	cin>>q;
	while(q--){
		int index;
		cin>>index;
		cout<<arr[index]<<endl;
	}
	return 0;
}
