
#include<bits/stdc++.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        long long int o=1,z=0,i;
        for(i=0;i<n;i++)
        {
            long long int temp=z;
            z=z+o;
            o=temp;
        }
        printf("%lld %lld\n",o,z);
    }
    return 0;
}
