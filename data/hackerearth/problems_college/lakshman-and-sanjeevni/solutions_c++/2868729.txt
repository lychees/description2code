#include <iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;

int A[10000001]={0};
long long int e[10000001]={0};
 void square_free()
{
 long int i,j;
 long long k;
 for(i=2;i<=3162;i++)
 {
 	k=i*i;
  if(A[i]==0)
   for(j=k;j<=10000000;j+=k)
    A[j]=-1;
  
 }
 
}
void divisor(long long int y)
{
	long long int p;
	
	for(p=1;p<=sqrtl(y);++p)
	{
		if(y%p==0)
		{
			
			if(A[p]!=-1)
				++e[y];
				
			if(p!=(y/p))
				if(A[y/p]!=-1)
					++e[y];
			
		}
	}
	
}
int main() {
	// your code goes here
	long long int n,i;
	scanf("%lld",&n);
	long long int a;
	long long int sum=0;
	square_free();
	for(i=0;i<n;++i)
	{
	   scanf("%lld",&a);
	   if(e[a]!=0)
	   	sum=sum+e[a];
	   else
		{
		divisor(a);
	//	cout<<e[a]<<" ";
		sum=sum+e[a];
		}
		
	}
	
	printf("%lld\n",sum);
	return 0;
}
