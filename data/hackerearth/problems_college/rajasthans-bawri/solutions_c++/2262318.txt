#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int px,py,bx,by;
		cin>>px>>py>>bx>>by;
		
		int flag=0;
		
		if(px>=bx && py>=by){
			flag=1;
		}else if(py>=bx && px>=by){
			flag=1;
		}
		
		if(flag==1){
			cout<<"Possible"<<"\n";
		}else{
			cout<<"Not Possible"<<"\n";
		}
	}
    return 0;
}
