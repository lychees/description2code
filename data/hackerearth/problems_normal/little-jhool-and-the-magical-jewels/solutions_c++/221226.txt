#include <iostream>
#include<string.h>
using namespace std;

int main()
{
 int t,i,r,u,b,y,l,c;
 char A[101];
 cin>>t;
 while(t)
 {
  cin>>A;
  l=strlen(A);
  c=0;
  r=0;
  u=0;
  b=0;
  y=0;
  for(i=0;i<l;i++)
  {
  	if(A[i]=='r')
  	 r++;
  	else if(A[i]=='u')
  	 u++;
  	else if(A[i]=='b')
  	 b++;
  	else if(A[i]=='y')
  	 y++;
  }
  if((r<=u)&&(r<=b)&&(r<=y))
   c=r;
  else if((u<=r)&&(u<=b)&&(u<=y))
   c=u;
  else if((b<=r)&&(b<=u)&&(b<=y))
   c=b;
  else if((y<=r)&&(y<=b)&&(y<=u))
   c=y;
  cout<<c<<"\n";
  t--;
 }
 return 0;
}
