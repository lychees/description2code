#include <iostream>
using namespace std;
void selection_sort(int a[],int n) {
  int i,j,temp,small;
  int index;
  for(i=0;i<n;i++)
  { small=a[i];
    index=i;
    for(j=i;j<n;j++) {
      if(a[j]<small) {
        small=a[j];
        index=j;
      }
    }
    temp=a[i];
    a[i]=small;
    a[index]=temp;
  }
}

int main()
{ int ceo_height,coo_height,cto_height;
  cin>>ceo_height>>coo_height>>cto_height;
  int n;
  cin>>n;
  int H[n];
  for(int i=0;i<n;i++)
  {
    cin>>H[i];
  }
  selection_sort(H,n);
  int A[n+3];
  int mid=(n+3)/2;
  A[mid]=ceo_height;
  if(coo_height>cto_height)
  {
    A[n+2]=coo_height;
    A[0]=cto_height;
  }
  else
  {
    A[n+2]=cto_height;
    A[0]=coo_height;
  }
  int j=0;
  for(int i=0;i<(n+3);i++)
  {
    if((i==0)||(i==mid)||(i==(n+2)))
    {
      continue;
    }
    else
    {
      A[i]=H[j];
      j++;
    }
  }
  int C[n+2];
  j=0;
  for(int i=0;i<n+2;i++)
  {
    C[j]=A[i+1]-A[i];
    if(C[j]<0)
    {
      C[j]=-1*C[j];
    }
    j++;
  }
  int sum=0;
  for(int i=0;i<n+2;i++)
  {
    sum=sum+C[i];
  }
  cout<<sum;
  return 0;
}
