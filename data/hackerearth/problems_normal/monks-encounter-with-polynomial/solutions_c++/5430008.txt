#include <iostream>
using namespace std;
long int a,b,c,k, ans;
int bin_search(long int low, long int high){
	long int x;
	
	if(c>=k){
		return 0;
	}
	else{
		x=1;
		low=0;
		high=x;
		while((((a*(x*x))+(b*x)+c))<k){
			low=x;
			x*=2;
			high=x;
		}
		
		while((high-low)>1){
			int mid=(high+low)/2;
			if((((a*(mid*mid))+(b*mid)+c))<k){
				low=mid;
			}
			if((((a*(mid*mid))+(b*mid)+c))>=k){
				high=mid;
			}
			
		}
		return high;
	}
}
int main()
{
	int t;
	cin >> t;
	while(t--){
		cin >> a;
		cin >> b;
		cin >> c;
		cin >> k;
		ans = bin_search(0, 1);
		cout << ans << endl;
	}
    return 0;
}
