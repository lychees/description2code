#include <bits/stdc++.h> // Firstest code for length.
using namespace std;
char st[100002];
int main()
{
    int t,n,len,ans;
    scanf("%d",&t);
    for(int j=1;j<=t;j++)
    {
        scanf("%s",st);
        ans=0;
        len=strlen(st);
        for(int i=0; i<len; i++)
        {
            if(st[i]!='B'&&st[i]!='D'&&st[i]!='S'&& st[i+1]!='B'&&st[i+2]!='B'&&st[i+1]!='S')
            {
                if(i>0 && st[i-1]!='S')
                ans++;
                else  if(i==0) ans++;

            }
        }
        printf("%ld\n",ans);
    }

}
