#include <bits/stdc++.h>

using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    vector <int> a(n);
    for (int i = 0; i < n; i++)
    	scanf("%d",&a[i]);
    sort(a.begin(),a.end());
    int ans = 987654321;
    int ctr = 0;
    for (int i = 0; i < n; i++)
    {
    	ctr = 0;
    	int num = a[i]*3;
    	int id;
    	id = upper_bound(a.begin(),a.end(),num)-a.begin();
    	ctr = i+(n-id);
    	ans = min(ans,ctr);
    }
    cout << ans;
}
