#include <bits/stdc++.h>
using namespace std;

#define ll long long int
int main() 
{
	ll a[1000000],n,i,q,l,r,k;
	scanf("%lld",&n);
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		
	ll *start,*end;
	start=new ll[n];
	end=new ll[n];
	
	scanf("%lld",&q);
	while(q--)
	{
		scanf("%lld %lld %lld",&l,&r,&k);
		start[l-1]+=k;
		end[r-1]+=k;
	}
	
	ll inc=0;
	for(i=0;i<n;i++)
	{
		inc+=start[i];
		printf("%lld\n",(a[i]+inc));
		inc-=end[i];
	}
	return 0;
}
