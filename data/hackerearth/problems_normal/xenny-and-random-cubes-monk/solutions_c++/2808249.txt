#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
	int n,k;
	char s[15],a;
	long long hash[42][42];
	for(int i=0;i<42;i++)
	for(int j=0;j<42;j++)
		hash[i][j]=0;
	cin>>n>>k;

	for(int i=0;i<n;i++)
		for(int j=0;j<6;j++)
		{
			cin>>a;
			hash[i][a-'a']++;
		}

	cin>>s;
	long long ans=0;
	for(int i=0;i<(1<<n);i++)
	{
	
		if(__builtin_popcount(i) == k )
		{
			vector<int> index;
			int count=0,val;
			 count=0;
			 val=i;
			 while(val)
			 {
			 	if(val & 1)
			 		index.push_back(count);
			 	count++;
			 	val>>=1;
			 }
		
			sort(index.begin(),index.end());
			do{
			 long long ways=1;
			 for(int q=0 ; q<k ; q++)
			 {
				int ch=s[q]-'a';
				int value=index[q];
				if(hash[value][ch]>0)
				{
					ways=(ways * hash[value][ch]);
					ways%=mod;
				}
				else ways=0;
			}
			ans=ans+ways;
					ans%=mod;
			}while(next_permutation(index.begin(),index.end()));
		}
		
	}
	cout<<ans%mod<<endl;
	return 0;
}
