void mulMat(ll A[2][2],ll B[2][2]){
    ll C[2][2];
    rep(i,2)
        rep(j,2){
            C[i][j]=0;
            rep(k,2)
                C[i][j]=(C[i][j]+ (A[i][k]*B[k][j])%mod)%mod;
        }
    rep(i,2)rep(j,2)A[i][j]=C[i][j];
}

void mul(ll M[2][2],ll x, ll ret[2][2]){
    rep(i,2)rep(j,2)ret[i][j]=(i==j);

    while(x > 0){
        if(x&1){
            x--;
            mulMat(ret,M);
        }
        else{
            x>>=1;
            mulMat(M,M);
        }
    }
}

ll solve(ll n){
    if(n<=1) return n;
    ll M[2][2]={{1,1},{1,0}};
    ll ret[2][2];
    mul(M,n-1,ret);
    
    /*rep(i,2){
        rep(j,2)cout<<ret[i][j]<<" ";
        cout<<endl;
    }*/
    
    ll fn = ret[0][0];
    ll fn_1 = (ret[0][0]+ret[1][0])%mod;
    return (fn * fn_1)%mod;
}
