    #include<cstdio>
    #include<iostream>
    using namespace std;
    int main()
    {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    		int n,i;
    		cin>>n;
    		if(n==3)
    		{
    		for(i=0;i<n;i++)
    		{
    			cout<<"5";
    			}
    			cout<<"\n";	
    		}
    		else
    		if(n==5)
    		{
    			for(i=0;i<n;i++)
    			{
    			cout<<"3";
    		}
    		cout<<"\n";
    		}
    		else
    		{
    			int x=0,y=0,flag =0;
    			for(i=3;i<=n;i=i+3)
    			{
    				if((n-i)%5==0)
    				{
    					flag =1;
    					x = i;
    					y = n-i;
    				}
    			}
    			for(i=0;i<x;i++)
    			{
    				cout<<"5";
    			}
    			for(i=0;i<y;i++)
    			{
    				cout<<"3";
    			}
    			if(flag ==0)
    			{
    				cout<<"-1";
    			}
    			cout<<"\n";
    		}
    	}
    }
