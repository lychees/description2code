def process(arr, ip, r, c):
	currentRow = 0
	currentCol = c-1
	while(currentRow>=0 and currentCol>=0 and currentRow < r and currentCol < c) :		
		if arr[currentRow][currentCol] == ip:
			return [currentRow,currentCol]
		elif arr[currentRow][currentCol] > ip:
			currentCol = currentCol-1			
		else:
			currentRow = currentRow+1
	return [-1,-1]

temp = raw_input()

for i in range(N):
