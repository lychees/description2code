#include <bits/stdc++.h>
using namespace std;
map<int,int> ma;
int main()
{
    string str; 
    int maxn;
    getline(cin,str);
    
    for(int i=0;str[i];i++)
    {
    	ma[str[i]]++;
		    	
    }
    maxn=0;
    map<int,int>::iterator itr;
    for(itr=ma.begin();itr!=ma.end();itr++)
    {
    	if(maxn<itr->second)
    		maxn=itr->second;
    }
    map<int,int>::iterator it;
    for(it=ma.begin();it!=ma.end();it++)
    {
    	if(it->second==maxn)
		{
			cout<<(char)it->first<<" "<<maxn<<endl;
			break;
		}
    }
    return 0;
}
