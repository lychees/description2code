#include<bits/stdc++.h>
using namespace std;
int a[100005];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a,b,c;
		cin>>a>>b>>c;
		int pos = 0,neg = 0;
		for(int i=0;i<n;i++)
		{
			int x,y;
			cin>>x>>y;
			if((a*x+b*y+c)==0)
			continue;
			else if((a*x+b*y+c)<0)
			neg++;
			else
			pos++;
		}
		cout<<max(pos,neg)<<endl;
		
	}
	return 0;
}
