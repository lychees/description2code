#include <bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
   int n;
   int t,i,x;
   scanf("%d",&t);
   while(t--)
   {
   	scanf("%d",&n);
   	vector<int> v;
   	for(i=0;i<n;i++)
   	{
   		scanf("%d",&x);
   		v.push_back(x);
   	}
   	sort(v.begin(),v.end());
   	printf("%d\n",2*(v[n-1]-v[0]));
   }
    return 0;
}
