#include <iostream>
using namespace std;
long long int reverse(long long int n)
{
	long long int rev=0;
	while(n>0)
	{
		rev=10*rev+n%10;
		n=n/10;
	}
	return (rev);
}
int main()
{
    int t;
    cin>>t;
    long long int n,r1;
    long long int p,q;
    while(t--)
    {
    	cin>>n;
    	r1=reverse(n);
    	p=r1*r1;
    	q=n*n;
    	if(n==r1)
    	{
    		cout<<"No"<<endl;
    	}
    	else
    	{
    		if(reverse(p)==q)
    		{
    			cout<<r1<<endl;
    		}
    		else
    		{
    			cout<<"No"<<endl;
    		}
    	}
    }
    return 0;
}
