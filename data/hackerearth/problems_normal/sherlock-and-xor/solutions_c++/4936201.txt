#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
   	while(t--){
   		int n;
   		cin >> n;
   		long int num;
   		long int even=0,odd=0;
   		long int odd_pairs=0;
   		for(int i=0;i<n;i++){
   			cin >> num;
   			if(num%2)
   				odd++;
   			else
   				even++;
   		}
   		odd_pairs = odd*even;
   		cout << odd_pairs << "\n";
   		
   	}
   	return 0;
}
