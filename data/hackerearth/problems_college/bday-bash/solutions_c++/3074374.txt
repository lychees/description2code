#include <stdio.h>

int main()
{
	int i, t, n1, n2, n3, cake, bf, sf, f;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
    	scanf("%d %d %d %d",&cake, &sf, &bf, &f);
    	cake = cake - (100*sf) - (50*bf) - (25*f);
    	n1 = 100*sf;
    	n2 = 50*bf;
    	n3 = 25*f;
    	
    	if(cake<0)
    		printf("No\n");
    		
    	else if (cake == 0)
    		printf("%d %d %d\n",100*sf, 50*bf, 25*f);
    		
    	else
    		{
    			while(cake>0)
    			{
    				
	    			if( cake >0 && (cake/100) >= sf)
	    				{
	    					n1 = n1 + 100*sf;
	    					cake -=  (sf*100);
	    				}
	    			if(cake > 0 && (cake/50) >= bf)
		    			{
		    				n2 += 50*bf;
		    				cake -= (bf*50);
		    			}
	    			
	    			if(cake > 0 && (cake/25) >= f)
		    			{
		    				n3 += 25*f;
		    				cake -= (f*25);
		    			}
		    		else 
		    			break;
    			}
    			if(cake>0)
    				printf("No\n");
    			
    			else
	    			printf("%d %d %d\n",n1, n2, n3);
    		}
    		
    //	printf("%lld %lld %lld\n",n1, n2, n3);
    }
}
