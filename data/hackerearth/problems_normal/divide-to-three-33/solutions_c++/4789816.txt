#include<stdio.h>

#define MAX 13
int N, ans;
int arr[MAX];

void solve(int s1, int s2, int s3, int idx)
{
//printf("s1 - %d s2 - %d s3 - %d\n",s1,s2,s3);
if(idx==N && (s1>=s2 && s2>=s3))
{
ans = (ans<s1)?ans:s1;
}
else
{
if(idx<N)
{	
solve(s1+arr[idx],s2,s3,idx+1);
solve(s1,s2+arr[idx],s3,idx+1);
solve(s1,s2,s3+arr[idx],idx+1);
}
}
}

int main()
{
int i,t1,t2,t3;
t1=t2=t3=0;
ans=9999;
scanf("%d",&N);
for(i=0;i<N;i++)
scanf("%d",&arr[i]);

solve(t1,t2,t3,0);
printf("%d\n",ans);
return 0;
}
