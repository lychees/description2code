#include<iostream>
using namespace std;
int main()
{
int n,min=-1;
while(cin>>n)
{
    int arr[n],sum=0,i;
    for(i=0;i<n;i++)
    {
    cin>>arr[i];
    sum=sum+arr[i];
    }
    if (sum%2==0)
    {
    for(i=0;i<n;i++)
    {
        if ((min==-1)&&(arr[i]%2==1))
        min=arr[i];
        if (min>arr[i])
        min=arr[i];
    }
    }
    if (sum%2==1) cout<<sum<<endl;
    else if(min==-1) cout<<":(\n";
    else cout<<sum-min<<endl;
}
return 0;
}
