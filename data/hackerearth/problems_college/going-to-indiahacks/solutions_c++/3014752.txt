#include <iostream>
#define mod 1000000007
#define ll long long
using namespace std;
ll arr[1005][1005];
int main()
{
	ll i, j, a, b, c, d, t;
    for(i=1;i<=1004;i++)
    {
    	arr[i][1]=1;arr[1][i]=1;
    }
    for(i=2;i<=1004;i++)
    {
    	for(j=2;j<=1004;j++)
    	{
    		arr[i][j]=(arr[i-1][j]+arr[i][j-1]+arr[i-1][j-1])%mod;
    	}
    }
    cin>>t;
    while(t--)
    {
    	cin>>a>>b>>c>>d;
    	cout<<arr[c-a+1][d-b+1]<<endl;
    }
    return 0;
}
