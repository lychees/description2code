#include<bits/stdc++.h>
#define N 200005
using namespace std;

long a[N],m[N];

int main(void)
{
    long t,n,k,c=-1;
    std::ios::sync_with_stdio(0);
    cin>>t;
    while(t--)
    {
        c=-1;
        long ans=0;
        cin>>n>>k;
        for(int i=0;i<n;i++)
        {
            m[i]=-1;
            cin>>a[i];
            if(a[i]>k) m[i]=i;
        }
        for(int i=n-1;i>=0;i--)
        {
            if(m[i]!=-1) c=m[i];
            m[i]=c;
        }
        for(int i=0;i<n;i++)
        {
            if(m[i]==i) ans+=(n-i);
            else if(m[i]!=-1) ans+=(n-m[i]);
        }
        cout<<ans<<'\n';
    }
    return 0;
}
