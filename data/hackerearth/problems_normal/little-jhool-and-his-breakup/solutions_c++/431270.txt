#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

int main()
{
	int count,i,j;
	char s[100000];
	count = 0;
	scanf("%s",s);
	for (i = 0; i < strlen(s); i++) {
		if (s[i] == 'l') {
			count++;
			break;
		}
	}
	for (j = i+1; j < strlen(s); j++) {
		if (s[j] == 'o') {
			count++;
			break;
		}
	}
	for (i = j+1; i < strlen(s); i++) {
		if (s[i] == 'v') {
			count++;
			break;
		}
	}
	for (j = i+1; j < strlen(s); j++) {
		if (s[j] == 'e') {
			count++;
			break;
		}
	}
	if (count == 4) {
		printf("I love you, too!\n");
	} else {
		printf("Let us breakup!\n");
	}
	
	return 0;
}
