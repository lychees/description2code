#include <iostream>
using namespace std;
#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
	char s[1000001];
	unsigned long long int t,i1=0;
	scanf("%llu",&t);
	getchar();
	while(i1<t)
	{
		gets(s);
		unsigned long long int c=0,i=0,d=0,c1=1,j=0;
		while(i<strlen(s))
		{
			if(s[i]=='H')
			{
				c+=1;
				if(i+1<strlen(s) && s[i+1]=='H')
				{
					c1=c+1;
					j=i+1;
					while(j<strlen(s))
					{
						if(s[j]=='T')
						{
							break;
						}
						else
						{
							i=j;
							c+=c1;c1+=1;
						}
						j+=1;
					}
						
				}
			}
			d+=c;
			c=0;
			i+=1;
		}
		i1+=1;
		printf("%llu\n",d);
	}
	return 0;
}
