#include<bits/stdc++.h>
using namespace std;
int main()
{
       int t;
       cin>>t;
       while(t--)
       {
            long long  n,m;
            cin>>n>>m;
            if(n<m or m==1)
            {
                cout<<n<<endl;
            }
            else
            {

                long long sum=0;
                while(n>0)
                {
                  sum=sum+(n%m);
                  n/=m;

                }
                cout<<sum<<endl;


            }
       }

       return 0;
}
