#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int arr[1000];
    arr[0]=1;
    arr[1]=1;

    for(int i=2;i<1000;i++){
         arr[i] = arr[i-1] + arr[i-2];
    }
    while(t--)
    {
    	int a,b,c=0;
    	cin>>a>>b;
    	
    	for(int i=a;i<=b;i++)
    	{
    		for(int j=0;j<1000;j++)
    		{
    			if(i==arr[j])
    			c++;
    		}
    	}
    	
    		if(c%2==0)
    		cout<<"VALID"<<endl;
    		else
    		cout<<"INVALID"<<endl;
    	
    }
    return 0;
}
