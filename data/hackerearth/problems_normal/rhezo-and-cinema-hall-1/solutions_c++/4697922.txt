#define bit(x,y) (((x)>>(y))&1)
const int mod = 1e9 + 7;
int n, m, k;
vector<int> a;
int dp[505][2505][2];

void prepare() {
	memset(dp, -1, sizeof(dp));
	for (int mask = 0; mask < (1 << m); mask++) {
		bool flag = true;
		for (int j = 0; j < m; j++) if (bit(mask, j) && bit(mask, j + 1)) {
				flag = false;
				break;
			}
		if (flag && __builtin_popcount(mask) >= 2) {
			a.push_back(__builtin_popcount(mask));
		}
	}
}

int cal(int level, int cur, bool ok) {
	if (level == 0) return ok;
	int &ret = dp[level][cur][ok];
	if (ret != -1) return ret;
	ret = 0;
	for (int i = 0; i < a.size(); i++) {
		bool _ok = ok; if (cur + a[i] >= k) _ok = true;
		ret += cal(level - 1, cur + a[i], _ok);
		if (ret >= mod) ret -= mod;
	}
	return ret;
}
int main() {
	ios_base::sync_with_stdio(false); cin.tie(0);
	cin >> n >> m >> k;
	prepare();
	cout << cal(n, 0, 0);
}
