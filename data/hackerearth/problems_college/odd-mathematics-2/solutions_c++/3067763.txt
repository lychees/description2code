#include<stdio.h>
int main ()
{
   int n;
   long long n1,i,j,sum;
   scanf("%d",&n);
   while(n--)
   {
     scanf("%lld",&n1);
     sum=0;
     while(n1)
     {
     if(n1%2==1)
        j=n1/2+1;
     else
        j=n1/2;
     sum+=j*j;
     n1=n1-j;
     }
     printf("%lld\n",sum);
   }
}
