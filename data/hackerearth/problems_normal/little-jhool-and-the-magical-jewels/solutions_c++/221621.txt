#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int min(int a,int b,int c,int d)
{
	int i;
	int var[4]={a,b,c,d};
    int minimum=var[0];
    for(i=1;i<4;i++)
    {
    	if(var[i]<minimum)
    	 minimum=var[i];
    }
    return minimum;
}
int main()
{
  int tc,len,i,minimum;
  char c[100];
  cin>>tc;
  while(tc)
  {
  	int countr=0,countu=0,countb=0,county=0;
  	cin>>c;
  	len=strlen(c);
  	for(i=0;i<len;i++)
  	{
  		if(c[i]=='r')
  		 countr++;
  		if(c[i]=='u')
  		 countu++;
  		if(c[i]=='b')
  		 countb++;
  		if(c[i]=='y')
  		 county++;
   	}
  	minimum=min(countr,countu,countb,county);
  	cout<<minimum<<"\n";
	tc--;   
  }
  
}
