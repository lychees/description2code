#include <bits/stdc++.h>
#define n1 1000001
using namespace std;

int main()
{
	long long int t,i,j,k,l,m,n,o,p;
	cin>>t;
	while(t--)
	{
		vector <bool> a(n1,false);
		cin>>n;
		p=0;
		o=0;
		for(i=0;i<n;i++)
		{
			cin>>j>>k;
			
			if(k==1)
			{
			if(a[j]==0)
			{
				a[j]=1;
				o++;
			}
			}
			
			else
			{
				p=max(p,o);
				o=0;
			}
			
		}
		
		p=max(o,p);
		
		cout<<p<<"\n";
	}
}
