#define rep(i,a,b) for(int i = a; i < b; i++)
#define S(x) scanf("%d",&x)
#define S2(x,y) scanf("%d%d",&x,&y)
#define P(x) printf("%d\n",x)
#define all(v) v.begin(),v.end()
#define FF first
#define SS second

typedef long long int LL;
typedef pair<int, int > pii;
typedef vector<int > vi;

const int N = 20005;

set<int > B;
int ans[N];

const int M = 300000;
int X[21][M];
vi res;

int main() {
  int n,K,m;
  scanf("%d%d%d",&n,&K,&m);
  rep(i,0,m) {
    int x;
    S(x);
    B.insert(x);
  }

  int idx = 1;

  K = min(K, 20);

  res.push_back(1);

  rep(i,1,K+1) X[i][1] = 1;

  rep(i,2,M) if(!X[K][i]) {
    if(B.find(i) != B.end()) continue;
    res.push_back(i);
    if(res.size() >= n) break;
    rep(j,1,K+1) X[j][i] = 1;
    for(int j = 2*i; j < M; j += i) {
      rep(k,1,K+1) {
        LL pw = i;
        int cnt = 1;
        X[k][j] |= X[k-1][j];
        while(pw < M && !X[k][j] && j % pw == 0 && cnt <= k) {
          X[k][j] |= X[k-cnt][j/pw];
          pw *= i;
          cnt++;
        }
      }
    }
  }

  rep(i,0,n) printf("%d ",res[i]); printf("\n");

  return 0;
}
