#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
    	string input;
    	cin >> input;
    	
    	int ans = 0;
    	int k = count(input.begin(), input.end(), 'K');
    	for(int i = 0, r = 0; i < input.size(); i++)
    	{
    		r += input[i] == 'R';
    		k -= input[i] == 'K';
    		ans = max(ans, min(r, k));
    	}
    	cout << ans * 2 << '\n';
    	
    }
    return 0;
}
