#include <iostream>
#include<map>
#include<vector>
using namespace std;
#define boost std::ios::sync_with_stdio(false);
int main()
{   boost;
    short int t,i,k;
    char s[1000000],b;
    long int j;
    map<char,int> ctr;
    map<int,vector<char>> ap;
    scanf("%ld",&t);
    for(i=0;i<t;i++){
    	scanf("%s",s);
    	k=-1;
    	for(j=0;s[j]!='\0';j++){
    		ctr[s[j]]++;
    	}
    
    	for(b='z';b>='a';b--){
    	auto it=ctr.find(b);
    		if(ctr.find(b)==ctr.end()){
    			ap[0].push_back(b);
    		}
    		else{
    			ap[it->second].push_back(b);
    		}
    		
    	}
    	for(auto it=ap.begin();it!=ap.end();it++){
    	  for(auto t1=(it->second).begin();t1!=(it->second).end();t1++){
    	  	printf("%c ",*t1);
    	  }
        }
        printf("\n");
    	
    	ap.clear();
    	ctr.clear();
    }
    return 0;
}
