#include "iostream"
#include "climits"
#include "cmath"
#include "cstring"
#define lli long long int
#define m 1000000007
using namespace std;

lli power(lli a, lli b)
{
lli x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>m) x%=m;
        }
        y = (y*y);
        if(y>m) y%=m;
        b /= 2;
    }
    return x;
}

int main()
{
    lli t,n,k,a[20],b,x,sum,q,add;
    cin>>t;
    while(t--)
    {
        memset(a,0,sizeof(a));
        lli max_=INT_MIN;
        lli min_=INT_MAX;
        lli maxl=INT_MIN;
        lli minl=INT_MAX;
        cin>>n>>k;
        for(int i=0;i<k;i++)
        {
            for(int j=0;j<n;j++)
            {
                cin>>x;
                a[j]=a[j]+x;
            }
        }
        b=pow(2,n);
        for(int i=1;i<b;i++)
        {
            x=1;
            sum=0;
            for(int j=0;j<n;j++)
            {
                if(i&(1<<j))
                {
                    x=((x%m)*(a[j]%m))%m;
                    sum++;
                }
            }
          //  cout<<"h "<<x<<"\n";
            q=power(sum,m-2);
            add=((x%m)*(q%m))%m;
            if(max_<add)
            {
                max_=add;
                maxl=sum;
            }
            if(min_>add)
            {
                min_=add;
                minl=sum;
            }
            if(max_==add && maxl>sum)
            {
                maxl=sum;
            }
            if(min_==add && minl<sum)
            {
                minl=sum;
            }
        }
     //   cout<<max_<<" "<<min_<<" "<<maxl<<" "<<minl<<"\n";
        lli ans1=(max_^min_);
        lli ans2=(maxl^minl);
        cout<<ans1<<" "<<ans2<<"\n";
    }
    return 0;
}
