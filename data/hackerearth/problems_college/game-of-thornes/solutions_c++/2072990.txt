#include<iostream>
#include <math.h>
#include <stdlib.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int even=0,odd=0,i;
    int a[26]={0};
    for(i=0;i<s.length();i++)
    {
        a[s[i]-97]++;
    }
    for(i=0;i<26;i++)
    {
        if(a[i]%2==0)
        {
            even++;
        }
        else{
            odd++;
        }
    }

    if(s.length()%2==0)
    {
        if(odd)
        {
            cout<<"NO"<<endl;
        }
        else{
            cout<<"YES"<<endl;
        }
    }
    else{
        if(odd!=1)
        {
            cout<<"NO"<<endl;
        }
        else{
            cout<<"YES"<<endl;
        }
    }

    return 0;
}
