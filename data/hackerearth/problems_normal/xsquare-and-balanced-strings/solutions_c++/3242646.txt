#include <iostream>
using namespace std;
int main()
{
    int array[26]={0};
    int T; cin >> T;
    while(T--)
    {
    	for(int k=0; k<26; k++) array[k] = 0;
    	string str; cin >> str;
    	for(int i=0; i<str.length(); i++)
    		array[str[i]-'a']++;
    	bool isBalanced = true;
    	for(int k=0; k<26; k++)
    	{
    		if(array[k]!=0)
    		{
    			if(array[k]%2!=0){  isBalanced = false; break; }
    		}
    	}
    	cout << (isBalanced?1:-1) << endl;
    }
    return 0;
}
