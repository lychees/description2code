#include <iostream>
#include <cstdlib>

using namespace std;

int main()

{
int t;
cin >>t;

while (t--)
{
long long int w,n;
long long int array[100000];
cin>>w>>n;
for (int i=0;i<n;i++)
{
cin>>array[i];
}
long long int max=-1;
for (int m=0;m<n;m++)
{
if (array[m]==1)
  {
max=m;
  }
}
max=max+1;
long long int s=max*2+(n-max);
long long int answer=w*n+s;
cout<<answer<<endl;
}
}
