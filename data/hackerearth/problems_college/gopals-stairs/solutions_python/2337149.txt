dp={1:1,2:2,3:4}
def ans(n):
    if n in dp:
        return dp[n]
    else:
        l=[]
        l.append(1)
        l.append(2)
        l.append(4)
        for i in range(4,n+1):
        	if i not in dp:
        		l.append(l[i-2]+l[i-3]+l[i-4])
        		dp[i]=l[i-2]+l[i-3]+l[i-4]
        	else:
        		l.append(dp[i])
        dp[n]=l[n-1]
        return l[n-1]
tc=int(input())
while tc>0:
    n=int(input())
    a=ans(n)
    print(a%(1000000007))
    tc=tc-1
