#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
ll ans[52][52],arr[52][52],C[52][52];
void multiply(ll A[][52],ll B[][52],int n){
for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
        C[i][j] = 0;
        for(int k=0;k<n;k++){
            C[i][j] = (C[i][j]+(A[i][k]*B[k][j])%MOD)%MOD;
        }
    }
}
for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
        A[i][j] = C[i][j];
    }
}
}
void solve(ll k,int n){
    while(k){
        if(k&1)
            multiply(ans,arr,n);
        multiply(arr,arr,n);
        k >>= 1;
    }
}
int main(){
int t,n,u,v;
ll k;
cin >> t;
while(t--){
    cin >> n >> k >> u >> v;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            scanf("%d",&arr[i][j]);
            ans[i][j] = arr[i][j];
        }
    }
    k--;
    solve(k,n);
    cout << ans[u][v] << endl;
}
return 0;
}
