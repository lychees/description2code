#include<cstdio>
#include<iostream>
#include<vector>
using namespace std;
const int INF=100000000;
int dp[600][600];
int rev[600][600];
int l[600];
int r[600];
int c[600];

int main(){
	int tc;
	scanf("%d",&tc);
	while(tc--){
		int n,m,k;
		cin >> n;
		cin >> m;
		int i;
		int j;
		for(i=0;i<=n;i++){
			for(j=1;j<=n+1;j++){
				rev[i][j]=INF;
				dp[i][j]=INF;
			}
		}
		for(i=0;i<m;i++){
			cin >> l[i];
			cin >> r[i];
			cin >> c[i];
		//	l[i]++;
		//	r[i]++;
			for(j=l[i];j<=r[i];j++){
				if(rev[j][j-l[i]] > c[i]){
					rev[j][j-l[i]]=c[i];
				}
			}
		}
		for(i=0;i<=n;i++){
			dp[i][0]=0;
		}
		for(i=1;i<=n;i++){
			dp[0][i]=INF;
		}
		dp[0][0]=0;
		int minn=INF;
		int p;
		int y;
		int cov;
		for(i=1;i<=n;i++){
			for(j=1;j<=i;j++){
				minn=INF;
				if(dp[i-1][j]  < minn){
					minn=dp[i-1][j];
				}
				for(p=0;p<=j;p++){
					if(dp[i-p][j-p]+rev[i][p] < minn){
						minn=dp[i-p][j-p]+rev[i][p];
					}
				}
				dp[i][j]=minn;
			}
		}
		if(dp[n][n]==INF){
			printf("-1\n");
		}
		else{
			cout << dp[n][n] << endl;
		}
	}
	return 0;
}
