#include<iostream>
using namespace std;
int gcd(int a,int b)   // calculating gcd for finding the lcm of A and B
{
    if(a==b)
        return a;
    else if(b>a)
        return gcd(a,b-a);
    else
        return gcd(b,a-b);
}
int main()
{
    int t,value,j,z,x,i,k;
    cin>>t;
    for(i=0;i<t;i++)
    {
    value=0,x=0;
        cin>>z;
        for(j=1;j<=z;j++)
        {
            if(gcd(z,j)==1)
            {
                x++;
                //cout<<x<<" "<<j<<"\n";
            }
        }
        //cout<<x<<"\n";
        for(k=2;k<x;k++)
        {
            if(x%k==0)
                value++;
        }
        //cout<<value<<"\n";
        if(value==0&&x!=1)
            cout<<"TRUE\n";
        else
            cout<<"FALSE\n";
    }
    return 0;
}
