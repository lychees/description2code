int main()
{
	long long num,n,a,sum=0;
	cin>>num;
	
	n=(num-1)/3;
	a=num-1;
	while(a)
	{
		if(a%3==0)
			break;
		a--;
	}
	sum=(n*(a+3))/2;
	
	n=(num-1)/5;
	a=num-1;
	while(a)
	{
		if(a%5==0)
			break;
		a--;
	}
	sum=sum + (n*(a+5))/2;
	
	n=(num-1)/15;
	a=num-1;
	while(a)
	{
		if(a%15==0)
			break;
		a--;
	}
	sum=sum-(n*(a+15))/2;
	
	cout<<sum;
	return 0;
}
