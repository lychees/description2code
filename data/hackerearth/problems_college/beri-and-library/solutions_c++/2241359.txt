 #include <iostream>
 using namespace std;
 int main()
{
	int t;
	cin>>t;
	while (t--)
	{
  		long long int n,remainder,quotient;
		int binaryNumber[100],i=1,j;
    	cin>>n;
    	quotient = n;
    	while(quotient!=0)
		{
         binaryNumber[i++]= quotient % 2;
         quotient = quotient / 2;
    	}	
		int count = 0;
 	    for(j = i -1 ;j> 0;j--)
 	    {
 	    	if (binaryNumber[j]==1)
 	    		count++;
 	    }
		cout<<count<<"\n";
	}
    return 0;
}
