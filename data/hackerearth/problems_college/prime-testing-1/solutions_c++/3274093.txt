#include <bits/stdc++.h>

using namespace std;
bool checkprime(int N) {
    int count = 0;
    for( int i = 1;i * i <=N;++i ) {
         if( N % i == 0) {
         if( i * i == N )
                     count++;
                 else                                                     
                     count += 2;
      }
    }
    if(count == 2)
      return true;

   else
    return false;
}

int main () {

int t;
cin >> t;
long long n;
while (t--) {
    cin >> n;
    long long  a[n];
   long long cnt = 0;
    for (int i = 0;i < n;i++){
        cin >> a[i];
       if (checkprime(a[i])){
        cnt++;
       }
    }
    cout << cnt << endl;
    }
}
