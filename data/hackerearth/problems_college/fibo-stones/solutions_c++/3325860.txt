#include "iostream"
using namespace std;
int main(int argc, char const *argv[])
{
	int m,n;
	cin>>n>>m;
	long long fib1=1,fib2=1,fib3;
	long long sum=2;
	for (int i = 3; i<=n; ++i)
	{
		fib3=fib1+fib2;
		sum+=fib3;
		fib1=fib2;		
		fib2=fib3;
	}
	long long tsum=0,temp;
	for (int i = 1; i<=m; ++i)
	{
		cin>>temp;
		tsum+=temp;
	}
	long long ans=sum-tsum;
	cout<<ans<<endl;
	return 0;
}
