#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >>t;
    while(t--)
    {
    	int n,a[100005],sell[100005]={0},ans=0;
    	cin >>n;
    	for(int i=0;i<n;i++)
    	 {cin >>a[i];sell[i]=a[i];}
    	 for(int i=0;i<n;i++)
    	 {
    	 	for(int j=i+1;j<n;j++)
    	 	{
    	 		sell[i]=max(sell[i],sell[j]);
    	 	}
    	 }
    	 for(int i=0;i<n;i++)
    	 {
    	 	if(sell[i]!=a[i])
    	 	{
    	 		ans+=sell[i]-a[i];
    	 	}
    	 }
    	 cout << ans <<endl; 
    	 
    }
    return 0;
}
