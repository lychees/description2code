#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while (t--) {
        long long F, G, p1, p2, p3, m;
        cin >> F >> G >> p1 >> p2 >> p3 >> m;
        long long maxs = 0;
        for (int i = 0 ; i <= F/100 ; i++) {
            for (int j = 0; j <= (F - i*100)/125 && j <= G/50; j++) {
                long long k = min((F - i*100 - j*125)/50, (G-j*50)/100);
                maxs = max(maxs, i * p1 + j * p2 + k * p3);
            }
        }
        if (m >= maxs)
            cout << -1 << endl;
        else
            cout << maxs - m << endl;
    }
}
