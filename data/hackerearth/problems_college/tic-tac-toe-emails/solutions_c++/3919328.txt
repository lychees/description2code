#include <bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   int val=t+1;
   t=t+1;
   
   while(t--)
   {
   		if(val!=t)
   		{
	   		map<string , int>m;
	   		string s;
	   		getline(cin,s);
	   		//cout<<s<<"\n";
	   		int count=0;
	   		string str[1000];
	   		for(int i=0;i<s.length();i++)
	   		{
	   			if(s[i]==' ')
	   			count++;
	   			else
	   			{
	   				str[count] +=s[i];
	   			}
	   		}
	   		for(int i=0;i<=count;i++)
	   		{
	   			//cout<<str[i]<<" ";
	   			m[str[i]]++;
	   			if(m[str[i]]==1)
	   			cout<<str[i]<<" ";
	   		}
	   		cout<<"\n";
   		}
   		
   }
}
