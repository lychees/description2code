#include <iostream>
#include<string>
using namespace std;

int main() {
	int n,k;
	string s;
	cin>>n>>k;
	int a[130];
	for(int i=0;i<130;i++) a[i]=1;
	int j=1,ans=0;
	for(int x=1;x<=n;x++)
	{
		cin>>s;
		for(int i=0;i<s.length();i++)
		{
			if(a[s[i]]==j) a[s[i]]++;
		}
		j++;
	}
	for(int i=0;i<130;i++) {  if(a[i]==n+1) ans++; } //cout<<ans<<" ";
	if(k==ans) cout<<":-)\n";
	else cout<<":-(\n";
	return 0;
}
