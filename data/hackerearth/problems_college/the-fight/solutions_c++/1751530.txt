#include <stdio.h>
typedef long long int ll;
ll a[1000010];
ll b[1000010];
int main()
{
    ll n,i,j,k,m;
    scanf("%lld",&n);
    for (i=1;i<=n;i++)
        scanf("%lld",&a[i]);
    scanf("%lld",&m);
    while(m--)
	{
        scanf("%lld",&i);
        scanf("%lld",&j);
        scanf("%lld",&k);
        b[i]+=k;
        b[j+1]-=k;
    }
    for(i=1;i<=n;i++)
        b[i]+=b[i-1];
    for(i=1;i<=n;i++)
        printf("%lld\n",a[i]+b[i]);
    return 0;
}
