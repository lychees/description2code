#define maX(a,b) (a) > (b) ? (a) : (b)
#define MAX 10101
int phi[MAX];
int A[MAX];
void solve()
{
    int i,j;
    phi[1]=1;
    for(i=2;i<MAX;i++)
    {
        if(!phi[i])
        {
            phi[i]=i-1;
            for(j=(i<<1);j<MAX;j+=i)
            {
                if(!phi[j]) phi[j]=j;
                phi[j] = (phi[j]/i)*(i-1);
            }
        }
    }
    A[0] = 0;
    for(int i=1;i<MAX;i++)
    {
        A[i] = A[i-1];
        A[i] += phi[i];
    }
}
int main()
{
	int t;
	solve();
	//for(int i=0;i<MAX;i++)
	//if(A[i] < 0)
    //    cout << A[i];
	scanf("%d",&t);
	while(t--)
	{
        int n;
        scanf("%d",&n);
        if(n > 10000)
            assert(0);
        if(n == 0)
            printf("%d\n",0);
        else
            printf("%d\n",A[n]-1);
	}
return 0;
}
