noOfTestCases = int(raw_input())

def modifiedString(string):
	newString = ""
	prevChar = ""
	curChar = ""
	for c  in string:
		curChar = c
		if(curChar != prevChar):
			newString += curChar
		prevChar = curChar
		
	print newString

for caseNo in range(1, noOfTestCases + 1):
	modifiedString(raw_input())
