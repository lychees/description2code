while True:

    try:
        program = raw_input()
        if program[0:2] == '//':
            print program

        else:
            x = program.split('//')
            if len(x) == 1:
                print program.replace('->', '.')

            else:
                x[0] = x[0].replace('->', '.')
                print "//".join(x)

    except EOFError:
        break
