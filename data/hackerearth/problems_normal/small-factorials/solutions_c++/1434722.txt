#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int tc,i,j,k,num,temp;
	int arr[20000];
	scanf("%d",&tc);
	
	while(tc--)
	{
		arr[0]=1;
		temp=0;
		k=0;	
		scanf("%d",&num);
		for(i=1;i<=num;i++)
		{
			for(j=0;j<=k;j++)
			{
				temp=arr[j]*i+temp;
				arr[j]=temp%10;
				temp/=10;	
			}
			while(temp)
			{
				k++;
				arr[k]=temp%10;
				temp/=10;
			}
		}
		for(i=k;i>=0;i--)
		printf("%d",arr[i]);
		printf("\n");
	}
}
