t = input()
while t:
    n = input()
    ans = 0
    strength = [int(x) for x in raw_input().split()]
    strength.sort(reverse=True)
    for i in range(0, n/2):
        ans = (ans + (n-1-2*i)*(strength[i]-strength[n-i-1])) % 1000000007
    ans = (ans*strength[0]) % 1000000007
    print(ans)
    t = t-1
