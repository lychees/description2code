#include <iostream>
#include <limits.h>
using namespace std;

int main()
{
    int n,k; cin>>n>>k; int m[n+2][k+2]; 
    for(int i=1;i<=n;i++){
    	for(int j=1;j<=k;j++){
    		cin>>m[i][j];
    	}
    }
    
    int q; cin>>q; 
    for(int i=1;i<=q;i++){
    	int x; cin>>x;  int s[x+2];
    	for(int j=1;j<=x;j++){
    		cin>>s[j];
    	}
    	int min,s2,ind;
    	int map[1002]={0};
    	for(int j=1;j<=x;j++){
    		 s2=s[j]; ind=j;
    		min=INT_MAX;
    		for(int k=1;k<=n;k++){
    			if(map[k]==j-1 && m[k][s2]<min){
    				min=m[k][s2];
    			}
    		}
    		int count=0; 
    		for(int k=1;k<=n;k++){
    			if(map[k]==j-1 && m[k][s2]==min){
    				map[k]++; count++; 
    			}
    		}
    		if(count==1){
    			break;
    		}
    	}
    	
    	for(int k=1;k<=n;k++){
    		if(map[k]==ind && m[k][s2]==min){
    		    cout<<k<<endl; break;
    		}
    	}
    }
    
    return 0;
}
