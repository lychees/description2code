#include<istream>
#include<iostream>
#include<stdio.h>

using namespace std;

long long int arr[1000003][11];
int main()
{
   long long int t,n,i,j;
   
   for(i=1;i<=1000001;i++)
   {
   	long long  sum=0;
   	for(j=0;j<=9;j++)
   	{
   		if(i==1)
   		{
   			arr[i][j]=1;
   			sum=10;
   		}
   		else if(j==0)
   		{
   			arr[i][j]=arr[i-1][1];
   		}
   		else if(j==9)
   		{
   			arr[i][j]=arr[i-1][8];
   			sum=(sum+arr[i][j])%1000000007;
   		}
   		else
   		{
   			arr[i][j]=(arr[i-1][j-1]+arr[i-1][j+1])%1000000007;
   			sum=(sum+arr[i][j])%1000000007;
   		}
   	}
   	arr[i][10]=sum;
   }
   
   cin>>t;
   while(t--)
   {
   	cin>>n;
   	cout<<arr[n][10]<<endl;
   }
   
  return 0; 
}
