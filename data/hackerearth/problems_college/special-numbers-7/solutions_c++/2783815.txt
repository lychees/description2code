#include<iostream>
using namespace std;
void sieve();
int p[100005];
int main()
{
    int arr[1005],count=0,i,tmp,pos=0;
    sieve();
    int num=30;
    while(count!=1000){
            //cout << num<<endl;
            tmp = 0;
            for(i=1;i*i<=num;i++) {
                if(num%i==0)
                {
                    if(p[i]==0 || p[num/i]==0)
                        tmp++;
                }
            }
            if(tmp>=3)
            {
                arr[pos]=num;
                pos++;
                count++;
                //cout << num<<endl;
            }
            num++;

    }
    int t;
    cin>>t;
    while(t--)
    {

        int n;
        cin>>n;
        cout<<arr[n-1]<<endl;
    }



    return 0;
}
void sieve()
{
    int i,j;
    p[0]=1;
    p[1]=1;
    for(i=2; i*i<100005; i++)
    {
        j=2;
        if(p[i]==0)
        {
            while(i*j<100005)
            {
                p[i*j]=1;
                j++;
            }

        }

    }}
