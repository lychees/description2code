#include <iostream>
using namespace std;

int main()
{
    int nt;
    cin >> nt;
    for(; nt > 0; nt --) {
    	int n;
    	cin >> n;
    	long ans = 0;
    	long long x, y, z = 0;
    	for(int i = 0; i < n; i ++) {
    		cin >> x >> y;
    		ans += ((x * y + 8) % 9) + 1;
    		z += y == 0;
    	}
    	ans = ((ans + 8) % 9) + 1;
    	if( z == n )
    		cout << 0 << endl;
    	else
    		cout << ans << endl;
    }
    return 0;
}
