#include<iostream>
using namespace std;
static int c=0;
void swap(int *arr,int j,int i)
{
	if(j==i)
	return;
	int t;
	c++;
	t=arr[j];
	arr[j]=arr[j+1];
	arr[j+1]=t;
	swap(arr,j+1,i);
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		c=0;
		int n;
		cin>>n;
		int arr[n];
		int brr[n];
		int i,j;
		for(i=0;i<n;i++)
		cin>>arr[i];
		for(i=0;i<n;i++)
		cin>>brr[i];
		for(i=n-1;i>=0;i--)
		{
			for(j=0;j<n;j++)
			{
				if(arr[j]==brr[i])
				break;
			}
			if(i!=j)
			swap(arr,j,i);
		}
		cout<<c<<endl;
	}
}
