#include <cstdio>
#include <map>
using namespace std;

int main()
{
	int t,n,v;
	map<int,int> m,c;
	map<int,int>::iterator i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		while(n--)
		{
			scanf("%d",&v);
			m[v]++;
		}
		v=0;
		while(m.size())
		{
			v++;
			for(i=m.begin();i!=m.end();i++)
				if(i->second>1)
					c[i->first]=i->second-1;
			m=c;
			c.clear();
		}
		printf("%d\n",v);
		m.clear();
	}
    return 0;
}
