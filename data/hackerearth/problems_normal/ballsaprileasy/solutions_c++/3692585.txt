#include<bits/stdc++.h>
using namespace std;
const int MAX = 1000000;
 int a[MAX];
int main()	{
int n;
scanf("%d",&n);

for(int i=0;i<n;i++)
scanf("%d",&a[i]);
sort(a, a+n);
long long int ans=0;
long long int layer=1,pos=0;
while(pos<n)	{
if(layer*layer<a[pos])	{
ans+=layer*layer;

}
else	{
ans+=layer*layer-a[pos];
pos++;

}
layer++;


}
printf("%lld\n",ans);
return 0;

}
