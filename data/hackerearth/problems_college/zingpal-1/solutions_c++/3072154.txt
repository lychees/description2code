#include<stdio.h>
#include<stdlib.h>
int visited[101]={0};
int cycle=0;
typedef struct node{
     int data;
     struct node *next;
}node;
node* g[101];
void dfs1(int u)
{
     visited[u]=1;
     node*temp=g[u];
     while(temp!=NULL)
     {
	  if(visited[temp->data]==0)
	       dfs1(temp->data);
	  temp=temp->next;
     }
}
int main()
{
     int n,m,u,v,ans,i;
     scanf("%d%d",&n,&m);
     for(i=0;i<m;i++)
     {
	  scanf("%d%d",&u,&v);
	  u--;v--;
	  node*np=(node*)malloc(sizeof(node));
	  np->data=v;
	  np->next=NULL;
	  node*temp=g[u];
	  g[u]=np;
	  g[u]->next=temp;
	  node*np1=(node*)malloc(sizeof(node));
	  np1->data=u;
	  np1->next=NULL;
	  temp=g[v];
	  g[v]=np1;
	  g[v]->next=temp;
     }
     if(n!=m)
	  printf("NO\n");
     else
     {
	  ans=1;
	  dfs1(0);
	  for(i=0;i<n;i++)
	       if(visited[i]==0)
	       {
		    ans=0;
		    break;
	       }
	  if(ans==0)
	  {
	       printf("NO\n");
	       return 0;
	  }
	  else
	       printf("FHTAGN!\n");
     }
 
}
