#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int t;
    cin>>t;
    while(t--){
    	long int a,b,c=0,flg=0;
    	cin>>a>>b;
    	while(a!=0&&b!=0){
    		if((a%10)+(b%10)+flg>9){
    			c++;
    			flg=1;
    		}
    		else flg=0;
    		a=a/10;
    		b=b/10;
    	}
    	if(!c)cout<<"No carry operation"<<endl;
    	else if(c==1) cout<<c<<" carry operation"<<endl;
    	else cout<<c<<" carry operations"<<endl;
    }
    return 0;
}
