N = int(input())
ans = []
for i in range(0, N):
    s = raw_input()
    s_arr = s.split(" ")
    l = int(s_arr[0])
    b = int(s_arr[1])
    h = int(s_arr[2])
    dimtr = min(l, b, h)
    dimtr = dimtr*10
    anss = str((dimtr/20))+"."+str((dimtr/2) % 10)
    ans.append(anss)
for s in ans:
    print s
