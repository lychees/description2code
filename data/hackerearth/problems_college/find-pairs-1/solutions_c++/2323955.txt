#include<bits/stdc++.h>
#define N 1000001
using namespace std;
int a[N], cnt[3*N];
int main(){
	int t;
	cin>>t;
	while(t--){
		memset(cnt,0,sizeof cnt);
		int n; cin>>n;
		for(int i=1; i<=n; i++){
			cin>>a[i];
			cnt[a[i]+N]++;
		} 
		long long int ans=0ll;
		for(int i=0; i<3*N -1; i++){
			long long int temp= (long long int) cnt[i]*(cnt[i]-1);
			temp/=2;
			ans+=temp;
			ans+=cnt[i];
		}
		cout<<ans<<endl;
	}
}
