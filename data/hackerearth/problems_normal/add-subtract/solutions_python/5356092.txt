import sys
 
T  = int(raw_input())
while T > 0:
    N,K = map(int,raw_input().split(" "))
    A = map(int,raw_input().split())
    ans = sys.maxint
    for i in range(1,101):
        cost = []
        for x in A:
            if i > x:
                cost.append((i-x) * 3)
            else:
                cost.append((x-i) * 5)
 
        cost =sorted(cost)
        temp = sum(cost[0:K])
        
        ans = min(ans,temp)
    print ans
    T -= 1
