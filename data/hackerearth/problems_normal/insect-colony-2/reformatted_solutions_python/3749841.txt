def analize_colony():
    tests_qtd = int(raw_input())
    for x in xrange(tests_qtd):
        colony = map(int, raw_input().split()[1:])
        print can_disapear(colony)


def impar(numero):
    if (numero % 2 == 0):
        return False
    else:
        return True


def can_disapear(colony):
    numImpar = 0

    for insect in colony:
        if impar(insect):
            numImpar += 1

    if impar(numImpar):
        return "No"
    else:
        return "Yes"


analize_colony()
