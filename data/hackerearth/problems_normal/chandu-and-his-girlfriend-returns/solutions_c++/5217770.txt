#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,m,n;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        int a[n+m];
        for(i=0;i<n+m;++i)
            cin>>a[i];
        sort(a,a+n+m,greater<int>());
        for(i=0;i<n+m;++i)
            cout<<a[i]<<" ";
        cout<<endl;
    }
}
