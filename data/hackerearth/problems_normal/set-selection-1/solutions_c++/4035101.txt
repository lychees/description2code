#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
int i,s,j,q,t,n;
cin>>n;
int a[n];
for(i=1;i<=n;i++)
cin>>a[i];

//sort(a,a+n);
bool dp[100002][102];

for(i=0;i<n;i++)
  dp[0][i]=true;

for(i=1;i<100002;i++)
    dp[i][0]=false;
    
for(i=1;i<100002;i++)
{
	for(j=1;j<=n;j++)
	{
			dp[i][j]=dp[i][j-1];
			if(i>=a[j])
				dp[i][j]=dp[i][j]||dp[i-a[j]][j-1];
	}
}
cin>>q;
while(q--)
{
	cin>>t;
	if(dp[t][n]==1)
	cout<<"Yes\n";
	
	else
	cout<<"No\n";
}
    return 0;
}
