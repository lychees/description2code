#include <iostream>
#include <cassert>
using namespace std;

int main()
{
	int Q, T;
	long long N;
	cin>>Q;
	assert(1<=Q && Q<=10);
	while(Q--){
		cin>>N;
		assert(1<=N && N<=1e16);
		cin>>T;
		assert(1<=T && T<=50);
		while(T--){
			if(N%2 == 0)
				N -= N/2;
			else
				N -= (N+1)/2;
			N -= N/4;
		}
		cout<<N<<endl;
	}    
    return 0;
}
