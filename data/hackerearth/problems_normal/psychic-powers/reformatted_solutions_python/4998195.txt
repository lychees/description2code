x = raw_input()
x = [int(i) for i in x]
track = [1, 1, 1, 1, 1, 1]
flag = 0
while len(x) >= 6:
    sum_list = [a and b for a, b in zip(x[:6], track)]
    s = sum(sum_list)
    if s == 0 or s == 6:
        flag = 1
        print "Sorry, sorry!"
        break
    else:
        x.pop(0)

if flag == 0:
    print "Good luck!"
