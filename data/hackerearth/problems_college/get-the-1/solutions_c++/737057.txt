#include <iostream>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	for(int i=1;i<=t;i++){
		int n;
		cin>>n;
		int num=n;
		int result=0;
		while(true){
			while(num>0){
				result+=(num%10)*(num%10);
				num/=10;
			}
			if(result<10){
				break;
			}
			else{
				num=result;result=0;
			}
		}
		if(result==1){
			cout<<"Case #"<<i<<":YES"<<endl;
		}
		else{
			cout<<"Case #"<<i<<":NO"<<endl;
		}
	}
}
