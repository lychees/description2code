#include <iostream>
using namespace std;

int n , ans = 0 ;
bool a[100][100] , flg[100][100];

void DFS( int i , int j )
{
  if( i >= 0 && i < n && j >= 0 && j < n && !flg[i][j] && a[i][j] == 0 )
  {
  	  if( i == n-1 && j == n-1 ) { ans++ ;  return ;}
  	  
  	  flg[i][j] = 1 ;
  	  DFS(i+1,j) ;   DFS(i-1,j) ;
  	  DFS(i,j+1) ;   DFS(i,j-1) ;
  	  flg[i][j] = 0 ;
  }
}

int main()
{
    scanf("%d",&n) ;
    for(int i = 0 ; i < n ; i++ )  
      for(int j = 0 ; j < n ; j++ ) scanf("%d",&a[i][j]) ;
      
    DFS(0,0) ;
    printf("%d\n",ans) ;  
  return 0;
}
