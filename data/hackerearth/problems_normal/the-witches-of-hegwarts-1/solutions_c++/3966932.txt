#include <iostream>
#include <set>
#include <queue>

using namespace std;

int get_ans(int);

int main()
{
	int t, n;
	cin >> t;

	for (; t; --t) {
		cin >> n;
		cout << get_ans(n) << endl;
	}

	return 0;
}

int get_ans(int n)
{
	if (n == 1) return 0;
	int a, b, c, x, y, z;
	set <int> s;
	queue <int> q;
	q.push(n);
	s.insert(n);
	z = 0;
	while (q.front() != 1) {
		z++;
		y = q.size();
		while (y--) {
			x = q.front();
			if (x == 1) {
				z--;
				break;
			}
			q.pop();

			a = b = c = -1;

			if ((x % 3) == 0) {
				a = x / 3;
				if (s.find(a) == s.end()) {
					q.push(a);
					s.insert(a);
				}
			}

			if ((x % 2) == 0) {
				a = x / 2;
				if (s.find(a) == s.end()) {
					q.push(a);
					s.insert(a);
				}
			}

			a = x - 1;
			if (s.find(a) == s.end()) {
				q.push(a);
				s.insert(a);
			}
		}
	}

	return z;
}
