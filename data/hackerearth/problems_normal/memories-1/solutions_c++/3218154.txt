#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <climits>
#include <cmath>
#include <vector>
using namespace std;

int main(){
	
	int T=0,i=0, j=0, k=0, N,P,A,B,count=0,lim;
	
	scanf("%d",&T);
	
	for(i=1; i<=T; i++){
			scanf("%d %d",&N,&P);
			lim = pow(2,N);
		//	printf("%d %d\n",N,P);
			std::vector<int> a(lim, 0);
			
			count=0;
			for(j=1; j<=P; j++){
				scanf("%d %d",&A,&B);
	//			printf("%d %d\n",A,B);
				
		//		printf("[");
				for(k=1; k<lim; k++){
					
					if( (k & (1<<(A-1))) && (k & (1<<(B-1)))){
		//				printf("(%d),",k);
						a[k] = a[k]+1;
					}
					else{
			//			printf("%d,",k);
						if(a[k]==0)
							count++;
					}
				}
	//			printf("]\n");
			}
			
			count=0;
			for(k=1; k<lim; k++)
				if(a[k]==0) count++;
				
			printf("%d\n",count);
	}

	return 0;
}
