#include <iostream>
using namespace std;

int main()
{
    int t,n,k;
    long long int sum,term;
    cin>>t;
    while(t--)
    {
    	cin>>n>>k;
    	sum=0;
    	if(k==1)
    	{
    		term=1;
    	}
    	else
    		term=(k-1)*n+1;
    	for(int j=0; j<n; ++j)
    	{
    		sum+=term+j*(n-k+1);
    	}
    	cout<<sum<<endl;
    }
    return 0;
}
