#include <bits/stdc++.h>
using namespace std;
string str[105];
bool taken[26][26];
bool present[26];
vector<int> vec[27];
vector<int> chr;
int indeg[27];
int main()
{
    int w,i,k,l,j,cnt;
    cin>>w;
    for(i=0;i<w;i++)
    {
        cin>>str[i];
        for(j=0;j<str[i].length();j++)
            present[str[i][j]-'a']=1;
    }
    for(i=0;i<w;i++)
    {
        for(j=i+1;j<w;j++)
        {
            l=min(str[i].length(),str[j].length());
            for(k=0;k<l;k++)
            {
                if(str[i][k]!=str[j][k])
                {
                    if(!taken[str[i][k]-'a'][str[j][k]-'a'])
                    {
                       taken[str[i][k]-'a'][str[j][k]-'a']=1;
                       vec[str[i][k]-'a'].push_back(str[j][k]-'a');
                    }
                    break;
                }
            }
        }
    }
    for(i=0;i<27;i++) indeg[i]=-1;
    for(i=0;i<26;i++)
    {
        //cout<<(char)(i+97)<<"-->";
        for(j=0;j<vec[i].size();j++)
        {
          //  cout<<(char)(vec[i][j]+97)<<" ";
            if(indeg[vec[i][j]]==-1)
                indeg[vec[i][j]]=1;
            else
                indeg[vec[i][j]]++;
        }
      //  cout<<endl;
    }
    /*for(i=0;i<26;i++)
        cout<<indeg[i]<<" ";
    cout<<endl;*/
    cnt=0;
    for(i=0;i<26;i++)
    {
        if(present[i]==1){
           cnt++;
           if(indeg[i]==-1)
            indeg[i]=0;
        }
    }
    //cout<<cnt<<endl;
    while(cnt>0)
    {
        chr.clear();
        for(i=0;i<26;i++)
        {
            if(indeg[i]==0)
            {
              // cout<<cnt<<" "<<(char)(i+97)<<" ";
               chr.push_back(i);
               indeg[i]=-1;
            }
        }
        for(i=0;i<chr.size();i++)
        for(j=0;j<vec[chr[i]].size();j++)
                indeg[vec[chr[i]][j]]--;
        cnt-=chr.size();
        for(i=0;i<chr.size();i++)
            cout<<(char)(97+chr[i]);
        cout<<endl;
        //cout<<cnt<<endl;
    }
    return 0;
}
