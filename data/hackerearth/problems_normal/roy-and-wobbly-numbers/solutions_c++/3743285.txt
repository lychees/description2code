#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    int n,k;
    while(t--)
    {
        int i,j;
        cin>>n>>k;
        if(k>81)
        {
            cout<<"-1\n";
            continue;
        }
        int start=k/9+(k%9!=0);
        int second=k%9;
        if(second==0)
            second=9;
        if(second<=start)
            second--;
            for(i=0,j=1;i<n;i++,j*=-1)
        {
            if(j==1)cout<<start;
            else
                cout<<second;
        }cout<<"\n";
    }
    return 0;
}
