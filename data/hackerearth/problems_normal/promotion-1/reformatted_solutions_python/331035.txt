n, m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
a.sort()
b.sort()
c = [0]*m
i = 0
for j in xrange(m):
    while i < n and a[i] <= b[j]:
        i = i+1
    c[j] = i
if m >= 2:
    p = c[m-1]-c[m-2]
else:
    p = c[m-1]
d = [0]*m
if n % m == 0:
    q = n/m
else:
    q = n/m+1
d[0] = q
for j in xrange(1, m):
    if (n-c[j-1]) % (m-j) == 0:
        q = (n-c[j-1])/(m-j)
    else:
        q = (n-c[j-1])/(m-j)+1
    d[j] = q
print max(d)*2-1
