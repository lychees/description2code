def fun(arr,letter):
	c=[0]*26
	for ind in xrange(26):
		c[ind]=arr[ind]
	c[ord(letter)-97]+=1
	return c



n,q=map(int,raw_input().split())
s=raw_input().strip()
a=[]
b=[0]*26
a=[[]]*(n+1)
a[0]=b
for i in xrange(1,n+1):
	a[i]=fun(a[i-1],s[i-1])

for i in xrange(q):
	l,r,k=map(int,raw_input().split())
	su=0
	fla=0
	for j in xrange(26):
		su=su+a[r][j]-a[l-1][j]
		if su>=k:
			fla=1
			print chr(97+j)
			break
	if fla==0:
		print "Out of range"
