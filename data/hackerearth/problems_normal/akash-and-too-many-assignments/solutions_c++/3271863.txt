const int cs = 0x186A1;

const int me = 1000005;
const ll mod = 1.e9 + 7;

int n, q, l, r, k, x;
string s, g;
int ft[26][me];
void add(int who, int pos, int v){
    while(pos < me){
        ft[who][pos] += v;
        pos += (pos & -pos);
    }
}
int get(int who, int pos){
    int s = 0;
    while(pos > 0){
        s += ft[who][pos];
        pos -= (pos & -pos);
    }
    return s;
}
int main()
{
    ios_base:: sync_with_stdio(false);
    cin.tie(false);

    cin >> n >> q;
    cin >> s; s = "$" + s;

    for(int i = 1 ; i <= n ; i ++)
        add(s[i] - 'a', i, 1);
    int type, pos;
    while(q --){
        cin >> type;

        if(type == 0){
            cin >> pos >> g;
            add(s[pos] - 'a', pos, -1);
            s[pos] = g[0];
            add(s[pos] - 'a', pos, 1);
        }
        else{
            cin >> l >> r >> k;

            if(r - l + 1 < k){
                cout << "Out of range" << endl;
                continue;
            }
            int sum = 0, re;

            for(int i = 0 ; i < 26 ; i ++){
                sum += get(i, r) - get(i, l - 1);

                if(sum >= k){
                    re = i;
                    break;
                }
            }
            cout << (char)(re + 'a') << endl;
        }
    }
    ///cout << "Elapsed time is: " << clock() / CLOCKS_PER_SEC << endl;

    return 0;
}
