from math import *


def check_prime(n):

    for i in range(2, int(n/2)+1):
        if n % i == 0:
            return False

    return True


def factorial(n):
    k = 1
    for i in range(2, n+1):
        k *= i
    return k


def main():
    n = int(raw_input())
    for i in range(2, n+1):
        if check_prime(i) is True:
            print str(factorial(i))


if __name__ == '__main__':
    main()
