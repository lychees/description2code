#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;
int a[1000000]={0};
int main()
{

	int n,cnt=0,flag=0,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}


	for(i=0;i<n;i++)
	{ 
		if(a[i]==i)
		{
			cnt++;
		}
	}

	for(i=0;i<n;i++)
	{
		if(a[a[i]]==i)
		{
			flag=1;
			break;
		}
	}

	if(cnt==n)
	{
		cout<<n-2;
	}
	else if(flag==1)
	{
		cout<<cnt+2;
	}
	else
	{
		cout<<cnt+1;
	}

	return 0;
}
