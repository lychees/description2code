#include<bits/stdc++.h>
using namespace std;
int main(){

   int t;
   cin >> t;
   while( t-- ){

     int n;
     cin >> n;
     int a[100001];
     a[0] = 1;
     int sz = 0;
     int rem  = 0;
     for(int i = 1;  i <= n ; ++i){
        int rem = 0;
      for(int j = 0; j <= sz ; ++j ){
        rem = a[j]*2 + rem;
        a[j] = rem%10;
        rem = rem/10;
      }
      while(rem){
        sz++;
        a[sz] = rem%10;
        rem = rem/10;
      }

     }
     for(int i = sz ; i >= 0 ; --i )
     cout << a[i];
     cout << endl;


   }

}
