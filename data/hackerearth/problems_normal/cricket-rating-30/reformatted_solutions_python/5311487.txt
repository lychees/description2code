def maxRating(ratingList, length):
    maxNow = 0
    maxEnd = 0
    for i in range(0, length):
        maxEnd = maxEnd + int(ratingList[i])
        if maxEnd < 0:
            maxEnd = 0
        if (maxNow < maxEnd):
            maxNow = maxEnd
    return maxNow


length = int(raw_input())
if length == 0:
    print 0
else:
    ratingList = raw_input().split(" ")
    ratingList = filter(None, ratingList)
    print maxRating(ratingList, length)
