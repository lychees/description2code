#include <iostream>
using namespace std;

bool oof(int n) {
    return (n&(n>>1)) == 0;
}

int foo(int n)
{
	bool bOne = false;
	int current = n;
	bool bFound = false;

	while(!bFound)
	{
		current = ++n;

		while(current != 0)
		{
			if(current & 0x1 == 1)
			{
				if(bOne)
				{
					bFound = false;
					break;
				}

				bOne = true;
			}else
			{
				bOne = false;
			}

			current = current >> 1;
		}

		if(current == 0)
			bFound = true;
	}

	return n;
}

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        if(oof(n) == true) {
            cout << n << endl;
        }
        else
            cout << foo(n) << endl;
    }
    return 0;
}
