using namespace std;
typedef long long int ll;
ll arr[25][30];
char cal(ll idx)
{
    int j,num=0,den=0,val,i,mn,sec;
    char ch;
    mn=1e9;
    for(j=0;j<26;j++)
    {
        sec=0;
        for(i=0;i<26;i++)
        {
            sec+=arr[idx][i]*(abs(i-j));
        }
        if(sec<mn)
        {
            mn=sec;
            val=j;
        }
    }
    ch=(char)(val+'a');
    return ch;
}
int main()
{
    ll tc,n,j,i,k;
    string s,ans;
    cin>>tc;
    while(tc--)
    {
        ans="";
        memset(arr,0,sizeof(arr));
        cin>>n>>k;
        for(j=0;j<n;j++){
        cin>>s;
        for(i=0;i<s.length();i++)
        {
            arr[i][s[i]-'a']++;
        }
        }
        for(i=0;i<k;i++)
        {
            ans=ans+cal(i);
        }
        cout<<ans<<endl;
    }
    return 0;
}
