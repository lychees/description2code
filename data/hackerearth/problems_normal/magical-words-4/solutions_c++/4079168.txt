#include <iostream>
using namespace std;
 
int main()
{
	int T=0;
	cin>>T;
	while(T--)
	{
	    string str;
	    cin>>str;
	    int slen = str.size();
	    long long sum=0;
	    for(int i=0; i<slen; i++)
	    {
	    	int j=i; 
	    	int k=i;
	    	while(j>=0 && k<slen && str[j] == str[k])
	    	{
	    		sum += (k-j+1)*(k-j+1);
	    		j--;
	    		k++;
	    	}
	    	
	    	j=i;
	    	k=i+1;
	    	while(j>=0 && k<slen && str[j] == str[k])
	    	{
	    		sum += (k-j+1)*(k-j+1);
	    		j--;
	    		k++;
	    	}
	    }
	    cout<<sum<<"\n";
	}
    return 0;
}
