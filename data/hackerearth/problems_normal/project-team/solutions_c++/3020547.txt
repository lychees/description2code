#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,mini,maxi,j;
    cin>>t;
    while(t--){
    	mini=10000000;
    	maxi=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	sort(a,a+n);
    	for(i=0;i<n/2;i++){
    		j=a[i]+a[n-1-i];
    		mini=min(j,mini);
    		maxi=max(j,maxi);
    	}
    	cout<<maxi-mini<<endl;
    }
    return 0;
}
