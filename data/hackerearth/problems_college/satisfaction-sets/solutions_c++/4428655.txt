#include <bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    int ar[505],vst[505];
    memset(ar,0,sizeof(ar));
    for(int i=0;i<n;i++)
    {
    	int m,num;
    	cin>>m;
    	memset(vst,0,sizeof(vst));
    	for(int i=0;i<m;i++)
    	{
    		cin>>num;
    		if(!vst[num])
    		{
    			ar[num]++;
    			vst[num]=1;
    		}
    	}
    }
    int diff=0,sum=0;
    for(int i=0;i<502;i++)
    {
    	if(ar[i]) 
    	{
    		sum+=i; diff+=i;
    	}
    	if(ar[i]==n) diff-=i;
    }
    cout<<(diff-sum)<<endl;
    return 0;
}
