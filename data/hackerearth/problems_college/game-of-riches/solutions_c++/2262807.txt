#include<bits/stdc++.h>

using namespace std;
int main()
{
	int t,g,n,a[100000],i;
	vector<pair<int,int> >v;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&g);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		v.clear();
		int my=a[0];
		int myp;
		sort(a,a+n);
		int z=1;
		for(i=n-1;i>=0;i--)
		{
			if(a[i]==my)
			{	//cout<<z<<endl;
				int count=0;
				for(int k=0;k<v.size();k++)
				{
					if((v[k].first==z)&&(v[k].second>a[i]))
					{	
						count++;
					//	cout<<v[k].first<<" "<<v[k].second<<endl;
					}
						
				}
				printf("%d\n",count);
				break;
			}
			v.push_back(make_pair(z,a[i]));
		//	cout<<z<<" "<<a[i]<<endl;
			if(z<g)
			z++;
			else
			z=1;
			
		}
	}
}
