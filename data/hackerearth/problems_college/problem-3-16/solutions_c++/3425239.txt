#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll t,i,j,k,l,n;
int main()
{
    cin>>t;
    while(t--)
    {
    	cin>>i>>j;
    	vector<int> v(j+1,0);
    	priority_queue<int> pq;
    	while(i--)
    	{
    		cin>>n;
    		v[n]++;
    	}
    	for(i=1;i<=j;i++)
    	{
    		pq.push(v[i]);
    	}
    	ll c=0;
    	pq.pop();
    	while(!pq.empty())
    	{
    		c+=pq.top();
    		pq.pop();
    	}
    	cout<<c<<endl;
    }
}
