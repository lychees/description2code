#include<iostream>
using namespace std;

double C[1001][1001];

void precompute()
{
	C[0][0]=1.0;
	int i,j;
	for(i=1;i<1001;i++)
	{
		C[i][0]=C[i][i]=1.0;
		for(j=1;j<i;j++)
		{
			C[i][j]=C[i-1][j]+C[i-1][j-1];
		}
	}
}

int main()
{
	int t,s,n,m,k,i;
	double num,den;
	precompute();
	cin>>t;
	while(t--)
	{
		cin>>s>>n>>m>>k;
		den=C[s][n];
		num=0;
		for(i=k;i<=m && i<=n;i++)
		{
			if(s-m>=0 && n-i>=0 && s-m>=n-i)
			{
				num+=C[m][i]*C[s-m][n-i];
			}
		}
		printf("%0.6lf\n",num/den);
	}
	return 0;
}
