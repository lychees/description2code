#include <bits/stdc++.h>
#include<string.h>
using namespace std;

int main()
{
    long long a,b,i,l,f,t;
    string s,d;
    cin>>t;
    while(t--)
    {
    	cin>>s;
    	l=s.length();
    	d=s;
    	reverse(d.begin(), d.end());
    	f=1;
    	for(i=0;i<l-1;i++)
    	{
    		a=abs(s[i+1]-s[i]);
    		b=abs(d[i+1]-d[i]);
    		if(a!=b)
    		{
    			f=0;
    			break;
    			
    		}
    		//cout<<a<<b<<"\n";
    	}
    	if(f==1)
    	{
    		cout<<"Funny\n";
    	}
        else
        {
        	
        	cout<<"Not Funny\n";
        }
    }
    return 0;
}
