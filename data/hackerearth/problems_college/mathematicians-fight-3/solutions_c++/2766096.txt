#include<bits/stdc++.h>
#define LL unsigned long long
using namespace std;
int main()
{
    LL Arr[100]={0};
    int cnt=0;
    Arr[0]=1;
    for(cnt=1;cnt<64;cnt++)
        Arr[cnt]=2*Arr[cnt-1];
    int T;
    scanf("%d",&T);
    while(T--)
    {
        LL n,p;
        scanf("%llu",&n);
        LL i;
        p=1;
        for(i=0;i<64;i++)
            if(Arr[i]>=n)
                break;
        LL Ans=1;
        if(Arr[i]!=n)
            Ans+=2*(n-Arr[i-1]);
        Ans=Ans+p-1;
        if(Ans>n)
        Ans=Ans-n;
        printf("%llu\n",Ans);
    }
    return 0;
}
