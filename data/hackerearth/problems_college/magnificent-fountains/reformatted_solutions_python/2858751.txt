def hcf(a, b):
    if (b == 0):
        return a
    return hcf(b, a % b)


def lcm(l):
    a = l[0]
    b = l[1]
    h = hcf(a, b)
    lc = (a*b)/h
    for i in xrange(2, len(l)):
        h = hcf(lc, l[i])
        lc = (lc*l[i])/h
    return lc
