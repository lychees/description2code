#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
#include <cstdio>
#include <stdlib.h>
#include <math.h>
#include <vector>
#include <list>
#define mod 1000000007

typedef long long int ll;
typedef long int l;
using namespace std;



int main()
{
  std::ios_base::sync_with_stdio(false);

int t;
cin>>t;
while(t--)
{
  ll m,n;
  cin>>n>>m;
  ll res=0;
  ll old_p=0,new_p=m;
  ll old_c=1,new_c=n-1;
  while(new_p && new_c)
  {
      res++;
    ll p=old_p+min(new_p,old_c);
    ll c=old_c+min(new_c,old_p);
    new_p-=min(new_p,old_c);
    new_c-=min(old_p,new_c);
    old_p=p;
    old_c=c;

  }
  if(new_c)
  {
      res+=(1+((new_c-1)/old_p));
  }
  cout<<res<<endl;
}

return 0;
}
