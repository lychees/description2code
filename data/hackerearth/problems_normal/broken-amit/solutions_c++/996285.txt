#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int n;
    cin>>n;
    long long arr[2][n],t;
    long long res=0;
    for(int i=0;i<n;i++)
    {
    	cin>>arr[0][i];
    	res=res+arr[0][i];
    }
    for(int i=0;i<n;i++)
    {
    cin>>t;
    arr[1][i]=t*i;
    }

    for(int i=0;i<n-1;i++)
    {
    for(int j=0;j<n-i-1;j++)
    {
    	if(arr[1][j]<arr[1][j+1])
    	{
    		long long temp=arr[1][j+1];
    		arr[1][j+1]=arr[1][j];
    		arr[1][j]=temp;
    		 temp=arr[0][j+1];
    		arr[0][j+1]=arr[0][j];
    		arr[0][j]=temp;
    	}
    }
    }
    int j;
    for(int i=1;i<n;i++)
    {
    	j=0;
    	while(j<i)
    	{
    		arr[0][i]-=arr[1][i];res+=arr[0][i];
    		j++;
    	}
    	if(j==0)
    	{
    	    res+=(n-1)*arr[0][i];
    	}
    	else
    	res+=(n-i-1)*arr[0][i];
    }
res+=(n-1)*arr[0][0];
  /*  for(int i=0;i<n;i++)
    res+=(n-i-1)*arr[0][i];*/
    cout<<res;


    return 0;
}
