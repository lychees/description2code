t = int(raw_input(''))

for p in range(t):
	nums = map(int, raw_input('').split());
	x = nums[0]
	y = nums[1]
	N = nums[2]

	first_terms = []
	for i in range(x):
		first_terms.append(y)

	for j in range(N-x):
		next_term = sum(first_terms[j:x+j])
		first_terms.append(next_term)

	print first_terms[N-1]
