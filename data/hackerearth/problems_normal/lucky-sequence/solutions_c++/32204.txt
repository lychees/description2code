void solve(ll k,ll l)
{
    if(l==0LL)
    return ;

    i=1LL<<(l-1LL);
    if(k>i)
    {
        printf("7");
        solve(k-i,l-1LL);
        return;
    }

    printf("4");
    solve(k,l-1LL);
}
