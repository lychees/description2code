#include<iostream>
#include<stdlib.h>
#include<string.h>
#define MAXN 105
#include<stdio.h>
#define MAXM 5005
using namespace std;
int mat[MAXN][MAXM];
int arr[MAXN];
int n,m,q;
int solve()
{
	int ans=m-1;
	for(int i=0;i<n;i++)
	{
		while(arr[i]<mat[i][ans])
			ans--;
	}
	return ans+1;
}
int main()
{
	cin>>n>>m>>q;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			scanf("%d",&mat[i][j]);
		}
	}
	while(q-->0)
	{
		for(int i=0;i<n;i++)
			scanf("%d",&arr[i]);
		printf("%d\n",solve());
	}
	return 0;
}
