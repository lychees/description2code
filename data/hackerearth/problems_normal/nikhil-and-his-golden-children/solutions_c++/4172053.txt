#include <iostream>
using namespace std;
typedef long long int ll;
ll f[1000001];
ll calc(ll a, ll b, ll mod) {
	ll ans = 1;
	while(b > 0) {
		if(b%2 == 1) {
			ans = (ans * a)%mod;
		}
		a = (a * a)%mod;
		b/=2;
	}
	return ans;
}
int main()
{
	ll mod = 1000000007;
   	ll max = 1000000;
   	ll i;
   	f[0] = 1;
   	for(i = 1; i <= max; i++) {
   		f[i] = (i * f[i - 1])%mod;
   	}
   	int t;
   	cin >> t;
   	while(t--) {
   		ll k, sum, n, r;
   		cin >> k >> sum;
   		for(i = 0; i < k; i++) {
   			cin >> n;
   			sum-=n;
   		}
   		if(sum < 0) {
   			cout << "0\n";
   		}
   		else {
   			ll ans;
   			sum = sum + k - 1;
   			r = k - 1;
   			ans = f[sum];
   			ans = ans * calc(f[r], mod - 2, mod);
   			ans = ans%mod;
   			ans = ans * calc(f[sum - r], mod - 2, mod);
   			ans = ans%mod;
   			cout << ans << endl;
   		}
   	}
    return 0;
}
