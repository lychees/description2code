#define MAXN 100005

int n, a[MAXN];

set<int> S;

void solve()
{
    S.clear();
    for(int i = 0; i < n; i++)
    {
        if (S.find(a[i]) == S.end())
        {
            S.insert(a[i]);
            set<int>::iterator k = S.find(a[i]);
            k++;
            if(k != S.end())
            {
                S.erase(k);
            }
        }
    }
    cout << n-S.size() << endl;
}

int main()
{
    int t;
    scanf("%d",&t);
    for(int j=1;j<=t;j++)
    {
        cin >> n;
        for(int i = 0; i < n; i++)
        {
            scanf("%d", a+i);
        }
        solve();
    }

}
