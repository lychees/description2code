#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    string s;
    while(t--)
    {
    	cin>>s;
    	int count=0;
    	long long int sum=0;
    	int i=0;
    	while(i<s.length())
    	{
    		while(((s[i]>=65&&s[i]<=67)||(s[i]>=97&&s[i]<=99))&&i<s.length())
    		{
    			count++;
    			i++;
    		}
    		sum+=(count*(count+1))/2;
    		count=0;
 			i++;
    	}
    	cout<<sum<<endl;
    }
}
