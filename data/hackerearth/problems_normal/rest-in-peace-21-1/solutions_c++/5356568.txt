#include <iostream>
#include <string>

using namespace std;

int main(int argc, char **argv) {
	int t;
	cin>>t;
	int n;
	for (int i = 0; i <t; ++i) {
		cin>>n;
		if(n%21==0){
			cout<<"The streak is broken!"<<endl;
		}else if (to_string(n).find("21")!=string::npos) {
			cout<<"The streak is broken!"<<endl;
		}else {
			cout<<"The streak lives still in our heart!"<<endl;
		}
	}
}
