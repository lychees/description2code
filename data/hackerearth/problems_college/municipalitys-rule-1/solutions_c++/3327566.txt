#include <iostream>
using namespace std;
typedef long long ull;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ull n;
        cin>>n;
        n--;
        ull a=n/6;
        ull b=(n+3)/6;
        ull ans=-(3*a*(a+1))/2 - (3*b*(b+1))/2;
        if(n&1)
        {
            ans+=(((n-1)/2)*(a+1))  +  ((b+1)*((n+3)/2));
        }
        else
        {
            ans+=((n/2)*(a+1))  +  ((b+1)*((n+2)/2));
        }
        cout<<ans<<"\n";
    }
    return 0;
}
