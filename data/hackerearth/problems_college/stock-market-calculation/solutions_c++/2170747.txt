#define f first
#define s second
#define mp make_pair
#define pb push_back
#define MAX 50000+100
ll cum[MAX];
pair<ll,int> A[MAX];
bool cmp(pii a,pii b)
{
    if(a.first==b.first)
        return a.second<b.second;
    return a.first>b.first;
}
int main()
{
    int t,n,last_index;
    ll ans=0;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        last_index=0;
        scanf("%d",&n);
        for(int i=1;i<=n;i++)
        {
            scanf("%lld",&A[i].f);
            A[i].s=i;
            cum[i]=A[i].f+cum[i-1];
        }
        sort(A+1,A+n+1,cmp);
        for(int i=1;i<=n;i++)
        {
            if(last_index>A[i].s)
                continue;
            int cost=cum[A[i].s-1]-cum[last_index];
            ans=ans+A[i].f*(A[i].s-1-last_index)-cost;
            last_index=A[i].s;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
