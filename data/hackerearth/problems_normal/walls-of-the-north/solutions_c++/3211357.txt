#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define pb push_back
#define pii pair<int,int>
#define maxn 100005
#define mod 1000000007

int main(){
	ios::sync_with_stdio(false);

	int t,n;
	cin>>t;
	while(t--){
		ll x,s1,s2,a1=1,a2=1;
		int f1=1,f2=-1;
		cin>>n>>x;
		s1=s2=x;
		for(int i=2;i<=n;i++){
			cin>>x;
			if(f1==1){
				if(x>=s1){
					a1++;
					s1=x;
					f1=-1;
				}
				else{
					s1=x;
				}
			}
			else if(f1==-1){
				if(x<=s1){
					s1=x;
					a1++;
					f1=1;
				}
				else{
					s1=x;
				}
			}
			if(f2==1){
				if(x>=s2){
					a2++;
					s2=x;
					f2=-1;
				}
				else{
					s2=x;
				}
			}
			else if(f2==-1){
				if(x<=s2){
					s2=x;
					a2++;
					f2=1;
				}
				else{
					s2=x;
				}
			}
		}
		cout<<max(a1,a2)<<endl;
	}

	return 0;
}
