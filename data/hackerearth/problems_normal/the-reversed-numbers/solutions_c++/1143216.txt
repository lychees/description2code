#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
void rev(int& a) {
    int ans=0;
    while(a) {
        int md=a%10;
        ans=ans*10+md;
        a/=10;
    }
    a=ans;
    return;
}
int main() {
    int tt;
    scanf("%d",&tt);
    while(tt--) {
        int a,b;
        scanf("%d%d",&a,&b);
        rev(a); rev(b);
        int val=a+b;
        rev(val);
        cout<<val<<endl;
    }
    return 0;
}
