#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int arr[2000],i,j,k,l,n,m;
    cin>>n;
    for(i=0;i<n;i++){
    	cin>>arr[i];
    }
    sort(arr,arr+n);
    cin>>k;
    l=0;
    for(i=n-1;i>=0;i--){
    	l+=arr[i];
    	if(l>=k){
    		cout<<n-i<<endl;
    		break;
    	}
    }
    return 0;
}
