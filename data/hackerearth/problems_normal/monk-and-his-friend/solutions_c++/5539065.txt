#include<iostream>
using namespace std;
int set(unsigned long long int x)
{
    int count=0;
    while(x)
    {
        if(x%2!=0)
            count++;
        x/=2;
    }
    return count;
}
int main(){
    int t;
    cin>>t;
    while(t--)
    {
        unsigned long long int a,b,x;
        cin>>a>>b;
       x=a^b;
        int ans = set(x);
        cout<<ans<<endl;
    }
    return 0;
}
