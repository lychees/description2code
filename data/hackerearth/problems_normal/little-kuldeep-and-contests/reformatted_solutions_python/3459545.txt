times = [0] * 24 * 60
ans = "Who needs a moderator?"
for _ in range(input()):
    ins = raw_input().split("-")
    hrs, mins = ins[0].split(":")
    start = int(hrs) * 60 + int(mins)
    hrs, mins = ins[1].split(":")
    end = int(hrs) * 60 + int(mins)
    for i in range(start, end):
        if times[i] == 1:
            ans = "Will need a moderator!"
            break
        else:
            times[i] = 1
print ans
