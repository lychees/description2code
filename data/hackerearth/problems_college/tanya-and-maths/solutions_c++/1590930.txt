vector<int> vec[N];
LL val[N];
int solve(){
    int a, b, g, i, j;
    LL ans = 0;
    cin>>a>>b;
    g = __gcd(a, b);
    for(i = 1; i <= sqrt(g + 1); i++){
        j = g / i;
        if(i * j == g){
            if(i <= j){
                ans += val[i];
            }
            if(i < j){
                ans += val[j];
            }
        }
    }
    cout<<ans<<endl;
}

int main(){
    int pr[N], i, j;
    REP(i, 1, N - 1){
        pr[i] = 1;
    }
    for(i = 2; i < N; i++){
        if(pr[i] == 1){
            for(j = i; j < N; j += i){
                vec[j].PB(i);
                pr[j] = 0;
            }
        }
    }
    val[1] = 1;
    for(i = 2; i < N; i++){
        val[i] = i;
        for(j = 0; j < vec[i].size(); j++){
            val[i] /= vec[i][j];
            val[i] *= (vec[i][j] - 1);
        }
    }

    //freopen("in.txt", "r", stdin);
    int t = 1;
    cin>>t;
    while(t--){
        solve();
    }
    return 0;
}
