#include <iostream>
#include <cmath>
using namespace std;

double area(double a,double b,double c)
{
	
		double s = (a+b+c)/2;
		double ar = s*(s-a)*(s-b)*(s-c);
		ar = sqrt(ar);
		return ar;
}


int main()
{
   int t = 0;
   cin>>t;
   while(t--)
   {
   	
   		double a,b,c;
   		cin>>a>>b>>c;
   		double ar ;
   		ar = area(a,b,c); 
   		double r = (a*b*c)/(4*ar);
   		ar = 3.1415*r*r;
   		printf("%.4f\n",ar);
   	
   }
    return 0;
}
