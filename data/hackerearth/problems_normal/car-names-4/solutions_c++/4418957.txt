#include <bits/stdc++.h>
#define ll long long
using namespace std;
//dar hale hazer daram ba donafar mosabege midam yeki yashare yeki RM e ;))
#define pb push_back
#define mp make_pair
#define F first
#define S second
int main()
{
	int T;cin>>T;
	while(T--)
	{
		string str;cin>>str;
		vector<pair<int,int> >V;
		V.pb(mp(str[0],1));
		int index=0;
		for(int i=1;i<(int)str.length();i++)
		{
			if(str[i] == V[index].F)V[index].S++;
			else
			{
				V.pb(mp(str[i],1));
				index++;
			}
		}
		if(index == 2 && V[0].F != V[2].F && V[0].S == V[1].S && V[1].S == V[2].S)
			cout<<"OK\n";
		else cout<<"Not OK\n";
	}
	return 0;
}
