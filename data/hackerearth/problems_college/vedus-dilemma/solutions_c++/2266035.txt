#include <bits/stdc++.h>
using namespace std;
#define FOR(i,a,n) for(int i=a;i<n;i++)
#define si(x) scanf("%d",&x)
#define sll(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define pi(x) printf("%d\n",x)
#define pll(x) printf("%lld\n",x)
#define ps(x) printf("%s\n",ch)
#define INF 2000000000
#define ll long long
#define mp make_pair
#define pb push_back
#define fr first
#define se second
ll gcd(ll a, ll b)
{
	if(b==0) return a;
	else return gcd(b,a%b);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n;
		sll(n);
		vector <ll> v(n+10);
		FOR(i,0,n)
			sll(v[i]);
		
		ll p,q;
		sll(p); sll(q);
		vector< pair<ll,ll> > diff;
		
		FOR(i,0,n)
		{
			ll x = fabs(p - v[i]);
			diff.pb(mp(x,v[i]));
		}
		sort(diff.begin(),diff.end());	
		
		//FOR(i,0,diff.size())
		//	pll(diff[i].se);
		ll c=0;
		vector<ll> ans;
		for(ll i=0;i<n && c<q;i++)
		{
			if(gcd(diff[i].se, p)!=1){
				ans.pb(diff[i].se);
				c++;
			}
		}
		sort(ans.begin(),ans.end());
		//cout<<ans.size()<<endl;
		//FOR(i,0,ans.size())
		//	pll(ans[i]);
		if(c==q)
		{
			FOR(i,0,q)
				printf("%lld ",ans[i]);
			printf("\n");
		}
		else
			printf("-1\n");
	}
	return 0;
}
