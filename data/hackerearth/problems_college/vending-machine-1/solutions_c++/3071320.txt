#include <cstdio>
#include <cstring>
using namespace std;
int base[30];
long long fifb[30];
char name[5][10]={
    "Sheldon", "Leonard", "Penny", "Rajesh", "Howard"
};
int main(){
    int n;
    for(int i=0;i<30;i++){base[i]=1<<i;fifb[i]=(i?fifb[i-1]:0)+5*base[i];}
   while( scanf("%d",&n)==1){
   for(int i=0;i<30;i++){
        if(fifb[i]>=n){
            int gap=(n-(i?fifb[i-1]+1:1))/base[i];
            puts(name[gap]);
            break;
        }
   }}
    return 0;
}
