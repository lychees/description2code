#include <iostream>
using namespace std;


int main() {
	long long T, L;
	cin >> T;
	for (long long t = 0; t < T; t++) {
		cin >> L;
		long long values[L];
		for (long long i = 0; i < L; i++)
		{
			cin >> values[i];
		}

		long long result[L + 1];
		result[0] = 0;
		for (long long j = 1; j <= L; j++)
		{
			long long max = 0;
			for (long long i = 1; i <= j; i++)
			{
				long long currentPt = j - i;
				long long tV = 0;
				if (currentPt >= 0)
				{
					tV = result[currentPt]+values[i-1];
				}
				if(tV > max)
				{
					max = tV;
				}
				result[j] = max;
			}
		}
		cout << result[L] << endl;
	}
	return 0;
}
