#include <bits/stdc++.h>
using namespace std;
int sr[405],sc[405];
int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,k;
    	cin>>n>>k;
    	int a[n][n],r[n],c[n];
    	memset(r,0,sizeof r);
    	memset(c,0,sizeof c);
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++){
    			cin>>a[i][j];
    			r[i]+=a[i][j];
    			c[j]+=a[i][j];
    		}
    	int ans=100000004;
    	for(int i=1;i<=k;i++){
    		sort(r,r+n);sort(c,c+n);
    		sr[i]=sr[i-1]+r[0];sc[i]=sc[i-1]+c[0];
    		r[0]+=n;c[0]+=n;
    	}
    	for(int i=0;i<=k;i++)	ans=min(ans,(sr[i]+sc[k-i]+i*(k-i)));
    	cout<<ans<<endl;
    }
    return 0;
}
