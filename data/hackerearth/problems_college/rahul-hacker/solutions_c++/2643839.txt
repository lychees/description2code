#include<bits/stdc++.h>
using namespace std;

int reverse(long long int num)
{
	long long int sum=0;
	while(num>0)
	{
		sum = (sum*10) + (num%10);
		num = num/10;
	}
	return sum;
}
int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
    	long long int a,b;
    	cin>>a>>b;
    	cout<<reverse(reverse(a)+reverse(b))<<endl;
    }
}
