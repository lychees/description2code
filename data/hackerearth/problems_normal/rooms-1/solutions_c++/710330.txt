#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
        queue<long long> free1;
        int i,n;
        cin>>n;
        int dept[100000],arri[100000];
        for(i=0;i<n;i++)
        {
            cin>>arri[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>dept[i];
            dept[i]+=arri[i];
            dept[i]--;
        }
        sort(arri,arri+n);
        sort(dept,dept+n);
        int max=0;
        for(i=0;i<n;i++)
        {
            int *q=lower_bound(dept,dept+n,arri[i]);
            if(q==dept+n)
                q=dept;
       //     cout<<*q<<"for"<<arri[i]<<endl;   
            int countleft=q-dept;
            
         //   cout<<"coountleft is "<<countleft<<endl;
            if(i-countleft+1>max)
            {
                max=i-countleft+1;
            }
        }
        cout<<max<<endl;
	}
	return 0;
}
