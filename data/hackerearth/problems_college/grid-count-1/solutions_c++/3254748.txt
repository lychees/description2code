#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MAX 1005

ll mat[MAX][MAX];

int main()
{
	int N,K;
	
	cin>>N>>K;
	
	for(int i=1;i<=N;i++)
	{
		for(int j=1;j<=N;j++)
		{
			cin>>mat[i][j];
		}
	}
	
	unordered_set <ll> S;
	unordered_set <ll> ::iterator it;
	
	for(ll i=1;i<=MAX;i++ )
	{
		ll value=i*i*i;
		
		S.insert(value);
	}
	
	int ans=0;
	
	for(int i=1;i<=(N-K+1);i++)
	{
		ll val=0;
		
		for(int j=1;j<=(N-K+1);j++)
		{
			if(j==1)
			{
				for(int l=i;l<=(i+K-1);l++)
				{
					for(int m=j;m<=(j+K-1);m++)
					{
						val+=mat[l][m];
					}
				}
			}
			else
			{
				for(int l=i;l<=(i+K-1);l++)
				val-=mat[l][j-1];
				
				for(int l=i;l<=(i+K-1);l++)
				val+=mat[l][j+K-1];
				
			}
			
			//cout<<val<<endl;
			
			it=S.find(val);
			
			if(it!=S.end())
			ans++;
		}
	}
	
	cout<<ans<<endl;
	
    return 0;
}
