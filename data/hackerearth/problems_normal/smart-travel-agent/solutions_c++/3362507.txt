
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,r;
    cin>>n>>r;
    vector< pair<int,int> >v[n+1];
    for(int i=0;i<r;i++)
    {
        int a,b,c;
        cin>>a>>b>>c;
        v[a].push_back(make_pair(c,b));
        v[b].push_back(make_pair(c,a));
    }
    int start,end1,time;
    cin>>start>>end1>>time;
    queue<int>q;
    q.push(start);
    int vis[n+1];
    int parent[n+1];
    fill(vis,vis+n+1,0);
    fill(parent,parent+n+1,0);
    vis[start]=INT_MAX;
    parent[start]=-1;
    while(!q.empty())
    {
        int k=q.front();
        q.pop();
        for(int i=0;i<v[k].size();i++)
        {
            if(min(v[k][i].first,vis[k])>vis[v[k][i].second])
            {
                //cout<< v[k][i].first << " " << vis[k] << " "<<vis[v[k][i].second]<<endl;
                vis[v[k][i].second]=min(v[k][i].first,vis[k]);
                q.push(v[k][i].second);
                parent[v[k][i].second]=k;
            }
        }
    }
    vector<int>v1;
    v1.push_back(end1);
    int k=parent[end1];
    while(parent[k]>0)
    {
        v1.push_back(k);
        k=parent[k];
    }
    v1.push_back(start);
    for(vector<int>::reverse_iterator it=v1.rbegin();it!=v1.rend();it++)
    {
        cout<<(*it)<<" ";
    }
    cout<<endl;
    cout<<((int)ceil((double)(time)/(vis[end1]-1)))<<endl;
}
