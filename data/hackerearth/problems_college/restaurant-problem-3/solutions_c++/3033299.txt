#include <stdio.h>
#include <stdlib.h>
#include <string.h>
int main()
{
    char s[1000002];
    long long int a[2][8]={0},kq=0;
    int fl=0,i,j;
    gets(s);
    for (i=0;i<strlen(s);i++)
    {
        fl=1-fl;
        for (j=0;j<8;j++) a[fl][j]=0;
        for (j=0;j<8;j++) a[fl][(j*10+s[i]-'0')%8]+=a[1-fl][j];
        if (s[i]=='0') kq++;
        else a[fl][(s[i]-'0')%8]++;
        kq+=a[fl][0];
    }
    printf("%lld",kq);
    return 0;
}
