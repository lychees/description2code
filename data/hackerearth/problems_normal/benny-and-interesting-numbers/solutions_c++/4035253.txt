void sieve()
{
	pr[1] = pr[0] = 1;
	for (int i = 2; i * i < MAX; i++)
	{
		if (!pr[i])
		{
			for (int j = i * i; j < MAX; j += i)
			{
				pr[j] = 1;
			}
		}
	}
	for (int i = 2; i < MAX; i++)
	{
		if (!pr[i])
		{
			v.pb(i * i);
		}
	}
	//cout << v.size() << endl;
}
