#include <bits/stdc++.h>
#define ll int
#define gt(i) scanf("%d", &i)

using namespace std;

ll a[27][27][27][27][27] = {0};

int main()
{
    ll n, i, j, k, p, cnt = 0;
    ll b[5];
    gt(n);
    string s;
    for(i = 0; i < n; i++) {
        cin >> s;
        p = 0;
        k = s.size();
        memset(b, 0, sizeof(b));
        for(j = 0; j < k; j++) {
            b[j] = (ll)s[j] - 96;
        }
        if(a[b[0]][b[1]][b[2]][b[3]][b[4]] == 0) {
        	a[b[0]][b[1]][b[2]][b[3]][b[4]] = 1;
        	cnt++;
        }
    }
    cout << cnt << endl;
    ll q, r;
    for(i = 0; i < 27; i++) {
        for(j = 0; j < 27; j++) {
            for(k = 0; k < 27; k++) {
                for(p = 0; p < 27; p++) {
                    for(q = 0; q < 27; q++) {
                        if(a[i][j][k][p][q] == 1) {
                            if(i != 0)
                                cout << (char)(i + 96);
                            if(j != 0)
                                cout << (char)(j + 96);
                            if(k != 0)
                                cout << (char)(k + 96);
                            if(p != 0)
                                cout << (char)(p + 96);
                            if(q != 0)
                                cout << (char)(q + 96);
                            cout << endl;
                        }
                    }
                }
            }
        }
    }
    return 0;
}
