def number_of_pairs(N):
    count = 0
    for i in range(1, N + 1):
        for j in range(i + 1, N + 1):
            if i ^ j <= N:
                count += 1
    return count

T = int(raw_input())
while T != 0:
    N = int(raw_input())
    print(number_of_pairs(N))
    T -= 1
