#include<bits/stdc++.h>
using namespace std;
int main(){
   int t;
   scanf("%d",&t);
   while(t--){
   	long long n,k,f=1;
   	int res=1;
   	scanf("%lld %lld",&n,&k);
   //	printf("%lld %lld\n",n,k);
   	for(int i=0;i<n;i++){
   	f*=2;}
   	if(k>f){
   	printf("-1\n");
   	continue;}
   	while(f>1){
   		if(k>f/2){
   			k=f-k;
   			res*=-1;
   		}
   		else
   		if(k==f/2){
   			if(res==1)
   			printf("G\n");
   			else printf("C\n");
   			break;
   		}
   		f/=2;
   	}
   	if(f==1){
   			if(res==1)
   			printf("G\n");
   			else printf("C\n");
   			break;
   		}
   }
  return 0;
}
