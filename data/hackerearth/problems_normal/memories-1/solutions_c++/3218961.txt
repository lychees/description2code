int main(){
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	bool pairing[15][15] = {0};
	while(t--){
		int n, p, i, j, k, a, b, ans = 0, flag;
		cin>>n>>p;
		fill(pairing, 0);
		/* initializing */
		int total = (1<<n) - 1; // 2^n - 1
		rep(i,0,p-1){
			cin>>a>>b;
			pairing[a-1][b-1] = 1;
			pairing[b-1][a-1] = 1;
		}
		rep(i,1,total){
			flag = 0;
			rep(j, 0, n-1){
				if(((1<<j)&i) == 0) continue;
				else{
					rep(k,0,n-1){
						if(((1<<k)&i) == 0) continue;
						else{
							if(pairing[k][j]){
								flag = 1;
								break;
							}
						}
					}
					if(flag) break;
				}
			}
			if(!flag) ans++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
