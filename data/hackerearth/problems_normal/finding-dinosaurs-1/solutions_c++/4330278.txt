#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,k,l,q;
    
    cin>>n>>k>>l>>q;
    map<vector<int>, string> mapping;
    for(int i=0;i<n;i++){
        string name;
    	cin>>name;
    	vector<int> s;
    	for(int j=0;j<k;j++){
    		int temp;
    		cin>>temp;
    		s.push_back(temp);
    	}
    	
    	mapping[s] = name;
    	
    }
    
    for(int i=0; i<q; i++){
    	vector<int> s;
    	for(int i=0;i<k; i++){
    		int temp;
    		cin>>temp;
    		s.push_back(temp);
    	}
    	if(mapping.find(s) != mapping.end()){
    		cout<<mapping[s]<<endl;
    	}
    	else
    		cout<<"You cant fool me :P"<<endl;
    	}
    
}
