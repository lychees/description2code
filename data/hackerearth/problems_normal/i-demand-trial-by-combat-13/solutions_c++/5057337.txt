#include <iostream>
using namespace std;

int nums[105];
int nums2[105];
int main()
{
	ios::sync_with_stdio(false);cin.tie(0); 
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	int m;
    	cin>>m;
    	//cout<<"scanned nos are"<<endl;
    	for(int i=0;i<n;i++){
    		cin>>nums[i];
    		//cout<<nums[i]<<" ";
    	}
    	//cout<<endl;
    	
    	for(int j=0;j<m;j++){
    		int flag=0;
    	//	cout<<"Temp zero assigned"<<nums[1]<<endl;
    		nums2[0]=nums[1];
    		//cout<<"Temp n-1 assigned"<<nums[n-2]<<endl;
    		nums2[n-1]=nums[n-2];
    	for(int i=1;i<=n-2;i++){
    		if(nums[i-1]==0||nums[i+1]==0){ nums2[i]=0;}
    		else nums2[i]=1;
    	}
    	for(int i=0;i<=n-1;i++){
    	if(nums2[i]!=nums[i]){flag=1;break;}
    	}
    	if(flag==0) break;
    	//cout<<"value after iternation"<<endl;
    	for(int i=0;i<=n-1;i++){
    		nums[i]=nums2[i];
    		//cout<<nums[i]<<" "<<endl;
    	}
    	}
    	for(int i=0;i<=n-1;i++){
    		cout<<nums[i]<<" ";
    	}
    	cout<<"\n";
    }
    return 0;
}
