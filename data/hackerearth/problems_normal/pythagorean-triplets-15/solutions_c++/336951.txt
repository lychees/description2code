using namespace std;
#include<iostream>
#include<math.h>

int gcd(int a,int b)
{
	if(b == 0)
		return a;
	else
		return gcd(b,a%b);
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int count=0,m,n,k,N;
		cin>>N;
		for(m=1;m<=sqrt(N);m++)
			for(n=1;n<m;n++)
			{
				int c=m*m+n*n;
				if(gcd(m,n) == 1 && (m-n)%2!=0 && c<= N)
				{
					k=1;
					while(k*c<=N)
						k++;
					count += k-1;
				}	
			}
		cout<<count<<endl;
	}
return 0;
}
