#include <iostream>
#include <fstream>
#include <cstring>
#include <algorithm>
#include <string>
#include <cstdio>
using namespace std;
char s[22];
int n;
int p[11];
long long dp[22][2][2];
long long poww[22];
void pre()
{
     poww[0] = 1;
     for ( int i = 1; i <= 18; i++ ) poww[i] = poww[i-1]*10;
     return;
}
long long f(int idx, int tight, int flag)
{
	if ( idx == n ) {
		if ( flag ) return 1;
		return 0;
	}
	long long &ans = dp[idx][tight][flag];
	if ( ans != -1 ) return ans;
	ans = 0;
	if ( tight == 0 ) {
        if ( flag ) {
            ans = poww[n-idx];
		    return ans;
	    } 
		for ( int i = 0; i <= 9; i++ ) {
			ans += f(idx+1, 0, flag | p[i]);
		}
	}
	else {
		for ( int i = 0; i <= s[idx]-49; i++ ) {
			ans += f(idx+1, 0, flag | p[i]);
		}
		ans += f(idx+1, 1, flag | p[s[idx]-48]);
	}
	return ans;
}
int main()
{
	p[2] = p[3] = p[5] = p[7] = 1;
	int t;
	pre();
//	freopen("inp5.txt", "r", stdin);
//	freopen("out5.txt", "w", stdout);
	scanf("%d", &t);
	while ( t-- ) {
        scanf("%s", s);  
		n = (int)(strlen(s));
		for ( int i = 0; i < n; i++ ) {
            for ( int j = 0; j < 2; j++ ) {
                for ( int k = 0; k < 2; k++ ) dp[i][j][k] = -1;
            }
        }
		long long ans = f(0,1,0);
		printf("%lld\n", ans);
	}
	return 0;
}
