#include <algorithm>
#include <iostream>
#define DIM 100000
using namespace std;

int n,k,a[DIM];

int main()
{
    cin>>n>>k;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    int ans=0,st=1,mij,dr=(a[0]+a[n-1])/2+5;
    while(st<=dr) {
        mij=(st+dr)/2;
        int p=a[0]+mij,mk=1;
        for(int i=0;i<n;i++)
            if(max(a[i],p)-min(a[i],p)>mij)
                mk++,p=a[i]+mij;
        if(mk>k)
            st=mij+1;
        else
            dr=mij-1,ans=mij;
    }
    cout<<ans;
    return 0;
}
