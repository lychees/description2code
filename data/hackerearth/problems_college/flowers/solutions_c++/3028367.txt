#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,k;
	cin>>n>>k;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	
	int c=0;
	long long int s=0; 
	for(int i=n-1;i>=0;i--)
	{
		c++;
		if(c<=k)
		s+=a[i];
		else
		{
			if(c%k!=0)
			{
				int x=c/k;
				s+=(x+1)*a[i];
			}
			else
			{
				int x=c/k;
				s+=x*a[i];
			}
		}
	}
	cout<<s<<"\n";
}
