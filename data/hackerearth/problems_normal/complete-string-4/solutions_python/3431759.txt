letters = 'qwertyuioplkjhgfdsazxcvbnm'
N = int(raw_input())

for i in range(N):
	complete = True
	D = {}
	s = raw_input()
	for l in s:
		D[l] = D.get(l, 0) + 1
	for l in letters:
		if D.get(l, 0) == 0:
			complete = False
			break
	if complete:
		print 'YES'
	else:
		print 'NO'
