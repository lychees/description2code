#include<cstdio>
#include<cmath>
#include<algorithm>
#define ll long long int
#define ip(n) scanf("%lld",&n);
#define op(n) printf("%lld\n",n);
#define For(a,n) for(int i=a;i<n;i++)
ll a[1003];
using namespace std;
int main()
{
 ll n,t;
 ip(t);
 while(t--)
 {
     ip(n);
   for(ll i=0;i<n;i++)
   {
       ip(a[i]);
   }
   sort(a,a+n);
   ll count =0;
   for(ll i=1;i<n;i++)
   {
       count+=a[0]+a[i];
   }
   op(count);
 }
 return 0;
}
