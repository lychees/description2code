def main():
    input_string = raw_input()
    output_string = ""
    for i in range(len(input_string)):
        if input_string[i].isupper():
            output_string += input_string[i].lower()
        else:
            output_string += input_string[i].upper()
    print output_string


if __name__ == "__main__":
    main()
