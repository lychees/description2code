#include <bits/stdc++.h>
using namespace std;
int prime(int x)
{if(x==1)
return -1;
for(int i=2;i<=sqrt(x);i++)
if(x%i==0)
return -1;
return 1;
}
int main() {
	int t;
	cin>>t;
     	while(t--)
	{int a;
	cin>>a;
	int f=0;int c=1;
	while(f==0)
	{if(prime(a)==1)
	{for(int i=1;i<a/2&&f==0;i++)
	{if((prime(i)!=1)&&(prime(a-i)!=1))
	{f=1;c++;}
	}
	if(f==0)
	{a--;c++;}}
	else
	{f=1;}
	}
	cout<<c<<endl;
	}
	
	
	return 0;
}
