#include <iostream>
using namespace std;

int main()
{
	int n,i,j,a,count;
	cin >> n;
	for(i=0;i<n;i++){
		cin >> a;
		count=0;
		for(j=1;j*j<=a;j++){
			if(a%j==0){
				if(j*j==a)
					count+=1;
				else 
					count+=2;
			}
		
		}
		if(count<4){
			cout << "NO" << endl;
		}else
			cout << "YES" << endl;
	}
    return 0;
}
