#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int n;
    cin>>n;
    vector<string>v;
    string a;
     for(int i=0;i<n;i++)	 
   	  {
   	  	cin>>a;
   	  	v.push_back(a);
   	  }	
   	 sort(v.begin(),v.end());
   	 int count=1;
   	 for(int i=1;i<n;i++)
   	 {
   	 	if(v[i]==v[i-1])
   	 	  count++;
   	 	else
   	 	 {
   	 	 	cout<<v[i-1]<<" "<<count<<endl;
   	 	 	count=1;
   	 	 } 
   	 }
   	cout<<v[n-1]<<" "<<count;
   
    return 0;
}
