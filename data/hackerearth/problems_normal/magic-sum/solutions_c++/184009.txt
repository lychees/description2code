#include<stdio.h>
#include<iostream>
using namespace std;
int magic(int tree[],int i,int j)
{
	int sum=0;
	while(1)
	{
		if(i==j)
		{
			sum+=tree[i];
			break;
		}
		else{
			sum+=tree[i];
			i/=2;
		
		
			sum+=tree[j];
			j/=2;
		}
	}
	return sum;
}
int main()
{
	int TC,n,i,j,sum,maxi,tree[521];
	cin>>TC;
	while(TC--)
	{
		cin>>n;
		for(i=1;i<=n;i++)
			cin>>tree[i];
		maxi=-10000000;
		for(i=(n+1)/2;i<=n;i++)
		{
			for(j=i;j<=n;j++)
			{
				sum=magic(tree,i,j);
				if(sum>maxi)
					maxi=sum;
			}
		}
		cout<<maxi<<endl;
	}
	return 0;
}
