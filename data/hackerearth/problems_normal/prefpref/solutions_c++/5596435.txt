#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s,t;
    long long i,j=0;
    cin>>s>>t;
    for(i=0;i<s.size();){
    	while(j<t.size() && t[j]!=s[i]){
    		j++;
    	}
    	if(j==t.size()){
    		break;
    	}
    	else{
    		i++;
    		j++;
    	}
    }
    cout<<i;
    return 0;
}
