#include <iostream>
#include <set>
#include <algorithm>
using namespace std;

int main()
{
    int n,sum=0;
    cin>>n;
    set <int> odd;
    while(n--)
    {
    	int rose;
    	cin>>rose;
    	sum+=rose;
    	if(rose%2!=0)
    	{odd.insert(rose);}
    }
    if(sum%2!=0){cout<<sum;}
    else
    {
    	if(odd.size()==0)
    	cout<<":(";
    	else
    	{
    		sum-=*(odd.begin());
    		cout<<sum;
    	}
    }
}
