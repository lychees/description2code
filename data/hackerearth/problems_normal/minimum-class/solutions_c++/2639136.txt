#include <bits/stdc++.h>
using namespace std;

main() {
	int t,n,ans;
	cin >> t;
	while(t-- ) {
		cin >> n;
		if(n==0)
			cout << "0" << endl;
		else if(n==2)
			cout << "2" << endl;
		else if(n==3)
			cout << "3" << endl;
		else {
			int x=sqrt(n);
			if(x*x==n) {
				int ans=x+x-1;
				cout<< ans << endl;
			}
			 else {
				ans=x+x;
				if(n<=(x*x+x))
					cout<< ans << endl;
				else
					cout<< ans+1 << endl;
			}
		}
	}
}
