#include<bits/stdc++.h>
using namespace std;
static int BIT[200001][30],n;
void update(int ind,char val){
	while(ind<=n){
		BIT[ind][val-'a']++;
		ind+=(ind&-ind);
	}
}
void update1(int ind,char old,char val){
	while(ind<=n){
		BIT[ind][val-'a']++;
		BIT[ind][old-'a']--;
		ind+=(ind&-ind);
	}
}
bool query(int ind1,int ind2){
	int ans[26];
	for(int i=0;i<26;i++)
	    ans[i]=0;
	while(ind2>0){
		for(int i=0;i<26;i++)
		   ans[i]-=BIT[ind2][i];
		ind2-=ind2&-ind2;
	}
	while(ind1>0){
		for(int i=0;i<26;i++)
		   ans[i]+=BIT[ind1][i];
		ind1-=ind1&-ind1;
	}
	int o=0;
	for(int i=0;i<26;i++){
		if(ans[i]%2!=0&&!o)
		  o++;
		  else
		  if(ans[i]%2!=0&&o)
		     return false;
	}
	return true;
}
static char m[101001];
int main(){
	int q;
	scanf("%d %d",&n,&q);
	scanf("%s",m);
	for(int i=0;i<n;i++){
		update(i+1,m[i]);
	}
	while(q--){
		int type,x,y;
		char c;
		scanf("%d",&type);
		if(type==1){
			scanf("%d %c",&x,&c);
			update1(x,m[x-1],c);
			m[x-1]=c;
		}
		else {
			scanf("%d %d",&x,&y);
			if(query(x-1,y))
			   printf("yes\n"); else printf("no\n");
		}
	}
	return 0;
}
