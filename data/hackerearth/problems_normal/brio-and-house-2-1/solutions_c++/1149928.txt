#include <iostream>
using namespace std;
#include<cmath>
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
  	double x1,x2,y1,y2,r;
  	cin>>x1>>y1>>x2>>y2>>r;
  	
  	double dx=x2-x1;
  	double dy=y2-y1;
    double dr=sqrt(dx*dx+dy*dy);
     double D=x1*y2-y1*x2;
  	
  	double result=r*r*dr*dr-D*D;
  	if(result>0)
  	cout<<"REPLANNING";
  	else if(result==0)
  	cout<<"JUST MISSED";
  	else
  	cout<<"SAFE";
  	cout<<endl;
  }
}
