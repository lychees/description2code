#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    long int n,i,b,g;
    cin>>t;
    string s;
    while(t--){
        cin>>n;
        cin>>s;
        b=0;
        g=0;
        for(i=0;i<s.size();i=i+2){
        if(s[i]=='B')
        b++;
        else
        g++;}
        
        if(b>=g)
        cout<<g<<endl;
        else
        cout<<b<<endl;
    }
    
}
