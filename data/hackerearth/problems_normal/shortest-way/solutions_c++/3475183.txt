#include <iostream>
using namespace std;

int main()
{
   
   int t;
   cin>>t;
   for(int i=0;i<t;i++)
   {
   	long  a,b;
   	cin>>a>>b;
   	long   j=0;
   	while(a!=1 && b!=1){
   		 
   		if(a>b){
   			j+=a/b;
   			a%=b;
   		}
   		else{
   			j+=b/a;
   			b%=a;
   		}
   	}
   	if(b==1){
j+=a-1;
   	}
   	else if (a==1)
   	j+=b-1;
   	cout<<j<<endl;
   }
    return 0;
}
