#include <iostream>
#include <string>
#include <assert.h>

using namespace std;

string str1, str2;

void calc(long int len){
    int cnt = 0;

    for(int i = 0; i < len; i++){
        if(str1[i] == str2[i])
        cnt++;
    }
    cout << cnt << endl;
}

int main(){

    int tc;
    long int len1, len2;

    cin >> tc;
    assert(1 <= tc && tc <= 50);

    while(tc--){
        str1.clear();
        str2.clear();
        cin >> str1;
        cin >> str2;
        len1 = str1.size();
        len2 = str2.size();
        if(len1 == len2){
            calc(len1);
        }else if(len1 > len2)
            calc(len2);
        else
            calc(len1);
    }
    return 0;
}
