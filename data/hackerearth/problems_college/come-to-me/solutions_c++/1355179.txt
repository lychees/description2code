using namespace std;

#define ll long long
#define vt vector

int main()
{
	int t;
	cin>>t;
	string temp;
	while(t--)
	{
		cin>>temp;
		int ans=1;
		for(int len=26;len>=2;len--)
		{
			if(temp.length()<len)
				continue;
			vector<int> vals(26,0);
			bool check=true;
			for(int i=0;i<len;i++)
				vals[temp[i]-65]++;
			for(int i=0;i<26;i++)
			{
				if(vals[i]>1)
					check=false;
			}
			if(check)
			{
				ans = max(ans,len);
				break;
			}
			for(int i=len;i<temp.length();i++)
			{
				check=true;
				vals[temp[i-len]-65]--;
				vals[temp[i]-65]++;
				for(int i=0;i<26;i++)
				{
					if(vals[i]>1)
						check=false;
				}
				if(check)
				{
					ans = max(ans,len);
					break;
				}
			}
			if(check)
				break;
		}
		cout << ans << endl;
	}
	return 0;
}
