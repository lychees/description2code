#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,m;
    cin>>n;
    for(m=0;m<=n;m++)
    {
        std::string s;
        std::getline(std::cin, s);
        int x = 0,i,j;
        int l = s.length();
        int r = pow(l,0.5);
        char arr[r][r];
        for(i=0;i<r;i++)
        {
            for(j=0;j<r;j++)
            {
                arr[j][i] = s[x];
                x++;
            }
        }
        for(i=0;i<r;i++)
        {
            for(j=0;j<r;j++)
            {
                cout<<arr[i][j];
            }
        }
        cout<<endl;
    }
    return 0;
}
