#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
int a[100005];
int hashTable[100005];
int main() {
    int n,q;
    cin>>n>>q;
    string str;
    cin>>str;
    for(int i=1;i<=n;i++) a[i]=str[i-1]-'0';
    int i;
    for(i=1;i<=n;i++) {
        if(a[i]==0) break;
    }
    int val=1;
    hashTable[i]=val;
    for(;i<=n;i++) {
        if(a[i]!=1) {
            if(hashTable[i]==0) {
                val++;
                hashTable[i]=val;
            }
            if(2*i<=n&&a[2*i]!=1) hashTable[2*i]=hashTable[i];
            if(2*i+1<=n&&a[2*i+1]!=1) hashTable[2*i+1]=hashTable[i];
        }
    }
    while(q--) {
        int low,high;
        cin>>low>>high;
        if(hashTable[low]!=0&&(hashTable[low]==hashTable[high])) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        //cout<<hashTable[low]<<" "<<hashTable[high]
    }
    return 0;
}
