#include <iostream>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int array[26];
    for(int i = 0;i<26;i++){
    	array[i] = 0;
    }
    for(int i = 0;i<s.length();i++){
    	array[int(s[i])-97]++;
    }
    int number = 0;
    for(int i = 0;i<26;i++){
    	if(array[i]%2 == 1){
    		number++;
    	}
    }
    if(s.length()%2 == 0){
    	if(number == 0){
    		cout<<"YES";
    	}
    	else{
    		cout<<"NO";
    	}
    }
    else{
    	if(number == 1){
    		cout<<"YES";
    	}
    	else{
    		cout<<"NO";
    	}
    }
    return 0;
}
