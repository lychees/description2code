#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		int n,set=0;
		cin>>s>>n;
		if(n%2!=0)
		n=n+1;
		for(int i=0;i<s.length();i++)
		{
			if(s[i]=='1')
			set++;
		}
		int p=ceil(float(n/2));
		if(p%2==0)
		cout<<set<<endl;
		else
		cout<<s.length()-set<<endl;
	}
}
