#include<bits/stdc++.h>
using namespace std;


int nim_game(int n) {
	vector<int> v;
	//cout<<"x\n";
	for(int i = 2 ; i*i <= n; i++) {
		int cnt = 0;
		while(n % i == 0) {
			n = n / i;
			cnt++;
		}
		if(cnt != 0)
			v.push_back(cnt);
	}
	if(n > 1) {
		v.push_back(1);
	}
	
	int ans = v[0];
	for(int i = 1 ; i < v.size(); i++) {
		ans ^= v[i];
	}
	return ans;
}

int main() {
	int t;
	cin>>t;
	while(t--) {
		int n;
		cin>>n;
		int ans = nim_game(n);
		if(ans == 0) {
			cout<<"Pappu Bhujia\n";
		} else {
			cout<<"Printf{}\n";
		}
	}
	return 0;
}
