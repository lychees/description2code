#include<bits/stdc++.h>
using namespace std;
struct ghost{
	int titles;
	int age;
};
struct myComp{
    bool operator()(const ghost& a, const ghost& b){
        if(a.titles==b.titles)
			return a.age<b.age;
		return a.titles<b.titles;
    }
};
int main(){
	cout.sync_with_stdio(0);
	cin.tie(0);
	int n,m,i,a;
	map<int,int> myMap;
	priority_queue<ghost,vector<ghost>,myComp> g;
	ghost tmp;
	cin>>n>>m;
	for(i=0;i<n;++i){
		cin>>a;
		if(myMap.count(a))
			++myMap[a];
		else
			myMap[a]=1;
		tmp.age=a;
		tmp.titles=myMap[a];
		g.push(tmp);
		cout<<g.top().age<<" "<<g.top().titles<<"\n";
	}
	return 0;
}
