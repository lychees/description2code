typedef long long ll;

ll pwr(ll base, ll p, ll mod = MOD){
ll ans = 1;while(p){if(p&1)ans=(ans*base)%mod;base=(base*base)%mod;p/=2;}return ans;
}



int main()
{
    ios_base::sync_with_stdio(0);

    int t;
    cin>>t;
    while(t--){

    	ll n;
    	cin>>n;
    	bool ok = true;
    	for(ll i=2;i*i<=n;i++)
    		if(n % i == 0){
    			n /= i;
    			if(n % i == 0)
    				ok = false;
    		}

    	if(ok)	cout<<"Yes\n";
    	else	cout<<"No\n";
    }

    return 0;
}
