#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

long long ta[100005],xa[100005],ya[100005];

int main()
{
    long long i,j,n,m,temp,d,speed=0,ans=0,t,x,y;
    cin>>n>>d;

    for(i=1;i<=100000;i++) ta[i]=-1;

    for(i=1;i<=n;i++) {
        cin>>t>>x>>y;
        if(ta[t]==-1) {
            ta[t]=1; xa[t]=x; ya[t]=y;
        } else if(ya[t]<y || (ya[t]==y && xa[t]>x)) xa[t]=x,ya[t]=y;
    }
    for(i=1;i<=100000;i++) {
        if(ta[i]!=-1 && ya[i]>speed) speed=ya[i],ans+=xa[i];
        d-=speed;
        if(d<=0) break;
    }
    cout<<ans;
    return 0;
}
