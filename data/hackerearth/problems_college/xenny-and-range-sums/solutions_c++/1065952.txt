#include <iostream>

using namespace std;

int main()
{
    long n,q,arr[1000000],l,r,sum,diff;
    cin>>n>>q;
    for (int i =1;i<=n;i++)
        {
            cin>>arr[i];
        }
    for (int j=0;j<q;j++)
        {
            cin>>l>>r;
            sum = 0;
            diff = arr[r];
            for (int k=l;k<=r;k++)
                {
                    sum+=arr[k];
                }
                diff = 2*arr[r]-sum;
            cout<<sum<<" "<<diff<<endl;
        }
}
