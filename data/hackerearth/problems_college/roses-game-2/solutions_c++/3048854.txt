int main()
{

        for (int i = 0;i <n;i++){
            cin >> a[i][0];
            a[i][1] = 0;
        }

        int count = 0;
        int k = 0;
        int i = 0;

        while(1)
        {
            if (a[i][0] == -1)
            {
                k = 1;

            }
