#include <iostream>
#include <string.h>

using namespace std;

int main()
{
	int t,n;

	cin>>t;
	
	while(t--){
		cin>>n;
		
		string str[n];
		
		for(int i=0;i<n;i++){
			cin>>str[i];
		}
		
		bool flag=true;
		
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				
				int pos1= str[i].find(str[j]);
				int pos2= str[j].find(str[i]);
				
				
				if(pos1==0 || pos2==0){
					flag=false;
					break;
				}
			}
		}
		
		
		if(flag==false){
			cout<<"NO\n";
		}else{
			cout<<"YES\n";
		}
	}
	return 0;
}
