#include <iostream>
#include <map>
#include <vector>
using namespace std;

int main()
{
	string s;
	map<string,int> mp;
	vector<string> v;
	vector<string>::iterator it;
   	while(cin>>s){
   		for(int i=0; i<s.size();i++){
   			if(s[i]>='a' && s[i]<='z')
   				s[i] = s[i]-'a'+'A';
   		}
   		v.push_back(s);
   		mp[s]++;
   	}
   	
	for(it=v.begin();it!=v.end();it++){
		if(mp[*it]!=0){
			cout<<*it<<" "<<mp[*it]<<endl;
			mp[*it] = 0;
		}
	}
   	
    return 0;
}
