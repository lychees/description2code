using namespace std;
int x,y,n,w;double p1,p2;
double dp[2000][2000];
double func(int sum,int ch)
{
    if(sum>w)
        sum=w;
    if(ch>=n)
    {
        if(sum>=w)
            return 1.0;
        else
            return 0.0;
    }
    if(dp[sum][ch]!=-1.0)
        return dp[sum][ch];
    else
        return dp[sum][ch]=max(p1*func(sum+x,ch+1)+(1-p1)*func(sum,ch+1),p2*func(sum+y,ch+1)+(1-p2)*func(sum,ch+1));
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        for(int i=0;i<2000;i++)
        {
            for(int j=0;j<2000;j++)
            {
                dp[i][j]=-1.00;
            }
        }
        cin>>x>>y>>n>>w>>p1>>p2;
        p1/=100.0;
        p2/=100.0;
        cout<<fixed;
        cout<<setprecision(6);
        cout<<(func(0,0)*100)<<endl;
    }
}
