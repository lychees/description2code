#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll MOD=1000000007;
ll a[1000005];
ll pow(ll b,ll p)
{
    if(p==1)
        return b;
    if(p%2==1)
        return(pow(b,p-1)*b)%MOD;
    else
    {
        long long res=pow(b,p/2);
        return (res*res)%MOD;
    }
}
int main()
{
    ll i,t,n,r,x,res;
    a[0]=a[1]=1;
    for(i=2; i<=1000001; i++)
        a[i]=(a[i-1]%MOD*i%MOD)%MOD;
    cin>>t;
    for(i=1; i<=t; i++)
    {
        cin>>n>>r;
        swap(n,r);
        //cout<<"Case "<<i<<": ";
            res=((a[n-r]%MOD))%MOD;
            cout<<((a[n]%MOD)*(pow(res,MOD-2)%MOD)%MOD)<<endl;
    }
}
