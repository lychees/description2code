#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long t,arr[26] = {0},len = 0,pos;
    string str;
    cin>>str;
    int x;
    for(x=0;x<str.length();x++)
    {
    	int j = x+1 , num = 0;
    	while(j<str.length() && (str[j] >= '0' && str[j] <= '9'))
    	{
    		num = num*10 + str[j] - '0';
    		j++;
		}
    	arr[str[x] - 'a'] += num;
    	len += num;
    	x = j-1;
    }

    cin>>t;
    while(t--)
    {
    	cin>>pos;
    	if(pos <= len)
    	{
    		pos--;
    		int i = 0;
    		while(pos>0)
    		{
    			pos -= arr[i];
    			if(pos < 0)
    				break;
    			if(pos == 0)
    			{
    				i++;
    				break;
				}
    			i++;
    		}
    		while(arr[i] == 0)
    			i++;
    		cout<<(char)(i+'a')<<"\n";
    	}
    	else
    		cout<<"-1\n";
    }
}
