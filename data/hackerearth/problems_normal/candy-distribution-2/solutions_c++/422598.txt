#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t,n,j;
    
    long long int sum;
    
    scanf("%d", &t);
    
    for (int i = 0; i < t; i++) {
    	scanf("%d", &n);
    	long long int a[n];
    	sum = 0;
    	for (j = 0; j < n; j++) {
    		scanf("%lld", &a[j]);
    		sum += a[j] - 1;
    	}
    	
    	sum += 1;
    	
    	printf("%lld\n", sum);
    	
    }
    
    return 0;
}
