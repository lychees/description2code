#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int l=s.length();
    for(int i=0;i<l;i++)
    {
    	int p=s.at(i);
    	int d=p;
    	if(d>96&&d<=123)
    	{
    	char c=d-32;
    	cout<<c;
    	}
    	else if(d>=65&&d<=90)
    	{
    			char c=d+32;
    	cout<<c;
    	}
    }
       return 0;
}
