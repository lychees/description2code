#include <iostream>
using namespace std;
int main()
{
    long long int a,b,ans;
    cin>>a>>b;
    long long int x[b+1],y[a+1];
    for(int i=1;i<=b;i++)
        x[i]=0;
    
    
    for(int i=1;i<=a;i++)
    {
        int n;
        cin>>n;
        y[i]=n;
        x[y[i]]++;
    }
    long int mini=x[1];
    ans=1;
    for(int i=2;i<=b;i++)
        if(x[i]>mini)
        {mini=x[i];
            ans=i;
        }
    cout<<ans<<endl;
    return 0;
}
