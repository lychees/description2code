#include <bits/stdc++.h>
using namespace std;
#define MAX 10000000
int main()
{
    int t,n,hash[MAX],x,count;
    cin>>t;
    while(t--){
    	cin>>n;
    	count=0;
    	memset(hash,0,sizeof(hash));
    	while(n--){
    		cin>>x;
    		if(hash[x%10]==1)
    			count++;
    		hash[x%10]=1;
    	}
    	cout<<count<<endl;
    	
    }
    return 0;
}
