#include <bits/stdc++.h>
using namespace std;

int BS(int b[], int l, int h, int k) {
	int m;
	while(l < h) {
		m = (h+l)/2;
		if(b[m] == k) {
			l = m; break;
		}
		else if(b[m] < k) l = m+1;
		else h = m-1;
	}
	if(b[l] >= k)return l+1;
	else return l+2;
}
int main()
{
    // cout << "Hello World!" << endl;
    int n, i;
    cin>>n;
    int a[n], b[n];
    for(i = 0; i < n; i++) {
    	cin>>a[i]; b[i]= 0;
    }
    b[0] = a[0];
    for(i = 1; i < n; i++) b[i] = b[i-1]+a[i];
    sort(b, b+n);
    int q, num, ans;
    cin>>q;
    while(q--) {
    	cin>>num;
    	if(num > b[n-1]) {
    		cout<<"-1\n"; continue;
    	}
    	ans = BS(b, 0, n-1, num);
    	cout<<ans<<endl;
    }
    return 0;
}
