#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int T,M,N,U,V,W,X,G,H,i;
	cin>>T;
	pair< int , int > o[100001],n[100001];
	while(T--)
	{
		cin >> M >> N;
		for(i=0;i<M;i++)
		{
			cin >> U >> V;
			n[i] = make_pair(U,V);
		}
		for(i=0;i<N;i++)
		{
			cin >> W >> X;
			o[i] = make_pair(W,X);
		}
		cin >> G >> H;
		sort(n,n+M);
		sort(o,o+N);
		int count=0,j=0;
		i=0;
		while(count<G && i<N && j < M)
		{
			if(n[j].first == o[i].first)
			{
				j++;
				i++;
				count++;
			}
			else if(o[i].first < n[j].first)
			{
				i++;
			}
			else
			{
				j++;
			}
		}
		if(count == G)
		{
			i=0;
			j=0;
			count=0;
			while(count<H && i<N && j < M)
		{
			if(n[j].first == o[i].first)
			{
	            if(n[j].second == o[i].second)
				{
				   count++;
				   i++;
				   j++;	
				}
				else if (n[j].second > o[i].second)
				{
				    i++;	
				
				}
				else
				{
				  j++;	
				}     
			}
			else if(o[i].first < n[j].first)
			{
			    i++;
			}
			else
			{
			    j++;
			}
		}
			if(count == H)
			{
				cout << "Great" << endl;
			}
			else cout << "Good" << endl;
		}
		else
		{
			cout << ":(" << endl;
		}
	}
	return 0;
}
