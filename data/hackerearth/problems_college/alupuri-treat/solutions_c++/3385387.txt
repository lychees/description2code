#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int m,sum1=0,sum2=0,diff,i,j;
	cin >> m;
	int a[m][m];
    for(i=0;i<m;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin >> a[i][j];
    	}
    }
    for(i=0;i<m;i++)
    {
    	sum1 = sum1 + a[i][i];
    	sum2 = sum2 + a[i][m-i-1];
    }
    diff = sum1 - sum2;
    cout << abs(diff) << endl;
    
    
    return 0;
}
