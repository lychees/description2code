#include <bits/stdc++.h>
using namespace std;
long long pow(int a, int b, int MOD)
{
    long long x=1,y=a; 
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD; 
        b /= 2;
    }
    return x;
}
long long InverseEuler(int n, int MOD)
{
    return pow(n,MOD-2,MOD);
}
int pr[1000001];
int x[1000001];
int main()
{

    int n,t;
    long long int mo=1000000007;    
    cin >> t;
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
            scanf("%d",&pr[i]);
        for(int i=0;i<n;i++)
            scanf("%d",&x[i]);
        long long int ans=1;
        for(int i=0;i<n;i++)
            ans=(ans*pr[i])%mo;
        for(int i=0;i<n;i++)
        {
            long long int temp=(pow(pr[i],x[i],mo)-1+mo)%mo;
            temp=(temp*InverseEuler(pr[i]-1,mo))%mo;
            ans=(ans*temp)%mo;
        }
        printf("%d\n",ans);


    }
    return 0;
}
