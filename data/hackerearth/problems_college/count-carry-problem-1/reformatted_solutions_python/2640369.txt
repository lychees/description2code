for _ in range(int(input())):
    a, b = map(str, raw_input().split())
    count = 0
    for i in range(len(a)):
        if int(a[i]) + int(b[i]) > 9:
            count += 1
    if count == 0:
        print('No carry operation')
    elif count == 1:
        print(str(count)+' carry operation')
    else:
        print(str(count)+' carry operations')
