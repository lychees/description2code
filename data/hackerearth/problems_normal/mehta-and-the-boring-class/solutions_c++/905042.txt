#include <iostream>
using namespace std;
#define MAX 1000007
int prime[1000007]={0};
void sieve()
{
	int i,j,k;
	for(i=2;i<=MAX;i++)
	{
		if(prime[i]!=0)
		continue;
		for(j=i;j<=MAX;j+=i)
		{
			k=j;
			while(k%i==0)
			{
				prime[j]++;
			    k/=i;
			}
		}
	}
	for(i=2;i<MAX;i++)
	{
		prime[i]+=prime[i-1];
	}
}
int main()
{
	int sum,count,i,j,a,b,t,k;
	sieve();
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&a,&b);
		printf("%d\n",prime[b]-prime[a-1]);
	}
    return 0;
}
