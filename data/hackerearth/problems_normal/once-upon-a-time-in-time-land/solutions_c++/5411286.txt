#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int t, n, k, i;
    
    cin >> t;
    
    while(t--){
    	
    	cin >> n >> k;
    	vector<int> v(n);
    	vector<long long> dp(n+k+1);
    	for(i = 0; i < n; i++)
    		cin >> v[i];
    	
    	for(i = n+k; i >= 0; i--){
    		if(i >= n) dp[i] = 0;
    		else dp[i] = max(dp[i+1], v[i]+dp[i+k+1]);
    	}
    	cout << dp[0] << endl;
    }
    
    return 0;
}
