#include <iostream>
#include <queue>
using namespace std;

int main()
{
    int i,m,n,x,money = 0;
    priority_queue <int> pq;
    cin>>n>>m;
    for(i = 0 ; i < n ; i++)
    {
    	cin>>x;
    	pq.push(x);
    }
    while(m--)
    {
    	money = money+pq.top();
    	int temp = pq.top();
    	pq.pop();
    	pq.push(temp-1);
    }
    cout<<money;
    return 0;
}
