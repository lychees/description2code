ll mod= 1e7 + 7, mod1=1e9+9;

{   io;

                a[i][j]=a[i-1][j];
                if((j^v1[i]) < (1<<11))a[i][j]+=a[i-1][j^v1[i]]%mod;

 	if(sz1>0)   a[sz1-1][0]--;
