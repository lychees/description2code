#include <iostream>
#include <algorithm>
using namespace std;

int a[100];

int dcost(int n, int k, int val) {
	int m = -1;
	for(int i=0; i+k-1<n; i++) {
		int c = 0;
		for(int j=0; j<k; j++) {
			if (a[i+j] < val) c += (val - a[i+j])*3;
			else c += (a[i+j] - val)*5;
		}
		if (m == -1 || m > c) m = c;
	}
	return m;
}

void solve() {
	int n, k;
	cin >> n >> k;
	for(int i=0; i<n; i++) cin >> a[i];
	sort(a,a+n);
	int mc = dcost(n,k,1);
	for(int val=2; val<101; val++) mc = min(mc, dcost(n, k, val));
	cout << mc << endl;
}

int main()
{
    int T; cin >> T;
    while (T--) solve();
    return 0;
}
