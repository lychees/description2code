#include<iostream>
#include<stdio.h>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	int i,n,j,x,y;
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		int *a=new int[n];
		int *b=new int[n];
		for(j=0;j<n;j++)
		{
			cin>>a[j];
			cin>>b[j];
		}
		sort(a,a+n);
		sort(b,b+n);
		x=a[n-1]-a[0];
		y=b[n-1]-b[0];
		if(x==0)
		cout<<y<<endl;
		else if(y==0)
		cout<<x<<endl;
		else 
		cout<<x*y<<endl;
		
	}
	return 0;
}
