#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,t,i,j,k,flag=0,sum=0;
	cin>>n;
	int arr[n];
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	cin>>t;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			for(k=0;k<n;k++)
			{
				if(arr[i]!=arr[j] && arr[j]!=arr[k] && arr[k]!=arr[j])
				{
					sum=sum+arr[i]+arr[j]+arr[k];
					
					if(sum==t)
					{	flag=1;
						cout<<"True"<<endl;
						break;
					}
					sum=0;	
				}
			
		}
		if(flag==1)
			break;
	}
	if(flag==1)
			break;
	}
	if(flag==0)
	cout<<"False"<<endl;
	}
