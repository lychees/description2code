#include <iostream>
#include<vector>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int v[1000001];
    v[0]=0;
    v[1]=1;
    v[2]=1;
    int val=1;
    int i;
    for(i=3;val<1000001;i++)
    {
    	val=v[i-1]+v[i-2];
    	v[i]=val;
    }
    int t;
    cin>>t;
    while(t--)
    {
    	int x,y;
    	i=0;
    	int count=0;
    	cin>>x>>y;
    	for(i=0;v[i]<=y;i++)
    	{
    		if(v[i]>=x && v[i]<=y)
    		count++;
    		
    	}
    	if(count%2==0)
    	cout<<"VALID\n";
    	else
    	cout<<"INVALID\n";
    	
    }
    
    return 0;
}
