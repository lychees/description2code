#include <iostream>
#define MOD 1000000007
using namespace std;

long long int fibo[1000005],sum[1000005],num,t;

int main()
{
    fibo[0]=1;
    fibo[1]=2;
    for(int i=2;i<1000005;i++)
    {
    	fibo[i]=(fibo[i-1]+fibo[i-2])%MOD;
    }
    sum[0]=1;
    sum[1]=2;
    for(int i=2;i<1000005;i++)
    {
    	sum[i]=(fibo[i]+sum[i-1])%MOD;
    }
    cin>>t;
    while(t--)
    {
    	cin>>num;
    	cout<<sum[num]<<endl;
    }
    return 0;
}
