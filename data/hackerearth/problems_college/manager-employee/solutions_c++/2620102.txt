#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int arr[107];
bool visited[107];
int main()
{
    int t; cin >> t;
    int n;
    while(t--){
    	cin >> n;
    	for(int i = 0; i < n; i++)cin >> arr[i];
    	sort(arr, arr+n);
    	memset(visited, 0, sizeof visited);
    	vector <int> ans;
    	for(int i = 0; i < n; i++){
    		if(visited[i])break;
    		int tmp = arr[i] + arr[n-1-i];
    		ans.push_back(tmp);
    		visited[i] = visited[n-1-i] = 1;
    	}
    	sort(ans.begin(), ans.end());
    	cout << ans[ans.size()-1] - ans[0] << endl;
    }
    return 0;
}
