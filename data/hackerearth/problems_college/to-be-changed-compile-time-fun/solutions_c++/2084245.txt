#include<bits/stdc++.h>
using namespace std;
vector<int> ar;
int main()
{
    int n,a;
    int ele;
    int maxi;
    long long ans;
    cin>>n;
    vector<int>::iterator it,t;
    for(int i=0;i<n;i++)
    {
        cin>>a;
       ar.push_back(a);
    }

    ans=0;
    maxi=0;
    while(n>2)
    {
    	maxi=0;
        for(it=ar.begin()+1;it<ar.end()-1;it++)
        {
          if(*(it-1)*(*(it+1))>maxi)
          {
              maxi=(*(it-1))*(*(it+1));
              t=it;

          }
    }
    ans+=maxi;
    ar.erase(t);
    n--;
    }
    cout<<ans;
    return 0;

}
