import sys
t = int(raw_input())
while t > 0:
    t -= 1
    a, b, c = map(int, raw_input().split())
    k = int(a/b)
    sys.stdout.write(str(k))
    if c > 0:
        sys.stdout.write(".")
    rem = a-b*k
    i = 0
    while rem != 0 and i < c:
        rem *= 10
        a = rem
        k = int(a/b)
        sys.stdout.write(str(k))
        rem = a-b*k
        i += 1
    while i < c:
        sys.stdout.write("0")
        i += 1
    print
