#include <iostream>
using namespace std;
string n,m;

int recur(int in,int im,int c)
{
	if(in<0 && im<0 && c==0)return 0;
	
	int add=((in>=0)?(n[in]-'0'):0) + ((im>=0)?(m[im]-'0'):0)  + c;
	
	int carry=add/10;
	
	return carry + recur(in-1,im-1,carry);
	
}

int main()
{
    
    int t;
    cin>>t;
    
    while(t--)
    {
    	cin>>n>>m;
    	
    	int ans=recur(n.size()-1,m.size()-1,0);
    	
    	if(ans==0)
    		cout<<"No carry operation\n";
    	else if(ans==1)
    		cout<<"1 carry operation\n";
    	else
    		cout<<ans<<" carry operations\n";
    
    }
    
    return 0;
}
