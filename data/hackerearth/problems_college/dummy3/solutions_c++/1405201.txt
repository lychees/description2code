#include<iostream>
#include<fstream>
#define MAX 10000000
using namespace std;
int main()
{
  int i,j,n,a[1005][1005],ans[1005][1005],final=MAX,k;
  cin>>n;
  for(i=1;i<=n;i++) {
    for(j=1;j<=n;j++) cin>>a[i][j],ans[i][j]=MAX;
  }
  for(i=1;i<=n;i++) ans[i][1]=a[i][1];
  for(i=2;i<=n;i++) {
    for(j=1;j<=n;j++) {
      ans[j][i]=min(ans[j][i],ans[j][i-1]+a[j][i]);
      if(ans[j-1][i]>ans[j][i] && j>1) {
        for(k=j-1;k>=1;k--) {
          if(ans[k+1][i]+a[k][i]<ans[k][i]) ans[k][i]=ans[k+1][i]+a[k][i];
          else break;
        }
      }
      else if(ans[j-1][i] < ans[j][i] && j>1) {
        if(ans[j-1][i]+a[j][i]<ans[j][i]) ans[j][i]=ans[j-1][i]+a[j][i];
      }
    }
  }
  for(i=1;i<=n;i++) {
    final=min(final,ans[i][n]);
  }
  cout<<final;
//  system("pause");
  return 0;
}
