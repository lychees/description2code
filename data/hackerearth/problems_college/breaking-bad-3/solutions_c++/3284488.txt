#include <iostream>
#include <algorithm>
using namespace std;
int n;
long int a[200020]={0};
long long int s,i;
long long int out=0;
int main()
{
	cin>>n;
	while(n--){
		s=0;
		out=0;
		cin>>s;
		for(i=0;i<s;i++)
			cin>>a[i];
		//	cout<<a[0]<<"\n";
		sort(begin(a), a+s);	
		for(i=s-1;i>=0;i-=2){
			out+=a[i]*a[i-1];
			//cout<<" "<<a[i];
		}
		cout<<out<<"\n";
		for(i=0;i<s;i++)
		a[i]=0;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
