        scanf("%d%d", &n, &k);
        REP(i, n) scanf("%d", a + i);
        int lo = 1, hi = 1e9 + 1, mid;
        while (lo < hi) {
            mid = (lo + hi) >> 1;
            int pos = 0;
            REP(i, k) {
                if (a[pos + 1] - a[pos] > mid) break;
                pos = lower_bound(a + pos + 1, a + n, a[pos] + mid + 1) - a - 1;
                if (pos == n - 1) break;
            }
            if (pos == n - 1) hi = mid;
            else lo = mid + 1;
        }
        printf("%d\n", lo);
    }

}
