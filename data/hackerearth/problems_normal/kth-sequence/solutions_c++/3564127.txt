#include <iostream>
#include <limits>
using namespace std;

typedef long long int ll;
constexpr ll MAX = numeric_limits<long long int>::max() / 2;


ll table[1024];

int main()
{
	table[0] = 1;
	table[1] = 2;
	for(int i = 2; i <= 1000; i++)
	    table[i] = min(MAX, table[i - 1] + table[i - 2]);

	    
	
	int T;
	cin >> T;
	while(T--)
	{
	    ll N, K;
	    cin >> N >> K;
	    
	    if(table[N] < K)
	        cout << -1 << '\n';
	    else
	    {
	    	for(int i = N; i > 0; i--)
	    	    if(table[i - 1] >= K)
	    	        cout << 'O';
	    	    else
	    	    {
	    	        cout << "Z";
	    	        if(i > 1)
	    	            cout << 'O';
	    	        K -= table[i - 1];
	    	        i--;
	    	    }
	    	cout << '\n';
	    }
	    
	}
    
    return 0;
}
