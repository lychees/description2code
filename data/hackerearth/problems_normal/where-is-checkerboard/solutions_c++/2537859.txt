#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,i,j,k;
    static char a[2001][2001];
    static int vert[2001][2001],hor[2001][2001];
    cin>>n>>m;
    
    for(i=0;i<n;i++)
    {
    	for(j=0;j<m;j++)
    	{
    		cin>>a[i][j];
    	}
    }
    
    for(i=n-1;i>=0;i--)
    {
    	for(j=0;j<m;j++)
    	{
    		if(i==n-1 || a[i][j]==a[i+1][j])
    		{
    			vert[i][j] = 1;
    		}
    		else
    		{
    			vert[i][j] = vert[i+1][j] + 1;
    		}
    	}
    }
    
    for(i=0;i<n;i++)
    {
    	for(j=m-1;j>=0;j--)
    	{
    		if(j==m-1 || a[i][j]==a[i][j+1])
    		{
    			hor[i][j] = 1;
    		}
    		else
    		{
    			hor[i][j] = hor[i][j+1] + 1;
    		}
    	}
    }
    
    int ans = 0,mi;
    
    for(i=0;i<=n-32;i++)
    {
    	for(j=0;j<=m-32;j++)
    	{
    		if(a[i][j]=='_' && vert[i][j]>=32)
    		{
    			mi = hor[i][j];
    			for(k=1;k<32;k++)
    			{
    				mi = min(mi,hor[i+k][j]);
    			}
    			if(mi>=32)
    			{
    				ans++;
    			}
    		}
    	}
    }
    
    cout<<ans;
    
    return 0;
}
