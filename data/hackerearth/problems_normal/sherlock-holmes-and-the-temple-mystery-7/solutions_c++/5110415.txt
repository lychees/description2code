#include <iostream>
using namespace std;

int main()
{
    int nTest;
    cin>>nTest;
    for(int n=0; n<nTest; n++)
    {
        int nItems;
        cin>>nItems;
        int cost[nItems];
        int sum = 0;

        for(int i=0; i<nItems; i++)
        {
            cin>>cost[i];
            sum += cost[i];
        }
        if(sum & 1)
            cout<<"NO"<<endl;
        else
        {
            int halfsum = sum/2;
            bool possible[halfsum+1];
            possible[0] = true;
            for(int i = 1; i<=halfsum; ++i)
                possible[i] = false;
            for(int idx = 0; idx < nItems; idx++)
            {
                for(int idy = halfsum; idy >= cost[idx]; idy--)
                {
                    possible[idy] |= possible[idy-cost[idx]];
                }
       			if(possible[halfsum])
       				break;
            }
            if(possible[halfsum])
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
        }
    }
    return 0;
}
