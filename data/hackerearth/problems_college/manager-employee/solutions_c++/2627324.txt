#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        int n;
        scanf("%d", &n);
        int a[101];
        for (int i = 0; i < n; i++) {
            scanf("%d", &a[i]);
        }
        sort(a, a+n);
        vector<int> v;
        for (int i = 0, j = n-1; i < j; i++, j--) {
            v.push_back(a[i]+a[j]);
        }
        sort(v.begin(), v.end());
        printf("%d\n", v[v.size()-1]-v[0]);
    }

    return 0;
}
