#include <bits/stdc++.h>
using namespace std;
int dp[205];
int flag;
int fnc(int pg)
{
	if(pg < 0) {
		flag = -1;
		return 0;
	}
	else if(pg == 0 || dp[pg] != 0)
		return dp[pg];
	dp[pg] = 1 + min( fnc(pg-10), fnc(pg-12));
	return dp[pg];
}

int main()
{
    int t,i,j;
    cin >> t;
    int arr[200];
    for(i=0;i<200;i++)
    	arr[i] = -1;
    for(i=0;i<=20;i++) {
    	for(j=0;(10*i+12*j)<200;j++) {
    		arr[10*i+12*j] = i+j;
    	}
    }
    while(t--) {
    	int n;
    	cin >> n;
    	flag = 0;
    	//int i,j;
    	int ans = fnc(n);
    	//if(flag == -1)
    	//	cout<<"-1\n";
    	//else
    	if(arr[n]!=-1)
    		cout<<ans<<endl;
    	else
    		cout<<-1<<endl;
    }
    return 0;
}
