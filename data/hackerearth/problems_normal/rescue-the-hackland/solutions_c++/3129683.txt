#include <iostream>
using namespace std;

int main()
{
	int count =1;
	int tests =0;
	int L,R,K =0;
	int low=0;
	cin >> tests;
	for (int i=0;i<tests;i++)
	{
		cin >> L >> R >> K;
		for(int j=L;j<R;j++)
		{
		if(j*K<R&&((low)*K<j))
		{
	     	low=j;
		    count++;
		}
		}
    cout << count << endl;
    count=1;
    low=0;
	}
    return 0;
}
