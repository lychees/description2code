#include <bits/stdc++.h>

using namespace std;

vector <int> V[100005];

void DFS (int *visited, int x)
{
	int i,j;
	visited[x]=1;
	
	for (i=0;i<V[x].size();++i)
		if (visited[V[x][i]]==0)
			DFS(visited, V[x][i]);
}

int main()
{
    int t,n,e,i,j,k,a,b;
    scanf ("%d",&t);
    while (t--)
    {
    	scanf ("%d %d",&n,&e);
    	while (e--)
    	{
    		scanf ("%d %d",&a,&b);
    		V[a].push_back(b);
    		V[b].push_back(a);
    	}
    	int visited[n],cnt=0;
    	for (i=0;i<n;++i)
    		visited[i]=0;
    		
    	for (i=0;i<n;++i)
    	{
    		if (visited[i]==0)
    		{
    			DFS(visited, i);
    			cnt++;
    		}
    	}
    	printf ("%d\n",cnt);
    	for (i=0;i<n;++i)
    		V[i].clear();
    }
    return 0;
}
