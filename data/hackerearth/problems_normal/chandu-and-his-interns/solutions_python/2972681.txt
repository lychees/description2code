import math
n = input()
for i in xrange(0,n):
	count=0
	pflag = False
	m = input()
	mm = math.sqrt(m)
	for j in xrange(1,int(mm)+1):
		if m%j == 0:
			if j*j == m:
				count += 1
			else:
				count += 2
			if count >= 4:
				print "YES"
				pflag = True
				break
	if pflag == False:
		print "NO"
