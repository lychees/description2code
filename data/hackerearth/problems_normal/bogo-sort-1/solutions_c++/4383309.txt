#include <iostream>
using namespace std;

void swap(int &a, int &b)
{
	int temp = a;
	a = b;
	b = temp;
}

int partition(int A[], int l, int r)
{
	int p = A[l], i = l + 1;
	for(int j = l + 1; j <= r; j++)
		if(A[j] < p)
		{
			swap(A[i], A[j]);
			i++;
		}
	swap(A[l], A[i - 1]);
	return i - 1;
}

void quick_sort(int A[], int l, int r)
{
	if(l < r)
	{
		int pivot = partition(A, l, r);
		quick_sort(A, l, pivot);
		quick_sort(A, pivot + 1, r);
	}
}

int main()
{
    int N;
    int A[10000];
    cin >> N;
    for(int i = 0; i < N; i++)
    {
    	cin >> A[i];
    }
    quick_sort(A, 0, N - 1);
    for(int i = 0; i < N; i++)
    {
    	cout << A[i] << endl;
    }
    return 0;
}
