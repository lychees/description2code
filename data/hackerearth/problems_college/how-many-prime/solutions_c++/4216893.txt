#include <iostream>
#include <cmath>
using namespace std;

#define LL long long int

int prime_check(int x){
    for(int i=2; i<=sqrt(x); i++){
        if(x%i == 0)
            return 1;
    }
    return 0;
}

int main()
{
	int tc;
	cin >> tc;
	while(tc--){
		int n, a, b;
		cin >> n;
		if(n%2==0){
			a = n/2-1;
			b = n/2+1;
		}
		else{
			a = (n+1)/2-1;
			b = (n-1)/2+1;
		}
		//	**	//
		while(1){
            if (a < 2){
                cout << "-1 -1" << endl;
                break;
            }
			else if(prime_check(a) == 0){
                    if(prime_check(b) == 0){
                        cout << a << " " << b << endl;
                        break;
                    }
                    else{
                        a--; b++;
                    }
            }
            else{
                a--; b++;
            }
        }
		//	**	//
	}
	return 0;
}
