#include <iostream>
#include <algorithm>
using namespace std;

typedef long long ll;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll a,b,c;
		cin>>a>>b>>c;
		ll i,j;
		//linear search for solution; no need for binary search or other algo
		i=1;
		bool fw = false; //does first player win?
		while(true){
			fw = !fw;
			ll k = c/b;
			if(c%b) k++;
			if(k<=i) break;
			i = b*i;
			/*if(c<=i) //prevent overflow of 64 bit int
				break;*/
			swap(a,b);
			}
		cout<<(fw ? "Adam\n" : "Bob\n");
	}
	return 0;
}
