#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,i,count=0,x[30],x1;
char str[1003];
cin>>t;
while(t--)
{
for(i=0;i<30;i++)
x[i]=0;

cin>>str;
x1=strlen(str);

for(i=0;i<x1;i++)
{
x[str[i]-97]++;
}
for(i=0;i<25;i++)
{
if(x[i]==x[i+1])
count++;
}
if(count==25)
cout<<"Yes\n";
else
cout<<"No\n";
count=0;
}
return 0;
}
