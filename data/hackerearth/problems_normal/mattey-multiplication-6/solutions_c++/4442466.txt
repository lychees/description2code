using namespace std;

#define ll long long

ll pow2(ll& M)
{
	ll x = 1, cnt = 0;
	while(x < M)
	{
		x <<= 1;
		cnt++;
	}
	if (x == M)
	{
		M -= x;
		return cnt;
	}
	else
	{
		M -= x/2;
		return cnt-1;
	}
}

int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		ll N, M;
		bool first = true;
		cin >> N >> M;
		while(M)
		{
			ll p = pow2(M);
			if (first)
			{
				first = false;
				cout << "(" << N << "<<" << p << ")";
			}
			else
				cout << " + (" << N << "<<" << p << ")";
		}
		cout << endl;
	}
	return 0;
}
