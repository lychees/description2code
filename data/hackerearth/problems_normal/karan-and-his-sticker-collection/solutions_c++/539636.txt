#include <iostream>
#include<cstdio>
using namespace std;

#define getcx getchar_unlocked
void inp( unsigned long long int &n )//fast input function
{
   n=0;
   int ch=getcx();
   while(  ch >= '0' && ch <= '9' )
   {
           n = (n<<3)+(n<<1) + ch-'0'; ch=getcx();
   }
}


int main()
{
    unsigned long long int k,t;
    inp(t);
    int cnt=0;
    while(t--)
    {
     inp(k);
     if(k < 300000000)
     {
     	cnt++;
     }
    }
    cout << cnt;
    return 0;
}
