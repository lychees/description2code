#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
unordered_map<ll,ll> mp;


ll modpow(ll base,ll pwr)
{
    if(pwr==0)
        return 1;
    else if(pwr&1)
        return (base*(modpow(base,pwr-1)))%mod;
    else
    {

        ll k = modpow(base,pwr/2);
        return (k*k)%mod;
    }

}
ll solve(ll n)
{
    if(n==1||n==2)
        return 1;
    if(mp[n])
        return mp[n];
    else
    {
        ll temp=n;
        n--;
        ll sum=1;

        for(ll i=2;i*i<=n;++i)
        {
            if(n%i==0)
            {
                if(i*i==n)
                    sum=(sum+modpow(solve(n/i),i))%mod;
                else
                sum=(sum+modpow(solve(n/i),i)+modpow(solve(i),n/i))%mod;
            }
        }
        return mp[temp]=sum;
    }
}
int main()
{
    ll n;
    cin>>n;

    cout<<solve(n);

}
