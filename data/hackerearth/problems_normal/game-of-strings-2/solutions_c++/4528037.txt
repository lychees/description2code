#define maxVal 100000000

using namespace std;

int n;
char a[maxN+5];
int lps[maxN+5];
map<int,int> m;

void compute()
{
	int i,j;
	lps[0]=0;
	for(i=1;i<n;i++)
	{
		j=lps[i-1];
		while(j>0&&a[i]!=a[j])
			j=lps[j-1];
		if(a[i]==a[j])
			j++;
		lps[i]=j;
	}
}

int main()
{
    int t,i,k,j,l;
    scanf("%d",&t);
    while(t--)
    {
    	m.clear();
    	scanf("%s%d",a,&k);
    	n=strlen(a);
    	compute();
    	for(i=(n-1);i>0;i=(lps[i]-1))
    		m[lps[i]]=1;
    	l=0;
    	for(i=1;i<k;i++)
    		if(lps[i]>l&&m[lps[i]])
    			l=lps[i];
    	if(l>0)
    	{
    		for(i=0;i<l;i++)
    			printf("%c",a[i]);
    		printf("\n");
    	}
    	else
    		printf("Puchi is a cheat!\n");
    }
    return 0;
}
