#include <stdio.h>
#define MAXN 1005

char grid[MAXN][MAXN];
int row[MAXN];
int col[MAXN];

int main(void) {
    int t;
    int n;

    scanf(" %d", &t);
    while(t--) {
        int sum = 0;
        scanf(" %d", &n);
        for (int i = 0; i < n; i++) {
            scanf(" %s", grid[i]);
            for (int j = 0; j < n; j++) {
                sum += grid[i][j] == '#';
            }
        }

        for (int i = 0; i < n; i++) {
            if (i == 0) {
                row[i] = 0;
            } else {
                row[i] = row[i - 1];
            }
            for (int j = 0; j < n; j++) {
                row[i] += grid[i][j] == '#';
            }
        }
        for (int j = 0; j < n; j++) {
            if (j == 0) {
                col[j] = 0;
            } else {
                col[j] = col[j - 1];
            }
            for (int i = 0; i < n; i++) {
                col[j] += grid[i][j] == '#';
            }
        }
        int ok = 0;
        for (int i = 0; i < n; i++) {
            if (row[i] == sum - row[i] || col[i] == sum - col[i]) {
                ok = 1;
                break;
            }
        }
        if (ok) {
            printf("YES\n");
        } else {
            printf("NO\n");
        }
    }
    return 0;
}
