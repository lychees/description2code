map = {}
#details = []
nTestCases = int(raw_input())
for i in range(0,(2**nTestCases)-1):
	matches = raw_input()
	#details.append(matches)
	racer1,racer2 = ''.join(matches.split(" ")[0]),''.join(matches.split(" ")[1])
	if racer1 in map:
		if map[racer1] == "false":
			map[racer1],map[racer2] = "false","false"
		else:
			map[racer1],map[racer2] = "true","false"
	else:
		map[racer1],map[racer2] = "true","false"

for k in map.keys():
    if map[k] == "true":
