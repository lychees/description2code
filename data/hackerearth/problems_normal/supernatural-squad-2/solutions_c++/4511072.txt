#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

long long int dp[205][205];
int k;

long long int solve(int n,int m){
	if(n == 0) return 1;
    if(m == 0) return 0;
    if(n < 0) return 0;
    if(n < k || m < k) return 0;
    if(dp[n][m] == -1)
    {
        dp[n][m] = solve(n, m - 1) + solve(n - m, m);
    }
    return dp[n][m];
}

int main()
{
	int t;
	cin>>t;
	
	
	
		while(t--){
				int i,j;
		for(i=0;i<205;i++)
			for(j=0;j<205;j++)
				dp[i][j] = -1;
			
			int n;
			cin>>n>>k;
			long long int ans = solve(n,n);
			cout<<ans<<endl;
		}	
	
	
    return 0;
}
