#include<iostream>

using namespace std;
int main(int argc, char const *argv[])
{
	int long long n=0,m=0,sum=2,sum2=0,last=0,second_last=0,current=0;
	cin>>n>>m;
	for(int i=0;i<n;i++){
		if(i==0){
			second_last = 1;
        	current =1;
		}else if(i==1){
			current =1;
        	last = 1;
		}else{
        current = last + second_last;
        second_last = last;
        last = current;
        sum =sum + last;
		}
	}

	for(int i =0;i<m;i++){
		int long long k;
		cin>>k;
		sum2 = sum2 + k;
	}

	cout<<(sum-sum2);


	/* code */
	return 0;
}
