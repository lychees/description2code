#include<bits/stdc++.h>
using namespace std;

int main()
{
int m,dia1=0,dia2=0,diff;
cin>>m;
int arr[m][m];
for(int i=0;i<m;i++)
{
	for(int j=0;j<m;j++)
	{
		cin>>arr[i][j];
		if(i==j)
			dia1=dia1+arr[i][j];
	}

}
//cout<<dia1<<endl;
for(int i=0,j=m-1;i<m,j>=0;i++,j--)
	dia2=dia2+arr[i][j];

diff=abs(dia1-dia2);
cout<<diff<<endl;
return 0;
}
