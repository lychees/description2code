#include<bits/stdc++.h>
using namespace std;

int main() {
	int N,tmp,sum=0,Q,X;
	scanf("%d",&N);
	vector<int>A;
	for(int i=0;i<N;++i) {
		scanf("%d",&tmp);
		sum+=tmp;
		A.push_back(sum);
	}
	scanf("%d",&Q);
	while(Q--) {
		scanf("%d",&X);
		vector<int>::iterator it=upper_bound(A.begin(),A.end(),X);
		if(*(it-1)==X && it-A.begin()<=N) printf("%ld\n",it-A.begin());
		else if(it-A.begin()+1<=N) printf("%ld\n",it-A.begin()+1);
		else printf("%d\n",-1);
	}
	return 0;
}
