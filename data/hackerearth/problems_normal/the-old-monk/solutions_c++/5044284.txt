#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int t,n,j;
    cin>>t;
    while(t--){
        cin>>n;
         long long a[n],b[n],maxm;
         for(int i=0;i<n;i++){
                cin>>a[i];
         }
          for(int i=0;i<n;i++){
                  
          	cin>>b[i];
          }
         maxm=0;
         for(int i=0;i<n;i++){
           j=i;
            while(j>=i && b[j]>=a[i]&&j<n){
                if((j-i)>maxm){
                    maxm = j-i;
                   // cout<<"hee"<<maxm<<endl;
                }
                j++;
            }
         }
        cout<<maxm<<endl;
    }
    return 0;
}
