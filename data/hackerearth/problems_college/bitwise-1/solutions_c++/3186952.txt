#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t, x, k, i, j, ans, c, ca;
    
    cin>>t;
    ca = 0;
    while(t--) {
        cin>>x>>k;
        
        c = 0;
        
        for (i = 1; i < 1000005; i++) {
            j = x + i;
            ans = x^i;
            if (j == ans) {
                c++;
            } 
            if (c == k) {
                ans = i;
                break;
            }
        }
        cout<<"Case #"<<++ca<<":"<<" "<<ans<<endl;
    }
    
    return 0;
}
