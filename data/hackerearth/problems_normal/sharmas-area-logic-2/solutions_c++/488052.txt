#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int T;
    cin >> T;
    while(T--){
    	int N;
    	cin >> N;
    	int Xmin = 500, Xmax = -500, Ymin = 500, Ymax = -500;
    	while(N--){
 			int X,Y;
 			cin >> X >> Y;
    		if(X < Xmin)
    			Xmin = X;
    		if(Y < Ymin)
    			Ymin = Y;
    		if(X > Xmax)
    			Xmax = X;
    		if(Y > Ymax)
    			Ymax = Y;
    	}
    	int area = (Xmax-Xmin)*(Ymax-Ymin);
    	if(area==0)
    		cout << abs((Xmax-Xmin)+(Ymax-Ymin))<<endl;
        else
        	cout << abs(area)<<endl; 
    }
    return 0;
}
