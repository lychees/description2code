#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    while(n--)
    {
    	int t;
    	cin>>t;
    	vector<int>v;
    	v.clear();
    	for(int i=0;i<t;i++)
    	{
    		int temp;
    		cin>>temp;
    		v.push_back(temp);
    	}
    	sort(v.begin(),v.end());
    	reverse(v.begin(),v.end());
    	for(int i=0;i<t;i++)
    	{
    		
    		cout<<v[i]<<" ";
    	}
    	cout<<endl;
    }
    return 0;
}
