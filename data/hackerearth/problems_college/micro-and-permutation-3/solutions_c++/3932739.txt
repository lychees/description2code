#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

bool ASC(long long a,long long b){
	 string i=to_string(a);
	 string j =to_string(b);
	 string ij = i+j;
	 string ji =j+i;
	 return ij.compare(ji) > 0 ? true: false;
}

int main()
{	
	int t,i;
	cin>>t;
	long long *a = new long long[t];
	
	for(i=0; i<t; i++)
		cin>>a[i];
	
	sort(a,a+t,ASC);
	
//	cout<<a[1];
	for(i=0; i<t; i++){
	 	cout<<a[i];
	}
	
    return 0;
}
