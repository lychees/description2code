#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
#define bitcount __builtin_popcountll
#define ll long long
int a[100005];
int main()
{
	int i,j,k;
	int arr[100000],x;
	for(i=2;i<=100000;i++)
    {
        if(a[i]==0)
        {
           for(j=i;j<=100000;j+=i)
           {
               k=j;
               while(k%i==0)
               {
                   a[j]++;
                   k/=i;
               }
           }
        }

    }
    int n;
    sd(n);
    for(i=0;i<n;i++)
    {
        sd(x);
        arr[i]=a[x];
    }
    int ans=0;
    for(i=0;i<n;i++)
        ans=ans^arr[i];
    //printf("%d\n",ans);
    if(ans==0)
    {
        printf("BOB\n");
    }
    else
    {
        printf("ALICE\n");
    }
	return 0;
}
