#include <iostream>
using namespace std;

int getSetBits(int number) {
	int counter = 0;
	while(number) {
		number &= (number - 1);
		++counter;
	}
	
	return counter;
}
int main()
{
	int testCases = 0;
	cin >> testCases;
	long long int N;
	while(testCases--) {
		cin >> N;
		cout << getSetBits(N) << "\n";
	}
    return 0;
}
