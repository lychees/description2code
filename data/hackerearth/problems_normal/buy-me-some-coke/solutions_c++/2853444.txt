#include<stdio.h>
#include<iostream>

using namespace std;

int tot;
int dp[155][155][55];
int vis[155][155][55];

int timer=1,c;

int find(int coke,int n5,int n10)
{
    if(coke==c)
        return 0;
    if(vis[coke][n5][n10]==timer)
        return dp[coke][n5][n10];
    vis[coke][n5][n10]=timer;
    dp[coke][n5][n10]=1e9;
    int n1=tot-coke*8;
    n1=n1-5*n5-10*n10;
    if(n1>=8)
        dp[coke][n5][n10]=min(dp[coke][n5][n10],find(coke+1,n5,n10)+8);
    if(n1>=3 && n5>=1)
        dp[coke][n5][n10]=min(dp[coke][n5][n10],find(coke+1,n5-1,n10)+4);
    if(n5>=2)
        dp[coke][n5][n10]=min(dp[coke][n5][n10],find(coke+1,n5-2,n10)+2);
    if(n10>=1)
        dp[coke][n5][n10]=min(dp[coke][n5][n10],find(coke+1,n5,n10-1)+1);
    if(n10>=1 && n1>=3)
        dp[coke][n5][n10]=min(dp[coke][n5][n10],find(coke+1,n5+1,n10-1)+4);
    return dp[coke][n5][n10];
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n1,n5,n10;
        cin>>c>>n1>>n5>>n10;
        tot=n1+n5*5+n10*10;
        if(c*8>tot)
        {
            cout<<-1<<endl;
            timer++;
            continue;
        }
        cout<<find(0,n5,n10)<<endl;
        timer++;
    }
    return 0;
}
