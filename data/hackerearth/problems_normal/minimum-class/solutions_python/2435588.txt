def f(n):
	if n % 2 == 0:
		return (n*n+2*n)/4
	else:
		return (n*n+2*n+1)/4

T = int(raw_input())
for _ in xrange(T):
	N = int(raw_input())
	lo = 1
	hi = N
	while lo < hi:
		mid = (lo + hi) / 2
		if f(mid) >= N:
			hi = mid
		else:
			lo = mid + 1
