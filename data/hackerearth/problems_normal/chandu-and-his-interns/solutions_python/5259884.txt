n = int(raw_input())


limit = 10**7
#limit = 1000
l =[1] * (limit+5)
l[0]=0
l[1]=1
for i in range(2,(limit) +1):
    if l[i] ==1 :
        for j in range((i*i), (limit) , i ):
            l[j] = 0
        if i*i <= limit:
            l[i*i] = 2

for i in range(n):
    x = int(raw_input())
    if l[x] == 1 or l[x] == 2:
        print 'NO'
    else:
        print 'YES'
