#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <set>
using namespace std;
set<int, greater<int>> s;
int n;
int main() {
  scanf("%d", &n);
  for (int i = 1, x; i <= n; i++) {
    scanf("%d", &x);
    while (s.find(x) != s.end()) {
      s.erase(x);
      x++;
    }
    s.insert(x);
  }
  printf("%d\n", *s.begin() - s.size() + 1);
  return 0;
}
