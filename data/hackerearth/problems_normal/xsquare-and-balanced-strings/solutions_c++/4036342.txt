#include<bits/stdc++.h>
using namespace std;
#define ll long long int
char str[1000100];
ll freq[30];
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		memset(freq,0,sizeof(freq));
		scanf(" %s",str);
		ll len=strlen(str);
		ll i,z=0,cnt=0;
		for(i=0; i<len ; i++)
		{
			freq[str[i]-97]++;
			if(freq[str[i]-97]==1)
				z++;
		}
		for(i=0; i<26; i++)
		{
			if(freq[i]%2==0 && freq[i]!=0)
			{
				cnt++;
			}
		}
		if(cnt==z)
		{
			printf("1\n");
		}
		else
		{
			printf("-1\n");
		}
	}
	return 0;
}
