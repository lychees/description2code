def solve(a):
    if a%12 == 0:
        return a/12
    x = a/12
    ra = range(x+1)
    for i in ra:
            h = a-12*(x-i)
            if h%10 == 0:
                return (x-i)+h/10
    return -1


for i in range(int(raw_input())):
	print solve(int(raw_input()))
