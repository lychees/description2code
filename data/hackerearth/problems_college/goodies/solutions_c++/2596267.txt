#include <iostream> 
#include <algorithm> 
using namespace std;

int main()
{
  int tc,i,top,sum;
  long n;
  cin>>tc;
  while(tc--)
  {	cin>>n;
  	sum=top=i=0;
   	long stack[n];
  	for(int i=0;i<n;i++)	cin>>stack[i];
  	//sorting array in ascending order, hence stack in descending order
  	std::sort(stack,stack+n);
  	top=n-1;
   	while(top>=0){
  		if(stack[top]==-1)	top--;
  		else
  		{	i=top-1;
  			while(i>=0){
  				if(stack[i]<stack[top]&&stack[i]!=-1){
	  			stack[top]=stack[i];
	  			stack[i]=-1;	//marking the ith element
	  			}
	  		i--;
  			}
  		top--;
  		}
  	}
  	for(int i=0;i<n;++i){
  		if(stack[i]!=-1){
  			sum++;				//counting the number of unmarked elements
  		}
  	}
  	cout<<sum<<endl;
  }
    return 0;
}
