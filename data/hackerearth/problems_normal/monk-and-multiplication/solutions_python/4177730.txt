import heapq

n=input()
a=map(int,raw_input().split())
minheap=[]
l=len(a)
for i in xrange(l):
    if len(minheap)<3:
        heapq.heappush(minheap,a[i])
    elif minheap[0]<a[i]:
        minheap[0]=a[i]
        heapq.heapify(minheap)

    if len(minheap)<3:
        print -1
    else:
        #print minheap[0],minheap[1],minheap[2]
        print minheap[0]*minheap[1]*minheap[2]
