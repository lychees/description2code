import sys
from fractions import gcd
T = input()
while T > 0:
    T -= 1
    N, X = [int(x) for x in raw_input().split()]
    arr = [int(x) for x in raw_input().split()]
    minus = True
    for k in arr:
        if gcd(X, k) == 1:
            minus = False
            sys.stdout.write(str(k) + " ")
    if minus:
        sys.stdout.write("-1")
    sys.stdout.write('\n')
