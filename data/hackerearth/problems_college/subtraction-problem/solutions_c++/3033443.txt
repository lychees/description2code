#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;
int s[10],a,b,m,i,j,k;
char p;
int main(int argc, char *argv[])
{
    cin>>a>>b;
    m=a-b;
   // cout<<m<<endl;
    p='+';
    if(m<0)
    { p='-';
      m=-m;
    }
   if(m==0)
       cout<<1<<endl;
   else
   {  i=0;
       while(m!=0)
       { s[i++]=m%10;
         m=m/10;
       }
      
     
    k=0;
    for(j=0;j<i;j++)
       if(s[j]<9)
       { s[j]++;
         k=1;
         break;
         
       }
    
    if(p=='-')
      cout<<p;
    if(k==0)
    { 
        cout<<1;
      for(j=i-1;j>=0;j--)
         cout<<0;
    }  
    else
      for(j=i-1;j>=0;j--)
         cout<<s[j];
           
   }
    
    
    return EXIT_SUCCESS;
}
