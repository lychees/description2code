#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	int n;
    	int odd = 0, even = 0;
    	int x;
    	cin >> n;
    	for(int i = 0; i < n; i ++) {
    		cin >> x;
    		odd += x % 2 == 1;
    		even += x % 2 == 0;
    	}
    	cout << ((long long)odd * even) << endl;
    }
    return 0;
}
