#include <iostream>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	cin.tie(NULL);
    long long t;
    cin>>t;
    while(t--)
    {
    long long  a,b;
    cin>>a>>b;
    long long  res=a;
    for(long long  i=a+1;i<=b;i++)
    {
    	res=res&i;
    }
    cout << res <<endl;
}
}
