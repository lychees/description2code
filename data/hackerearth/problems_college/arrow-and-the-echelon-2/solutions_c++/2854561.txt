#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while (t!=0){
        int n,i;
        cin>>n;
        int rem = n%3;
        if((n==1)||(n==2)){cout<<"-1"<<endl;}
        else{
            if(rem==0){
                for(i=0;i<n;i++){cout<<"5";}
                cout<<endl;
            }
            if(rem==2){
                for(i=0;i<n-5;i++){cout<<"5";}
                for(i=0;i<5;i++){cout<<"3";}
                cout<<endl;
            }
            if(rem==1){cout<<"-1"<<endl;}
        }
        t--;
    }
    return 0;
}
