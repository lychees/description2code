#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>
#include <stack>
using namespace std;


bool compare (pair <int, int> a, pair <int, int> b){
	
	if(a.second != b.second){
		
		return a.second > b.second;
	}
	else {
		return a.first > b.first;
	}
}
int main()
{
    long int r;
    int n;
    int s, q;
    int g;
    int m;
    
    vector < pair<int, int> > v;
    cin>>n;
    while(n--){
    	
    	cin>>s>>q;
    	v.push_back(make_pair(s, q));
    }
    sort ( v.begin(), v.end(), compare );
    cin>>g;
    
    r = 0;
    while(g--){
    	
    	cin>>m;
    	
    	int i=0;
    	while( i < v.size() && v[i].first < m ){
    		i++;
    	}
    	
    	if(i != v.size()){
    		r += (v[i].second) / m;
    	}
    }
    
    cout<<r*100;
    return 0;
}
