#include <bits/stdc++.h>
using namespace std;

const int sz = 123456;

int main(void) {
    
    int T = 0;
    scanf("%d", &T);
    while(T--) {
    	int N = 0;
    	scanf("%d", &N);
    	vector<int> share(N);
    	for(int i = 0; i < N; ++i)
    		scanf("%d", &share[i]);
    	
    	int current = 0;
    	long long int res = 0;
    	for(int i = N - 1; i >= 0; --i) {
    		current = max(current, share[i]);
    		res += current - share[i];
    	}    	
    	printf("%lld\n", res);
    }
    
    return 0;
}
