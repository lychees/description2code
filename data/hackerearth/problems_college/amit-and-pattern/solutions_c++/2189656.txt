#include<iostream>
using namespace std;
int main()
{
	int m,n;
	cin>>m>>n;
	int a[m][n];
	for(int i=0;i<m;i++)
		for(int j=0;j<n;j++)
			cin>>a[i][j];
	int x=0,y;
	for(int i=0;i<n;i++)
	{
		x=i;
		y=0;
		while(x>0&&y<m-1)
		{
			cout<<a[y][x]<<" ";
			x--;
			y++;
		}
		cout<<a[y][x];
		cout<<endl;
	}
	for(int i=1;i<m;i++)
	{
		x=n-1;
		y=i;
		while(y<m-1&&x>0)
		{
			cout<<a[y][x]<<" ";
			x--;
			y++;
		}
		cout<<a[y][x];
		cout<<endl;
	}
	return 0;
}
