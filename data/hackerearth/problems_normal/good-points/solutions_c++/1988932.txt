#include <iostream>
using namespace std;

int main()
{
  	int n;
  	cin>>n;
  	if(n==4)
  	{
  		cout<<"YES\n";
  		cout<<"0 0\n0 1\n1 1\n1 0";
  	}
  	else if(n==1)
  	{
  		cout<<"YES\n";
  		cout<<"0 0";
  	}
  	else if(n==2)
  	{
  		cout<<"YES\n";
  		cout<<"0 0\n1 0";
  	}
  	else
  		cout<<"NO\n";
    return 0;
}
