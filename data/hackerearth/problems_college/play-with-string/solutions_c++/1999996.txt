#include <bits/stdc++.h>
using namespace std;
int main()
{
   	long long int t;
   	cin>>t;
   	string s1,s2;
   	while(t--)
   	{
   		int flag=0;
   		cin>>s2;
   		s1.assign(s2);
   		int len=s1.length();
   		reverse(s2.begin(),s2.end());
   		for(int i=1;i<len;i++)
   		{
   			if(abs(s1[i]-s1[i-1])!=abs(s2[i]-s2[i-1]))
   			{
   				flag=1;
   				break;
   			}
   		}
   		if(flag==0)
   		{
   			cout<<"Funny"<<endl;
   		}
   		else
   		{
   			cout<<"Not Funny"<<endl;
   		}
   	}
   	return 0;
}
