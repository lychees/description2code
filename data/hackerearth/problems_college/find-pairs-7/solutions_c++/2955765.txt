#include<bits/stdc++.h>

using namespace std;

int a[2000010];
int main()
{
int t,n,i,x;
scanf("%d",&t);
long long int ans=0;
while(t--)
{
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&x);
a[x+1000000]++;
}
for(i=0;i<2000010;i++)
{
if(a[i]>0)
ans=ans+a[i]+(a[i]*(a[i]-1))/2;
}
printf("%lld\n",ans);
ans=0;
for(i=0;i<2000010;i++)
a[i]=0;
}
return 0;
}
