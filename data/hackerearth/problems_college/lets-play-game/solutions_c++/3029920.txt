#include<iostream>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
			long long int n,x,i,y,k;
			cin>>n>>k;
				long long int a[n];
			for(i=0;i<n;i++){
				cin>>x;
				a[i]=k/x;
			}
				for(i=0;i<n;i++){
				cin>>y;
				a[i]=a[i]*y;
			}
			sort(a,a+n);
			cout<<a[n-1]<<"\n";
			
	}
	return 0;
}
