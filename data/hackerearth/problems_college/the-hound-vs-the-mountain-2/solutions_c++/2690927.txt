#include <iostream>
#include <math.h>
using namespace std;

int binary_decimal(int n) /* Function to convert binary to decimal.*/
{
    int decimal=0, i=0, rem;
    while (n!=0)
    {
        rem = n%10;
        n/=10;
        decimal += rem*pow(2,i);
        ++i;
    }
    return decimal;
}

int po(int a,int b){
    int prod=1;
    for(int i=1;i<=b;i++)
    prod*=a;
    return a;
}
