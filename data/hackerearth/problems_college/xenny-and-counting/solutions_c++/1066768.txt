#include <cmath>
#include <cstdio>
#include <map>
#include <vector>
#include <cstring>
using namespace std;

typedef long long ll;
typedef vector<int> vi;
typedef map<int, int> mii;

int main()
{
	int t, n, k;
	long long ans, div;
	
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d%d", &n, &k);
		ans = div = 1LL;
		for(int i=k+1; i<=k+n-1; i++)
		{
			ans *= (long long)i;
			while(ans%div == 0 && div<n)
			{
				ans = ans/div;
				div++;
			}
		}
		printf("%lld\n", ans);
	}
	
	return 0;
}
