#include <vector>
#include<algorithm>
#include <cstdio>
#define F first
#define S second
using namespace std;

int main()
{
	int t,n,i,md;
	typedef pair<int,int> pii;
	vector<pii> v(10001);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=1;i<=n;i++)
		{
			scanf("%d",&md);
			v[i]=pii(md,i);
		}
		sort(v.begin()+1,v.begin()+n+1);
		md=abs(1-v[1].S);
		for(i=2;i<=n;i++)
			if(md<abs(i-v[i].S)) md=abs(i-v[i].S);
		printf("%d\n",md);
	}
	return 0;
}
