#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,tt,a=1,b=1,ctr=0,aa,bb;
        cin>>n>>m>>tt;
        while(tt--)
        {
            cin>>aa>>bb;
            while(a+aa<=n&&b+bb<=m&&a+aa>0&&b+bb>0)
            {
                //cout<<a<<" "<<b<<endl;
                ctr++;
                a=a+aa;b=b+bb;
            }
        }
        cout<<ctr<<endl;
    }
}
