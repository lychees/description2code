#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,i,j,count,sum,a[150],b[10000];
    cin>>t;
    
    while(t--)
    {
        cin>>n;
        count=1;
        sum=0;
        for(j=0;j<10000;j++)b[j]=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            b[i]=0;
            sum=sum+a[i];
        }
        b[0]=1;
        for(i=0;i<n;i++)
        {
            for(j=sum;j>=a[i];j--)
            {
                if(b[j-a[i]]==1 && b[j]!=1)
                {
                    count++;
                    b[j]=1;
                }
            }
            
        }
        
        cout<<count<<"\n";
    }
    return(0);
}
