#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int main(){
    int t;
    cin>>t;
    unordered_map<string,int>mp;
    int grp=0;
    while(t--){
        string a;
        cin>>a;
        int n = a.length();
        string tmp =a;
        sort(tmp.begin()+1, tmp.end()-1);
        if(mp.find(tmp)==mp.end()){
            mp[tmp]=1;
            grp+=1;
        }
        else mp[tmp]+=1;
    }
    cout<<grp<<endl;
    return 0;

}
