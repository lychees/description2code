n = int(raw_input())
a = [int(i) for i in raw_input().split()]

a.sort()

front = []
back = []

i = 0
j = n-1

small = []
big = []

while i < j :
	small.append(a[i])
	small.append(a[j])
	big.append(a[j])
	big.append(a[i])

	i+=1
	j-=1

	if i == j :
		small.append(a[i])
		big.append(a[i])
		break
	if i > j :
		break

x = 0
y = 0

for i in range(1,n) :
	x += abs(small[i] - small[i-1])
	y += abs(big[i] - big[i-1])

print max(x,y)
