#include <limits.h>
#define mod 1000000007
using namespace std;
long long anslast[5009];
long long ans[5009];

long long sumlast[5009];
long long sum[5009];


int main()
{
    int n,j,a,b;
    int temp,curr;
    scanf("%d %d %d %d",&n,&a,&b,&j);
    a--;b--;
    
    
    
   for(int i=0;i<n;i++){
           if(i==b)
             anslast[i]=0;
           else 
             anslast[i]=1;
           
           if(i==0)
              sumlast[i]=anslast[i];
           else 
              sumlast[i]=anslast[i]+sumlast[i-1];
   }
   
   for(int k=1;k<=j;k++){
        for(int i=0;i<n;i++){
                if(i==b){
                    ans[i]=0;
                    continue;
                }
                if(i==0){
                         ans[i]=(sumlast[b]-anslast[i])%mod;
                         while(ans[i]<0) ans[i]+=mod;
                         continue;
                }
                if(i==n-1){
                          ans[i]=(sumlast[i-1]-sumlast[b])%mod;
                          while(ans[i]<0) ans[i]+=mod;
                          continue;
                }
                if(i<b){
                       temp=b-i;
                       ans[i]=(sumlast[b]-(i-temp>=0?sumlast[i-temp]:0)-anslast[i])%mod;
                       while(ans[i]<0) ans[i]+=mod;
                       continue;
                }
                if(i>b){
                        temp=i-b;
                        temp--;
                        ans[i]=((i+temp<n?sumlast[i+temp]:sumlast[n-1])-sumlast[b]-anslast[i])%mod;
                        while(ans[i]<0) ans[i]+=mod;
                        continue;
                }
        }
        
        sum[0]=ans[0];
        for(int i=1;i<n;i++){
                sum[i]=(sum[i-1]+ans[i])%mod;
                while(sum[i]<0) sum[i]+=mod;
        }
        
        for(int i=0;i<n;i++){
           sumlast[i]=sum[i];
           anslast[i]=ans[i];
        }
   }
    printf("%lld\n",ans[a]);
   return 0;
}
