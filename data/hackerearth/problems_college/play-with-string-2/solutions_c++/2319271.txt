#include<bits/stdc++.h>
using namespace std;
int main(){
    int T,i,j,counter,len;
    string s;
    cin>>T;
    while(T--){
        cin>>s;
        len=s.size();
        counter=0;
        for(i=1,j=len-2;i<=len/2,j>=len/2;i++,j--){
            if(abs(s[i]-s[i-1])!=(abs(s[j]-s[j+1]))){
                counter=1;
                break;
            }
        }
        if(counter)
            cout<<"Not Funny\n";
        else
            cout<<"Funny\n";
    }
    return 0;
}
