#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll t,n;
    cin>>t;
    ll arr[t];
    for(int i=0;i<t;++i)
    	cin>>arr[i];
    sort(arr,arr+t);
    ll res=0;
    ll i=0,j=1;
    while(i<t){
    	if(j*j<arr[i]){
    		res+=(j*j);
    	}
    	else{
    		res+=((j*j-arr[i]));
    		i++;
    	}
    	j++;
    }
    cout<<res<<endl;
    return 0;
}
