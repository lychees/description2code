using namespace std;
ll t,m,k,a[100005],sum,x,y,prd,i;
ll pow(ll a, ll b, ll MOD)
{
	ll x = 1, y = a;
    while(b > 0) 
	{
        if(b%2 == 1) 
		{
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
 
ll modInverse(ll a, ll m) {
    return pow(a,m-2,m);
}
int main()
{
	sl(t);
	while(t--)
	{
		sl(k);
		sl(m);
		sum=0;
		rep(i,0,k)
		{
			sl(a[i]);
			sum+=a[i];
		}
		m-=sum;
		if(m<0)
		{
			cout<<"0"<<endl;
		}
		else
		{
		x=m+k-1;
		y=k-1;
		prd=1;
		for(i=x;i>y;i--)
		{
			prd*=(i);
			if(prd>e)
			prd%=e;
		}
		for(i=1;i<x-y+1;i++)
		{
			prd=(prd*modInverse(i,e))%e;
		}
		pl(prd);
	}
	}
	return 0;
}
