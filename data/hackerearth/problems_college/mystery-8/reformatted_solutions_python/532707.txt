N = input()

for _ in range(N):
    s = raw_input()
    slist = list(s)
    punc = '.,?! '
    for item in punc:
        while item in slist:
            slist.remove(item)
    srev = slist[::-1]
    for i in range(len(slist)):
        if srev[i] == '*':
            print slist[i]
            break
