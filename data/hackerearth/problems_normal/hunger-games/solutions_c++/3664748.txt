#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,i,num,maxi=0,a[1002],b[3];
    cin>>n;
    for(i=0;i<n;i++)
    {   
    	cin>>a[i];
    }
   sort(a,a+n);
   b[0]=a[0];
   b[1]=a[1];
   b[2]=a[2];
   maxi=b[2]-b[0];
    for(i=3;i<n;i++)
    {
    	int temp=b[1];
    	b[1]=a[i];
    	if(b[0]<b[2])
    	{
    		b[0]=temp;
    	}
    	else
    	b[2]=temp;
    	
    	maxi=max(maxi,max(b[1]-b[0],b[1]-b[2]));
    }
    cout<<maxi;
    return 0;
}
