#include <iostream>
using namespace std;

int main()
{
    int t;
    int n,i;
    long int c[100004]={0};
    c[0]=1;c[1]=1;c[2]=2;
    for(i=3;i<=100000;i++)
    {
    	c[i]=(c[i-1]+c[i-2]+c[i-3])%1000000007;
    }
    cin >> t;
    while(t--)
   	{
   		cin >> n;
   		cout << c[n] << '\n';
   	}
    return 0;
}
