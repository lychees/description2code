def is_prime(x):
    if x == 1:
        return False
    elif x == 2:
        return True
    else:
        div = int(x**0.5)+1
        for i in range(2, div):
            if x % i == 0:
                return False
        return True


t = int(raw_input())
for i in range(t):
    n, m = map(int, raw_input().split(' '))
    s = 0
    m = m+1
    for i in range(n, m):
        if is_prime(i):
            s = s+i
    print s
