#include<iostream>
#include<cmath>
using namespace std;
long long int money(int n,int k,long long int m)
{
     unsigned long long int cost=0;
     int i=n*(n+1)/2;

     cost=k*pow(i,2);
     if(m>=cost)
     return 0;
     else
     return (cost-m);
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int N,K;
        long long int M;
    cin>>N>>K>>M;
    long long int Ans=money(N,K,M);
    cout<<Ans<<endl;
    }

    return 0;
}
