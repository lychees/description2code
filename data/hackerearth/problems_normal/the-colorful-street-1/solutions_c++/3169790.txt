#include<bits/stdc++.h>
using namespace std;;
const int MAXN = 25;
long long dp[3][MAXN];
int main() {
	int t;
	cin>>t;
	while(t--) {
		int n;
		dp[0][0]=dp[1][0]=dp[2][0]=0;
		scanf("%d",&n);
		for(int i=1;i<=n;i++) {
			int a,b,c;
			scanf("%d%d%d",&a,&b,&c);
			dp[0][i]=a+dp[1][i-1];
			dp[0][i]=min(dp[0][i],a+dp[2][i-1]);
			dp[1][i]=b+dp[0][i-1];
			dp[1][i]=min(dp[1][i],b+dp[2][i-1]);
			dp[2][i]=c+dp[0][i-1];
			dp[2][i]=min(dp[2][i],c+dp[1][i-1]);
		}
		printf("%lld\n",min(min(dp[0][n],dp[1][n]),dp[2][n]));
	}
	return 0;
}
