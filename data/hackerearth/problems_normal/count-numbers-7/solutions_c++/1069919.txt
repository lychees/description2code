#include<iostream>
using namespace std;
int a[11];
int main()
{
    int T,K,A,B;
    cin>>K>>T;
    for(int i=0;i<K;i++)
    {
        cin>>a[i];
    }
    while(T--)
    {
        cin>>A>>B;
        int ans=0;
        for(int i=1;i<(1<<K);i++)
        {
        	
        	long long int diwide = 1;
        	int count=0;
        	for(int j=0;j<K;j++)
        	{
        		if(diwide>B)
        		{
        			break;
        		}
        		if((i>>j) & 1)
        		{
        			diwide*=a[j];
        			++count;
        		}
        	}
        	if(diwide>B)
        	{
        		continue;
        	}
        	count=count&1;
        	if(count)
        	{
        		ans += (B/diwide - (A-1)/diwide);
        	}
            else
            {
            	ans -= (B/diwide - (A-1)/diwide);
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
