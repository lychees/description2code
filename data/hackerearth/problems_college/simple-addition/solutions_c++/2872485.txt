    #include<bits/stdc++.h>
    #define sc(n) scanf("%d",&n)
    #define pr(n) printf("%d\n",n)
    #define N 2147483647
    using namespace std;
    int cal(int a,int b)
    {
        int count=0;
        while(b>0)
        {
            int tmp=a/b;
            count+=tmp;
            tmp=a-tmp*b;
            a=b;
            b=tmp;
        }
        if(a==1)
            return count-1;
        else
            return N;
    }
    int main()
    {
        int t,l,ans;
        sc(t);
        while(t--)
        {
            sc(l);
            ans=N;
            if(l==1)
            {
                pr(0);
                continue;
            }
            for(int i=1;i<l;i++)
                ans=min(ans,cal(l,i));
            pr(ans);
        }
        return 0;
    }
