#include <iostream>
#include <algorithm>
using namespace std;

int n, a[10004], t,flag,b[10004];

int main()
{
	
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{cin>>a[i];b[i]=a[i];}
    	flag  =0 ;
    	
    	sort(b,b+n);
    	
    	int i=0,j,k;
    	
    	for(;i<n;i++)
    	if(a[i]!=b[i]){flag =1; break;}
    	
    	
    	for(j=n-1; j>-1;j--)
    	if(a[j]!=b[j]){break;}
    	
    	//cout<<i<<" "<<j<<endl;
    	for(k = i+1; k<j+1; k++)
    	if(a[k]>a[k-1]){flag  =0 ; break;}
    	
    	//if(i+1  == j)flag =0;
    	if(flag == 1)cout<<i+1<<" "<<j+1<<endl;
    	else cout<<-1<<" "<<-1<<endl;
    	
    }
    return 0;
}
