#include<iostream>
using namespace std;

int reverseNo(int);

   int  main( ) {
       int t=0;
        int a = 0, b = 0;
        cin>>t;
       
       
       
        while(t>0){
        
        
       cin>>a>>b;
       
       
        cout<< reverseNo( reverseNo(a)+reverseNo(b) )<<"\n";
            
        
        t--;
        
        }   
        
          return 0;  
     
    }
      int reverseNo(int a){
        int digit=0;
        int reverse=0;
        while(a>0){
            
            digit=a%10;
            reverse=reverse*10+digit;                   
           a=a/10;       
            
        }
        return reverse ;
    }
