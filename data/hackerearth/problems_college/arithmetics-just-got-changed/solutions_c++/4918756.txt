#include <iostream>
using namespace std;

long long int rev(long long int N)
{
	long long int r=0;
	while(N)
	{
		r=r*10+N%10;
		N/=10;
	}
	return r;
}

int main()
{
    long int T;
    long long int n1,n2;
    cin>>T;
    while(T--)
    {
    	cin>>n1>>n2;
    	cout<<rev(rev(n1)+rev(n2))<<endl;
    }
    return 0;
}
