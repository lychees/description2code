#include<bits/stdc++.h>
#define ft first
#define sd second
#define pb push_back

using namespace std;

typedef __int128 ll;

const int N=200000;
ll n,g0,a[N],pref[N],s,opt,wynik,tmp;

ll f(ll k){
  if(pref[k] >= g0 - (k+1)*(k+2)/2) return 0;
  ll minn = 0;
  ll mnoz =10000;
  ll maks = mnoz*mnoz*mnoz;
  ll sr;
  //cout<<(long long)k;
  while(minn+1<maks){
    //cout<<(long long)minn<<" "<<(long long)maks<<"\n";
    ll sr = (minn+maks)/2;
    if(sr * s + pref[k] >= g0 - (n*sr + k+1)*(n*sr + k + 2)/(__int128)2){
      maks = sr;
    }
    else minn = sr;
  }
  //cout<<(long long)maks<<"\n";
  return maks;
}

int main(){
  long long wczyt;
  cin>>wczyt;
  n = wczyt;
  cin>>wczyt;
  g0 = wczyt;;
  for(int i=0;i<n;i++){
    cin>>wczyt;
    a[i]  = wczyt;
    pref[i]+=a[i];
    pref[i+1]=pref[i];
    s+=a[i];
  }
  wynik = f(0);
  for(int i=0;i<n;i++){
    tmp = f(i);
    if(tmp<wynik){
      wynik = tmp;
      opt = i;
    }
  }
  cout<<(long long)(wynik*n+opt + 1);
}
