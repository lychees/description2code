#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    long t, n, i, brav, temp;
    bool strongest;
    vector <long> warrior, horse;
    cin >> t;
    while(t--){
    	warrior.clear();
    	horse.clear();
    	cin >> n;
    	for (i = 0; i < n; i++){
    		cin >> temp;
    		if (i == 0)
    			brav = temp;
    		else
    			warrior.push_back(temp);
    	}
    	for (i = 0; i < n; i++){
    		cin >> temp;
    		horse.push_back(temp);
    	}
    	sort(warrior.begin(), warrior.end());
    	sort(horse.rbegin(), horse.rend());
    	for (i = 1, strongest = true; i < n; i++){
    		if (warrior[i-1] * horse[i] >= brav * horse[0])
    			strongest = false;
    	}
    	if (strongest)
    		cout << "YES" << endl;
    	else
    		cout << "NO" << endl;
    }
    return 0;
}
