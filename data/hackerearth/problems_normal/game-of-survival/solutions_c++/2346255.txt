#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n ;
		cin>>n;
		int sharp[n],stren[n];
		for(int i=0;i<n;i++)
		{
			cin>>sharp[i];
		}
		for(int i=0;i<n;i++)
		{
			cin>>stren[i];
		}
		sort(sharp,sharp+n);
		sort(stren,stren+n);
		int x=0;
		bool flag = true;
		for(int i=0;i<n;i++)
		{
			if(sharp[i]<stren[i])
			{
				flag = false;
				break;
			}
		}
		if(flag == true)
		{
			cout<<"YES"<<endl;
		}
		else
		{
			cout<<"NO"<<endl;
		}
	}
    return 0;
}
