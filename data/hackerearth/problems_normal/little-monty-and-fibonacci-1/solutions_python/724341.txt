fib = [1,1]
[a,b] = [1,1]

count = {}
count[1] = 1
for x in xrange(1001):
	fib.append(a+b)
	count[a+b] = 0
	a = b
	b = fib[-1]

lookup = [1]
for i in xrange(1,1001):
	for j in xrange(1,i+1):
		if fib[i]%fib[j]==0:
			count[fib[j]] += 1
	lookup.append(sum([count[key] for key in fib[0:i+1]]))

T = int(raw_input())
for t in xrange(T):
	n = int(raw_input())
	print lookup[n-1]
