#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,i;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		int a,b,c,d,s;
		scanf("%d %d %d %d %d",&a,&b,&c,&d,&s);
		int l = c-s;
		if(l>=a&&l<=b)
		printf("%d\n",l);
		else if(l>b)
		printf("%d\n",b);
		else
		printf("STAY IN LONDON\n");
	}
    // cout << "Hello World!" << endl;
    return 0;
}
