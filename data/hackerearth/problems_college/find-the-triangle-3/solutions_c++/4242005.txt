#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t, n;
    cin >> t;
    int arr[100010];
    while(t--){
    	cin >> n;
    	for(int i = 0; i < n; i++){
    		cin >> arr[i];
    	}
    	sort(arr, arr+n);
    	int i = n-1;
    	bool flag = false;
    	while(i >= 2){
    		if(arr[i] < arr[i-1]+arr[i-2]){
    			flag = true;
    			break;
    		}
    		i--;
    	}
    	if (flag){
    		cout << arr[i-2] <<  " " << arr[i-1] << " " << arr[i] << endl;
    	}else{
    		cout << "-1\n";
    	}
    }
}
