#include <bits/stdc++.h>
using namespace std;
typedef long long lln;
#define MOD 100007
int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	lln t,n,m,num; cin>>t;
	while(t--){
		set<lln> myset;
		cin>>n>>m;
		for(int i=0;i<n;i++) cin>>num,myset.insert(num);
		while(m--){
			cin>>num;
			if(myset.find(num)!=myset.end()) cout<<"YES\n"; 
			else {cout<<"NO\n";myset.insert(num);}
		}
	}
	return 0;
}
