#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int t=0;
	cin>>t;
	while(t--)
	{
		int n=0;
		cin>>n;
		int W = 0;
		cin>>W;
		
		vector<int> values(n,0);
		vector<int> weights(n,0);
		for(int i=0;i<n;i++)
		{
			cin>>weights[i];
		}
		for(int i=0;i<n;i++)
		{
			cin>>values[i];
		}
		
		vector<vector<int>> dp(n+1,vector<int>(W+1,0));
		for(int i=0;i<=n;i++)
		{
			for(int j=0;j<=W;j++)
			{
				if(i==0 || j==0)
					dp[i][j] = 0;
				else if(weights[i-1] <= j)
				{
					dp[i][j] = max(values[i-1] + dp[i-1][j-weights[i-1]], dp[i-1][j]);
				}
				else
					dp[i][j] = dp[i-1][j];
			}
		}
		cout<<dp[n][W]<<endl;;
	}
    // cout << "Hello World!" << endl;
    return 0;
}
