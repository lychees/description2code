#include <cstdio>
#include <iostream>
using namespace std;
int min(int a, int b)
{
	return a<b?a:b;
	}
int main()
{
    int t,n;
    char arr[10005];
    scanf("%d",&t);
    int count1,count2;
    while(t--){
    	count1=0;
    	count2=0;
    	scanf("%d%s",&n,arr);
    	for(int i=0;i<n;i++)
    	{
    		if(i%2==0){
    		if(arr[i] != 'Y')
    			count1++;
    		}
    		else{
    			if(arr[i]!='N')
    				count1++;
    		}
    		if(i%2==0){
    		if(arr[i] != 'N')
    			count2++;
    		}
    		else{
    			if(arr[i]!='Y')
    				count2++;
    		}
    	}
    	printf("%d\n",min(count1,count2));
    }
    return 0;
}
