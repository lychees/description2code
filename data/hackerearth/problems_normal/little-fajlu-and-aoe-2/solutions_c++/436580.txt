#include<iostream>
#include<cstdio>
#include<vector>
#include<queue>
#include <algorithm>
#include<climits>
#include<cmath>

#include<sstream>
#include<fstream>

using namespace std;

int main()
{
    //Code starts here
    int t;
    cin>>t;

    while(t--)
    {
        int f,g,s,p,c,power1,power2,maxPower=0,fr=0,gr=0;
        scanf("%d %d %d %d %d %d",&f,&g,&s,&p,&c,&power1);

        for(int ns=0; ns*100<=f; ns++)
        {
            for(int np=0; np*125<=f && np*50<=g ; np++)
            {
                for(int nc=0; nc*50<=f && nc*100<=g; nc++)
                {
                    fr=ns*100+ np*125 + nc*50;
                    gr=ns*0+ np*50 + nc*100;

                    if(!(fr>f || gr> g))
                    {
                        power2=ns*s + np*p + nc*c;

                        if(power2>maxPower)
                            maxPower=power2;
                    }
                }
            }
        }

        int req=maxPower-power1;
        if(req<=0) req=-1;

        printf("%d\n",req);
    }
    //Code ends here
    return 0;
}
