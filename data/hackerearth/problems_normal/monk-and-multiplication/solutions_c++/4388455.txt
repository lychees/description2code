#include<cstdio>
#include<queue>
using namespace std;

int main()
{




int n;
scanf("%d",&n);
priority_queue<long long int> q;

for(long long int i=0;i<n;i++)

{ long long int ans=1;
long int ele;
scanf("%lld",&ele);
q.push(ele);

if(i<=1)
printf("-1\n");

else
{
long long int ans,x,y,z;
x=q.top();
q.pop();

y=q.top();
q.pop();

z=q.top();
q.pop();
ans=x*y*z;
q.push(x);
q.push(y);
q.push(z);

printf("%lld\n",ans);

}
}





return 0;
}
