#include <bits/stdc++.h>

using namespace std;

# define pi           3.14159265358979323846

int main()
{
    int t;
    cin >> t;

    while(t--) {

        double v;

        cin >> v;

        double r = cbrt(v/(2*pi));
        //cout << r <<endl;

        double A = 2*pi*r*r + (2*v)/r;

        printf("%0.9f\n", A);

        //cout << A <<endl;
    }

}
