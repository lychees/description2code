#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    string s,p;
    cin>>s>>p;
    int j=0;
    int m=p.length(),n=s.length();
    for(int i=0;i<n && j<m;i++){
    		if(p[j]==s[i]){
    			j++;
    		}
    }
    //cout<<j<<endl;
    
    if(j==p.length()){
    	cout<<"YES"<<endl;
    }
    else{
    	cout<<"NO"<<endl;
    }
    }
    return 0;
}
