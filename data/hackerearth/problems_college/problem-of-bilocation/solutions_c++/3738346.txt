using namespace std;

int main(){
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	int t;
	cin>>t;
	for(int i=0;i<t;i++){
		int a,b,c,d,s;
		cin>>a;
		cin>>b;
		cin>>c;
		cin>>d;
		cin>>s;

		int gap = c-s;
		if(gap<b && gap>=a){
			cout<<gap<<"\n";
		}
		else if(gap>=b){
			cout<<b<<"\n";
		}
		else{
			cout<<"STAY IN LONDON\n";
		}
	}

}
