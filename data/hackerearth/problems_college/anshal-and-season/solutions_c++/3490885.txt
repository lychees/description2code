#include<stdio.h>
#include<string.h>
int main(){
	char a[1000001], b[1000001]; int ai[26] = {0}, bi[26] = {0}, al, bl, i;
	scanf("%s%s", a, b); al = strlen(a), bl = strlen(b);
	for(i = 0; i < al; i++)	ai[a[i] - 65]++;
	for(i = 0; i < bl; i++)	bi[b[i] - 65]++;
	for(i = 0; i <= 25; i++){
		if(ai[i] != bi[i]){printf("WRONG\n");return 0;}
	}		
	printf("CORRECT\n");
	return 0;
}
