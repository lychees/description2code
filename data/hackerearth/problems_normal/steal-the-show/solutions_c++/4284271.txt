#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{ int a,b,c;
  
  vector <int> test;
  cin>>a>>b>>c;
  int n;
  cin>>n;
  int i;
  for(i=0;i<n;i++)
  { int val;
    cin>>val;
    test.push_back(val);
  }
  sort(test.begin(),test.end());
  vector <int> v(n+3);
  if(b<c)
  { v[0]=b;
    v[v.size()-1]=c;
    v[v.size()/2]=a;
  }
  else
  { v[0]=c;
    v[v.size()-1]=b;
    v[v.size()/2]=a;
  }
  int j=0;
  for(i=1;i<v.size()-1;i++)
  { if(i==v.size()/2)
       continue;
    v[i]=test[j];
    j++;
  }
  int sum=0;
  for(i=0;i<v.size()-1;i++)
  { int s=v[i]-v[i+1];
    if(s<0)
      s=s*-1;
    sum=sum+s;
  }
  cout<<sum<<"\n";
  return 0;
}
