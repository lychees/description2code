#include <bits/stdc++.h>
#define MOD 1000000007 
using namespace std;

typedef pair<int,int> PII;

vector< pair<int, PII> > edges;
int v[1501];
long long int prod;

int root(int a){
	if(v[a]!=a){
		v[a] = root(v[a]);
	}
	return v[a];
}

void apply_kruskal(){
	for(int i=0;i<edges.size();i++){
		int a,b;
		a = root(edges[i].second.first);
		b = root(edges[i].second.second);
		if(a==b)
			continue;
		v[a] = b;
		prod*=edges[i].first;
		prod = prod%MOD;
	}
}

int main(){
	int t;
	cin >> t;
	while(t--){
		prod = 1;
		edges.clear();
		int n,m;
		cin >> n >> m;
		for(int i=1;i<=n;i++){
			v[i]=i;
		}
		for(int i=0;i<m;i++){
			int a,b,c;
			cin >> a >> b >> c;
			edges.push_back(make_pair(c,make_pair(a,b)));
		}
		sort(edges.begin(),edges.end());
		apply_kruskal();
		cout << prod << endl;
	}
}
