#include <iostream>
using namespace std;

int main()
{
    long int n,i;
	cin>>n;
	long int a[n];
	long int sum[3]={0,0,0};
	int count=0;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
		sum[count]+=a[i];
		count++;
		if(count>=3)
			count=count%3;
	}
	cout<<sum[0]<<' '<<sum[1]<<' '<<sum[2];
	return 0;
    
}
