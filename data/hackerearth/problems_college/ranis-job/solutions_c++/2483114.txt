#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int arr[100000]={NULL};
int main()
{
int test;
scanf("%d",&test);
long int n,sum=0,k=0,count=0,i,s;
while(test--)
{
sum=0;k=0;count=0;
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%d",&s);
sum=sum+s;
}
for(i=0;i<n;i++)
scanf("%d",&arr[i]);
sort(arr,arr+n);
i=n-1;
while(k<sum)
{
k=k+arr[i--];
count++;
}
printf("%ld\n",count);
}
return 0;
}
