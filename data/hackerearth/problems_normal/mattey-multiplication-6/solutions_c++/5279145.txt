#include <iostream>
#include<algorithm>
#include<climits>
#include<cstring>

using namespace std;

#define llu long long int

int main()
{
	llu t;

	cin >> t;
	
	for(llu test_case=1 ; test_case<=t; test_case++ )
	 {
	 	 llu n , m;
		  
		 cin >> n >> m;
		 
		 int i;
		  
		 for(i=63; i>=0; i--)
		  {
		  	 if( (m&(1LL<<i)) > 0 )
		  	   {
		  	   	 cout << "(" << n << "<<" << i << ")" ;
		  	   	 i--;
		  	   	 break;
		  	   }
		  }
		  
		 for(    ; i>=0; i--)
		  {
		  	 if( (m&(1LL<<i)) > 0 )
		  	   {
		  	   	 cout << " + (" << n << "<<" << i << ")" ;
		  	   }
		  }
		  
		  cout <<"\n";
	 }
	 
	 return 0;
}
