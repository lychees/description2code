#include <bits/stdc++.h>
using namespace std;
int main()
{
    int te,n,i;
    int val[20];
    int nn,np,nz;
    vector<int> vn,vp;
    long long mi,ma;
    scanf("%d",&te);
    while(te--)
    {
        nn=0;
        np=0;
        nz=0;
        mi=1;
        ma=1;
        vn.clear();
        vp.clear();
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&val[i]);
            if(val[i]<0)
            {
                nn++;
                vn.push_back(val[i]);
            }
            else if(val[i]>0)
            {
                np++;
                vp.push_back(val[i]);
            }
            else
            {
                nz++;
            }
        }
        sort(vn.begin(),vn.end());
        sort(vp.begin(),vp.end());
        //cout<<np<<" "<<nz<<" "<<nn<<endl;
        if(nn>0)
        {
            if(nn%2==0)
            {
                for(i=0;i<nn-1;i++)
                    mi=mi*vn[i];
            }
            else if(nn%2==1)
            {
                for(i=0;i<nn;i++)
                    mi=mi*vn[i];
            }
            for(i=0;i<np;i++)
                mi=mi*vp[i];
        }
        else if(nn==0&&nz>0)
            mi=0;
        else
            mi=vp[0];

        if(nn>0)
        {
            if(nn%2==0)
            {
                for(i=0;i<nn;i++)
                    ma=ma*vn[i];
            }
            else if(nn%2==1)
            {
                for(i=0;i<nn-1;i++)
                    ma=ma*vn[i];
            }
            for(i=0;i<np;i++)
                ma=ma*vp[i];
        }
        else if(np>0)
        {
            for(i=0;i<np;i++)
                ma=ma*vp[i];
        }
        if((np==0)&&(nz>0)&&(nn<=1))
            ma=0;
        if((np==0)&&(nn==1)&&(nz==0))
            ma=vn[0];
        printf("%lld %lld\n",ma,mi);
    }
    return 0;
}
