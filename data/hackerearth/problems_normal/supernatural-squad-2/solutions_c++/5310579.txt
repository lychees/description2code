#include <bits/stdc++.h>
using namespace std;

long long dp[203][203];

long long rec (int n, int k)
{
	if(n<2*k)
		return !(n<k);

	long long &ret=dp[n][k];

	if(ret!=-1)
		return ret;

	ret=1;

	for(int i=k ; i<=n ; i++)
		ret+=rec(n-i,i);

	return ret;
}

int main()
{
	ios::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);

	int t, n, k;

	cin >> t;

	while (t--)
	{
		memset(dp,-1,sizeof (dp));
		cin >> n >> k;
		cout << rec(n,k) << '\n';
	}
	return 0;
}
