#include <iostream>
using namespace std;

int main()
{
    int a,m;
    scanf("%d",&a);
    bool n[100]={};
    scanf("%d",&m);
    while(m--)
    {
    	int x,y,z;
    	cin>>x>>y>>z;
    	if(x==0)
    	{
    		for(int i=y;i<=z ;i++)
    		n[i]=1;
    	}
    	else
    	{   int cnt=0;
    		for(int i=y;i<=z;i++)
    		{if(n[i]==1)cnt++;}
    		cout<<cnt<<"\n";
    	}
    }
    return 0;
}
