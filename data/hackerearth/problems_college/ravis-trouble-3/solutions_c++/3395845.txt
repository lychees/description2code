#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		char m[10001],n[10001];
		scanf("%s %s",m,n);
		int s=0;
		for(int i=0;n[s]!='\0'&&m[i]!='\0';i++){
			if(m[i]==n[s])
			s++;
		}
		if(n[s]=='\0')
		  printf("YES\n");
		  else
		  printf("NO\n");
	}
	return 0;
}
