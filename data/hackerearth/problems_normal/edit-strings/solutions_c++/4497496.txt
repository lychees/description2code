#include<bits/stdc++.h>
using namespace std;
int main()
{
	string a;
	char str;
	unsigned long long i,count,l,n,pos;
	cin>>l>>n;
	for(i=1; i<=l; i++)
	cin>>a[i];
	while(n--)
	{
		cin>>pos>>str;
		a[pos]=str;
		count=0;
		for(i=1; a[i]!='\0'; i++)
		{
			if(a[i]=='.' && a[i+1]=='.')
			count++;
		}
		cout<<count<<endl;
	}
}
