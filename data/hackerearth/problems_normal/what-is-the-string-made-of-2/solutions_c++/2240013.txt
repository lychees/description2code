#include <iostream>
using namespace std;

int main(){
    string message; //0-48 9-57
    cin >> message;
    int dashes[] = {6,2,5,5,4,5,6,3,7,6};
    int i=0;
    int count = 0;
    for (i=0; i<message.length(); i++) {
        count += dashes[message[i]-48];
    }
    cout << count << endl;
    return 0;
}
