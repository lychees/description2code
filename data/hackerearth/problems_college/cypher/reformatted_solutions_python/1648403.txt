def cypher(s):
    out = ''
    for c in range(len(s)):
        o = ord(s[c])
        if o+c <= 122:
            c = chr(o+c)
        else:
            x = abs((o+c)-122)
            c = chr(96+x)
        out += c
    print out


if __name__ == "__main__":
    t = int(raw_input())
    if t >= 1 and t <= 10000:
        while t > 0:
            s = raw_input()
            if len(s) >= 1 and len(s) <= 100:
                cypher(s)
            t -= 1
