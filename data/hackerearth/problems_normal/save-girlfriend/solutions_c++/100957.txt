#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <memory>
#include <functional>
#include <numeric>
#include <bitset>
#include <set>
#include <map>
#include <vector>
#include <iostream>
#define sz(a) (a.size())
#define FOR(i,lb,ub) for(i=lb;i<=ub;i++)
using namespace std;
typedef unsigned long long int uint64;
typedef long long int int64;
typedef long int in32;

int main(int argc,const char *argv[])
{
    int t,i,j,sv[10004];
     cin>>t;
     while (t--)
     {
     	int a,b;
     	cin>>a>>b;
     	int limit = b - a + 1;
     	FOR(i,0,limit)
     		sv[i] = 0;
     	for (i=1;i*i<=b;i++)
     	{
     		int sq = i*i;
     		for (j = ( (a-1) / i + 1) * i; j <= b; j += i)
     		{
     			if (j<sq)
     				continue;
     			else if (j==sq)
     			{
     				sv[j-a]++;
     			}
     			else sv[j-a] += 2;
     		}
     	}

     	int mx,ans=0;
     	FOR(i,a,b)
     	{
     		if (i==a)
     			mx = sv[i-a];
     		else if (mx<sv[i-a])
     			mx = sv[i-a];
     	}
     	FOR(i,a,b) if (sv[i-a]==mx) ans++;
     	cout<<ans+1<<"\n";
     }
return 0;
}
