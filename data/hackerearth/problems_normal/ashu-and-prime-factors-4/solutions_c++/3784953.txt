#include <iostream>
using namespace std;
typedef long long int ll;
ll a[1000001];
ll arr[1000001];
int main()
{
    ll i, j;
    ll max = 1000000;
    for(i = 2; i <= max; i++) {
    	if(a[i] == 0) {
    		for(j = i; j <= max; j+=i) {
    			if(a[j] == 0) {
    				a[j] = i;
    			}
    		}
    	}
    }
    for(i = 2; i <= max; i++) {
    	arr[a[i]]++;
    }
    ll t;
    cin >> t;
    while(t--) {
    		ll x;
    		cin >> x;
    		cout << arr[x] << endl;
    }
    return 0;
}
