#include<bits/stdc++.h>
#include <iostream>
#define ll long long int
#define MOD 1000000007
using namespace std;
ll A[51][51],B[51][51],C[51][51];
void matrix_multiplication(ll A[][51],ll B[][51],int n)
{
	ll i,j,k;
	for(i=0;i< n;i++)
	{
		for(j=0;j< n;j++)
		{
			C[i][j]=0;
			for(k=0;k<n;k++)
			{
				C[i][j]+=(A[i][k]*B[k][j]);
				if(C[i][j]>=MOD)
				{
					C[i][j]=C[i][j]%MOD;
				}
				
			}
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			A[i][j]=C[i][j];
		}
	}
}
int main()
{
    int t,n,u,v,i,j;
    ll k,path;
    cin >> t;
    while(t--)
    {
    	cin >> n >> k >> u >> v;
    	for(i=0;i < n; i++)
    	{
    		for(j=0; j < n; j++)
    		{
    			cin >> path;
    			A[i][j]=path;
    			B[i][j]=path;
    			
    		}
    	
    	}
    	k--;
    	while(k)
    	{
    		if(k%2!=0)
    		{
    			matrix_multiplication(A,B,n);
    		}
    		matrix_multiplication(B,B,n);
    		k=k/2;
    	}
    /*	for(i=0;i < n; i++)
    	{
    		for(j=0; j < n; j++)
    		{
    			cout << A[i][j] << ' ';
    			
    		}
    		cout << "\n";
    	}   */
    	
    	cout<< A[u][v] <<"\n";	
    }
    return 0;
}
