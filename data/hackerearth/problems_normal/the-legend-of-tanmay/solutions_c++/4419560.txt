#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long ll;

ll arr[20];

int main()
{
	int t,n,z;
	ll min_res,max_res;
	scanf("%d", &t);
	while(t--) {
		z = -1;
		max_res = min_res = 1;
		scanf("%d", &n);
		for(int i = 1; i <= n; i++)
			scanf("%lld", &arr[i]);

		sort(arr+1, arr+n+1);

		for(int i = 1; i <= n; i++)
			if(arr[i] >= 0) {
				z = i;
				break;
			}


		if(n == 1) {
			min_res = max_res = arr[1];
		}
		else {
			if(z > 0) {
				if(z > 1) {
					for(int i = 1; i <= z-1; i++)
						min_res *= arr[i];
					if(min_res > 0) {
						max_res = min_res;
						min_res /= arr[z-1];	
					}
					else {
						max_res = min_res / arr[z-1];
					}

					if(z == 2 && arr[n] == 0)
							max_res = 0;

					while(z <= n) {
						if(arr[z] == 0) {
							z++;
							continue;
						}

						max_res *= arr[z];
						min_res *= arr[z];
						z++;
					}

				}
				else {
					min_res = arr[1];
					for(int i = 1; i <= n; i++) {
						if(arr[i] == 0)
							continue;
						max_res *= arr[i];
					}

					if(arr[n] == 0)
						max_res = 0;
				}
			}
			else {
				for(int i = 1; i <= n-1; i++)
					min_res *= arr[i];
			
				if(min_res > 0) {
					max_res = min_res;
					min_res *= arr[n];	
				}
				else {
					max_res = min_res * arr[n];
				}
			}

		}
		printf("%lld %lld\n", max_res, min_res);
	}
}
