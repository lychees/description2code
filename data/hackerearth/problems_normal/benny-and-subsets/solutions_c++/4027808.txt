using namespace std;


#define LL long long int
#define sc1(x) scanf("%d", &x)
#define sc2(x, y) scanf("%d%d", &x, &y)
#define sc3(x, y, z) scanf("%d%d%d", &x, &y, &z)
#define pr1(x) printf("%d\n", x)
#define FOR(i, a, n) for(int i=a;i<n;i++)
#define pp pair<int, int>
#define pb push_back 

#define MAX 1000001
#define LIMIT 2097154
#define MOD 10000007

int a[1001], b[1001];

int mat[LIMIT], tempM[LIMIT];

map<int, int> myMap, temp;


int main() {

	int n, x, y, A, B;

	sc2(n, x);
	A = B = 0;
	FOR(i, 0, n) {
		sc1(y);
		if(y > 1024) a[A++] = y;
		else b[B++] = y;
	}

 	FOR(i, 0, A) {
		FOR(j, 0, LIMIT) tempM[j] = 0;

		FOR(j, 0, LIMIT) {
			tempM[j^a[i]] = mat[j];
		}
		tempM[a[i]]++;

		FOR(j, 0, LIMIT) {
			mat[j] = (mat[j] + tempM[j]) % MOD;
		}
	}

	FOR(i, 0, B) {
		temp.clear();

		for(auto elem: myMap) {
			temp[b[i]^elem.first] = elem.second;
		}
		temp[b[i]]++;

		for(auto elem: temp) {
			myMap[elem.first] = (myMap[elem.first] + elem.second) % MOD;
		}
	}
	//myMap[0]++;

	//mat[0]--;
	LL result = mat[x];
	if(x==0) result++;
	//cout<<result<<endl;
	for(auto &elem: myMap) {
		LL cnt = elem.second;

		//cout<<elem.first<<" "<<elem.second<<endl;

		if(elem.first == x) result = (result + cnt) % MOD;
		result = (result + (cnt * mat[x^elem.first])) % MOD;
	}

	cout<<result<<endl;

	return 0;
}
