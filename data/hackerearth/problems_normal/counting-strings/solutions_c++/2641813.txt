#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	string s;
    	cin>>s;
    	long long int count=0;
    	long long int ans=0;
    	long long int n=s.size();
    	long long int sub=((n+1)*n)/2;
    	for(long int i=0;i<n;i++){
    		if(s[i]=='a'||s[i]=='z'){
    			ans+=(count*(count+1))/2;
    			count=0;
    		}
    		else{
    			count++;
    		}
    	}
    	ans+=(count*(count+1))/2;
    	long long int res=sub-ans;
    	cout<<res<<endl;;
    }
    return 0;
}
