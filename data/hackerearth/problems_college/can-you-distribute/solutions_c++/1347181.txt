using namespace std;
typedef unsigned long long ull;
#define mod 10000007
int dp[101][101];
int func(int n, int r)
{
	if(dp[n][r]!=-1) return dp[n][r];
	if(n==r) return 1;
	if(r==1) return n;
	if(n<r) return 0;
	dp[n][r]=(func(n-1,r)%mod+func(n-1,r-1)%mod)%mod;
	return dp[n][r];
}

int main()
{
	int n,r,i,j;
	cin>>n>>r;
	if(n<r) cout<<"-1"<<endl;
	else
	{		
		memset(dp,-1,sizeof(dp));
		cout<<func(n-1,r-1)<<endl;
	}

}
