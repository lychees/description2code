n = int(raw_input())
 
while n > 0 :
	n-=1
	x = int(raw_input())
	i = 2
	good = 0
	count = 2
	
	while i * i < x:
		if x % i ==0 :
			count += 2
			if count >= 4:
				good = 1
				break
		i+=1
	if x % i == 0 :
		count +=1
		if count >=4:
			good = 1
	if not good:
		print 'NO'
	else:
		print 'YES'
