#include <iostream>
using namespace std;

int main()
{
	long long int n,flag=0,r,c;
	cin>>n;
	for(long long int i=0;i<n;i++)
	{
		cin>>r>>c;
		string s[r];
		for(long long int j=0;j<r;j++)
			cin>>s[j];
		for(long long int j=0;j<r;j++)
		{
			flag=0;
			for(long long int k=j+1;k<r;k++)
				if(s[j]==s[k])
					{cout<<"No\n";flag=1;break;}
						if(flag==1)
					break;
		}
	
			if(flag==0)
		cout<<"Yes\n";
	}
	
    
    
    return 0;
}
