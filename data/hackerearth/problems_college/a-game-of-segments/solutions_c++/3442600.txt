#include <iostream>
using namespace std;
int max(int a,int b){ return (a>b?a:b); }
int totalWays(int ar[],int sum[],int st,int end,int summ)
{
	if(st==end) return 1;
	int ans=0;
	for(int i=st;i<end;i++)
	{
		
		{
			if(st==0)
			{
				summ=sum[i];
				ans+=totalWays(ar,sum,i+1,end,summ);
			}
			else if(summ<=(sum[i]-sum[max(st-1,0)]))
			{
				summ=sum[i]-sum[st-1];
				ans+=totalWays(ar,sum,i+1,end,summ);
		
			}
		}	
	}
	//cout<<ans<<endl;
	return ans;
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	char str[50];
    	int ar[50],len,sum[50];
    	cin>>str;
    	ar[0]=sum[0]=str[0]-48;
    	for(int i=1;str[i]!='\0';i++)
    	{
    		ar[i]=(str[i]-48);
    		//cout<<ar[i]<<" ";
    		sum[i]=ar[i]+sum[i-1];
    		//cout<<sum[i]<<" ";
    		len=i;
    	}
    	//cout<<len<<endl;
    	cout<<(totalWays(ar,sum,0,len+1,0))<<endl;
    }
    return 0;
}
