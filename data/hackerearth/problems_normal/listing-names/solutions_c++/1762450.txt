#include<bits/stdc++.h>
using namespace std;
int main(){
	long t;
	cin>>t;
	vector<string> v;
	while(t--){
		string str;
		cin>>str;
		v.push_back(str);
	}
	sort(v.begin(),v.end());
	for(int i=0;i<v.size();i++){
		long count=1;
		if(i!=(v.size()-1)){
		while(v[i]==v[i+1]){
			i++;
			count++;
			if(i==v.size()-1){
				break;
			}
		}
		cout<<v[i]<<" "<<count<<endl;
		}
		else{
			cout<<v[i]<<" "<<count<<endl;
		}
	}
}
