int gcd(int a, int b)
{
    while(b) b ^= a ^= b ^= a %= b;
    return a;
}

template <class T> T& get(T &n){
    static int p;
    static bool f;
    f=false;
    n = 0;
    do{
        p=getcx();
    }while( !isdigit(p) && p!='-');
    if(p=='-'){
        f=true;
        p=getcx();
    }
    while ( isdigit(p) ){
        n = (n << 3) + (n << 1) + (p - '0');
        p = getcx();
    }
    n = (f ? -n : n);
    return n;
}

}
