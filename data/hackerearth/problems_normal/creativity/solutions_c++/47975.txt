#include <iostream>
#include <sstream>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    string s1,s2;
    ios_base::sync_with_stdio(false);
    cin>>t;
    getline(cin,s1);
    while(t--)
    {
    	getline(cin,s1);
    	getline(cin,s2);
    	if(s2.find(s1)!=string::npos) cout<<"possible"<<endl;
    	else cout<<"not possible"<<endl;
    	s1.clear();s2.clear();
    }
    return 0;
}
