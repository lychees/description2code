inline LL mod(LL x){
    if(x > 0) return x;
    return -x;
}

inline void Solve(){
    LL n, xp, yp, xk, yk, m;
    cin>>n>>yp>>xp>>yk>>xk>>m;
    if(yp > yk + 1){
        cout<<"White Wins"<<endl;
        return;
    }
    if(yp == yk + 1 and m == 0){
        cout<<"White Wins"<<endl;
        return;
    }
    LL dp = (n - yp), dk = mod(xp - xk);
    if(m) dk--;
    if(dp < dk){
        cout<<"White Wins"<<endl;
    }
    else{
        cout<<"Draw"<<endl;
    }

    return;
}

int main(){
    LL t = 1;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
