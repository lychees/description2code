#include <bits/stdc++.h>
using namespace std;
double dp[1<<20];
int main()
{
    int n;
    double p;
    cin>>n>>p;
    p /= 100.0;
    dp[0]=1;
    for(int i=1;i<=n;i++)
    {
    	if (i-2>=0)
    		dp[i] += dp[i-2]*p;
    	if (i-3>=0)
    		dp[i] += dp[i-3]*(1.0-p);
    }
    cout<<fixed<<setprecision(6)<<dp[n]<<endl;
    return 0;
}
