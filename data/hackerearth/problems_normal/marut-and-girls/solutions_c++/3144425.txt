#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int a[11111];
char s[1000001];
int main(void){
//	std::ios::sync_with_stdio(false);cin.tie(0);
	int n,m,i;
	int x;
	int re=0;
	int count;
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		scanf("%d",&x);
		a[x]=1;
	}
	scanf("%d",&n);
	gets(s);
	char c;
	int num;
	while(n--)
	{
		count=0;
		i=0;
		num=0;
		gets(s);
		while(s[i]!='\0')
        {
        	c=s[i];
            if(c==' ')
            {
                if(a[num]==1)
                {
                	count++;
                }
                num=0;
            }
            else
            {
                num=num*10+(c-'0');
            }
            i++;
        }
        if(num!=0)
        {
        	if(a[num]==1)
                count++;
        }
		if(count>=m)
			re++;
	}
	cout<<re<<endl;

	return 0;
}
