#include<bits/stdc++.h>
#define mp(a,b) make_pair(a,b)
#define pb(a) push_back(a)
#define scan(a) scanf("%d",&a)
#define scanl(a) scanf("%lld",&a)
#define print(a) printf("%d",a)
#define printl(a) printf("%lld",a)
#define MAX 100
using namespace std;

typedef pair<int,int>pii;
typedef long long LL;

bool cmp(const pii &left,const pii &right)
{
	return left.second<right.second;
}
int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else
		return gcd(b,a%b);
}
LL f[MAX];
LL cnt[MAX];
void fib()
{
	f[0]=0;
	f[1]=1;
	LL k=1;
	
	cnt[0]=0;
	for(int i=2;i<75;i++)
	{
		f[i]=f[i-1]+f[i-2];
		if(i>3)
		{
			cnt[k]=f[i-1]-1+cnt[k-1];
			k++;
		}
	}
	//cout<<f[74]<<endl;
}
int main()
{
	int t;
	scan(t);
	fib();
	while(t--)
	{
		LL x;
		scanl(x);
		int l=4;
		while(l<75)
		{
			LL temp;
			if(cnt[l-4]+f[l+1]-f[l]-1<x)
			{
				l++;
			}
			else
			{
				if(cnt[l-4]==x)
				{
					cout<<f[l]-1<<endl;
					break;
				}
				else
				{
					temp=x-cnt[l-4];
					//cout<<temp<<" "<<l<<endl;
					cout<<f[l]+temp<<endl;
					break;
				}
			}
		}
	}
}
