#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int L[5001][5001];

int lcs( string X, string Y, int m, int n )
{
   
   int i, j;
  
   for (i=0; i<=m; i++)
   {
     for (j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         L[i][j] = 0;
  
       else if (X[i-1] == Y[j-1] )
         L[i][j] = L[i-1][j-1] + 1;
  
       else
         L[i][j] = max(L[i-1][j], L[i][j-1]);
     }
   }
    
   return L[m][n];
}

int main() {
   string str1,str2,s,s1;
  cin>>s>>s1;
  for(int i=0;i<s.length();i++)
      if( s[i]!='i'&& s[i]!='a'&& s[i]!='e'&& s[i]!='o' && s[i]!='u')
      str1=str1+s[i];
      for(int i=0;i<s1.length();i++)
      if( s1[i]!='i'&& s1[i]!='a'&& s1[i]!='e'&& s1[i]!='o' && s1[i]!='u')
      str2=str2+s1[i];
cout<<lcs(str1,str2,str1.length(),str2.length())<<endl;

}
