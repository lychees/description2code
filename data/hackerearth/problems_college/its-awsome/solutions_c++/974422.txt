#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;

while(t--)
{

    char ch[100];
    cin>>ch;

    int fg=0,l=strlen(ch),gg=1,xx=1;

    for(int i=0;i<l;i++)
    {
            if(ch[i]=='.')
                continue;

           if(ch[i]=='7' || ch[i]=='4')
            fg++;
   }

//cout<<"fg is"<<fg<<endl;
xx=0;
        do {
    int digit = fg % 10;
    if(digit!=4 && digit !=7)
    {
        xx=1;
        break;
    }
    fg /= 10;
} while (fg > 0);


        if(!xx)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;





}


return 0;
}
