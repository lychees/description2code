#include<iostream>

using namespace std;

int main()
{
	int t,ctr,d;
	long a,b,x,rev;
	cin>>t;
	while(t--)
	{
		ctr=0;
		cin>>a>>b;
		while(a<=b)
		{
			x=a;
			rev=0;
			while(x)
			{
				d=x%10;
				rev=rev*10+d;
				x=x/10;
			}
			if(rev==a)
				ctr++;
			a++;
		}
		printf("%d\n",ctr);
	}
	return 0;
}
