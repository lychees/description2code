#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
	long long int t,n,root;
	long long a,b;
	cin>>t;
	while(t--)
	{
		cin>>n;
		root=sqrt(n);
		a=root*root;
		b=(root+1)*(root+1);
		if(root*root == n)
		{
			cout<<2*root-1<<endl;
		}
		else
		{
			if(n>=(b-sqrt(b)+1))
			{
				cout<<2*sqrt(b)-1<<endl;
			}
			else
				cout<<2*sqrt(a)<<endl;

		}
	}
	return 0;
}
