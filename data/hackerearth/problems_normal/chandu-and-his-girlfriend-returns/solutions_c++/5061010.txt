void merge(int arrM[],int arr1[],int arr2[],int N,int M)
{
	int i = 0;
	int j = 0;
	int c[N+M];
	int z = 0;
	while(i<N && j<M)
	{
		if(arr1[i] > arr2[j])
		{
			c[z++] = arr1[i];
			i++;
		}
		else
		{
			c[z++] = arr2[j];
			j++;
		}
	}
	while(i<N)
	{
		c[z++] = arr1[i];
		i++;
	}
	while(j<M)
	{
		c[z++] = arr2[j];
	}
	for(int i=0;i<z;i++)
	{
		arrM[i] = c[i];
	}
}
