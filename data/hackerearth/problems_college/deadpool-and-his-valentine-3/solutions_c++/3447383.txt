#include <iostream>
using namespace std;

int main()
{
    int k, m;
	cin>>k>>m;
	int flg=1;
	int b[m];
	for(int i=0;i<m;i++)
        b[i]=0;
	for(int r=0;r<k;r++)
    {
        int l;
        cin>>l;
        for(int c=0;c<l;c++)
        {
            int p;
            cin>>p;
            b[p-1]=1;
        }
    }
    for(int i=0;i<m;i++)
    {
        if(b[i]==0)
        {
            cout<<"NO"<<endl;
            flg=0;
            break;
        }
        
    }
    if(flg)
        cout<<"YES"<<endl;
    return 0;
}
