#include<iostream>
#include<cstdio>
#define ll long long
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
#define mod 1000000009
ll power(ll a,ll b)
{
	ll mul=1;
	while(b)
	{
		if(b&1)
			mul=(mul*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return mul;
}
int main()
{
	ll t,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		printf("%lld\n",power(26,(n+1)/2));
	}
	return 0;
}
