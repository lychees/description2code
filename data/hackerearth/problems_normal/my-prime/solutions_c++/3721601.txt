#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	
	int n, i, j;
	
	cin>>n;
	
	int a[n];
	
	for(i=0;i<n;i++)
		cin>>a[i];
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(a[i]%a[j]==0 && i!=j)
					break;
				
		}
		
		if(j==n)
			cout<<a[i]<<" ";
	}
		
	return 0;
}
