#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
int c=0;
int bs(long long int a[],int low,int high,int x)
{
	if(low<=high)
	{
		int mid=(low+high)/2;
		if(a[mid]==x)
			return mid;
		else if(a[mid]<x)
			return bs(a,mid+1,high,x);
		else
			return bs(a,low,mid-1,x);
	}
	c=1;
	return low;
}
int main()
{
	int n,i,q,k=0;
	scanf("%d",&n);
	long long int a[n],x,p[n];
	for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
	scanf("%d",&q);
	sort(a,a+n);
	p[k++]=a[0];
	for(i=1;i<n;i++)
	{
		if(a[i]==a[i-1])
			continue;
		else
			p[k++]=a[i];
	}
	while(q--)
	{
		scanf("%lld",&x);
		int s=bs(p,0,k-1,x);
		if(c!=1)
		{
			if(s!=k-1)
				printf("YES %lld %lld\n",p[s],p[s+1]);
			else
				printf("YES %lld -1\n",p[s]);
		}
		else
			printf("NO %lld %lld\n",p[s],p[s]);
		c=0;
	}
	return 0;
}
