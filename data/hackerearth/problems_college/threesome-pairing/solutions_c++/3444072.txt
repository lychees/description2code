#include <iostream>
using namespace std;

int main()
{
    int n,t,i,c=0,j;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    		scanf("%d",&a[i]);
    	}
    	for(i=0;i<n;i++)
    	{
    		for(j=i+1;j<n;j++)
    		{
    			if((a[i]+a[j])%3==0)
    			c++;
    		}
    	}
    	printf("%d\n",c);
    	c=0;
    }
    
    return 0;
}
