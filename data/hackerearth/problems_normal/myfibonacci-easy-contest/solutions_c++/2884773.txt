#include <iostream>
using namespace std;
int fibonacii_no[11] = {-1};

int get_fib(int n)
{
	if (n <= 1) {
		return fibonacii_no[n-1];
	}
	if (fibonacii_no[n-1] != -1) {
	//	cout <<n <<":" <<fibonacii_no[n-1]<<endl;
		return fibonacii_no[n-1];
	}
	int temp = get_fib(n-1)+ get_fib(n-2);
	fibonacii_no[n-1] = temp;
	//cout <<n<<": "<<temp<<endl;
	return temp;
}

int main()
{
	int N;
	cin >> fibonacii_no[0];
	cin >> fibonacii_no[1];
	for (int i=2; i<11; i++)
		fibonacii_no[i] = -1;
    cin >> N;
    cout << get_fib(N);
    return 0;
}
