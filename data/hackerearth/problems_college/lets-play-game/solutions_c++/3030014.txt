#include <iostream>
#define ll long long
using namespace std;

int main()
{
	ll t, n, k, i, d, min;
    cin>>t;
    while(t--)
    {
    	min=0;
    	cin>>n>>k;
    	ll a[n], b[n];
    	for(i=0;i<n;i++)
    		cin>>a[i];
    	for(i=0;i<n;i++)
    		cin>>b[i];
    	for(i=0;i<n;i++)
    	{
    		d=k/a[i];
    		if((d*b[i])>min)
    			min=d*b[i];
    	}
    	cout<<min<<endl;
    }
    return 0;
}
