import math
test = int(raw_input())
while test>0 :
	n = int(raw_input())
	sq = math.sqrt(n)
	f = math.floor(sq)
	c = math.ceil(sq)
	if int(f)**2 == n:
		print("YES");
	else:
		f = int(f)**2
		c = int(c)**2
		c = c - n
		f = n - f
		if c<f:
			print('+'+str(c))
		else:
			print('-'+str(f))
	test-=1
