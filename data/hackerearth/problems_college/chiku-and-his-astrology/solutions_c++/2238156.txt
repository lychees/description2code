#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
    	string str;
    	cin >> str;
    	int n = str.size();
    	stack<char>s;
    	while(!s.empty()) {
    		s.pop();
    	}
    	for(int i = 0; i < n; i++) {
    		if(s.empty()) {
    			s.push(str[i]);
    		}
    		else {
    			if(str[i] == '0' && s.top() == '1') {
    				s.pop();
    			}
    			else if(str[i] == '1' && s.top() == '0') {
    				s.pop();
    			} 
    			else {
    				s.push(str[i]);
    			}
    		}
    	}
    	int sz = 0;
    	while(!s.empty()) {
    		s.pop();
    		sz++;
    	}
    	cout << sz << endl;
    }
    return 0;
}
