#include<iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;

    while(t--)
    {

        string s;
        cin>>s;
        int a=s.size();

        int x=0;
        int O=0;
        int e=0;

        for(int i=0;i<a;i++)
            {
                if(s[i]=='X')
                    x++;

                if(s[i]=='O')
                    O++;

                if(s[i]=='*')
                {
                    if(x==0)
                        e+=O;

                    O=0;
                    x=0;
                }
            }

            if(x==0)
                e+=O;

            cout<<e<<endl;

    }


}
