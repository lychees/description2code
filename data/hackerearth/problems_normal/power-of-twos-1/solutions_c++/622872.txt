#include <bits/stdc++.h>
using namespace std;

#define N 1000000
long long a[1000005];

int main()
{
	int i,j,t;
	
	memset(a,0,sizeof(a));

	for(i=1;i<=N/2;i++)
	{
		for(j=2*i;j<=N;j=j+i)
		{
			a[j]++;
		}
	}
	for(i=2;i<=N;i++)
	{
		a[i]=a[i]+a[i-1];
	}
	long long x;

	for(scanf("%d\n",&t);t--; )
	{
		scanf("%lld",&x);
		printf("%lld\n",a[x]);
	}
	return 0;
}
