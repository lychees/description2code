#include <iostream>

#define ll long long int

using namespace std;

ll mod=1000000007;

void multiply(ll f[2][2],ll m[2][2])
{
	ll a,b,c,d;
	a=f[0][0]*m[0][0]+f[0][1]*m[1][0];
	b=f[0][0]*m[0][1]+f[0][1]*m[1][1];
	c=f[1][0]*m[0][0]+f[1][1]*m[1][0];
	d=f[1][0]*m[0][1]+f[1][1]*m[1][1];
	f[0][0]=a%mod;
	f[0][1]=b%mod;
	f[1][0]=c%mod;
	f[1][1]=d%mod;
}

void power(ll f[2][2],ll n)
{
	if(n==0||n==1) return;
	ll m[2][2]={{1,1},{1,0}};
	power(f,n/2);
	multiply(f,f);
	if(n%2==1) {
		multiply(f,m);
	}
}

ll fibo(ll n)
{
	ll f[2][2]={{1,1},{1,0}};
	if(n==0) return 0;
	if(n>0)
		power(f,n-1);
	return f[0][0];
}

int main()
{
	ll t,ans,k,n;
	t=0;
	cin >> t;
	while(t--) {
		cin>>n;
		n=n+1;
		n=n/2;
		k=fibo(n);
		ans=(k*k)%mod;
		cout << ans << endl;
	}
	return 0;
}
