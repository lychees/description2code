#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int i,n;
    scanf("%d",&n);
    int w[n],l[n],t[n],s[n];
    for(i=1;i<=n;i++)
    	{scanf("%d%d%d",&w[i],&t[i],&l[i]);
    	}
    vector<int> v;
    for(i=1;i<=n;i++)
    	{s[i]=(w[i]*3)+(t[i]*1)-(l[i]*1);
    	v.push_back(s[i]);
    	}
    vector<int> ::iterator itr =max_element(v.begin(),v.end());
    printf("%d",*itr);
    
    return 0;
}
