#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int s[100001] , q , l1 , l2 , r1 , r2 , i;
	string str;
    cin >> str;
    cin >> q;
    s[0] = 0;
    for(i = 1; i <= str.length(); i++)
    	s[i] = str[i-1] - 'a' + s[i-1];
    while(q--)
    {
    	cin >> l1 >> r1 >> l2 >> r2;
    	if((r1 - l1) != (r2 - l2))
    	{
    		cout << "No\n";
    		continue;
    	}
    	if((s[r1] - s[l1]) != (s[r2] - s[l2]) || (str[l1-1] != str[l2-1]))
    		cout << "No\n";
    	else
    		cout << "Yes\n";
    }
    return 0;
}
