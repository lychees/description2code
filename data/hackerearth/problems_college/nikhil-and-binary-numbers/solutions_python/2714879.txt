def getXOR(a):
    result = 0
    for x in a:
        result ^= int(x)
    return result


noc = input()
while noc:
    noc -= 1
    lens = input()
    ins = map(int, raw_input().split())
    cnt = 0
    for x in range(len(ins)):
        b = ins[:]
        b.remove(ins[x])
        if getXOR(b)==0:
            cnt += 1
    print cnt
