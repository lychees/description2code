#include<bits/stdc++.h>
using namespace std;

int n,k,arr[10010],dp[10010];

int fun(int index)
{
    if(index>=n)
        return 0;

    if( dp[index] != -1 )
        return dp[index];

    int a=0,b=0;

    if( index  < n )
    a = arr[index] + fun(index+k+1);

    b = fun(index+1);

    return dp[index] = max(a,b);

}

int main()
{
    int t;
    cin>>t;

     while(t--)
     {
         cin>>n>>k;
         for(int i=0;i<n;i++)
            cin>>arr[i];
    memset(dp,-1,sizeof(dp));
     cout<<fun(0)<<endl;

     }
}
