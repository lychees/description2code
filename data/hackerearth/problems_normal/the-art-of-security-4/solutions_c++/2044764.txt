#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;
int ans = 0,ent1,ent2;
int A[13],n;

void solve(int num)
{
	if(ent1 > ent2)
		return;

	if(num == n)
	{
		ans++;
		return;
	}

	if(ent1 + A[num] <= ent2)
	{
		ent1 = ent1 + A[num];
		solve(num + 1);
		ent1 -= A[num];
	}
	ent2 += A[num];
	solve(num + 1);
	ent2 -= A[num];

	return;
}

int main(int argc, char const *argv[])
{
	scanf("%d",&n);
	int i = 0;
	int sum = 0;
	while(i < n)
	{
		scanf("%d",&A[i++]);
		sum = sum + A[i-1];
	}
	sort(A,A+n);
	ent1 = 0;
	ent2 = 0;
	ans = 0;
	solve(0);

	while(next_permutation(A, A+n))
	{
		ent1 = 0; ent2 = 0;
		solve(0);
	}
	printf("%d %d\n",ans,sum );
	if(ans > sum)
	printf("We will win!");
	else
	printf("Got no way out!");
	return 0;
}
