#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <map>
#include <queue>
#include <set>
#include <utility>
#include <vector>
using namespace std;

int main() {
  long long hei, wei;
  cin >> hei >> wei;
  pair<long long, long long> ans = make_pair(1, 1);
  for (long long h = 1, w; h <= hei; h <<= 1) {
    w = min(5 * h / 4, wei);
    if (4 * h <= 5 * w) {
      ans = max(make_pair(h * w, h), ans);
    }
  }
  for (long long w = 1, h; w <= wei; w <<= 1) {
    h = min(5 * w / 4, hei);
    if (4 * w <= 5 * h) {
      ans = max(make_pair(h * w, h), ans);
    }
  }
  cout << ans.second << " " << ans.first / ans.second;
  return 0;
}
