ll howMany(ll a, ll b, ll p)
{
    ll one = ceil((double)a/(double)p);
    ll two = b/p;
    return two-one+1;
}

int main(){
    int k,t;
    ll a,b,p[12];
    scanf("%d %d",&k,&t);
    rep(i,k)scanf("%lld",&p[i]);
    while(t--){
        scanf("%lld %lld",&a,&b);
        ll ans = 0;
        for(int i=1;i<(1<<k);i++)
        {
            bool odd = __builtin_popcount(i)&1;
            ll pro=1;
            rep(j,k)
            {
                if(pro > b)
                    break;                
                if(i & (1<<j))                
                    pro*=p[j];                
            }
            if(odd) ans+=howMany(a,b,pro);
            else ans-=howMany(a,b,pro);
        }
        printf("%lld\n",ans);
    }
}
