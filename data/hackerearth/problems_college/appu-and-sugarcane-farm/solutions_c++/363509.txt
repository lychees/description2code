#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

int gcd(int a,int b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}
int main()
{
    int n,h[50],comm=0;
    double l;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
    	scanf("%d",(h+i));
    	comm=gcd(h[i],comm);
    }
    for(int i=0;i<n;i++)
    {
    	h[i]/=comm;
    	l=log2(h[i]);
    	if(l!=(int)l)
    	{
    		printf("NO");
    		return 0;
    	}
    }
    printf("YES");
    return 0;
}
