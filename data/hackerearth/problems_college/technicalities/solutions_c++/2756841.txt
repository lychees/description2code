#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ll int test;
	cin>>test;
	while(test--)
	{
		ll int i,j,l;
		string s;
		cin>>l>>s;
		ll int ans=0;
		for(i=0;i<l;i++)
		{
			ans=(ans*10 +(s.at(i)-'0'))%mod;
		}
		cout<<ans<<endl;
	}
	return 0;
}
