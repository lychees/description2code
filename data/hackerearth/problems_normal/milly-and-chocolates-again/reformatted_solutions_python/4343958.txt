def bin(ci, ind, lower, upper):
    # print ci
    while (lower <= upper):
        if lower+1 == upper or lower == upper:
            if ind <= ci[lower]:
                # print 'returning lower'
                return lower
            # print 'returing upper'
            return upper
        mid = (lower+upper)/2
        if ci[mid] >= ind and ci[mid-1] < ind:
            # print 'here'
            return mid
        elif ci[mid] > ind and ci[mid-1] >= ind:
            # print 'first half'
            upper = mid-1
        else:
            # print 'second half'
            lower = mid+1
    return lower


n = input()
ci = map(int, raw_input().split())
q = input()
for i in range(1, n):
    ci[i] = ci[i]+ci[i-1]
for i in range(q):
    ind = input()
    print bin(ci, ind, 0, n-1)+1
