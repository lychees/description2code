#include<iostream>
#include<set>
#include<string>
using namespace std;
int main()
{   int t,n,m;
cin>>t;
string x;
set<string> ss;
while(t--)
	{cin>>n>>m;
	for(int i=0;i<n;i++)
	{cin>>x;
	ss.insert(x);
	}
	if(ss.size()==n)
	cout<<"Yes"<<endl;
	else
	cout<<"No"<<endl;
	ss.clear();	
	}
	
}
