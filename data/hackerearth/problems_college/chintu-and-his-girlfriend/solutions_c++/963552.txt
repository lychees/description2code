#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
bool hashTable[105][30];
int main() {
    string str;
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++) {
        cin>>str;
        for(int j=0;j<str.size();j++) hashTable[i][str[j]-'a']=true;
    }
    int count=0;
    for(int i=0;i<26;i++) {
        bool flag=false;
        for(int j=0;j<n;j++) {
            if(hashTable[j][i]==false) flag=true;
        }
        if(flag==false) count++;
    }
    if(count>=k) printf(":-)");
    else printf(":-(");
    return 0;
}
