#include <iostream>
#include<map>
#include<cstring>
using namespace std;


map<long int,long int> mp;
//long long int a[200000],mp[200000];
long long int t,n,a[2000000],maxx=-9,start,ss,ee;
int main()
{ // long long int t,n,a[200000],maxx=-9,start,end,ss,ee,mp[200000];
     cin>>t;
     while(t--)
     {  maxx=-9;
       mp.clear();
        start=0;
        //end=0;
     	cin>>n;
       for(int i=0;i<n;i++)
       {
       	cin>>a[i];
       }
       
      // memset(mp,-5,sizeof(mp));
      /*for(int i=0;i<200000;i++)
      {
      	mp[i]=-5;
      }*/
       
       for(int i=0;i<n;i++)
       {
       	if(mp.find(a[i])==mp.end() || mp[a[i]]<start)
       	{  //end++;
       	//  cout<<i<<" "<<mp[a[i]]<<endl;
       		mp[a[i]]=i;
       		 // cout<<a[i]<<" "<<start<<" "<<end<<endl;
       		if(i-start+1>maxx)
       		{   ss=start;
       		    ee=i;
       			maxx=i-start+1;
       		}
       		//end++;
       	}
       	
       	else
       	{   // cout<<start<<" ";
       	    // cout<<i<<endl;
       		while(a[start]!=a[i])
       		{
       			start++;
       		}
       		start=start+1;
       		mp[a[i]]=i;
       	//	end++;
       	//	cout<<start<<endl;
       	}
       }
       
       cout<<ss+1<<" "<<ee+1<<endl;
     }


return 0;
}
