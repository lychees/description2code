#include<bits/stdc++.h>
using namespace std;
vector< string > all(65666);
#define lld long long int
void cal(int n){
    string x = "";
    int val = n;
    char test;
    while(n>0){
        test = n%2 + '0';
        x = test+x;
        n/=2;
    }
    while(x.size()<16)
        x ='0'+x;

    all[val] = x;
}
int main(){
    int test;
    scanf("%d", &test);
    for(int i=0; i<65536; ++i) cal(i);
    while(test--){
        int n;
        scanf("%d", &n);
        lld arr[n];
        for(int i=0; i<n; ++i)
            scanf("%lld", &arr[i]);
        int total = 2;
        for(int i=1; i<n; ++i)
            total *= 2;
        //cout<<total<<endl;

        priority_queue< pair<lld, int> > pq;
        int p = 0;
        for(int i=1; i<total; ++i){
            lld xr = 0, an;
            int cnt = 0;
            bool f = false;
            for(int j=15; j>=0; --j)
                if(all[i][j] =='1'){
                    cnt++;
                    xr = xr ^ arr[15-j];
                    if(f)
                        an = an & arr[15-j];
                    else
                        an = arr[15-j], f = true;
                }
                lld t = xr | an;
                //cout<<t<<endl;
                pq.push(make_pair(t, cnt));

        }
        pair<lld, int> tm;
        tm = pq.top();
        cout<<tm.first<<" "<<tm.second<<endl;
        //cout<<total;
    }
}
