#include <stdio.h>
#include <string.h>
#define LL long long
int main(){
    int t,len,i;
    scanf("%d",&t);
    while(t--){
        char s[100005];
        scanf("%s",s);
        len=strlen(s);
        LL ans=s[len-1]-'0';
        int temp;
        for(i=len-3;i>=0;i-=2){
            temp=s[i]-'0';
            if(s[i+1]=='+')
				ans+=(LL)temp;
            if(s[i+1]=='-')
				ans-=(LL)temp;
            if(s[i+1]=='*')
				ans*=(LL)temp;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
