#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    while(t--){
    	string s;
    	int r=0,u=0,b=0,y=0,temp;
    	cin>>s;
    	
    	for(i=0;i<s.length();i++){
    		temp=s[i]-'r';
    		if(temp==0)
    			r++;
    		temp=s[i]-'u';
    		if(temp==0)
    			u++;
    		temp=s[i]-'b';
    		if(temp==0)
    			b++;
    		temp=s[i]-'y';
    		if(temp==0)
    			y++;
    	}
    	
    	temp=min(r,u);
    	temp=min(temp,b);
    	temp=min(temp,y);
    	cout<<temp<<endl;
    	
    	
    }
    return 0;
}
