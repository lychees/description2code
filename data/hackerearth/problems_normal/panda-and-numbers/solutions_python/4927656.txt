from sys import stdin
s = set()
for i in xrange(1,8):
	s.add(i**i)
def val(n):
	if n in s:
		return 1
	a = str(n)
	for i in a:
		j = ord(i) - ord('0')
		pr = n - j*j
		#print "checking ",pr,"for ",n
		if pr in s:
			return 1
	return 0

hi = 10**6 + 5
for i in xrange(1,hi):
	if val(i):
		s.add(i)

a = stdin.readlines()
for i in a[1:]:
	if int(i) in s:
		print "Yes"
	else:
		print "No"
