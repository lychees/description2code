#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define d1(a)cout<<#a<<": "<<a<<"\n";
#define d2(a,b)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<"\n";
#define d3(a,b,c)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<"\n";
#define d4(a,b,c,d)cout<<#a<<": "<<a<<" , "<<#b<<": "<<b<<" , "<<#c<<": "<<c<<" , "<<#d<<": "<<d<<"\n";

ll a[100005];
ll dp[100005];

int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	dp[0] = 1;
	for(int i=1;i<n;i++){
		if(a[i] > a[i-1]){
			dp[i] = dp[i-1] + 1;
		}
		else{
			dp[i] = 1;
		}
	}
	for(int i=(n-2);i>=0;i--){
		if(a[i] > a[i+1] && dp[i]<=dp[i+1]){
			dp[i] = dp[i+1] + 1;
		}
	}
	ll ans1 = 0;
	for(int i=0;i<n;i++){
		ans1 = ans1 + dp[i];
	}
	cout<< ans1;
	return 0;
}
