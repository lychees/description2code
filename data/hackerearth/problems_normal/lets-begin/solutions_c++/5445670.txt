#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	long long  a[999999];
	a[0]=0;
	a[1]=1;
	a[2]=1;
	a[3]=1;
	a[4]=2;
	a[5]=1;
	a[6]=2;
	a[7]=1;
	a[8]=2;
	for(int i=9;i<999999;i++)
	{

		a[i]=min(i/7+a[i%7],i/5+a[i%5]);
		a[i]=min(a[i],i/3+a[i%3]);
		a[i]=min(a[i],i/2+a[i%2]);
	}
	for(int i=0;i<t;i++)
	{
		int n;
		cin>>n;
		if(n==1||n==0)
            cout<<"-1"<<endl;
        else
		cout<<a[n]<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
