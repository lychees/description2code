#include <bits/stdc++.h>
using namespace std;

int main()
{   
    int t;
   scanf("%d",&t);
    while(t--){
    	long long int x,y,z;
    	scanf("%lld%lld%lld",&z,&x,&y);
    	 long long int ans=0;
    	for(long long int i=1;i*i<=z && (i<=x||i<=y);++i)
    	   if(z%i==0)
    	     {if(i<=x && z/i<=y)
    	  	    ++ans;
    	  	  if(i<=y && z/i<=x) ++ans;
    	  	  if((i<=x && z/i<=y )&& i<=y && z/i<=x && i==z/i) --ans;
    	     }
    	printf("%lld\n",ans);
    }
    return 0;
}
