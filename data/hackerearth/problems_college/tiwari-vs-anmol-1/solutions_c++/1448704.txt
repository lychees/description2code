int prime[1000010];
int p[1000010];
int sum[1000010];
void sieve()
{
    int lim= 1000010;
    memset(prime,true,sizeof(prime));
    for(int i=2;i*i<=lim;i++)
    {
        if(prime[i])
        {

            for(int j=i*i;j<=lim;j+=i)
            {
                if(p[j]==0)p[j]=i;
                prime[j]=false;
            }
        }
    }
    int mod = 1000000007;
    for(int i=2;i<=lim;i++)
    {
        sum[i]= (sum[i-1]+p[i])%mod;
    }
}
