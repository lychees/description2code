int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int i,j,k,n,m;
		scanf("%d%d",&n,&m);
		int a[n][m],b[n][m];
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
				scanf("%d",&a[i][j]);
		}
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
				scanf("%d",&b[i][j]);
		}
		
		int row[n],col[m];
		
		for(i=0;i<n;i++)
		{
			row[i]=INT_MIN;
			
			for(j=0;j<m;j++)
				row[i]=max(row[i],b[i][j]);
				
		}
		
		for(i=0;i<m;i++)
		{
			col[i]=INT_MIN;
			
			for(j=0;j<n;j++)
				col[i]=max(col[i],a[j][i]);
				
		}
		int co=0;
		
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				if(a[i][j]==col[j] && b[i][j]==row[i])
					co++;
			}
		}
		
		cout<<co<<"\n";
	}
	return 0;
}
