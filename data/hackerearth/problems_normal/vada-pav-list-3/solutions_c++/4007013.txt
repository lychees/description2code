#include <bits/stdc++.h>
using namespace std;
int main()
{
    long i,n;
    char arr[1000100];
    set<string> str;
    scanf("%ld",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s",arr);
        string s(arr);
        str.insert(s);
    }
    set<string> ::iterator it;
    printf("%ld\n",str.size());
    for(it=str.begin();it!=str.end();it++)
    {
        printf("%s\n",(*it).c_str());
    }
    return 0;
}
