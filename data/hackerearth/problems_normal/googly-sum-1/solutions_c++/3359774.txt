#include<bits/stdc++.h>

#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;


ll a[50000];
ll sum[10000000];

int main(){

	ll i,p,m,n,j,c,s;

	SET(sum,-1);

	cin>>n;

	for(i=0;i<n;i++)
		cin>>a[i];
    
   
	s=300000;
   
    c=0;

	for(i=0;i<n;i++){

		for(j=0;j<i;j++){

			if(sum[a[i]-a[j]+s]!=-1){

				c++;
				break;
			}
		}

		for(j=0;j<=i;j++){

			sum[a[i]+a[j]+s]=1;
		}
	}

	cout<<c<<endl;

   return 0;
}
