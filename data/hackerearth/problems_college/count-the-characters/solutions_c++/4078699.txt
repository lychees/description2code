    #include <iostream>
    #include <string.h>
    using namespace std ;
    int main()
     {
        	int n, i, j, c= 0, total = 0;
        	cin >> n ;
    	    char s[n][100], ch;
            for (i = 0; i < n; i++)
                cin >> s[i] ;
    	    for(ch = 97 ; ch <= 122 ; ch++)
    	    {
    		    for(i = 0; i < n; i++)
    		    {
    			   for(j = 0; j < strlen(s[i]); j++)
    			   {
    			       if(ch == s[i][j])
                        {
                            c++;
                            break;
                        }
                    }
    		    }
    		    if(c== n)
                        total++;
                c = 0;
           }
    	cout << total << endl ;
    	return 0;
    }
