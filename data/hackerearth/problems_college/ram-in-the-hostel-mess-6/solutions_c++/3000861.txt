#include<bits/stdc++.h>
using namespace std;
long long fib (long long n) 
{
    if (n==1)
        return 2;
        if(n==2)
        return 3;
    return fib(n-1) + fib(n-2);
}
int main()
{
	long long t,n;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n==0)
		cout<<"1"<<endl;
		long long ans=fib(n);
		cout<<ans*ans<<endl;
	}
	return 0;
}
