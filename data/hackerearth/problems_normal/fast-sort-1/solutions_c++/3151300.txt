#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string a,b;
    	int arr[26]={0};
    	cin>>a>>b;
    	for(int i=0;i<a.size();i++)
    	{
    		char p=a[i];
    		arr[p]=i+1;
    	}
    
    	vector<pair<int,char> > v;
    	for(int i=0;i<b.size();i++)
    	{
    		char m=b[i];
    		v.push_back(make_pair(arr[m],m));
    	}
    	
    	sort(v.begin(),v.end());
    	
    	for(int i=0;i<v.size();i++)
    	{
    		cout<<v[i].second;
    	}
    	cout<<endl;
    }
    
    return 0;
}
