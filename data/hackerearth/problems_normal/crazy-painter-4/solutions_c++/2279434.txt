#include<stdio.h>
#include<stdlib.h>


int main()
{

int nT;
scanf("%d",&nT);
while(nT--)
{

 unsigned long long int n;
scanf("%llu",&n);
int c[26];
for(int i =0;i<26;i++)
 scanf("%d",&c[i]);


unsigned long long int res =12*n;
unsigned long long int q = res/26;
unsigned long long int r = res%26;
unsigned long long int sum =0;
if(q!=0)
{
   
   for(int i =0;i<26;i++)
    {
     if(r!=0 && i<r)
     sum =sum+((q+1)*c[i] +((q+1)*q)/2 );
     else
     sum =sum+(q*c[i] +((q-1)*q)/2 );
    }
  

}
else
{
 for(int i =0;i<res;i++)
   sum+=c[i];

}

printf("%llu\n",sum);
}
return 0;
}
