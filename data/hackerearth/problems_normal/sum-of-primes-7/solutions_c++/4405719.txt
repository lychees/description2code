#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define lli long long int
lli a[1000005],s[1000005];
void sieve()
{
	lli i,j;
	for(i=2;i*i<=1000000;i++)
	{
		if(!a[i])
		{
			for(j=i*i;j<=1000000;j+=i)
			{
				a[j]=1;
			}
		}
	}
	s[1]=0;
	for(i=2;i<=1000000;i++)
	{
		s[i]=s[i-1];
		if(!a[i])
		{
			s[i]+=i;
		}
	}
}
int main()
{
	sieve();
    lli n,l,r;
    cin>>n;
    while(n--)
    {
    	cin>>l>>r;
    	cout<<s[r]-s[l-1]<<endl;
    }
    return 0;
}
