#include <iostream>     
#include <algorithm> 
#include <vector>
using namespace std;
int arr[100010];

int lis(int n)
{
	int i,len=0,k=0;
	vector<int> temp;
	vector<int>::iterator low;
	temp.push_back(arr[0]);
	len=1;
	for(i=1;i<n;i++)
	{
		if(arr[i]<temp[0])
		{
			temp[0]=arr[i];
		}
		else if(arr[i]>temp[len-1])
		{
			temp.push_back(arr[i]);
			len++;
		}
		else 
		{
			low=lower_bound(temp.begin(),temp.end(),arr[i]); // return's the index of element to before element. 
			k=low-temp.begin();
			temp[k]=arr[i];
		}
	}
	return len;
}
int main()
{
	int i,j,n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
		}
		j=lis(n);
		cout<<n-j<<endl;
	}
	return 0;
}
