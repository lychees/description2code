#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long int sum;
    while(t--)
    {
    	  int a,b,i,k,flag;
    	  cin>>a>>b;
    	  sum=0;
    	  if(a==1)
    	  i=a+1;
    	  else
    	  i=a;
    	  for(;i<=b;i++)
    	  {
    	  	//cout<<"i"<<i;
    	  flag=0;
    	  	 	   for(int k=2;k<=sqrt(i);k++)
    	  	 	   {
    	  	 	   	
    	  	 	   
    	  	 	   if(i%k==0)
    	  	 	   {
    	  	 	   	flag=1;
    	  	 	   	break;
    	  	 	   }
    	  	 	   }
    	  	 	   
    	  	 	//   cout<<"count"<<count;
    	  	 	   
    	  	 	   if(!flag)
    	  	 	   {
    	  	 	   sum=sum+i;
    	  	 	  // cout<<"sum"<<sum;
    	  	 	   }
    	  	 }
    	  	 
    	  	 
    	  
    	  cout<<sum<<"\n";
    }
    return 0;
}
