#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t=0;
	scanf("%d",&t);
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		int c=0,m=INT_MAX,i=0,j=0;
		if(s1.size()==s2.size())
		{
			for(i=0;i<s1.size();i++)
			{
				if(s1[i]!=s2[i])
				c++;
			}
			m=c;
		}
		else
		{
			if(s1.size()>s2.size())
			{
				string t=s1;
				s1=s2;
				s2=t;
 
			}
			for(i=0;i<=(s2.size()-s1.size());i++)
			{
			    c=0;
				for(int j=0;j<s1.size();j++)
				{
					if(s1[j]!=s2[j+i])
					++c;
				}
				if(c<m)
				m=c;
			}
		}
		cout<<m<<endl;
	}
    
    return 0;
}
