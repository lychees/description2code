#include <iostream>
using namespace std;
int fact(int n)
{
	if(n==1)
	return 1;
	else
	return n*fact(n-1);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	if(n==1)
    	{
    		cout<<"0"<<endl;
    	}
    	else
    	{
   
    	cout<<(n-1)*n/2<<endl;
    	}
    }
    return 0;
}
