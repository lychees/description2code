#include <cstdio>
#define M 100000
using namespace std;

long long x,b[M+5];
int main()
{
    long long i,t;
   
    b[1]=1;
    b[2]=16;
    for(i=3;i<=M;i++)
    {
    	b[i]=b[i-2] + (i-2)*1LL*(i-2)*8 + 4*(3*(i-1)+1)*1LL*(i-1);
    }
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&x);
    	printf("%.6lf\n",1.0*b[x]/(1.0*x*1LL*x));
    }
    return 0;
}
