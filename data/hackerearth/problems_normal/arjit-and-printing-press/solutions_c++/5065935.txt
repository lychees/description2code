#include <iostream>
using namespace std;
#include <algorithm>



int main()
{
    int t;
    cin >> t;
    while(t--){
        string W, R;
        cin >> W >> R;
        int rlen = R.length();
        sort(R.begin(),R.end());
        int wlen = W.length();
        int j = 0;
        for(int i = 0; i < wlen; i++){
            
            while(j < rlen){
                if(W[i] > R[j]){ W[i] = R[j++]; break;}
                else if(W[i] <= R[j]){break;}
               
            }
        }
        cout << W << endl;
    }
    return 0;
}
