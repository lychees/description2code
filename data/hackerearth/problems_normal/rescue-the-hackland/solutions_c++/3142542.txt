#include<iostream>
#include<stdio.h>
#include<malloc.h>
#include<vector>
using namespace std;
int main()
{
	
	int n;
	cin>>n;

for(int m=0;m<n;m++)
{	std::vector< int > ar;
	int l,r,k;
	cin>>l>>r>>k;
	if(l!=r)
	{
		ar.push_back(l);
	}
	for (int i=l;i<r+1;i++)
	{
		int p=1;
		for (int j=0;j<ar.size();j++)
		{
			if(i<=ar[j]*k && ar[j]<=i*k)
				{
					p=0;
				break;
			}
		}
	
		if(p!=0)
			ar.push_back(i);
	}
	cout<<ar.size()<<"\n";
}
return 0;
}
