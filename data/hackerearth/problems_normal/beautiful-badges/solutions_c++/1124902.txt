#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cerr<<#x<<":"<<x<<"\n"
long long s;
int n,i,j,k,a,b,c,d;
int A[100001],B[100001];
int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
  {
    scanf("%d%d",&c,&d);
    if(c==1)
      A[a++]=d;
    else
      B[b++]=d;
  }
  sort(A,A+a);
  sort(B,B+b);
  for(i=j=0;i<a;i++)
  {
    while(j<b && B[j]*sqrt(2)<A[i])
      j++;
    if(j<b && B[j]/sqrt(2)<A[i])
    {
      s++;
      j++;
    }
  }
  cout<<s<<"\n";
  return 0;
}
