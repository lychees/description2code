#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
long long int dp(long long int arr[],int n)
{
   long long int d[10000],i;
   d[0]=arr[0];
   d[1]=max(arr[0],arr[1]);
   for(i=2;i<n;i++)
   d[i] =max(d[i-1],d[i-2]+arr[i]);
   return d[n-1];
   	
}
int main()
{
	int t;
	scanf("%d",&t);
	int j=0;
	while(t--)
	{
		int n;
		long long int arr[10000],ans,i;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%lld",&arr[i]);
		ans=dp(arr,n);
		printf("Case %d: %lld\n",++j,ans);
	}
}
