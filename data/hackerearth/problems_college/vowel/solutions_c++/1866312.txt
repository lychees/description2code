#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	string s;
	int l,i;
	while(t--)
	
	{
		cin>>s;
		l=s.length();
		int c=0;
		for(i=0;i<=l-1;i++)
		{
			
			if(tolower(s[i])=='i' || tolower(s[i])=='a'||tolower(s[i])=='e'||tolower(s[i])=='u'||tolower(s[i])=='o')
			{
				c++;
				cout<<s[i];
			}
			
		}
		if(c==0)
			{
				cout<<"No"<<'\n';
			}
			else
			{
				cout<<'\n';
			}
		
	}
	return 0;
}
