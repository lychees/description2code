#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int c[3];
    	c[0]=c[1]=c[2]=-1;
    	char str[MAX];
    	scanf("%s",str);
    	long int n;
    	n=strlen(str);
    	long int i;
    	long int ans=0;
    	for(i=0;i<n;i++)
    	{
    		c[str[i]-'x']=i;
    		if(str[i]=='x')
    		{
    			ans+=i-min(c[1],c[2]);
    		}
    		else if(str[i]=='y')
    		{
    			ans+=i-min(c[0],c[2]);
    		}
    		else
    		{
    			ans+=i-min(c[0],c[1]);
    		}
    	}
    	cout<<ans<<endl;
    }
}
