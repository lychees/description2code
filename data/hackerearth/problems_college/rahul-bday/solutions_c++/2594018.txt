#include <iostream>
using namespace std;

int main()
{
       int t;
       long int x,y,z,b,w;
       long long int s,s1,s2;
       cin>>t;
       while(t--)
       {
       	   cin>>b>>w>>x>>y>>z;
       	   s=b*x+w*y;
       	   if(x<y)
       	   {
       	   	  s1=b*x;
       	   	  s2=w*(x+z);
       	   	  if((s1+s2)<s)
       	   	   s=s1+s2;
       	   }
       	   else if(y<x)
       	   {
       	   	 s1=w*y;
       	   	 s2=b*(y+z);
       	   	 if((s1+s2)<s)
       	   	  s=s1+s2;
       	   }
       	   cout<<s<<endl;
       }
       
    return 0;
}
