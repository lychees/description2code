#include <iostream>
#include<string>
using namespace std;

int main()
{
    string a;
    int t;
    cin>>t;
    if(t<1||t>10){
        cout<<"Invalid Test\n";
        }
        else
            {
    while(t--)
    {
        int cs=0,cc=0,csp=0,cn=0;
        cin>>a;
        for(int i=0;i<a.length();i++)
            if((a[i]>='a')&&(a[i]<='z'))
            cs++;
            else if((a[i]>='A')&&(a[i]<='Z'))
                cc++;
            else if((a[i]>='0')&&(a[i]<='9'))
                cn++;
                    else
                    csp++;
                    if((cs==0)&&(cc==0)||a.length()>100)
                        cout<<"Invalid Input\n";

                    else
                        cout<<min(cs,cc)<<"\n";
    }
        }
    return 0;
}
