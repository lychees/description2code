#include <bits/stdc++.h>
using namespace std;

int f(int a[], int d[], int n)
{
   sort(a, a+n);
   sort(d, d+n);
   int p = 1, r = 1;
   int i = 1, j = 0;
   while (i < n && j < n)
   {
      if (a[i] < d[j])
      {
          p++;i++;
          if (p > r) r = p;
      }
      else 
      {
          p--;j++;
      }
   }
 
   return r;
}

int main()
{
    int t,n;
    cin>>t;
    while(t--){
    	cin>>n;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)cin>>a[i];
    	for(int i=0;i<n;i++){
    		cin>>b[i];
    		b[i]+=a[i];
    	}
    	cout<<f(a,b,n)<<endl;
    }
    return 0;
}
