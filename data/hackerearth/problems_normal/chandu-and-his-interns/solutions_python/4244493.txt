import math
def find(n):
    count=0
    temp=int(math.ceil(n**0.5))
    for x in range(2,temp+1):
      if n%x==0:
          if x!=temp:
              count+=2
          else:
              count+=1
      if count>=2:
          break
    if count>=2:
        return False
    else:
        return True
