#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n,m;
    	int dp[300][300],i,j,k;
    	
    	
    	scanf("%d %d",&n,&m);
    	for(i=0;i<=n;i++)
    	{
    		for(j=0;j<=n;j++)
    		if(i==j)
    		dp[i][j]=0;
    		else
    		dp[i][j]=100000000;
    	}
    	for(i=0;i<m;i++)
    	{
    		int a,b,cost;
    		scanf("%d %d %d",&a,&b,&cost);
    		for(j=a;j<b;j++)
    		{
    			for(k=j+1;k<=b;k++)
    			{
    				dp[j][k]=min(dp[j][k],cost);
    			}
    		}
    	}
    	
    	for(k=0;k<=n;k++)
    	{
    		for(i=0;i<=n;i++)
    		{
    			for(j=0;j<=n;j++)
    			{
    				dp[i][j]=min(dp[i][j],dp[i][k]+dp[k][j]);
    			}
    		}
    	}
    	
    	if(dp[0][n]>=100000000)
    	printf("-1\n");
    	else
    	printf("%d\n",dp[0][n]);
    }
    return 0;
}
