#include <iostream>
using namespace std;

int main()
{
    long long int t,n,i;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	long long int a[n],ctr=n;
    	for(i=0;i<n;++i)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;++i)
    	{
    		for(long long j=i+1;j<n;++j)
    		{
    		    if(a[i]==a[j])
    		    ++ctr;
    		}
    	}
    	cout<<ctr<<"\n";

    }
    return 0;
}
