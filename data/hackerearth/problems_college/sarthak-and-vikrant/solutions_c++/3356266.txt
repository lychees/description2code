#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int i,lt,j,ans=0;
    scanf("%d",&lt);
    vector<bool> v(lt+1);
    for(i=4;i<=lt;i+=2)
    	v[i]=true;
    for(i=6;i<=lt;i+=3)
    	v[i]=true;;
    for(i=6;i<=lt;i=i+6)
    {
    	if(!v[i-1])
    	{
    		if((ll)(i-1)*(ll)(i-1)<=lt)
    		{
    			for(j=(i-1)*(i-1);j<=lt;j=j+2*(i-1))
    			{
    				v[j]=true;
    			}
    		}
    	}
    	if(!v[i+1])
    	{
    		if((ll)(i+1)*(ll)(i+1)<=lt)
    		{
    			for(j=(i+1)*(i+1);j<=lt;j=j+2*(i+1))
    			{
    				v[j]=true;
    			}
    		}
    	}
    }
    for(i=2;i<=lt;i++)
    {
    	if(!v[i])
    		ans++;
    }
    ll f=ans;
    f=(f*(f-1))/2;
    printf("%lld",f);
    return 0;
}
