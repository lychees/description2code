#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,q,l,r;
	cin>>n>>l>>r>>q;
	int a[n+1];
	for(int i=0; i<n; i++) cin>>a[i];
	while(q--){
		int x;
		cin>>x;
		int mini=99999;
		for(int i=l; i<=r; i++){
			if(a[i]>x) mini=min(mini,a[i]);
		}
		if(mini<99999) cout<<mini<<endl;
		else cout<<-1<<endl;
	}	
}
