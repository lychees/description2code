#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int dp[101][10001][2]; // [0] profit , [1] amt spent on fertilizing
int a[101], profit[101];
int main() {
	int t;
	cin >> t;
	while(t--){
		int n,p,i,j,k;
		cin >> n >> p;
		for(i = 1; i <= n; i++){
			cin >> j >> k;
			profit[i] = j - k;
			a[i] = k;
		}
		memset(dp,0,sizeof(dp));
		for(i = 1; i <= n; i++){
			for(j = 1; j <= p; j++){
				if(j >= a[i]){ //if I can fertilize the ith plant
					//either I choose to fertilize it, or I do not.
					k = dp[i-1][j-a[i]][0] + profit[i];
					if(dp[i-1][j][0] < k){
						dp[i][j][0] = k;
						dp[i][j][1] = dp[i-1][j-a[i]][1] + a[i];
					}
					else if(dp[i-1][j][0] == k){
						dp[i][j][0] = k;
						dp[i][j][1] = min(dp[i-1][j][1], dp[i-1][j-a[i]][1] + a[i]);
					}
					else{
						dp[i][j][0] = dp[i-1][j][0];
						dp[i][j][1] = dp[i-1][j][1];
					}
				}
				else{
					dp[i][j][0] = dp[i-1][j][0];
					dp[i][j][1] = dp[i-1][j][1];
				}
			}
		}
		cout << dp[n][p][1] << " " << dp[n][p][0] + p << "\n";
	}
	return 0;
}
