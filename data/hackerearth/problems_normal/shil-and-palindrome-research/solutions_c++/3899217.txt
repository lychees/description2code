#include<iostream>
#include<algorithm>
 
using namespace std;
 
int const N = 1e5 + 41;
 
int F[26][N], n, q, l, r, cntodd;
char s[N];
 
void upd(int id, int x, int v){
	while(x < n){
		F[id][x] += v;
		x |= (x+1);
	}
}
 
int get(int id, int x){
	int ret = 0;
	while(x >= 0){
		ret += F[id][x];
		x = (x&(x+1))-1;
	}
	return ret;
}
 
int main(){
	scanf("%d%d",&n,&q);
	getchar();
	for(int i=0;i<n;++i){
		char t = getchar();
		upd(t-'a', i, 1);
		s[i] = t;
	}
	for(int i=0;i<q;++i){
		int t;
		scanf("%d",&t);
		if(t==1){
			char ch;
			int pos;
			scanf("%d %c",&pos, &ch);
			--pos;
			upd(s[pos]-'a', pos, -1);
			upd(ch-'a', pos, 1);
			s[pos] = ch;
		}else{
			int l, r;
			scanf("%d%d",&l,&r);
			--l;
			--r;
			cntodd = 0;
			for(int j=0;j<26;++j){
				if( (get(j, r) - get(j, l-1)) % 2 == 1){
					++cntodd;
				}
			}
			if(cntodd > 1){
				printf("no\n");
				continue;
			}
		
			printf("yes\n");
		}
	}
 
 
 
	return 0;
}
