#include <iostream>
#define MAX 100000
using namespace std;
long long n,a[MAX],q[MAX];
long long f(int i)
{if(i>n-1)
 return 0;
  if(q[i]!=0)
  return q[i];
  return q[i]=max(f(i+2)+a[i],f(i+3)+a[i]);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    { cin>>n;
    
    for(int i=0;i<n;i++)
    q[i]=0;
     for(int i=0;i<n;i++)
     cin>>a[i];
     
     long long x=f(0);
    
     long long y=f(1);
     cout<<max(x,y)<<endl;
     
    }

}
