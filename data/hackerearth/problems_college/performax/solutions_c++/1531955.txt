#include <iostream>
using namespace std;

int main()
{
    int T,D;
    cin>>T;
    cin>>D;
    int a[T];
    long long prev=0;
    for(int i=0;i<T;i++){
    	cin>>a[i];
    }
    int ans=0;
    prev=a[0];
    for(int i=1;i<T;i++){
    	if(a[i]>prev){prev=a[i];continue;}
    	else {
    		
    		if((((prev-a[i])/D)*D)+a[i]>prev){ans+=((prev-a[i])/D);prev=((int)((prev-a[i])/D)*D)+a[i];cout<<"tt";}
    		else {ans+=((prev-a[i])/D)+1;prev=(((prev-a[i])/D)*(D))+a[i]+D;}
    		//cout<<prev<<endl;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
