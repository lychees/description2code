#include<iostream>
using namespace std;

class pattern
{

    int m,n;
    int a[100][100];
 
   public :
         inline void get_data()
          {
            cin>>n>>m;
           }
        inline void get_matrix()
        {
           for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
              cin>>a[i][j];

         }
         void print_pattern()
        {

          	int l,k,i=0;
	      for(int j=0;j<m;j++)
	        {
	          k=j;l=i;
	   
 			    while(l<n && k>=0)
			         {
				        cout<<a[l][k]<<" ";
                         l++;k--;
		        	}
		
	           	   if(j==m-1)
		           {
		               j=j-1;
		               i=i+1;
                    if(i==n)
                     break;	
	               	}
	               	
	               	cout<<"\n";
	        }
	    }

};

int main()
{
	pattern p;

p.get_data();
p.get_matrix();
p.print_pattern();
return 0;
}
