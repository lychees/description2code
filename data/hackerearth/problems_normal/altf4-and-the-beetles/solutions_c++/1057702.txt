#include <bits/stdc++.h>

typedef long long li;
 
using namespace std;
 
#define F(i, n) for(i = 0;i < n; ++i)
#define FI(i, st, ft) for(i = st;i <= ft; ++i)
#define pb(a, b) a.push_back(b)
//#define s second
//#define f first
#define MAXN 100015
#define mk make_pair
#define mod 1000000007
#define inf 1000000000000ll

map <li, li> mp;

int main()
{
	//freopen("input.txt", "r", stdin);
	
	li t, n, a, b, c, i, t1, t2;
	scanf("%lld", &t);
	while(t--){
		mp.clear();
		scanf("%lld%lld%lld%lld", &n, &a, &b, &c);
		F(i, n){
			scanf("%lld%lld", &t1, &t2);
			mp[t1] += b - a;
			mp[t2 + 1] += c - b;
		}
		
		li tt = n * a;
		li ans = 0;
		
		li pre = 1;
		
		map <li, li>::iterator it;
		
		for(it = mp.begin(); it != mp.end(); ++it){
			if(it -> first > pre){
				ans = max(tt, ans);	
			}
			tt += it -> second;
			ans = max(tt, ans);
			pre = it -> first;
		}
		printf("%lld\n", ans);
	}

	return 0;
}
