#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int ar[100],flag;
    for(int i=0;i<n;i++)
    cin>>ar[i];
    for(int i=0;i<n;i++)
    {
    	flag=0;
    	for(int j=0;j<n;j++)
    	{
    		if(i==j)
    		continue;
    		else if(ar[i]%ar[j]==0)
    		{
    			flag=1;
    			break;
    		}
    	}
    	if(flag==0)
    	printf("%d ",ar[i]);
    }
}
