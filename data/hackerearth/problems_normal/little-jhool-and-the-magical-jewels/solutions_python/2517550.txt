for i in range(input()):
    s=str(raw_input())
    r=0
    u=0
    b=0
    y=0
    for k in range (0,len(s)):
        if s[k]=='r':
            r+=1
        if s[k]=='u':
            u+=1
        if s[k]=='b':
            b+=1
        if s[k]=='y':
            y+=1
    print min(r,u,b,y)
