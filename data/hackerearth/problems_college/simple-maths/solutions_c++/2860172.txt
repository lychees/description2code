#include <iostream>
using namespace std;

long long gcd(long long a, long long b)
{
    while(b){
    	long long flag = a;
    	a = b;
    	b = flag % b;
    }
    return a;
}

long long lcm (long long a, long long b){

	long long ans = 1,i = 2;
	bool flag;
	while(a != 1 && b != 1){
		flag = 0;
		if(a % i == 0){
			a /= i;
			flag = 1;
		}
		if(b % i == 0){
			b /= i;
			flag = 1;
		}
		if(flag){
			ans *= i;
			i = 1;
		}
		i++;
		//cout<<a <<"  "<< b <<"  ";
	}
	ans *= (a * b);
	return ans;
}

int main()
{
	long long n,a,b,l;
	int t;
	cin>>t;
	while(t--){
		
		cin>>n>>a>>b;
		if ( a == 0 || b == 0){
			cout<<-1<<endl;
			continue;
		}
		l = (a*b)/gcd(a,b);
		a = n/a;
		b = n/b;
		l = n/l;
		cout<<a + b - l<<endl;
	}
    //cout << "Hello World!" << endl;
    return 0;
}
