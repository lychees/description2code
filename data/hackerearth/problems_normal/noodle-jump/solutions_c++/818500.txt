#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
    int n,k,ans;
    
    scanf("%d%d", &n, &k);
    
    int a[n];
    
    for (int i = 0; i < n; i++) {
    	scanf("%d", &a[i]);
    }
    
    sort(a,a+n);
    
    ans = a[0];
   	
    for (int i = 1; i < n; i++) {
    	if (a[i] - a[i-1] <= k) {
    		ans = a[i];
    	} else {
    		break;
    	}
    }
    
    printf("%d", ans);
    
    return 0;
}
