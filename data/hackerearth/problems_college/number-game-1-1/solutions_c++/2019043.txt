#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    long long int n,count;
    while(t--)
    {
    	count=0;
    	cin>>n;
    	while(n!=1){
    	if(n%3==0)
    	{
    		count++;
    		n/=3;
    	}
    	else if(n%2==0)
    	{
    		count++;
    		n/=2;
    	}
    	else
    	{
    		count++;
    		n--;
    	}
    	}
    	cout<<count<<endl;
    }
}
