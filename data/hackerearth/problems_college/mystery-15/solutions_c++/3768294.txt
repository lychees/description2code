#include <iostream>
#include<math.h>
using namespace std;
int main()
{
    int t, k, a[3], i;
    cin>>t;
    cout<<"\n";
    for(i=0; i<t; ++i)
    {
    	cin>>k;
    	int sum=0;
    	for(int j=0; j<3; ++j)
    	{
    		cin>>a[j];
    		sum+=a[j];
    	}
    	long long int ans=pow(sum,k);
    	cout<<ans<<"\n";
    }
    return 0;
    
}
