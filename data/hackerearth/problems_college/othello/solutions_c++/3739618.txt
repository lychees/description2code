#include<iostream>
#include<string>
#include<algorithm>

using namespace std;

const int MAXN = 1e3 + 10;

string s[MAXN];
int n, m;
int c[MAXN], r[MAXN];

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n >> m;
	for (int i = 0; i < n; i++)	cin >> s[i];
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)	if (s[i][j] == 'B')	r[i]++;
	for (int j = 0; j < m; j++)
		for (int i = 0; i < n; i++)	if (s[i][j] == 'B')	c[j]++;
	int ans = 0;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			if ((r[i] + c[j] - int(s[i][j] == 'B')) & 1)	ans++;
	cout << ans << "\n";
	return	0;
}
