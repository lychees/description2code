def is_prime(N):
    if N == 2:
        return True
    if N % 2 == 0:
        return False
    if N == 1:
        return False
    limit = N**0.5 + 1
    i = 3
    while i <= limit:
        if N % i == 0:
            return False
        i += 2
    return True
