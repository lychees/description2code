#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int q=7;
    	int tim=24*60*60*7;
    	int total=0;
    	vector<int> v;
    	while(q--)
    	{
    	int n,beg,en;
    	cin>>n;
    	total+=n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>beg>>en;
    		int diff=en-beg;
    		v.push_back(diff);
    	}
    	}
    	sort(v.begin(),v.end());
    	int min=ceil(0.75*total);
    	for(int i=0;i<min;i++)
    	{
    		tim-=v[i];
    	}
    	cout<<tim<<"\n";
    }
    return 0;
}
