#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define vec vector
#define pb push_back
#define s(x) scanf("%lld",&x)
#define all(x)	x.begin(),x.end()
#define pii pair<lli,lli>
#define fi first
#define se second
#define SZ 100005 //1e5
lli n,q,x,minD=1000000002;
vec<pii> adj[10002];
lli d[1002];


vec<lli>  dijktras()
{
	vec<lli> dist(10004,1e18);
	dist[0]=0;
	priority_queue< pii > Q;
	Q.push({0,0});
	
	while (!Q.empty())
	{
		pii Top=Q.top();Q.pop();
		
		for(auto to:adj[Top.se])
			if(dist[to.se]>dist[Top.se]+to.fi)
			{
			 dist[to.se]=dist[Top.se]+to.fi;
			 Q.push({-dist[to.se],to.se});	
			}		
	}	
	return dist;
}

int main()
{
	//freopen("input.txt","r",stdin);
	cin>>n>>q;
	
	for(int i=0; i<n; i++)
	{
		s(d[i]);
		minD=min(minD,d[i]);
	}
	
	for(int i=0; i<=minD-1; i++)
	{
		for(int j=0; j<n; j++)
		{
			if (d[j]!=minD)
			{
				int to=(i+d[j])%minD;
				adj[i].pb({d[j],to});
			}
			
		}
		
	}
	vec<lli> dist=dijktras();
	while (q--)
	{
		s(x);
		lli rem=x%minD;
		if (x>=dist[rem])
		printf("YES\n");
		else
		printf("NO\n");
	}
}
