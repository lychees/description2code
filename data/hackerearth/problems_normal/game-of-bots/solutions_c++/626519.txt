#include<cstdio>
int pos[1000005][2];
int main(){
	int t,counta,countb,a,b;
    bool flag;
	char stra[1000005],strb[1000005];
	scanf("%d ",&t);
    while(t--){
        flag=a=b=false;
        scanf("%s %s",stra,strb);
        for(int i=0;stra[i];i++){
            if(stra[i]!='#')    pos[a++][0]=i;
            if(strb[i]!='#')    pos[b++][1]=i;
        }
        if(a!=b)    flag=true;
        else{
            for(int i=0;i<a;i++)
                if(stra[pos[i][0]]==strb[pos[i][1]]){
                    if(stra[pos[i][0]]=='A' && pos[i][0]<pos[i][1]){
                        flag=true;
                        break;
                    }
                    if(stra[pos[i][0]]=='B' && pos[i][0]>pos[i][1]){
                        flag=true;
                        break;
                    }
                }
                else{
                    flag=true;      
                    break;
                }      
        }
        if(flag)    printf("No\n");
        else    printf("Yes\n");
    }
	return 0;
}
