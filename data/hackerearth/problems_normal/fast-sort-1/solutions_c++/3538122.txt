#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--) {
    	string p,s;
    	int a[26];
    	memset(a,0,sizeof(a));
    	cin>>p>>s;
    	int i;
    	for(i=0;i<int(s.size());i++) {
    		a[(s[i]-'a')]++;
    	}
    	for(i=0;i<26;i++) {
    		int in=p[i]-'a';
    		while(a[in]>0) {
    			cout<<p[i];
    			a[in]--;
    		}
    	}
    	cout<<endl;
    }
    return 0;
}
