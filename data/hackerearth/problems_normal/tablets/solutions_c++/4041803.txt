#include <iostream>

using namespace std;

int main() {
    int n;
    cin>>n;
    long answer = 0;
    long priorities[n+2];
    priorities[0] = 1;

    long dp[n + 2];
    dp[0] = 0;
    dp[n+1] = 0;

    for (int i = 1; i <= n; ++i) {
        cin>>priorities[i];
    }
    priorities[n+1] = priorities[n];
    for (int i = 1; i <= n; ++i) {
        if (priorities[i] > priorities[i-1]) {
            dp[i] = dp[i-1] + 1;
        } else {
            dp[i] = 1;
        }
    }

    for (int i = n; i >= 1; --i) {
        if (priorities[i] > priorities[i+1]) {
            if (dp[i+1] + 1 > dp[i]) {
                dp[i] = dp[i + 1] + 1;
            }
        }
    }

    for (int i = 1; i <= n; ++i) {
        answer += dp[i];
    }
    cout<<answer;
}
