#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;

        int arr[n+1];
        int i ,j,cnt;
        for(i=0;i<n;i++)
        {
            cin >> arr[i];
        }
        cnt=0;
        for ( i=0;i<n;i++)
        {
            for(j=i;j<n;j++)
            {
                if ( j >= i && arr[j]==arr[i])
                {
                    cnt++;
                }
            }
        }
        cout <<cnt <<endl;
    }
}
