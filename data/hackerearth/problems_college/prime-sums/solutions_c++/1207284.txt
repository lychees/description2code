#include<iostream>
#include<cmath>
#include<numeric>
#include<vector>
using namespace std;
typedef unsigned long long int ll;
int main(void)
{
	ll n = 0;
	cin>>n;
	vector<bool> v(n+1,true);
	vector<bool>::iterator it;
	vector<ll> p;
	vector<ll>::iterator itl;
	for(ll i=2;i<=n;i++)
	{
		if(v[i]==true)
		{
			p.push_back(i);
			for(ll j=i;i*j<n+1;j++)
				v[i*j] = false;
		}
	}
	ll sum = 0;
	for(itl=p.begin();itl!=p.end();itl++)
	{
		if(*itl<n)
			sum+=*itl;
		else
			break;
	}
	cout<<sum<<endl;
	return 0;
}
