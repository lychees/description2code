#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t, n, k;
    cin>>t;
    while(t--){
    	cin>>n>>k;
    	int a[n];
    	for(int i = 0; i<n; ++i)
    		cin>>a[i];
    	if(k == 0)
    		cout<<"0\n";
    	else if(k >= n){
    		long long sum = 0;
    		for(int i = 0; i<n; ++i)
    			sum += a[i];
    		cout<<sum<<endl;
    	}
    	else{
    		long long sum = 0;
    		sort(a, a+n);
    		for(int i = n-1; i>=n-k; --i)
    			sum += a[i];
    		cout<<sum<<endl;
    	}
    }
    return 0;
}
