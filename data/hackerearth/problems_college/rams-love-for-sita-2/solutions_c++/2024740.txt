#include<bits/stdc++.h>
using namespace std;
int main()
{
   long long int t,n,c,m,i;
   cin>>t;
   while(t--){
    cin>>n>>c>>m;
    long long int sum=0;
    sum=n/c;
    for(i=0;i>=0;i++)
    if(sum-i*m>0)
        sum++;
    else
        break;
    cout<<sum-1<<endl;
   }
   return 0;
}
