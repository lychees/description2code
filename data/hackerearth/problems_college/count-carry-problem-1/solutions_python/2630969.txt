t=input()
i=0
while i<t:
    count=0
    a,b=map(int,raw_input().split(" "))
    while a!=0 and b!=0:
        c=a%10
        d=b%10
        a=a/10
        b=b/10
        e=c+d
        if e>=10:
            count=count+1
        else:
            continue
    if count==1:
            print count,'carry operation'
    elif count>1:
        print count,'carry operations'
    elif count==0:
        print 'No carry operation'
        
    i=i+1
