#print 'Hello World!'
n= long(raw_input());
s=[long(0)]*1000001;

for i in range(0,n):
	b,c=raw_input().split();
	s[long(b)] = long(c);

m=long(raw_input());
r=long(0);
for i in range(999999,-1,-1):
	s[i]=max(s[i],s[i+1]);
	
for i in range(0,m):
	num = long(raw_input());
	r += long(s[num]/num);
				

	
print (r*100);
