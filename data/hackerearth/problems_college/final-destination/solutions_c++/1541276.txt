#include <cstdio>
#include <queue>

#define MAX	1000000

typedef struct
{
	int n;
	int d;
} s_t;

int main(void)
{
	int t;
	s_t x, z;
	int y;
	int m, d, a, s;
	bool mark[MAX + 1];
	std::queue<s_t> q;

	for(scanf("%d", &t); t --;)
	{
		for(m = MAX + 1; -- m;)
			mark[m] = false;

		scanf("%d %d %d %d %d %d", &x.n, &y, &m, &d, &a, &s);

		mark[x.n] = true;

		x.d = 0;
		q.push(x);

		while(!q.empty())
		{
			x = q.front();
			q.pop();

			if(x.n == y)
				break;

			z.d = x.d + 1;

			z.n = x.n + a;
			if(z.n >= 2 && z.n <= MAX && !mark[z.n])
			{
				mark[z.n] = true;
				q.push(z);
			}

			z.n = x.n - s;
			if(z.n >= 2 && z.n <= MAX && !mark[z.n])
			{
				mark[z.n] = true;
				q.push(z);
			}

			z.n = x.n * m;
			if(z.n >= 2 && z.n <= MAX && !mark[z.n])
			{
				mark[z.n] = true;
				q.push(z);
			}

			z.n = x.n / d;
			if(z.n >= 2 && z.n <= MAX && !mark[z.n])
			{
				mark[z.n] = true;
				q.push(z);
			}
		}

		while(!q.empty())
			q.pop();

		if(x.n == y)
			printf("%d\n", x.d);
		else
			puts("-1");
	}

	return 0;
}
