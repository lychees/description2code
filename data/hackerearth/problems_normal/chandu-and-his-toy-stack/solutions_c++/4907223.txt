#include<bits/stdc++.h>
using namespace std;
int a[100005],b[100005];
int main()
{
    int t,n,i,ans,inc,dec;
    cin>>t;
    while(t--)
    {
        cin>>n>>inc>>dec;
        ans=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i]>>b[i];
        }
        sort(a,a+n);
        sort(b,b+n);
        for(i=0;i<n;i++)
        {
            if(a[i]>b[i])
            {
                ans+=(a[i]-b[i])*dec;
            }
            else
                ans+=(b[i]-a[i])*inc;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
