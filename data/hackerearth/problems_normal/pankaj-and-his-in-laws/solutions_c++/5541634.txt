#include <iostream>
#include <vector>
using namespace std;

void i_to_b(int n)
{
	vector<int> s;
	while(n>0)
	{
		s.push_back(n%2);
		n/=2;
	}
	for(int j=s.size()-1;j>=0;j--)
	 cout<<s[j];
}

int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
     cin>>a[i];
    int maxlen=0;
    int len=0;
    int l[101]={0};
    l[0] = -1;
    for(int i=0;i<n;i++)
     {
       	int beg=1,end=maxlen;
       	int mid;
       	while(beg<=end)
       	 {
       	 	mid = (beg+end+1)/2;
       	 	if(a[l[mid]]<a[i])
       	 	 beg=mid+1;
       	 	else
       	 	 end=mid-1;
       	 }
       	 //cout<<beg<<' ';
       	 len = beg;
       	 maxlen=max(len,maxlen);
       	 l[len]=i;
       	 //for(int j=0;j<=n;j++)cout<<l[j]<<' '; cout<<"#"<<len;cout<<endl;
       }
       
     
     i_to_b(maxlen);
    
    return 0;
}
