#include<bits/stdc++.h>
using namespace std;
#define ll long long
long double pi=acos(-1.0);
long double EPS=10e-5;
ll r,s;
long double f(long double mid)  //mid=h
{
    long double R=sqrt(s*s-mid*mid)+(long double)r;
    long double vol=((long double)(pi*mid)/3.0)*(R*R + r*R + r*r );
    return vol;
}
int main()
{
    double lo,hi,mid;
    cin>>r>>s;
    long double mx=-1.0;
    for(long double h=0.0;h<=s+EPS;h+=0.001)
    {
        //cout<<h<<" "<<f(h)<<" "<<mx<<"\n";
        //getchar();
        if(f(h)>mx+EPS)
        {
            mx=f(h);
        }
    }
    cout<<std::fixed;
    cout<<std::setprecision(2)<<mx;
    return 0;
}
