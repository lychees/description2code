#include<bits/stdc++.h>
using namespace std;

long long int dp[100001];

void solve()
{
	
	memset(dp,0,sizeof(dp));
	
	dp[1] = 12;
	
	for(int i = 2;i <= 100000;i++)
	  dp[i] += dp[i-1] + 12;
	
}

main()
{
	
	int t;
	
	scanf("%d",&t);
	
	//solve();
	
	while(t--)
	{
		
		long long int n;
		
		scanf("%lld",&n);
		
		long long int c[27],sum=0;
		
		for(int i = 1;i <= 26;i++)
		{
		      scanf("%lld",&c[i]);
		      
		      sum += c[i];
		}
		   
		long long int ans = 0;
		long long int temp = (12*n)/26;
		
		if(temp)
		    ans += temp*sum + 26*(temp*(temp-1))/2;
		    
		for(int i = 1 ; i <= (12*n)%26 ; i++)
		{
			
			ans += c[i] + temp;
	    
		}   
		
		printf("%lld\n",ans);
		 
	}
}
