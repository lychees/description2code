#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	do
	{
		string a,b;
	cin>>a;
	cin>>b;
	int count=0;
	for(int i=0;i<a.size();i++)
	{
		for(int j=0;j<b.size();j++)
		{
			if(a.at(i)==b.at(j))
		     count++;
		}
	}
   if(count>0)
   cout<<"YES"<<"\n";
   else
   cout<<"NO"<<"\n";
	 t--;
	}while(t>0);
	
}
