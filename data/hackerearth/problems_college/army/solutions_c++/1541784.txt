inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
	int n=scan();
	vector<int> v[10];
	for(int i=1;i<=n;i++){
		int temp=i;
		while(temp){
			int t=temp%10;
			v[t].push_back(i);
			temp/=10;
		}
	}
	int q=scan();
    bool a[100001];
    int count;
	while(q--){
		int m=scan();
		count=0;
		memset(a,0,100001);
		for(int i=0;i<m;i++){
			int x=scan();
			int t=v[x].size();
			for(int j=0;j<t;j++){
                if(!a[v[x][j]])count++;
				a[v[x][j]]=1;
			}
		}
		printf("%d\n",count);
	}
}
