#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {
    	long long s=0,x,y,temp,n,k;
    	scanf("%lld %lld %lld",&x,&y,&n);
    	s+=(x+y);
    	for(i=0;i<n-2;i++)
    	{
    		temp=abs(x-y);
    		x=y;
    		y=temp;
    		s+=temp;
    		if(y==0) break;
    	}
    	k=temp=n-2-i-1;
    	if(k>0){
    	temp/=3;
    	s+=((x+x)*temp);
    	k=k%3;
    	if(k==1) s+=x;
    	else if(k==2) s+=(x+x);
    	}
    	cout<<s<<endl;
    }
    return 0;
}
