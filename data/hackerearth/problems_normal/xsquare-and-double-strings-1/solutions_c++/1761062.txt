#include <iostream>
#include <cstring>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
   	char s[100];
   	cin>>s;
   	int c=0;
   	int j;
   	int n=strlen(s);
   	for(int i=0;i<n;i++)
   	{
   		for( j=i+1;j<n;j++)
   		{
   			if(s[i]==s[j])
   			{c++;
   			break;}
   		}
   		if(s[i]==s[j])
   		break;
   	}
   	if(c==0)
   	cout<<"No"<<endl;
   	else
   	cout<<"Yes"<<endl;
   	
   	
   }
    return 0;
}
