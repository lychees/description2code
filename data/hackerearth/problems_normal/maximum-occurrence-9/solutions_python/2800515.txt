
array=raw_input()
check={}
for eachelement in array:
    if eachelement not in check:
        check[eachelement]=1
    else:
        check[eachelement]+=1
allvalues=[key for key,val in check.iteritems() if val == max(check.values())]

minimumasc=99999
for asci in allvalues:

    ascivalue=map(ord, asci)
    ascivalue=ascivalue[0]
    if ascivalue < minimumasc:
        minimumasc=ascivalue

finalkey= chr(minimumasc)
finalval=check[finalkey]

print finalkey,finalval
