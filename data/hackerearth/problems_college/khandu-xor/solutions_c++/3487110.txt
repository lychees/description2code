#include <iostream>
using namespace std;

int main()
{
	int t;
	cin >> t;
	string s1, s2, s3;
	while(t--){
		cin >> s1;
		cin >> s2;
		s3 = s1 + s2;
		int l = s3.length(), flag1 = 1;
		for(int i = 0; i < l; i++) {
			if(s3[i] != s3[l - i - 1]) {
				flag1 = 0;
				break;
			}
		}
		
		s3 = s2 + s1;
		l = s3.length();
		int flag2 = 1;
		for(int i = 0; i < l; i++) {
			if(s3[i] != s3[l - i - 1]) {
				flag2 = 0;
				break;
			}
		}
		
		if(flag1 == 0 && flag2 == 0) {
			cout << "NO" << endl;
		} else {
			cout << "YES" << endl;
		}
	}
    return 0;
}
