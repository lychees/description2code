#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <functional>
#include <algorithm>

using namespace std;

int alist[26];

int main()
{
	string s;
	int l,count;
	
	cin >> s;

	l = s.length();

	memset(alist,0,26);
	count =0;

	for(int i=0; i < l ; i++)
	{
		if( alist[int(s[i]) - 97] == 0 )
		{
			alist[int(s[i]) - 97 ] = 1;	
		}
	}

	for( int j =0 ; j<26 ; j++)
	{
		count = count + alist[j];
	}

	

	if( count % 2 == 0)
	{
		cout << "CHAT WITH HER! " << endl;
	}
	else
	{
		cout << "IGNORE HIM!" << endl;
	}
	

	return 0;
}
