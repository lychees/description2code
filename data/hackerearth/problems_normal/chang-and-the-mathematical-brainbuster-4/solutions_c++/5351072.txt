int binary_search(int arr[],int l,int r,int key)
{
	if(l==r)
	{
		return l;
	}
	if(r-l==1)
	{
		if(arr[r]-3*key<=0)
			return r;
		else
			return l;
	}
	int mid=(l+r)/2;
	if(arr[mid]-3*key<=0)
		return binary_search(arr,mid,r,key);
	else
		return binary_search(arr,l,mid,key);
}
