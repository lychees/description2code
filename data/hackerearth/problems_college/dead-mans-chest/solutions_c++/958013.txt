#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n;
    string s;
    scanf("%d", &t);
    while ( t-- ) {
        cin >> s;
        n = (int)s.size();
        map <char, int> m;
        map <char, int> :: iterator it;
        for ( int i = 0; i < n; i++ ) m[s[i]]++;
        int cnt = 0;
        for ( it = m.begin(); it != m.end(); it++ ) {
            if ( ((*it).second)%2 != 0 ) cnt++;
        }
        if ( cnt <= 1 ) printf("JACK\n");
        else {
            cnt--;
            if ( cnt%2 == 0 ) printf("JACK\n");
            else printf("DAVY JONES\n");
        }
    }
    return 0;
}
