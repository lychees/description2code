def rev(n):
    return int(n[::-1].rstrip('0'))


cases = int(raw_input())
for z in range(cases):
    sum = 0
    num = raw_input().split(" ")
    for y in range(len(num)):
        sum += rev(num[y])
    print rev(str(sum))
