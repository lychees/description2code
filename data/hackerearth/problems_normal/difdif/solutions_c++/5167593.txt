#include <bits/stdc++.h>
using namespace std;

int main()
{
    string s;
    cin>>s;
    int l,i,j;
    l=s.length();
    set<string> st;
    for(i=0;i<l;i++)
    {
    	string r="";
    	r+=s[i];
    	st.insert(r);
    	for(j=i+1;j<l;j++)
    	{
    		r+=s[j];
    		st.insert(r);
    	}
    }
    cout<<st.size()<<endl;
    return 0;
}
