#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int x,min=2147483647,scnt=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			if(min>x)
				min=x;
			scnt+=x-1;	
		}
		printf("%d %d\n",min-1,scnt);
	}
	return 0;
}
