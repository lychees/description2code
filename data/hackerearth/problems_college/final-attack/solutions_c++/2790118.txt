typedef struct node{
    long long first;
    long long second;
    long long weight;
}Edgey;

Edgey E[100003];

lld bellman(lld x,lld n,lld m){
    lld i,j,u,v,w;
    for(i=1;i<=n;i++) D[i]=INT_MAX;
    D[x]=0;
    for(i=1;i<=n;i++){
        for(j=0;j<m;j++){
            u=E[j].first;
            v=E[j].second;
            w=E[j].weight;
            if(D[v]>D[u]+w){
                if(i==n) return 0;
                D[v]=D[u]+w;
            }
        }
    }
    return 1;
}
