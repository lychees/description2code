#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	int l,i,tc,d,count;
	char str[200];
	cin>>tc;
	while(tc--)
	{
		cin>>str;
		l=strlen(str);
		count=0;
		for(i=0;i<l;i++)
		{
			if(str[i]=='a'|| str[i]=='e'||str[i]=='i'||str[i]=='o'||str[i]=='u')
			count++;
		}
		d=l-(3+count);
		
		cout<<d<<"/"<<l<<endl; 
	}
    
    return 0;
}
