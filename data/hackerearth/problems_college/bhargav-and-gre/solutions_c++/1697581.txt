#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
	int i,cnt=0,cnt2=0,n;
	char x[100],s[100];
	cin>>x;
	
	cin>>n;
	for(i=0;i<n;i++)
	{
		
	cin>>s;
	if(strlen(x)==strlen(s))
	{
	if(s[0]==x[0])
	cnt++;
    else if(x[0]+3==s[0])
	cnt2++;	
	}
	}
	cout<<cnt<<" "<<cnt2;
	
	
	return 0;
}
