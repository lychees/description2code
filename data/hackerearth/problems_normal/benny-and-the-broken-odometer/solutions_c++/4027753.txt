#include <iostream>
using namespace std;

int main()
{
	int n ,x;
	cin >>n ;
	for(int i = 0 ;i<n;i++){
		cin >> x ;
		int ans = 0 ;
		int j = 1 ;
		while((x-(x%10))/10!=0){
			if(x%10>3) ans+=((x%10)-1)*j ;
			else ans+= (x%10)*j ;
			x = (x-(x%10))/10 ;
			j*=9;
		}
        ans+= (x%10)*j;
        if(x%10>3) ans-=(j);
		cout << ans<<endl ;
	}
	    return 0;
}
