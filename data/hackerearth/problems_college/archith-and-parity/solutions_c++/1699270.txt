#include<iostream>
    using namespace std;
int main()
{
    int i,j,k,T,n,count;
    cin>>T;
    while(T--)
    {
        cin>>n;
        count=0;
        while(n>=1)
        {
            if(n%2==1)
                count++;
            n/=2;
        }
        if(count%2==0)
            cout<<"even"<<endl;
        else
            cout<<"odd"<<endl;
    }
    return(0);
}
