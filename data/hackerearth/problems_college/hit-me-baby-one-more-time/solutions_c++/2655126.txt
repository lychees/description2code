#include<iostream>
#include<cmath>
#include<cstdlib>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int p = (n-1)/3;
	long long int q = (n-1)/5;
	long long int r = (n-1)/15;
	long long int sum1 = (2*p*3 + p*(p-1)*3)/2;
	long long int sum2 = (2*q*5 + q*(q-1)*5)/2;
	long long int sum3 = (2*r*15 + r*(r-1)*15)/2;
	cout<<sum1+sum2-sum3;
}
