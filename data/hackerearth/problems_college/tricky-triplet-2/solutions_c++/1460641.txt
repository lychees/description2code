#include<iostream>
using namespace std;

int main()
{
	int n,m=0;
	int z[100];
	cin>>n;
	for(int i=0;i<n;i++)
	cin>>z[i];
	if(n==3&&(z[1]-z[0])==(z[2]-z[1]))
	{m=1;
		}
		else{
	for(int j=0;j<n;j++)
	{
		for(int k=j+1;k<n;k++)
		{
			for(int l=k+1;l<n;l++)
			{   if((z[k]-z[j])==(z[l]-z[k]))
			    {
			    	m++;
				}
				
			}
		}
	}
}
	cout<<m;
	return(0);
}
