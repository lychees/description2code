#include <iostream>
using namespace std;

bool cities[10001];

int main()
{
	int e, x, y, t;
	
	for (cin >> t; t--;)
	{
		cin >> e;
		
		for (int i = 0; i < 10001; ++i) cities[i] = false;
		
		for (int i = 0; i < e; ++i)
		{
			cin >> x >> y;
			
			if (cities[x] == false) cities[x] = true;
			if (cities[y] == false) cities[y] = true;
		}
		
		int count = 0;
		for (int i = 1; i < 10001; ++i) if (cities[i] == true) ++count;
		
		cout << count << endl;
	}
	
	return 0;
}
