#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		long int i=0;
		char m[26];
		long int no[26];
		while(i<26)
		{
			m[i]=char(122-i);
			no[i]=0;
			i++;
		}
		i=0;
		long int n=s.length();
		while(i<n)
		{
			no[25-(s[i]-'a')]++;
			i++;
		}
		
		i=0;
		int k=26,j=0;;
		while(i<26)
		{
			j=0;
			while((j+1)<k)
			{
				if(no[j] > no[j+1])
				{
					swap(no[j],no[j+1]);
					swap(m[j],m[j+1]);
				}
				j++;
			}
			i++;
		}
		i=0;
		while(i<26)
		{
			cout<<(m[i])<<" ";
			i++;
		}
		cout<<endl;
	}
}
