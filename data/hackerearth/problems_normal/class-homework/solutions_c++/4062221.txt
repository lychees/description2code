#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
long long int tc,n,sum,i,j,k,a,b,c,d,x,y,maxi=1e12;;
cin>>tc;
while(tc--)
{
	string s;
	cin>>s;
	sum=INT_MIN;
	n=s.size();
	
	if(n<4)
	cout<<"unlucky\n";
	
	else
	{
	for(i=0;i<=n-4;i++)
	{
		for(j=i+1;j<=n-3;j++)
		{
			for(k=j+1;k<=n-2;k++)
			{
				a=b=c=d=0;
				int flag=0;
				for(x=0;x<=i;x++)
				{
					
					a=a*10+(s[x]-'0');
				
				}
				
				for(x=i+1;x<=j;x++)
				{
					
					b=b*10+(s[x]-'0');
				
				}
				
				for(x=j+1;x<=k;x++)
				{
					
					c=c*10+(s[x]-'0');
						
				}
				
				for(x=k+1;x<=n-1;x++)
				{
					
					d=d*10+(s[x]-'0');
						
				}
				
				if(i>=1&&s[0]=='0')flag=1;
    				if(j-i>1&&s[i+1]=='0')flag=1;
    				if(k-j>1&&s[j+1]=='0')flag=1;
    				if(n-1-k>1&&s[k+1]=='0')flag=1;
    				if(a>maxi||b>maxi||c>maxi||d>maxi)flag=1;
    				if(flag==1)
    				continue;
    				//cout<<a<<" "<<b<<" "<<c<<" "<<d;
    				sum=max(sum,a+b+c+d);
			}
		}
	}

    if(sum==INT_MIN)
    cout<<"unlucky\n";
    
	else
	cout<<sum<<"\n";
	}
}
    return 0;
}
