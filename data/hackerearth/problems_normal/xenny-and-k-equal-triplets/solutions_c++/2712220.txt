#include <iostream>

using namespace std;
long int gcd(long int n,long int m)
{
	int gcd, remain;
 
	while (n != 0)
	{
		remain = m % n;
		m = n;
		n = remain;
	}
 
	gcd = m;
 
	return gcd;
}
long int ncthree(long int n){
	
	return(n*(n-1)*(n-2)/6)	;
}
int main()
{
	int t;
	long int n,k,numbers,num1,num2,gcf,countk,i;
	
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&k);
		countk=0;
		for (i=0;i<n;i++){
        scanf("%d",&numbers);
        if(numbers==k){
			countk++;
		}
	    }
			num1=ncthree(countk);
			num2=ncthree(n);
			gcf=gcd(num1, num2);
			num1=num1/gcf;
			num2 = num2/gcf;
			cout << num1<<"/"<<num2<<endl;		
	}
    return 0;
}
