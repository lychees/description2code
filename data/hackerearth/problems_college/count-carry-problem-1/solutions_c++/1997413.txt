#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    long int t,i,j,x,y;
    char a[100099];
    char b[100099];
    cin>>t;
    while(t--)
{
    cin>>a;
    cin>>b;
    x=strlen(a);
    y=strlen(b);
    long int flag=0,count=0;
    for(i=0;i<x;i++)
    {
    	j=i;
    	a[i]=a[i]-48;
    	b[i]=b[i]-48;
    	//cout<<a[i]+b[i]<<endl;
    	if(a[i]+b[i]>=10&&flag==0)
    	{
    		count++;
    		flag=(a[i]+b[j])/10;
    	}
    	else if(a[i]+b[j]+flag>10||a[i]+b[j]>=10)
    	{
    		count++;
    		flag=(a[i]+b[j]+flag)/10;
    	}
    	else if(a[i]+b[j]+flag<=10)
    	{
    		flag=0;
    	}
    }
    if(count==1)
    cout<<count<<" carry operation"<<endl;
    else if(count>1)
    {
    	cout<<count<<" carry operations"<<endl;
    }
    else
    cout<<"No carry operation"<<endl;
}
    return 0;
}
