#include<iostream>
#include<set>
using namespace std;
long long facts[1000007];
const long long MOD = 1000000007;
int main()
{
	int n, m;
	cin >> n >> m;
	facts[0] = 1;
	for (int i = 1; i <= n*m; i++)
	{
		facts[i] = (facts[i - 1] * i) % MOD;
	}
	multiset<int>s;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			s.insert(i^j);
		}
	}
	long long ans = 1;
	set<int>t;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			if (!t.count(i^j))
			{
				ans *= facts[s.count(i^j)];
				ans %= MOD;
			}
			t.insert(i^j);
		}
	}
	ans %= MOD;
	cout << ans << endl;
	return 0;
}
