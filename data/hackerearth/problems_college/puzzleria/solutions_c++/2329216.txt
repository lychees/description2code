#include<bits/stdc++.h>
using namespace std;
long long fact(int n)
{ if(n==0) return 1;
  else return (n*fact(n-1));
}
int main()
{ int t;long long int x;
cin>>t;
 while(t--)
 { int a,b;
   cin>>a>>b;
 x=(fact(7)*fact(5)*fact(a+b))/(fact(a)*fact(b)*fact(7-a)*fact(5-b));
 cout<<x<<endl;
 }


}
