#include <iostream>
using namespace std;
int main()
{
    int t; cin>>t;
    while(t--)
    {
        int N,M; cin>>N>>M;
    	int T=0;
    	int Ph=0; int Pdh=M;
    	int Ch=1; int Cdh=N-1;
    	// Ph pendrive have, Pdh pendrive dont have, Ch computer have, Cdh computer dont have,
    	while(Pdh && Cdh)
    	{
    		// cCh temp Computer have, cPh temp Peddrive have
    		T++;
    		int cPh = Ph + min(Pdh,Ch);
    		int cCh = Ch + min(Cdh,Ph);
    		Pdh -= min(Pdh,Ch);
    		Cdh -= min(Cdh,Ph);
    		Ph=cPh;
    		Ch=cCh;
    	}
    	if(Cdh)
    		T += (1 + ((Cdh-1)/Ph));
    	cout<<T<<endl;
    }
    return 0;
}
