#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
    	int n;
    	cin>>n;
    	int a[n];
    	for(int i=0;i<n;i++)
    	cin>>a[i];
    	sort(a,a+n);
    	int count=0;
    	for(int i=0;i<n;i++)
    	{
    		count++;
    		if(i!=n-1 && a[i+1]==a[i]+1)
    		i++;
    	}
    	cout<<count<<endl;
    }
    return 0;
}
