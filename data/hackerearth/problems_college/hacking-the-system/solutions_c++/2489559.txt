#include <bits/stdc++.h>
using namespace std;

int main()
{
	int cnt=0;
    char num1[900000],num2[900000];
    cin>>num1>>num2;
    int l1=strlen(num1);
    int l2=strlen(num2);
    if(abs(l1-l2)>1)
    {
    	cout<<"loses"<<endl;
    }
    else
    {
    	for(int i=0;num1[i]!='\0';i++){
    		if(num1[i]!=num2[i])
    			cnt++;
    	}
    }
    
    if(cnt>1)
    	cout<<"loses"<<endl;
    else
    	cout<<"wins"<<endl;
    return 0;
}
