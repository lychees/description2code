#include <iostream>
using namespace std;
#define ll long long 
int main()
{
    int t, n, m, x;
    scanf("%d", &t);
    while(t--){
    	scanf("%d %d %d", &n, &m, &x);
    	ll sum=0, a;
    	for(int i=0; i<n; i++){
    		cin>>a;
    		sum+=a;
    	}
    	ll req = (ll)x*(n+1);
    	ll d = req - sum;
    	if(d<=m){
    		cout<<max(1ll, d)<<"\n";
    	}
    	else{
    		cout<<"Impossible\n";
    	}
    }
    return 0;
}
