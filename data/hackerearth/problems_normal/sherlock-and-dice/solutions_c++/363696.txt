#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <bits/stdc++.h>
#include <string.h>
#include <sstream>
#include <math.h>
#include <limits.h>
#include <assert.h>

using namespace std;

double hashe[51][2505];
int main()
{
	int t;
	cin>>t;
	assert(t<=10);
	int n,m,k;
		
	
	while(scanf("%d%d%d",&m,&n,&k)!=EOF)
	{
		assert(m<=50 && n<=50);
	
		if(k>n*m || k <m)
		printf("0.000 0\n");
		
		else
		{
			for(int i=1;i<=m;i++)
			 for(int j=1;j<=n*m;j++)
			  hashe[i][j]=0;
			  
			for(int i=1;i<=n ;i++)
			 hashe[1][i]=1;
			 
			// all things have occured once
			for(int i=2;i<=m;i++)
			{
			  for(int j=1;j<=n;j++)
			  {
			  	for(int x=1;x+j<=n*m;x++)
			  	  hashe[i][x+j]+=hashe[i-1][x];
			  }
			}
			double ans=hashe[m][k];
			
			//printf("%lf--\n",ans);
			
			for(int i=1;i<=m;i++)
			 ans/=n;
			 
			 int count=0;
			 
			 if(ans>0)
			 {
			 while(ans<1.000)
			 {
			  ans*=10;
			  count++;
			 }
			 }
			 printf("%.3lf %d\n",ans,count);
		}
	}
		
	return 0;
}
