#include <iostream>
#include<math.h>
#include<stdio.h>
#include<string>
#include<string.h>
using namespace std;

int main() 
{
	
	long long int t,flag,r,x,y,a[4][2];
	double d;
	cin>>t;
	while(t--)
	{
		cin>>r>>x>>y;
		for(int i=0;i<4;i++)
		cin>>a[i][0]>>a[i][1];
		flag=0;
		for(int i=0;i<4;i++)
		{
			d=sqrt((x-a[i][0])*(x-a[i][0])+(y-a[i][1])*(y-a[i][1]));
			
			if(d>r)
			{
				flag=1;
				break;
			}
			
		}
		if(flag==0)
		cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
		
	}
	
	
	
	return 0;
}
