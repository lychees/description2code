#include<bits/stdc++.h>
using namespace std;
int n;
const int MAXN = 1e3+5;
int dp[MAXN][MAXN];
int ar[MAXN];
int go(int idx, int previ) {
	if(idx==n+1) return 0;
	if(dp[idx][previ]!=-1) return dp[idx][previ];
	int ret=0;
	ret=max(ret,go(idx+1,previ));
	if(previ==0 or ar[idx]%ar[previ]==0) ret=max(ret,1+go(idx+1,idx));
	return dp[idx][previ]=ret;
}

int main() {
	cin>>n;
	for(int i=1;i<=n;i++) scanf("%d",&ar[i]);
	sort(ar+1,ar+1+n);
	memset(dp,-1,sizeof(dp));
	int ans=go(1,0);
	if(ans<2) ans=-1;
	cout<<ans<<endl;
	return 0;
}
