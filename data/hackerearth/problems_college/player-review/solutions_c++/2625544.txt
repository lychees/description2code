#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	long long a[n];
	int i;int c=0;
	for(i=0;i<n;i++)
	scanf("%lld",&a[i]);
	for(i=0;i<n;i++)
	{
		if(a[i]>0)
		c++;
		
	}
	if(c==0)
	printf("0\n");
	else
	{
		long long s=0,sum=0;
		for(i=0;i<n;i++)
		{
			sum=sum+a[i];
			if(sum<0)
			{
				sum=0;
			}
			s=max(s,sum);
		}
		printf("%lld\n",s);
	}
	return 0;
}
