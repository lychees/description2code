#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
    long int n,num,item;
    cin>>n;
    num=n;
    vector <long int> v1(n),v2(n);
    vector <long int>::iterator itr1=v1.begin(),itr2=v2.begin();
    while(n--){
        cin>>item;
        *itr1=item;
        itr1++;
    }
    n=num;
    while(n--){
        cin>>item;
        *itr2=item;
        itr2++;
    }
    sort(v1.begin(),v1.end());
    sort(v2.begin(),v2.end());
    itr1=v1.begin();
    itr2=v2.begin();
    n=num;
    long long sum=0;
    while(n--){
        sum+=(*itr1)*(*itr2);
        ++itr1;
        ++itr2;
    }
cout<<sum;
}
