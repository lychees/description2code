int LCM(int m, int n)
	{
     int a=m;
    int  b=n; 
      while (a != b)  
          {if (a < b)  
               a= a + m;  
           else  
            b= b + n;
          }
    return a;
    }

int isPrime(int n)
   {
    int i;
    if(n==1)
        return 0;
    if (n==2)
        return 1;

    if (n%2==0)
        return 0;

    for (i=3;i<=sqrt(n);i+=2)
       { if (n%i==0)
            return 0;
       }
    return 1;
   }
