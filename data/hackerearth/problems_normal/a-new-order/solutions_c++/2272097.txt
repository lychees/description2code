#include<bits/stdc++.h>

using namespace std;
int check(string s1,string s2)
{
	int x=s1.length(),y=s2.length();
	x=x>y?y:x;
	for(int i=0;i<x;i++) if(s1[i]!=s2[i]) return i;
	return -1;
}
set<char> a[130],s,p;
int main()
{

	for(int i=2;i<=n;i++) {
		cin>>s2;
		xy = s2.length();
		for(int j=0;j<xy;j++) p.insert(s2[j]);
		int x=check(s1,s2);
		if(x!=-1) {
			a[s2[x]].insert(s1[x]);
		}
		s1=s2;
	}
	//for(auto &i:p) cout<<i;cout<<endl;
	while(1) {
		s.clear();
		for(auto &i:p) {
			if(a[i].empty()) s.insert(i);
		}
		for(auto &it:s) {
