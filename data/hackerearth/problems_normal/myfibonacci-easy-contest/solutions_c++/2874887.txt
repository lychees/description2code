#include <bits/stdc++.h>
#define ll long long int

using namespace std ;

int f[11] ;

void compute(int A , int B)
{ 
	
	f[0] = A ;
	f[1] = B ;
	
	for(int i = 2 ; i < 10 ; i++)
	 f[i] = f[i - 1] + f[i - 2] ;
}

int main()
{
	int A , B , N ;
	
	cin >> A >> B >> N ;
	
	compute(A , B) ;
	
	cout << f[N - 1] << endl ;
	
	return 0 ;
}
