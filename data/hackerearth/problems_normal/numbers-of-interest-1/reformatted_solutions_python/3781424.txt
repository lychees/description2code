test_cases = int(raw_input(""))
result = []
inputs = []
temp = []
for i in range(test_cases):
    inputs = map(int, raw_input("").split(" ")[:3])
    temp = [inputs[1] for j in range(inputs[0])]
    if inputs[2] > inputs[0]:
        for j in range(inputs[2]-inputs[0]):
            new = sum(temp[k] for k in range(inputs[0]))
            temp.pop(0)
            temp.append(new)
    else:
        new = temp[inputs[2]-1]
    result.append(new)
for each in result:
    print each
