#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll arr[10000],arrm[10000],arrf[10000];
ll root(ll i)
{
    while(i!=arr[i])
        i=arr[arr[i]];
    return i;
}
ll root1(ll i)
{
    while(i!=arrm[i])
        i=arrm[arrm[i]];
    return i;
}
ll root2(ll i)
{
    while(i!=arrf[i])
        i=arrf[arrf[i]];
    return i;
}
main()
{
    ll n,m,a,b,c,ans=0,i;
    cin>>n>>m;
    for(i=1;i<n+1;i++)
    {
        arr[i]=i;
        arrm[i]=i;
        arrf[i]=i;
    }
    pair<ll,pair<ll,ll> > p[m];
    for(i=0;i<m;i++)
    {
        cin>>a>>b>>c;
        p[i]=make_pair(c,make_pair(a,b));
    }
    for(i=0;i<m;i++)
    {
        if(p[i].first==3)
        {
            if(root(p[i].second.first)!=root(p[i].second.second))
            {
                ++ans;
                arr[root(p[i].second.first)]=root(p[i].second.second);
                arrm[root1(p[i].second.first)]=root1(p[i].second.second);
                arrf[root2(p[i].second.first)]=root2(p[i].second.second);
            }
        }
    }
    for(i=0;i<m;i++)
    {
        if(p[i].first==1)
        {
            if(root1(p[i].second.first)!=root1(p[i].second.second))
            {
                ++ans;
                arrm[root1(p[i].second.first)]=root1(p[i].second.second);
            }
        }
    }
    a=root1(1);
    for(i=2;i<n+1;i++)
    {
        if(root1(i)!=a)
        {
            cout<<-1<<endl;
            return 0;
        }
    }
    for(i=0;i<m;i++)
    {
        if(p[i].first==2)
        {
            if(root2(p[i].second.first)!=root2(p[i].second.second))
            {
                ++ans;
                arrf[root2(p[i].second.first)]=root2(p[i].second.second);
            }
        }
    }
    a=root2(1);
    for(i=2;i<n+1;i++)
    {
        if(root2(i)!=a)
        {
            cout<<-1<<endl;
            return 0;
        }
    }
    cout<<m-ans<<endl;
    return 0;
}
