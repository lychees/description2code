#include <iostream>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int k,r,n;
    	scanf("%d%d",&n,&k);
    	k++;
    	r=k/n;
    	if(k%n!=0)
    	r++;
    	k=k%n;
    	if(k%n==0)
    	k=n;
    	k--;
    	
    	if(k==r)
    	printf("0\n");
    	else
    	printf("1\n");
    	
    }
    return 0;
}
