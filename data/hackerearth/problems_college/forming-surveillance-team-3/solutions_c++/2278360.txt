#include<bits/stdc++.h>
using namespace std;
int main()
{
    int test;
    long int n,num;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%ld",&num);
        set<long int> st;
        for(long int i=0;i<num;i++)
        {
            scanf("%ld",&n);
            st.insert(n);
        }
        cout<<st.size()<<endl;
    }
}
