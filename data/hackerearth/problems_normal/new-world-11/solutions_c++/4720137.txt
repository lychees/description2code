#include <iostream>
using namespace std;
 
int find(int x[],int r,int n,int k){
	int i,p=0,q=0,a=0;
	for(i=0;i<n;i++){
		while(p<n&&x[p]-x[q]<=r)
			p++;
		a++;
		if(p==n)
			break;
		q=p-1;
	}
	if(p!=n)
		return 0;
	if(a<=k)
		return 1;
	return 0;
}
int main()
{
	int t;cin>>t;
	while(t--){
    	int n,k;cin>>n>>k;
    	int x[n],i;
    	for(i=0;i<n;i++)	cin>>x[i];
    	int mid,s=x[n-1]-x[0],left=0,right=s;
    	while(left<right){
    		mid=left+(right-left)/2;
    		int ret=find(x,mid,n,k);
    		if(ret==1)
    			right=mid;
    		else
    			left=mid+1;
    	}
    	cout<<right<<"\n";
	}
    return 0;
}
