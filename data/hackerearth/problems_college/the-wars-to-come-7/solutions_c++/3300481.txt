#include <bits/stdc++.h>

using namespace std;

int main()
{
    long int n,d,i,a[100000],j,sum=0,c = 0;
    scanf("%ld",&n);
    for(i=0;i<n;i++)
    {
    	scanf("%ld",&a[i]);
    }
    scanf("%ld",&d);

    sort(a,a+n);
    for(i=n; i >= 0; i--)
    {
    	sum = sum + a[i];
    	c++;
    	if(sum >= d)
    	{
    	    c = c-1;
    	break;
    	}
    }
    printf("%ld",c);
    return 0;
}
