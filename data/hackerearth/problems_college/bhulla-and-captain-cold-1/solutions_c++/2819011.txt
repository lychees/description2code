#include<bits/stdc++.h>
#define sc(n) scanf("%d",&n)
#define rep(i,a,b) for(i=a;i<b;i++)
#define ll long long
#define pr(n) printf("%lld\n",n)
using namespace std;
int ct[1000000][3];
ll int dp[1000000][3];
int main()
{
    int t,n,i,j;
    sc(t);
    while(t--)
    {
        sc(n);
        rep(i,0,n)
        {
            sc(ct[i][0]);
            sc(ct[i][1]);
            sc(ct[i][2]);
        }
        rep(i,0,3)
            dp[0][i]=ct[0][i];
        rep(i,1,n)
        {
            rep(j,0,3)
                dp[i][j]=min(dp[i-1][(j+1)%3],dp[i-1][(j+2)%3])+ct[i][j];
        }
        pr(min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2])));
    }
    return 0;
}
