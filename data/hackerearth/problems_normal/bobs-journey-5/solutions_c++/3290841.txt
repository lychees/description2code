#include<iostream>
using namespace std;

int main()
{
int t,count,flag,n	;
char c;
cin>>t;

for(int z=0;z<t;z++)
{
count=flag=0;
cin>>n;
string arr[n];
for(int i=0;i<n;i++)
	cin>>arr[i];

char imp[n];
for(int i=0;i<n;i++)
	imp[i]=arr[i][0];
	
for(int i=0;i<n;i++)
{
	c=imp[i];
	count=0;
	for(int j=0;j<n;j++)
	{
		if(imp[j]==c)
			count++;
	}
	if(count!=1)
	{
		cout<<"NO"<<endl;
		flag=1;
		break;
	}
}
if(flag!=1)
{
	cout<<"YES"<<endl;
}


}

return 0;
}
