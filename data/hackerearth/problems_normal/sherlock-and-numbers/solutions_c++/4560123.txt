#include<bits/stdc++.h>
#define FOR(i,j,k) for(int i=j;i<=k;i++)
#define N 1000005
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		int n,k,p;
		int a[N];
		scanf("%d %d %d",&n,&k,&p);
		FOR(i,1,k) {
			scanf("%d",&a[i]);
		}
		sort(a+1,a+1+k);
		FOR(i,1,k) {
			if(a[i]<=p) {
				p++;
			}
		}
		printf("%d\n",p>n?-1:p);
	}
	
}
