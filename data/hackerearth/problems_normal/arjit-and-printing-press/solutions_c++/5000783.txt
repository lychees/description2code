#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    string w,r;
    while(t--)
    {	int j=0;
    	cin>>w>>r;
    	int v=r.size();
    	 int hs[v];
    	 for(i=0;i<r.size();i++)
    	 	hs[i]=0;
    	sort(r.begin(),r.end());
    	for(i=0;i<w.length();i++)
    	{
    		if(r[j]<w[i] && j<r.size())
    			{w[i]=r[j];
    			hs[j]=1;
    			}
    			if(hs[j]>0)
    			j++;
    		if(j>=r.size())
    			break;
    	}
    	cout<<w<<endl;
    }
    return 0;
}
