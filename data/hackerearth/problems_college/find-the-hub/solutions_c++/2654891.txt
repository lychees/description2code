int check(int a)
{
    int til=a/2;
    for(int i=2;i<=til;i++)
        if((a%i)==0) return 0;
    return 1;
}
int main()
{
    int n;
    si(n);
    int a[n];
    REP(i,n)
        si(a[i]);
    int ans=0;
    REP(i,n)
    {
        if(check(a[i]) && a[i]>ans) {ans=a[i];}
    }
    int ct=0;

    }
