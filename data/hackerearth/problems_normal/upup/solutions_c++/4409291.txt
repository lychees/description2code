#include <iostream>
using namespace std;

bool lowercase(char ch) {
	int c = (int)ch;
	if(c >= 97 && c <= 123) {
		return true;
	}
	return false;
}
int main()
{
    string str;
    getline(cin, str);

    if(lowercase(str[0])) {
    	str[0] = (char)((int)str[0] - 32);
    }

    for (int i = 1; i < str.size(); i++) {

    	if (str[i] == ' ' && lowercase(str[i + 1])) {
    		str[i + 1] = (char)((int)str[i + 1] - 32);
    	}
    }
    cout << str << endl;
    return 0;
}
