#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,a,b;
	cin>>n;
	while(n--)
	{
		long count=0;
	cin>>a>>b;
	while(a&&b)
	{
	if(((a%10)+(b%10))>9)
	count++;
	a=a/10;
	b=b/10;
	}
	if(count==0)
	cout<<"No carry operation"<<endl;
	else if(count==1)
	cout<<"1 carry operation"<<endl;
	else
	cout<<count<<" carry operations"<<endl;
	}
return 0;
}
