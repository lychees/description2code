#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
#define F(i,a,b) for(int i = (int)(a); i <= (int)(b); i++)
#define RF(i,a,b) for(int i = (int)(a); i >= (int)(b); i--)
int A[105][105];
int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
        freopen("output.txt","w",stdout);
    #endif
    int T,N,M,nC;
    scanf("%d",&T);
    while(T--)
    {
    	scanf("%d%d",&N,&M);
    	char S[105];
    	nC = 0;
    	memset(A,0,sizeof(A));
    	F(i,1,N)
    	{
    		scanf("%s",S);
    		F(j,1,M)
    		{
    			A[i][j] = S[j-1] - '0';
    			if(S[j-1] == '1') nC++;
    		}
    	}
    	if(nC == 0)
    	{
    		printf("no 2\n");
    		continue;
    	}
    	if(nC == 1)
    	{
    		printf("no 1\n");
    		continue;
    	}
    	bool flag = 0;
    	F(i,1,N)
    	{
    		F(j,1,M)
    		{
    			if(A[i][j] == 1)
    			{
    				if(A[i+1][j] == 1 || A[i-1][j] == 1 || A[i][j-1] == 1 || A[i][j+1] == 1)
    				{
    					flag = 1;
    				}
    			}
    		}
    	}
    	if(flag == 1) printf("yes 0\n");
    	else printf("no 1\n");
    }
    return 0;
}
