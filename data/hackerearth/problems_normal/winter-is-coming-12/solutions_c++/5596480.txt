#include<bits/stdc++.h>
#define int long long
using namespace std;

vector<int> v[100005];
int n,t,c[100005],a[100005];

 main()
{
    cin>>t;
    while(t--)
    {

cin>>n;
for(int i=0;i<n;i++)
{cin>>a[i];
c[i]=a[i];
if(i>0)c[i]+=c[i-1];
}

for(int i=0;i<n;i++)
v[i].clear();

for(int i=0;i<n;i++)
{
    c[i]%=n;
    //cout<<c[i]<<endl;
    v[c[i]].push_back(i);
}

int l,r,mn=INT_MAX;

int rx;

if(v[0].size()>0){l=-1;r=v[0][0];mn=r-l;}

for(int i=1;i<n;i++)
{
    rx=v[i].size();
    --rx;
    for(int j=0;j<(rx);j++)
    {
        //cout<<j<<" "<<v[i].size();
        int d=v[i][j+1]-v[i][j];
       // cout<<d<<" "<<i<<" "<<j<<" "<<r<<endl;
        if(d<mn)
        {
            mn=d;l=v[i][j];r=v[i][j+1];
        }
        else if(d==mn&&l>v[i][j])
        {
            l=v[i][j];r=v[i][j+1];
        }
//cout<<"DC";

    }
}
cout<<l+2<<" "<<r+1<<endl;

//cout<<endl;

}
return 0;
}
