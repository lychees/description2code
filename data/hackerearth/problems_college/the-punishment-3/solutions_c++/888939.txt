#include <iostream>
#include <stdio.h>

int main()
{
    int t,n,k,i,a,b,l;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d",&n,&k);
    	a=k;
    	b=n-k;
    	if(a==b)
    	{
    		for(i=1;i<=a-1;i++)
    			printf("9");
    		printf("8");
    		for(i=1;i<=a-1;i++)
    			printf("0");
    	}
    	else if(a==1||b==1)
    	{
    		printf("8");
    		for(i=2;i<=n-1;i++)
    			printf("9");	
    	}
    	else
    	{
    		if(a>b)
    		{
    			for(i=1;i<=b-1;i++)
    				printf("9");
    			printf("8");
    			l=a-b;
    			while(l--)
    				printf("9");
    			for(i=1;i<=b-1;i++)
    				printf("0");
    		}
    		else
    		{
    			for(i=1;i<=a-1;i++)
    				printf("9");
    			printf("8");
    			l=b-a;
    			while(l--)
    				printf("9");
    			for(i=1;i<=a-1;i++)
    				printf("0");
    		}
    	}
    	printf("1\n");
    }
    return 0;
}
