#include <bits/stdc++.h>
 
using namespace std;

//#define LOCAL

int mat[1002][1002];
int rsum[1002],csum[1002];

int main(){
    #ifdef LOCAL
        freopen("input.txt","r",stdin);
		clock_t start1=clock();
	#endif
	int n,m,i,j;
    	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			scanf("%d",&mat[i][j]);
	for(i=0;i<n;i++){
		rsum[i]=0;
		for(j=0;j<m;j++){
			rsum[i]+=mat[i][j];
		}
	}
	for(j=0;j<m;j++){
		csum[j]=0;
		for(i=0;i<n;i++){
			csum[j]+=mat[i][j];
		}
	}
	int mx=-1;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			mx=max(mx,rsum[i]^csum[j]);
		}
	}
	printf("%d\n",mx);
	#ifdef LOCAL
		printf("%.3lf seconds\n",(double)(clock()-start1)/CLOCKS_PER_SEC);
	#endif
	return 0;
}
