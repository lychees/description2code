#include <bits/stdc++.h>
using namespace std;
int func(int k)
{
	int cnt=0;
	while(k)
	{
		k=k&(k-1);
		cnt++;
	}
	return cnt;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k;
    	cin>>n>>k;
    	int ans[n];
    	for(int i=0;i<n;i++)
    	{
    		int k;cin>>k;
    		ans[i]=func(k);
    	}
    	sort(ans,ans+n,greater<int>());
    	int as=0;
    	for(int i=0;i<k;i++)
    	{
    		as+=ans[i];
    	}
    	cout<<as<<endl;
    }
}
