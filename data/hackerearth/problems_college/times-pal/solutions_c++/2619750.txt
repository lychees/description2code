#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--) {
		string a,b;
		cin>>a>>b;
		int n = a.length();
		int cnt = 0;
		for(int i = 0; i <= n; i++) {
			int len1 = i - 0;
			int len2 = n - len1;
			string fi = a.substr(0, len1);
			string se = a.substr(i, len2);
			string pa1 = fi + b + se;
			//cout<<pa1<<"\n";
			string pa2 = pa1;
			reverse(pa2.begin(), pa2.end() ) ;
			if(pa1 == pa2) {
				cnt++;
			}
		}
		cout<<cnt<<"\n";
	}
	return 0;
}
