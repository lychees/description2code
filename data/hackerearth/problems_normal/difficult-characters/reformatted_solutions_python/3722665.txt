from operator import itemgetter
from string import lowercase
for _t in xrange(int(input())):
    D = {}
    name = raw_input()
    for char in name:
        if char not in D:
            D[char] = 1
        else:
            D[char] += 1
    for char in lowercase:
        if char not in D:
            D[char] = 0
    out = ''
    for t in sorted(sorted(D.items(), reverse=True), key=itemgetter(1)):
        out += t[0]
    for i in out:
        print i,
    print
