#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

main () {
int N;
cin >> N;

int K;
cin >> K;

vector <int> price(N);

for (int i = 0; i < N; ++i) {
cin >> price[i];	
	
}

sort (price.begin(), price.end());

int num = 0, sum = 0;
for (; num < N, sum < K; ++num)
{
	sum += price[num];
	
}

cout << num-1 << endl;
return 0;
}
