#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	long long int left[n];
	long long int right[n];
	left[0]=1;
	right[n-1]=1;
	for(int i=1;i<n;i++)
	{
		if(a[i]>a[i-1])
		left[i]=left[i-1]+1;
		else left[i]=1;
		if(a[n-1-i]>a[n-i])
		right[n-i-1]=right[n-i]+1;
		else right[n-i-1]=1;
	}
	long long int sum=0;
	for(int i=0;i<n;i++)
	{
		sum+=max(left[i],right[i]);
	}
	cout<<sum<<endl;
    return 0;
}
