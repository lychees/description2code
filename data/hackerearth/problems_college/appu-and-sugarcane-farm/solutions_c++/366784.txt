#include <iostream>
using namespace std;
int check(unsigned long long int a)
{
	int r;
     while(a!=0&&r==0)
     {
     	r=a%2;
     	a=a/2;
     }
     if(a!=0&&r==1)
     return 0;
     else
     return 1;
}
int main()
{
	int n,i,flag;
	unsigned long long int a[55],max;
    cin>>n;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(max<a[i])
    	max=a[i];
    }
    flag=1;
    for(i=0;i<n&&flag;i++)
    {
    	if(max%a[i]==0&&check(max/a[i]))
    	;
    	else
    	flag=0;
    	
    }
    if(flag)
    cout<<"YES"<<endl;
    else
    cout<<"NO"<<endl;
    return 0;
}
