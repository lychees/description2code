#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	vector <int> v;
    int m,t,n,k,l;
    cin>>t;
    while(t--)
    {
    	v.clear();
    	cin>>n>>k;
    	for(int i=0;i<n;i++)
    	{
    		cin>>l;
    		v.push_back(l);
    	}
    	sort(v.begin(),v.end());
    	m=v.size();
    	for(int i=0;i<(m/2);i++)
    	{
    		v[i]+=v[n-i-1];
    	}
    	sort(v.begin(),v.begin()+m/2);
    	int a=v[m/2-1]-v[0];
    	cout<<a<<endl;
    	if(a<k)
    	cout<<"Chick magnet Jhool!"<<endl;
    	if(a==k)
    	cout<<"Lucky chap!"<<endl;
    	if(a>k)
    	cout<<"No more girlfriends!"<<endl;
    }
    return 0;
}
