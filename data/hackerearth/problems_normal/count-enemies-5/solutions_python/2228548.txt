def  getAlive(line):
	TmpKill = 0
	Kill = 0
	KillSwich = 0
	Enemy = 0
	for ch in line:
		if ch == 'O':
			Enemy = Enemy+1
			if KillSwich == 1:
				Kill = Kill+1
			else:
				TmpKill = TmpKill+1
		elif ch == 'X':
			Kill = Kill + TmpKill
			TmpKill = 0
			KillSwich = 1
		elif ch == '*':
			TmpKill = 0
			KillSwich = 0
	return Enemy - Kill
	
T = raw_input()
T = int(T)
while T>0:
	LineStr = raw_input()
	print getAlive(LineStr)
	T = T - 1
