int max(int i , int j)
{
	 if(i > j )
	 return i;
	 else
	 return i ;
}
int main()
{
    long long int t , i, j , k , n;
    long long int s[200] , e[200] , p[200] ;
    cin>>t;
    map< pair<long long int ,long long  int> , long long int > M ;
    while(t--)
    {
    	  M.clear() ;
    	  pair<long long int,long long int>g[200] ;

    	  cin>>n;
    	  for(i = 0 ; i < n ; i++){
    	   
    	    cin>>s[i]>>e[i]>>p[i] ;
    	    g[i] =  make_pair(e[i] , s[i]) ;	
    	    M[g[i]] = p[i] ;
    	    
         	}
    	  
    	  
    	  sort( g , g + n) ;
    	  double ans = 0 ;
    	  double m = 0 ;
    	  double dp[200] ;
    	  dp[0] = M[g[0]] ;
    	  
    	  for( i = 1 ; i < n ;i++)
    	  {  
    	  	    m = 0 ;
    	        for(j = 0 ; j < i ; j++)
    	        {
    	        	
    	          if(g[i].second >= g[j].first)
    	          {
    	          	 if( dp[j]  > m ) 
    	          	    m = dp[j] ;
    	          }
    	        } 
    	     
    	    
    	     dp[i] = max( dp[i-1] , m + M[g[i]] ) ; //dp[last] + M[g[i]]) ;
    	    
    	     
    	  }
    	  
    	  //cout<<setprecision(2)<<fixed<<dp[n-1]/100<<endl;
