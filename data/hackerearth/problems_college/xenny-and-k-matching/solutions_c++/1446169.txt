 
#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;
int s[1000000];
int main()
{ 
	 int n,k,cnt=0;
	 cin>>n>>k;
	 for(int i=0;i<n;i++)cin>>s[i];
	 
	 int i=0,j=k;
	 while(j<n)
	 {
	 	int x=abs(s[j]-s[i]);
	 	int y=abs(s[j]+s[i]);
	 	if(x%k==0||y%k==0)
	 	cnt++;
	 	
	 	
	 	i++;
	 	j++;
	 }
	 cout<<cnt<<endl;

return 0;
}
