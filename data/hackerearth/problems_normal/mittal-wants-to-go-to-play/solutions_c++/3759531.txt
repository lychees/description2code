#include <bits/stdc++.h>
using namespace std;
int n,m,d[10002];
vector< vector<int>> vec;
set <pair<int,int> > se;
map < int, map <int,int> > mp;


int main()
{
    int t,i,a,b,c;
    cin>>t;
    while(t--)
    {
    
    	vec.clear();
    	se.clear();
    	mp.clear();
    		cin>>n>>m;
    		vec.resize(m+1);
    	for(i=1;i<=n;i++)
    	d[i]=100000000;
    	d[1]=0;
    	for(i=0;i<m;i++)
    	{
    		cin>>a>>b>>c;
    	   vec[a].push_back(b);
    		vec[b].push_back(a);
    		if(mp[a][b])
    		{
    			mp[a][b]=min(c,mp[a][b]);
    			mp[b][a]=min(c,mp[b][a]);
    		}
    		else{
    		mp[a][b]=c;
    		mp[b][a]=c;
    		}
    	}
    	
    	
    
    	se.insert(make_pair(0,1));
    
    while(!se.empty())
    {    pair <int ,int > toper;
    	 toper=*(se.begin());
    	
    	se.erase(se.begin());
    	
    	int u=toper.second;
    	
    	
    	
    	
    	int l=vec[u].size();
    	for(i=0;i<l;i++)
    	{
    		if(d[vec[u][i]]>d[u]+mp[u][vec[u][i]])
    		{
    			if(d[vec[u][i]]!= 100000000)
    			se.erase(se.find(make_pair(d[vec[u][i]],vec[u][i])));
    			
    			d[vec[u][i]]=d[u]+mp[u][vec[u][i]];
    			se.insert(make_pair(d[vec[u][i]],vec[u][i]));
    		}
    	}
    	
    }
    	
    	
    	
    	
    	
    	int q;
    	cin>>q;
    	
    	while(q--)
    	{
    		cin>>a>>b;
    		
    		if((b-2*d[a])<=0)
    		cout<<"0"<<endl;
    		else
    		cout<<(b-2*d[a])<<endl;
    	}
    	
    }
    return 0;
}
