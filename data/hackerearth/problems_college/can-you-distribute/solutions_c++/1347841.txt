#include <bits/stdc++.h>
#define M 10000007
#define M1 941
#define M2 10627
#define ll long long int
using namespace std;
ll p(ll a,ll b,ll c)
{
	ll ans=1;
//	cout<<"a="<<a<<" b="<<b<<endl;
	while(b)
	{
		if(b&1)
			ans=(ans*a)%c;
		a=(a*a)%c;
		b=b/2;
	}
//	cout<<"Ans="<<ans<<endl;
	return ans;
}
int m[50000000];
int main()
{
	ll n,r,fact[200],i,ans,ans1,ans2;
	ios_base::sync_with_stdio(0);
    scanf("%lld%lld",&n,&r);
    fact[0]=1;
    for(i=1;i<=100;i++)
    {
    	fact[i]=(fact[i-1]*i)%M;
    }
    n--;
    r--;
    if(r>(n/2) and r<=n)
    	r=n-r;
    if(r<=n)
   	{
    ans1=(fact[n]*p((fact[n-r]*fact[r])%M1,M1-2,M1))%M1;
    ans2=(fact[n]*p((fact[n-r]*fact[r])%M2,M2-2,M2))%M2;
   	i=ans1;
    while(i<=M)
    {
    	m[i]=1;
  		i=i+M1;
   	}
   	i=ans2;
   	while(i<=M)
   	{
   		if(m[i])
   		{
   			ans=i;
   			break;
   		}
   		i=i+M2;
   	}
   	}
    else
    ans=-1;
    printf("%lld\n",ans);
    return 0;
}
