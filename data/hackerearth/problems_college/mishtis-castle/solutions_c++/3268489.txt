#include <iostream>
#include <cstring>
using namespace std;
int n, A[102][102], memo[102][102];
int f(int i,int j)
{
	if(memo[i][j]!=-1)
		return memo[i][j];
	if(i>=n-1&&j>=n-1)
		return A[n-1][n-1];
	int a, b;
	a=b=0;
	if(i<n-1)
		a=A[i][j]+f(i+1, j);
	if(j<n-1)
		b=A[i][j]+f(i, j+1);
	memo[i][j]=max(a, b);
	return memo[i][j];
}
int main()
{
	int t, c, ans, i, j;
	cin>>t;
	while(t--)
	{
		cin>>n>>c;
		for(i=0;i<=101;i++)
		for(j=0;j<=101;j++)
		{
			memo[i][j]=-1;
			A[i][j]=0;
		}
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		cin>>A[i][j];
		ans=f(0, 0);
		ans=ans-c;
		if(ans<0)
			cout<<"-1\n";
		else 
			cout<<ans<<endl;
	}
    return 0;
}
