#include <iostream>
using namespace std;
#define ull unsigned long long int

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ull n;
    	cin>>n;
    	if(n<2)
    	{
    		cout<<0<<endl;
    	}
    	else if(n<3)
    	{
    		cout<<2<<endl;
    	}
    	else
    	{
    		ull s1 = 1;
    		ull s2 = 2;
    		ull temp = s1+s2;
    		ull sum = 2;
    		while(temp<n)
    		{
    			if(temp%2 == 0)
    			{
    				sum += temp;
    			}
    			s1 = s2;
    			s2 = temp;
    			temp = s1 + s2;
    		}
    		cout<<sum<<endl;
    	}
    		
    }
    return 0;
}
