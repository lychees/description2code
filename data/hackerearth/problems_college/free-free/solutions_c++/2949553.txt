#include <iostream>
using namespace std;
int main()
{
	int t,arr[200],i,j,k,count,ans[200];
	char index,string[200];
	for(k=0;k<200;k++)
		arr[k]=0;
	cin>>t;
	for(i=0;i<t;i++){
		for(k=0;k<200;k++){
		arr[k]=0;
		}
		count=0;
		cin>>string;
		for(j=0;string[j]!='\0';j++){
			index=string[j];
			arr[int(index)]++;
		}
		for(j=0;j<200;j++){
			count+=arr[j]/2+arr[j]%2;
		}
		ans[i]=count;
	}
	for(i=0;i<t;i++)
		cout<<ans[i]<<endl;
	return 0;
}
