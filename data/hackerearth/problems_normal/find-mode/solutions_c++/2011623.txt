#include <bits/stdc++.h>
using namespace std;
#define MAX 1000001
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long int n,i,temp;
    	scanf("%ld",&n);
    	long int seen[MAX]={0},max=0;
       	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&temp);
    		seen[temp]++;
    		if(max<seen[temp])
    		{
    			max=seen[temp];
    		}
    	}
    	priority_queue<long int>mypq;
    	for(i=0;i<MAX;i++)
    	{
    		if(seen[i]==max)
    		{
    			mypq.push(i);
    		}
    	}
   	  while (!mypq.empty())
  	  {
     	cout<<mypq.top()<<" ";
    	 mypq.pop();
  		}

       	cout<<endl;
    }
}
