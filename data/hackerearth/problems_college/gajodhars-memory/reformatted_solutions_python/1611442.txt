t = int(raw_input())

while t > 0:
    t -= 1
    n = int(raw_input())
    li = [0 for i in range(0, 256)]
    vis = [False for i in range(0, 256)]
    for i in range(0, n):
        x = ord(raw_input()[0])
        if vis[x]:
            li[x] = 0
        else:
            li[x] = 1
            vis[x] = True
    print sum(li)
