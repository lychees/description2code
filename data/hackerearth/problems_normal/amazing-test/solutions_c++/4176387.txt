#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(false);
    ll t,i,j,n,x,s;
    cin>>t;
    while(t--)
    {
        cin>>n>>x;
        ll dp[n+1][x+1];
        s=0;
        for(i=0;i<=x;i++)
        {
            dp[0][i]=0;
        }
        ll a[n+1];
        for(i=1;i<=n;i++)
        {
            cin>>a[i];
            s+=a[i];
            dp[i][0]=0;
        }
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=x;j++)
            {
                if(j<a[i])
                {
                    dp[i][j]=dp[i-1][j];
                }
                else
                {
                    dp[i][j]=max(dp[i-1][j],a[i]+dp[i-1][j-a[i]]);
                }
            }
        }
        j=dp[n][x];
        //cout<<j<<" "<<s<<endl;
        if((s-j)<=x)
        cout<<"YES\n";
        else
        cout<<"NO\n";
        
    }
    return 0;
}
