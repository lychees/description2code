long long int read(){
	long long int num=0;
	register char charNumber=get;
	while(!(charNumber >= '0' && charNumber <= '9')){
		charNumber=get;
	}
	while(charNumber>='0' && charNumber<='9'){
		num = num*10 + charNumber - '0';
		charNumber=get;
	}
	return num;
}
long long int gcd(long long int a, long long int b)
{
       long long int r;
        while(b != 0)
        {
                r = a % b;
                a = b;
                b = r;
        }
        return a;
}
