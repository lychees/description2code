'''

t = int(raw_input())
for i in range(t):
    str_input = list(raw_input())
    str_r = sorted(raw_input())
    input_length = len(str_input)
    j = 0

    k = 0
    for j in range(input_length):
        try:
            if str_input[j] > str_r[k] and str_r[k] != "\0":
                str_input[j] = str_r[k]
                k = k+1
        except BaseException:
            break
    print ''.join(str_input)
