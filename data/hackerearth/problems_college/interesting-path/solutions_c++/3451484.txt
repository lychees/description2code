#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define inf 1000000007

using namespace std;

int main()
{
    int t, n;
    int A[1001];
    int P[1001];

    cin >> t;

    while (t--) {
        cin >> n;

        for (int i = 1; i <= n; i++) {
            cin >> A[i];
            P[i] = INT_MAX;
        }
        P[0] = P[1] = 0;

        for (int i = 1; i <= n; i++) {
            for (int j = i; j <= min(A[i] + i, n); j++) {
                P[j] = min(P[j], P[i] + 1);
            }
        }

        /*for (int i = 1; i <= n; i++) {
            cout << P[i] << " ";
        }*/
        cout << P[n] << endl;
    }
    return 0;
}
