#include <iostream>
using namespace std;

int main()
{
	string line;
    while (getline(cin, line)) {
    	int l = line.length(), i = 0;
    	bool cm = false;
    	while (i < l) {
    		if (i < l-1 && !cm && line[i] == '-' && line[i+1] == '>') {
    			cout << '.'; i++;
    		} else if (i < l-1 && line[i] == '/' && line[i+1] == '/') {
    			cout << "//";
    			i++; cm = true;
    		} else {
    			cout << line[i];
    		}
    		i++;
    	}
    	cout << endl;
    }
    return 0;
}
