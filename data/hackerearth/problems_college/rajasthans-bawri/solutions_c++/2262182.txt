#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,x,y,a,b;
    
    scanf("%d",&t);
    while(t--)
    {
    	
    	scanf("%d %d %d %d\n",&x,&y,&a,&b);
    	if((x>=(a))&&(y>=(b)))
    	printf("Possible\n");
    	else 
    	if((x>=(b))&&(y>=(a)))
    		printf("Possible\n");
    		else printf("Not Possible\n");
    	
    	
    }
    return 0;
}
