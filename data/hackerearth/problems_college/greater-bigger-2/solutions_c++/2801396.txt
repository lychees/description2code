#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string str;
    	int j,i;
    	cin>>str;
    	int n=str.length();
    for(j=n-1;j>0;j--)
    {
    	if(str[j-1]<str[j])
    	break;
    }
    if(j==0)
    {
    cout<<"no answer"<<"\n";
    continue;
    }
     
    int small1=j;
      int smaller=str[j-1];
    for(i=j+1;i<n;i++)
    {
    	if(str[i]>smaller && str[i]<str[small1])
    	small1=i;
    }
    //cout<<j;
    swap(str[j-1],str[small1]);
    sort(str.begin()+j,str.end());
    cout<<str<<"\n";
    }
    
}
