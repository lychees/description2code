#include <iostream>
using namespace std;

long int sum1=0,sum2=0,sum3=0;

int main()
{
	long int size;
	cin>>size;
	int count = 0;	
	for(long int i=0;i<size;i++)
	{
		long int num;
		cin>>num;
		if(i==0)
		{
			sum1+=num;
			continue;
		}
		if(i==1)
		{
			sum2+=num;
			continue;
		}
		if(i==2)
		{
			sum3+=num;
			continue;
		}
		if(count==0)
		{
			sum1+=num;
			
		}
		if(count==1)
		{
			sum2+=num;
		}
		if(count==2)
		{
			sum3+=num;
		}
		if(count ==2)
		{
			count=0;
		}
		else
		{
			count++;
		}
	}
	cout<<sum1<<" "<<sum2<<" "<<sum3<<endl;
	sum1=0;sum2=0;sum3=0;
	return 0;
}
