#include <cstdlib>
#include <cstdio>
#include <cstring>
using namespace std;

int palindrome(char *s) {
    char rev[10], i = 0, len;
    len = strlen(s);

    while (i < len) {
        rev[i] = s[len - i - 1];
        i++;
    }
    rev[len] = '\0';
    if (strcmp(s, rev) == 0) {
        return 1;
    }
    return 0;
}

int main(int argc, char** argv) {
    char *s;
    int len, t;
    unsigned long long value = 1;

    scanf("%d", &t);

    while (t > 0) {
        s = (char *) malloc(10 * sizeof (char));
        scanf("%s", s);
        if (palindrome(s)) {
            printf("Palindrome\n");
        } else {
            len = strlen(s);
            while (len > 0) {
                value *= (s[len - 1] - 'a' + 1);
                len--;
            }
            printf("%lld\n", value);
        }
        free(s);
        t--;
        value = 1;
    }
    return 0;
}
