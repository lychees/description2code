#include <bits/stdc++.h>
using namespace std;

int sum(int n){
	int ans=0;
	while(n!=0){
		ans+=n%10;
		n/=10;
	}
	return ans;
}

int main(){
	int n;
	cin>>n;
	int arr[n];
	for(int i=0;i<n;i++){
		cin>>arr[i];
	}
	int l,b,area;
	cin>>l>>b;
	area=l*b;
	long long int amt=0;
	for(int i=0;i<n;i++){
		if(arr[i]<=10)
			amt+= area*arr[i]*sum(arr[i])*0.75;
		else if(arr[i]<=20)
			amt+= area*arr[i]*sum(arr[i])*0.7;
		else
			amt+= area*arr[i]*sum(arr[i])*0.8;
		//cout<<amt<<endl;;
	}
	cout<<amt<<endl;
}
