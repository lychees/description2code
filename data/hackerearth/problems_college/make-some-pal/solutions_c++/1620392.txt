#include <iostream>
#include<cstring>
using namespace std;
int main()
{
    int t,i,le,p,j,l,r; 
    char s[1000]; 
    scanf("%d",&t);
    while(t--){
    	scanf("%s",s);
    	le=strlen(s);
    	p=0; 
    	int longest=0;
 
    	for(i=le-1;i>=0;i--){
    		p=1;
    		l=i-1; r=i+1;
    		while(l>=0 && r<=le-1 && s[r]==s[l]){
    			 p+=2;
    			 r++; l--;
                // cout<<s[i]<<" ";
    		} 
    		if(r==le && p>longest){
                 longest=p;    			
    		}
    		p=0;
    		l=i; r=i+1;
    		while(l>=0 && r<=le-1 && s[r]==s[l]){
    			 p+=2;
    			 r++; l--;
                // cout<<s[i]<<" ";
    		} 
    		if(r==le && p>longest){
                 longest=p;    			
    		}
    	}
      //	cout<<"longest="<<longest<<endl;
    	printf("%d\n",2*(le-longest)+longest);
    }  
    return 0;
}
