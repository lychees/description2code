#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
char arr[100005];

int main()  {

    int t;scanf("%d",&t);
    while(t--)
    {
        scanf("%s",arr);
        int len = strlen(arr);
        bool flag = 0;
        for(int i=0;i<len;++i)
        {
            if(arr[i]=='R')
            {
                if(i+1<len && arr[i+1]=='Y')
                {
                    if(i+2<len && arr[i+2]=='Y')
                    {
                        i += 2;
                    }
                    else
                    {
                        ++i;
                    }
                }
            }
            else {flag = 1;break;}
        }
        if(flag==0)printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
