#include <iostream>
#include <cmath>
#include <cstdio>
#include <cstring>
#include <functional>
#include <algorithm>

using namespace std;

int main()
{
	int n ;
	int s,d;
	cin >> n;

	while(n--)
	{
		cin >> s >> d;

		if(d>s || ((s+d)%2 == 1))
		{
			cout << "impossible" << endl;
		}
		else
		{
			cout << max((s+d)/2,(s-d)/2 ) << " " << min((s+d)/2,(s-d)/2 ) << endl;
		}
	}

	return 0;
}
