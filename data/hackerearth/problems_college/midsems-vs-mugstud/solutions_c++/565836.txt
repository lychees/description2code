#include <iostream>
#include<cstdio>
#define ll long long
using namespace std;
ll A[51][51][51],B[51][51][51];
int main()
{
    int i,j,k;
    for(i=0;i<=50;i++)
    {
    	for(j=0;j<=50;j++)
    	{
    		A[0][i][j]=0;
    		A[i][0][j]=0;
    		A[i][j][0]=0;
    	}
    }
    int t,n;
    ll sum;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(k=1;k<=n;k++)
    	{
    		for(i=1;i<=n;i++)
    		{
    			for(j=1;j<=n;j++)
    			{
    				scanf("%lld",&A[k][i][j]);
    			}
    		}
    	}
    	for(k=1;k<=n;k++)
    	{
    		for(i=1;i<=n;i++)
    		{
    			sum=0;
    			for(j=1;j<=n;j++)
    			{
    				B[k][i][j]=A[k][i][j]-A[k-1][i][j]-A[k][i-1][j]-sum+A[k-1][i-1][j];
    				sum+=B[k][i][j];
    			}
    		}
    	}
    	for(k=1;k<=n;k++)
    	{
    		for(i=1;i<=n;i++)
    		{
    			for(j=1;j<=n;j++)
    			{
    				printf("%lld ",B[k][i][j]);
    			}
    			printf("\n");
    		}
    	}
    }
    return 0;
}
