def magicNumber():
	tests = int(raw_input())
	for t in xrange(tests):
		m = int(raw_input())
		first = [1,0]
		second = [0,1]
		if m == 0:
			print first[0], first[1]
		elif m == 1:
			print second[0], second[1]
			
		else:
			for i in xrange(m-1):
				c = [first[0]+second[0],first[1]+second[1]]
				first = second
				second = c
			print c[0], c[1]

magicNumber()
