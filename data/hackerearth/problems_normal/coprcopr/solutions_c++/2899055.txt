#include<bits/stdc++.h>
using namespace std;
long long totient[10000009]={0};
void computetotient()     
{
	long long  i,j;
	totient[1]=1;
	for(i=2;i<=10000006;i++)
	{
		if(totient[i]==0)				//If i is prime
		{
			totient[i]=i-1;
			for(j=2*i;j<=10000006;j+=i)		
			{
				if(totient[j]==0)
				totient[j]=j;
				totient[j]=(totient[j]/i)*(i-1);
			}
		}
	}	
}
int  main()
{
	long long  t,n;
	scanf("%lld",&t);
	computetotient();
	for(int i=2;i<=10000006;i++)
	totient[i]+=totient[i-1];
	while(t--)
	{
		scanf("%lld",&n);
		printf("%lld\n",totient[n]);
	}	
	return 0;
}
