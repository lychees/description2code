#include <iostream>
#include <string>
using namespace std;

int findStreak(string s)
{
	int i,max=-1,len;
	len = s.length();
	int *arr = new int[len+1];
	s = "X"+s;
	arr[0]=0;
	for(i=1;i<=len;i++)
	{
		if(s[i]=='C')
			arr[i] = arr[i-1]+1;
		else
			arr[i]=0;
		
		if(arr[i]>max)
			max = arr[i];
	}
	
	return max;
}

int main()
{
	int n;
	string final,part;
	final = "";
	int curStreak,dayStreak = -1,totalStreak = -1;
	cin>>n;
	while(n--)
	{
		cin>>part;
		curStreak = findStreak(part);
		dayStreak = (dayStreak > curStreak? dayStreak : curStreak);
		final = final + part;
	}
	totalStreak = findStreak(final);
	
	cout<<dayStreak<<" "<<totalStreak<<endl;
    return 0;
}
