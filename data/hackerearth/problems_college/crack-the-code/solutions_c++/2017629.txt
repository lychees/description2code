#include<bits/stdc++.h>
#define ll long long
using namespace std;
int F[100000];
ll fib(ll n)
{
    if(n==0)return 0;
    if(n==1)return 1;
    if(F[n]!=0)return F[n];
    return F[n]=fib(n-1)+fib(n-2);
}
int main()
{
    ll T,X,Y,i,cnt,ans;
    cin>>T;
    F[0]=F[1]=1;
    ans=fib(10000);
    while(T--)
    {
        cin>>X>>Y;
        cnt=0;
        i=0;
        while(F[i]<=X)i++;
        if(F[i-1]==X)i--;
        if(F[i]==1)i--;
        while(F[i]<=Y)
        {
            cnt++;
            i++;
        }
        if(cnt%2==0)
            cout<<"VALID\n";
        else
            cout<<"INVALID\n";


    }
}
