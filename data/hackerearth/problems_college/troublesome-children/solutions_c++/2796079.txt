#include <bits/stdc++.h>
using namespace std;

 int main(){
  
 long long int k,start=0;
  cin>>k;
 long long int h[k+1],a[k+1];
  for(long long int i=0;i<=k;i++)a[i]=0;
  for(long long int i=1;i<=k;i++)cin>>h[i];
  for(long long int i=1;i<=k;i++){
    long long  int tmp=h[i];
  long long    int left=k-i+1;
      tmp%=left;
      if(tmp==0)tmp=left;
  long long int count=0,flag=0;
 
      for(long long int j=start+1;j<=k;j++){
          if(a[j]!=0)continue;
          count++;
          if(count==tmp){start=j;a[j]=i;cout<<j<<endl;flag=1;break;}
      }
      for(long long int j=1;j<=start && count<tmp;j++){
          if(a[j]!=0)continue;
          count++;
          if(count==tmp){start=j;a[j]=i;cout<<j<<endl;flag=1;break;}
      }
      
  }
  
  //for(long long int i=1;i<=k;i++)cout<<a[i]<<endl;
  
  
  
    return 0;
}
