#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin >> t;
	while ( t-- ) {
		string s;
		cin >> s;
		int n = (int)s.size();
		int idx = 0;
		int ans = 0;
		for ( int i = 0; i < n; i++ ) {
			if ( s[i] == '|' ) goto p1;
		}
		cout << "Bob" << endl;
		continue;
		p1: { }
		while ( idx < n ) {
			if ( s[idx] == '_' ) {
				int cnt = 0;
				while ( idx < n && s[idx] == '_' ) cnt++, idx++;
				ans = ans^cnt;
			}
			else idx++;
		}
		if ( !ans ) cout << "Bob" << endl;
		else cout << "Alice" << endl;
	}
	return 0;
}
