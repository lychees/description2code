#include <bits/stdc++.h>
using namespace std;
long long dp[26];
#define ll long long
int main()
{
	long long int n;
    string s;
    cin>>n;
    cin>>s;
    int i,j,k,q;
    char x,y;
 
    for(i=0;i<n;i++)
    {
    	dp[s[i]-'a']++;
    }
    cin>>q;
    
   // for(i=0;i<26;i++)
   
    
    while(q--)
    {
    cin>>x>>y;
    ll a=dp[x-'a'];
    ll b=dp[y-'a'];
    if(x!=y)
    cout<<a*b<<endl;
    else
    cout<<(a*(a-1))/2<<endl;
    }
    return 0;
}
