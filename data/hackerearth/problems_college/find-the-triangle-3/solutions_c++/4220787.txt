#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int t,n,i,j,k,a[100005];
    sd(t);
    while(t--)
    {
    	sd(n);
    	for(i=0;i<n;i++)
    		sd(a[i]);
    	sort(a,a+n);
    	j=-1;
    	for(i=n-1;i>=2;i--)
    	{
    		if(a[i]<a[i-1]+a[i-2])
    		{
    			j=i;
    			break;
    		}
    	}
    	if(j==-1)
    		printf("-1\n");
    	else
    		printf("%d %d %d\n",a[j-2],a[j-1],a[j]);
    }
    return 0;
}
