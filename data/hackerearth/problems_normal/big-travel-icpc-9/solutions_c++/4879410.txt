#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    vector<int> x,y;
    int tempx,tempy;
    for(int i=0;i<n;i++)
    {
    	cin>>tempx>>tempy;
    	x.push_back(tempx);
    	y.push_back(tempy);
    }
    sort(x.begin(),x.end());
    sort(y.begin(),y.end());
    long long int ans=0;
    long long int j=n-1,x1,y1;
    for(int i=n-1;i>=0;i--)
    {
    	x1=(x[i]*j)%1000000007;
    	y1=(y[i]*j)%1000000007;
    	ans=(ans+y1+x1)%1000000007;
    	j=j-2;
    }
    cout<<ans;
    return 0;
}
