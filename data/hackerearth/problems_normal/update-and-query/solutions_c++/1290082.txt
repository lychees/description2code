#include <iostream>
#include <cmath>
#include <set>
#include <iomanip>
#include <cstdio>
#include <vector>
using namespace std;

int a[100500], n, m;

int abs(int a) { return a < 0 ? -a : a; }

int main()
{
    cin.sync_with_stdio(false);
    cin >> n >> m;
    for(int i = 0; i < n; i++)
        cin >> a[i];
    for(int i = 0; i < m; i++) {
        char c;
        int x, y;
        cin >> c >> x >> y;
        if(c == 'Q') {
            int st = x - 1;
            int tt = 0;
            while(st < n and abs(a[st] - a[x - 1]) <= y) {
                tt = max(tt, abs(a[st] - a[x - 1]));
                st++;
            }

            if(st == x - 1)
                cout << "-1 -1\n";
            else
                cout << st - x + 1 << " " << tt << "\n";
        } else {
            a[x - 1] = y;
        }
    }
    return 0;
}
