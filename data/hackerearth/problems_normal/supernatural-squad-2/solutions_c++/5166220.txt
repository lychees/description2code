#include<bits/stdc++.h>
using namespace std;

long long dp[210][210];
int n,k;

long long rec(int numLeft,int minTake)
{
	if(numLeft == minTake)
	return 1;
	if(numLeft<minTake || numLeft <k)
	return 0;
	if(dp[numLeft][minTake] != -1)
	return dp[numLeft][minTake];
	return dp[numLeft][minTake] = rec(numLeft-minTake,minTake) + rec(numLeft,minTake+1);
}

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		memset(dp,-1,sizeof(dp));
		cin >> n >> k;
		long long ans = rec(n,k);
		cout << ans << endl;
	}
	return 0;
}
