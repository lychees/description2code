#include <bits/stdc++.h>
using namespace std;
long long a[555],pr[555];
long long ans[555];
void go()
{
	for(int i=2;i<501;i++)
	{
		if(!a[i])
		{
			for(int j=2*i;j<501;j+=i)
				a[j]++;
		}
	}
	pr[0] = 0;
	pr[1] = 0;
	pr[2] = 1;
	for(int i=3;i<501;i++)
	{
		pr[i] = pr[i-1];
		if(a[i]==0)
			pr[i]++;
	}
}

int main()
{
	go();
	long long n;
	cin>>n;
	ans[0]=ans[1] = 0;
	ans[2] = 1;
	ans[3] = 3;
	for(int i=4;i<=n;i++)
	{
		ans[i] = ans[i-1];
		ans[i] += pow(2,pr[i]-1);
	}
	cout<<ans[n];
	return 0;
}
