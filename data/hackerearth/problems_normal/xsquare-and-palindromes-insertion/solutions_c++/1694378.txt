#include <iostream>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--){
    string s;
    cin>>s;
    int a[26];
    for(int i=0;i<26;i++) a[i]=0; 
    for(int i=0;i<s.size();i++) 
     {
     	a[s[i]-'a']++;
     }
    int k=0; 
    for(int i=0;i<26;i++)
     {
     	if(a[i]%2!=0)
     	 k++;
     }
    if(k>0) cout<<k-1<<endl;
    else cout<<0<<endl;
}
}
