#include<bits/stdc++.h>
using namespace std;
# define ll long long
bool hsh[1111111];
int fact[1111111];
void sieve(int n)
{
    int i,a;
    for(i=1;i<=n;i++)
        hsh[i]=true;
    hsh[1]=false;
    for(i=2;i<=n;i++)
    if(hsh[i])
    {
        fact[i]=i;
        a=2;
        while(a*i<=n)
        {
            hsh[a*i]=false;
            fact[a*i]=i;
            a++;
        }
    }
}
int factorise(int n)
{
    set <int> s;
    while(n>1)
    {
        s.insert(fact[n]);
        n/=fact[n];
    }
    return s.size();
}
int main()
{
    cin.sync_with_stdio(false);
    int t,n,m,i,k;
    sieve(1000001);
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        n*=m;
        k=factorise(n);
        if(hsh[k])
            puts("Yes");
        else
            puts("No");
    }
    return 0;
}
