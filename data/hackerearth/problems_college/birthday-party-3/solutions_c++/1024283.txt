#include <iostream>
using namespace std;

long long R[1001][1001], B[1001][1001];
int P, Q, X, Y;
#define mod 179424673
int main()
{
    int T;
    cin >> T;
    for (int it = 0; it < T; it++) {
    	cin >> P >> Q >> X >> Y;
    	R[0][0] = B[0][0] = 1;
    	
    	for (int j = 1; j <= Q; j++) {
    		R[0][j] = 0;
    		B[0][j] = (j <= Y);
    	}
    		
    	
    	for (int i = 1; i <= P; i++) {
    		if (i <= X)
    			R[i][0] = 1;
    		else
    			R[i][0] = 0;
    		B[i][0] = 0;
    		for (int j = 1; j <= Q; j++) {
    			if (i <= X)
    				R[i][j] = (R[i-1][j] + B[i-1][j])%mod;
    			else
    				R[i][j] = (R[i-1][j] + B[i-1][j] - B[i-X-1][j] + mod)%mod;
    				
    			if (j <= Y)
    				B[i][j] = (B[i][j-1] + R[i][j-1])%mod;
    			else
    				B[i][j] = (B[i][j-1] + R[i][j-1] - R[i][j-Y-1] + mod)%mod;
			}
    	}
    	
    	cout << (B[P][Q] + R[P][Q])%mod << endl;
    	
    }
    return 0;
}
