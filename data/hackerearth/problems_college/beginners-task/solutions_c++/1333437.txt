#include <cstdio>
using namespace std;

int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
    	int num,count=0;
    	scanf("%d",&num);
    	while(num--)
    	{
    		int temp;
    		scanf("%d",&temp);
    		if(temp%2==0) count++;
    	}
    	printf("%d\n",count);
    }
    return 0;
}
