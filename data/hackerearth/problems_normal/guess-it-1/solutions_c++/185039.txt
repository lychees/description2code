#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char str[201];
        cin>>str;
        int i,sum=0;
        for(i=0;str[i]!='\0';i++)
        {
            if(str[i]=='_')
                continue;
            else
                sum=sum+(str[i]-'0');
        }
        for(i=0;i<=9;i++)
        {
            sum=sum+i;
            if(sum%9==0)
                break;
            sum=sum-i;
        }
        if(i==0||i==9)
            cout<<"0 or 9"<<endl;
        else
            cout<<i<<endl;
    }
    return 0;
}
