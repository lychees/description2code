#include <bits/stdc++.h>
using namespace std;

int main() {
	unsigned long long t,pro,n,m,n1,n2,temp,c=0;
	cin>>t;
	while(t--)
	{c=0;
		cin>>pro>>n>>m;
	
		for(n1=1;n1<=sqrt(pro);n1++)
	    {
	    	n2=pro/n1;
	    	
	    	if(n1*n2==pro)
	    	{
	    		if(n2>=1 && n2<=m)
	    		c++;
	    		
	    		if(n2<=n && n2>=1 && n2!=n1)
	    		c++;
	    	}
	    }
	    cout<<c<<endl;
	}
	return 0;
}
