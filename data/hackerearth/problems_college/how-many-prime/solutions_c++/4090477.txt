#include<bits/stdc++.h>
using namespace std;

long int prime[1010100] = {0};
long int ans[1000000] = {0};
int main()
 {
    int t;

     scanf("%d",&t);

     long int i,j,k =0,x,y,m,h;

        for(i = 2; i<1010100; i++)
          if(!prime[i])
            {

              ans[k] = i;

                //cout<<ans[k]<<" ";
                k++;
               for(j = 2*i; j<1010100; j+=i)
                 prime[j] = 1;
            }
  //cout<<"hey"<<endl;
     while(t--)
      {
        int n,flag = 0;

        scanf("%d",&n);

         long int l = 0;
         long int q = 1000000;
         for(l = 0; l<k && n>ans[l]; l++)
          {
            long int x = n - ans[l];


            if(prime[x] == 0)
             {
                //cout<<x<<endl;
                flag = 1;
                   if((abs(x-ans[l]) <  q ) && x!=ans[l])
                    {
                        q = abs(x-ans[l]);
                        //cout<<ans[l]<<endl;
                        h = ans[l];
                        m = x;
                    }
             }

        }

      if(flag)
        {
            if(m > h)
                cout<<h<<" "<<m<<endl;

           else
             cout<<m<<" "<<h<<endl;
        }
      else if(!flag)
       cout<<-1<<" "<<-1<<endl;
      }
  return 0;
 }
