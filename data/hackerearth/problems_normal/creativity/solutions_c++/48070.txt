#include<iostream>
#include<string>
#include<stdio.h>
using namespace std;
int main()
{
    int t;
    string a,b;
    cin>>t;
    getchar();
    while(t!=0){
      getline(cin,a);
      getline(cin,b);
      size_t f=b.find(a);
      if(f!=std::string::npos)
      {
       cout<<"possible"<<"\n";
      }
      else
      {
        cout<<"not possible"<<"\n";
      }
      t--;
    }




}
