#include<bits/stdc++.h>
using namespace std;
#define in(x) scanf("%lld",&x)
int main()
{
    long long int i,t,x,j,a,b,ans,y;
    in(j);
    vector<pair<long long int,long long int> > p,n;
    for(i=0;i<j;i++)
    {
        in(t);
        in(x);
        if(t<=x)
            p.push_back(make_pair(t,x));
        else if(t>x)
            n.push_back(make_pair(x,t));
    }
    sort(p.begin(),p.end());
    sort(n.begin(),n.end());
    reverse(n.begin(),n.end());
    a=p.size(),b=n.size();
    if(a!=0)
        ans=p[0].first,y=p[0].second;
    else if(b!=0)
        ans=0,y=0;
    for(i=1;i<a;i++)
    {
        t=p[i].first;
        x=p[i].second;
        if(y>=t)
            y=y-t+x;
        else if(y<t)
            ans=ans+t-y,y=x;
    }
    for(i=0;i<b;i++)
    {
        x=n[i].first;
        t=n[i].second;
        if(y>=t)
            y=y-t+x;
        else if(y<t)
            ans=ans+t-y,y=x;
    }
    cout << ans << endl;
}
