#include <iostream>
#include<string.h>
#include <vector>
#include <iterator>
#include <algorithm>
using namespace std;
 
int main()
{
    int n;
    cin>>n;
    int a[26];
    
    cin.ignore();
    while(n--){
    	memset(a,0,sizeof a);
    string s,k;
    getline(cin,s);
    getline(cin,k);
    char c;
    for(int i=0;i<s.length();i++)
    {  c = s[i];
      
      if(!isalpha(c))
        continue;
       if(int(tolower(c)-'a')>=0 && int(tolower(c)-'a')<26)
          a[int (tolower(c)-'a')]++;
    }
    for(int i=0;i<k.length();i++)
    {  
    	c = k[i];
       if(!isalpha(c))
        continue;
       if(int(tolower(c)-'a')>=0 && int(tolower(c)-'a')<26)
         a[int (tolower(c)-'a')]--;
    }
    int count1=0,countp=0,countn=0,i=0,odd=0;
    for(;i<26;i++)
    {
    	if(abs(a[i])%2==1)
    	odd++;
    	//cout<<a[i]<<" ";
    	if(a[i]==0)
    	  count1++;
     	else if(a[i]>0)
    	  countp++;
    	else if(a[i]<0)
    	  countn++;
    	if(odd>=2)
    	break;  
    }
     string t="";
    if(i!=26)
    cout<<"NO LUCK\n";
    else if((countp>0 && countn>0))
    cout<<"NO LUCK\n";
    else if(count1==26)
    cout<<"YES\n";
    else
    {
    	for(int i=0;i<26;i++)
    	{
    		a[i] = abs(a[i]);
    	}
    	int k=-1;
    	for(int i=0;i<26;i++)
    	{
    	    if((a[i])%2==1)
    		 {
    		 	k=i;
    		 	a[i]--;
    		 }
    	    if((a[i])>1)
    		{   
    			int u=(a[i])/2;
    			while(u<(a[i]))
    			{
    				c = 'a'+i;
    				t = t+c;
    				(a[i])--;
    			}
    		}
    	}
    	string final =t;
    	if(k!=-1){
    	c = 'a'+k;
    	final = t+c;
    	}
    	reverse(t.begin(),t.end());
    	final = final+t;
    	cout<<final<<"\n";
    }
 }
}
