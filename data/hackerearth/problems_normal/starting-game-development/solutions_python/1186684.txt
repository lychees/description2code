import bisect

N,M,Q = map(int,raw_input().split())
levels = []
for _ in xrange(N):
	levels.append(map(int,raw_input().split()))
	
for _ in xrange(Q):
	q=map(int,raw_input().split())
	m=bisect.bisect_right(levels[0],q[0])
	#print "initial",m
	for i in xrange(1,N):
		if m==0:
			break
		while q[i]<levels[i][m-1]:
			m-=1
	print m
