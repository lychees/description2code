#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define MIN(a,b) (a>b?b:a)
#define MAX(a,b) (a>b?a:b)
LL n,dp[5000000];
LL Dp(LL ind){
	if(ind>=n){return 1;}
	return dp[ind];
}
int main(){
	LL t;cin>>t;
	while(t--){
		cin>>n;
		dp[n-1]=2;
		for(LL i=n-2;i>=0;i--){
			dp[i]=Dp(i+1)+Dp(i+2);
		}
		cout<<dp[0]*dp[0]<<"\n";
	}
	return 0;
}
