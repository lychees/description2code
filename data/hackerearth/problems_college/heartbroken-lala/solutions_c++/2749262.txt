#include <iostream>
using namespace std;
typedef  unsigned long long ll;

int countSetBits(ll n)
{
	int count = 0;
	while (n != 0) {	
		count++;
		n &= n - 1;
	}
	return count;
}

bool check(ll k)
{	
	while (k != 0) { 
		if (k & 1 == 1) {
			k = k >> 1;
			return k & 1 == 1;	
		}
		k = k >> 1;
	}
}

int solve(ll k) {
	while (k > 3) {
		if (countSetBits(k) == 1) return 0;
		else {
			ll n = k, prev = k;
			while (n != 0) {
				prev = n;
				n &= n - 1;
			}

			k -= prev;
			
			if ((countSetBits(k) == 2) && check(k)) return 1;
		}
	}

	if (k == 1 || k == 2) return 0;
	else return 1;
}

int main()
{
	int t;
	cin>>t;
	while(t--) {
		ll k;
		cin>>k;
		cout<< solve(k)<<endl;
	}
	
	return 0;
}
