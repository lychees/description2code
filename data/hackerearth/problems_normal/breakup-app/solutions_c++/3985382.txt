#include <bits/stdc++.h>
using namespace std;

int main()
{
	int N;cin>>N;
	cin.ignore();
	vector<int> count;
	count.assign(40, 0);
	while(N--) {
		string str;
		getline(cin, str);
		size_t j = -1;
		do {
			j = str.find(" ", j+1);
			if(isdigit(str[j+1])) {
				if(isdigit(str[j+2])) {
					//cout<<str.substr(j+1, 2)<<endl;
					if(str[0] == 'G')
						count[stoi(str.substr(j+1, 2))] += 2;
					else 
						count[stoi(str.substr(j+2, 2))] += 1;
				} else {
					//cout<<str.substr(j+1, 1)<<endl;
					if(str[0] == 'G')
						count[stoi(str.substr(j+1, 1))] += 2;
					else 
						count[stoi(str.substr(j+1, 1))] += 1;
				}
			}
		}while(j != string::npos);
	}
	
	int maxDay = distance(count.begin(), max_element(count.begin(), count.end()));
	//cout<<maxDay<<endl;
	if(maxDay == 19 || maxDay == 20) {
		cout<<"Date"<<endl;
	} else {
		cout<<"No Date"<<endl;
	}
    return 0;
}
