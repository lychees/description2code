#include <iostream>
using namespace std;

int main()
{
    int t;
    long long int m, p, x, b;
    cin>>t;
    while(t--) {
    	b=1;
    	cin>>m>>p;
    	x = m^p;
    	if(x == 0)
    		cout<<0<<endl;
    	else {
	    	while((x=x&(x-1)))
	    		b++;
	    	cout<<b<<endl;
    	}
    }
    return 0;
}
