#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
   long long int t,a,c,b,b1,a1=180;
    cin>>t;
    while(t--)
    {
    	cin>>a;
    	a=a%360;
    	c=__gcd(a,a1);
    		b=a/c;
    		b1=180/c;
    		if(a==180)
    		{
    			cout<<"$"<<endl;
    			continue;
    		}
    	    if(b==1)
    	    {
    	    cout<<"$/"<<b1<<endl;
    	    continue;
    	    }
    	    if(b1==1)
    	    	{
    	    		cout<<b<<"$"<<endl;
    	    		continue;
    	    	}
    	    else
      		cout<<b<<"$/"<<b1<<endl;
    	
    }
}
