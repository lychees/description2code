import sys
import math

n = int(raw_input())

primes = set([2,3])

for i in range(2,100000):
	isPrime = True
	maxi = int(math.sqrt(i))
	if (maxi < 4):
		maxi = 4
	for n in range(2,maxi+1):
		if i % n == 0:
			isPrime = False
			break
	if isPrime:
		primes.add(i)


def isCombination(num):
	for i in primes:
		if (num-i) in primes:
			return True
	return False

for line in sys.stdin:
	if isCombination(int(line)):
		print "Deepa"
	else:
		print "Arjit"
