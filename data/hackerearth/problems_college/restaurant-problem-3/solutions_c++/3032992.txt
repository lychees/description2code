#include <bits/stdc++.h>
using namespace std;
//kise
int main() 
{
	char s[1000000];
	long long int j,n,i;
	int k,val;
	j=0;
	i=0;
	k=0;
	gets(s);
	n=strlen(s);
	
	while(i<n)
	{
		if(s[i]=='8'||s[i]=='0')
		j++;
		if(i>0)
		{
			val=(s[i-1]-48)*10 + (s[i]-48);
			if(val%8==0&&val>10)
			j++;
		}
		
		if(i>1)
		{
			if(i>2&&s[i-3]!='0')
			k++;
			
			val+=(s[i-2]-48)*100;
			if(val%8==0)
			{
				if(val>100)
				{
					j++;
				}
			    j+=k;
			}
			
		}
		i++;
	}
	
	printf("%lld",j);
	
	return 0;
}
