#include<iostream>
using namespace std;
int main(){
	long long int n,arr[2000],count=0,i,j;
	cin>>n;
	//scanf("%lld",&n);
	for(i=0;i<n;i++){
		cin>>arr[i];
		//scanf("%lld",&arr[i]);
	}
	for(i=0;i<n;i++){
		for(j=i+1;j<n;j++){
			if(arr[i]>arr[j]){
				count++;
			}
		}
	}
	cout<<count;
	//printf("%lld",count);
	return 0;
}
