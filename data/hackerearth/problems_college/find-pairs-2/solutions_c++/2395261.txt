#include <iostream>
#include <bits/stdc++.h>
#define ll long long int 
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	ll n,c=0,k=0;
    	cin>>n;
    	int a[n],b[n];
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+n);
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]==a[i+1])
    		{
    			c=c+1;
    		}
    		else
    		{  k=((c*(c+1))/2) +k;
    			c=0;
    		}
    	}
    	cout<<k+n<<endl;
    }
    return 0;
}
