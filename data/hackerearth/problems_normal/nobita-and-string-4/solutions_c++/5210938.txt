#include "bits/stdc++.h"
using namespace std;
int main()
{
	long int tc;
	cin >> tc;
	char a1[2];
	cin.getline(a1,2);
	while(tc--)
	{
		char arr[100001];
		long int arrC=0;
		cin.getline(arr,100001);
		for(arrC=0;arr[arrC]!='\0';arrC++);
		
		for(long int j=arrC-1;j>=0;)
		{
			char brr[100001];
			long int  brrC=0;
			
			long int i=j;
			while(i>=0 && arr[i]!=' ')
			{
				brr[brrC++]=arr[i];
				i--;
			}
			for(long int i=brrC-1;i>=0;i--)
				cout << brr[i];
			while(arr[i] == ' ' && i>=0)
			{
				cout << ' ';
				i--;	
			}	
			j=i;
		}
		cout << endl;
	}
	return 0;
}
