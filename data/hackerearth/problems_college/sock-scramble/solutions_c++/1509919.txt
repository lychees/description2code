#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n,num[1000000],a[1000000]={0};
    	cin>>n;
    	for(int i=0;i<n;i++){
    		int temp;
    		cin>>num[i];
    		temp = num[i];
    		a[temp]++;
    	}
    	for(int i =0;i<n;i++){
    		if(a[num[i]] == 1){
    			cout<<num[i]<<endl;
    			break;
    		}
    	}
    	
    	
    }
    return 0;
}
