#include <iostream>
using namespace std;

int main()
{    int i,f=0;
    char a[1000002];
     char b[1000002];
    cin>>a;
    cin>>b;
    int q[26]={0};
    int w[26]={0};
    for(i=0;a[i]!='\0';i++)
    q[a[i]-65]++;
    for(i=0;b[i]!='\0';i++)
    w[b[i]-65]++;
    for(i=0;i<26;i++)
    {
    	if(q[i]!=w[i])
    	{
    		f=1;
    		break;
    	}
    }
    if(f==1)
    cout<<"WRONG";
    else
    cout<<"CORRECT";
    return 0;
}
