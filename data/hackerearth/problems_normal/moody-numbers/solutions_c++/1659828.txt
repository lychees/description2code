typedef unsigned long long int uint64;

int dp[200];
int vis[200];
int counter=1;
int sum_dig(int64 x){
	int ret=0;
	while(x){
		ret+=x%10;
		x/=10;
	}
	return ret;
}

int go(int val){
	if(dp[val]!=-1)
	return dp[val];
	if(vis[val]==counter)
	return 0;
	vis[val]=counter;
	if((val==1)||(val==4)){
		dp[val]=1;
		return 1;
	}
	int sqr=val*val;
	int x=sum_dig(sqr);
	dp[x]=go(x);
}

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int64 n;
		scanf("%lld",&n);
		bool flag=false;
		while(1){
			if((n==1)||(n==4)){
				flag=true;
				break;
			}
			n=n*n;
			n=sum_dig(n);
			if(vis[n]==counter)
			break;
			vis[n]=counter;
		}
		counter++;
		if(flag)
		printf("YES\n");
		else
		printf("NO\n");
	}
	return 0;
}
