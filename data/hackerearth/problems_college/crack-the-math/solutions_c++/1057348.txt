#include<iostream>

using namespace std;

char xor1(char a,char b)
{
        if(a==b)
                return '0';
        else
                return '1';
}

int main()
{
        int t;
        cin >> t;
        while(t--)
        {
                char a[200],b[200];
                cin >> a;
                cin >> b;
                for(int i=0;a[i] !='\0';i++)
                        cout << xor1(a[i],b[i]);
                cout << endl;
        }
}
