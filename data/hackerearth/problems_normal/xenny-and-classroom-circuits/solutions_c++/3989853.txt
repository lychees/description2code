#include <bits/stdc++.h>
using namespace std;

int main()
{
  long long n,i,t;
  string str;
  cin>>t;

  while(t--)
  {
  	int count1,count2=count1=0;
  	cin>>n;
  	cin>>str;
  	for(i=0;i<2*n;i+=2)
  	{
  		if(str[i]!='B')
  		{
  			count1++;
  		}
  	}
  	for(i=1;i<2*n;i+=2)
  	{
  		if(str[i]!='B')
  		{
  			count2++;
  		}
  	}
  	cout<<min(count1,count2)<<endl;
  }

  return 0;
}
