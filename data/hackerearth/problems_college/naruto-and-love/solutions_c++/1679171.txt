#include <bits/stdc++.h>
#define ll long long

using namespace std;

int a[200005];
int b[200005];
int n, m;

bool pre(int tt)
{
    int i = 0, j = 0;

    while(i < n && j < m)
    {
        if(abs(a[i]-b[j]) <= tt)
        {
            i ++;
            j ++;
        }
        else
        {
            j ++;
        }
    }

    if(i == n)
        return 1;

    else
        return 0;
}


int main()
{
    int t;

    scanf("%d", &t);

    while (t--) {
        int i;

        scanf("%d %d", &n, &m);

        for(i = 0; i < n; i ++)
            scanf("%d", &a[i]);

        for(i = 0; i < m; i ++)
            scanf("%d", &b[i]);

        sort(a, a + n);
        sort(b, b + m);

        int l = 0, r = 500000007, mid;
        while(l < r)
        {
            mid = (l+r)/2;

            if(pre(mid) == 1)
                r = mid;

            else
                l = mid+1;
        }

        printf("%d\n", l);
    }

    return 0;
}
