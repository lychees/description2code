#include<bits/stdc++.h>
using namespace std;
int n;
int a[5005];
int dp[5005];
int main()
{   scanf("%d",&n);
    int i,j;
    for(i=1;i<=n;i++)
    {   scanf("%d",&a[i]);
	}
	for(i=1;i<=n;i++)
	{   int temp=0;
	   for(j=1;j<i;j++)
	   {  if(((a[i]>0&&a[j]<0)||(a[i]<0&&a[j]>0))&&(abs(a[i])>abs(a[j])))
	      temp=max(temp,dp[j]);
	   }
	   dp[i]=temp+1;
	}
	int megh=1;
	for(i=1;i<=n;i++)
	megh=max(megh,dp[i]);
	printf("%d\n",megh);
	return 0;
    
}
