#include<iostream>
#include<string>
using namespace std;
int main()
{int t,countlower=0,countupper=0,countspecial=0;
 string s;
 cin>>t;
 if(t>10 || t<1)
  cout<<"Invalid Test"<<endl;
 else
  {while(t>0) 
    {cin>>s;
     countlower=0;
     countupper=0;
     countspecial=0;
     if(s.length()>100)
       cout<<"Invalid Input"<<endl;
     else
      {for(int i=0;i<s.length();i++)
        {if(s.at(i)>=65 && s.at(i)<=90)
          countupper++;
         else if(s.at(i)>=97 && s.at(i)<=122)
          countlower++;
         else
          countspecial++;
        }
       if(countupper==0 && countlower==0)
         cout<<"Invalid Input"<<endl;
       else if(countupper==0 || countlower==0)
         cout<<0<<endl;
       else if(countupper<countlower)
         cout<<countupper<<endl;
       else
         cout<<countlower<<endl;
      }
     t--; 
    }
  }
 return 0; 
}
