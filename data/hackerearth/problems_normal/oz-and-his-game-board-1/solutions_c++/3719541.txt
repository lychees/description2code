#include <iostream>
using namespace std;
typedef long long int lli;
int main()
{
	int t;
	cin>>t;
	while(t--){
		lli n, m, a, b, c;
		cin>>n>>m>>a>>b>>c;
		lli ans1, ans2;
		if(n>m){
			ans1 = c*m*2;
			ans1 = ans1 + (n-m)*a;
		} else {
			ans1 = c*n*2;
			ans1 = ans1 + (m-n)*b;
		}
		ans2 = a*n + b*m;
		if(ans1>ans2){
			cout<<ans1<<endl;
		} else {
			cout<<ans2<<endl;
		}
	}
    return 0;
}
