#include<iostream>
using namespace std;

int solve(int n,int d,int r){
    int rem=n,ans;
    for(int i=1;i<=r;i++){
        if(rem==0) return 0;
        ans=(10*rem)/d;
        rem=(10*rem)%d;
    }
    return ans;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,d,r;
        cin>>n>>d>>r;
        cout<<solve(n,d,r)<<"\n";
    }
    return 0;
}
