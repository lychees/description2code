#include <iostream>
using namespace std;

int main()
{
	long T,i,j;
	long A[200],B[200];
    cin>>T;
    for(i=0;i<T;i++){
    	cin>>A[i]>>B[i];
    }
    for(i=0;i<T;i++){
    	for(j=A[i];j<=B[i];j++){
    		A[i]=A[i]&B[i];
    	}
    }
    for(i=0;i<T;i++)
    {
    cout<<A[i]<<endl;
    }
    return 0;
}
