#include<iostream>
#include<vector>
#include <algorithm>
int main(){
	std::string input;
	size_t testCase;
	size_t n;
	std::cin>>testCase;
	for ( size_t loop = 0; loop < testCase ; ++ loop ){
		std::cin>>input;
		std::cin>>n;
		std::vector<bool> bb;
		for ( std::string::iterator it=input.begin() ; it != input.end() ; ++it){
//			std::cout<<*it<<std::endl;
			int k = (int)*it;
//			std::cout<<k<<std::endl;
			if ( k == 49 ){
//				std::cout<<"true\n";
				bb.push_back(true);
			}
			else{

//				std::cout<<"false\n";
				bb.push_back(false);
			}
		}
		for ( size_t loop2 = 1 ; loop2 <= n ; ++loop2 ){
			if ( loop2 %2 == 1){
				bb.flip();
			}
		}

		std::cout<<(size_t)std::count(bb.begin(),bb.end(), true )<<std::endl;
	}
}
