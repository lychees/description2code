 #include"bits/stdc++.h"
     
    using namespace std;
     
    int main() {
        int N,ans;
        cin >> N;
        while (N--) {
            string a, b;
            cin >> a >> b;
            if (a.size() > b.size()) swap(a, b);
             ans = a.size();
            for (int i = 0; i < b.size() - a.size() + 1; i++) {
    			int d = 0;
    			for (int j=0; j<a.size();j++) {
    				if (a[j] != b[j + i]) d++;
    			}
    			ans = min(ans,d);
    		}
    		cout << ans << endl;
        }
        return 0;
    }
