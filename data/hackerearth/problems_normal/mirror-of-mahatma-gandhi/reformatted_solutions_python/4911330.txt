t = int(raw_input())
for i in range(t):
    y = raw_input()
    x = int(y)
    c = 0
    b = list()
    while (x > 0):
        if (x % 10 in [2, 3, 4, 5, 6, 7, 9]):
            c = 1
            break
        b.append(x % 10)
        x = x/10
    while (len(y) > len(b)):
        b.append(0)

    for j in range(len(b)/2):
        if (b[j] != b[len(b)-1-j]):
            c = 1
            break
    if (c == 0):
        print "YES"
    else:
        print "NO"
