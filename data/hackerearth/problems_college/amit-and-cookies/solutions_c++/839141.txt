#include<bits/stdc++.h>
using namespace std;

int main()
{
	int N,K,i,sum;
	cin>>N>>K;
	int a[N];
	for(i=0;i<N;i++)
		cin>>a[i];
	sort(a,a+N);
	sum = 0;
	for(i=0;i<N;i++)
	{
		sum += a[i];
		if(sum > K)
			break;	
	}
	cout<<i<<endl;
 	return 0;
}
