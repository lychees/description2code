#include <bits/stdc++.h>
using namespace std;

int main()
{	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		if(n%21==0)
		{
			cout<<"The streak is broken!\n";
			continue;
		}
		stringstream ss;
		ss << n;
		string s = ss.str();
		int flag=0;
		for(int i=0;i<s.size()-1;i++)
		{
			if(s[i]=='2'&&s[i+1]=='1')
			{
				flag=1;
				break;
			}
			
		}
		if(flag==0)
		{
			cout<<"The streak lives still in our heart!\n";
		}
		else
		{
			cout<<"The streak is broken!\n";
		}
		
	}
    return 0;
}
