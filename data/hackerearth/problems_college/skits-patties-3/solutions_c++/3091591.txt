#include<bits/stdc++.h>
#define L 10000005
using namespace std ;
bool visited[L],possible[L] ;
bool ispossible(int x)
{
    if(x<0) return false ;
    if(visited[x]) return possible[x] ;
    visited[x]=true ;
    possible[x]=ispossible(x-6) | ispossible(x-9) | ispossible(x-20) ;
    return possible[x] ;
}
int main()
{
    visited[0]=true ; possible[0]=true ;
    int T ; scanf("%d",&T );
    while(T--)
    {
        int A ; scanf("%d",&A);
        if(ispossible(A)) printf("True\n" );
        else printf("False\n" );
    }
    return 0 ;
}
