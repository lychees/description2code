#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll  r[1<<20];
int main()
{
	ios::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);
	ll n;
	cin>>n;
	for(int i=1;i<=n;++i)
	{
		cin>>r[i];
	}
	for(int i=2;i<=n;++i)
	{
		r[i]+=r[i-1];
	}
	int q;
	cin>>q;
	while(q--)
	{
		int a,b;
		cin>>a>>b;
		cout<<r[b]-r[a-1]<<"\n";
	}

	return 0;
}
