#include <iostream>
using namespace std;

int main()
{
    int n,x,a[10000000],i,unsolved=0,solved=0;
    cin>>n>>x;
    for(i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(a[i]>x)
    	{
    		unsolved++;
    		if(unsolved==2)
    		break;
    	}
    	else
    	solved++;
    }
    cout<<solved;
    return 0;
}
