#include<cstdio>
using namespace std;
typedef long long ll;
int wp[1001];//width possible
int dp[1001];
const int mod = 1000000007;
int f(int n)
{
    for (int k=1;k<=n;k++) {
	ll out=0;
	for (int i=0;i<k;i++)
	{
	    if (wp[i])
		out=(out+dp[k-i-1])%mod;
	}
	if (wp[k])
	    out=(out+1)%mod;
	dp[k]=out;
    }
    return dp[n];

}
int main()
{
    int n,k,w[1001];
    dp[0]=1;
    scanf("%d %d",&n,&k);
    for (int i=0;i<k;i++)
	scanf("%d",&w[i]);
    wp[0]=1;
    for (int i=0;i<k;i++)
    {
	for (int j=0;j+w[i]<=n;j++)
	    if (wp[j])
		wp[j+w[i]]=1;
    }
    int out=f(n);
    printf("%d",out);

    return 0;
}
