#include <bits/stdc++.h>
using namespace std;
long long int a[500006];
long long int mod = 1000000007;
int main()
{
    int t,i,j;
    scanf("%d",&t);
    for(i=0;i<=500004;i++)
    a[i] = i;
    for(i=2;i<=500000;i++) {
    	j = i + i;
    	while(j<=500000) {
    		a[j] = a[j]/a[i];
    		j = j + i;
    	}
    }
    for(i=2;i<=500000;i++) {
    	a[i] = (a[i]*a[i-1])%mod;
    }
    while(t--) {
    	int b;
    	scanf("%d",&b);
    	printf("%lld\n",a[b]);
    }
    return 0;
}
