def seq(n):
    ret_val = 0
    if (n <= 2):
        if (n == 1):
            ret_val = 1
        if (n == 2):
            ret_val = 2
    else:
        if (n % 2 != 0):
            ret_val = (int(seq(n-1))*int(seq(n-2)))
        if (n % 2 == 0):
            ret_val = (int(seq(n-1))+int(seq(n-2)))
    return ret_val


t = int(input())
while (t > 0):
    n = int(input())
    final = int(seq(n))
    print(final)
    t = t-1
