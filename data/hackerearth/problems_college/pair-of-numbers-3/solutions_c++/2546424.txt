#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a,c=0;
		long long int b;
		for(int i=1;i<=n;i++)
		{
			for(int j=i+1;j<=n;j++)
			{
				a=i+j;
				b=i*j;
				if(b%a==0)
				c++;
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
