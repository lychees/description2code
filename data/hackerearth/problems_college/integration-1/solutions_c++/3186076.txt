#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    long n[t],p[t],u[t],r[t],s[t],count=0;;
    for(int i=0;i<t;i++)
    cin>>n[i]>>p[i]>>u[i]>>r[i]>>s[i];	
    for(int i=0;i<t;i++)
    {  
    	for(long j=0;j<n[i];j++)
    	{
    		if(p[i]+r[i]<=u[i])
    		p[i]=p[i]+r[i];
    	  else	if(p[i]-s[i]>=0)
    		p[i]=p[i]-s[i];
    	
    	  
    	}
    	cout<<"Case #"<<i+1<<": "<<p[i]<<endl;
    }
    
    return 0;
}
