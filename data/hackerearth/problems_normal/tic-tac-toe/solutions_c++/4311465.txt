ll M=1000000007;
 
 
ll fast_pow(ll b,ll n,ll m)
{
	if(n==0)
		return 1;
	if(n==1)
		return b;
	ll halfn=fast_pow(b,n/2,m);
	if(n%2==0)
	return (halfn*halfn)%m;
	else
	return (((halfn*halfn)%m)*b)%m;
}
 
ll mmi_fermat(ll n,ll m)
{
	return fast_pow(n,m-2,m);
}
 
int main()
{
	ios_base::sync_with_stdio(false);
	ll t,n,max,min;
	cin>>t;
	while(t--)
	{
		cin>>n;
	//	min=((n*(n-1)*(n-1))/4)%1000000009;
	//	max=((n*(n-1)*(2*n-1))/6)%1000000009;
		min = (((((n*(n-1))%M)*(n-1))%M)*mmi_fermat(4,M))%M;
		max = (((((n*(n-1))%M)*(2*n-1))%M)*mmi_fermat(6,M))%M;
		cout<<min<<" "<<max<<endl;	
	}
	return 0;
}
