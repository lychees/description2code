#include <bits/stdc++.h>
using namespace std;

int main() {
	long long int t,i,j,k,l,m,n,o;
	cin>>t;
	vector<pair<long int, long int> >v;
	pair<long int, long int> p,y;
	while(t--)
	{
		cin>>i>>j;
		p=make_pair(i,j);
		v.push_back(p);
	}
	
	sort(v.begin(),v.end());
	
	vector<long int> u;
	
	for(i=0;i<v.size();i++)
	{
		p=v[i];
		y=v[i+1];
		
		if(p.first==y.first)
		{
			u.push_back(p.second);
		}
		
		else
		{
			u.push_back(p.second);
			
			k=u.size();
			
			sort(u.begin(),u.end());
			
			for(j=k-1;j>=0;j--)
			{
				cout<<p.first<<" "<<u[j]<<"\n";
			}
			
			u.clear();
		}
		
		
		
		//cout<<p.first<<" "<<p.second<<"\n";
	}
	
	
	
	
}
