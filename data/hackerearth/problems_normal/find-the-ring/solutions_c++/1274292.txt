#include<iostream>
//#include<cstdio>
//#include<cstring>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int i, n;
        cin>>i>>n;
        if(i==0)
        {
            if(n%2==0)
            cout<<0<<endl;
            else
            cout<<1<<endl;
        }
        else if(i==1)
        {
            if(n%2==0)
            cout<<1<<endl;
            else
            cout<<0<<endl;
        }
        else if(i==2)
        {
            if(n==0)
            cout<<2<<endl;
            else if(n%2==1)
            cout<<1<<endl;
            else 
            cout<<0<<endl;
        }

    }

    return 0;
}
