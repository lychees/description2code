 #include<bits/stdc++.h>
     using namespace std;
      int main(){
        int i,j,t,c,n;
        int a[1001];
        int b[1001];
        cin>>t;
        while(t--)
        {
          c=0;
        scanf("%d", &n);
           for(i = 0;i <n;i++)
            {
            cin>>a[i];
            }
            for(int j=0;j<n;j++)
            {
            cin>>b[j];

            }
            sort(a,a+n);
            sort(b,b+n);
            for(int k=0;k<n;k++)
            {
            c+=a[k]*b[k];
            }
            cout<<c<<endl;
            }
 }
