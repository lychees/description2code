#include <bits/stdc++.h>
using namespace std;
 
int main()
{
	int i, t, a_size, b_size;
	
	string a, b;
	
	cin>>t;
	
	char x;
	
		x=getchar();
	int ar[26], br[26];
	while(t--)
	{
	
		getline(cin, a);
	//	x=getchar();
		getline(cin, b);
		
		for(i=0;i<26;i++)
			ar[i]=br[i]=0;
	//	cout<<a<<" "<<b<<endl;
			
		a_size=a.size();
		b_size=b.size();
		
		for(i=0;i<a_size;i++)
			ar[a[i]-'a']++;
		
		for(i=0;i<b_size;i++)
			br[b[i]-'a']++;
		
		
		for(i=0;i<26;i++)
		{
			if((ar[i]>=1) && br[i]>=1)
				break;
		}
		
		if(i==26)
			cout<<"NO\n";
		else
			cout<<"YES\n";
	}
    return 0;
}
