#include<cstdio>
#include<iostream>
#include<vector>
using namespace std;

#define Lim 1000000
vector<bool> b((Lim>>1)+100);
vector<int> p;

void sieve(){
    for(int i=3;i<=9257;i+=2)
        if(!b[(i-3)>>1])
            for(int j=i*i;j<Lim;j+=i)
                if((j&1)==1)
                    b[(j-3)>>1]=true;
    p.push_back(2);
    for(int i=0;2*i<Lim;i++)
        if(!b[i])
            p.push_back(2*i+3);
}

int leastFactor(int n){
    for(int i = 0; p[i]*p[i] <= n; i++) {
        if(n% p[i] ==0)
            return p[i];
    }
    return n;
}

int main(){
		sieve();
		int k,q;
		for(scanf("%d",&q);q>0;q--){
            scanf("%d",&k) ,
            printf("%d\n",k-leastFactor(k));
        }
}
