#include <stdio.h>
#include <string.h>
#include <iostream>

using namespace std;

void solve(char *s)
{
  int i = 0, n = 0;
  int l = strlen(s);

  while(s[i])
  {
    if(s[i] != '_') n += s[i] - '0';
    ++i;
  }

  if(n % 9 == 0 && (n + 9) % 9 == 0)
  {
    printf("0 or 9\n");
    return;
  }

  for(int i = 1; i <= 8; ++i)
   if((n + i) % 9 == 0)
   {
     printf("%d\n", i);
     return;
   }
}

int main()
{
  int t;
  char s[1000];

  scanf("%d", &t);
  while(t--)
  {
    scanf("%s", &s);
    solve(s);
  }

  return 0;
}
