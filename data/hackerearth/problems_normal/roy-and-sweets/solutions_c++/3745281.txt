#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll int arr[1000001],dp[1000001];

int main()
{
    int n;
    cin>>n;

    for(int i=0;i<n;i++)
    {
        int s,q;
        cin>>s>>q;
        arr[s]=q;
    }

    dp[1000000]=arr[1000000];
    for(int i=1000000-1;i>=0;i--)
    {
        dp[i]=max(dp[i+1],arr[i]);
    }

    int g;
    cin>>g;
    ll int ans=0;
    while(g--)
    {
        int m;
        cin>>m;

        int aa=dp[m]/m;
        ans+=100*aa;
    }
    cout<<ans<<endl;
}
