#include <iostream>
using namespace std;
#include <stdio.h>

typedef unsigned long long int INT;

INT function1( INT *a)
{
  INT num=0;
   while(*a)
   {
       num=num*10+(*a)%10;
       (*a)=(*a)/10;
   }
   return num;
}
int main()
{
     INT N,a,b,res;
     cin>>N;
     while(N--)
     {
        cin>>a>>b;
         a=function1(&a);
         b=function1(&b);
         res=a+b;
         res=function1(&res);
         cout<<res<<endl;
     }

    return(0);
}
