#include<bits/stdc++.h>
using namespace std;
string s;
int n;
int cal_eql()
	{
		int cnt1[26]={0};
		int cnt2[26]={0};
		string w=s;
		//sort(w.begin(),w.begin()+n/2);
		//sort(w.begin()+n/2,w.end());
		for(int i=0;i<n/2;++i)
			{
				int x=w[i]-97;
				cnt1[x]++;
			}
		for(int i=n/2;i<n;++i)
			{
				int x=w[i]-97;
				cnt2[x]++;
			}
		int min_sum=0;
		for(int i=0;i<26;++i)
			{
				min_sum=min_sum+min(cnt1[i],cnt2[i]);
			}
		min_sum=n/2-min_sum;
		return min_sum;
	}
int cal_grt()
	{
		int i=0;
		int j=n/2;
		int good=0;
		int xtra_cost=0;
		string w=s;
		
		for(int i=0;i<n/2;++i)
			{
				if(w[i]=='a')
					{
						w[i]='z';
						xtra_cost++;
					}
			}
		for(int j=n/2;j<n;++j)
			{
				if(w[j]=='z')
					{
						w[j]='a';
						xtra_cost++;
					}
			}
		sort(w.begin(),w.begin()+n/2);
		sort(w.begin()+n/2,w.end());
		i=0;j=n/2;
		while(i<n/2 && j<n)
			{
				if(w[i]>w[j])
					{
						++good;
						++i;
						++j;
					}
				else
					++i;
			}
		int res=(n-2*good)/2+xtra_cost;
		return res;
	}
		
int cal_less()
	{
		int i,j,good,xtra_cost;
		xtra_cost=0;
		string w=s;
		
		for(int i=0;i<n/2;++i)
			{
				if(w[i]=='z')
					{
						xtra_cost++;
						w[i]='a';
					}
			}
		for(int i=n/2;i<n;++i)
			{
				if(w[i]=='a')
					{
						xtra_cost++;
						w[i]='z';
					}
			}
		sort(w.begin(),w.begin()+n/2);
		sort(w.begin()+n/2,w.end());
		good=0;
		i=0;
		j=n/2;
		while(i<n/2 && j<n)
			{
				if(w[i]<w[j])
					{
						++good;
						++i;
						++j;
					}
				else
					++j;
			}
		int res=(n-2*good)/2+xtra_cost;
		return res;
	}

int main()
	{
		
		cin>>n;
		//string s;
		
		cin>>s;
		
		int res1,res2,res3,res;
		res1=cal_eql();
		res2=cal_grt();
		res=min(res1,res2);
		res3=cal_less();
		res=min(res,res3);
		cout<<res<<endl;
		//cout<<s2<<endl;
	}
