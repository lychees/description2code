#include <iostream>
#include <string.h>
#include <string>
#include <vector>
using namespace std;

int main()
{
    int length;
    cin>>length;

    if(length < 1 && length > 100)
    {
        return 0;
    }

    int i=0;
    vector<string> var;
    string var1;
    for(i=0; i<length; ++i)
    {
        cin>>var1;
        var.push_back(var1);
    }

    for(i=0; i<length; ++i)
    {
        cout<<strlen(var[i].c_str())<<endl;
    }

    return 0;
}
