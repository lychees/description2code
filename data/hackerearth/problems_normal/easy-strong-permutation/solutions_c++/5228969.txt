#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    long int a[n+5],b[n+5];
    for(int i = 0;i < n;i++)
    	cin>>a[i];
    sort(a,a+n);	
    int i = 0,j = n-1,k = 0;
    while(1)
    {
    	b[k++] = a[i++];
    	if(k==n)
    		break;
    	b[k++] = a[j--];
    	if(k==n)
    		break;
    }
    long long int ans = 0;
    for(i = 0;i < n-1;i++)
    	ans += abs(b[i] - b[i + 1]);
    ans += abs(b[0] - b[n - 1]);
    cout<<ans<<endl;
    return 0;
}
