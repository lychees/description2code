#include<iostream>
#include<stdlib.h>
using namespace std;
int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}


int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,a[100000];
		cin>>n;
		for(int i=0;i<n;i++)
		cin>>a[i];
		qsort (a,n, sizeof(int), compare);
		for(int i=0;i<n;i++)
		cout<<a[i]<<" ";
		cout<<endl;		
	}
}
