#include <stdio.h>
int primes[100001];
int main()
{
	primes[1]=1;
	for(int i=2;i<=400;i++)
    {
    	if(!primes[i])
    	{
    		for(int j=i;j*i<=100000;j++)
    			primes[j*i]=1;
    	}	
    }
    int t,n;
    scanf("%d",&t);
    while(t--){
	    scanf("%d",&n);
	    if(n==3)
	    {
	    	puts("3");
	    	continue;
	    }
	    if(primes[n])
	    	puts("1");
	    else
	    	puts("2");
    }
    return 0;
}
