#include <iostream>
using namespace std;
 
int tries(int x)
{
int a=3,c=1;
if(x<=1)return 0;
if(x==2)return 1;
while(x>a)
{
 c++;
 a*=3;
 
}
 
return c;
 
}
 
int main()
 
{
   
   int t,i,b;
   cin>>t;
   for(i=0;i<t;i++)
   {
   	cin>>b;
   	cout<<tries(b)<<"\n";
   	
   }
   
   return 0;
}
