#include <iostream>
using namespace std;

int main()
{
	int n,test;
	long int x,y,j,fou=0;
    cin>>n>>test;
    long int arr[n];
    for(int i=0;i<n;i++)
    {
    	cin>>x;
    	arr[i]=x;
    }
    for (int i=0;i<test;i++)
    {
    	cin>>x;
    	if(x==1)
    	{
    		cin>>y;
    		for(int p=0;p<n;p++)
    		{
    			if(arr[p]>=y)
    			{
    				fou=p+1;
    				break;
    			}
    			}
    		if(fou>0)
    			cout<<fou<<endl;
    		else
    			cout<<"-1"<<endl;
    		fou=0;
    		
    		}
    	if(x==0)
    	{
    		cin>>y>>j;
    		arr[y-1]=j;
    	}
    }
    return 0;
}
