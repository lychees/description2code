for _ in xrange(int(raw_input())):
    w = raw_input()
    r = sorted(raw_input())
    n = len(w)
    m = len(r)
    index = 0
    s = n*['0']
    for i in xrange(n):
        s[i] = w[i]
    for i in xrange(n):
        if s[i] > r[index]:
            s[i] = r[index]
            index += 1
        if index == m:
            break
    print ''.join(s)
