#include <bits/stdc++.h>
using namespace std;
bool prime[100001];

int phi(int n)
{ 
int result = n; 
for (int p=2; p*p<=n; ++p)
{
if (n % p == 0)
{
while (n % p == 0)
n /= p;
result -= result / p;
}
}

if (n > 1)
result -= result / n;
return result;
}
void Sieve()
{
prime[1] = false;
memset(prime, true, sizeof(prime));

for (long p=2; p*p<=100000; p++)
{
if (prime[p] == true)
{
for (long i=p*2; i<=100000; i += p)
prime[i] = false;
}
}
}

int main()
{
int t ;
int flag=1 ;
Sieve() ;
scanf("%d",&t) ;
while(t--)
{
int n ;
scanf("%d",&n);
if(n==2||n==1)
{
printf("Incorrect\n");
continue;
}
if(!prime[phi(n)])
printf("Incorrect\n");
else
printf("Correct\n");
}	
return 0;
}
