#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int t,k,val;
    
    cin>>t;
    
    while(t--) {
        int n,i,j,arr[4005] = {0};
        
        cin>>n>>k;
        
        arr[0] = 1;
        
        while(n--) {
            cin>>val;
                      
            for(i=0;    i+val<=3000;    i++) {
                if(arr[i]) {
                    arr[i+val] = 1;
                }
            }
        }
        
        for(i=k;    i>=0;    i--) {
            if(arr[i]) {
                cout<<i<<"\n";
                break;
            }
        }
    }
    
    return 0;
}
