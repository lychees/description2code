#include<iostream>
using namespace std;

void change(char game[][101] , int n , int m , int i , int j,char game2[][101])
{
	//cout<<"a\n"<<i<<" "<<j<<" ";
	int count = 0,f;
	
	if( (i < 0 || j < 0 || i >= n || j >= m) )
	{
		return ;
	}
	if(game[i][j]=='*' || game2[i][j]==1)
		return;
	if( i-1 >= 0 )
	{
		if( j-1 >= 0 )
		{
			game[i-1][j-1]=='*' ? count++ : f=1; 
		}
		if( j+1 < m )
		{
			game[i-1][j+1]=='*' ? count++ : f=1; 
		}
		game[i-1][j]=='*' ? count++ : f=1 ;
	}
	if( i+1 < n )
	{
		if( j-1 >= 0 )
		{
			game[i+1][j-1]=='*' ? count++ : f=1; 
		}
		if( j+1 < m )
		{
			game[i+1][j+1]=='*' ? count++ : f=1; 
		}
		game[i+1][j]=='*' ? count++ : f=1 ;
	}
	if( j-1 >=0 )
		game[i][j-1]=='*' ? count++ : f=1 ;
		
	if( j+1 < m )
		game[i][j+1]=='*' ? count++ : f=1 ;
	game[i][j] = count + 48;
	game2[i][j] =1;
	if(count==0)
	{
		change(game,n,m,i-1,j,game2);
        change(game,n,m,i,j-1,game2);
        change(game,n,m,i+1,j,game2);
        change(game,n,m,i,j+1,game2);
        
        change(game,n,m,i-1,j-1,game2);
        change(game,n,m,i-1,j+1,game2);
        change(game,n,m,i+1,j-1,game2);
        change(game,n,m,i+1,j+1,game2);
	}
	return;
}

int main()
{
    int test_cases;
    cin>>test_cases;
    while(test_cases>0)
    {
        int X,Y,i,X1,Y1,flag=0;
        char game[100][101] , game2[100][101];
        cin>>X>>Y;
        for(i=0;i<X;i++)
        {
        	for(int j=0;j<Y;j++)
        		game2[i][j] = 0;
        }
        for(i=0;i<X;i++)
            cin>>game[i];
        for(i=0;i<X;i++)
        {
        	for(int j=0;j<Y;j++)
        	{
        		if(game[i][j]=='c')
        		{
        			flag=1;
        			X1 = i;
        			Y1 = j;
        			break;
        		}
        	}
        	if(flag==1)
        		break;
        }
        change(game,X,Y,X1,Y1,game2);
        for(i=0;i<X;i++)
            cout<<game[i]<<endl;
        
        test_cases--;
    }
    return 0;
}
