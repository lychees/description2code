for _ in xrange(int(raw_input())):
    S = raw_input()
    R = S[::-1]
    flag = 0
    N = len(R)
    for i in xrange(1, N):
        if abs(ord(R[i])-ord(R[i-1])) != abs(ord(S[i])-ord(S[i-1])):
            flag = 1
            break
    if flag:
        print "Not Funny"
    else:
        print "Funny"
