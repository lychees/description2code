#include<iostream>
#include<cstring>
#define MOD 1000000007
using namespace std;
long long int m(long long int a)
{
	
		return (a%1000000007+1000000007)%1000000007;
}
long long int fn(long long int n)
{
long long fib[2][2]= {{0,1},{-1,3}},ret[2][2]= {{1,0},{0,1}},tmp[2][2]= {{0,0},{0,0}};
    int i,j,k;
    if(n==1)
    return 1;
    else
    {
    	n=n-1;
    while(n)
    {
        if(n&1)
        {
            memset(tmp,0,sizeof tmp);
            for(i=0; i<2; i++)
			 for(j=0; j<2; j++) 
			 for(k=0; k<2; k++)
                        tmp[i][j]=m(tmp[i][j]+m(ret[i][k]*fib[k][j]));
            for(i=0; i<2; i++)
			 for(j=0; j<2; j++) 
			 ret[i][j]=tmp[i][j];
        }
        memset(tmp,0,sizeof tmp);
        for(i=0; i<2; i++) 
		for(j=0; j<2; j++) 
		for(k=0; k<2; k++)
                    tmp[i][j]=m(tmp[i][j]+m(fib[i][k]*fib[k][j]));
        for(i=0; i<2; i++) for(j=0; j<2; j++) 
		fib[i][j]=tmp[i][j];
        n=n>>1;
    }
//    cout<<ret[0][0]+3*ret[0][1]<<endl;
    return m(ret[0][0]+ret[0][1]*3);
    }
}
int main()
{
	long long int n,t;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cout<<m(fn(n))<<endl;
		
	}
	return 0;
}
