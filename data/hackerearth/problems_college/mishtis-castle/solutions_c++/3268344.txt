#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define maxn 1001
ll a[maxn][maxn],dp[maxn][maxn];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,c,x;
		cin>>n>>c;
		for(ll i=1;i<=n;i++)
		{
			for(ll j=1;j<=n;j++)
			cin>>a[i][j];
		}
		for(ll i=1;i<=n;i++)
		{
			for(ll j=1;j<=n;j++)
			{
				dp[i][j]=a[i][j];
				x=max(dp[i-1][j],dp[i][j-1]);
				dp[i][j]+=x;
			}
		}
		if(dp[n][n]<c)cout<<"-1"<<endl;
		else cout<<dp[n][n]-c<<endl;
	}
	return 0;
}
