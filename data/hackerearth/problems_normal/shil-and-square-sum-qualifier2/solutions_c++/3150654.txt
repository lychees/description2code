#include<iostream>
#include<algorithm>
#include<vector>
#define MOD 1000000007
using namespace std;

long long a[1000005],b[1000005];

int main()
{
    long long i,j,n,m,temp,k,toSub=0,ans=0,cur=1;
    cin>>n>>k;
    for(i=1;i<=n;i++) cin>>a[i];
    b[0]=0;
    for(i=1;i<=n;i++) b[i]=(b[i-1]+a[i])%MOD;
    for(i=1;i<=n-k+1;i++) {
        if(i==1) {
            for(j=1;j<=k;j++) {
                ans=(ans+((j*j)%MOD*a[j])%MOD)%MOD;
                toSub=(toSub+(cur*a[j])%MOD)%MOD;
                cur+=2;
            }
        } else {
            ans=(ans+((k*k)%MOD*a[i+k-1])%MOD)%MOD;
            ans=(ans-toSub+MOD)%MOD;
            toSub=(toSub-a[i-1])%MOD;
            toSub=(toSub-(2*(b[i+k-2]-b[i-1]))%MOD+MOD)%MOD;
            toSub=(toSub+((2*k-1)*a[i+k-1])%MOD)%MOD;
        }
        cout<<ans<<" ";
    }
    return 0;
}
