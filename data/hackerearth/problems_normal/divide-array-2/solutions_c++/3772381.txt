#include<bits/stdc++.h>






using namespace std;
int main()
{
	long long int n,*a,q,d,divide=1;
	cin>>n;
    a=(long long int *)malloc(sizeof(long long int )*n);
	for(long long int i=0;i<n;i++)	{
	   scanf("%lld",(a+i));
	
	}
	cin>>q;
	for(long long  int i=0;i<q;i++)
	{
	   cin>>d;
	   divide*=d;
	  
	 if( divide>1000000)
	 break;
	
	
	}
	for (long long int i = 0; i <n; i++)
	{
	  if(divide>a[i])
	  cout<<0<<" ";
	  else
	  cout<<a[i]/divide<<" ";
	}
	
return 0;
}
