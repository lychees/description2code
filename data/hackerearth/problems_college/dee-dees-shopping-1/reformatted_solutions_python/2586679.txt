for i in range(0, x):
    y = input()
    c = 0
    t = y
    while True:
        if t % 8 == 0:
            t -= 8
            c += 1
        elif t % 6 == 0:
            t -= 6
            c += 1
        else:
            t -= 8
            c += 1
        if t == 0:
            print c
            break
        if t < 6:
            print "-1"
            break
