#include <iostream>
using namespace std;
#define MAX 10009 
long long int g[101][101];

void initialize(int n)
{
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			if(i==j)
			{g[i][j]=0;}
			else
			{g[i][j]=MAX;}
		}
}

int main()
{
    long long int c,max;
    int t,n,m,u,v,s,a,h;
    cin>>t;
    while(t--)
     {	
     	max=0;
    	cin>>n>>m;
    	initialize(n);
    	for(int i=1;i<=m;i++)
    	{
    		cin>>u>>v>>c;
    		g[u][v]=min(c,g[u][v]);
    		g[v][u]=min(c,g[v][u]);
    	}
    	
		for(int k=1;k<=n;k++)
			for(int i=1;i<=n;i++)
				for(int j=1;j<=n;j++)
					{
						g[i][j]=min(g[i][j],g[i][k]+g[k][j]);
					}
		
		cin>>s>>a>>h;
		
		for(int i=1;i<=n;i++)
		{
			if(i==s||i==a||i==h)
			continue;
			
			c=g[s][i]+g[i][a]+g[a][i]+g[i][h];
			if(c>max)
			max=c;
		}
		
		cout<<max<<"\n";
		
    }
    return 0;
}
