#include<bits/stdc++.h>
using namespace std;
long long pow1(long long a,long long b)
{
	if(b==0)
	return 1;
	else if(b==1)
	return a;
	else
	{
		if(b%2==0)
		{
			long long x=pow1(a,b/2);
			return x*x;
		}
		else
		return (a*pow1(a,b-1));
	}
}
int main()
{
	long long arr[11][51];
	for(long long i=0;i<11;i++)
	{
		for(long long j=0;j<51;j++)
		{
			arr[i][j]=0;
		}
	}
	for(long long i=2;i<=10;i++)
	{
		for(long long j=1;j<=50;j++)
		{
			if(j<i)
			{
				arr[i][j]=pow1(2,j);
			}
			else if(j==i)
			{
				arr[i][j]=pow1(2,j)-1;
			}
			else
			{
				long long k=i;
				long long l=j-1;
				while(k>0)
				{
					k--;
					arr[i][j]+=arr[i][l--];
				}
			}
		}
	}
	long long t;
	cin>>t;
	while(t--)
	{
		long long m,n;
		cin>>m>>n;
		cout<<arr[m][n]<<endl;
	}
}
