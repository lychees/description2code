#include <iostream>
using namespace std;

int main()
{
	int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	int i,a[n],j,count=1;
    	int s[10001]={0};
    	s[0]=1;
    	for(i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
    	for(i=0;i<n;i++)
    	{
    		if(a[i]==0) continue;
    		for(j=10000;j>0;j--)
    		{
    			if(j-a[i]>=0)
    			{
	    			if(s[j-a[i]]==1)
	    			{
	    				if(s[j]==0)
	    				{ 
	    					s[j]=1;
	    					count++;
	    				}
	    			}
    			}
    			
    		}
    	}
    	cout<<count<<endl;
    }
    return 0;
}
