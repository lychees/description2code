#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,i;
    scanf("%d",&n);
    map<string,int> m;
    pair<int,string> p[n];
    string tmp;
    for(i=0;i<n;i++){
    	cin>>tmp;
    	m[tmp]++;
    }    
    i=0;
    for(map<string,int>::iterator it=m.begin();it!=m.end();it++){
    	p[i]=make_pair(it->second,it->first);
    	i++;
    }
    n=i;
    sort(p,p+n);
    for(i=n-1;i>=0;i--)cout<<p[i].second<<' '<<p[i].first<<endl;
    return 0;
}
