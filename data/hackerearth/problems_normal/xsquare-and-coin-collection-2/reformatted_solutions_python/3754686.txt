t = int(raw_input())
for _ in xrange(t):
    n, k = map(int, raw_input().split())
    a = map(int, raw_input().split())
    ans = 0
    val = 0
    for i in xrange(n):
        if a[i] <= k:
            val += a[i]
        if val > ans:
            ans = val
        if a[i] > k:
            val = 0
    print ans
