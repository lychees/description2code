#include<iostream>
#include<stdlib.h>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,count1=0;
		double count=1;
		cin>>n;
		for(int i=1;i<=n;i++)
		count+=(log10(i));
		
		for(int i=5;i<=n;i*=5)
		count1+=n/i;
		
		cout<<(int(count)-count1)<<endl;
	}
}
