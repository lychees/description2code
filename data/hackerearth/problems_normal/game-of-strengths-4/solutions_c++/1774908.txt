#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll_t;
typedef unsigned long long ull_t;
#define PRE_CPP do { ios_base::sync_with_stdio(0);	cin.tie(0); } while(0)

const ull_t MOD = 1e9 + 7;
int main()
{
	PRE_CPP;
	int testcase, N;
	cin >> testcase;
	
	while (testcase--) {
		cin >> N;
		vector<ll_t> A(N);
		ll_t max = 0;
		for (int i = 0; i < N; i++) {
			cin >> A[i];
			if (A[i] > max) {
				 max = A[i];
			}
		}
		sort(A.begin(), A.end());

		ll_t sum = 0, product_n = N-1;
		for(int i = N - 1; i >= 0; i--)   {
            sum += A[i] * product_n;
            product_n -= 2;
        }
        sum %= MOD;
        
		cout << (sum * max) % MOD<< "\n";	
	}
 	return 0;
}
