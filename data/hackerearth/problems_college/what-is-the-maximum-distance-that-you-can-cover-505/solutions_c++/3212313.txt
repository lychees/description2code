#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;
    int arr[n];
    
    for(int i=0;i<n;i++)
    {
    	cin >> arr[i];
    }
    int m,l;
    double k=0;
    for(int i=0;i<n;i++)
    {
    	cin >> m;
    	double t=(double)arr[i]/(double)m;
    	if(t>k)
    	{
    		k=t;
    		//l=m;
    	}
    }
    cin >> l;
	long long int t=(l*k*1000);   
    cout.precision(3);
    cout<< fixed <<(double) t/1000 << endl;
    //printf("%.3f\n",l*k);
    
    
    
    return 0;
}
