#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    	int t;
	cin>>t;
while(t--)
	{
	int n;
	cin>>n;
	int *arr;
	arr=(int *)malloc(sizeof(int)*n);

	for(int i=0;i<n;i++)
	{
		int q;
		cin>>q;
		arr[i]=q;
	}

      sort(arr,arr+n);

	for(int i=n-1;i>=0;i--)
	{
		cout<<arr[i]<<" ";
	}
	cout<<endl;

}

    return 0;
}
