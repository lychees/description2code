#include<iostream>
 using namespace std;
 int main()
 {
     int n,m;
     int a[100000], b[100000];
     cin >> n >> m;
     long sum1 = 0, sum2 = 0;
     for(int i = 0; i < n; i++)
     {
         cin >> a[i];
         sum1 += a[i];
     }
     for(int i = 0; i < m; i++)
     {
         cin >> b[i];
         sum2 += b[i];
     }
     
     cout << sum1 * sum2 << endl;
     return 0;
 }
