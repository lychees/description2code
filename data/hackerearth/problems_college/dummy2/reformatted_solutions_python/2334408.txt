for t in range(input()):
    s = raw_input()
    ch = ""
    l = len(s)
    for i in range(0, l, 1):
        if (s[i] >= 'a' and s[i] <= 'z') or (s[i] >= 'A' and s[i] <= 'Z'):
            ch = ch+s[i]
        else:
            x = ch
            limit = int(s[i])-1
            for j in range(0, limit, 1):
                ch = ch+x
    print ch
