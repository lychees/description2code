#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll fun(map<ll,ll> &m,ll n)
{
	if(m.find(n)!=m.end())
	return m[n];
	ll var=max(n,fun(m,n/2)+fun(m,n/3)+fun(m,n/4));
	m[n]=var;
	return m[n];
}
int main()
{
	ll n,ans;
	map<ll,ll> m;
	m.insert(make_pair(0,0));
	while(scanf("%lld",&n)>0)
	{
		ans=max(n,fun(m,n/2)+fun(m,n/3)+fun(m,n/4));
		cout<<ans<<endl;
	}
    return 0;
}
