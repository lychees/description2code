//int dx[]= {-1,0,1,0};
//int dy[]= {0,1,0,-1};
int n,k;
int x1[100005],_y1[100005],x2[100005],y2[100005],c[100005];
vector<int> V[100005];
int main()
{
    scanf("%d %d",&n,&k);
    vector<int> vv;
    for(int i=1;i<=n;i++)
    {
        vv.pb(i);
    }
    for(int i=1;i<=n;i++)
    {
        V[i] = vv;
    }
    FOR(i,k)
    {
        char ss[2];
        scanf("%d %d %d %d %s",&x1[i],&_y1[i],&x2[i],&y2[i],ss);
        if(x1[i] > x2[i])
            swap(x1[i],x2[i]);
        if(_y1[i] > y2[i])
            swap(_y1[i],y2[i]);
        if(ss[0] == 'b')
            c[i] = 0;
        else
            c[i] = 1;
    }
    int ans = 0;
    for(int i=k-1;i>=0;i--)
    {
        int X1 = x1[i];
        int Y1 = _y1[i];
        int X2 = x2[i];
        int Y2 = y2[i];
        int C = c[i];
        for(int j=X1;j<=X2;j++)
        {

            vector<int>::iterator itlow = lower_bound(V[j].begin(),V[j].end(),Y1);
           // vector<int>::iterator itup = upper_bound(V[j].begin(),V[j].end(),Y2);
            if(itlow == V[j].end() || *itlow > Y2)
                continue;
            int cnt = 0;
            for(vector<int>::iterator it = itlow;it!=V[j].end();)
            {
                if(*it <= Y2)
                {
                    it = V[j].erase(it);
                    cnt++;
                }
                else
                    break;
            }
            if(!C)
            {
                ans += (cnt);
            }
            //if(cnt)
           // V[j].erase(itlow,itlow+cnt);
        }
    }
    cout << (n*n)-ans;
return 0;
}
