#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int temp;
    vector<int>f,m;
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	f.push_back(temp);
    }
    for(int i=0;i<n;i++)
    {
    	cin>>temp;
    	m.push_back(temp);
    }
    sort(f.begin(),f.end());
    sort(m.begin(),m.end());
    reverse(m.begin(),m.end());
    int i,j,count=0,c=0;
    i=0;j=0;
    while(i<n&&j<n)
    {
    	if(f[i]>m[j])
    	{
    		i++;
    		j++;
    		count++;
    	}
    	else if(f[i]==f[j])
    	{	
    		i++;
    		j++;
    		c++;
    		
    	}
    	else
    	{
    		i++;
    		j++;
    	}
    }
    cout<<count*100+c*50<<endl;
}
