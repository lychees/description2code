using namespace std;

int vis[3001],nextN,maxm;
vector<int>v[3001];

void dfs( int n, int d)
{
    vis[n]=1;
    if(d>maxm)
    {
        maxm=d;
        nextN=n;
    }
    vector<int>::iterator it;
    for(it=v[n].begin();it!=v[n].end();it++)
    {
        int x=(*it);
        if(!vis[x]) dfs(x,d+1);
    }
    vis[n]=0;
}

void minDist(int x)
{
    nextN=-1;
    maxm=0;
    dfs(x,0);
    //printf("%d\n",nextN);
    dfs(nextN,0);
}

int main()
{
    int t,i,n,a,b,w;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<=n;i++) v[i].clear();
        for(i=0;i<n-1;i++)
        {
            maxm=0;
            scanf("%d%d",&a,&b);
            v[a].push_back(b);
            v[b].push_back(a);
        }
        minDist(0);
        printf("%d\n",maxm+1);
    }
    return 0;
}
