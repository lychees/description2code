#include <iostream>
using namespace std;
 
/* Function to get sum of digits */
int getSum(int n)
{
    int sum;
    /*Single line that calculates sum*/
    for(sum=0; n > 0; sum+=n%10,n/=10);
    return sum;
}
 
int main()
{
  long long int temp=0,sum=0;
  int n ;
  cin >> n;
  int arr[n];
  for(int i=0;i<n;i++)
  {
  	cin >> arr[i];	
  }
  int x=0;
  int y=0;
  cin >> x >> y;
  for(int j=0;j<n;j++)
  {
  	temp=arr[j]*getSum(arr[j]);
  	temp=temp*x*y;
  	if(arr[j]<=10 && arr[j]>=1) temp=0.75*temp;
  	else if(arr[j]<=20 && arr[j]>=11) temp=temp*0.70;
  	else temp=temp*0.80;
  	sum+=temp;
  }
 
  cout<< sum << endl;
  return 0;
}
