#include <bits/stdc++.h>
using namespace std;

#define MOD 1000000007
long long H[1000005],E[1000005];
long long ans[1000005];
int main()
{
    int T,n;
    cin >> T;
    H[1]=1;
    E[1]=1;
    ans[1]=2;
    for(int i=2;i<=1000000;i++)
    {
    	H[i]=(H[i-1]%MOD+E[i-1]%MOD)%MOD;
    	E[i]=H[i-1]%MOD;
    	ans[i]=(ans[i-1]+(H[i]%MOD)+(E[i]%MOD))%MOD;
    }
    while(T--)
    {
    	cin >> n;
    	cout << ans[n] << endl;
    }
    return 0;
}
