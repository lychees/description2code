#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,c;
    long long n;
    int a[100001];
    cin>>t;
    while(t--)
    {
    	c=0;
    	cin>>n;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    		if(a[i]==1)
    		c++;
    	}
    	if(c%2==1)
    	cout<<c<<"\n";
    	else
    	cout<<(n-c)<<"\n";
    	
    	
    }
    return 0;
}
