#include <iostream>
using namespace std;

int main()
{
    int n,k,a;
    cin>>n>>k;
    int angles[n];
    
    int poss[360];
    for(int i=0;i<360;i++)
    poss[i]=0;
    poss[0]=1;
        
    for(int i=0;i<n;i++){
    cin>>angles[i];
    angles[i]=angles[i]%360;
    
    int a=angles[i];
    while(a!=0){
    	poss[a]=1;
    	a=(a+angles[i])%360;    	
    }    
    }
    
    /*
    for(int i=0;i<360;i++){
    if(poss[i]==0) continue;	
    for(int j=i+1;j<360;i++){
    	if(poss[j]==0) continue;
    	poss[(i+j)%360]=1;
    	//poss[abs(i-j)%360]=1;
    }	
    }
    */
    for(int i=0;i<k;i++){
    	cin>>a;
    	if(poss[a]==0)
    	cout<<"NO"<<endl;
    	else
    	cout<<"YES"<<endl;    	
    }   
    
    return 0;
}
