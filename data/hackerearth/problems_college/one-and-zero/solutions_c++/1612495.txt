#include <bits/stdc++.h>
using namespace std;

void reverse(string &a){	
	int l = a.length();
	for(int i=0,j=l-1;i<j;i++,j--){
		char ch = a[i];
		a[i] = a[j];
		a[j] = ch;
	}
}


string add(string a , string b){
	string c="";
	reverse(a);
	reverse(b);
	
	int i=0,j=0 , carry=0;
	while(i<a.length() && j<b.length()){
		int ans = a[i] + b[j] - '0' - '0'+ carry;
		
		c += (ans%10) +'0';
		carry = ans/ 10;
		i++;
		j++;
	}
	
	if(i<a.length()){
		int ans = a[i] - '0'+ carry;
		
		c += (ans%10)+'0';
		carry = ans/ 10;
		i++;
	}
	
	if(j < b.length() ) {
		int ans = b[j] - '0'+ carry;
		
		c += (ans%10)+'0';
		carry= ans/ 10;
		j++;
	}
	
	if(carry!=0)
		c += carry+'0';
	
	reverse(c);
	return c;
}


int main()
{
    int t;
    cin>>t;
    while(t--){
    	int n;
    	cin>>n;
    	string a = "1" , b = "1" , c;
    	    		
    	for(int i=2;i<=n;i++){    	
    		c = add(a,b);
    		a = b;
    		b = c;
    	}
    	cout<<add(a,b)<<endl;
    }
    return 0;
}
