#include <iostream>
#include <map>
#include <string>
using namespace std;

int main() {
  map<string, int> x;
  string s, n;
  cin >> n;
  for (int i = 0; i < 10; i++) {
    cin >> s, x[s] = i;
  }
  for (int i = 0; i < n.size(); i += 10) {
    cout << x[n.substr(i, 10)];
  }
  cout << endl;
  return 0;
}
