#define sort(v) sort(v.begin(),v.end())

#define len(x) (sizeof(x)/sizeof(x[0]))
using namespace std;
typedef long long int ll;

ll t,n,a[111111][3],dp[111111][3];


int main()
{
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			for (int j = 0; j < 3; j++)
			{
				cin>>a[i][j];
			}
		}
		for (int i = 0; i < 3; i++)
		{
			dp[0][i] = a[0][i];
		}
		for (int i = 1; i < n; i++)
		{
			for (int j = 0; j < 3; j++)
			{
				dp[i][j] = min(dp[i-1][(j+2)%3],dp[i-1][(j+4)%3]) + a[i][j];
			}
		}
		cout<<min(dp[n-1][0],min(dp[n-1][1],dp[n-1][2]))<<"\n";
	}
}
