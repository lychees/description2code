#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar_unlocked
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007

set<string> st;

int main(){
    string s,t;
    cin>>s>>t;
    int ans = 0;
    for(int i=0;i<s.size();i++){
        string tmp = "";
        for(int j=i;j<s.size();j++){
            tmp += s[j];
            st.insert(tmp);
        }
    }
    for(auto&it : st){
        if(it.find(t) != string::npos)
            ans++;
    }
    cout<<ans<<"\n";
    return 0;
}
