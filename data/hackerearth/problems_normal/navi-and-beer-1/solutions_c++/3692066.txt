#include <iostream>
using namespace std;

int main()
{
    int T;
    cin>>T;
    long long int answer[T];
    for(int i=0;i<T;i++)
    {
    	long long int N,K,M;
        cin>>N>>K>>M;
        long long int sum = (N*(N+1)*N*(N+1))/4;
        answer[i] = (K*sum)-M;
        if(answer[i]<0)
        	answer[i]=0;
    }
    for (int i=0;i<T;i++)
    {
        	cout<<answer[i]<<endl;
    }
    return 0;
}
