#include <bits/stdc++.h>
#define ll long long
#define MAXN 10007
#define MOD 1000000007
using namespace std;
ll dp[MAXN][12];

void solve() {
	int n,p;
	cin>>n>>p;
	cout<<dp[n+1][p+1]%MOD<<"\n";
}


void pre() {
	for(int i=0;i<MAXN;i++) {
		dp[i][0] = 0;
	}
	for(int j=0;j<12;j++) {
		dp[0][j]=0;
	}
	for(int i=1;i<MAXN;i++) {
		dp[i][1] = 1;
	}
	for(int j=1;j<12;j++) {
		dp[1][j] = 1;
	}
	for(int p=1;p<12;p++) {
		for(int i=2;i<MAXN;i++) {
			for(int j=i-1;j>=0;j--) {
				if(i-j < p) {
					dp[i][p] += dp[j][p];
					dp[i][p] %= MOD;
				}
			}
		}
	}
}


int main()
{
	pre();
    int t;
    cin>>t;
    while(t--) solve();
}
