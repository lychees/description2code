#include <bits/stdc++.h>
using namespace std;
long long int dp[213][213],n,k;
long long int solve (long long int n, long long int m) {

	if (n == 0) return 1;
    if (m == 0) return 0;
    if (n < 0 || n < k || m < k) return 0;
    if (dp[n][m] == -1)
        dp[n][m] = solve(n, m - 1) + solve(n - m, m);
    return dp[n][m];

}
int main () {
	int tc;
	scanf("%d",&tc);
	while (tc--)
    {

        memset(dp,-1,sizeof(dp));
		scanf("%Ld %Ld",&n,&k);
		long long ans = solve(n,n);
		printf("%Ld\n",ans);
	}
	return 0;
}
