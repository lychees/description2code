import sys
from sets import Set
t = int(raw_input())
while (t > 0):
    n, m = map(int, sys.stdin.readline().split(" "))
    a = map(int, sys.stdin.readline().split(" "))
    b = Set([])
    for i in range(0, n):
        b.add(a[i])

    for i in range(n, len(a)):
        if (a[i] in b):
            print "YES"
        else:
            b.add(a[i])
            print "NO"
    t -= 1
