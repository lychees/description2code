#include <iostream>
#include<cstring>
#include<algorithm>
using namespace std;

int main(){                                             
    int t,z;
    cin>>t;
    while(t--){
    	string str;cin>>str;
    	string ans="";
    	int n=str.size();
    	while(n>0){
    		char copy1[1000];
    		if(n&1){
    			z=(n+1)/2;
    		}
    		else{
    		    z=n/2;
    		    if(str[z-1]>str[z]){swap(str[z-1],str[z]);}
    		}    
    		ans+=str[z-1];
    		size_t length=str.copy(copy1,n-z,z);
            copy1[length]='\0';
            str.erase(str.begin()+z-1,str.end());
    		str.append(copy1);
    		n=str.size();
    		//cout<<str<<endl;
    	}
    	cout<<ans<<endl;
    }
    return 0;
}
