#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	string s,t;
    	cin>>t>>s;
    	int a[26] = {0},ans=0;
    	int n = s.length();
    	for(int i=0;i<n;i++)
    	{
    		a[s[i]-'a']++;
    	}
    	n = t.length();
    	for(int i=0;i<n;i++)
    	{
    		int  z = t[i]-'a';
    		//if(a[z]==0) ans++;
    		a[z]--;
    	}
    	for(int i=0;i<26;i++)
    	{
    		if(a[i]<0) {ans=-1;break;}
    		else if(a[i]>0) ans++;
    	}
    	printf("%d\n",ans);
    }
    return 0;
}
