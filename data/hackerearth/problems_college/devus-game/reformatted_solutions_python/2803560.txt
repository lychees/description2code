x = input()
for i in range(x):
    y = input()
    l = map(int, raw_input().split())
    x = l[0]
    con = 0
    c = 0
    while x < y:
        if x == -1:
            print "Win"
            con = 1
            break
        else:
            c += 1
            x = l[x]
        if c > y:
            break
    if con == 0:
        print "Lose"
