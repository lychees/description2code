#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t > 0) {
    	int i,count=0;
    	string s1,s2;
    	cin>>s1>>s2;
    	int len;
    	if(s1.size() < s2.size()) {
    		len = s1.size();
    	} else {
    		len = s2.size();
    	}
    	for(i = 0; i < len; i++) {
    		if(s1[i] == s2[i]) {
    			count++;
    		}
    	}
    	cout<<count<<endl;
    	t -= 1;
    }
    return 0;
}
