#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
struct node{
    long long p,c;
};
int cmp(node a,node b)
{
    return a.c<b.c;
}
int main()
{
    int n;
    cin>>n;
    vector<node> nd(n);
    for(int i=0;i<n;i++)
    cin>>nd[i].p>>nd[i].c;
    sort(nd.begin(),nd.end(),cmp);
    int t;
    cin>>t;
    long long ar[t];
    for(int i=0;i<t;i++)
    cin>>ar[i];
    for(int i=t-1;i>0;i--)
    ar[i]-=ar[i-1];
    long long ans=0LL;
    int i=0,j=0;
    long long f=1LL;
    while(1)
    {
        if(i>=n)
        break;

        long long mn=nd[i].p;
        if(j<t)
        if(mn>ar[j])
        mn=ar[j];
        ans+=(f*mn*nd[i].c);
        ar[j]-=mn;nd[i].p-=mn;
        if(nd[i].p<=0)
        i++;
        if(j<t)
        if(ar[j]<=0)
        j++,f+=1LL;
    }
    cout<<ans<<"\n";
}
