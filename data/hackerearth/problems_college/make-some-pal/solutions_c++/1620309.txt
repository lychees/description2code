#include<bits/stdc++.h>
using namespace std;

#define mod 1000000007
typedef set<string> ss;
typedef vector<int> vs;
typedef map<int,char> msi;
typedef pair<int,int> pa;
typedef long long int ll;

ll u,v,i,l,ans;
char s[52];
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		l=strlen(s);
		ans=mod;
		for(i=0;i<l;i++)
		{
			u=i; v=i; 
			while(u>=0 && v<l && s[u]==s[v])
			u--,v++;
			if(v==l)
			ans=min(ans,u+1);
			u=i; v=i+1;
			while(u>=0 && v<l && s[u]==s[v])
			u--,v++;
			if(v==l)
			ans=min(ans,u+1);
		}
		cout<<l+ans<<"\n";
	}
	return 0;
}
