#include <iostream>
#include<stdio.h>
#include <iomanip>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
    	int k,n,l;
    	long double p =1.0;
    	cin >> k >> n;
    	l = n;
    	if(k<=n)
    	{
    	    int i;
    		for(i=0; i<k; ++i)
    		{
    			p *= (long double)l/n;
    			l--;
    		//	cout << p << endl;
    		}
    		std::cout << std::fixed;
    		cout << std::setprecision(9) << 1-p << endl ;
    		//printf("%.9lf\n",1-p);
    	}
    	else
    	std::cout << std::fixed <<  std::setprecision(9) << p << endl ;
    }
    return 0;
}
