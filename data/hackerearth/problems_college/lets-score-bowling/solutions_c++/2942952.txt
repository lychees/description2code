#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--){
		int a,b,c;
		cin>>a>>b>>c;
		int s=0;
		if(a>0 && a<7){s=s+1;}
		else if(a>=7 && a<9){s=s+2;}
		else if(a==9){s=s+10;}
		if(b>0 && b<7 && a!=9){s=s+1;}
		else if(b>=7 && b<9  && a!=9){s=s+2;}
		else if(b==9  && a!=9){s=s+10;s=s-3;}
		if(c>0 && c<7 && a!=9 && b!=9){s=s+1;}
		else if(c>=7 && c<9  && a!=9 && b!=9){s=s+2;}
		else if(c==9  && a!=9 && b!=9){s=s+10;s=s-6;}
		if(a!=9 && b!=9 && c!=9){s=0;}
		cout << s << endl;
	}
    
    return 0;
}
