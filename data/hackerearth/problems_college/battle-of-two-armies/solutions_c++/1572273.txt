#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;


bool isPr(int n)
{
if(n == 1) return false;
int i,flag = 0;
int x = sqrt(n);
for(i=2;i<=x;i++)
{
if(n%i == 0) {flag = 1; break;}
}

if(flag == 0) return true;
else return false;

}


int main()
{


int t;
cin>>t;


while(t--)
{
int n,i,j,x;
cin>>n;

int c[10],p[10];

for(i=0;i<n;i++) cin>>c[i];
for(i=0;i<n;i++) cin>>p[i];

int comp = 0, prim = 0;


for(i=0;i<n;i++)
{
if(isPr(c[i]) && isPr(p[i])) prim+=p[i];
else if(!isPr(c[i]) && !isPr(p[i])) comp+=c[i];


else
{
if(c[i] == p[i]) continue;
if(c[i] > p[i]) comp+= (c[i] - p[i]);
else prim+= (p[i] - c[i]);
}

}

if(comp == prim) cout<<"Tie\n";
if(comp > prim) cout<<"The Composites\n";
if(comp < prim) cout<<"The Primes\n";



}

return 0;
}
