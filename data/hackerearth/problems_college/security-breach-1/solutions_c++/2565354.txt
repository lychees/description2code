#include <bits/stdc++.h>
using namespace std;
#define ll unsigned long long
int main()
{
    int t;cin>>t;
    map<int,string> m;
    m[2]="cde";m[6]="b3";m[7]="acf";m[9]="c6a";
    while(t--)
    {
    	ll n;
    	cin>>n;
    	int a[20];
    	int j=0;
    	while(n>0)
    	{
    		int temp=n%10;
    		if(temp==2 || temp==6 || temp==7 ||temp==9)
    		{a[j]=temp;j++;}
    		n/=10;
    	}
    
    
    string s="";
    for(int i=j-1;i>=0;i--)
    {
    	s+=m[a[i]];
    }
    cout<<s<<endl;
    }
    return 0;
}
