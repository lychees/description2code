#include<iostream>
#include<map>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	int n;
	cin>>n;
	map<string,int> mp,mp2;
	string str;
	vector<string> vec;
	for(int i=0;i<n;i++){
		cin>>str;
		vec.push_back(str);
		mp[str]++;
		mp2[str]=0;
	}
	sort(vec.begin(),vec.end());
	for(int i=vec.size()-1;i>=0;i--){
		if(mp2[vec[i]]==0){
			cout<<vec[i]<<" "<<mp[vec[i]]<<"\n";
			mp2[vec[i]]=1;	
		}
	}
	return 0;
}
