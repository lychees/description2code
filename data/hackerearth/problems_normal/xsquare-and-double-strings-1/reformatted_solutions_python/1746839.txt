def checkRemove2DoubleString(string):
    cmap = {}

    result = False
    for l in string:
        if l not in cmap:
            cmap[l] = 1
        else:
            return True

    return result


numofTest = int(raw_input().strip())

    inputString = raw_input()

        print "Yes"
