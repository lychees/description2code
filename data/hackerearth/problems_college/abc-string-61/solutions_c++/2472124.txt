#include <iostream>
using namespace std;

long long int INF = 1000000007;

long long int func(long long int a,long long int b)
{
    long long int ans = 1;
    while (b > 0)
       {
           if (b & 1)
           {
               a=a%INF;
               ans = (ans * a)%INF;
               ans=ans%INF;
           }
        b=b>>1;
        a=a%INF;
        a = (a*a)%INF;
        a=a%INF;
       }
    return ans;
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
    	long long int n;
    	scanf("%lld", &n);
    	long long int ans = func(3, n-1);
    	ans = (ans - func(2, n)%INF + INF)%INF;
    	ans = (ans + 1)%INF;
    	ans = (ans * 3)%INF;
    	printf("%lld\n", ans);
    }
    return 0;
}
