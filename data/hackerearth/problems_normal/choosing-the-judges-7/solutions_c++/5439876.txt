#include <bits/stdc++.h>
using namespace std;
int a[10000];
long long int dp[10005];
int main() {
	int t;
	cin>>t;
	for(int p=1;p<=t;p++){
		int n;
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		dp[0]=0;dp[1]=a[0];
		for(int i=2;i<=n;i++){
			dp[i]=max(dp[i-1],a[i-1]+dp[i-2]);
		}
		cout<<"Case "<<p<<": "<<dp[n]<<endl;
	}
	return 0;
}
