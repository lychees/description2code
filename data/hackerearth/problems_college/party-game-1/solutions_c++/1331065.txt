#include <cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int t,n,a[10005],i,x,y;
    long long ch,jh;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld%lld%d",&ch,&jh,&n);
    	for(i=0;i<n;i++){
    		scanf("%d",&a[i]);
    	}
    	int c1=0,c2=0;
    	sort(&a[0],&a[n]);
    	for(i=0;i<n;i++){
    	if(ch-a[i]>=0){
    	ch=ch-a[i];
    	c1++;
    	}
    	if(jh-a[i]>=0){
    	jh=jh-a[i];
    	c2++;
    	}
    	}
    	if(c1==c2) printf("Tie\n");
    	else if(c1>c2) printf("Chef Won\n");
    	else printf("Jhool Won\n");
    }
    return 0;
}
