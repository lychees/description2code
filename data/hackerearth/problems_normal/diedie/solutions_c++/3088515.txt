#include <iostream>
using namespace std;
long long int mod=1000000007;
long long int pwr(long long int a,long long int b)
{
	long long int res=1;
	while(b>0)
	{
		if((b&1)==1)
			res=((res%mod)*(a%mod))%mod;
		a=((a%mod)*(a%mod))%mod;
		b/=2;
	}
	return res;
}
int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	cout<<pwr(pwr(2,n-1),mod-2)<<"\n";
    }
    return 0;
}
