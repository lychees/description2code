    #include<iostream>
    #include<cstdio>
    #include<cmath>
    using namespace std;
    #define max 10000001
    typedef long long int ll;
    ll dp[max];
    ll euler[max];
    void build()
    {
     
    	ll i,j;
    	for(i=0;i<max;++i)
    	euler[i]=i;
    	
     
    	for(i=2;i<max;++i)
    	{
    		if(euler[i]==i)
    		{
    			for(j=i;j<max;j+=i)
    			euler[j]-=euler[j]/i;
    		}
    	}
    }

    int main()
    {
        int t;
        ll n,i;
        build();
        dp[0]=0;
        dp[1]=1;
        for(i=2;i<max;++i)
        {
            dp[i]=dp[i-1]+euler[i];
        }
        cin>>t;
        
        
        while(t--)
        {
            cin>>n;
            
            cout<<dp[n]<<"\n";
           
        }
    }
