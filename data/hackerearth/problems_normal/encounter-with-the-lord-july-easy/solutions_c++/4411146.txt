struct Djikstra {
    const int n;
    const vector<vpii> &edg;
    vi distances;
    const int root;
    Djikstra ( const vector<vpii> &_edg,int _root ) :n ( _edg.size() ),edg ( _edg ) ,distances ( n,INF ),root ( _root ) {
        build();
    }
    void build() {
        priority_queue<pii> pq;
        pq.push ( mp ( 0,root ) );
        distances[root]=0;
        int found=0;
        while ( !pq.empty() && found<n ) {
            pll nxt=pq.top();
            int no=nxt.second;
            pq.pop();
            if ( -nxt.first>distances[no] ) continue;
            found++;
            for ( pii p: edg[no] ) {
                if ( distances[p.first]>distances[no]+p.second ) {
                    pq.push ( mp ( nxt.first-p.second,p.first ) );
                    distances[p.first]=distances[no]+p.second;
                }
            }
        }
    }
};
struct Hungarian {
    static constexpr int MAX_VALUE=INF;

    int solveMax ( const vector<vi> &gain ) {
        int n=gain.size();
        int m=gain[0].size();
        assert ( n<=m );
        vector<vi> cost2 ( n,vi ( m ) );
        int maxc=-INF;
        rep ( i,n ) rep ( j,m ) {
            maxc=max ( maxc,gain[i][j] );
        }
        rep ( i,n ) rep ( j,m ) {
            cost2[i][j]=maxc-gain[i][j];
        }
        int ans=solveMin ( cost2 );
        return n*maxc-ans;

    }

    vi matchingR;
    int solveMin ( const vector<vi> &cost ) {
        int n=cost.size();
        int m=cost[0].size();
        matchingR =vi ( m,-1 );
        vi pL ( n ),pR ( m ),min ( m ),back ( m );
        int ans = 0;
        rep ( i,n ) {
            fill ( all ( min ), MAX_VALUE );
            for ( int ci = i, cj = -1;; ci = matchingR[cj] ) {
                int delta = MAX_VALUE;
                int nj = -1;
                for ( int j = 0; j < m; j++ ) {
                    if ( min[j] < 0 ) {
                        continue;
                    }
                    int v = cost[ci][j] - pL[ci] - pR[j];
                    if ( v < min[j] ) {
                        min[j] = v;
                        back[j] = cj;
                    }
                    if ( min[j] < delta ) {
                        delta = min[j];
                        nj = j;
                    }
                }
                cj = nj;
                pL[i] += delta;
                rep ( j,m ) {
                    if ( min[j] < 0 ) {
                        pR[j] -= delta;
                        pL[matchingR[j]] += delta;
                    } else
                        min[j] -= delta;

                }
                ans += delta;
                if ( matchingR[cj] < 0 ) {
                    while ( back[cj] >= 0 ) {
                        matchingR[cj] = matchingR[back[cj]];
                        cj = back[cj];
                    }
                    matchingR[cj] = i;
                    break;
                }
                min[cj] = -1;
            }
        }
        return ans;
    }
};

        vector<vpii> adj ( n );
        rep ( i,m ) {

            adj[x].push_back ( mp ( y,c ) );

        }

                costs[i][j]=min ( dj.distances[n-1-j],10000 );

            }
        }
        Hungarian h;

    }
