#include<bits/stdc++.h>
using namespace std;
map<int,long> arr;
long rec(long n)
{
    if(n==0||n==1)
        return n;
    if(arr.count(n))
        return arr[n];
    return arr[n]=max(n,rec(n/2)+rec(n/3)+rec(n/4));
}
int main()
{
    long n;
    while(scanf("%ld",&n)!=EOF)
    {
        long res=rec(n);
        cout<<res<<"\n";
    }
}
