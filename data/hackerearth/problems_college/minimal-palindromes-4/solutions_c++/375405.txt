#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int gcd(int a,int b){
	if(!b) return a;
	return gcd(b,a%b);
}

int main(){
	int t,l,a,j,b,i,c;
	string s;
	cin>>t;
	while(t--){
		cin>>s;
		l = s.size();
		a = s[0] - '0';
		for(i=1;i<l;++i){
			b = gcd(a,s[i]-'0');
			a = b;
		}

		for(i=0;i<l;++i){
			c = s[i] - '0';
			c/=a;
			s[i] = c + '0';
		}
		cout<<s<<endl;
	}
	return 0;
}
