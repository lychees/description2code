#include<cstdio>
#include<iostream>
using namespace std;
main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		long long int n;
		scanf("%lld", &n);
		n = n*12;
		int q = n/26;
		int r = n%26;
		long long ans = 0;
		for(int i = 1; i<=26; i++)
		{
			long long int temp;
			scanf("%lld", &temp);
			long long int count = q+(i<=r?1:0);
			ans+=(count*(2*temp+count-1))/2;
		}
		printf("%lld\n", ans);
	}
}
