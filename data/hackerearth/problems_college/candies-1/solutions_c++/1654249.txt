#include <iostream>
#define ll long long

using namespace std;

int main()
{
	cin.tie(0);
    int T;
    cin >> T;
    while(T--) {
    	ll N,sum=0;
    	cin >> N;
    	ll temp;
    	for(int i=0;i<N;i++) {
    		cin >> temp;
    		sum+=temp;
    	}
    	if(sum%N==0) {
    		cout << "YES" << endl;
    	}
    	else {
    		cout << "NO" << endl;
    	}
    	cout << sum/N  << " " << sum%N << endl;
    }
    return 0;
}
