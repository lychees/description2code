
#include<bits/stdc++.h>
 
using namespace std ;
#define ll long long int
ll dp[101][101] ;
ll gcd(ll a, ll b)
{
    if(b == 0) return a ;
    else return gcd(b,a%b) ;
}
int main()
{
    ll i,j,k,n,m,p,t ;
    cin >> n >> k ;
    ll a[101] ;
    p = 2 ;
    memset(dp,0,sizeof(dp)) ;
 
    for(i=1;i<62;i++) {
        a[i] = p ;
        p *= 2 ;
    }
    p = 2 ;
    for(i=1;i<=n;i++) {
        dp[i][1] = p - 1 ;
        p *= 2 ;
    }
    for(i=2;i<=n;i++) {
        for(j=2;j<=i;j++) {
            if(i == j) dp[i][j] = 1 ;
            else {
                p = 1 ;
                while(p<=k) {
                   if(i>=p) dp[i][j] += dp[i-p][j] ;
                    p++ ;
                }
                if(i>=k)dp[i][j] += a[i-k] ;
            }
           // cout << dp[i][j] << " " ;
        }
      //  cout << endl ;
    }
    p = gcd(dp[n][k],a[n]) ;
 
    cout << dp[n][k]/p << "/" << a[n]/p << endl ;
 
    return 0;
}
