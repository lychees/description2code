for _ in xrange(input()):
    strB = raw_input()
    strA = raw_input()

    l = len(strA)
    minCount = 10000
    for j in range(0, len(strB)-l+1):
        count = 0

        for i in range(j, l+j):
            if strA[i-j] != strB[i]:
                # print strA[i-j]+" != "+strB[i];
                count += 1
        # print "Check complete : "+strB[j:l+j];
        if count < minCount:
            minCount = count
    print minCount
