#include<iostream>
using namespace std;
int minInsrt(string s, int l, int h)
{
    if (l > h) return 999999999;
    if (l == h) return 0;
    if (l == h - 1) return (s[l] == s[h])? 0 : 1;
   return (s[l] == s[h])? minInsrt(s, l + 1, h - 1):(min(minInsrt(s, l, h - 1),minInsrt(s, l + 1, h)) + 1);
}
int min(int a, int b)
{  return a < b ? a : b; }

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        string s;
        cin>>s;
        cout<<minInsrt(s,0,n-1)<<endl;
    }
    return 0;
}
