#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    int ans[1000001];
    ans[0]=0;
    for(int i=1;i<1000001;i++)
     ans[i] = -1;
    int p[4] = {2,3,5,7};
    for(int i=2;i<=1000001;i++)
     for(int j=0;j<4;j++)
      {
      	if(i>=p[j])
      	 {
      	 	if(ans[i-p[j]]!=-1)
      	 	 if(ans[i]==-1||ans[i]>1+ans[i-p[j]])
      	 	  ans[i] = 1+ans[i-p[j]];
      	 }
      }
    while(t--)
    {
    	int n;
    	cin>>n;
    	cout<<ans[n]<<endl;
    }
    return 0;
}
