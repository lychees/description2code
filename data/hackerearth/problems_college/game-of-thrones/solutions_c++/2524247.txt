#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    long int tc,a[100000],i,b,c;
    cin>>tc;
    while(tc--)
    {c=0;
    	cin>>b;
    	for(i=0;i<b;i++)
    	{
    		cin>>a[i];
    	}
    	sort(a,a+b);
    for(i=0;i<b;i++)
    {
    	if(a[i]==a[i+1])
    	c++;
    }
    cout<<c<<"\n";
    }
    return 0;
}
