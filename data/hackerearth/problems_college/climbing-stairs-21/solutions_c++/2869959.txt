typedef long long LL;

using namespace std;

LL dp[1005];

LL func(LL ct,LL n)
{
	if(ct==n)
	return 1;
	
	else if(ct>n)
	return 0;
	
	if(dp[ct])
	return dp[ct];
	
	else 
	dp[ct]= func(ct+1,n)+func(ct+2,n);
	
	return dp[ct];
}

main()
{
	 int t;
	 cin>>t;
	 
	 while(t--)
	 {
		LL n;
		sll(n);
		
		printf("%lld\n",func(0,n));
		memset(dp,0,sizeof(dp));
	 }
}
