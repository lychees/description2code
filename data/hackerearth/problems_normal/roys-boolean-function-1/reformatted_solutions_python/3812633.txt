def hcf(a, b):
    if a % b == 0:
        return b
    else:
        return hcf(b, a % b)


def num_primes(num):
    count = 0
    flag = "TRUE"
    for item in range(num):
        if hcf(num, item+1) == 1:
            count += 1

    if count == 1:
        return "FALSE"
    for item in range(2, count):
        if count % item == 0:
            flag = "FALSE"
            break
    return flag

    print num_primes(int(raw_input()))
