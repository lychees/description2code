#include <iostream>
using namespace std;
long long ar[100009];
int main()
{
    int n,m;
    long long low,high,mid,ans,tot=0,cur,count,maxval=0;
    cin>>n>>m;
    for(int i=1;i<=n;i++){
    	cin>>ar[i];
    	tot+=ar[i];
    	maxval=max(maxval,ar[i]);
    }
    low=maxval;
    high=tot;
    while(low<=high){
    	mid=(low+high)/2;
    	count=0;
    	cur=0;
    	for(int i=1;i<=n;i++){
    		if(cur+ar[i]>mid){
    			count++;
    			cur=ar[i];
    		}
    		else{
    			cur+=ar[i];
    		}
    	}
    	if(count>=m){
    		low=mid+1;
    	}
    	else{
    		high=mid-1;
    		ans=mid;
    	}
    }
    cout<<ans<<endl;
    return 0;
}
