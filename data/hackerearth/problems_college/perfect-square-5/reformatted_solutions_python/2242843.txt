import math


def findclosestsquare(num):
    num_sqrt = math.sqrt(num)
    num_prev = int(num_sqrt)
    num_next = int(num_sqrt + 1)
    sqr_prev = num_prev * num_prev
    sqr_next = num_next * num_next
    if (num - sqr_prev < sqr_next - num):
        return sqr_prev
    else:
        return sqr_next
