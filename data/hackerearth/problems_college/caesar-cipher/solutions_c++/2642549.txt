#include <iostream>
using namespace std;

int main()
{
    int t,i,k;
    string s;
    cin>>t;
    while(t--){
    	cin>>k;
    	cin>>s;
    	
    	for(i=0;i<s.size();i++){
    		if(isupper(s[i])){
    			s[i] = (char)(((s[i]-'A'+k)%26)+'A');
    		}else if(islower(s[i])){
    			s[i] = (char)(((s[i]-'a'+k)%26)+'a');
    		}
    	}
    	
    	cout<<s<<endl;
    }
    return 0;
}
