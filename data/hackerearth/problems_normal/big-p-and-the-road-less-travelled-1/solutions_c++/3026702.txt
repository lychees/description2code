#include <iostream>
#include<bits/stdc++.h>
using namespace std;

vector<int>v[100001];
int dp[100001];

void dfs(int x)
{
	if(dp[x]!=-1)
	return ;
	dp[x]=0;
	for(int i=0;i<v[x].size();i++)
	{
		dfs(v[x][i]);
		dp[x]+=dp[v[x][i]];
	}
}	

int main()
{
	int i,j,n,x,y;
	cin>>n;
	while(1)
	{
		cin>>x>>y;
		if(!x && !y)
		break;
		v[x].push_back(y);
	}
	for(i=1;i<=n;i++)
	dp[i]=-1;
	dp[n]=1;
	dfs(1);
	cout<<dp[1];
	return 0;
}
