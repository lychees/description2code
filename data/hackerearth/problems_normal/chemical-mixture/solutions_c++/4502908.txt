#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	long long i,j,m,n1,n2;
    	cin>>m>>n1>>n2;
    	long long a1[n1],a2[n2];
    	vector<long long> sum;
    	for(i=0;i<n1;i++)
    	   cin>>a1[i];
    	for(i=0;i<n2;i++)
    	   cin>>a2[i];
    	for(i=0;i<(1<<n2);i++)
    	{
    		long long tmp=0;
    		for(j=0;j<n2;j++)
    		{
    			if(i&(1<<j))
    			  tmp+=a2[j];
    		}
    		sum.push_back(tmp);
    	}
    	sort(sum.begin(),sum.end());
    	sort(a1,a1+n1);
    	long flag=0;
    	vector<long long> ans;
    	for(i=0;i<sum.size();i++)
    	{
    		if(m>=sum[i])
    		{
    			long long tmp=0,val=m-sum[i];
    			for(j=n1-1;j>=0;j--)
    			{
    				if(a1[j]<=val)
    				{
    					val-=a1[j];
    					ans.push_back(a1[j]);
    				}
    			}
    			if(val==0)
    			{
    				flag=1;
    				break;
    			}
    		}
    		ans.clear();
    	}
    	if(flag){
    		cout<<"YES\n";
    		sort(ans.begin(),ans.end());
    		for(i=0;i<ans.size();i++)
    		{
    			cout<<ans[i]<<" ";
    		}
    		cout<<"\n";
    	}
    	else
    	{
    		cout<<"NO\n";
    	}
    	
    	
    }
    return 0;
}
