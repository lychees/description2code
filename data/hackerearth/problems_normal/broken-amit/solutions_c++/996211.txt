#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
using namespace std;
typedef long long ll;
const int inf = 1e9+9;
int a[20];
int b[20];
bool visited[20];
int main() {
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++) scanf("%d",&b[i]);
    ll sum=0;
    for(int i=0;i<n;i++) sum+=a[i];
    for(int i=0;i<n-1;i++) {
        int idx;
        int maxVal=-1;
        for(int j=0;j<n;j++) {
            if(visited[j]==false) {
                if(b[j]*j>maxVal) {
                    maxVal=b[j]*j;
                    idx=j;
                }
            }
        }
        visited[idx]=true;
        for(int j=0;j<n;j++) if(visited[j]==false) a[j]-=b[j]*j;
        for(int j=0;j<n;j++) sum+=a[j];
    }
    printf("%lld\012",sum);
    return 0;
}
