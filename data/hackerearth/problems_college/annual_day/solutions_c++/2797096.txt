#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,i;
		cin>>n>>k;
		int num ,a[n];
		for(i=0;i<n;i++)
		{
			int c=0;
			cin>>num;
			while(num!=0)
			{
				if(num%2==1)
				{
					c++;
				}
				num=num/2;
			}
			a[i]=c;
		}
		sort(a,a+n,greater<int>());
		
		int sum =0;
		for(i=0;i<k;i++)
		{
			sum = sum + a[i];
		}
		cout<<sum<<"\n";
	}
}
