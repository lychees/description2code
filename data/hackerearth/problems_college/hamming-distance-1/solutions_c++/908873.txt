#include <iostream>
using namespace std;
int hammingDistance (uint64_t x, uint64_t y) {
        uint64_t res = x ^ y;
        return __builtin_popcountll (res);
}
int main()
{
	long int t;
	cin>>t;
	while(t--)
	{
		long a=0,b=0,c=0;
		cin>>a>>b;
		c=hammingDistance(a,b);
		cout<<c<<endl;
		
	}
return 0;
}
