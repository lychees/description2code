#include <bits/stdc++.h>
using namespace std;

int lcs( int *X, int *Y, int m, int n )
{
   int L[m+1][n+1];
   int i, j;
   for(i=0;i<=m;i++)
   	for(j=0;j<=n;j++)
   		L[i][j]=0;
  
   for (i=0; i<=m; i++)
   {
     for (j=0; j<=n; j++)
     {
       if (i == 0 || j == 0)
         L[i][j] = 0;
  
       else if (X[i-1] == Y[j-1])
         L[i][j] = L[i-1][j-1] + 1;
  
       else
         L[i][j] = max(L[i-1][j], L[i][j-1]);
     }
   }
   return L[m][n];
}

int main()
{
    int t,n,p,i,j;
    cin>>t;
    while(t--){
    	vector <pair <int,int> > va,vb;
    	cin>>n;
    	for(i=0;i<n;i++){
    		cin>>p;
    		va.push_back(make_pair(p,i));
    	}
    	for(i=0;i<n;i++){
    		cin>>p;
    		vb.push_back(make_pair(p,i));
    	}
    	sort(va.begin(),va.end());
    	sort(vb.begin(),vb.end());
    	
    	int X[2000],Y[2000];
    	
    	for(i=0;i<n;i++)
    		X[i]=va[i].second;
    	for(i=0;i<n;i++)
    		Y[i]=vb[i].second;
    	
    	cout<<lcs(X,Y,n,n)<<endl;
    }
    return 0;
}
