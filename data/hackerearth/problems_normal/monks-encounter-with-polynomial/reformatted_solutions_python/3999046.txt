import math


def fx(A, B, C, x):
    return A*x*x + B*x + C


T = int(raw_input())

for e in xrange(T):
    A, B, C, K = (float(i) for i in raw_input().split())
    ub = int(math.floor(math.sqrt(K/A)))

    # s=-1
    if K <= C:
        print 0
    else:
        while ub > 0:
            if fx(int(A), int(B), int(C), ub) >= int(K):
                ub -= 1
            else:
                break
        print ub+1
