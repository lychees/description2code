#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	char a[105];
	gets(a);
	
	int i,c,l = strlen(a);
	
	long int sum=0;
	for(i=0;i<l;i++){
		sum+=a[i];
	}
	
	sum=(int)(sum/l);
	int max[105],m;
	
	if(sum%2!=0){
			m=0;
		for(i=0;i<l;i++){
			max[i]=0;
			for(int j=i;j<l;j++){
				if(a[i]==a[j]) max[i]++;
			}
			if(max[i]>m){
				 m=max[i];
				 c=i;
			}
		}
		cout<<a[c]<<endl;
	}
	else{
		for(i=l-1;i>=0;i--){
			cout<<a[i];
		}
		cout<<endl;
	}
	return 0;
}
