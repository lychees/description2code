#include <iostream>
#include<string>
using namespace std;

int main()
{
    int n=0 , m , sum1=0 , sum2=0 , counter=0 ,ch;
    string st;
    cin>>m;
    cin.ignore(1);
    for(int i=0;i<m;i++)
    {
    	getline(cin,st);
    	n=st.size();
    	for(int  j=0;j<n;j++)
    	{
    		if(j%2==0)
    		sum1+=st.at(j);
    		else
    		sum2+=st.at(j);
		}
		if(sum1==sum2)
    		{
    			counter++;
    			
			}
    	sum1=0;
    	sum2=0;
	}
	
	cout<<counter<<endl;
    return 0;
}
