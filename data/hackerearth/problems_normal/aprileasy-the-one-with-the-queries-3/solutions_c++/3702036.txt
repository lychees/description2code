#include <bits/stdc++.h>
using namespace std;

string S;
vector<int> next_pos[26];

int main()
{
    cin >> S;
    
    int N = S.size();
    
    for(int i=0; i<N; i++)
    	next_pos[S[i]-'a'].push_back(i);
    
    int Q;
    scanf("%d", &Q);
    
    while(Q--)
    {
    	cin >> S;
    	
    	int prev = -1, flag = 0;
    	
    	for(int i=0; i<S.size(); i++)
    	{
    		int indx = upper_bound(next_pos[S[i]-'a'].begin(), next_pos[S[i]-'a'].end(), prev) - next_pos[S[i]-'a'].begin();
    		
    		if(indx==next_pos[S[i]-'a'].size())
    		{
    			flag = 1;
    			break;
    		}
    		
    		prev = next_pos[S[i]-'a'][indx];
    	}
    	
    	if(flag)
    		cout << "No\n";
    	else
    		cout << "Yes\n";
    }
    
    return 0;
}
