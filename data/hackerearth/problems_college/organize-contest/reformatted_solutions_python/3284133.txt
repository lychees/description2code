t = input()
c = []
d = {}
g = 0
p = []
for i in range(t):
    v = raw_input()
    p = v.split()
    f = p[0]
    n = int(p[1])
    if d.get(f, -1) != -1:
        d[f] += n
    else:
        c.append(f)
        d[f] = n
    if g == 0:
        k = d[f]
        m = f
        g = 1
    elif g == 1 and d[f] > k:
        k = d[f]
        m = f
