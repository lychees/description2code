#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
long long cs,n,i,s;
long long A[50001],MX[50001];
int main()
{
  scanf("%lld",&cs);
  while(cs--)
  {
    scanf("%lld",&n);
    for(i=0;i<n;i++)
      scanf("%lld",&A[i]);
    for(i=n-2,MX[n-1]=A[n-1];i>=0;i--)
      MX[i]=max(MX[i+1],A[i]);
    for(i=s=0;i<n;i++)
      s+=MX[i]-A[i];
    printf("%lld\n",s);
  }
  return 0;
}
