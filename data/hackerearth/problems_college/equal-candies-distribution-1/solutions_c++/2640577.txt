#include<bits/stdc++.h>
using namespace std;
long int f(long int a[],long int n)
{
	long int ans=0;
	for(long int i=0;i<n;i++)
	{
		ans+=a[i]/5 + (a[i]%5)/2 + (a[i]%5%2);
	}
	return ans;
}

{

		for(i=0;i<n;i++)
			cin>>a[i];
