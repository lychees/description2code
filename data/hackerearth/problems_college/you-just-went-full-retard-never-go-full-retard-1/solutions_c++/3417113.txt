#include<bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
	ll tc;
	cin>>tc;
	while(tc--)
	{

		ll n;
		cin>>n;
		if(n==1 || n==0)
		{
			puts("NO");
		}
		else
		{
			bool flag=true;
			for(ll i=2;i*i<=n;++i)
			{
				if(n%i==0){
					flag=false;
					break;
				}

			}
			(flag==true)?puts("YES"):puts("NO");
		}

	}
	return 0;}
