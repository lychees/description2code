#include <iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t,l,r,x;
    cin>>t;
    while(t--)
    {
    	cin>>l>>r;
    	vector<int> v;
    	for(int i=l;i<=r;i++)
    	{
    		for(int j=l;j<=r;j++)
    		{
    			x=(i^j);
    			if(x==0)
    			continue;
    			else if(x!=0)
    			{
    			
    				v.push_back(x);
    			}
    		}
       	
    		
    		
    	}
    	
    	sort(v.begin(),v.end());
    	
    	cout<<v[v.size()-1]<<endl;
    }
    
    return 0;
}
