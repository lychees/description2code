struct bird
{
  long long ai;
  int ci;
};
bird arr[101];

int cmp(const void *p1,const void *p2)
{
  bird *a=(bird *)p1;
  bird *b=(bird *)p2;
  return ((a->ci)-(b->ci));
}

int main()
{
int n;
long long  p[101];
scanf("%d",&n);
for(int i=0;i<n;i++)
{
scanf("%lld%d",&arr[i].ai,&arr[i].ci);
}
qsort(arr,n,sizeof(bird),cmp);

for(int i=0;i<t;i++)
{
  scanf("%lld",&p[i]);
}
int k=0;
for(int i=t-1;i>0;i--)
{
p[i]=p[i]-p[i-1];
}
long long sum=0;
int i;
for(i=0;i<n && k<t;)
{
  if(p[k]>arr[i].ai)
  {
    p[k]=p[k]-arr[i].ai;
   sum+=(arr[i].ai*arr[i].ci)*(k+1); 
   i++;
  }
  else if(p[k]<arr[i].ai)
  {
    arr[i].ai=arr[i].ai-p[k];
   sum+=p[k]*arr[i].ci*(k+1);
  k++;
  }
 else { sum+=p[k]*arr[i].ci*(k+1); k++; i++;}

}
for(;i<n;i++)
sum+=arr[i].ai*arr[i].ci*(k+1);

}
