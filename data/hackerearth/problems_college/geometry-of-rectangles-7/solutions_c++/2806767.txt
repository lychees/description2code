#include <iostream>
#include <algorithm>
using namespace std;
struct rect
{
	int x1,x2,y1,y2;
};	
int main()
{
	rect r1,r2;
	int t;
    cin >> t;
    while(t--)
    { 
      cin >> r1.x1 >> r1.y1 >> r1.x2 >> r1.y2;
      cin >> r2.x1 >> r2.y1 >> r2.x2 >> r2.y2;
      
      if((((r1.x2 - r1.x1)+(r2.x2 - r2.x1)) < (max(r2.x2,r1.x2) - min(r2.x1,r1.x1))) || (((r1.y2 - r1.y1)+(r2.y2 - r2.y1)) < (max(r2.y2,r1.y2) - min(r2.y1,r1.y1))))
        cout << "Apart\n";
      else
      if((((r1.x2 - r1.x1)+(r2.x2 - r2.x1)) == (max(r2.x2,r1.x2) - min(r2.x1,r1.x1))) && (((r1.y2 - r1.y1)+(r2.y2 - r2.y1)) == (max(r2.y2,r1.y2) - min(r2.y1,r1.y1))))
      cout << "Point\n";
      else
        if((((r1.x2 - r1.x1)+(r2.x2 - r2.x1)) == (max(r2.x2,r1.x2) - min(r2.x1,r1.x1))) || (((r1.y2 - r1.y1)+(r2.y2 - r2.y1)) == (max(r2.y2,r1.y2) - min(r2.y1,r1.y1))))
      cout << "Line\n";
      else
        cout << "Rectangle\n";
      
      
    }
}
