def func(n):
    l = [2, 3, 4, 5, 6, 7, 9]
    for i in xrange(len(l)):
        if n.count(str(l[i])) > 0 or n != n[::-1]:
            return "NO"
    return "YES"


t = input()
while (t > 0):
    n = raw_input().strip()
    print func(n)
    t -= 1
