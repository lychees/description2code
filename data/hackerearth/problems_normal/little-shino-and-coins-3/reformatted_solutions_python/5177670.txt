
n = int(raw_input())
s = raw_input()
c = 0
res = 0
for i in xrange(len(s)):
    st = s[i]
    c = 0
    for j in xrange(i, len(s)):
        if s[j] not in st:
            c = c+1
            if c > n:
                break
            else:
                st = st+s[j]
        if len(st) == n:
            res = res+1
print res
