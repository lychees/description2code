#include <bits/stdc++.h>

using namespace std;

int bits[1 << 16];
long long ways[1 << 16];
long long f[1 << 16];

int main() {
  // freopen("in", "r", stdin); freopen("out", "w", stdout);
  int tt;
  scanf("%d", &tt);
  while (tt--) {
    int n;
    scanf("%d", &n);
    bits[0] = 0;
    for (int i = 1; i < (1 << n); ++i) {
      bits[i] = bits[i & (i - 1)] + 1;
    }
    ways[0] = 1;
    f[0] = 0;
    for (int i = 1; i < (1 << n); ++i) {
      ways[i] = 0;
      f[i] = 0;
      for (int j = 0; j < n; ++j) {
        if (j + 1 == bits[i]) {
          continue;
        }
        if (((i >> j) & 1) == 0) {
          continue;
        }
        int ni = i ^ (1 << j);
        ways[i] += ways[ni];
        f[i] += f[ni] + ways[ni] * bits[i & ~((1 << (j + 1)) - 1)];
      }
    }
    printf("%lld\n", f[(1 << n) - 1]);
  }
  return 0;
}
