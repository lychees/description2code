for i in xrange(int(raw_input())):
    num = int(raw_input())
    arr = raw_input().split()
    for j in xrange(num):
        arr[j] = int(arr[j])
    st = ""
    count = 0
    while (len(arr) >= 1):
        if (count % 2 == 0):
            x = min(arr)
            st += str(x) + " "
            arr.remove(x)
            count += 1
        else:
            y = max(arr)
            st += str(y) + " "
            arr.remove(y)
            count += 1
    print st
