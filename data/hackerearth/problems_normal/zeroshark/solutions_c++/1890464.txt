#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int dp[2][111111] ;
const int mod = 1e9 + 7 ;

void pre_compute(){
		dp[1][1] = dp[0][1] = 1 ;
		dp[1][0] = dp[0][0] = 1 ;
		for(int i=2;i<=100000;i++){
			dp[1][i] = (dp[1][i-1] + dp[0][i-1]) % mod ; 
			dp[0][i] = dp[1][i-1] ;
		}
}
int main()
{
    // cout << "Hello World!" << endl;
    pre_compute() ;
    int t ; cin >> t ;
    while( t-- ){
    	int n ; cin >> n ;
    	int ans = 0 ;
    	for(int i=1;i<n;i++){
    		ans += (1LL * dp[1][i-1] * dp[1][n-i-1]) % mod ; 
    		if( ans >= mod ) ans -= mod ; 
    	}
    	// ans = 1LL * ans * n % mod ;
    	cout << ans << endl ;
    }
    return 0;
}
