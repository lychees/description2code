t=input()

paliCount=[[]]

def cPalis(s,i,j):
	if j<i: return 0
	if i==j: return 1
	if paliCount[i][j]: return paliCount[i][j]
	su=cPalis(s,i,j-1)
	for k in range(i,j+1):
		if s[k]==s[j]: 
			su+=1+cPalis(s,k+1,j-1)
			#print k,j,s[k],s[j]
	return su	

for _ in range(t):
	s=raw_input()
	n=len(s)
	paliCount=[[0]*n for i in range(n)]
	r=cPalis(s,0,n-1)
