#include <bits/stdc++.h>
#define ll long long int

using namespace std;

ll A[1000005];

int main()
{
    ll t, n;

    cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> A[i];
        }
        queue<ll> Q;
        Q.push(0);
        bool mark[n + 1];
        memset(mark, false, sizeof(mark));

        while (!Q.empty()) {
            ll v = Q.front();
            Q.pop();

            if (v == -1) {
                cout << "Win\n";
                break;
            }
            if (mark[v]) {
                cout << "Lose\n";
                break;
            }
            mark[v] = true;
            Q.push(A[v]);
        }
    }
    return 0;
}
