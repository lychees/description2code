#include <bits/stdc++.h>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	
	map<string,int> m1;
	long count = 0;
	int n;
	cin >> n;
	while(n--)
	{
		string s;
		cin >> s;
		count += m1[s];
		
		reverse(s.begin(),s.end());
		
		m1[s]++;
		
		
		
		
	}
	
	cout << count << endl;
    //cout << "Hello World!" << endl;
    return 0;
}
