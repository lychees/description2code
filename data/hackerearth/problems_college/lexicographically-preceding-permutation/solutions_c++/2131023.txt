#include <bits/stdc++.h>

using namespace std;
const int N=1000001;
int ans,t,n,a[N];
int main()
{
    cin>>t;
    while (t--)
    {
        cin>>n;
        for (int i=1;i<=n;i++)
            cin>>a[i];
        for (int i=n-1;i>=1;i--)
            if (a[i]>a[i+1])
        {
            for (int j=n;j>i;j--)
                if (a[j]<a[i])
            {
                swap(a[j],a[i]);
                break;
            }
            sort(a+i+1,a+n+1);
            reverse(a+i+1,a+n+1);
            break;
        }
        for (int i=1;i<=n;i++)
            cout<<a[i]<<' ';
        cout<<endl;
    }
    return 0;
}
