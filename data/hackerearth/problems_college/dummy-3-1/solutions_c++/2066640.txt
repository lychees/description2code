#include <iostream>
using namespace std;

int main()
{
   int *arr,*len,i,t,n,flag;
   cin>>t;
   while(t--)
   {
   	flag=0;
   	cin>>n;
   	arr=new int[n+1];
   	len=new int[n+1];
   	for(i=1;i<=n;i++)
   	  cin>>arr[i];
   	for(i=1;i<=n;i++)
   	{
   		len[i]=arr[arr[i]];
   	}
   	for(i=1;i<n;i++)
   	if(len[i]!=i)
   	{
   		cout<<"not inverse"<<endl;
   		flag=1;
   		break;
   	}
   	if(!flag)
   	 cout<<"inverse"<<endl;
   }
    return 0;
}
