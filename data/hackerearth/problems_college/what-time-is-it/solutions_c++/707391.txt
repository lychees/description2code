#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    float hh ;
    float mm ;

    scanf("%f:%f",&hh,&mm);

    float  hr = mm/60;

    float hrdeg = (hh+hr)*30;

    float minu = mm*6;

    float d = fabsf(hrdeg - minu);

   if (d>180)
   {
       d= 360-d;
   }

    printf("%.6f",d);



}
