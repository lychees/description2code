#include<iostream>
using namespace std;
void part(int[],int,int);
void merge(int[],int,int,int);
void parte(int[],int,int);
void mergee(int[],int,int,int);
int main()
{
    int i,j,t,num,prod,men[10000],women[10000];
    long long sum;
    cin>>t;//no of test cases
    for(i=0;i<t;i++)
    {
        sum=0;
        cin>>num;//number of men or women
        for(j=0;j<num;j++)
        {
            cin>>men[j];//rating of men (10points)
        }
        for(j=0;j<num;j++)
        {
            cin>>women[j];//rating of women 10 pts
        }
        part(men,0,num-1);
        parte(women,0,num-1);
        for(j=0;j<num;j++)
        {
            prod=men[j]*women[j];//product of ratings
            sum=sum+prod;
        }
        cout<<sum<<"\n";//sum of pproducts of ratings
    }
return 0;
}
void part(int men[],int min,int max)
{
 int mid;
 if(min<max)
 {
   mid=(min+max)/2;
   part(men,min,mid);
   part(men,mid+1,max);
   merge(men,min,mid,max);
 }
}
void merge(int men[],int min,int mid,int max)
{
  int tmp[10000];
  int i,j,k,m;
  j=min;
  m=mid+1;
  for(i=min; j<=mid && m<=max ; i++)
  {
     if(men[j]<=men[m])
     {
         tmp[i]=men[j];
         j++;
     }
     else
     {
         tmp[i]=men[m];
         m++;
     }
  }
  if(j>mid)
  {
     for(k=m; k<=max; k++)
     {
         tmp[i]=men[k];
         i++;
     }
  }
  else
  {
     for(k=j; k<=mid; k++)
     {
        tmp[i]=men[k];
        i++;
     }
  }
  for(k=min; k<=max; k++)
     men[k]=tmp[k];
}
void parte(int women[],int min,int max)
{
 int mid;
 if(min<max)
 {
   mid=(min+max)/2;
   part(women,min,mid);
   part(women,mid+1,max);
   merge(women,min,mid,max);
 }
}
