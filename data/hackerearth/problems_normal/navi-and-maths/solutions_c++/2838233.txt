#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
ll fast(ll base,ll ex)
{
	ll res = 1;
	while(ex)
	{
		if(ex&1) res *= base;
		
		res %= mod;
		ex = ex >> 1;
		base = base* base;
		base %= mod;
	}
	return res;
}


ll  f(ll  a[],ll  last,ll  count,ll  pro,ll  sum)
{
	if(last>=0)
	{

 return max(f(a,last-1,count,pro,sum),
 	f(a,last-1,count+1,(pro*a[last])%mod,sum+a[last] ) ); 
	}
	if( count>=2)
	return ( (pro%mod) * fast(sum,mod-2)%mod) % mod;
	return -1;
}
int main()
{
    ll  t,n; cin >> t;
    ll  i=1;
   // cout<<fast(4,mod-2)<<endl;
    while(t--)
    {
    	cin >> n;
    	ll  a[n];
    	for(ll  i=0;i<n;i++) cin >> a[i];
    	cout<<"Case #"<<i<<": "<<f(a,n-1,0,1,0)<<endl;
    	i++;
    }
    return 0;
}
