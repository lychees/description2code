#include<bits/stdc++.h>
using namespace std;
long long int dp[1000000];
void pre(){
	long long int ans=0;
	for(int i=1;i<=1000000;i++){
		ans=0;
		for(int j=5;(i/j)>=1;j=j*5){
			ans+=(i/j);
		}
		dp[i]=dp[i-1]+ans;
	}
}
int main(){
	int t,a,b;
	cin>>t;
	pre();
	while(t--){
		cin>>a>>b;
		//if(a=0)cout<<dp[b]<<endl;
		cout<<dp[b]-dp[a-1]<<endl;
		//cout<<ans<<endl;
	}
	
	return 0;
}
