#include <bits/stdc++.h>

using namespace std;

vector <int> primes;
int len;

void pre(){
	primes.push_back(2);
	bool isPrime;
	for(int i = 3; i<1000000; i++){
		isPrime = true;
		for(int j = 0; primes[j]*primes[j]<=i; j++){
			if(i%primes[j]==0){
				isPrime = false;
				break;
			}
		}
		if(isPrime) primes.push_back(i);
	}
	len = primes.size();
}

bool checkPrime(int n){
	bool isPrime = true;
	int j = 0;
	while(n>1 && j<len){
		if(n==primes[j]) return true;
		if(n%primes[j]==0) return false;
		j++;
	}
	return true;
}

void solve(){
	int a,b;
	int ans = 0;
	cin>>a>>b;
	if(a==1) a = 2;
	if(a==2){
		for(int i = a+1; i<=b; i++){
			if(binary_search(primes.begin(),primes.end(),i)) ans++;
		}
	}
	cout<<ans<<endl;
}

int main(){
	pre();
	int t;
	cin>>t;
	while(t--) solve();
	return 0;
}
