#include<bits/stdc++.h>
using namespace std;

int a[10][10];

bool isSafeBox(int row,int col,int val) {
    int rowVal=row-row%3;
    int colVal=col-col%3;
    for(int i=rowVal;i<rowVal+3;i++) {
        for(int j=colVal;j<colVal+3;j++) {
            if(a[i][j]==val) {
                return false;
            }
        }
    }
    return true;
}
bool isSafeRow(int row,int val) {
    for(int i=0;i<9;i++) {
        if(a[row][i]==val) {
            return false;
        }
    }
    return true;
}
bool isSafeCol(int col,int val) {
    for(int i=0;i<9;i++) {
        if(a[i][col]==val) {
            return false;
        }
    }
    return true;
}
bool isSafe(int row,int col,int i) {
    if(isSafeCol(col,i)==true&&isSafeRow(row,i)==true&&isSafeBox(row,col,i)==true) {
        return true;
    }
    return false;
}
bool check(int& row,int& col) {
    for(int i=0;i<9;i++) {
        for(int j=0;j<9;j++) {
            if(a[i][j]==0) {
                row=i;
                col=j;
                return false;
            }
        }
    }
    return true;
}
bool solve() {
    int row=-1,col=-1;
    if(check(row,col)==true) {
        return true;
    } else {
        for(int i=1;i<=9;i++) {
            if(isSafe(row,col,i)==true) {
                a[row][col]=i;
                if(solve()==true) {
                    return true;
                }
                a[row][col]=0;
            }
        }
    }
    return false;
}
int main() {
    int tt;
    scanf("%d",&tt);
    int sum=0;
    while(tt--) {
        string str;
        for(int i=0;i<9;i++) {
            cin>>str;
            for(int j=0;j<9;j++) {
                a[i][j]=str[j]-'0';
            }
        }
        /*for(int i=0;i<9;i++) {
            for(int j=0;j<9;j++) {
                scanf("%d",&a[i][j]);
            }
        }*/
        if(solve()==false) {
            printf("Not Possible");
        } else {
            for(int i=0;i<9;i++) {
                sum+=a[i][i];
            }
        }
    }
    printf("%d\012",sum);
    return 0;
}
