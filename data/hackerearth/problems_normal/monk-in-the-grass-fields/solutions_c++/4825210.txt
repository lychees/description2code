using namespace std;
typedef vector<int> V;

const long long LINF = (long long)1e16 + 5;
const int INF =1e9;
const int MOD=1000000007;
const int maxlen=1e5;
int mat[201][201],row[201],col[201],n,sr[400],sc[400];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int i,j,k,sum,x;
        cin>>n>>k;
        mset(row,0);
        mset(col,0);
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                cin>>mat[i][j];
                row[i]+=mat[i][j];
                col[j]+=mat[i][j];
            }
        }
       int ans=INF;
       mset(sr,0);
       mset(sc,0);
       for(i=1;i<=k;i++)
       {
          x=0;
          for(j=0;j<n;j++)
          {
              if(row[j]<row[x])
              {
                  x=j;
              }
          }
          sr[i]=sr[i-1]+row[x];
          row[x]+=n;
       }
       for(i=1;i<=k;i++)
       {
          x=0;
          for(j=0;j<n;j++)
          {
              if(col[j]<col[x])
              {
                  x=j;
              }
          }
          sc[i]=sc[i-1]+col[x];
          col[x]+=n;
       }
       for(i=0;i<=k;i++)
       {
           ans=min(ans,sr[i]+sc[k-i]+i*(k-i));
       }
       cout<<ans<<endl;
        
    }
}
