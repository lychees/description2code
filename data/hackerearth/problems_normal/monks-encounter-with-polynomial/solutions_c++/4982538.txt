#include <iostream>
using namespace std;

class Polynomial{
	public:
		Polynomial(int A, int B, int C):A(A),B(B),C(C){}
		int operator()(int x) const{
			return (A*x*x+B*x+C);
		}
	private:
		int A;
		int B;
		int C;
};

int main()
{
    int tests;
    cin >> tests;
    for(int test(0) ; test<tests ; ++test){
    	int A,B,C,K;
    	cin >> A >> B >> C >> K;
    	Polynomial poly(A,B,C);
    	int x(0);
    	while(true){
    		if(poly(x) >= K){
    			cout << x << endl;
    			break;
    		}
    		++x;
    	}
    }
    return 0;
}
