#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int count=0,num;
        cin>>num;
        int a,hash[100000]={0};
        for(int i=0;i<num;i++)
        {
            cin>>a;
            if(hash[a]==0)
            {
                count++;
                hash[a]=1;
            }

        }
        cout<<count<<endl;
    }
}
