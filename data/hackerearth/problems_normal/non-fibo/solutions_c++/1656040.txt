#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long a,b,c,fib[100],n;
	int count=0,t;
	a=b=1,c=1;
	while(c<100000000000000000ll)
	{
		fib[count]=c;
		a=b;
		b=c;
		c=a+b;
		count++;
	}
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		long long beg = -1,mid,end=max(100ll,n+n);
		mid = (beg+end)/2;

		while(beg+1<end)
		{	
		//	cout<<beg<<" "<<mid<<" "<<end<<"\n";
		
			if(mid-distance(fib,lower_bound(fib,fib+count,mid+1))>=n)
				end = mid;
			else
				beg = mid;
			mid = (beg+end)/2;
		}
		printf("%lld\n",end);
	}
}
