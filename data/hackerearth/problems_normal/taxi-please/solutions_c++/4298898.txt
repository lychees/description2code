bool func(const pair<int,pair<int,int> > &p1,const pair<int,pair<int,int> > &p2)
{
	if(p1.second.first<p2.second.first)

	else if(p1.second.first==p2.second.first)
	{
		if(p1.second.second<p2.second.second)
		return true;
	}
	return false;
}
