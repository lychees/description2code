import operator
n = input()
ranks = []
while(n>0) :
	n -= 1
	idno, z, p, l, c, s = map(int, raw_input().split())
	newz = (p*50 + c*10 + l*5 + s*20)
	ranks.append([idno, newz, newz-z])
ranks = sorted(ranks, key=operator.itemgetter(0))
ranks = sorted(ranks, key=operator.itemgetter(2))
ranks = ranks[-5:]
ranks.reverse()
for i in range(5) :
	print ranks[i][0], ranks[i][1]
