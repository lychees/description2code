#include <iostream>
#include <map>
using namespace std;

map<char,int> mymap;
typedef map<char,int> :: iterator IT;
int main(){
	int test;
	cin>>test;
	while(test--){
		int N;
		string str;
		cin>>N;
		
		for(int i =0;i<N;i++){
			cin>>str;
			
			IT itr = mymap.find(str[0]);
			if(itr==mymap.end()){
				mymap.insert(make_pair(str[0],1));
			}
			else
			mymap[str[0]]+=1;
		}
		bool check = true;
		for(IT itr =mymap.begin();itr!=mymap.end();++itr){
			if((*itr).second>1){
			    check = false;
			    break;
		    }
		}
		
		if(check)
		   cout<<"YES"<<'\n';
		else
		   cout<<"NO"<<'\n';
		 
		mymap.clear();
	}
	return 0;
	
}
