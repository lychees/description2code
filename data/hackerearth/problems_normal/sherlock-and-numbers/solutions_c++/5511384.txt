#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t,n,k,p,temp,i;
    scanf("%d",&t);
   	while(t--){
   		scanf("%d%d%d",&n,&k,&p);
   		if((n-k)>=p){
   			for(i=0;i<k;i++){
   				scanf("%d",&temp);
   				if(temp<=p)
   					p++;
   			}
   			printf("%d\n",p);
   			continue;
        }
   		else if((n-k)<p){
   			for(i=0;i<k;i++)
   				scanf("%d",&temp);
   			printf("-1\n");
   			continue;
   		}
   	}
    return 0;
}
