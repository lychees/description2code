#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		bool b=false;
		int* ar=new int[n];
		int* sum=new int[n];
		cin>>ar[0];
		sum[0]=ar[0];
		for(int i=1;i<n;i++)
		{
			cin>>ar[i];
			sum[i]=sum[i-1]+ar[i];
		}
		for(int i=1;i<n;i++)
		{
			if(sum[i-1]==sum[n-1]-sum[i-1]-ar[i])
			{
			b=true;
			break;
			}
		}
		if(b==false)
		cout<<"NO"<<endl;
		else
		cout<<"YES"<<endl;
	}
}
