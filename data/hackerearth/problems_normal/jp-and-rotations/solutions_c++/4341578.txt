#include<iostream>
#include<algorithm>
#include<cmath>
#include<map>
#include<vector>
#include<queue>
using namespace std;
#define ll long long int
ll a[100003],am[100001];
char mo[100003];
ll foo(ll p,ll n)
{
    while(p>n)
    {
        p-=n;
    }
    return p;
}
ll f1(ll p,ll n)
{
    while(p<1)
    {
        p+=n;
    }
    return p;
}
int main()
{
ll x,n,i,j,k,q,t,m,b;
cin>>n>>m;
for(i=0;i<n;i++)
{
    cin>>a[i];
}
ll p=a[0];
for(i=0;i<n;i++)
{
    cin>>b;
    if(b==p)
    {
        q=i+1;
    }
}
for(i=0;i<m;i++)
{
    cin>>mo[i]>>am[i];
}
ll re=0;
p=1;
for(i=0;i<m;i++)
{
    if(mo[i]=='R')
    {
        p=p+am[i];
        if(p>n)
        {
            p=foo(p,n);
        }
        re+=1;
        if(p==q)
        {
            break;
        }
    }
    else
    {
        p=p-am[i];
        if(p<1)
        {
            p=f1(p,n);
        }
        re+=1;
        if(p==q)
        {
            break;
        }
    }
}
if(i==m)
{
    cout<<"-1";
}
else
{
cout<<re;
}
return 0;
}
