test_cases = int(raw_input())
for i in range(test_cases):
    length = int(raw_input())
    a = [0]*length
    a = raw_input().split()
    for i in range(length):
        a[i] = int(a[i])
    flag = 0
    for i in range(length-1):
        if a[i] > a[i+1]:
            flag += 1
    if flag == 0:
        print "YES"
    else:
        print "NO"
