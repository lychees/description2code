#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <vector>
#include <iostream>
#include <algorithm>
#include <iomanip>
using namespace std;
int n,w;
const int maxn = 51;
int dp[maxn][maxn];
int main() {
    string s;
    cin>>s;
    int len = s.length();
    for(int i=0;i<maxn;i++)
        for(int j=0;j<maxn;j++)
          dp[i][j]=0;
    for(int i=1;i<=len;i++)
        dp[i][i] = 1;
    int ans = len;
    for(int i=1;i<=len;i++)
      if(s[i]==s[i-1]) 
        {
         dp[i][i+1] = 1;
         ans++;
        }
    for(int l=3;l<=len;l++)
    {
       for(int i=1;i<=len-l+1;i++)
           {
            int j=i+l-1;
            if(s[i-1]==s[j-1] && dp[i+1][j-1]==1) {dp[i][j] = 1;ans++;} 
           }
    }
    cout<<ans<<endl;
  return 0;
}
