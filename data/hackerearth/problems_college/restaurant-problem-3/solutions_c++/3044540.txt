#include <iostream>
#include<cstring>
using namespace std;

int main()
{
   char str[1000001];
   cin>>str;
   int d1,d2,d3,count=0,div=0;
   int l=strlen(str);
   for(int i=0;i<l;i++)
   {
	d1=str[i]-'0';
	if(d1%8==0)
	div++;
	if(i<l-1)
	{
		d2=d1*10+(str[i+1]-'0');
		if(d2%8==0 && d2>10)
		div++;
	}
	if(i<l-2)
	{
		d3=d2*10+(str[i+2]-'0');
		if(d1>0)
		count++;
		if(d3%8==0)
		div+=count;
	}
   }
   cout<<div<<endl;
   return 0;
}
