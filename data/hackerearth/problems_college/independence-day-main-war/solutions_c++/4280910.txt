#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
   	int t;
   	cin>>t;
   	while(t--)
   	{
   		float n,x,y,sumx=0,sumy=0;
   		cin>>n;

   		for(int i=0;i<n;i++)
   		{
   			cin>>x>>y;
   			sumx+=x;
   			sumy+=y;
   		}
   		cout<<"(";
   		cout<<fixed<<setprecision(2)<<sumx/n;
   		cout<<" , ";
   		cout<<fixed<<setprecision(2)<<sumy/n;
   		cout<<")"<<endl;
   	}
    return 0;
}
