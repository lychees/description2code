#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int t,ad[26],bd[26],i,c;
    char a[100001],b[100001];
    cin>>t;
    while(t--){
    	for(i=0;i<26;i++)
    	{ad[i]=0;bd[i]=0;}
    	c=0;
    	cin>>a>>b;
    	for(i=0;i<strlen(a);i++)
    		ad[a[i]-97]++;
    		for(i=0;i<strlen(b);i++)
    		bd[b[i]-97]++;
    		for(i=0;i<26;i++){
    			if(ad[i]>0&&bd[i]>0)
    			{c=1;break;}
    		}
    		if(c==1)
    		cout<<"YES\n";
    		else
    		cout<<"NO\n";
    }
    return 0;
}
