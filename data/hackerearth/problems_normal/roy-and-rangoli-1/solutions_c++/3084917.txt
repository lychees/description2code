#include <iostream>
#include <vector>
using namespace std;
vector<long long int> v1,v2;
long long int mod=1000000007;
int main()
{
    int n,p1,p2;
    cin>>n;
    v1.resize(2*n+1,0);
    for(p1=2;p1<=2*n;p1++)
    {
    	if(v1[p1]==0)
    	{
    		v1[p1]=v1[p1-1]+1;
    		for(p2=2*p1;p2<=2*n;p2+=p1)
    			v1[p2]=-1;
    	}
    	else
    		v1[p1]=v1[p1-1];
    }
    v2.resize(n+1,0);
    if(n==1)
    {
    	cout<<0;
    	return 0;
    }
    if(n==2)
    {
    	cout<<1;
    	return 0;
    }
    v2[0]=0;
    v2[1]=0;
    v2[2]=1;
    for(p1=3;p1<=n;p1++)
    {
    	p2=p1-1;
    	v2[p1]=(((v1[p2*2]-v1[p2-1])*2)%mod+v2[p1-1]%mod)%mod;
    }
    cout<<v2[n];
    return 0;
}
