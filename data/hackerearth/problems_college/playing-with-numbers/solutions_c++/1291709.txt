#include<bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
inline void scanllint(long long int &x)
{
    long long int c = gc();
    x = 0;
    long long int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int main()
{
    long long int t,n,o=1;
    scanllint(t);
    while(t--)
    {
        scanllint(n);
        long long int c=0;
        while(n)
        {
            if(n&1)
                c++;
            n>>=1;
        }
        if(c%2==1)
        {
            if(o==1)
                printf("Sherlock\n");
            else
            {
                printf("Watson\n");
                o=2;
            }
        }
        else
        {
            if(o==2)
                printf("Sherlock\n");
            else printf("Watson\n");
            o=(o==1)?2:1;
        }
    }
    return 0;
}
