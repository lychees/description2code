#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n;
		int x5 = 0,x3=0;
	
		if(n%3==0)
		{
			for(k=0;k<n;k++)
				cout<<5;
			cout<<endl;
		}
		else if(n%5==0)
		{
			for(k=0;k<n;k++)
				cout<<3;
			cout<<endl;
		}
		else
		{
			int l;
			k = n;
			while(k>=0)
			{
				x5 = n-k;
				if(x5%3==0&&k%5==0)
				{
					l = x5;	
				}
				k-=3;
			}
			x3 = n - l;
			if(l%3==0&&x3%5==0)
			{
				for(k=0;k<l;k++)
					cout<<5;
				for(k=0;k<x3;k++)
					cout<<3;
				cout<<endl;
			}
			else
			{
				cout<<"-1\n";
			}
		}
	}
	return 0;
}
