#include <iostream>
using namespace std;
void compute(unsigned int *a,int n){
	a[0]=0;
	a[1]=1;
	for(int i=2;i<n;i++){
		a[i]=a[i-1]+a[i-2];
	}
}
int main()
{
    int t;
    cin>>t;
    unsigned int *a= new unsigned int[28];
    compute(a,28);
    while(t--){
    	int x,y;
    	cin>>x;
    	cin>>y;
    	int count =0;
    	for(int i=0;i<28;i++){
    		if(a[i]>=x && a[i]<=y){
    			count++;
    		}
    		else if(a[i]>y){
    			break;
    		}
    	}
    	if(count %2 ==0){
    		cout<<"VALID"<<endl;
    	}
    	else{
    		cout<<"INVALID"<<endl;
    	}
    }
    delete []a;
    return 0;
}
