#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
const int N = 1e6+5;
int n, s, q, a[N];
ll R = 0;

int main()
{
	ios::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	cin >> n;
	while (n--) cin >> s >> q, a[s] = q;
	for (int i = N - 5; i >= 0; --i) a[i] = max(a[i], a[i + 1]);
	cin >> n;
	while (n--)
	{
		cin >> s;
		R += ((ll)a[s] / (ll)s) * 100LL;
	}
	cout << R;
    return 0;
}
