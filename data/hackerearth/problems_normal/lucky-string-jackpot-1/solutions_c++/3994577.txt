#include <iostream>
#include <math.h>
using namespace std;
typedef long long int ll;
int main()
{
    ll t;
    cin >> t;
    while(t--) {
    	ll n;
    	ll val;
    	cin >> n; 
    	if(n > 4) {
    		double k = log2(n - 2);
    		k = k - 1;
    		ll x = (ll)(k);
    		while(true) {
    			val = pow(2, x + 1) * ( x - 1)	 + 2;
    			if(val > n) {
    				x--;
    			}
    			else {
    				break;
    			}
    		}
	 		ll max = 2 * x - 1;
	 		val = n - val;
	 		ll p = 2 * x + 1;
    		if(val >= p) {
    			max = p; 
    		}
    		else {
    			if(val > max) {
    				max = val;
    			}
    		}
    		cout << max << endl;
	  }
	  else {
	  	if(n <= 3) {
	  		cout << "1\n";
	  	}
	  	else {
	  		cout << "2\n";
	  	}
	  }
    }
    return 0;
}
