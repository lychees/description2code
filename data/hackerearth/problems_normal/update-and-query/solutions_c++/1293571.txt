#include <iostream>
using namespace std;
int n,m;
long long a[200001];
int main()
{
    scanf("%d%d",&n,&m);
    char op;
    int x;
    long long y,c;
    for(int i=1;i<=n;i++) scanf("%lld",&a[i]);
    while(m--){
    	cin>>op;
    	if (op=='U') {
    		scanf("%d%lld",&x,&y);
    		a[x] = y;
    	}
    	else {
    		scanf("%d%lld",&x,&c);
    		if (c<0){
    			cout << "-1 -1\n";
    			continue;
    		}
    		long long low = a[x]-c,high = a[x]+c;
    		int ans = 1;
    		long long cbest = 0;
    		for(int i=x+1;i<=n;i++){
    			if (a[i]>=low && a[i]<=high) {
    				ans++;
    				cbest = max(cbest,abs(a[x]-a[i]));
    			}
    			else break;
    		}
    		cout << ans << " " << min(cbest,c) << "\n";
    	}
    }
    return 0;
}
