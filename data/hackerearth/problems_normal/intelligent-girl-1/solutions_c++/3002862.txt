#include <iostream>
using namespace std;
#include <stdio.h>
#include<string.h>
#include<malloc.h>
#define MAX_SIZE 100000
int main()
{
 char s[MAX_SIZE];
 int count,i=0,j=0,d,len=0;
 
gets(s);
len=strlen(s);
for(i=0;i<len;i++)
{
	count=0;
	for(j=i;j<len;j++)
	{
		d=(int)s[j];
	
		if(d%2==0)
		count++;
	}
	cout<<count<<" ";
}

}
