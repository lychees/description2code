#include <iostream>
using namespace std;

int main()
{
	long long int t,n1,n2,c,s,i;
	int a[20],b[20];
   cin>>t;
   while(t--){
   	s=0;
   	c=0;
   	for(i=0;i<10;i++)
   	{
    a[i]=0;
   	b[i]=0;
   	}
cin>>n1>>n2;
while(n1!=0){
	a[c++]=n1%10;
	n1=n1/10;
}
c=0;
while(n2!=0){
	b[c++]=n2%10;
	n2=n2/10;
}
c=0;
 i=0;
while(i<10){
	if(c+a[i]+b[i]>=10){
	
	     c=1;
		s++;
	}
	else
	c=0;
	i++;
}
	if(s==0)
	{
		cout<<"No carry operation"<<endl;
	}
	else if(s==1)
	cout<<"1 carry operation"<<endl;
	else
	cout<<s<<" carry operations"<<endl;
}
    
   	
   	
   
    return 0;
}
