#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t,n;
    vector<ll>v;
    cin>>t;
    while(t--){
    	cin>>n;
    	v.clear();
    	v.resize(n);
    	ll xor_;
    	ll and_;
    	bool b=0;
    	ll m=-1000000;
    	for(int i=0;i<n;i++)cin>>v[i];
    	ll pow_set_size = pow(2,n);
    	ll sz=0,new_sz=0;;
    	for(int i=0;i<pow_set_size;i++){
    		sz=0;
    		b=0;
    		xor_=-1;
    		and_=-1;
    		for(int j=0;j<n;j++){
    			if(i & (1<<j)){
    				if(b==0){
    					xor_=v[j];
    					and_=v[j];
    					sz++;
    					b=1;
    				}
    				else{
    					xor_=xor_^v[j];
    					and_=and_&v[j];
    					sz++;
    				}
    			}
    		}
    		ll or_=(xor_ | and_);
    		if(m==or_){
    			new_sz=max(new_sz,sz);
    		}
    		else if(m<or_){
    			m=or_;
    			new_sz=sz;
    		}
    	}
    	cout<<m<<" "<<new_sz<<endl;
    }
}
