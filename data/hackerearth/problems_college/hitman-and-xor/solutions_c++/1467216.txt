#include <iostream>
using namespace std;

int main()
{
    int tc;
    scanf("%d",&tc);
    while(tc--)
    {
    	long int n,even=0,odd=0,num,i;
    	scanf("%ld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%ld",&num);
    		if(num%2==0)
    		even++;
    		else
    		odd++;
    	}
    	printf("%ld\n",even*odd);
    }
    return 0;
}
