using namespace std;

#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a<b?a:b)
#define FOR(i,n) for(i=0;i<n;i++)
#define FORJAS(i,j,n) for(i=j;i<n;i++)
#define FORS(i,n) for(i=1;i<=n;i++)
#define FORJS(i,j,n) for(i=j;i<=n;i++)
#define LEN 105
#define INF 1000005
#define I int
#define LL long long
#define L long

pair<LL,LL> princes[INF];
LL hardles[INF];

int main (){
	LL t,n,k,i,j,tmp;
	// cin>>t;
	scanf("%lld",&t);
	while(t--){
		// cin>>n>>k;
		scanf("%lld %lld",&n,&k);
		FOR(i,n){
			scanf("%lld",&tmp);
			// cin>>tmp;
			princes[i]=make_pair(tmp,i);
		}
		FOR(i,k){
			scanf("%lld",&hardles[i]);
			// cin>>hardles[i];
		}
		sort(princes,princes+n);
		LL ans=0;
		for(i=0,j=0;i < n && j < k;j++){
			LL minInx = INT_MAX;
			while(i < n && princes[i].first < hardles[j]){
				minInx = MIN(minInx,princes[i].second);
				i++;
			}
			ans = minInx;
		}
		if(i < n){
			ans = INT_MAX;
			while(i<n){
				ans = MIN(ans,princes[i].second);
				i++;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
