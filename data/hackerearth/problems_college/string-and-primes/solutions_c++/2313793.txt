#include <iostream>
#include<string>
#include<math.h>
using namespace std;

int is_prime(int n)
{
	if(n>2 && n%2==0)
	return 0;
	
	for(int i=3; i<=sqrt(n); i+=2)
	{
		if(n%i == 0)
		return 0;
	}
	return 1;
}

int main()
{
    int t;
    cin>>t;
    string str;
    while(t--)
    {
    	cin>>str;
    	int index = 0,flag = 0;
    	for(int i=0; i<str.length(); i++)
    	{
    		if(!(is_prime(str[i]))){
    		cout<<str[i];	
    		flag = 1;
    		}
    	}
    	if(flag == 0)
    	cout<<"NULL";
        cout<<"\n";
    }
    
    return 0;
}
