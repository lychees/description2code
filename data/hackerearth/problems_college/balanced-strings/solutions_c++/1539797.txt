#include <iostream>
#include<string>
using namespace std;

int main()
{
   int T;
   cin>>T;
   while(T--){
   	string str;
   	cin>>str;
   	int arr[26];
   	for(int i=0;i<26;i++)arr[i]=0;
   	for(int i=0;i<str.length();i++)arr[str[i]-'a']++;
   	int ma=arr[0];
   	bool flag=0;
   	for(int i=1;i<26;i++){
   		if(ma!=arr[i]){
   			flag=1;
   		}
   	}
   	if(flag)cout<<"No"<<endl;
   	else cout<<"Yes"<<endl;
   }
    return 0;
}
