#include <iostream>
#include <vector>


using namespace std;

int main()
{
	int n , k ; 

	cin >> n >> k ; 

	vector <int> know_angle ;
	
	for(int i = 0 ; i <  n ; i ++)
	{
		int temp ; 
		cin >> temp ; 
		know_angle.push_back(temp%360) ; 
	}
	
	
	for(int i = 0 ; i < k ; i ++)
	{
		int temp ; 
		cin >> temp ;
		int x = 0 ; 
		
		for(int i = 0 ; i < n ; i++)
		{
			int ang  = know_angle[i] ;
			for(int j = 1 ; j <= 360 ; j++ )
			{
				if(ang == temp)
				{
					cout << "YES" << endl ; 
					x = 1 ;
					break ; 
				}
				ang += know_angle[i] ; 
				ang = ang%360 ;
			}
			if(x == 1)
				break ; 
		}
		
		if(x == 0)
			cout << "NO" << endl ; 
		
	}
	

	
	
	
	return 0;
}
