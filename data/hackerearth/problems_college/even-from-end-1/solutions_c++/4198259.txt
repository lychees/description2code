#include <iostream>
using namespace std;

int main()
{
	int n;
	while(cin>>n){
		int a[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		
		int flag=1;
		for(int i=n-1;i>=0;i--){
			if(a[i]%2==0){
				cout<<a[i]<<" ";
				flag=0;
			}
		}
		if(flag==1)cout<<"None to display"<<endl;
		else cout<<endl;
	}
    return 0;
}
