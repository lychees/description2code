#include <iostream>
using namespace std;
int rev(string str){
	int l = str.size();
	int s=0;
	int e = l-1;
	
	while(s<l){
		if(str[s]!=str[e])
			return 0;
			
		s++;
		e--;
	}
	return 1;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
    	string str;
    	cin>>str;
    	if(rev(str))
    		cout<<"YES"<<endl;
    	else
    		cout<<"NO"<<endl;
    }
}
