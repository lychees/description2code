long long gcd(long long a,long long b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);	
}
int main()
{
	int t;
	cin>>t;
	long long a,b,c;
	while(t--)
	{
		cin>>a>>b>>c;
		long long l=gcd(gcd(a,b),c);
		a/=l;
		b/=l;
		c/=l;
		unsigned long long ans=(((a*b)%1000000007)*c)%1000000007;
		cout<<l<<" "<<ans<<endl;
	}
