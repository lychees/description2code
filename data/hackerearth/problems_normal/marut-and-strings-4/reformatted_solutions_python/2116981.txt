import sys

small = 'abcdefghijklmnopqrstuvwxyz'
caps = small.upper()


def main():
    countLine = 0
    noTestCases = 0
    file_brk = sys.stdin.readlines()
    noActualTestCase = len(file_brk) - 1
    for line in file_brk:
        line = line.rstrip('\n')
        if countLine == 0:
            countLine += 1
            if not line.isdigit():
                print 'Invalid Test'
                break
            else:
                noTestCases = int(line)
                if noTestCases < 1 or noTestCases > 10:
                    print 'Invalid Test'
                    break
        else:
            countLine += 1
            if countLine > noTestCases+1:
                break
            print doStringThing(line)


def doStringThing(lineVal):
    noOfUpper = 0
    noOfLower = 0
    noLetters = True
    if not lineVal or len(lineVal) > 100 or ' ' in lineVal:
        return 'Invalid Input'
    else:
        for letter in lineVal:
            if letter in small:
                noOfLower += 1
                noLetters = False
            elif letter in caps:
                noOfUpper += 1
                noLetters = False
    if noLetters:
        return 'Invalid Input'
    elif noOfUpper > noOfLower:
        return str(noOfLower)
    else:
        return str(noOfUpper)


if __name__ == '__main__':
    main()
