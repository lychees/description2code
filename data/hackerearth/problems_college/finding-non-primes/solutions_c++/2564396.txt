using namespace std;

const int SIZE = 1e4 + 10;

int N;
bool prime[ SIZE ];

int main(){
    int i,k;

    scanf("%d", &N);

    for ( i=2; i<=N; i++)
        if ( prime[ i ] ) printf("%d\n", i);
        else for ( k = i+i; k<=N; k+=i )
                prime[ k ] = true;
}
