#include<iostream>
#include<bits/stdc++.h>
using namespace std;
bool check(int n)
{
	while(n!=0)
	{
		int r1=n%2;
		n=n/2;
		int r2=n%2;
		if(r1&r2==1)
		return false;
		
	}
	return true;
	}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		while(!check(n))
		n++;
		cout<<n<<endl;
	}
	return 0;
	}
