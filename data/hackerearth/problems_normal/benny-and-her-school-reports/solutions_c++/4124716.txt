#include <bits/stdc++.h>
using namespace std;

int main()
{
    long int t=0;
    cin>>t;
    while(t--){
        long int m=0,n=0,x=0,y=0;
        int a[100000],p=0;
        cin>>p>>m>>n;
        for(int i=0;i<p;i++) {
                cin>>a[i];
                x=x+a[i];
        }
        y=n*(p+1)-x;
        if(1<=y&&y<=m) cout<<y<<endl;
        else if(y<1) cout<<"1"<<endl;
        else cout<<"Impossible"<<endl;
            }
    return 0;
}
