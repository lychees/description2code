def is_complete(str):
    arr = sorted(set(str))
    if len(arr) == 26:
        return 1
    else:
        return 0


N = input()
for i in range(N):
    str = raw_input().lower()
    if is_complete(str) == 1:
        print 'YES'
    else:
        print 'NO'
