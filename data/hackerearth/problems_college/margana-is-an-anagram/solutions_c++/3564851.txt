#include <bits/stdc++.h>

using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int flag=1;
		string first;
		string second;
		cin>>first>>second;
		int count1[27],count2[27];
		memset(count1,0,sizeof(int));
		memset(count2,0,sizeof(int));
		for(int i=0;i<27;i++)
		{
			count1[i]=0;
			count2[i]=0;
		}
		for(int i=0;i<first.length();i++)
		{
			count1[first[i]-97]++;
			//cout<<first[i]-97<<" ";
		}

		for(int i=0;i<second.length();i++)
		{
			count2[second[i]-97]++;
			//cout<<second[i]-97<<" ";
		}
		for(int i=0;i<27;i++)
		{
			if(count1[i]%2==0 && count2[i]%2==0 || count1[i]%2==1 && count2[i]%2==1 )
			{
				flag=1;
			}
			else
			{
				flag=0;
				break;
			}
			
		}
		if(flag==1)
			cout<<"Yes"<<endl;
		else
			cout<<"No"<<endl;
	}
	return 0;
}
