#include<bits/stdc++.h>
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,m,i,j,incl,excl;
    cin>>n>>m;
    ll availC[m];
    for(i=0;i<m;i++)
    {
        cin>>availC[i];
    }
    ll dp[n+1][m];
    for(i=0;i<m;i++)
    {
        dp[0][i]=1;
    }
    for(i=1;i<n+1;i++)
    {
        for(j=0;j<m;j++)
        {
            if( (i-availC[j])>=0 )
                incl=dp[i-availC[j]][j];
            else
                incl=0;
            if(j>=1)
                excl=dp[i][j-1];
            else
                excl=0;
            dp[i][j]=incl+excl;
        }
    }
    cout<<dp[n][m-1];
    return 0;
}
