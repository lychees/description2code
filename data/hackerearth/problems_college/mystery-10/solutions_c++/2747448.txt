#include <iostream>
using namespace std;

int main()
{
    int t,i;
    char s[100000];
    char h[26] = {'q','w','e','r','t','y','u','i','o','p','a','s','d','f','g','h','j','k','l','z','x','c','v','b','n','m'};
    
    scanf("%d",&t);
    while(t--){
    	scanf(" %[^\n]s",&s);
    	i=0;
    	while(s[i]!='\0'){
    		if(isalpha(s[i])){
    			printf("%c",h[s[i]-'a']);
    		}else{
    			printf(" ");
    		}
    		i++;
    	}
    	printf("\n");
    }
    return 0;
}
