#include<iostream>
#include<cstdio>
using namespace std;
#define mod 1000000007
long long arr2[10000001];
long long arr3[10000001];
int main(){
	arr2[1]=2;
	arr3[1]=3;
	for(int i=2;i<=10000000;i++){
		arr2[i]=(arr2[i-1]*2)%mod;
		arr3[i]=(arr3[i-1]*3)%mod;
	}
	int t,n;
	long long ans;
	scanf("%d",&t);
	while(t--){
		ans = 0;
		scanf("%d",&n);
		if(n==2 || n==1) ;
		else{
		ans += arr3[n];
		ans = ans+3;
		ans = ans%mod;
		ans = ans-(arr2[n]*3)%mod;
		if(ans<0) ans+=mod;}
		cout<<ans<<endl;
		
	}
}
