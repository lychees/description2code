#include<iostream>

using namespace std;

int isPrime(int x)
{
	bool ans=1;
	if(x==1)
	return 0;
	for(int i=2;i<x;i++)
	if(x%i==0)
	{ans=0;break;}
	return ans;
}
int gcdr ( int a, int b )
{
  if ( a==0 ) return b;
  return gcdr ( b%a, a );
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n==1)
		{
		cout<<"FALSE\n";
		cin>>n;
		t--;
		}
		int rbf=0;
		for(int k=1;k<n;k++)
		if(gcdr(n,k)==1)
		rbf++;
		if(isPrime(rbf))
		cout<<"TRUE\n";
		else
		cout<<"FALSE\n";
	}
}
