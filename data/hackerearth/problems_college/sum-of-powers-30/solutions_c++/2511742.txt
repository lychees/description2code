#include<bits/stdc++.h>
typedef long long LL;
LL arr[100000000];
int hasIt(LL value,int size)
{
    int i;
    for(i=0;i<size;i++)
        if(arr[i]==value)
            return 1;
    return 0;
}
 
main()
{
	
    LL i,j,k,test,val,pw,ans;
    double sq,lg;
    int counter=0;
    scanf("%lld",&test);
    while(test--)
    {
        scanf("%lld",&val);
        if(val==2){
            printf("Yes\n");
            continue;
        }
        sq=sqrt(val);
        counter=0;
        ans=0;
        arr[counter++]=1;
        for(i=2;i<=sq&&ans==0;i++)
        {
            lg=log(val)/log(i);
            for(j=2;j<=lg&&ans==0;j++)
            {
                pw=pow(i,j);
                if(hasIt(val-pw,counter)||pw+pw==val){
                    ans=1;
                    break;
                }
                arr[counter++]=pw;
            }
        }
        if(ans==1)
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
