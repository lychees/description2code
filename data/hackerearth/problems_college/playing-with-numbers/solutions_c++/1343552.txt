#include <iostream>
using namespace std;

int main()
{
   int t;
   int n;
   bool ok=0;
   scanf("%d",&t);
   while(t--){
   	scanf("%d",&n);
   	if(__builtin_popcount(n)%2){
   		if(!ok){
   		puts("Sherlock");
   		ok=0;
   		}
   		else{
   			puts("Watson");
   			ok=1;
   		}
   	}
   	else{
   		if(!ok){
   		puts("Watson");
   		ok=1;
   		}
   		else{
   			puts("Sherlock");
   			ok=0;
   		}
   	}
   }
    return 0;
}
