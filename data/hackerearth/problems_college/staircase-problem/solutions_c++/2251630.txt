#include <iostream>
#include <stdio.h>

using namespace std;

const int maxn = 112345;

int ar[maxn];

int main()
{
	int flag=0,add=1;
	int i=0;
	while(flag+add < maxn)
	{
		ar[flag+add]=++i;
		flag+=add;
		add++;
	}
	for(int i=1;i<maxn;i++)
	{
		if(ar[i]) flag=ar[i];
		ar[i]=flag;
	}
	int t;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		int n;
		scanf("%d",&n);
		printf("%d\n", ar[n]);
	}
	return 0;
}
