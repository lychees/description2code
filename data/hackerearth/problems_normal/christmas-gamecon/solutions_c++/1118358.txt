#include <iostream>
#include <cstdio>
using namespace std;
long long arr[101024];

int main(){
	int T,ans;
	scanf("%d",&T);
	while(T--){
		int N,i;
		scanf("%d",&N);
		for(i=0;i<N;i++){
			scanf("%lld",&arr[i]);
			arr[i]=arr[i]%5;
		}
		ans=0;
		for(i=0;i<N;i++){
			ans=ans^arr[i];
		}
		if(ans)
		printf("Aniruddha\n");
		else
		printf("Andrew\n");
	}
	return(0);
}
