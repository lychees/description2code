#include <iostream>
#define ul unsigned long

using namespace std;

ul a,b,c,k;

ul calc(ul x)
{
	return a*x*x + b*x + c;
}

int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    
    ul t,s,f,mid,ans;
    cin >> t;
    while (t--)
    {
    	cin >> a >> b >> c >> k;
    	if (c >= k)
    	{
    		cout << 0 << "\n";
    		continue;
    	}
    	s = 1;
    	f = 1000000;
    	while (s <= f)
    	{
    		mid = (s + f) / 2;
    		ans = calc(mid);
    		if (ans >= k && calc(mid-1) < k)
    		 break;
    		if (ans > k)
    		 f = mid - 1;
    		else s = mid + 1;
    	}
    	cout << mid << "\n";
    }
    return 0;
}
