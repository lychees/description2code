#include <iostream>
using namespace std;

int main()
{  int t;
     cin>>t;
     while(t--)
     {int j,m;
     cin>>j>>m;
     int a[j];
     int count=0;
     for(int i=0;i<j;i++)
     {
     cin>>a[i];
     if(a[i]<=0)
     {count++;}
    
     }
     if(m<=count)
     {
     	cout<<"NO \n";
     }
     else
     {
     	cout<<"YES \n";
     }
     
     }
    
   return 0;
}
