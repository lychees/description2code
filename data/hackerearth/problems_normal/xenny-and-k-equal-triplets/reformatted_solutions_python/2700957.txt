t = input()


def gcd(a, b):
    if b == 0:
        return a
    else:
        return gcd(b, a % b)


for tests in range(0, t):
    r = raw_input().split()
    n, x = int(r[0]), r[1]
    data = raw_input().split()
    count = 0
    for i in data:
        if i == x:
            count += 1
    num = count*(count-1)*(count-2)
    den = n*(n-1)*(n-2)
    g = gcd(num, den)
    num = num/g
    den = den/g
    s = str(num)+"/"+str(den)
    print(s)
