#include<bits/stdc++.h>
using namespace std;
typedef long long lld;

#define MAX 100005

lld BIT[MAX][26];
lld n,q;
void update(lld index,lld val,lld pos)
{
   lld x=index;
   for(;x<=n;x+=(x&(-x)))
   {    
      BIT[x][pos]+=val;
   }

}

lld query(lld index,lld pos)
{
   lld x=index;
   lld sum=0;
   for(;x>0;x-=(x&(-x)))
   { 
   	 sum+=BIT[x][pos];
   }
   return sum;
}




int main()
{
     cin>>n>>q;

   string s;
   cin>>s;
   
   lld i,j;
   for(i=1;i<=n;i++)
   {
       update(i,1,s[i-1]-'a');
   }
   
  while(q--)   
  {
     lld type;
     cin>>type;

     if(type==1)
     {  
     	lld x;
          cin>>x;
        char ch;
          cin>>ch;
        
        update(x,1,ch-'a');
        update(x,-1,s[x-1]-'a');
        s[x-1]=ch;

     }	
     else
     {  lld l,r;
     	cin>>l>>r;
     	lld odd=0;
        for(i=0;i<26;i++)
        {   
        	lld total=query(r,i)-query(l-1,i);
               if(total%2)   //---odd
                odd++;
        }	
         if(odd<=1)
         	cout<<"yes\n";
         else
         	cout<<"no\n";

     }	
  }
 


}
