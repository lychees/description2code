noOfTestCases = int(raw_input())


def outPut(string):
    sum = 0
    for j in string:
        if (j.isdigit()):
            sum += int(j)

    string = ''.join([i for i in string if not i.isdigit()])
    tokenstrings = string.split(" ")
    sorteTokenstrings = sorted(tokenstrings)
    outputString = ""
    curLen = 0
    for t in sorteTokenstrings:
        curLen += len(t)
        lenToShow = sum - curLen
        if (lenToShow < 0):
            lenToShow = - lenToShow
        outputString += t + " " + str(lenToShow) + " "
    print outputString


for no in range(1, noOfTestCases + 1):
    outPut(raw_input())
