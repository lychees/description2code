#include<bits/stdc++.h>
using namespace std;
long gcd(long a, long b)
{
    // base case
    if (a == b)
        return a;

    // a is greater
    if (a > b)
        return gcd(a-b, b);
    return gcd(a, b-a);
}
int  main()
{

 long s,q,n;

 cin>>s>>q;
 bool aa[100004]={0};
 while(q--)
 {
     cin>>n;
long ss = gcd(s,n);
if(aa[ss]==0)
{cout<<ss<<endl;
aa[ss]=1;
}
else cout<<"-1"<<endl;


 }

    return 0;
}
