#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	ll i, t, n, m, x, a, s;
	cin>>t;
	while(t--){
		cin>>n>>m>>x;
		//cout<<n<<" "<<m<<" "<<x<<endl;
		s=0;
		for(i=0; i<n; i++){
			cin>>a;
			s += a;
		}

		//cout<<s<<endl;
		a = x*(n+1) - s;
	//	cout<<"a: "<<a<<" "<<"m: "<<m<<endl;
		if(a<=m){
			if(a<=0)
				cout<<"1\n";
			else
				cout<<a<<endl;
		}
		else
			cout<<"Impossible\n";
	}
	return 0;
}
