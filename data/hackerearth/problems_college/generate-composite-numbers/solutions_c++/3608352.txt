#include <iostream>
using namespace std;
int prime[10005];
void sieve(int t)
{
	int i;
	for(i=4;i<=t;i++) prime[i]=1;
	int p=2,j;
	while(p*p<=t)
	{
		j=p*p;
		while(j<=t)
		{
			prime[j]=0;
			j=j+p;
		}
		p++;
	}
	
}
int main()
{
    int t,i;
    scanf("%d",&t);
    sieve(t);
    for(i=4;i<=t;i++)
    {
    	if(prime[i]!=1) printf("%d ",i);
    }
    cout<<endl;
    return 0;
}
