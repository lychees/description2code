#include<iostream>
using namespace std;
class fibs
{
public:
	int one,zero;
	int fibo(int );
	fibs();
	void display();
};
fibs::fibs()
{
	one=zero=0;
}
void fibs::display()
{
cout<<zero<<" "<<one<<endl;
}
int fibs ::fibo(int n)
{
	{
	   if (n == 0)
	{
	   zero++;
	   return 0;
	}
	if (n == 1)
	{
	   one++;
	   return 1;
	}
	return (fibo(n - 1) + fibo(n - 2));
	}

}
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		fibs obj;
		cin>>n;
		obj.fibo(n);
		obj.display();
	}

	return 0;
}
