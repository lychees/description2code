#include <bits/stdc++.h>
using namespace std ;
#define ll long long int 
int main()
{
   ll  t ;
   cin >> t ;
   while ( t--)
   {

      string s ;
      cin >> s ;
      ll l = s.length()  ;
      ll r = 0 ;
      ll  i ;
      for ( i = 0 ; i<=l/2 ; i++ )
      {
        if ( s[i] != s[l-i-1])
        {
          r=  1 ;
          break;
        }

      }
      if ( r == 0 )
      {
       cout << "Palindrome" <<endl;
      }
      else{
      ll  sum = 1;
         for ( i = 0; i< l ; i++)
         {
            int y = (int )(s[i]-'a'+1) ;
            sum = sum*y ;

         }
         cout << sum <<endl;
      }


   }
  return 0 ;

}
