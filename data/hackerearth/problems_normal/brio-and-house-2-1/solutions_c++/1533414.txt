#include<iostream>
#include<math.h>
using namespace std;

void result(double x1,double y1,double x2,double y2,double r){
	double x;
	x=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
	double d=x1*y2-x2*y1;
	double del=r*r*x*x-d*d;
	if(del<0){
		cout<<"SAFE";
	}
	else if(del==0){
		cout<<"JUST MISSED";
	}
	else{
		cout<<"REPLANNING";
	}
}

int main(){
	int T;
	double x1,y1,x2,y2,r;
	cin>>T;
	while(T){
		cin>>x1>>y1>>x2>>y2>>r;
		result(x1,y1,x2,y2,r);
		cout<<endl;
		--T;
	}
return 0;
}
