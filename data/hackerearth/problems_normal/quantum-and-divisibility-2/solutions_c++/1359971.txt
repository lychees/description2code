#include<bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(false);
using namespace std;
map<int, long long> fun(int n)
{
    map<int, long long>r;
    long long c;
    for(int i=2;i<=sqrt(n);i++){
            c=0;
        if(n%i==0){
            while(n%i==0){
                c++;
                n/=i;
            }
        }
        r[i]=c;
    }
    if(n>1){
        r[n]=1;
    }
    return r;
}
int main()
{_
    long long t,a,b,c,d,flag;
    cin>>t;
    while(t--){
            flag=1;
       cin>>a>>b>>c>>d;
       map<int, long long> x=fun(a),y=fun(c);
       map<int ,long long>::iterator p;
       for(p=y.begin();p!=y.end();p++){
          if((p->second)*d>x[p->first]*b){
              flag=0;break;
          }
       }
    if(flag){
        cout<<"Divisible\n";
    }
    else{
        cout<<"Not divisible\n";
    }
  }
    return 0;
}
