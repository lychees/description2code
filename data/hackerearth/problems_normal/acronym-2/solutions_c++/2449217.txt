#include<iostream>
#include<unordered_map>
#include<cstdio>
using namespace std;
int main(void)
{
   // freopen("in.txt","r",stdin);
    int n;
    cin>>n;
    unordered_map<string,int> m;
    for(int i=0;i<n;++i)
    {
        string temp;
        cin>>temp;
        m[temp]++;
    }
    cin>>n;
    string ans;
    while(n--)
    {
        string temp;
        cin>>temp;
        if(m[temp]==0)
        {
            char tem=temp[0];
            if(tem>='a' && tem<='z')
                tem=tem-'a'+'A';
            ans.push_back(tem);
        }
    }
    for(int i=0;i<ans.size()-1;++i)
        cout<<ans[i]<<".";
    cout<<ans[ans.size()-1];
    m.clear();
    return 0;
}
