t = int(raw_input())
for i in range(t):
    d = int(raw_input())
    x = (2*d)**0.5
    x = int(x)
    if (x*(x-1) < 2*d <= x*(x+1)):
        x = x % 3
    elif ((x-2)*(x-1) < 2*d <= x*(x-1)):
        x = (x-1) % 3
    else:
        x = (x+1) % 3
    if (x == 0):
        print("SL")
    elif (x == 1):
        print("LB")
    else:
        print("BS")
