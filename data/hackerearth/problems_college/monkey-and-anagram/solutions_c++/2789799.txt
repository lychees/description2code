#include <bits/stdc++.h>
using namespace std;
int count[255];
int main()
{
	int t1;
	cin>>t1;
	string s,t;
	for(int j=0;j<t1;j++)
	{	int count[255]={ };
		cin>>s>>t;
		int l=s.length();
		int ll1=t.length();
		for(int i=0;i<l;i++)
		{	int ll=t.length();
			for(int k=0;k<ll;k++)
			{
				if(s[i]==t[k])
				{
					t.erase(t.begin()+k);
					k=ll;
				}
			}
		}
		if(t.length()!=ll1-l)
		cout<<"-1\n";
		else
		{
		for(int i=0;i<t.length();i++)
		{
			count[int(t[i])]=1;
		}
		int c=0;
		for(int i=0;i<255;i++)
		{	if(count[i]==1)
			c++;
		}
		cout<<c<<"\n";
	}
	}
}
