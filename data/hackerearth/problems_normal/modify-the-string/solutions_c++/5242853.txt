const double pi = acos(-1.0);

int count_one(int n) {
	int count = 0;
	while(n > 0)
	{
		n = n & (n - 1);
		count++;
	}
	return count;
}
const int maxn = 105;
char s[maxn];

int main()
{
	// printf("%d\n", sizeof(uint32_t));
	// printf("%d\n", count_one(1024));
	
	scanf("%s", s);
	int m = strlen(s);
	for(int i = 0; i < m ; i++)
	{
		if(isupper(s[i])) s[i] = tolower(s[i]);
		else s[i] = toupper(s[i]);
	}
	printf("%s\n", s);
	return 0;
}
