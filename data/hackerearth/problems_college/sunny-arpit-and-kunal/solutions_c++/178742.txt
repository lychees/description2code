#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    
    while(t--)
    {
    	
    	
    	int A,B,C;
    	
    	cin>>A>>B>>C;
    	
    	vector<int> arr;
    	arr.push_back(A);
    	arr.push_back(B);
    	arr.push_back(C);
    	sort(arr.begin(),arr.end());
    	
    	A=arr[0];
    	B=arr[1];
    	C=arr[2];
    	if(A==B||B==C||A==C)
    	{
    		
    		cout<<0<<endl;
    	}
    	else
    	{
    		int ans=max(B-A-1,C-B-1);
    		
    		if(ans>=0)
    		{
    			cout<<ans<<endl;
    		}
    		else
    	    {
    	    	cout<<0<<endl;
    	    }
    	}
    }
    return 0;
}
