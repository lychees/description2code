#include <bits/stdc++.h>
using namespace std;
# define MOD 1000000007
long long power(long long a,long long b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	long long temp=power(a,b/2);
	temp=(temp*temp)%MOD;
	if(b&1)
		temp=(temp*a)%MOD;
	return temp;
}
map<long,int> fmap;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	fmap.clear();
    	long long n;
    	cin>>n;
    	long a;
    	long long count=0;
    	for(int i=0;i<n;i++)
    	{
    		cin>>a;
    		if(!fmap[a])
    		{
    			count++;
    			fmap[a]=1;
    		}
    	}
    	long long ans;
    	ans=power(n,count);
    	cout<<ans<<endl;
    }
    return 0;
}
