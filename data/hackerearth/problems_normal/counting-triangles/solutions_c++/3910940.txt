#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin>>n;
    set<vector<long long int> > s1;
    set<vector<long long int> > s2;
    long long int x,c=n,f=0;
    for(int i=0;i<n;i++){
    	vector<long long int> vi;
    	for(int j=0;j<3;j++){
    		cin>>x;
    		vi.push_back(x);
    	}
    	sort(vi.begin(),vi.end());
    	if(s1.find(vi) == s1.end())
    		s1.insert(vi);
    	else{
    		s2.insert(vi);
    		c--;
    	}
    	vi.clear();
    }
    c -= s2.size();
    cout<<c<<endl;
    return 0;
}
