#include <bits/stdc++.h>
using namespace std;

int main() {
    ios_base::sync_with_stdio(false);

    int T, N, a[100];
    cin >> T;
	
    while (T--){
        cin >> N;
        for (int i = 0; i < N; i++){
            cin >> a[i];
        }
        sort(a, a + N);
        
        int minK = 100000;
        int maxK = 0;
        for (int i = 0, j = N - 1; i < j; i++, j--) {
        	minK = min(minK, a[i] + a[j]);
        	maxK = max(maxK, a[i] + a[j]);
        }
        cout << (maxK - minK) << endl;
    }
    return 0;
}
