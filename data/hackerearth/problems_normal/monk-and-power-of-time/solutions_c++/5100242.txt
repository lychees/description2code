#include <iostream>
#include<queue>
using namespace std;

int main()
{
    int i,j,k,l,m,n;
    scanf("%d",&n);
    queue<int> q1;
    queue<int> q2;
    for(i=0;i<n;i++)
    {
    	scanf("%d",&m);
    	q1.push(m);
    }
    for(j=0;j<n;j++)
    {
    	scanf("%d",&m);
    	q2.push(m);
    }
    int c=0;
    while(!q2.empty())
    {
    	while(q1.front()!=q2.front())
    	{
    		m=q1.front();
    		q1.pop();
    		q1.push(m);
    		c++;
    	}
    	q1.pop();
    	q2.pop();
    	c++;
    }
    printf("%d",c);
    return 0;
}
