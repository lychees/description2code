#include <bits/stdc++.h>

using namespace std;

int main(){
    int t;
    long long a,b;

    std::ios::sync_with_stdio(false);
    cin >> t;
    while(t){
        t--;
        int c=0;
        cin >> a >> b;
        do{
            if(a%10 + b%10 >=10){
                c++;
            }
            a = a/10;
            b = b/10;
        }while(a > 0 || b > 0);
        if(c == 1)
            cout << "1 carry operation" << endl;
        else if (c == 0)
            cout << "No carry operation" << endl;
        else
            cout << c << " carry operations" << endl;
    }
}
