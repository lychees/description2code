#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,j;
    cin>>t;
    long long count;
    while(t--)
    {
    	count=0;
    	cin>>n;
       int *p=(int *)malloc(n*sizeof(int));
       for(i=0;i<n;i++)
       cin>>p[i];
       for(i=0;i<n-1;i++)
       {
       	for(j=i+1;j<n;j++)
       	if(p[i]==p[j])
       	count++;
       }
       count+=n;
       cout<<count<<"\n";
       free(p);
    }
    return 0;
}
