#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,count=0;
		cin>>n;
		int i,j,flag=0;
		if(n<6)
		{
			cout<<"-1"<<endl;
			continue;
		}
		else
		{
		
			for(i=0;i<=n/6;i++)
			{
				if((n-6*i)%8==0)
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			{
				count=i+(n-6*i)/8;
				cout<<count<<endl;
			}
			else
			{
				cout<<"-1"<<endl;
			}
		}
	}
	return 0;
}
