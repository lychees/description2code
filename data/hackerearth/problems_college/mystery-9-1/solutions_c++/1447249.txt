#include <iostream>

using namespace std;

bool ispalin(string s)
{

    auto start = s.begin();
    auto last  = s.end()-1;

    while(start<last)
    {
        if( (*start) != (*last) )
            return false;

        start++ , last--;
    }

    return true;
}

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        string s;
        cin>>s;

        if(!ispalin(s))
            cout<<-1;
        else
            cout<<int(s[s.size()/2]);

        cout<<endl;
    }
    return 0;
}
