#include <stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int a,b,rev1=0,rev2=0,rev=0,sum,d1,d2,d;
    	scanf("%lld %lld",&a,&b);
    	while(a!=0)
		{
			d1=a%10;
			rev1=rev1*10+d1;
			a=a/10;
		}
		while(b!=0)
		{
			d2=b%10;
			rev2=rev2*10+d2;
			b=b/10;
		}
		sum=rev1+rev2;
		while(sum!=0)
		{
			d=sum%10;
			rev=rev*10+d;
			sum=sum/10;
		}
		printf("%lld\n",rev);
    }
    return 0;
}
