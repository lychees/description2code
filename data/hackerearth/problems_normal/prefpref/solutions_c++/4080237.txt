#include <iostream>
using namespace std;

int main()
{
    string s, t;
    cin>>s;
    cin>>t;
    long ls=s.size(), ts=t.size();
    long i, j, k=0, ans=0;
    for(i=0; i<ls; i++){
    	for(j=k; j<ts; j++){
    		if(s[i]==t[j]){
    			//printf("j=%ld\n",j);
    			k=j+1;
    			ans++;
    			break;
    		}
    	}
    	if(k>=ts-1 || j>=ts-1)
    		break;
    }
    printf("%ld", ans);
    return 0;
}
