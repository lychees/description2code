#include<bits/stdc++.h>
using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
    	string str;
    	cin>>str;
    	int l = str.size();
    	vector<char>v;
    	for(int i=0;i<l;i++)
    	{
    		v.push_back(str[i]);
    	}
    	for(int i=0;i<l;i++)
    	{
    		cout<<v.back();
    		v.pop_back();
    	}
    	cout<<endl;
    }
    return 0;
}
