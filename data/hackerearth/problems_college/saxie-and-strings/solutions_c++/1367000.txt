#include<bits/stdc++.h>
using namespace std;
typedef struct fun{
	string str;
};
int main()
{
int t;
cin>>t;
while(t--)
{
	string s;
	cin>>s;
int len=s.size();
int flag=1,i;
for (i=0;i<len;i++) {
		if(s[i]!=s[len-i-1])
		  {   
		  flag=0;
		  break;
		  }
		  
	}
	if(flag)
	cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
}
return 0;}
