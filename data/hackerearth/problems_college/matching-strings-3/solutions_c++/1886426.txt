#include<bits/stdc++.h>
using namespace std;
#define s(a) scanf("%d", &a)
#define ll unsigned long long
int main()
{
	int t,l1,l2,min,i;
	s(t);
	while(t--)
	{
		char s1[1005]="\0", s2[1005]="\0";
		scanf("%s", s1);
		scanf("%s", s2);
		l1=strlen(s1);
		l2=strlen(s2);
		if(l1<l2)
			min=l1;
		else
			min=l2;
		int count = 0;
		for(i=0;i<min;i++)
		if(s1[i]==s2[i])
			count++;
		printf("%d\n", count);
	}
	return 0;
}
