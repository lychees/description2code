int a[105][105];
int b[105][105];
int main(){
	int c,t,n,i,j;
	cin>>t;
	while(t--){
		cin>>n>>c;
		for(i=1;i<=n;i++){
			for(j=1;j<=n;j++)
			cin>>a[i][j];
		}
		b[1][1]=a[1][1];
		for(i=2;i<=n;i++){
		b[i][1]=b[i-1][1]+a[i][1];
		b[1][i]=b[1][i-1]+a[1][i];}
		for(i=2;i<=n;i++){
			for(j=2;j<=n;j++){
				b[i][j]=max(b[i-1][j],b[i][j-1])+a[i][j];
			}
		}
		if(b[n][n]<c)
		cout<<-1<<endl;
		else
		cout<<b[n][n]-c<<endl;
	}
	
	return 0;
}
