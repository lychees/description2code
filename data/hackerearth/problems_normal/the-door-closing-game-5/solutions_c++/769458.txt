#include<iostream>

using namespace std;


int main(){
	
	
	int t;
	cin>>t;
	while(t--){
	
	int c = 0;
	string s;
	cin>>s;	
	bool fl = 0;
	int ans = 0;
	
		for(int i=0;i<s.size();i++){
			int tt = (s[i]=='|'?0:1);
		
			if(tt==0){  ans=ans^c; c=0; fl=1;}
			else if(tt==1){
				c++;
			} 
		}
		
		if(fl){
			ans = ans^c;
		}
		
		if(ans) cout<<"Alice\n";
		else cout<<"Bob\n";
		
	}
	return 0;
}
