#include <iostream>
#include <bits/stdc++.h>
using namespace std;
#define max 1000005
int v[max];
void sieve()
{
	int i,j;
	v[0]=1;
	v[1]=1;
	for(i=2;i*i<max;i++)
	{
		if(!v[i])
		{
			for(j=i*i;j<max;j+=i)
			{
				v[j]=1;
			}
		}
	}
}
int check(int a,int b,int k,int m)
{
	int i,j,c=0;
	for(i=a;i<a+m;i++)
	{
		c+=!v[i];	
	}
	if(c<k)
	{
		return 0;
	}
	for(j=a;i<=b;i++,j++)
	{
		c-=!v[j];
		c+=!v[i];
		if(c<k)
		{
			return 0;
		}
	}
	return 1;
}
int  b_search(int a,int b,int k)
{
	int l,m,r;
	l=1;
	r=b-a+1;
	while(r-l>1)
	{
		m=(l+r)/2;
		if(check(a,b,k,m))
		{
			r=m;
		}
		else
		{
			l=m+1;
		}
	}
	if(check(a,b,k,l))
	{
		return l;
	}
	if(check(a,b,k,r))
	{
		return r;
	}
	return -1;
}
int main()
{
	int t,a,b,k;
	sieve();
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>k;
		cout<<b_search(a,b,k)<<endl;
	}
   	return 0;
}
