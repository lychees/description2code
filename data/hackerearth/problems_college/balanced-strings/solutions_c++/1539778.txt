#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		char s[500]; cin>>s;
		int l=strlen(s);
		int a[27]={0};
		for(int i=0; i<l; i++) a[s[i]-'a']++;
		int val=a[0]; int f=1;
		for(int i=1; i<26; i++){
			if(a[i]!=val){
				f=0;
				break;
			}
		}
		if(f) cout<<"Yes\n";
		else cout<<"No\n";
	}
}
