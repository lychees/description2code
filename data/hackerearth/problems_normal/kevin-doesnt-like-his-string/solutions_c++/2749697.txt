template <class T> inline void Min(T &a,T b){if(a>b)a=b;}

char s[N];
set<int>v[26],q[26];
set<int>::iterator it;
int use[N];
int main(){
    int T,i,j,k,m,n,K;
    scanf("%d%d%s",&n,&K,s);
    if(K==1){
        puts(s);return 0;
    }
    for(i=0;i<n;i++)v[s[i]-'a'].insert(i);
    m=K;
    for(i=0;i<n-1&&m>=0;i++){
        if(!use[i])v[s[i]-'a'].erase(i);
        for(j=0;j<s[i]-'a';j++){
            if(q[j].size()){
                it=q[j].end();it--;
                if(*it>i){
                    int x=*it;q[j].erase(x);
                    if(use[i]){    
                        q[s[i]-'a'].erase(i);
                        q[s[i]-'a'].insert(x);
                        swap(s[i],s[x]);
                        break;
                    }
                    else if(m>=1){
                        m--;
                        q[s[i]-'a'].insert(x);
                        swap(s[i],s[x]);
                        break;
                    }
                }
            }
            if(v[j].size()==0)continue;
            it=v[j].end();it--;
            int x=*it;
            if(x<=i)continue;
            if(use[i]&&m>=1){
                v[j].erase(x);q[s[i]-'a'].erase(i);q[s[i]-'a'].insert(x);
                swap(s[i],s[x]);use[x]=1;
                m--;break;
            }
            else if(!use[i]&&m>=2){
                v[j].erase(x);q[s[i]-'a'].insert(x);
                swap(s[i],s[x]);use[x]=1;
                m-=2;break;
            }
        }
        if(m==0){
            vector<char>v;vi vv;
            for(j=i+1;j<n;j++)if(use[j]){
                v.pb(s[j]);vv.pb(j);
            }
            sort(all(v));
            for(j=0;j<v.size();j++)s[vv[j]]=v[j];
                break;
        }
        //puts(s);
    }
    puts(s);
    return 0;
}
