#include <iostream>
#include <iomanip>
using namespace std;

int main()
{
    int N,P,T,i,v;
    
    cin>>T;
    
    cout.precision(8);
    
    for(;T;T--)
    {
    	cin>>N;
    	v=0;
    	for(i=1;i<=N;i++)
    	{
    		cin>>P;
    		v+=P;
    	}
    	cout<<fixed<<double(v)/double(N)<<endl;
    }
    
    return 0;
}
