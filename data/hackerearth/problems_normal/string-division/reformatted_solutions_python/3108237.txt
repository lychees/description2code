import itertools as it


def split(str, split):
    a, b, c = split
    s1 = str[:a]
    s2 = str[a:b]
    s3 = str[b:c]
    s4 = str[c:]
    return s1, s2, s3, s4


def unique(s):
    for i in range(4):
        for j in range(4):
            if i != j:
                if s[i] == s[j]:
                    return False
    return True


def isSplit(str):
    if len(str) >= 10:
        return True
    combs = it.combinations(range(1, len(str)), 3)
    for i in combs:
        if unique(split(str, i)):
            return True
    return False


for i in range(input()):
    str = raw_input()
    if isSplit(str):
        print 'YES'
    else:
        print 'NO'
