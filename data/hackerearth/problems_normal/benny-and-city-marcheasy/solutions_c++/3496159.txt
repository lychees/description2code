int a[22][22];

int main(){
  ios_base::sync_with_stdio(false);
  cout<<fixed<<setprecision(0);
  int n,m,c;
  cin>>n>>m>>c;
  rep(i,n) rep(j,m) cin>>a[i][j];
  ll re=1e18;
  rep(s,1<<m){
    ll sum=1ll*c*__builtin_popcount(s);
    rep(i,n){
      int tmp=0;
      rep(j,m) if(s>>j&1){
	sum+=a[i][j];
      }else{
	tmp+=a[i][j];
      }
      sum+=min(0,tmp+c);
    }
    re=min(re,sum);
  }
  re*=-1;
  //cout<<re<<endl;
  rep(i,n) rep(j,m) re+=a[i][j];
  cout<<re<<endl;
  return 0;
