#include<iostream>
using namespace std;
int main()
{
       int n , d ;
       cin >> n >> d;
         int a[n+1];
         for(int i = 0 ; i < n ; ++i )
             cin >> a[i];
           int ans = 0;
         for(int i = 1 ; i < n ; ++i )
         {
             if(a[i] < a[i-1] )
                {
                    int tmp = d;
                 int val= 1;
                  while(true)
                  {
                      if(tmp + a[i] > a[i-1])
                         break;
                         else
                            val++;
                        tmp+=d;

                  }
                  ans = ans + val;
                  a[i] = tmp+a[i];
             }
         }
         cout << ans << endl;
}
