#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    int t;cin>>t;
    while(t--)
    {
    	int l,r;cin>>l>>r;
    	int c_xor=0,max_xor=0;
    	for(int i=l;i<r;i++)
    	{
    	for(int j=l+1;j<=r;j++)
    	{
    		c_xor=i^j;
    		if(c_xor>max_xor)
    		{
    			max_xor=c_xor;
    		}
    	}
    	}
    	cout<<max_xor<<endl;
    }
    return 0;
}
