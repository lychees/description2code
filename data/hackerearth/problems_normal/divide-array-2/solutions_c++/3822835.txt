#include <bits/stdc++.h>
using namespace std;

int main()
{
    int q,n,i,j;
    double d2;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++)
    {scanf("%d",&a[i]);}
    cin>>q;
    int d[q];
    d2=1;
    for(i=0;i<q;i++)
    {scanf("%d",&d[i]);
    	d2*=d[i];
    }
   
    
    for(i=0;i<n;i++)
    {a[i]/=d2;
    
    printf("%d ",a[i]);}
    
    return 0;
}
