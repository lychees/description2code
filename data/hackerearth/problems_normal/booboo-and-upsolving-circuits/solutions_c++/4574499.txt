const int N = 300000;
const int INF = (int)1e7;
const double eps = 1e-6;
const long long MOD = 1e9 + 7;
const double PI = M_PI;


int n,m;
ll a[N];
ll max_time;
ll ans = 1e18;

bool check(ll t){
	ll s = 0;
	int cnt = 1;
	for(int i = 0; i < n; ++i){
		if(s + a[i] > t){
			cnt++;
			s = a[i];
		}
		else{
			s += a[i];
		}
	}
	return cnt <= m;
}

void solve(){
	cin >> n >> m;
	for(int i = 0; i < n; ++i){
		cin >> a[i];
		max_time = max(max_time,a[i]);
	}
	ll l = max_time - 1; ll r = 2e17;
	while(r - l > 1){
		ll mid = (l + r) >> 1;
		if(check(mid)){
			r = mid;
		}
		else{
			l = mid;
		}
	}
	if(check(r)){
		cout << r;
	}	
}

int main(){
	std::ios::sync_with_stdio(false);
	//freopen("input.in","r",stdin);

	int t = 1;
	//scanf("%d", &t);
	while(t--){
		solve();
	}	
	return 0;
}
