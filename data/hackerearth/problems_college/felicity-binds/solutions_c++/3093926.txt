#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;
class MyData
{
public:
  int m_iData;
  string m_strSomeOtherData;
  bool operator<(const MyData &rhs) const { return m_iData < rhs.m_iData; }
};
int main()
{
	int t,n,j,i,x;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	vector<int> men;
    	vector<int> wmen;
    	for(i=0;i<2;i++)
    		for(j=0;j<n;j++)
    		{
    			if(i==0)
    			{
    			cin>>x;
    			men.push_back(x);
    			}
    			else
    			{
    			cin>>x;
    			wmen.push_back(x);
    			}
    		}
    	sort(men.begin(), men.end());
    	sort(wmen.begin(), wmen.end());
    	//sum
    	int sum=0;
    	for(i=0;i<n;i++)
    	{
    		sum = sum+(men[i]*wmen[i]);
    	}
    	cout<<sum<<endl;
    }
    
    return 0;
}
