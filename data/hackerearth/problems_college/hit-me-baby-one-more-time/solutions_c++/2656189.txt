#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    long int n,sum=0,q;
    cin>>n;
    n--;
    q=n/3;
    sum=sum+3*((q*(q+1))/2);
    //cout<<sum;
    q=n/5;
    sum=sum+5*((q*(q+1))/2);
    //cout<<sum;
    q=n/15;
    sum=sum-15*((q*(q+1))/2);
    cout<<sum<<endl;
    return 0;
}
