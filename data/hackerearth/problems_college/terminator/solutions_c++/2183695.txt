#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int start,p,t;//start & pending
		p=0;
		t=0;
		string s,r;
		cin>>s;
		r="";
		int i,j,l=s.length();
		for(i=0;i<l;i++)
		{
			if(s.at(i)=='*')//Life saver
			{
				t=0;
				if(p)
				{
					for(j=start;j<i;j++)
					r+=s.at(j);
				}
				r+=s.at(i);
				p=0;
			}
			else if(s.at(i)=='x')//Terminator
			{
				t=1;
				r+=s.at(i);
				p=0;
			}
			else
			{
				if(t)
				continue;
				if(p)
				continue;
				else
				{
					p=1;
					start=i;
				}
			}
		}
		if(p)
		{
			for(j=start;j<i;j++)
			r+=s.at(j);
		}
		cout<<r<<endl;
	}
	return 0;
}
