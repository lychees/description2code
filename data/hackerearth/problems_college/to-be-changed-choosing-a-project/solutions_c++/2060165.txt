#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,r;
    cin>>n>>r;
    vector<int> v1(n);
    for(int i=0;i<n;i++)
        v1[i]=i+1;

    int p,cp=0;
    while(n>1)
    {
        p=(r-1)%n;
        cp=(cp+p)%n;
       // cout<<v1[cp]<<" ";

        v1.erase(v1.begin()+cp);
        n--;
    }
    cout<<v1[0]<<endl;
}
