#include <iostream>
using namespace std;
int main()
 {
   int t; cin>>t;
    while(t--)
   {
     int x; cin>>x;
     int i=x/3;
     i=i*3; int flag=0;
     for(i=i;i>=0;i=i-3)
      {
         if((x-i)%5==0)
          {
            flag=1; break;
          }
      }
     if(flag)
      {
         for(int k=0;k<i;k++)
          cout<<"5";
         for(int k=0;k<x-i;k++)
         cout<<"3";
         cout<<endl;
      }
     else 
         cout<<"-1\n";
   }
return 0;
}
