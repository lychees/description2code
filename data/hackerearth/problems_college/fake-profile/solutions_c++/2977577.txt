#include <iostream>
#include <vector>
using namespace std;

int main()
{
    string s;
    vector<int> alph;
    int res = 0;
    alph.assign(26, 0);
    
    cin >> s;
    for (int i=0; i<s.size(); i++)
    {
    	int ind = (int)(s[i]-'a');
    	alph[ind]++;
    }
    for (int i=0; i<26; i++)
    	if (alph[i]>0)
    		res++;
    		
    if (res%2==0)
    	cout << "CHAT WITH HER!";
    else
    	cout << "IGNORE HIM!";
    return 0;
}
