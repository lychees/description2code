#include<bits/stdc++.h>
using namespace std;
int n,cnt;
string str;
struct TrieNode
{
	bool leaf=false;
	unordered_map<char,TrieNode *> child;
};

void addWord(TrieNode *root,string str,int pos)
{
	if(pos==str.length())
	{
		root->leaf=true;
	}
	else
	{
		if(root->child[str[pos]]==NULL)
		{
			cnt++;
			root->child[str[pos]]=new TrieNode();
		}
		addWord(root->child[str[pos]],str,pos+1);
	}
}

int main(void)
{
	TrieNode *root=new TrieNode();
	cin >> n;
	cnt=0;
	while(n--)
	{
		cin >> str;
		addWord(root,str,0);
	}
	cout << cnt+1 << "\n";
	return 0;
}
