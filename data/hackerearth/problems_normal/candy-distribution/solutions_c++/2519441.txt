#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll hi = 1000000007;
ll po(ll a,ll b)
{
	if(b==0)return 1;
	if(a==1)return 1;
	ll x = po(a,b/2);
	x = (x*x)%hi;
	if(b%2)x = (x*a)%hi;
	return x;
}
ll w[1000005];
ll get(ll mm)
{
	if(mm<=1000000)return w[mm-1];
	mm--;

		vector<ll> fa;
 	fa.push_back(1);

 	ll ta = mm;
 		int i;

 	ll nn;
 	for(nn=2;nn*nn<=mm;nn++)
 	if(mm%nn==0)
 	{

 		int l = fa.size();
 		ll cur = 1;
 		while(mm%nn==0)
 		{
 			mm/=nn;
 			cur*=nn;

 			for(i=0;i<l;i++)
 			fa.push_back(cur*fa[i]);
 		}
 	}
 

 	int l = fa.size();
 	if(mm!=1)
 	for(i=0;i<l;i++)fa.push_back(mm*fa[i]);
 	sort(fa.begin(),fa.end());
 	ll ans = 0;
 	 l = fa.size();
 //	printf("factor of %lld are \n",mm);
 	for(i=0;i<l-1;i++)
 	{
 	//	printf("%lld ",fa[i]);
 		ll sss = get(fa[i]);
 		ll pp = ta/fa[i];
 		if(pp>1)
 		ans += po(sss,pp);
 		ans%=hi;
 	}
 	return ans;
}
