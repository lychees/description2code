import math
def findlen(n):
	i=0
	while n>=math.pow(10,i):
		i+=1
	return i


t=input()
while t>0:
	t-=1
	l=input()
	length=findlen(l)
	#print length
	if length==1:
		if l%2==0:
			print "EVEN"
		else:
			print "ODD"
			
	else:
		length-=1
		f=math.floor(l/math.pow(10,length))
		if f%2==0:
			print "EVEN"
		else:
			print "ODD"
