#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,D,P;
	cin>>t;
	while(t--)
	{
		cin>>D>>P;
	long long int i,j,x,x1;
		x=pow(D,2)+4*P;
		x1=sqrt(x);
		if(x1*x1!=x)
		 cout<<"0\n";
		 else 
		 {
		 	i=(x1+D)/2;
		    j=x1-i;
		    if(i!=j)
		     cout<<"4\n";
		     else cout<<"2\n";
		 }
	}
}
