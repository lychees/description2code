#include<iostream>
#include<cstring>
using namespace std;
int main()
{
          long long int t;
          cin>>t;
          while(t--)
          {
          string str;
          cin>>str;
          long long int l=str.length();
          long long int i,r=0;
          for(i=1;i<l;i++)
          {
                    if(str[i]==str[i-1])
                              r++;
          }
          cout<<r<<endl;
          }
          
          return 0;
}
