#include <iostream>
#include <cstdio>
using namespace std;
#define MOD 1000000007
#define ll long long int
int N, L, R;
ll d[105];
ll fact[105];
ll ifact[105];
ll pow_mod(ll a,ll n)
{
	ll res = 1;
	while(n)
	{
		if(n&1)
		{
			res *= a;
			res %= MOD;
		}
		a *= a;
		a %= MOD;
		n = n>>1;
	}
	return res;
}
void comp()
{
	d[0] = 1;
	d[1] = 0;
	for(int i = 2;i<=N;i++)
	{
		d[i] = ((i-1)*(d[i-1] + d[i-2]))%MOD;
	}
}
int main()
{
    scanf("%d %d %d", &N, &L, &R);
    comp();
    ll ans = 0;
    ll C[N+5][N+5];
    for(int i = 0;i<=N;i++)
    {
    	for(int j = 0;j<=i;j++)
    	{
    		if(i == j || j == 0)
    		{
    			C[i][j] = 1;
    		}
    		else
    		{
    			C[i][j] = (C[i-1][j] + C[i-1][j-1])%MOD;
    		}
    	}
    }
    ll temp;
    ll q;
    for(int i = 0;i<=N;i++)
    {
    	temp = (d[i]*C[N][i])%MOD;
    	if(i>= L && i<=R)
    	{
    	 
    	  ans = (ans+temp)%MOD;
        }
        q += temp;
        q %= MOD;
    }
    ans = (ans*pow_mod(q, MOD-2))%MOD;
    printf("%lld", ans);
    return 0;
}
