#include <iostream>
#include <math.h>
using namespace std;
int a[2000][2000];
int list[2000];
int main()
{
    int N;
    cin>>N;
    int len = pow(2,N)-1;
    int jj=1;
    for(int i=2;i<(2+len);i++)
    {
    	for(int j=1;j<=jj;j++)
    	{
    		cin>>a[i][j];
    	}
    	++jj;
    }
    for(int i=1;i<=len+1;i++)
    {
    	list[i]=i;
    }
    int tot = len+1;
    for(int i=0;i<N;i++)
    {
    	tot = pow(2,N-i);
    	int jj=1;
    	for(int j=1;j<=tot;j+=2)
    	{
    		if(a[list[j+1]][list[j]] == 1)
    		{
    			list[jj]=list[j+1];
    		}
    		else
    		{
    			list[jj]=list[j];
    		}
    		++jj;
    	}
    }
    cout<<list[1]<<endl;
    return 0;
}
