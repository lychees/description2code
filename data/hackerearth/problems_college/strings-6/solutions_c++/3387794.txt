#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
int main() 
{
    char A[1000002],B[1000002];
    long long t,cot,i,j,lena,lenb;
    cin>>t;
    getchar();
    while(t--)
    {
        cin>>A>>B;
        cot=1;
        lena=strlen(A);
        lenb=strlen(B);
        int hash[257]={0};
        for(i=0;i<lena;i++)
        {
            hash[A[i]]++;
        }
        for(i=0;i<lenb;i++)
        {
           cot=((cot%M)*(hash[B[i]]%M))%M;
           hash[B[i]]--;
        }
        cout<<cot<<endl;
    }
	return 0;
}
