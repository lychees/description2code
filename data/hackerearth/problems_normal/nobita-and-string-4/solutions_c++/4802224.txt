#include <bits/stdc++.h>
using namespace std;
int main() {
		string s;
	getline(cin,s);

	while(getline(cin,s))
	{
		vector<string> v;
		string temp;
		for(int j=0;j<s.size();j++)
		{
			if(s[j]==' ')
			 {
			 	v.push_back(temp);
			 	temp.clear();
			 }
			 else
			 {
			 	temp+=s[j];
			 }
		}
		v.push_back(temp);
		string ans;
		int start=0,end=v.size()-1;
		while(start<end)
		{
			string t=v[start];
			v[start]=v[end];
			v[end]=t;
			start++;
			end--;
		}
		for(int k=0;k<v.size();k++)
		{
			cout<<v[k]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
