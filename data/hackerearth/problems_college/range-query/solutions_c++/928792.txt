#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<cmath>
#include<set>
using namespace std;
typedef long long ll;
int main()
{
    cin.tie(NULL);
    ios::sync_with_stdio(false);
    ll n,i,j,m,T,x,y,val,ans;
    cin>>n;
    ll Arr[n][101];
    for(i=0;i<n;i++)
        for(j=0;j<101;j++)
            Arr[i][j]=0;
    ll A[n];
    for(i=0;i<n;i++)
        cin>>A[i];
    Arr[0][A[0]-1]=1;
    for(i=1;i<n;i++)
    {
       for(j=0;j<101;j++)
            Arr[i][j]=Arr[i-1][j];
       Arr[i][A[i]-1]++;
    }
    cin>>T;
    while(T--)
    {
        cin>>x>>y>>val;
        if(x==0)
            ans=Arr[y][val-1];
        else ans=Arr[y][val-1]-Arr[x-1][val-1];
        cout<<ans<<endl;
    }
    return 0;
}
