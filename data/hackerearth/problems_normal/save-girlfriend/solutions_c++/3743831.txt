#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
vector<ll> v[100005];
bool isprime[100005];
unordered_map<ll,ll> mp;
vector<ll> prime;
void sieve()
{
    for(int i=2;i<10005;i++)
        isprime[i]=true;

   for(int i=2;i<10005;i++)
   {
       if(isprime[i])
       {
           for(int j=i+i;j<10005;j+=i)
            isprime[j]=false;
       }
   }
   for(int i=2;i<10005;i++)
   {
       if(isprime[i])
        prime.push_back(i);
   }

}
int main()
{
    sieve();

    int t;
    ll a,b;
    ll maxi;


    cin>>t;

    while(t--)
    {
        maxi=0;
        cin>>a>>b;
        for(int i=0;(i<prime.size() && prime[i]<=b);i++)
        {
            ll k = ceil(double(a)/double(prime[i]));
            k = prime[i]*k;


            ll init = k-a;


            for(int j=init;j<b-a+1;j+=prime[i])
            {
               if(j+a <10001 && isprime[j+a])
                    continue;
                    {

                v[j].push_back(prime[i]);

                    }
            }

        }
        for(int i=0;i<b-a+1;i++)
        {
            ll div=1;

            ll x=a+i;
            ll temp=x;

       //  cout<<"Size"<<v[i].size()<<"\n";
            for(int j=0;j<v[i].size();j++)
            {



                ll tmp=0;
               while(x!=1 && x%v[i][j]==0)
               {
                   x/=v[i][j];
                   tmp++;

               }
               div=(div*(tmp+1))%mod;
            }

            if(div==1 &&temp!=1)
                div=2;
            //cout<<div<<"\n";

            mp[div]++;
            if(div>maxi)
                maxi=div;
        }
        cout<<mp[maxi]+1<<"\n";
        mp.clear();
        for(int i=0;i<b-a+1;i++)
            v[i].clear();

    }
}
