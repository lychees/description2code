#include <iostream>

using namespace std;
#define s(a) scanf("%d\n",&(a))


int main()
{
    //freopen("inp.txt","r",stdin);
    int t;
    s(t);
    char str[1000001];
    int prev;
    long long int count=0;
    while(t--)
    {
        count=0;
        gets(str);//puts(str);
        int len=0;
        for(len=0;str[len]!='\0';len++);
        prev=-1;
        if(str[len-1]=='a' || str[len-1]=='z')prev=len-1;
        for(int i=len-1;i>=0;i--)
        {
            if(str[i]=='a' || str[i]=='z'){prev=i;}
            if(prev!=-1)count+=(len-prev);
        }
        cout<<count<<endl;
    }
    return 0;
}
