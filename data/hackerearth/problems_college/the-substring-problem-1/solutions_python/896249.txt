from string import ascii_lowercase
t = input()
while t:
    n = input()
    a = ['']*100
    for i in xrange(n):
        a[i] = raw_input()
    ans = [0]*26
    x = 0
    for i in ascii_lowercase:
        for j in a:
            if i in j: ans[x]+=1
        x+=1
    final,c = 0,'a'
    for i in xrange(26):
        if ans[i]>final:
            final = ans[i]
            c = ascii_lowercase[i]
    print c
    t -= 1
