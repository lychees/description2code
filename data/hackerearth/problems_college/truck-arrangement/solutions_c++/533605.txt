int num[10001]={0};


int main(int argc,char* argv[])
{
	int T,N,i;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		int arr[N],count=0,k=0;
		for(i=0;i<10001;i++)
		{
			num[i]=0;
		}
		vector<int> vec[10001];
		for(i=0;i<N;i++)
		{
			scanf("%d",&arr[i]);
			num[arr[i]]++;
			vec[arr[i]].push_back(i);
		}
		int temp;
		for(i=0;i<10001;i++)
		{
			if(num[i]!=0)
			{
						for(std::vector<int>::iterator it=vec[i].begin();it!=vec[i].end();it++)
						{
							temp=*it-k;
							if(temp<0)
							{
								temp=-1*temp;
							}
							if(temp>count)
							{
								count=temp;
							}
							k++;
						}
				}
			}
			printf("%d\n",count);
		}

}
