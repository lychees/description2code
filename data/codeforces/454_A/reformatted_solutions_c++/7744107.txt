#include <cmath>
#include <iostream>
using namespace std;
int main() {
  int n, i, j, k, x;
  cin >> n;
  x = (n / 2) + 1;
  for (i = 1; i <= n; i++) {
    for (j = 1; j <= n; j++) {
      if (abs(i - x) + abs(j - x) <= (n / 2))
        cout << "D";
      else
        cout << "*";
    }
    cout << endl;
  }
}
