N = int(raw_input())
classroom = {}
for x in range(N):
    inp = raw_input().split()
    if int(inp[1]) in classroom:
        classroom[int(inp[1])].append(inp[0])
    else:
        classroom[int(inp[1])] = []
        classroom[int(inp[1])].append(inp[0])
marks = sorted(classroom.keys(), reverse=True)
for num in marks:
    names = sorted(classroom[num])
    for val in names:
        print val + " " + str(num)
