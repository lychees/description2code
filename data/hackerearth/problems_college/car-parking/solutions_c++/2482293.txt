#include<iostream>
#include<cstdio>
#define ll long long
bool arr[1000000];
using namespace std;
int main()
{
	int i,count=0;
	ll t;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		count=0;
		int len=s.length();
		for(i=0;i<len;i++)
		arr[i]=true;
		for(i=0;i<len;i++)
		{
			if(s[i]=='D')
			arr[i]=false;
			else if(s[i]=='B')
			{
				arr[i]=false;
				if(i>0)
				arr[i-1]=false;
				if(i>1)
				arr[i-2]=false;
			}
			else if(s[i]=='S')
			{
				arr[i]=false;
				if(i>0)
				arr[i-1]=false;
				if(i<len-1)
				arr[i+1]=false;
			}
		}
		for(i=0;i<len;i++)
		{
			if(arr[i]==true)
			count++;
		}
		cout<<count<<"\n";
	}
return 0;
}
