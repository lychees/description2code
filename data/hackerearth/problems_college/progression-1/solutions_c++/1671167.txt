#include<iostream>

using namespace std;


int main()
{
	int t, n,flag;
	cin >> t;
	while (t--)
	{
		flag = 0;
		cin >> n;
		
		for (int i = 0;i<=500; i++)
		{
			for (int j =0;j<=500; j++)
			{
				if (6 * i + 45 * j == n)
				{
					flag = 1;
					break;
				}
				
			}
			if (flag)
				break;
		}
		if(n<45)
		flag=0;
		if (flag)
			cout << "CORRECT\n";
		else
			cout << "INCORRECT\n";
	}

	return 0;
}
