#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,m,i,j,q,start,end,mid,flag;
    long long int x;
    cin>>n>>m;
    long int arr[n][m];
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
        cin>>arr[i][j];

     cin>>q;
     while(q--)
     {
         cin>>x;
         flag=0;
         for(i=0;i<n;i++)
         {
             if(x>=arr[i][0] && x<=arr[i][m-1])
             {
                 start=0;end=m-1;
                 while(start<=end)
                 {
                     mid=(start+end)/2;
                     if(arr[i][mid]==x)
                     {
                         cout<<i<<" "<<mid<<endl;
                         flag=1;
                         break;
                     }
                     else if(arr[i][mid]>x)
                     {
                         end=mid-1;
                     }
                     else if(arr[i][mid]<x)
                     {
                         start=mid+1;
                     }
                 }
             }

             if(flag==1)
                break;
         }

         if(flag==0)
            cout<<"-1 -1\n";
     }

     return(0);
}
