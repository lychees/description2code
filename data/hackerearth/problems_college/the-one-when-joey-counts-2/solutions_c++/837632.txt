#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t;
	cin>>t;
	int p=0;
	while(p<t)
	{
		long int n,k;
		int i=0;
		cin>>n;
		k=n;
		int a[100];
		while(k!=0)
		{
			long int r=k%3;
			k=k/3;
			if(r==0)
			{
				a[i]=4;
				k=k-1;
			}
			else if(r==1)
			a[i]=2;
			else
			a[i]=3;
			i++;
		}
		for(int k=i-1;k>=0;k--)
		cout<<a[k];
		cout<<endl;
		p++;
	}
	return 0;
}
