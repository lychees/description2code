#include <bits/stdc++.h>
using namespace std;

	
	int main()
	{
	    int t;
	    scanf("%d",&t);
	    while(t--)
	    {
	    	int n;
	    	scanf("%d",&n);
	    	int a[n],i,j;
	    	for(i=0;i<n;i++)
	    	{
	    		scanf("%d",&a[i]);
	    	}
	    	sort(a,a+n);
	    	int c=0;
	    	for(i=0;i<n-1;i++)
	    	{
	    		for(j=i+1;j<n;j++)
	    		{
	    			int k=a[i]+a[j];
	    			if(a[i]!=a[j] && k%2==0)
	    			{
	    				c++;
	    			}
	    		}
	    	}
	    	printf("%d\n",c);
	    }
	    return 0;
	}
