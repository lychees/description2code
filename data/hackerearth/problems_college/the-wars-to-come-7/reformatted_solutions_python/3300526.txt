N = int(raw_input())


def getint(list):
    templist = []
    for i in range(len(list)):
        templist = templist+[int(list[i])]
    return templist


if (N >= 1 and N <= 1000):
    temp = raw_input("").split()
    temp = getint(temp)
    mini = int(raw_input())
    temp.sort()
    sum = 0
    count = 0
    for i in range(len(temp)-1, 0, -1):
        sum += temp[i]
        count += 1
        if (sum >= mini):
            print(count)
            break
