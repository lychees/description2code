#include <iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    	int i,key;
    	cin>>key;
    	key=key%26;
    	string str;
    	cin>>str;
    	for(i=0;str[i];i++)
    	{
    		if(str[i]>='a'&&str[i]<='z')
    		str[i]= 'a'+ char  ((str[i]-'a'+key)%26) ;

    		else if(str[i]>='A'&&str[i]<='Z')
    		str[i]= 'A'+ char((str[i]-'A'+key)%26);

    	}
    	cout<<str<<endl;


    }
    return 0;
}
