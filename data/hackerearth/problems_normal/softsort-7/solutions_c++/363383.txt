#include<cstdio>
#define MOD 1000000007
#define ll unsigned long long
using namespace std;
ll f[1000001];
void fact(){
	f[0]=0;
	f[1]=1;f[2]=2;f[3]=6;
	for(int i=4;i<=1000000;i++){
		f[i]=(f[i-1]*i)%MOD;
	}
}

int main(){
	fact();
	int t;scanf("%d",&t);
	while(t--){
		int n;scanf("%lld",&n);
		printf("%lld\n",((3*f[n]%MOD)+3)%MOD);
	}
	return 0;
}
