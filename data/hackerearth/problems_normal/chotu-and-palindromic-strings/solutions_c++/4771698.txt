#include <iostream>
#include <stack>

using namespace std;

bool pal(string s) {
    stack<char> st;
    int i=0;
    while (i<s.length()) {
        st.push(s[i++]);
    }
    string c;
    while (!st.empty()) {
        c+=st.top();
        st.pop();
    }
    if (s==c)
        return true;
    else
        return false;
}

int main()
{
    int t;
    cin>>t;
    while (t--) {
        string s1, s2;
        cin>>s1>>s2;
        string s3 = s1+s2;
        string s4 = s2+s1;
        bool ans = pal(s3)||pal(s4);
        (ans==1)?cout<<"YES":cout<<"NO";
        cout<<endl;
   
	}    
	return 0;
}
