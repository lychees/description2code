#include <iostream>
#include <stdlib.h>
using namespace std;

int main()
{
    int n,k;
    cin>>n>>k;
    int arr[n+1];
    for(int i=0;i<n;i++)
    {
    	cin>>arr[i];
    }
    
    int count=0;
    for(int i=0;i+k<n ;i++)
    {
    	
    	long long a=abs(arr[i]+arr[i+k]);
    	long long b=abs(arr[i]-arr[i+k]);
    	if((!(a%k)) || (!(b%k)))
    		count++;
    		
    }
    
    cout<<count;
}
