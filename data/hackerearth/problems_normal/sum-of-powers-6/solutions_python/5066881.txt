from sys import stdin
s = set()
s.add(1)
for i in xrange(2,1001):
	j = 2
	while True:
		if i**j > 10**6:
			break
		s.add(i**j)
		j+=1
s = list(s)
s.sort()
dp = [0]*(10**6 + 5)
dp[0] = 1
for i in s:
	for j in s:
		if i + j > 10**6:
			break
		dp[i+j] = 1
t = int(stdin.readline())
for _ in xrange(t):
	n = int(stdin.readline())
	if dp[n]:
		print 'Yes'
	else:
		print 'No'
