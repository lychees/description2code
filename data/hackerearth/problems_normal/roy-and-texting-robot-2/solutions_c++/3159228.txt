#include<bits/stdc++.h>
using namespace std;
map<char,int>m;
map<char,int>m1;
int main()
{
	int t,i,j;
	scanf("%d",&t);
	m1['.']=m1[',']=m1['?']=m1['!']=m1['1']=1;
	m1['_']=m1['0']=0;
	m['_']=1;m['0']=2;
	m['.']=1;m[',']=2;m['?']=3;m['!']=4;m['1']=5;
	char c='a';
	for(i=2;i<=9;i++)
	{
		if(i==7||i==9)
		{
			for(j=1;j<=4;j++){
				m[c]=j;m1[c++]=i;}
			m[i+'0']=5;m1[i+'0']=i;
		}
		else
		{
			for(j=1;j<=3;j++){
				m[c]=j;m1[c++]=i;}
			m[i+'0']=4;m1[i+'0']=i;
		}
	}
	while(t--)
	{
		char str[1005];
		scanf("%s",str);int ini=1,ans=0;
		for(i=0;i<strlen(str);i++)
		{
			if(m1[str[i]]==ini)
				ans+=m[str[i]];
			else{
				ans+=m[str[i]]+1;ini=m1[str[i]];}
		}
		cout<<ans<<"\n";
	}
}
