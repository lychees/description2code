#include <iostream>
#include <vector>
#define ll long long
using namespace std;

int main()
{
	ll a,b;
	vector <ll> vec;
	
    cin >> a >> b;
    
    for(int i=a+1 ; i<b ; i++)
    {
    	if(i%3==0 && i%5==0)
    	{
    		vec.push_back(i);
    		continue;
    	}
    	else if(i%3==0 && i%5!=0)
    	continue;
    	else if(i%3!=0 && i%5==0)
    	continue;
    	else
    	vec.push_back(i);	    					
    }
    for(int i=0 ; i<vec.size() ; i++)
    {
    	cout << vec[i];
    	if(i!=vec.size()-1)
    	cout << ",";
    }
    return 0;
}
