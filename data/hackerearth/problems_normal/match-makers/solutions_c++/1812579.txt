#include <bits/stdc++.h>
using namespace std;
int fun(int i,int j)
{
	return (i>j);
}
int main()
{
    int t,count;
    cin>>t;
    int n;
    while(t--)
    {
    	cin>>n;
    	count=0;
    	vector <int> a(n);
    	vector <int> b(n);
    	for(int i=0;i<n;i++)
    	{
    		cin>>a[i];
    	}
       	for(int i=0;i<n;i++)
    	{
    		cin>>b[i];
    	}
    	sort(a.begin(),a.end());
    	sort(b.begin(),b.end(),fun);
    	for(int i=0;i<n;i++)
    	{
    		if(a[i]%b[i]==0 || b[i]%a[i]==0)
    		{
    			count++;
    		}
    	}
    	cout<<count<<endl;
    }
}
