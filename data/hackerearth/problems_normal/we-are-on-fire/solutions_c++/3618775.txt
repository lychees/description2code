#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int planet[1005][1005];
bool vis[1005][1005];
int spread(int x,int y)
{
    if(vis[x][y] || planet[x][y]==0 || planet[x][y]==-1)
        return 0;
    vis[x][y]=1;
    return (1+spread(x-1,y)+spread(x,y+1)+spread(x+1,y)+spread(x,y-1));
}
int main()
{
    int n,m,q,x,y,i,j,unburnt=0,cnt;
    scanf("%d %d %d",&n,&m,&q);
    for(i=0;i<=n;i++)planet[i][0]=-1;
    for(i=0;i<=n;i++)planet[i][m+1]=-1;
    for(i=0;i<=m;i++)planet[0][i]=-1;
    for(i=0;i<=n;i++)planet[n+1][i]=-1;
    for(i=0;i<n+3;i++)
        for(j=0;j<m+3;j++)
            vis[i][j]=0;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
            scanf("%d",&planet[i][j]);
            if(planet[i][j]==1)
                unburnt++;
        }
    }
    while(q--)
    {
        scanf("%d %d",&x,&y);
        cnt=spread(x,y);
        unburnt-=cnt;
        printf("%d\n",unburnt);
    }
    return 0;
}
