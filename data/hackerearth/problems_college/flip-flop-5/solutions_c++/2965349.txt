#include<bits/stdc++.h>
using namespace std;
char s[100005];
int n;
int Y(int);
int X(int i)
{
  if(i==n)
  return 0;
  if(s[i]=='X')
  return 1+Y(i+1);
  return X(i+1);
}

int Y(int i)
{
  if(i==n)
  return 0;
  if(s[i]=='Y')
  return 1+X(i+1);
  return Y(i+1);
}

int main()
{
  int t;
  cin>>t;
  while(t--)
  {
     scanf("%s",s);
     n = strlen(s);
     cout<<n-max(X(0),Y(0))<<endl;
  }
  return 0;
}
