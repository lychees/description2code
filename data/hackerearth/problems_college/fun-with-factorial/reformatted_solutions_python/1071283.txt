def fact(N):
    x = 1
    for i in range(1, N+1):
        x = x*i
    return x


def make_5(lis):
    newlis = ""
    if (len(lis) == 5):
        return lis
    else:
        for j in range(5-len(lis)):
            newlis = newlis+'0'
    return newlis+lis


x = input()
lis = []
for i in range(x):
    z = input()
    lis.append(z)
for i in lis:
    out = fact(i) % 100000
    out = str(out)
    print make_5(out)
