#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long int
#define pb push_back
#define mk make_pair
ll power(ll a, ll b) {
ll x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>mod) x%=mod;
        }
        y = (y*y);
        if(y>mod) y%=mod;
        b /= 2;
    }
    return x;
}
map<int,int> mp;
int main()
{
	int i,t,n;
	cin>>t;
	while(t--) {
		cin>>n;
		int a[n];
		vector<pair<int,int> >dat;
		for(i = 0; i < n; i++) {
			cin>>a[i];
			dat.pb({a[i],i});
		}
		sort(dat.begin(),dat.end());
		for(i = 0; i < n; i++) {
			mp[dat[i].first] = i+1;
		}
		for(i = 0; i < n; i++) {
			cout<<n-mp[a[i]]+1<<" ";
		}
		mp.clear();
		cout<<endl;
	}
	return 0;
}
