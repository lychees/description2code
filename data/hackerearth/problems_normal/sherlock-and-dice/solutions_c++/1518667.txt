//HackerEarth--SherlockAndDice
#include<bits/stdc++.h>
#define MAX 55

using namespace std;

int last,reqdSum,faces;
double dp[MAX][MAX*MAX];

double solve(int idx,int sum){
	
	if(idx==last){
		
		if(sum==reqdSum) return 1.0;
		else return 0.0;
		
	}
	
	if(dp[idx][sum]>=0.0) return dp[idx][sum];
	
	double prob=0.0;
	int i;
	
	for(i=1;i<=faces;i++)	
		prob+=(1/(double )faces)*solve(idx+1,sum+i);
		
	return dp[idx][sum]=prob;
	
}

int main(void){
	
	int testCases;
	
	scanf("%d",&testCases);
	
	while(testCases--){
		
		
		cin>>last>>faces>>reqdSum;
		
		memset(dp,-1.0,sizeof dp);
		
	    double prob=solve(0,0);
				
		int count=0;
		for(int i=0;i<100;i++){
		    
			if(prob>=1.0)
		      break;
		    
			prob*=10.0;
			
			count++;
			  
	    }
	    
	    if(prob==0.0)
	      count=0;
	    
	    printf("%.3lf %d\n",prob,count);
	    
	    
	}

	return 0;
	
}
