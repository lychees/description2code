#include<cstdio>
using namespace std;
int check(int n)
{
	int j=n;
	while(n!=0)
	{
		int a=n%2;
		n=n/2;
		int b=n%2;
		if(a==1 && b==1)
			return 1;
	}
	printf("%d\n",j);
	return 0;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		while(check(n))
			n++;
	}
}
