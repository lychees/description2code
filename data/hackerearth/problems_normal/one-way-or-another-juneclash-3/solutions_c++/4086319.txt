#include<iostream>
using namespace std;
int main(){
       int t;
       cin>>t;
       long long int r,c,x;
        int flag=0,ctr=0;
        long int n;
        for (int i=0;i<t;i++)
        {
        	cin>>n;
        	cin>>r>>c>>x;
        	if ((r+c)%2!=0 && x%2!=0)
        	flag=1;
        	else if ((r+c)%2==0 && x%2==0)
        	flag=2;
        	else if ((r+c)%2!=0 && x%2==0)
        	flag=3;
        	else if ((r+c)%2==0 && x%2!=0)
        	flag=4;
			ctr=0;       	
        	for (long int j=1;j<n;j++)
        	{
        		cin>>r>>c>>x;
        		if (ctr==0)
        		{
        			if (flag==1 || flag==2)
        			{
        				if ( ((r+c)%2!=0 && x%2==0) || ((r+c)%2==0 && x%2!=0) )
        				{	
        					ctr++;
        				}
        			}
        			else if (flag==3 || flag==4)
        			{
        				if ( ((r+c)%2==0 && x%2==0) || ((r+c)%2==0 && x%2==0) )
        				{
        					ctr++;
        				}
        			}
        		}
        	}
        	if (ctr==0)
        	cout<<"Yes"<<endl;
        	else
        	cout<<"No"<<endl;
        				
        }
    
}
