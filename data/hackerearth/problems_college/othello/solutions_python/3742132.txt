m, n = map(int, raw_input().split())
y_cnt = [0] * m
x_cnt = [0] * n
black = set()
for y in xrange(m):
	s = raw_input()
	for x, c in enumerate(s):
		if c == "B":
			x_cnt[x] += 1
			y_cnt[y] += 1
			black.add((x, y))

ans = 0
for y in xrange(m):
	for x in xrange(n):
		ans += (x_cnt[x] + y_cnt[y] + ((x, y) in black)) & 1
print ans
