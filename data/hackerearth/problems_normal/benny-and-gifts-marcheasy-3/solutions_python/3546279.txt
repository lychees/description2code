string = raw_input()
iceList = [[0,0]]
X,Y = 0,0
counter = 0
strlen = 0
for move in string:
    strlen += 1
    if move == 'L':
        Y -= 1
    elif move == 'R':
        Y += 1
    elif move  == 'U':
        X -= 1
    elif move == 'D':
        X  += 1
    iceList.append([X,Y])
iceList =  sorted(iceList)
for e in range(len(iceList)-1):
    if iceList[e] == iceList[e+1]:
        counter += 1
print counter
