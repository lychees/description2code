from sys import stdin

    
def main():
    T = int(stdin.readline())
    
    for i in range(T):
        R,C,L = map(int,stdin.readline().split())
        x = C/L
        
        if C%L == 0:
            result = (R*x) + L -1
        else:
            result = (R*x) + (2*L) -1

        print "Case #"+str(i+1)+": "+str(result)
main()
