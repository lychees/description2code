#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;

long long int max(long long int a, long long int b)
{
	return a > b ? a : b;
}

int main()
{
    int N;
    scanf("%d", &N);
    long long int i, t, max1 = 0, max2 = 0, min1 = 0, min2 = 0;
    for(i=0;i<N;++i)
    {
    	cin >> t;
    	if(t >= 0)
    	{
    		if(t > max1)
    		{
    			max2 = max1;
    			max1 = t;
    		}
    		else if(t > max2)
    		{
    			max2 = t;
    		}
    	}
    	else
    	{
    		if(t < min1)
    		{
    			min2 = min1;
    			min1 = t;
    		}
    		else if(t < min2)
    		{
    			min2 = t;
    		}
    	}
    }
    long long ans = (max1 * max2 > min1 * min2) ? max1 * max2 : min1 * min2;
    cout << ans << endl;
    return 0;
}
