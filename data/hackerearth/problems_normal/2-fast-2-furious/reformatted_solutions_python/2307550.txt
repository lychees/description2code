n = int(raw_input())
s = map(int, raw_input().split())
l = map(int, raw_input().split())
d = 0
f = 0
g = []
h = []
for i in range(n-1):
    d = abs(s[i+1]-s[i])
    f = abs(l[i+1]-l[i])
    if d > f:
        g.append(d)
    else:
        if d < f:
            h.append(f)
        else:
            h.append(f)
            g.append(d)
if max(g) > max(h):
    print "Dom\n%d" % (max(g))
else:
    if max(g) < max(h):
        print "Brian\n%d" % (max(h))
    else:
        print "Tie\n%d" % (max(g))
