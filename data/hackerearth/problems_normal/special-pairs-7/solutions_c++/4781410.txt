#include <bits/stdc++.h>
#define F first
#define S second
#define MOD 1000000007
#define pb push_back
#define MAX 200002
#define REP(i,n) for(int i=0;i<n;i++)
#define FOR(i,a,b) for(int i=a;i<b;i++)
#define mp make_pair
#define CLEAR(a) memset(a,0,sizeof a)
#define pii pair< int , int >
#define fr freopen("input.txt", "r", stdin)
#define fw freopen("output.txt", "w", stdout)
using namespace std;
int d[(1<<22)],f[1<<22],dp[(1<<22)][22];

int main()
{
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    int T;
    cin >> T;
    while(T--){
        int n;
        cin >> n;
        REP(i,n){
            cin >> d[i];
            f[d[i]]++;
        }

        for(int i=0;i<(1<<22);i++){
            dp[i][0] = f[i];
            if(i&1) dp[i][0] += f[i^1];

            for(int j=1;j<22;j++){
                dp[i][j] = dp[i][j-1];
                if(i&(1<<j)) dp[i][j] += dp[i^(1<<j)][j-1];
            }
        }

        long long int ret = 0;

        for(int i=0;i<n;i++){
            ret += dp[(1<<22)-1-d[i]][21];
            f[d[i]] = 0;
        }
        cout << ret <<"\n";
    }
}
