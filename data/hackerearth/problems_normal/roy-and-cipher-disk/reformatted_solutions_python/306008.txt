numOfTestCase = int(raw_input())
for i in range(numOfTestCase):
    strInput = raw_input()
    strInput = 'a'+strInput
    for j in range(len(strInput)-1):
        first = ord(strInput[j])
        sec = ord(strInput[j+1])

        if sec >= first:
            gap = sec-first
            if gap <= 13:
                print gap,
            else:
                print -(13-(gap-13)),
        else:
            gap = first-sec
            if gap < 13:
                print -gap,
            else:
                print 13-(gap-13),
    print ''
