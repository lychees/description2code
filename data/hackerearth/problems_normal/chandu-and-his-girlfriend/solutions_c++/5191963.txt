#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

bool comp(int a, int b) {
	return a>b;
}
int main()
{
    int T;
    cin >> T;
    for (int i = 0; i < T; i++) {
    	int N;
    	cin >> N;
    	vector<int> A(N);
    	for (int j = 0; j < N; j++)
    		cin >> A[j];
    	sort(A.begin(), A.end(), comp);
    	for (int j = 0; j < N; j++)
    		cout << A[j] << " ";
    	cout << endl;
    }
    return 0;
}
