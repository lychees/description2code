#include<iostream>
using namespace std;
bool isprime(int n)
{
	int i;
	for(i=2;i<n;i++)
	{
		if(n%i==0)
		break;
	}
	if(i==n)
	return true;
	else
	return false;
}
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int l=2;
		int r=n-2;
		while(l<=r)
		{
			if(isprime(l) && isprime(r))
			{
				cout<<"Deepa"<<endl;
				break;
			}
			l++;r--;
		}
		if(l>r)
		cout<<"Arjit"<<endl;
	}
}
