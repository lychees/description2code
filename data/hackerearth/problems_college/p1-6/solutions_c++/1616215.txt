#include <bits/stdc++.h>
using namespace std;

#define ll long long int

char str[1000010];

int main()
{
	ll i,n,ans,total,upto,after;

	scanf("%s", str);

	n = strlen(str);

	total = 0;

	for(i=0;i<n;i++) {
		if(str[i] == '1') total++;
	}
	upto = 0;
	ans = 0;

	for(i=0;i<n;i++) {
		if(str[i] == '1') upto++;
		else if(str[i]=='0') {
			after = total - upto;
			ans = ans + (upto * after);
		}
	}

	printf("%lld\n", ans);

	return 0;
}
