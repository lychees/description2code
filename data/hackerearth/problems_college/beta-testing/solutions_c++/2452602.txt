#include <iostream>
#include<string.h>
using namespace std;

int present(string name[],string s,int top){
	if(top == 0) return -1;
	int i;
	for(i=1;i<=top;i++){
		if(s==name[i]) return i;
	}
	return -1;
}

int main()
{
    int i,j,n,top=0,t,v;
    cin >> n;
    string name[n+1],s,ts;
    int count[n+1];
    for(i=1;i<=n;i++) count[i] = 0;
    for(i=1;i<=n;i++){
    	cin >> s;
    	v = present(name,s,top);
    	if(v==-1){
    		top++;
    		name[top] = s;
    		count[top]++;
    	}
    	else{
    		count[v]++;
    	}
    }
    //sort
    for(i=1;i<=top-1;i++){
    	for(j=1;j<=top-i;j++){
    		if(count[j]<count[j+1]){
    			t = count[j];
    			count[j] = count[j+1];
    			count[j+1] = t;
    			ts = name[j];
    			name[j] = name[j+1];
    			name[j+1] = ts;
    		}
    	}
    }
    for(i=1;i<=top;i++){
    	cout << name[i] << " " << count [i] << "\n";
    }
    
}
