#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
ll dp[1000001];
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(false);
    ll i,fg,t,n;
    dp[0]=0;
    dp[1]=-1;
    for(i=2;i<=1000000;i++)
    {
        fg=INT_MAX;
        if(i-2>=0&&dp[i-2]!=-1)
        {
            fg=min(fg,1+dp[i-2]);
        }
        if(i-3>=0&&dp[i-3]!=-1)
        {
            fg=min(fg,1+dp[i-3]);
        }
        if(i-5>=0&&dp[i-5]!=-1)
        {
            fg=min(fg,1+dp[i-5]);
        }
        if(i-7>=0&&dp[i-7]!=-1)
        {
            fg=min(fg,1+dp[i-7]);
        }
        if(fg!=INT_MAX)
        dp[i]=fg;
        else
        dp[i]=-1;
    }
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<dp[n]<<"\n";
    }
    return 0;
}
