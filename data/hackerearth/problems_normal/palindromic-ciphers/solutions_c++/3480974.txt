#include <iostream>
#include <string>
#include <algorithm>
#include <iterator>
using namespace std;

int main()
{
	unsigned long long mul=1;
    int T;
    string s,t;
    cin>>T;
    while(T--){
    cin>>s;
    t=s;
    mul=1;
    reverse(begin(s),end(s));
    if(s.compare(t)==0) cout<<"Palindrome\n";
    else{
    	for(int i=0;i<s.length();i++)
    	mul*=(s[i]-96);
    	
    	cout<<mul<<endl;
    }
    }
    
    return 0;
}
