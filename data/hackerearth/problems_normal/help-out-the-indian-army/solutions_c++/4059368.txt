#include<bits/stdc++.h>
using namespace std;
#define in(x) scanf("%lld",&x)
int main()
{
    long long int n,s,e,i,x,y,a,b,ans;
    in(n),in(s),in(e);
    vector<pair<long long int,long long int> >v;
    for(i=0;i<n;i++)
    {
        in(x),in(y);
        if(x-y<e && x+y>s)
        {
            a=max(s,x-y),b=min(e,x+y);
            v.push_back(make_pair(a,b));
        }
    }
    sort(v.begin(),v.end());
    ans=0,y=s;
    n=v.size();
    for(i=0;i<n;i++)
    {
        a=v[i].first;
        b=v[i].second;
        if(y<a)
            ans=ans+a-y,y=b;
        else if(y>=a)
            y=max(y,b);
    }
    ans=ans+e-y;
    cout << ans << endl;
    return 0;
}
