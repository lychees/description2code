#include <iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    string str;
    int a[26],b[26],c[26];
    char prev;
    for(int i=0;i<n;i++){
    	 for(int i=0;i<26;i++){
    	    	a[i]=0;
    	    	b[i]=0;
    	    	c[i]=0;
    	    }
    	cin>>str;
    	prev=str.at(0);
    	for(int j=0;j<str.length();j++){
    		int k=str.at(j);
    		if(a[k-97]==0){
    			b[k-97]=j;
    		}
    		if(prev!=str.at(j)){
    			int k2=prev;
    			c[k2-97]=j-1;
    			prev=str.at(j);
    		}
    		a[k-97]++;
    	}
    	int k2=prev;
    	c[k2-97]=str.length()-1;
    int count=0,flag=0,count2=0;
    for(int i=0;i<26;i++){
    	if(a[i]!=0){
    		if(count==0){
    			count2=a[i];
    		}
    		else if(count2!=a[i]){
    			flag=1;
    			break;
    		}
    		count++;
    		if(count>3){
    			flag=1;
    			break;
    		}
    		if(c[i]-b[i]+1!=count2){
    			flag=1;
    			break;
    		}
    	}
    }
    if(flag==1||count<3){
    	cout<<"Not OK"<<endl;
    }
    else{
    	cout<<"OK"<<endl;
    }
   
    }
    return 0;
}
