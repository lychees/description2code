#include<iostream>
using namespace std;
typedef long long ll;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		ll l=s.length();
		ll i,count=0,sum=0;
		for(i=0;i<l;i++)
		{
			if(s[i]=='T')
			{
				count=0;
			}
			else
			{
				count++;
				sum=sum+count;
			}
			
		}
		cout<<sum<<endl;
	}
}
