long long int a[200005];

void max_heapify(int j){
	int left=2*j;
	int right=2*j + 1;
	int largest;
	if(left<=i && a[left]>a[j])
		largest=left;
	else
		largest=j;
	if(right<=i && a[right]>a[largest])
		largest=right;
     if(largest!=j){
     	long long x=a[largest];
     	a[largest]=a[j];
     	a[j]=x;
     	max_heapify(largest);
     }
}
