#include <iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,i;
	scanf("%d",&n);
	int s[n],c[n];
	long long ans=0;
	for(i=0;i<n;i++)
	scanf("%d",&s[i]);
	for(i=0;i<n;i++)
	scanf("%d",&c[i]);
	sort(s,s+n);
	sort(c,c+n);
	for(i=0;i<n;i++)
	ans+=1ll*s[i]*c[i];
	printf("%lld\n",ans);
}
