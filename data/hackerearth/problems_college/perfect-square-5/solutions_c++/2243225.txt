#include<math.h>
#include <iostream>
using namespace std;

int main()
{
    unsigned long long int n,t,j,f,c,x,y;
    long  double a;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	a=sqrt(n);
    	f=floor(a);
    	c=ceil(a);
    	if(f==c)
    	{
    		cout<<"YES\n";
    	}
    	else
    	{
    		x=c*c;
    		y=f*f;
    		if(n-y<x-n)
    			cout<<"-"<<n-y<<endl;
    		else
    			cout<<"+"<<x-n<<endl;
    	}
    }
    return 0;
}
