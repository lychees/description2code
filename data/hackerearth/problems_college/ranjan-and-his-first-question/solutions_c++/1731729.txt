#include <iostream>
using namespace std;

int main() {
	unsigned long long t,n,m,k,temp,temp2,u,sum=0;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		k=m-1;
		k=(m*(m-1))/2;
		temp=n/m;
		
		temp2=n%m;
	
		sum=0;
		u=1;
		if(temp2!=0)
		temp2=(temp2*(temp2+1))/2;
		sum=temp2;
	
		sum=sum+(temp*k);
		cout<<sum<<endl;
	
	}
	return 0;
}
