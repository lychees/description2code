#include<bits/stdc++.h>


#define LL long long int
#define mod   1000000007
using namespace std;

LL bmod(LL a,LL b)
{
      if(b==(LL)0)
      {
           return (1);
      }
      else if(b%(LL)2==(LL)0)
      {
            LL ans=bmod(a,b/(LL)2);
           return((ans)*(ans))%mod;
      }
      else if(b%(LL)2!=(LL)0)
      {
            LL k=a%mod;
            LL kk=bmod(a,b-1);
            return ((k)*(kk))%mod;
      }
}
