t = input()
for i in xrange(t):
	n = input()
	
	a = map(int, raw_input().split())
	
	M = input()
	c = 0
	z = 0
	k = sum(a)
	y = True
	#print M,k
	if M==0:
		c = 1
		y = False
	elif M%k==0:
		c = 0
		for u in xrange(n):
			if a[u]!=0:
				c = u
		c+=1
		y = False
	elif M>k:
		q = M-((M/k)*k)
	else:
		q = M	
	for j in a:
			if not y:
				break
			if z>=q:
				break
			c+=1
			z+=j
			if z>=q:
				break
			
	
	print c
