#include <bits/stdc++.h>
using namespace std;

int main()
{
	bool flag;
    long long int i,j,t,n,a[100001];
    scanf("%lld",&t);
    while(t--){
    	flag = true;
    	scanf("%lld",&n);
    	for(i=0;i<n;i++)
    		scanf("%lld",&a[i]);
    	i=0;
    	while(1){
    		if(a[i]==-2){
    			flag = false;
    			break;
    		}
    		if(a[i]==-1)
    			break;
    		j = a[i];
    		a[i] = -2;
    		i=j;    		
    	}
    	if(flag)
    		printf("Win\n");
    	else
    		printf("Lose\n");
    }
    return 0;
}
