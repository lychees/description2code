#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
#include <vector>
using namespace std;

int main () {
	int t;
	cin >> t;
	vector < string > results(t);
	for (int i=0; i<t; i++) {
		long long int n,k,points=0,num;
		cin >> n >> k;
		for (int j=0; j<n; j++) {
			//cin >> num;
			scanf ("%lld",&num);
			if (num == k)	points++;
		}

		if (points >= 3) {
			long long int den = (n*(n-1)*(n-2))/6;
			long long int nem = (points*(points-1)*(points-2))/6;
			long long int gcd = __gcd(den,nem);

			nem = nem/gcd;
			den = den/gcd;
			results[i] = to_string(nem) + "/" + to_string(den);
		}

		else {	
			results[i] = "0/1"; 
			
		}

	}

	for (int i=0; i<t; i++)
		cout << results[i] << endl;
	return 0;
}
