#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string s;int n;
		cin>>n;
		//cout<<s;
		cin>>s;
		n%=26;
		for(int i=0;i<s.length();i++) 
		{
			if(char(s[i]-n)<'A')
			{
				int x=s[i]-'A';
				x=n-x-1;
				s[i]=char('Z'-x);
			}
			else
			s[i]=char(s[i]-n);
			cout<<s[i];
		}
		cout<<endl;
	}
	return 0;
}
