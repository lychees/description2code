for tc in xrange(input()):
	n, s = raw_input().split()
	n = int(n)
	count = [0] * 26
	for c in s:
		count[ord(c) - 97] += 1
	# print count
	winner = 1
	bestScore = -1
	mem = 0
	for i in xrange(n):
		score = 0;
		s = raw_input()
		for c in s:
			# print c,
			score += count[ord(c) - 97]
		# print score
		if score > bestScore or (score == bestScore and len(s) < mem):
			bestScore = score
			mem = len(s)
			winner = i + 1
	print winner
