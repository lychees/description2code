#include<bits/stdc++.h>
using namespace std;
string foo(int n)
{
    string ret="";
    ret+=(n/10)%10 +'0';
    ret+=n%10 + '0';
    return ret;
}
int ar[10]={};
bool check(int h, int m , int s)
{
    memset(ar,0,sizeof(ar));
    ar[(h/10)%10]++;
    ar[h%10]++;
    ar[(m/10)%10]++;
    ar[(m)%10]++;
    ar[(s/10)%10]++;
    ar[(s)%10]++;
    for(int i=0; i<10; i++)
	if(ar[i]>1)return false;
    return true;
}
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
	int h,m,s,i,j,k;
	scanf("%d:%d:%d",&h,&m,&s);
	while(1)
	{
	    if(check(h,m,s))break;
	    s=s+1;
	    if(s==60)m++,s=0;
	    if(m==60)h++,m=0;
	    if(h==24)h=0;
	}
	cout << foo(h) << ":" <<foo(m) << ":" << foo(s) << endl;
    }
    return 0;
}
