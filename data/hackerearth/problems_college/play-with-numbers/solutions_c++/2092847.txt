#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int x,y,z,i,j,count=0;
    	scanf("%lld%lld%lld",&z,&x,&y);
    	for(i=1;i<=sqrt(z);i++)
    	{
    		if(z%i==0)
    		{
    			long long val1=i,val2=(z/i);
    			if(val1<=x && val2<=y)
    			{
    				//printf("v1 = %lld\n",val1);
    				//printf("v2 = %lld\n",val2);
    				count++;
    			}
    			if(val1<=y && val2<=x && val1!=val2)
    			{
    				//printf("v1 = %lld\n",val1);
    				//printf("v2 = %lld\n",val2);
    				count++;
    			}
    		}
    	}
    	printf("%lld\n",count);
    }
    return 0;
}
