#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define mod 1000000007
int calarea(int x1, int y1, int x2, int y2, int x3, int y3)
{
    int area= (x1*(y3-y2)+x2*(y1-y3)+x3*(y2-y1));
    if(area<0)
        area*=-1;
    return area;
}
int main()
{
    int T=0,x1,y1,x2,y2,x3,y3,x4,y4;
    int a1,a2,a3,a4;
    sd(T);
    while(T--)
    {
        sd(x1);
        sd(y1);
        sd(x2);
        sd(y2);
        sd(x3);
        sd(y3);
        sd(x4);
        sd(y4);
        a1=calarea(x1,y1,x2,y2,x4,y4);
        a2=calarea(x1,y1,x3,y3,x4,y4);
        a3=calarea(x3,y3,x2,y2,x4,y4);
        a4=calarea(x1,y1,x2,y2,x3,y3);
        //printf("%lf %lf %lf %lf\n",a1,a2,a3,a4);
        if(a1+a2+a3==a4)
            printf("INSIDE\n");
        else
            printf("OUTSIDE\n");
    }
	return 0;
}
