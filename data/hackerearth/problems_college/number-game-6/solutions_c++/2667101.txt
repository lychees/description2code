#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t, i;
	long long int n;
	long long int a[30001];

	for(i = 0; i < 30001; i++)
	{
		a[i] = (long long int)(3*i*i + 11*i + 10)*(i)*(i+1) / 12;
	}



	//cout << a[25147] << ' ' <<  a[25148];// <<  ' ' << a[2] << ' ' << a[30000];

	//r/eturn 0;
	scanf("%d", &t);

	while(t--)
	{
		scanf("%lld", &n);
		if(n < 4)
		{
			printf("0\nDraw\n");
		}
		else
		{
			
			int l = 0, r = 27000, mid;
			
			while(l <= r)
			{
				mid = (l + r) / 2;
				if(a[mid - 1] <= n && a[mid] > n)
					break;
				else if(a[mid] > n)
				{
					r = mid - 1;
				}
				else
					l = mid + 1;

			}
			
			if(mid % 2 == 0)
			{
				printf("%d\nDhruv\n", mid-1);
			}
			
			else
			{
				printf("%d\nKetan\n", mid-1);
			}
		}
	}
	return 0;
}
