#include <iostream>
using namespace std;

int main()
{
	int n;
	cin>>n;
	int b[n];
	int a[n];
	for(int i=0;i<n;i++)
		cin>>b[i];
	int s=0;
	for(int i=0;i<n;i++)
	{
		
		if(i==0)
			a[i]=b[i];
		else
		{
			s+=a[i-1];
		//	cout<<"SUM"<<s<<endl;
			a[i]=(i+1)*b[i]-s;	
		}
	}
	for(int i=0;i<n;i++)
		cout<<a[i]<<" ";
		
    //cout << "Hello World!" << endl;
    return 0;
}
