#include <iostream>
using namespace std;

int main()
{
	long int t,t1,t2,T;
	long int a,b;
	cin>>T;
	for(int i=0;i<T;i++)
	{
		cin>>a>>b;
		t1=a;
		t2=b;
		while(b!=0)
		{
			t=a%b;
			a=b;
			b=t;
		}
		if(a==1)
		{
			cout<<"\nIs a Co-Prime";
		}
		else
		{
			cout<<"\nNot a Co-Prime";
		}
	}
    //cout << "Hello World!" << endl;
    return 0;
}
