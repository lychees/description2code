#include <bits/stdc++.h>
#define ll long long int
#define mod (int)(1e9 + 7)
using namespace std;

ll x, n;
ll a[1000005];
ll i, j, sz;

int main()
{
	cin >> n >> x;
	
	for (i = 0; i < n; i++) {
		cin >> a[i];
	}
	sz = pow(2, n);
	vector<int> r;
	for (i = 1; i < sz; i++) {
		vector<int> tmp;
		ll sum = 0;
		for (j = 0; j < n; j++) {
			if (i & (1 << j)) {
				sum += a[j];
				tmp.push_back(j);
			}
		}
		if (sum == x) {
			/*for (j = 0; j < tmp.size(); j++) {
				cout << tmp[j] << " ";
			}
			cout << endl;*/
			if (r.size() == 0) {
				r = tmp;
			} else if (r.size() > tmp.size()) {
				r = tmp;
			} else if (r.size() == tmp.size()) {
				int flag = 0;
				for (j = 0; j < r.size(); j++) {
					if (r[j] <= tmp[j]) {
						continue;
					}
					flag = 1;
					break;
				}
				if (flag == 1) {
					r = tmp;
				}
			}			
		}
	}
	if (r.size() == 0) {
		cout << "-1\n";
	} else {
		for (i = 0; i < r.size(); i++) {
			cout << r[i] + 1 << " ";
		}
		cout << endl;
	}
	
	return 0;
}
