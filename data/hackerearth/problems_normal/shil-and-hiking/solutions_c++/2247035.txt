#include <bits/stdc++.h>
using namespace std;

double a, b;

int X[1003], H[1003];

bool beautiful(int i, int j, int flag)
{
	a = X[j]-X[i];
	b = H[j]-H[i];
	
	double allowed_y;
	
	if(flag)
	{
		int k = j-1;
		
		allowed_y = (double)(b*X[k]-(b*X[i]-a*H[i]))/a;
		
		if(allowed_y<(double)H[k])
			return 0;
		
		return 1;
	}
	
	for(int k=i+1; k<j; k++)
	{
		allowed_y = (double)(b*X[k]-(b*X[i]-a*H[i]))/a;
		
		if(allowed_y<(double)H[k])
			return 0;
	}
	
	return 1;
}

int main()
{
    int N;
    scanf("%d", &N);
    
    for(int i=1; i<=N; i++)
    	scanf("%d %d", &X[i], &H[i]);
    
    int ans = 0;
    
    for(int i=1; i<=N; i++)
    {
    	int flag = 0;
    	
    	for(int j=i+2; j<=N; j++)
    	{
    		if(!flag)
    		{
    			if(beautiful(i, j, 0))
    			{
    				ans++;
    				flag = 1;
    			}
    			else
    				flag = 0;
    		}
    		else
    		{
    			if(beautiful(i, j, 1))
    			{
    				ans++;
    				flag = 1;
    			}
    			else
    				flag = 0;
    		}
    	}
    }
    
    printf("%d\n", ans+(N-1));
    
    return 0;
}
