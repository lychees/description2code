def jhole():
    msg = raw_input()
    mstr ,d = ['l','o','v','e'], ""
    for i in msg:
        if mstr[len(d)] == i:
            d += i
            if len(d) == 4:
                break

    if "love" == d:
        print "I love you, too!"
    else:
        print "Let us breakup!"

if __name__ == '__main__':
    jhole()
