t= int(raw_input())

for i in range(t):
	n, x = map(int,raw_input().split())
	a= []
	for j in range(n):
		a.append(int(raw_input()))
	st=0
	flag = 0
	s = a[0]
	for j in range(1,n):
		while s>x:
			s-=a[st]
			st+=1
		if(s==x):
			flag=1
			break;
		s += a[j]
	while s>x:
			s-=a[st]
			st+=1
	if(s==x):
		flag=1
	if (flag==1):
		print('YES')
	else:
		print('NO')
