ll n;
vector < ll > g[N];
ll depth[N], ans[N];
void dfs(ll u, ll pre, ll d)
{
    depth[u] = d;
    for(ll v : g[u])
    {
        if(v == pre) continue;
        dfs(v, u, d + 1);
    }
}
void dfs2(ll u, ll pre)
{
    ans[u] = 1;
    for(ll v : g[u])
    {
        if(v == pre) continue;
        dfs2(v, u);
        ans[u] += ans[v];
    }
}

        cin >> u >> v;
        g[u].pb(v);
        g[v].pb(u);
    }
    dfs2(1, 0);
    ans[1]--;
    rep(i, 1, n) cout << ans[i] << " ";
    return 0;
}
