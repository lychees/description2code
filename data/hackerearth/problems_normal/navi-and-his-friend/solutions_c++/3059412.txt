#include <bits/stdc++.h>
#define mp make_pair
using namespace std;
vector<pair<int,int> >A(20);
int ind[20];
int n,p,w,C,d;
long long W;
int main()
{
	scanf("%d",&d);
   for(int a=1;a<=d;a++)
   {
   		printf("For Day #%d:\n",a);
	    scanf("%d",&n);
	    for(int i=0;i<n;i++)
	    {
	    	scanf("%d %d",&p,&w);
	    	A[i]=mp(p,w);
	    }
	    scanf("%d %d",&W,&C);
	    int y=1;
	    long long price=0;
	    for(int i=1;i<(1<<n);i++)
	    {
	    	int cnt=0;
	    	for(int j=0;j<y;j++)
	    	{
	    		if(i&(1<<j))
	    		ind[cnt++]=j;
	    	}
	    	if(i==(1<<y)-1)
	    	y++;
	    	if(cnt>C)
	    	continue;
	    	long long weight=0,temp=0;
	    	for(int j=0;j<cnt;j++)
	    	{
	    		weight+=A[ind[j]].second;
	    		if(weight>W)
	    		{
	    			temp=0;
	    			break;
	    		}
	    		temp+=A[ind[j]].first;
	    	}
	    	price=max(price,temp);
	    }
	    if(price==0)
	    price=-1;
	    cout<<price<<endl;
	    
   }
    
}
