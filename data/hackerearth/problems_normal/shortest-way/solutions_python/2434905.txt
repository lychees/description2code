if __name__ == "__main__":

    T = int(raw_input())

    for t in xrange(T):

        u, v = map(int, raw_input().split())

        CT = 0

        while u > 1 or v > 1:
            CT += u/v
            CT -= 0 if v != 1 else 1
            t = u%v
            u = v
            v = t

        print CT
