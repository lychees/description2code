def days(n):
    if( not(n-(-n&n)) ):
        return 1;
    power=0
    check=False
    i=0
    while(check is False):
        if(2**i > n):
            power=i-1
            check=True
        i+=1

    return 1+days(n-2**power)
t=int(raw_input())
while t:
    t-=1
    n=int(raw_input())
    if(n==0):
    	print(0)
    else:
    	print(days(n))
