#include<iostream>
#include<vector>
#include<string>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    vector<string>row,col;
    while(tc--)
    {
        int n,flag2=1,flag3=1;
        cin>>n;
        int temp=n;
        while(temp--)
        {
            string pb;
            cin>>pb;
            row.push_back(pb);
        }
        for(int i=0;i<n;i++)
        {
            string nw="";
            for(int j=0;j<n;j++)
            {
                string temp=row[j];
                nw+=temp[i];
            }
            col.push_back(nw);
        }
        for(int i=0;i<n/2;i++)
        {
            if(row[i]!=row[n-i-1])
            {
                flag2=0;
            }
            if(col[i]!=col[n-i-1])
            {
                flag3=0;
            }

        }
    if(flag2==1&&flag3==1)
        cout<<"BOTH\n";
    else if(flag2==1)
        cout<<"HORIZONTAL\n";
    else if(flag3==1)
        cout<<"VERTICAL\n";
    else
        cout<<"NO\n";

    row.erase(row.begin(),row.end());
    col.erase(col.begin(),col.end());
    }
}
