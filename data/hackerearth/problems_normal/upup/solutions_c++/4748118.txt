#include <iostream>
using namespace std;

int main()
{
    //cout << "Hello World!" << endl;
    string s;
    getline(cin,s);
    if(s[0]>='a'&&s[0]<='z')
    	s[0]=s[0]-'a'+'A';
    for(int i=1;i<s.length();i++){
    	if(s[i-1]==' '){
    		if(s[i]>='a'&&s[i]<='z')
    			s[i]=s[i]-'a'+'A';
    	}
    }
    
    cout<<s;
    return 0;
}
