#define pb push_back

#define f first
#define maxn 10100
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18+1)
#define mod 1000000007
#define sz(x) (int) x.size()
using namespace  std;
ll arr[maxn];
int main(int argc, char const *argv[])
{
   int t,n,k;
   cin>>t;
   while(t--)
    {
      memset(arr,0,sizeof arr);
cin>>n>>k;
int val = n/k;
int v2 = n%k;
for(int i=0;i<k;i++)
{
  arr[i] = val;
  if(i<=v2&&i!=0)
    arr[i]++;
}
ll ans=0;
for(int i=0;i<k;i++)
{
  if(i==0)
    ans+=(arr[i]*(arr[i]-1))/2;
  else
  {
    ll vv = (arr[i]*arr[k-i]);
    if(i==k-i)
    vv = (arr[i]*(arr[i]-1))/2;
    ans+=vv;
    arr[i]=0;arr[k-i]=0;
  }
}

cout<<ans<<"\n";
}
}
