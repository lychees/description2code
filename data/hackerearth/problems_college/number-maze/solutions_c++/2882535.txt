#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long long int a,b;
	cin>>a>>b;
	for(int i=a+1;i<b;i++)
	{	
		if(i==b-1)
		{
		if(i%3==0 && i%5==0)
		{
			cout<<i;
			break;
		}
		if(i%3==0 || i%5==0)
		{
			break;
		}
		else
		{
			cout<<i;
			break;
		}
	}
		else
		{
		if(i%3==0 && i%5==0)
		{
			cout<<i<<",";
		}
		if(i%3==0 || i%5==0)
		{
			continue;
		}
		else
		{
			cout<<i<<",";
		}
		
	}
}
}
