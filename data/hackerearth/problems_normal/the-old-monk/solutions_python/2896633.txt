def b_search(begin,end,arr,value):
	while (end-begin)>0:
		mid = (begin+end)/2
		if arr[mid]<value:
			end = mid
		else:	
			begin = mid+1
	if arr[end]!=value:
		end=end-1
	return end

t = int(raw_input())
while t>0:
	n = int(raw_input())
	A=map(int,raw_input().split())
	B=map(int,raw_input().split())
	res=-1
	maxim=0
	for i in xrange(n):
		res = b_search(i,n-1,B,A[i])
		res = res - i
		if res>maxim:
			maxim = res
	print maxim
	t-=1
