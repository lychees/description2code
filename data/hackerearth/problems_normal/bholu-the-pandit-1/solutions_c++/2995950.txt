#include <bits/stdc++.h>
using namespace std;

int main()
{

	long long a, b, c;
	int t;
	
	cin >> t;
	long long p;
	
	long long mod = 1000000007;
	long long vol;
	
	while (t--)
	{
		cin >> a >> b >> c;
		p = __gcd(a,__gcd(b,c));
		cout << p << " ";
		
		vol = (a / p) * (b / p); 	
		vol %= mod;
		
		vol *= (c / p);
		vol %= mod;
		
		cout << vol << endl;
	}


	return 0;
}
