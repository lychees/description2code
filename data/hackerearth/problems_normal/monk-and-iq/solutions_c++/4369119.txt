
#include<bits/stdc++.h>
using namespace std;

int c,p,n,x[1<<20],y[1<<20];
set<pair<int, int> >S;
set<pair<int, int> >::iterator it;
vector<int> vec[1<<20];

int add_student(int val)
{
	it=S.begin();
	int id=(*it).second;
	vec[id].push_back(val);
	int v1=vec[id].size();
	int  v2=0;
	for (int i=v1-1;i>=0&&i>=v1-2;i--)
		v2+=vec[id][i];
	S.erase(it);
	S.insert(make_pair(v1*v2,id));
	return id;
}

int main(){

ios_base::sync_with_stdio(0);


cin>>c>>p>>n;

for (int i=1;i<=n;i++)
	cin>>x[i];

for (int i=1;i<=p;i++)
	cin>>y[i];

for (int i=1;i<=n;i++)
//	int t=add_student(x[i]);
{
	vec[i].push_back(x[i]);
}

for (int i=1;i<=c;i++)
	S.insert(make_pair(x[i],i));

for (int i=1;i<=p;i++)
{
	int t=add_student(y[i]);
	
	if (i>1)cout<<" ";
	cout<<t;
//	cout<<t<<" ";
}
cout<<endl;

//cin.get();cin.get();
return 0;}
