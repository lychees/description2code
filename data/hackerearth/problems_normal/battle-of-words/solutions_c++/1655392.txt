#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define ll long long
#define MAX 100010

char s[MAX];
int a[300];
int t, n, r1, r2;

int main(){
	scanf("%d\n", &t);
	while (t--){
		memset(a, 0, sizeof(a));
		gets(s);
		n = strlen(s);
		for (int i = 0; i < n; i++)
			a[s[i]]++;
		gets(s);
		n = strlen(s);
		for (int i = 0; i < n; i++)
			a[s[i]]--;
		r1 = r2 = 0;
		for (int i = 'a'; i <= 'z'; i++){
			if (a[i] > 0) r1++;
			if (a[i] < 0) r2++;
		}
		if (r1 && r2 == 0)
			printf("You win some.\n");
		else
			if (r1 == 0 && r2)
				printf("You lose some.\n");
			else
				printf("You draw some.\n");
	}

	return 0;
}
