Matrix = []
for i in xrange(110):
    Matrix.append([])
    for j in xrange(1015):
        Matrix[i].append(0)

for i in range(1, 105):
    Matrix[i][0] = 1

for i in range(1, 105):
    for j in range(1, 1005):
        Matrix[i][j] = (Matrix[i][j - 1] * i)

t = input()
t = int(t)

while t:
    t = t-1
    x, m = raw_input().split()
    x = int(x)
    m = int(m)
    x = Matrix[x][m]
    ans = 0
    s = str(x)
    for k in range(len(s)):
        ans = ans + (int(s[k]))
    print ans
