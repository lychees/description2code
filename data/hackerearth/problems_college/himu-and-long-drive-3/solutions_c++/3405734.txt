#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	
	while(t--)
	{
		long long int n;
       cin>>n;
       
       if((n & (n-1)) ==0 )
       {
       	cout<<((n)^(n-1))<<"\n";
	   }
	   else
	   {
	   int c=0;
		while(n!=0)
		{
			n = n/2;
			c++;
		}
		long long sum = pow(2,c)-1;
		cout<<sum<<"\n";
	   }
	}
}
