
#include<bits/stdc++.h>
using namespace std;

# define mod 1000000007
long long f,inv1,inv2,inv;
long long power(long long a,long long b)
{
    long long res=1;
    while(b)
    {
        if(b%2)
        {
            res=(res*a)%mod;
        }
        a=(a*a)%mod;
        b/=2;
    }
    return res;
}
int main()
{
    long long n,k,a,b;
    cin>>a>>b>>n>>k;

    long long i;
    f=1;
    for(i=1;i<=n;i++)
    {
       f=(f*i)%mod;
    }
    inv=1;
    for(i=1;i<=max(k,n-k);i++)
    {
        inv=(inv*power(i,mod-2))%mod;
        if(i==k) inv1=inv;
        if(i==n-k) inv2=inv;
    }
    long long ans=1;

ans=(ans*f)%mod;
ans=(ans*inv1)%mod;
ans=(ans*inv2)%mod;
ans=(ans*power(a,k))%mod;
ans=(ans*power(b,n-k))%mod;
    cout<<ans;
    return 0;
}
