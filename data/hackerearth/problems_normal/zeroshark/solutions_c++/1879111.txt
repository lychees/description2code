#include <bits/stdc++.h>

using namespace std;

const int N=1e5+5;
const int MOD=1e9+7;
int test,n,f[2][N][2]={0};
void prepare(){
	f[0][0][1]=1;
	for(int i=1;i<N;i++){
		for(int j=0;j<=1;j++) f[j][i][1]=(f[j][i-1][0]+f[j][i-1][1])%MOD;
		f[0][i][0]=f[0][i-1][1];
		f[1][i][0]=(f[0][i-1][0]+f[1][i-1][1])%MOD;
	}
}
int ans(int x){
	return (f[1][x][0]+f[1][x][1])%MOD;	
}
int main()
{
	prepare();
    scanf("%d",&test);
    while (test--){
    	scanf("%d",&n);
    	printf("%d\n",ans(n));
    }
}
