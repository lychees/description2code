#include <iostream>
#include<vector>
using namespace std;

int main()
{
    vector<int> arr;
    long long int n,q,ans=1,abc;
    cin>>n;
    for(long long int i=0;i<n;i++)
    {
    	long long int num;
    	cin>>num;
    	arr.push_back(num);
    }
    cin>>q;
    for(long long int i=0;i<q;i++)
    {
    	cin>>abc;
    	ans=ans*abc;
    	
    }
    for(long long int i;i<n;i++)
    {
    	if(ans!=0)	cout<<arr[i]/ans<<" ";
    	if(ans==0) cout<<0<<" ";
    }
    return 0;
}
