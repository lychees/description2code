#include <iostream>
using namespace std;

int main()
{
    int i, t, n;
    long x, s, c, a, b;
    cin>>t;
    while(t--)
    {
    	cin>>n>>x;
    	s = 0;
    	cin>>a;
    	for(i=1; i<n; ++i)
    	{
    		cin>>b;
    		if(b<=a)
    		{
    			c = (a+x-b)/x;
    			b+=c*x;
    			s+=c;
    		}
    		a = b;
    	}
    	cout<<s<<endl;
    }
    return 0;
}
