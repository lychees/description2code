#include <iostream>
using namespace std;

int main()
{
    int t,n,m,ctr,i,j;
    cin>>t;
    while(t--)
    {
    	cin>>n>>m;
    	char a[n][m];
    	for(i=0;i<n;i++)
    	{
    		for(j=0;j<m;j++)
    		{
    			cin>>a[i][j];
    		}
    	}
    	ctr=0;
    	for(i=0;i<n-1;i++)
    	{
    		for(j=0;j<m-1;j++)
    		{
    			if(a[i][j]=='/'&&a[i+1][j]=='\\'&&a[i][j+1]=='\\'&&a[i+1][j+1]=='/')
    			{
    				ctr++;
    			}
    		}
    	}
    	cout<<ctr<<endl;
    }
    return 0;
}
