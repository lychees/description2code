n = int(raw_input())
s = raw_input().split()
s = map(int, s)
g = min(s)
while s != []:
    i = 0
    count = 0
    while i < n:
        s[i] -= g
        count += 1
        if s[i] == 0:
            s.remove(s[i])
            i -= 1
            n -= 1
        i += 1
    try:
        g = min(s)
    except BaseException:
        pass
    print count
