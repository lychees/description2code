#include <iostream>
using namespace std;

int main(){
    int n;
    cin >> n;
    while(n--){
    	int A[26]={};
    	string s;
    	cin >> s;
    	for(int i=0;i<s.size();i++){
    		A[s[i]-'a']++;
    	}
    	int cnt=0;
    	for(int i=0;i<26;i++){
    		if(A[i]%2) cnt++;
    	}
    	cout << max(0,cnt-1) << endl;
    }
    return 0;
}
