#include <iostream>
using namespace std;

void find_no(int M)
{
	long long int m,n;
	if(M>0){
	m = 0;
	n = 1;
	M = M-1;
	}
	
	if(M>0){
		m = n;
		M--;
	}
	long long int m1 = m, n1 = n;
	while(M>0)
	{
	    m = n;
	    n = m1+n;
	    m1 = m;
		M--;
	}
	cout<<m<<" "<<n;
}

int main()
{
    int T,M;
    cin>>T;
    for(int i=0; i<T; i++)
    {
    	cin>>M;
    	find_no(M);
    	cout<<"\n";
    }
    return 0;
}
