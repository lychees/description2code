// CP template
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	for(int i=1;i<=t; i++){
		string a,b;
		cin>>a;
		cin>>b;
		int count = 0;
		int totalLength= a.length() + b.length();
		for(int j=0; j<=a.length(); j++){
			string temp = a.substr(0,j) + b + a.substr(j);
			//cout<<temp<<endl;
			bool check = true;
			int i=0;
			while(check && i<totalLength/2){
				if(temp[i] != temp [totalLength-1 -i])
					check =false;
				i++;
			}
			if(check)
				count++;
		}
		printf("%d\n",count);
	}	
	return 0;
}
