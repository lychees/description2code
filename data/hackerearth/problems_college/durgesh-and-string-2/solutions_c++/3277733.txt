#include <bits/stdc++.h>
#define endl '\n'
using namespace std;

const int maxN = 18;

struct str1
{
	string s1;
	int cntOne, cntZero;

	str1()
	{
		cntOne = 0;
		cntZero = 0;
	}
};

str1 s[maxN];

bool cmp(str1 x, str1 y)
{
	return abs(x.cntOne - x.cntZero) < abs(y.cntOne - y.cntZero);
}

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int n;

	cin>>n;
	for(int i = 0; i < n; i++)
	{
		string s1;

		cin>>s1;
		s[i].s1 = s1;
		for(int j = 0; j < s1.size(); j++)
		{
			if(s[i].s1[j] == '1') s[i].cntOne++;
			else s[i].cntZero++;
		}
	}

	sort(s, s + n, cmp);

	int a, b;
	cin>>a>>b;

	int ans = 0;
	for(int i = 0; i < n; i++)
	{
		if(a >= s[i].cntOne && b >= s[i].cntZero)
		{
			ans++;
			a -= s[i].cntOne;
			b -= s[i].cntZero;
		}
	}
	cout<<ans<<endl;
	return 0;
}
