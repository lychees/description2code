#include <bits/stdc++.h>
using namespace std;

int main()
{
    long long int  n,a[100001],ans[100001],sum=0;
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    sort(a,a+n);
    int j=0;
	for(int i=0;i<n;i+=2)
	{
		ans[i]=a[j];
		ans[i+1]=a[n-j-1];
		j++;
	}
	//for(int i=0;i<n;i++) cout<<ans[i]<<" "; cout<<endl;
	for(int i=0;i<n-1;i++)
	{
		sum+=abs(ans[i]-ans[i+1]);
	}
	sum+=abs(ans[0]-ans[n-1]);
	cout<<sum;
    return 0;
}
