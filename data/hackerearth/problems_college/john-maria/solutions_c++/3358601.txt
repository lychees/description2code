#include <iostream>
using namespace std;

int main()
{
	int x,y,i,j,min,max,a[1000][1000];
    cin>>x>>y;
    max=0;
    for(i=0;i<x;i++)
    {
    	for(j=0;j<y;j++)
    	{
    		cin>>a[i][j];
    		if(j==0)
    		{min=a[i][0];}
    		if(a[i][j]<=min)
    		min=a[i][j];
    	}
    	if(min>max)
    	max=min;
    }
    cout<<max;
    return 0;
}
