#include <iostream>
using namespace std;
int _min(int i,int j){return i<j?i:j;}
#include<string>
string s;
int fun(int i,int j)
{
	if(j<i)
	return 0;
	
	if(s[i]!=s[j])
	return _min(fun(i+1,j),fun(i,j-1))+1;
	else
	return fun(i+1,j-1);
}
	
	
	  
int main()
{
   // cout << "Hello World!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
    	int n;
    	cin>>n;
    	cin>>s;
    	cout<<fun(0,n-1)<<'\n';
    }
    
    return 0;
}
