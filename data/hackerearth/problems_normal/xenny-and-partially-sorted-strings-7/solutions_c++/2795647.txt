#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
#include<utility>
#include<cstdlib>
#include<cstring>
using namespace std;
int t,n,k,m;
string temp;
vector< pair< string , int > > inp;

bool cmp( pair<string ,int >a, pair<string ,int >b ){

for(int i=0;i<m;i++){
	if(a.first[i]<b.first[i])
		return 1;
	else if(a.first[i]>b.first[i])
		return 0;
}
return a.second < b.second;
}

int main()
{
std::ios::sync_with_stdio(false);

cin>>t;
while(t--){

inp.clear();

cin>>n>>k>>m;

for(int i=0;i<n;i++){
	cin>>temp;
	inp.push_back(make_pair(temp,i));
}
sort(inp.begin(),inp.end(),cmp);

cout<<inp[k-1].first<<endl;

}

return 0;
}
