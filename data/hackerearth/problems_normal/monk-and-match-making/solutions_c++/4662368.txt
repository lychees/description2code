#include<bits/stdc++.h>
using namespace std;
long long int b1=89,b2=199,mod1=1e9+7,mod2=1e9+9;
int main()
{
	char w[100050];
	scanf("%s",w);
	long long int i,j,le[100500],le2[100500],ha[100500],ha2[100500],n,q;
	n=strlen(w);
	le[0]=le2[0]=ha[0]=ha2[0]=1;
	for(i=1;i<=n;i++)
	{
		le[i]=(b1*le[i-1])%mod1;
		le2[i]=(b2*le2[i-1])%mod2;
	}
	for(i=1;i<=n;i++)
	{
		ha[i]=(ha[i-1]*b1+w[i-1]-'a'+1)%mod1;
		ha2[i]=(ha2[i-1]*b2+w[i-1]-'a'+1)%mod2;
	}
	scanf("%lld",&q);
	while(q--)
	{
		long long int l,r,a,b,c,d,len,l1,r1;
		scanf("%lld%lld%lld%lld",&l,&r,&l1,&r1);
		if(r-l==r1-l1)
		{
			len=r-l+1;
			a=(ha[r]-(ha[l-1]*le[len])%mod1+mod1)%mod1;
			b=(ha2[r]-(ha2[l-1]*le2[len])%mod2+mod2)%mod2;
			l=l1;
			r=r1;
			c=(ha[r]-(ha[l-1]*le[len])%mod1+mod1)%mod1;
			d=(ha2[r]-(ha2[l-1]*le2[len])%mod2+mod2)%mod2;
		//	cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
			      pair < long long int , long long int > H1(a,b);
                pair < long long int , long long int > H2(c,d);
     
			if(H1==H2)
			printf("Yes\n");
			else
			printf("No\n");

		}
		else
		printf("No\n");
	}
}
