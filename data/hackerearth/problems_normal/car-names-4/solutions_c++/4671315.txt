#include <iostream>
using namespace std;

int main()
{
    int n;
	cin >> n;
	while (n--)
	{
		string s;
		cin >> s;
		int l = s.length();
		int i, j,x=0, c=1;
		int a[500];
		for (i = 0; i < l; i++)
		{
			c = 1;
			for (j = i+1; j < l; j++)
			{

				if (s[i] == s[j])
				{
					c++;
				}
				
			}
			a[x] = c;
			i = i + (c - 1);
			x++;
			//cout << "c" << c << endl;
			//cout << "x" << x << endl;
		}
		//cout << x;
		if (x > 3  || x<3)
		{
			cout << "Not OK"<<endl;
		}
		else
		{
			int flag = 0;
			int y = 0;
			for (y; y < 2; y++)
			{
				if (a[y] == a[y + 1])
				{
					flag = 1;
				}
			}
			if (flag)
			{
				cout << "OK"<<endl;
			}
			else
			{
				cout << "Not OK"<<endl;
			}
		}
	}
    return 0;
}
