#include <iostream>
#include <cstring>
#include <algorithm>
#include <map>
using namespace std;
int n;
map<pair<int,int>, int> mp;
char s1[100001],s2[100001];
int solve(int i1, int i2){
	if (i1==n || i2==n) return 0;
	if (mp.count({i1,i2})) return mp[{i1,i2}];
	if (s1[i1]==s2[i2]){
		return mp[{i1,i2}] = 1+solve(i1+1,i2+1);
	}
	return mp[{i1,i2}] = max(solve(i1,i2+1),solve(i1+1,i2));
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    	//string s1,s2;
    	scanf("%d",&n);
    	//cin>>s1;
    	//cin>>s2;
    	scanf("%s",&s1);
    	scanf("%s",&s2);
    	mp.clear();
    	printf("%d\n",solve(0,0));
    }
    return 0;
}
