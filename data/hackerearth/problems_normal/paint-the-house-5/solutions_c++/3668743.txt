struct st
{
	ll t,x,y;
}a[100001];	

bool way(st a,st b)
{
	if(a.t==b.t)
	{
		if(a.y==b.y)
		return a.x<b.x;
		return a.y>b.y;
	}
	return a.t<b.t;
}
int main()
{
	ll i,j,n,d,cost=0,max_speed;
	cin>>n>>d;
	for(i=1;i<=n;i++)
	{
		cin>>a[i].t;
		cin>>a[i].x;//x is cost
		cin>>a[i].y;//y is speed
	}
	sort(a+1,a+n+1,way);
	cost+=a[1].x;
	d-=(a[2].t-a[1].t)*(a[1].y);
	max_speed=a[1].y;
	for(i=2;i<=n;i++)
	{
		if(d<=0)
		break;
		if(a[i].y<=max_speed)
		d-=(a[i+1].t-a[i].t)*max_speed;
		else 
		{
			max_speed=a[i].y;
			d-=(a[i+1].t-a[i].t)*max_speed;
			cost+=a[i].x;
		}
	}
	cout<<cost;
	return 0;
}
