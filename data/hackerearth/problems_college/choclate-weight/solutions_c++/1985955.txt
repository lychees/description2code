#include<bits/stdc++.h>
using namespace std;
#define MAX 100000
void Pairs(int arr[], int arr_size, int sum)
{
  int i, temp,d=0;
  bool binMap[MAX] = {0};

  for(i = 0; i < arr_size; i++)
  {
    temp = sum - arr[i];
    if(temp >= 0 && binMap[temp] == 1)
    {
        d++;
      cout << "I Like You" << endl;
      break;
    }
    binMap[arr[i]] = 1;
  }
  if(d == 0)
  cout << "I Hate You" << endl;
}

int main()
{
    int n;
    cin >> n;
    int a[n];
    for(int i=0;i<n;i++) {
        cin >> a[i];
    }
        int t;
        cin >> t;
        while(t--)  {
            int m,c=0;
            cin >> m;
            for(int i=0;i<n;i++)  {
                if(m == a[i])  {
                    c++;
                    break;
                }
            }
            if( c == 0)
                Pairs(a,n,m);
                else
                    cout << "I Like You" << endl;

        }
    }
