#include <iostream>

using namespace std;

int main()
{
	int i, j, n, k, d, sum, ans = 0;
	cin >> n >> k >> d;
	long long int sums[n];
	int nums[n];
	for(i = 0; i < n; i++)
		cin >> nums[i];
//	long long int sums[n];
	for(i = 0; i <= n - k; i++)
	{
		sum = 0;
		for(j = i; j < i + k; j++)
			sum += nums[j];
		sums[i] = sum;
//		cout << nums[i] << " ";
	}
//	cout << endl;
	int prev = 0;
	for(i = k - 1; i < n; i += 2*k-1)
	{
		while(i >= 0 && nums[i] == 0)
			i--;
		if(i < prev)
		{
			ans = -1; 
			break;
		}
		ans++;
		prev = i + k - 1;
	}
	cout << ans << endl;
	return 0;
}
