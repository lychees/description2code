struct TurboMatching{  
	int n;
	VI g[max_n];   
	VI left;
	int m1[max_n], m2[max_n];   
	bool c[max_n];

	void add_vertex(int side){
		if(side == 0) left.pb(n);
		n++;
	}

	void add_edge(int u,int v){
		g[u].pb(v);
		g[v].pb(u);
	}

	bool dfs(int u) {
    if (u<0) return true;
    if (c[u]) return false; else c[u]=true;
    FORE(v, g[u]) 
        if (dfs(m2[*v])) { m1[u] = *v; m2[*v] = u; return true; }
    return false;
	}

	int matching() {
    FOR(i,0,n) m1[i]=-1;
    FOR(i,0,n) m2[i]=-1;
    bool changed;
    do {
        changed = 0;
        FORE(i,left) c[*i]=false;
        FORE(i,left) if (m1[*i]<0) changed |= dfs(*i);
    } while(changed);
    int siz = 0;
    FORE(i,left) siz += (m1[*i]!=-1);
    return siz;
	}
};


int n, m;
VI v;
vector<PII> dupa;

bool check(int k) {
	TurboMatching match;
	match.n = 0;
	FOR(i,0,n) match.add_vertex(0);
	FOR(i,0,n) match.add_vertex(1);
	FOR(i,0,n) FOR(j,0,n) FOR(u,0,k) {
		if (v[i]+j == dupa[u].nd) match.add_edge(i,j+n);		
	}
	if (match.matching()==n) return true;
	return false;
}
