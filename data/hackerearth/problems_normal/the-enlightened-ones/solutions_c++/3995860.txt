#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,b) for(int i=a;i<b;i++)
int check(int arr[], int n, int mid, int k)
{
	int temp;
	temp=arr[0]+mid;
	k--;
	rep(i,1,n+1)
	{
	  if(temp+mid>=arr[i])
	  {
	  	continue;
	  }
	  else
	  {
	  	temp=arr[i]+mid;
	  	k--;
	  }
	}
	if(k>=0)
	return 1;
	else
	return 0;
}
int main()
{
    int n,k,mid;
    cin>>n>>k;
    int arr[n];
    rep(i,0,n)
    cin>>arr[i];
    sort(arr,arr+n);
    int l=1;
    int h=10000000;
    while(l<=h)
    {
    	mid=(h+l)/2;
    	if(check(arr,n-1,mid,k)==1&&check(arr,n-1,mid-1,k)==1)
    	{
    		h=mid-1;
    	}
    	else if(check(arr,n-1,mid,k)==1&&check(arr,n-1,mid-1,k)==0)
    	{
    		cout<<mid<<endl;
    		return 0;
    	}
    	else if(check(arr,n-1,mid,k)==0)
    	{
    		l=mid+1;
    	}
    }
    return 0;
}
