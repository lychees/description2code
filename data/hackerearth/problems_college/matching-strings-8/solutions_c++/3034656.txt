#include <iostream>
using namespace std;

int main()
{
    int T;
    cin >> T;
    
    while(T--)
    {
    	string a, b;
    	cin >>a >> b;
    	int count = 0;
    	int len = (a.size() < b.size())? a.size():b.size();
    	for (int i = 0; i < len; ++i)
    	{
    		if (a.c_str()[i] == b.c_str()[i])
    			count++;
    	}
    	
    	cout << count << endl;
    }
    return 0;
}
