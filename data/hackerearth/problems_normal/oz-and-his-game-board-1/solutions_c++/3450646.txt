#include <iostream>

using namespace std;


int main() {
        long long int T, sum1, sum2;
        cin >> T;
        while(T--) {
                long long int N, M, A, B, C;
                cin >> N >> M >> A >> B >> C;
                sum1 = N*A+M*B;
                //cout << sum1 << endl;
                if(N>M) {
                        sum2 = 2*M*C + (N-M)*A;

                        //cout << sum2 << endl;
                } else {
                        sum2 = 2*N*C + (M-N)*B;
                        //cout << sum2 << endl;
                }
                if(sum1>sum2) {
                        cout << sum1 << endl;
                } else {
                        cout << sum2 << endl;
                }
        }
}
