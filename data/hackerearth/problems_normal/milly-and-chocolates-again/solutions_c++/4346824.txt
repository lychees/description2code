int search(long a[],int start,int end,int item){
	int mid = (start+end)/2;
	int index = 0;
	if(mid==0){
		return 0 ;
	}
	if(item == a[mid]){
		return mid;
	}
	if(item< a[mid] && item> a[mid-1]){
		return mid;
	}
	
	if(item > a[mid]){
		index = search(a,mid+1,end,item);
	}else{
		index = search(a,start,mid-1,item);
	}
	return index;
}
