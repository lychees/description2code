#include<bits/stdc++.h>
using namespace std;

int main()
{
	std::map< string, int > mymap;
	string str,temp;
	getline(cin,str);
	transform(str.begin(), str.end(), str.begin(), ::toupper);
    temp=str;
	stringstream ss,s;
	ss.str(str);
	while(ss>>str)
	{
	  if(!mymap.count(str))
	  mymap[str]=1;
	  else
	  {
	  	mymap.at(str)=++(mymap.find(str)->second);
	  }
	}
	s.str(temp);
	while(s>>temp)
	{
		if(mymap.count(temp))
		{
			cout<<temp<<" "<<mymap.at(temp)<<endl;
			mymap.erase (temp);
		}
	}
	
	  
	return 0;
}
