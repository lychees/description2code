#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		int k;
		cin>>k;
		k--;
		int prefix[s.length()+1];
		memset(prefix,0,sizeof(prefix));
		int i,j=0;
    	prefix[0]=0;
    	int max1=0;
    	for(i=1;i<s.size();i++){
        	while(j && s[i]!=s[j])
        		j=prefix[j-1];
        	if(s[i]==s[j])
        		j++;
        	prefix[i]=j;
        	if(i<=k) max1=max(max1,prefix[i]);
    	}
		int k1=prefix[s.length()-1];
		while(k1>max1 && k1)k1=prefix[k1-1];
		if(k1==0){
			cout<<"Puchi is a cheat!"<<endl;
		}else{
			for(int i=0;i<k1;i++)cout<<s[i];
			cout<<endl;
		}
		//for(int i=0;i<s.length();i++)cout<<prefix[i]<<" ";
	}
	return 0;
}
