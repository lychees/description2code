#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t, n, k, po;
    cin >> t;
    while(t--)
    {
        cin >> n >> k;
        po = pow(2,n);
        if(k%po == po-1)
        {
            cout << "YES" << endl;
        }
        else
        {
            cout << "NO" << endl;
        }
    }
    return 0;
}
