#include <iostream>
using namespace std;

int max(int a , int b){
	return a>b?a:b;
}

int min(int a , int b){
	return a>b?b:a;
}

int main()
{
	 int t ; cin >> t ; 
	 
	 while(t--){
	 	
	int F,G,S,P,C,ll,ans=0;
	cin >>F>>G>>S>>P>>C>>ll;
	
	for(int s=0; s*100<=F;s++){
		for(int p=0;p*125+s*100<=F && p*50<=G;p++){
				int food = F - s*100 - p*125;
				int gold = G - p*50;
			int c = min(food/50,gold/100);
			ans = max(ans,s*S+p*P+c*C);
		}
	}
	
	if(ans>ll)
	cout << ans-ll << endl ;	
	else
	cout << "-1" << endl ;
	 	
	 }
    
    return 0;
}
