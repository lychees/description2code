#include <bits/stdc++.h>
using namespace std;

char a[1005];

int main() {
	scanf("%s",a);
	int l = strlen(a);
	if(isalpha(a[0])){
		a[0]=	toupper(a[0]);
	}
	for(int i=0;i<l-1;i++){
		if(a[i]=='_'){
			if(isalpha(a[i+1])){
				//a[i+1]-=32;
				a[i+1]=	toupper(a[i+1]);
			}
		}
	}
	printf("%s\n",a);
	return 0;
}
