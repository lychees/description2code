#include <iostream>

using namespace std;

long long int arr[100001];

int main()
{
	long long int t, i, j, k, n, m, cur, max;
	cin >> t;
	while (t--) {
		cin >> n >> k;
		for (i = 0; i < n; ++i) {
			cin >> arr[i];
		}
		cur = max = 0;
		for (i = 0; i < n; ++i) {
			if (arr[i] <= k) {
				cur += arr[i];
			} else {
				if (cur > max) max = cur;
				cur = 0;
			}	
		}
		if (cur > max) max = cur;
		cout << max << endl;
	}

	return 0;
}
