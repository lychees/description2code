#include <iostream>

//#define f(i,0,n){for(i=0;i<n;i++)}
typedef long int nt;
using namespace std;
char str[55];
int a[55][55];
int fun(int i,int j)
{if(i>=j){return 1;}
  if(a[i][j]!=-1)return a[i][j];
	if(str[i]==str[j])
	{a[i][j]=fun(i+1,j-1);
	 return a[i][j];
	}
	else {a[i][j]=0;
		return 0;
	}
}
int main()
{   int i,j;
 //char str[55];
    //cout << "Hello World!" << endl;
    ss(str);
    int n=strlen(str);
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {a[i][j]=-1;
     if(i==j)a[i][j]=1;
    } 
    nt count=0;
    for(i=0;i<n;i++)
    for(j=i;j<n;j++)
    {if(fun(i,j)==1)count++;
    }
    cout<<count<<endl;
    return 0;
}
