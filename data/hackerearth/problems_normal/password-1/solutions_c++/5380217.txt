#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n,i,j,k,l;
    char a[101][15];
    cin>>n;
    for(i=0;i<n;i++)
    {
    	scanf("%s",a[i]);
    	for(j=0;j<i;j++)
    	{
    		if(strlen(a[i])==strlen(a[j]))
    		{
    			l=strlen(a[i]);
    			for(k=0;k<l;k++)
    			if(a[i][k]!=a[j][l-k-1])
    			goto next;
    			cout<<l<<" "<<a[i][l/2];
    			return 0;
    		}
    		next:;
    	}
    }
    return 0;
}
