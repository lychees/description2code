#include <iostream>
#include <cstdio>
#include <cmath>

using namespace std;

int main()
{
	int t;
	long r,c;
	cin >> t;
	while(t--) {
		cin >> r >> c;
		int step = min(r,c);
		long res1 = step * (step*(r+1) + 1 - r) / 2;
		long res2 = step * (step*(c+1) + 1 - c) / 2;
		cout << res1 + res2 << endl;
	}
	return 0;
}
