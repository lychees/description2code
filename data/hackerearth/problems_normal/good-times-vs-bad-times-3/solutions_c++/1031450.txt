#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int X[24*60*60+5];

int main() {
	for(int h = 0; h < 24; h++) {
		for(int m = 0; m < 60; m++) {
			for(int s = 0; s < 60; s++) {
				int g = 0;
				if(h) g = __gcd(g, h);
				if(m) g = __gcd(g, m);
				if(s) g = __gcd(g, s);
				if(g != 1) X[h*60*60+m*60+s] = 1;
			}
		}
	}

	for(int i = 1; i < 24*60*60; i++)
		X[i] += X[i-1];

	int t;
	cin >> t;
	while(t--) {
		int h,m,s;
		cin >> h >> m >> s;
		int num = X[24*60*60-1] - X[h*60*60+m*60+s-1];
		int den = 24*60*60 - (h*60*60+m*60+s) - num;
		int k = __gcd(num, den);
		if(!num) k = 1;
		printf("%d:%d\n",num/k,den/k);
	}
	return 0;
}
