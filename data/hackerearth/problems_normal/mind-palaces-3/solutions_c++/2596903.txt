#include <iostream>
#include<cstdio>
using namespace std;
int arr[1001][1001];
int main()
{
	int n,m,q,x;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		cin>>arr[i][j];
	}
	cin>>q;
	while(q--)
	{int flag=0;
		cin>>x;
		int i=0,j=m-1;
		while(j>=0 && i<=n-1)
		{
			if(arr[i][j]==x)
			{
				cout<<i<<" "<<j<<endl;
				flag=1;
				break;
			}
			else if(arr[i][j]<x)
			i++;
			else
			j--;
		}
		if(!flag)
		cout<<"-1 -1"<<endl;
	}
    return 0;
}
