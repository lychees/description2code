#include<bits/stdc++.h>
#define ll long long int
#define inp(x) scanf("%lld",&x)
using namespace std;

int main(){
	ll n,k;
	inp(n);inp(k);
	ll c = 0;
	for(ll i=k;i<=n;++i){
		ll r = 0,x = i;
		for(ll j=2;j*j<=x;++j){
			if(x%j==0){r+=j;}
			while(x%j==0){
				x/=j;
			}
		}
		if(x!=1) {r += x;}
		if(r==k){
			++c;
		}
	}
	printf("%lld\n",c);
	return 0;
}
