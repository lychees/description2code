#include<bits/stdc++.h>
using namespace std;
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int a,b;
        int h,m;
        cin>>h>>m;
        if(h==12) h=0;
        if(m==60) m=0;
        a = (h*60 + m)/2;
        b = 6*m;
        int ans = abs(a-b);
        ans = min(360-ans,ans);
        cout<<ans<<endl;
    }
}
