#include<bits/stdc++.h>
#define max_limit 1000000
using namespace std;
long str_to_int(string str);
int visited[max_limit];
queue<string> q;
int main()
{
    int t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--){
        long i,j,n,a,h,minm;
        string str,ans;
        cin>>str;
        cin>>a>>h;
        n=str.size();
        h=h%n;
        for(i=0;i<max_limit;i++)
            visited[i]=0;
        minm=str_to_int(str);
        ans=str;
        q.push(str);
        while(!q.empty())
        {
            string tmp2,tmp=q.front();
            q.pop();
            long num=str_to_int(tmp);
            //cout<<"popped "<<num<<"\n";
            if(num<minm){
                    ans=tmp;
                    minm=num;
            }
            tmp2=tmp;
            if(visited[num])
                continue;
            visited[num]=1;
            for(i=1;i<tmp2.size();i+=2)
            {
                long j=tmp2[i]-'0';
                j=(j+a)%10;
                tmp2[i]=j+'0';
            }
            i=str_to_int(tmp2);
            if(!visited[i])
            {
                q.push(tmp2);
            }
            tmp2=tmp;
            std::rotate(tmp2.begin(),tmp2.begin()+n-h,tmp2.end());
            i=str_to_int(tmp2);
            if(!visited[i])
            {
                q.push(tmp2);
            }
        }
        cout<<ans<<"\n";
    }
    return 0;
}
long str_to_int(string str)
{
    long i,j;
    i=0;
    for(j=0;j<str.size();j++)
        {
            i=i*10+str[j]-'0';
        }
    return i;
}
