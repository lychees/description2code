#include<bits/stdc++.h>

using namespace std;
long arr[100005];
int main()
{
    long t,n,k;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(int i=0;i<n;++i)
            cin>>arr[i];
           long low=0;
            for(int i=1;i<n;i++)
                low=max(low,arr[i]-arr[i-1]);


        long high =1e9;

        long mid;
        while(low<high)
        {
            mid=(high+low)>>1;


            long point=0;
            long jump=0;
            for(int i=1;i<n;i++)
            {
                if(arr[i]-arr[point]>mid)
                {
                    point=i-1;
                    jump++;
                }

            }
            jump++;
            if(jump<=k)
                 high=mid;
            else
                low=mid+1;

        }
        cout<<low<<"\n";
    }

}
