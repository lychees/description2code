using namespace std;

ll b[100020];

void sieve()
{
	ll i,j;
	for(i=0;i<=100010;i++)
		b[i]=1;
	b[0]=0;
	b[1]=0;
	for(i=2;i<=100010/2;i++)
	{
		if(b[i]==1)
		{
			for(j=i+i;j<=100010;j=j+i)
				b[j]=0;
		}
	}
}
int main()
{
	ll t;
	sieve();
	scan(t);
	while(t--)
	{
		ll n,i;
		scan(n);
		if(n==0)
			printf("4\n");
		else
		{
			rep(i,n+1,100010)
			{
				if(b[i]==0)
					break;
			}
			printf("%lld\n",i);
		}
	}
	return 0;
}
