#include <bits/stdc++.h>
using namespace std;

int main()
{
    int tc;
    cin >> tc;
    while(tc--)
    {
    	long long n;
    	cin >> n;
    	if(n%3 == 0)
    	{
    		long long ans=n/3;
    		cout << ans*(n-1)*(n+1);
    	}
    	else if((n-1)%3 == 0)
    	{
    		long long ans=(n-1)/3;
    		cout << ans*(n)*(n+1) ;
    	}
    	else
    	{
    		long long ans=(n+1)/3;
    		cout << ans * n * (n-1);
    	}
    	cout << endl; 
    }
    return 0;
}
