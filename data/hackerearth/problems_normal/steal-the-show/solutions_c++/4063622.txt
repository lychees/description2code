#include <iostream>
#include <vector>
#include <algorithm>
#include <math.h>
using namespace std;

int main()
{
	int ceo,cto,coo,i,num,N,index=0,total=0;
    cin>>ceo>>coo>>cto>>N;
    vector<int>V,T;
    for(i=0;i<N;i++)
    {
        cin>>num;
        T.push_back(num);
    }
    int minimum=coo<=cto?coo:cto;
    V.push_back(minimum);
    int maximum=coo>=cto?coo:cto;
    sort(T.begin(),T.end());
    for(i=1;i<N+3;i++)
    {
        if(i==(N+3)/2)
        {
           V.push_back(ceo);
           continue;
        }
        if(i==N+2)
            {
            V.push_back(maximum);
            continue;
            }
        V.push_back(T[index++]);
    }
    for(i=1;i<N+3;i++)
        total+=abs(V[i]-V[i-1]);
    cout<<total;    return 0;
}
