inline LD calcPow(LD p, int x){
    if(x == 0){
        return 1.0;
    }
    LD ans = calcPow(p, x / 2);
    ans *= ans;
    if(x % 2 == 1) ans *= p;
    return ans;
}

inline LD calcRoot(LD p, LD t){
    int count = 0;
    LD low = 0.0000000000, high = 1.0000000000, mid, val;
    while(count <= 10000){
        mid = (high + low) / 2.0;
        val = calcPow(mid, t);
        //cout<<"mid = "<<mid<<" val = "<<val<<endl;
        if(val < p){
            low = mid;
        }
        else{
            high = mid;
        }
        count++;
    }
    return mid;
}

    p1 = calcRoot(1 - p, 1440 / ti);
    LD ans = 1 - p1;
    printf("%0.4LF\n", ans);
    return;
}
