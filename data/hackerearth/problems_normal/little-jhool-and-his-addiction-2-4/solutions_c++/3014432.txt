{

	    vector<int>A(n);
	    for(int i=0;i<n;i++)
	    	scanf("%d",&A[i]);
	    sort(A.begin(),A.end());
	    int maxi=INT_MIN,mini=INT_MAX;
	    for(int i=0;i<n/2;i++)
	    {
	    	maxi=max(maxi,A[n-i-1]+A[i]);
	    	mini=min(mini,A[n-i-1]+A[i]);
	    }

	    printf("%d\nNo more girlfriends!\n",maxi-mini);

}
